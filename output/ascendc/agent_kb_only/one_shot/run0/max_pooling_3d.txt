namespace {

static inline int64_t floor_div(int64_t a, int64_t b) {
    return a / b;
}

static inline uint32_t CeilDivU32(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static inline uint32_t RoundUpU32(uint32_t a, uint32_t b) {
    return ((a + b - 1u) / b) * b;
}

} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernel_size = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride      = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding     = attrs->GetAttrPointer<int32_t>(2);

    // Input shape: [N, C, D, H, W]
    const gert::Shape *xShape = context->GetInputShape(0);
    // Robustly read 5D, fallback to flattened size if unknown
    uint32_t n = 1, c = 1, di = 1, hi = 1, wi = 1;
    if (xShape->GetOriginShape().GetDimNum() >= 5) {
        n  = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
        c  = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
        di = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
        hi = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));
        wi = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(4));
    } else {
        uint64_t total = xShape->GetOriginShape().GetShapeSize();
        // Fallback as 5D tensor with D=H=W=1
        wi = 1; hi = 1; di = 1; c = 1; n = static_cast<uint32_t>(total);
    }

    const uint32_t k = static_cast<uint32_t>(*kernel_size);
    const uint32_t s = static_cast<uint32_t>(*stride);
    const uint32_t p = static_cast<uint32_t>(*padding);

    // Output dims (ceil_mode = false)
    const int64_t do_ = floor_div(static_cast<int64_t>(di) + 2LL * p - static_cast<int64_t>(k), static_cast<int64_t>(s)) + 1;
    const int64_t ho  = floor_div(static_cast<int64_t>(hi) + 2LL * p - static_cast<int64_t>(k), static_cast<int64_t>(s)) + 1;
    const int64_t wo  = floor_div(static_cast<int64_t>(wi) + 2LL * p - static_cast<int64_t>(k), static_cast<int64_t>(s)) + 1;

    const uint32_t outD = static_cast<uint32_t>(do_ > 0 ? do_ : 0);
    const uint32_t outH = static_cast<uint32_t>(ho  > 0 ? ho  : 0);
    const uint32_t outW = static_cast<uint32_t>(wo  > 0 ? wo  : 0);
    const uint64_t totalOut64 = static_cast<uint64_t>(n) * c * outD * outH * outW;
    const uint32_t totalOut = static_cast<uint32_t>(totalOut64);

    // workLocal size for ReduceMax (float)
    // elementsPerBlock=8, elementsPerRepeat=64
    auto CeilDiv = [](uint32_t a, uint32_t b){ return (a + b - 1u) / b; };
    const uint32_t maxWin = k * k * k;
    uint32_t firstMaxRepeat = CeilDiv(maxWin, 64u);
    if (firstMaxRepeat == 0) firstMaxRepeat = 1;
    uint32_t iter1OutputCount = firstMaxRepeat * 2u;
    uint32_t workLocalElems = RoundUpU32(iter1OutputCount, 8u);

    // BlockDim: simple mapping: one output voxel per thread block in a grid-stride loop
    uint32_t blockDim = totalOut > 0 ? totalOut : 1u;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_di(di);
    tiling.set_hi(hi);
    tiling.set_wi(wi);
    tiling.set_do_(outD);
    tiling.set_ho(outH);
    tiling.set_wo(outW);
    tiling.set_kernel(k);
    tiling.set_stride(s);
    tiling.set_padding(p);
    tiling.set_workLocalElems(workLocalElems);
    tiling.set_maxWin(maxWin);
    tiling.set_totalOut(totalOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernel_size = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride      = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding     = attrs->GetAttrPointer<int32_t>(2);

    const gert::Shape *x = context->GetInputShape(0);
    gert::Shape *y = context->GetOutputShape(0);

    int64_t n = 1, c = 1, di = 1, hi = 1, wi = 1;
    if (x->GetOriginShape().GetDimNum() >= 5) {
        n  = x->GetOriginShape().GetDim(0);
        c  = x->GetOriginShape().GetDim(1);
        di = x->GetOriginShape().GetDim(2);
        hi = x->GetOriginShape().GetDim(3);
        wi = x->GetOriginShape().GetDim(4);
    } else {
        n = x->GetOriginShape().GetShapeSize();
        c = 1; di = 1; hi = 1; wi = 1;
    }

    const int64_t k = static_cast<int64_t>(*kernel_size);
    const int64_t s = static_cast<int64_t>(*stride);
    const int64_t p = static_cast<int64_t>(*padding);

    const int64_t do_ = (di + 2 * p - k) / s + 1;
    const int64_t ho  = (hi + 2 * p - k) / s + 1;
    const int64_t wo  = (wi + 2 * p - k) / s + 1;

    std::vector<int64_t> outDims = {
        n,
        c,
        do_ > 0 ? do_ : 0,
        ho  > 0 ? ho  : 0,
        wo  > 0 ? wo  : 0
    };
    y->SetOriginShape(ge::Shape(outDims));
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

// This kernel computes 3D MaxPooling (N, C, D, H, W) -> (N, C, Do, Ho, Wo) with uniform kernel/stride/padding.
// ceil_mode=false, return_indices=false. Data type: float.

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t di, uint32_t hi, uint32_t wi,
                                uint32_t do_, uint32_t ho, uint32_t wo,
                                uint32_t kernel, uint32_t stride, uint32_t padding,
                                uint32_t workLocalElems, uint32_t maxWin, uint32_t totalOut)
    {
        n_ = n; c_ = c; di_ = di; hi_ = hi; wi_ = wi;
        do__ = do_; ho_ = ho; wo_ = wo;
        k_ = kernel; s_ = stride; p_ = padding;
        workLocalElems_ = workLocalElems;
        maxWin_ = maxWin;
        totalOut_ = totalOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * c_ * di_ * hi_ * wi_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * c_ * do__ * ho_ * wo_);

        // Queues for temporary buffers
        pipe_.InitBuffer(windowQue_, 1, maxWin_ * sizeof(float));          // window buffer
        pipe_.InitBuffer(reduceDstQue_, 1, 2u * sizeof(float));            // ReduceMax dst (value + index space)
        pipe_.InitBuffer(workQue_, 1, workLocalElems_ * sizeof(float));    // ReduceMax workLocal
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || do__ == 0 || ho_ == 0 || wo_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t outIdx = blockIdx; outIdx < totalOut_; outIdx += blockNum) {
            ComputeOne(outIdx);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t outLinearIdx)
    {
        // Decode (n, c, od, oh, ow) from linear index
        uint32_t ow = outLinearIdx % wo_;
        uint32_t t  = outLinearIdx / wo_;
        uint32_t oh = t % ho_;
        t /= ho_;
        uint32_t od = t % do__;
        t /= do__;
        uint32_t c  = t % c_;
        uint32_t n  = t / c_;

        const int32_t dStart = static_cast<int32_t>(od) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);
        const int32_t hStart = static_cast<int32_t>(oh) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);
        const int32_t wStart = static_cast<int32_t>(ow) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);

        const int32_t dEnd = dStart + static_cast<int32_t>(k_);
        const int32_t hEnd = hStart + static_cast<int32_t>(k_);
        const int32_t wEnd = wStart + static_cast<int32_t>(k_);

        const uint64_t baseNC = (static_cast<uint64_t>(n) * c_ + c) * di_ * hi_ * wi_;

        AscendC::LocalTensor<float> winLocal = windowQue_.AllocTensor<float>();
        uint32_t writePos = 0;

        // Iterate valid ranges and copy contiguous W-runs into window buffer
        const int32_t d0 = dStart < 0 ? 0 : dStart;
        const int32_t d1 = dEnd   > static_cast<int32_t>(di_) ? static_cast<int32_t>(di_) : dEnd;

        const int32_t h0 = hStart < 0 ? 0 : hStart;
        const int32_t h1 = hEnd   > static_cast<int32_t>(hi_) ? static_cast<int32_t>(hi_) : hEnd;

        const int32_t w0base = wStart < 0 ? 0 : wStart;
        const int32_t w1base = wEnd   > static_cast<int32_t>(wi_) ? static_cast<int32_t>(wi_) : wEnd;

        for (int32_t dd = d0; dd < d1; ++dd) {
            for (int32_t hh = h0; hh < h1; ++hh) {
                const int32_t w0 = w0base;
                const int32_t w1 = w1base;
                const uint32_t len = static_cast<uint32_t>((w1 > w0) ? (w1 - w0) : 0);
                if (len == 0) continue;
                const uint64_t gmOffset = baseNC + (static_cast<uint64_t>(dd) * hi_ + static_cast<uint64_t>(hh)) * wi_ + static_cast<uint64_t>(w0);
                AscendC::DataCopy(winLocal[writePos], xGm_[gmOffset], len);
                writePos += len;
            }
        }

        if (writePos == 0) {
            // No valid inputs (should not happen with standard pooling); write -inf
            AscendC::LocalTensor<float> dstLocal = reduceDstQue_.AllocTensor<float>();
            // Use ClampMax-like path to set a large negative value via DataCopy from a constant GM? Fallback: zero copy is fine for placeholder.
            // Here we set to 0 as a safe fallback.
            dstLocal.SetValue(0, 0.0f);
            AscendC::DataCopy(yGm_[GetOutOffset(n, c, od, oh, ow)], dstLocal, 1);
            reduceDstQue_.FreeTensor(dstLocal);
            windowQue_.FreeTensor(winLocal);
            return;
        }

        // ReduceMax on window buffer
        AscendC::LocalTensor<float> dstLocal = reduceDstQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> workLocal = workQue_.AllocTensor<float>();
        AscendC::ReduceMax<float>(dstLocal, winLocal, workLocal, static_cast<int32_t>(writePos), false);

        // Write the first element (max value) to output
        const uint64_t outOffset = GetOutOffset(n, c, od, oh, ow);
        AscendC::DataCopy(yGm_[outOffset], dstLocal, 1);

        workQue_.FreeTensor(workLocal);
        reduceDstQue_.FreeTensor(dstLocal);
        windowQue_.FreeTensor(winLocal);
    }

    __aicore__ inline uint64_t GetOutOffset(uint32_t n, uint32_t c, uint32_t od, uint32_t oh, uint32_t ow) const
    {
        return (((static_cast<uint64_t>(n) * c_ + c) * do__ + od) * ho_ + oh) * wo_ + ow;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> windowQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> reduceDstQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{1}, c_{1}, di_{1}, hi_{1}, wi_{1};
    uint32_t do__{1}, ho_{1}, wo_{1};
    uint32_t k_{1}, s_{1}, p_{0};
    uint32_t workLocalElems_{0}, maxWin_{0}, totalOut_{0};
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tilingData.n, tilingData.c, tilingData.di, tilingData.hi, tilingData.wi,
            tilingData.do_, tilingData.ho, tilingData.wo,
            tilingData.kernel, tilingData.stride, tilingData.padding,
            tilingData.workLocalElems, tilingData.maxWin, tilingData.totalOut);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor max_pooling_3d_custom_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding) {
    at::Tensor result;

    // Output shape computation mirrors PyTorch MaxPool3d with ceil_mode=false
    auto sizes = self.sizes();
    TORCH_CHECK(sizes.size() == 5, \"Input must be 5D (N, C, D, H, W)\");
    int64_t N = sizes[0], C = sizes[1], D = sizes[2], H = sizes[3], W = sizes[4];

    int64_t Do = (D + 2 * padding - kernel_size) / stride + 1;
    int64_t Ho = (H + 2 * padding - kernel_size) / stride + 1;
    int64_t Wo = (W + 2 * padding - kernel_size) / stride + 1;
    Do = std::max<int64_t>(Do, 0);
    Ho = std::max<int64_t>(Ho, 0);
    Wo = std::max<int64_t>(Wo, 0);

    result = at::empty({N, C, Do, Ho, Wo}, self.options());

    EXEC_NPU_CMD(aclnnMaxPooling3dCustom, self, static_cast<int64_t>(kernel_size),
                 static_cast<int64_t>(stride), static_cast<int64_t>(padding), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"max_pooling_3d_custom\", &max_pooling_3d_custom_impl_npu);
}
