host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB budget for our kernel usage
static constexpr uint32_t ALIGN_ELEMS = 2048;            // DMA-friendly alignment in elements
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid very large per-iter tiles

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const int64_t rank = static_cast<int64_t>(in_shape->GetOriginShape().GetDims().size());
    if (rank <= 0) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int64_t axis = static_cast<int64_t>(*dimAttr);
    if (axis < 0) axis += rank;
    if (axis < 0 || axis >= rank) {
        return ge::GRAPH_PARAM_INVALID;
    }

    // Only support reduction along the last dimension for this optimized kernel.
    if (axis != (rank - 1)) {
        return ge::GRAPH_PARAM_INVALID;
    }

    // Compute rows and cols by flattening all but the last dimension
    uint64_t colsU64 = static_cast<uint64_t>(in_shape->GetOriginShape().GetDim(rank - 1));
    uint64_t rowsU64 = 1;
    for (int64_t i = 0; i < rank - 1; ++i) {
        rowsU64 *= static_cast<uint64_t>(in_shape->GetOriginShape().GetDim(i));
    }
    if (colsU64 == 0 || rowsU64 == 0) {
        return ge::GRAPH_PARAM_INVALID;
    }
    const uint32_t cols = static_cast<uint32_t>(colsU64);
    const uint32_t totalRows = static_cast<uint32_t>(rowsU64);

    // UB model:
    // - inQueueX: 2 * tileCols
    // - outQueueY: 2 * tileCols
    // - tmpQueue: 1 * tileCols
    // Total floats in UB for one tile: 5 * tileCols => bytes = 20 * tileCols
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileCols = maxTile;
    if (cols < tileCols) {
        const uint32_t aligned = AlignDown(cols, ALIGN_ELEMS);
        tileCols = (aligned > 0) ? aligned : cols;
    }

    // BlockDim based on row count
    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_cols(cols);
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);
    tiling.set_axis(static_cast<int32_t>(axis));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;  // no workspace needed
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(-1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t cols, uint32_t tileCols)
    {
        rows_ = totalRows;
        cols_ = cols;
        tileCols_ = tileCols;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(rows_) * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(rows_) * cols_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileCols_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0 || tileCols_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            float maxVal = -3.402823e+38f;  // approx -FLT_MAX
            // Pass 1: find per-row max
            for (uint32_t colBase = 0; colBase < cols_; colBase += tileCols_) {
                const uint32_t len = ((colBase + tileCols_) <= cols_) ? tileCols_ : (cols_ - colBase);
                CopyIn(row, colBase, len);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                float chunkMax = AscendC::ReduceMax(xLocal, len);
                if (chunkMax > maxVal) maxVal = chunkMax;
                inQueueX_.FreeTensor(xLocal);
            }

            // Pass 2: compute sum of exp(x - max)
            float sumExp = 0.0f;
            for (uint32_t colBase = 0; colBase < cols_; colBase += tileCols_) {
                const uint32_t len = ((colBase + tileCols_) <= cols_) ? tileCols_ : (cols_ - colBase);
                CopyIn(row, colBase, len);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

                // tmp = x - maxVal
                AscendC::Adds(tmpLocal, xLocal, -maxVal, len);
                // tmp = exp(tmp)
                AscendC::Exp(tmpLocal, tmpLocal, len);
                float chunkSum = AscendC::ReduceSum(tmpLocal, len);
                sumExp += chunkSum;

                tmpQueue_.FreeTensor(tmpLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            // Pass 3: compute y = (x - max) - log(sumExp)
            for (uint32_t colBase = 0; colBase < cols_; colBase += tileCols_) {
                const uint32_t len = ((colBase + tileCols_) <= cols_) ? tileCols_ : (cols_ - colBase);
                CopyIn(row, colBase, len);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

                // yLocal = x - maxVal
                AscendC::Adds(yLocal, xLocal, -maxVal, len);

                // tmpLocal = log(sumExp) broadcast
                AscendC::Dup(tmpLocal, sumExp, len);
                AscendC::Ln(tmpLocal, tmpLocal, len);

                // yLocal = yLocal - tmpLocal  => x - max - log(sumExp)
                AscendC::Sub(yLocal, yLocal, tmpLocal, len);

                outQueueY_.EnQue<float>(yLocal);
                tmpQueue_.FreeTensor(tmpLocal);
                inQueueX_.FreeTensor(xLocal);

                CopyOut(row, colBase, len);
                AscendC::LocalTensor<float> yLocalDeq = outQueueY_.DeQue<float>();
                outQueueY_.FreeTensor(yLocalDeq);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row, uint32_t colBase, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        const uint64_t gmOffset = static_cast<uint64_t>(row) * cols_ + colBase;
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row, uint32_t colBase, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        const uint64_t gmOffset = static_cast<uint64_t>(row) * cols_ + colBase;
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileCols_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.cols, tiling_data.tileCols);
    op.Process();
}
"""
