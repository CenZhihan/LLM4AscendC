host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint64_t ProductUpTo(const gert::Shape& shape, int64_t end) {
    // product of dims [0, end)
    uint64_t p = 1;
    for (int64_t i = 0; i < end; ++i) {
        p *= static_cast<uint64_t>(shape.GetDim(i));
    }
    return p;
}

static inline uint64_t ProductFrom(const gert::Shape& shape, int64_t start) {
    // product of dims [start, rank)
    uint64_t p = 1;
    const int64_t rank = shape.GetDimNum();
    for (int64_t i = start; i < rank; ++i) {
        p *= static_cast<uint64_t>(shape.GetDim(i));
    }
    return p;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint32_t b) {
    return static_cast<uint32_t>((a + b - 1u) / b);
}
} // namespace

namespace optiling {

// UB model for this kernel per core (worst concurrent):
//   inQueueX: 1 * tile
//   outQueueY: 1 * tile
//   tmpQueue: 1 * tile
// Total floats ~= 3 * tile => bytes ~= 12 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative share per-core
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimAttr != nullptr) ? *dimAttr : 1;

    const gert::Shape& origin = context->GetInputShape(0)->GetOriginShape();
    const int64_t rank = origin.GetDimNum();
    if (rank <= 0) {
        // Scalar edge-case: treat as 1x1
        tiling.set_rows(1);
        tiling.set_cols(1);
        tiling.set_tileCols(1);
        tiling.set_blockDim(1);
        tiling.set_dim(0);
        context->SetBlockDim(1);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    if (dim < 0) dim += static_cast<int32_t>(rank);
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = static_cast<int32_t>(rank - 1);

    const uint64_t rows64 = ProductUpTo(origin, dim);
    const uint64_t cols64 = ProductFrom(origin, dim);

    // tileCols from UB budget: bytes = 12 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    if (maxTile == 0) maxTile = ALIGN_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileCols = static_cast<uint32_t>(maxTile);
    if (cols64 < tileCols) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(cols64), ALIGN_ELEMS);
        tileCols = (aligned > 0u) ? aligned : static_cast<uint32_t>(cols64);
    }

    // Choose blockDim based on number of rows
    uint32_t blockDim = static_cast<uint32_t>(rows64);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_rows(static_cast<uint32_t>(rows64));
    tiling.set_cols(static_cast<uint32_t>(cols64));
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    gert::Shape *y_shape = context->GetOutputShape(0);

kernel_src="""

#include "kernel_operator.h"
#include <cfloat>
#include <cmath>

constexpr int32_t BUFFER_NUM = 1;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t tileCols)
    {
        rows_ = rows;
        cols_ = cols;
        tileCols_ = tileCols;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(rows_) * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(rows_) * cols_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileCols_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0 || tileCols_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            const uint64_t rowBase = static_cast<uint64_t>(row) * cols_;

            // Pass 1: find row max
            float rowMax = -FLT_MAX;
            for (uint32_t c = 0; c < cols_; c += tileCols_) {
                uint32_t len = cols_ - c;
                if (len > tileCols_) len = tileCols_;
                CopyIn(rowBase + c, len);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                float tmax = AscendC::ReduceMax(xLocal, len);
                rowMax = fmaxf(rowMax, tmax);
                inQueueX_.FreeTensor(xLocal);
            }

            // Pass 2: compute exp(x - rowMax), accumulate sum, and store numerator
            float rowSum = 0.0f;
            for (uint32_t c = 0; c < cols_; c += tileCols_) {
                uint32_t len = cols_ - c;
                if (len > tileCols_) len = tileCols_;
                CopyIn(rowBase + c, len);

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // tLocal = x - rowMax
                AscendC::Adds(tLocal, xLocal, -rowMax, len);
                // yLocal = exp(tLocal)
                AscendC::Exp(yLocal, tLocal, len);

                float part = AscendC::ReduceSum(yLocal, len);
                rowSum += part;

                // write numerators to y GM
                outQueueY_.EnQue(yLocal);
                CopyOut(rowBase + c, len);

                // release locals
                outQueueY_.FreeTensor(yLocal);
                tmpQueue_.FreeTensor(tLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            // Pass 3: normalize by sum
            const float invSum = 1.0f / rowSum;
            for (uint32_t c = 0; c < cols_; c += tileCols_) {
                uint32_t len = cols_ - c;
                if (len > tileCols_) len = tileCols_;

                // load previously stored numerators
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, yGm_[rowBase + c], len);

                AscendC::Muls(yLocal, yLocal, invSum, len);

                AscendC::DataCopy(yGm_[rowBase + c], yLocal, len);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileCols_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.rows, tiling_data.cols, tiling_data.tileCols);
    op.Process();
}
