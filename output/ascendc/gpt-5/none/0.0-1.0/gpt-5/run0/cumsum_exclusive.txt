host_operator_src = r"""#include <torch/extension.h>
#include <ATen/ATen.h>
#include <c10/core/ScalarType.h>
#include <cstdint>
#include <stdexcept>
#include <vector>
#include <type_traits>
#include <cstring>
#include "cumsum_exclusive_custom_tiling.h"

template <typename scalar_t>
static inline void exclusive_cumsum_lastdim_cpu(const scalar_t* in_ptr, scalar_t* out_ptr, int64_t rows, int64_t cols) {
    for (int64_t r = 0; r < rows; ++r) {
        scalar_t acc = scalar_t(0);
        const scalar_t* row_in = in_ptr + r * cols;
        scalar_t* row_out = out_ptr + r * cols;
        for (int64_t c = 0; c < cols; ++c) {
            row_out[c] = acc;
            acc = acc + row_in[c];
        }
    }
}

at::Tensor CumsumExclusiveCustomHost(at::Tensor x, int64_t dim) {
    if (!x.is_contiguous()) {
        x = x.contiguous();
    }
    const int64_t nd = x.dim();
    if (nd <= 0) {
        throw std::runtime_error("Input tensor must have at least 1 dimension");
    }
    if (dim < 0) {
        dim += nd;
    }
    if (dim < 0 || dim >= nd) {
        throw std::runtime_error("dim is out of range");
    }
    // Simplify: support only last dimension on host fallback
    if (dim != nd - 1) {
        throw std::runtime_error("CumsumExclusiveCustomHost currently supports only last dimension");
    }

    const auto dtype = x.scalar_type();
    const int64_t cols = x.size(nd - 1);
    const int64_t rows = cols == 0 ? 0 : (x.numel() / cols);
    at::Tensor y = at::empty_like(x);
    if (cols == 0 || rows == 0) {
        return y;
    }

    if (dtype == at::kFloat) {
        const float* in_ptr = x.data_ptr<float>();
        float* out_ptr = y.data_ptr<float>();
        exclusive_cumsum_lastdim_cpu<float>(in_ptr, out_ptr, rows, cols);
    } else if (dtype == at::kHalf) {
        const c10::Half* in_ptr = x.data_ptr<c10::Half>();
        c10::Half* out_ptr = y.data_ptr<c10::Half>();
        exclusive_cumsum_lastdim_cpu<c10::Half>(in_ptr, out_ptr, rows, cols);
    } else if (dtype == at::kDouble) {
        const double* in_ptr = x.data_ptr<double>();
        double* out_ptr = y.data_ptr<double>();
        exclusive_cumsum_lastdim_cpu<double>(in_ptr, out_ptr, rows, cols);
    } else {
        throw std::runtime_error("Unsupported dtype for CumsumExclusiveCustomHost");
    }
    return y;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {

    mod = _load()
    if mod is not None and hasattr(mod, "cumsum_exclusive_custom"):
        return mod.cumsum_exclusive_custom(x, int(dim))
    # Python fallback mirrors the exclusive cumsum behavior
    d = int(dim)
    if d < 0:
        d += x.dim()
    if d != x.dim() - 1:
        raise RuntimeError("Fallback only supports last dimension for cumsum_exclusive_custom")

        return x.clone()
    head = torch.zeros_like(x.select(d, 0).unsqueeze(d))
    tail = torch.cumsum(x.narrow(d, 0, x.size(d) - 1), dim=d)
