host_tiling_src = r"""#ifndef CUMPROD_CUSTOM_TILING_H
#define CUMPROD_CUSTOM_TILING_H
#include <cstdint>

struct CumprodCustomTiling {
    int64_t outer;
    int64_t axis_len;
    int64_t inner;
    int64_t elem_size;
    int64_t dtype_code;
    int64_t reserved0;
    int64_t reserved1;
    int64_t reserved2;
};

static inline void CumprodCustomMakeTiling(CumprodCustomTiling &t,
                                           const int64_t *shape,
                                           int64_t ndims,
                                           int64_t axis,
                                           int64_t elem_size_bytes,
                                           int64_t dtype_code) {
    int64_t outer = 1;
    for (int64_t i = 0; i < axis; ++i) {
        outer *= shape[i];
    }
    int64_t axis_len = shape[axis];
    int64_t inner = 1;
    for (int64_t i = axis + 1; i < ndims; ++i) {
        inner *= shape[i];
    }
    t.outer = outer;
    t.axis_len = axis_len;
    t.inner = inner;
    t.elem_size = elem_size_bytes;
    t.dtype_code = dtype_code;
    t.reserved0 = 0;
    t.reserved1 = 0;
    t.reserved2 = 0;
}

#endif
"""

kernel_src = r"""// AscendC kernel for cumulative product along a given axis.
// This is a simplified illustrative implementation skeleton.
#include <cstdint>
#include <cmath>
#include "cumprod_custom_tiling.h"

extern "C" __attribute__((visibility("default"))) void cumprod_custom(void* x_ptr,
                                                                      void* y_ptr,
                                                                      void* /*workspace*/,
                                                                      void* tiling_ptr) {
    // This CPU-like fallback implementation is for illustration/testing environments only.
    // Real AscendC kernels run on device and use device intrinsics.
    if (!x_ptr || !y_ptr || !tiling_ptr) {
        return;
    }
    CumprodCustomTiling t = *reinterpret_cast<CumprodCustomTiling*>(tiling_ptr);
    const int64_t outer = t.outer;
    const int64_t axis_len = t.axis_len;
    const int64_t inner = t.inner;
    const int64_t elem_size = t.elem_size;
    const int64_t dtype_code = t.dtype_code;

    if (dtype_code == 0) {
        // float16 path using float compute for accumulation
        const uint16_t* x = reinterpret_cast<const uint16_t*>(x_ptr);
        uint16_t* y = reinterpret_cast<uint16_t*>(y_ptr);
        auto fp16_to_fp32 = [](uint16_t h) -> float {
            uint16_t h_exp = (h & 0x7C00u);
            uint16_t h_sig = (h & 0x03FFu);
            uint32_t sign = (h & 0x8000u) ? 0x80000000u : 0u;
            uint32_t exp;
            uint32_t mant;
            if (h_exp == 0) {
                if (h_sig == 0) {
                    exp = 0;
                    mant = 0;
                } else {
                    int e = -1;
                    uint16_t sig = h_sig;
                    while ((sig & 0x0400u) == 0) {
                        sig <<= 1;
                        e--;
                    }
                    sig &= 0x03FFu;
                    exp = static_cast<uint32_t>(127 - 15 + e) << 23;
                    mant = static_cast<uint32_t>(sig) << 13;
                }
            } else if (h_exp == 0x7C00u) {
                exp = 0x7F800000u;
                mant = static_cast<uint32_t>(h_sig) << 13;
            } else {
                int32_t e = (h_exp >> 10) - 15 + 127;
                exp = static_cast<uint32_t>(e) << 23;
                mant = static_cast<uint32_t>(h_sig) << 13;
            }
            uint32_t u = sign | exp | mant;
            float f;
            std::memcpy(&f, &u, sizeof(float));
            return f;
        };
        auto fp32_to_fp16 = [](float f) -> uint16_t {
            uint32_t u;
            std::memcpy(&u, &f, sizeof(uint32_t));
            uint32_t sign = (u >> 16) & 0x8000u;
            int32_t exp = ((u >> 23) & 0xFF) - 127 + 15;
            uint32_t mant = u & 0x7FFFFFu;
            if (exp <= 0) {
                if (exp < -10) return static_cast<uint16_t>(sign);
                mant = (mant | 0x800000u) >> (1 - exp);
                return static_cast<uint16_t>(sign | (mant + 0x1000u) >> 13);
            } else if (exp == 0xFF - 127 + 15) {
                if (mant == 0) {
                    return static_cast<uint16_t>(sign | 0x7C00u);
                } else {
                    return static_cast<uint16_t>(sign | 0x7C00u | (mant >> 13));
                }
            } else if (exp > 30) {
                return static_cast<uint16_t>(sign | 0x7C00u);
            }
            return static_cast<uint16_t>(sign | (static_cast<uint32_t>(exp) << 10) | ((mant + 0x1000u) >> 13));
        };

        for (int64_t o = 0; o < outer; ++o) {
            for (int64_t i = 0; i < inner; ++i) {
                int64_t base = (o * axis_len * inner) + i;
                float acc = 1.0f;
                for (int64_t a = 0; a < axis_len; ++a) {
                    int64_t idx = base + a * inner;
                    float xv = fp16_to_fp32(x[idx]);
                    acc = acc * xv;
                    y[idx] = fp32_to_fp16(acc);
                }
            }
        }
    } else {
        // float32
        const float* x = reinterpret_cast<const float*>(x_ptr);
        float* y = reinterpret_cast<float*>(y_ptr);
        for (int64_t o = 0; o < outer; ++o) {
            for (int64_t i = 0; i < inner; ++i) {
                int64_t base = (o * axis_len * inner) + i;
                float acc = 1.0f;
                for (int64_t a = 0; a < axis_len; ++a) {
                    int64_t idx = base + a * inner;
                    acc = acc * x[idx];
                    y[idx] = acc;
                }
            }
        }
    }
    (void)elem_size;
}
"""

from torch.library import Library

# Define custom op schema
lib = Library("cumprod_custom", "DEF")
lib.define("cumprod_custom(Tensor x, int dim) -> Tensor")

# Meta implementation for shape inference
@lib.impl("cumprod_custom", "Meta")
def cumprod_custom_meta(x: torch.Tensor, dim: int):
    if dim < 0:
        dim = dim + x.dim()
    if dim < 0 or dim >= x.dim():
        raise RuntimeError("Invalid dim for cumprod_custom")
    return x.new_empty(x.shape)

# CPU fallback implementation using torch.cumprod
@lib.impl("cumprod_custom", "CPU")
def cumprod_custom_cpu(x: torch.Tensor, dim: int):
    return torch.cumprod(x, dim)

# Optional fallback for CompositeImplicitAutograd to catch other devices in Python
try:

except Exception:
    pass
