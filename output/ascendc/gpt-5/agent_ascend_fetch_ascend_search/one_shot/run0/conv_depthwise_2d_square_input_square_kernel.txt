{

}

{
    ConvDepthwise2dSquareInputSquareKernelCustomTilingData tiling;

    auto *x_shape = context->GetInputShape(0);
    auto *w_shape = context->GetInputShape(1);

    // Expect x: [N, C, H, W], w: [C, 1, kH, kW]

        return ge::GRAPH_FAILED;

    const uint32_t W = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const uint32_t wC  = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t wC1 = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    const uint32_t kH  = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    const uint32_t kW  = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(3));

    const int32_t *strideH = attrs->GetAttrPointer<int32_t>(0);

    const int32_t *dilH    = attrs->GetAttrPointer<int32_t>(4);
    const int32_t *dilW    = attrs->GetAttrPointer<int32_t>(5);
    const int32_t *groups  = attrs->GetAttrPointer<int32_t>(6);

        return ge::GRAPH_FAILED;

    // Only support dilation 1 and padding 0, stride 1 for this custom kernel
    if (*dilH != 1 || *dilW != 1 || *padH != 0 || *padW != 0 || *strideH != 1 || *strideW != 1) {

    const uint32_t outW = W - kW + 1;

    const uint64_t totalOut = static_cast<uint64_t>(N) * C * outH * outW;
    uint32_t blockDim = CeilDiv(totalOut, static_cast<uint64_t>(16384)); // heuristic
    if (blockDim < 1u)  blockDim = 1u;

    tiling.set_c(C);
    tiling.set_h(H);
    tiling.set_w(W);

    tiling.set_kW(kW);

    tiling.set_dilationW(static_cast<uint32_t>(*dilW));

kernel_src="""

#include "kernel_operator.h"

class KernelConvDepthwise2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t kH, uint32_t kW,
                                uint32_t outH, uint32_t outW)
    {
        n_ = n; c_ = c; h_ = h; w_ = w_in;
        kH_ = kH; kW_ = kW; outH_ = outH; outW_ = outW;

        // Flatten total sizes
        uint64_t xSize = static_cast<uint64_t>(n_) * c_ * h_ * w_;
        uint64_t wSize = static_cast<uint64_t>(c_) * kH_ * kW_;
        uint64_t ySize = static_cast<uint64_t>(n_) * c_ * outH_ * outW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // Initialize small UB buffers
        pipe_.InitBuffer(wQueue_, 1, kH_ * kW_ * sizeof(float));
        pipe_.InitBuffer(patchQueue_, 1, kH_ * kW_ * sizeof(float));
        pipe_.InitBuffer(prodQueue_, 1, kH_ * kW_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint64_t totalOut = static_cast<uint64_t>(n_) * c_ * outH_ * outW_;
        if (totalOut == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t idx = blockIdx; idx < totalOut; idx += blockNum) {
            uint64_t t = idx;
            const uint32_t ow = static_cast<uint32_t>(t % outW_); t /= outW_;
            const uint32_t oh = static_cast<uint32_t>(t % outH_); t /= outH_;
            const uint32_t c  = static_cast<uint32_t>(t % c_);    t /= c_;
            const uint32_t n  = static_cast<uint32_t>(t);

            ComputeOne(n, c, oh, ow);
        }
    }

private:
    __aicore__ inline void LoadWeight(uint32_t c, AscendC::LocalTensor<float>& wLocal)
    {
        // weight layout: [C, 1, kH, kW] contiguous -> per-channel 9 contiguous
        const uint64_t wBase = static_cast<uint64_t>(c) * kH_ * kW_;
        AscendC::DataCopy(wLocal, wGm_[wBase], kH_ * kW_);
    }

    __aicore__ inline void LoadPatch(uint32_t n, uint32_t c, uint32_t oh, uint32_t ow,
                                     AscendC::LocalTensor<float>& patchLocal)
    {
        // NCHW layout, no padding, stride=1
        const uint64_t ncBase = (static_cast<uint64_t>(n) * c_ + c) * h_ * w_;
        // Copy kH rows of kW width from input starting at (oh, ow)
        for (uint32_t kh = 0; kh < kH_; ++kh) {
            const uint64_t rowOff = ncBase + static_cast<uint64_t>(oh + kh) * w_ + ow;
            AscendC::DataCopy(patchLocal[kh * kW_], xGm_[rowOff], kW_);
        }
    }

    __aicore__ inline void ComputeOne(uint32_t n, uint32_t c, uint32_t oh, uint32_t ow)
    {
        // Allocate locals
        AscendC::LocalTensor<float> wLocal    = wQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> patchLocal= patchQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> prodLocal = prodQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> yScalar   = scalarQueue_.AllocTensor<float>();

        // Load weight and patch
        LoadWeight(c, wLocal);
        LoadPatch(n, c, oh, ow, patchLocal);

        // Elementwise product
        AscendC::Mul(prodLocal, patchLocal, wLocal, kH_ * kW_);

        // Reduce sum to scalar using iterative accumulation
        AscendC::DataCopy(yScalar, prodLocal, 1);
        for (uint32_t i = 1; i < kH_ * kW_; ++i) {
            AscendC::Add(yScalar, yScalar, prodLocal[i], 1);
        }

        // Store scalar to GM
        const uint64_t yOff = (((static_cast<uint64_t>(n) * c_ + c) * outH_ + oh) * outW_ + ow);
        AscendC::DataCopy(yGm_[yOff], yScalar, 1);

        // Free locals
        scalarQueue_.FreeTensor(yScalar);
        prodQueue_.FreeTensor(prodLocal);
        patchQueue_.FreeTensor(patchLocal);
        wQueue_.FreeTensor(wLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> patchQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> prodQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t kH_{0}, kW_{0}, outH_{0}, outW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kH, tiling_data.kW,
            tiling_data.outH, tiling_data.outW);
    op.Process();
}
