host_operator_src="""

#include "matmul_scaling_residual_add_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model for post-scale pass:
//   inQueueY: 2 * tile
//   outQueueY: 2 * tile
// Total floats = 4 * tile => bytes = 16 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulScalingResidualAddCustomTilingData tiling;

    // Shapes
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    const gert::Shape* b_shape = context->GetInputShape(2);
    if (x_shape->GetStorageShape().GetDimNum() < 2 ||
        w_shape->GetStorageShape().GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(0)); // weight is [N,K] (out,in)

    // Validate bias
    if (b_shape->GetStorageShape().GetShapeSize() != N) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t totalLength = M * N;

    // Attr: scale
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *pScale = attrs->GetAttrPointer<float>(0);
    const float scale = (pScale == nullptr) ? 0.0f : *pScale;
    const float scaleFactor = 1.0f + scale;

    // tileLen from UB budget: bytes = 16 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 16u;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_scaleFactor(scaleFactor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *x_shape = context->GetInputShape(0); // [M,K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N,K] (out,in)

    if (x_shape->GetOriginShape().GetDimNum() < 2 ||

    const int64_t M = x_shape->GetOriginShape().GetDim(0);

    out.SetDimNum(2);
    out.SetDim(0, M);
    out.SetDim(1, N);

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

constexpr int32_t BUFFER_NUM = 2;

using namespace AscendC;

class KernelMatmulScalingResidualAddCustom {
public:
    __aicore__ inline KernelMatmulScalingResidualAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t totalLen, uint32_t tileLen,
                                float scaleFactor)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        totalLen_ = totalLen;
        tileLen_ = tileLen;
        scaleFactor_ = scaleFactor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_); // stored as [N,K]
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void ComputeMatmul()
    {
        // Define Matmul types: A[M,K], B stored as [N,K] with isTrans=true to represent KxN,
        // C[M,N], Bias[1,N], all in GM, ND format, float32.
        typedef MatmulType<TPosition::GM, CubeFormat::ND, float> aType;
        typedef MatmulType<TPosition::GM, CubeFormat::ND, float, true> bType; // transposed storage [N,K]
        typedef MatmulType<TPosition::GM, CubeFormat::ND, float> cType;
        typedef MatmulType<TPosition::GM, CubeFormat::ND, float> biasType;

        AscendC::Matmul<aType, bType, cType, biasType> mm;

        // Configure shapes and tensors
        // Notes:
        // - B provided as [N, K], bType::isTrans = true so it is treated as [K, N]
        // - Bias is [1, N]
        mm.SetShape(M_, N_, K_);
        mm.SetTensorA(xGm_.GetAddr());
        mm.SetTensorB(wGm_.GetAddr());
        mm.SetBias(bGm_.GetAddr());
        mm.SetTensorC(yGm_.GetAddr());

        // Use default Normal config
        auto cfg = AscendC::GetNormalConfig(M_, N_, K_);
        mm.SetConfig(cfg);

        // Execute matmul: C = A * B + Bias
        mm.Iterate();
        // Results are stored in C (GM position for cType)
    }

    __aicore__ inline void PostScale()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Prefetch first tile
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            // Prefetch next tile
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            ComputeScale(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void ComputeScale(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocalIn = inQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> yLocalOut = outQueueY_.AllocTensor<float>();

        // yOut = yIn * scaleFactor_
        AscendC::Muls(yLocalOut, yLocalIn, scaleFactor_, len);

        outQueueY_.EnQue<float>(yLocalOut);
        inQueueY_.FreeTensor(yLocalIn);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    float scaleFactor_{1.0f};
};

extern "C" __global__ __aicore__ void matmul_scaling_residual_add_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelMatmulScalingResidualAddCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.scaleFactor);

    // GEMM + bias
    op.ComputeMatmul();
    // Post scale: y = y * (1 + scale)
    op.PostScale();
}
