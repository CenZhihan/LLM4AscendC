kernel_src="""
#include "kernel_operator.h"
#include <cmath>

class KernelConv2dSubtractHardSwishMaxPoolMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractHardSwishMaxPoolMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t K, uint32_t P,
                                uint32_t outH, uint32_t outW,
                                float subtractValue, uint32_t hasBias)
    {
        xGm_ = reinterpret_cast<__gm__ float*>(x);
        wGm_ = reinterpret_cast<__gm__ float*>(w);
        bGm_ = reinterpret_cast<__gm__ float*>(b);
        yGm_ = reinterpret_cast<__gm__ float*>(y);

        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; K_ = K; P_ = P;
        outH_ = outH; outW_ = outW;
        subVal_ = subtractValue;
        hasBias_ = hasBias;
    }

    __aicore__ inline float HardSwishScalar(float x) const
    {
        float t = x + 3.0f;
        if (t < 0.0f) t = 0.0f;
        if (t > 6.0f) t = 6.0f;
        return x * (t * (1.0f / 6.0f));
    }

    __aicore__ inline float MishScalar(float x) const
    {
        // mish(x) = x * tanh(softplus(x)), softplus = ln(1 + exp(x))
        float sp = logf(1.0f + expf(x));
        float e2s = expf(2.0f * sp);
        float th = (e2s - 1.0f) / (e2s + 1.0f);
        return x * th;
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || outH_ == 0 || outW_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalTasks = N_ * Cout_;
        for (uint32_t task = blockIdx; task < totalTasks; task += blockNum) {
            const uint32_t n = task / Cout_;
            const uint32_t oc = task % Cout_;
            const float biasVal = (hasBias_ && bGm_ != nullptr) ? bGm_[oc] : 0.0f;

            for (uint32_t oh2 = 0; oh2 < outH_; ++oh2) {
                for (uint32_t ow2 = 0; ow2 < outW_; ++ow2) {
                    // Compute pooled tile base in conv-out domain
                    const uint32_t baseH = oh2 * P_;
                    const uint32_t baseW = ow2 * P_;
                    const uint32_t patchH = P_ + K_ - 1u;
                    const uint32_t patchW = P_ + K_ - 1u;

                    // Accumulate P*P conv outputs within this pool tile
                    // We'll keep them as a small stack array
                    // Note: Assuming reasonable P (e.g., 2/3/4). For larger P, consider tiling further.
                    const uint32_t tileSize = P_ * P_;
                    float accMax = -3.402823466e+38F; // -FLT_MAX

                    // For each location within the P*P pooling window, compute convolution
                    for (uint32_t ph = 0; ph < P_; ++ph) {
                        for (uint32_t pw = 0; pw < P_; ++pw) {
                            float acc = 0.0f;
                            // Convolution sum over Cin and KxK
                            for (uint32_t c = 0; c < Cin_; ++c) {
                                for (uint32_t kh = 0; kh < K_; ++kh) {
                                    const uint32_t ih = baseH + ph + kh;
                                    const uint32_t xRowBase = (((n * Cin_ + c) * H_ + ih) * W_);
                                    const uint32_t wRowBase = (((oc * Cin_ + c) * K_ + kh) * K_);
                                    for (uint32_t kw = 0; kw < K_; ++kw) {
                                        const uint32_t iw = baseW + pw + kw;
                                        const float xv = xGm_[xRowBase + iw];
                                        const float wv = wGm_[wRowBase + kw];
                                        acc += xv * wv;
                                    }
                                }
                            }
                            // Add bias if any
                            acc += biasVal;
                            // Subtract scalar
                            acc -= subVal_;
                            // HardSwish
                            float hs = HardSwishScalar(acc);
                            // Max pool over P*P: update running max
                            if (hs > accMax) accMax = hs;
                        }
                    }
                    // Mish on the pooled value
                    float outv = MishScalar(accMax);

                    // Write to output y[n, oc, oh2, ow2]
                    const uint32_t yOff = (((n * Cout_ + oc) * outH_ + oh2) * outW_ + ow2);
                    yGm_[yOff] = outv;
                }
            }
        }
    }

private:
    __gm__ float* xGm_ {nullptr};
    __gm__ float* wGm_ {nullptr};
    __gm__ float* bGm_ {nullptr};
    __gm__ float* yGm_ {nullptr};

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, K_{0}, P_{0};
    uint32_t outH_{0}, outW_{0};
    float subVal_{0.0f};
    uint32_t hasBias_{0};
};

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractHardSwishMaxPoolMishCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.K, tiling_data.P,
            tiling_data.outH, tiling_data.outW,
            tiling_data.subtractValue, tiling_data.hasBias);
    op.Process();
}
