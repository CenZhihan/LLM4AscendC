host_operator_src="""

#include "matmul_with_transposed_b_custom_tiling.h"
#include "register/op_def_registry.h"

// Matmul tiling headers
#include "tiling/platform/platform_ascendc.h"
#include "lib/matmul_tiling.h"

namespace {

static inline uint64_t GetDimOr(const gert::Shape* s, size_t idx, uint64_t def) {
    if (s == nullptr) return def;
    if (s->GetOriginShape().GetDimNum() <= idx) return def;
    return static_cast<uint64_t>(s->GetOriginShape().GetDim(idx));
}

} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Shapes
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    // A: (M, K)
    // B: (N, K) but we compute A * (B^T) => effective right matrix is (K, N)
    const uint32_t M = static_cast<uint32_t>(GetDimOr(a_shape, 0, 1));
    const uint32_t K = static_cast<uint32_t>(GetDimOr(a_shape, 1, 1));
    const uint32_t N = static_cast<uint32_t>(GetDimOr(b_shape, 0, 1)); // from B(N, K)

    // Platform and Matmul tiling
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling tiling(ascendcPlatform);

    // Data types and formats for A/B/C. Use half for inputs, float for output (accumulate in fp32).
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(AscendC::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);

    // Transpose config: A not transposed, B is transposed at compute time (since input B is (N,K))
    tiling.SetAIsTranspose(false);
    tiling.SetBIsTranspose(true);

    // Shapes
    tiling.SetM(M);
    tiling.SetN(N);
    tiling.SetK(K);

    // Alpha/Beta
    tiling.SetAlpha(1.0f);
    tiling.SetBeta(0.0f);

    // Get tiling data
    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Block dim from matmul tiling if available, otherwise use a safe default.
    uint32_t blockDim = 0;
    // Prefer to query from tilingData if supported; otherwise fallback
    // Note: In typical Matmul tiling, blockDim will be encoded; if not, default to 24
    blockDim = 24u;
    context->SetBlockDim(blockDim);

    // Save tiling
    tilingData.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tilingData.GetDataSize());

    // Workspace: Matmul high-level API manages L2 requirements via tiling. No explicit extra workspace.
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0); // (M, K)
    const gert::Shape *b_shape = context->GetInputShape(1); // (N, K)
    if (a_shape == nullptr || b_shape == nullptr) return GRAPH_FAILED;

    const int64_t M = a_shape->GetOriginShape().GetDim(0);
    const int64_t N = b_shape->GetOriginShape().GetDim(0);

    gert::Shape *c_shape = context->GetOutputShape(0);
    gert::Shape new_shape;
    new_shape.SetDimNum(2);
    new_shape.SetDim(0, M);
    new_shape.SetDim(1, N);
    c_shape->SetOriginShape(new_shape);
    c_shape->SetStorageShape(new_shape);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output promoted to float (fp32)
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

class MatmulWithTransposedBCustom : public OpDef {
public:
    explicit MatmulWithTransposedBCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulWithTransposedBCustom);

} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using namespace AscendC;

extern \"C\" __global__ __aicore__ void matmul_with_transposed_b_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    // Retrieve Matmul tiling
    GET_TILING_DATA(tilingData, tiling);

    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    // Create GM tensors
    A_TYPE aGm;
    B_TYPE bGm;
    C_TYPE cGm;
    BIAS_TYPE biasGm;

    aGm.SetGlobalBuffer((__gm__ half*)A, 0);
    bGm.SetGlobalBuffer((__gm__ half*)B, 0);
    cGm.SetGlobalBuffer((__gm__ float*)C, 0);
    biasGm.SetGlobalBuffer((__gm__ float*)nullptr, 0);

    // Create Matmul object and configure transpose for B
    AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> mm;
    mm.Init(tilingData);

    // Explicitly configure transpose options: A not transposed, B transposed
    mm.SetIsTransposeA(false);
    mm.SetIsTransposeB(true);

    // Alpha/Beta
    mm.SetAlphaBeta(1.0f, 0.0f);

    // Bind GM tensors
    mm.SetTensorA(aGm);
    mm.SetTensorB(bGm);
    mm.SetTensorC(cGm);
    mm.SetBias(biasGm);

    // Run
    mm.Run();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_npu_dtype(const at::Tensor& t, at::ScalarType dt) {
    if (t.dtype() == dt && t.device().type() == at::kPrivateUse1) return t;
    return t.to(t.options().device(torch_npu::current_device()).dtype(dt));
}

at::Tensor matmul_with_transposed_b_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "Inputs must be 2D");
    TORCH_CHECK(A.size(1) == B.size(1), "K dimension must match: A(M,K), B(N,K)");

    // Matmul kernel expects half inputs and returns float output
    at::Tensor A_half = to_npu_dtype(A, at::kHalf);
    at::Tensor B_half = to_npu_dtype(B, at::kHalf);

    // Output shape (M, N) where N = B.size(0)
    auto out_sizes = at::IntArrayRef({A_half.size(0), B_half.size(0)});
    at::Tensor C = at::empty(out_sizes, A_half.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnMatmulWithTransposedBCustom, A_half, B_half, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_with_transposed_b_custom", &matmul_with_transposed_b_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_with_transposed_b_custom", &matmul_with_transposed_b_npu,
          "Matmul with internal B^T (MatmulWithTransposedBCustom on NPU)");
}
