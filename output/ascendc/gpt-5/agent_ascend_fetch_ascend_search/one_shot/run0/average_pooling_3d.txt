#include "average_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static inline int64_t FloorDiv(int64_t a, int64_t b)
{
    return (a >= 0) ? (a / b) : -(( -a + b - 1) / b);
}
}

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling3dCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const std::vector<int64_t> &dims = in_shape->GetOriginShape().GetDims();
    if (dims.size() != 5) {
        return ge::GRAPH_PARAM_INVALID;
    }
    uint32_t N = static_cast<uint32_t>(dims[0]);
    uint32_t C = static_cast<uint32_t>(dims[1]);
    uint32_t Din = static_cast<uint32_t>(dims[2]);
    uint32_t Hin = static_cast<uint32_t>(dims[3]);
    uint32_t Win = static_cast<uint32_t>(dims[4]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernelSizePtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stridePtr     = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *paddingPtr    = attrs->GetAttrPointer<int32_t>(2);
    const bool    *countIncPad   = attrs->GetAttrPointer<bool>(3);

    int32_t k = *kernelSizePtr;
    int32_t s = *stridePtr;
    int32_t p = *paddingPtr;
    if (s <= 0) {
        s = k;  // default stride = kernel_size when not provided
    }

    uint32_t kD = static_cast<uint32_t>(k);
    uint32_t kH = static_cast<uint32_t>(k);
    uint32_t kW = static_cast<uint32_t>(k);
    uint32_t sD = static_cast<uint32_t>(s);
    uint32_t sH = static_cast<uint32_t>(s);
    uint32_t sW = static_cast<uint32_t>(s);
    uint32_t pD = static_cast<uint32_t>(p);
    uint32_t pH = static_cast<uint32_t>(p);
    uint32_t pW = static_cast<uint32_t>(p);

    // Output dims: floor((in + 2p - k)/s) + 1
    uint32_t Dout = static_cast<uint32_t>(FloorDiv(static_cast<int64_t>(Din + 2 * pD - kD), static_cast<int64_t>(sD)) + 1);
    uint32_t Hout = static_cast<uint32_t>(FloorDiv(static_cast<int64_t>(Hin + 2 * pH - kH), static_cast<int64_t>(sH)) + 1);
    uint32_t Wout = static_cast<uint32_t>(FloorDiv(static_cast<int64_t>(Win + 2 * pW - kW), static_cast<int64_t>(sW)) + 1);

    uint32_t totalOutLen = N * C * Dout * Hout * Wout;

    // UB model: we only store y tile and a small scalar buffer
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOutLen < tileLen) {
        const uint32_t aligned = AlignDown(totalOutLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutLen;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_dIn(Din);
    tiling.set_hIn(Hin);
    tiling.set_wIn(Win);
    tiling.set_dOut(Dout);
    tiling.set_hOut(Hout);
    tiling.set_wOut(Wout);
    tiling.set_totalOutLen(totalOutLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_kD(kD);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_sD(sD);
    tiling.set_sH(sH);
    tiling.set_sW(sW);
    tiling.set_pD(pD);
    tiling.set_pH(pH);
    tiling.set_pW(pW);
    tiling.set_countIncludePad((*countIncPad) ? 1u : 0u);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> &dims = x_shape->GetOriginShape().GetDims();
    if (dims.size() != 5) {
        return GRAPH_PARAM_INVALID;
    }
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernelSizePtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stridePtr     = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *paddingPtr    = attrs->GetAttrPointer<int32_t>(2);

    int64_t N = dims[0], C = dims[1], Din = dims[2], Hin = dims[3], Win = dims[4];
    int64_t k = static_cast<int64_t>(*kernelSizePtr);
    int64_t s = static_cast<int64_t>(*stridePtr);
    int64_t p = static_cast<int64_t>(*paddingPtr);
    if (s <= 0) s = k;

    int64_t Dout = (Din + 2 * p - k) / s + 1;
    int64_t Hout = (Hin + 2 * p - k) / s + 1;
    int64_t Wout = (Win + 2 * p - k) / s + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> out_dims{N, C, Dout, Hout, Wout};
    y_shape->SetOriginShape(gert::OriginShape(out_dims));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelAveragePooling3dCustom {
public:
    __aicore__ inline KernelAveragePooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, const optiling::AveragePooling3dCustomTilingData &td)
    {
        n_ = td.n;
        c_ = td.c;
        dIn_ = td.dIn;
        hIn_ = td.hIn;
        wIn_ = td.wIn;
        dOut_ = td.dOut;
        hOut_ = td.hOut;
        wOut_ = td.wOut;

        totalOutLen_ = td.totalOutLen;
        tileLen_ = td.tileLen;

        kD_ = td.kD; kH_ = td.kH; kW_ = td.kW;
        sD_ = td.sD; sH_ = td.sH; sW_ = td.sW;
        pD_ = td.pD; pH_ = td.pH; pW_ = td.pW;
        countIncludePad_ = (td.countIncludePad != 0);

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * c_ * dIn_ * hIn_ * wIn_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, 1024 * sizeof(float)); // small scratch for scalar copies
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutLen_) return;

        uint32_t curLen = totalOutLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        while (true) {
            ComputeAndStore(base, curLen);
            CopyOut(base, curLen);

            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalOutLen_) break;
            base = nextBase;
            curLen = totalOutLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
        }
    }

private:
    __aicore__ inline void ComputeAndStore(uint32_t gmOutOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

        for (uint32_t j = 0; j < len; ++j) {
            uint32_t outIndex = gmOutOffset + j;

            // Map linear index to (n, c, od, oh, ow)
            uint32_t ow = outIndex % wOut_;
            uint32_t t = outIndex / wOut_;
            uint32_t oh = t % hOut_;
            t = t / hOut_;
            uint32_t od = t % dOut_;
            t = t / dOut_;
            uint32_t c = t % c_;
            uint32_t n = t / c_;

            // Compute input region
            int32_t dStart = static_cast<int32_t>(od) * static_cast<int32_t>(sD_) - static_cast<int32_t>(pD_);
            int32_t hStart = static_cast<int32_t>(oh) * static_cast<int32_t>(sH_) - static_cast<int32_t>(pH_);
            int32_t wStart = static_cast<int32_t>(ow) * static_cast<int32_t>(sW_) - static_cast<int32_t>(pW_);

            float sum = 0.0f;
            uint32_t validCount = 0;

            for (uint32_t kd = 0; kd < kD_; ++kd) {
                int32_t id = dStart + static_cast<int32_t>(kd);
                if (id < 0 || id >= static_cast<int32_t>(dIn_)) {
                    if (countIncludePad_) {
                        // include padded zeros in denominator
                    }
                    continue;
                }
                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    int32_t ih = hStart + static_cast<int32_t>(kh);
                    if (ih < 0 || ih >= static_cast<int32_t>(hIn_)) {
                        if (countIncludePad_) {
                        }
                        continue;
                    }
                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        int32_t iw = wStart + static_cast<int32_t>(kw);
                        if (iw < 0 || iw >= static_cast<int32_t>(wIn_)) {
                            if (countIncludePad_) {
                            }
                            continue;
                        }
                        // Compute input linear index
                        uint32_t inIndex = (((n * c_ + c) * dIn_ + static_cast<uint32_t>(id)) * hIn_ + static_cast<uint32_t>(ih)) * wIn_ + static_cast<uint32_t>(iw);

                        // Copy single element into tmpLocal and read back
                        AscendC::DataCopy(tmpLocal, xGm_[inIndex], 1);
                        float v = tmpLocal.GetValue(0);
                        sum += v;
                        validCount += 1;
                    }
                }
            }

            uint32_t denom = countIncludePad_ ? (kD_ * kH_ * kW_) : validCount;
            float avg = (denom > 0) ? (sum / static_cast<float>(denom)) : 0.0f;
            yLocal.SetValue(j, avg);
        }

        tmpQueue_.FreeTensor(tmpLocal);
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, dIn_{0}, hIn_{0}, wIn_{0};
    uint32_t dOut_{0}, hOut_{0}, wOut_{0};
    uint32_t totalOutLen_{0};
    uint32_t tileLen_{0};

    uint32_t kD_{1}, kH_{1}, kW_{1};
    uint32_t sD_{1}, sH_{1}, sW_{1};
    uint32_t pD_{0}, pH_{0}, pW_{0};
    bool countIncludePad_{false};
};

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling3dCustom op;
    op.Init(x, y, tiling_data);
    op.Process();
}
