namespace {
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core conservative
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static size_t ShapeSize(const gert::Shape& s) {
    size_t sz = 1;
    for (size_t i = 0; i < s.GetDimNum(); ++i) sz *= static_cast<size_t>(s.GetDim(i));
    return sz;
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SqueezeNetFireModuleCustomTilingData tiling;

    const gert::Shape xShape = context->GetInputShape(0)->GetOriginShape();
    if (xShape.GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N  = static_cast<uint32_t>(xShape.GetDim(0));
    const uint32_t C  = static_cast<uint32_t>(xShape.GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(xShape.GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(xShape.GetDim(3));

    const gert::Shape wSqShape = context->GetInputShape(1)->GetOriginShape();  // [Csq, Cin, 1, 1]
    const uint32_t Csq = static_cast<uint32_t>(wSqShape.GetDim(0));
    const uint32_t Cin_check = static_cast<uint32_t>(wSqShape.GetDim(1));

    const gert::Shape wE1Shape = context->GetInputShape(3)->GetOriginShape();  // [Ce1, Csq, 1, 1]
    const uint32_t Ce1 = static_cast<uint32_t>(wE1Shape.GetDim(0));
    const uint32_t Csq_check = static_cast<uint32_t>(wE1Shape.GetDim(1));

    const gert::Shape wE3Shape = context->GetInputShape(5)->GetOriginShape();  // [Ce3, Csq, 3, 3]
    const uint32_t Ce3 = static_cast<uint32_t>(wE3Shape.GetDim(0));
    const uint32_t Csq_check2 = static_cast<uint32_t>(wE3Shape.GetDim(1));

    if (Cin_check != C || Csq_check != Csq || Csq_check2 != Csq) {
        return ge::GRAPH_FAILED;
    }

    // Budget model in floats for a tile of size (TH, TW):
    //   xPatch: Cin * (TH+2) * (TW+2)
    //   sExt  : Csq * (TH+2) * (TW+2)
    //   e1    : Ce1 * TH * TW
    //   e3    : Ce3 * TH * TW
    // Choose a conservative candidate that fits within UB_BUDGET_BYTES
    const uint32_t candidates[][2] = { {8, 8}, {8, 4}, {4, 8}, {4, 4}, {2, 8}, {8, 2}, {2, 4}, {4, 2}, {2, 2}, {1, 8}, {8, 1}, {1, 4}, {4,1}, {1,2}, {2,1}, {1,1} };
    uint32_t tileH = 1, tileW = 1;
    for (auto &cand : candidates) {
        const uint32_t TH = cand[0];
        const uint32_t TW = cand[1];
        const uint64_t floats = static_cast<uint64_t>(C) * (TH + 2u) * (TW + 2u)
                              + static_cast<uint64_t>(Csq) * (TH + 2u) * (TW + 2u)
                              + static_cast<uint64_t>(Ce1 + Ce3) * TH * TW;
        const uint64_t bytes = floats * sizeof(float);
        if (bytes <= UB_BUDGET_BYTES) {
            tileH = TH; tileW = TW;
            break;
        }
    }

    if (tileH == 0 || tileW == 0) {
        tileH = 1; tileW = 1;
    }

    const uint32_t tilesH = CeilDiv(H, tileH);
    const uint32_t tilesW = CeilDiv(W, tileW);
    const uint32_t totalTiles = N * tilesH * tilesW;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cIn(C);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_cSqueeze(Csq);
    tiling.set_cE1(Ce1);
    tiling.set_cE3(Ce3);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_tilesH(tilesH);
    tiling.set_tilesW(tilesW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelSqueezeNetFireModuleCustom {
public:
    __aicore__ inline KernelSqueezeNetFireModuleCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wSq, GM_ADDR bSq, GM_ADDR wE1, GM_ADDR bE1,
                                GM_ADDR wE3, GM_ADDR bE3, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t w,
                                uint32_t cSqueeze, uint32_t cE1, uint32_t cE3,
                                uint32_t tileH, uint32_t tileW,
                                uint32_t tilesH, uint32_t tilesW)
    {
        N_ = n; C_ = cIn; H_ = h; W_ = w; Csq_ = cSqueeze; Ce1_ = cE1; Ce3_ = cE3;
        tileH_ = tileH; tileW_ = tileW; tilesH_ = tilesH; tilesW_ = tilesW;

        // Global buffers
        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * C_ * H_ * W_);
        wSqGm_.SetGlobalBuffer((__gm__ float*)wSq, static_cast<uint64_t>(Csq_) * C_);
        bSqGm_.SetGlobalBuffer((__gm__ float*)bSq, Csq_);
        wE1Gm_.SetGlobalBuffer((__gm__ float*)wE1, static_cast<uint64_t>(Ce1_) * Csq_);
        bE1Gm_.SetGlobalBuffer((__gm__ float*)bE1, Ce1_);
        wE3Gm_.SetGlobalBuffer((__gm__ float*)wE3, static_cast<uint64_t>(Ce3_) * Csq_ * 3u * 3u);
        bE3Gm_.SetGlobalBuffer((__gm__ float*)bE3, Ce3_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * (Ce1_ + Ce3_) * H_ * W_);

        // Init weight/bias local buffers
        pipe_.InitBuffer(wSqQ_, BUFFER_NUM, Csq_ * C_ * sizeof(float));
        pipe_.InitBuffer(bSqQ_, BUFFER_NUM, Csq_ * sizeof(float));
        pipe_.InitBuffer(wE1Q_, BUFFER_NUM, Ce1_ * Csq_ * sizeof(float));
        pipe_.InitBuffer(bE1Q_, BUFFER_NUM, Ce1_ * sizeof(float));
        pipe_.InitBuffer(wE3Q_, BUFFER_NUM, Ce3_ * Csq_ * 3u * 3u * sizeof(float));
        pipe_.InitBuffer(bE3Q_, BUFFER_NUM, Ce3_ * sizeof(float));

        wSqLocal_ = wSqQ_.AllocTensor<float>();
        bSqLocal_ = bSqQ_.AllocTensor<float>();
        wE1Local_ = wE1Q_.AllocTensor<float>();
        bE1Local_ = bE1Q_.AllocTensor<float>();
        wE3Local_ = wE3Q_.AllocTensor<float>();
        bE3Local_ = bE3Q_.AllocTensor<float>();

        AscendC::DataCopy(wSqLocal_, wSqGm_, Csq_ * C_);
        AscendC::DataCopy(bSqLocal_, bSqGm_, Csq_);
        AscendC::DataCopy(wE1Local_, wE1Gm_, Ce1_ * Csq_);
        AscendC::DataCopy(bE1Local_, bE1Gm_, Ce1_);
        AscendC::DataCopy(wE3Local_, wE3Gm_, Ce3_ * Csq_ * 9u);
        AscendC::DataCopy(bE3Local_, bE3Gm_, Ce3_);

        // Init scratch queues per-tile
        const uint32_t xPatchElems  = C_   * (tileH_ + 2u) * (tileW_ + 2u);
        const uint32_t sExtElems    = Csq_ * (tileH_ + 2u) * (tileW_ + 2u);
        const uint32_t e1PatchElems = Ce1_ * tileH_ * tileW_;
        const uint32_t e3PatchElems = Ce3_ * tileH_ * tileW_;
        pipe_.InitBuffer(xPatchQ_, BUFFER_NUM, xPatchElems * sizeof(float));
        pipe_.InitBuffer(sExtQ_,   BUFFER_NUM, sExtElems   * sizeof(float));
        pipe_.InitBuffer(e1PatchQ_,BUFFER_NUM, e1PatchElems* sizeof(float));
        pipe_.InitBuffer(e3PatchQ_,BUFFER_NUM, e3PatchElems* sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tilesPerBatch = tilesH_ * tilesW_;
        const uint32_t totalTiles = N_ * tilesPerBatch;

        for (uint32_t tileLinear = blockIdx; tileLinear < totalTiles; tileLinear += blockNum) {
            const uint32_t n = tileLinear / tilesPerBatch;
            const uint32_t rem = tileLinear % tilesPerBatch;
            const uint32_t th = rem / tilesW_;
            const uint32_t tw = rem % tilesW_;
            const uint32_t h0 = th * tileH_;
            const uint32_t w0 = tw * tileW_;

            ComputeTile(n, h0, w0);
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const {
        return (((static_cast<uint64_t>(n) * C_ + c) * H_ + h) * W_ + w);
    }
    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t Cout) const {
        return (((static_cast<uint64_t>(n) * Cout + c) * H_ + h) * W_ + w);
    }
    __aicore__ inline uint64_t OffsetWsq(uint32_t csq, uint32_t c) const {
        return static_cast<uint64_t>(csq) * C_ + c;
    }
    __aicore__ inline uint64_t OffsetWe1(uint32_t co, uint32_t csq) const {
        return static_cast<uint64_t>(co) * Csq_ + csq;
    }
    __aicore__ inline uint64_t OffsetWe3(uint32_t co, uint32_t csq, uint32_t ky, uint32_t kx) const {
        return (((static_cast<uint64_t>(co) * Csq_ + csq) * 3u + ky) * 3u + kx);
    }

    __aicore__ inline void ComputeTile(uint32_t n, uint32_t h0, uint32_t w0)
    {
        AscendC::LocalTensor<float> xPatch = xPatchQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> sExt   = sExtQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> e1Patch= e1PatchQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> e3Patch= e3PatchQ_.AllocTensor<float>();

        const uint32_t extH = tileH_ + 2u;
        const uint32_t extW = tileW_ + 2u;

        // 1) Load x patch for valid rows/cols into UB (extended by halo)
        for (uint32_t c = 0; c < C_; ++c) {
            for (uint32_t rExt = 0; rExt < extH; ++rExt) {
                const int32_t hGlob = static_cast<int32_t>(h0) + static_cast<int32_t>(rExt) - 1;
                if (hGlob < 0 || hGlob >= static_cast<int32_t>(H_)) {
                    continue;
                }
                // iwExt valid range [iwMin, iwMax] inclusive
                const int32_t iwMin = AscendC::Max(0, 1 - static_cast<int32_t>(w0));
                const int32_t iwMax = AscendC::Min(static_cast<int32_t>(extW) - 1, static_cast<int32_t>(W_) - static_cast<int32_t>(w0));
                if (iwMin > iwMax) { continue; }
                const uint32_t len = static_cast<uint32_t>(iwMax - iwMin + 1);
                const uint32_t wGlobStart = static_cast<uint32_t>(static_cast<int32_t>(w0) - 1 + iwMin);

                const uint64_t gmOff = OffsetX(n, c, static_cast<uint32_t>(hGlob), wGlobStart);
                // Local offset: ((c * extH) + rExt) * extW + iwMin
                const uint64_t ubOff = ((static_cast<uint64_t>(c) * extH + rExt) * extW + static_cast<uint32_t>(iwMin));
                AscendC::DataCopy(xPatch[ubOff], xGm_[gmOff], len);
            }
        }

        // 2) Compute sExt (ReLU(conv1x1(x) + b_squeeze)) on extended region; invalid positions -> 0
        for (uint32_t rExt = 0; rExt < extH; ++rExt) {
            const int32_t hGlob = static_cast<int32_t>(h0) + static_cast<int32_t>(rExt) - 1;
            for (uint32_t iwExt = 0; iwExt < extW; ++iwExt) {
                const int32_t wGlob = static_cast<int32_t>(w0) + static_cast<int32_t>(iwExt) - 1;
                const bool valid = (hGlob >= 0 && hGlob < static_cast<int32_t>(H_) &&
                                    wGlob >= 0 && wGlob < static_cast<int32_t>(W_));
                for (uint32_t csq = 0; csq < Csq_; ++csq) {
                    float val = 0.0f;
                    if (valid) {
                        val = bSqLocal_.GetValue(csq);
                        for (uint32_t c = 0; c < C_; ++c) {
                            const uint64_t ubOffX = ((static_cast<uint64_t>(c) * extH + rExt) * extW + iwExt);
                            const float xv = xPatch.GetValue(ubOffX);
                            const float wv = wSqLocal_.GetValue(OffsetWsq(csq, c));
                            val = AscendC::FMA(wv, xv, val); // val += wv * xv
                        }
                        if (val < 0.0f) val = 0.0f;
                    } else {
                        val = 0.0f;
                    }
                    const uint64_t ubOffS = ((static_cast<uint64_t>(csq) * extH + rExt) * extW + iwExt);
                    sExt.SetValue(ubOffS, val);
                }
            }
        }

        // 3) Compute expand1 (1x1 on center region), ReLU
        for (uint32_t co = 0; co < Ce1_; ++co) {
            for (uint32_t ih = 0; ih < tileH_; ++ih) {
                const uint32_t rExt = ih + 1u;
                const uint32_t hGlob = h0 + ih;
                if (hGlob >= H_) { continue; }
                for (uint32_t iw = 0; iw < tileW_; ++iw) {
                    const uint32_t cExt = iw + 1u;
                    const uint32_t wGlob = w0 + iw;
                    if (wGlob >= W_) { continue; }
                    float sum = bE1Local_.GetValue(co);
                    for (uint32_t csq = 0; csq < Csq_; ++csq) {
                        const float sVal = sExt.GetValue(((static_cast<uint64_t>(csq) * extH + rExt) * extW + cExt));
                        const float wv = wE1Local_.GetValue(OffsetWe1(co, csq));
                        sum = AscendC::FMA(wv, sVal, sum);
                    }
                    if (sum < 0.0f) sum = 0.0f;
                    const uint64_t ubOffE1 = ((static_cast<uint64_t>(co) * tileH_ + ih) * tileW_ + iw);
                    e1Patch.SetValue(ubOffE1, sum);
                }
            }
        }

        // 4) Compute expand3 (3x3 with padding on sExt), ReLU
        for (uint32_t co = 0; co < Ce3_; ++co) {
            for (uint32_t ih = 0; ih < tileH_; ++ih) {
                const uint32_t hGlob = h0 + ih;
                if (hGlob >= H_) { continue; }
                for (uint32_t iw = 0; iw < tileW_; ++iw) {
                    const uint32_t wGlob = w0 + iw;
                    if (wGlob >= W_) { continue; }
                    float sum = bE3Local_.GetValue(co);
                    const uint32_t rExtCenter = ih + 1u;
                    const uint32_t cExtCenter = iw + 1u;
                    for (uint32_t csq = 0; csq < Csq_; ++csq) {
                        for (uint32_t ky = 0; ky < 3u; ++ky) {
                            for (uint32_t kx = 0; kx < 3u; ++kx) {
                                const uint32_t rExt = rExtCenter + ky - 1u;
                                const uint32_t cExt = cExtCenter + kx - 1u;
                                const float sVal = sExt.GetValue(((static_cast<uint64_t>(csq) * extH + rExt) * extW + cExt));
                                const float wv = wE3Local_.GetValue(OffsetWe3(co, csq, ky, kx));
                                sum = AscendC::FMA(wv, sVal, sum);
                            }
                        }
                    }
                    if (sum < 0.0f) sum = 0.0f;
                    const uint64_t ubOffE3 = ((static_cast<uint64_t>(co) * tileH_ + ih) * tileW_ + iw);
                    e3Patch.SetValue(ubOffE3, sum);
                }
            }
        }

        // 5) Store outputs to GM: concat [e1, e3] along channel dim
        const uint32_t Cout = Ce1_ + Ce3_;
        // Write e1
        for (uint32_t co = 0; co < Ce1_; ++co) {
            for (uint32_t ih = 0; ih < tileH_; ++ih) {
                const uint32_t hGlob = h0 + ih;
                if (hGlob >= H_) { continue; }
                const int32_t iwMin = 0;
                const int32_t iwMax = AscendC::Min(static_cast<int32_t>(tileW_) - 1, static_cast<int32_t>(W_) - 1 - static_cast<int32_t>(w0));
                if (iwMin > iwMax) { continue; }
                const uint32_t len = static_cast<uint32_t>(iwMax - iwMin + 1);
                const uint32_t wGlobStart = w0 + static_cast<uint32_t>(iwMin);
                const uint64_t gmOff = OffsetY(n, co, hGlob, wGlobStart, Cout);
                const uint64_t ubOff = ((static_cast<uint64_t>(co) * tileH_ + ih) * tileW_ + static_cast<uint32_t>(iwMin));
                AscendC::DataCopy(yGm_[gmOff], e1Patch[ubOff], len);
            }
        }
        // Write e3
        for (uint32_t co = 0; co < Ce3_; ++co) {
            for (uint32_t ih = 0; ih < tileH_; ++ih) {
                const uint32_t hGlob = h0 + ih;
                if (hGlob >= H_) { continue; }
                const int32_t iwMin = 0;
                const int32_t iwMax = AscendC::Min(static_cast<int32_t>(tileW_) - 1, static_cast<int32_t>(W_) - 1 - static_cast<int32_t>(w0));
                if (iwMin > iwMax) { continue; }
                const uint32_t len = static_cast<uint32_t>(iwMax - iwMin + 1);
                const uint32_t wGlobStart = w0 + static_cast<uint32_t>(iwMin);
                const uint32_t coOut = Ce1_ + co;
                const uint64_t gmOff = OffsetY(n, coOut, hGlob, wGlobStart, Cout);
                const uint64_t ubOff = ((static_cast<uint64_t>(co) * tileH_ + ih) * tileW_ + static_cast<uint32_t>(iwMin));
                AscendC::DataCopy(yGm_[gmOff], e3Patch[ubOff], len);
            }
        }
