host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model for this kernel:
// - inQueue: 1 * tile
// - calcQueue: 1 * tile
// - outQueue: 1 * tile
// Total floats ~ 3 * tile => bytes ~ 12 * tile (plus small scalars)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const int64_t dimNum = shape->GetOriginShape().GetDimNum();
    uint64_t totalSize = static_cast<uint64_t>(shape->GetOriginShape().GetShapeSize());

    uint32_t cols = 1;
    if (dimNum >= 1) {
        cols = static_cast<uint32_t>(shape->GetOriginShape().GetDim(dimNum - 1));
    }
    if (cols == 0) cols = 1;
    uint32_t rows = 1;
    if (cols > 0) {
        rows = static_cast<uint32_t>(totalSize / cols);
    }

    // tileCols from UB budget: bytes ≈ 12 * tileCols
    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileCols = maxTile;
    if (cols < tileCols) {
        const uint32_t aligned = AlignDown(cols, ALIGN_ELEMS);
        tileCols = (aligned > 0) ? aligned : cols;
    }
    if (tileCols == 0) tileCols = cols;

    // blockDim based on number of rows
    uint32_t blockDim = rows;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t tileCols)
    {
        rows_ = rows;
        cols_ = cols;
        tileCols_ = tileCols;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(rows_ * cols_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(rows_ * cols_));

        pipe_.InitBuffer(inQueue_, IN_BUF_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUT_BUF_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(calcQueue_, 1, tileCols_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0 || tileCols_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows_; row += (blockNum == 0 ? 1 : blockNum)) {
            float M = -INFINITY;
            float S = 0.0f;

            // Pass 1: compute row max (M) and sum of exp-shifted (S)
            for (uint32_t base = 0; base < cols_; base += tileCols_) {
                uint32_t len = cols_ - base;
                if (len > tileCols_) len = tileCols_;

                CopyIn(row, base, len);
                AscendC::LocalTensor<float> xLocal = inQueue_.DeQue<float>();

                // tileMax = ReduceMax over this tile
                AscendC::LocalTensor<float> tMax = scalarQueue_.AllocTensor<float>();
                uint32_t srcShape[2] = {1u, len};
                AscendC::ReduceMax<float, AscendC::Pattern::Reduce::AR>(tMax, xLocal, srcShape, false);
                float tileMax = tMax.GetValue(0);
                scalarQueue_.FreeTensor(tMax);

                AscendC::LocalTensor<float> tVec = calcQueue_.AllocTensor<float>();

                if (tileMax > M) {
                    // Compute sum(exp(x - tileMax))
                    AscendC::Adds(tVec, xLocal, -tileMax, len);
                    AscendC::Exp(tVec, tVec, len);

                    AscendC::LocalTensor<float> tSum = scalarQueue_.AllocTensor<float>();
                    AscendC::ReduceSum<float, AscendC::Pattern::Reduce::AR>(tSum, tVec, srcShape, false);
                    float tileSum = tSum.GetValue(0);
                    scalarQueue_.FreeTensor(tSum);

                    float coeff = 0.0f;
                    if (M == -INFINITY) {
                        coeff = 0.0f;
                    } else {
                        coeff = expf(M - tileMax);
                    }
                    S = S * coeff + tileSum;
                    M = tileMax;
                } else {
                    // Compute sum(exp(x - M))
                    AscendC::Adds(tVec, xLocal, -M, len);
                    AscendC::Exp(tVec, tVec, len);

                    AscendC::LocalTensor<float> tSum = scalarQueue_.AllocTensor<float>();
                    uint32_t srcShape2[2] = {1u, len};
                    AscendC::ReduceSum<float, AscendC::Pattern::Reduce::AR>(tSum, tVec, srcShape2, false);
                    float tileSum = tSum.GetValue(0);
                    scalarQueue_.FreeTensor(tSum);

                    S += tileSum;
                }

                calcQueue_.FreeTensor(tVec);
                inQueue_.FreeTensor(xLocal);
            }

            // Avoid division by zero in degenerate cases
            if (S == 0.0f) {
                S = 1.0f;
            }
            const float invS = 1.0f / S;

            // Pass 2: compute final outputs: y = exp(x - M) / S
            for (uint32_t base = 0; base < cols_; base += tileCols_) {
                uint32_t len = cols_ - base;
                if (len > tileCols_) len = tileCols_;

                CopyIn(row, base, len);
                AscendC::LocalTensor<float> xLocal = inQueue_.DeQue<float>();

                AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
                // yLocal = exp(xLocal - M) * invS
                AscendC::Adds(yLocal, xLocal, -M, len);
                AscendC::Exp(yLocal, yLocal, len);
                AscendC::Muls(yLocal, yLocal, invS, len);

                outQueue_.EnQue<float>(yLocal);
                inQueue_.FreeTensor(xLocal);

                CopyOut(row, base, len);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row, uint32_t baseCol, uint32_t len)
    {
        const uint32_t gmOffset = row * cols_ + baseCol;
        AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueue_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row, uint32_t baseCol, uint32_t len)
    {
        const uint32_t gmOffset = row * cols_ + baseCol;
        AscendC::LocalTensor<float> yLocal = outQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_{0};
    uint32_t cols_{0};
    uint32_t tileCols_{0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.rows, tiling_data.cols, tiling_data.tileCols);
    op.Process();
}
"""
