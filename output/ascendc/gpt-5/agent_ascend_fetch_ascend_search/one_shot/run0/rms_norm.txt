host_operator_src="""

#include "rms_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// UB budget and alignment considerations:
// We need three tile buffers in UB: in, out, acc => 3 * tileLen * sizeof(float) bytes.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core UB share
static constexpr uint32_t BYTES_PER_ELEM = 4;
static constexpr uint32_t BUFFERS_PER_TILE = 3;          // in, out, acc
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    RmsNormCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    const auto& ori_shape = in_shape->GetOriginShape();
    const int64_t dim_num = ori_shape.GetDimNum();

    uint64_t N = 1, C = 1, S = 1;
    if (dim_num >= 1) {
        N = static_cast<uint64_t>(ori_shape.GetDim(0));
    }
    if (dim_num >= 2) {
        C = static_cast<uint64_t>(ori_shape.GetDim(1));
    }
    if (dim_num >= 3) {
        S = 1;
        for (int64_t i = 2; i < dim_num; ++i) {
            S *= static_cast<uint64_t>(ori_shape.GetDim(i));
        }
    } else {
        S = 1;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps_ptr = attrs->GetAttrPointer<float>(0);
    const float eps = (eps_ptr != nullptr) ? *eps_ptr : 1e-5f;

    uint64_t maxTileElems = UB_BUDGET_BYTES / (BUFFERS_PER_TILE * BYTES_PER_ELEM);
    if (maxTileElems > HARD_CAP_ELEMS) maxTileElems = HARD_CAP_ELEMS;
    uint32_t tileLen = static_cast<uint32_t>(AlignDown(static_cast<uint32_t>(maxTileElems), ALIGN_ELEMS));
    if (tileLen == 0) tileLen = ALIGN_ELEMS;

    if (S < tileLen) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(S), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(S);
    }

    const uint64_t tilesPerN_u64 = (tileLen == 0) ? 0ull : CeilDiv(static_cast<uint32_t>(S), tileLen);
    const uint64_t totalTiles_u64 = N * tilesPerN_u64;

    uint32_t blockDim = static_cast<uint32_t>(totalTiles_u64);
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_C(static_cast<uint32_t>(C));
    tiling.set_S(static_cast<uint32_t>(S));
    tiling.set_tileLen(tileLen);
    tiling.set_tilesPerN(static_cast<uint32_t>(tilesPerN_u64));
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class RmsNormCustom : public OpDef {
public:
    explicit RmsNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("eps").AttrType(OPTIONAL).Float(1e-5);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(RmsNormCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelRmsNormCustom {
public:
    __aicore__ inline KernelRmsNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t S,
                                uint32_t tileLen, uint32_t tilesPerN,
                                float eps)
    {
        N_ = N;
        C_ = C;
        S_ = S;
        tileLen_ = tileLen;
        tilesPerN_ = tilesPerN;
        eps_ = eps;

        const uint64_t totalElems = static_cast<uint64_t>(N_) * static_cast<uint64_t>(C_) * static_cast<uint64_t>(S_);
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElems);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElems);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpAcc_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || S_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalTiles = static_cast<uint64_t>(N_) * static_cast<uint64_t>(tilesPerN_);
        for (uint64_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t nIdx = static_cast<uint32_t>(tileId / tilesPerN_);
            const uint32_t tileIdx = static_cast<uint32_t>(tileId % tilesPerN_);
            const uint32_t tStart = tileIdx * tileLen_;
            uint32_t len = tileLen_;
            if (tStart + len > S_) len = S_ - tStart;

            // acc buffer for sum of squares over channels
            AscendC::LocalTensor<float> acc = tmpAcc_.AllocTensor<float>();
            AscendC::Duplicate(acc, 0.0f, len);

            // Pass 1: accumulate sum of squares across C channels
            for (uint32_t c = 0; c < C_; ++c) {
                const uint64_t gmOffset = Offset(nIdx, c, tStart);
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xComp = inQueueX_.DeQue<float>();
                AscendC::Mul(xComp, xComp, xComp, len);
                AscendC::Add(acc, acc, xComp, len);
                inQueueX_.FreeTensor(xComp);
            }

            // Compute inv_rms = 1 / sqrt(mean + eps) = rsqrt(sum_sq / C + eps)
            if (C_ > 1) {
                const float invC = 1.0f / static_cast<float>(C_);
                AscendC::Muls(acc, acc, invC, len);
            }
            AscendC::Adds(acc, acc, eps_, len);
            AscendC::Rsqrt(acc, acc, len);

            // Pass 2: normalize each channel using inv_rms and write out
            for (uint32_t c = 0; c < C_; ++c) {
                const uint64_t gmOffset = Offset(nIdx, c, tStart);
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xComp = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Mul(yLocal, xComp, acc, len);
                outQueueY_.EnQue(yLocal);
                inQueueX_.FreeTensor(xComp);

                AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[gmOffset], yOut, len);
                outQueueY_.FreeTensor(yOut);
            }

            tmpAcc_.FreeTensor(acc);
        }
    }

private:
    __aicore__ inline uint64_t Offset(uint32_t n, uint32_t c, uint32_t tStart) const
    {
        // Linear offset for (n, c, spatial tStart..tStart+len-1) in NCHW-like layout
        // Offset = ((n * C + c) * S) + tStart
        return (static_cast<uint64_t>(n) * static_cast<uint64_t>(C_) + static_cast<uint64_t>(c)) * static_cast<uint64_t>(S_) + static_cast<uint64_t>(tStart);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpAcc_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0};
    uint32_t C_{0};
    uint32_t S_{0};
    uint32_t tileLen_{0};
    uint32_t tilesPerN_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNormCustom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.S,
            tiling_data.tileLen, tiling_data.tilesPerN,
            tiling_data.eps);
    op.Process();
}
"""
