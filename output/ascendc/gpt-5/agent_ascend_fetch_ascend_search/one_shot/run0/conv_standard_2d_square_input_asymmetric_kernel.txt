static inline uint32_t CeilDiv(uint64_t a, uint64_t b)

//   - input patch: K = C*KH*KW elements
//   - weight tile: tileOc * K elements
//   - temporary/output scalars: negligible
// bytes ~= 4 * (K + tileOc*K)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t SAFETY_BYTES   = 8 * 1024;     // safety headroom

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t C  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t OC = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t IC = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t KH = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t KW = static_cast<uint32_t>(w_shape->GetDim(3));

    if (C != IC) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t OH = (H >= KH) ? (H - KH + 1u) : 0u;
    const uint32_t OW = (W >= KW) ? (W - KW + 1u) : 0u;

    const uint64_t totalPositions = static_cast<uint64_t>(N) * OH * OW;

    // choose tileOc by UB budget
    const uint64_t K = static_cast<uint64_t>(C) * KH * KW;
    uint32_t tileOc = (OC > 0) ? 1u : 0u;
    if (K > 0) {

            tileOc = static_cast<uint32_t>(maxTileOc64);
        }

    if (tileOc > OC) tileOc = OC;

    uint32_t blockDim = (totalPositions == 0) ? 1u : static_cast<uint32_t>(totalPositions);
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;

    tiling.set_c(C);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_oc(OC);
    tiling.set_ic(IC);
    tiling.set_kh(KH);
    tiling.set_kw(KW);
    tiling.set_oh(OH);
    tiling.set_ow(OW);

    tiling.set_blockDim(blockDim);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t PATCH_BUF_NUM = 1;
constexpr int32_t W_TILE_BUF_NUM = 1;

class KernelConvStandard2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t wd,
                                uint32_t oc, uint32_t ic, uint32_t kh, uint32_t kw,
                                uint32_t oh, uint32_t ow, uint32_t tileOc)
    {
        n_ = n; c_ = c; h_ = h; w_ = wd;
        oc_ = oc; ic_ = ic; kh_ = kh; kw_ = kw;
        oh_ = oh; ow_ = ow; tileOc_ = tileOc;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_) * c_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(oc_) * ic_ * kh_ * kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_) * oc_ * oh_ * ow_);

        K_ = c_ * kh_ * kw_;

        // UB buffers:
        // patch: K elements
        // weight tile: tileOc * K elements
        uint32_t patchBytes = K_ * sizeof(float);
        uint32_t wTileBytes = tileOc_ * K_ * sizeof(float);
        pipe_.InitBuffer(patchQ_, PATCH_BUF_NUM, patchBytes);
        pipe_.InitBuffer(wTileQ_, W_TILE_BUF_NUM, wTileBytes);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        if (oh_ == 0 || ow_ == 0 || oc_ == 0 || c_ == 0) return;

        const uint64_t totalPos = static_cast<uint64_t>(n_) * oh_ * ow_;

        // grid-stride loop over output positions
        for (uint64_t pos = blockIdx; pos < totalPos; pos += blockNum) {
            uint32_t tmp = static_cast<uint32_t>(pos);
            const uint32_t owIdx = tmp % ow_;
            tmp /= ow_;
            const uint32_t ohIdx = tmp % oh_;
            const uint32_t nIdx  = tmp / oh_;

            // Build input patch vector of length K into local UB
            AscendC::LocalTensor<float> patch = patchQ_.AllocTensor<float>();
            uint32_t pOffset = 0;
            for (uint32_t icIdx = 0; icIdx < c_; ++icIdx) {
                for (uint32_t khIdx = 0; khIdx < kh_; ++khIdx) {
                    const uint32_t ih = ohIdx + khIdx;
                    const uint32_t iw = owIdx;
                    const uint64_t xBase = (((uint64_t)nIdx * c_ + icIdx) * h_ + ih) * w_ + iw;
                    // Copy KW contiguous elements for this row in the window
                    AscendC::DataCopy(patch[pOffset], xGm_[xBase], kw_);
                    pOffset += kw_;
                }
            }
            // patch filled with K elements
            patchQ_.EnQue(patch);

            // Process OC in tiles
            for (uint32_t ocStart = 0; ocStart < oc_; ocStart += tileOc_) {
                const uint32_t curOc = AscendC::min(tileOc_, oc_ - ocStart);

                // Load weight tile [curOc, K] into UB
                AscendC::LocalTensor<float> wTile = wTileQ_.AllocTensor<float>();
                // For each oc in tile, copy its filter vector of length K as IC x KH rows of KW contiguous segments
                for (uint32_t t = 0; t < curOc; ++t) {
                    const uint32_t ocIdx = ocStart + t;
                    uint32_t wLocalBase = t * K_;
                    for (uint32_t icIdx = 0; icIdx < ic_; ++icIdx) {
                        for (uint32_t khIdx = 0; khIdx < kh_; ++khIdx) {
                            const uint64_t wBase = ((((uint64_t)ocIdx * ic_ + icIdx) * kh_ + khIdx) * kw_);
                            const uint32_t locOff = wLocalBase + icIdx * kh_ * kw_ + khIdx * kw_;
                            AscendC::DataCopy(wTile[locOff], wGm_[wBase], kw_);
                        }
                    }
                }
                wTileQ_.EnQue(wTile);

                // Dequeue local buffers
                AscendC::LocalTensor<float> patchLocal = patchQ_.DeQue<float>();
                AscendC::LocalTensor<float> wLocal     = wTileQ_.DeQue<float>();

                // Compute dot products for each oc in tile
                for (uint32_t t = 0; t < curOc; ++t) {
                    float acc = 0.0f;
                    const uint32_t wOff = t * K_;
                    // naive accumulation across K
                    for (uint32_t k = 0; k < K_; ++k) {
                        // Note: Element-wise accessors are conceptual; replace with appropriate vector reduce in optimized implementation.
                        float xv = patchLocal.GetValue(k);
                        float wv = wLocal.GetValue(wOff + k);
                        acc += xv * wv;
                    }
                    // Write scalar result to GM at y[n, ocStart+t, oh, ow]
                    const uint64_t yOffset = ((((uint64_t)nIdx * oc_) + (ocStart + t)) * oh_ + ohIdx) * ow_ + owIdx;
                    // Write one element using a temporary 1-element local
                    AscendC::LocalTensor<float> oneVal = wTileQ_.AllocTensor<float>();
                    oneVal.SetValue(0, acc);
                    AscendC::DataCopy(yGm_[yOffset], oneVal, 1);
                    wTileQ_.FreeTensor(oneVal);
                }

                // Free local tiles
                wTileQ_.FreeTensor(wLocal);
                patchQ_.FreeTensor(patchLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, PATCH_BUF_NUM> patchQ_;
    AscendC::TQue<AscendC::TPosition::VECIN,    W_TILE_BUF_NUM> wTileQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t oc_{0}, ic_{0}, kh_{0}, kw_{0};
    uint32_t oh_{0}, ow_{0};
    uint32_t tileOc_{1};
    uint32_t K_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.oc, tiling_data.ic, tiling_data.kh, tiling_data.kw,
            tiling_data.oh, tiling_data.ow, tiling_data.tileOc);
    op.Process();
}
