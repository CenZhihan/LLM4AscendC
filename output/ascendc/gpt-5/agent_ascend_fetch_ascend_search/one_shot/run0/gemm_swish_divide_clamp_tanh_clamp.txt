#include "gemm_swish_divide_clamp_tanh_clamp_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : ((a + b - 1u) / b);
}
} // anonymous namespace

namespace optiling {

// Post-processing UB model for this kernel (after GEMM):
//   inQueue: 2 * tile
//   outQueue: 2 * tile
//   tmpSigmoid: 1 * tile
// Total floats ≈ 5 * tile => bytes ≈ 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmSwishDivideClampTanhClampCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    if (xShape == nullptr || wShape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    // x: [M, K], w: [N, K] (PyTorch linear uses w^T in math), b: [N]
    const uint32_t dimsX = static_cast<uint32_t>(xShape->GetOriginShape().GetDimNum());
    const uint32_t dimsW = static_cast<uint32_t>(wShape->GetOriginShape().GetDimNum());
    if (dimsX < 2 || dimsW < 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t M = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(0));

    const uint32_t totalLength = M * N;

    // tileLen from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmSwishDivideClampTanhClampCustom {
public:
    __aicore__ inline KernelGemmSwishDivideClampTanhClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t totalLength, uint32_t tileLength)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        // UB buffers for post-processing
        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) {
            return;
        }
        // 1. GEMM: y = x * w^T + b
        RunGemm_();

        if (totalLen_ == 0 || tileLen_ == 0) return;

        // 2. Fused post-processing on y in GM:
        // y = tanh(clamp((y * sigmoid(y) / 2), -1, 1)) then clamp again to [-1, 1]
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Prefetch first tile
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn_(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn_(nextBase, nextLen);
            }

            ComputeFused_(curLen);
            CopyOut_(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void RunGemm_()
    {
        // Types for Matmul: A=x in GM, B=w in GM (treated as transposed), C=y in GM, Bias in GM.
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false> aType;
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true>  bType;   // w stored as [N,K], use trans
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>        cType;
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>        biasType;

        AscendC::Matmul<aType, bType, cType, biasType> mm;

        // Initialize and bind tensors.
        // The Matmul high-level API manages tiling internally for ND float.
        mm.SetTensorA((__gm__ float*)xGm_.GetGlobalBuffer(), M_, K_);
        mm.SetTensorB((__gm__ float*)wGm_.GetGlobalBuffer(), N_, K_);
        mm.SetBias((__gm__ float*)bGm_.GetGlobalBuffer(), N_);
        mm.SetDst((__gm__ float*)yGm_.GetGlobalBuffer(), M_, N_);
        mm.SetM(M_);
        mm.SetN(N_);
        mm.SetK(K_);
        mm.Compute();
        mm.End();
    }

    __aicore__ inline void CopyIn_(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> local = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(local, yGm_[gmOffset], len);
        inQueue_.EnQue(local);
    }

    __aicore__ inline void ComputeFused_(uint32_t len)
    {
        AscendC::LocalTensor<float> vin = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> vout = outQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

        // tmp = sigmoid(vin)
        AscendC::Sigmoid(tmp, vin, len);
        // vout = vin * tmp
        AscendC::Mul(vout, vin, tmp, len);
        // vout = vout / 2.0f
        AscendC::Muls(vout, vout, 0.5f, len);
        // clamp to [-1, 1]
        AscendC::Maxs(vout, vout, -1.0f, len);
        AscendC::Mins(vout, vout, 1.0f, len);
        // vout = tanh(vout)
        AscendC::Tanh(vout, vout, len);
        // clamp again to [-1, 1]
        AscendC::Maxs(vout, vout, -1.0f, len);
        AscendC::Mins(vout, vout, 1.0f, len);

        tmpQueue_.FreeTensor(tmp);
        outQueue_.EnQue<float>(vout);
        inQueue_.FreeTensor(vin);
    }

    __aicore__ inline void CopyOut_(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> vout = outQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], vout, len);
        outQueue_.FreeTensor(vout);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void gemm_swish_divide_clamp_tanh_clamp_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSwishDivideClampTanhClampCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
