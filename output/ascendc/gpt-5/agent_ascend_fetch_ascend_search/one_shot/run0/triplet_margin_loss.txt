#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return (x + a - 1u) / a * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t NextPow2(uint32_t v) {
    if (v == 0u) return 1u;
    v--;
    v |= v >> 1;
    v |= v >> 2;
    v |= v >> 4;
    v |= v >> 8;
    v |= v >> 16;
    v++;
    return v;
}
} // namespace

namespace optiling {

// UB budget and element alignment assumptions:
// We need buffers for: anchor chunk, positive chunk, negative chunk, work diff chunk.
// We'll also reserve small scalar buffers. Use conservative budget.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // per-core UB budget slice
static constexpr uint32_t BYTES_PER_ELEM = 4;             // float
static constexpr uint32_t NUM_VECS = 4;                   // a, p, n, diff
static constexpr uint32_t PER_ELEM_BYTES = NUM_VECS * BYTES_PER_ELEM; // 16 bytes per element

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;
    const gert::Shape *a_shape = context->GetInputShape(0);
    // Expect 2D: [nRows, nCols]
    uint32_t nRows = 1;
    uint32_t nCols = 1;
    if (a_shape->GetStorageShape().GetDimNum() >= 2) {
        nRows = static_cast<uint32_t>(a_shape->GetStorageShape().GetDim(0));
        nCols = static_cast<uint32_t>(a_shape->GetStorageShape().GetDim(1));
    } else if (a_shape->GetStorageShape().GetDimNum() == 1) {
        nRows = 1;
        nCols = static_cast<uint32_t>(a_shape->GetStorageShape().GetDim(0));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *marginPtr = attrs->GetAttrPointer<float>(0);
    float margin = (marginPtr != nullptr) ? *marginPtr : 1.0f;

    // Last-axis must be aligned to 32B in some high-level interfaces; here we do internal padding logic.
    const uint32_t alignedCols = AlignUp(nCols, 1u); // we handle padding ourselves; keep element alignment at 1
    // Determine chunk capacity within UB budget, rounded up to next power of 2 to ease in-UB reductions
    uint32_t maxCapElems = UB_BUDGET_BYTES / PER_ELEM_BYTES; // elements we can afford for 4 vectors
    if (maxCapElems == 0) maxCapElems = 1;
    // We will process a row in chunks of up to chunkCap elements. We choose chunkCap as a power-of-two not exceeding maxCapElems.
    uint32_t chunkCap = NextPow2(maxCapElems);
    if (chunkCap > maxCapElems) {
        // Ensure chunkCap does not exceed budget
        while (chunkCap > maxCapElems) chunkCap >>= 1;
        if (chunkCap == 0) chunkCap = 1;
    }
    // Also cap chunkCap to at least 8 for better vectorization
    if (chunkCap < 8u) chunkCap = 8u;

    // Use single block to simplify final mean reduction across rows
    uint32_t blockDim = 1u;
    context->SetBlockDim(blockDim);

    tiling.set_nRows(nRows);
    tiling.set_nCols(alignedCols);
    tiling.set_chunkCap(chunkCap);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y,
                                uint32_t nRows, uint32_t nCols, uint32_t chunkCap, float margin)
    {
        nRows_ = nRows;
        nCols_ = nCols;
        chunkCap_ = chunkCap;
        margin_ = margin;

        aGm_.SetGlobalBuffer((__gm__ float*)anchor, static_cast<uint64_t>(nRows_) * nCols_);
        pGm_.SetGlobalBuffer((__gm__ float*)positive, static_cast<uint64_t>(nRows_) * nCols_);
        nGm_.SetGlobalBuffer((__gm__ float*)negative, static_cast<uint64_t>(nRows_) * nCols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        // Queues for chunk buffers and scalar accumulators
        pipe_.InitBuffer(aQ_, BUFFER_NUM, chunkCap_ * sizeof(float));
        pipe_.InitBuffer(pQ_, BUFFER_NUM, chunkCap_ * sizeof(float));
        pipe_.InitBuffer(nQ_, BUFFER_NUM, chunkCap_ * sizeof(float));
        pipe_.InitBuffer(diffQ_, BUFFER_NUM, chunkCap_ * sizeof(float));

        // A small queue for scalar locals (1-element tensors)
        pipe_.InitBuffer(scalarQ_, 4, 8 * sizeof(float)); // enough space for few scalars
    }

    __aicore__ inline void Process()
    {
        if (nRows_ == 0 || nCols_ == 0) {
            // Write zero result
            AscendC::LocalTensor<float> yLocal = scalarQ_.AllocTensor<float>();
            AscendC::Dup(yLocal, 0.0f, 1);
            AscendC::DataCopy(yGm_[0], yLocal, 1);
            scalarQ_.FreeTensor(yLocal);
            return;
        }

        // loss accumulator across all rows
        AscendC::LocalTensor<float> lossAcc = scalarQ_.AllocTensor<float>();
        AscendC::Dup(lossAcc, 0.0f, 1);

        // constant margin local
        AscendC::LocalTensor<float> marginLocal = scalarQ_.AllocTensor<float>();
        AscendC::Dup(marginLocal, margin_, 1);

        for (uint32_t r = 0; r < nRows_; ++r) {
            // accumulators for this row
            AscendC::LocalTensor<float> accAp = scalarQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> accAn = scalarQ_.AllocTensor<float>();
            AscendC::Dup(accAp, 0.0f, 1);
            AscendC::Dup(accAn, 0.0f, 1);

            uint64_t rowBase = static_cast<uint64_t>(r) * nCols_;

            for (uint32_t c = 0; c < nCols_; c += chunkCap_) {
                uint32_t remain = nCols_ - c;
                uint32_t len0 = (remain < chunkCap_) ? remain : chunkCap_;
                // Compute next power of two length for reduction
                uint32_t curLen = 1u;
                while (curLen < len0) curLen <<= 1u;

                AscendC::LocalTensor<float> aLocal = aQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> pLocal = pQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> nLocal = nQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> diffLocal = diffQ_.AllocTensor<float>();

                // Load chunk
                AscendC::DataCopy(aLocal, aGm_[rowBase + c], len0);
                AscendC::DataCopy(pLocal, pGm_[rowBase + c], len0);
                AscendC::DataCopy(nLocal, nGm_[rowBase + c], len0);

                // diffAP = a - p; square into diffLocal
                AscendC::Sub(diffLocal, aLocal, pLocal, len0);
                AscendC::Mul(diffLocal, diffLocal, diffLocal, len0);
                // zero padding for reduction tail
                if (curLen > len0) {
                    AscendC::LocalTensor<float> tail = diffLocal[len0];
                    AscendC::Dup(tail, 0.0f, curLen - len0);
                }
                // reduce sum to scalar in-place via halving
                uint32_t runLen = curLen;
                while (runLen > 1u) {
                    uint32_t half = runLen >> 1u;
                    AscendC::LocalTensor<float> secondHalf = diffLocal[half];
                    AscendC::Add(diffLocal, diffLocal, secondHalf, half);
                    runLen = half;
                }
                // accAp += diffLocal[0]
                AscendC::Add(accAp, accAp, diffLocal, 1);

                // diffAN = a - n; reuse buffers
                AscendC::Sub(diffLocal, aLocal, nLocal, len0);
                AscendC::Mul(diffLocal, diffLocal, diffLocal, len0);
                if (curLen > len0) {
                    AscendC::LocalTensor<float> tail2 = diffLocal[len0];
                    AscendC::Dup(tail2, 0.0f, curLen - len0);
                }
                runLen = curLen;
                while (runLen > 1u) {
                    uint32_t half = runLen >> 1u;
                    AscendC::LocalTensor<float> secondHalf2 = diffLocal[half];
                    AscendC::Add(diffLocal, diffLocal, secondHalf2, half);
                    runLen = half;
                }
                // accAn += diffLocal[0]
                AscendC::Add(accAn, accAn, diffLocal, 1);

                diffQ_.FreeTensor(diffLocal);
                nQ_.FreeTensor(nLocal);
                pQ_.FreeTensor(pLocal);
                aQ_.FreeTensor(aLocal);
            }

            // distAp = sqrt(accAp); distAn = sqrt(accAn)
            AscendC::Sqrt(accAp, accAp, 1);
            AscendC::Sqrt(accAn, accAn, 1);

            // hinge = relu(distAp - distAn + margin)
            AscendC::LocalTensor<float> tmp = scalarQ_.AllocTensor<float>();
            AscendC::Sub(tmp, accAp, accAn, 1);
            AscendC::Add(tmp, tmp, marginLocal, 1);
            AscendC::Relu(tmp, tmp, 1);

            // lossAcc += hinge
            AscendC::Add(lossAcc, lossAcc, tmp, 1);

            scalarQ_.FreeTensor(tmp);
            scalarQ_.FreeTensor(accAn);
            scalarQ_.FreeTensor(accAp);
        }

        // mean over rows
        float invRows = 1.0f / static_cast<float>(nRows_);
        AscendC::Muls(lossAcc, lossAcc, invRows, 1);

        // write out scalar
        AscendC::LocalTensor<float> yLocal = scalarQ_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, lossAcc, 1);
        AscendC::DataCopy(yGm_[0], yLocal, 1);

        scalarQ_.FreeTensor(yLocal);
        scalarQ_.FreeTensor(marginLocal);
        scalarQ_.FreeTensor(lossAcc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> pQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> nQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> diffQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> scalarQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> pGm_;
    AscendC::GlobalTensor<float> nGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t nRows_{0};
    uint32_t nCols_{0};
    uint32_t chunkCap_{0};
    float margin_{1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y,
            tiling_data.nRows, tiling_data.nCols,
            tiling_data.chunkCap, tiling_data.margin);
    op.Process();
}
