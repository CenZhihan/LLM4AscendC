kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Od, uint32_t Oh, uint32_t Ow,
                                uint32_t tileC, uint32_t tileOw)
    {
        n_ = N; cin_ = Cin; cout_ = Cout;
        d_ = D; h_ = H; w_ = W;
        kd_ = Kd; kh_ = Kh; kw_ = Kw;
        od_ = Od; oh_ = Oh; ow_ = Ow;
        tileC_ = (tileC == 0) ? 1u : tileC;
        tileOw_ = (tileOw == 0) ? 1u : tileOw;

        // strides for input x: N,C,D,H,W
        x_stride_w_ = 1u;
        x_stride_h_ = w_;
        x_stride_d_ = h_ * x_stride_h_;
        x_stride_c_ = d_ * x_stride_d_;
        x_stride_n_ = cin_ * x_stride_c_;

        // strides for output y: N,Cout,Od,Oh,Ow
        y_stride_w_ = 1u;
        y_stride_h_ = ow_;
        y_stride_d_ = oh_ * y_stride_h_;
        y_stride_c_ = od_ * y_stride_d_;
        y_stride_n_ = cout_ * y_stride_c_;

        kvol_ = cin_ * kd_ * kh_ * kw_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * cin_ * d_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cout_) * kvol_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * cout_ * od_ * oh_ * ow_);

        // UB buffers
        pipe_.InitBuffer(wTileQ_, BUFFER_NUM, tileC_ * kvol_ * sizeof(float));
        pipe_.InitBuffer(yTileQ_, BUFFER_NUM, tileC_ * tileOw_ * sizeof(float));
        pipe_.InitBuffer(xVecQ_, BUFFER_NUM, tileOw_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQ_, BUFFER_NUM, tileOw_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (cout_ == 0 || n_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t coPerBlock = (cout_ + blockNum - 1u) / blockNum;
        const uint32_t coStart = blockIdx * coPerBlock;
        uint32_t coEnd = coStart + coPerBlock;
        if (coStart >= cout_) return;
        if (coEnd > cout_) coEnd = cout_;

        for (uint32_t coBase = coStart; coBase < coEnd; coBase += tileC_) {
            const uint32_t curTileC = AscendC::min((uint32_t)tileC_, coEnd - coBase);

            // Load weight tile [curTileC, Cin, Kd, Kh, Kw] contiguous
            AscendC::LocalTensor<float> wLocal = wTileQ_.AllocTensor<float>();
            const uint64_t wOffset = static_cast<uint64_t>(coBase) * kvol_;
            AscendC::DataCopy(wLocal, wGm_[wOffset], curTileC * kvol_);

            // Allocate reusable vectors
            AscendC::LocalTensor<float> yLocal = yTileQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> xVec = xVecQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpVec = tmpVecQ_.AllocTensor<float>();

            // Iterate outer loops: n, od, oh, owBase (tiles)
            for (uint32_t n = 0; n < n_; ++n) {
                for (uint32_t od = 0; od < od_; ++od) {
                    for (uint32_t oh = 0; oh < oh_; ++oh) {
                        for (uint32_t owBase = 0; owBase < ow_; owBase += tileOw_) {
                            const uint32_t currLen = AscendC::min((uint32_t)tileOw_, ow_ - owBase);

                            // Zero initialize yLocal for all co in current tile
                            for (uint32_t cti = 0; cti < curTileC; ++cti) {
                                AscendC::LocalTensor<float> yVec = yLocal[cti * tileOw_];
                                AscendC::Duplicate(yVec, 0.0f, currLen);
                            }

                            // Accumulate over input channels and kernel window
                            for (uint32_t ci = 0; ci < cin_; ++ci) {
                                for (uint32_t kd = 0; kd < kd_; ++kd) {
                                    const uint32_t in_d = od + kd; // stride=1, pad=0
                                    for (uint32_t kh = 0; kh < kh_; ++kh) {
                                        const uint32_t in_h = oh + kh;
                                        for (uint32_t kw = 0; kw < kw_; ++kw) {
                                            const uint32_t in_w = owBase + kw;

                                            // GM offset for input vector of length currLen
                                            const uint64_t xOffset =
                                                static_cast<uint64_t>(n) * x_stride_n_ +
                                                static_cast<uint64_t>(ci) * x_stride_c_ +
                                                static_cast<uint64_t>(in_d) * x_stride_d_ +
                                                static_cast<uint64_t>(in_h) * x_stride_h_ +
                                                static_cast<uint64_t>(in_w) * x_stride_w_;

                                            AscendC::DataCopy(xVec, xGm_[xOffset], currLen);

                                            // For each output channel in tile, y += xVec * w_scalar
                                            const uint32_t wBaseIdx = ((ci * kd_ + kd) * kh_ + kh) * kw_ + kw; // index within kvol for fixed ci,kd,kh,kw
                                            for (uint32_t cti = 0; cti < curTileC; ++cti) {
                                                const uint32_t wIdx = cti * kvol_ + wBaseIdx;
                                                // load scalar weight from UB
                                                float wVal = wLocal.GetValue(wIdx);
                                                AscendC::LocalTensor<float> yVec = yLocal[cti * tileOw_];
                                                AscendC::Muls(tmpVec, xVec, wVal, currLen);
                                                AscendC::Add(yVec, yVec, tmpVec, currLen);
                                            }
                                        } // kw
                                    } // kh
                                } // kd
                            } // ci

                            // Write yLocal to GM for each output channel in tile
                            for (uint32_t cti = 0; cti < curTileC; ++cti) {
                                const uint32_t co = coBase + cti;
                                const uint64_t yBase =
                                  static_cast<uint64_t>(n) * y_stride_n_ +
                                  static_cast<uint64_t>(co) * y_stride_c_ +
                                  static_cast<uint64_t>(od) * y_stride_d_ +
                                  static_cast<uint64_t>(oh) * y_stride_h_ +
                                  static_cast<uint64_t>(owBase) * y_stride_w_;

                                AscendC::LocalTensor<float> yVec = yLocal[cti * tileOw_];
                                AscendC::DataCopy(yGm_[yBase], yVec, currLen);
                            }
                        } // owBase
                    } // oh
                } // od
            } // n

            // Free locals
            yTileQ_.FreeTensor(yLocal);
            xVecQ_.FreeTensor(xVec);
            tmpVecQ_.FreeTensor(tmpVec);
            wTileQ_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> wTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> xVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpVecQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    // shapes
    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t d_{0}, h_{0}, w_{0};
    uint32_t kd_{0}, kh_{0}, kw_{0};
    uint32_t od_{0}, oh_{0}, ow_{0};
    uint32_t tileC_{0}, tileOw_{0};

    // strides
    uint32_t x_stride_w_{0}, x_stride_h_{0}, x_stride_d_{0}, x_stride_c_{0}, x_stride_n_{0};
    uint32_t y_stride_w_{0}, y_stride_h_{0}, y_stride_d_{0}, y_stride_c_{0}, y_stride_n_{0};

    uint32_t kvol_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.od, tiling_data.oh, tiling_data.ow,
            tiling_data.tileC, tiling_data.tileOw);
    op.Process();
}
