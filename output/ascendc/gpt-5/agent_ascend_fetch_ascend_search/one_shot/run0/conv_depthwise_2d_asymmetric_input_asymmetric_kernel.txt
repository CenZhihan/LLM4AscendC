    const uint32_t dw = static_cast<uint32_t>(*dil_w);

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{

    const gert::Shape *w_shape = context->GetInputShape(1);

    // attrs

    const int64_t stride_w = context->GetAttrs()->GetAttrValue<int64_t>(1, 1);
    const int64_t pad_h    = context->GetAttrs()->GetAttrValue<int64_t>(2, 0);
    const int64_t pad_w    = context->GetAttrs()->GetAttrValue<int64_t>(3, 0);
    const int64_t dil_h    = context->GetAttrs()->GetAttrValue<int64_t>(4, 1);
    const int64_t dil_w    = context->GetAttrs()->GetAttrValue<int64_t>(5, 1);

    const int64_t n  = x_shape->GetDim(0);
    const int64_t c  = x_shape->GetDim(1);
    const int64_t h  = x_shape->GetDim(2);
    const int64_t w  = x_shape->GetDim(3);

    const int64_t kh = w_shape->GetDim(2);
    const int64_t kw = w_shape->GetDim(3);

    const int64_t hout = (h + 2 * pad_h - dil_h * (kh - 1) - 1) / stride_h + 1;
    const int64_t wout = (w + 2 * pad_w - dil_w * (kw - 1) - 1) / stride_w + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> out_dims = {n, c, std::max<int64_t>(0, hout), std::max<int64_t>(0, wout)};
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelConvDepthwise2dAsymInAsymKerCustom {
public:
    __aicore__ inline KernelConvDepthwise2dAsymInAsymKerCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t in_w,
                                uint32_t kh, uint32_t kw,
                                uint32_t strideH, uint32_t strideW,
                                int32_t padH, int32_t padW,
                                uint32_t dilH, uint32_t dilW,
                                uint32_t hout, uint32_t wout, uint32_t tileW)
    {
        n_ = n;
        c_ = c;
        h_ = h;
        w_ = in_w;
        kh_ = kh;
        kw_ = kw;
        sh_ = strideH;
        sw_ = strideW;
        ph_ = padH;
        pw_ = padW;
        dh_ = dilH;
        dw_ = dilW;
        hout_ = hout;
        wout_ = wout;
        tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * c_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)c_ * kh_ * kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * c_ * hout_ * wout_);

        // Two VECCALC buffers for acc and temp input
        const uint32_t elemSize = tileW_ > 0 ? tileW_ : 1;
        pipe_.InitBuffer(vecQueue_, 2, elemSize * sizeof(float));
        const uint32_t kSize = kh_ * kw_;
        pipe_.InitBuffer(wQueue_, 1, (kSize > 0 ? kSize : 1) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || h_ == 0 || w_ == 0 || hout_ == 0 || wout_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t tasks = n_ * c_;

        for (uint32_t t = blockIdx; t < tasks; t += blockNum) {
            const uint32_t nIdx = t / c_;
            const uint32_t cIdx = t % c_;

            // Load this channel's kernel weights into UB
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            const uint64_t wOff = static_cast<uint64_t>(cIdx) * kh_ * kw_;
            AscendC::DataCopy(wLocal, wGm_[wOff], kh_ * kw_);

            for (uint32_t oh = 0; oh < hout_; ++oh) {
                for (uint32_t ow0 = 0; ow0 < wout_; ow0 += tileW_) {
                    const uint32_t curLen = AscendC::min(wout_ - ow0, tileW_);

                    AscendC::LocalTensor<float> acc = vecQueue_.AllocTensor<float>();
                    AscendC::Dup(acc, 0.0f, curLen);

                    // Iterate kernel spatial
                    for (uint32_t khIdx = 0; khIdx < kh_; ++khIdx) {
                        const int32_t hi = static_cast<int32_t>(oh) * static_cast<int32_t>(sh_) - ph_ + static_cast<int32_t>(khIdx) * static_cast<int32_t>(dh_);
                        if (hi < 0 || hi >= static_cast<int32_t>(h_)) {
                            continue;
                        }
                        for (uint32_t kwIdx = 0; kwIdx < kw_; ++kwIdx) {
                            // weight scalar from local buffer
                            const uint32_t kIndex = khIdx * kw_ + kwIdx;
                            float wv = 0.0f;
                            // Read weight value from local tensor (assumes API supports scalar read)
                            wv = wLocal.GetValue(kIndex);

                            AscendC::LocalTensor<float> tmp = vecQueue_.AllocTensor<float>();
                            AscendC::Dup(tmp, 0.0f, curLen);

                            const int32_t wBase = static_cast<int32_t>(ow0) * static_cast<int32_t>(sw_) - pw_ + static_cast<int32_t>(kwIdx) * static_cast<int32_t>(dw_);

                            if (sw_ == 1) {
                                int32_t left = 0;
                                int32_t right = static_cast<int32_t>(curLen);
                                if (wBase < 0) {
                                    left = -wBase;
                                }
                                if (wBase + right > static_cast<int32_t>(w_)) {
                                    right = static_cast<int32_t>(w_) - wBase;
                                }
                                const int32_t valid = right - left;
                                if (valid > 0) {
                                    const uint64_t inOff = (((static_cast<uint64_t>(nIdx) * c_ + cIdx) * h_ + static_cast<uint32_t>(hi)) * w_) + static_cast<uint32_t>(wBase + left);
                                    AscendC::DataCopy(tmp[left], xGm_[inOff], valid);
                                }
                            } else {
                                for (uint32_t i = 0; i < curLen; ++i) {
                                    const int32_t wi = wBase + static_cast<int32_t>(i) * static_cast<int32_t>(sw_);
                                    if (wi >= 0 && wi < static_cast<int32_t>(w_)) {
                                        const uint64_t inOff = (((static_cast<uint64_t>(nIdx) * c_ + cIdx) * h_ + static_cast<uint32_t>(hi)) * w_) + static_cast<uint32_t>(wi);
                                        AscendC::DataCopy(tmp[i], xGm_[inOff], 1);
                                    }
                                }
                            }

                            // acc += tmp * wv
                            AscendC::Muls(tmp, tmp, wv, curLen);
                            AscendC::Add(acc, acc, tmp, curLen);

                            vecQueue_.FreeTensor(tmp);
                        }
                    }

                    // Write back to GM
                    const uint64_t outOff = (((static_cast<uint64_t>(nIdx) * c_ + cIdx) * hout_ + oh) * wout_) + ow0;
                    AscendC::DataCopy(yGm_[outOff], acc, curLen);
                    vecQueue_.FreeTensor(acc);
                }
            }

            wQueue_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> vecQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t kh_{0}, kw_{0};
    uint32_t sh_{1}, sw_{1};
    int32_t ph_{0}, pw_{0};
    uint32_t dh_{1}, dw_{1};
    uint32_t hout_{0}, wout_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymInAsymKerCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kh, tiling_data.kw,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilH, tiling_data.dilW,
            tiling_data.hout, tiling_data.wout, tiling_data.tileW);
    op.Process();
}
