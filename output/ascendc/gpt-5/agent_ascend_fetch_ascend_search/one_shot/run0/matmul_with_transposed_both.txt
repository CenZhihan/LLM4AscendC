host_operator_src="""

#include "matmul_with_transposed_both_custom_tiling.h"
#include "register/op_def_registry.h"

// Matmul high-level API headers
#include "lib/matmul_intf.h"
#include "lib/matmul_tiling_intf.h"
#include "lib/tiling_platform.h"

namespace {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Input shapes:
    //   A: (K, M)    (we will compute A^T in kernel via transpose flags)
    //   B: (N, K)    (we will compute B^T in kernel via transpose flags)
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);
    if (aShape == nullptr || bShape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t K = static_cast<uint32_t>(aShape->GetOriginShape().GetDim(0));
    const uint32_t M = static_cast<uint32_t>(aShape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(bShape->GetOriginShape().GetDim(0));

    // Build Matmul API tiling with transpose for both inputs.
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling cubeTiling(ascendcPlatform);

    cubeTiling.SetAType(AscendC::TPosition::GM, CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    cubeTiling.SetBType(AscendC::TPosition::GM, CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    cubeTiling.SetCType(AscendC::TPosition::GM, CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);

    // Logical matmul shape after transpose: (M, K) x (K, N) -> (M, N)
    cubeTiling.SetShape(M, N, K);
    cubeTiling.SetOrgShape(M, N, K);

    // Indicate that both A and B are transposed relative to provided buffers
    cubeTiling.EnableTransposeA(true);
    cubeTiling.EnableTransposeB(true);

    // Let API use full buffer space available on the target
    cubeTiling.SetBufferSpace(-1, -1, -1);

    // Get tiling
    MatmulCustomTilingData mmTiling;
    if (cubeTiling.GetTiling(mmTiling.cubeTilingData) == -1) {
        return ge::GRAPH_FAILED;
    }

    // BlockDim recommendation: derive from tiles if possible; fallback to conservative default
    uint32_t blockDim = 24;
    // Save tiling to raw buffer
    mmTiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                          context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(mmTiling.GetDataSize());

    context->SetBlockDim(blockDim);

    // No workspace
    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // Input A: (K, M), B: (N, K), Output: (M, N)
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);
    if (aShape == nullptr || bShape == nullptr) {
        return GRAPH_FAILED;
    }
    const int64_t K = aShape->GetOriginShape().GetDim(0);
    const int64_t M = aShape->GetOriginShape().GetDim(1);
    const int64_t N = bShape->GetOriginShape().GetDim(0);

    gert::Shape *yShape = context->GetOutputShape(0);
    std::vector<int64_t> dims = {M, N};
    yShape->SetDimNum(dims.size());
    for (size_t i = 0; i < dims.size(); ++i) {
        yShape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType adt = context->GetInputDataType(0);
    context->SetOutputDataType(0, adt);
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

kernel_src="""
#define ASCENDC_CUBE_ONLY
#include "lib/matmul_intf.h"
#include "kernel_operator.h"

using AscendC::TPosition;
using AscendC::MatmulType;

typedef MatmulType<TPosition::GM, CubeFormat::ND, float> aType;
typedef MatmulType<TPosition::GM, CubeFormat::ND, float> bType;
typedef MatmulType<TPosition::GM, CubeFormat::ND, float> cType;

class KernelMatmulWithTransposedBothCustom {
public:
    __aicore__ inline KernelMatmulWithTransposedBothCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR Y, GM_ADDR tiling)
    {
        // Bind GM buffers
        a_.SetGlobalBuffer((__gm__ float*)A);
        b_.SetGlobalBuffer((__gm__ float*)B);
        c_.SetGlobalBuffer((__gm__ float*)Y);

        // Load Matmul tiling from host
        GET_TILING_DATA(mmTiling_, tiling);
        // Configure transpose for both inputs (A^T @ B^T)
        mmObj_.SetTransposeA(true);
        mmObj_.SetTransposeB(true);
        mmObj_.SetTilingData(mmTiling_);
    }

    __aicore__ inline void Process()
    {
        mmObj_.Compute(a_, b_, c_);
    }

private:
    AscendC::Matmul<aType, bType, cType> mmObj_;
    aType a_;
    bType b_;
    cType c_;
    MatmulCustomTilingData mmTiling_;
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR A, GM_ADDR B, GM_ADDR Y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelMatmulWithTransposedBothCustom op;
    op.Init(A, B, Y, tiling);
    op.Process();
}

#include <torch/library.h>

at::Tensor matmul_with_transposed_both_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Expected inputs:
    //   A: (K, M)
    //   B: (N, K)
    // Output:
    //   Y: (M, N) equal to (A^T) @ (B^T)
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, \"Inputs must be 2-D matrices\");
    TORCH_CHECK(A.size(0) == B.size(1), \"Inner dimensions must match: A[0] vs B[1]\");
    auto opts = A.options();
    at::Tensor Y = at::empty({A.size(1), B.size(0)}, opts);
    EXEC_NPU_CMD(aclnnMatmulWithTransposedBothCustom, A, B, Y);
    return Y;
}

    m.impl(\"matmul_with_transposed_both_custom\", &matmul_with_transposed_both_impl_npu);
