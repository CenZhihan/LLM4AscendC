host_operator_src="""
#include "cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    if (a == 0) return x;
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous

namespace optiling {

// UB usage model (per core, per tile):
//   inQueueX: 1 * tile
//   outQueueY: 1 * tile
// Temporary for in-UB prefix-sum: reuse outQueue
// Total floats ~= 2 * tile => bytes ~= 8 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();
    const int64_t rank = origin.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimPtr != nullptr) ? *dimPtr : static_cast<int32_t>(rank - 1);

    // Normalize negative dims
    if (dim < 0) dim += static_cast<int32_t>(rank);

    // Compute segments along dim (assume row-major contiguous)
    uint64_t numSegments64 = 1;
    for (int64_t i = 0; i < dim; ++i) {
        numSegments64 *= static_cast<uint64_t>(origin.GetDim(i));
    }
    uint64_t segmentLen64 = (dim >= 0 && dim < rank) ? static_cast<uint64_t>(origin.GetDim(dim)) : 1ULL;
    uint64_t totalLen64 = numSegments64 * segmentLen64;

    // Choose tile length bounded by UB and segmentLen
    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t segmentLen = static_cast<uint32_t>(segmentLen64);
    uint32_t numSegments = static_cast<uint32_t>(numSegments64);
    uint32_t totalLength = static_cast<uint32_t>(totalLen64);

    if (segmentLen == 0 || numSegments == 0 || totalLength == 0) {
        segmentLen = 0;
        numSegments = 0;
        totalLength = 0;
    }

    uint32_t tileLen = (segmentLen < maxTile) ? segmentLen : maxTile;
    if (tileLen == 0 && segmentLen > 0) tileLen = segmentLen;

    // Set blockDim based on number of segments
    uint32_t blockDim = numSegments;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_segmentLen(segmentLen);
    tiling.set_numSegments(numSegments);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumsumCustom : public OpDef {
public:
    explicit CumsumCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumsumCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 1;

class KernelCumsumCustom {
public:
    __aicore__ inline KernelCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t segmentLen,
                                uint32_t numSegments, uint32_t tileLength,
                                int32_t dim)
    {
        totalLen_ = totalLength;
        segLen_ = segmentLen;
        numSegs_ = numSegments;
        tileLen_ = tileLength;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQue_, IN_BUF_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQue_, OUT_BUF_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || segLen_ == 0 || numSegs_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Assign segments to cores in a grid-stride fashion
        for (uint32_t seg = blockIdx; seg < numSegs_; seg += blockNum) {
            const uint32_t segBase = seg * segLen_;
            float carry = 0.0f;

            uint32_t processed = 0;
            while (processed < segLen_) {
                uint32_t len = segLen_ - processed;
                if (len > tileLen_) len = tileLen_;

                // Copy in
                CopyIn(segBase + processed, len);

                // Compute cumsum tile with carry-in, along last dimension (contiguous)
                Compute(len, carry);

                // Copy out
                CopyOut(segBase + processed, len);

                processed += len;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQue_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQue_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len, float &carry)
    {
        AscendC::LocalTensor<float> xLocal = inQue_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQue_.AllocTensor<float>();

        // In-UB inclusive scan using iterative doubling
        // yLocal = xLocal
        AscendC::DataCopy(yLocal, xLocal, len);

        // For step sizes 1,2,4,... perform y[i] += y[i-step] for i >= step
        for (uint32_t step = 1; step < len; step <<= 1) {
            // Add shifted yLocal onto itself starting from 'step'
            // yLocal[step : step+len-step] += yLocal[0 : len-step]
            AscendC::Add(yLocal[step], yLocal[step], yLocal, len - step);
        }

        // Add carry to the entire tile if non-zero
        if (carry != 0.0f) {
            AscendC::Adds(yLocal, yLocal, carry, len);
        }

        // Update carry with last element of this tile
        // Assume LocalTensor supports element access via GetValue/SetValue
        float lastVal = yLocal.GetValue(len - 1);
        carry = lastVal;

        outQue_.EnQue(yLocal);
        inQue_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t segLen_{0};
    uint32_t numSegs_{0};
    uint32_t tileLen_{0};
    int32_t dim_{-1};
};

extern "C" __global__ __aicore__ void cumsum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.segmentLen,
            tiling_data.numSegments,
            tiling_data.tileLength,
            tiling_data.dim);
    op.Process();
}
"""
