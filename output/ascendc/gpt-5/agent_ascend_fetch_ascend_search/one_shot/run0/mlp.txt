host_operator_src="""
#include "mlp_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// UB model for one layer tile processing:
//   yTile: 1 * tileOut
//   wTile: 1 * tileOut
//   xTile: 1 * tileK
// Total floats ~= 2 * tileOut + tileK
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_BLOCKS = 32;
static constexpr uint32_t DEFAULT_TILEK = 64;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MlpCustomTilingData tiling;
    const gert::Shape* x_shape = context->GetInputShape(0);
    if (x_shape->GetOriginShape().GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t batch = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t inSize = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t* inputSizeAttr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t* h1Attr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t* h2Attr = attrs->GetAttrPointer<int32_t>(2);
    const int32_t* outSizeAttr = attrs->GetAttrPointer<int32_t>(3);
    if (inputSizeAttr == nullptr || h1Attr == nullptr || h2Attr == nullptr || outSizeAttr == nullptr) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t inputSize = static_cast<uint32_t>(*inputSizeAttr);
    const uint32_t h1 = static_cast<uint32_t>(*h1Attr);
    const uint32_t h2 = static_cast<uint32_t>(*h2Attr);
    const uint32_t outSize = static_cast<uint32_t>(*outSizeAttr);
    // Basic consistency check
    if (inputSize != inSize) {
        // Fallback to inSize from tensor if mismatch
    }

    uint32_t tileK = DEFAULT_TILEK;
    if (tileK > inSize) tileK = inSize;
    tileK = AlignDown(tileK, 8);
    if (tileK == 0) tileK = 8;

    // bytes needed roughly = 4 * (2*tileOut + tileK)
    uint32_t maxTileOut = (UB_BUDGET_BYTES / 4);
    if (maxTileOut > tileK) {
        maxTileOut -= tileK;
    }
    // approximate half for safety
    maxTileOut = maxTileOut / 2;
    maxTileOut = AlignDown(maxTileOut, ALIGN_ELEMS);
    if (maxTileOut < ALIGN_ELEMS) maxTileOut = ALIGN_ELEMS;

    const uint32_t minOutDim = std::min(h1, std::min(h2, outSize));
    uint32_t tileOut = std::min(maxTileOut, minOutDim);
    tileOut = AlignDown(tileOut, ALIGN_ELEMS);
    if (tileOut == 0) tileOut = ALIGN_ELEMS;

    uint32_t blockDim = batch;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_inSize(inSize);
    tiling.set_h1(h1);
    tiling.set_h2(h2);
    tiling.set_outSize(outSize);
    tiling.set_tileOut(tileOut);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    // workspace per sample stores hidden1 (h1) and hidden2 (h2)
    workspaceSizes[0] = static_cast<size_t>(batch) * static_cast<size_t>(h1 + h2) * sizeof(float);
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape::ShapeType &oshape = x_shape->GetOriginShape();
    int64_t b = (oshape.GetDimNum() >= 1) ? oshape.GetDim(0) : 1;
    // Get output size from attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t* outSizeAttr = attrs->GetAttrPointer<int32_t>(3);
    int64_t outSize = (outSizeAttr != nullptr) ? static_cast<int64_t>(*outSizeAttr) : 1;
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, b);
    y_shape->SetDim(1, outSize);
    return GRAPH_SUCCESS;
}

    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMlpCustom {
public:
    __aicore__ inline KernelMlpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                GM_ADDR workspace,
                                uint32_t batch, uint32_t inSize,
                                uint32_t h1, uint32_t h2, uint32_t outSize,
                                uint32_t tileOut, uint32_t tileK)
    {
        batch_ = batch;
        inSize_ = inSize;
        h1_ = h1;
        h2_ = h2;
        outSize_ = outSize;
        tileOut_ = tileOut;
        tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(batch_) * inSize_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(inSize_) * h1_ + static_cast<uint64_t>(h1_) * h2_ + static_cast<uint64_t>(h2_) * outSize_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, static_cast<uint64_t>(h1_) + h2_ + outSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(batch_) * outSize_);
        wsGm_.SetGlobalBuffer((__gm__ float*)workspace, static_cast<uint64_t>(batch_) * (h1_ + h2_));

        // Buffers: yTile, wTile, xTile
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileOut_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, tileOut_ * sizeof(float));
        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileK_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t b = blockIdx; b < batch_; b += blockNum) {
            ComputeSample(b);
        }
    }

private:
    __aicore__ inline void ComputeSample(uint32_t b)
    {
        const uint64_t xRowOfs = static_cast<uint64_t>(b) * inSize_;
        const uint64_t yRowOfs = static_cast<uint64_t>(b) * outSize_;
        const uint64_t wsBaseOfs = static_cast<uint64_t>(b) * (h1_ + h2_);

        // Layer 1: inSize_ -> h1_
        const uint64_t w1Base = 0ULL;
        const uint64_t b1Base = 0ULL;
        ComputeLayer(xGm_, xRowOfs, inSize_, w1Base, h1_, bGm_, b1Base, wsGm_, wsBaseOfs, true);

        // Layer 2: h1_ -> h2_
        const uint64_t w2Base = static_cast<uint64_t>(inSize_) * h1_;
        const uint64_t b2Base = static_cast<uint64_t>(h1_);
        ComputeLayer(wsGm_, wsBaseOfs, h1_, w2Base, h2_, bGm_, b2Base, wsGm_, wsBaseOfs + h1_, true);

        // Layer 3: h2_ -> outSize_
        const uint64_t w3Base = static_cast<uint64_t>(inSize_) * h1_ + static_cast<uint64_t>(h1_) * h2_;
        const uint64_t b3Base = static_cast<uint64_t>(h1_) + h2_;
        ComputeLayer(wsGm_, wsBaseOfs + h1_, h2_, w3Base, outSize_, bGm_, b3Base, yGm_, yRowOfs, false);
    }

    // xSrcGm: row-major vector source (length inDim) at offset xRowBase
    // wBase: column-major packed weights base offset for this layer
    // outDim: output dimension for this layer
    // bSrcGm offset bBase points to bias for this layer
    // yDstGm at yBase gets the full output vector
    __aicore__ inline void ComputeLayer(AscendC::GlobalTensor<float>& xSrcGm, uint64_t xRowBase, uint32_t inDim,
                                        uint64_t wBase, uint32_t outDim,
                                        AscendC::GlobalTensor<float>& bSrcGm, uint64_t bBase,
                                        AscendC::GlobalTensor<float>& yDstGm, uint64_t yBase,
                                        bool applyRelu)
    {
        if (outDim == 0) {
            return;
        }
        for (uint32_t outBase = 0; outBase < outDim; outBase += tileOut_) {
            uint32_t curOut = outDim - outBase;
            if (curOut > tileOut_) curOut = tileOut_;

            // Allocate locals
            AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();

            // Initialize with bias tile
            AscendC::DataCopy(yLocal, bSrcGm[bBase + outBase], curOut);

            // Accumulate outer products
            for (uint32_t kBase = 0; kBase < inDim; kBase += tileK_) {
                uint32_t curK = inDim - kBase;
                if (curK > tileK_) curK = tileK_;
                AscendC::DataCopy(xLocal, xSrcGm[xRowBase + kBase], curK);

                for (uint32_t t = 0; t < curK; ++t) {
                    float xval = xLocal.GetValue(t);
                    // Load the contiguous vector for weight column segment
                    const uint64_t wColSeg = wBase + static_cast<uint64_t>(kBase + t) * outDim + outBase;
                    AscendC::DataCopy(wLocal, wGm_[wColSeg], curOut);
                    AscendC::Muls(wLocal, wLocal, xval, curOut);
                    AscendC::Add(yLocal, yLocal, wLocal, curOut);
                }
            }

            if (applyRelu) {
                AscendC::Maxs(yLocal, yLocal, 0.0f, curOut);
            }

            // Store result tile
            AscendC::DataCopy(yDstGm[yBase + outBase], yLocal, curOut);

            // Free locals
            xQueue_.FreeTensor(xLocal);
            wQueue_.FreeTensor(wLocal);
            yQueue_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wsGm_;

    uint32_t batch_{0};
    uint32_t inSize_{0};
    uint32_t h1_{0};
    uint32_t h2_{0};
    uint32_t outSize_{0};
    uint32_t tileOut_{0};
    uint32_t tileK_{0};
};

extern "C" __global__ __aicore__ void mlp_custom(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMlpCustom op;
    op.Init(x, w_packed, b_packed, y, workspace,
            tiling_data.batch, tiling_data.inSize, tiling_data.h1, tiling_data.h2, tiling_data.outSize,
            tiling_data.tileOut, tiling_data.tileK);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'mlp_custom'

class ModelNew(torch.nn.Module):
    """
    Fused 3-layer MLP using custom AscendC kernel. Expects weights repacked to column-major per layer.
    """
    def __init__(self, input_size, layer_sizes, output_size):
        super(ModelNew, self).__init__()
        self.input_size = int(input_size)
        self.layer_sizes = [int(layer_sizes[0]), int(layer_sizes[1])]
        self.output_size = int(output_size)

    def forward(self, x, w_packed, b_packed):
        # Unpack row-major packed weights into matrices
        in_sz = self.input_size
        h1, h2 = self.layer_sizes
        out_sz = self.output_size

        p = 0
        w1n = h1 * in_sz
        w1 = w_packed[p : p + w1n].view(h1, in_sz)
        p += w1n
        w2n = h2 * h1
        w2 = w_packed[p : p + w2n].view(h2, h1)
        p += w2n
        w3n = out_sz * h2
        w3 = w_packed[p : p + w3n].view(out_sz, h2)

        # Repack to column-major per layer: concatenate columns
        # This is achieved by taking transpose and flattening in row-major
        w1_cm = w1.t().contiguous().flatten()
        w2_cm = w2.t().contiguous().flatten()
        w3_cm = w3.t().contiguous().flatten()
        w_packed_cm = torch.cat([w1_cm, w2_cm, w3_cm])

        # Biases remain as-is
        y = getattr(custom_ops_lib, _FN)(
            x, w_packed_cm, b_packed,
            self.input_size, h1, h2, out_sz
        )
        return y
'''
