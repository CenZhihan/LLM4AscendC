host_operator_src="""

#include "matmul_with_transposed_a_custom_tiling.h"
#include "register/op_def_registry.h"

// Matmul tiling side headers
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Input shapes:
    // A: (K, M) as provided by user
    // B: (K, N)
    // Compute C = A^T * B, so logical shapes for Matmul are:
    //   A^T: (M, K) with isTransA=true when A is (K, M)
    //   B:   (K, N)
    //   C:   (M, N)
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    if (a_shape->GetOriginShape().GetDimNum() != 2 ||
        b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const int64_t K = a_shape->GetOriginShape().GetDim(0);
    const int64_t M = a_shape->GetOriginShape().GetDim(1);
    const int64_t N = b_shape->GetOriginShape().GetDim(1);
    // Basic validation
    if (K <= 0 || M <= 0 || N <= 0) {
        return ge::GRAPH_FAILED;
    }
    if (b_shape->GetOriginShape().GetDim(0) != K) {
        return ge::GRAPH_FAILED;
    }

    // Build tiling via MatmulApiTiling
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling tiling(ascendcPlatform);

    // A is transposed logically (we receive A as (K, M) but want (M, K))
    // ND format on GM, half precision
    tiling.SetAType(matmul_tiling::TPosition::GM,
                    matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16,
                    true);  // isTransA = true

    // B is not transposed
    tiling.SetBType(matmul_tiling::TPosition::GM,
                    matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16,
                    false);

    // C in GM, ND, half
    tiling.SetCType(matmul_tiling::TPosition::GM,
                    matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);

    // No bias in this example
    // Prepare shape M, N, K to tiling object
    tiling.SetMNK(static_cast<uint32_t>(M),
                  static_cast<uint32_t>(N),
                  static_cast<uint32_t>(K));

    // Get tiling config
    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Configure blocks: use tiler's suggestion when available; otherwise set a conservative default.
    uint32_t blockDim = tilingData.GetBlockDim() > 0 ? tilingData.GetBlockDim() : 20u;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 32u) blockDim = 32u;
    context->SetBlockDim(blockDim);

    // Save tiling buffer
    tilingData.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tilingData.GetDataSize());

    // No workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // Inputs:
    //   A: (K, M)
    //   B: (K, N)
    // Output C = A^T * B -> (M, N)
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 ||
        b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    int64_t K = a_shape->GetOriginShape().GetDim(0);
    int64_t M = a_shape->GetOriginShape().GetDim(1);
    int64_t N = b_shape->GetOriginShape().GetDim(1);
    if (b_shape->GetOriginShape().GetDim(0) != K) {
        return GRAPH_FAILED;
    }

    gert::Shape *c_shape = context->GetOutputShape(0);
    gert::Shape out;
    out.SetDimNum(2);
    out.SetDim(0, M);
    out.SetDim(1, N);
    *c_shape = out;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // All tensors are half
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using AscendC::CubeFormat;
using AscendC::TPosition;

// Use half-precision inputs/outputs for high performance on AI Core
using A_TYPE = AscendC::MatmulType<TPosition::GM, CubeFormat::ND, half>;
using B_TYPE = AscendC::MatmulType<TPosition::GM, CubeFormat::ND, half>;
using C_TYPE = AscendC::MatmulType<TPosition::GM, CubeFormat::ND, half>;

extern \"C\" __global__ __aicore__ void matmul_with_transposed_a_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    // Load unified matmul tiling data produced on host side
    GET_TILING_DATA(tiling_data, tiling);

    // Initialize Matmul operator
    AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE> mm;

    // Set tiling data
    mm.SetTilingData(tiling_data);

    // We receive A in (K, M) and compute A^T * B, so set transpose flag for A
    mm.SetTransposeA(true);
    // B remains non-transposed by default

    // Bind GM pointers
    A_TYPE a_gm;
    B_TYPE b_gm;
    C_TYPE c_gm;
    a_gm.SetGlobalBuffer((__gm__ half*)A);
    b_gm.SetGlobalBuffer((__gm__ half*)B);
    c_gm.SetGlobalBuffer((__gm__ half*)C);

    // Run compute
    mm.Run(a_gm, b_gm, c_gm);
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_contiguous_half_if_needed(const at::Tensor& t) {
    at::Tensor x = t;
    if (x.scalar_type() != at::kHalf) {
        x = x.to(at::kHalf);
    }
    if (!x.is_contiguous()) {
        x = x.contiguous();
    }
    return x;
}

// C = A^T * B, where A: (K,M), B: (K,N), C: (M,N)
at::Tensor matmul_with_transposed_a_impl_npu(const at::Tensor& A_in, const at::Tensor& B_in) {
    TORCH_CHECK(A_in.dim() == 2 && B_in.dim() == 2, \"Inputs must be 2D tensors\");
    TORCH_CHECK(A_in.size(0) == B_in.size(0), \"K dimension must match between A (K, M) and B (K, N)\");
    auto K = A_in.size(0);
    auto M = A_in.size(1);
    auto N = B_in.size(1);

    at::Tensor A = to_contiguous_half_if_needed(A_in);
    at::Tensor B = to_contiguous_half_if_needed(B_in);

    // Output is half
    at::Tensor C = at::empty({M, N}, A.options().dtype(at::kHalf));

    // Custom op invocation: A and B in GM, result C
    // Contract name must match project json operator and kernel naming
    EXEC_NPU_CMD(aclnnMatmulWithTransposedACustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"matmul_with_transposed_a_custom\", &matmul_with_transposed_a_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"matmul_with_transposed_a_custom\", &matmul_with_transposed_a_impl_npu,
          \"Matmul with A transposed (MatmulWithTransposedACustom on NPU)\");
}
