host_operator_src="""

#include "matmul_mish_mish_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous

namespace optiling {

// UB model for Mish pass:
//   in: 1 * tile
//   out: 1 * tile
//   tmp: 1 * tile
// Total floats = 3 * tile => bytes = 12 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMishMishCustomTilingData tiling;

    // Shapes:
    // x: [M, K]
    // weight: [N, K] (PyTorch weight is [N, K] and linear does x @ weight^T; in kernel we will set transB=true)
    // bias: [N]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetStorageShape().GetDimNum() < 2 || w_shape->GetStorageShape().GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(0)); // out_features

    // UB tile for Mish pass
    uint32_t maxTile = UB_BUDGET_BYTES / 12; // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint64_t totalLen64 = static_cast<uint64_t>(M) * static_cast<uint64_t>(N);
    uint32_t totalLen = totalLen64 > 0xFFFFFFFFull ? 0xFFFFFFFFu : static_cast<uint32_t>(totalLen64);

    uint32_t tileLen = maxTile;
    if (totalLen < tileLen) {
        const uint32_t aligned = AlignDown(totalLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLen;
    }

    // Conservative blockDim
    // Use up to 24 cores (AIC count on 910B). Keep within [8, 24].
    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_blockDim(blockDim);
    tiling.set_tileLength(tileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() < 2 || w_shape->GetOriginShape().GetDimNum() < 2) {
        return GRAPH_FAILED;
    }
    int64_t M = x_shape->GetOriginShape().GetDim(0);
    int64_t N = w_shape->GetOriginShape().GetDim(0);

    gert::Shape *y_shape = context->GetOutputShape(0);
    gert::Shape out;
    out.SetDimNum(2);
    out.SetDim(0, M);
    out.SetDim(1, N);
    *y_shape = out;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MatmulMishMishCustom : public OpDef {
public:
    explicit MatmulMishMishCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

            .ParamType(REQUIRED)

    }

OP_ADD(MatmulMishMishCustom);

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using AType = float;
using BType = float;
using CType = float;
using BiasType = float;

using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, AType>;
using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BType>;
using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, CType>;
using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BiasType>;
using CFG_NORM = AscendC::MatmulCfgNorm;

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulMishMishCustom {
public:
    __aicore__ inline KernelMatmulMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileLength)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        totalLen_ = M_ * N_;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));

        // Matmul object (M x K) * (K x N), with B transposed since PyTorch weight is [N, K] for linear
        // Bias is length N and will be added by Matmul obj.
        // Note: The matmul API abstracts detailed tiling internally.
        // The following is a representative configuration; actual API may differ by version.
        matmulObj_.SetTransA(false);
        matmulObj_.SetTransB(true);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        // Phase 1: GEMM + bias
        // Compute: y = x @ weight^T + bias
        // Using Matmul High-level API (NORMAL layout)
        matmulObj_.SetTensorA<A_TYPE>(xGm_);
        matmulObj_.SetTensorB<B_TYPE>(wGm_);
        matmulObj_.SetTensorC<C_TYPE>(yGm_);
        matmulObj_.SetBias<BIAS_TYPE>(bGm_);
        matmulObj_.IterateBatch(); // single batch NORMAL; tiling config baked by host

        // Phase 2: Apply Mish twice over y in-place.
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            // First Mish
            ComputeMish(curLen);
            // Second Mish
            ComputeMish(curLen);

            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueue_.EnQue(yLocal);
    }

    __aicore__ inline void ComputeMish(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

        // tmp = exp(y)
        AscendC::Exp(tmpLocal, yLocal, len);
        // tmp = tmp + 1
        AscendC::Adds(tmpLocal, tmpLocal, 1.0f, len);
        // tmp = ln(tmp)
        AscendC::Ln(tmpLocal, tmpLocal, len);
        // tmp = tanh(tmp)
        AscendC::Tanh(tmpLocal, tmpLocal, len);
        // y = y * tmp
        AscendC::Mul(yLocal, yLocal, tmpLocal, len);

        tmpQueue_.FreeTensor(tmpLocal);
        inQueue_.EnQue(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        inQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};

    AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE, CFG_NORM> matmulObj_;
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                               GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMishMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileLength);
    op.Process();
}

#include <torch/library.h>

at::Tensor matmul_mish_mish_impl_npu(const at::Tensor& x,
                                     const at::Tensor& weight,
                                     const at::Tensor& bias) {
    // Output shape: [x.size(0), weight.size(0)]
    TORCH_CHECK(x.dim() == 2 && weight.dim() == 2, \"Inputs must be 2D\");
    TORCH_CHECK(bias.dim() == 1, \"Bias must be 1D\");
    TORCH_CHECK(x.size(1) == weight.size(1), \"in_features mismatch\");
    TORCH_CHECK(weight.size(0) == bias.size(0), \"out_features mismatch\");

    at::Tensor out = at::empty({x.size(0), weight.size(0)}, x.options());
    EXEC_NPU_CMD(aclnnMatmulMishMishCustom, x, weight, bias, out);
    return out;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_mish_mish_custom", &matmul_mish_mish_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_mish_mish_custom", &matmul_mish_mish_impl_npu,
          "Fused Matmul + Mish + Mish (MatmulMishMishCustom on NPU)");
