#include "conv_standard_2d_square_input_square_kernel_custom_tiling.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (a == 0) ? x : (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

// UB model:
// - weights: ocChunk * (cIn * kH * kW)
// - accumulators: ocChunk * posChunk
// - xVec: posChunk
// - tmpVec: posChunk
// bytes = 4 * (ocChunk * K + ocChunk * P + P + P) = 4 * (ocChunk*(K+P) + 2P)
static constexpr uint32_t UB_BUDGET_BYTES = 192u * 1024u;
static constexpr uint32_t UB_SAFETY_BYTES = 8u * 1024u;
static constexpr uint32_t ALIGN_P = 128u;
static constexpr uint32_t MIN_P = 64u;
static constexpr uint32_t MAX_BLOCKDIM = 24u;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dSquareInputSquareKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const uint32_t n   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t cIn = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t h   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t w   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const uint32_t cOut = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t kH   = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    const uint32_t kW   = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(3));

    // Fixed parameters for this custom kernel: stride=(1,1), padding=(0,0), dilation=(1,1), groups=1
    const uint32_t outH = h - kH + 1;
    const uint32_t outW = w - kW + 1;
    const uint32_t K    = cIn * kH * kW;

    // Choose ocChunk and posChunk to fit UB
    uint32_t ocChunk = (cOut >= 32u) ? 32u : ((cOut >= 16u) ? 16u : cOut);
    uint32_t bestPosChunk = 0;

    // Try to find a feasible ocChunk and posChunk
    for (; ocChunk >= 1u; ocChunk /= 2u) {
        const uint32_t bytesWeights = 4u * ocChunk * K;
        if (bytesWeights + UB_SAFETY_BYTES >= UB_BUDGET_BYTES) continue;

        const uint32_t remBytes = UB_BUDGET_BYTES - UB_SAFETY_BYTES - bytesWeights;
        // 4 bytes per float: ocChunk*P (acc) + P (x) + P (tmp) = (ocChunk + 2) * P
        uint32_t pMax = remBytes / (4u * (ocChunk + 2u));
        if (pMax > outW) pMax = outW;
        pMax = AlignDown(pMax, ALIGN_P);
        if (pMax >= MIN_P) {
            bestPosChunk = pMax;
            break;
        }
    }

    if (ocChunk == 0u || bestPosChunk == 0u) {
        // Fallback minimal tiling
        ocChunk = (cOut > 1u) ? 1u : cOut;
        bestPosChunk = (outW >= 1u) ? 1u : outW;
    }

    const uint32_t numOcChunks = CeilDiv(cOut, ocChunk);
    const uint32_t numTasks = n * outH * numOcChunks;

    uint32_t blockDim = numTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCKDIM) blockDim = MAX_BLOCKDIM;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_cIn(cIn);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_cOut(cOut);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_kTotal(K);
    tiling.set_ocChunk(ocChunk);
    tiling.set_posChunk(bestPosChunk);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvStandard2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t wi,
                                uint32_t cOut, uint32_t kH, uint32_t kW,
                                uint32_t outH, uint32_t outW, uint32_t K,
                                uint32_t ocChunk, uint32_t posChunk)
    {
        n_ = n; cIn_ = cIn; h_ = h; w_ = wi; cOut_ = cOut; kH_ = kH; kW_ = kW;
        outH_ = outH; outW_ = outW; K_ = K;
        ocChunk_ = ocChunk; posChunk_ = posChunk;

        // total sizes for GM tensor binding
        uint32_t xSize = n_ * cIn_ * h_ * w_;
        uint32_t wSize = cOut_ * K_;
        uint32_t ySize = n_ * cOut_ * outH_ * outW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // Allocate local buffers
        // weights: ocChunk * K
        pipe_.InitBuffer(weightsQ_, BUFFER_NUM, ocChunk_ * K_ * sizeof(float));
        // accumulator: ocChunk * posChunk
        pipe_.InitBuffer(accQ_, BUFFER_NUM, ocChunk_ * posChunk_ * sizeof(float));
        // xVec and tmpVec: posChunk
        pipe_.InitBuffer(xVecQ_, BUFFER_NUM, posChunk_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQ_, BUFFER_NUM, posChunk_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cIn_ == 0 || h_ == 0 || w_ == 0 || cOut_ == 0) return;

        const uint32_t numOcChunks = (cOut_ + ocChunk_ - 1u) / ocChunk_;
        const uint32_t numTasks = n_ * outH_ * numOcChunks;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t task = blockIdx; task < numTasks; task += blockNum) {
            // Map task => (nIdx, ocChunkIdx, oh)
            uint32_t t = task;
            const uint32_t rowSpan = outH_;
            const uint32_t nSpan = numOcChunks * rowSpan;

            const uint32_t nIdx = t / nSpan;
            t = t - nIdx * nSpan;
            const uint32_t ocChunkIdx = t / rowSpan;
            const uint32_t oh = t - ocChunkIdx * rowSpan;

            const uint32_t ocStart = ocChunkIdx * ocChunk_;
            uint32_t ocCount = cOut_ - ocStart;
            if (ocCount > ocChunk_) ocCount = ocChunk_;

            // Load weight chunk: shape [ocCount, K]
            AscendC::LocalTensor<float> wLocal = weightsQ_.AllocTensor<float>();
            for (uint32_t oc = 0; oc < ocCount; ++oc) {
                const uint32_t wBaseGm = (ocStart + oc) * K_; // contiguous K elements
                AscendC::DataCopy(wLocal[oc * K_], wGm_[wBaseGm], K_);
            }

            // Process across output width in segments of posChunk_
            for (uint32_t owStart = 0; owStart < outW_; owStart += posChunk_) {
                uint32_t posLen = outW_ - owStart;
                if (posLen > posChunk_) posLen = posChunk_;

                // Initialize accumulator to zeros
                AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
                AscendC::Dup(acc, 0.0f, ocCount * posChunk_);

                AscendC::LocalTensor<float> xVec = xVecQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpVec = tmpVecQ_.AllocTensor<float>();

                // Compute convolution: accumulate over K = cIn * kH * kW
                for (uint32_t k = 0; k < K_; ++k) {
                    const uint32_t c = k / (kH_ * kW_);
                    const uint32_t kRem = k % (kH_ * kW_);
                    const uint32_t kh = kRem / kW_;
                    const uint32_t kw = kRem % kW_;

                    // Input base offset for this (nIdx, c, oh+kh, owStart+kw)
                    const uint32_t xStrideW = 1u;
                    const uint32_t xStrideH = w_;
                    const uint32_t xStrideC = h_ * w_;
                    const uint32_t xStrideN = cIn_ * h_ * w_;

                    const uint32_t hIn = oh + kh;
                    const uint32_t wInStart = owStart + kw;

                    const uint32_t xBase = nIdx * xStrideN + c * xStrideC + hIn * xStrideH + wInStart * xStrideW;
                    // Contiguous along width for posLen
                    AscendC::DataCopy(xVec, xGm_[xBase], posLen);

                    // For each oc in chunk, yVec += xVec * wScalar
                    for (uint32_t oc = 0; oc < ocCount; ++oc) {
                        // Read scalar weight from local weights buffer
                        float wScalar = wLocal.GetValue(oc * K_ + k);  // assume API to get scalar from LocalTensor
                        AscendC::Muls(tmpVec, xVec, wScalar, posLen);
                        AscendC::Add(acc[oc * posChunk_], acc[oc * posChunk_], tmpVec, posLen);
                    }
                }

                // Store results to y GM
                for (uint32_t oc = 0; oc < ocCount; ++oc) {
                    const uint32_t yStrideW = 1u;
                    const uint32_t yStrideH = outW_;
                    const uint32_t yStrideC = outH_ * outW_;
                    const uint32_t yStrideN = cOut_ * outH_ * outW_;
                    const uint32_t yBase = nIdx * yStrideN + (ocStart + oc) * yStrideC + oh * yStrideH + owStart * yStrideW;
                    AscendC::DataCopy(yGm_[yBase], acc[oc * posChunk_], posLen);
                }

                // Free locals for this segment
                xVecQ_.FreeTensor(xVec);
                tmpVecQ_.FreeTensor(tmpVec);
                accQ_.FreeTensor(acc);
            }

            // Free weight local
            weightsQ_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> weightsQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> accQ_;
    AscendC::TQue<AscendC::TPosition::VECIN,   BUFFER_NUM> xVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpVecQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, h_{0}, w_{0};
    uint32_t cOut_{0}, kH_{0}, kW_{0};
    uint32_t outH_{0}, outW_{0}, K_{0};
    uint32_t ocChunk_{0}, posChunk_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cIn, tiling_data.h, tiling_data.w,
            tiling_data.cOut, tiling_data.kH, tiling_data.kW,
            tiling_data.outH, tiling_data.outW, tiling_data.kTotal,
            tiling_data.ocChunk, tiling_data.posChunk);
    op.Process();
}
