// Budget for UB per core (bytes). Keep conservative.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// Aim for contiguous DMA on width; favor >=256 elements.
static constexpr uint32_t ALIGN_ELEMS = 256;
// Max tile oc to bound per-iter latency and UB usage.
static constexpr uint32_t MAX_TILE_OC = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return ge::GRAPH_PARAM_INVALID;
    }
    // NCHW
    const uint64_t N = x_shape->GetDim(0);
    const uint64_t C = x_shape->GetDim(1);
    const uint64_t H = x_shape->GetDim(2);
    const uint64_t W = x_shape->GetDim(3);

    const uint64_t OC = w_shape->GetDim(0);
    const uint64_t Cw = w_shape->GetDim(1);
    const uint64_t KH = w_shape->GetDim(2);
    const uint64_t KW = w_shape->GetDim(3);

    if (C != Cw || KH != 3 || KW != 3) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const uint64_t strideH = 1, strideW = 1;
    const uint64_t padH = 0, padW = 0;
    const uint64_t dilH = 1, dilW = 1;

    const uint64_t HOut = (H + 2 * padH - dilH * (KH - 1) - 1) / strideH + 1;
    const uint64_t WOut = (W + 2 * padW - dilW * (KW - 1) - 1) / strideW + 1;

    // UB float capacity
    const uint32_t ubFloats = UB_BUDGET_BYTES / 4u;

    // Choose tile width (Tw) along output width and tileOc along output channels.
    // UB usage approx: yAccum (tileOc * Tw) + vIn (Tw) + temp (Tw) <= ubFloats
    uint32_t Tw = static_cast<uint32_t>(WOut);
    if (Tw > 2048u) Tw = 1024u;
    Tw = AlignDown(Tw, ALIGN_ELEMS);
    if (Tw == 0u) Tw = static_cast<uint32_t>(WOut);

    uint32_t tileOc = MAX_TILE_OC;
    uint32_t maxTileOcByUb = 1;
    if (Tw > 0) {
        const uint32_t used = 2u * Tw; // vIn + temp
        if (ubFloats > used) {
            maxTileOcByUb = (ubFloats - used) / Tw;
            if (maxTileOcByUb == 0) maxTileOcByUb = 1;
        }
    }
    if (tileOc > maxTileOcByUb) tileOc = maxTileOcByUb;
    if (tileOc > static_cast<uint32_t>(OC)) tileOc = static_cast<uint32_t>(OC);
    if (tileOc == 0) tileOc = 1;

    const uint64_t wTiles = CeilDiv(static_cast<uint32_t>(WOut), Tw);
    const uint64_t ocTiles = CeilDiv(static_cast<uint32_t>(OC), tileOc);
    const uint64_t totalTasks = N * HOut * wTiles * ocTiles;

    uint32_t blockDim = static_cast<uint32_t>(totalTasks);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_C(static_cast<uint32_t>(C));
    tiling.set_H(static_cast<uint32_t>(H));
    tiling.set_W(static_cast<uint32_t>(W));
    tiling.set_OC(static_cast<uint32_t>(OC));
    tiling.set_KH(static_cast<uint32_t>(KH));
    tiling.set_KW(static_cast<uint32_t>(KW));
    tiling.set_HOut(static_cast<uint32_t>(HOut));
    tiling.set_WOut(static_cast<uint32_t>(WOut));
    tiling.set_tileW(Tw);
    tiling.set_tileOc(tileOc);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t MAX_TILE_OC = 32;

class KernelConvStandard2dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t OC, uint32_t KH, uint32_t KW,
                                uint32_t HOut, uint32_t WOut,
                                uint32_t tileW, uint32_t tileOc)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        OC_ = OC; KH_ = KH; KW_ = KW;
        HOut_ = HOut; WOut_ = WOut;
        tileW_ = tileW; tileOc_ = tileOc;
        if (tileOc_ > MAX_TILE_OC) tileOc_ = MAX_TILE_OC;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * C_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)OC_ * C_ * KH_ * KW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * OC_ * HOut_ * WOut_);

        pipe_.InitBuffer(vecInQ_, 2, tileW_ * sizeof(float));
        pipe_.InitBuffer(vecTmpQ_, 2, tileW_ * sizeof(float));
        pipe_.InitBuffer(vecAccQ_, MAX_TILE_OC, tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || OC_ == 0 || HOut_ == 0 || WOut_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t wTiles = CeilDiv(WOut_, tileW_);
        const uint32_t ocTiles = CeilDiv(OC_, tileOc_);
        const uint64_t totalTasks = (uint64_t)N_ * HOut_ * wTiles * ocTiles;

        for (uint64_t task = blockIdx; task < totalTasks; task += blockNum) {
            // decode task id
            uint64_t tmp = task;
            const uint32_t t_w = static_cast<uint32_t>(tmp % wTiles); tmp /= wTiles;
            const uint32_t t_oc = static_cast<uint32_t>(tmp % ocTiles); tmp /= ocTiles;
            const uint32_t t_h = static_cast<uint32_t>(tmp % HOut_); tmp /= HOut_;
            const uint32_t t_n = static_cast<uint32_t>(tmp % N_);

            const uint32_t wStart = t_w * tileW_;
            uint32_t Tw = tileW_;
            if (wStart + Tw > WOut_) Tw = WOut_ - wStart;

            const uint32_t ocStart = t_oc * tileOc_;
            uint32_t ocount = tileOc_;
            if (ocStart + ocount > OC_) ocount = OC_ - ocStart;

            ComputeTile(t_n, t_h, wStart, Tw, ocStart, ocount);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) const
    {
        return (a + b - 1u) / b;
    }

    __aicore__ inline uint64_t XIndex(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * C_ + c) * H_ + h) * W_ + w;
    }

    __aicore__ inline uint64_t YIndex(uint32_t n, uint32_t oc, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * OC_ + oc) * HOut_ + h) * WOut_ + w;
    }

    __aicore__ inline uint64_t WIndex(uint32_t oc, uint32_t c, uint32_t kh, uint32_t kw) const
    {
        return (((uint64_t)oc * C_ + c) * KH_ + kh) * KW_ + kw;
    }

    __aicore__ inline void ComputeTile(uint32_t n, uint32_t hout, uint32_t wStart, uint32_t Tw,
                                       uint32_t ocStart, uint32_t ocount)
    {
        // Allocate accumulators for each oc in tile and zero-init
        AscendC::LocalTensor<float> yAcc[MAX_TILE_OC];
        for (uint32_t i = 0; i < ocount; ++i) {
            yAcc[i] = vecAccQ_.AllocTensor<float>();
            AscendC::Muls(yAcc[i], yAcc[i], 0.0f, Tw);
        }
        AscendC::LocalTensor<float> vIn = vecInQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> vTmp = vecTmpQ_.AllocTensor<float>();

        // Iterate over K = C * KH * KW
        for (uint32_t c = 0; c < C_; ++c) {
            for (uint32_t kh = 0; kh < KH_; ++kh) {
                const uint32_t ih = hout + kh; // stride=1, pad=0
                for (uint32_t kw = 0; kw < KW_; ++kw) {
                    const uint32_t iw0 = wStart + kw;

                    // Load input slice for this (c,kh,kw) across Tw
                    const uint64_t xOff = XIndex(n, c, ih, iw0);
                    AscendC::DataCopy(vIn, xGm_[xOff], Tw);

                    // Accumulate for each oc in tile
                    for (uint32_t o = 0; o < ocount; ++o) {
                        const uint32_t oc = ocStart + o;
                        const uint64_t wOff = WIndex(oc, c, kh, kw);
                        // Load weight scalar to a tiny local buffer (len 1), then use as scalar
                        // For simplicity, reuse vTmp's first element as scalar holder
                        AscendC::LocalTensor<float> wScalarBuf = vecTmpQ_.AllocTensor<float>();
                        // Copy single element weight
                        AscendC::DataCopy(wScalarBuf, wGm_[wOff], 1);
                        // Read scalar value from local buffer index 0
                        float wScalar = wScalarBuf.GetValue(0);
                        vecTmpQ_.FreeTensor(wScalarBuf);

                        AscendC::Muls(vTmp, vIn, wScalar, Tw);
                        AscendC::Add(yAcc[o], yAcc[o], vTmp, Tw);
                    }
                }
            }
        }

        // Store results
        for (uint32_t o = 0; o < ocount; ++o) {
            const uint32_t oc = ocStart + o;
            const uint64_t yOff = YIndex(n, oc, hout, wStart);
            AscendC::DataCopy(yGm_[yOff], yAcc[o], Tw);
        }

        // Free locals
        for (uint32_t i = 0; i < ocount; ++i) {
            vecAccQ_.FreeTensor(yAcc[i]);
        }
        vecTmpQ_.FreeTensor(vTmp);
        vecInQ_.FreeTensor(vIn);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> vecInQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> vecTmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, MAX_TILE_OC> vecAccQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t OC_{0}, KH_{0}, KW_{0};
    uint32_t HOut_{0}, WOut_{0};
    uint32_t tileW_{0}, tileOc_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dAsymmetricInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.OC, tiling_data.KH, tiling_data.KW,
            tiling_data.HOut, tiling_data.WOut,
            tiling_data.tileW, tiling_data.tileOc);
    op.Process();
}
