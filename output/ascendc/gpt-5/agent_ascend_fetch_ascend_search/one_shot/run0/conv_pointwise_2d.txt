host_operator_src="""

#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

// UB model for this kernel:
//   - Weight buffer: C_out * C_in floats
//   - Accumulator y buffer: tilePos * C_out floats
//   - Input slice buffer: tilePos floats
//   - Temp column buffer: C_out floats
//   - Some extra slop
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative per-core UB budget
static constexpr uint32_t ALIGN_TILE = 64;                // favor DMA alignment for plane segments
static constexpr uint32_t MIN_TILE = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect x: [N, C_in, H, W], weight: [C_out, C_in, 1, 1]
    const uint32_t N  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Ci = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(x_shape->GetDim(3));
    const uint32_t Co = static_cast<uint32_t>(w_shape->GetDim(0));

    const uint64_t plane = static_cast<uint64_t>(H) * static_cast<uint64_t>(W);

    // UB budget-based tile size along the HW plane
    const uint32_t weightBytes = Co * Ci * sizeof(float);
    uint32_t avail = (UB_BUDGET_BYTES > weightBytes) ? (UB_BUDGET_BYTES - weightBytes) : 0;
    // y buffer + x slice + temp column
    // bytes = tilePos * Co * 4 + tilePos * 4 + Co * 4
    // => tilePos * (Co + 1) * 4 + Co * 4 <= avail
    uint32_t denom = (Co + 1) * 4;
    uint32_t base = (avail > Co * 4) ? ((avail - Co * 4) / denom) : 0;
    uint32_t tilePos = base;
    if (tilePos < MIN_TILE) tilePos = MIN_TILE;
    // Bound by plane
    if (plane < tilePos) tilePos = static_cast<uint32_t>(plane);
    // Align down to favor DMA
    tilePos = AlignDown(tilePos, ALIGN_TILE);
    if (tilePos == 0) tilePos = MIN_TILE < plane ? MIN_TILE : static_cast<uint32_t>(plane);

    const uint32_t segPerN = CeilDiv(plane, tilePos);
    const uint32_t totalSeg = N * segPerN;
    uint32_t blockDim = totalSeg;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cIn(Ci);
    tiling.set_cOut(Co);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_tilePos(tilePos);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *w_shape = context->GetInputShape(1);

    y_shape->SetDimNum(4);
    y_shape->SetDim(0, x_shape->GetDim(0));
    y_shape->SetDim(1, w_shape->GetDim(0));
    y_shape->SetDim(2, x_shape->GetDim(2));
    y_shape->SetDim(3, x_shape->GetDim(3));

}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t Co,
                                uint32_t H, uint32_t W, uint32_t tilePos)
    {
        n_ = N; ci_ = Ci; co_ = Co; h_ = H; w_ = W; tilePos_ = tilePos;
        plane_ = static_cast<uint32_t>(H) * static_cast<uint32_t>(W);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(N) * Ci * plane_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, Co * Ci);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(N) * Co * plane_);

        // UB buffers:
        // - Full weight matrix (Co * Ci)
        // - Input slice (tilePos)
        // - Accumulator y (tilePos * Co) row-major (s, co)
        // - Temp w_col (Co)
        pipe_.InitBuffer(wFullQ_, 1, Co * Ci * sizeof(float));
        pipe_.InitBuffer(xSliceQ_, BUFFER_NUM, tilePos_ * sizeof(float));
        pipe_.InitBuffer(yRowQ_, 1, tilePos_ * Co * sizeof(float));
        pipe_.InitBuffer(wColQ_, 1, Co * sizeof(float));
        pipe_.InitBuffer(tmpVecQ_, 1, Co * sizeof(float));
        pipe_.InitBuffer(yColQ_, 1, tilePos_ * Co * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || ci_ == 0 || co_ == 0 || plane_ == 0 || tilePos_ == 0) { return; }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t segPerN = CeilDiv(plane_, tilePos_);
        const uint32_t totalSeg = n_ * segPerN;

        for (uint32_t seg = blockIdx; seg < totalSeg; seg += blockNum) {
            const uint32_t n = seg / segPerN;
            const uint32_t posIdx = seg % segPerN;
            const uint32_t b0 = posIdx * tilePos_;
            if (b0 >= plane_) { continue; }
            uint32_t curLen = plane_ - b0;
            if (curLen > tilePos_) curLen = tilePos_;

            // Load full weight matrix to UB (Co x Ci)
            AscendC::LocalTensor<float> wFull = wFullQ_.AllocTensor<float>();
            AscendC::DataCopy(wFull, wGm_, co_ * ci_);
            wFullQ_.EnQue(wFull);

            // Allocate and zero Y row-major buffer [curLen x Co]
            AscendC::LocalTensor<float> yRow = yRowQ_.AllocTensor<float>();
            AscendC::SetFloat(yRow, 0.0f, curLen * co_);

            // For each input channel ci, accumulate contributions
            for (uint32_t c = 0; c < ci_; ++c) {
                // Copy input slice for this ci: length curLen, contiguous at fixed n,c
                AscendC::LocalTensor<float> xSlice = xSliceQ_.AllocTensor<float>();
                const uint32_t xOffset = (n * ci_ + c) * plane_ + b0;
                AscendC::DataCopy(xSlice, xGm_[xOffset], curLen);

                // Build w_col (Co) for this ci from wFull (Co x Ci) with inner stride Ci
                AscendC::LocalTensor<float> wCol = wColQ_.AllocTensor<float>();
                // Gather column with scalar loop
                for (uint32_t co = 0; co < co_; ++co) {
                    // index in wFull: co * ci_ + c
                    float val = wFull.GetValue(co * ci_ + c);
                    wCol.SetValue(co, val);
                }

                // Accumulate into yRow: for each position s in [0, curLen)
                AscendC::LocalTensor<float> tmpVec = tmpVecQ_.AllocTensor<float>();
                for (uint32_t s = 0; s < curLen; ++s) {
                    const float alpha = xSlice.GetValue(s);
                    // tmpVec = wCol * alpha
                    AscendC::Muls(tmpVec, wCol, alpha, co_);
                    // yRow segment starting at s * Co of length Co
                    AscendC::Add(yRow[s * co_], yRow[s * co_], tmpVec, co_);
                }
                tmpVecQ_.FreeTensor(tmpVec);

                xSliceQ_.FreeTensor(xSlice);
                wColQ_.FreeTensor(wCol);
            }

            // Transpose yRow [curLen x Co] -> yCol [Co x curLen] for contiguous GM store
            AscendC::LocalTensor<float> yCol = yColQ_.AllocTensor<float>();
            for (uint32_t s = 0; s < curLen; ++s) {
                for (uint32_t co = 0; co < co_; ++co) {
                    float v = yRow.GetValue(s * co_ + co);
                    yCol.SetValue(co * curLen + s, v);
                }
            }

            // Write back per Co as contiguous segments of length curLen
            for (uint32_t co = 0; co < co_; ++co) {
                const uint32_t yOffset = (n * co_ + co) * plane_ + b0;
                AscendC::DataCopy(yGm_[yOffset], yCol[co * curLen], curLen);
            }

            yColQ_.FreeTensor(yCol);
            yRowQ_.FreeTensor(yRow);
            wFullQ_.FreeTensor(wFull);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wFullQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xSliceQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wColQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVecQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yColQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, ci_{0}, co_{0}, h_{0}, w_{0}, plane_{0}, tilePos_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cIn, tiling_data.cOut,
            tiling_data.h, tiling_data.w, tiling_data.tilePos);
    op.Process();
}
