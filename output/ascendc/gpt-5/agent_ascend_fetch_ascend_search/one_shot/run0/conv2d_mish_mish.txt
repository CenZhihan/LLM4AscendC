static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)

    const uint64_t W  = (x_shape->GetOriginShape().GetDimNum() > 3) ? x_shape->GetOriginShape().GetDim(3) : 0;

    const uint64_t Cout = (w_shape->GetOriginShape().GetDimNum() > 0) ? w_shape->GetOriginShape().GetDim(0) : 0;
    const uint64_t Kh   = (w_shape->GetOriginShape().GetDimNum() > 2) ? w_shape->GetOriginShape().GetDim(2) : 0;
    const uint64_t Kw   = (w_shape->GetOriginShape().GetDimNum() > 3) ? w_shape->GetOriginShape().GetDim(3) : 0;

    const uint64_t Hout = (H >= Kh) ? (H - Kh + 1) : 0;
    const uint64_t Wout = (W >= Kw) ? (W - Kw + 1) : 0;

    const uint64_t totalOutElems64 = N * Cout * Hout * Wout;
    const uint32_t totalOutElems = static_cast<uint32_t>(totalOutElems64);

    // Simple tiling across linearized output elements.
    // UB usage is small per-output; choose a large tile but keep DMA-friendly alignment.
    static constexpr uint32_t ALIGN_ELEMS = 1024;
    static constexpr uint32_t HARD_CAP_ELEMS = 32768;

    uint32_t tileLen = HARD_CAP_ELEMS;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;
    if (totalOutElems < tileLen) {
        uint32_t aligned = AlignDown(totalOutElems, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutElems;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutElems(totalOutElems);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_Cin(static_cast<uint32_t>(Cin));
    tiling.set_Cout(static_cast<uint32_t>(Cout));
    tiling.set_H(static_cast<uint32_t>(H));
    tiling.set_W(static_cast<uint32_t>(W));
    tiling.set_Kh(static_cast<uint32_t>(Kh));
    tiling.set_Kw(static_cast<uint32_t>(Kw));
    tiling.set_Hout(static_cast<uint32_t>(Hout));
    tiling.set_Wout(static_cast<uint32_t>(Wout));

kernel_src="""

#include "kernel_operator.h"

// Fused conv2d (stride=1, padding=0, dilation=1, groups=1) + mish + mish.
// Mish(x) = x * tanh(softplus(x)), softplus(x) = ln(1 + exp(x))
constexpr int32_t BUF_ONE = 1;

class KernelConv2dMishMishCustom {
public:
    __aicore__ inline KernelConv2dMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalOutElems, uint32_t tileLength,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t Kh, uint32_t Kw,
                                uint32_t Hout, uint32_t Wout, uint32_t hasBias)
    {
        totalOutElems_ = totalOutElems;
        tileLen_ = tileLength;

        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; Kh_ = Kh; Kw_ = Kw;
        Hout_ = Hout; Wout_ = Wout;
        hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, Cout_ * Cin_ * Kh_ * Kw_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * Cout_ * Hout_ * Wout_);

        // Allocate UB buffers for one patch (Cin*Kh*Kw), weights patch, tmp product, and small scalars.
        patchElems_ = Cin_ * Kh_ * Kw_;
        if (patchElems_ == 0) { patchElems_ = 1; }

        pipe_.InitBuffer(xPatchQue_, BUF_ONE, patchElems_ * sizeof(float));
        pipe_.InitBuffer(wPatchQue_, BUF_ONE, patchElems_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQue_, BUF_ONE, patchElems_ * sizeof(float));

        // Scalar work buffers (length 1) and temporary bytes for Exp/Tanh/Ln
        pipe_.InitBuffer(scalarQue_, BUF_ONE, 4 * sizeof(float));
        pipe_.InitBuffer(tmpByteQue_, BUF_ONE, 4096); // shared tmp for math functions
    }

    __aicore__ inline void Process()
    {
        if (totalOutElems_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutElems_) return;

        uint32_t curLen = totalOutElems_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        while (true) {
            ComputeRange(base, curLen);

            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalOutElems_) break;

            uint32_t nextLen = totalOutElems_ - nextBase;
            if (nextLen > tileLen_) nextLen = tileLen_;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void ComputeRange(uint32_t base, uint32_t len)
    {
        AscendC::LocalTensor<float> xPatch = xPatchQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> wPatch = wPatchQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpVec = tmpVecQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> scalarBuf = scalarQue_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tmpBytes = tmpByteQue_.AllocTensor<uint8_t>();

        for (uint32_t t = 0; t < len; ++t) {
            uint32_t idx = base + t;

            // Decode linear index into (n, co, oh, ow): fastest varying ow, then oh, then co, then n.
            uint32_t ow = idx % Wout_; idx /= Wout_;
            uint32_t oh = idx % Hout_; idx /= Hout_;
            uint32_t co = idx % Cout_; idx /= Cout_;
            uint32_t n  = idx;

            // Fill xPatch and wPatch in order [ci, kh, kw]
            uint32_t offset = 0;
            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    // Copy Kw contiguous elements for both input and weight
                    const uint32_t xBase = (((n * Cin_ + ci) * H_ + (oh + kh)) * W_ + ow);
                    const uint32_t wBase = (((co * Cin_ + ci) * Kh_ + kh) * Kw_);
                    AscendC::DataCopy(xPatch[offset], xGm_[xBase], Kw_);
                    AscendC::DataCopy(wPatch[offset], wGm_[wBase], Kw_);
                    offset += Kw_;
                }
            }

            // tmpVec = xPatch * wPatch
            AscendC::Mul(tmpVec, xPatch, wPatch, patchElems_);

            // Reduce tmpVec to scalar sum in-place in tmpVec[0]
            uint32_t cur = patchElems_;
            // If odd length, fold extra element each iteration
            while (cur > 1) {
                if (cur & 1u) {
                    // Add the last element into the first element
                    AscendC::Add(tmpVec, tmpVec, tmpVec[cur - 1], 1);
                    --cur;
                }
                const uint32_t half = cur >> 1;
                // tmpVec[0:half] = tmpVec[0:half] + tmpVec[half:cur]
                AscendC::Add(tmpVec, tmpVec, tmpVec[half], half);
                cur = half;
            }

            // scalarBuf[0] holds conv sum
            // Copy tmpVec[0] -> scalarBuf[0]
            AscendC::DataCopy(scalarBuf, tmpVec, 1);

            // Add bias if present: scalarBuf[0] += b[co]
            if (hasBias_) {
                AscendC::LocalTensor<float> bLocal = scalarQue_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[co], 1);
                AscendC::Add(scalarBuf, scalarBuf, bLocal, 1);
                scalarQue_.FreeTensor(bLocal);
            }

            // Apply Mish twice: z = mish(mish(z))
            // First mish
            ApplyMishInplace(scalarBuf, tmpBytes);
            // Second mish
            ApplyMishInplace(scalarBuf, tmpBytes);

            // Write to output
            const uint32_t yOff = (((n * Cout_ + co) * Hout_ + oh) * Wout_ + ow);
            AscendC::DataCopy(yGm_[yOff], scalarBuf, 1);
        }

        tmpByteQue_.FreeTensor(tmpBytes);
        scalarQue_.FreeTensor(scalarBuf);
        tmpVecQue_.FreeTensor(tmpVec);
        wPatchQue_.FreeTensor(wPatch);
        xPatchQue_.FreeTensor(xPatch);
    }

    __aicore__ inline void ApplyMishInplace(AscendC::LocalTensor<float>& z, AscendC::LocalTensor<uint8_t>& tmpBytes)
    {
        // softplus = ln(1 + exp(z))
        AscendC::LocalTensor<float> eLocal = scalarQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> onepLocal = scalarQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> spLocal = scalarQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tanhLocal = scalarQue_.AllocTensor<float>();
        AscendC::Exp<float, 12>(eLocal, z, tmpBytes, 1);
        // onepLocal = eLocal + 1
        AscendC::Adds(onepLocal, eLocal, 1.0f, 1);
        // spLocal = ln(onepLocal)
        AscendC::Ln(spLocal, onepLocal, tmpBytes, 1);
        // tanhLocal = tanh(spLocal)
        AscendC::Tanh(tanhLocal, spLocal, tmpBytes, 1);
        // z = z * tanhLocal
        AscendC::Mul(z, z, tanhLocal, 1);

        scalarQue_.FreeTensor(tanhLocal);
        scalarQue_.FreeTensor(spLocal);
        scalarQue_.FreeTensor(onepLocal);
        scalarQue_.FreeTensor(eLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_ONE> xPatchQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_ONE> wPatchQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_ONE> tmpVecQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_ONE> scalarQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_ONE> tmpByteQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutElems_{0};
    uint32_t tileLen_{0};

    uint32_t N_{0}, Cin_{0}, Cout_{0}, H_{0}, W_{0}, Kh_{0}, Kw_{0}, Hout_{0}, Wout_{0};
    uint32_t hasBias_{0};
    uint32_t patchElems_{1};
};

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMishMishCustom op;
    op.Init(x, w, b, y,
            tiling_data.totalOutElems, tiling_data.tileLength,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Hout, tiling_data.Wout, tiling_data.hasBias);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline std::vector<int64_t> infer_output_shape_2d_conv_nchw(const at::Tensor& x, const at::Tensor& weight)
{
    // stride=1, padding=0, dilation=1
    const int64_t N = x.size(0);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);
    const int64_t Cout = weight.size(0);
    const int64_t Kh = weight.size(2);
    const int64_t Kw = weight.size(3);
    const int64_t Hout = H - Kh + 1;
    const int64_t Wout = W - Kw + 1;
    return {N, Cout, Hout, Wout};
}

at::Tensor conv2d_mish_mish_impl_npu(const at::Tensor& x, const at::Tensor& weight, const c10::optional<at::Tensor>& bias_opt)
{
    TORCH_CHECK(x.dim() == 4, "x must be 4D NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be 4D OIHW");
    const std::vector<int64_t> out_shape = infer_output_shape_2d_conv_nchw(x, weight);
    at::Tensor y = at::empty(out_shape, x.options());

    if (bias_opt.has_value()) {
        const at::Tensor& bias = bias_opt.value();
        EXEC_NPU_CMD(aclnnConv2dMishMishCustom, x, weight, bias, y);
    } else {
        at::Tensor empty_bias; // undefined tensor for optional input
        EXEC_NPU_CMD(aclnnConv2dMishMishCustom, x, weight, empty_bias, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_mish_mish_custom", &conv2d_mish_mish_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_mish_mish_custom", &conv2d_mish_mish_impl_npu,
          "Fused Conv2D + Mish + Mish (Conv2dMishMishCustom on NPU)",
