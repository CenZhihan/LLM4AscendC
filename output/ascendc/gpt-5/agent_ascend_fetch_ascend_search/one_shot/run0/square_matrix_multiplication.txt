#include "square_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t SafeU32(size_t v) {
    return static_cast<uint32_t>(v);
}

} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = SafeU32(a_shape->GetOriginShape().GetDim(0));
    const uint32_t K = SafeU32(a_shape->GetOriginShape().GetDim(1));
    const uint32_t Kb = SafeU32(b_shape->GetOriginShape().GetDim(0));
    const uint32_t N = SafeU32(b_shape->GetOriginShape().GetDim(1));
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    SquareMatrixMultiplicationCustomTilingData tiling;
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);

    // Run full GEMM in one block by default; Matmul API manages cores internally.
    uint32_t blockDim = 1u;
    context->SetBlockDim(blockDim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = a_shape->GetOriginShape().GetDim(0);
    const int64_t K = a_shape->GetOriginShape().GetDim(1);
    const int64_t Kb = b_shape->GetOriginShape().GetDim(0);
    const int64_t N = b_shape->GetOriginShape().GetDim(1);
    if (K != Kb) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    gert::ShapeShape out;
    out.SetDimNum(2);
    out.SetDim(0, M);
    out.SetDim(1, N);
    c_shape->SetOriginShape(out);
    c_shape->SetStorageShape(out);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != ge::DT_FLOAT || b_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""

#define ASCENDC_CUBE_ONLY
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using namespace AscendC;

class KernelSquareMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelSquareMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t M, uint32_t N, uint32_t K)
    {
        a_ = a;
        b_ = b;
        c_ = c;
        M_ = M;
        N_ = N;
        K_ = K;
    }

    __aicore__ inline void Process()
    {
        // Single-core launch of high-level Matmul
        if (GetBlockIdx() != 0) {
            return;
        }

        typedef MatmulType<TPosition::GM, CubeFormat::ND, float> aType;
        typedef MatmulType<TPosition::GM, CubeFormat::ND, float> bType;
        typedef MatmulType<TPosition::GM, CubeFormat::ND, float> cType;
        typedef MatmulType<TPosition::GM, CubeFormat::ND, float> biasType;

        Matmul<aType, bType, cType, biasType> mm;

        // Init Matmul object
        mm.Init();

        // Set original shape M, K, N
        mm.SetOrgShape(M_, K_, N_);

        // Set input/output tensors in GM
        mm.SetTensorA((__gm__ float*)a_, M_, K_);
        mm.SetTensorB((__gm__ float*)b_, K_, N_);
        mm.SetTensorC((__gm__ float*)c_, M_, N_);

        // No Bias in this op
        mm.SetBias(nullptr, 0);

        // Run Matmul C = A * B
        mm.Run();

        // End Matmul
        mm.End();
    }

private:
    GM_ADDR a_{nullptr};
    GM_ADDR b_{nullptr};
    GM_ADDR c_{nullptr};
    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
};

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSquareMatrixMultiplicationCustom op;
    op.Init(A, B, C, tiling_data.M, tiling_data.N, tiling_data.K);
    op.Process();
}
"""
