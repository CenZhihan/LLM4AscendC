#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using half = AscendC::half;

class KernelMatmulScaleResidualAddClampLogSumExpMishCustom {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMishCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR bias,
                                GM_ADDR scaling, GM_ADDR clamp_min, GM_ADDR clamp_max,
                                GM_ADDR y, GM_ADDR workspace,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileN)
    {
        M_ = M; N_ = N; K_ = K; tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ half*)a, static_cast<uint64_t>(M_) * K_);
        bGm_.SetGlobalBuffer((__gm__ half*)b, static_cast<uint64_t>(K_) * N_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, N_);
        sGm_.SetGlobalBuffer((__gm__ float*)scaling, 1);
        loGm_.SetGlobalBuffer((__gm__ float*)clamp_min, 1);
        hiGm_.SetGlobalBuffer((__gm__ float*)clamp_max, 1);

        // Temp buffer for C[M,N]
        cGm_.SetGlobalBuffer((__gm__ float*)workspace, static_cast<uint64_t>(M_) * N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M_);

        // Queues for post-processing tiles over N
        pipe_.InitBuffer(vecIn_, 2, tileN_ * sizeof(float));
        pipe_.InitBuffer(vecTmp1_, 2, tileN_ * sizeof(float));
        pipe_.InitBuffer(vecTmp2_, 1, tileN_ * sizeof(float));

        pipe_.InitBuffer(scalarBuf_, 1, sizeof(float) * 8); // hold a few scalars
    }

    __aicore__ inline void RunMatmul(const optiling::MatmulScaleResidualAddClampLogSumExpMishCustomTilingData &tdata)
    {
        // Define Matmul types: A, B in half; C and Bias in float
        using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half>;
        using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half>;
        using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
        using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

        AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> mm;
        mm.SetInfo(tdata.cubeTilingData());
        mm.SetTensorA(aGm_.GetGlobalBuffer());
        mm.SetTensorB(bGm_.GetGlobalBuffer());
        mm.SetTensorC(cGm_.GetGlobalBuffer());
        mm.SetTensorBias(biasGm_.GetGlobalBuffer());
        mm.Compute();
    }

    __aicore__ inline void PostProcess()
    {
        // Load scalars
        AscendC::LocalTensor<float> scalars = scalarBuf_.AllocTensor<float>();
        // s
        AscendC::DataCopy(scalars[0], sGm_[0], 1);
        // clamp min
        AscendC::DataCopy(scalars[1], loGm_[0], 1);
        // clamp max
        AscendC::DataCopy(scalars[2], hiGm_[0], 1);
        scalarBuf_.EnQue(scalars);
        scalarBuf_.DeQue<float>(); // keep in local

        const float s = scalars.GetValue(0);
        const float lo = scalars.GetValue(1);
        const float hi = scalars.GetValue(2);
        const float one_plus_s = 1.0f + s;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            float sumExpRow = 0.0f;

            for (uint32_t n0 = 0; n0 < N_; n0 += tileN_) {
                uint32_t len = N_ - n0;
                if (len > tileN_) len = tileN_;

                AscendC::LocalTensor<float> v = vecIn_.AllocTensor<float>();
                AscendC::DataCopy(v, cGm_[static_cast<uint64_t>(m) * N_ + n0], len);

                // y = y * (1 + s)
                AscendC::Muls(v, v, one_plus_s, len);

                // Clamp to [lo, hi]
                AscendC::Mins(v, v, hi, len);
                AscendC::Maxs(v, v, lo, len);

                // exp(v)
                AscendC::Exp(v, v, len);

                // Reduce sum into scalar temp
                AscendC::LocalTensor<float> tmp = vecTmp1_.AllocTensor<float>();
                // Use vector reduce sum; assume API exists
                AscendC::ReduceSum(tmp, v, len);
                float partSum = tmp.GetValue(0);
                sumExpRow += partSum;

                vecTmp1_.FreeTensor(tmp);
                vecIn_.FreeTensor(v);
            }

            // y = log(sumExpRow)
            float yScalar = AscendC::Log(sumExpRow);

            // mish(y) = y * tanh(softplus(y)), softplus = log(1 + exp(y))
            float expy = AscendC::Exp(yScalar);
            float softplus = AscendC::Log(1.0f + expy);
            float mish = yScalar * AscendC::Tanh(softplus);

            float out = yScalar * mish;
            yGm_[m] = out;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> vecIn_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> vecTmp1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> vecTmp2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBuf_;

    AscendC::GlobalTensor<half> aGm_;
    AscendC::GlobalTensor<half> bGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> sGm_;
    AscendC::GlobalTensor<float> loGm_;
    AscendC::GlobalTensor<float> hiGm_;
    AscendC::GlobalTensor<float> cGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileN_{0};
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scaling, GM_ADDR clamp_min, GM_ADDR clamp_max,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{

    op.Init(x, weight, bias, scaling, clamp_min, clamp_max, y, workspace,
            tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileN);
    op.RunMatmul(tiling_data);
    op.PostProcess();
}
