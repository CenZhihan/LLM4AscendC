host_operator_src="""

#include "cumsum_reverse_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint64_t ShapeSize(const gert::Shape& s) {
    uint64_t prod = 1;
    for (size_t i = 0; i < s.GetDimNum(); ++i) {
        prod *= static_cast<uint64_t>(s.GetDim(i));
    }
    return prod;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 160 * 1024;  // use single buffer to be conservative
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MAX_BLOCKS = 32;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumReverseCustomTilingData tiling;

    const gert::Shape* inShape = context->GetInputShape(0);
    const int64_t rank = static_cast<int64_t>(inShape->GetDimNum());

    // Read axis attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t axis =  -1;
    if (attrs != nullptr && attrs->GetNum() > 0) {
        const int64_t* axisPtr = attrs->GetAttrPointer<int64_t>(0);
        axis = (axisPtr != nullptr) ? *axisPtr : -1;
    }
    if (axis < 0) axis += rank;
    if (axis < 0) axis = rank - 1;
    if (axis >= rank) axis = rank - 1;

    // Only optimize along last dim; for other axes, we fall back to treating last dim.
    const int64_t axisNorm = axis;
    const uint32_t K = static_cast<uint32_t>(inShape->GetDim(axisNorm));
    const uint64_t totalSize = ShapeSize(*inShape);
    const uint32_t outer = (K == 0) ? 0u : static_cast<uint32_t>(totalSize / K);

    // Tile along last dimension with a single buffer
    uint32_t maxTileElems = UB_BUDGET_BYTES / sizeof(float);
    maxTileElems = AlignDown(maxTileElems, ALIGN_ELEMS);
    if (maxTileElems == 0) maxTileElems = ALIGN_ELEMS;

    uint32_t tileInner = (K <= maxTileElems) ? K : maxTileElems;
    if (tileInner == 0) tileInner = 1;

    // blockDim based on outer rows
    uint32_t blockDim = outer == 0 ? 1u : (outer < MAX_BLOCKS ? outer : MAX_BLOCKS);
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_outerSize(outer);
    tiling.set_lastDim(K);
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);
    tiling.set_axis(static_cast<int32_t>(axisNorm));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:

kernel_src="""

#include "kernel_operator.h"

class KernelCumsumReverseCustom {
public:
    __aicore__ inline KernelCumsumReverseCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerSize, uint32_t lastDim,
                                uint32_t tileInner)
    {
        outer_ = outerSize;
        lastDim_ = lastDim;
        tileInner_ = tileInner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(outer_) * lastDim_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(outer_) * lastDim_);

        // Use a single VECCALC buffer for both read and write of a segment
        pipe_.InitBuffer(calcQueue_, 1, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outer_ == 0 || lastDim_ == 0 || tileInner_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < outer_; row += blockNum) {
            ComputeRow(row);
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t gmOffset, uint32_t len, AscendC::LocalTensor<float>& buf)
    {
        AscendC::DataCopy(buf, xGm_[gmOffset], len);
    }

    __aicore__ inline void CopyOut(uint64_t gmOffset, uint32_t len, AscendC::LocalTensor<float>& buf)
    {
        AscendC::DataCopy(yGm_[gmOffset], buf, len);
    }

    __aicore__ inline void ComputeRow(uint32_t row)
    {
        const uint64_t rowBase = static_cast<uint64_t>(row) * static_cast<uint64_t>(lastDim_);
        const uint32_t numSegs = (tileInner_ == 0) ? 0u : ((lastDim_ + tileInner_ - 1u) / tileInner_);
        float carry = 0.0f;

        for (int32_t segIdx = static_cast<int32_t>(numSegs) - 1; segIdx >= 0; --segIdx) {
            const uint32_t start = static_cast<uint32_t>(segIdx) * tileInner_;
            uint32_t len = lastDim_ - start;
            if (len > tileInner_) len = tileInner_;
            const uint64_t gmOffset = rowBase + start;

            AscendC::LocalTensor<float> seg = calcQueue_.AllocTensor<float>();
            CopyIn(gmOffset, len, seg);

            // Reverse cumulative sum within this segment, adding carry from the right side.
            float acc = carry;
            // Scalar loop on local tensor
            for (int32_t i = static_cast<int32_t>(len) - 1; i >= 0; --i) {
                float v = seg.GetValue(i);
                acc += v;
                seg.SetValue(i, acc);
            }
            carry = acc;

            CopyOut(gmOffset, len, seg);
            calcQueue_.FreeTensor(seg);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_ {0};
    uint32_t lastDim_ {0};
    uint32_t tileInner_ {0};
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverseCustom op;
    op.Init(x, y,
            tiling_data.outerSize,
            tiling_data.lastDim,
            tiling_data.tileInner);
    op.Process();
}
