host_operator_src="""
#include "matmul_with_irregular_shapes_custom_tiling.h"
#include "register/op_def_registry.h"

#include "lib/matmul_tiling.h"
#include "tiling/platform/platform_ascendc.h"

namespace {

static inline uint64_t GetDimSafe(const gert::Shape* shape, size_t idx, uint64_t default_val = 1) {
    if (shape == nullptr || shape->GetDimNum() <= idx) return default_val;
    int64_t d = shape->GetDim(idx);
    return d > 0 ? static_cast<uint64_t>(d) : default_val;
}

} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    if (context == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);
    if (aShape == nullptr || bShape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t M = GetDimSafe(aShape, 0, 1);
    const uint64_t K_a = GetDimSafe(aShape, 1, 1);
    const uint64_t K_b = GetDimSafe(bShape, 0, 1);
    const uint64_t N = GetDimSafe(bShape, 1, 1);

    if (K_a != K_b) {
        return ge::GRAPH_FAILED;
    }

    platform_ascendc::PlatformAscendC ascendcPlatform(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling cubeTiling(ascendcPlatform);

    cubeTiling.SetAType(AscendC::TPosition::GM, CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    cubeTiling.SetBType(AscendC::TPosition::GM, CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    cubeTiling.SetCType(AscendC::TPosition::GM, CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    cubeTiling.SetBiasType(AscendC::TPosition::GM, CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);

    cubeTiling.SetShape(static_cast<uint32_t>(M), static_cast<uint32_t>(N), static_cast<uint32_t>(K_a));
    cubeTiling.SetOrgShape(static_cast<uint32_t>(M), static_cast<uint32_t>(N), static_cast<uint32_t>(K_a));
    cubeTiling.SetBufferSpace(-1, -1, -1);

    MatmulWithIrregularShapesCustomTilingData tiling;
    // Serialize matmul tiling into our tiling buffer
    if (cubeTiling.GetTiling(tiling.mutable_cubeTiling()) == -1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t blk = cubeTiling.GetBlockDim();
    tiling.set_blockDim(blk);
    tiling.set_m(static_cast<uint32_t>(M));
    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_k(static_cast<uint32_t>(K_a));
    tiling.set_cubeTilingSize(cubeTiling.GetDataSize());

    context->SetBlockDim(blk);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape* aShape = context->GetInputShape(0);
    const gert::Shape* bShape = context->GetInputShape(1);
    if (aShape == nullptr || bShape == nullptr || aShape->GetDimNum() != 2 || bShape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = aShape->GetDim(0);
    const int64_t K1 = aShape->GetDim(1);
    const int64_t K2 = bShape->GetDim(0);
    const int64_t N = bShape->GetDim(1);
    if (K1 != K2) {
        return GRAPH_FAILED;
    }
    gert::Shape* cShape = context->GetOutputShape(0);
    cShape->SetDimNum(2);
    cShape->SetDim(0, M);
    cShape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""
#define ASCENDC_CUBE_ONLY
#include "lib/matmul_intf.h"
#include "kernel_operator.h"

using AscendC::TPosition;
using AscendC::GlobalTensor;

// Define Matmul data types and positions
typedef AscendC::MatmulType<TPosition::GM, CubeFormat::ND, float> aType;
typedef AscendC::MatmulType<TPosition::GM, CubeFormat::ND, float> bType;
typedef AscendC::MatmulType<TPosition::GM, CubeFormat::ND, float> cType;

class KernelMatmulWithIrregularShapesCustom {
public:
    __aicore__ inline KernelMatmulWithIrregularShapesCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                __gm__ uint8_t* tilingBuf, uint32_t tilingSize)
    {
        aGm_.SetGlobalBuffer((__gm__ float*)a, 0);
        bGm_.SetGlobalBuffer((__gm__ float*)b, 0);
        cGm_.SetGlobalBuffer((__gm__ float*)c, 0);

        // Initialize Matmul operator with tiling
        mm_.SetTilingData(tilingBuf, tilingSize);
        mm_.BindA(aGm_);
        mm_.BindB(bGm_);
        mm_.BindC(cGm_);
    }

    __aicore__ inline void Process()
    {
        mm_.Run();
    }

private:
    GlobalTensor<float> aGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> cGm_;

    AscendC::Matmul<aType, bType, cType> mm_;
};

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelMatmulWithIrregularShapesCustom op;
    op.Init(A, B, C,
            (__gm__ uint8_t*)tiling_data.cubeTiling, tiling_data.cubeTilingSize);
    op.Process();
}

    return C;
