kernel_src="""
#include "kernel_operator.h"

class KernelConvTransposed3dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Din, uint32_t Hin, uint32_t Win,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Dout, uint32_t Hout, uint32_t Wout)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Din_ = Din; Hin_ = Hin; Win_ = Win;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        Dout_ = Dout; Hout_ = Hout; Wout_ = Wout;

        xTotal_ = static_cast<uint64_t>(N_) * Cin_ * Din_ * Hin_ * Win_;
        wTotal_ = static_cast<uint64_t>(Cin_) * Cout_ * Kd_ * Kh_ * Kw_;
        yTotal_ = static_cast<uint64_t>(N_) * Cout_ * Dout_ * Hout_ * Wout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xTotal_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wTotal_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yTotal_);

        // Tiny scalar queues to implement a correct, general fallback.
        pipe_.InitBuffer(xQ_, 1, sizeof(float));
        pipe_.InitBuffer(wQ_, 1, sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, sizeof(float));
        pipe_.InitBuffer(yQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (yTotal_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over flattened output elements
        for (uint64_t idx = blockIdx; idx < yTotal_; idx += blockNum) {
            ComputeOneOutput(idx);
        }
    }

private:
    __aicore__ inline void ComputeOneOutput(uint64_t yIndex)
    {
        // Decode yIndex into (n, cout, d, h, w)
        uint64_t tmp = yIndex;

        const uint32_t w = static_cast<uint32_t>(tmp % Wout_); tmp /= Wout_;
        const uint32_t h = static_cast<uint32_t>(tmp % Hout_); tmp /= Hout_;
        const uint32_t d = static_cast<uint32_t>(tmp % Dout_); tmp /= Dout_;
        const uint32_t cout = static_cast<uint32_t>(tmp % Cout_); tmp /= Cout_;
        const uint32_t n = static_cast<uint32_t>(tmp);

        AscendC::LocalTensor<float> yLocal = yQ_.AllocTensor<float>();
        AscendC::Duplicate(yLocal, 0.0f, 1);

        // Accumulate over cin and kernel window
        for (uint32_t cin = 0; cin < Cin_; ++cin) {
            for (uint32_t kd = 0; kd < Kd_; ++kd) {
                int64_t id = static_cast<int64_t>(d) - static_cast<int64_t>(kd);
                if (id < 0 || id >= static_cast<int64_t>(Din_)) continue;

                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    int64_t ih = static_cast<int64_t>(h) - static_cast<int64_t>(kh);
                    if (ih < 0 || ih >= static_cast<int64_t>(Hin_)) continue;

                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        int64_t iw = static_cast<int64_t>(w) - static_cast<int64_t>(kw);
                        if (iw < 0 || iw >= static_cast<int64_t>(Win_)) continue;

                        // x offset: [N, Cin, Din, Hin, Win]
                        uint64_t xOff = (
                            (((static_cast<uint64_t>(n) * Cin_ + cin) * Din_ + static_cast<uint32_t>(id)) * Hin_
                              + static_cast<uint32_t>(ih)) * Win_ + static_cast<uint32_t>(iw)
                        );

                        // w offset: [Cin, Cout, Kd, Kh, Kw]
                        uint64_t wOff = (
                            (((static_cast<uint64_t>(cin) * Cout_ + cout) * Kd_ + kd) * Kh_ + kh) * Kw_ + kw
                        );

                        AscendC::LocalTensor<float> xLocal = xQ_.AllocTensor<float>();
                        AscendC::DataCopy(xLocal, xGm_[xOff], 1);

                        AscendC::LocalTensor<float> wLocal = wQ_.AllocTensor<float>();
                        AscendC::DataCopy(wLocal, wGm_[wOff], 1);

                        AscendC::LocalTensor<float> tLocal = tmpQ_.AllocTensor<float>();
                        AscendC::Mul(tLocal, xLocal, wLocal, 1);

                        wQ_.FreeTensor(wLocal);
                        xQ_.FreeTensor(xLocal);

                        AscendC::Add(yLocal, yLocal, tLocal, 1);
                        tmpQ_.FreeTensor(tLocal);
                    }
                }
            }
        }

        // Store result
        AscendC::DataCopy(yGm_[yIndex], yLocal, 1);
        yQ_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Din_{0}, Hin_{0}, Win_{0};
    uint32_t Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};
    uint64_t xTotal_{0}, wTotal_{0}, yTotal_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dAsymmetricInputSquareKernelCustom op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.din, tiling_data.hin, tiling_data.win,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.dout, tiling_data.hout, tiling_data.wout);
    op.Process();
}
