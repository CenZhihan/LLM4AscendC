static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous

namespace optiling {

// Conservative UB budget per-core
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // 192KB

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dAsymmetricInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect NCHW for x and OIHW for weight
    const uint32_t N  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Ci = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Co = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Ci_w = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t Kh = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetDim(3));

    // Basic checks
    if (Ci != Ci_w) {
        return ge::GRAPH_FAILED;
    }

    // stride=(1,1), padding=(0,0), dilation=(1,1)
    const uint32_t Hout = (H - Kh) + 1;
    const uint32_t Wout = (W - Kw) + 1;

    // Determine tileCout based on UB budget
    // Buffers (floats):
    //   weightsTrans: Ct * K
    //   weightsTmp: K
    //   yVec: Ct
    //   tmpVec: Ct
    //   patchRow: Kw
    // Total floats ≈ Ct*K + K + 2*Ct + Kw
    const uint32_t K = Ci * Kh * Kw;
    const uint32_t budgetFloats = UB_BUDGET_BYTES / 4;

    uint32_t maxCt = 1;
    if (K > 0) {
        // Ct <= (budget - (K + Kw)) / (K + 2)
        uint32_t numer = (budgetFloats > (K + Kw)) ? (budgetFloats - (K + Kw)) : 0;
        uint32_t denom = (K + 2);
        maxCt = (denom > 0) ? (numer / denom) : 1;
        if (maxCt == 0) maxCt = 1;
    }
    if (maxCt > 32u) maxCt = 32u;         // keep conservative
    if (maxCt > Co) maxCt = Co;
    // Align Ct down to a friendly factor
    const uint32_t ALIGN_CT = 8u;
    uint32_t tileCout = AlignDown(maxCt, ALIGN_CT);
    if (tileCout == 0) tileCout = (Co < ALIGN_CT ? Co : ALIGN_CT);
    if (tileCout > Co) tileCout = Co;

    // Block dim: distribute over output positions
    const uint64_t totalPos = static_cast<uint64_t>(N) * static_cast<uint64_t>(Hout) * static_cast<uint64_t>(Wout);
    uint32_t blockDim = static_cast<uint32_t>(totalPos);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cin(Ci);
    tiling.set_cout(Co);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_kh(Kh);
    tiling.set_kw(Kw);
    tiling.set_hout(Hout);
    tiling.set_wout(Wout);
    tiling.set_tileCout(tileCout);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t wIn,
                                uint32_t kh, uint32_t kw,
                                uint32_t hout, uint32_t wout,
                                uint32_t tileCout)
    {
        n_ = n;
        cin_ = cin;
        cout_ = cout;
        h_ = h;
        w_ = wIn;
        kh_ = kh;
        kw_ = kw;
        hout_ = hout;
        wout_ = wout;
        kSize_ = cin_ * kh_ * kw_;
        tileCout_ = tileCout;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cin_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, cout_ * kSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * cout_ * hout_ * wout_);

        // Init UB buffers according to tiling
        pipe_.InitBuffer(wTmpQ_, 1, kSize_ * sizeof(float));
        pipe_.InitBuffer(wTransQ_, 1, tileCout_ * kSize_ * sizeof(float));
        pipe_.InitBuffer(patchQ_, 1, kw_ * sizeof(float));
        pipe_.InitBuffer(yQ_, 2, tileCout_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tileCout_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, 2, sizeof(float)); // 1-element buffers for scalar ops
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || kSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalPos = static_cast<uint64_t>(n_) * static_cast<uint64_t>(hout_) * static_cast<uint64_t>(wout_);

        for (uint32_t ocStart = 0; ocStart < cout_; ocStart += tileCout_) {
            const uint32_t curCt = AscendC::min(tileCout_, cout_ - ocStart);

            AscendC::LocalTensor<float> wTrans = wTransQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> wTmp = wTmpQ_.AllocTensor<float>();

            // Build transposed weight: wTrans[k * curCt + ct] = weight[ocStart+ct][k]
            // First, copy each oc's K block to wTmp, then scatter into wTrans
            for (uint32_t ct = 0; ct < curCt; ++ct) {
                const uint32_t oc = ocStart + ct;
                const uint32_t wOff = oc * kSize_;
                AscendC::DataCopy(wTmp, wGm_[wOff], kSize_);
                // Scatter into transposed layout
                for (uint32_t k = 0; k < kSize_; ++k) {
                    // wTrans[k*curCt + ct] = wTmp[k]
                    AscendC::DataCopy(wTrans[k * tileCout_ + ct], wTmp[k], 1);
                }
            }

            // Iterate positions in a grid-stride manner
            for (uint64_t pos = blockIdx; pos < totalPos; pos += blockNum) {
                // Decode pos -> n, hOut, wOut
                uint64_t tmp = pos;
                const uint32_t nIdx = static_cast<uint32_t>(tmp / (static_cast<uint64_t>(hout_) * static_cast<uint64_t>(wout_)));
                tmp = tmp % (static_cast<uint64_t>(hout_) * static_cast<uint64_t>(wout_));
                const uint32_t hOut = static_cast<uint32_t>(tmp / wout_);
                const uint32_t wOut = static_cast<uint32_t>(tmp % wout_);

                // yVec initialized to zeros
                AscendC::LocalTensor<float> yVec = yQ_.AllocTensor<float>();
                AscendC::Muls(yVec, yVec, 0.0f, curCt);

                AscendC::LocalTensor<float> patchRow = patchQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpVec = tmpQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> pScalar = scalarQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> wScalar = scalarQ_.AllocTensor<float>(); // reuse second slot

                for (uint32_t c = 0; c < cin_; ++c) {
                    for (uint32_t kh = 0; kh < kh_; ++kh) {
                        // Copy a contiguous row of length kw_ from input for this (nIdx, c, hOut+kh, wOut..wOut+kw-1)
                        const uint32_t inOff = (((nIdx * cin_ + c) * h_ + (hOut + kh)) * w_) + wOut;
                        AscendC::DataCopy(patchRow, xGm_[inOff], kw_);

                        for (uint32_t kw = 0; kw < kw_; ++kw) {
                            // kIndex for this (c, kh, kw)
                            const uint32_t kIndex = ((c * kh_) + kh) * kw_ + kw;

                            // Load patch scalar p into pScalar[0]
                            AscendC::DataCopy(pScalar, patchRow[kw], 1);

                            // Compute accumulation: yVec += p * wVec(kIndex)
                            // Load contiguous weight vector of length curCt for this kIndex
                            AscendC::DataCopy(tmpVec, wTrans[kIndex * tileCout_], curCt);

                            // Multiply by scalar p (length-1), fallback to len-1 operations per channel if necessary
                            // Here we use scalar-by-scalar per channel accumulation to avoid needing an immediate scalar
                            for (uint32_t ct = 0; ct < curCt; ++ct) {
                                // wScalar = tmpVec[ct]
                                AscendC::DataCopy(wScalar, tmpVec[ct], 1);
                                // tmp = p * wScalar (len 1)
                                AscendC::Mul(wScalar, wScalar, pScalar, 1);
                                // yVec[ct] += tmp (len 1)
                                AscendC::Add(yVec[ct], yVec[ct], wScalar, 1);
                            }
                        }
                    }
                }

                // Write back yVec to output GM for this position across curCt output channels
                for (uint32_t ct = 0; ct < curCt; ++ct) {
                    const uint32_t oc = ocStart + ct;
                    const uint32_t yOff = (((nIdx * cout_ + oc) * hout_ + hOut) * wout_) + wOut;
                    AscendC::DataCopy(yGm_[yOff], yVec[ct], 1);
                }

                scalarQ_.FreeTensor(wScalar);
                scalarQ_.FreeTensor(pScalar);
                tmpQ_.FreeTensor(tmpVec);
                patchQ_.FreeTensor(patchRow);
                yQ_.FreeTensor(yVec);
            }

            wTmpQ_.FreeTensor(wTmp);
            wTransQ_.FreeTensor(wTrans);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wTmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wTransQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> patchQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> yQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t h_{0}, w_{0};
    uint32_t kh_{0}, kw_{0};
    uint32_t hout_{0}, wout_{0};
    uint32_t kSize_{0};
    uint32_t tileCout_{1};
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w,
            tiling_data.kh, tiling_data.kw,
            tiling_data.hout, tiling_data.wout,
            tiling_data.tileCout);
    op.Process();
}

static inline at::Tensor conv2d_valid_output_like(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, C, H, W], w: [Co, C, Kh, Kw]
    auto sizes = x.sizes();
    int64_t N = sizes[0];
    int64_t C = sizes[1];
    int64_t H = sizes[2];
    int64_t W = sizes[3];
    int64_t Co = w.size(0);
    int64_t Ci = w.size(1);
    int64_t Kh = w.size(2);
    int64_t Kw = w.size(3);

    int64_t Hout = H - Kh + 1;
    int64_t Wout = W - Kw + 1;
    TORCH_CHECK(Hout > 0 && Wout > 0, "Invalid output size for given kernel");
    return at::empty({N, Co, Hout, Wout}, x.options());
}
