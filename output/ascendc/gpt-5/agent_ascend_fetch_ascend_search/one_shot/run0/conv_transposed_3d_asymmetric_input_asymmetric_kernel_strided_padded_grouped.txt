static ge::graphStatus InferShape(gert::InferShapeContext *context)

    const int64_t groups = *context->GetAttrs()->GetAttrPointer<int64_t>(3);
    const int64_t *dilation = context->GetAttrs()->GetAttrPointer<int64_t>(4);

    const int64_t n  = x_shape->GetDim(0);
    const int64_t cI = x_shape->GetDim(1);
    const int64_t dI = x_shape->GetDim(2);
    const int64_t hI = x_shape->GetDim(3);
    const int64_t wI = x_shape->GetDim(4);

    const int64_t cW1 = w_shape->GetDim(1);
    const int64_t kD  = w_shape->GetDim(2);
    const int64_t kH  = w_shape->GetDim(3);
    const int64_t kW  = w_shape->GetDim(4);

    const int64_t cO = cW1 * groups;

    const int64_t dO = (dI - 1) * stride[0] - 2 * padding[0] + dilation[0] * (kD - 1) + outpad[0] + 1;
    const int64_t hO = (hI - 1) * stride[1] - 2 * padding[1] + dilation[1] * (kH - 1) + outpad[1] + 1;
    const int64_t wO = (wI - 1) * stride[2] - 2 * padding[2] + dilation[2] * (kW - 1) + outpad[2] + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, cO);
    y_shape->SetDim(2, dO);
    y_shape->SetDim(3, hO);
    y_shape->SetDim(4, wO);

    return GRAPH_SUCCESS;
}
