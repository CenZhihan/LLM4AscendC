{

    const gert::Shape *w_shape = context->GetInputShape(1); // [Cout, Cin, Kh, Kw]

    const int64_t W   = x_shape->GetDim(3);

    const int64_t Cout = w_shape->GetDim(0);
    const int64_t Kh   = w_shape->GetDim(2);
    const int64_t Kw   = w_shape->GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *poolK_ptr = attrs->GetAttrPointer<int64_t>(2);
    const int64_t poolK = *poolK_ptr;

    const int64_t Hc = H - Kh + 1;
    const int64_t Wc = W - Kw + 1;

    int64_t Hp = 0;
    int64_t Wp = 0;
    if (Hc >= poolK && Wc >= poolK && poolK > 0) {
        Hp = (Hc - poolK) / poolK + 1;
        Wp = (Wc - poolK) / poolK + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Hp);
    y_shape->SetDim(3, Wp);
    return GRAPH_SUCCESS;
}

#include "kernel_operator.h"
#include <math.h>

class KernelConv2dSubtractTanhSubtractAvgPoolCustom {
public:
    __aicore__ inline KernelConv2dSubtractTanhSubtractAvgPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t Kh, uint32_t Kw,
                                uint32_t poolK, bool hasBias,
                                float sub1, float sub2)
    {
        xGm_ = (__gm__ float*)x;
        wGm_ = (__gm__ float*)w;
        bGm_ = (__gm__ float*)b;
        yGm_ = (__gm__ float*)y;

        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; Kh_ = Kh; Kw_ = Kw;
        poolK_ = poolK;
        hasBias_ = hasBias;
        sub1_ = sub1;
        sub2_ = sub2;
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0) return;
        if (H_ < Kh_ || W_ < Kw_ || poolK_ == 0) return;

        const int32_t Hc = static_cast<int32_t>(H_ - Kh_ + 1);
        const int32_t Wc = static_cast<int32_t>(W_ - Kw_ + 1);
        if (Hc < static_cast<int32_t>(poolK_) || Wc < static_cast<int32_t>(poolK_)) return;

        const int32_t Hp = static_cast<int32_t>((Hc - poolK_) / poolK_ + 1);
        const int32_t Wp = static_cast<int32_t>((Wc - poolK_) / poolK_ + 1);

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        // Parallelize across output channels using grid-stride
        for (uint32_t ocBase = blockIdx; ocBase < Cout_; ocBase += blockNum) {
            const uint32_t oc = ocBase;

            for (uint32_t n = 0; n < N_; ++n) {
                for (int32_t ohp = 0; ohp < Hp; ++ohp) {
                    for (int32_t owp = 0; owp < Wp; ++owp) {

                        float poolSum = 0.0f;
                        for (uint32_t ph = 0; ph < poolK_; ++ph) {
                            for (uint32_t pw = 0; pw < poolK_; ++pw) {
                                const int32_t oh = ohp * (int32_t)poolK_ + (int32_t)ph;
                                const int32_t ow = owp * (int32_t)poolK_ + (int32_t)pw;

                                float acc = hasBias_ ? bGm_[oc] : 0.0f;

                                // Convolution at (n, oc, oh, ow)
                                for (uint32_t ic = 0; ic < Cin_; ++ic) {
                                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                                        const int32_t iy = oh + (int32_t)kh;
                                        const int32_t xRowBase = iy * (int32_t)W_;
                                        const uint32_t wRowBase = (((oc * Cin_) + ic) * Kh_ + kh) * Kw_;
                                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                            const int32_t ix = ow + (int32_t)kw;
                                            const int64_t xIndex =
                                                (((int64_t)n * Cin_ + ic) * H_ + (uint32_t)iy) * W_ + (uint32_t)ix;
                                            const int64_t wIndex =
                                                wRowBase + kw + 0; // expanded above
                                            const float xv = xGm_[xIndex];
                                            const float wv = wGm_[wIndex];
                                            acc += xv * wv;
                                        }
                                    }
                                }

                                // Fused post operations at conv output
                                float after = acc - sub1_;
                                float t = tanhf(after);
                                t = t - sub2_;
                                poolSum += t;
                            }
                        }

                        const float invCount = 1.0f / (float)(poolK_ * poolK_);
                        const float yv = poolSum * invCount;

                        const int64_t yIndex =
                            (((int64_t)n * Cout_ + oc) * (int64_t)Hp + (uint32_t)ohp) * (int64_t)Wp + (uint32_t)owp;
                        yGm_[yIndex] = yv;
                    }
                }
            }
        }
    }

private:
    __gm__ float* xGm_{nullptr};
    __gm__ float* wGm_{nullptr};
    __gm__ float* bGm_{nullptr};
    __gm__ float* yGm_{nullptr};

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, Kh_{0}, Kw_{0};
    uint32_t poolK_{0};
    bool hasBias_{false};
    float sub1_{0.0f}, sub2_{0.0f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractTanhSubtractAvgPoolCustom op;
    const bool hasBias = (tiling_data.hasBias != 0u);
    op.Init(x, weight, conv_bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.Kh, tiling_data.Kw,
            tiling_data.poolK, hasBias,
            tiling_data.subtract1, tiling_data.subtract2);
    op.Process();
}
