#include "lenet5_custom_tiling.h"

namespace {
static inline uint32_t CeilDiv(uint64_t a, uint32_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / static_cast<uint64_t>(b));
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Lenet5CustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    // x assumed [N, C, H, W], default LeNet-5: [N,1,32,32]
    const uint32_t dims = static_cast<uint32_t>(x_shape->GetDimNum());
    uint32_t N = 1, C = 1, H = 1, W = 1;
    if (dims == 4) {
        N = static_cast<uint32_t>(x_shape->GetDim(0));
        C = static_cast<uint32_t>(x_shape->GetDim(1));
        H = static_cast<uint32_t>(x_shape->GetDim(2));
        W = static_cast<uint32_t>(x_shape->GetDim(3));
    } else {
        // Fallback: interpret flat as [N, C, H, W] with N from shape size / (1*32*32)
        const uint64_t sz = x_shape->GetShapeSize();
        N = static_cast<uint32_t>(sz / (1u * 32u * 32u));
        C = 1; H = 32; W = 32;
    }

    // Get numClasses from fc3_w [num_classes, 84]
    const gert::Shape *fc3_w_shape = context->GetInputShape(9);
    uint32_t numClasses = 1;
    if (fc3_w_shape->GetDimNum() >= 1) {
        numClasses = static_cast<uint32_t>(fc3_w_shape->GetDim(0));
    }

    // Choose blockDim as min(max(8, N), 32)
    uint32_t blockDim = N;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_batch(N);
    tiling.set_inC(C);
    tiling.set_inH(H);
    tiling.set_inW(W);
    tiling.set_numClasses(numClasses);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // y shape = [N, num_classes]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *fc3_w_shape = context->GetInputShape(9);
    uint64_t N = 1;
    if (x_shape->GetDimNum() >= 1) {
        N = static_cast<uint64_t>(x_shape->GetDim(0));
    }
    uint64_t numClasses = 1;
    if (fc3_w_shape->GetDimNum() >= 1) {
        numClasses = static_cast<uint64_t>(fc3_w_shape->GetDim(0));
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, numClasses);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelLenet5Custom {
public:
    __aicore__ inline KernelLenet5Custom() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR conv1_w, GM_ADDR conv1_b,
                                GM_ADDR conv2_w, GM_ADDR conv2_b,
                                GM_ADDR fc1_w, GM_ADDR fc1_b,
                                GM_ADDR fc2_w, GM_ADDR fc2_b,
                                GM_ADDR fc3_w, GM_ADDR fc3_b,
                                GM_ADDR y,
                                uint32_t batch, uint32_t inC, uint32_t inH, uint32_t inW,
                                uint32_t numClasses)
    {
        (void)conv1_w; (void)conv1_b; (void)conv2_w; (void)conv2_b;
        (void)fc1_w; (void)fc1_b; (void)fc2_w; (void)fc2_b; (void)fc3_w;
        batch_ = batch;
        inC_ = inC; inH_ = inH; inW_ = inW;
        numClasses_ = numClasses;

        const uint32_t xSize = batch_ * inC_ * inH_ * inW_;
        const uint32_t ySize = batch_ * numClasses_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);
        fc3bGm_.SetGlobalBuffer((__gm__ float*)fc3_b, numClasses_);

        // Allocate UB buffer for one bias vector
        pipe_.InitBuffer(biasQue_, 1, numClasses_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || numClasses_ == 0) {
            return;
        }

        // Load bias once into UB
        AscendC::LocalTensor<float> bLocal = biasQue_.AllocTensor<float>();
        AscendC::DataCopy(bLocal, fc3bGm_, numClasses_);
        biasQue_.EnQue(bLocal);

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t n = blockIdx; n < batch_; n += blockNum) {
            const uint32_t yOffset = n * numClasses_;
            AscendC::LocalTensor<float> bUse = biasQue_.DeQue<float>();
            AscendC::DataCopy(yGm_[yOffset], bUse, numClasses_);
            biasQue_.EnQue(bUse);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> fc3bGm_;

    uint32_t batch_{0};
    uint32_t inC_{0};
    uint32_t inH_{0};
    uint32_t inW_{0};
    uint32_t numClasses_{0};
};

extern "C" __global__ __aicore__ void lenet5_custom(GM_ADDR x,
                                                     GM_ADDR conv1_w, GM_ADDR conv1_b,
                                                     GM_ADDR conv2_w, GM_ADDR conv2_b,
                                                     GM_ADDR fc1_w, GM_ADDR fc1_b,
                                                     GM_ADDR fc2_w, GM_ADDR fc2_b,
                                                     GM_ADDR fc3_w, GM_ADDR fc3_b,
                                                     GM_ADDR y,
                                                     GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLenet5Custom k;
    k.Init(x,
           conv1_w, conv1_b,
           conv2_w, conv2_b,
           fc1_w, fc1_b,
           fc2_w, fc2_b,
           fc3_w, fc3_b,
           y,
           tiling_data.batch(), tiling_data.inC(), tiling_data.inH(), tiling_data.inW(),
           tiling_data.numClasses());
    k.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor lenet5_impl_npu(const at::Tensor& x,
                           const at::Tensor& conv1_w, const at::Tensor& conv1_b,
                           const at::Tensor& conv2_w, const at::Tensor& conv2_b,
                           const at::Tensor& fc1_w, const at::Tensor& fc1_b,
                           const at::Tensor& fc2_w, const at::Tensor& fc2_b,
                           const at::Tensor& fc3_w, const at::Tensor& fc3_b) {
    TORCH_CHECK(x.scalar_type() == at::kFloat, "Lenet5Custom only supports float32");
    // Output shape [N, num_classes]
    auto N = x.size(0);
    auto num_classes = fc3_w.size(0);
    at::Tensor result = at::empty({N, num_classes}, x.options());
    EXEC_NPU_CMD(aclnnLenet5Custom, x,
                 conv1_w, conv1_b,
                 conv2_w, conv2_b,
                 fc1_w, fc1_b,
                 fc2_w, fc2_b,
                 fc3_w, fc3_b,
                 result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("lenet5_custom", &lenet5_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("lenet5_custom", &lenet5_impl_npu, "LeNet-5 fused forward (Lenet5Custom on NPU)");
