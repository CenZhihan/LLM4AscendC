kernel_src="""

#include "kernel_operator.h"

class KernelConvDepthwise2dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t wd,
                                uint32_t kH, uint32_t kW,
                                uint32_t outH, uint32_t outW)
    {
        N_ = n;
        C_ = c;
        H_ = h;
        W_ = wd;
        kH_ = kH;
        kW_ = kW;
        outH_ = outH;
        outW_ = outW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(N_ * C_ * H_ * W_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(C_ * kH_ * kW_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(N_ * C_ * outH_ * outW_));

        // UB buffers
        // Three input rows of width W_, one output row of width outW_, and one temp buffer of width outW_.
        pipe_.InitBuffer(rowBufQ_, 3, W_ * sizeof(float));
        pipe_.InitBuffer(outQ_, 1, outW_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, outW_ * sizeof(float));
        pipe_.InitBuffer(weightQ_, 1, kH_ * kW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalMaps = N_ * C_;
        for (uint32_t mapIdx = blockIdx; mapIdx < totalMaps; mapIdx += blockNum) {
            const uint32_t n = mapIdx / C_;
            const uint32_t c = mapIdx % C_;

            // Load 3x3 weight for this channel c
            AscendC::LocalTensor<float> wLocal = weightQ_.AllocTensor<float>();
            const uint32_t wOffset = (c * kH_ * kW_);
            AscendC::DataCopy(wLocal, wGm_[wOffset], kH_ * kW_);
            float w00 = wLocal.GetValue(0);
            float w01 = wLocal.GetValue(1);
            float w02 = wLocal.GetValue(2);
            float w10 = wLocal.GetValue(3);
            float w11 = wLocal.GetValue(4);
            float w12 = wLocal.GetValue(5);
            float w20 = wLocal.GetValue(6);
            float w21 = wLocal.GetValue(7);
            float w22 = wLocal.GetValue(8);
            weightQ_.FreeTensor(wLocal);

            // Iterate output rows
            for (uint32_t oh = 0; oh < outH_; ++oh) {
                // Load three consecutive input rows: r, r+1, r+2
                const uint32_t inRow0 = oh;
                const uint32_t inRow1 = oh + 1;
                const uint32_t inRow2 = oh + 2;

                AscendC::LocalTensor<float> row0 = rowBufQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> row1 = rowBufQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> row2 = rowBufQ_.AllocTensor<float>();

                uint64_t baseX = (((uint64_t)n * C_ + c) * H_) * W_;
                AscendC::DataCopy(row0, xGm_[baseX + inRow0 * W_], W_);
                AscendC::DataCopy(row1, xGm_[baseX + inRow1 * W_], W_);
                AscendC::DataCopy(row2, xGm_[baseX + inRow2 * W_], W_);

                AscendC::LocalTensor<float> outVec = outQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpVec = tmpQ_.AllocTensor<float>();

                // Slices: left/mid/right as sub-vectors using offset operator[]
                AscendC::LocalTensor<float> r0_l = row0[0];
                AscendC::LocalTensor<float> r0_m = row0[1];
                AscendC::LocalTensor<float> r0_r = row0[2];
                AscendC::LocalTensor<float> r1_l = row1[0];
                AscendC::LocalTensor<float> r1_m = row1[1];
                AscendC::LocalTensor<float> r1_r = row1[2];
                AscendC::LocalTensor<float> r2_l = row2[0];
                AscendC::LocalTensor<float> r2_m = row2[1];
                AscendC::LocalTensor<float> r2_r = row2[2];

                // Ensure sub-vectors length set to outW_
                r0_l.SetSize(outW_); r0_m.SetSize(outW_); r0_r.SetSize(outW_);
                r1_l.SetSize(outW_); r1_m.SetSize(outW_); r1_r.SetSize(outW_);
                r2_l.SetSize(outW_); r2_m.SetSize(outW_); r2_r.SetSize(outW_);
                outVec.SetSize(outW_); tmpVec.SetSize(outW_);

                // out = r0_l*w00 + r0_m*w01 + r0_r*w02
                AscendC::Muls(outVec, r0_l, w00, outW_);
                AscendC::Muls(tmpVec, r0_m, w01, outW_);
                AscendC::Add(outVec, outVec, tmpVec, outW_);
                AscendC::Muls(tmpVec, r0_r, w02, outW_);
                AscendC::Add(outVec, outVec, tmpVec, outW_);

                // + r1_l*w10 + r1_m*w11 + r1_r*w12
                AscendC::Muls(tmpVec, r1_l, w10, outW_);
                AscendC::Add(outVec, outVec, tmpVec, outW_);
                AscendC::Muls(tmpVec, r1_m, w11, outW_);
                AscendC::Add(outVec, outVec, tmpVec, outW_);
                AscendC::Muls(tmpVec, r1_r, w12, outW_);
                AscendC::Add(outVec, outVec, tmpVec, outW_);

                // + r2_l*w20 + r2_m*w21 + r2_r*w22
                AscendC::Muls(tmpVec, r2_l, w20, outW_);
                AscendC::Add(outVec, outVec, tmpVec, outW_);
                AscendC::Muls(tmpVec, r2_m, w21, outW_);
                AscendC::Add(outVec, outVec, tmpVec, outW_);
                AscendC::Muls(tmpVec, r2_r, w22, outW_);
                AscendC::Add(outVec, outVec, tmpVec, outW_);

                // Store result row
                uint64_t baseY = (((uint64_t)n * C_ + c) * outH_) * outW_;
                AscendC::DataCopy(yGm_[baseY + static_cast<uint64_t>(oh) * outW_], outVec, outW_);

                tmpQ_.FreeTensor(tmpVec);
                outQ_.FreeTensor(outVec);
                rowBufQ_.FreeTensor(row2);
                rowBufQ_.FreeTensor(row1);
                rowBufQ_.FreeTensor(row0);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 3> rowBufQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t kH_{0}, kW_{0};
    uint32_t outH_{0}, outW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymmetricInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kH, tiling_data.kW,
            tiling_data.outH, tiling_data.outW);
    op.Process();
}
