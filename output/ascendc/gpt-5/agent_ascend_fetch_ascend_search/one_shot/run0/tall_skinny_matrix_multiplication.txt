#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

// UB budget and alignment policies
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core UB share
static constexpr uint32_t ALIGN_ELEMS = 16;              // for vector alignment
static constexpr uint32_t MIN_TILE_M = 64;
static constexpr uint32_t MAX_BLOCKS = 24;
static constexpr uint32_t MIN_BLOCKS = 8;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    // Expect A: [M, K], B: [K, N]
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t M64 = a_shape->GetOriginShape().GetDim(0);
    const uint64_t K64 = a_shape->GetOriginShape().GetDim(1);
    const uint64_t Kb64 = b_shape->GetOriginShape().GetDim(0);
    const uint64_t N64 = b_shape->GetOriginShape().GetDim(1);

    if (K64 != Kb64) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(M64);
    const uint32_t K = static_cast<uint32_t>(K64);
    const uint32_t N = static_cast<uint32_t>(N64);

    // Choose tileN favoring small N (tall-skinny), align to 16
    uint32_t tileN = (N <= 64u) ? N : 64u;
    tileN = AlignDown(tileN, ALIGN_ELEMS);
    if (tileN == 0u) tileN = (N < ALIGN_ELEMS ? N : ALIGN_ELEMS);

    // Choose kTile moderately sized to enable reuse, align to 16
    uint32_t kTile = (K >= 128u) ? 128u : (K >= 64u ? 64u : K);
    kTile = AlignDown(kTile, ALIGN_ELEMS);
    if (kTile == 0u) kTile = (K < ALIGN_ELEMS ? K : ALIGN_ELEMS);

    // UB bytes usage: C(tileM*tileN) + B(kTile*tileN) + Atemp(kTile) + tmpRow(tileN)
    // Add safety factor (extra 4KB)
    uint32_t bytesPerElem = 4u;
    uint32_t overheadBytes = 4 * 1024;
    uint32_t fixedBytes = bytesPerElem * (kTile * tileN + kTile + tileN) + overheadBytes;

    uint32_t maxCMBytes = (UB_BUDGET_BYTES > fixedBytes) ? (UB_BUDGET_BYTES - fixedBytes) : 0u;
    uint32_t tileM = (tileN > 0 && bytesPerElem > 0) ? (maxCMBytes / (bytesPerElem * tileN)) : 0u;

    if (tileM < MIN_TILE_M) tileM = MIN_TILE_M;
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    if (tileM == 0u) tileM = ALIGN_ELEMS;
    if (tileM > M) tileM = AlignDown(M, ALIGN_ELEMS) > 0 ? AlignDown(M, ALIGN_ELEMS) : M;

    // BlockDim based on number of M tiles
    uint32_t mTiles = CeilDiv(M, tileM);
    uint32_t blockDim = mTiles;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    optiling::TallSkinnyMatrixMultiplicationCustomTiling tiling;
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_kTile(kTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t kTile)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN; kTile_ = kTile;

        aGm_.SetGlobalBuffer((__gm__ float*)a, (uint32_t)((uint64_t)M_ * (uint64_t)K_));
        bGm_.SetGlobalBuffer((__gm__ float*)b, (uint32_t)((uint64_t)K_ * (uint64_t)N_));
        cGm_.SetGlobalBuffer((__gm__ float*)c, (uint32_t)((uint64_t)M_ * (uint64_t)N_));

        // Buffers: B block (kTile * tileN), A chunk (kTile), C accum (tileM * tileN), temp row (tileN)
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, kTile_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(inQueueAChunk_, BUFFER_NUM, kTile_ * sizeof(float));
        pipe_.InitBuffer(accumQueue_, 1, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpRowQueue_, 1, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid stride over M tiles
        uint32_t strideM = blockNum * tileM_;
        for (uint32_t baseM = blockIdx * tileM_; baseM < M_; baseM += strideM) {
            uint32_t curM = M_ - baseM;
            if (curM > tileM_) curM = tileM_;

            for (uint32_t baseN = 0; baseN < N_; baseN += tileN_) {
                uint32_t curN = N_ - baseN;
                if (curN > tileN_) curN = tileN_;

                ComputeTile(baseM, curM, baseN, curN);
            }
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t baseM, uint32_t curM,
                                       uint32_t baseN, uint32_t curN)
    {
        // Allocate accumulation buffer and zero it
        AscendC::LocalTensor<float> cLocal = accumQueue_.AllocTensor<float>();
        AscendC::Dup(cLocal, 0.0f, curM * curN);

        // Iterate over K dimension with kTile
        for (uint32_t kBase = 0; kBase < K_; kBase += kTile_) {
            uint32_t curKTile = K_ - kBase;
            if (curKTile > kTile_) curKTile = kTile_;

            // Prefetch B block (curKTile x curN)
            LoadBBlock(kBase, curKTile, baseN, curN);

            // For each row in M tile
            for (uint32_t m = 0; m < curM; ++m) {
                // Load A chunk for this row: A[baseM + m, kBase : kBase + curKTile]
                AscendC::LocalTensor<float> aChunk = inQueueAChunk_.AllocTensor<float>();
                DataCopyAChunk(aChunk, baseM + m, kBase, curKTile);

                // Get cRow view
                AscendC::LocalTensor<float> cRow = cLocal[m * curN];

                // Accumulate: for t in curKTile: cRow += B_row_t * aVal
                AccumulateRow(cRow, curN, curKTile);

                inQueueAChunk_.FreeTensor(aChunk);
            }

            // Free the B block buffer after this kTile
            AscendC::LocalTensor<float> bBlock = inQueueB_.DeQue<float>();
            inQueueB_.FreeTensor(bBlock);
        }

        // Copy C tile back to GM with row-wise copies
        StoreCTile(cLocal, baseM, curM, baseN, curN);

        accumQueue_.FreeTensor(cLocal);
    }

    __aicore__ inline void LoadBBlock(uint32_t kBase, uint32_t curKTile,
                                      uint32_t baseN, uint32_t curN)
    {
        AscendC::LocalTensor<float> bBlock = inQueueB_.AllocTensor<float>();
        // Copy each row t of size curN from GM to local at offset t*tileN_
        for (uint32_t t = 0; t < curKTile; ++t) {
            uint32_t gmOffset = (kBase + t) * N_ + baseN;
            AscendC::LocalTensor<float> bRowLocal = bBlock[t * tileN_];
            AscendC::DataCopy(bRowLocal, bGm_[gmOffset], curN);
        }
        inQueueB_.EnQue(bBlock);
    }

    __aicore__ inline void DataCopyAChunk(AscendC::LocalTensor<float> &aChunk,
                                          uint32_t rowIdx, uint32_t kBase, uint32_t curKTile)
    {
        uint32_t gmOffset = rowIdx * K_ + kBase;
        AscendC::DataCopy(aChunk, aGm_[gmOffset], curKTile);
    }

    __aicore__ inline void AccumulateRow(AscendC::LocalTensor<float> &cRow, uint32_t curN, uint32_t curKTile)
    {
        // Read B block from queue (peek without removing)
        AscendC::LocalTensor<float> bBlock = inQueueB_.DeQue<float>();

        // Temporary row buffer
        AscendC::LocalTensor<float> tmpRow = tmpRowQueue_.AllocTensor<float>();

        // For each t in [0, curKTile)
        for (uint32_t t = 0; t < curKTile; ++t) {
            // Fetch scalar aVal from aChunk
            // Assume LocalTensor supports GetValue to read scalar at index t
            AscendC::LocalTensor<float> aChunk = inQueueAChunk_.GetCurTensor<float>();
            float aVal = aChunk.GetValue(t);

            // bRow local view
            AscendC::LocalTensor<float> bRow = bBlock[t * tileN_];

            // tmpRow = bRow * aVal
            AscendC::Muls(tmpRow, bRow, aVal, curN);
            // cRow += tmpRow
            AscendC::Add(cRow, cRow, tmpRow, curN);
        }

        tmpRowQueue_.FreeTensor(tmpRow);
        // Keep bBlock until curKTile loop ends; freed outside
    }

    __aicore__ inline void StoreCTile(AscendC::LocalTensor<float> &cLocal,
                                      uint32_t baseM, uint32_t curM,
                                      uint32_t baseN, uint32_t curN)
    {
        for (uint32_t m = 0; m < curM; ++m) {
            AscendC::LocalTensor<float> cRow = cLocal[m * curN];
            uint32_t gmOffset = (baseM + m) * N_ + baseN;
            AscendC::DataCopy(cGm_[gmOffset], cRow, curN);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAChunk_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accumQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpRowQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0}, N_ {0}, K_ {0};
    uint32_t tileM_ {0}, tileN_ {0}, kTile_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.kTile);
    op.Process();
}
