#include "gemm_scaling_hard_tanh_gelu_custom_tiling.h"

#include "tiling/platform/platform_ascendc.h"
#include "graph/tensor_desc.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// Post-matmul vector fusion UB model (per core):
//   in/out queues: 2 * tile
//   tmp queues: ~4 * tile (y^2, y^3, tmp, tanh)
// Total floats ~= 6 * tile => bytes ~= 24 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative
static constexpr uint32_t ALIGN_ELEMS     = 1024;         // DMA-friendly
static constexpr uint32_t HARD_CAP_ELEMS  = 32768;        // latency cap

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmScalingHardTanhGeluCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }
    const auto& x_os = x_shape->GetOriginShape();
    const auto& w_os = w_shape->GetOriginShape();

    // Expect x: [M, K], w: [N, K] (PyTorch linear uses w.T inside)
    if (x_os.GetDimNum() < 2 || w_os.GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_os.GetDim(x_os.GetDimNum() - 2));
    const uint32_t K = static_cast<uint32_t>(x_os.GetDim(x_os.GetDimNum() - 1));
    const uint32_t N = static_cast<uint32_t>(w_os.GetDim(w_os.GetDimNum() - 2));

    const uint32_t yLength = M * N;

    // Vector fusion tile from UB budget: bytes = 24 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 24;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (yLength < tileLen) {
        const uint32_t aligned = AlignDown(yLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : yLength;
    }

    // Choose blockDim based on number of vector tiles
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(yLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // Fetch attributes
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scaling = attrs->GetAttrPointer<float>(0);
    const float *htMin   = attrs->GetAttrPointer<float>(1);
    const float *htMax   = attrs->GetAttrPointer<float>(2);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_yLength(yLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_scaling((scaling != nullptr) ? *scaling : 1.0f);
    tiling.set_hardtanhMin((htMin != nullptr) ? *htMin : -2.0f);
    tiling.set_hardtanhMax((htMax != nullptr) ? *htMax : 2.0f);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // no additional workspace
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
// Matmul high-level API header (mixed mode: cube + vector)
#include "lib/matmul_intf.h"

constexpr int32_t BUFFER_NUM = 2;

typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> aType;
typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> bType;
typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> cType;
typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

class KernelGemmScalingHardTanhGeluCustom {
public:
    __aicore__ inline KernelGemmScalingHardTanhGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t yLength, uint32_t tileLength,
                                float scaling, float htMin, float htMax)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        yLen_ = yLength;
        tileLen_ = tileLength;

        scaling_ = scaling;
        htMin_ = htMin;
        htMax_ = htMax;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(M_ * K_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(N_ * K_)); // expects w:[N,K]
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yLen_);

        // UB queues for fused vector post-pass
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue3_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Stage 1: GEMM with bias into yGm_
        // Compute: y = x @ w.T + b
        // Note: Matmul high-level API expects B:[K,N]. Our input w is [N,K], so we rely on API transposition handling.
        // If transposition is not supported, w should be pre-transposed offline.
        {
            AscendC::Matmul<aType, bType, cType, biasType> mm;
            mm.SetOrgShape(M_, N_, K_);
            mm.SetA(xGm_);
            mm.SetB(wGm_);   // expects B as (K,N) or implement transpose in API if available
            mm.SetBias(bGm_);
            mm.SetC(yGm_);
            mm.Iterate();    // perform full GEMM according to default/internal tiling
        }

        // Stage 2: Fused vector post-ops over y = GELU(hardtanh(y * scaling))
        if (yLen_ == 0 || tileLen_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= yLen_) return;

        uint32_t curLen = yLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < yLen_) {
                nextLen = yLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocalIn = inQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> yLocalOut = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> t1 = tmpQueue1_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2 = tmpQueue2_.AllocTensor<float>();
        AscendC::LocalTensor<float> t3 = tmpQueue3_.AllocTensor<float>();

        // y = y * scaling_
        AscendC::Muls(yLocalIn, yLocalIn, scaling_, len);

        // y = clamp(y, htMin_, htMax_)
        AscendC::Maxs(yLocalIn, yLocalIn, htMin_, len);
        AscendC::Mins(yLocalIn, yLocalIn, htMax_, len);

        // GELU approx: 0.5 * y * (1 + tanh( sqrt(2/pi)*(y + 0.044715*y^3) ))
        // t1 = y^2
        AscendC::Mul(t1, yLocalIn, yLocalIn, len);
        // t2 = y^3
        AscendC::Mul(t2, t1, yLocalIn, len);
        // t2 = 0.044715 * y^3
        AscendC::Muls(t2, t2, 0.044715f, len);
        // t3 = y + t2
        AscendC::Add(t3, yLocalIn, t2, len);
        // t3 = c0 * t3, where c0 = sqrt(2/pi)
        AscendC::Muls(t3, t3, 0.7978845834732056f, len);
        // t3 = tanh(t3)
        AscendC::Tanh(t3, t3, len);
        // t3 = 1 + t3
        AscendC::Adds(t3, t3, 1.0f, len);
        // t3 = 0.5 * t3
        AscendC::Muls(t3, t3, 0.5f, len);
        // yOut = y * t3
        AscendC::Mul(yLocalOut, yLocalIn, t3, len);

        tmpQueue3_.FreeTensor(t3);
        tmpQueue2_.FreeTensor(t2);
        tmpQueue1_.FreeTensor(t1);

        outQueueY_.EnQue<float>(yLocalOut);
        inQueueY_.FreeTensor(yLocalIn);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    // Pipes and queues for vector fusion
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue3_;

    // Global tensors
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    // Shapes and tiling
    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t yLen_{0};
    uint32_t tileLen_{0};

    // Attributes
    float scaling_{1.0f};
    float htMin_{-2.0f};
    float htMax_{2.0f};
};

extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(GM_ADDR x,
                                                                          GM_ADDR w,
                                                                          GM_ADDR b,
                                                                          GM_ADDR y,
                                                                          GM_ADDR workspace,
                                                                          GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScalingHardTanhGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.yLength, tiling_data.tileLength,
            tiling_data.scaling, tiling_data.hardtanhMin, tiling_data.hardtanhMax);
    op.Process();
}
