#include "batched_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    BatchedMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    if (a_shape->GetOriginShape().GetDimNum() != 3 || b_shape->GetOriginShape().GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t B = a_shape->GetOriginShape().GetDim(0);
    const uint64_t M = a_shape->GetOriginShape().GetDim(1);
    const uint64_t K_a = a_shape->GetOriginShape().GetDim(2);

    const uint64_t B2 = b_shape->GetOriginShape().GetDim(0);
    const uint64_t K_b = b_shape->GetOriginShape().GetDim(1);
    const uint64_t N = b_shape->GetOriginShape().GetDim(2);

    if (B != B2 || K_a != K_b) {
        return ge::GRAPH_FAILED;
    }

    uint32_t batch = static_cast<uint32_t>(B);
    uint32_t m = static_cast<uint32_t>(M);
    uint32_t n = static_cast<uint32_t>(N);
    uint32_t k = static_cast<uint32_t>(K_a);

    // Choose blockDim based on batch parallelism; cap for 910B
    uint32_t blockDim = batch;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    // Strides for contiguous batches in ND: A(b,:,:), B(b,:,:), C(b,:,:)
    uint32_t strideA = m * k;
    uint32_t strideB = k * n;
    uint32_t strideC = m * n;

    tiling.set_batch(batch);
    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_blockDim(blockDim);
    tiling.set_batchStrideA(strideA);
    tiling.set_batchStrideB(strideB);
    tiling.set_batchStrideC(strideC);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 3 || b_shape->GetOriginShape().GetDimNum() != 3) {
        return GRAPH_FAILED;
    }

    const int64_t B = a_shape->GetOriginShape().GetDim(0);
    const int64_t M = a_shape->GetOriginShape().GetDim(1);
    const int64_t K = a_shape->GetOriginShape().GetDim(2);
    const int64_t B2 = b_shape->GetOriginShape().GetDim(0);
    const int64_t K2 = b_shape->GetOriginShape().GetDim(1);
    const int64_t N = b_shape->GetOriginShape().GetDim(2);

    if (B != B2 || K != K2) {
        return GRAPH_FAILED;
    }

    gert::Shape *c_shape = context->GetOutputShape(0);
    gert::Shape out;
    out.SetDimNum(3);
    out.SetDim(0, B);
    out.SetDim(1, M);
    out.SetDim(2, N);
    *c_shape = out;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using namespace AscendC;

class KernelBatchedMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelBatchedMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t batch, uint32_t m, uint32_t n, uint32_t k,
                                uint32_t strideA, uint32_t strideB, uint32_t strideC)
    {
        batch_ = batch;
        m_ = m;
        n_ = n;
        k_ = k;
        strideA_ = strideA;
        strideB_ = strideB;
        strideC_ = strideC;

        aGm_.SetGlobalBuffer((__gm__ float*)A, (uint64_t)batch_ * strideA_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, (uint64_t)batch_ * strideB_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, (uint64_t)batch_ * strideC_);

        // Matmul config for NORMAL layout (batch, m, k) x (batch, k, n) -> (batch, m, n)
        cfg_.SetComputeDataType(DataType::DT_FLOAT);
        cfg_.SetAType(TPosition::GM, CubeFormat::ND, DataType::DT_FLOAT);
        cfg_.SetBType(TPosition::GM, CubeFormat::ND, DataType::DT_FLOAT);
        cfg_.SetCType(TPosition::GM, CubeFormat::ND, DataType::DT_FLOAT);
        cfg_.SetTransposeA(false);
        cfg_.SetTransposeB(false);
        cfg_.SetBaseMNK(m_, n_, k_);
        cfg_.SetBatchInfoForNormal(m_, n_, k_, batch_, batch_); // A/B both have same batch
        cfg_.SetOutputTranspose(false);

        matmul_.SetConfig(cfg_);

        pipe_.InitBuffer(bufA_, 1, 1);
        pipe_.InitBuffer(bufB_, 1, 1);
        pipe_.InitBuffer(bufC_, 1, 1);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = (uint32_t)GetBlockNum();
        const uint32_t blockIdx = (uint32_t)GetBlockIdx();

        for (uint32_t b = blockIdx; b < batch_; b += blockNum) {
            __gm__ float* aPtr = aGm_.GetPhyAddr() + (uint64_t)b * strideA_;
            __gm__ float* bPtr = bGm_.GetPhyAddr() + (uint64_t)b * strideB_;
            __gm__ float* cPtr = cGm_.GetPhyAddr() + (uint64_t)b * strideC_;

            // one-shot matmul on (m x k) * (k x n) -> (m x n) for this batch
            matmul_.Compute(aPtr, bPtr, cPtr, m_, n_, k_);
        }
    }

private:
    TPipe pipe_;
    TBuf<TPosition::VECCALC> bufA_;
    TBuf<TPosition::VECCALC> bufB_;
    TBuf<TPosition::VECCALC> bufC_;

    GlobalTensor<float> aGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> cGm_;

    uint32_t batch_{0};
    uint32_t m_{0};
    uint32_t n_{0};
    uint32_t k_{0};
    uint32_t strideA_{0};
    uint32_t strideB_{0};
    uint32_t strideC_{0};

    matmul_tiling::MatmulConfig cfg_;
    matmul_tiling::Matmul matmul_;
};

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBatchedMatrixMultiplicationCustom op;
    op.Init(A, B, C,
            tiling_data.batch, tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.batchStrideA, tiling_data.batchStrideB, tiling_data.batchStrideC);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor bmm_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 3 && B.dim() == 3, "A and B must be 3D (batch, m, k) and (batch, k, n)");
    TORCH_CHECK(A.size(0) == B.size(0) && A.size(2) == B.size(1),
                "Incompatible batch or inner dimensions for batched matmul");
    auto sizes = std::vector<int64_t>{A.size(0), A.size(1), B.size(2)};
    at::Tensor C = at::empty(sizes, A.options());
    EXEC_NPU_CMD(aclnnBatchedMatrixMultiplicationCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("batched_matrix_multiplication_custom", &bmm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("batched_matrix_multiplication_custom", &bmm_impl_npu,
          "Batched MatMul (BatchedMatrixMultiplicationCustom on NPU)");
}
