static inline uint32_t AlignDown(uint32_t x, uint32_t a)

    return (x / a) * a;

{
    return (a + b - 1u) / b;

#include "kernel_operator.h"

constexpr int32_t IN_BUFFER_NUM = 2;

class KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom {
public:
    __aicore__ inline KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom() {}

    __aicore__ inline void Init(GM_ADDR y_in, GM_ADDR y_out,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        n_ = n; c_ = c; d_ = d; h_ = h; w_ = w;
        spatialSize_ = d_ * h_ * w_;

        inGm_.SetGlobalBuffer((__gm__ float*)y_in, totalLen_);
        outGm_.SetGlobalBuffer((__gm__ float*)y_out, n_ * c_); // output N*C scalars

        pipe_.InitBuffer(inQueue_, IN_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpReduceQueue_, 1, tileLen_ * sizeof(float));
        // two 1-element scalar buffers for accumulators
        pipe_.InitBuffer(scalarQueueA_, 1, sizeof(float));
        pipe_.InitBuffer(scalarQueueB_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        // Force single-core behavior (tiling sets blockDim=1)
        if (AscendC::GetBlockIdx() != 0) return;

        // Pass 1: compute global maximum across all elements
        AscendC::LocalTensor<float> sMaxLocal = scalarQueueA_.AllocTensor<float>();
        AscendC::Dup(sMaxLocal, -3.4028235e38f, 1);  // approx -FLT_MAX

        uint32_t offset = 0;
        while (offset < totalLen_) {
            uint32_t len = totalLen_ - offset;
            if (len > tileLen_) len = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, inGm_[offset], len);

            AscendC::LocalTensor<float> tileMax = scalarQueueB_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpBuf = tmpReduceQueue_.AllocTensor<float>();

            AscendC::ReduceMax(tileMax, xLocal, tmpBuf, static_cast<int32_t>(len), false);
            AscendC::Max(sMaxLocal, sMaxLocal, tileMax, 1);

            tmpReduceQueue_.FreeTensor(tmpBuf);
            scalarQueueB_.FreeTensor(tileMax);
            inQueue_.FreeTensor(xLocal);

            offset += len;
        }

        // Pass 2: for each (n, c), compute average over D*H*W, multiply by global max, and clamp
        const float invSpatial = (spatialSize_ > 0) ? (1.0f / static_cast<float>(spatialSize_)) : 0.0f;

        for (uint32_t ni = 0; ni < n_; ++ni) {
            for (uint32_t ci = 0; ci < c_; ++ci) {
                AscendC::LocalTensor<float> sumLocal = scalarQueueB_.AllocTensor<float>();
                AscendC::Dup(sumLocal, 0.0f, 1);

                // base offset for this (n,c) plane
                uint64_t base = (((uint64_t)ni * c_ + ci) * (uint64_t)spatialSize_);

                uint32_t consumed = 0;
                while (consumed < spatialSize_) {
                    uint32_t len = spatialSize_ - consumed;
                    if (len > tileLen_) len = tileLen_;

                    AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, inGm_[base + consumed], len);

                    AscendC::LocalTensor<float> tileSum = scalarQueueA_.AllocTensor<float>();
                    AscendC::LocalTensor<float> tmpBuf = tmpReduceQueue_.AllocTensor<float>();

                    AscendC::ReduceSum(tileSum, xLocal, tmpBuf, static_cast<int32_t>(len));
                    AscendC::Add(sumLocal, sumLocal, tileSum, 1);

                    tmpReduceQueue_.FreeTensor(tmpBuf);
                    scalarQueueA_.FreeTensor(tileSum);
                    inQueue_.FreeTensor(xLocal);

                    consumed += len;
                }

                // avg = (sum / spatialSize)
                AscendC::Muls(sumLocal, sumLocal, invSpatial, 1);
                // scale by global max
                AscendC::Mul(sumLocal, sumLocal, sMaxLocal, 1);
                // clamp to [0, 1]
                AscendC::Maxs(sumLocal, sumLocal, 0.0f, 1);
                AscendC::Mins(sumLocal, sumLocal, 1.0f, 1);

                // write result
                uint64_t outIndex = ((uint64_t)ni * c_) + ci;
                AscendC::DataCopy(outGm_[outIndex], sumLocal, 1);

                scalarQueueB_.FreeTensor(sumLocal);
            }
        }

        scalarQueueA_.FreeTensor(sMaxLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpReduceQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueueA_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueueB_;

    AscendC::GlobalTensor<float> inGm_;
    AscendC::GlobalTensor<float> outGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t n_{0}, c_{0}, d_{0}, h_{0}, w_{0};
    uint32_t spatialSize_{0};
};

extern "C" __global__ __aicore__ void conv_transpose3d_multiply_max_global_avg_pool_clamp_custom(GM_ADDR y_in, GM_ADDR y_out, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom op;
    op.Init(y_in, y_out,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w);
    op.Process();
}
