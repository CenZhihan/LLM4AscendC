host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}

static inline int64_t SafeProd(const std::vector<int64_t> &v, size_t begin, size_t end)
{
    int64_t r = 1;
    for (size_t i = begin; i < end; ++i) r *= v[i];
    return r;
}

} // namespace

namespace optiling {

// UB model for this kernel processing inner-chunks per outer:
// Buffers per chunk: val(float)*tile + x(float)*tile + idx(int32)*tile = (4+4+4)*tile = 12*tile bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t BYTES_PER_ELEM = 12;           // val + x + idx
static constexpr uint32_t ALIGN_ELEMS = 512;             // DMA alignment preference
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // limit single-iter UB

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);

    const gert::Shape *shape = context->GetInputShape(0);
    const auto &origin = shape->GetOriginShape();

    const int32_t rank = static_cast<int32_t>(origin.GetDimNum());
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }

    std::vector<int64_t> dims(rank);
    for (int32_t i = 0; i < rank; ++i) {
        dims[i] = origin.GetDim(i);
    }

    int32_t axis = (dimAttr != nullptr) ? *dimAttr : 0;
    if (axis < 0) axis += rank;
    if (axis < 0 || axis >= rank) {
        return ge::GRAPH_FAILED;
    }

    const int64_t outer64 = SafeProd(dims, 0, static_cast<size_t>(axis));
    const int64_t axisLen64 = dims[axis];
    const int64_t inner64 = SafeProd(dims, static_cast<size_t>(axis + 1), static_cast<size_t>(rank));

    if (outer64 <= 0 || axisLen64 <= 0 || inner64 <= 0) {
        // Degenerate shapes are not supported
        return ge::GRAPH_FAILED;
    }

    const uint64_t innerU = static_cast<uint64_t>(inner64);
    uint32_t maxTile = static_cast<uint32_t>(UB_BUDGET_BYTES / BYTES_PER_ELEM);
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = ALIGN_ELEMS;

    uint32_t tileInner = maxTile;
    if (innerU < maxTile) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(innerU), ALIGN_ELEMS);
        tileInner = (aligned > 0) ? aligned : static_cast<uint32_t>(innerU);
    }

    const uint64_t outerU = static_cast<uint64_t>(outer64);
    uint32_t blockDim = static_cast<uint32_t>(outerU);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_outer(static_cast<uint32_t>(outer64));
    tiling.set_axisLen(static_cast<uint32_t>(axisLen64));
    tiling.set_inner(static_cast<uint32_t>(inner64));
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin = x_shape->GetOriginShape();

    const int32_t rank = static_cast<int32_t>(origin.GetDimNum());
    if (rank <= 0) return GRAPH_FAILED;

    // Get dim attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t axis = (dimAttr != nullptr) ? *dimAttr : 0;
    if (axis < 0) axis += rank;
    if (axis < 0 || axis >= rank) return GRAPH_FAILED;

    std::vector<int64_t> out_dims;
    out_dims.reserve(static_cast<size_t>(rank > 0 ? rank - 1 : 0));
    for (int32_t i = 0; i < rank; ++i) {
        if (i == axis) continue;
        out_dims.push_back(origin.GetDim(i));
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(static_cast<uint32_t>(out_dims.size()));
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT32);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices,
                                uint32_t outer, uint32_t axisLen,
                                uint32_t inner, uint32_t tileInner)
    {
        outer_ = outer;
        axisLen_ = axisLen;
        inner_ = inner;
        tileInner_ = tileInner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(outer_ * axisLen_ * inner_));
        idxGm_.SetGlobalBuffer((__gm__ int32_t*)indices, static_cast<uint32_t>(outer_ * inner_));

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(outQueueIdx_, OUT_BUF_NUM, tileInner_ * sizeof(int32_t));
        pipe_.InitBuffer(calcBufFloat_, 2, tileInner_ * sizeof(float));   // val buffer and tmp
        pipe_.InitBuffer(calcBufInt_, 1, tileInner_ * sizeof(int32_t));   // idx buffer
        pipe_.InitBuffer(maskBuf_, 1, tileInner_ * sizeof(uint8_t));      // mask buffer (byte per elem)
    }

    __aicore__ inline void Process()
    {
        if (outer_ == 0 || axisLen_ == 0 || inner_ == 0 || tileInner_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t outerIdx = blockIdx; outerIdx < outer_; outerIdx += blockNum) {
            const uint64_t baseOuterOffset = static_cast<uint64_t>(outerIdx) * axisLen_ * inner_;
            // Process inner dimension in chunks
            for (uint32_t innerBase = 0; innerBase < inner_; innerBase += tileInner_) {
                const uint32_t curTile = (innerBase + tileInner_ <= inner_) ? tileInner_ : (inner_ - innerBase);

                // Alloc persistent buffers for current tile
                AscendC::LocalTensor<float> valLocal = calcBufFloat_.AllocTensor<float>();
                AscendC::LocalTensor<int32_t> idxLocal = calcBufInt_.AllocTensor<int32_t>();
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();

                // k = 0 initialization
                {
                    const uint64_t gmOff0 = baseOuterOffset + static_cast<uint64_t>(0) * inner_ + innerBase;
                    AscendC::DataCopy(xLocal, xGm_[gmOff0], curTile);
                    AscendC::DataCopy(valLocal, xLocal, curTile);
                    AscendC::Duplicate(idxLocal, static_cast<int32_t>(0), curTile);
                }

                // iterate over k = 1 .. axisLen_-1
                for (uint32_t k = 1; k < axisLen_; ++k) {
                    const uint64_t gmOff = baseOuterOffset + static_cast<uint64_t>(k) * inner_ + innerBase;
                    AscendC::DataCopy(xLocal, xGm_[gmOff], curTile);

                    // tmpVal = max(valLocal, xLocal)
                    AscendC::LocalTensor<float> tmpVal = calcBufFloat_.AllocTensor<float>();
                    AscendC::Max(tmpVal, valLocal, xLocal, curTile);

                    // mask = xLocal > valLocal
                    AscendC::LocalTensor<uint8_t> mask = maskBuf_.AllocTensor<uint8_t>();
                    AscendC::Greater(mask, xLocal, valLocal, curTile);

                    // prepare scalar k vector
                    AscendC::LocalTensor<int32_t> kVec = outQueueIdx_.AllocTensor<int32_t>();
                    AscendC::Duplicate(kVec, static_cast<int32_t>(k), curTile);

                    // idxLocal = select(mask, kVec, idxLocal)
                    AscendC::Select(idxLocal, mask, kVec, idxLocal, curTile);

                    // valLocal = tmpVal
                    AscendC::DataCopy(valLocal, tmpVal, curTile);

                    // free temporaries
                    outQueueIdx_.FreeTensor(kVec);
                    maskBuf_.FreeTensor(mask);
                    calcBufFloat_.FreeTensor(tmpVal);
                }

                // write indices for this tile
                AscendC::LocalTensor<int32_t> outLocal = outQueueIdx_.AllocTensor<int32_t>();
                AscendC::DataCopy(outLocal, idxLocal, curTile);
                const uint64_t outBase = static_cast<uint64_t>(outerIdx) * inner_ + innerBase;
                AscendC::DataCopy(idxGm_[outBase], outLocal, curTile);

                // release locals
                outQueueIdx_.FreeTensor(outLocal);
                inQueueX_.FreeTensor(xLocal);
                calcBufInt_.FreeTensor(idxLocal);
                calcBufFloat_.FreeTensor(valLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueIdx_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> calcBufFloat_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcBufInt_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maskBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int32_t> idxGm_;

    uint32_t outer_ {0};
    uint32_t axisLen_ {0};
    uint32_t inner_ {0};
    uint32_t tileInner_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, indices,
            tiling_data.outer, tiling_data.axisLen,
            tiling_data.inner, tiling_data.tileInner);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline std::vector<int64_t> compute_argmax_out_sizes(const at::Tensor& self, int64_t dim) {
    auto sizes = self.sizes().vec();
    const int64_t ndim = static_cast<int64_t>(sizes.size());
    if (dim < 0) dim += ndim;
    std::vector<int64_t> out;
    out.reserve(static_cast<size_t>(ndim > 0 ? ndim - 1 : 0));
    for (int64_t i = 0; i < ndim; ++i) {
        if (i == dim) continue;
        out.push_back(sizes[static_cast<size_t>(i)]);
    }
    return out;
}

at::Tensor argmax_over_a_dimension_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    auto out_sizes = compute_argmax_out_sizes(self, dim);
    at::Tensor result_int32 = at::empty(out_sizes, self.options().dtype(at::kInt));
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, static_cast<int>(dim), result_int32);
    // Match PyTorch's default index dtype: int64
    return result_int32.to(at::kLong);
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_over_a_dimension_custom_impl_npu,
          "Argmax over a dimension (ArgmaxOverADimensionCustom on NPU)");
}
