host_operator_src="""

#include "average_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// Memory model for this kernel (float32):
//   srcLocal:  tileSrc  = tileOut * stride + (kernelSize - 1)
//   sumLocal:  tileConv = tileSrc - kernelSize + 1 = tileOut * stride
//   yLocal:    tileOut
// Total floats = tileSrc + tileConv + tileOut
// Bytes = 4 * ( (tileOut * stride + (k-1)) + (tileOut * stride) + tileOut )
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_TILE = 512;              // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_TILE_OUT = 32768;     // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling1dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const int64_t dims = x_shape->GetOriginShape().GetDimNum();
    if (dims != 3) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t n = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t c = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t length = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t totalRows = n * c;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kptr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *sptr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *pptr = attrs->GetAttrPointer<int32_t>(2);
    const uint32_t kernelSize = static_cast<uint32_t>(*kptr);
    const uint32_t stride = static_cast<uint32_t>(sptr ? *sptr : 1);
    const uint32_t padding = static_cast<uint32_t>(pptr ? *pptr : 0);

    // Compute output length (ceil_mode=false)
    int64_t numer = static_cast<int64_t>(length) + 2LL * static_cast<int64_t>(padding) - static_cast<int64_t>(kernelSize);
    int64_t outL = (numer >= 0) ? (numer / static_cast<int64_t>(stride) + 1) : 0;
    if (outL < 0) outL = 0;
    const uint32_t outLength = static_cast<uint32_t>(outL);

    // Compute tileOut from UB budget
    // bytes = 4 * ( tileOut*(2*stride + 1) + (kernelSize - 1) )
    uint32_t denom = (2u * stride + 1u);
    uint32_t maxTileOut = 0;
    if (denom > 0) {
        if (UB_BUDGET_BYTES > 4u * (kernelSize > 0 ? (kernelSize - 1u) : 0u)) {
            maxTileOut = (UB_BUDGET_BYTES / 4u - (kernelSize > 0 ? (kernelSize - 1u) : 0u)) / denom;
        } else {
            maxTileOut = ALIGN_TILE;
        }
    }
    if (maxTileOut == 0) maxTileOut = ALIGN_TILE;
    maxTileOut = AlignDown(maxTileOut, ALIGN_TILE);
    if (maxTileOut > HARD_CAP_TILE_OUT) maxTileOut = AlignDown(HARD_CAP_TILE_OUT, ALIGN_TILE);

    uint32_t tileOut = maxTileOut;
    if (outLength < tileOut) {
        const uint32_t aligned = AlignDown(outLength, ALIGN_TILE);
        tileOut = (aligned > 0) ? aligned : outLength;
    }
    if (tileOut == 0 && outLength > 0) {
        tileOut = outLength;
    }

    // Blocks across rows
    uint32_t blockDim = totalRows;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_length(length);
    tiling.set_outLength(outLength);
    tiling.set_totalRows(totalRows);
    tiling.set_tileOut(tileOut);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t SRC_BUFFERS = 1;
constexpr int32_t SUM_BUFFERS = 1;
constexpr int32_t OUT_BUFFERS = 1;

class KernelAveragePooling1dCustom {
public:
    __aicore__ inline KernelAveragePooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t length, uint32_t outLength,
                                uint32_t tileOut,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding)
    {
        n_ = n;
        c_ = c;
        length_ = length;
        outLength_ = outLength;
        tileOut_ = tileOut == 0 ? outLength : tileOut;
        k_ = kernelSize;
        s_ = stride;
        p_ = padding;

        totalRows_ = n_ * c_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(totalRows_ * length_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(totalRows_ * outLength_));

        uint32_t maxTileSrc = tileOut_ * s_ + (k_ > 0 ? (k_ - 1) : 0);
        uint32_t maxTileConv = (maxTileSrc >= k_) ? (maxTileSrc - k_ + 1) : 0;

        pipe_.InitBuffer(srcQueue_, SRC_BUFFERS, maxTileSrc * sizeof(float));
        pipe_.InitBuffer(sumQueue_, SUM_BUFFERS, maxTileConv * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUT_BUFFERS, tileOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (length_ == 0 || outLength_ == 0 || k_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            const uint64_t xBase = static_cast<uint64_t>(row) * length_;
            const uint64_t yBase = static_cast<uint64_t>(row) * outLength_;

            for (uint32_t outStart = 0; outStart < outLength_; outStart += tileOut_) {
                uint32_t curOut = outLength_ - outStart;
                if (curOut > tileOut_) curOut = tileOut_;

                const uint32_t tileSrc = curOut * s_ + (k_ - 1);
                const uint32_t tileConv = (tileSrc >= k_) ? (tileSrc - k_ + 1) : 0;

                AscendC::LocalTensor<float> srcLocal = srcQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> yLocal   = outQueue_.AllocTensor<float>();

                // Zero initialize locals
                AscendC::Dup(srcLocal, 0.0f, tileSrc);
                if (tileConv > 0) {
                    AscendC::Dup(sumLocal, 0.0f, tileConv);
                }
                AscendC::Dup(yLocal, 0.0f, curOut);

                // Compute input slice bounds
                // startX = outStart * s - p
                int64_t startX = static_cast<int64_t>(outStart) * static_cast<int64_t>(s_) - static_cast<int64_t>(p_);
                int64_t srcBegin = startX;
                int64_t srcEnd = startX + static_cast<int64_t>(tileSrc);

                int64_t left = srcBegin < 0 ? 0 : srcBegin;
                int64_t right = srcEnd > static_cast<int64_t>(length_) ? static_cast<int64_t>(length_) : srcEnd;

                if (right > left) {
                    uint32_t copyLen = static_cast<uint32_t>(right - left);
                    uint32_t destOff = static_cast<uint32_t>(left - srcBegin);
                    AscendC::DataCopy(srcLocal[destOff], xGm_[xBase + static_cast<uint64_t>(left)], copyLen);
                }

                // Compute sliding window sum (stride-1 domain)
                if (tileConv > 0) {
                    for (uint32_t j = 0; j < k_; ++j) {
                        AscendC::Add(sumLocal, sumLocal, srcLocal[j], tileConv);
                    }
                }

                // Subsample and scale by kernel size
                const float invK = 1.0f / static_cast<float>(k_);
                if (s_ == 1) {
                    // Directly use the contiguous part
                    AscendC::Muls(yLocal, sumLocal, invK, curOut);
                } else {
                    // General stride: sample every s_ elements
                    for (uint32_t i = 0; i < curOut; ++i) {
                        // y[i] = sum[i * s_] * invK
                        AscendC::DataCopy(yLocal[i], sumLocal[i * s_], 1);
                    }
                    AscendC::Muls(yLocal, yLocal, invK, curOut);
                }

                // Write back
                AscendC::DataCopy(yGm_[yBase + outStart], yLocal, curOut);

                outQueue_.FreeTensor(yLocal);
                sumQueue_.FreeTensor(sumLocal);
                srcQueue_.FreeTensor(srcLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, SRC_BUFFERS> srcQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, SUM_BUFFERS> sumQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFERS> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0};
    uint32_t length_{0}, outLength_{0}, tileOut_{0};
    uint32_t k_{0}, s_{1}, p_{0};
    uint32_t totalRows_{0};
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling1dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.length, tiling_data.outLength,
            tiling_data.tileOut,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t compute_out_length(int64_t length, int64_t kernel_size, int64_t stride, int64_t padding) {
    const int64_t numer = length + 2 * padding - kernel_size;
    if (numer < 0) return 0;
    return numer / stride + 1;
}

at::Tensor average_pooling_1d_impl_npu(const at::Tensor& x, int64_t kernel_size, int64_t stride, int64_t padding) {
    TORCH_CHECK(x.dim() == 3, "average_pooling_1d_custom expects 3D tensor (N, C, L)");
    auto sizes = x.sizes();
    const int64_t N = sizes[0];
    const int64_t C = sizes[1];
    const int64_t L = sizes[2];
    const int64_t outL = compute_out_length(L, kernel_size, stride, padding);
    at::Tensor result = at::empty({N, C, outL}, x.options());
    EXEC_NPU_CMD(aclnnAveragePooling1dCustom, x, static_cast<int>(kernel_size), static_cast<int>(stride), static_cast<int>(padding), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("average_pooling_1d_custom", &average_pooling_1d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("average_pooling_1d_custom", &average_pooling_1d_impl_npu, "1D Average Pooling (AveragePooling1dCustom on NPU)");
