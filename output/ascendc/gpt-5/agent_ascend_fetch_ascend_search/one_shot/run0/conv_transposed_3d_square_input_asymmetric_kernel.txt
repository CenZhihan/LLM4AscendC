kernel_src="""

#include "kernel_operator.h"
#include "lib/conv_backprop/conv3d_bp_input_api.h"

// Implement conv_transpose3d using Conv3DBackpropInput high-level API.
// Mapping:
//  - gradOutput (N, Cout_fwd, D_out_fwd, H_out_fwd, W_out_fwd) <= x (input to conv_transpose)
//  - weight (Cout_fwd, Cin_fwd, kd, kh, kw) <= weight (Cin_transpose, Cout_transpose, kd, kh, kw) == forward conv weight
//  - gradInput (N, Cin_fwd, D_in_fwd, H_in_fwd, W_in_fwd) => y (output of conv_transpose)

using gradType   = float;
using weightType = float;
using inputSizeType = int32_t;

using weightDxType    = ConvBackpropApi::ConvType<ConvCommonApi::TPosition::GM, ConvCommonApi::ConvFormat::FRACTAL_Z_3D, weightType>;
using inputSizeDxType = ConvBackpropApi::ConvType<ConvCommonApi::TPosition::GM, ConvCommonApi::ConvFormat::ND, inputSizeType>;
using gradOutputDxType= ConvBackpropApi::ConvType<ConvCommonApi::TPosition::GM, ConvCommonApi::ConvFormat::NDC1HWC0, gradType>;
using gradInputDxType = ConvBackpropApi::ConvType<ConvCommonApi::TPosition::GM, ConvCommonApi::ConvFormat::NCDHW, gradType>;

class KernelConvTransposed3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t n, uint32_t c_in_trans, uint32_t c_out,
                                uint32_t d_in, uint32_t h_in, uint32_t w_in,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t d_out, uint32_t h_out, uint32_t w_out,
                                uint32_t stride_d, uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_d, uint32_t pad_h, uint32_t pad_w,
                                uint32_t dil_d, uint32_t dil_h, uint32_t dil_w,
                                uint32_t groups)
    {
        x_gm_      = (__gm__ gradType*)x;
        w_gm_      = (__gm__ weightType*)weight;
        y_gm_      = (__gm__ gradType*)y;

        n_ = n;
        cout_fwd_ = c_in_trans; // Cout of forward equals in_channels of conv_transpose
        cin_fwd_  = c_out;      // Cin of forward equals out_channels of conv_transpose

        d_out_fwd_ = d_in;  // forward output spatial equals x spatial
        h_out_fwd_ = h_in;
        w_out_fwd_ = w_in;

        kd_ = kd; kh_ = kh; kw_ = kw;

        d_in_fwd_ = d_out;
        h_in_fwd_ = h_out;
        w_in_fwd_ = w_out;

        stride_d_ = stride_d; stride_h_ = stride_h; stride_w_ = stride_w;
        pad_d_ = pad_d; pad_h_ = pad_h; pad_w_ = pad_w;
        dil_d_ = dil_d; dil_h_ = dil_h; dil_w_ = dil_w;
        groups_ = groups;
    }

    __aicore__ inline void Process()
    {
        // Create Conv3DBackpropInput object with GM pointers and configuration.
        ConvBackpropApi::Conv3DBackpropInput<weightDxType, inputSizeDxType, gradOutputDxType, gradInputDxType> op;

        // Configure convolution parameters
        ConvBackpropApi::Conv3DBackpropInputConfig cfg;
        cfg.n = n_;
        cfg.cin = cin_fwd_;
        cfg.cout = cout_fwd_;
        cfg.d_in = d_in_fwd_;
        cfg.h_in = h_in_fwd_;
        cfg.w_in = w_in_fwd_;
        cfg.kd = kd_;
        cfg.kh = kh_;
        cfg.kw = kw_;
        cfg.stride_d = stride_d_;
        cfg.stride_h = stride_h_;
        cfg.stride_w = stride_w_;
        cfg.pad_d = pad_d_;
        cfg.pad_h = pad_h_;
        cfg.pad_w = pad_w_;
        cfg.dil_d = dil_d_;
        cfg.dil_h = dil_h_;
        cfg.dil_w = dil_w_;
        cfg.groups = groups_;

        op.Init(cfg);

        // Set GM buffers: gradOutput (x), weight, gradInput (y)
        op.SetGradOutput(x_gm_, n_, cout_fwd_, d_out_fwd_, h_out_fwd_, w_out_fwd_);
        op.SetWeight(w_gm_, cout_fwd_, cin_fwd_, kd_, kh_, kw_);
        op.SetGradInput(y_gm_, n_, cin_fwd_, d_in_fwd_, h_in_fwd_, w_in_fwd_);

        // Execute convolution backprop input (equivalent to conv_transpose)
        op.Run();

        op.DeInit();
    }

private:
    __gm__ gradType*   x_gm_{nullptr};
    __gm__ weightType* w_gm_{nullptr};
    __gm__ gradType*   y_gm_{nullptr};

    uint32_t n_{0}, cout_fwd_{0}, cin_fwd_{0};
    uint32_t d_in_fwd_{0}, h_in_fwd_{0}, w_in_fwd_{0};
    uint32_t d_out_fwd_{0}, h_out_fwd_{0}, w_out_fwd_{0};
    uint32_t kd_{0}, kh_{0}, kw_{0};
    uint32_t stride_d_{1}, stride_h_{1}, stride_w_{1};
    uint32_t pad_d_{0}, pad_h_{0}, pad_w_{0};
    uint32_t dil_d_{1}, dil_h_{1}, dil_w_{1};
    uint32_t groups_{1};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvTransposed3dSquareInputAsymmetricKernelCustom kernel;
    kernel.Init(
        x, weight, y,
        tiling_data.n, tiling_data.c_in_trans, tiling_data.c_out,
        tiling_data.d_in, tiling_data.h_in, tiling_data.w_in,
        tiling_data.kd, tiling_data.kh, tiling_data.kw,
        tiling_data.d_out, tiling_data.h_out, tiling_data.w_out,
        tiling_data.stride_d, tiling_data.stride_h, tiling_data.stride_w,
        tiling_data.pad_d, tiling_data.pad_h, tiling_data.pad_w,
        tiling_data.dil_d, tiling_data.dil_h, tiling_data.dil_w,
        tiling_data.groups
    );
    kernel.Process();
}
