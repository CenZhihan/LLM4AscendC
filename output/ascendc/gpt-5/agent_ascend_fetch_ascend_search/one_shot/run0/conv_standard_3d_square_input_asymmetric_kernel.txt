#include "conv_standard_3d_square_input_asymmetric_kernel_custom_tiling.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

} // anonymous

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative UB budget
static constexpr uint32_t MIN_BLOCKS = 8;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard3dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape &w_shape = context->GetInputShape(1)->GetOriginShape();

    if (x_shape.GetDimNum() != 5 || w_shape.GetDimNum() != 5) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N  = static_cast<uint32_t>(x_shape.GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape.GetDim(1));
    const uint32_t W  = static_cast<uint32_t>(x_shape.GetDim(2));
    const uint32_t H  = static_cast<uint32_t>(x_shape.GetDim(3));
    const uint32_t D  = static_cast<uint32_t>(x_shape.GetDim(4));

    const uint32_t Cout = static_cast<uint32_t>(w_shape.GetDim(0));
    const uint32_t Cin_w = static_cast<uint32_t>(w_shape.GetDim(1));
    const uint32_t Kw = static_cast<uint32_t>(w_shape.GetDim(2));
    const uint32_t Kh = static_cast<uint32_t>(w_shape.GetDim(3));
    const uint32_t Kd = static_cast<uint32_t>(w_shape.GetDim(4));

    if (Cin != Cin_w) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Wo = (W >= Kw) ? (W - Kw + 1u) : 0u;
    const uint32_t Ho = (H >= Kh) ? (H - Kh + 1u) : 0u;
    const uint32_t Do = (D >= Kd) ? (D - Kd + 1u) : 0u;

    if (Wo == 0 || Ho == 0 || Do == 0) {
        return ge::GRAPH_FAILED;
    }

    // Choose tile on D-out and oc tile based on UB budget:
    // UB usage (floats) ~= (tileOc * tileD)  // y accumulators
    //                   + tileD              // x vector
    //                   + tileD              // temp vector
    //                   + 1                  // scalar weight
    // bytes = 4 * (tileOc * tileD + 2 * tileD + 1)
    // Strategy: start with tileD = Do (cap aligned), then compute tileOc; if tileOc < 1, reduce tileD.
    uint32_t tileD = Do;
    // encourage alignment but allow tail
    const uint32_t ALIGN_ELEMS = 16u;
    if (tileD > ALIGN_ELEMS) tileD = AlignDown(tileD, ALIGN_ELEMS);
    if (tileD == 0) tileD = Do; // small Do

    auto computeTileOc = [&](uint32_t tD)->uint32_t {
        const uint32_t denom = 4u * tD; // bytes per oc lane per tileD
        uint32_t maxOc = 1u;
        if (denom > 0) {
            const uint32_t capFloats = UB_BUDGET_BYTES / 4u;
            if (capFloats > (2u * tD + 1u)) {
                maxOc = (capFloats - (2u * tD + 1u)) / tD;
                if (maxOc == 0) maxOc = 1u;
            }
        }
        if (maxOc > Cout) maxOc = Cout;
        return maxOc;
    };

    uint32_t tileOc = computeTileOc(tileD);
    if (tileOc == 0u) tileOc = 1u;

    // If oc tile is too small, try to reduce tileD to increase oc tile (not strictly necessary here).
    if (tileOc < 1u) {
        // Fallback to minimal tileD
        tileD = (Do >= ALIGN_ELEMS) ? ALIGN_ELEMS : Do;
        tileOc = computeTileOc(tileD);
        if (tileOc == 0u) tileOc = 1u;
    }

    const uint32_t numDTiles = CeilDiv(Do, tileD);
    const uint64_t tasks64 = static_cast<uint64_t>(N) * Wo * Ho * numDTiles;
    uint32_t blockDim = static_cast<uint32_t>(tasks64 > UINT32_MAX ? MAX_BLOCKS : (tasks64));
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_W(W);
    tiling.set_H(H);
    tiling.set_D(D);
    tiling.set_Kw(Kw);
    tiling.set_Kh(Kh);
    tiling.set_Kd(Kd);
    tiling.set_Wo(Wo);
    tiling.set_Ho(Ho);
    tiling.set_Do(Do);
    tiling.set_tileD(tileD);
    tiling.set_tileOc(tileOc);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvStandard3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t W, uint32_t H, uint32_t D,
                                uint32_t Kw, uint32_t Kh, uint32_t Kd,
                                uint32_t Wo, uint32_t Ho, uint32_t Do,
                                uint32_t tileD, uint32_t tileOc)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        W_ = W; H_ = H; D_ = D;
        Kw_ = Kw; Kh_ = Kh; Kd_ = Kd;
        Wo_ = Wo; Ho_ = Ho; Do_ = Do;
        tileD_ = tileD; tileOc_ = tileOc;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_*Cin_*W_*H_*D_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cout_*Cin_*Kw_*Kh_*Kd_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_*Cout_*Wo_*Ho_*Do_);

        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileD_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileD_ * sizeof(float));
        pipe_.InitBuffer(yTileQueue_, BUFFER_NUM, tileOc_ * tileD_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || Wo_ == 0 || Ho_ == 0 || Do_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numDTiles = (tileD_ == 0) ? 1u : CeilDiv(Do_, tileD_);
        const uint64_t totalTasks = static_cast<uint64_t>(N_) * Wo_ * Ho_ * numDTiles;

        for (uint64_t task = blockIdx; task < totalTasks; task += blockNum) {
            // Decode task -> (nIdx, wOut, hOut, dTileIdx)
            uint64_t tmp = task;
            const uint32_t dTileIdx = static_cast<uint32_t>(tmp % numDTiles); tmp /= numDTiles;
            const uint32_t hOut = static_cast<uint32_t>(tmp % Ho_); tmp /= Ho_;
            const uint32_t wOut = static_cast<uint32_t>(tmp % Wo_); tmp /= Wo_;
            const uint32_t nIdx = static_cast<uint32_t>(tmp);

            const uint32_t dBase = dTileIdx * tileD_;
            uint32_t curLenD = Do_ - dBase;
            if (curLenD > tileD_) curLenD = tileD_;

            for (uint32_t ocBase = 0; ocBase < Cout_; ocBase += tileOc_) {
                const uint32_t ocChunk = MinU32(tileOc_, Cout_ - ocBase);

                AscendC::LocalTensor<float> yTile = yTileQueue_.AllocTensor<float>();
                AscendC::Dup(yTile, 0.0f, ocChunk * curLenD);

                for (uint32_t c = 0; c < Cin_; ++c) {
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const uint32_t wIn = wOut + kw;
                        for (uint32_t kh = 0; kh < Kh_; ++kh) {
                            const uint32_t hIn = hOut + kh;
                            for (uint32_t kz = 0; kz < Kd_; ++kz) {
                                const uint32_t dInBase = dBase + kz;

                                // Load input vector x[nIdx, c, wIn, hIn, dInBase : dInBase + curLenD]
                                const uint32_t xOffset = ((((nIdx * Cin_ + c) * W_ + wIn) * H_ + hIn) * D_) + dInBase;
                                AscendC::LocalTensor<float> xVec = xQueue_.AllocTensor<float>();
                                AscendC::DataCopy(xVec, xGm_[xOffset], curLenD);

                                // For each oc in current chunk, load scalar weight and accumulate
                                for (uint32_t o = 0; o < ocChunk; ++o) {
                                    const uint32_t oc = ocBase + o;
                                    const uint32_t wOffset = ((((oc * Cin_ + c) * Kw_ + kw) * Kh_ + kh) * Kd_) + kz;

                                    AscendC::LocalTensor<float> wScalar = scalarQueue_.AllocTensor<float>();
                                    AscendC::DataCopy(wScalar, wGm_[wOffset], 1);

                                    // Multiply and accumulate: yVec(o) += xVec * wVal
                                    AscendC::LocalTensor<float> tmpVec = tmpQueue_.AllocTensor<float>();
                                    const float wVal = wScalar.GetValue(0); // read local scalar
                                    AscendC::Muls(tmpVec, xVec, wVal, curLenD);

                                    AscendC::LocalTensor<float> yVec = yTile[o * tileD_];
                                    AscendC::Add(yVec, yVec, tmpVec, curLenD);

                                    tmpQueue_.FreeTensor(tmpVec);
                                    scalarQueue_.FreeTensor(wScalar);
                                }

                                xQueue_.FreeTensor(xVec);
                            }
                        }
                    }
                }

                // Write out accumulators to Y
                for (uint32_t o = 0; o < ocChunk; ++o) {
                    const uint32_t oc = ocBase + o;
                    AscendC::LocalTensor<float> yVec = yTile[o * tileD_];
                    const uint32_t yOffset = ((((nIdx * Cout_ + oc) * Wo_ + wOut) * Ho_ + hOut) * Do_) + dBase;
                    AscendC::DataCopy(yGm_[yOffset], yVec, curLenD);
                }

                yTileQueue_.FreeTensor(yTile);
            }
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) const {
        return (a + b - 1u) / b;
    }
    __aicore__ inline uint32_t MinU32(uint32_t a, uint32_t b) const {
        return a < b ? a : b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yTileQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t W_{0}, H_{0}, D_{0};
    uint32_t Kw_{0}, Kh_{0}, Kd_{0};
    uint32_t Wo_{0}, Ho_{0}, Do_{0};
    uint32_t tileD_{0}, tileOc_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dSquareInputAsymmetricKernelCustom op;
    op.Init(
        x, weight, y,
        tiling_data.N, tiling_data.Cin, tiling_data.Cout,
        tiling_data.W, tiling_data.H, tiling_data.D,
        tiling_data.Kw, tiling_data.Kh, tiling_data.Kd,
        tiling_data.Wo, tiling_data.Ho, tiling_data.Do,
        tiling_data.tileD, tiling_data.tileOc
    );
    op.Process();
}
