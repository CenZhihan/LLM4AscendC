#include "kernel_operator.h"
#include "lib/conv/conv3d/conv3d_api.h"

using namespace AscendC;

class KernelConvStandard3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, const optiling::ConvStandard3dSquareInputSquareKernelCustomTilingData &t)
    {
        n_  = t.n();
        ci_ = t.ci();
        d_  = t.d();
        h_  = t.h();
        w_  = t.w();
        co_ = t.co();
        kd_ = t.kd();
        kh_ = t.kh();
        kw_ = t.kw();
        od_ = t.od();
        oh_ = t.oh();
        ow_ = t.ow();
        sd_ = t.stride_d();
        sh_ = t.stride_h();
        sw_ = t.stride_w();
        pd_ = t.pad_d();
        ph_ = t.pad_h();
        pw_ = t.pad_w();
        dd_ = t.dil_d();
        dh_ = t.dil_h();
        dw_ = t.dil_w();

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * ci_ * d_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, co_ * ci_ * kd_ * kh_ * kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * co_ * od_ * oh_ * ow_);
    }

    __aicore__ inline void Process()
    {
        // Use Conv3D library API for computation. Assumes groups=1, stride=1, padding=0, dilation=1.
        conv3d::Conv3DParams params;
        params.n = n_;
        params.c = ci_;
        params.d = d_;
        params.h = h_;
        params.w = w_;
        params.k = co_;
        params.kd = kd_;
        params.kh = kh_;
        params.kw = kw_;
        params.stride_d = sd_;
        params.stride_h = sh_;
        params.stride_w = sw_;
        params.pad_d = pd_;
        params.pad_h = ph_;
        params.pad_w = pw_;
        params.dilation_d = dd_;
        params.dilation_h = dh_;
        params.dilation_w = dw_;
        params.groups = 1;

        conv3d::TensorDesc x_desc{n_, ci_, d_, h_, w_};
        conv3d::TensorDesc w_desc{co_, ci_, kd_, kh_, kw_};
        conv3d::TensorDesc y_desc{n_, co_, od_, oh_, ow_};

        conv3d::Conv3DRunConfig run_cfg;
        run_cfg.block_num = static_cast<uint32_t>(GetBlockNum());
        run_cfg.block_idx = static_cast<uint32_t>(GetBlockIdx());

        // Execute
        conv3d::Conv3DCompute<float>(xGm_.GetGM(), wGm_.GetGM(), yGm_.GetGM(), x_desc, w_desc, y_desc, params, run_cfg);
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, ci_{0}, d_{0}, h_{0}, w_{0};
    uint32_t co_{0}, kd_{0}, kh_{0}, kw_{0};
    uint32_t od_{0}, oh_{0}, ow_{0};
    uint32_t sd_{1}, sh_{1}, sw_{1};
    uint32_t pd_{0}, ph_{0}, pw_{0};
    uint32_t dd_{1}, dh_{1}, dw_{1};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

}

    auto sizes_x = x.sizes();
    auto sizes_w = w.sizes();
    int64_t N  = sizes_x[0];

    int64_t D  = sizes_x[2];
    int64_t H  = sizes_x[3];

    int64_t Co = sizes_w[0];
    int64_t Kd = sizes_w[2];
    int64_t Kh = sizes_w[3];
    int64_t Kw = sizes_w[4];

    int64_t stride_d = 1, stride_h = 1, stride_w = 1;
    int64_t pad_d = 0, pad_h = 0, pad_w = 0;
