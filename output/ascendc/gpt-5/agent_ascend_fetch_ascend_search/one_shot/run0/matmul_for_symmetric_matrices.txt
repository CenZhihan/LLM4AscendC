host_operator_src="""
#include "register/op_def_registry.h"
#include "matmul_for_symmetric_matrices_custom_tiling.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForSymmetricMatricesCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t M = a_shape->GetOriginShape().GetDim(0);
    const uint64_t K = a_shape->GetOriginShape().GetDim(1);
    const uint64_t K2 = b_shape->GetOriginShape().GetDim(0);
    const uint64_t N = b_shape->GetOriginShape().GetDim(1);

    if (K != K2 || M != N || M != K) {
        // This kernel targets square symmetric matrices; fall back if not square-matched.
        return ge::GRAPH_FAILED;
    }

    const uint32_t N32 = static_cast<uint32_t>(N);

    // UB budget and simple tile heuristic for float32:
    // UB stores: A(tileM*tileK) + B(tileK*tileN) + C(tileM*tileN) + tmp(tileN)
    // bytes = 4 * (tileM*tileK + tileK*tileN + tileM*tileN + tileN)
    // Choose conservative tile sizes to fit < 160KB of the ~192KB per-core UB.
    uint32_t tileM = 64;
    uint32_t tileN = 64;
    uint32_t tileK = 64;

    // Adjust for small problems
    if (N32 < tileM) tileM = N32;
    if (N32 < tileN) tileN = N32;
    if (N32 < tileK) tileK = N32;

    // Compute blockDim based on number of tiles
    const uint32_t mTiles = CeilDiv(N32, tileM);
    const uint32_t nTiles = CeilDiv(N32, tileN);
    uint32_t totalTiles = mTiles * nTiles;
    if (totalTiles == 0) totalTiles = 1;

    uint32_t blockDim = totalTiles;
    if (blockDim < 4u) blockDim = 4u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N32);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = a_shape->GetOriginShape().GetDim(0);
    const int64_t K = a_shape->GetOriginShape().GetDim(1);
    const int64_t K2 = b_shape->GetOriginShape().GetDim(0);
    const int64_t N = b_shape->GetOriginShape().GetDim(1);
    if (K != K2 || M != N || M != K) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    gert::Shape out;
    out.SetDimNum(2);
    out.SetDim(0, M);
    out.SetDim(1, N);
    *c_shape = out;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include <cstdint>

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulForSymmetricMatricesCustom {
public:
    __aicore__ inline KernelMatmulForSymmetricMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileM,
                                uint32_t tileN, uint32_t tileK)
    {
        N_ = N;
        tM_ = tileM;
        tN_ = tileN;
        tK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(N_) * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(N_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(N_) * N_);

        // UB buffers for one tile of A, B, C and a temp vector
        pipe_.InitBuffer(aQueue_, BUFFER_NUM, static_cast<uint32_t>(tM_) * tK_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, BUFFER_NUM, static_cast<uint32_t>(tK_) * tN_ * sizeof(float));
        pipe_.InitBuffer(cQueue_, BUFFER_NUM, static_cast<uint32_t>(tM_) * tN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, static_cast<uint32_t>(tN_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t mTiles = (N_ + tM_ - 1u) / tM_;
        const uint32_t nTiles = (N_ + tN_ - 1u) / tN_;
        const uint32_t totalTiles = mTiles * nTiles;

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t mi = tileId / nTiles;
            const uint32_t ni = tileId % nTiles;
            const uint32_t mStart = mi * tM_;
            const uint32_t nStart = ni * tN_;
            const uint32_t mLen = AscendC::Min(tM_, N_ - mStart);
            const uint32_t nLen = AscendC::Min(tN_, N_ - nStart);

            AscendC::LocalTensor<float> aLocal = aQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> cLocal = cQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpVec = tmpQueue_.AllocTensor<float>();

            // Initialize C tile to zero
            AscendC::Muls(cLocal, cLocal, 0.0f, tM_ * tN_);

            for (uint32_t kStart = 0; kStart < N_; kStart += tK_) {
                const uint32_t kLen = AscendC::Min(tK_, N_ - kStart);

                // Load A tile (mLen x kLen)
                for (uint32_t i = 0; i < mLen; ++i) {
                    AscendC::DataCopy(aLocal[i * tK_], aGm_[(mStart + i) * N_ + kStart], kLen);
                }

                // Load B tile (kLen x nLen)
                for (uint32_t p = 0; p < kLen; ++p) {
                    AscendC::DataCopy(bLocal[p * tN_], bGm_[(kStart + p) * N_ + nStart], nLen);
                }

                // Accumulate: for each k within the k-tile, do outer-product-style update
                for (uint32_t p = 0; p < kLen; ++p) {
                    // For each row in current C tile
                    for (uint32_t i = 0; i < mLen; ++i) {
                        // scalar from A row i, column p of the k-tile
                        float a_scalar = aLocal.GetValue(i * tK_ + p);

                        // tmpVec = a_scalar * B_row(p)
                        AscendC::Muls(tmpVec, bLocal[p * tN_], a_scalar, nLen);

                        // C_row(i) += tmpVec
                        AscendC::Add(cLocal[i * tN_], cLocal[i * tN_], tmpVec, nLen);
                    }
                }
            }

            // Store back C tile (mLen x nLen)
            for (uint32_t i = 0; i < mLen; ++i) {
                AscendC::DataCopy(cGm_[(mStart + i) * N_ + nStart], cLocal[i * tN_], nLen);
            }

            tmpQueue_.FreeTensor(tmpVec);
            cQueue_.FreeTensor(cLocal);
            bQueue_.FreeTensor(bLocal);
            aQueue_.FreeTensor(aLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> aQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> cQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t tM_{0};
    uint32_t tN_{0};
    uint32_t tK_{0};
};

extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForSymmetricMatricesCustom op;
    op.Init(A, B, C,
            tiling_data.N,
            tiling_data.tileM,
            tiling_data.tileN,
            tiling_data.tileK);
    op.Process();
}
