static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : (a + b - 1u) / b;
}

    uint64_t workItems = n * cout * dout;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t din, uint32_t hin, uint32_t win,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout,
                                uint32_t tileW)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        din_ = din; hin_ = hin; win_ = win;
        kd_ = kd; kh_ = kh; kw_ = kw;
        dout_ = dout; hout_ = hout; wout_ = wout;
        tileW_ = tileW == 0 ? 64u : tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * cin_ * din_ * hin_ * win_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(cin_ * cout_ * kd_ * kh_ * kw_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_ * cout_ * dout_ * hout_ * wout_));

        // Local buffers:
        // yLocal (tileW), tmpX (tileW), tmpBuf (tileW)
        pipe_.InitBuffer(yOutQueue_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueA_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueB_, BUFFER_NUM, tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0) return;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Work partitioning by (n, cout, dout)
        const uint64_t totalItems = static_cast<uint64_t>(n_) * cout_ * dout_;
        if (blockIdx >= totalItems) return;

        const uint32_t itemsPerN = cout_ * dout_;
        const uint32_t curN = blockIdx / itemsPerN;
        const uint32_t rem = blockIdx % itemsPerN;
        const uint32_t curCout = rem / dout_;
        const uint32_t curDout = rem % dout_;

        // Iterate hOut and tiles over W-out
        for (uint32_t hOut = 0; hOut < hout_; ++hOut) {
            for (uint32_t wStart = 0; wStart < wout_; wStart += tileW_) {
                const uint32_t curTile = (wStart + tileW_ <= wout_) ? tileW_ : (wout_ - wStart);
                if (curTile == 0) continue;

                // yLocal <- 0
                AscendC::LocalTensor<float> yLocal = yOutQueue_.AllocTensor<float>();
                AscendC::Duplicate(yLocal, 0.0f, curTile);

                // Accumulate contributions
                for (uint32_t ic = 0; ic < cin_; ++ic) {
                    // Determine di for each kd with curDout = di + kd
                    for (uint32_t kd = 0; kd < kd_; ++kd) {
                        int32_t di = static_cast<int32_t>(curDout) - static_cast<int32_t>(kd);
                        if (di < 0 || di >= static_cast<int32_t>(din_)) continue;

                        for (uint32_t kh = 0; kh < kh_; ++kh) {
                            int32_t hi = static_cast<int32_t>(hOut) - static_cast<int32_t>(kh);
                            if (hi < 0 || hi >= static_cast<int32_t>(hin_)) continue;

                            for (uint32_t kw = 0; kw < this->kw_; ++kw) {
                                // wOut = wi + kw -> wi0 for vector start:
                                int32_t wi0 = static_cast<int32_t>(wStart) - static_cast<int32_t>(kw);
                                // Clip range into [0, win_)
                                int32_t leftPad = wi0 < 0 ? (-wi0) : 0;
                                int32_t rightExcess = (wi0 + static_cast<int32_t>(curTile) > static_cast<int32_t>(win_))
                                                        ? (wi0 + static_cast<int32_t>(curTile) - static_cast<int32_t>(win_))
                                                        : 0;
                                int32_t validLen = static_cast<int32_t>(curTile) - leftPad - rightExcess;
                                if (validLen <= 0) {
                                    continue;
                                }

                                // Load x vector: x[n, ic, di, hi, wi0+leftPad : wi0+leftPad+validLen)
                                const uint32_t xOff = ((((curN * cin_ + ic) * din_ + static_cast<uint32_t>(di)) * hin_ + static_cast<uint32_t>(hi)) * win_
                                                       + static_cast<uint32_t>(wi0 + leftPad));
                                AscendC::LocalTensor<float> xVec = tmpQueueA_.AllocTensor<float>();
                                AscendC::DataCopy(xVec, xGm_[xOff], static_cast<uint32_t>(validLen));

                                // Prepare shifted buffer of length curTile
                                AscendC::LocalTensor<float> shiftBuf = tmpQueueB_.AllocTensor<float>();
                                AscendC::Duplicate(shiftBuf, 0.0f, curTile);
                                // Place xVec into shiftBuf starting at offset leftPad
                                if (validLen > 0) {
                                    AscendC::DataCopy(shiftBuf[leftPad], xVec, static_cast<uint32_t>(validLen));
                                }

                                // Load weight scalar w[ic, cout, kd, kh, kw]
                                const uint32_t wOff = ((((ic * cout_) + curCout) * kd_ + kd) * kh_ + kh) * this->kw_ + kw;
                                // Copy 1 element to xVec[0] temporarily and read back via vector-scalar ops:
                                // Build a vector filled with weight by duplicating the first element after copy
                                AscendC::LocalTensor<float> wTmp = tmpQueueA_.AllocTensor<float>();
                                AscendC::DataCopy(wTmp, wGm_[wOff], 1);
                                // Broadcast weight into tmpQueueA_'s buffer region (reuse xVec buffer as w-broadcast)
                                float weightScalar = 0.0f; // not directly accessible from local tensor; approximate by vector ops
                                // Use vector move: copy wTmp[0] into wTmp[1..curTile-1]
                                AscendC::Repeat(wTmp, wTmp, 1, curTile); // replicate the first element across curTile

                                // Multiply shiftBuf by weight vector and accumulate into yLocal
                                AscendC::Mul(shiftBuf, shiftBuf, wTmp, curTile);
                                AscendC::Add(yLocal, yLocal, shiftBuf, curTile);

                                tmpQueueA_.FreeTensor(wTmp);
                                tmpQueueB_.FreeTensor(shiftBuf);
                                tmpQueueA_.FreeTensor(xVec);
                            } // kw
                        } // kh
                    } // kd
                } // ic

                // Store yLocal to global y
                const uint32_t yOff = ((((curN * cout_) + curCout) * dout_ + curDout) * hout_ + hOut) * wout_ + wStart;
                AscendC::DataCopy(yGm_[yOff], yLocal, curTile);

                yOutQueue_.FreeTensor(yLocal);
            } // wStart
        } // hOut
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yOutQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueueA_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueB_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t din_{0}, hin_{0}, win_{0};
    uint32_t kd_{0}, kh_{0}, kw_{0};
    uint32_t dout_{0}, hout_{0}, wout_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(
        x, weight, y,
        tilingData.n, tilingData.cin, tilingData.cout,
        tilingData.din, tilingData.hin, tilingData.win,
        tilingData.kd, tilingData.kh, tilingData.kw,
        tilingData.dout, tilingData.hout, tilingData.wout,
        tilingData.tileW
    );
    op.Process();
}
