kernel_src="""

#include "kernel_operator.h"
// Optionally include high-level conv backprop API headers if available
// #include "conv_backprop_api.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t ci, uint32_t hi, uint32_t wi,
                                uint32_t co, uint32_t kh, uint32_t kw,
                                uint32_t ho, uint32_t wo,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padTop, uint32_t padLeft,
                                uint32_t padBottom, uint32_t padRight,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups,
                                uint32_t totalOutElems, uint32_t tileLen)
    {
        n_ = n; ci_ = ci; hi_ = hi; wi_ = wi;
        co_ = co; kh_ = kh; kw_ = kw;
        ho_ = ho; wo_ = wo;
        sH_ = strideH; sW_ = strideW;
        pT_ = padTop; pL_ = padLeft; pB_ = padBottom; pR_ = padRight;
        dH_ = dilationH; dW_ = dilationW;
        groups_ = groups;
        totalOutElems_ = totalOutElems;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * ci_ * hi_ * wi_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(ci_) * co_ * kh_ * kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * co_ * ho_ * wo_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, 16 * sizeof(float)); // small scratch
    }

    __aicore__ inline void Process()
    {
        if (totalOutElems_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutElems_) return;

        uint32_t curLen = totalOutElems_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        ComputeAndWrite(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalOutElems_) break;
            uint32_t nextLen = totalOutElems_ - nextBase;
            if (nextLen > tileLen_) nextLen = tileLen_;
            ComputeAndWrite(nextBase, nextLen);
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void ComputeAndWrite(uint32_t outBase, uint32_t len)
    {
        // Allocate local output tile buffer
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        // Initialize to zeros
        AscendC::Dup(yLocal, 0.0f, len);

        // For each element in the output tile, compute direct accumulation
        for (uint32_t i = 0; i < len; ++i) {
            const uint32_t yIndex = outBase + i;
            uint32_t t = yIndex;
            const uint32_t wo = t % wo_;
            t /= wo_;
            const uint32_t ho = t % ho_;
            t /= ho_;
            const uint32_t co = t % co_;
            const uint32_t n  = t / co_;

            float acc = 0.0f;

            // Iterate all input channels and kernel spatial positions
            for (uint32_t ci = 0; ci < ci_; ++ci) {
                for (uint32_t r = 0; r < kh_; ++r) {
                    // effective input spatial based on transpose formula
                    int64_t hi_src = static_cast<int64_t>(ho) + static_cast<int64_t>(pT_) - static_cast<int64_t>(r) * static_cast<int64_t>(dH_);
                    // Since strides are 1, we don't need to check divisibility
                    if (hi_src < 0 || hi_src >= static_cast<int64_t>(hi_)) continue;
                    for (uint32_t s = 0; s < kw_; ++s) {
                        int64_t wi_src = static_cast<int64_t>(wo) + static_cast<int64_t>(pL_) - static_cast<int64_t>(s) * static_cast<int64_t>(dW_);
                        if (wi_src < 0 || wi_src >= static_cast<int64_t>(wi_)) continue;

                        const uint64_t xOff =
                            (((static_cast<uint64_t>(n) * ci_ + ci) * hi_ + static_cast<uint32_t>(hi_src)) * wi_ + static_cast<uint32_t>(wi_src));

                        const uint64_t wOff =
                            ((((static_cast<uint64_t>(ci) * co_) + co) * kh_) + r) * kw_ + s;

                        // Load scalar x and w, accumulate
                        float x_val = 0.0f;
                        float w_val = 0.0f;
                        // scratch 1-element local tensors
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, xGm_[xOff], 1);
                        x_val = tmp.GetValue(0);
                        AscendC::DataCopy(tmp, wGm_[wOff], 1);
                        w_val = tmp.GetValue(0);
                        tmpQueue_.FreeTensor(tmp);
                        acc += x_val * w_val;
                    }
                }
            }
            // Store acc to yLocal[i]
            yLocal.SetValue(i, acc);
        }

        // Write back
        AscendC::DataCopy(yGm_[outBase], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, ci_{0}, hi_{0}, wi_{0};
    uint32_t co_{0}, kh_{0}, kw_{0};
    uint32_t ho_{0}, wo_{0};
    uint32_t sH_{1}, sW_{1};
    uint32_t pT_{0}, pL_{0}, pB_{0}, pR_{0};
    uint32_t dH_{1}, dW_{1};
    uint32_t groups_{1};

    uint32_t totalOutElems_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.ci, tiling_data.hi, tiling_data.wi,
            tiling_data.co, tiling_data.kh, tiling_data.kw,
            tiling_data.ho, tiling_data.wo,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padTop, tiling_data.padLeft,
            tiling_data.padBottom, tiling_data.padRight,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups,
            tiling_data.totalOutElems, tiling_data.tileLength);
    op.Process();
}
