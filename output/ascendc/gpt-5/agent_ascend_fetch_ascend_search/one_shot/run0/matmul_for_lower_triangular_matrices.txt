host_operator_src="""
#include "matmul_for_lower_triangular_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t AlignUp(uint32_t x, uint32_t a)
{
    return ((x + a - 1u) / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB budget assumptions (per-core)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 16;              // prefer multiples of 16 for vector alignment
static constexpr uint32_t MAX_BLOCKS = 24;               // upper bound for block dim

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForLowerTriangularMatricesCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    if (a_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const int64_t N0 = a_shape->GetOriginShape().GetDim(0);
    const int64_t N1 = a_shape->GetOriginShape().GetDim(1);
    if (N0 <= 0 || N1 <= 0 || N0 != N1) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(N0);

    // We allocate three tiles in UB: A(m x k), B(k x n), C(m x n), plus a temp vector length n.
    // bytes = 4 * (m*k + k*n + m*n + n)
    // We choose square tiling m=n=k=tile so bytes ~= 4 * (3*tile^2 + tile).
    // Solve 4 * (3*tile^2 + tile) <= UB_BUDGET_BYTES -> tile bound.
    // We'll search a reasonable tile size aligned to 16.
    uint32_t maxTile = AlignDown(N, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = ALIGN_ELEMS;

    // Simple search decreasing by ALIGN_ELEMS to meet UB budget.
    uint32_t tile = maxTile;
    while (tile >= ALIGN_ELEMS) {
        uint64_t bytes = 4ull * (3ull * tile * tile + tile);
        if (bytes <= UB_BUDGET_BYTES) break;
        tile -= ALIGN_ELEMS;
    }
    if (tile < ALIGN_ELEMS) tile = ALIGN_ELEMS;
    if (tile > N) tile = AlignDown(N, ALIGN_ELEMS) > 0 ? AlignDown(N, ALIGN_ELEMS) : N;

    const uint32_t tileM = tile;
    const uint32_t tileN = tile;
    const uint32_t tileK = tile;

    // Determine grid of tiles
    const uint32_t mTiles = CeilDiv(N, tileM);
    const uint32_t nTiles = CeilDiv(N, tileN);
    uint32_t totalTiles = mTiles * nTiles;

    // Set blockDim based on number of tiles
    uint32_t blockDim = totalTiles;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim < 8u) blockDim = (totalTiles < 8u ? totalTiles : 8u);
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace_sizes = context->GetWorkspaceSizes(1);
    workspace_sizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulForLowerTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForLowerTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(N_) * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(N_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(N_) * N_);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t mTiles = (N_ + tileM_ - 1u) / tileM_;
        const uint32_t nTiles = (N_ + tileN_ - 1u) / tileN_;
        const uint32_t totalTiles = mTiles * nTiles;

        // Grid-stride over (mTile, nTile) tiles
        for (uint32_t tileIndex = blockIdx; tileIndex < totalTiles; tileIndex += blockNum) {
            uint32_t mi = tileIndex / nTiles;    // row tile index
            uint32_t nj = tileIndex % nTiles;    // col tile index

            uint32_t rowStart = mi * tileM_;
            uint32_t colStart = nj * tileN_;
            uint32_t curM = MinUInt(tileM_, N_ - rowStart);
            uint32_t curN = MinUInt(tileN_, N_ - colStart);

            // Allocate C tile and zero-initialize
            AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
            AscendC::Dup(cLocal, 0.0f, curM * curN);

            // K-loop
            for (uint32_t kStart = 0; kStart < N_; kStart += tileK_) {
                uint32_t curK = MinUInt(tileK_, N_ - kStart);

                AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
                AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();

                // Copy A submatrix: [curM x curK] from GM to UB, row by row
                for (uint32_t r = 0; r < curM; ++r) {
                    const uint32_t gmOff = (rowStart + r) * N_ + kStart;
                    AscendC::DataCopy(aLocal[r * tileK_], aGm_[gmOff], curK);
                }

                // Copy B submatrix: [curK x curN] from GM to UB, row by row
                for (uint32_t p = 0; p < curK; ++p) {
                    const uint32_t gmOff = (kStart + p) * N_ + colStart;
                    AscendC::DataCopy(bLocal[p * tileN_], bGm_[gmOff], curN);
                }

                // Compute C += A * B for this k block
                // Row-wise accumulation: cRow += sum_p (a[i,p] * b[p,:])
                AscendC::LocalTensor<float> tmpVec = tmpQueue_.AllocTensor<float>();
                for (uint32_t i = 0; i < curM; ++i) {
                    AscendC::LocalTensor<float> cRow = cLocal[i * tileN_];
                    for (uint32_t p = 0; p < curK; ++p) {
                        float a_val = aLocal.GetValue(i * tileK_ + p);
                        AscendC::LocalTensor<float> bRow = bLocal[p * tileN_];
                        // tmpVec = bRow * a_val
                        AscendC::Muls(tmpVec, bRow, a_val, curN);
                        // cRow += tmpVec
                        AscendC::Add(cRow, cRow, tmpVec, curN);
                    }
                }
                tmpQueue_.FreeTensor(tmpVec);

                inQueueA_.FreeTensor(aLocal);
                inQueueB_.FreeTensor(bLocal);
            }

            // Apply lower-triangular mask within this tile: zero elements where j_global > i_global
            for (uint32_t i = 0; i < curM; ++i) {
                int64_t i_global = static_cast<int64_t>(rowStart) + static_cast<int64_t>(i);
                // Maximum valid j in this row: j_global <= i_global => j_local <= i_global - colStart
                int64_t maxValidLocal = i_global - static_cast<int64_t>(colStart);
                if (maxValidLocal < 0) {
                    // Entire row in this tile is above diagonal; set all to 0 (already zero, but enforce)
                    AscendC::LocalTensor<float> rowView = cLocal[i * tileN_];
                    AscendC::Dup(rowView, 0.0f, curN);
                } else if (maxValidLocal < static_cast<int64_t>(curN)) {
                    // Zero the tail range (maxValidLocal+1 .. curN-1)
                    uint32_t start = static_cast<uint32_t>(maxValidLocal + 1);
                    if (start < curN) {
                        AscendC::LocalTensor<float> tailView = cLocal[i * tileN_ + start];
                        AscendC::Dup(tailView, 0.0f, curN - start);
                    }
                }
            }

            // Write C tile back to GM
            for (uint32_t r = 0; r < curM; ++r) {
                const uint32_t gmOff = (rowStart + r) * N_ + colStart;
                AscendC::LocalTensor<float> rowView = cLocal[r * tileN_];
                AscendC::DataCopy(cGm_[gmOff], rowView, curN);
            }

            outQueueC_.FreeTensor(cLocal);
        }
    }

private:
    __aicore__ inline uint32_t MinUInt(uint32_t a, uint32_t b) { return a < b ? a : b; }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t tileM_{0};
    uint32_t tileN_{0};
    uint32_t tileK_{0};
};

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForLowerTriangularMatricesCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
