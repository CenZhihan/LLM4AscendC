    // Expect NCHW for x: [N, Cin, H, W]
    const int64_t N  = x_shape->GetOriginShape().GetDim(0);

    const int64_t CinW = w_shape->GetOriginShape().GetDim(1);

    const int64_t Kw  = w_shape->GetOriginShape().GetDim(3);

    if (CinW != Cin) {
        return ge::GRAPH_FAILED;
    }

    // No padding, stride=1
    const int64_t OH = H - Kh + 1;
    const int64_t OW = W - Kw + 1;

                              static_cast<uint64_t>(OH) * static_cast<uint64_t>(OW);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float subtract1 = 0.5f;
    float subtract2 = 0.2f;
    if (attrs != nullptr) {
        if (attrs->GetNumAttrs() >= 1) {
            const float *p1 = attrs->GetAttrPointer<float>(0);
            if (p1 != nullptr) subtract1 = *p1;
        }
        if (attrs->GetNumAttrs() >= 2) {
            const float *p2 = attrs->GetAttrPointer<float>(1);
            if (p2 != nullptr) subtract2 = *p2;
        }
    }

    // Determine blockDim conservatively: up to 24 cores for stability
    uint32_t blockDim = static_cast<uint32_t>(totalLen);
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim == 0u) blockDim = 1u;

    tiling.set_Cin(static_cast<uint32_t>(Cin));
    tiling.set_Cout(static_cast<uint32_t>(Cout));

    tiling.set_Kh(static_cast<uint32_t>(Kh));
    tiling.set_Kw(static_cast<uint32_t>(Kw));
    tiling.set_OH(static_cast<uint32_t>(OH));
    tiling.set_OW(static_cast<uint32_t>(OW));
    tiling.set_K(static_cast<uint32_t>(Cin * Kh * Kw));
    tiling.set_totalLength(static_cast<uint32_t>(totalLen)); // note: clamp if > uint32_t, typical sizes fit

    const int64_t Cin = x_shape->GetDim(1);
    const int64_t H = x_shape->GetDim(2);
    const int64_t W = x_shape->GetDim(3);

    const int64_t Cout = w_shape->GetDim(0);
    const int64_t Kh = w_shape->GetDim(2);
    const int64_t Kw = w_shape->GetDim(3);

    const int64_t OH = H - Kh + 1;
    const int64_t OW = W - Kw + 1;
    if (OH <= 0 || OW <= 0) {

    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, OH);
    y_shape->SetDim(3, OW);
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

class KernelConv2dSubtractSubtractMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractSubtractMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t Kh, uint32_t Kw,
                                uint32_t OH, uint32_t OW, uint32_t K,
                                uint32_t biasLen, float subtract1, float subtract2)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout; H_ = H; W_ = W;
        Kh_ = Kh; Kw_ = Kw; OH_ = OH; OW_ = OW; K_ = K;
        hasBias_ = (biasLen > 0);
        subtract1_ = subtract1;
        subtract2_ = subtract2;
        totalOut_ = (uint64_t)N_ * (uint64_t)Cout_ * (uint64_t)OH_ * (uint64_t)OW_;

        xStrideN_ = Cin_ * H_ * W_;
        xStrideC_ = H_ * W_;
        xStrideH_ = W_;
        xStrideW_ = 1;

        yStrideN_ = Cout_ * OH_ * OW_;
        yStrideC_ = OH_ * OW_;
        yStrideH_ = OW_;
        yStrideW_ = 1;

        // Set GM buffers
        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cout_ * Cin_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Cout_ * OH_ * OW_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)Cout_);
        }

        // Initialize local buffers for patch computation
        pipe_.InitBuffer(xPatchQ_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(wPatchQ_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(mulQ_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, 2, sizeof(float)); // for yLocal and temp
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t idx = blockIdx; idx < totalOut_; idx += blockNum) {
            // Decode coordinates: idx -> (n, co, oh, ow)
            uint64_t tmp = idx;
            const uint32_t ow = static_cast<uint32_t>(tmp % OW_); tmp /= OW_;
            const uint32_t oh = static_cast<uint32_t>(tmp % OH_); tmp /= OH_;
            const uint32_t co = static_cast<uint32_t>(tmp % Cout_); tmp /= Cout_;
            const uint32_t n  = static_cast<uint32_t>(tmp);

            // Prepare local tensors
            AscendC::LocalTensor<float> xPatch = xPatchQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> wPatch = wPatchQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> mulBuf = mulQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = scalarQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = scalarQ_.AllocTensor<float>();

            // Load weight vector for this output channel: contiguous K elements
            const uint64_t wBase = (uint64_t)co * (uint64_t)K_;
            AscendC::DataCopy(wPatch, wGm_[wBase], K_);

            // Gather input patch (Cin x Kh x Kw) for this (n, oh, ow)
            uint32_t off = 0;
            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const uint64_t xBase = (uint64_t)n * xStrideN_
                                          + (uint64_t)ci * xStrideC_
                                          + (uint64_t)(oh + kh) * xStrideH_
                                          + (uint64_t)ow * xStrideW_;
                    // copy Kw contiguous elements along width
                    AscendC::DataCopy(xPatch[off], xGm_[xBase], Kw_);
                    off += Kw_;
                }
            }

            // Dot product: sum(xPatch * wPatch)
            AscendC::Mul(mulBuf, xPatch, wPatch, K_);
            AscendC::ReduceSum(yLocal, mulBuf, K_);

            // Add bias if present
            if (hasBias_) {
                AscendC::DataCopy(tLocal, bGm_[co], 1);
                AscendC::Add(yLocal, yLocal, tLocal, 1);
            }

            // Subtractions
            AscendC::Adds(yLocal, yLocal, -subtract1_, 1);
            AscendC::Adds(yLocal, yLocal, -subtract2_, 1);

            // Mish: y = x * tanh(ln(1 + exp(x)))
            AscendC::Exp(tLocal, yLocal, 1);
            AscendC::Adds(tLocal, tLocal, 1.0f, 1);
            AscendC::Ln(tLocal, tLocal, 1);
            AscendC::Tanh(tLocal, tLocal, 1);
            AscendC::Mul(yLocal, yLocal, tLocal, 1);

            // Store result
            const uint64_t yOff = (uint64_t)n * yStrideN_
                                + (uint64_t)co * yStrideC_
                                + (uint64_t)oh * yStrideH_
                                + (uint64_t)ow * yStrideW_;
            AscendC::DataCopy(yGm_[yOff], yLocal, 1);

            // Free locals
            scalarQ_.FreeTensor(tLocal);
            scalarQ_.FreeTensor(yLocal);
            mulQ_.FreeTensor(mulBuf);
            wPatchQ_.FreeTensor(wPatch);
            xPatchQ_.FreeTensor(xPatch);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xPatchQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wPatchQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> mulQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> bGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0}, H_{0}, W_{0}, Kh_{0}, Kw_{0}, OH_{0}, OW_{0}, K_{0};
    uint32_t xStrideN_{0}, xStrideC_{0}, xStrideH_{0}, xStrideW_{0};
    uint32_t yStrideN_{0}, yStrideC_{0}, yStrideH_{0}, yStrideW_{0};
    uint64_t totalOut_{0};
    bool hasBias_{false};
    float subtract1_{0.5f}, subtract2_{0.2f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_subtract_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractSubtractMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.Kh, tiling_data.Kw,
            tiling_data.OH, tiling_data.OW, tiling_data.K,
            tiling_data.biasLen, tiling_data.subtract1, tiling_data.subtract2);
    op.Process();
}
