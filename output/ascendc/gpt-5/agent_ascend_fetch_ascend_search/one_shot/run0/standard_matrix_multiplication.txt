host_operator_src="""

#include "standard_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// Choose a conservative row tile to balance occupancy and UB usage.
// For GEMM, rows should be multiple of 16 for better cube scheduling.
static constexpr uint32_t ROW_TILE_ALIGN = 16;
static constexpr uint32_t MIN_ROW_TILE = 64;
static constexpr uint32_t MAX_ROW_TILE = 256;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    StandardMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);
    if (shapeA->GetOriginFormat() != ge::FORMAT_ND ||
        shapeB->GetOriginFormat() != ge::FORMAT_ND) {
        return ge::GRAPH_FAILED;
    }

    const auto dimsA = shapeA->GetOriginShape().GetDims();
    const auto dimsB = shapeB->GetOriginShape().GetDims();
    if (dimsA.size() != 2 || dimsB.size() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(dimsA[0]);
    const uint32_t K = static_cast<uint32_t>(dimsA[1]);
    const uint32_t Kb = static_cast<uint32_t>(dimsB[0]);
    const uint32_t N = static_cast<uint32_t>(dimsB[1]);
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    // Pick tileM based on M; aim for around 16-32 blocks for good occupancy.
    uint32_t tileM = MAX_ROW_TILE;
    if (M < tileM) {
        const uint32_t aligned = AlignDown(M, ROW_TILE_ALIGN);
        tileM = (aligned > 0) ? aligned : M;
    } else {
        // Scale down tile for very large M to keep more blocks.
        const uint32_t targetBlocks = 32u;
        const uint32_t tentative = AlignDown(CeilDiv(M, targetBlocks), ROW_TILE_ALIGN);
        if (tentative >= MIN_ROW_TILE && tentative <= MAX_ROW_TILE) {
            tileM = tentative;
        }
    }
    if (tileM < MIN_ROW_TILE) tileM = MIN_ROW_TILE;
    tileM = AlignDown(tileM, ROW_TILE_ALIGN);
    if (tileM == 0) tileM = ROW_TILE_ALIGN;

    uint32_t numTiles = CeilDiv(M, tileM);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *A = context->GetInputShape(0);
    const gert::Shape *B = context->GetInputShape(1);
    const auto dimsA = A->GetOriginShape().GetDims();
    const auto dimsB = B->GetOriginShape().GetDims();
    if (dimsA.size() != 2 || dimsB.size() != 2) {
        return GRAPH_FAILED;
    }
    if (dimsA[1] != dimsB[0]) {
        return GRAPH_FAILED;
    }

    gert::Shape *C = context->GetOutputShape(0);
    gert::Shape result;
    result.SetOriginFormat(A->GetOriginFormat());
    result.SetStorageShape({dimsA[0], dimsB[1]});
    result.SetOriginShape({dimsA[0], dimsB[1]});
    *C = result;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != ge::DT_FLOAT || b_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class StandardMatrixMultiplicationCustom : public OpDef {
public:
    explicit StandardMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(StandardMatrixMultiplicationCustom);
} // namespace ops

kernel_src="""

// 纯cube模式（只有矩阵计算）场景下，需要设置该代码宏，并且必须在#include "lib/matmul_intf.h"之前设置
#define ASCENDC_CUBE_ONLY

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

class KernelStandardMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelStandardMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM)
    {
        M_ = M;
        K_ = K;
        N_ = N;
        tileM_ = tileM;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<size_t>(M_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<size_t>(K_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<size_t>(M_) * N_);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideRows = blockNum * tileM_;

        uint32_t baseRow = blockIdx * tileM_;
        if (baseRow >= M_) return;

        // Matmul type definitions: all tensors use GM position, ND format, float type.
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> aType;
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> bType;
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> cType;
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> biasType;
        AscendC::Matmul<aType, bType, cType, biasType> mm;

        while (true) {
            uint32_t rows = M_ - baseRow;
            if (rows > tileM_) rows = tileM_;

            // Each block computes a distinct [rows x N] slab of C.
            // A_sub: [rows, K], B: [K, N], C_sub: [rows, N].
            mm.Init();
            mm.SetA(aGm_[baseRow * K_], rows, K_, /*isTransA*/false);
            mm.SetB(bGm_[0], K_, N_, /*isTransB*/false);
            mm.SetC(cGm_[baseRow * N_], rows, N_);
            // No bias for standard matmul
            mm.Compute();
            mm.End();

            const uint32_t nextBase = baseRow + strideRows;
            if (nextBase >= M_) break;
            baseRow = nextBase;
        }
    }

private:
    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0};
    uint32_t K_{0};
    uint32_t N_{0};
    uint32_t tileM_{0};
};

extern "C" __global__ __aicore__ void standard_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelStandardMatrixMultiplicationCustom op;
    op.Init(A, B, C, tiling_data.M, tiling_data.K, tiling_data.N, tiling_data.tileM);
    op.Process();
}
"""

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    const auto M = A.size(0);
    const auto K = A.size(1);
    TORCH_CHECK(B.size(0) == K, "Incompatible matmul dimensions: A(M,K) and B(K,N)");
    const auto N = B.size(1);

    at::Tensor C = at::empty({M, N}, A.options());
    EXEC_NPU_CMD(aclnnStandardMatrixMultiplicationCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("standard_matrix_multiplication_custom", &matmul_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("standard_matrix_multiplication_custom", &matmul_impl_npu, "Standard MatMul (StandardMatrixMultiplicationCustom on NPU)");
}
