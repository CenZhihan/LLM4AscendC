static inline uint32_t AlignDown(uint32_t x, uint32_t a)

    return (x / a) * a;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// Conservative UB budget per core; used to select tile sizes.

static constexpr uint32_t ALIGN_ELEMS = 256;

        k = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));

        k = static_cast<uint32_t>(sz);

    // Buffers (floats) we reserve:

    //  - wrow[K] (optional reuse)
    //  - xChunk[tileK]
    //  - mulTmp[tileK]
    //  - yTile[tileRows]
    // Approx bytes = 4*(K + K + tileK + tileK + tileRows)
    // For safety, cap tileK to min(k, 2048) and align to 256.
    uint32_t tileK = k;
    const uint32_t capTileK = 2048u;
    if (tileK > capTileK) tileK = capTileK;
    tileK = AlignDown(tileK, ALIGN_ELEMS);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom {
public:
    __aicore__ inline KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileRows, uint32_t tileK)
    {
        m_ = m;
        n_ = n;
        k_ = k;
        tileRows_ = tileRows;
        tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(m_) * k_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(n_) * k_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, m_); // output [m, 1] flattened

        // Output tiles
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * sizeof(float));

        // Working buffers
        pipe_.InitBuffer(wsumQue_, 1, k_ * sizeof(float));        // sum_j W[j,:]
        pipe_.InitBuffer(wrowQue_, 1, k_ * sizeof(float));        // single W row
        pipe_.InitBuffer(xbufQue_, 1, tileK_ * sizeof(float));    // X chunk along K
        pipe_.InitBuffer(mulQue_, 1, tileK_ * sizeof(float));     // tmp mul
        pipe_.InitBuffer(scalQue_, 2, sizeof(float));             // scalar reduce buffers
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || k_ == 0) return;

        // Compute w_sum vector = sum over N rows of W, and b_sum scalar = sum(b)
        ComputeWSumAndBSum();

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideRows = blockNum * tileRows_;
        uint32_t baseRow = blockIdx * tileRows_;
        if (baseRow >= m_) return;

        // Prefetch first tile
        uint32_t curRows = m_ - baseRow;
        if (curRows > tileRows_) curRows = tileRows_;
        ComputeTile(baseRow, curRows);

        while (true) {
            const uint32_t nextBase = baseRow + strideRows;
            uint32_t nextRows = 0;
            if (nextBase < m_) {
                nextRows = m_ - nextBase;
                if (nextRows > tileRows_) nextRows = tileRows_;
                ComputeTile(nextBase, nextRows);
            }

            if (nextRows == 0) break;
            baseRow = nextBase;
            curRows = nextRows;
        }
    }

private:
    __aicore__ inline void ComputeWSumAndBSum()
    {
        // wsum = 0
        AscendC::LocalTensor<float> wsum = wsumQue_.AllocTensor<float>();
        AscendC::Dup(wsum, 0.0f, k_);

        // accumulate rows of W
        for (uint32_t j = 0; j < n_; ++j) {
            AscendC::LocalTensor<float> wrow = wrowQue_.AllocTensor<float>();
            AscendC::DataCopy(wrow, wGm_[static_cast<uint64_t>(j) * k_], k_);
            AscendC::Add(wsum, wsum, wrow, k_);
            wrowQue_.FreeTensor(wrow);
        }

        // b_sum = sum(b)
        bSum_ = 0.0f;
        uint32_t done = 0;
        while (done < n_) {
            uint32_t len = n_ - done;
            if (len > tileK_) len = tileK_;
            AscendC::LocalTensor<float> bchunk = xbufQue_.AllocTensor<float>();
            AscendC::DataCopy(bchunk, bGm_[done], len);
            AscendC::LocalTensor<float> btmp = scalQue_.AllocTensor<float>();
            AscendC::ReduceSum(btmp, bchunk, len);
            // Adds supports variable scalar; retrieve 1-element as scalar for accumulation
            float bpartial = 0.0f;
            AscendC::GetValue(bpartial, btmp, 0);
            bSum_ += bpartial;
            scalQue_.FreeTensor(btmp);
            xbufQue_.FreeTensor(bchunk);
            done += len;
        }

        wsumQue_.EnQue(wsum); // keep wsum enqueued for later DeQue use
    }

    __aicore__ inline void ComputeTile(uint32_t rowBase, uint32_t numRows)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Dup(yLocal, 0.0f, numRows);

        AscendC::LocalTensor<float> wsum = wsumQue_.DeQue<float>();

        // For each row in tile, compute dot(x[row], wsum) + b_sum
        for (uint32_t r = 0; r < numRows; ++r) {
            // Accumulator scalar
            float acc = 0.0f;
            uint32_t done = 0;
            while (done < k_) {
                uint32_t len = k_ - done;
                if (len > tileK_) len = tileK_;

                AscendC::LocalTensor<float> xchunk = xbufQue_.AllocTensor<float>();
                AscendC::LocalTensor<float> wchunk = xbufQue_.AllocTensor<float>();
                AscendC::LocalTensor<float> mulTmp = mulQue_.AllocTensor<float>();

                // Load x chunk
                const uint64_t xOffset = (static_cast<uint64_t>(rowBase + r) * k_) + done;
                AscendC::DataCopy(xchunk, xGm_[xOffset], len);
                // Copy wsum segment into wchunk
                AscendC::DataCopy(wchunk, wsum[done], len);
                // mul and reduce
                AscendC::Mul(mulTmp, xchunk, wchunk, len);
                AscendC::LocalTensor<float> tmpScal = scalQue_.AllocTensor<float>();
                AscendC::ReduceSum(tmpScal, mulTmp, len);
                float part = 0.0f;
                AscendC::GetValue(part, tmpScal, 0);
                acc += part;

                scalQue_.FreeTensor(tmpScal);
                mulQue_.FreeTensor(mulTmp);
                xbufQue_.FreeTensor(wchunk);
                xbufQue_.FreeTensor(xchunk);

                done += len;
            }
            acc += bSum_;
            // store acc into yLocal[r]
            AscendC::SetValue(yLocal, r, acc);
        }

        // Enqueue result and write back
        outQueueY_.EnQue(yLocal);
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[rowBase], yOut, numRows);
        outQueueY_.FreeTensor(yOut);

        // Re-enqueue wsum for next tiles
        wsumQue_.EnQue(wsum);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wsumQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wrowQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xbufQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> mulQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> scalQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t m_{0}, n_{0}, k_{0};
    uint32_t tileRows_{0}, tileK_{0};
    float bSum_{0.0f};
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom op;
    op.Init(x, w, b, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileRows, tiling_data.tileK);
    op.Process();
}
