kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvDepthwiseSeparable2dCustom {
public:
    __aicore__ inline KernelConvDepthwiseSeparable2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wdw, GM_ADDR wpw, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t w, uint32_t cOut,
                                uint32_t segLen, uint32_t segsPerRow, uint32_t ocChunk)
    {
        n_ = n; cIn_ = cIn; h_ = h; w_ = w; cOut_ = cOut;
        segLen_ = segLen; segsPerRow_ = segsPerRow; ocChunk_ = ocChunk;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_)*cIn_*h_*w_);
        wdwGm_.SetGlobalBuffer((__gm__ float*)wdw, static_cast<uint64_t>(cIn_) * 1u * 3u * 3u);
        wpwGm_.SetGlobalBuffer((__gm__ float*)wpw, static_cast<uint64_t>(cOut_) * cIn_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * cOut_ * h_ * w_);

        // UB buffers
        // yTile: ocChunk * segLen
        pipe_.InitBuffer(bigQueue_, BUFFER_NUM, ocChunk_ * segLen_ * sizeof(float));
        // wChunk: ocChunk * cIn
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, ocChunk_ * cIn_ * sizeof(float));
        // seg buffers for three rows and dw val: allocate generously
        uint32_t segBufSize = (segLen_ + 2u) * 3u + segLen_ + 16u;
        pipe_.InitBuffer(segQueue_, BUFFER_NUM, segBufSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Work partition: units over N * H * segsPerRow
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t unitsTotal = static_cast<uint64_t>(n_) * h_ * segsPerRow_;
        for (uint64_t unit = blockIdx; unit < unitsTotal; unit += blockNum) {
            uint32_t n = static_cast<uint32_t>(unit / (static_cast<uint64_t>(h_) * segsPerRow_));
            uint32_t rem = static_cast<uint32_t>(unit % (static_cast<uint64_t>(h_) * segsPerRow_));
            uint32_t row = rem / segsPerRow_;
            uint32_t segIdx = rem % segsPerRow_;
            uint32_t w0 = segIdx * segLen_;
            uint32_t posEff = w_ - w0;
            if (posEff > segLen_) posEff = segLen_;

            ComputeUnit(n, row, w0, posEff);
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((static_cast<uint64_t>(n) * cIn_ + c) * h_ + h) * w_ + w);
    }
    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((static_cast<uint64_t>(n) * cOut_ + c) * h_ + h) * w_ + w);
    }
    __aicore__ inline uint64_t OffsetWdw(uint32_t c) const
    {
        return static_cast<uint64_t>(c) * 9u;
    }
    __aicore__ inline uint64_t OffsetWpw(uint32_t oc, uint32_t c) const
    {
        return static_cast<uint64_t>(oc) * cIn_ + c;
    }

    __aicore__ inline void ZeroLocal(AscendC::LocalTensor<float> &buf, uint32_t len)
    {
        AscendC::Dup(buf, 0.0f, len);
    }

    __aicore__ inline void ComputeUnit(uint32_t n, uint32_t row, uint32_t w0, uint32_t posEff)
    {
        for (uint32_t ocStart = 0; ocStart < cOut_; ocStart += ocChunk_) {
            uint32_t ocEff = cOut_ - ocStart;
            if (ocEff > ocChunk_) ocEff = ocChunk_;

            AscendC::LocalTensor<float> yTile = bigQueue_.AllocTensor<float>();
            ZeroLocal(yTile, ocEff * segLen_);

            AscendC::LocalTensor<float> wChunk = wQueue_.AllocTensor<float>();
            // Copy contiguous block of pointwise weights: shape [Cout, Cin], row-major
            AscendC::DataCopy(wChunk, wpwGm_[ocStart * cIn_], ocEff * cIn_);

            // Streaming over input channels
            for (uint32_t c = 0; c < cIn_; ++c) {
                // Load depthwise 3x3 weights for channel c
                AscendC::LocalTensor<float> wdw9 = segQueue_.AllocTensor<float>();
                AscendC::DataCopy(wdw9, wdwGm_[OffsetWdw(c)], 9);

                // Prepare three row buffers with padding
                AscendC::LocalTensor<float> rowTop = segQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> rowMid = segQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> rowBot = segQueue_.AllocTensor<float>();

                ZeroLocal(rowTop, segLen_ + 2u);
                ZeroLocal(rowMid, segLen_ + 2u);
                ZeroLocal(rowBot, segLen_ + 2u);

                LoadRowWithPad(n, c, (row == 0u) ? (uint32_t)0u : row - 1u, (row == 0u), w0, rowTop);
                LoadRowWithPad(n, c, row, false, w0, rowMid);
                LoadRowWithPad(n, c, (row + 1u >= h_) ? (h_ - 1u) : (row + 1u), (row + 1u >= h_), w0, rowBot);

                // Depthwise conv result for this channel
                AscendC::LocalTensor<float> dwVal = segQueue_.AllocTensor<float>();
                ZeroLocal(dwVal, segLen_);

                ComputeDepthwise1x3Row(rowTop, rowMid, rowBot, wdw9, dwVal, posEff);

                // Accumulate into yTile via pointwise weights
                for (uint32_t ocRel = 0; ocRel < ocEff; ++ocRel) {
                    float wpc = wChunk[ocRel * cIn_ + c];
                    uint32_t yOff = ocRel * segLen_;
                    for (uint32_t i = 0; i < posEff; ++i) {
                        yTile[yOff + i] += dwVal[i] * wpc;
                    }
                }

                // Free temp buffers
                segQueue_.FreeTensor(dwVal);
                segQueue_.FreeTensor(rowBot);
                segQueue_.FreeTensor(rowMid);
                segQueue_.FreeTensor(rowTop);
                segQueue_.FreeTensor(wdw9);
            }

            // Write back yTile for oc range
            for (uint32_t ocRel = 0; ocRel < ocEff; ++ocRel) {
                uint32_t oc = ocStart + ocRel;
                uint64_t yOff = OffsetY(n, oc, row, w0);
                AscendC::DataCopy(yGm_[yOff], yTile[ocRel * segLen_], posEff);
            }

            wQueue_.FreeTensor(wChunk);
            bigQueue_.FreeTensor(yTile);
        }
    }

    // Helper: load a contiguous segment (len segLen_+2) of a given row with zero padding at boundaries
    __aicore__ inline void LoadRowWithPad(uint32_t n, uint32_t c, uint32_t row, bool isPadRow, uint32_t w0,
                                          AscendC::LocalTensor<float> &rowBuf)
    {
        if (isPadRow) {
            // already zeroed
            return;
        }
        // local buffer indexing aligned so that center at index i+1 corresponds to x[..., w0 + i]
        int64_t startW = static_cast<int64_t>(w0) - 1;
        uint32_t localStart = 0;
        uint32_t copyLen = segLen_ + 2u;

        uint32_t gmStartW = 0;
        if (startW < 0) {
            localStart = static_cast<uint32_t>(-startW);
            gmStartW = 0;
            if (localStart >= copyLen) return;
            copyLen -= localStart;
        } else {
            gmStartW = static_cast<uint32_t>(startW);
        }
        if (gmStartW >= w_) return;

        uint32_t maxCopy = w_ - gmStartW;
        if (copyLen > maxCopy) copyLen = maxCopy;

        uint64_t xOff = OffsetX(n, c, row, gmStartW);
        AscendC::DataCopy(rowBuf[localStart], xGm_[xOff], copyLen);
    }

    // Compute depthwise 3x3 output along width segment using three rows and 9 weights
    __aicore__ inline void ComputeDepthwise1x3Row(AscendC::LocalTensor<float> &rowTop,
                                                  AscendC::LocalTensor<float> &rowMid,
                                                  AscendC::LocalTensor<float> &rowBot,
                                                  AscendC::LocalTensor<float> &wdw9,
                                                  AscendC::LocalTensor<float> &dwVal,
                                                  uint32_t posEff)
    {
        float w00 = wdw9[0], w01 = wdw9[1], w02 = wdw9[2];
        float w10 = wdw9[3], w11 = wdw9[4], w12 = wdw9[5];
        float w20 = wdw9[6], w21 = wdw9[7], w22 = wdw9[8];

        for (uint32_t i = 0; i < posEff; ++i) {
            float t0 = rowTop[i];
            float t1 = rowTop[i + 1];
            float t2 = rowTop[i + 2];
            float m0 = rowMid[i];
            float m1 = rowMid[i + 1];
            float m2 = rowMid[i + 2];
            float b0 = rowBot[i];
            float b1 = rowBot[i + 1];
            float b2 = rowBot[i + 2];
            float v = t0 * w00 + t1 * w01 + t2 * w02
                    + m0 * w10 + m1 * w11 + m2 * w12
                    + b0 * w20 + b1 * w21 + b2 * w22;
            dwVal[i] = v;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bigQueue_; // yTile
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> wQueue_;   // pointwise weights chunk
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> segQueue_; // row buffers and temp

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wdwGm_;
    AscendC::GlobalTensor<float> wpwGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, h_{0}, w_{0}, cOut_{0};
    uint32_t segLen_{0}, segsPerRow_{0}, ocChunk_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_separable_2d_custom(GM_ADDR x, GM_ADDR w_depthwise, GM_ADDR w_pointwise, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwiseSeparable2dCustom op;
    op.Init(x, w_depthwise, w_pointwise, y,
            tiling_data.n, tiling_data.cIn, tiling_data.h, tiling_data.w, tiling_data.cOut,
            tiling_data.segLen, tiling_data.segsPerRow, tiling_data.ocChunk);
    op.Process();
}
