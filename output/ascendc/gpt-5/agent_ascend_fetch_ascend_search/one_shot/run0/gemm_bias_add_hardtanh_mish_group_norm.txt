kernel_src="""
#include "kernel_operator.h"
#include <cmath>

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
                                GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN, uint32_t tileK,
                                uint32_t numGroups, float eps)
    {
        M_ = M; N_ = N; K_ = K;
        tN_ = tileN; tK_ = tileK;
        G_ = numGroups; eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, static_cast<uint64_t>(N_) * K_);
        lbGm_.SetGlobalBuffer((__gm__ float*)lin_bias, N_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, N_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, N_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        // Buffers: xChunk (tK), wBlock (tN * tK), yTile (tN), gammaTile (tN), betaTile (tN), groupTmp(3*G)
        pipe_.InitBuffer(xBuf_, 1, tK_ * sizeof(float));
        pipe_.InitBuffer(wBuf_, 1, (uint32_t)(tN_ * tK_) * sizeof(float));
        pipe_.InitBuffer(yBuf_, 1, tN_ * sizeof(float));
        pipe_.InitBuffer(gammaBuf_, 1, tN_ * sizeof(float));
        pipe_.InitBuffer(betaBuf_, 1, tN_ * sizeof(float));
        pipe_.InitBuffer(groupBuf_, 1, (uint32_t)(3 * G_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        // Each block processes rows in a grid-stride manner
        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            // Initialize group accumulators: sums and sqSums
            AscendC::LocalTensor<float> gLocal = groupBuf_.AllocTensor<float>();
            float* groupSums = gLocal.GetData();
            float* groupSqSums = groupSums + G_;
            float* groupInvStd = groupSqSums + G_;
            for (uint32_t gi = 0; gi < G_; ++gi) {
                groupSums[gi] = 0.0f;
                groupSqSums[gi] = 0.0f;
                groupInvStd[gi] = 0.0f;
            }

            // First pass: compute Y = X*W^T + lin_bias + bias, then hardtanh + mish; accumulate group stats; write Y to GM
            for (uint32_t n0 = 0; n0 < N_; n0 += tN_) {
                uint32_t curTN = (n0 + tN_ <= N_) ? tN_ : (N_ - n0);

                AscendC::LocalTensor<float> yLocal = yBuf_.AllocTensor<float>();
                float* yPtr = yLocal.GetData();
                for (uint32_t i = 0; i < curTN; ++i) yPtr[i] = 0.0f;

                // Accumulate over K in chunks
                for (uint32_t k0 = 0; k0 < K_; k0 += tK_) {
                    uint32_t curTK = (k0 + tK_ <= K_) ? tK_ : (K_ - k0);

                    // Load x chunk: x[m, k0:k0+curTK]
                    AscendC::LocalTensor<float> xLocal = xBuf_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[m * K_ + k0], curTK);

                    // Load weight block: rows n0..n0+curTN, cols k0..k0+curTK
                    AscendC::LocalTensor<float> wLocal = wBuf_.AllocTensor<float>();
                    // Copy by rows for contiguity
                    for (uint32_t i = 0; i < curTN; ++i) {
                        const uint64_t wRowOff = (uint64_t)(n0 + i) * K_ + k0;
                        AscendC::DataCopy(wLocal[i * curTK], wGm_[wRowOff], curTK);
                    }

                    // Compute partial matmul for tile yPtr += w_row · x_chunk
                    float* xv = xLocal.GetData();
                    float* wv = wLocal.GetData();
                    for (uint32_t i = 0; i < curTN; ++i) {
                        float acc = 0.0f;
                        float* wrow = wv + i * curTK;
                        for (uint32_t kk = 0; kk < curTK; ++kk) {
                            acc += wrow[kk] * xv[kk];
                        }
                        yPtr[i] += acc;
                    }

                    wBuf_.FreeTensor(wLocal);
                    xBuf_.FreeTensor(xLocal);
                }

                // Add fused biases: lin_bias and bias, then hardtanh and mish, accumulate group stats
                // Load biases slice
                AscendC::LocalTensor<float> gBias = gammaBuf_.AllocTensor<float>(); // reuse buffer
                AscendC::LocalTensor<float> bBias = betaBuf_.AllocTensor<float>();  // reuse buffer
                AscendC::DataCopy(gBias, lbGm_[n0], curTN);
                AscendC::DataCopy(bBias, bGm_[n0], curTN);
                float* lbPtr = gBias.GetData();
                float* bPtr = bBias.GetData();

                for (uint32_t i = 0; i < curTN; ++i) {
                    float v = yPtr[i] + lbPtr[i] + bPtr[i];
                    // hardtanh clamp to [-1, 1]
                    if (v < -1.0f) v = -1.0f;
                    if (v > 1.0f) v = 1.0f;
                    // mish activation: v * tanh(softplus(v)), softplus(v) = log1p(exp(v))
                    float sp = log1pf(expf(v));
                    float mv = v * tanhf(sp);
                    yPtr[i] = mv;

                    // accumulate group stats
                    uint32_t ch = n0 + i;
                    uint32_t gi = (G_ > 0) ? (ch * G_) / N_ : 0; // group index by partitioning channels
                    groupSums[gi] += mv;
                    groupSqSums[gi] += mv * mv;
                }

                // Write intermediate y (pre-norm) to GM
                AscendC::DataCopy(yGm_[m * N_ + n0], yLocal, curTN);

                betaBuf_.FreeTensor(bBias);
                gammaBuf_.FreeTensor(gBias);
                yBuf_.FreeTensor(yLocal);
            }

            // Finish stats: compute inv std for each group
            const uint32_t Cpg = N_ / G_;
            for (uint32_t gi = 0; gi < G_; ++gi) {
                float mean = groupSums[gi] / (float)Cpg;
                float var = groupSqSums[gi] / (float)Cpg - mean * mean;
                if (var < 0.0f) var = 0.0f;
                groupSums[gi] = mean;
                groupInvStd[gi] = 1.0f / sqrtf(var + eps_);
            }

            // Second pass: normalize with group stats, apply gamma/beta and write back to y
            for (uint32_t n0 = 0; n0 < N_; n0 += tN_) {
                uint32_t curTN = (n0 + tN_ <= N_) ? tN_ : (N_ - n0);

                AscendC::LocalTensor<float> yLocal = yBuf_.AllocTensor<float>();
                AscendC::LocalTensor<float> gLocalTile = gammaBuf_.AllocTensor<float>();
                AscendC::LocalTensor<float> bLocalTile = betaBuf_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, yGm_[m * N_ + n0], curTN);
                AscendC::DataCopy(gLocalTile, gammaGm_[n0], curTN);
                AscendC::DataCopy(bLocalTile, betaGm_[n0], curTN);

                float* yPtr = yLocal.GetData();
                float* gPtr = gLocalTile.GetData();
                float* bPtr = bLocalTile.GetData();

                for (uint32_t i = 0; i < curTN; ++i) {
                    uint32_t ch = n0 + i;
                    uint32_t gi = (G_ > 0) ? (ch * G_) / N_ : 0;
                    float mean = groupSums[gi];
                    float invStd = groupInvStd[gi];
                    float norm = (yPtr[i] - mean) * invStd;
                    yPtr[i] = norm * gPtr[i] + bPtr[i];
                }

                AscendC::DataCopy(yGm_[m * N_ + n0], yLocal, curTN);

                betaBuf_.FreeTensor(bLocalTile);
                gammaBuf_.FreeTensor(gLocalTile);
                yBuf_.FreeTensor(yLocal);
            }

            groupBuf_.FreeTensor(gLocal);
        }
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, 1> xBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> gammaBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> betaBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> groupBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> lbGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tN_{0}, tK_{0};
    uint32_t G_{1};
    float eps_{1e-5f};
};

            tiling_data.tileN, tiling_data.tileK,
