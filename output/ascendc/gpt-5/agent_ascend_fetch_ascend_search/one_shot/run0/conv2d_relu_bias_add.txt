namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // [N,Cin,H,W]
    const gert::Shape *w_shape = context->GetInputShape(1); // [Cout,Cin,Kh,Kw]
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;
    }
    auto xs = x_shape->GetOriginShape();
    auto ws = w_shape->GetOriginShape();
    if (xs.GetDimNum() != 4 || ws.GetDimNum() != 4) {
        return GRAPH_FAILED;
    }

    int64_t N = xs.GetDim(0);
    int64_t Cin = xs.GetDim(1);
    int64_t H = xs.GetDim(2);
    int64_t W = xs.GetDim(3);

    int64_t Cout = ws.GetDim(0);
    int64_t Cin_w = ws.GetDim(1);
    int64_t Kh = ws.GetDim(2);
    int64_t Kw = ws.GetDim(3);

    if (Cin != Cin_w) return GRAPH_FAILED;
    if (H < Kh || W < Kw) return GRAPH_FAILED;

    int64_t Hout = H - Kh + 1;
    int64_t Wout = W - Kw + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    gert::Shape out;
    out.SetDimNum(4);
    out.SetDim(0, N);
    out.SetDim(1, Cout);
    out.SetDim(2, Hout);
    out.SetDim(3, Wout);
    *y_shape = out;
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelConv2dReluBiasAddCustom {
public:
    __aicore__ inline KernelConv2dReluBiasAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR conv_b, GM_ADDR post_b, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t w_in, uint32_t kh, uint32_t kw,
                                uint32_t hout, uint32_t wout, uint32_t hasConvBias)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        h_ = h; w_ = w_in; kh_ = kh; kw_ = kw; hout_ = hout; wout_ = wout;
        hasConvBias_ = hasConvBias != 0;

        uint64_t xSize = static_cast<uint64_t>(n_) * cin_ * h_ * w_;
        uint64_t wSize = static_cast<uint64_t>(cout_) * cin_ * kh_ * kw_;
        uint64_t ySize = static_cast<uint64_t>(n_) * cout_ * hout_ * wout_;
        uint64_t bSize = static_cast<uint64_t>(cout_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);
        postBiasGm_.SetGlobalBuffer((__gm__ float*)post_b, bSize);
        if (hasConvBias_) {
            convBiasGm_.SetGlobalBuffer((__gm__ float*)conv_b, bSize);
        }

        // We compute per-output element; need small local buffers of length 1
        pipe_.InitBuffer(calcQ_, 4, 1 * sizeof(float)); // reuse tiny buffers
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        const uint64_t totalOut = static_cast<uint64_t>(n_) * cout_ * hout_ * wout_;
        for (uint64_t linear = blockIdx; linear < totalOut; linear += blockNum) {
            ComputeOne(linear);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint64_t linear)
    {
        // Decode linear index to (n, co, ho, wo)
        uint64_t tmp = linear;
        const uint32_t wo = static_cast<uint32_t>(tmp % wout_); tmp /= wout_;
        const uint32_t ho = static_cast<uint32_t>(tmp % hout_); tmp /= hout_;
        const uint32_t co = static_cast<uint32_t>(tmp % cout_); tmp /= cout_;
        const uint32_t n  = static_cast<uint32_t>(tmp);

        LocalTensor<float> acc = calcQ_.AllocTensor<float>();
        LocalTensor<float> tmpMul = calcQ_.AllocTensor<float>();
        LocalTensor<float> xVal = calcQ_.AllocTensor<float>();
        LocalTensor<float> wVal = calcQ_.AllocTensor<float>();
        LocalTensor<float> bVal = calcQ_.AllocTensor<float>();

        // acc = conv_bias[co] if present else 0
        if (hasConvBias_) {
            DataCopy(bVal, convBiasGm_[co], 1);
            // acc <- bVal
            Add(acc, bVal, bVal, 1); // acc = bVal + bVal (temporary)
            // but we need acc = bVal; we can set acc=0 then add bVal
            // Re-initialize: set acc=0, then add bVal
            Dup(acc, 0.0f, 1);
            Add(acc, acc, bVal, 1);
        } else {
            Dup(acc, 0.0f, 1);
        }

        // Convolution accumulate
        for (uint32_t ci = 0; ci < cin_; ++ci) {
            for (uint32_t r = 0; r < kh_; ++r) {
                for (uint32_t c = 0; c < kw_; ++c) {
                    const uint32_t ih = ho + r;
                    const uint32_t iw = wo + c;

                    uint64_t xOff = (((static_cast<uint64_t>(n) * cin_ + ci) * h_ + ih) * w_ + iw);
                    uint64_t wOff = ((((static_cast<uint64_t>(co) * cin_ + ci) * kh_ + r) * kw_ + c));

                    DataCopy(xVal, xGm_[xOff], 1);
                    DataCopy(wVal, wGm_[wOff], 1);

                    Mul(tmpMul, xVal, wVal, 1);
                    Add(acc, acc, tmpMul, 1);
                }
            }
        }

        // ReLU
        Maxs(acc, acc, 0.0f, 1);

        // + post bias (broadcast per channel)
        DataCopy(bVal, postBiasGm_[co], 1);
        Add(acc, acc, bVal, 1);

        // Store to output
        uint64_t yOff = (((static_cast<uint64_t>(n) * cout_ + co) * hout_ + ho) * wout_ + wo);
        DataCopy(yGm_[yOff], acc, 1);

        calcQ_.FreeTensor(bVal);
        calcQ_.FreeTensor(wVal);
        calcQ_.FreeTensor(xVal);
        calcQ_.FreeTensor(tmpMul);
        calcQ_.FreeTensor(acc);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, 4> calcQ_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> convBiasGm_;
    GlobalTensor<float> postBiasGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t h_{0}, w_{0}, kh_{0}, kw_{0}, hout_{0}, wout_{0};
    bool hasConvBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x,
                                                                  GM_ADDR weight,
                                                                  GM_ADDR conv_bias,
                                                                  GM_ADDR bias,
                                                                  GM_ADDR y,
                                                                  GM_ADDR workspace,
                                                                  GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv2dReluBiasAddCustom op;
    op.Init(x, weight, conv_bias, bias, y,
            tilingData.n, tilingData.cin, tilingData.cout,
            tilingData.h, tilingData.w, tilingData.kh, tilingData.kw,
            tilingData.hout, tilingData.wout, tilingData.hasConvBias);
    op.Process();
}
