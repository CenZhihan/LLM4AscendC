host_operator_src="""

#include "matmul_for_upper_triangular_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core conservative
static constexpr uint32_t ELEMENT_BYTES = 4;            // float32

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const uint64_t N0 = static_cast<uint64_t>(a_shape->GetDim(0));
    const uint64_t N1 = static_cast<uint64_t>(a_shape->GetDim(1));
    const uint64_t M0 = static_cast<uint64_t>(b_shape->GetDim(0));
    const uint64_t M1 = static_cast<uint64_t>(b_shape->GetDim(1));
    if (N0 != N1 || M0 != M1 || N0 != M0) {
        // Require square matrices with identical N
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(N0);

    // Choose tiles to fit UB:
    // Buffers: A (tm*tk), B (tk*tn), C (tm*tn). Double buffer A and B -> 2*(tm*tk + tk*tn) + (tm*tn) floats.
    // bytes = ELEMENT_BYTES * (2*(tm*tk + tk*tn) + tm*tn)
    // Choose tm = tn = tk = baseTile as large as allowed; start with 64.
    uint32_t baseTile = 64;

    // Try to fit the buffer budget; reduce if needed.
    auto Fits = [&](uint32_t t)->bool{
        uint64_t a = static_cast<uint64_t>(t) * static_cast<uint64_t>(t);
        uint64_t b = a;
        uint64_t c = a;
        uint64_t floats = 2ull * (a + b) + c;
        uint64_t bytes = floats * ELEMENT_BYTES;
        return bytes <= UB_BUDGET_BYTES;
    };
    while (baseTile > 16 && !Fits(baseTile)) baseTile >>= 1;

    uint32_t tileM = baseTile;
    uint32_t tileN = baseTile;
    uint32_t tileK = baseTile;

    // Clamp to N if smaller.
    if (N < tileM) tileM = N;
    if (N < tileN) tileN = N;
    if (N < tileK) tileK = N;

    const uint32_t numTileRows = CeilDiv(N, tileM);
    // Estimate number of upper-triangular tiles ~ numTileRows*(numTileRows+1)/2. Use numTileRows to size blockDim.
    uint32_t blockDim = numTileRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    MatmulForUpperTriangularMatricesCustomTilingData tiling;
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

        return GRAPH_FAILED;

    // Output is NxN (same as inputs)

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulForUpperTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForUpperTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        N_ = N;
        tM_ = tileM;
        tN_ = tileN;
        tK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(N_) * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(N_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(N_) * N_);

        // Init UB buffers
        pipe_.InitBuffer(aQue_, BUFFER_NUM, static_cast<uint32_t>(tM_ * tK_ * sizeof(float)));
        pipe_.InitBuffer(bQue_, BUFFER_NUM, static_cast<uint32_t>(tK_ * tN_ * sizeof(float)));
        pipe_.InitBuffer(cQue_, 1, static_cast<uint32_t>(tM_ * tN_ * sizeof(float)));
        pipe_.InitBuffer(tmpQue_, 1, static_cast<uint32_t>(tN_ * sizeof(float)));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numTileRows = CeilDiv(N_, tM_);
        const uint32_t numTileCols = CeilDiv(N_, tN_);

        for (uint32_t tr = blockIdx; tr < numTileRows; tr += blockNum) {
            const uint32_t rowStart = tr * tM_;
            const uint32_t mLen = Min(tM_, N_ - rowStart);
            for (uint32_t tc = tr; tc < numTileCols; ++tc) {
                const uint32_t colStart = tc * tN_;
                const uint32_t nLen = Min(tN_, N_ - colStart);

                // Allocate C tile and zero-init
                AscendC::LocalTensor<float> cTile = cQue_.AllocTensor<float>();
                AscendC::Duplicate(cTile, 0.0f, static_cast<uint32_t>(tM_ * tN_));
                // Effective region (mLen x nLen) within cTile

                if (tr > tc) {
                    // Lower triangular tile: stay zeros and write out
                    StoreCTile(cTile, rowStart, colStart, mLen, nLen);
                    cQue_.FreeTensor(cTile);
                    continue;
                }

                // Accumulate A*B for this tile across K
                for (uint32_t ks = 0; ks < N_; ks += tK_) {
                    const uint32_t kLen = Min(tK_, N_ - ks);

                    AscendC::LocalTensor<float> aTile = aQue_.AllocTensor<float>();
                    AscendC::LocalTensor<float> bTile = bQue_.AllocTensor<float>();

                    // Load A tile: rows [rowStart, rowStart+mLen), cols [ks, ks+kLen)
                    for (uint32_t i = 0; i < mLen; ++i) {
                        const uint32_t gmOff = (rowStart + i) * N_ + ks;
                        AscendC::DataCopy(aTile[i * tK_], aGm_[gmOff], kLen);
                    }
                    // Load B tile: rows [ks, ks+kLen), cols [colStart, colStart+nLen)
                    for (uint32_t k = 0; k < kLen; ++k) {
                        const uint32_t gmOff = (ks + k) * N_ + colStart;
                        AscendC::DataCopy(bTile[k * tN_], bGm_[gmOff], nLen);
                    }

                    // Compute cTile[mLen x nLen] += aTile[mLen x kLen] * bTile[kLen x nLen]
                    for (uint32_t i = 0; i < mLen; ++i) {
                        // Pointer to C row segment
                        for (uint32_t k = 0; k < kLen; ++k) {
                            // Scale B row by a scalar and add to C row
                            AscendC::LocalTensor<float> tmp = tmpQue_.AllocTensor<float>();
                            // tmp = bRow * a_ik
                            float a_ik = aTile.GetValue(i * tK_ + k);
                            AscendC::Muls(tmp, bTile[k * tN_], a_ik, nLen);
                            // cRow += tmp
                            AscendC::Add(cTile[i * tN_], cTile[i * tN_], tmp, nLen);
                            tmpQue_.FreeTensor(tmp);
                        }
                    }

                    aQue_.FreeTensor(aTile);
                    bQue_.FreeTensor(bTile);
                }

                // If diagonal tile, zero its strictly-lower part within tile
                if (tr == tc) {
                    for (uint32_t i = 0; i < mLen; ++i) {
                        const uint32_t numToZero = Min(i, nLen);
                        if (numToZero > 0) {
                            AscendC::Muls(cTile[i * tN_], cTile[i * tN_], 0.0f, numToZero);
                        }
                    }
                }

                // Store C tile
                StoreCTile(cTile, rowStart, colStart, mLen, nLen);
                cQue_.FreeTensor(cTile);
            }
        }
    }

private:
    __aicore__ static inline uint32_t Min(uint32_t a, uint32_t b) { return a < b ? a : b; }
    __aicore__ static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

    __aicore__ inline void StoreCTile(AscendC::LocalTensor<float>& cTile,
                                      uint32_t rowStart, uint32_t colStart,
                                      uint32_t mLen, uint32_t nLen)
    {
        for (uint32_t i = 0; i < mLen; ++i) {
            const uint32_t gmOff = (rowStart + i) * N_ + colStart;
            AscendC::DataCopy(cGm_[gmOff], cTile[i * tN_], nLen);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bQue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> cQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t tM_{0};
    uint32_t tN_{0};
    uint32_t tK_{0};
};

extern "C" __global__ __aicore__ void matmul_for_upper_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForUpperTriangularMatricesCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
