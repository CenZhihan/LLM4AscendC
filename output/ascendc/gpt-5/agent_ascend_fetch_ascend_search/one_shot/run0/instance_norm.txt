host_operator_src="""
#include "instance_norm_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/utils/type_utils.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0U) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0U) ? 0U : (a + b - 1U) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    InstanceNormCustomTilingData tiling;

    // Extract shape NCHW
    const gert::Shape *inShape = context->GetInputShape(0);
    const std::vector<int64_t> &shapeVec = inShape->GetOriginShape().GetDims();
    if (shapeVec.size() != 4) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(shapeVec[0]);
    const uint32_t C = static_cast<uint32_t>(shapeVec[1]);
    const uint32_t H = static_cast<uint32_t>(shapeVec[2]);
    const uint32_t W = static_cast<uint32_t>(shapeVec[3]);

    const uint32_t groups = N * C;
    const uint32_t groupSize = H * W;

    // Attrs
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    const float eps = (epsPtr != nullptr) ? *epsPtr : 1e-5f;

    // Query ReduceSum tmp size for a candidate tile
    // We'll iterate tileLength from a safe upper bound down to fit UB given tmp buffer.
    uint32_t tileLen = 0;
    uint32_t reduceTmpBytes = 0;

    auto queryReduceTmp = [&](uint32_t len) -> uint32_t {
        // src shape is [1, len], reduce last dim
        ge::Shape rShape({1, static_cast<int64_t>(len)});
        uint32_t maxValue = 0, minValue = 0;
        bool isSrcInnerPad = ((len * sizeof(float)) % 32U) == 0U;
        bool isReuseSource = false;
        AscendC::GetReduceSumMaxMinTmpSize(
            rShape, ge::DataType::DT_FLOAT, AscendC::ReducePattern::AR,
            isSrcInnerPad, isReuseSource, maxValue, minValue);
        // doc says max == min currently
        return maxValue;
    };

    const uint32_t maxCandidate = AlignDown(std::min(groupSize, HARD_CAP_ELEMS), ALIGN_ELEMS);
    uint32_t candidate = (maxCandidate > 0U) ? maxCandidate : std::min(groupSize, HARD_CAP_ELEMS);
    if (candidate == 0U) {
        candidate = 1U;
    }

    // UB usage model:
    // - inQueueX: 2 * tileLen * 4
    // - outQueueY: 2 * tileLen * 4
    // - calcQueue: 2 * tileLen * 4
    // - reduce tmp: reduceTmpBytes
    // Total = 6 * tileLen * 4 + reduceTmpBytes
    while (candidate >= 1U) {
        uint32_t tmpBytes = queryReduceTmp(candidate);
        uint64_t needBytes = static_cast<uint64_t>(candidate) * 24ULL + static_cast<uint64_t>(tmpBytes);
        if (needBytes <= UB_BUDGET_BYTES) {
            tileLen = candidate;
            reduceTmpBytes = tmpBytes;
            break;
        }
        if (candidate <= ALIGN_ELEMS) {
            candidate = candidate / 2U;
        } else {
            candidate = AlignDown(candidate - ALIGN_ELEMS, ALIGN_ELEMS);
        }
    }

    if (tileLen == 0U) {
        // fallback to minimal tile 1 with its tmp
        tileLen = 1U;
        reduceTmpBytes = queryReduceTmp(tileLen);
    }
    if (tileLen > groupSize) {
        tileLen = groupSize;
    }

    // Blocks: one per group tile ideally; but upper/lower bounds for stability
    const uint32_t tilesPerGroup = (tileLen == 0U) ? 1U : CeilDiv(groupSize, tileLen);
    const uint32_t targetBlocks = std::min<uint32_t>(groups * tilesPerGroup, 24U);
    uint32_t blockDim = targetBlocks;
    if (blockDim < 8U) blockDim = 8U;

    context->SetBlockDim(blockDim);

    tiling.set_groups(groups);
    tiling.set_groupSize(groupSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_reduceTmpBytes(reduceTmpBytes);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace
    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

// Helper to check 32-byte alignment of inner axis
__aicore__ inline bool IsInnerPad32(uint32_t elems)
{
    return ((elems * sizeof(float)) % 32U) == 0U;
}

class KernelInstanceNormCustom {
public:
    __aicore__ inline KernelInstanceNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t groups, uint32_t groupSize,
                                uint32_t tileLength, uint32_t reduceTmpBytes,
                                float eps)
    {
        groups_ = groups;
        groupSize_ = groupSize;
        tileLen_ = tileLength;
        reduceTmpBytes_ = reduceTmpBytes;
        eps_ = eps;

        const uint64_t totalElems = static_cast<uint64_t>(groups_) * static_cast<uint64_t>(groupSize_);
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElems);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElems);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(calcQueue_, BUFFER_NUM, tileLen_ * sizeof(float));     // for square or constants
        pipe_.InitBuffer(tmpReduceQueue_, 1, reduceTmpBytes_);
    }

    __aicore__ inline void Process()
    {
        if (groups_ == 0 || groupSize_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        for (uint32_t g = blockIdx; g < groups_; g += blockNum) {
            const uint64_t base = static_cast<uint64_t>(g) * static_cast<uint64_t>(groupSize_);

            // Pass 1: compute sum and sumsq over the whole group
            float sum = 0.0f;
            float sumsq = 0.0f;

            uint32_t processed = 0;
            // Prefetch first tile
            uint32_t len0 = MinU32(groupSize_, tileLen_);
            if (len0 > 0) {
                CopyIn(base + 0, len0);
            }
            uint32_t remain = groupSize_ - len0;

            while (true) {
                // Dequeue and compute partial sums
                LocalTensor<float> xLocal = inQueueX_.DeQue<float>();

                // Reduce sum(xLocal)
                float s = ReduceLocalSum(xLocal, len0);
                sum += s;

                // Prepare square and reduce
                LocalTensor<float> sqLocal = calcQueue_.AllocTensor<float>();
                Mul(sqLocal, xLocal, xLocal, len0);
                float ss = ReduceLocalSum(sqLocal, len0);
                sumsq += ss;
                calcQueue_.FreeTensor(sqLocal);

                inQueueX_.FreeTensor(xLocal);

                if (remain == 0) break;

                // Prefetch next tile
                const uint32_t curOff = processed + len0;
                const uint32_t nextLen = MinU32(remain, tileLen_);
                CopyIn(base + curOff, nextLen);

                processed += len0;
                len0 = nextLen;
                remain -= nextLen;
            }

            const float m = static_cast<float>(groupSize_);
            const float mean = sum / m;
            float var = sumsq / m - mean * mean;
            if (var < 0.0f) var = 0.0f;
            const float invStd = 1.0f / sqrtf(var + eps_);

            // Pass 2: normalize and write back
            processed = 0;
            uint32_t len1 = MinU32(groupSize_, tileLen_);
            if (len1 > 0) {
                CopyIn(base + 0, len1);
            }
            remain = groupSize_ - len1;

            while (true) {
                LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // y = (x - mean) * invStd
                Adds(yLocal, xLocal, -mean, len1);
                Muls(yLocal, yLocal, invStd, len1);

                outQueueY_.EnQue(yLocal);
                inQueueX_.FreeTensor(xLocal);

                // copy out
                LocalTensor<float> yLocal2 = outQueueY_.DeQue<float>();
                DataCopy(yGm_[base + processed], yLocal2, len1);
                outQueueY_.FreeTensor(yLocal2);

                if (remain == 0) break;

                // prefetch next tile
                const uint32_t curOff = processed + len1;
                const uint32_t nextLen = MinU32(remain, tileLen_);
                CopyIn(base + curOff, nextLen);

                processed += len1;
                len1 = nextLen;
                remain -= nextLen;
            }
        }
    }

private:
    __aicore__ inline uint32_t MinU32(uint32_t a, uint32_t b) { return (a < b) ? a : b; }

    __aicore__ inline void CopyIn(uint64_t gmOffset, uint32_t len)
    {
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    // Reduce a local tensor to scalar sum using high-level ReduceSum with shared tmp.
    __aicore__ inline float ReduceLocalSum(LocalTensor<float>& src, uint32_t len)
    {
        // prepare dst (1 element) using a calc buffer
        LocalTensor<float> dst = calcQueue_.AllocTensor<float>();
        // Only 1 element of dst will be valid.
        uint32_t shape[2] = {1U, len};
        const bool srcInnerPad = IsInnerPad32(len);

        LocalTensor<uint8_t> tmp = tmpReduceQueue_.AllocTensor<uint8_t>();
        ReduceSum<float, Pattern::Reduce::AR, false>(dst, src, tmp, shape, srcInnerPad);
        tmpReduceQueue_.FreeTensor(tmp);

        // Read first element as scalar. LocalTensor provides indexer in device context.
        float val = dst.GetValue(0);
        calcQueue_.FreeTensor(dst);
        return val;
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, BUFFER_NUM> calcQueue_;
    TQue<TPosition::VECCALC, 1> tmpReduceQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t groups_{0};
    uint32_t groupSize_{0};
    uint32_t tileLen_{0};
    uint32_t reduceTmpBytes_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void instance_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelInstanceNormCustom op;
    op.Init(x, y,
            tiling_data.groups, tiling_data.groupSize,
            tiling_data.tileLength, tiling_data.reduceTmpBytes,
            tiling_data.eps);
    op.Process();
}
