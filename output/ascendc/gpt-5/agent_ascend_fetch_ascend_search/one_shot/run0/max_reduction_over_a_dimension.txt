host_operator_src="""

#include "max_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB budget and tiling policy:
// We maintain two local tiles (accumulator and current slice):
// bytes = 2 * tileLen * sizeof(float)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // encourage larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxReductionOverADimensionCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dimAttr = (dimPtr != nullptr) ? *dimPtr : 0;

    const gert::Shape *inShapeObj = context->GetInputShape(0);
    const auto &origin = inShapeObj->GetOriginShape();
    const int64_t rank = origin.GetDimNum();

    if (rank <= 0) {
        // Degenerate tensor
        context->SetBlockDim(1);
        tiling.set_outerLen(1);
        tiling.set_reduceLen(1);
        tiling.set_innerLen(1);
        tiling.set_tileLen(1);
        tiling.set_numInnerTiles(1);
        tiling.set_blockDim(1);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    int64_t dim = dimAttr;
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) {
        // fallback safe
        dim = rank - 1;
    }

    uint64_t outer = 1, reduce = 1, inner = 1;
    for (int64_t i = 0; i < dim; ++i) {
        outer *= static_cast<uint64_t>(origin.GetDim(i));
    }
    reduce = static_cast<uint64_t>(origin.GetDim(dim));
    for (int64_t i = dim + 1; i < rank; ++i) {
        inner *= static_cast<uint64_t>(origin.GetDim(i));
    }

    // Compute tileLen along the innermost (inner) dimension
    uint32_t maxTile = UB_BUDGET_BYTES / (2u * static_cast<uint32_t>(sizeof(float)));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t innerLen = static_cast<uint32_t>(inner);
    uint32_t tileLen = maxTile;
    if (innerLen < tileLen) {
        const uint32_t aligned = AlignDown(innerLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : innerLen;
    }
    if (tileLen == 0) tileLen = 1; // safe guard

    const uint64_t numTiles64 = CeilDiv(static_cast<uint32_t>(innerLen), tileLen);
    const uint64_t totalTasks64 = (outer == 0) ? 0 : outer * ((numTiles64 == 0) ? 1 : numTiles64);
    uint32_t blockDim = static_cast<uint32_t>(totalTasks64 == 0 ? 1 : totalTasks64);
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_outerLen(static_cast<uint32_t>(outer));
    tiling.set_reduceLen(static_cast<uint32_t>(reduce));
    tiling.set_innerLen(innerLen);
    tiling.set_tileLen(tileLen);
    tiling.set_numInnerTiles(static_cast<uint32_t>(numTiles64));
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin = x_shape->GetOriginShape();
    const int64_t rank = origin.GetDimNum();

    int64_t dimAttr = 0;
    if (context->GetAttrs() != nullptr) {
        const int64_t *dimPtr = context->GetAttrs()->GetAttrPointer<int64_t>(0);
        if (dimPtr != nullptr) dimAttr = *dimPtr;
    }
    int64_t dim = dimAttr;
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) {
        dim = (rank > 0) ? (rank - 1) : 0;
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> out_dims;
    out_dims.reserve(rank > 0 ? static_cast<size_t>(rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out_dims.push_back(origin.GetDim(i));
    }
    y_shape->SetDimNum(static_cast<int64_t>(out_dims.size()));
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(static_cast<int64_t>(i), out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

class KernelMaxReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMaxReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerLen, uint32_t reduceLen,
                                uint32_t innerLen, uint32_t tileLen,
                                uint32_t numInnerTiles)
    {
        outerLen_ = outerLen;
        reduceLen_ = reduceLen;
        innerLen_ = innerLen;
        tileLen_ = tileLen;
        numInnerTiles_ = (numInnerTiles == 0) ? ((innerLen_ + tileLen_ - 1) / tileLen_) : numInnerTiles;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(outerLen_) * reduceLen_ * innerLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(outerLen_) * innerLen_);

        // Two VECCALC buffers: accumulator (yLocal) and current slice (xLocal)
        pipe_.InitBuffer(calcQ_, 2, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outerLen_ == 0 || reduceLen_ == 0 || innerLen_ == 0 || tileLen_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalTasks = static_cast<uint64_t>(outerLen_) * numInnerTiles_;
        for (uint64_t t = blockIdx; t < totalTasks; t += blockNum) {
            const uint32_t outerIdx = static_cast<uint32_t>(t / numInnerTiles_);
            const uint32_t innerTileIdx = static_cast<uint32_t>(t % numInnerTiles_);

            const uint32_t innerOffset = innerTileIdx * tileLen_;
            uint32_t curLen = innerLen_ - innerOffset;
            if (curLen > tileLen_) curLen = tileLen_;

            // Base offset for this (outerIdx, innerOffset)
            const uint64_t base = (static_cast<uint64_t>(outerIdx) * reduceLen_ * innerLen_) + innerOffset;

            // Allocate local buffers
            AscendC::LocalTensor<float> yLocal = calcQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> xLocal = calcQ_.AllocTensor<float>();

            // Initialize accumulator with first row r=0
            AscendC::DataCopy(yLocal, xGm_[base], curLen);

            // Accumulate max across remaining rows
            for (uint32_t r = 1; r < reduceLen_; ++r) {
                const uint64_t rowBase = base + static_cast<uint64_t>(r) * innerLen_;
                AscendC::DataCopy(xLocal, xGm_[rowBase], curLen);
                AscendC::Max(yLocal, yLocal, xLocal, curLen);
            }

            // Write result
            const uint64_t yBase = static_cast<uint64_t>(outerIdx) * innerLen_ + innerOffset;
            AscendC::DataCopy(yGm_[yBase], yLocal, curLen);

            // Free locals
            calcQ_.FreeTensor(xLocal);
            calcQ_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> calcQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outerLen_ {0};
    uint32_t reduceLen_ {0};
    uint32_t innerLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numInnerTiles_ {0};
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.outerLen,
            tiling_data.reduceLen,
            tiling_data.innerLen,
            tiling_data.tileLen,
            tiling_data.numInnerTiles);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor max_reduce_dim_impl_npu(const at::Tensor& self, int64_t dim) {
    at::Tensor result;

    // Reduce removes the specified dimension.
    // Construct the expected output shape by removing 'dim'
    int64_t rank = self.dim();
    int64_t d = dim < 0 ? dim + rank : dim;
    TORCH_CHECK(d >= 0 && d < rank, \"dim out of range\");
    std::vector<int64_t> out_sizes;
    out_sizes.reserve(static_cast<size_t>(rank > 0 ? rank - 1 : 0));
    for (int64_t i = 0; i < rank; ++i) {
        if (i == d) continue;
        out_sizes.push_back(self.size(i));
    }
    result = at::empty(out_sizes, self.options());

    EXEC_NPU_CMD(aclnnMaxReductionOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_reduction_over_a_dimension_custom", &max_reduce_dim_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_reduction_over_a_dimension_custom", &max_reduce_dim_impl_npu, "Max reduction over a dimension (MaxReductionOverADimensionCustom on NPU)");
}
