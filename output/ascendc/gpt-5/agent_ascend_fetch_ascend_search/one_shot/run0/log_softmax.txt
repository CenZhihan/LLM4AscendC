host_operator_src="""
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
}  // namespace

namespace optiling {

// UB budget for this kernel per-core. We need:
//   inQueue: 2 * tileCols * 4B
//   outQueue: 2 * tileCols * 4B
//   tmpQueue: 2 * tileCols * 4B (exp buffer + reduce tmp)
//   scalar/reduce tmp: small, accounted in tmpQueue generously
// Total ~ 6 * tileCols * 4B => bytes = 24 * tileCols
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ALIGN_ELEMS = 2048;            // DMA-friendly
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // latency guard

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimAttr != nullptr) ? *dimAttr : 1;

    const gert::Shape *shape = context->GetInputShape(0);
    const auto &origin = shape->GetOriginShape();

    // Derive rows and cols given reduction along dim (default last dimension)
    int32_t dimNum = origin.GetDimNum();
    if (dim < 0) dim += dimNum;
    if (dim < 0) dim = dimNum - 1;
    if (dim >= dimNum) dim = dimNum - 1;

    uint64_t totalSize = origin.GetShapeSize();
    uint64_t cols = static_cast<uint64_t>(origin.GetDim(dim));
    uint64_t rows = (cols == 0) ? 0ULL : (totalSize / cols);

    // Compute tileCols from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 24;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileCols = static_cast<uint32_t>((cols < maxTile) ? cols : maxTile);
    if (cols < maxTile) {
        uint32_t aligned = AlignDown(static_cast<uint32_t>(cols), ALIGN_ELEMS);
        tileCols = (aligned > 0) ? aligned : static_cast<uint32_t>(cols);
    }

    // blockDim based on rows
    uint32_t blockDim = static_cast<uint32_t>(rows);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_rows(static_cast<uint32_t>(rows));
    tiling.set_cols(static_cast<uint32_t>(cols));
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t tileCols, int32_t dim)
    {
        rows_ = rows;
        cols_ = cols;
        tileCols_ = tileCols;
        dim_ = dim;

        totalLen_ = rows_ * cols_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileCols_ * sizeof(float));   // exp buf + shared reduce tmp
        pipe_.InitBuffer(scalarQueue_, 1, tileCols_ * sizeof(float)); // reuse for reduce dst (first elem)
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0 || tileCols_ == 0) return;

        // Only support reduction along last dimension
        if (dim_ != -1 && dim_ != static_cast<int32_t>(1)) {
            // Fallback: treat as last dimension
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            const uint32_t rowBase = row * cols_;

            // Pass1: row max
            float rowMax = -3.402823466e+38F; // -FLT_MAX
            for (uint32_t colBase = 0; colBase < cols_; colBase += tileCols_) {
                uint32_t len = cols_ - colBase;
                if (len > tileCols_) len = tileCols_;

                // Copy in
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + colBase], len);
                inQueueX_.EnQue(xLocal);

                // Reduce max on this chunk
                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> dstLocal = scalarQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceMax(dstLocal, xLocal, tmpLocal, static_cast<int32_t>(len), false);
                // read first element as scalar
                float chunkMax = AscendC::GetValue(dstLocal, 0);
                if (chunkMax > rowMax) rowMax = chunkMax;

                tmpQueue_.FreeTensor(tmpLocal);
                scalarQueue_.FreeTensor(dstLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            // Pass2: sum(exp(x - rowMax))
            float sumExp = 0.0f;
            for (uint32_t colBase = 0; colBase < cols_; colBase += tileCols_) {
                uint32_t len = cols_ - colBase;
                if (len > tileCols_) len = tileCols_;

                // Copy in
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + colBase], len);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

                // expLocal = exp(xLocal - rowMax)
                AscendC::Sub(expLocal, xLocal, rowMax, len);
                AscendC::Exp(expLocal, expLocal, len);

                // Reduce sum of exp
                AscendC::LocalTensor<float> dstLocal = scalarQueue_.AllocTensor<float>();
                AscendC::ReduceSum(dstLocal, expLocal, tmpLocal, static_cast<int32_t>(len));
                float part = AscendC::GetValue(dstLocal, 0);
                sumExp += part;

                scalarQueue_.FreeTensor(dstLocal);
                tmpQueue_.FreeTensor(tmpLocal);
                tmpQueue_.FreeTensor(expLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            // Compute log(sumExp)
            float logSum = logf(sumExp);

            // Pass3: y = x - rowMax - logSum
            for (uint32_t colBase = 0; colBase < cols_; colBase += tileCols_) {
                uint32_t len = cols_ - colBase;
                if (len > tileCols_) len = tileCols_;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + colBase], len);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                AscendC::Sub(yLocal, xLocal, rowMax, len);
                AscendC::Sub(yLocal, yLocal, logSum, len);

                outQueueY_.EnQue(yLocal);
                inQueueX_.FreeTensor(xLocal);

                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[rowBase + colBase], yLocal, len);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_{0};
    uint32_t cols_{0};
    uint32_t tileCols_{0};
    int32_t dim_{1};
    uint32_t totalLen_{0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.rows, tiling_data.cols,
            tiling_data.tileCols, tiling_data.dim);
    op.Process();
}
"""
