kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t w_in, uint32_t kh, uint32_t kw,
                                uint32_t oh, uint32_t ow,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups,
                                uint32_t tileOw)
    {
        N_ = n; Cin_ = cin; Cout_ = cout; H_ = h; W_ = w_in;
        Kh_ = kh; Kw_ = kw; Oh_ = oh; Ow_ = ow;
        strideH_ = strideH; strideW_ = strideW; padH_ = padH; padW_ = padW;
        dilH_ = dilationH; dilW_ = dilationW; groups_ = groups;
        tileOw_ = tileOw;

        // GM sizes in elements
        const uint64_t xSize = static_cast<uint64_t>(N_) * Cin_ * H_ * W_;
        const uint64_t wSize = static_cast<uint64_t>(Cout_) * Cin_ * Kh_ * Kw_;
        const uint64_t ySize = static_cast<uint64_t>(N_) * Cout_ * Oh_ * Ow_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // UB buffers
        // yBuf: Cout * tileOw
        pipe_.InitBuffer(yBufQueue_, 1, Cout_ * tileOw_ * sizeof(float));
        pipe_.InitBuffer(inVecQueue_, 1, tileOw_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQueue_, 1, tileOw_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || Oh_ == 0 || Ow_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tilesPerRow = CeilDiv(Ow_, tileOw_);
        const uint32_t totalTasks = N_ * Oh_ * tilesPerRow;

        for (uint32_t task = blockIdx; task < totalTasks; task += blockNum) {
            // Decode task -> (n, oh, tileId)
            uint32_t tmp = task;
            const uint32_t n = tmp / (Oh_ * tilesPerRow);
            tmp = tmp % (Oh_ * tilesPerRow);
            const uint32_t oh = tmp / tilesPerRow;
            const uint32_t tileId = tmp % tilesPerRow;

            const uint32_t owStart = tileId * tileOw_;
            uint32_t curTileLen = Ow_ - owStart;
            if (curTileLen > tileOw_) curTileLen = tileOw_;

            // UB allocations
            AscendC::LocalTensor<float> yBuf = yBufQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> inVec = inVecQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpVec = tmpVecQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> wScalarBuf = scalarQueue_.AllocTensor<float>();

            // Initialize all output rows to 0 for current tile
            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                AscendC::LocalTensor<float> yRow = yBuf[oc * tileOw_];
                AscendC::Duplicate(yRow, 0.0f, curTileLen);
            }

            // Convolution accumulation
            for (uint32_t ic = 0; ic < Cin_; ++ic) {
                for (uint32_t r = 0; r < Kh_; ++r) {
                    // input h index
                    int64_t hIn = static_cast<int64_t>(oh) * strideH_ + static_cast<int64_t>(r) * dilH_ - static_cast<int64_t>(padH_);
                    if (hIn < 0 || hIn >= static_cast<int64_t>(H_)) {
                        continue;
                    }
                    for (uint32_t s = 0; s < Kw_; ++s) {
                        // Build input vector for current (ic, r, s) across tile of output width
                        int64_t wStartIn = static_cast<int64_t>(owStart) * strideW_ + static_cast<int64_t>(s) * dilW_ - static_cast<int64_t>(padW_);
                        // Zero-fill the input vector
                        AscendC::Duplicate(inVec, 0.0f, curTileLen);

                        // Determine valid contiguous range
                        int64_t leftInvalid = 0;
                        int64_t rightInvalid = 0;
                        if (wStartIn < 0) leftInvalid = -wStartIn;
                        int64_t wEndIn = wStartIn + static_cast<int64_t>(curTileLen) - 1;
                        if (wEndIn >= static_cast<int64_t>(W_)) rightInvalid = wEndIn - static_cast<int64_t>(W_) + 1;
                        int64_t validLen = static_cast<int64_t>(curTileLen) - leftInvalid - rightInvalid;

                        if (validLen > 0) {
                            uint32_t vOff = static_cast<uint32_t>(leftInvalid);
                            uint32_t vLen = static_cast<uint32_t>(validLen);
                            uint32_t wBase = static_cast<uint32_t>(wStartIn + leftInvalid);

                            // GM offset for X: (((n*Cin + ic)*H + hIn) * W + wBase)
                            uint64_t xOff = ((((uint64_t)n * Cin_ + ic) * H_ + static_cast<uint32_t>(hIn)) * W_) + wBase;
                            AscendC::DataCopy(inVec[vOff], xGm_[xOff], vLen);
                        }

                        // Accumulate over output channels
                        for (uint32_t oc = 0; oc < Cout_; ++oc) {
                            // weight offset: (((oc*Cin + ic)*Kh + r)*Kw + s)
                            uint64_t wOff = ((((uint64_t)oc * Cin_ + ic) * Kh_ + r) * Kw_) + s;
                            // Load scalar weight into UB
                            AscendC::DataCopy(wScalarBuf, wGm_[wOff], 1);
                            // Read scalar from UB buffer to register
                            float wVal = wScalarBuf.GetValue(0);

                            // tmpVec = inVec * wVal
                            AscendC::Muls(tmpVec, inVec, wVal, curTileLen);
                            // yRow += tmpVec
                            AscendC::LocalTensor<float> yRow = yBuf[oc * tileOw_];
                            AscendC::Add(yRow, yRow, tmpVec, curTileLen);
                        }
                    }
                }
            }

            // Write results back to Y
            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                uint64_t yOff = ((((uint64_t)n * Cout_ + oc) * Oh_ + oh) * Ow_) + owStart;
                AscendC::LocalTensor<float> yRow = yBuf[oc * tileOw_];
                AscendC::DataCopy(yGm_[yOff], yRow, curTileLen);
            }

            // Free UB buffers
            scalarQueue_.FreeTensor(wScalarBuf);
            tmpVecQueue_.FreeTensor(tmpVec);
            inVecQueue_.FreeTensor(inVec);
            yBufQueue_.FreeTensor(yBuf);
        }
    }

private:
    static __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

private:
    // GM tensors
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    // UB queues
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yBufQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inVecQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVecQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    // dims and params
    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, Kh_{0}, Kw_{0}, Oh_{0}, Ow_{0};
    uint32_t strideH_{1}, strideW_{1}, padH_{0}, padW_{0}, dilH_{1}, dilW_{1}, groups_{1};
    uint32_t tileOw_{1};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w, tiling_data.kh, tiling_data.kw,
            tiling_data.oh, tiling_data.ow,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups,
            tiling_data.tileOw);
    op.Process();
}
