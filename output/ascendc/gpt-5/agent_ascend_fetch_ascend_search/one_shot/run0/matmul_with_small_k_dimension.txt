host_tiling_src="""

#include "register/tilingdata_base.h"
#include "register/op_def_registry.h"
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"

namespace optiling {
// Register TCubeTiling as the tiling data type for this op
REGISTER_TILING_DATA_CLASS(MatmulWithSmallKDimensionCustom, TCubeTiling)
}  // namespace optiling

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Read input shapes
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);
    if (aShape == nullptr || bShape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    // Expect 2D: A (M,K), B (K,N)
    const int64_t M = aShape->GetDim(0);
    const int64_t K = aShape->GetDim(1);
    const int64_t Kb = bShape->GetDim(0);
    const int64_t N = bShape->GetDim(1);
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    // Build matmul tiling using AscendC Matmul API
    platform_ascendc::PlatformAscendC ascendcPlatform(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling tiling(ascendcPlatform);

    // Types and formats
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);

    // Shapes and attributes: no transpose
    tiling.SetMNK(static_cast<uint32_t>(M), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    tiling.SetTranspose(false, false);

    // Optimize for small K by hinting preferKSmall if available (safe no-op if not supported)
    tiling.PreferSmallK(true);

    optiling::TCubeTiling tilingData;
    const int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Set block dim from tiling
    context->SetBlockDim(tilingData.GetBlockDim());

    // Save tiling buffer
    tilingData.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tilingData.GetDataSize());

    // Workspace requirement
    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = tilingData.GetWorkspaceSize();
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);
    if (aShape == nullptr || bShape == nullptr) return GRAPH_FAILED;
    if (aShape->GetDimNum() != 2 || bShape->GetDimNum() != 2) return GRAPH_FAILED;
    const int64_t M = aShape->GetDim(0);
    const int64_t K = aShape->GetDim(1);
    const int64_t Kb = bShape->GetDim(0);
    const int64_t N = bShape->GetDim(1);
    if (K != Kb) return GRAPH_FAILED;

    gert::Shape *cShape = context->GetOutputShape(0);
    cShape->SetDimNum(2);
    cShape->SetDim(0, M);
    cShape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) return GRAPH_FAILED;
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MatmulWithSmallKDimensionCustom : public OpDef {
public:
    explicit MatmulWithSmallKDimensionCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulWithSmallKDimensionCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                                                           GM_ADDR workspace, GM_ADDR tiling)
{
    // Use high-level Matmul API with precomputed tiling
    AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> mm;
    mm.Init(tiling);
    mm.SetWorkspace((__gm__ uint8_t*)workspace);

    // Bias is optional; pass nullptr to disable
    mm.Process((__gm__ float*)a, (__gm__ float*)b, (__gm__ float*)c, (__gm__ float*)nullptr);
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor matmul_with_small_k_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "Inputs must be 2D matrices");
    TORCH_CHECK(A.size(1) == B.size(0), "Incompatible matmul dimensions");
    at::Tensor C = at::empty({A.size(0), B.size(1)}, A.options());
    EXEC_NPU_CMD(aclnnMatmulWithSmallKDimensionCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_with_small_k_dimension_custom", &matmul_with_small_k_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_with_small_k_dimension_custom", &matmul_with_small_k_npu,
          "MatMul optimized for small K (MatmulWithSmallKDimensionCustom on NPU)");
