host_operator_src="""

#include "conv_standard_1d_dilated_strided_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static inline uint32_t MinU(uint32_t a, uint32_t b)
{
    return a < b ? a : b;
}

static inline uint32_t ComputeLout(uint32_t lin, uint32_t k, uint32_t stride, uint32_t dilation)
{
    // padding = 0
    // lout = floor((lin - dilation*(k-1) - 1)/stride + 1)
    if (lin <= dilation * (k - 1) + 1) {
        return 0;
    }
    return (uint32_t)((lin - dilation * (k - 1) - 1) / stride + 1);
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard1dDilatedStridedCustomTilingData tiling;

    // Fixed params as per model: stride=3, dilation=4, padding=0, groups=1
    const uint32_t stride = 3;
    const uint32_t dilation = 4;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t Lin = static_cast<uint32_t>(x_shape->GetDim(2));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t WCin = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t K    = static_cast<uint32_t>(w_shape->GetDim(2));

    if (WCin != Cin) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Lout = ComputeLout(Lin, K, stride, dilation);

    // UB budgeting (floats)
    // UB usage approx: 4 * (tileCo*Cin*K + tileCo*tileT + segLen)
    // segLen = (tileT-1)*stride + (K-1)*dilation + 1
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative share
    static constexpr uint32_t MAX_BLOCKS = 24;

    // choose a conservative tileT and tileCo
    uint32_t tileT = 128;
    if (Lout < tileT) tileT = Lout > 0 ? Lout : 1;

    // Try to pick tileCo so UB fits
    uint32_t segLen = (tileT > 0) ? ((tileT - 1u) * stride + (K - 1u) * dilation + 1u) : 1u;

    auto Fits = [&](uint32_t tCo)->bool {
        uint64_t floats = (uint64_t)tCo * Cin * K + (uint64_t)tCo * tileT + segLen;
        uint64_t bytes = floats * sizeof(float);
        return bytes < UB_BUDGET_BYTES;
    };

    uint32_t tileCo = 8;
    while (tileCo > 1 && !Fits(tileCo)) tileCo >>= 1;
    if (!Fits(tileCo)) {
        // fallback shrink tileT
        while (tileT > 16) {
            tileT >>= 1;
            segLen = ((tileT - 1u) * stride + (K - 1u) * dilation + 1u);
            if (Fits(tileCo)) break;
        }
        if (!Fits(tileCo)) {
            // As a last resort, tileCo=1
            tileCo = 1;
        }
    }
    if (tileCo > Cout) tileCo = Cout == 0 ? 1u : Cout;

    // blockDim based on number of tasks (N * ceil_div(Cout, tileCo))
    const uint32_t numCoTiles = CeilDiv(Cout, tileCo);
    const uint32_t totalTasks = N * numCoTiles;
    uint32_t blockDim = totalTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    // Save tiling to context
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_ci(Cin);
    tiling.set_co(Cout);
    tiling.set_lin(Lin);
    tiling.set_lout(Lout);
    tiling.set_k(K);
    tiling.set_stride(stride);
    tiling.set_dilation(dilation);
    tiling.set_tileT(tileT);
    tiling.set_tileCo(tileCo);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // [N, Cin, Lin]
    const gert::Shape *w_shape = context->GetInputShape(1); // [Cout, Cin, K]

    const int64_t N   = x_shape->GetDim(0);
    const int64_t Cin = x_shape->GetDim(1);
    const int64_t Lin = x_shape->GetDim(2);

    const int64_t Cout = w_shape->GetDim(0);
    const int64_t WCin = w_shape->GetDim(1);
    const int64_t K    = w_shape->GetDim(2);

    if (Cin != WCin) return GRAPH_FAILED;

    const int64_t stride = 3;
    const int64_t dilation = 4;
    const int64_t pad = 0;
    const int64_t Lout = (Lin + 2 * pad - dilation * (K - 1) - 1) / stride + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Lout);
    return GRAPH_SUCCESS;
}

}

kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard1dDilatedStridedCustom {
public:
    __aicore__ inline KernelConvStandard1dDilatedStridedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Lin, uint32_t Lout, uint32_t K,
                                uint32_t stride, uint32_t dilation,
                                uint32_t tileT, uint32_t tileCo)
    {
        N_ = N;
        Cin_ = Cin;
        Cout_ = Cout;
        Lin_ = Lin;
        Lout_ = Lout;
        K_ = K;
        stride_ = stride;
        dilation_ = dilation;
        tileT_ = tileT;
        tileCo_ = tileCo;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint32_t)((uint64_t)N_ * Cin_ * Lin_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint32_t)((uint64_t)Cout_ * Cin_ * K_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint32_t)((uint64_t)N_ * Cout_ * Lout_));

        // UB buffers
        // weight buffer: tileCo * Cin * K
        // x segment buffer: segLen = (tileT - 1)*stride + (K - 1)*dilation + 1
        // y buffer: tileCo * tileT
        uint32_t segLen = (tileT_ > 0) ? ((tileT_ - 1u) * stride_ + (K_ - 1u) * dilation_ + 1u) : 1u;
        pipe_.InitBuffer(weightQ_, 1, tileCo_ * Cin_ * K_ * sizeof(float));
        pipe_.InitBuffer(xQ_, 1, segLen * sizeof(float));
        pipe_.InitBuffer(yQ_, 1, tileCo_ * tileT_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || Lout_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numCoTiles = CeilDiv(Cout_, tileCo_);
        const uint32_t totalTasks = N_ * numCoTiles;

        for (uint32_t taskId = blockIdx; taskId < totalTasks; taskId += blockNum) {
            const uint32_t n = taskId / numCoTiles;
            const uint32_t coTileIdx = taskId % numCoTiles;
            const uint32_t coStart = coTileIdx * tileCo_;
            const uint32_t coEff = MinU(tileCo_, Cout_ - coStart);

            // Load weight tile [coEff, Cin, K] contiguous
            const uint32_t wTileElems = coEff * Cin_ * K_;
            AscendC::LocalTensor<float> wLocal = weightQ_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[(uint64_t)coStart * Cin_ * K_], wTileElems);
            weightQ_.EnQue(wLocal);

            for (uint32_t t0 = 0; t0 < Lout_; t0 += tileT_) {
                const uint32_t tl = MinU(tileT_, Lout_ - t0);
                const uint32_t segLen = (tl > 0) ? ((tl - 1u) * stride_ + (K_ - 1u) * dilation_ + 1u) : 1u;

                AscendC::LocalTensor<float> yLocal = yQ_.AllocTensor<float>();
                // zero-init yLocal (coEff * tl)
                for (uint32_t i = 0; i < coEff * tl; ++i) {
                    yLocal.SetValue(i, 0.0f);
                }

                // compute contributions per input channel
                for (uint32_t ci = 0; ci < Cin_; ++ci) {
                    // copy x segment for this channel
                    const uint32_t lbase = t0 * stride_;
                    const uint64_t xBaseOffset = (((uint64_t)n * Cin_ + ci) * Lin_) + lbase;
                    AscendC::LocalTensor<float> xLocal = xQ_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[xBaseOffset], segLen);

                    // get weight tile local again
                    AscendC::LocalTensor<float> wLocalRead = weightQ_.DeQue<float>();

                    // compute for each co in the tile
                    for (uint32_t coRel = 0; coRel < coEff; ++coRel) {
                        const uint64_t wOff = ((uint64_t)coRel * Cin_ + ci) * K_;

                        for (uint32_t i = 0; i < tl; ++i) {
                            float acc = yLocal.GetValue(coRel * tileT_ + i);
                            // K typically small (3)
                            for (uint32_t kk = 0; kk < K_; ++kk) {
                                const uint32_t xIdx = i * stride_ + kk * dilation_;
                                const float xv = xLocal.GetValue(xIdx);
                                const float ww = wLocalRead.GetValue(wOff + kk);
                                acc += xv * ww;
                            }
                            yLocal.SetValue(coRel * tileT_ + i, acc);
                        }
                    }

                    weightQ_.EnQue(wLocalRead);
                    xQ_.FreeTensor(xLocal);
                }

                // write back yLocal to GM for each co in tile
                for (uint32_t coRel = 0; coRel < coEff; ++coRel) {
                    const uint32_t coAbs = coStart + coRel;
                    const uint64_t yBaseOffset = (((uint64_t)n * Cout_ + coAbs) * Lout_) + t0;
                    AscendC::DataCopy(yGm_[yBaseOffset], yLocal[coRel * tileT_], tl);
                }

                yQ_.FreeTensor(yLocal);
            }

            weightQ_.FreeTensor(wLocal);
        }
    }

private:
    __aicore__ static inline uint32_t MinU(uint32_t a, uint32_t b) { return a < b ? a : b; }
    __aicore__ static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> weightQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Lin_{0}, Lout_{0}, K_{0};
    uint32_t stride_{1}, dilation_{1};
    uint32_t tileT_{1}, tileCo_{1};
};

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dDilatedStridedCustom op;
    op.Init(
        x, weight, y,
        tiling_data.n, tiling_data.ci, tiling_data.co,
        tiling_data.lin, tiling_data.lout, tiling_data.k,
        tiling_data.stride, tiling_data.dilation,
        tiling_data.tileT, tiling_data.tileCo);
    op.Process();
}
