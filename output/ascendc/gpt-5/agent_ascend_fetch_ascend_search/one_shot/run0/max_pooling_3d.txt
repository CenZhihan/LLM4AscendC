namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1) / b);
}
static inline int64_t FloorDiv(int64_t a, int64_t b)
{
    return static_cast<int64_t>(std::floor(static_cast<double>(a) / static_cast<double>(b)));
}
static inline int64_t CeilDivS(int64_t a, int64_t b)
{
    return static_cast<int64_t>(std::ceil(static_cast<double>(a) / static_cast<double>(b)));
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData t;

    const gert::Shape *in_shape = context->GetInputShape(0);
    if (in_shape->GetOriginShape().GetDimNum() != 5) {
        return ge::GRAPH_FAILED;
    }
    const int64_t N = in_shape->GetOriginShape().GetDim(0);
    const int64_t C = in_shape->GetOriginShape().GetDim(1);
    const int64_t Din = in_shape->GetOriginShape().GetDim(2);
    const int64_t Hin = in_shape->GetOriginShape().GetDim(3);
    const int64_t Win = in_shape->GetOriginShape().GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernel_size_ptr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride_ptr      = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding_ptr     = attrs->GetAttrPointer<int32_t>(2);
    const bool    *ceil_mode_ptr   = attrs->GetAttrPointer<bool>(3);

    int64_t Kd = static_cast<int64_t>(*kernel_size_ptr);
    int64_t Kh = static_cast<int64_t>(*kernel_size_ptr);
    int64_t Kw = static_cast<int64_t>(*kernel_size_ptr);

    int64_t Sd = (stride_ptr && *stride_ptr > 0) ? static_cast<int64_t>(*stride_ptr) : Kd;
    int64_t Sh = (stride_ptr && *stride_ptr > 0) ? static_cast<int64_t>(*stride_ptr) : Kh;
    int64_t Sw = (stride_ptr && *stride_ptr > 0) ? static_cast<int64_t>(*stride_ptr) : Kw;

    int64_t Pd = padding_ptr ? static_cast<int64_t>(*padding_ptr) : 0;
    int64_t Ph = padding_ptr ? static_cast<int64_t>(*padding_ptr) : 0;
    int64_t Pw = padding_ptr ? static_cast<int64_t>(*padding_ptr) : 0;

    const bool ceil_mode = (ceil_mode_ptr != nullptr) ? *ceil_mode_ptr : false;

    auto compute_out_axis = [&](int64_t in, int64_t k, int64_t s, int64_t p, bool ceilm) -> int64_t {
        if (ceilm) {
            int64_t out = CeilDivS(in + 2 * p - k + 1, s);
            // Adjust if window starts in right/bottom/back padding region
            if ((out - 1) * s >= in + p) {
                out -= 1;
            }
            return std::max<int64_t>(out, 0);
        } else {
            int64_t out = FloorDiv(in + 2 * p - k + 1, s);
            out = out + 1;
            if ((out - 1) * s >= in + p) {
                out -= 1;
            }
            return std::max<int64_t>(out, 0);
        }
    };

    int64_t Dout = compute_out_axis(Din, Kd, Sd, Pd, ceil_mode);
    int64_t Hout = compute_out_axis(Hin, Kh, Sh, Ph, ceil_mode);
    int64_t Wout = compute_out_axis(Win, Kw, Sw, Pw, ceil_mode);

    const uint64_t totalOutLen_u64 =
        static_cast<uint64_t>(N) * static_cast<uint64_t>(C) *
        static_cast<uint64_t>(Dout) * static_cast<uint64_t>(Hout) * static_cast<uint64_t>(Wout);
    const uint32_t totalOutLen = static_cast<uint32_t>(totalOutLen_u64);

    // UB budget: yLocal tile + small tmp buffer; choose conservative
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t ALIGN_ELEMS = 1024;
    static constexpr uint32_t HARD_CAP_ELEMS = 65536;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOutLen < tileLen) {
        const uint32_t aligned = AlignDown(totalOutLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutLen;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    t.set_totalOutLen(totalOutLen);
    t.set_tileLen(tileLen);
    t.set_blockDim(blockDim);

    t.set_N(static_cast<uint32_t>(N));
    t.set_C(static_cast<uint32_t>(C));
    t.set_Din(static_cast<uint32_t>(Din));
    t.set_Hin(static_cast<uint32_t>(Hin));
    t.set_Win(static_cast<uint32_t>(Win));

    t.set_Kd(static_cast<uint32_t>(Kd));
    t.set_Kh(static_cast<uint32_t>(Kh));
    t.set_Kw(static_cast<uint32_t>(Kw));

    t.set_Sd(static_cast<uint32_t>(Sd));
    t.set_Sh(static_cast<uint32_t>(Sh));
    t.set_Sw(static_cast<uint32_t>(Sw));

    t.set_Pd(static_cast<uint32_t>(Pd));
    t.set_Ph(static_cast<uint32_t>(Ph));
    t.set_Pw(static_cast<uint32_t>(Pw));

    t.set_Dout(static_cast<uint32_t>(Dout));
    t.set_Hout(static_cast<uint32_t>(Hout));
    t.set_Wout(static_cast<uint32_t>(Wout));

    t.set_ceilMode(ceil_mode ? 1u : 0u);

    t.SaveToBuffer(context->GetRawTilingData()->GetData(),
                   context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(t.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalOutLen, uint32_t tileLen,
                                uint32_t N, uint32_t C, uint32_t Din, uint32_t Hin, uint32_t Win,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Sd, uint32_t Sh, uint32_t Sw,
                                uint32_t Pd, uint32_t Ph, uint32_t Pw,
                                uint32_t Dout, uint32_t Hout, uint32_t Wout)
    {
        totalOutLen_ = totalOutLen;
        tileLen_ = tileLen;

        N_ = N; C_ = C; Din_ = Din; Hin_ = Hin; Win_ = Win;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        Sd_ = Sd; Sh_ = Sh; Sw_ = Sw;
        Pd_ = Pd; Ph_ = Ph; Pw_ = Pw;
        Dout_ = Dout; Hout_ = Hout; Wout_ = Wout;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(N_ * C_ * Din_ * Hin_ * Win_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutLen_) return;

        while (base < totalOutLen_) {
            uint32_t curLen = totalOutLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            ComputeTile(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t baseOut, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        // Initialize with a very small value
        AscendC::Dup(yLocal, -3.4028235e38f, len);

        AscendC::LocalTensor<float> xScalar = tmpQueue_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            uint32_t outIndex = baseOut + i;

            // Decode linear index to (n,c,od,oh,ow)
            uint32_t ow = outIndex % Wout_;
            uint32_t tmp = outIndex / Wout_;
            uint32_t oh = tmp % Hout_;
            tmp = tmp / Hout_;
            uint32_t od = tmp % Dout_;
            tmp = tmp / Dout_;
            uint32_t c = tmp % C_;
            uint32_t n = tmp / C_;

            // Compute start indices
            int32_t d0 = static_cast<int32_t>(od) * static_cast<int32_t>(Sd_) - static_cast<int32_t>(Pd_);
            int32_t h0 = static_cast<int32_t>(oh) * static_cast<int32_t>(Sh_) - static_cast<int32_t>(Ph_);
            int32_t w0 = static_cast<int32_t>(ow) * static_cast<int32_t>(Sw_) - static_cast<int32_t>(Pw_);

            int32_t dStart = d0 < 0 ? 0 : d0;
            int32_t hStart = h0 < 0 ? 0 : h0;
            int32_t wStart = w0 < 0 ? 0 : w0;

            int32_t dEnd = d0 + static_cast<int32_t>(Kd_);
            if (dEnd > static_cast<int32_t>(Din_)) dEnd = static_cast<int32_t>(Din_);
            int32_t hEnd = h0 + static_cast<int32_t>(Kh_);
            if (hEnd > static_cast<int32_t>(Hin_)) hEnd = static_cast<int32_t>(Hin_);
            int32_t wEnd = w0 + static_cast<int32_t>(Kw_);
            if (wEnd > static_cast<int32_t>(Win_)) wEnd = static_cast<int32_t>(Win_);

            // Iterate over valid window and update max
            for (int32_t id = dStart; id < dEnd; ++id) {
                for (int32_t ih = hStart; ih < hEnd; ++ih) {
                    for (int32_t iw = wStart; iw < wEnd; ++iw) {
                        uint32_t inOffset =
                            (((n * C_ + c) * Din_ + static_cast<uint32_t>(id)) * Hin_ + static_cast<uint32_t>(ih)) * Win_ +
                            static_cast<uint32_t>(iw);

                        AscendC::DataCopy(xScalar, xGm_[inOffset], 1);

                        // yLocal[i] = max(yLocal[i], xScalar[0])
                        AscendC::LocalTensor<float> ySlot = yLocal[i];
                        AscendC::Max(ySlot, ySlot, xScalar, 1);
                    }
                }
            }
        }

        AscendC::DataCopy(yGm_[baseOut], yLocal, len);
        tmpQueue_.FreeTensor(xScalar);
        outQueueY_.EnQue(yLocal);
        AscendC::LocalTensor<float> yDeq = outQueueY_.DeQue<float>();
        outQueueY_.FreeTensor(yDeq);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLen_{0};
    uint32_t tileLen_{0};

    uint32_t N_{0}, C_{0}, Din_{0}, Hin_{0}, Win_{0};
    uint32_t Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Sd_{0}, Sh_{0}, Sw_{0};
    uint32_t Pd_{0}, Ph_{0}, Pw_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.totalOutLen, tiling_data.tileLen,
            tiling_data.N, tiling_data.C, tiling_data.Din, tiling_data.Hin, tiling_data.Win,
            tiling_data.Kd, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Sd, tiling_data.Sh, tiling_data.Sw,
            tiling_data.Pd, tiling_data.Ph, tiling_data.Pw,
            tiling_data.Dout, tiling_data.Hout, tiling_data.Wout);
    op.Process();
}
