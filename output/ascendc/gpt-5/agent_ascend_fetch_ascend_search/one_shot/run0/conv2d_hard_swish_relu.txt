    const uint32_t Kw   = static_cast<uint32_t>(w_shape->GetDim(3));

    uint32_t blockDim = 24;

    tiling.set_kh(Kh);
    tiling.set_kw(Kw);
    tiling.set_ho(Ho);
    tiling.set_wo(Wo);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t VEC_BUFS = 4;

class KernelConv2dHardSwishReluCustom {
public:
    __aicore__ inline KernelConv2dHardSwishReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t w_in,
                                uint32_t kh, uint32_t kw,
                                uint32_t ho, uint32_t wo,
                                bool hasBias)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        h_ = h; wIn_ = w_in; kh_ = kh; kw_ = kw;
        ho_ = ho; wo_ = wo; hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * cin_ * h_ * wIn_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)cout_ * cin_ * kh_ * kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * cout_ * ho_ * wo_);

        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, cout_);
        }

        // Buffers sized by one output row length (wo_)
        uint32_t rowBytes = wo_ * sizeof(float);
        pipe_.InitBuffer(vecQ_, VEC_BUFS, rowBytes);
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || ho_ == 0 || wo_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t totalTasks = static_cast<uint64_t>(n_) * cout_ * ho_;

        for (uint64_t task = blockIdx; task < totalTasks; task += blockNum) {
            // Decode (n, co, ho)
            uint64_t tmp = task;
            uint32_t co = static_cast<uint32_t>(tmp % cout_);
            tmp /= cout_;
            uint32_t ho = static_cast<uint32_t>(tmp % ho_);
            uint32_t n  = static_cast<uint32_t>(tmp / ho_);

            ComputeRow(n, co, ho);
        }
    }

private:
    __aicore__ inline uint64_t XIndex(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * cin_ + c) * h_ + h) * wIn_ + w;
    }

    __aicore__ inline uint64_t WIndex(uint32_t co, uint32_t ci, uint32_t kh, uint32_t kw) const
    {
        return (((uint64_t)co * cin_ + ci) * kh_ + kh) * kw_ + kw;
    }

    __aicore__ inline uint64_t YIndex(uint32_t n, uint32_t co, uint32_t ho, uint32_t wo) const
    {
        return (((uint64_t)n * cout_ + co) * ho_ + ho) * wo_ + wo;
    }

    __aicore__ inline void ComputeRow(uint32_t n, uint32_t co, uint32_t hoRow)
    {
        // Allocate accum buffer yAcc for this row (length wo_)
        AscendC::LocalTensor<float> yAcc = vecQ_.AllocTensor<float>();
        bool initialized = false;

        // Accumulate convolution: yAcc = sum over (ci, ky, kx) of x * w
        for (uint32_t ci = 0; ci < cin_; ++ci) {
            for (uint32_t ky = 0; ky < kh_; ++ky) {
                // For each kernel col
                for (uint32_t kx = 0; kx < kw_; ++kx) {
                    // Load input row segment of length wo_: x[n, ci, hoRow+ky, kx : kx+wo_-1]
                    AscendC::LocalTensor<float> xRow = vecQ_.AllocTensor<float>();
                    const uint64_t xoff = XIndex(n, ci, hoRow + ky, kx);
                    AscendC::DataCopy(xRow, xGm_[xoff], wo_);

                    // Load weight scalar w[co,ci,ky,kx] -> use as scalar in muls
                    // Copy one element to local, then read as scalar value
                    AscendC::LocalTensor<float> wLocal = vecQ_.AllocTensor<float>();
                    AscendC::DataCopy(wLocal, wGm_[WIndex(co, ci, ky, kx)], 1);
                    float wVal = wLocal.GetValue(0);
                    vecQ_.FreeTensor(wLocal);

                    if (!initialized) {
                        // yAcc = xRow * wVal
                        AscendC::Muls(yAcc, xRow, wVal, wo_);
                        initialized = true;
                    } else {
                        AscendC::LocalTensor<float> tmp = vecQ_.AllocTensor<float>();
                        AscendC::Muls(tmp, xRow, wVal, wo_);
                        AscendC::Add(yAcc, yAcc, tmp, wo_);
                        vecQ_.FreeTensor(tmp);
                    }
                    vecQ_.FreeTensor(xRow);
                }
            }
        }

        // Bias add if present: yAcc += bias[co]
        if (hasBias_) {
            AscendC::LocalTensor<float> bLocal = vecQ_.AllocTensor<float>();
            AscendC::DataCopy(bLocal, bGm_[co], 1);
            float bVal = bLocal.GetValue(0);
            vecQ_.FreeTensor(bLocal);
            AscendC::Adds(yAcc, yAcc, bVal, wo_);
        }

        // Fused HardSwish then ReLU:
        // t = yAcc + 3
        AscendC::LocalTensor<float> t = vecQ_.AllocTensor<float>();
        AscendC::Adds(t, yAcc, 3.0f, wo_);
        // t = min(t, 6)
        AscendC::Mins(t, t, 6.0f, wo_);
        // t = max(t, 0)
        AscendC::Maxs(t, t, 0.0f, wo_);
        // out = yAcc * t * (1/6)
        AscendC::LocalTensor<float> outRow = vecQ_.AllocTensor<float>();
        AscendC::Mul(outRow, yAcc, t, wo_);
        AscendC::Muls(outRow, outRow, (1.0f / 6.0f), wo_);
        // ReLU: out = max(out, 0)
        AscendC::Maxs(outRow, outRow, 0.0f, wo_);

        // Store to GM: y[n, co, hoRow, 0:wo_-1]
        const uint64_t yoff = YIndex(n, co, hoRow, 0);
        AscendC::DataCopy(yGm_[yoff], outRow, wo_);

        vecQ_.FreeTensor(outRow);
        vecQ_.FreeTensor(t);
        vecQ_.FreeTensor(yAcc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, VEC_BUFS> vecQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t h_{0}, wIn_{0}, kh_{0}, kw_{0};
    uint32_t ho_{0}, wo_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_hard_swish_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dHardSwishReluCustom kernel;
    const bool hasBias = (tiling_data.hasBias != 0);
    kernel.Init(x, weight, bias, y,
                tiling_data.n, tiling_data.cin, tiling_data.cout,
                tiling_data.h, tiling_data.w,
                tiling_data.kh, tiling_data.kw,
                tiling_data.ho, tiling_data.wo,
                hasBias);
    kernel.Process();
}
