#include "conv_transposed_1d_dilated_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB budget
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed1dDilatedCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    if (xShape->GetDimNum() != 3 || wShape->GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(xShape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(xShape->GetDim(1));
    const uint32_t Lin = static_cast<uint32_t>(xShape->GetDim(2));

    const uint32_t W_Cin = static_cast<uint32_t>(wShape->GetDim(0));
    const uint32_t Cout  = static_cast<uint32_t>(wShape->GetDim(1));
    const uint32_t K     = static_cast<uint32_t>(wShape->GetDim(2));

    if (W_Cin != Cin) {
        return ge::GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *strideAttr   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *paddingAttr  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilationAttr = attrs->GetAttrPointer<int32_t>(2);

    const uint32_t stride   = static_cast<uint32_t>(*strideAttr);
    const uint32_t padding  = static_cast<uint32_t>(*paddingAttr);
    const uint32_t dilation = static_cast<uint32_t>(*dilationAttr);

    // Lout = (Lin - 1) * stride - 2*padding + dilation*(K - 1) + 1
    const int64_t tmpLout = static_cast<int64_t>(Lin - 1) * static_cast<int64_t>(stride)
                          - static_cast<int64_t>(2 * padding)
                          + static_cast<int64_t>(dilation) * static_cast<int64_t>(K - 1)
                          + 1;
    const uint32_t Lout = tmpLout > 0 ? static_cast<uint32_t>(tmpLout) : 0u;

    // UB budget for yTile + xTile (worst-case s=1): 2 * tileLen * 4 bytes
    uint32_t maxTile = (UB_BUDGET_BYTES / 8);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLenOut = maxTile;
    if (Lout < tileLenOut) {
        const uint32_t aligned = AlignDown(Lout, ALIGN_ELEMS);
        tileLenOut = (aligned > 0) ? aligned : Lout;
    }
    if (tileLenOut == 0) tileLenOut = 1;

    const uint32_t groups = N * Cout;
    uint32_t blockDim = (groups == 0) ? 1u : groups;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_lin(Lin);
    tiling.set_kernel(K);
    tiling.set_lout(Lout);
    tiling.set_tileLenOut(tileLenOut);
    tiling.set_blockDim(blockDim);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_dilation(dilation);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    if (xShape->GetDimNum() != 3 || wShape->GetDimNum() != 3) {
        return GRAPH_FAILED;
    }

    const int64_t N   = xShape->GetDim(0);
    const int64_t Cin = xShape->GetDim(1);
    const int64_t Lin = xShape->GetDim(2);

    const int64_t W_Cin = wShape->GetDim(0);
    const int64_t Cout  = wShape->GetDim(1);
    const int64_t K     = wShape->GetDim(2);

    if (W_Cin != Cin) {
        return GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t stride   = *attrs->GetAttrPointer<int32_t>(0);
    const int32_t padding  = *attrs->GetAttrPointer<int32_t>(1);
    const int32_t dilation = *attrs->GetAttrPointer<int32_t>(2);

    const int64_t Lout = (Lin - 1) * stride - 2 * padding + dilation * (K - 1) + 1;
    if (Lout <= 0) {
        return GRAPH_FAILED;
    }

    gert::Shape *yShape = context->GetOutputShape(0);
    yShape->SetDimNum(3);
    yShape->SetDim(0, N);
    yShape->SetDim(1, Cout);
    yShape->SetDim(2, Lout);

    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelConvTransposed1dDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed1dDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t kernel,
                                uint32_t lout, uint32_t tileLenOut,
                                uint32_t stride, uint32_t padding, uint32_t dilation)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        lin_ = lin; kernel_ = kernel; lout_ = lout;
        tileLenOut_ = tileLenOut;
        stride_ = stride; padding_ = padding; dilation_ = dilation;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * cin_ * lin_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)cin_ * cout_ * kernel_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * cout_ * lout_);

        pipe_.InitBuffer(inQueueX_, 1, tileLenOut_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, 1, kernel_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileLenOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || lin_ == 0 || kernel_ == 0 || lout_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t groups = n_ * cout_;

        for (uint32_t gid = blockIdx; gid < groups; gid += blockNum) {
            const uint32_t nIdx = gid / cout_;
            const uint32_t cOut = gid % cout_;

            for (uint32_t outBase = 0; outBase < lout_; outBase += tileLenOut_) {
                uint32_t curLen = lout_ - outBase;
                if (curLen > tileLenOut_) curLen = tileLenOut_;

                // Allocate and zero y tile
                AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();
                AscendC::Muls(yTile, yTile, 0.0f, curLen);

                for (uint32_t cIn = 0; cIn < cin_; ++cIn) {
                    // Load weights for this (cIn, cOut)
                    const uint64_t wBase = ((uint64_t)cIn * cout_ + cOut) * kernel_;
                    AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
                    AscendC::DataCopy(wLocal, wGm_[wBase], kernel_);

                    if (stride_ == 1u) {
                        // Optimized path for stride=1: process contiguous x segments
                        for (uint32_t k = 0; k < kernel_; ++k) {
                            const int64_t t_in_start = static_cast<int64_t>(outBase) + static_cast<int64_t>(padding_) - static_cast<int64_t>(k) * static_cast<int64_t>(dilation_);
                            int64_t valid_begin_in = t_in_start;
                            int64_t valid_end_in = t_in_start + static_cast<int64_t>(curLen);
                            if (valid_begin_in < 0) valid_begin_in = 0;
                            if (valid_end_in > static_cast<int64_t>(lin_)) valid_end_in = static_cast<int64_t>(lin_);

                            const int64_t valid_len64 = valid_end_in - valid_begin_in;
                            if (valid_len64 <= 0) {
                                continue;
                            }
                            const uint32_t valid_len = static_cast<uint32_t>(valid_len64);
                            const int64_t y_offset64 = valid_begin_in - t_in_start;
                            const uint32_t y_offset = static_cast<uint32_t>(y_offset64);

                            const uint64_t xBase = ((uint64_t)nIdx * cin_ + cIn) * lin_ + static_cast<uint64_t>(valid_begin_in);
                            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[xBase], valid_len);

                            // scale and accumulate
                            const float wScalar = wLocal.GetValue(k);
                            AscendC::Muls(xLocal, xLocal, wScalar, valid_len);
                            AscendC::LocalTensor<float> ySeg = yTile[y_offset];
                            AscendC::Add(ySeg, ySeg, xLocal, valid_len);

                            inQueueX_.FreeTensor(xLocal);
                        }
                    } else {
                        // Generic path for stride != 1: process element-wise
                        for (uint32_t k = 0; k < kernel_; ++k) {
                            const float wScalar = wLocal.GetValue(k);
                            for (uint32_t i = 0; i < curLen; ++i) {
                                const int64_t t_out = static_cast<int64_t>(outBase) + static_cast<int64_t>(i);
                                const int64_t numerator = t_out + static_cast<int64_t>(padding_) - static_cast<int64_t>(k) * static_cast<int64_t>(dilation_);
                                if (numerator % static_cast<int64_t>(stride_) != 0) {
                                    continue;
                                }
                                const int64_t t_in = numerator / static_cast<int64_t>(stride_);
                                if (t_in < 0 || t_in >= static_cast<int64_t>(lin_)) {
                                    continue;
                                }
                                const uint64_t xIdx = ((uint64_t)nIdx * cin_ + cIn) * lin_ + static_cast<uint64_t>(t_in);
                                AscendC::LocalTensor<float> xLocal1 = inQueueX_.AllocTensor<float>();
                                AscendC::DataCopy(xLocal1, xGm_[xIdx], 1);
                                // yTile[i] += xLocal1[0] * wScalar
                                AscendC::Muls(xLocal1, xLocal1, wScalar, 1);
                                AscendC::LocalTensor<float> yElem = yTile[i];
                                AscendC::Add(yElem, yElem, xLocal1, 1);
                                inQueueX_.FreeTensor(xLocal1);
                            }
                        }
                    }

                    inQueueW_.FreeTensor(wLocal);
                }

                const uint64_t yBase = ((uint64_t)nIdx * cout_ + cOut) * lout_ + outBase;
                AscendC::DataCopy(yGm_[yBase], yTile, curLen);
                outQueueY_.FreeTensor(yTile);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t lin_{0}, kernel_{0}, lout_{0};
    uint32_t tileLenOut_{0};
    uint32_t stride_{1}, padding_{0}, dilation_{1};
};

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1dDilatedCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.kernel,
            tiling_data.lout, tiling_data.tileLenOut,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation);
    op.Process();
}
