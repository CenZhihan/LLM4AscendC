kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t X_LINE_LEN = 64;   // input W
constexpr int32_t OUT_LINE0_LEN = 64; // segments [0..63], [1..64], [2..65] all length 64

class KernelConvTransposed3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t din, uint32_t hin, uint32_t win,
                                uint32_t kd, uint32_t kh, uint32_t kw)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        din_ = din; hin_ = hin; win_ = win;
        kd_ = kd; kh_ = kh; kw_ = kw;
        dout_ = din_ + kd_ - 1;
        hout_ = hin_ + kh_ - 1;
        wout_ = win_ + kw_ - 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_*cin_*din_*hin_*win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)cin_*cout_*kd_*kh_*kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_*cout_*dout_*hout_*wout_);

        // Initialize UB buffers
        pipe_.InitBuffer(calcQ_, 6, X_LINE_LEN * sizeof(float));  // x_line, tmp, zeros and reuse
        pipe_.InitBuffer(wQ_, 1, cout_ * sizeof(float));          // weight vector across cout
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalTasks = static_cast<uint64_t>(n_) * dout_ * hout_;
        if (totalTasks == 0) return;

        for (uint64_t task = blockIdx; task < totalTasks; task += blockNum) {
            // Decode (n, zo, yo)
            uint64_t tmp = task;
            uint32_t nIdx = static_cast<uint32_t>(tmp / (dout_ * hout_));
            tmp = tmp % (static_cast<uint64_t>(dout_) * hout_);
            uint32_t zo = static_cast<uint32_t>(tmp / hout_);
            uint32_t yo = static_cast<uint32_t>(tmp % hout_);

            // Precompute base line index factor for output
            // y index: ((((n*cout + co)*dout + zo)*hout + yo)*wout + xo)
            const uint64_t perNStrideY   = static_cast<uint64_t>(cout_) * dout_ * hout_ * wout_;
            const uint64_t perCoStrideY  = static_cast<uint64_t>(dout_) * hout_ * wout_;
            const uint64_t perZoStrideY  = static_cast<uint64_t>(hout_) * wout_;
            const uint64_t baseLineBase0 = static_cast<uint64_t>(nIdx) * perNStrideY
                                         + static_cast<uint64_t>(zo) * perZoStrideY
                                         + static_cast<uint64_t>(yo) * wout_; // add co offset later

            // Prepare zeros vector for clearing segments
            LocalTensor<float> zeros = calcQ_.AllocTensor<float>();
            AscendC::Duplicate(zeros, 0.0f, OUT_LINE0_LEN);

            // Clear three segments of output lines for each co (overwrite mode)
            AscendC::SetAtomicNone();
            for (uint32_t co = 0; co < cout_; ++co) {
                const uint64_t coBase = baseLineBase0 + static_cast<uint64_t>(co) * perCoStrideY;
                // Segment starts at xo = 0, 1, 2; Length 64
                AscendC::DataCopy(yGm_[coBase + 0], zeros, OUT_LINE0_LEN);
                AscendC::DataCopy(yGm_[coBase + 1], zeros, OUT_LINE0_LEN);
                AscendC::DataCopy(yGm_[coBase + 2], zeros, OUT_LINE0_LEN);
            }
            AscendC::PipeBarrier();
            calcQ_.FreeTensor(zeros);

            // Accumulation with atomic add
            AscendC::SetAtomicAdd();

            // Precompute input strides
            const uint64_t perNStrideX  = static_cast<uint64_t>(cin_) * din_ * hin_ * win_;
            const uint64_t perCiStrideX = static_cast<uint64_t>(din_) * hin_ * win_;
            const uint64_t perZiStrideX = static_cast<uint64_t>(hin_) * win_;
            const uint64_t perHiStrideX = static_cast<uint64_t>(win_);

            // Precompute weight strides: [cin, cout, kd, kh, kw]
            const uint64_t perCinStrideW = static_cast<uint64_t>(cout_) * kd_ * kh_ * kw_;
            const uint64_t perCoStrideW  = static_cast<uint64_t>(kd_) * kh_ * kw_;
            const uint64_t perKdStrideW  = static_cast<uint64_t>(kh_) * kw_;
            const uint64_t perKhStrideW  = static_cast<uint64_t>(kw_);

            for (uint32_t kd = 0; kd < kd_; ++kd) {
                // input z index
                int32_t zi = static_cast<int32_t>(zo) - static_cast<int32_t>(kd);
                if (zi < 0 || zi >= static_cast<int32_t>(din_)) {
                    continue;
                }
                for (uint32_t kh = 0; kh < kh_; ++kh) {
                    int32_t hi = static_cast<int32_t>(yo) - static_cast<int32_t>(kh);
                    if (hi < 0 || hi >= static_cast<int32_t>(hin_)) {
                        continue;
                    }

                    for (uint32_t ci = 0; ci < cin_; ++ci) {
                        // Load one input line x[nIdx, ci, zi, hi, 0..63]
                        const uint64_t xBase = static_cast<uint64_t>(nIdx) * perNStrideX
                                             + static_cast<uint64_t>(ci) * perCiStrideX
                                             + static_cast<uint64_t>(zi) * perZiStrideX
                                             + static_cast<uint64_t>(hi) * perHiStrideX;

                        LocalTensor<float> xLine = calcQ_.AllocTensor<float>();
                        AscendC::DataCopy(xLine, xGm_[xBase], X_LINE_LEN);

                        // For each kw, we will load weights across cout and then accumulate for all co
                        for (uint32_t kw = 0; kw < kw_; ++kw) {
                            const uint64_t wBase = static_cast<uint64_t>(ci) * perCinStrideW
                                                  + static_cast<uint64_t>(kd) * perKdStrideW
                                                  + static_cast<uint64_t>(kh) * perKhStrideW
                                                  + static_cast<uint64_t>(kw);
                            // Load weight vector over cout (stride is perCoStrideW)
                            // Memory layout contiguous along cout dimension for fixed cin,kd,kh,kw
                            LocalTensor<float> wVec = wQ_.AllocTensor<float>();
                            AscendC::DataCopy(wVec, wGm_[wBase], cout_);

                            // For each co, compute tmp = xLine * wScalar and atomic-add to y line segment
                            for (uint32_t co = 0; co < cout_; ++co) {
                                // Read weight scalar from local buffer
                                // Note: Accessing a single element from LocalTensor to scalar variable
                                // is used here to feed Duplicate/Muls.
                                float wScalar = wVec.GetValue(co);

                                LocalTensor<float> tmp = calcQ_.AllocTensor<float>();
                                // tmp = xLine * wScalar
                                AscendC::Muls(tmp, xLine, wScalar, X_LINE_LEN);

                                // Accumulate to output: line segment starts at xo = kw (0,1,2), length 64
                                const uint64_t coBase = baseLineBase0 + static_cast<uint64_t>(co) * perCoStrideY;
                                AscendC::DataCopy(yGm_[coBase + kw], tmp, OUT_LINE0_LEN);

                                calcQ_.FreeTensor(tmp);
                            }

                            wQ_.FreeTensor(wVec);
                        }

                        calcQ_.FreeTensor(xLine);
                    } // ci
                } // kh
            } // kd

            AscendC::SetAtomicNone();
            AscendC::PipeBarrier();
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 6> calcQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t din_{0}, hin_{0}, win_{0};
    uint32_t kd_{0}, kh_{0}, kw_{0};
    uint32_t dout_{0}, hout_{0}, wout_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvTransposed3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.din, tiling_data.hin, tiling_data.win,
            tiling_data.kd, tiling_data.kh, tiling_data.kw);
    op.Process();
}
