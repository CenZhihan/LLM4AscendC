kernel_src="""

#include "kernel_operator.h"

// We implement a simple per-batch-core RNN cell unrolled over T time steps.
// Each block handles one batch item (or more with grid-stride if B < blockDim).

constexpr int32_t INBUF_NUM = 1;
constexpr int32_t OUTBUF_NUM = 1;
constexpr int32_t WORKBUF_NUM = 1;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0,
                                GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o,
                                GM_ADDR y,
                                uint32_t T, uint32_t B, uint32_t I,
                                uint32_t H, uint32_t O, uint32_t K)
    {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O; K_ = K;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)T_ * B_ * I_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, (uint64_t)B_ * H_);
        wI2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, (uint64_t)H_ * K_);
        bI2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);
        wH2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, (uint64_t)O_ * H_);
        bH2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)T_ * B_ * O_);

        // Allocate UB queues for working tensors
        // inpLocal: size K
        pipe_.InitBuffer(inpQ_, INBUF_NUM, K_ * sizeof(float));
        // hLocal: size H
        pipe_.InitBuffer(hQ_, INBUF_NUM, H_ * sizeof(float));
        // yLocal: size O
        pipe_.InitBuffer(yQ_, OUTBUF_NUM, O_ * sizeof(float));
        // mulTmp buffer: max(K, H) for tiled dot partials and misc tmp (shared)
        uint32_t maxVec = K_ > H_ ? K_ : H_;
        if (maxVec < O_) maxVec = O_;
        pipe_.InitBuffer(workQ_, WORKBUF_NUM, maxVec * sizeof(float));
        // small tmp for tanh/calculations
        pipe_.InitBuffer(tmpScalarQ_, WORKBUF_NUM, 8 * sizeof(float)); // a few scalars
        pipe_.InitBuffer(tmpTanhQ_, WORKBUF_NUM, 256); // shared tmp for Tanh when needed
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride across batch dimension
        for (uint32_t b = blockIdx; b < B_; b += blockNum) {
            RunBatch(b);
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t t, uint32_t b) const
    {
        // [T, B, I]
        return ((uint64_t)t * B_ + b) * I_;
    }
    __aicore__ inline uint64_t OffsetY(uint32_t t, uint32_t b) const
    {
        // [T, B, O]
        return ((uint64_t)t * B_ + b) * O_;
    }
    __aicore__ inline uint64_t OffsetW_i2h_row(uint32_t hRow) const
    {
        // [H, K] row-major
        return (uint64_t)hRow * K_;
    }
    __aicore__ inline uint64_t OffsetW_h2o_row(uint32_t oRow) const
    {
        // [O, H] row-major
        return (uint64_t)oRow * H_;
    }

    // Compute dot product s = dot(a[0:len], b[0:len])
    __aicore__ inline void Dot(const AscendC::LocalTensor<float>& a,
                               const AscendC::GlobalTensor<float>& bGm, uint64_t bOff,
                               uint32_t len,
                               AscendC::LocalTensor<float>& scalarOut)
    {
        // We'll accumulate in scalarOut[0]
        // Initialize to zero
        AscendC::LocalTensor<float> sumLocal = scalarOut;
        AscendC::Dup(sumLocal, 0.0f, 1);

        // Tile length for elementwise mul before reduction
        // Use chunks that fit in workQ_
        const uint32_t CHUNK = 2048u;
        AscendC::LocalTensor<float> mulBuf = workQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpSum = tmpScalarQ_.AllocTensor<float>(); // single-element temp

        uint32_t done = 0;
        while (done < len) {
            uint32_t cur = len - done;
            if (cur > CHUNK) cur = CHUNK;

            // Load b tile to mulBuf
            // For simplicity, reuse mulBuf as GM to Local copy buffer for w tile
            AscendC::DataCopy(mulBuf, bGm[bOff + done], cur);
            // Elementwise multiply a[done:done+cur] * mulBuf
            // Reuse mulBuf for the result
            AscendC::Mul(mulBuf, a[done], mulBuf, cur);
            // Reduce sum of mulBuf into tmpSum[0]
            AscendC::ReduceSum(tmpSum, mulBuf, cur);
            // Accumulate into sumLocal
            AscendC::Add(sumLocal, sumLocal, tmpSum, 1);

            done += cur;
        }

        tmpScalarQ_.FreeTensor(tmpSum);
        workQ_.FreeTensor(mulBuf);
    }

    __aicore__ inline void RunBatch(uint32_t b)
    {
        // Local working tensors
        AscendC::LocalTensor<float> hLocal = hQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> inpLocal = inpQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = yQ_.AllocTensor<float>();

        // Initialize h with h0[b]
        AscendC::DataCopy(hLocal, h0Gm_[((uint64_t)b) * H_], H_);

        // Prepare reusable 1-element buffers
        AscendC::LocalTensor<float> scalar0 = tmpScalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> tanhTmp = tmpTanhQ_.AllocTensor<uint8_t>();

        // Main time loop
        for (uint32_t t = 0; t < T_; ++t) {
            // Load x[t, b, :]
            AscendC::LocalTensor<float> xLocal = workQ_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[OffsetX(t, b)], I_);

            // Build inp = [x, h]
            AscendC::DataCopy(inpLocal, xLocal, I_);
            AscendC::DataCopy(inpLocal[I_], hLocal, H_);

            // Compute pre-activation -> hLocal = tanh(inp @ w_i2h^T + b)
            for (uint32_t j = 0; j < H_; ++j) {
                // s = dot(inpLocal, w_i2h[j])
                AscendC::LocalTensor<float> s = scalar0; // alias 1-element buffer
                Dot(inpLocal, wI2hGm_, OffsetW_i2h_row(j), K_, s);
                // add bias
                AscendC::Add(s, s, bI2hGm_[j], 1);
                // tanh
                AscendC::LocalTensor<float> sAct = tmpScalarQ_.AllocTensor<float>();
                AscendC::Tanh(sAct, s, tanhTmp, 1);
                // write into hLocal[j]
                AscendC::DataCopy(hLocal[j], sAct, 1);
                tmpScalarQ_.FreeTensor(sAct);
            }

            // Compute y_t = hLocal @ w_h2o^T + b_h2o
            for (uint32_t o = 0; o < O_; ++o) {
                AscendC::LocalTensor<float> s = scalar0;
                Dot(hLocal, wH2oGm_, OffsetW_h2o_row(o), H_, s);
                AscendC::Add(s, s, bH2oGm_[o], 1);
                AscendC::DataCopy(yLocal[o], s, 1);
            }

            // Store y[t, b, :]
            AscendC::DataCopy(yGm_[OffsetY(t, b)], yLocal, O_);

            // Free temporary x
            workQ_.FreeTensor(xLocal);
        }

        // Free locals
        tmpTanhQ_.FreeTensor(tanhTmp);
        tmpScalarQ_.FreeTensor(scalar0);
        yQ_.FreeTensor(yLocal);
        inpQ_.FreeTensor(inpLocal);
        hQ_.FreeTensor(hLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, INBUF_NUM> inpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, INBUF_NUM> hQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, OUTBUF_NUM> yQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, WORKBUF_NUM> workQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, WORKBUF_NUM> tmpScalarQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, WORKBUF_NUM> tmpTanhQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> wI2hGm_;
    AscendC::GlobalTensor<float> bI2hGm_;
    AscendC::GlobalTensor<float> wH2oGm_;
    AscendC::GlobalTensor<float> bH2oGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t T_{0}, B_{0}, I_{0}, H_{0}, O_{0}, K_{0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0,
                                                                GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                GM_ADDR w_h2o, GM_ADDR b_h2o,
                                                                GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I,
            tiling_data.H, tiling_data.O, tiling_data.K);
    op.Process();
}
