static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Shapes:
    // x: [M, K]
    // w: [N, K] (PyTorch linear uses w.t() inside)
    // b: [N]
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    const gert::Shape *bShape = context->GetInputShape(2);

    if (xShape->GetDimNum() != 2 || wShape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(xShape->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(xShape->GetDim(1));
    const uint32_t N = static_cast<uint32_t>(wShape->GetDim(0));

    // Basic sanity checks.
    if (static_cast<uint32_t>(wShape->GetDim(1)) != K) {
        return ge::GRAPH_FAILED;
    }
    if (bShape->GetDimNum() != 1 || static_cast<uint32_t>(bShape->GetDim(0)) != N) {
        return ge::GRAPH_FAILED;
    }

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);

    // Configure A/B/C types and positions
    tiling->SetAType(matmul_tiling::TPosition::GM,
                     matmul_tiling::CubeFormat::ND,
                     matmul_tiling::DataType::DT_FLOAT);
    tiling->SetBType(matmul_tiling::TPosition::GM,
                     matmul_tiling::CubeFormat::ND,
                     matmul_tiling::DataType::DT_FLOAT);
    tiling->SetCType(matmul_tiling::TPosition::GM,
                     matmul_tiling::CubeFormat::ND,
                     matmul_tiling::DataType::DT_FLOAT);
    tiling->SetBiasType(matmul_tiling::TPosition::GM,
                        matmul_tiling::CubeFormat::ND,
                        matmul_tiling::DataType::DT_FLOAT);

    tiling->SetShape(M, N, K);
    tiling->SetOrgShape(M, N, K);
    tiling->SetBias(true);
    // default buffer space
    tiling->SetBufferSpace(-1, -1, -1);
    // Do NOT enable SplitK because bias is used (per API constraints)
    tiling->EnableMultiCoreSplitK(false);

    // Suggest core usage equal to number of output tiles in N dimension or platform AICore number
    // Here we pick a conservative blockDim heuristic based on M*N size
    uint64_t totalOut = static_cast<uint64_t>(M) * static_cast<uint64_t>(N);
    uint32_t blockDim = 24;
    if (totalOut < 4096) blockDim = 8;
    if (totalOut < 65536) blockDim = 16;
    tiling->SetDim(blockDim);

    optiling::TCubeTiling tilingData;
    int ret = tiling->GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Final block dim for kernel launch
    context->SetBlockDim(blockDim);

    // Save tiling buffer
    tilingData.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tilingData.GetDataSize());

    // workspace requirement: none for this fused kernel
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // x: [M,K], w:[N,K] => y:[M,N]
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    if (xShape->GetDimNum() != 2 || wShape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = xShape->GetDim(0);
    const int64_t Kx = xShape->GetDim(1);
    const int64_t N = wShape->GetDim(0);
    const int64_t Kw = wShape->GetDim(1);
    if (Kx != Kw) {
        return GRAPH_FAILED;
    }
    gert::Shape *yShape = context->GetOutputShape(0);
    yShape->SetDimNum(2);
    yShape->SetDim(0, M);
    yShape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t UB_BUFFER_NUM = 2;
constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
constexpr uint32_t ALIGN_ELEMS = 2048;
constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline __aicore__ uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

class KernelMatmulSubtractMultiplyReluCustom {
public:
    __aicore__ inline KernelMatmulSubtractMultiplyReluCustom() {}

    // Matmul will be computed using cube tiling; then apply (y - sub) * mul -> relu as epilogue.
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR sub, GM_ADDR mul, GM_ADDR y,
                                const optiling::TCubeTiling *tiling)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, 0);
        wGm_.SetGlobalBuffer((__gm__ float*)w, 0);
        bGm_.SetGlobalBuffer((__gm__ float*)b, 0);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 0);

        subGm_ = (__gm__ float*)sub;
        mulGm_ = (__gm__ float*)mul;

        tiling_ = *tiling;

        // Infer output size from tiling (M, N)
        M_ = tiling_.GetM();
        N_ = tiling_.GetN();

        totalOut_ = static_cast<uint32_t>(M_) * static_cast<uint32_t>(N_);

        // UB tiling for epilogue over y
        uint32_t maxTileElems = UB_BUDGET_BYTES / sizeof(float);
        maxTileElems = AlignDown(maxTileElems, ALIGN_ELEMS);
        if (maxTileElems < ALIGN_ELEMS) maxTileElems = ALIGN_ELEMS;
        if (maxTileElems > HARD_CAP_ELEMS) maxTileElems = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
        tileLen_ = maxTileElems;
        if (totalOut_ < tileLen_) {
            const uint32_t aligned = AlignDown(totalOut_, ALIGN_ELEMS);
            tileLen_ = (aligned > 0) ? aligned : totalOut_;
        }

        pipe_.InitBuffer(epiInQ_, UB_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(epiOutQ_, UB_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // 1) Compute Matmul + Bias into y using cube tiling
        ComputeMatmul();

        // 2) Load scalars
        float subVal = 0.0f;
        float mulVal = 1.0f;
        // Load scalar values from GM
        subVal = subGm_[0];
        mulVal = mulGm_[0];

        // 3) Epilogue: y = relu((y - subVal) * mulVal)
        if (totalOut_ == 0 || tileLen_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOut_) return;

        uint32_t curLen = totalOut_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalOut_) {
                nextLen = totalOut_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Epilogue(curLen, subVal, mulVal);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void ComputeMatmul()
    {
        // Using built-in Matmul cube compute guided by tiling.
        // Inputs:
        //   A = x [M,K] from GM
        //   B = w [N,K] from GM, matmul expects W^T internally via tiling
        //   C = y [M,N] in GM
        // Bias:
        //   b [N] from GM
        // The MultiCoreMatmulTiling on host side has SetBias(true), so the compute will add bias.
        MatmulAPI<float, float, float> mm;
        mm.Init(&tiling_, xGm_, wGm_, yGm_, bGm_);
        mm.Compute();
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        auto yLocal = epiInQ_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        epiInQ_.EnQue(yLocal);
    }

    __aicore__ inline void Epilogue(uint32_t len, float subVal, float mulVal)
    {
        auto inLocal = epiInQ_.DeQue<float>();
        auto outLocal = epiOutQ_.AllocTensor<float>();
        auto tmpLocal = tmpQ_.AllocTensor<float>();

        AscendC::DataCopy(outLocal, inLocal, len);
        // outLocal = outLocal - subVal
        AscendC::Adds(outLocal, outLocal, -subVal, len);
        // outLocal = outLocal * mulVal
        AscendC::Muls(outLocal, outLocal, mulVal, len);
        // outLocal = relu(outLocal) = max(outLocal, 0)
        AscendC::Maxs(outLocal, outLocal, 0.0f, len);

        tmpQ_.FreeTensor(tmpLocal);
        epiOutQ_.EnQue(outLocal);
        epiInQ_.FreeTensor(inLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        auto outLocal = epiOutQ_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], outLocal, len);
        epiOutQ_.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, UB_BUFFER_NUM> epiInQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, UB_BUFFER_NUM> epiOutQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    __gm__ float* subGm_ {nullptr};
    __gm__ float* mulGm_ {nullptr};

    optiling::TCubeTiling tiling_;
    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t totalOut_ {0};
    uint32_t tileLen_ {0};
};

// Kernel entry
extern "C" __global__ __aicore__ void matmul_subtract_multiply_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR sub, GM_ADDR mul, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSubtractMultiplyReluCustom op;
    op.Init(x, w, b, sub, mul, y, &tiling_data);
    op.Process();
}
