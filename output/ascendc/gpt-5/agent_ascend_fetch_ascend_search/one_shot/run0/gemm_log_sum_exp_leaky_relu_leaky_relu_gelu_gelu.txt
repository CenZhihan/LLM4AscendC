namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// We stream one row across column tiles, keeping small UB footprint.
// UB usage per iteration ~ 2 * tileCols * sizeof(float).
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // per-core conservative UB budget
static constexpr uint32_t ALIGN_ELEMS = 2048;             // favor large aligned DMA
static constexpr uint32_t HARD_CAP_ELEMS = 32768;         // avoid too large tiles

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmLogSumExpLeakyReluLeakyReluGeluGeluCustomTilingData tiling;

    // x shape: [rows, cols]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const uint32_t dims = static_cast<uint32_t>(x_shape->GetOriginDimNum());
    if (dims < 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t rows = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t cols = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float negativeSlope = 0.01f;
    if (attrs != nullptr && attrs->GetNum() > 0) {
        const float *ns = attrs->GetAttrPointer<float>(0);
        if (ns != nullptr) negativeSlope = *ns;
    }

    // Compute tile over columns by UB budget: bytes = 2 * tileCols * 4
    uint32_t maxTile = UB_BUDGET_BYTES / (2u * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileCols = maxTile;
    if (cols < tileCols) {
        uint32_t aligned = AlignDown(cols, ALIGN_ELEMS);
        tileCols = (aligned > 0) ? aligned : cols;
    }

    // Block dim based on rows
    uint32_t blockDim = rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);
    tiling.set_negativeSlope(negativeSlope);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t IN_BUFFER_NUM = 2;

class KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom {
public:
    __aicore__ inline KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR /*w*/, GM_ADDR /*b*/, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t tileCols,
                                float negativeSlope)
    {
        rows_ = rows;
        cols_ = cols;
        tileCols_ = tileCols;
        negativeSlope_ = negativeSlope;
        negCorr_ = negativeSlope_ - 1.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(rows_) * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(rows_));

        pipe_.InitBuffer(inQueueX_, IN_BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileCols_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0 || tileCols_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            // Pass 1: reduce max along columns for numerical stability of logsumexp
            float rowMax = -3.4028235e+38f; // -FLT_MAX
            for (uint32_t c = 0; c < cols_; c += tileCols_) {
                uint32_t len = cols_ - c;
                if (len > tileCols_) len = tileCols_;
                CopyInRowTile(row, c, len);

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> redLocal = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceMax(redLocal, xLocal, len);
                // Assume reduction result is stored at index 0
                float tileMax = redLocal.GetValue(0);
                if (tileMax > rowMax) rowMax = tileMax;

                tmpQueue_.FreeTensor(redLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            // Pass 2: compute sum(exp(x - rowMax))
            float sumExp = 0.0f;
            for (uint32_t c = 0; c < cols_; c += tileCols_) {
                uint32_t len = cols_ - c;
                if (len > tileCols_) len = tileCols_;
                CopyInRowTile(row, c, len);

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

                // t = x - rowMax
                AscendC::Adds(tLocal, xLocal, -rowMax, len);
                // t = exp(t)
                AscendC::Exp(tLocal, tLocal, len);

                AscendC::LocalTensor<float> redLocal = outQueueY_.AllocTensor<float>();
                AscendC::ReduceSum(redLocal, tLocal, len);
                float tileSum = redLocal.GetValue(0);
                sumExp += tileSum;

                outQueueY_.FreeTensor(redLocal);
                tmpQueue_.FreeTensor(tLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            // LogSumExp scalar for the row
            float lse = std::log(sumExp) + rowMax;

            // Two LeakyReLU (same slope)
            lse = leaky_relu_scalar(lse);
            lse = leaky_relu_scalar(lse);

            // Two GELU (approximation)
            lse = gelu_scalar(lse);
            lse = gelu_scalar(lse);

            // Store result to y[row]
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Dup(yLocal, lse, 1);
            AscendC::DataCopy(yGm_[row], yLocal, 1);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    __aicore__ inline float leaky_relu_scalar(float v) const
    {
        float minv = (v < 0.0f) ? v : 0.0f;
        return v + negCorr_ * minv;
    }

    __aicore__ inline float gelu_scalar(float x) const
    {
        // tanh approximation
        const float c0 = 0.5f;
        const float c1 = 0.7978845608028654f;  // sqrt(2/pi)
        const float c2 = 0.044715f;
        float x3 = x * x * x;
        float t = c1 * (x + c2 * x3);
        float th = tanhf(t);
        return c0 * x * (1.0f + th);
    }

    __aicore__ inline void CopyInRowTile(uint32_t row, uint32_t colStart, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        uint64_t gmOffset = static_cast<uint64_t>(row) * cols_ + colStart;
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_{0};
    uint32_t cols_{0};
    uint32_t tileCols_{0};
    float negativeSlope_{0.01f};
    float negCorr_{-0.99f};
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom op;
    op.Init(x, w, b, y, tiling_data.rows, tiling_data.cols, tiling_data.tileCols, tiling_data.negativeSlope);
    op.Process();
}
