kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMultiplyLeakyreluCustom {
public:
    __aicore__ inline KernelGemmMultiplyLeakyreluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileN, uint32_t tileK,
                                float multiplier, float negativeSlope)
    {
        M_ = M; K_ = K; N_ = N;
        tileN_ = tileN; tileK_ = tileK;
        multiplier_ = multiplier;
        negativeSlope_ = negativeSlope;

        xGm_.SetGlobalBuffer((__gm__ half*)x, M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ half*)w, K_ * N_);
        bGm_.SetGlobalBuffer((__gm__ half*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ half*)y, M_ * N_);

        // Buffers:
        // - wTile: tileK * tileN
        // - aSeg: tileK
        // - rowVec: tileN
        // - tmpRow: tileN
        pipe_.InitBuffer(inQueueWT_, 1, tileK_ * tileN_ * sizeof(half));
        pipe_.InitBuffer(inQueueASeg_, BUFFER_NUM, tileK_ * sizeof(half));
        pipe_.InitBuffer(outQueueRow_, BUFFER_NUM, tileN_ * sizeof(half));
        pipe_.InitBuffer(tmpRowQ_, 1, tileN_ * sizeof(half));
        pipe_.InitBuffer(biasVecQ_, 1, tileN_ * sizeof(half));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Split rows across blocks as evenly as possible.
        const uint32_t rowsPerBlock = (M_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowStart >= M_) return;
        if (rowEnd > M_) rowEnd = M_;

        for (uint32_t nBase = 0; nBase < N_; nBase += tileN_) {
            const uint32_t nCur = (nBase + tileN_ <= N_) ? tileN_ : (N_ - nBase);

            // Preload bias tile for this N-window
            AscendC::LocalTensor<half> biasTile = biasVecQ_.AllocTensor<half>();
            AscendC::DataCopy(biasTile, bGm_[nBase], nCur);
            if (nCur < tileN_) {
                // Zero the tail to avoid garbage when using vector ops
                AscendC::Dup(biasTile[nCur], (half)0.0f, tileN_ - nCur);
            }

            for (uint32_t r = rowStart; r < rowEnd; ++r) {
                // Initialize row accumulator
                AscendC::LocalTensor<half> rowVec = outQueueRow_.AllocTensor<half>();
                AscendC::Dup(rowVec, (half)0.0f, tileN_);

                for (uint32_t kBase = 0; kBase < K_; kBase += tileK_) {
                    const uint32_t kCur = (kBase + tileK_ <= K_) ? tileK_ : (K_ - kBase);

                    // Load W tile [kCur, nCur]
                    AscendC::LocalTensor<half> wTile = inQueueWT_.AllocTensor<half>();
                    // Copy each k-row of size nCur
                    for (uint32_t kk = 0; kk < kCur; ++kk) {
                        AscendC::DataCopy(wTile[kk * tileN_], wGm_[(kBase + kk) * N_ + nBase], nCur);
                        if (nCur < tileN_) {
                            AscendC::Dup(wTile[kk * tileN_ + nCur], (half)0.0f, tileN_ - nCur);
                        }
                    }
                    if (kCur < tileK_) {
                        // Zero remaining k-rows to keep vector ops safe
                        const uint32_t remain = tileK_ - kCur;
                        AscendC::Dup(wTile[kCur * tileN_], (half)0.0f, remain * tileN_);
                    }

                    // Load A segment for this row: [kCur]
                    AscendC::LocalTensor<half> aSeg = inQueueASeg_.AllocTensor<half>();
                    AscendC::DataCopy(aSeg, xGm_[r * K_ + kBase], kCur);
                    if (kCur < tileK_) {
                        AscendC::Dup(aSeg[kCur], (half)0.0f, tileK_ - kCur);
                    }

                    // Accumulate: rowVec += sum_{kk} aSeg[kk] * wTile[kk, :]
                    AscendC::LocalTensor<half> tmpRow = tmpRowQ_.AllocTensor<half>();
                    for (uint32_t kk = 0; kk < tileK_; ++kk) {
                        // Scalar aVal from aSeg[kk] approximated by broadcasting via Muls with scalar
                        // Local variable will be fetched with intrinsic during codegen
                        half aVal = aSeg.GetValue(kk);
                        AscendC::Muls(tmpRow, wTile[kk * tileN_], static_cast<float>(aVal), tileN_);
                        AscendC::Add(rowVec, rowVec, tmpRow, tileN_);
                    }
                    tmpRowQ_.FreeTensor(tmpRow);
                    inQueueASeg_.FreeTensor(aSeg);
                    inQueueWT_.FreeTensor(wTile);
                }

                // Add bias
                AscendC::Add(rowVec, rowVec, biasTile, tileN_);

                // Multiply by scalar multiplier
                AscendC::Muls(rowVec, rowVec, multiplier_, tileN_);

                // LeakyReLU
                // tmp = ns * min(rowVec, 0)
                AscendC::LocalTensor<half> tmpRow2 = tmpRowQ_.AllocTensor<half>();
                AscendC::Mins(tmpRow2, rowVec, 0.0f, tileN_);
                AscendC::Muls(tmpRow2, tmpRow2, negativeSlope_, tileN_);
                // rowVec = max(rowVec, 0) + tmpRow2
                AscendC::Maxs(rowVec, rowVec, 0.0f, tileN_);
                AscendC::Add(rowVec, rowVec, tmpRow2, tileN_);
                tmpRowQ_.FreeTensor(tmpRow2);

                // Copy out
                AscendC::DataCopy(yGm_[r * N_ + nBase], rowVec, nCur);
                outQueueRow_.FreeTensor(rowVec);
            }

            biasVecQ_.FreeTensor(biasTile);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueWT_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueASeg_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueRow_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpRowQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasVecQ_;

    AscendC::GlobalTensor<half> xGm_;
    AscendC::GlobalTensor<half> wGm_;
    AscendC::GlobalTensor<half> bGm_;
    AscendC::GlobalTensor<half> yGm_;

    uint32_t M_{0}, K_{0}, N_{0};
    uint32_t tileN_{128}, tileK_{256};
    float multiplier_{1.0f};
    float negativeSlope_{0.01f};
};

extern "C" __global__ __aicore__ void gemm_multiply_leakyrelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMultiplyLeakyreluCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileN, tiling_data.tileK,
            tiling_data.multiplier, tiling_data.negativeSlope);
    op.Process();
}

static inline at::Tensor to_contig_half(const at::Tensor& t) {
    return t.to(at::kHalf).contiguous();
}

at::Tensor gemm_mul_leakyrelu_impl_npu(const at::Tensor& x,
                                       const at::Tensor& w,
                                       const at::Tensor& b,
                                       const at::Tensor& multiplier,
                                       const at::Tensor& negative_slope) {
    // Prepare inputs: convert to fp16 and ensure proper shapes.
    // Matmul expects w of shape [K, N], so transpose from [N, K].
    at::Tensor x_h = to_contig_half(x);
    at::Tensor w_t_h = to_contig_half(w.transpose(0, 1)); // [K, N]
    at::Tensor b_h = to_contig_half(b);

    // Output in half, will cast back to float to align with original PyTorch behavior.
    std::vector<int64_t> y_sizes = {x_h.size(0), w_t_h.size(1)};
    at::Tensor y_h = at::empty(y_sizes, x_h.options());

    const double mult = multiplier.item<double>();
    const double ns = negative_slope.item<double>();

    EXEC_NPU_CMD(aclnnGemmMultiplyLeakyreluCustom, x_h, w_t_h, b_h, static_cast<float>(mult), static_cast<float>(ns), y_h);

    return y_h.to(at::kFloat);
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("gemm_multiply_leakyrelu_custom", &gemm_mul_leakyrelu_impl_npu);
