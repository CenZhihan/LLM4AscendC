host_operator_src="""
#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// UB model for this kernel:
//   - inQueueSlice: 2 * tilePost
//   - accBuffer   : 1 * tilePost
// Total floats = 3 * tilePost => bytes = 12 * tilePost
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape* inShape = context->GetInputShape(0);
    const std::vector<int64_t> originDims = inShape->GetOriginShape().GetDims();
    const uint32_t rank = static_cast<uint32_t>(originDims.size());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    if (dimPtr == nullptr) {
        return ge::GRAPH_FAILED;
    }
    int64_t dimAttr = *dimPtr;
    int64_t cdim = dimAttr;
    if (cdim < 0) cdim += static_cast<int64_t>(rank);
    if (cdim < 0 || cdim >= static_cast<int64_t>(rank)) {
        return ge::GRAPH_FAILED;
    }

    uint64_t pre64 = 1, red64 = 1, post64 = 1;
    for (int64_t i = 0; i < cdim; ++i) pre64 *= static_cast<uint64_t>(originDims[i]);
    red64 = static_cast<uint64_t>(originDims[cdim]);
    for (int64_t i = cdim + 1; i < static_cast<int64_t>(rank); ++i) post64 *= static_cast<uint64_t>(originDims[i]);

    // Safeguard against overflow to uint32_t, though typical shapes are safe.
    uint32_t pre = static_cast<uint32_t>(pre64);
    uint32_t reduceDim = static_cast<uint32_t>(red64);
    uint32_t post = static_cast<uint32_t>(post64);

    // tilePost from UB budget: bytes = 12 * tilePost
    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tilePost = maxTile;
    if (post < tilePost) {
        // choose either aligned down value or exact post if too small
        uint32_t aligned = AlignDown(post, ALIGN_ELEMS);
        tilePost = (aligned > 0u) ? aligned : post;
    }

    const uint32_t postTiles = (tilePost == 0u) ? 1u : CeilDiv(post, tilePost);
    uint32_t outerGroups = (pre == 0u) ? 0u : pre * postTiles;
    uint32_t blockDim = outerGroups;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_pre(pre);
    tiling.set_reduceDim(reduceDim);
    tiling.set_post(post);
    tiling.set_tilePost(tilePost);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t>& dims = x_shape->GetOriginShape().GetDims();
    const uint32_t rank = static_cast<uint32_t>(dims.size());

    int64_t dimAttr = 0;
    if (!context->GetAttrs()->GetAttrValue(0, dimAttr)) {
        return GRAPH_FAILED;
    }
    int64_t cdim = dimAttr;
    if (cdim < 0) cdim += static_cast<int64_t>(rank);
    if (cdim < 0 || cdim >= static_cast<int64_t>(rank)) {
        return GRAPH_FAILED;
    }

    std::vector<int64_t> outDims;
    outDims.reserve(rank > 0 ? rank - 1 : 0);
    for (int64_t i = 0; i < static_cast<int64_t>(rank); ++i) {
        if (i == cdim) continue;
        outDims.push_back(dims[i]);
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetOriginShape(outDims);
    y_shape->SetStorageShape(outDims);
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUFFERS = 1;
constexpr int32_t OUT_BUFFERS = 1;
constexpr int32_t TMP_BUFFERS = 1;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t pre, uint32_t reduceDim,
                                uint32_t post, uint32_t tilePost)
    {
        pre_ = pre;
        reduceDim_ = reduceDim;
        post_ = post;
        tilePost_ = tilePost;
        invReduce_ = (reduceDim_ > 0) ? (1.0f / static_cast<float>(reduceDim_)) : 0.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(pre_) * reduceDim_ * post_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(pre_) * post_);

        pipe_.InitBuffer(inQueue_, IN_BUFFERS, tilePost_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUT_BUFFERS, tilePost_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMP_BUFFERS, tilePost_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (post_ == 0 || (pre_ == 0 && reduceDim_ != 0)) return;

        const uint32_t postTiles = (tilePost_ == 0u) ? 1u : CeilDiv(post_, tilePost_);
        const uint32_t outerGroups = pre_ * postTiles;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t g = blockIdx; g < outerGroups; g += blockNum) {
            const uint32_t i = (postTiles == 0u) ? 0u : (g / postTiles);
            const uint32_t t = (postTiles == 0u) ? 0u : (g % postTiles);

            const uint32_t pBase = t * tilePost_;
            uint32_t currTp = post_ - pBase;
            if (currTp > tilePost_) currTp = tilePost_;

            ReduceOneGroup(i, pBase, currTp);
        }
    }

private:
    __aicore__ inline static uint32_t CeilDiv(uint32_t a, uint32_t b)
    {
        return (a + b - 1u) / b;
    }

    __aicore__ inline void ReduceOneGroup(uint32_t i, uint32_t pBase, uint32_t currTp)
    {
        // Base offsets in GM for this group
        const uint64_t xBase = static_cast<uint64_t>(i) * reduceDim_ * post_ + pBase;
        const uint64_t yBase = static_cast<uint64_t>(i) * post_ + pBase;

        // Accumulator buffer
        AscendC::LocalTensor<float> acc = tmpQueue_.AllocTensor<float>();
        // Load first slice into accumulator
        AscendC::DataCopy(acc, xGm_[xBase], currTp);

        // Accumulate remaining slices
        for (uint32_t k = 1; k < reduceDim_; ++k) {
            AscendC::LocalTensor<float> inLocal = inQueue_.AllocTensor<float>();
            AscendC::DataCopy(inLocal, xGm_[xBase + static_cast<uint64_t>(k) * post_], currTp);
            AscendC::Add(acc, acc, inLocal, currTp);
            inQueue_.FreeTensor(inLocal);
        }

        // Scale by 1/reduceDim to compute mean
        AscendC::Muls(acc, acc, invReduce_, currTp);

        // Store to output
        AscendC::DataCopy(yGm_[yBase], acc, currTp);

        tmpQueue_.FreeTensor(acc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFERS> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFERS> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUFFERS> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t pre_{0};
    uint32_t reduceDim_{0};
    uint32_t post_{0};
    uint32_t tilePost_{0};
    float invReduce_{0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y, tiling_data.pre, tiling_data.reduceDim, tiling_data.post, tiling_data.tilePost);
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t CanonicalDim(int64_t dim, int64_t rank) {
    if (dim < 0) dim += rank;
    return dim;
}

at::Tensor mean_reduction_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    TORCH_CHECK(self.scalar_type() == at::kFloat, \"Only float32 is supported by MeanReductionOverADimensionCustom\");
    int64_t rank = self.dim();
    int64_t cdim = CanonicalDim(dim, rank);
    TORCH_CHECK(cdim >= 0 && cdim < rank, \"dim out of range\");

    std::vector<int64_t> outSizes = self.sizes().vec();
    outSizes.erase(outSizes.begin() + cdim);

    at::Tensor result = at::empty(outSizes, self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, static_cast<int>(cdim), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"mean_reduction_over_a_dimension_custom\", &mean_reduction_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"mean_reduction_over_a_dimension_custom\", &mean_reduction_over_a_dimension_impl_npu,
          \"Mean reduction over a dimension (MeanReductionOverADimensionCustom on NPU)\");
}
