host_operator_src="""

#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous namespace

namespace optiling {

// UB model for ReLU post-pass only:
//   inQueue: 1 * tile
//   outQueue: 1 * tile
// Total floats = 2 * tile => bytes = 8 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    // Shapes: x [M, K], w [N, K] (PyTorch's F.linear uses w as [N, K], with matmul using w^T)
    const uint64_t M64 = x_shape->GetOriginShape().GetDim(0);
    const uint64_t Kx64 = x_shape->GetOriginShape().GetDim(1);
    const uint64_t N64 = w_shape->GetOriginShape().GetDim(0);
    const uint64_t Kw64 = w_shape->GetOriginShape().GetDim(1);
    if (Kx64 != Kw64) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(M64);
    const uint32_t K = static_cast<uint32_t>(Kx64);
    const uint32_t N = static_cast<uint32_t>(N64);

    // Compute relu tiling based on total elements M*N
    const uint64_t totalLen64 = static_cast<uint64_t>(M) * static_cast<uint64_t>(N);
    const uint32_t totalLen = static_cast<uint32_t>(totalLen64);

    uint32_t maxTile = UB_BUDGET_BYTES / 8;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLen < tileLen) {
        const uint32_t aligned = AlignDown(totalLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLen;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_reluTile(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;
    }
    const int64_t M = x_shape->GetOriginShape().GetDim(0);
    const int64_t N = w_shape->GetOriginShape().GetDim(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    std::vector<int64_t> out_dims = {M, N};
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"
// Use mix mode to allow both Matmul (cube) and vector for ReLU.
#include "lib/matmul_intf.h"

constexpr int32_t RELU_BUFFER_NUM = 2;

class KernelGemmAddReluCustom {
public:
    __aicore__ inline KernelGemmAddReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t reluTile)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileLen_ = reluTile;

        xPtr_ = (__gm__ float*)x;
        wPtr_ = (__gm__ float*)w;
        bPtr_ = (__gm__ float*)bias;
        yPtr_ = (__gm__ float*)y;

        totalLen_ = M_ * N_;

        // ReLU UB pipes
        pipe_.InitBuffer(inQueue_, RELU_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, RELU_BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void ComputeGemmBias()
    {
        // Define Matmul types: A: x [M,K], B: w [N,K] with transpose flag so logical B is [K,N].
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> aType;
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true> bType; // isTrans = true
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> cType;
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> biasType;
        AscendC::Matmul<aType, bType, cType, biasType> mm;

        // Initialize and set tensors; treat bias as [1, N], broadcast on rows.
        // The exact API signatures may vary by CANN version; following common pattern.
        mm.Init();
        mm.SetA(xPtr_, M_, K_, K_);      // A [M,K], lda = K
        mm.SetB(wPtr_, K_, N_, K_);      // B logical [K,N], ldb = K (original w is [N,K], isTrans=true)
        mm.SetC(yPtr_, M_, N_, N_);      // C [M,N], ldc = N
        mm.SetBias(bPtr_, N_);           // Bias [N], broadcast per row
        mm.Run();
        mm.End();
    }

    __aicore__ inline void ReluPost()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Prefetch first tile
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yPtr_ + gmOffset, len);
        inQueue_.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> inLocal = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> outLocal = outQueue_.AllocTensor<float>();

        AscendC::Maxs(outLocal, inLocal, 0.0f, len);

        outQueue_.EnQue<float>(outLocal);
        inQueue_.FreeTensor(inLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> outLocal = outQueue_.DeQue<float>();
        AscendC::DataCopy(yPtr_ + gmOffset, outLocal, len);
        outQueue_.FreeTensor(outLocal);
    }

private:
    // Matmul params
    __gm__ float* xPtr_ {nullptr};
    __gm__ float* wPtr_ {nullptr};
    __gm__ float* bPtr_ {nullptr};
    __gm__ float* yPtr_ {nullptr};
    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};

    // ReLU post-process
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, RELU_BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, RELU_BUFFER_NUM> outQueue_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddReluCustom op;
    op.Init(x, w, bias, y, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.reluTile);
    op.ComputeGemmBias();
    op.ReluPost();
}
