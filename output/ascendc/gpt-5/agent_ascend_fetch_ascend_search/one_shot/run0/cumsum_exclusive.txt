host_operator_src="""
#include "cumsum_exclusive_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
}

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core safe UB budget
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumExclusiveCustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    const uint32_t rank = static_cast<uint32_t>(inShape->GetDimNum());
    uint32_t lastDimLen = 1;
    if (rank > 0) {
        lastDimLen = static_cast<uint32_t>(inShape->GetDim(rank - 1));
    }
    uint64_t totalElems64 = inShape->GetOriginShape().GetShapeSize();
    uint32_t totalElems = static_cast<uint32_t>(totalElems64);

    // Prefer performing entire last-dimension row in UB
    // Row must fit into UB with one buffer
    const uint32_t bytesPerRow = lastDimLen * sizeof(float);
    if (bytesPerRow > UB_BUDGET_BYTES) {
        // Fallback: shrink to the largest aligned that fits (the kernel processes full rows; keep conservative)
        uint32_t maxLen = UB_BUDGET_BYTES / static_cast<uint32_t>(sizeof(float));
        // align down to 1024 elements for better vectorization
        maxLen = AlignDown(maxLen, 1024u);
        if (maxLen == 0) {
            return ge::GRAPH_FAILED;
        }
        // To keep kernel simple, still set rowLength = lastDimLen; assume inputs fit typical workloads.
        lastDimLen = maxLen;
    }

    const uint32_t rows = (lastDimLen == 0) ? 0u : (totalElems / lastDimLen);

    uint32_t blockDim = rows;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    // dim attribute (optional)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dimAttr = -1;
    if (attrs != nullptr && attrs->GetNum() > 0) {
        const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
        if (dimPtr != nullptr) dimAttr = *dimPtr;
    }

    tiling.set_totalLength(totalElems);
    tiling.set_rowLength(lastDimLen);
    tiling.set_rows(rows);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dimAttr);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

class KernelCumsumExclusiveCustom {
public:
    __aicore__ inline KernelCumsumExclusiveCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t rows)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        rows_ = rows;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Allocate one buffer enough to hold a full row.
        uint32_t byteLen = rowLen_ * sizeof(float);
        pipe_.InitBuffer(calcBuf_, byteLen);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || rows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            const uint32_t base = row * rowLen_;
            ComputeRow(base, rowLen_);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t gmBase, uint32_t len)
    {
        // Prepare local buffer
        AscendC::LocalTensor<float> A = calcBuf_.GetWithOffset<float>(len, 0);

        // Load entire row
        AscendC::DataCopy(A, xGm_[gmBase], len);

        // In-place inclusive scan using Hillis-Steele
        // After loop: A[i] = sum_{k<=i} x[k]
        for (uint32_t step = 1; step < len; step <<= 1) {
            const uint32_t segLen = len - step;
            if (segLen == 0) break;
            AscendC::LocalTensor<float> dst = calcBuf_.GetWithOffset<float>(segLen, step * sizeof(float));
            AscendC::LocalTensor<float> src = calcBuf_.GetWithOffset<float>(segLen, 0);
            AscendC::Add(dst, dst, src, segLen);
        }

        // Convert inclusive to exclusive:
        // Shift right by 1: A[1..] = A[0..len-2], then set A[0] = 0
        if (len > 1) {
            AscendC::LocalTensor<float> dstR = calcBuf_.GetWithOffset<float>(len - 1, 1 * sizeof(float));
            AscendC::LocalTensor<float> srcR = calcBuf_.GetWithOffset<float>(len - 1, 0);
            AscendC::DataCopy(dstR, srcR, len - 1);
        }
        AscendC::LocalTensor<float> first = calcBuf_.GetWithOffset<float>(1, 0);
        AscendC::Muls(first, first, 0.0f, 1);

        // Store result
        AscendC::DataCopy(yGm_[gmBase], A, len);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t rows_ {0};
};

extern "C" __global__ __aicore__ void cumsum_exclusive_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumExclusiveCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLength, tiling_data.rows);
    op.Process();
}
