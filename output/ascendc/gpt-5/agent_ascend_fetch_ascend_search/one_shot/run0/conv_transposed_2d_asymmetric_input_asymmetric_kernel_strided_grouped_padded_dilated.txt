class KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Hin, uint32_t Win,
                                uint32_t Cout, uint32_t Groups,
                                uint32_t KH, uint32_t KW,
                                uint32_t StrideH, uint32_t StrideW,
                                uint32_t PadH, uint32_t PadW,
                                uint32_t OutPadH, uint32_t OutPadW,
                                uint32_t DilationH, uint32_t DilationW,
                                uint32_t OutH, uint32_t OutW,
                                uint32_t TileH)
    {
        N_ = N; Cin_ = Cin; Hin_ = Hin; Win_ = Win;
        Cout_ = Cout; Groups_ = Groups;
        KH_ = KH; KW_ = KW;
        StrideH_ = StrideH; StrideW_ = StrideW;
        PadH_ = PadH; PadW_ = PadW;
        OutPadH_ = OutPadH; OutPadW_ = OutPadW;
        DilationH_ = DilationH; DilationW_ = DilationW;
        OutH_ = OutH; OutW_ = OutW;
        TileH_ = (TileH == 0) ? 1u : TileH;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Cin_ * Hin_ * Win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cin_ * (Cout_ / Groups_) * KH_ * KW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Cout_ * OutH_ * OutW_);

        // Buffers: y tile, x row, kernel (KH*KW)
        pipe_.InitBuffer(yTileQ_, 1, TileH_ * OutW_ * sizeof(float));
        pipe_.InitBuffer(xRowQ_, 1, Win_ * sizeof(float));
        pipe_.InitBuffer(kTileQ_, 1, KH_ * KW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t totalTasks = N_ * Cout_;
        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t task = blockIdx; task < totalTasks; task += blockNum) {
            const uint32_t n = task / Cout_;
            const uint32_t oc = task % Cout_;

            const uint32_t CoutG = Cout_ / Groups_;
            const uint32_t CinG = Cin_ / Groups_;
            const uint32_t g = oc / CoutG;
            const uint32_t oc_g = oc % CoutG;

            for (uint32_t ohBase = 0; ohBase < OutH_; ohBase += TileH_) {
                uint32_t rows = TileH_;
                if (ohBase + rows > OutH_) rows = OutH_ - ohBase;
                uint32_t tileElems = rows * OutW_;

                AscendC::LocalTensor<float> yTile = yTileQ_.AllocTensor<float>();
                // Zero initialize y tile
                for (uint32_t i = 0; i < tileElems; ++i) {
                    yTile.SetValue(i, 0.0f);
                }

                for (uint32_t ic_g = 0; ic_g < CinG; ++ic_g) {
                    uint32_t ic = g * CinG + ic_g;

                    // Load kernel K[KH*KW] for (ic, oc_g)
                    AscendC::LocalTensor<float> kLocal = kTileQ_.AllocTensor<float>();
                    uint64_t kOffset = (((uint64_t)ic * CoutG + oc_g) * KH_) * KW_;
                    AscendC::DataCopy(kLocal, wGm_[kOffset], KH_ * KW_);

                    for (uint32_t kh = 0; kh < KH_; ++kh) {
                        for (uint32_t oh = ohBase; oh < ohBase + rows; ++oh) {
                            int32_t t_h = (int32_t)oh + (int32_t)PadH_ - (int32_t)kh * (int32_t)DilationH_;
                            if (t_h % (int32_t)StrideH_ != 0) {
                                continue;
                            }
                            int32_t ih = t_h / (int32_t)StrideH_;
                            if (ih < 0 || ih >= (int32_t)Hin_) {
                                continue;
                            }

                            // Load x row for (n, ic, ih, :)
                            AscendC::LocalTensor<float> xRow = xRowQ_.AllocTensor<float>();
                            uint64_t xRowOffset = ((((uint64_t)n * Cin_ + ic) * Hin_) + (uint32_t)ih) * Win_;
                            AscendC::DataCopy(xRow, xGm_[xRowOffset], Win_);

                            for (uint32_t ow = 0; ow < OutW_; ++ow) {
                                for (uint32_t kw = 0; kw < KW_; ++kw) {
                                    int32_t t_w = (int32_t)ow + (int32_t)PadW_ - (int32_t)kw * (int32_t)DilationW_;
                                    if (t_w % (int32_t)StrideW_ != 0) {
                                        continue;
                                    }
                                    int32_t iw = t_w / (int32_t)StrideW_;
                                    if (iw < 0 || iw >= (int32_t)Win_) {
                                        continue;
                                    }
                                    float xv = xRow.GetValue((uint32_t)iw);
                                    float kv = kLocal.GetValue(kh * KW_ + kw);
                                    uint32_t yIndex = (oh - ohBase) * OutW_ + ow;
                                    float yv = yTile.GetValue(yIndex);
                                    yTile.SetValue(yIndex, yv + xv * kv);
                                }
                            }

                            xRowQ_.FreeTensor(xRow);
                        }
                    }

                    kTileQ_.FreeTensor(kLocal);
                }

                // Store y tile back to GM per row
                for (uint32_t r = 0; r < rows; ++r) {
                    uint64_t yRowOffset = ((((uint64_t)n * Cout_ + oc) * OutH_) + (ohBase + r)) * OutW_;
                    AscendC::DataCopy(yGm_[yRowOffset], yTile[r * OutW_], OutW_);
                }

                yTileQ_.FreeTensor(yTile);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> kTileQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Hin_{0}, Win_{0};
    uint32_t Cout_{0}, Groups_{1};
    uint32_t KH_{0}, KW_{0};
    uint32_t StrideH_{1}, StrideW_{1};
    uint32_t PadH_{0}, PadW_{0};
    uint32_t OutPadH_{0}, OutPadW_{0};
    uint32_t DilationH_{1}, DilationW_{1};
    uint32_t OutH_{0}, OutW_{0};
    uint32_t TileH_{1};
};
