kernel_src="""
#include "kernel_operator.h"

class KernelConv2dMinAddMultiplyCustom {
public:
    __aicore__ inline KernelConv2dMinAddMultiplyCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR cbias, GM_ADDR pbias,
                                GM_ADDR cval, GM_ADDR sval, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t K, uint32_t hasBias)
    {
        xPtr_ = (__gm__ const float*)x;
        wPtr_ = (__gm__ const float*)w;
        cbiasPtr_ = (__gm__ const float*)cbias;
        pbiasPtr_ = (__gm__ const float*)pbias;
        cvalPtr_ = (__gm__ const float*)cval;
        svalPtr_ = (__gm__ const float*)sval;
        yPtr_ = (__gm__ float*)y;

        N_ = N; Cin_ = Cin; H_ = H; W_ = W; Cout_ = Cout; K_ = K;
        hasBias_ = (hasBias != 0u);

        HOut_ = H_ - K_ + 1;
        WOut_ = W_ - K_ + 1;
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || K_ == 0) return;
        if (HOut_ <= 0 || WOut_ <= 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const float cap = cvalPtr_[0];
        const float scale = svalPtr_[0];

        for (uint32_t co = blockIdx; co < Cout_; co += ((blockNum==0)?1:blockNum)) {
            const float b = hasBias_ ? cbiasPtr_[co] : 0.0f;
            const float pb = pbiasPtr_[co];  // broadcast [Cout,1,1] -> scalar per channel

            for (uint32_t n = 0; n < N_; ++n) {
                for (uint32_t ho = 0; ho < (uint32_t)HOut_; ++ho) {
                    for (uint32_t wo = 0; wo < (uint32_t)WOut_; ++wo) {
                        float acc = b;
                        // Convolution inner product
                        for (uint32_t ci = 0; ci < Cin_; ++ci) {
                            for (uint32_t r = 0; r < K_; ++r) {
                                const uint32_t hi = ho + r;
                                const int64_t xRowBase = (((int64_t)n * Cin_ + ci) * H_ + hi) * W_;
                                const int64_t wRowBase = (((int64_t)co * Cin_ + ci) * K_ + r) * K_;
                                for (uint32_t s = 0; s < K_; ++s) {
                                    const uint32_t wi = wo + s;
                                    const float xv = xPtr_[xRowBase + wi];
                                    const float wv = wPtr_[wRowBase + s];
                                    acc += xv * wv;
                                }
                            }
                        }
                        // Post-ops: min with constant cap, add post_bias, multiply by scaling
                        float v = (acc < cap) ? acc : cap;
                        v = v + pb;
                        v = v * scale;

                        const int64_t yIndex = (((int64_t)n * Cout_ + co) * HOut_ + ho) * WOut_ + wo;
                        yPtr_[yIndex] = v;
                    }
                }
            }
        }
    }

private:
    __gm__ const float* xPtr_ {nullptr};
    __gm__ const float* wPtr_ {nullptr};
    __gm__ const float* cbiasPtr_ {nullptr};
    __gm__ const float* pbiasPtr_ {nullptr};
    __gm__ const float* cvalPtr_ {nullptr};
    __gm__ const float* svalPtr_ {nullptr};
    __gm__ float* yPtr_ {nullptr};

    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0}, Cout_{0}, K_{0};
    int32_t HOut_{0}, WOut_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(GM_ADDR x,
                                                                     GM_ADDR weight,
                                                                     GM_ADDR conv_bias,
                                                                     GM_ADDR post_bias,
                                                                     GM_ADDR constant_value,
                                                                     GM_ADDR scaling_factor,
                                                                     GM_ADDR y,
                                                                     GM_ADDR workspace,
                                                                     GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinAddMultiplyCustom op;
    op.Init(x, weight, conv_bias, post_bias, constant_value, scaling_factor, y,
            tiling_data.N, tiling_data.Cin, tiling_data.H, tiling_data.W,
            tiling_data.Cout, tiling_data.K, tiling_data.hasBias);
    op.Process();
}

#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor conv2d_min_add_multiply_impl_npu(const at::Tensor& x,
                                                   const at::Tensor& weight,
                                                   c10::optional<at::Tensor> conv_bias_opt,
                                                   const at::Tensor& post_bias,
                                                   const at::Tensor& constant_value,
                                                   const at::Tensor& scaling_factor)
{
    TORCH_CHECK(x.dim() == 4, "x must be 4D [N,C,H,W]");
    TORCH_CHECK(weight.dim() == 4, "weight must be 4D [Cout,Cin,K,K]");
    TORCH_CHECK(post_bias.numel() == weight.size(0), "post_bias shape must be [Cout,1,1] broadcastable");

    const int64_t N = x.size(0);
    const int64_t Cin = x.size(1);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);
    const int64_t Cout = weight.size(0);
    const int64_t Kin = weight.size(1);
    const int64_t K = weight.size(2);
    TORCH_CHECK(K == weight.size(3), "Kernel must be square");
    TORCH_CHECK(Cin == Kin, "Cin mismatch");

    const int64_t HOut = H - K + 1;
    const int64_t WOut = W - K + 1;
    TORCH_CHECK(HOut > 0 && WOut > 0, "Invalid output spatial size; require H>=K and W>=K");

    at::Tensor result = at::empty({N, Cout, HOut, WOut}, x.options());

    if (conv_bias_opt.has_value()) {
        const at::Tensor& conv_bias = conv_bias_opt.value();
        EXEC_NPU_CMD(aclnnConv2dMinAddMultiplyCustom, x, weight, conv_bias, post_bias, constant_value, scaling_factor, result);
    } else {
        // Pass an undefined tensor for optional bias
        at::Tensor empty_bias;
        EXEC_NPU_CMD(aclnnConv2dMinAddMultiplyCustom, x, weight, empty_bias, post_bias, constant_value, scaling_factor, result);
    }
    return result;
}
