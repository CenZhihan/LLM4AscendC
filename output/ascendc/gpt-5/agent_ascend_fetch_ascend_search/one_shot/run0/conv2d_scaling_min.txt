host_operator_src="""

#include "conv2d_scaling_min_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : (a + b - 1u) / b;
}
}

namespace optiling {

// UB model for this kernel (per-core):
//   - xQueue: 1 * tileLen floats
//   - minBuf: 1 * tileLen floats
// Total bytes ~= 8 * tileLen
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dScalingMinCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const std::vector<int64_t> dims = xShape->GetOriginShape().GetDims();

    if (dims.size() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t n = static_cast<uint32_t>(dims[0]);
    const uint32_t c = static_cast<uint32_t>(dims[1]);
    const uint32_t h = static_cast<uint32_t>(dims[2]);
    const uint32_t w = static_cast<uint32_t>(dims[3]);
    const uint32_t hw = h * w;

    // Determine tile length along the spatial/batch-flattened axis (per image).
    uint32_t maxTile = UB_BUDGET_BYTES / 8; // two float buffers
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (hw < tileLen) {
        const uint32_t aligned = AlignDown(hw, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : hw;
    }

    // Number of tiles per image
    const uint32_t tilesPerImg = (tileLen == 0) ? 1u : CeilDiv(hw, tileLen);
    uint32_t totalTiles = tilesPerImg * n;

    // Choose blockDim conservatively
    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_hw(hw);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> inDims = x_shape->GetOriginShape().GetDims();
    if (inDims.size() != 4) {
        return GRAPH_FAILED;
    }
    std::vector<int64_t> outDims = {inDims[0], 1, inDims[2], inDims[3]};
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) {
        y_shape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;

class KernelConv2dScalingMinCustom {
public:
    __aicore__ inline KernelConv2dScalingMinCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR /*weight*/, GM_ADDR /*bias*/,
                                GM_ADDR scale, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t hw, uint32_t tileLen)
    {
        n_ = n;
        c_ = c;
        hw_ = hw;
        tileLen_ = tileLen;
        tilesPerImg_ = (tileLen_ == 0) ? 1u : (hw_ + tileLen_ - 1u) / tileLen_;
        totalTiles_ = tilesPerImg_ * n_;

        // Load scalar scale from GM (assumes numel == 1)
        scale_ = *(__gm__ float*)scale;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint32_t)(n_ * c_ * hw_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint32_t)(n_ * hw_)); // y has shape [n,1,h,w]

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float)); // min buffer
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || hw_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t t = blockIdx; t < totalTiles_; t += blockNum) {
            // Decode tile index to (nIdx, tileIdxInImg)
            const uint32_t nIdx = t / tilesPerImg_;
            const uint32_t tileIdxInImg = t % tilesPerImg_;
            const uint32_t tileStart = tileIdxInImg * tileLen_;
            uint32_t len = hw_ - tileStart;
            if (len > tileLen_) len = tileLen_;
            if (len == 0) continue;

            ComputeTile(nIdx, tileStart, len);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t nIdx, uint32_t tileStart, uint32_t len)
    {
        // Allocate min buffer
        AscendC::LocalTensor<float> minLocal = tmpQueue_.AllocTensor<float>();

        // Process first channel: minLocal = x0 * scale
        {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            const uint32_t off0 = ((nIdx * c_ + 0u) * hw_) + tileStart;
            AscendC::DataCopy(xLocal, xGm_[off0], len);
            AscendC::Muls(xLocal, xLocal, scale_, len);
            AscendC::DataCopy(minLocal, xLocal, len);
            inQueueX_.FreeTensor(xLocal);
        }

        // Accumulate min across remaining channels
        for (uint32_t ci = 1; ci < c_; ++ci) {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            const uint32_t off = ((nIdx * c_ + ci) * hw_) + tileStart;
            AscendC::DataCopy(xLocal, xGm_[off], len);
            AscendC::Muls(xLocal, xLocal, scale_, len);
            AscendC::Min(minLocal, minLocal, xLocal, len);
            inQueueX_.FreeTensor(xLocal);
        }

        // Write result to y: y[n,0,h,w] contiguous segment
        const uint32_t yOff = nIdx * hw_ + tileStart;
        AscendC::DataCopy(yGm_[yOff], minLocal, len);

        tmpQueue_.FreeTensor(minLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_ {0};
    uint32_t c_ {0};
    uint32_t hw_ {0};
    uint32_t tileLen_ {0};
    uint32_t tilesPerImg_ {0};
    uint32_t totalTiles_ {0};
    float scale_ {1.0f};
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                GM_ADDR scale, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dScalingMinCustom op;
    op.Init(x, weight, bias, scale, y,
            tiling_data.n, tiling_data.c, tiling_data.hw, tiling_data.tileLength);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv2d_scaling_min_impl_npu(const at::Tensor& x,
                                       const at::Tensor& weight,
                                       c10::optional<at::Tensor> bias_opt,
                                       const at::Tensor& scale) {
    // x is expected to be the output of conv2d: [N, C_out, H_out, W_out]
    // weight/bias are passed to match the interface; kernel ignores them.
    at::Tensor y = at::empty({x.size(0), 1, x.size(2), x.size(3)}, x.options());
    if (bias_opt.has_value()) {
        EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, bias_opt.value(), scale, y);
    } else {
        // Pass an empty tensor when bias is not provided
        at::Tensor empty_bias;
        EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, empty_bias, scale, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_scaling_min_custom", &conv2d_scaling_min_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_scaling_min_custom", &conv2d_scaling_min_impl_npu,

model_src='''
import torch
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

_FN = 'conv2d_scaling_min_custom'

class ModelNew(torch.nn.Module):
    """Uses fused custom kernel for scaling + channelwise min, conv runs via PyTorch."""
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor):
        super(ModelNew, self).__init__()

    def forward(self, x, weight, bias_opt, scale):
        # Compute conv2d using PyTorch (NPU-accelerated if available)
        conv_b = bias_opt if bias_opt is not None else None
        x_conv = F.conv2d(x, weight, conv_b, stride=1, padding=0, dilation=1)

        # Call custom kernel to do: y = min( x_conv * scale, dim=1, keepdim=True )
        fn = getattr(custom_ops_lib, _FN)
        return fn(x_conv, weight, bias_opt, scale)
