static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
}

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;        // per-core conservative UB budget
static constexpr uint32_t ALIGN_ELEMS = 256;                    // prefer DMA alignment
static constexpr uint32_t MIN_BLOCKS = 8;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const uint32_t c = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t h = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t w = static_cast<uint32_t>(x_shape->GetDim(3));
    const uint32_t k = static_cast<uint32_t>(w_shape->GetDim(2));

    if (k == 0 || h < k) {

        context->SetBlockDim(1);
        tiling.set_n(n);
        tiling.set_c(c);
        tiling.set_h(h);

        tiling.set_tileW(0);

        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

        size_t *workspace = context->GetWorkspaceSizes(1);
        workspace[0] = 0;
        return ge::GRAPH_SUCCESS;

    // Memory per width tile = (K input rows + 2 working rows) * tileW * sizeof(float)
    // inQueueX: K * tileW, outQueueY: 1 * tileW, tmpQueue: 1 * tileW
    const uint32_t bytesPerElem = 4;

        maxTileW = (aligned > 0) ? aligned : w;

    if (tileW == 0) tileW = w;

    const uint64_t totalTasks = static_cast<uint64_t>(n) * static_cast<uint64_t>(c) * static_cast<uint64_t>(hOut);
    uint32_t blockDim = static_cast<uint32_t>(totalTasks);
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_k(k);
    tiling.set_hOut(hOut);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUFFER_NUM = 2;
constexpr int32_t OUT_BUFFER_NUM = 2;

class KernelConvDepthwise2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t W,
                                uint32_t k, uint32_t hOut, uint32_t tileW)
    {
        n_ = n; c_ = c; h_ = h; w_ = W; k_ = k; hOut_ = hOut; tileW_ = tileW;
        totalX_ = n_ * c_ * h_ * w_;
        totalY_ = (hOut_ == 0 ? 0u : n_ * c_ * hOut_ * w_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalX_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, c_ * k_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalY_);

        // UB buffers: in = k * tileW, out = tileW, tmp = tileW
        pipe_.InitBuffer(inQueueX_, IN_BUFFER_NUM, k_ * tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(weightQueue_, 1, k_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalY_ == 0 || tileW_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t tasks = static_cast<uint64_t>(n_) * static_cast<uint64_t>(c_) * static_cast<uint64_t>(hOut_);

        for (uint64_t t = blockIdx; t < tasks; t += blockNum) {
            uint32_t ho = static_cast<uint32_t>(t / (static_cast<uint64_t>(n_) * c_));
            uint64_t rem = t % (static_cast<uint64_t>(n_) * c_);
            uint32_t ni = static_cast<uint32_t>(rem / c_);
            uint32_t ci = static_cast<uint32_t>(rem % c_);

            curN_ = ni;
            curC_ = ci;
            curHOut_ = ho;

            // Prefetch first width tile
            uint32_t baseW = 0;
            uint32_t curLen = (w_ < tileW_) ? w_ : tileW_;

            // Load channel weights once for this task into local
            weightLocal_ = weightQueue_.AllocTensor<float>();
            AscendC::DataCopy(weightLocal_, wGm_[ci * k_], k_);

            CopyIn(baseW, curLen);

            while (true) {
                // Next prefetch
                uint32_t nextBaseW = baseW + tileW_;
                uint32_t nextLen = 0;
                if (nextBaseW < w_) {
                    nextLen = w_ - nextBaseW;
                    if (nextLen > tileW_) nextLen = tileW_;
                    CopyIn(nextBaseW, nextLen);
                }

                Compute(curLen);
                CopyOut(baseW, curLen);

                if (nextLen == 0) break;
                baseW = nextBaseW;
                curLen = nextLen;
            }

            weightQueue_.FreeTensor(weightLocal_);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t baseW, uint32_t len)
    {
        if (len == 0) return;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        // Pack K rows x[h_out + k, baseW:baseW+len] contiguously
        for (uint32_t kk = 0; kk < k_; ++kk) {
            uint64_t xBase = (((static_cast<uint64_t>(curN_) * c_ + curC_) * h_ + (curHOut_ + kk)) * w_) + baseW;
            AscendC::LocalTensor<float> dstRow = xLocal[kk * tileW_];
            AscendC::DataCopy(dstRow, xGm_[xBase], len);
        }
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        if (len == 0) return;
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

        // y = sum_{k}( x_k * w[k] )
        // First term initialize yLocal
        {
            float wk0 = weightLocal_.GetValue(0);
            AscendC::LocalTensor<float> x0 = xLocal[0];
            AscendC::Muls(yLocal, x0, wk0, len);
        }
        for (uint32_t kk = 1; kk < k_; ++kk) {
            float wkk = weightLocal_.GetValue(kk);
            AscendC::LocalTensor<float> xk = xLocal[kk * tileW_];
            AscendC::Muls(tmpLocal, xk, wkk, len);
            AscendC::Add(yLocal, yLocal, tmpLocal, len);
        }

        tmpQueue_.FreeTensor(tmpLocal);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t baseW, uint32_t len)
    {
        if (len == 0) return;
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint64_t yBase = (((static_cast<uint64_t>(curN_) * c_ + curC_) * hOut_ + curHOut_) * w_) + baseW;
        AscendC::DataCopy(yGm_[yBase], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    AscendC::LocalTensor<float> weightLocal_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0}, k_{0}, hOut_{0}, tileW_{0};
    uint32_t totalX_{0}, totalY_{0};

    // Current task indices
    uint32_t curN_{0}, curC_{0}, curHOut_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.k, tiling_data.hOut, tiling_data.tileW);
    op.Process();
}
