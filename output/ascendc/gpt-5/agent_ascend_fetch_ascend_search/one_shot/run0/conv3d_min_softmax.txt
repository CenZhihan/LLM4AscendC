kernel_src="""

#include "kernel_operator.h"
#include <cmath>

extern \"C\" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x,
                                                                  GM_ADDR weight,
                                                                  GM_ADDR bias,
                                                                  GM_ADDR y,
                                                                  GM_ADDR workspace,
                                                                  GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N   = tiling_data.N;
    const uint32_t Cin = tiling_data.Cin;
    const uint32_t D   = tiling_data.D;
    const uint32_t H   = tiling_data.H;
    const uint32_t W   = tiling_data.W;

    const uint32_t Cout = tiling_data.Cout;
    const uint32_t Kd   = tiling_data.Kd;
    const uint32_t Kh   = tiling_data.Kh;
    const uint32_t Kw   = tiling_data.Kw;

    const uint32_t Dout = tiling_data.Dout;
    const uint32_t Hout = tiling_data.Hout;
    const uint32_t Wout = tiling_data.Wout;

    if (Dout == 0 || Hout == 0 || Wout == 0) {
        return;
    }

    __gm__ const float* xGm = reinterpret_cast<__gm__ const float*>(x);
    __gm__ const float* wGm = reinterpret_cast<__gm__ const float*>(weight);
    __gm__ const float* bGm = reinterpret_cast<__gm__ const float*>(bias);
    __gm__ float* yGm       = reinterpret_cast<__gm__ float*>(y);

    const uint64_t positions = static_cast<uint64_t>(N) * static_cast<uint64_t>(Hout) * static_cast<uint64_t>(Wout);

    // Strides for ND layout: x [N, Cin, D, H, W], y [N, Cout, Hout, Wout], w [Cout, Cin, Kd, Kh, Kw]
    const uint64_t xStrideN = static_cast<uint64_t>(Cin) * D * H * W;
    const uint64_t xStrideC = static_cast<uint64_t>(D) * H * W;
    const uint64_t xStrideD = static_cast<uint64_t>(H) * W;
    const uint64_t xStrideH = static_cast<uint64_t>(W);
    const uint64_t xStrideW = 1ull;

    const uint64_t yStrideN = static_cast<uint64_t>(Cout) * Hout * Wout;
    const uint64_t yStrideC = static_cast<uint64_t>(Hout) * Wout;
    const uint64_t yStrideH = static_cast<uint64_t>(Wout);
    const uint64_t yStrideW = 1ull;

    const uint64_t wStrideCo = static_cast<uint64_t>(Cin) * Kd * Kh * Kw;
    const uint64_t wStrideCi = static_cast<uint64_t>(Kd) * Kh * Kw;
    const uint64_t wStrideKd = static_cast<uint64_t>(Kh) * Kw;
    const uint64_t wStrideKh = static_cast<uint64_t>(Kw);
    const uint64_t wStrideKw = 1ull;

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    for (uint64_t pos = blockIdx; pos < positions; pos += blockNum) {
        const uint64_t hw_span = static_cast<uint64_t>(Hout) * Wout;
        const uint32_t n  = static_cast<uint32_t>(pos / hw_span);
        const uint64_t rem = pos % hw_span;
        const uint32_t h  = static_cast<uint32_t>(rem / Wout);
        const uint32_t w  = static_cast<uint32_t>(rem % Wout);

        // First pass: compute min over depth for each output channel and store to y as temporary buffer.
        for (uint32_t co = 0; co < Cout; ++co) {
            float minVal = 1.0e30f;

            for (uint32_t od = 0; od < Dout; ++od) {
                float acc = bGm[co];

                // Convolution at (n, co, od, h, w)
                for (uint32_t ci = 0; ci < Cin; ++ci) {
                    for (uint32_t kd = 0; kd < Kd; ++kd) {
                        const uint64_t x_d = static_cast<uint64_t>(od + kd);
                        for (uint32_t kh = 0; kh < Kh; ++kh) {
                            const uint64_t x_h = static_cast<uint64_t>(h + kh);
                            for (uint32_t kw = 0; kw < Kw; ++kw) {
                                const uint64_t x_w = static_cast<uint64_t>(w + kw);

                                const uint64_t x_idx =
                                    static_cast<uint64_t>(n) * xStrideN +
                                    static_cast<uint64_t>(ci) * xStrideC +
                                    x_d * xStrideD +
                                    x_h * xStrideH +
                                    x_w * xStrideW;

                                const uint64_t w_idx =
                                    static_cast<uint64_t>(co) * wStrideCo +
                                    static_cast<uint64_t>(ci) * wStrideCi +
                                    static_cast<uint64_t>(kd) * wStrideKd +
                                    static_cast<uint64_t>(kh) * wStrideKh +
                                    static_cast<uint64_t>(kw) * wStrideKw;

                                const float xv = xGm[x_idx];
                                const float ww = wGm[w_idx];
                                acc += xv * ww;
                            }
                        }
                    }
                }
                if (acc < minVal) {
                    minVal = acc;
                }
            }

            const uint64_t y_idx =
                static_cast<uint64_t>(n) * yStrideN +
                static_cast<uint64_t>(co) * yStrideC +
                static_cast<uint64_t>(h) * yStrideH +
                static_cast<uint64_t>(w) * yStrideW;

            yGm[y_idx] = minVal; // store temporary min for softmax
        }

        // Second pass: softmax across channel dimension at fixed (n, h, w)
        float maxVal = -1.0e30f;
        for (uint32_t co = 0; co < Cout; ++co) {
            const uint64_t y_idx =
                static_cast<uint64_t>(n) * yStrideN +
                static_cast<uint64_t>(co) * yStrideC +
                static_cast<uint64_t>(h) * yStrideH +
                static_cast<uint64_t>(w) * yStrideW;
            const float v = yGm[y_idx];
            if (v > maxVal) maxVal = v;
        }

        float sumExp = 0.0f;
        for (uint32_t co = 0; co < Cout; ++co) {
            const uint64_t y_idx =
                static_cast<uint64_t>(n) * yStrideN +
                static_cast<uint64_t>(co) * yStrideC +
                static_cast<uint64_t>(h) * yStrideH +
                static_cast<uint64_t>(w) * yStrideW;
            const float v = yGm[y_idx];
            const float e = expf(v - maxVal);
            // Overwrite with unnormalized exp to save memory traffic
            yGm[y_idx] = e;
            sumExp += e;
        }

        const float invSum = 1.0f / sumExp;
        for (uint32_t co = 0; co < Cout; ++co) {
            const uint64_t y_idx =
                static_cast<uint64_t>(n) * yStrideN +
                static_cast<uint64_t>(co) * yStrideC +
                static_cast<uint64_t>(h) * yStrideH +
                static_cast<uint64_t>(w) * yStrideW;
            yGm[y_idx] = yGm[y_idx] * invSum;
        }
    }
}

at::Tensor conv3d_min_softmax_custom_impl_npu(const at::Tensor& x,
                                              const at::Tensor& weight,
                                              const at::Tensor& bias) {
    TORCH_CHECK(x.dim() == 5, \"x must be 5D [N, C, D, H, W]\");
    TORCH_CHECK(weight.dim() == 5, \"weight must be 5D [Cout, Cin, Kd, Kh, Kw]\");
    TORCH_CHECK(bias.dim() == 1, \"bias must be 1D [Cout]\");
    auto N = x.size(0);
    auto D = x.size(2);

    auto W = x.size(4);
    auto Cout = weight.size(0);
    auto Kd = weight.size(2);
    auto Kh = weight.size(3);
    auto Kw = weight.size(4);
    TORCH_CHECK(D >= Kd && H >= Kh && W >= Kw, \"Kernel larger than input\");
    auto Dout = D - Kd + 1;
    auto Hout = H - Kh + 1;
    auto Wout = W - Kw + 1;

    at::Tensor y = at::empty({N, Cout, Hout, Wout}, x.options());
    EXEC_NPU_CMD(aclnnConv3dMinSoftmaxCustom, x, weight, bias, y);
    return y;
}
