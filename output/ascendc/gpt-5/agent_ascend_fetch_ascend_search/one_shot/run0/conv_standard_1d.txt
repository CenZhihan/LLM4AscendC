host_operator_src="""
#include "conv_standard_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
}

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core UB share
static constexpr uint32_t VEC_STRIDE_NUM = 3;            // xVec, yVec, tmpVec
static constexpr uint32_t ELEM_SIZE = 4;                 // float32
static constexpr uint32_t ALIGN_ELEMS = 256;             // favor large aligned DMA
static constexpr uint32_t MAX_BLOCKS = 24;
static constexpr uint32_t MIN_BLOCKS = 8;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard1dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 3 || w_shape->GetOriginShape().GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t Lin = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t CinW = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    const uint32_t K    = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));

    if (CinW != Cin) {
        return ge::GRAPH_FAILED;
    }

    // stride=1, padding=0, dilation=1
    if (Lin < K) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t Lout = Lin - K + 1;

    // tile along output-length dimension
    uint32_t maxTilePos = UB_BUDGET_BYTES / (VEC_STRIDE_NUM * ELEM_SIZE);  // elements
    maxTilePos = AlignDown(maxTilePos, ALIGN_ELEMS);
    if (maxTilePos < ALIGN_ELEMS) maxTilePos = ALIGN_ELEMS;
    if (maxTilePos > Lout) {
        const uint32_t aligned = AlignDown(Lout, ALIGN_ELEMS);
        maxTilePos = (aligned > 0) ? aligned : Lout;
    }

    const uint32_t tilePos = (maxTilePos == 0) ? 1u : maxTilePos;

    const uint32_t tilesPerN = CeilDiv(Lout, tilePos);
    uint32_t totalTiles = N * tilesPerN;
    if (totalTiles == 0) totalTiles = 1;

    uint32_t blockDim = totalTiles;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_lin(Lin);
    tiling.set_lout(Lout);
    tiling.set_k(K);
    tiling.set_tilePos(tilePos);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 3 || w_shape->GetOriginShape().GetDimNum() != 3) {
        return GRAPH_FAILED;
    }

    const int64_t N   = x_shape->GetDim(0);
    const int64_t Cin = x_shape->GetDim(1);
    const int64_t Lin = x_shape->GetDim(2);

    const int64_t Cout = w_shape->GetDim(0);
    const int64_t CinW = w_shape->GetDim(1);
    const int64_t K    = w_shape->GetDim(2);

    if (Cin != CinW || Lin < K) {
        return GRAPH_FAILED;
    }

    const int64_t Lout = Lin - K + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Lout);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    const ge::DataType w_dtype = context->GetInputDataType(1);
    if (x_dtype != ge::DT_FLOAT || w_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvStandard1dCustom {
public:
    __aicore__ inline KernelConvStandard1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t lout, uint32_t k,
                                uint32_t tilePos)
    {
        n_ = n;
        cin_ = cin;
        cout_ = cout;
        lin_ = lin;
        lout_ = lout;
        k_ = k;
        tilePos_ = tilePos;
        kcin_ = cin_ * k_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * cin_ * lin_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(cout_ * cin_ * k_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_ * cout_ * lout_));

        pipe_.InitBuffer(inX_, BUFFER_NUM, tilePos_ * sizeof(float));
        pipe_.InitBuffer(yBuf_, BUFFER_NUM, tilePos_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, BUFFER_NUM, tilePos_ * sizeof(float));
        pipe_.InitBuffer(wRowBuf_, BUFFER_NUM, kcin_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || lin_ == 0 || lout_ == 0 || k_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tilesPerN = CeilDiv(lout_, tilePos_);
        uint32_t totalTiles = n_ * tilesPerN;
        if (totalTiles == 0) totalTiles = 1;

        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            const uint32_t nIdx = t / tilesPerN;
            const uint32_t posTileIdx = t % tilesPerN;
            const uint32_t posStart = posTileIdx * tilePos_;
            uint32_t curLen = lout_ - posStart;
            if (curLen > tilePos_) curLen = tilePos_;

            // Loop over output channels one by one to reduce UB usage
            for (uint32_t co = 0; co < cout_; ++co) {
                AscendC::LocalTensor<float> yLocal = yBuf_.AllocTensor<float>();
                AscendC::Duplicate(yLocal, 0.0f, curLen);

                // Load weight row for this output channel: length = cin_ * k_
                AscendC::LocalTensor<float> wRowLocal = wRowBuf_.AllocTensor<float>();
                const uint32_t wRowBase = co * kcin_;
                AscendC::DataCopy(wRowLocal, wGm_[wRowBase], kcin_);

                for (uint32_t ci = 0; ci < cin_; ++ci) {
                    for (uint32_t kk = 0; kk < k_; ++kk) {
                        // x offset: ((nIdx * cin_ + ci) * lin_) + (posStart + kk)
                        const uint32_t xBase = ((nIdx * cin_ + ci) * lin_) + (posStart + kk);

                        AscendC::LocalTensor<float> xLocal = inX_.AllocTensor<float>();
                        AscendC::DataCopy(xLocal, xGm_[xBase], curLen);

                        // w scalar from local weight row
                        const uint32_t wIdx = ci * k_ + kk;
                        float wScalar = wRowLocal.GetValue(wIdx);

                        AscendC::LocalTensor<float> tmpLocal = tmpBuf_.AllocTensor<float>();
                        AscendC::Muls(tmpLocal, xLocal, wScalar, curLen);
                        AscendC::Add(yLocal, yLocal, tmpLocal, curLen);

                        tmpBuf_.FreeTensor(tmpLocal);
                        inX_.FreeTensor(xLocal);
                    }
                }

                // write back y for this output channel
                const uint32_t yBase = ((nIdx * cout_ + co) * lout_) + posStart;
                AscendC::DataCopy(yGm_[yBase], yLocal, curLen);

                yBuf_.FreeTensor(yLocal);
                wRowBuf_.FreeTensor(wRowLocal);
            }
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wRowBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0};
    uint32_t cin_{0};
    uint32_t cout_{0};
    uint32_t lin_{0};
    uint32_t lout_{0};
    uint32_t k_{0};
    uint32_t kcin_{0};
    uint32_t tilePos_{0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.lout, tiling_data.k,
            tiling_data.tilePos);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv1d_standard_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    TORCH_CHECK(x.dim() == 3, \"x must be 3D (N, Cin, Lin)\");
    TORCH_CHECK(weight.dim() == 3, \"weight must be 3D (Cout, Cin, K)\");
    TORCH_CHECK(x.scalar_type() == at::kFloat && weight.scalar_type() == at::kFloat,
                \"Only float32 supported\");

    const auto N = x.size(0);
    const auto Cin = x.size(1);
    const auto Lin = x.size(2);
    const auto Cout = weight.size(0);
    const auto CinW = weight.size(1);
    const auto K = weight.size(2);

    TORCH_CHECK(Cin == CinW, \"Input Cin must match weight Cin\");
    TORCH_CHECK(Lin >= K, \"Input length must be >= kernel size\");
    const auto Lout = Lin - K + 1;

    at::Tensor y = at::empty({N, Cout, Lout}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard1dCustom, x, weight, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv_standard_1d_custom\", &conv1d_standard_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"conv_standard_1d_custom\", &conv1d_standard_impl_npu, \"1D standard convolution (ConvStandard1dCustom on NPU)\");
}
