namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : ((x / a) * a);
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : ((a + b - 1u) / b);
}
} // namespace

namespace optiling {

// UB budget per core; conservative
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// prefer DMA-friendly tile width
static constexpr uint32_t ALIGN_ELEMS = 256;
// avoid excessively long per-iteration latency
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

// Estimate per-element UB bytes: minBuf (4B) + outBuf (4B) + xLine (4B) + tanh tmp factor (~7*4B conservatively)
static constexpr uint32_t PER_ELEM_BYTES = 4u * (3u + 7u); // 40 bytes/elem

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMinTanhTanhCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    // Expect NCHW for x, OIHW for weight
    const auto x_dims = x_shape->GetOriginShape().GetDimSizes();
    const auto w_dims = w_shape->GetOriginShape().GetDimSizes();
    if (x_dims.size() != 4 || w_dims.size() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_dims[0]);
    const uint32_t Cin = static_cast<uint32_t>(x_dims[1]);
    const uint32_t H   = static_cast<uint32_t>(x_dims[2]);
    const uint32_t W   = static_cast<uint32_t>(x_dims[3]);
    const uint32_t Cout = static_cast<uint32_t>(w_dims[0]);
    const uint32_t CinW = static_cast<uint32_t>(w_dims[1]);
    const uint32_t kH = static_cast<uint32_t>(w_dims[2]);
    const uint32_t kW = static_cast<uint32_t>(w_dims[3]);

    if (CinW != Cin) {
        return ge::GRAPH_FAILED;
    }

    // Stride=1, pad=0, dilation=1
    if (H < kH || W < kW) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t Hout = H - kH + 1;
    const uint32_t Wout = W - kW + 1;

    // Determine if bias is provided
    bool hasBias = false;
    const gert::Shape* b_shape = context->GetInputShape(2);
    if (b_shape != nullptr && b_shape->GetOriginShape().GetShapeSize() > 0) {
        hasBias = true;
    }

    // Tile along output width; account for small constant overhead for (kW - 1) lookahead in input line
    uint32_t availBytes = UB_BUDGET_BYTES;
    if (kW > 1) {
        const uint32_t constOver = (kW - 1u) * 4u;
        if (availBytes > constOver) availBytes -= constOver;
    }

    uint32_t maxTile = (availBytes / PER_ELEM_BYTES);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileW = maxTile;
    if (Wout < tileW) {
        const uint32_t aligned = AlignDown(Wout, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : Wout;
    }
    if (tileW == 0) tileW = 1;

    // Parallelize across N * Hout rows
    const uint32_t totalRows = N * Hout;
    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cIn(Cin);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_cOut(Cout);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_hOut(Hout);
    tiling.set_wOut(Wout);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1u : 0u);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

// Fused Conv2D (stride=1,pad=0,dilation=1) + channel-min + tanh + tanh
// Input x: [N, Cin, H, W]
// Weight:  [Cout, Cin, kH, kW]
// Bias:    [Cout] (optional)
// Output y: [N, 1, H-kH+1, W-kW+1]
constexpr int32_t BUFFER_NUM = 1;

class KernelConv2dMinTanhTanhCustom {
public:
    __aicore__ inline KernelConv2dMinTanhTanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t wIn,
                                uint32_t cOut, uint32_t kH, uint32_t kW,
                                uint32_t hOut, uint32_t wOut, uint32_t tileW, uint32_t hasBias)
    {
        n_ = n; cin_ = cIn; h_ = h; w_ = wIn;
        cout_ = cOut; kH_ = kH; kW_ = kW;
        hout_ = hOut; wout_ = wOut;
        tileW_ = tileW;
        hasBias_ = (hasBias != 0);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * cin_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cout_) * cin_ * kH_ * kW_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * 1u * hout_ * wout_);

        // Buffers: minBuf(T), outBuf(T), xLine(T), tanhTmp(~T*16B conservative)
        pipe_.InitBuffer(minQ_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQ_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(xLineQ_, BUFFER_NUM, tileW_ * sizeof(float));
        // Conservative temporary buffer for Tanh: 7x input bytes + small extra
        uint32_t tanhTmpBytes = tileW_ * sizeof(float) * 8u;
        if (tanhTmpBytes < 256u) tanhTmpBytes = 256u;
        pipe_.InitBuffer(tmpQ_, BUFFER_NUM, tanhTmpBytes);
        pipe_.InitBuffer(scalarQ_, BUFFER_NUM, 16); // for 1-2 scalars
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || hout_ == 0 || wout_ == 0 || tileW_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalRows = n_ * hout_;
        if (blockIdx >= totalRows) return;

        for (uint32_t row = blockIdx; row < totalRows; row += blockNum) {
            const uint32_t n = row / hout_;
            const uint32_t ho = row % hout_;

            for (uint32_t wb = 0; wb < wout_; wb += tileW_) {
                const uint32_t len = AscendC::Min((uint32_t)tileW_, (uint32_t)(wout_ - wb));

                AscendC::LocalTensor<float> minLocal = minQ_.AllocTensor<float>();
                bool minInitialized = false;

                // Iterate over output channels; accumulate per-channel output vector, then update min
                for (uint32_t co = 0; co < cout_; ++co) {
                    AscendC::LocalTensor<float> outLocal = outQ_.AllocTensor<float>();
                    // outLocal = 0
                    AscendC::Dup(outLocal, 0.0f, len);

                    // Accumulate convolution for this co
                    for (uint32_t ci = 0; ci < cin_; ++ci) {
                        for (uint32_t kh = 0; kh < kH_; ++kh) {
                            const uint32_t hi = ho + kh;
                            for (uint32_t kw = 0; kw < kW_; ++kw) {
                                const uint32_t wi = wb + kw;

                                // Load input line segment xLine: length = len
                                AscendC::LocalTensor<float> xLine = xLineQ_.AllocTensor<float>();
                                uint64_t xOffset = ((((uint64_t)n * cin_ + ci) * h_ + hi) * w_) + wi;
                                AscendC::DataCopy(xLine, xGm_[xOffset], len);

                                // Load weight scalar wVal
                                AscendC::LocalTensor<float> sLocal = scalarQ_.AllocTensor<float>();
                                uint64_t wOffset = ((((uint64_t)co * cin_ + ci) * kH_ + kh) * kW_) + kw;
                                AscendC::DataCopy(sLocal, wGm_[wOffset], 1);

                                // Fetch scalar from local tensor
                                float wVal = 0.0f;
                                // emulate sLocal[0] read: copy to a 1-length temp and use GetValue-like op via Vector to scalar conversion
                                // AscendC API allows Dup with scalar; assume we can fetch to scalar via GetValue
                                wVal = sLocal.GetValue(0);

                                // temp = xLine * wVal; outLocal += temp
                                AscendC::Muls(xLine, xLine, wVal, len);
                                AscendC::Add(outLocal, outLocal, xLine, len);

                                xLineQ_.FreeTensor(xLine);
                                scalarQ_.FreeTensor(sLocal);
                            }
                        }
                    }

                    // Add bias if present
                    if (hasBias_) {
                        AscendC::LocalTensor<float> bLocal = scalarQ_.AllocTensor<float>();
                        AscendC::DataCopy(bLocal, bGm_[co], 1);
                        float bVal = bLocal.GetValue(0);
                        AscendC::Adds(outLocal, outLocal, bVal, len);
                        scalarQ_.FreeTensor(bLocal);
                    }

                    if (!minInitialized) {
                        AscendC::DataCopy(minLocal, outLocal, len);
                        minInitialized = true;
                    } else {
                        AscendC::Min(minLocal, minLocal, outLocal, len);
                    }

                    outQ_.FreeTensor(outLocal);
                }

                // Apply tanh twice
                AscendC::LocalTensor<uint8_t> tmpLocal = tmpQ_.AllocTensor<uint8_t>();
                AscendC::Tanh(minLocal, minLocal, tmpLocal, len);
                AscendC::Tanh(minLocal, minLocal, tmpLocal, len);

                // Store to output y: [N,1,Hout,Wout]
                uint64_t yOffset = ((((uint64_t)n) * 1u + 0u) * hout_ + ho) * wout_ + wb;
                AscendC::DataCopy(yGm_[yOffset], minLocal, len);

                tmpQ_.FreeTensor(tmpLocal);
                minQ_.FreeTensor(minLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> minQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> outQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> xLineQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, h_{0}, w_{0};
    uint32_t cout_{0}, kH_{0}, kW_{0};
    uint32_t hout_{0}, wout_{0};
    uint32_t tileW_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanhCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.n, tiling_data.cIn, tiling_data.h, tiling_data.w,
            tiling_data.cOut, tiling_data.kH, tiling_data.kW,
            tiling_data.hOut, tiling_data.wOut, tiling_data.tileW, tiling_data.hasBias);
    op.Process();
}
