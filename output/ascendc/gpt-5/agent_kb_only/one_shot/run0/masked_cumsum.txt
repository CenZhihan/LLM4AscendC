#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimAttr != nullptr) ? *dimAttr : 1;

    // Default to 2D [outter, inner]
    const gert::Shape *x_shape = context->GetInputShape(0);
    int64_t outter64 = 1;
    int64_t inner64  = static_cast<int64_t>(x_shape->GetOriginShape().GetShapeSize());

    // Try to obtain 2D dims if available
    // Note: If actual API differs, adapt GetDim calls accordingly.
    if (x_shape->GetOriginShape().GetDimNum() >= 2) {
        outter64 = x_shape->GetOriginShape().GetDim(0);
        inner64  = x_shape->GetOriginShape().GetDim(1);
    }

    // CumSum requires inner * sizeof(T) to be 32-byte aligned. Assume inputs are aligned or handled by framework.
    uint32_t outter = static_cast<uint32_t>(outter64);
    uint32_t inner  = static_cast<uint32_t>(inner64);

    uint32_t isLastAxis = (dim == 1) ? 1u : 0u;

    context->SetBlockDim(1);

    tiling.set_outter(outter);
    tiling.set_inner(inner);
    tiling.set_isLastAxis(isLastAxis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

template <typename T>
class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t outter, uint32_t inner, uint32_t isLastAxis)
    {
        outer_ = outter;
        inner_ = inner;
        isLastAxis_ = isLastAxis != 0;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ T*>(x), static_cast<uint64_t>(outer_) * inner_);
        mGm_.SetGlobalBuffer(reinterpret_cast<__gm__ T*>(mask), static_cast<uint64_t>(outer_) * inner_);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ T*>(y), static_cast<uint64_t>(outer_) * inner_);

        // For simplicity, allocate one-shot buffers sized to the full tensor.
        // In production, tile across outer/inner to fit UB.
        uint64_t totalElems = static_cast<uint64_t>(outer_) * inner_;
        pipe_.InitBuffer(inQueueX_, 1, totalElems * sizeof(T));
        pipe_.InitBuffer(inQueueM_, 1, totalElems * sizeof(T));
        pipe_.InitBuffer(outQueueY_, 1, totalElems * sizeof(T));
        pipe_.InitBuffer(lastRowQueue_, 1, inner_ * sizeof(T)); // required by CumSum, may be ignored
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX_.AllocTensor<T>();
        AscendC::LocalTensor<T> mLocal = inQueueM_.AllocTensor<T>();
        AscendC::DataCopy(xLocal, xGm_, static_cast<uint64_t>(outer_) * inner_);
        AscendC::DataCopy(mLocal, mGm_, static_cast<uint64_t>(outer_) * inner_);
        inQueueX_.EnQue(xLocal);
        inQueueM_.EnQue(mLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX_.DeQue<T>();
        AscendC::LocalTensor<T> mLocal = inQueueM_.DeQue<T>();
        AscendC::LocalTensor<T> yLocal = outQueueY_.AllocTensor<T>();
        AscendC::LocalTensor<T> lastRowLocal = lastRowQueue_.AllocTensor<T>();

        // Elementwise mask: x = x * mask
        AscendC::Mul(xLocal, xLocal, mLocal, static_cast<uint64_t>(outer_) * inner_);

        // CumSum along the required axis
        AscendC::CumSumInfo info{outer_, inner_};
        if (isLastAxis_) {
            // last axis
            static constexpr AscendC::CumSumConfig cfg{true, false, false};
            AscendC::CumSum<T, cfg>(yLocal, lastRowLocal, xLocal, info);
        } else {
            // first axis
            static constexpr AscendC::CumSumConfig cfg{false, false, false};
            AscendC::CumSum<T, cfg>(yLocal, lastRowLocal, xLocal, info);
        }

        outQueueY_.EnQue<T>(yLocal);
        lastRowQueue_.EnQue<T>(lastRowLocal);

        inQueueX_.FreeTensor(xLocal);
        inQueueM_.FreeTensor(mLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<T> yLocal = outQueueY_.DeQue<T>();
        AscendC::DataCopy(yGm_, yLocal, static_cast<uint64_t>(outer_) * inner_);
        outQueueY_.FreeTensor(yLocal);

        AscendC::LocalTensor<T> lastRowLocal = lastRowQueue_.DeQue<T>();
        lastRowQueue_.FreeTensor(lastRowLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueM_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> lastRowQueue_;

    AscendC::GlobalTensor<T> xGm_;
    AscendC::GlobalTensor<T> mGm_;
    AscendC::GlobalTensor<T> yGm_;

    uint32_t outer_{1};
    uint32_t inner_{1};
    bool isLastAxis_{true};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom<float> op;
    op.Init(x, mask, y, tiling_data.outter, tiling_data.inner, tiling_data.isLastAxis);
    op.Process();
}
"""

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor masked_cumsum_impl_npu(const at::Tensor& self, const at::Tensor& mask, int64_t dim) {
    at::Tensor mask_cast = mask;
    if (mask.dtype() != self.dtype()) {
        mask_cast = mask.to(self.dtype());
    }
    at::Tensor result = at::empty_like(self);
    EXEC_NPU_CMD(aclnnMaskedCumsumCustom, self, mask_cast, (int)dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("masked_cumsum_custom", &masked_cumsum_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("masked_cumsum_custom", &masked_cumsum_impl_npu, "Masked cumsum (MaskedCumsumCustom on NPU)");
}
