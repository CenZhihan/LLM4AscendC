host_operator_src="""

#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
}

namespace optiling {

// UB model per-core (float):
//   xVec: 1 * tile
//   yVec: 1 * tile
//   tmpVec: 1 * tile
//   w_o (per-output-channel weights): C
// Total bytes ~= 3 * 4 * tile + 4 * C
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core budget
static constexpr uint32_t ALIGN_ELEMS = 128;             // favor vector-friendly alignment
static constexpr uint32_t MIN_TILE = 128;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    // Expect x: [N, C, H, W], weight: [O, C, 1, 1]
    const int64_t N = xShape->GetOriginShape().GetDim(0);
    const int64_t C = xShape->GetOriginShape().GetDim(1);
    const int64_t H = xShape->GetOriginShape().GetDim(2);
    const int64_t W = xShape->GetOriginShape().GetDim(3);
    const int64_t O = wShape->GetOriginShape().GetDim(0);

    const uint64_t S = static_cast<uint64_t>(H) * static_cast<uint64_t>(W);

    // tileSpatLen from UB: 3 * 4 * tile + 4 * C <= UB
    uint64_t maxTileByUb = 0;
    if (UB_BUDGET_BYTES > static_cast<uint64_t>(C) * 4ULL) {
        maxTileByUb = (UB_BUDGET_BYTES - static_cast<uint64_t>(C) * 4ULL) / (3ULL * 4ULL);
    }
    uint32_t tile = static_cast<uint32_t>(maxTileByUb);
    if (tile > static_cast<uint32_t>(S)) tile = static_cast<uint32_t>(S);
    tile = AlignDown(tile, ALIGN_ELEMS);
    if (tile < MIN_TILE) tile = (S < MIN_TILE) ? static_cast<uint32_t>(S) : MIN_TILE;

    const uint32_t tilesPerImg = (tile == 0) ? 1u : CeilDiv(S, tile);
    const uint64_t totalTiles = static_cast<uint64_t>(N) * static_cast<uint64_t>(tilesPerImg);

    uint32_t blockDim = static_cast<uint32_t>(totalTiles);
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_c(static_cast<uint32_t>(C));
    tiling.set_h(static_cast<uint32_t>(H));
    tiling.set_w(static_cast<uint32_t>(W));
    tiling.set_o(static_cast<uint32_t>(O));
    tiling.set_tileSpatLen(tile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    gert::Shape *yShape = context->GetOutputShape(0);

    const int64_t N = xShape->GetOriginShape().GetDim(0);
    const int64_t H = xShape->GetOriginShape().GetDim(2);
    const int64_t W = xShape->GetOriginShape().GetDim(3);
    const int64_t O = wShape->GetOriginShape().GetDim(0);

    gert::Shape outShape;
    outShape.SetDimNum(4);
    outShape.SetDim(0, N);
    outShape.SetDim(1, O);
    outShape.SetDim(2, H);
    outShape.SetDim(3, W);
    *yShape = outShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t o,
                                uint32_t tileSpatLen)
    {
        n_ = n; c_ = c; h_ = h; w_ = w; o_ = o;
        s_ = static_cast<uint64_t>(h_) * static_cast<uint64_t>(w_);
        tile_ = (tileSpatLen == 0) ? static_cast<uint32_t>(s_) : tileSpatLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * c_ * s_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, static_cast<uint64_t>(o_) * c_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * o_ * s_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tile_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, c_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tile_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tile_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || o_ == 0 || s_ == 0 || tile_ == 0) return;

        const uint32_t tilesPerImg = static_cast<uint32_t>((s_ + tile_ - 1ULL) / tile_);
        const uint64_t totalTiles = static_cast<uint64_t>(n_) * tilesPerImg;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t t = blockIdx; t < totalTiles; t += blockNum) {
            const uint32_t nIdx = static_cast<uint32_t>(t / tilesPerImg);
            const uint32_t tileIdx = static_cast<uint32_t>(t % tilesPerImg);
            const uint64_t sBase = static_cast<uint64_t>(tileIdx) * tile_;
            uint32_t curLen = static_cast<uint32_t>(s_ - sBase);
            if (curLen > tile_) curLen = tile_;

            for (uint32_t oIdx = 0; oIdx < o_; ++oIdx) {
                // Load weight vector for this output channel (length C, contiguous)
                AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
                const uint64_t wOff = static_cast<uint64_t>(oIdx) * c_;
                AscendC::DataCopy(wLocal, wGm_[wOff], c_);
                inQueueW_.EnQue(wLocal);
                wLocal = inQueueW_.DeQue<float>();

                // Accumulator yLocal for this (nIdx, oIdx, spatial tile)
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                bool first = true;

                for (uint32_t cIdx = 0; cIdx < c_; ++cIdx) {
                    // Copy x vector for this input channel and spatial tile
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    const uint64_t xOff = ( (static_cast<uint64_t>(nIdx) * c_ + cIdx) * s_ ) + sBase;
                    AscendC::DataCopy(xLocal, xGm_[xOff], curLen);
                    inQueueX_.EnQue(xLocal);
                    xLocal = inQueueX_.DeQue<float>();

                    // Fetch weight scalar w[oIdx, cIdx] from UB
                    float wScalar = wLocal.GetValue(cIdx);

                    if (first) {
                        // y = x * wScalar
                        AscendC::Muls(yLocal, xLocal, wScalar, curLen);
                        first = false;
                    } else {
                        // tmp = x * wScalar; y += tmp
                        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                        AscendC::Muls(tmpLocal, xLocal, wScalar, curLen);
                        AscendC::Add(yLocal, yLocal, tmpLocal, curLen);
                        tmpQueue_.FreeTensor(tmpLocal);
                    }

                    inQueueX_.FreeTensor(xLocal);
                }

                // Store result yLocal -> GM
                const uint64_t yOff = ( (static_cast<uint64_t>(nIdx) * o_ + oIdx) * s_ ) + sBase;
                AscendC::DataCopy(yGm_[yOff], yLocal, curLen);

                outQueueY_.FreeTensor(yLocal);
                inQueueW_.FreeTensor(wLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0}, o_{0};
    uint64_t s_{0};
    uint32_t tile_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w, tiling_data.o,
            tiling_data.tileSpatLen);
    op.Process();
}
