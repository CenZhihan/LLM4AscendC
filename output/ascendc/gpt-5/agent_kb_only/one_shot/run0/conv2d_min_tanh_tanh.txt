kernel_src="""
#include "kernel_operator.h"

// This kernel computes y = tanh(tanh(min_c_out(conv2d(x, weight, bias))))
// Assumptions fixed: stride=1, padding=0, dilation=1.
// Tiling strategy: process 1 output element (tileW=1) per inner loop for simplicity.

class KernelConv2dMinTanhTanhCustom {
public:
    __aicore__ inline KernelConv2dMinTanhTanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t HOut, uint32_t WOut, uint32_t tileW, uint32_t hasBias)
    {
        N_ = N; Cin_ = Cin; H_ = H; W_ = W;
        Cout_ = Cout; Kh_ = Kh; Kw_ = Kw;
        HOut_ = HOut; WOut_ = WOut;
        tileW_ = tileW;
        hasBias_ = hasBias != 0;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_*Cin_*H_*W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, Cout_*Cin_*Kh_*Kw_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_*1u*HOut_*WOut_);

        // We only need small scalar buffers since tileW==1.
        // Use VECCALC queue as working local buffers.
        pipe_.InitBuffer(calcQue_, 6, sizeof(float));  // acc, tmp, minv, xScal, wScal, mid
        pipe_.InitBuffer(outQue_, 1, sizeof(float));   // out scalar
    }

    __aicore__ inline void Process()
    {
        if (HOut_ == 0 || WOut_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalRows = N_ * HOut_;
        // Grid-stride over rows
        for (uint32_t row = blockIdx; row < totalRows; row += blockNum) {
            const uint32_t n = row / HOut_;
            const uint32_t ho = row % HOut_;

            for (uint32_t wo = 0; wo < WOut_; wo += tileW_) {
                ComputeOneTile(n, ho, wo);
            }
        }
    }

private:
    __aicore__ inline void ComputeOneTile(uint32_t n, uint32_t ho, uint32_t wo)
    {
        // Local buffers (length=1)
        AscendC::LocalTensor<float> acc  = calcQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp  = calcQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> minv = calcQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> xScal = calcQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> wScal = calcQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> mid  = calcQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLoc = outQue_.AllocTensor<float>();

        bool firstOc = true;

        for (uint32_t oc = 0; oc < Cout_; ++oc) {
            bool firstTerm = true;

            for (uint32_t c = 0; c < Cin_; ++c) {
                for (uint32_t ky = 0; ky < Kh_; ++ky) {
                    for (uint32_t kx = 0; kx < Kw_; ++kx) {
                        const uint32_t hi = ho + ky;
                        const uint32_t wi = wo + kx;

                        // x offset: (((n * Cin + c) * H) + hi) * W + wi
                        uint32_t xOff = ((((n * Cin_) + c) * H_) + hi) * W_ + wi;
                        AscendC::DataCopy(xScal, xGm_[xOff], 1);

                        // w offset: ((((oc) * Cin + c) * Kh) + ky) * Kw + kx
                        uint32_t wOff = ((((oc * Cin_) + c) * Kh_) + ky) * Kw_ + kx;
                        AscendC::DataCopy(wScal, wGm_[wOff], 1);

                        // tmp = x * w (length=1)
                        AscendC::Mul(tmp, xScal, wScal, 1);

                        if (firstTerm) {
                            // acc = tmp
                            AscendC::DataCopy(acc, tmp, 1);
                            firstTerm = false;
                        } else {
                            // acc += tmp
                            AscendC::Add(acc, acc, tmp, 1);
                        }
                    }
                }
            }

            // Add bias if present: acc += b[oc]
            if (hasBias_) {
                AscendC::LocalTensor<float> bLoc = calcQue_.AllocTensor<float>();
                AscendC::DataCopy(bLoc, bGm_[oc], 1);
                AscendC::Add(acc, acc, bLoc, 1);
                calcQue_.FreeTensor(bLoc);
            }

            if (firstOc) {
                // minv = acc
                AscendC::DataCopy(minv, acc, 1);
                firstOc = false;
            } else {
                // minv = element-wise min(minv, acc)
                AscendC::Min(minv, minv, acc, 1);
            }
        }

        // y = tanh(tanh(minv))
        AscendC::Tanh(mid, minv, 1);
        AscendC::Tanh(yLoc, mid, 1);

        // Store to GM: y offset: (((n * 1) + 0) * HOut + ho) * WOut + wo
        uint32_t yOff = ((((n * 1u) + 0u) * HOut_) + ho) * WOut_ + wo;
        AscendC::DataCopy(yGm_[yOff], yLoc, 1);

        outQue_.FreeTensor(yLoc);
        calcQue_.FreeTensor(mid);
        calcQue_.FreeTensor(wScal);
        calcQue_.FreeTensor(xScal);
        calcQue_.FreeTensor(minv);
        calcQue_.FreeTensor(tmp);
        calcQue_.FreeTensor(acc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 6> calcQue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, Kh_{0}, Kw_{0};
    uint32_t HOut_{0}, WOut_{0};
    uint32_t tileW_{1};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanhCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.H, tiling_data.W,
            tiling_data.Cout, tiling_data.Kh, tiling_data.Kw,
            tiling_data.HOut, tiling_data.WOut, tiling_data.tileW, tiling_data.hasBias);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor allocate_output_like_conv2d_min(const at::Tensor& x,
                                                         const at::Tensor& weight)
{
    // Shapes: x [N,Cin,H,W], weight [Cout,Cin,Kh,Kw]
    auto N = x.size(0);
    auto H = x.size(2);
    auto W = x.size(3);
    auto Kh = weight.size(2);
    auto Kw = weight.size(3);
    auto HOut = H - Kh + 1;
    auto WOut = W - Kw + 1;
    return at::empty({N, 1, HOut, WOut}, x.options());
}

at::Tensor conv2d_min_tanh_tanh_impl_npu(const at::Tensor& x,
                                         const at::Tensor& weight,
                                         c10::optional<at::Tensor> bias_opt)
{
    at::Tensor y = allocate_output_like_conv2d_min(x, weight);
    if (bias_opt.has_value() && bias_opt.value().defined()) {
        const at::Tensor& b = bias_opt.value();
        EXEC_NPU_CMD(aclnnConv2dMinTanhTanhCustom, x, weight, b, y);
    } else {
        // Pass an undefined/empty tensor to represent no bias
        at::Tensor empty_bias;
        EXEC_NPU_CMD(aclnnConv2dMinTanhTanhCustom, x, weight, empty_bias, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_min_tanh_tanh_custom", &conv2d_min_tanh_tanh_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_min_tanh_tanh_custom", &conv2d_min_tanh_tanh_impl_npu,
          "Conv2d + channel-min + tanh + tanh fused (Conv2dMinTanhTanhCustom on NPU)");
