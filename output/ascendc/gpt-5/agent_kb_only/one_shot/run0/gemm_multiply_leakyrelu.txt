static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GemmMultiplyLeakyreluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_PARAM_INVALID;

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));

    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0)); // w: [N, K]

    const uint32_t totalLength = M * N;

    // UB budget for post-processing: inQueueY 2*tile + outQueueY 2*tile + tmp 1*tile = 5 * tile floats
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t ALIGN_ELEMS = 2048;
    static constexpr uint32_t HARD_CAP_ELEMS = 32768;

    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    auto aicNum = ascendcPlatform.GetCoreNumAic();
    auto aivNum = ascendcPlatform.GetCoreNumAiv();

    const uint32_t numTiles = (tileLen == 0u) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = ascendcPlatform.CalcTschBlockDim(numTiles, aicNum, aivNum);
    if (blockDim == 0u) blockDim = 1u;
    context->SetBlockDim(blockDim);

    TCubeTiling &ct = tiling.get_cubeTiling();
    // Zero initialize
    uint8_t *ctBytes = reinterpret_cast<uint8_t*>(&ct);

    ct.singleCoreK = K;
    ct.singleCoreN = N;
    ct.ALayoutInfoB = 1;   // batch A
    ct.BLayoutInfoB = 1;   // batch B
    ct.CLayoutInfoB = 1;   // batch C
    ct.isBias = 1;         // enable bias add
    ct.shareMode = 0;
    ct.shareL1Size = 512 * 1024;
    ct.shareL0CSize = 128 * 1024;
    ct.shareUbSize = 0;

    tiling.set_K(K);
    tiling.set_N(N);

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using AscendC::LocalTensor;
using AscendC::GlobalTensor;
using AscendC::TPipe;

constexpr int32_t VEC_BUFFER_NUM = 2;

struct PostTiling {
    uint32_t M;
    uint32_t K;
    uint32_t N;
    uint32_t totalLength;
    uint32_t tileLength;
    uint32_t blockDim;
    TCubeTiling cubeTiling;
};

class KernelGemmMultiplyLeakyreluCustom {
public:
    __aicore__ inline KernelGemmMultiplyLeakyreluCustom() {}

    __aicore__ inline void Init(GM_ADDR xGM, GM_ADDR wGM, GM_ADDR bGM,
                                GM_ADDR multiplierGM, GM_ADDR negativeSlopeGM,
                                GM_ADDR yGM, const PostTiling &pt)
    {
        M_ = pt.M;
        K_ = pt.K;
        N_ = pt.N;
        totalLen_ = pt.totalLength;
        tileLen_ = pt.tileLength;

        aGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(xGM), M_ * K_);
        bGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(wGM), N_ * K_); // stored as [N, K]
        biasGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bGM), N_);
        yGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(yGM), M_ * N_);
        multiplierGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(multiplierGM), 1);
        negativeSlopeGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(negativeSlopeGM), 1);

        pipe_.InitBuffer(inQueuePostIn_, VEC_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueuePostOut_, VEC_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));

        cubeTiling_ = pt.cubeTiling;
    }

    __aicore__ inline void Process(GM_ADDR workspaceGM)
    {
        // 1) Matmul: y = x @ w^T + b
        MatmulCompute(workspaceGM);

        // 2) Load scalar parameters
        float localMul = 1.0f;
        float localNs = 0.0f;
        {
            LocalTensor<float> mulLocal = inQueuePostIn_.AllocTensor<float>();
            AscendC::DataCopy(mulLocal, multiplierGlobal_, 1);
            localMul = mulLocal.GetValue(0);
            inQueuePostIn_.FreeTensor(mulLocal);

            LocalTensor<float> nsLocal = inQueuePostIn_.AllocTensor<float>();
            AscendC::DataCopy(nsLocal, negativeSlopeGlobal_, 1);
            localNs = nsLocal.GetValue(0);
            inQueuePostIn_.FreeTensor(nsLocal);
        }
        negCorr_ = localNs - 1.0f;

        // 3) Post-process: y = LeakyReLU(y * multiplier, negative_slope)
        PostProcess(localMul);
    }

private:
    __aicore__ inline void MatmulCompute(GM_ADDR workspaceGM)
    {
        // Types: all float, NORMAL layout
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> aType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL> bType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> cType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

        set_atomic_none();

        // Set GM tensors with exact sizes
        GlobalTensor<float> gm_a;
        gm_a.SetGlobalBuffer(const_cast<__gm__ float*>(aGlobal_.GetPhyAddr()), M_ * K_);
        GlobalTensor<float> gm_b;
        gm_b.SetGlobalBuffer(const_cast<__gm__ float*>(bGlobal_.GetPhyAddr()), K_ * N_);
        GlobalTensor<float> gm_c;
        gm_c.SetGlobalBuffer(const_cast__gm__ float*>(yGlobal_.GetPhyAddr()), M_ * N_);
        GlobalTensor<float> gm_bias;
        gm_bias.SetGlobalBuffer(const_cast<__gm__ float*>(biasGlobal_.GetPhyAddr()), N_);

        constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
        matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;

        TPipe cubePipe;
        g_cubeTPipePtr = &cubePipe;
        REGIST_MATMUL_OBJ(&cubePipe, GetSysWorkSpacePtr(), mm1);
        mm1.Init(&cubeTiling_);
        mm1.SetTensorA(gm_a, false);
        mm1.SetTensorB(gm_b, true); // w is [N,K], transpose to [K,N]
        mm1.SetWorkspace(workspaceGM, 0);
        mm1.SetBias(gm_bias);
        // Single batch multiply
        int32_t batchA = 1, batchB = 1;
        mm1.IterateBatch(gm_c, batchA, batchB, false);
    }

    __aicore__ inline void PostProcess(float multiplier)
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyInY(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyInY(nextBase, nextLen);
            }

            ComputeTile(curLen, multiplier);
            CopyOutY(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

    __aicore__ inline void CopyInY(uint32_t gmOffset, uint32_t len)
    {
        LocalTensor<float> inLocal = inQueuePostIn_.AllocTensor<float>();
        AscendC::DataCopy(inLocal, yGlobal_[gmOffset], len);
        inQueuePostIn_.EnQue(inLocal);
    }

    __aicore__ inline void ComputeTile(uint32_t len, float multiplier)
    {
        LocalTensor<float> inLocal = inQueuePostIn_.DeQue<float>();
        LocalTensor<float> outLocal = outQueuePostOut_.AllocTensor<float>();
        LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

        // scale: z = in * multiplier
        AscendC::Muls(outLocal, inLocal, multiplier, len);

        // LeakyReLU: y = z + (ns - 1) * min(z, 0)
        AscendC::Mins(tmpLocal, outLocal, 0.0f, len);
        AscendC::Muls(tmpLocal, tmpLocal, negCorr_, len);
        AscendC::Add(outLocal, outLocal, tmpLocal, len);

        tmpQueue_.FreeTensor(tmpLocal);
        outQueuePostOut_.EnQue<float>(outLocal);
        inQueuePostIn_.FreeTensor(inLocal);
    }

    __aicore__ inline void CopyOutY(uint32_t gmOffset, uint32_t len)
    {
        LocalTensor<float> outLocal = outQueuePostOut_.DeQue<float>();
        AscendC::DataCopy(yGlobal_[gmOffset], outLocal, len);
        outQueuePostOut_.FreeTensor(outLocal);
    }

private:
    // GM tensors
    GlobalTensor<float> aGlobal_;
    GlobalTensor<float> bGlobal_;
    GlobalTensor<float> biasGlobal_;
    GlobalTensor<float> yGlobal_;
    GlobalTensor<float> multiplierGlobal_;
    GlobalTensor<float> negativeSlopeGlobal_;

    // Vector queues
    TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, VEC_BUFFER_NUM> inQueuePostIn_;
    AscendC::TQue<AscendC::TPosition::VECOUT, VEC_BUFFER_NUM> outQueuePostOut_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    // Shapes and tiling
    uint32_t M_{0};
    uint32_t K_{0};
    uint32_t N_{0};
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    float negCorr_{0.0f};
    TCubeTiling cubeTiling_;
};

extern "C" __global__ __aicore__ void gemm_multiply_leakyrelu_custom(GM_ADDR xGM, GM_ADDR wGM, GM_ADDR bGM,
                                                                      GM_ADDR multiplierGM, GM_ADDR negativeSlopeGM,
                                                                      GM_ADDR yGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    GET_TILING_DATA(tilingData, tilingGM);
    PostTiling pt;
    pt.M = tilingData.M;
    pt.K = tilingData.K;
    pt.N = tilingData.N;
    pt.totalLength = tilingData.totalLength;
    pt.tileLength = tilingData.tileLength;
    pt.blockDim = tilingData.blockDim;
    pt.cubeTiling = tilingData.cubeTiling;

    KernelGemmMultiplyLeakyreluCustom op;
    op.Init(xGM, wGM, bGM, multiplierGM, negativeSlopeGM, yGM, pt);
    op.Process(workspaceGM);
}
