host_tiling_src="""
#include <cstring>
#include "register/tilingdata_base.h"
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"

namespace optiling {

static inline int64_t GetDim(const gert::Shape* shape, int idx) {
    return static_cast<int64_t>(shape->GetDim(idx));
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    auto *A_shape = context->GetInputShape(0);
    auto *B_shape = context->GetInputShape(1);

    if (A_shape->GetOriginShape().GetDimNum() != 2 || B_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const int64_t A_dim0 = GetDim(&A_shape->GetOriginShape(), 0); // K
    const int64_t A_dim1 = GetDim(&A_shape->GetOriginShape(), 1); // M
    const int64_t B_dim0 = GetDim(&B_shape->GetOriginShape(), 0); // N
    const int64_t B_dim1 = GetDim(&B_shape->GetOriginShape(), 1); // K

    if (A_dim0 <= 0 || A_dim1 <= 0 || B_dim0 <= 0 || B_dim1 <= 0) {
        return ge::GRAPH_FAILED;
    }
    if (A_dim0 != B_dim1) {
        return ge::GRAPH_FAILED;
    }

    const int32_t M = static_cast<int32_t>(A_dim1);
    const int32_t K = static_cast<int32_t>(A_dim0);
    const int32_t N = static_cast<int32_t>(B_dim0);

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);

    tiling.SetDim(1);
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);

    tiling.SetShape(M, N, K);
    tiling.SetOrgShape(M, N, K);

    tiling.SetBias(false);
    tiling.SetBatchInfoForNormal(1, 1, M, N, K);
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    int32_t coreNum = tiling.GetCoreNum();
    if (coreNum <= 0) coreNum = 1;
    context->SetBlockDim(static_cast<uint32_t>(coreNum));

    std::memcpy(context->GetRawTilingData()->GetData(), &tilingData, sizeof(tilingData));
    context->GetRawTilingData()->SetDataSize(sizeof(tilingData));

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *A_shape = context->GetInputShape(0);
    const gert::Shape *B_shape = context->GetInputShape(1);

    if (A_shape->GetOriginShape().GetDimNum() != 2 || B_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }

    const int64_t M = A_shape->GetOriginShape().GetDim(1);
    const int64_t N = B_shape->GetOriginShape().GetDim(0);

    gert::Shape *C_shape = context->GetOutputShape(0);
    C_shape->SetDimNum(2);
    C_shape->SetDim(0, M);
    C_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Inputs: float16, float16; Output: float32
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

} // namespace ge

#include "register/op_def_registry.h"
namespace ops {
class MatmulWithTransposedBothCustom : public OpDef {
public:
    explicit MatmulWithTransposedBothCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(MatmulWithTransposedBothCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR aGM, GM_ADDR bGM,
                                                                          GM_ADDR cGM, GM_ADDR workspaceGM,
                                                                          GM_ADDR tilingGM)
{
    // Define MatmulType for NORMAL layout: A and B inputs in GM, C output in GM.
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, true,  LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    // Load tiling
    TCubeTiling tiling;
    auto tgm = (__gm__ uint32_t*)tilingGM;
    auto tl  = (uint32_t*)&tiling;
    for (int i = 0; i < static_cast<int>(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tgm, ++tl) {
        *tl = *tgm;
    }

    // Initialize GM buffers using original shapes from tiling: A in (K, M), B in (N, K), C in (M, N)
    AscendC::GlobalTensor<half> aGlobal;
    AscendC::GlobalTensor<half> bGlobal;
    AscendC::GlobalTensor<float> cGlobal;

    int32_t sizeA_bytes = tiling.Ka * tiling.M * static_cast<int32_t>(sizeof(half));
    int32_t sizeB_bytes = tiling.N  * tiling.Kb * static_cast<int32_t>(sizeof(half));
    int32_t sizeC_bytes = tiling.M  * tiling.N  * static_cast<int32_t>(sizeof(float));

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(aGM), sizeA_bytes);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(bGM), sizeB_bytes);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cGM), sizeC_bytes);

    // Set reserved share fields conservatively
    tiling.shareMode  = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize = 0;

    // Slice views for single core computation
    AscendC::GlobalTensor<half> gm_a;
    AscendC::GlobalTensor<half> gm_b;
    AscendC::GlobalTensor<float> gm_c;

    gm_a.SetGlobalBuffer(const_cast<__gm__ half*>(aGlobal[0].GetPhyAddr()), tiling.singleCoreM * tiling.singleCoreK);
    gm_b.SetGlobalBuffer(const_cast<__gm__ half*>(bGlobal[0].GetPhyAddr()), tiling.singleCoreK * tiling.singleCoreN);
    gm_c.SetGlobalBuffer(const_cast<__gm__ float*>(cGlobal[0].GetPhyAddr()), tiling.singleCoreM * tiling.singleCoreN);

    // Create and initialize Matmul instance
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;

    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);
    mm1.Init(&tiling);

    // Set tensors with both inputs transposed: A is (K,M) -> treated as A^T (M,K), B is (N,K) -> treated as B^T (K,N)
    const uint32_t isTransposeA = 1U;
    const uint32_t isTransposeB = 1U;
    mm1.SetTensorA(gm_a, isTransposeA);
    mm1.SetTensorB(gm_b, isTransposeB);
    mm1.SetWorkspace(workspaceGM, 0);

    // Single-batch compute
    const uint32_t batchA = 1U;
    const uint32_t batchB = 1U;
    const bool enSequentialWrite = false;
    mm1.IterateBatch(gm_c, batchA, batchB, enSequentialWrite);
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor matmul_with_transposed_both_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Expected input shapes: A[K, M], B[N, K]; kernel computes (A^T) @ (B^T) -> [M, N]
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, \"Inputs must be 2D\");
    TORCH_CHECK(A.size(0) == B.size(1), \"K dimension mismatch\");
    const int64_t M = A.size(1);
    const int64_t N = B.size(0);

    at::Tensor Ah = A.to(at::kHalf);
    at::Tensor Bh = B.to(at::kHalf);
    at::Tensor C = at::empty({M, N}, Ah.options().dtype(at::kFloat)); // output float32

    EXEC_NPU_CMD(aclnnMatmulWithTransposedBothCustom, Ah, Bh, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"matmul_with_transposed_both_custom\", &matmul_with_transposed_both_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"matmul_with_transposed_both_custom\", &matmul_with_transposed_both_impl_npu,
          \"Matmul with both inputs transposed (MatmulWithTransposedBothCustom on NPU)\");
}
