#include "conv_standard_1d_dilated_strided_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative
static constexpr uint32_t ALIGN_ELEMS = 256;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard1dDilatedStridedCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0)->GetOriginShapePtr();
    const gert::Shape *w_shape = context->GetInputShape(1)->GetOriginShapePtr();

    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    uint32_t n    = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t cin  = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t lin  = static_cast<uint32_t>(x_shape->GetDim(2));

    uint32_t cout = static_cast<uint32_t>(w_shape->GetDim(0));
    uint32_t cinw = static_cast<uint32_t>(w_shape->GetDim(1));
    uint32_t k    = static_cast<uint32_t>(w_shape->GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stride_p   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *dilation_p = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding_p  = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *groups_p   = attrs->GetAttrPointer<int32_t>(3);

    uint32_t stride   = static_cast<uint32_t>(*stride_p);
    uint32_t dilation = static_cast<uint32_t>(*dilation_p);
    uint32_t padding  = static_cast<uint32_t>(*padding_p);
    uint32_t groups   = static_cast<uint32_t>(*groups_p);

    // Basic validation (groups==1 supported)
    if (groups != 1 || cinw * groups != cin) {
        // Fallback to simple assumptions while keeping shape correctness
        groups = 1;
    }

    // Compute output length (same as PyTorch conv1d)
    // lout = floor((lin + 2*padding - dilation*(k-1) - 1) / stride) + 1
    int64_t effective = static_cast<int64_t>(dilation) * static_cast<int64_t>(k - 1) + 1;
    int64_t temp = static_cast<int64_t>(lin) + 2LL * static_cast<int64_t>(padding) - effective;
    int64_t lout64 = (temp >= 0) ? (temp / static_cast<int64_t>(stride) + 1) : 0;
    uint32_t lout = static_cast<uint32_t>(lout64);

    // Tile only along time dimension (output length)
    const uint32_t weightBytes = cin * k * sizeof(float);
    uint32_t maxTileElems = 0;
    if (UB_BUDGET_BYTES > (weightBytes + 1024)) {
        maxTileElems = (UB_BUDGET_BYTES - weightBytes - 1024) / sizeof(float);
    } else {
        maxTileElems = 1024 / sizeof(float);
    }
    maxTileElems = std::max<uint32_t>(ALIGN_ELEMS, AlignDown(maxTileElems, ALIGN_ELEMS));
    uint32_t tileT = (lout > 0) ? std::min<uint32_t>(lout, maxTileElems) : ALIGN_ELEMS;
    tileT = (tileT == 0) ? ALIGN_ELEMS : tileT;

    // Parallelize across batch dimension
    uint32_t blockDim = n;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_cin(cin);
    tiling.set_lin(lin);
    tiling.set_cout(cout);
    tiling.set_k(k);
    tiling.set_lout(lout);
    tiling.set_tileT(tileT);
    tiling.set_blockDim(blockDim);
    tiling.set_stride(stride);
    tiling.set_dilation(dilation);
    tiling.set_padding(padding);
    tiling.set_groups(groups);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    int64_t n    = x_shape->GetDim(0);
    int64_t cin  = x_shape->GetDim(1);
    int64_t lin  = x_shape->GetDim(2);
    int64_t cout = w_shape->GetDim(0);
    int64_t k    = w_shape->GetDim(2);

    int64_t stride   = context->GetAttrs()->GetAttrValue<int32_t>(0);
    int64_t dilation = context->GetAttrs()->GetAttrValue<int32_t>(1);
    int64_t padding  = context->GetAttrs()->GetAttrValue<int32_t>(2);
    int64_t groups   = context->GetAttrs()->GetAttrValue<int32_t>(3);
    (void)groups; // groups=1 expected

    int64_t effective = dilation * (k - 1) + 1;
    int64_t temp = lin + 2 * padding - effective;
    int64_t lout = (temp >= 0) ? (temp / stride + 1) : 0;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, cout);
    y_shape->SetDim(2, lout);
    return GRAPH_SUCCESS;
}

    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelConvStandard1dDilatedStridedCustom {
public:
    __aicore__ inline KernelConvStandard1dDilatedStridedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t lin,
                                uint32_t cout, uint32_t k, uint32_t lout,
                                uint32_t tileT,
                                uint32_t stride, uint32_t dilation, uint32_t padding, uint32_t groups)
    {
        (void)groups; // groups=1 supported
        n_ = n; cin_ = cin; lin_ = lin;
        cout_ = cout; k_ = k; lout_ = lout;
        tileT_ = tileT; stride_ = stride; dilation_ = dilation; padding_ = padding;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<int32_t>(n_ * cin_ * lin_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<int32_t>(cout_ * cin_ * k_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<int32_t>(n_ * cout_ * lout_));

        // Buffers: y-tile and calc buffer (weights + 1 scalar slot)
        pipe_.InitBuffer(outQueueY_, 1, tileT_ * sizeof(float));
        uint32_t calcFloats = cin_ * k_ + 1;
        pipe_.InitBuffer(calcBufs_, calcFloats * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || lin_ == 0 || cout_ == 0 || k_ == 0 || lout_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t nIdx = blockIdx; nIdx < n_; nIdx += blockNum) {
            for (uint32_t tBase = 0; tBase < lout_; tBase += tileT_) {
                const uint32_t curTile = AscendC::Min(tileT_, lout_ - tBase);

                for (uint32_t co = 0; co < cout_; ++co) {
                    // Prepare buffers
                    AscendC::LocalTensor<float> calcLocal = calcBufs_.Get<float>();
                    calcLocal.SetSize(cin_ * k_ + 1);
                    AscendC::LocalTensor<float> wLocal = calcLocal;
                    wLocal.SetSize(cin_ * k_);
                    AscendC::LocalTensor<float> xScalar = calcLocal[cin_ * k_];
                    xScalar.SetSize(1);

                    // Load weights for this output channel
                    const uint32_t wOff = co * (cin_ * k_);
                    AscendC::DataCopy(wLocal, wGm_[wOff], cin_ * k_);

                    AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                    yLocal.SetSize(curTile);
                    // Initialize output tile to 0
                    for (uint32_t i = 0; i < curTile; ++i) {
                        yLocal.SetValue(i, 0.0f);
                    }

                    // Accumulate over input channels and kernel taps
                    for (uint32_t ci = 0; ci < cin_; ++ci) {
                        const uint32_t wCiBase = ci * k_;
                        for (uint32_t kk = 0; kk < k_; ++kk) {
                            const float wval = wLocal.GetValue(wCiBase + kk);
                            for (uint32_t i = 0; i < curTile; ++i) {
                                const uint32_t tOut = tBase + i;
                                // x index along length: tOut*stride - padding + kk*dilation
                                int64_t xPos = static_cast<int64_t>(tOut) * static_cast<int64_t>(stride_) -
                                               static_cast<int64_t>(padding_) +
                                               static_cast<int64_t>(kk) * static_cast<int64_t>(dilation_);
                                if (xPos < 0 || xPos >= static_cast<int64_t>(lin_)) {
                                    continue; // out of bounds due to padding
                                }
                                const uint32_t xIdx = static_cast<uint32_t>(xPos);
                                const uint32_t xOff = ((nIdx * cin_ + ci) * lin_) + xIdx;

                                AscendC::DataCopy(xScalar, xGm_[xOff], 1);
                                const float xval = xScalar.GetValue(0);
                                const float acc = yLocal.GetValue(i) + xval * wval;
                                yLocal.SetValue(i, acc);
                            }
                        }
                    }

                    // Write back output tile for this (nIdx, co)
                    const uint32_t yOff = ((nIdx * cout_ + co) * lout_) + tBase;
                    AscendC::DataCopy(yGm_[yOff], yLocal, curTile);
                    outQueueY_.FreeTensor(yLocal);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBufs_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, lin_{0}, cout_{0}, k_{0}, lout_{0}, tileT_{0};
    uint32_t stride_{1}, dilation_{1}, padding_{0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dDilatedStridedCustom op;
    op.Init(
        x, w, y,
        tiling_data.n, tiling_data.cin, tiling_data.lin,
        tiling_data.cout, tiling_data.k, tiling_data.lout,
        tiling_data.tileT,
        tiling_data.stride, tiling_data.dilation, tiling_data.padding, tiling_data.groups
    );
    op.Process();
}
