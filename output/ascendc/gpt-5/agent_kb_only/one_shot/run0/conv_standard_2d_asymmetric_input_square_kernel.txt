#include "conv_standard_2d_asymmetric_input_square_kernel_custom_tiling.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// UB budget for vectors (bytes)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// We store per-oc working buffers: yAcc(tileW), xRow(tileW), tmp(tileW) (floats) and K weights (floats).
// bytes ~= 4 * (3 * tileW + K)
static constexpr uint32_t TILE_ALIGN = 128;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    const int64_t n64  = xShape->GetOriginShape().GetDim(0);
    const int64_t c64  = xShape->GetOriginShape().GetDim(1);
    const int64_t h64  = xShape->GetOriginShape().GetDim(2);
    const int64_t w64  = xShape->GetOriginShape().GetDim(3);
    const int64_t cout64 = wShape->GetOriginShape().GetDim(0);
    const int64_t c_w64  = wShape->GetOriginShape().GetDim(1);
    const int64_t kh64 = wShape->GetOriginShape().GetDim(2);
    const int64_t kw64 = wShape->GetOriginShape().GetDim(3);

    const uint32_t n  = static_cast<uint32_t>(n64);
    const uint32_t c  = static_cast<uint32_t>(c64);
    const uint32_t h  = static_cast<uint32_t>(h64);
    const uint32_t wv = static_cast<uint32_t>(w64);
    const uint32_t cout = static_cast<uint32_t>(cout64);
    const uint32_t cw = static_cast<uint32_t>(c_w64);
    const uint32_t kh = static_cast<uint32_t>(kh64);
    const uint32_t kw = static_cast<uint32_t>(kw64);

    // Basic checks for expected shapes: groups=1, stride=1, padding=0, dilation=1
    if (cw != c) {
        return ge::GRAPH_FAILED;
    }
    if (kh == 0 || kw == 0 || h < kh || wv < kw) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t hout = h - kh + 1;
    const uint32_t wout = wv - kw + 1;

    // Reduction dim K
    const uint32_t K = c * kh * kw;

    // Compute tileW from UB budget: 4 bytes * (3*tileW + K) <= UB_BUDGET_BYTES
    uint32_t maxTileW = 0;
    if (UB_BUDGET_BYTES > 4 * K) {
        maxTileW = (UB_BUDGET_BYTES / 4 - K) / 3;
    }
    if (maxTileW < 1) maxTileW = 1;
    maxTileW = AlignDown(maxTileW, TILE_ALIGN);
    if (maxTileW < TILE_ALIGN) maxTileW = TILE_ALIGN;

    uint32_t tileW = maxTileW;
    if (wout < tileW) {
        uint32_t aligned = AlignDown(wout, TILE_ALIGN);
        tileW = (aligned > 0) ? aligned : wout;
    }

    // Decide blockDim and ocPerBlock: split on Cout
    uint32_t blockDim = cout;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim < 1u) blockDim = 1u;
    uint32_t ocPerBlock = CeilDiv(cout, blockDim);

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(wv);
    tiling.set_cout(cout);
    tiling.set_kh(kh);
    tiling.set_kw(kw);
    tiling.set_hout(hout);
    tiling.set_wout(wout);
    tiling.set_tileW(tileW);
    tiling.set_ocPerBlock(ocPerBlock);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConv2dS1P0 {
public:
    __aicore__ inline KernelConv2dS1P0() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t win,
                                uint32_t cout, uint32_t kh, uint32_t kw,
                                uint32_t hout, uint32_t wout,
                                uint32_t tileW, uint32_t ocPerBlock)
    {
        n_ = n; c_ = c; h_ = h; win_ = win;
        cout_ = cout; kh_ = kh; kw_ = kw;
        hout_ = hout; wout_ = wout;
        tileW_ = tileW; ocPerBlock_ = ocPerBlock;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)w);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Buffers:
        // xRow: tileW
        // yAcc: tileW
        // tmp:  tileW
        // wBlock: K = c*kh*kw
        K_ = c_ * kh_ * kw_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(calcQueue_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, K_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (cout_ == 0 || n_ == 0 || hout_ == 0 || wout_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t ocStart = blockIdx * ocPerBlock_;
        if (ocStart >= cout_) return;
        const uint32_t ocEnd = AscendC::Min(ocStart + ocPerBlock_, cout_);

        // Strides for linear indexing (NCHW)
        const uint64_t xStrideN = static_cast<uint64_t>(c_) * h_ * win_;
        const uint64_t xStrideC = static_cast<uint64_t>(h_) * win_;
        const uint64_t xStrideH = static_cast<uint64_t>(win_);
        const uint64_t xStrideW = 1ull;

        const uint64_t yStrideN = static_cast<uint64_t>(cout_) * hout_ * wout_;
        const uint64_t yStrideC = static_cast<uint64_t>(hout_) * wout_;
        const uint64_t yStrideH = static_cast<uint64_t>(wout_);
        const uint64_t yStrideW = 1ull;

        for (uint32_t n = 0; n < n_; ++n) {
            for (uint32_t oh = 0; oh < hout_; ++oh) {
                for (uint32_t oc = ocStart; oc < ocEnd; ++oc) {
                    // Load full weight block for this oc: size = K_
                    AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                    const uint64_t wOffset = static_cast<uint64_t>(oc) * K_;
                    AscendC::DataCopy(wLocal, wGm_[wOffset], K_);
                    // Keep wLocal alive for this oc
                    for (uint32_t owBase = 0; owBase < wout_; owBase += tileW_) {
                        uint32_t curW = wout_ - owBase;
                        if (curW > tileW_) curW = tileW_;

                        // Allocate y accumulator and zero it
                        AscendC::LocalTensor<float> yAcc = outQueueY_.AllocTensor<float>();
                        float zeroVal = 0.0f;
                        AscendC::Duplicate(yAcc, zeroVal, curW);

                        // Reduction over k = c*kh*kw
                        for (uint32_t ic = 0; ic < c_; ++ic) {
                            for (uint32_t r = 0; r < kh_; ++r) {
                                const uint32_t inH = oh + r; // stride=1
                                for (uint32_t s = 0; s < kw_; ++s) {
                                    const uint32_t inWStart = owBase + s; // stride=1
                                    const uint64_t xBaseOfs =
                                        n * xStrideN + ic * xStrideC + inH * xStrideH + inWStart * xStrideW;

                                    // Load input row slice
                                    AscendC::LocalTensor<float> xRow = inQueueX_.AllocTensor<float>();
                                    AscendC::DataCopy(xRow, xGm_[xBaseOfs], curW);

                                    // Multiply and accumulate: yAcc += xRow * w_scalar
                                    AscendC::LocalTensor<float> tmp = calcQueue_.AllocTensor<float>();
                                    const uint32_t kIndex = ic * (kh_ * kw_) + r * kw_ + s;
                                    const float wScalar = wLocal.GetValue(kIndex);
                                    AscendC::Muls(tmp, xRow, wScalar, curW);
                                    AscendC::Add(yAcc, yAcc, tmp, curW);

                                    calcQueue_.FreeTensor(tmp);
                                    inQueueX_.FreeTensor(xRow);
                                }
                            }
                        }

                        // Store result back to GM
                        const uint64_t yBaseOfs =
                            n * yStrideN + oc * yStrideC + oh * yStrideH + static_cast<uint64_t>(owBase) * yStrideW;
                        AscendC::DataCopy(yGm_[yBaseOfs], yAcc, curW);
                        outQueueY_.FreeTensor(yAcc);
                    }
                    wQueue_.FreeTensor(wLocal);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> calcQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, win_{0};
    uint32_t cout_{0}, kh_{0}, kw_{0};
    uint32_t hout_{0}, wout_{0};
    uint32_t tileW_{0}, ocPerBlock_{1};
    uint32_t K_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dS1P0 op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.cout, tiling_data.kh, tiling_data.kw,
            tiling_data.hout, tiling_data.wout,
            tiling_data.tileW, tiling_data.ocPerBlock);
    op.Process();
}

at::Tensor conv2d_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    // Output shape: N, Cout, Hout, Wout with stride=1, pad=0, dilation=1
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, \"Inputs must be 4D tensors\");
    int64_t N = x.size(0);
    int64_t C = x.size(1);
    int64_t H = x.size(2);
    int64_t W = x.size(3);
    int64_t Cout = weight.size(0);
    int64_t Kh = weight.size(2);
    int64_t Kw = weight.size(3);
    TORCH_CHECK(weight.size(1) == C, \"Weight C must match input C\");
    TORCH_CHECK(H >= Kh && W >= Kw, \"Invalid kernel size for input\");
    int64_t Hout = H - Kh + 1;
    int64_t Wout = W - Kw + 1;

    at::Tensor result = at::empty({N, Cout, Hout, Wout}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard2dAsymmetricInputSquareKernelCustom, x, weight, result);
    return result;
