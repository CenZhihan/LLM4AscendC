kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelSqueezeNetFireModuleCustom {
public:
    __aicore__ inline KernelSqueezeNetFireModuleCustom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR y,
        GM_ADDR workspace,
        GM_ADDR wsq, GM_ADDR bsq,
        GM_ADDR wex1, GM_ADDR bex1,
        GM_ADDR wex3, GM_ADDR bex3,
        uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
        uint32_t Cs, uint32_t Ce1, uint32_t Ce3,
        uint32_t wsPerSampleBytes)
    {
        N_ = N; Cin_ = Cin; H_ = H; W_ = W; Cs_ = Cs; Ce1_ = Ce1; Ce3_ = Ce3;
        wsPerSampleFloats_ = wsPerSampleBytes / static_cast<uint32_t>(sizeof(float));
        Cout_ = Ce1_ + Ce3_;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), static_cast<uint64_t>(N_) * Cin_ * H_ * W_);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), static_cast<uint64_t>(N_) * Cout_ * H_ * W_);
        wsGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(workspace), static_cast<uint64_t>(N_) * wsPerSampleFloats_);

        wsqGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(wsq), static_cast<uint64_t>(Cs_) * Cin_);
        bsqGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bsq), Cs_);
        wex1Gm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(wex1), static_cast<uint64_t>(Ce1_) * Cs_);
        bex1Gm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bex1), Ce1_);
        wex3Gm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(wex3), static_cast<uint64_t>(Ce3_) * Cs_ * 9);
        bex3Gm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bex3), Ce3_);

        // Allocate UB buffers
        // Row-sized buffers
        pipe_.InitBuffer(qRow_, 1, W_ * sizeof(float));
        pipe_.InitBuffer(qTemp_, 1, W_ * sizeof(float));
        // Squeeze row buffer: Cs * W
        pipe_.InitBuffer(qSRow_, 1, Cs_ * W_ * sizeof(float));
        // Y tile buffer for expand parts (max 8 channels at once)
        ocTile1_ = 8;
        ocTile3_ = 8;
        if (ocTile1_ > Ce1_) ocTile1_ = Ce1_;
        if (ocTile3_ > Ce3_) ocTile3_ = Ce3_;
        pipe_.InitBuffer(qYTile_, 1, static_cast<uint32_t>(Max(ocTile1_, ocTile3_) * W_) * sizeof(float));
        // Shift buffer for 3x3
        pipe_.InitBuffer(qShift_, 1, W_ * sizeof(float));
        // Small parameter buffers
        pipe_.InitBuffer(qSmall_, 1, Max((uint32_t)(Cs_*Cin_), (uint32_t)(ocTile3_*Cs_*9)) * sizeof(float));
        pipe_.InitBuffer(qBias_, 1, Max(Max(Cs_, Ce1_), Ce3_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());
        if (blockIdx >= N_) return;

        // Preload squeeze weights [Cs * Cin] and bias [Cs]
        LocalTensor<float> wsqLocal = qSmall_.AllocTensor<float>();
        LocalTensor<float> bsqLocal = qBias_.AllocTensor<float>();
        DataCopy(wsqLocal, wsqGm_, Cs_ * Cin_);
        DataCopy(bsqLocal, bsqGm_, Cs_);

        // Stage 1: squeeze 1x1 + ReLU -> write to workspace
        for (uint32_t h = 0; h < H_; ++h) {
            // ySRow holds [Cs, W]
            LocalTensor<float> ySRow = qSRow_.AllocTensor<float>();
            // Initialize with bias
            for (uint32_t oc = 0; oc < Cs_; ++oc) {
                LocalTensor<float> seg = ySRow[oc * W_];
                float bval = bsqLocal.GetValue(oc);
                Duplicate(seg, bval, W_);
            }

            for (uint32_t k = 0; k < Cin_; ++k) {
                // Load x row for (b, k, h, :)
                LocalTensor<float> xRow = qRow_.AllocTensor<float>();
                const uint64_t xBase = ((((uint64_t)blockIdx * Cin_ + k) * H_ + h) * W_);
                DataCopy(xRow, xGm_[xBase], W_);
                // Temp row for mul
                LocalTensor<float> tRow = qTemp_.AllocTensor<float>();
                for (uint32_t oc = 0; oc < Cs_; ++oc) {
                    float wval = wsqLocal.GetValue(oc * Cin_ + k);
                    Muls(tRow, xRow, wval, W_);
                    LocalTensor<float> seg = ySRow[oc * W_];
                    Add(seg, seg, tRow, W_);
                }
                qTemp_.FreeTensor(tRow);
                qRow_.FreeTensor(xRow);
            }

            // ReLU and store to workspace
            for (uint32_t oc = 0; oc < Cs_; ++oc) {
                LocalTensor<float> seg = ySRow[oc * W_];
                Maxs(seg, seg, 0.0f, W_);
            }

            // Store squeeze row to workspace at (b, oc, h, :)
            for (uint32_t oc = 0; oc < Cs_; ++oc) {
                const uint64_t wsBaseSample = (uint64_t)blockIdx * wsPerSampleFloats_;
                const uint64_t wsBase = wsBaseSample + ((uint64_t)oc * H_ + h) * W_;
                LocalTensor<float> seg = ySRow[oc * W_];
                DataCopy(wsGm_[wsBase], seg, W_);
            }
            qSRow_.FreeTensor(ySRow);
        }

        // Stage 2: expand1 1x1 + ReLU -> write to y [0..Ce1-1]
        for (uint32_t oc0 = 0; oc0 < Ce1_; oc0 += ocTile1_) {
            const uint32_t tile = (oc0 + ocTile1_ <= Ce1_) ? ocTile1_ : (Ce1_ - oc0);
            LocalTensor<float> wTile = qSmall_.AllocTensor<float>();     // size >= ocTile1 * Cs
            LocalTensor<float> bTile = qBias_.AllocTensor<float>();      // size >= ocTile1
            const uint64_t wOff = (uint64_t)oc0 * Cs_;
            DataCopy(wTile, wex1Gm_[wOff], (uint64_t)tile * Cs_);
            DataCopy(bTile, bex1Gm_[oc0], tile);

            for (uint32_t h = 0; h < H_; ++h) {
                LocalTensor<float> yTile = qYTile_.AllocTensor<float>(); // layout [tile, W]
                // init with bias
                for (uint32_t t = 0; t < tile; ++t) {
                    LocalTensor<float> seg = yTile[t * W_];
                    float bval = bTile.GetValue(t);
                    Duplicate(seg, bval, W_);
                }
                for (uint32_t k = 0; k < Cs_; ++k) {
                    // s row at channel k
                    LocalTensor<float> sRow = qRow_.AllocTensor<float>();
                    const uint64_t wsBaseSample = (uint64_t)blockIdx * wsPerSampleFloats_;
                    const uint64_t sOff = wsBaseSample + ((uint64_t)k * H_ + h) * W_;
                    DataCopy(sRow, wsGm_[sOff], W_);

                    LocalTensor<float> tRow = qTemp_.AllocTensor<float>();
                    for (uint32_t t = 0; t < tile; ++t) {
                        float wval = wTile.GetValue(t * Cs_ + k);
                        Muls(tRow, sRow, wval, W_);
                        LocalTensor<float> seg = yTile[t * W_];
                        Add(seg, seg, tRow, W_);
                    }
                    qTemp_.FreeTensor(tRow);
                    qRow_.FreeTensor(sRow);
                }
                // ReLU and store
                for (uint32_t t = 0; t < tile; ++t) {
                    LocalTensor<float> seg = yTile[t * W_];
                    Maxs(seg, seg, 0.0f, W_);
                    const uint64_t yBase = ((((uint64_t)blockIdx * Cout_ + (oc0 + t)) * H_ + h) * W_);
                    DataCopy(yGm_[yBase], seg, W_);
                }
                qYTile_.FreeTensor(yTile);
            }
            qBias_.FreeTensor(bTile);
            qSmall_.FreeTensor(wTile);
        }

        // Stage 3: expand3 3x3 + ReLU -> write to y [Ce1..Ce1+Ce3-1]
        for (uint32_t oc0 = 0; oc0 < Ce3_; oc0 += ocTile3_) {
            const uint32_t tile = (oc0 + ocTile3_ <= Ce3_) ? ocTile3_ : (Ce3_ - oc0);
            LocalTensor<float> wTile = qSmall_.AllocTensor<float>();  // size >= tile * Cs * 9
            LocalTensor<float> bTile = qBias_.AllocTensor<float>();   // size >= tile
            const uint64_t wOff = (uint64_t)oc0 * Cs_ * 9;
            DataCopy(wTile, wex3Gm_[wOff], (uint64_t)tile * Cs_ * 9);
            DataCopy(bTile, bex3Gm_[oc0], tile);

            for (uint32_t h = 0; h < H_; ++h) {
                LocalTensor<float> yTile = qYTile_.AllocTensor<float>(); // [tile, W]
                // init with bias
                for (uint32_t t = 0; t < tile; ++t) {
                    LocalTensor<float> seg = yTile[t * W_];
                    float bval = bTile.GetValue(t);
                    Duplicate(seg, bval, W_);
                }

                for (int32_t dy = -1; dy <= 1; ++dy) {
                    const int32_t hs = static_cast<int32_t>(h) + dy;
                    const bool rowValid = (hs >= 0 && hs < static_cast<int32_t>(H_));
                    if (!rowValid) {
                        continue; // padding zeros
                    }
                    // Load s row for each input channel k at row hs
                    for (uint32_t k = 0; k < Cs_; ++k) {
                        // sRow
                        LocalTensor<float> sRow = qRow_.AllocTensor<float>();
                        const uint64_t wsBaseSample = (uint64_t)blockIdx * wsPerSampleFloats_;
                        const uint64_t sOff = wsBaseSample + ((uint64_t)k * H_ + (uint32_t)hs) * W_;
                        DataCopy(sRow, wsGm_[sOff], W_);

                        // Build shifted row for each dx
                        for (int32_t dx = -1; dx <= 1; ++dx) {
                            LocalTensor<float> sShift = qShift_.AllocTensor<float>();
                            // fill zeros
                            Duplicate(sShift, 0.0f, W_);
                            if (dx == 0) {
                                DataCopy(sShift, sRow, W_);
                            } else if (dx == -1) {
                                if (W_ > 1) {
                                    // dst[1..W-1] = src[0..W-2]
                                    DataCopy(sShift[1], sRow, W_ - 1);
                                }
                            } else { // dx == +1
                                if (W_ > 1) {
                                    // dst[0..W-2] = src[1..W-1]
                                    DataCopy(sShift, sRow[1], W_ - 1);
                                }
                            }

                            // Accumulate into yTile
                            const uint32_t kxy = (uint32_t)((dy + 1) * 3 + (dx + 1)); // 0..8
                            LocalTensor<float> tRow = qTemp_.AllocTensor<float>();
                            for (uint32_t t = 0; t < tile; ++t) {
                                const uint32_t baseIdx = t * (Cs_ * 9) + k * 9;
                                const float wval = wTile.GetValue(baseIdx + kxy);
                                Muls(tRow, sShift, wval, W_);
                                LocalTensor<float> seg = yTile[t * W_];
                                Add(seg, seg, tRow, W_);
                            }
                            qTemp_.FreeTensor(tRow);
                            qShift_.FreeTensor(sShift);
                        }

                        qRow_.FreeTensor(sRow);
                    } // k
                } // dy

                // ReLU and store
                for (uint32_t t = 0; t < tile; ++t) {
                    LocalTensor<float> seg = yTile[t * W_];
                    Maxs(seg, seg, 0.0f, W_);
                    const uint64_t yBase = ((((uint64_t)blockIdx * Cout_ + (Ce1_ + oc0 + t)) * H_ + h) * W_);
                    DataCopy(yGm_[yBase], seg, W_);
                }

                qYTile_.FreeTensor(yTile);
            } // h

            qBias_.FreeTensor(bTile);
            qSmall_.FreeTensor(wTile);
        }
    }

private:
    template <typename T>
    __aicore__ inline T Max(T a, T b) { return a > b ? a : b; }

private:
    // Global tensors
    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> wsGm_;
    GlobalTensor<float> wsqGm_, bsqGm_;
    GlobalTensor<float> wex1Gm_, bex1Gm_;
    GlobalTensor<float> wex3Gm_, bex3Gm_;

    // Queues and buffers
    TPipe pipe_;
    TQue<QuePosition::VECCALC, 1> qRow_;
    TQue<QuePosition::VECCALC, 1> qTemp_;
    TQue<QuePosition::VECCALC, 1> qSRow_;
    TQue<QuePosition::VECCALC, 1> qYTile_;
    TQue<QuePosition::VECCALC, 1> qShift_;
    TQue<QuePosition::VECCALC, 1> qSmall_;
    TQue<QuePosition::VECCALC, 1> qBias_;

    // Params
    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0};
    uint32_t Cs_{0}, Ce1_{0}, Ce3_{0}, Cout_{0};
    uint32_t wsPerSampleFloats_{0};
    uint32_t ocTile1_{8}, ocTile3_{8};
};
