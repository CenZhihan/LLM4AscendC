class KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Hin, uint32_t Win,
                                uint32_t Kh, uint32_t Kw,
                                uint32_t G,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t outPadH, uint32_t outPadW,
                                uint32_t dilH, uint32_t dilW,
                                uint32_t Hout, uint32_t Wout)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout; Hin_ = Hin; Win_ = Win;
        Kh_ = Kh; Kw_ = Kw; G_ = G;
        sH_ = strideH; sW_ = strideW; pH_ = padH; pW_ = padW;
        oH_ = outPadH; oW_ = outPadW; dH_ = dilH; dW_ = dilW;
        Hout_ = Hout; Wout_ = Wout;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_*Cin_*Hin_*Win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cin_*(Cout_/G_)*Kh_*Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_*Cout_*Hout_*Wout_);

        // Buffers: one scalar for x, one scalar for w, one scalar temp, and one row zeroing
        pipe_.InitBuffer(xScalarQ_, 1, sizeof(float));
        pipe_.InitBuffer(wScalarQ_, 1, sizeof(float));
        pipe_.InitBuffer(tmpScalarQ_, 1, sizeof(float));
        pipe_.InitBuffer(rowBufQ_, 1, Wout_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || Hout_ == 0 || Wout_ == 0) return;

        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint64_t totalRows = (uint64_t)N_ * (uint64_t)Cout_ * (uint64_t)Hout_;

        for (uint64_t row = blockIdx; row < totalRows; row += blockNum) {
            // Decode (n, cOut, hOut)
            uint64_t tmp = row;
            const uint32_t hOut = (uint32_t)(tmp % Hout_); tmp /= Hout_;
            const uint32_t cOut = (uint32_t)(tmp % Cout_); tmp /= Cout_;
            const uint32_t n    = (uint32_t)(tmp);

            // Zero the output row y[n, cOut, hOut, :]
            {
                AscendC::LocalTensor<float> rowLocal = rowBufQ_.AllocTensor<float>();
                AscendC::Duplicate(rowLocal, 0.0f, Wout_);
                const uint64_t yBase = ((((uint64_t)n * Cout_) + cOut) * Hout_ + hOut) * Wout_;
                AscendC::DataCopy(yGm_[yBase], rowLocal, Wout_);
                rowBufQ_.FreeTensor(rowLocal);
            }

            const uint32_t CoutPerG = Cout_ / G_;
            const uint32_t groupId = cOut / CoutPerG;
            const uint32_t cOutG   = cOut % CoutPerG;
            const uint32_t CinPerG = Cin_ / G_;

            // Accumulate contributions into y using atomic add per element
            for (uint32_t cInG = 0; cInG < CinPerG; ++cInG) {
                const uint32_t cIn = groupId * CinPerG + cInG;

                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    // Compute corresponding input h index
                    int64_t hInNum = (int64_t)hOut + (int64_t)pH_ - (int64_t)kh * (int64_t)dH_;
                    if (hInNum < 0) continue;
                    if ((hInNum % (int64_t)sH_) != 0) continue;
                    const int64_t hIn = hInNum / (int64_t)sH_;
                    if (hIn < 0 || hIn >= (int64_t)Hin_) continue;

                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        // Load weight scalar w[cIn, cOutG, kh, kw]
                        const uint64_t wOff = ((((uint64_t)cIn * CoutPerG) + cOutG) * Kh_ + kh) * Kw_ + kw;
                        AscendC::LocalTensor<float> wLocal = wScalarQ_.AllocTensor<float>();
                        AscendC::DataCopy(wLocal, wGm_[wOff], 1);

                        // Iterate input width
                        for (uint32_t wIn = 0; wIn < Win_; ++wIn) {
                            // Compute output w index
                            int64_t wOut = (int64_t)wIn * (int64_t)sW_ - (int64_t)pW_ + (int64_t)kw * (int64_t)dW_;
                            if (wOut < 0 || wOut >= (int64_t)Wout_) continue;

                            // x scalar at [n, cIn, hIn, wIn]
                            const uint64_t xOff = ((((uint64_t)n * Cin_) + cIn) * Hin_ + (uint64_t)hIn) * Win_ + wIn;
                            AscendC::LocalTensor<float> xLocal = xScalarQ_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[xOff], 1);

                            // tmp = x * w
                            AscendC::LocalTensor<float> tmpLocal = tmpScalarQ_.AllocTensor<float>();
                            AscendC::Mul(tmpLocal, xLocal, wLocal, 1);

                            // Atomic add tmp into y[n, cOut, hOut, wOut]
                            const uint64_t yOff = ((((uint64_t)n * Cout_) + cOut) * Hout_ + hOut) * Wout_ + (uint64_t)wOut;
                            AscendC::SetAtomicAdd<float>();
                            AscendC::DataCopy(yGm_[yOff], tmpLocal, 1);
                            AscendC::SetAtomicNone();

                            tmpScalarQ_.FreeTensor(tmpLocal);
                            xScalarQ_.FreeTensor(xLocal);
                        }

                        wScalarQ_.FreeTensor(wLocal);
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xScalarQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wScalarQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpScalarQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> rowBufQ_;

    AscendC::GlobalTensor<float> xGm_, wGm_, yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0}, Hin_{0}, Win_{0};
    uint32_t Kh_{0}, Kw_{0}, G_{1};
    uint32_t sH_{1}, sW_{1}, pH_{0}, pW_{0}, oH_{0}, oW_{0}, dH_{1}, dW_{1};
    uint32_t Hout_{0}, Wout_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
