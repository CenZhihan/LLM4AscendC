static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MlpCustomTilingData t;
    // Attrs
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *inputSize = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *h1       = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *h2       = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *outSize  = attrs->GetAttrPointer<int32_t>(3);

    // Shapes
    const gert::Shape *x_shape = context->GetInputShape(0);
    // Expect ND: [batch, inSize]
    uint32_t batch = 1;
    if (x_shape->GetOriginShape().GetDimNum() >= 1) {
        batch = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    }

    const uint32_t in_sz  = static_cast<uint32_t>(*inputSize);
    const uint32_t h1_sz  = static_cast<uint32_t>(*h1);
    const uint32_t h2_sz  = static_cast<uint32_t>(*h2);
    const uint32_t out_sz = static_cast<uint32_t>(*outSize);

    // Offsets in packed buffers (in elements)
    const uint64_t w1n = static_cast<uint64_t>(h1_sz) * static_cast<uint64_t>(in_sz);
    const uint64_t w2n = static_cast<uint64_t>(h2_sz) * static_cast<uint64_t>(h1_sz);
    const uint64_t w3n = static_cast<uint64_t>(out_sz) * static_cast<uint64_t>(h2_sz);
    const uint64_t b1n = static_cast<uint64_t>(h1_sz);
    const uint64_t b2n = static_cast<uint64_t>(h2_sz);
    const uint64_t b3n = static_cast<uint64_t>(out_sz);

    const uint64_t w1_off = 0;
    const uint64_t w2_off = w1_off + w1n;
    const uint64_t w3_off = w2_off + w2n;

    const uint64_t b1_off = 0;
    const uint64_t b2_off = b1_off + b1n;
    const uint64_t b3_off = b2_off + b2n;

    // Prepare three Matmul tilings (NORMAL layout, float)
    platform_ascendc::PlatformAscendC plat(context->GetPlatformInfo());

    auto get_mm_tiling = [&](uint32_t M, uint32_t N, uint32_t K, optiling::TCubeTiling &td) -> bool {
        matmul_tiling::MatmulApiTiling api(plat);
        api.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
        api.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
        api.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
        api.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
        api.SetShape(M, N, K);
        api.SetOrgShape(M, N, K);
        api.SetBias(true);
        api.SetBufferSpace(-1, -1, -1);
        return api.GetTiling(td) != -1;
    };

    optiling::TCubeTiling mm1_t, mm2_t, mm3_t;
    bool ok = true;
    ok = ok && get_mm_tiling(batch, h1_sz, in_sz, mm1_t);
    ok = ok && get_mm_tiling(batch, h2_sz, h1_sz, mm2_t);
    ok = ok && get_mm_tiling(batch, out_sz, h2_sz, mm3_t);
    if (!ok) {
        return ge::GRAPH_FAILED;
    }

    // Fill header
    t.set_batch(batch);
    t.set_inSize(in_sz);
    t.set_h1(h1_sz);
    t.set_h2(h2_sz);
    t.set_outSize(out_sz);
    t.set_w1Offset(w1_off);
    t.set_w2Offset(w2_off);
    t.set_w3Offset(w3_off);
    t.set_b1Offset(b1_off);
    t.set_b2Offset(b2_off);
    t.set_b3Offset(b3_off);

    // Append 3 TCubeTiling blobs after header in RawTilingData buffer
    auto *raw = context->GetRawTilingData();
    // Save header first
    t.SaveToBuffer(raw->GetData(), raw->GetCapacity());
    const uint32_t headerSize = t.GetDataSize();

    const uint32_t needBytes = headerSize + 3 * static_cast<uint32_t>(sizeof(optiling::TCubeTiling));
    if (needBytes > raw->GetCapacity()) {
        return ge::GRAPH_FAILED;
    }

    uint8_t *dst = reinterpret_cast<uint8_t *>(raw->GetData());
    std::memcpy(dst + headerSize, &mm1_t, sizeof(optiling::TCubeTiling));
    std::memcpy(dst + headerSize + sizeof(optiling::TCubeTiling), &mm2_t, sizeof(optiling::TCubeTiling));
    std::memcpy(dst + headerSize + 2 * sizeof(optiling::TCubeTiling), &mm3_t, sizeof(optiling::TCubeTiling));

    // Update header.tilingBytes and re-save header
    MlpCustomTilingData t2 = t;
    t2.set_tilingBytes(3u * static_cast<uint32_t>(sizeof(optiling::TCubeTiling)));
    t2.SaveToBuffer(raw->GetData(), raw->GetCapacity());

    raw->SetDataSize(needBytes);

    // Workspace: two intermediate activations [batch,h1] and [batch,h2]
    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = (static_cast<size_t>(batch) * h1_sz + static_cast<size_t>(batch) * h2_sz) * sizeof(float);

    // Block dim: conservative choice
    context->SetBlockDim(1);
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    std::vector<int64_t> dims;
    for (int i = 0; i < x_shape->GetOriginShape().GetDimNum(); ++i) {
        dims.push_back(x_shape->GetOriginShape().GetDim(i));
    }

    // attrs for out size
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *outSize = attrs->GetAttrPointer<int32_t>(3);
    if (!dims.empty()) {
        dims.back() = static_cast<int64_t>(*outSize);
    } else {
        dims.push_back(static_cast<int64_t>(*outSize));
    }

    y_shape->SetDimNum(static_cast<int>(dims.size()));
    for (size_t i = 0; i < dims.size(); ++i) {
        y_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""
#include \"kernel_operator.h\"
#include \"lib/matmul_intf.h\"
#include \"mlp_custom_tiling.h\"

using A_T = float;
using B_T = float;
using C_T = float;
using BiasT = float;

typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType;
typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

static __aicore__ inline void AppendReluGM(const AscendC::GlobalTensor<float> &gm, uint64_t elems)
{
    if (elems == 0) return;
    constexpr int32_t BUFS = 2;
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFS> qin;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFS> qout;
    uint32_t tile = 32768; // 128KB UB budget for in+out
    if (tile > elems) tile = static_cast<uint32_t>(elems);
    pipe.InitBuffer(qin, BUFS, tile * sizeof(float));
    pipe.InitBuffer(qout, BUFS, tile * sizeof(float));

    uint64_t base = 0;
    while (base < elems) {
        uint32_t len = static_cast<uint32_t>(AscendC::Min<uint64_t>(tile, elems - base));
        // copy in
        auto xLoc = qin.AllocTensor<float>();
        AscendC::DataCopy(xLoc, gm[base], len);
        qin.EnQue(xLoc);

        // compute
        auto inL = qin.DeQue<float>();
        auto outL = qout.AllocTensor<float>();
        AscendC::Relu(outL, inL, len);
        qout.EnQue(outL);
        qin.FreeTensor(inL);

        // copy out
        auto yLoc = qout.DeQue<float>();
        AscendC::DataCopy(gm[base], yLoc, len);
        qout.FreeTensor(yLoc);

        base += len;
    }
}

extern \"C\" __global__ __aicore__ void mlp_custom(GM_ADDR x,
                                                  GM_ADDR w_packed,
                                                  GM_ADDR b_packed,
                                                  GM_ADDR y,
                                                  GM_ADDR workspaceGM,
                                                  GM_ADDR tiling)
{
    // Parse header
    GET_TILING_DATA(til, tiling);
    const uint32_t batch  = til.batch;
    const uint32_t inSize = til.inSize;
    const uint32_t h1     = til.h1;
    const uint32_t h2     = til.h2;
    const uint32_t outSz  = til.outSize;

    const uint64_t w1Off  = til.w1Offset;
    const uint64_t w2Off  = til.w2Offset;
    const uint64_t w3Off  = til.w3Offset;
    const uint64_t b1Off  = til.b1Offset;
    const uint64_t b2Off  = til.b2Offset;
    const uint64_t b3Off  = til.b3Offset;

    // Locate appended TCubeTiling arrays
    const uint32_t headerSize = sizeof(MlpCustomTilingData);
    __gm__ uint8_t *tilBase = reinterpret_cast<__gm__ uint8_t *>(tiling);
    __gm__ uint8_t *blob    = tilBase + headerSize;

    optiling::TCubeTiling mm1_t;
    optiling::TCubeTiling mm2_t;
    optiling::TCubeTiling mm3_t;

    // Copy blobs from GM to local structures
    {
        __gm__ uint32_t *p = reinterpret_cast<__gm__ uint32_t *>(blob);
        uint32_t *d = reinterpret_cast<uint32_t *>(&mm1_t);
        for (uint32_t i = 0; i < sizeof(optiling::TCubeTiling)/sizeof(uint32_t); ++i) d[i] = p[i];
    }
    {
        __gm__ uint32_t *p = reinterpret_cast<__gm__ uint32_t *>(blob + sizeof(optiling::TCubeTiling));
        uint32_t *d = reinterpret_cast<uint32_t *>(&mm2_t);
        for (uint32_t i = 0; i < sizeof(optiling::TCubeTiling)/sizeof(uint32_t); ++i) d[i] = p[i];
    }
    {
        __gm__ uint32_t *p = reinterpret_cast<__gm__ uint32_t *>(blob + 2 * sizeof(optiling::TCubeTiling));
        uint32_t *d = reinterpret_cast<uint32_t *>(&mm3_t);
        for (uint32_t i = 0; i < sizeof(optiling::TCubeTiling)/sizeof(uint32_t); ++i) d[i] = p[i];
    }

    // GM tensors
    AscendC::GlobalTensor<float> xGM;
    AscendC::GlobalTensor<float> wGM;
    AscendC::GlobalTensor<float> bGM;
    AscendC::GlobalTensor<float> yGM;
    xGM.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), static_cast<uint64_t>(batch) * inSize);
    // Set total capacities for packed buffers; actual sub-tensors will be carved by offset.
    wGM.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w_packed), 0); // capacity not used by sub SetGlobalBuffer
    bGM.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b_packed), 0);
    yGM.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), static_cast<uint64_t>(batch) * outSz);

    // Sub GM for weights and biases
    AscendC::GlobalTensor<float> w1GM;
    AscendC::GlobalTensor<float> w2GM;
    AscendC::GlobalTensor<float> w3GM;
    AscendC::GlobalTensor<float> b1GM;
    AscendC::GlobalTensor<float> b2GM;
    AscendC::GlobalTensor<float> b3GM;

    // Build base views to whole packed buffers so GetPhyAddr works
    AscendC::GlobalTensor<float> wBase;
    AscendC::GlobalTensor<float> bBase;
    // Approximate total sizes to cover offsets
    uint64_t wTot = w1Off + static_cast<uint64_t>(h1) * inSize + static_cast<uint64_t>(h2) * h1 + static_cast<uint64_t>(outSz) * h2;
    uint64_t bTot = b1Off + static_cast<uint64_t>(h1) + static_cast<uint64_t>(h2) + static_cast<uint64_t>(outSz);
    wBase.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w_packed), wTot);
    bBase.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b_packed), bTot);

    w1GM.SetGlobalBuffer(const_cast<__gm__ float*>(wBase[w1Off].GetPhyAddr()),
                         static_cast<uint64_t>(h1) * inSize);
    w2GM.SetGlobalBuffer(const_cast<__gm__ float*>(wBase[w2Off].GetPhyAddr()),
                         static_cast<uint64_t>(h2) * h1);
    w3GM.SetGlobalBuffer(const_cast<__gm__ float*>(wBase[w3Off].GetPhyAddr()),
                         static_cast<uint64_t>(outSz) * h2);

    b1GM.SetGlobalBuffer(const_cast<__gm__ float*>(bBase[b1Off].GetPhyAddr()),
                         static_cast<uint64_t>(h1));
    b2GM.SetGlobalBuffer(const_cast<__gm__ float*>(bBase[b2Off].GetPhyAddr()),
                         static_cast<uint64_t>(h2));
    b3GM.SetGlobalBuffer(const_cast<__gm__ float*>(bBase[b3Off].GetPhyAddr()),
                         static_cast<uint64_t>(outSz));

    // Workspace for two intermediate tensors
    AscendC::GlobalTensor<float> tmp1GM;
    AscendC::GlobalTensor<float> tmp2GM;
    const uint64_t tmp1Elems = static_cast<uint64_t>(batch) * h1;
    const uint64_t tmp2Elems = static_cast<uint64_t>(batch) * h2;
    tmp1GM.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(workspaceGM), tmp1Elems);
    tmp2GM.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(workspaceGM) + tmp1Elems, tmp2Elems);

    // Create Matmul instances
    matmul::Matmul<aType, bType, cType, biasType> mm1;
    matmul::Matmul<aType, bType, cType, biasType> mm2;
    matmul::Matmul<aType, bType, cType, biasType> mm3;

    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1, &mm1_t, mm2, &mm2_t, mm3, &mm3_t);

    // Layer 1: X [B, in] * W1^T [in, h1] + b1 -> tmp1
    mm1.Init(&mm1_t);
    mm1.SetTensorA(xGM, false);
    mm1.SetTensorB(w1GM, true);
    mm1.SetWorkspace(workspaceGM, 0);
    mm1.SetBias(b1GM);
    mm1.IterateBatch(tmp1GM, 1, 1, false);

    // ReLU
    AppendReluGM(tmp1GM, tmp1Elems);

    // Layer 2: tmp1 [B, h1] * W2^T [h1, h2] + b2 -> tmp2
    mm2.Init(&mm2_t);
    mm2.SetTensorA(tmp1GM, false);
    mm2.SetTensorB(w2GM, true);
    mm2.SetWorkspace(workspaceGM, 0);
    mm2.SetBias(b2GM);
    mm2.IterateBatch(tmp2GM, 1, 1, false);

    // ReLU
    AppendReluGM(tmp2GM, tmp2Elems);

    // Layer 3: tmp2 [B, h2] * W3^T [h2, out] + b3 -> y
    mm3.Init(&mm3_t);
    mm3.SetTensorA(tmp2GM, false);
    mm3.SetTensorB(w3GM, true);
    mm3.SetWorkspace(workspaceGM, 0);
    mm3.SetBias(b3GM);
    mm3.IterateBatch(yGM, 1, 1, false);
}

_FN = 'mlp_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, layer_sizes, output_size):
        super(ModelNew, self).__init__()

        self.output_size = output_size
