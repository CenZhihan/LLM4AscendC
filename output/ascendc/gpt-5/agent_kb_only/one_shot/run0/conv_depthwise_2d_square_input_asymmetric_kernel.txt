static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t MinU32(uint32_t a, uint32_t b)
{
    return (a < b) ? a : b;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    // Expect x: [N,C,H,W], weight: [C,1,K,1]
    const uint32_t n = static_cast<uint32_t>(xShape->GetDim(0));
    const uint32_t c = static_cast<uint32_t>(xShape->GetDim(1));
    const uint32_t h = static_cast<uint32_t>(xShape->GetDim(2));
    const uint32_t w = static_cast<uint32_t>(xShape->GetDim(3));

    const uint32_t k = static_cast<uint32_t>(wShape->GetDim(2));
    const uint32_t oh = h >= k ? (h - k + 1) : 0;
    const uint32_t ow = w;

    // UB model per-core (float):
    //   - inQueueX: 1 * tileW
    //   - acc buffer: 1 * tileW
    //   - tmp buffer: 1 * tileW
    //   - weight buffer: k
    // Total bytes ≈ (3 * tileW + k) * 4
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t HARD_CAP_ELEMS = 32768;
    static constexpr uint32_t ALIGN_ELEMS = 128;

    uint32_t maxTileW = 0;
    if (UB_BUDGET_BYTES > (k * 4)) {
        maxTileW = (UB_BUDGET_BYTES - k * 4) / (3u * 4u);
    }
    if (maxTileW == 0) {
        maxTileW = ALIGN_ELEMS;
    }
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;

    uint32_t tileW = MinU32(w, maxTileW);
    tileW = AlignDown(tileW, ALIGN_ELEMS);
    if (tileW == 0) tileW = MinU32(w, ALIGN_ELEMS);

    // Parallelize over N*C
    uint32_t blockDim = n * c;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_k(k);
    tiling.set_oh(oh);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelConvDepthwise2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t k, uint32_t oh, uint32_t tileW)
    {
        n_ = n; c_ = c; h_ = h; w_ = w; k_ = k; oh_ = oh; tileW_ = tileW;

        const uint32_t xSize = n_ * c_ * h_ * w_;
        const uint32_t wSize = c_ * k_;
        const uint32_t ySize = n_ * c_ * oh_ * w_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // Queues and UB buffers
        pipe_.InitBuffer(inQueueX_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpAccQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpTmpQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, k_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || oh_ == 0 || w_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalNC = n_ * c_;

        for (uint32_t nc = blockIdx; nc < totalNC; nc += blockNum) {
            const uint32_t nIdx = nc / c_;
            const uint32_t cIdx = nc % c_;

            // Load weights for this channel [k]
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[cIdx * k_], k_);

            // Iterate over output rows
            for (uint32_t ohIdx = 0; ohIdx < oh_; ++ohIdx) {
                // Tile over width
                for (uint32_t wb = 0; wb < w_; wb += tileW_) {
                    uint32_t len = w_ - wb;
                    if (len > tileW_) len = tileW_;

                    // Accumulator buffer initialized to 0
                    AscendC::LocalTensor<float> accLocal = tmpAccQueue_.AllocTensor<float>();
                    AscendC::Duplicate(accLocal, (float)0, len);

                    // Accumulate over kernel height k
                    for (uint32_t kh = 0; kh < k_; ++kh) {
                        const uint32_t inH = ohIdx + kh;
                        const uint32_t xOffset = (((nIdx * c_ + cIdx) * h_ + inH) * w_) + wb;

                        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                        AscendC::DataCopy(xLocal, xGm_[xOffset], len);
                        inQueueX_.EnQue(xLocal);
                        AscendC::LocalTensor<float> xUse = inQueueX_.DeQue<float>();

                        // tmp = x * weight_scalar
                        AscendC::LocalTensor<float> tmpLocal = tmpTmpQueue_.AllocTensor<float>();

                        // Read the scalar weight from local weight buffer
                        // Note: Accessing a single scalar stored in local tensor
                        float wScalar = reinterpret_cast<const float*>(wLocal.GetAddr())[kh];

                        AscendC::Muls(tmpLocal, xUse, wScalar, static_cast<int32_t>(len));
                        AscendC::Add(accLocal, accLocal, tmpLocal, static_cast<int32_t>(len));

                        tmpTmpQueue_.FreeTensor(tmpLocal);
                        inQueueX_.FreeTensor(xUse);
                    }

                    const uint32_t yOffset = (((nIdx * c_ + cIdx) * oh_ + ohIdx) * w_) + wb;
                    AscendC::DataCopy(yGm_[yOffset], accLocal, len);
                    tmpAccQueue_.FreeTensor(accLocal);
                }
            }

            wQueue_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpAccQueue_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpTmpQueue_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0}, k_{0}, oh_{0}, tileW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvDepthwise2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tilingData.n, tilingData.c, tilingData.h, tilingData.w,
            tilingData.k, tilingData.oh, tilingData.tileW);
    op.Process();
}
