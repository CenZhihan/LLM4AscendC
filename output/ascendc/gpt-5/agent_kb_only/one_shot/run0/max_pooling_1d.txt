namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape* inShape = context->GetInputShape(0);
    const auto& origin = inShape->GetOriginShape();
    uint32_t n = static_cast<uint32_t>(GetDimOr(*inShape, 0, 1));
    uint32_t c = static_cast<uint32_t>(GetDimOr(*inShape, 1, 1));
    uint32_t l = static_cast<uint32_t>(GetDimOr(*inShape, 2, 1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t ksize = *(attrs->GetAttrPointer<int32_t>(0));
    int32_t stride = *(attrs->GetAttrPointer<int32_t>(1));
    int32_t padding = *(attrs->GetAttrPointer<int32_t>(2));
    const uint32_t kernelSize = static_cast<uint32_t>(ksize);
    if (stride <= 0) stride = ksize;

    // Compute output length with zero padding semantics
    // outL = floor((L + 2*P - K)/S) + 1, guard against negatives
    int64_t numer = static_cast<int64_t>(l) + 2LL * static_cast<int64_t>(padding) - static_cast<int64_t>(kernelSize);
    int64_t outL_tmp = (numer >= 0) ? (numer / static_cast<int64_t>(stride) + 1) : 0;
    uint32_t outL = static_cast<uint32_t>(outL_tmp > 0 ? outL_tmp : 0);

    uint32_t totalRows = n * c;

    uint32_t blockDim = totalRows;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 32u) blockDim = 32u;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_l(l);
    tiling.set_outL(outL);
    tiling.set_totalRows(totalRows);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(static_cast<uint32_t>(stride));
    tiling.set_padding(static_cast<uint32_t>(padding));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Tensor *x_tensor = context->GetInputTensor(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t ksize = *(attrs->GetAttrPointer<int32_t>(0));
    int32_t stride = *(attrs->GetAttrPointer<int32_t>(1));
    int32_t padding = *(attrs->GetAttrPointer<int32_t>(2));
    if (stride <= 0) stride = ksize;

    int64_t N = x_shape->GetOriginShape().GetDim(0);
    int64_t C = x_shape->GetOriginShape().GetDim(1);
    int64_t L = x_shape->GetOriginShape().GetDim(2);

    int64_t numer = L + 2LL * padding - static_cast<int64_t>(ksize);
    int64_t outL = (numer >= 0) ? (numer / static_cast<int64_t>(stride) + 1) : 0;
    if (outL < 0) outL = 0;

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> y_dims = {N, C, outL};
    y_shape->SetDimNum(3);
    for (int i = 0; i < 3; ++i) y_shape->SetDim(i, y_dims[i]);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t l, uint32_t outL,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding)
    {
        n_ = n; c_ = c; l_ = l; outL_ = outL;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        totalRows_ = n_ * c_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * c_ * l_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * c_ * outL_);

        // UB buffers for one row
        pipe_.InitBuffer(inQueueX_, 1, l_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, outL_ * sizeof(float));

        // Temporary buffers
        pipe_.InitBuffer(workQueue_, 1, l_ * sizeof(float));            // workLocal for ReduceMax
        pipe_.InitBuffer(winQueue_, 1, kernelSize_ * sizeof(float));    // window buffer
        pipe_.InitBuffer(resQueue_, 1, 2 * sizeof(float));              // result (max, idx)
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || l_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            const uint64_t xBase = static_cast<uint64_t>(row) * l_;
            const uint64_t yBase = static_cast<uint64_t>(row) * outL_;
            CopyInRow(xBase);
            ComputeRow();
            CopyOutRow(yBase);
        }
    }

private:
    __aicore__ inline void CopyInRow(uint64_t gmOffsetX)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffsetX], l_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::LocalTensor<float> workLocal = workQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> winLocal  = winQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> resLocal  = resQueue_.AllocTensor<float>();

        for (uint32_t t = 0; t < outL_; ++t) {
            int64_t s = static_cast<int64_t>(t) * static_cast<int64_t>(stride_) - static_cast<int64_t>(padding_);
            int64_t a = s < 0 ? 0 : s;
            int64_t b = s + static_cast<int64_t>(kernelSize_);
            if (b > static_cast<int64_t>(l_)) b = static_cast<int64_t>(l_);
            int32_t len = static_cast<int32_t>(b - a);
            if (len <= 0) {
                // Entire window is padding => max is 0
                AscendC::Duplicate(yLocal[t], 0.0f, 1);
                continue;
            }
            // Copy the valid part of the window to winLocal
            AscendC::Copy(winLocal, xLocal[a], static_cast<uint32_t>(len));
            // ReduceMax on the window
            AscendC::ReduceMax(resLocal, winLocal, workLocal, len, false);
            // Store max to yLocal[t]
            AscendC::Copy(yLocal[t], resLocal, 1);
            // Include zero padding: clamp with 0
            AscendC::Maxs(yLocal[t], yLocal[t], 0.0f, 1);
        }

        resQueue_.FreeTensor(resLocal);
        winQueue_.FreeTensor(winLocal);
        workQueue_.FreeTensor(workLocal);

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutRow(uint64_t gmOffsetY)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffsetY], yLocal, outL_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> winQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> resQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, l_{0}, outL_{0};
    uint32_t kernelSize_{1}, stride_{1}, padding_{0};
    uint32_t totalRows_{0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.l, tiling_data.outL,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_1d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0, return_indices: bool = False):
        super(ModelNew, self).__init__()
        self.kernel_size = int(kernel_size)
        self.stride = int(stride) if stride is not None else None
        self.padding = int(padding)
        self.return_indices = bool(return_indices)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        stride = self.stride if self.stride is not None else self.kernel_size
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, int(self.kernel_size), int(stride), int(self.padding), bool(self.return_indices))
