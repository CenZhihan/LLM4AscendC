kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dAsym {
public:
    __aicore__ inline KernelConvTransposed3dAsym() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t din, uint32_t hin, uint32_t win,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout,
                                uint32_t totalOutLen, uint32_t tileLen)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        din_ = din; hin_ = hin; win_ = win;
        kd_ = kd; kh_ = kh; kw_ = kw;
        dout_ = dout; hout_ = hout; wout_ = wout;
        totalOutLen_ = totalOutLen;
        tileLen_ = tileLen;

        const uint64_t xElems = static_cast<uint64_t>(n_) * cin_ * din_ * hin_ * win_;
        const uint64_t wElems = static_cast<uint64_t>(cin_) * cout_ * kd_ * kh_ * kw_;
        const uint64_t yElems = static_cast<uint64_t>(n_) * cout_ * dout_ * hout_ * wout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(xElems));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(wElems));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(yElems));

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, 2 * sizeof(float)); // scratch for single-element fetches
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * tileLen_;
        uint64_t base = static_cast<uint64_t>(blockIdx) * tileLen_;
        if (base >= totalOutLen_) return;

        while (base < totalOutLen_) {
            uint32_t curLen = static_cast<uint32_t>(totalOutLen_ - base);
            if (curLen > tileLen_) curLen = tileLen_;
            ComputeAndStore(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        return ((((static_cast<uint64_t>(n) * cin_ + c) * din_ + d) * hin_ + h) * win_ + w);
    }
    __aicore__ inline uint64_t WOffset(uint32_t ci, uint32_t co, uint32_t kz, uint32_t ky, uint32_t kx) const
    {
        return ((((static_cast<uint64_t>(ci) * cout_ + co) * kd_ + kz) * kh_ + ky) * kw_ + kx);
    }
    __aicore__ inline void UnflattenY(uint64_t idx, uint32_t &n, uint32_t &co,
                                      uint32_t &od, uint32_t &oh, uint32_t &ow) const
    {
        uint64_t t = idx;
        ow = static_cast<uint32_t>(t % wout_); t /= wout_;
        oh = static_cast<uint32_t>(t % hout_); t /= hout_;
        od = static_cast<uint32_t>(t % dout_); t /= dout_;
        co = static_cast<uint32_t>(t % cout_); t /= cout_;
        n  = static_cast<uint32_t>(t);
    }

    __aicore__ inline void ComputeAndStore(uint64_t base, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> scratch = tmpQueue_.AllocTensor<float>(); // size 2
        yLocal.SetSize(len);

        for (uint32_t i = 0; i < len; ++i) {
            uint32_t n, co, od, oh, ow;
            UnflattenY(base + i, n, co, od, oh, ow);

            float acc = 0.0f;

            // For conv_transpose3d with stride=1, pad=0, dilation=1:
            // iz = od - (kd - 1 - kz), etc. Only accumulate when indices within bounds.
            for (uint32_t ci = 0; ci < cin_; ++ci) {
                for (uint32_t kz = 0; kz < kd_; ++kz) {
                    int32_t iz = static_cast<int32_t>(od) - static_cast<int32_t>(kd_ - 1 - kz);
                    if (iz < 0 || iz >= static_cast<int32_t>(din_)) continue;
                    for (uint32_t ky = 0; ky < kh_; ++ky) {
                        int32_t iy = static_cast<int32_t>(oh) - static_cast<int32_t>(kh_ - 1 - ky);
                        if (iy < 0 || iy >= static_cast<int32_t>(hin_)) continue;
                        for (uint32_t kx = 0; kx < kw_; ++kx) {
                            int32_t ix = static_cast<int32_t>(ow) - static_cast<int32_t>(kw_ - 1 - kx);
                            if (ix < 0 || ix >= static_cast<int32_t>(win_)) continue;

                            uint64_t xoff = XOffset(n, ci, static_cast<uint32_t>(iz),
                                                    static_cast<uint32_t>(iy),
                                                    static_cast<uint32_t>(ix));
                            uint64_t woff = WOffset(ci, co, kz, ky, kx);

                            // Load single elements from GM into UB scratch
                            AscendC::DataCopy(scratch, xGm_[xoff], 1);
                            AscendC::DataCopy(scratch[1], wGm_[woff], 1);
                            float xv = scratch.GetValue(0);
                            float wv = scratch.GetValue(1);
                            acc += xv * wv;
                        }
                    }
                }
            }

            yLocal.SetValue(i, acc);
        }

        outQueueY_.EnQue<float>(yLocal);
        // Copy out
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[base], yOut, len);
        outQueueY_.FreeTensor(yOut);
        tmpQueue_.FreeTensor(scratch);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_, cin_, cout_;
    uint32_t din_, hin_, win_;
    uint32_t kd_, kh_, kw_;
    uint32_t dout_, hout_, wout_;
    uint32_t totalOutLen_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvTransposed3dAsym op;
    op.Init(x, weight, y,
            tilingData.n, tilingData.cin, tilingData.cout,
            tilingData.din, tilingData.hin, tilingData.win,
            tilingData.kd, tilingData.kh, tilingData.kw,
            tilingData.dout, tilingData.hout, tilingData.wout,
            tilingData.totalOutLen, tilingData.tileLen);
    op.Process();
}
