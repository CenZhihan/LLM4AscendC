static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1ull) / b);
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Fixed parameters per problem statement
    constexpr uint32_t STRIDE_H = 1;
    constexpr uint32_t STRIDE_W = 1;
    constexpr uint32_t PAD_H = 2;
    constexpr uint32_t PAD_W = 4;
    constexpr uint32_t DIL_H = 2;
    constexpr uint32_t DIL_W = 3;

    const gert::Shape* xShape = context->GetInputShape(0);   // [N, Cin, H, W]
    const gert::Shape* wShape = context->GetInputShape(1);   // [Cout, Cin, Kh, Kw]

    const uint32_t N   = static_cast<uint32_t>(xShape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(xShape->GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(xShape->GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(xShape->GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(wShape->GetDim(0));
    const uint32_t CinW = static_cast<uint32_t>(wShape->GetDim(1));
    const uint32_t Kh   = static_cast<uint32_t>(wShape->GetDim(2));
    const uint32_t Kw   = static_cast<uint32_t>(wShape->GetDim(3));

    if (Cin != CinW) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t effKh = (Kh - 1u) * DIL_H + 1u;
    const uint32_t effKw = (Kw - 1u) * DIL_W + 1u;

    const uint32_t Hout = (H + 2u * PAD_H - effKh) / STRIDE_H + 1u;
    const uint32_t Wout = (W + 2u * PAD_W - effKw) / STRIDE_W + 1u;

    const uint64_t totalOut64 = static_cast<uint64_t>(N) * Cout * Hout * Wout;
    const uint32_t totalOut = static_cast<uint32_t>(totalOut64);

    // Tile sizing: compute in tiles of output elements. Scalar UB usage is minimal.
    // Choose a moderate tile size to balance kernel launch occupancy vs. per-iter overhead.
    uint32_t tileOut = 4096;
    if (tileOut > totalOut) tileOut = totalOut;
    if (tileOut == 0u) tileOut = 1u;

    uint32_t blockDim = CeilDiv(totalOut64, tileOut);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    ConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustomTilingData tiling;
    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_hin(H);
    tiling.set_win(W);
    tiling.set_cout(Cout);
    tiling.set_kh(Kh);
    tiling.set_kw(Kw);
    tiling.set_padH(PAD_H);
    tiling.set_padW(PAD_W);
    tiling.set_dilH(DIL_H);
    tiling.set_dilW(DIL_W);
    tiling.set_strideH(STRIDE_H);
    tiling.set_strideW(STRIDE_W);
    tiling.set_hout(Hout);
    tiling.set_wout(Wout);
    tiling.set_totalOut(totalOut);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0; // no extra workspace
    return ge::GRAPH_SUCCESS;
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape* x = context->GetInputShape(0); // [N, Cin, H, W]
    const gert::Shape* w = context->GetInputShape(1); // [Cout, Cin, Kh, Kw]
    if (x->GetDimNum() != 4 || w->GetDimNum() != 4) {
        return GRAPH_FAILED;
    }

    const int64_t N   = x->GetDim(0);
    const int64_t Cin = x->GetDim(1);
    const int64_t H   = x->GetDim(2);
    const int64_t W   = x->GetDim(3);

    const int64_t Cout = w->GetDim(0);
    const int64_t CinW = w->GetDim(1);
    const int64_t Kh   = w->GetDim(2);
    const int64_t Kw   = w->GetDim(3);
    if (Cin != CinW) {
        return GRAPH_FAILED;
    }

    // Fixed parameters
    constexpr int64_t STRIDE_H = 1;
    constexpr int64_t STRIDE_W = 1;
    constexpr int64_t PAD_H = 2;
    constexpr int64_t PAD_W = 4;
    constexpr int64_t DIL_H = 2;
    constexpr int64_t DIL_W = 3;

    const int64_t effKh = (Kh - 1) * DIL_H + 1;
    const int64_t effKw = (Kw - 1) * DIL_W + 1;

    const int64_t Hout = (H + 2 * PAD_H - effKh) / STRIDE_H + 1;
    const int64_t Wout = (W + 2 * PAD_W - effKw) / STRIDE_W + 1;

    gert::Shape* y = context->GetOutputShape(0);
    y->SetDimNum(4);
    y->SetDim(0, N);
    y->SetDim(1, Cout);
    y->SetDim(2, Hout);
    y->SetDim(3, Wout);
    return GRAPH_SUCCESS;
}

#include "kernel_operator.h"

class KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t hin, uint32_t win,
                                uint32_t cout, uint32_t kh, uint32_t kw,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilH, uint32_t dilW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t hout, uint32_t wout,
                                uint32_t totalOut, uint32_t tileOut)
    {
        n_ = n; cin_ = cin; hin_ = hin; win_ = win;
        cout_ = cout; kh_ = kh; kw_ = kw;
        padH_ = padH; padW_ = padW;
        dilH_ = dilH; dilW_ = dilW;
        strideH_ = strideH; strideW_ = strideW;
        hout_ = hout; wout_ = wout;
        totalOut_ = totalOut; tileOut_ = tileOut;

        const uint32_t xSize = n_ * cin_ * hin_ * win_;
        const uint32_t wSize = cout_ * cin_ * kh_ * kw_;
        const uint32_t ySize = totalOut_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(tmpCalc_, 1, 3 * sizeof(float)); // small scratch area
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileOut_;
        uint32_t base = blockIdx * tileOut_;
        if (base >= totalOut_) return;

        while (true) {
            uint32_t curLen = totalOut_ - base;
            if (curLen > tileOut_) curLen = tileOut_;
            ComputeRange(base, curLen);

            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalOut_) break;
            base = nextBase;
        }
    }

private:
    __aicore__ inline void ComputeRange(uint32_t base, uint32_t len)
    {
        // Allocate small reusable scalars in VECCALC
        AscendC::LocalTensor<float> scalarX = tmpCalc_.AllocTensor<float>();
        AscendC::LocalTensor<float> scalarW = tmpCalc_.AllocTensor<float>();
        AscendC::LocalTensor<float> scalarY = tmpCalc_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            const uint32_t outLinear = base + i;

            // Decode [n, oc, oh, ow] from linear index with order [N, Cout, Hout, Wout]
            uint32_t t = outLinear;
            const uint32_t ow = t % wout_;
            t /= wout_;
            const uint32_t oh = t % hout_;
            t /= hout_;
            const uint32_t oc = t % cout_;
            const uint32_t n  = t / cout_;

            float acc = 0.0f;

            for (uint32_t ic = 0; ic < cin_; ++ic) {
                for (uint32_t kh = 0; kh < kh_; ++kh) {
                    // input y index: oh * strideH - padH + kh * dilH
                    int32_t iy = static_cast<int32_t>(oh) * static_cast<int32_t>(strideH_) -
                                 static_cast<int32_t>(padH_) +
                                 static_cast<int32_t>(kh) * static_cast<int32_t>(dilH_);
                    if (iy < 0 || iy >= static_cast<int32_t>(hin_)) {
                        continue;
                    }
                    for (uint32_t kw = 0; kw < kw_; ++kw) {
                        int32_t ix = static_cast<int32_t>(ow) * static_cast<int32_t>(strideW_) -
                                     static_cast<int32_t>(padW_) +
                                     static_cast<int32_t>(kw) * static_cast<int32_t>(dilW_);
                        if (ix < 0 || ix >= static_cast<int32_t>(win_)) {
                            continue;
                        }

                        // Compute flattened offsets
                        const uint32_t xOff =
                            (((n * cin_ + ic) * hin_ + static_cast<uint32_t>(iy)) * win_) +
                            static_cast<uint32_t>(ix);
                        const uint32_t wOff =
                            ((((oc * cin_ + ic) * kh_) + kh) * kw_) + kw;

                        // Load one x and one w scalar from GM
                        AscendC::DataCopy(scalarX, xGm_[xOff], 1);
                        AscendC::DataCopy(scalarW, wGm_[wOff], 1);
                        float xv = scalarX.GetValue(0);
                        float wv = scalarW.GetValue(0);
                        acc += xv * wv;
                    }
                }
            }

            // Store result to GM
            scalarY.SetValue(0, acc);
            AscendC::DataCopy(yGm_[outLinear], scalarY, 1);
        }

        tmpCalc_.FreeTensor(scalarY);
        tmpCalc_.FreeTensor(scalarW);
        tmpCalc_.FreeTensor(scalarX);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpCalc_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, hin_{0}, win_{0};
    uint32_t cout_{0}, kh_{0}, kw_{0};
    uint32_t padH_{0}, padW_{0}, dilH_{0}, dilW_{0};
    uint32_t strideH_{0}, strideW_{0};
    uint32_t hout_{0}, wout_{0};
    uint32_t totalOut_{0}, tileOut_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom op;
    op.Init(
        x, w, y,
        tiling_data.n, tiling_data.cin, tiling_data.hin, tiling_data.win,
        tiling_data.cout, tiling_data.kh, tiling_data.kw,
        tiling_data.padH, tiling_data.padW,
        tiling_data.dilH, tiling_data.dilW,
        tiling_data.strideH, tiling_data.strideW,
        tiling_data.hout, tiling_data.wout,
        tiling_data.totalOut, tiling_data.tileOut
    );
    op.Process();
}
