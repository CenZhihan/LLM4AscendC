namespace {
static inline int64_t FloorDiv(int64_t a, int64_t b) {
    // assume b > 0
    if (a >= 0) return a / b;
    return -(( -a + b - 1 ) / b);
}
static inline int64_t CalcOut(int64_t in, int64_t pad, int64_t dilation, int64_t k, int64_t stride) {
    // floor((in + 2*pad - dilation*(k-1) -1)/stride + 1)
    const int64_t num = (in + 2 * pad - dilation * (k - 1) - 1);
    return FloorDiv(num, stride) + 1;
}
} // namespace

    const int32_t *stride_w = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding_h = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *padding_w = attrs->GetAttrPointer<int32_t>(3);
    const int32_t *dilation_h = attrs->GetAttrPointer<int32_t>(4);
    const int32_t *dilation_w = attrs->GetAttrPointer<int32_t>(5);

    const int64_t Sh = static_cast<int64_t>(*stride_h);
    const int64_t Sw = static_cast<int64_t>(*stride_w);
    const int64_t Ph = static_cast<int64_t>(*padding_h);
    const int64_t Pw = static_cast<int64_t>(*padding_w);
    const int64_t Dh = static_cast<int64_t>(*dilation_h);

    y_shape->SetDim(3, Ow);

#include "kernel_operator.h"

class KernelConvDepthwise2dAsym {
public:
    __aicore__ inline KernelConvDepthwise2dAsym() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t kh, uint32_t kw,
                                uint32_t sh, uint32_t sw,
                                uint32_t ph, uint32_t pw,
                                uint32_t dh, uint32_t dw,
                                uint32_t oh, uint32_t ow)
    {
        n_ = n; c_ = c; h_ = h; w_ = w_in;
        kh_ = kh; kw_ = kw;
        sh_ = sh; sw_ = sw;
        ph_ = ph; pw_ = pw;
        dh_ = dh; dw_ = dw;
        oh_ = oh; ow_ = ow;

        xSize_ = n_ * c_ * h_ * w_;
        wSize_ = c_ * kh_ * kw_;
        ySize_ = n_ * c_ * oh_ * ow_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize_);

        // UB queues: weight buffer (kh*kw), scalar x, scalar y
        pipe_.InitBuffer(wQueue_, 1, kh_ * kw_ * sizeof(float));
        pipe_.InitBuffer(xQueue_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(yQueue_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || h_ == 0 || w_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t tasks = n_ * c_;

        for (uint32_t bc = blockIdx; bc < tasks; bc += blockNum) {
            const uint32_t b = bc / c_;
            const uint32_t ch = bc % c_;

            // Load weight for channel ch to UB
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            const uint32_t wOff = ch * kh_ * kw_;
            AscendC::DataCopy(wLocal, wGm_[wOff], kh_ * kw_);

            // Traverse output spatial locations
            for (uint32_t oh = 0; oh < oh_; ++oh) {
                for (uint32_t ow_pos = 0; ow_pos < ow_; ++ow_pos) {
                    float acc = 0.0f;

                    // Convolution window
                    for (uint32_t r = 0; r < kh_; ++r) {
                        const int32_t in_y = static_cast<int32_t>(oh * sh_) + static_cast<int32_t>(r * dh_) - static_cast<int32_t>(ph_);
                        if (in_y < 0 || in_y >= static_cast<int32_t>(h_)) {
                            continue;
                        }
                        for (uint32_t s = 0; s < kw_; ++s) {
                            const int32_t in_x = static_cast<int32_t>(ow_pos * sw_) + static_cast<int32_t>(s * dw_) - static_cast<int32_t>(pw_);
                            if (in_x < 0 || in_x >= static_cast<int32_t>(w_)) {
                                continue;
                            }

                            const uint32_t xOff = (((b * c_ + ch) * h_ + static_cast<uint32_t>(in_y)) * w_ + static_cast<uint32_t>(in_x));
                            AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[xOff], 1);
                            float xval = xLocal.GetValue(0);
                            xQueue_.FreeTensor(xLocal);

                            const uint32_t kIdx = r * kw_ + s;
                            float wval = wLocal.GetValue(kIdx);

                            acc += xval * wval;
                        }
                    }

                    // Store y
                    const uint32_t yOff = (((b * c_ + ch) * oh_ + oh) * ow_ + ow_pos);
                    AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                    // Fill one element and write back
                    // Use Duplicate for portability
                    AscendC::Duplicate(yLocal, acc, 1);
                    AscendC::DataCopy(yGm_[yOff], yLocal, 1);
                    yQueue_.FreeTensor(yLocal);
                }
            }

            wQueue_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> wQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> xQueue_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> yQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t kh_{0}, kw_{0};
    uint32_t sh_{1}, sw_{1};
    uint32_t ph_{0}, pw_{0};
    uint32_t dh_{1}, dw_{1};
    uint32_t oh_{0}, ow_{0};
    uint32_t xSize_{0}, wSize_{0}, ySize_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvDepthwise2dAsym op;
    op.Init(x, weight, y,
            tilingData.n(), tilingData.c(), tilingData.h(), tilingData.w(),
            tilingData.kh(), tilingData.kw(),
            tilingData.sh(), tilingData.sw(),
            tilingData.ph(), tilingData.pw(),
            tilingData.dh(), tilingData.dw(),
            tilingData.oh(), tilingData.ow());
    op.Process();
}
