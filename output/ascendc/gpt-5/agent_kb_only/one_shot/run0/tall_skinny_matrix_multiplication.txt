host_operator_src="""
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include "optiling/cube_tiling.h"
#include "optiling/ascendc/platform_ascendc.h"
#include "optiling/tiling.h"

namespace optiling {

static inline int64_t GetDim(const gert::Shape *shape, int i) {
    return shape->GetOriginShape().GetDim(i);
}

static inline int64_t GetRank(const gert::Shape *shape) {
    return shape->GetOriginShape().GetDimNum();
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    auto *a_shape = context->GetInputShape(0);
    auto *b_shape = context->GetInputShape(1);
    if (GetRank(a_shape) != 2 || GetRank(b_shape) != 2) {
        return ge::GRAPH_FAILED;
    }

    const int64_t M = GetDim(a_shape, 0);
    const int64_t K = GetDim(a_shape, 1);
    const int64_t Kb = GetDim(b_shape, 0);
    const int64_t N = GetDim(b_shape, 1);
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());

    // Use multi-core tiling to handle tall/skinny efficiently
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);
    // Set number of cores allowed; a conservative number favors stability.
    tiling.SetDim(24);

    // A: GM, ND, FP16; B: GM, ND, FP16; C: GM, ND, FP32
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBias(false);

    // Set shapes
    tiling.SetShape(static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K));
    tiling.SetSingleShape(static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K));
    tiling.SetOrgShape(static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K));

    // Use all available buffer spaces by default
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling cubeTiling;
    int ret = tiling.GetTiling(cubeTiling);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    uint32_t blockDim = static_cast<uint32_t>(tiling.GetCoreNum());
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    // Save raw TCubeTiling to device
    auto *raw = context->GetRawTilingData();
    const uint32_t need = sizeof(optiling::TCubeTiling);
    if (raw->GetCapacity() < need) {
        return ge::GRAPH_FAILED;
    }
    std::memcpy(raw->GetData(), &cubeTiling, need);
    raw->SetDataSize(need);

    // No workspace needed
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = a_shape->GetOriginShape().GetDim(0);
    const int64_t N = b_shape->GetOriginShape().GetDim(1);
    if (a_shape->GetOriginShape().GetDim(1) != b_shape->GetOriginShape().GetDim(0)) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output is float32 accumulation
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

// Kernel performs C = A * B using AscendC high-level Matmul API.
// A: FP16 ND on GM (M x K)
// B: FP16 ND on GM (K x N)
// C: FP32 ND on GM (M x N)
// Tiling is provided as raw TCubeTiling in tilingGM.

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM, GM_ADDR tilingGM, GM_ADDR workspaceGM)
{
    // Load TCubeTiling from tilingGM
    TCubeTiling tiling;
    auto tilingGMU32 = reinterpret_cast<__gm__ uint32_t*>(tilingGM);
    auto tilingU32 = reinterpret_cast<uint32_t*>(&tiling);
    for (int i = 0; i < static_cast<int>(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i) {
        tilingU32[i] = tilingGMU32[i];
    }

    // Define Matmul types
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    // Initialize GM tensors
    AscendC::GlobalTensor<half> aGlobal;
    AscendC::GlobalTensor<half> bGlobal;
    AscendC::GlobalTensor<float> cGlobal;

    // Size in elements for SetGlobalBuffer: using total element counts from tiling layouts
    // For NORMAL layout, these sizes are sufficient placeholders; data movement is handled by Matmul.
    uint64_t sizeA = static_cast<uint64_t>(tiling.M) * static_cast<uint64_t>(tiling.Ka);
    uint64_t sizeB = static_cast<uint64_t>(tiling.Kb) * static_cast<uint64_t>(tiling.N);
    uint64_t sizeC = static_cast<uint64_t>(tiling.M) * static_cast<uint64_t>(tiling.N);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cGM), sizeC);

    // Create Matmul instance
    matmul::Matmul<aType, bType, cType, biasType> mm;
    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);

    // Initialize and bind tensors
    mm.Init(&tiling);
    mm.SetTensorA(aGlobal[0], false);
    mm.SetTensorB(bGlobal[0], false);
    mm.SetTensorC(cGlobal[0]);
    mm.SetWorkspace(workspaceGM, 0);

    // Compute
    mm.IterateAll();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include \"pytorch_npu_helper.hpp\"

static at::Tensor matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Convert to FP16 for matmul, accumulate FP32, then cast back to input dtype if needed.
    auto A_half = A.to(at::kHalf);
    auto B_half = B.to(at::kHalf);
    // Output is float per operator definition
    auto out_options = A.options().dtype(at::kFloat);
    at::Tensor C = at::empty({A.size(0), B.size(1)}, out_options);
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, A_half, B_half, C);
    // Cast back to original dtype of A if needed
    if (A.dtype() != at::kFloat) {
        return C.to(A.dtype());
    }
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"tall_skinny_matrix_multiplication_custom\", &matmul_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"tall_skinny_matrix_multiplication_custom\", &matmul_impl_npu,
          \"Tall-Skinny Matrix Multiplication (TallSkinnyMatrixMultiplicationCustom on NPU)\");
}
