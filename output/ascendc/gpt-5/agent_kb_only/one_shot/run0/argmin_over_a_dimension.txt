namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgminOverADimensionCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    // attr 0 -> dim
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);

    // Get rank and dims
    const gert::Shape &origin = x_shape->GetOriginShape();
    const int64_t rank = origin.GetDimNum();
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : 0;
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    uint64_t outer = 1, inner = 1, reduce = 1;
    for (int64_t i = 0; i < dim; ++i) {
        outer *= static_cast<uint64_t>(origin.GetDim(i));
    }
    reduce = static_cast<uint64_t>(origin.GetDim(dim));
    for (int64_t i = dim + 1; i < rank; ++i) {
        inner *= static_cast<uint64_t>(origin.GetDim(i));
    }

    const uint64_t segs = outer * inner;
    uint32_t blockDim = static_cast<uint32_t>(segs);
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_outerSize(static_cast<uint32_t>(outer));
    tiling.set_reduceSize(static_cast<uint32_t>(reduce));
    tiling.set_innerSize(static_cast<uint32_t>(inner));
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);

    const gert::Shape &origin = x_shape->GetOriginShape();
    const int64_t rank = origin.GetDimNum();
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : 0;
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    std::vector<int64_t> outDims;
    outDims.reserve(rank > 0 ? static_cast<size_t>(rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        outDims.emplace_back(origin.GetDim(i));

    y_shape->SetDimNum(static_cast<uint32_t>(outDims.size()));
    for (size_t i = 0; i < outDims.size(); ++i) {
        y_shape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelArgminOverADimensionCustom {
public:
    __aicore__ inline KernelArgminOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerSize, uint32_t reduceSize, uint32_t innerSize)
    {
        outer_ = outerSize;
        reduce_ = reduceSize;
        inner_ = innerSize;
        totalSeg_ = (outer_ == 0 || inner_ == 0) ? 0 : outer_ * inner_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)outer_ * reduce_ * inner_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)totalSeg_);

        // Local buffers:
        // - srcLocal: reduce_ elements (input slice)
        // - workLocal: reduce_ elements (temp for ReduceMin)
        // - dstLocal: 2 elements (min, index encoded in float bits)
        pipe_.InitBuffer(inQueue_, 1, reduce_ * sizeof(float));
        pipe_.InitBuffer(workQueue_, 1, reduce_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, 1, 2 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalSeg_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t seg = blockIdx; seg < totalSeg_; seg += blockNum) {
            // Map seg -> (outerIdx, innerIdx)
            const uint32_t outerIdx = inner_ == 0 ? 0 : (seg / inner_);
            const uint32_t innerIdx = inner_ == 0 ? 0 : (seg % inner_);

            // Copy gather along reduce dim into contiguous srcLocal
            AscendC::LocalTensor<float> srcLocal = inQueue_.AllocTensor<float>();
            for (uint32_t r = 0; r < reduce_; ++r) {
                uint64_t gmOffset = ((uint64_t)outerIdx * reduce_ + r) * inner_ + innerIdx;
                AscendC::DataCopy(srcLocal[r], xGm_[gmOffset], 1);
            }
            inQueue_.EnQue(srcLocal);

            // Compute ReduceMin with index
            AscendC::LocalTensor<float> src = inQueue_.DeQue<float>();
            AscendC::LocalTensor<float> work = workQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> dst = outQueue_.AllocTensor<float>();

            // calIndex = true
            AscendC::ReduceMin(dst, src, work, static_cast<int32_t>(reduce_), true);

            // Extract encoded index (stored as float bits), convert to numeric float
            float idxEncoded = dst.GetValue(1);
            uint32_t idxU32 = *reinterpret_cast<uint32_t*>(&idxEncoded);
            float idxAsFloat = static_cast<float>(idxU32);
            dst.SetValue(0, idxAsFloat);

            // Cleanup and enqueue result
            workQueue_.FreeTensor(work);
            outQueue_.EnQue(dst);
            inQueue_.FreeTensor(src);

            // Copy out first element (index as float value)
            AscendC::LocalTensor<float> outLocal = outQueue_.DeQue<float>();
            AscendC::DataCopy(yGm_[seg], outLocal, 1);
            outQueue_.FreeTensor(outLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_{0};
    uint32_t reduce_{0};
    uint32_t inner_{0};
    uint32_t totalSeg_{0};
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgminOverADimensionCustom op;
    op.Init(x, y, tiling_data.outerSize, tiling_data.reduceSize, tiling_data.innerSize);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor argmin_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    // Normalize dim
    int64_t rank = self.dim();
    int64_t adim = dim < 0 ? dim + rank : dim;

    // Build output sizes with the reduction dimension removed
    std::vector<int64_t> outSizes;
    outSizes.reserve(rank > 0 ? static_cast<size_t>(rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == adim) continue;
        outSizes.emplace_back(self.size(i));
    }

    // Kernel outputs indices encoded as float values; convert to long afterwards.
    at::Tensor result_float = at::empty(outSizes, self.options().dtype(at::kFloat));
    // Call custom kernel
    EXEC_NPU_CMD(aclnnArgminOverADimensionCustom, self, static_cast<int>(adim), result_float);

    // Convert numeric float indices to long dtype
    // Note: kernel stores numeric float indices (not bit-encoded).
    at::Tensor result_long = result_float.to(at::kLong);
    return result_long;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmin_over_a_dimension_custom", &argmin_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmin_over_a_dimension_custom", &argmin_over_a_dimension_impl_npu, "Argmin over a dimension (ArgminOverADimensionCustom on NPU)");
}
