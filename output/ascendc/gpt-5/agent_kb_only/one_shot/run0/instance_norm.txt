host_operator_src="""
#include "instance_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    InstanceNormCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const auto &origin = xShape->GetOriginShape();

    // Expect 4D NCHW
    const uint32_t n = static_cast<uint32_t>(origin.GetDim(0));
    const uint32_t c = static_cast<uint32_t>(origin.GetDim(1));
    const uint32_t h = static_cast<uint32_t>(origin.GetDim(2));
    const uint32_t w = static_cast<uint32_t>(origin.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilon = attrs->GetAttrPointer<float>(0);

    // Use a conservative blockDim for stability
    uint32_t blockDim = 8u;
    const uint64_t tiles = static_cast<uint64_t>(n) * static_cast<uint64_t>(c);
    if (tiles > 0) {
        uint32_t target = CeilDiv(tiles, 1);
        if (target < 8u) blockDim = 8u;
        else if (target > 24u) blockDim = 24u;
        else blockDim = target;
    }
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_blockDim(blockDim);
    tiling.set_epsilon(*epsilon);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *g_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    // Basic validation: gamma/beta are 1-D of length C (best-effort)
    (void)g_shape;
    (void)b_shape;

    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

template <typename dataType, bool isReuseSource = false>
class KernelInstanceNormCustom {
public:
    __aicore__ inline KernelInstanceNormCustom() {}

    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR gammaGm, GM_ADDR betaGm, GM_ADDR yGm,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w, float epsilon)
    {
        this->n_ = n;
        this->c_ = c;
        this->h_ = h;
        this->w_ = w;
        this->g_ = c;
        this->epsilon_ = epsilon;

        bshwLen_ = static_cast<uint32_t>(n_) * c_ * h_ * w_;
        cnLen_   = static_cast<uint32_t>(c_) * n_;

        inputXGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(xGm), bshwLen_);
        gammaGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(gammaGm), c_);
        betaGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(betaGm), c_);
        outputGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(yGm), bshwLen_);

        uint32_t hwAlignSize = (sizeof(dataType) * h_ * w_ + ONE_BLK_SIZE - 1) / ONE_BLK_SIZE * ONE_BLK_SIZE / sizeof(dataType);

        pipe_.InitBuffer(inQueueX_, 1, sizeof(dataType) * n_ * c_ * hwAlignSize);
        pipe_.InitBuffer(inQueueGamma_, 1, (sizeof(dataType) * c_ + 31) / 32 * 32);
        pipe_.InitBuffer(inQueueBeta_, 1, (sizeof(dataType) * c_ + 31) / 32 * 32);
        pipe_.InitBuffer(outQueueY_, 1, sizeof(dataType) * n_ * c_ * hwAlignSize);
        pipe_.InitBuffer(meanBuffer_, (sizeof(dataType) * g_ * n_ + 31) / 32 * 32);
        pipe_.InitBuffer(varBuffer_, (sizeof(dataType) * g_ * n_ + 31) / 32 * 32);
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<dataType> inputXLocal = inQueueX_.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> gammaLocal  = inQueueGamma_.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> betaLocal   = inQueueBeta_.AllocTensor<dataType>();

        uint32_t hwAlignSize = (sizeof(dataType) * h_ * w_ + ONE_BLK_SIZE - 1) / ONE_BLK_SIZE * ONE_BLK_SIZE / sizeof(dataType);

        AscendC::DataCopyParams copyParams{static_cast<uint16_t>(n_ * c_), static_cast<uint16_t>(h_ * w_ * sizeof(dataType)), 0, 0};
        AscendC::DataCopyPadParams padParams{true, 0, static_cast<uint8_t>(hwAlignSize - h_ * w_), 0};
        AscendC::DataCopyPad(inputXLocal, inputXGlobal_, copyParams, padParams);

        AscendC::DataCopyParams copyParamsGamma{1, static_cast<uint16_t>(c_ * sizeof(dataType)), 0, 0};
        AscendC::DataCopyPadParams padParamsGamma{false, 0, 0, 0};
        AscendC::DataCopyPad(gammaLocal, gammaGlobal_, copyParamsGamma, padParamsGamma);
        AscendC::DataCopyPad(betaLocal,  betaGlobal_,  copyParamsGamma, padParamsGamma);

        inQueueX_.EnQue(inputXLocal);
        inQueueGamma_.EnQue(gammaLocal);
        inQueueBeta_.EnQue(betaLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<dataType> inputXLocal = inQueueX_.DeQue<dataType>();
        AscendC::LocalTensor<dataType> gammaLocal  = inQueueGamma_.DeQue<dataType>();
        AscendC::LocalTensor<dataType> betaLocal   = inQueueBeta_.DeQue<dataType>();
        AscendC::LocalTensor<dataType> outputLocal = outQueueY_.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> meanLocal   = meanBuffer_.Get<dataType>();
        AscendC::LocalTensor<dataType> varianceLocal = varBuffer_.Get<dataType>();

        AscendC::PipeBarrier<AscendC::PIPE_ALL>();

        uint32_t stackBufferSize = 0;
        {
            AscendC::LocalTensor<float> stackBuffer;
            bool ans = AscendC::PopStackBuffer<float, AscendC::TPosition::LCM>(stackBuffer);
            (void)ans;
            stackBufferSize = stackBuffer.GetSize();
        }

        AscendC::GroupNormTiling groupNormTiling;
        uint32_t inputShape[4] = {n_, c_, h_, w_};
        AscendC::DataFormat dataFormat = AscendC::DataFormat::ND;
        AscendC::ShapeInfo shapeInfo{ (uint8_t)4, inputShape, (uint8_t)4, inputShape, dataFormat };
        AscendC::GetGroupNormNDTillingInfo(shapeInfo, stackBufferSize, sizeof(dataType), isReuseSource, g_, groupNormTiling);

        AscendC::GroupNorm<dataType, isReuseSource>(outputLocal, meanLocal, varianceLocal, inputXLocal,
                                                    gammaLocal, betaLocal, (dataType)epsilon_, groupNormTiling);

        AscendC::PipeBarrier<AscendC::PIPE_ALL>();

        outQueueY_.EnQue<dataType>(outputLocal);
        inQueueX_.FreeTensor(inputXLocal);
        inQueueGamma_.FreeTensor(gammaLocal);
        inQueueBeta_.FreeTensor(betaLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<dataType> outputLocal = outQueueY_.DeQue<dataType>();

        AscendC::DataCopyParams copyParams{static_cast<uint16_t>(n_ * c_), static_cast<uint16_t>(h_ * w_ * sizeof(dataType)), 0, 0};
        AscendC::DataCopyPadParams padParams{false, 0, 0, 0};
        AscendC::DataCopyPad(outputGlobal_, outputLocal, copyParams, padParams);

        outQueueY_.FreeTensor(outputLocal);
        AscendC::PipeBarrier<AscendC::PIPE_ALL>();
    }

private:
    uint32_t n_ {0};
    uint32_t c_ {0};
    uint32_t h_ {0};
    uint32_t w_ {0};
    uint32_t g_ {0};
    float epsilon_ {1e-5f};

    uint32_t bshwLen_ {0};
    uint32_t cnLen_ {0};

    AscendC::GlobalTensor<dataType> inputXGlobal_;
    AscendC::GlobalTensor<dataType> gammaGlobal_;
    AscendC::GlobalTensor<dataType> betaGlobal_;
    AscendC::GlobalTensor<dataType> outputGlobal_;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueGamma_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueBeta_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY_;
    AscendC::TBuf<AscendC::QuePosition::VECCALC> meanBuffer_, varBuffer_;
};

extern "C" __global__ __aicore__ void instance_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelInstanceNormCustom<float, false> op;
    op.Init(x, gamma, beta, y,
            tilingData.n, tilingData.c, tilingData.h, tilingData.w,
            tilingData.epsilon);
    op.Process();
}
"""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor instance_norm_impl_npu(const at::Tensor& self) {
    TORCH_CHECK(self.dim() == 4, \"InstanceNormCustom expects 4D NCHW input\");

    auto C = self.size(1);
    auto options = self.options().dtype(self.scalar_type()).device(self.device());

    at::Tensor gamma = at::ones({C}, options);
    at::Tensor beta  = at::zeros({C}, options);
    at::Tensor result = at::empty_like(self);

    float epsilon = 1e-5f;
    EXEC_NPU_CMD(aclnnInstanceNormCustom, self, gamma, beta, epsilon, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"instance_norm_custom\", &instance_norm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"instance_norm_custom\", &instance_norm_impl_npu, \"InstanceNorm2d (InstanceNormCustom on NPU)\");
}
