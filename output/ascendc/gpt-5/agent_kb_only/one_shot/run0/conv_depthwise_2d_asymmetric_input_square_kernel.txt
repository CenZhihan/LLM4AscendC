#include "conv_depthwise_2d_asymmetric_input_square_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}

} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB budget
static constexpr uint32_t ALIGN_ELEMS = 256;             // DMA-friendly alignment
static constexpr uint32_t HARD_CAP_ELEMS = 8192 * 2;     // conservative cap

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    // Expect NCHW for x and [C_out, 1, 3, 3] for weight
    const uint32_t n = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    const uint32_t c = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    const uint32_t h = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
    const uint32_t w = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));

    const uint32_t kh = 3;
    const uint32_t kw = 3;

    // Valid conv, stride=1, dilation=1, padding=0
    const uint32_t hout = (h >= kh) ? (h - kh + 1) : 0u;
    const uint32_t wout = (w >= kw) ? (w - kw + 1) : 0u;

    // Tile over output width for vectorization
    // UB floats needed ~= 3*(tileW+2) input rows + tileW (y) + tileW (tmp)
    // -> ~ (5*tileW + 6) floats => bytes ~= 4*(5*tileW + 6)
    uint32_t maxTileW = 0;
    if (UB_BUDGET_BYTES > 24) {
        maxTileW = (UB_BUDGET_BYTES - 24) / (5u * sizeof(float));
    } else {
        maxTileW = 1024;
    }
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;
    if (maxTileW > HARD_CAP_ELEMS) maxTileW = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileW = maxTileW;
    if (wout < tileW) {
        const uint32_t aligned = AlignDown(wout, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : wout;
    }

    // Work items across N*C*Hout. Each work item processes one (n,c,h_out) row
    const uint64_t totalWork = static_cast<uint64_t>(n) * static_cast<uint64_t>(c) * static_cast<uint64_t>(hout);
    uint32_t blockDim = static_cast<uint32_t>(totalWork == 0 ? 1 : (totalWork < 8 ? totalWork : 8));
    // Prefer 8~24 cores
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_kh(kh);
    tiling.set_kw(kw);
    tiling.set_hout(hout);
    tiling.set_wout(wout);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x = context->GetInputShape(0);
    const gert::Shape *w = context->GetInputShape(1);
    // NCHW for x, [C_out, 1, 3, 3] for w
    const int64_t N = x->GetDim(0);
    const int64_t C = x->GetDim(1);
    const int64_t H = x->GetDim(2);
    const int64_t W = x->GetDim(3);

    const int64_t KH = w->GetDim(2);
    const int64_t KW = w->GetDim(3);

    const int64_t HOut = (H >= KH) ? (H - KH + 1) : 0;
    const int64_t WOut = (W >= KW) ? (W - KW + 1) : 0;

    gert::Shape *y = context->GetOutputShape(0);
    std::vector<int64_t> y_dims = {N, C, HOut, WOut};

        y->SetDim(i, y_dims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2D {
public:
    __aicore__ inline KernelConvDepthwise2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t kh, uint32_t kw, uint32_t h_out, uint32_t w_out,
                                uint32_t tileW)
    {
        n_ = n; c_ = c; h_ = h; w_ = w_in;
        kh_ = kh; kw_ = kw;
        hOut_ = h_out; wOut_ = w_out;
        tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<size_t>(n_) * c_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<size_t>(c_) * kh_ * kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<size_t>(n_) * c_ * hOut_ * wOut_);

        // Queues: three input rows (tileW+2), one output row (tileW), one tmp (tileW), one weight (9)
        pipe_.InitBuffer(inRowQueue_, 3, (tileW_ + 2) * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, 9 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || hOut_ == 0 || wOut_ == 0 || tileW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t workPerNC = hOut_;
        const uint64_t totalWork = static_cast<uint64_t>(n_) * c_ * workPerNC;

        for (uint64_t idx = blockIdx; idx < totalWork; idx += blockNum) {
            uint64_t tmp = idx;
            const uint32_t n = static_cast<uint32_t>(tmp / (static_cast<uint64_t>(c_) * hOut_));
            tmp = tmp % (static_cast<uint64_t>(c_) * hOut_);
            const uint32_t ch = static_cast<uint32_t>(tmp / hOut_);
            const uint32_t oh = static_cast<uint32_t>(tmp % hOut_);

            // Load channel weights (3x3)
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            const uint32_t wBase = ch * kh_ * kw_;
            AscendC::DataCopy(wLocal, wGm_[wBase], 9);
            wQueue_.EnQue(wLocal);

            for (uint32_t owBase = 0; owBase < wOut_; owBase += tileW_) {
                uint32_t curW = wOut_ - owBase;
                if (curW > tileW_) curW = tileW_;

                // Copy in three contiguous rows from input: each length curW + 2
                AscendC::LocalTensor<float> row0 = inRowQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> row1 = inRowQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> row2 = inRowQueue_.AllocTensor<float>();

                const uint32_t inBase0 = IndexX(n, ch, oh + 0, owBase + 0);
                const uint32_t inBase1 = IndexX(n, ch, oh + 1, owBase + 0);
                const uint32_t inBase2 = IndexX(n, ch, oh + 2, owBase + 0);

                AscendC::DataCopy(row0, xGm_[inBase0], curW + 2);
                AscendC::DataCopy(row1, xGm_[inBase1], curW + 2);
                AscendC::DataCopy(row2, xGm_[inBase2], curW + 2);

                inRowQueue_.EnQue(row0);
                inRowQueue_.EnQue(row1);
                inRowQueue_.EnQue(row2);

                // Compute y = sum_{r=0..2} sum_{s=0..2} w[r,s] * x_row[r][s: s+curW]
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

                AscendC::LocalTensor<float> wLocal9 = wQueue_.DeQue<float>();
                AscendC::LocalTensor<float> r0 = inRowQueue_.DeQue<float>();
                AscendC::LocalTensor<float> r1 = inRowQueue_.DeQue<float>();
                AscendC::LocalTensor<float> r2 = inRowQueue_.DeQue<float>();

                bool first = true;

                // Helper lambda to process one (row, shift, weight_index)
                auto accumulate = [&](AscendC::LocalTensor<float>& row, uint32_t shift, uint32_t widx) {
                    AscendC::LocalTensor<float> sub = row[shift];
                    if (first) {
                        AscendC::Muls(yLocal, sub, wLocal9.GetValue(widx), curW);
                        first = false;
                    } else {
                        AscendC::Muls(tLocal, sub, wLocal9.GetValue(widx), curW);
                        AscendC::Add(yLocal, yLocal, tLocal, curW);
                    }
                };

                // r0c0..r2c2
                accumulate(r0, 0, 0);
                accumulate(r0, 1, 1);
                accumulate(r0, 2, 2);
                accumulate(r1, 0, 3);
                accumulate(r1, 1, 4);
                accumulate(r1, 2, 5);
                accumulate(r2, 0, 6);
                accumulate(r2, 1, 7);
                accumulate(r2, 2, 8);

                // Free temporaries
                tmpQueue_.FreeTensor(tLocal);
                wQueue_.FreeTensor(wLocal9);
                inRowQueue_.FreeTensor(r0);
                inRowQueue_.FreeTensor(r1);
                inRowQueue_.FreeTensor(r2);

                // Copy out
                const uint32_t yBase = IndexY(n, ch, oh, owBase);
                AscendC::DataCopy(yGm_[yBase], yLocal, curW);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline uint32_t IndexX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((n * c_ + c) * h_ + h) * w_ + w);
    }

    __aicore__ inline uint32_t IndexY(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((n * c_ + c) * hOut_ + h) * wOut_ + w);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 3> inRowQueue_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t kh_{3}, kw_{3};
    uint32_t hOut_{0}, wOut_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(GM_ADDR x,
                                                                                               GM_ADDR weight,
                                                                                               GM_ADDR y,
                                                                                               GM_ADDR workspace,
                                                                                               GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvDepthwise2D op;
    op.Init(x, weight, y,
            tilingData.n, tilingData.c, tilingData.h, tilingData.w,
            tilingData.kh, tilingData.kw, tilingData.hout, tilingData.wout,
            tilingData.tileW);
    op.Process();
}
