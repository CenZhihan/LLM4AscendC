host_operator_src="""
#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"
#include "conv2d_min_add_multiply_custom_tiling.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// UB model for this kernel (double-buffer for in/out):
//   inQueueX: 2 * tile
//   inQueueB: 2 * tile
//   outQueueY: 2 * tile
// Total floats = 6 * tile => bytes = 24 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ALIGN_ELEMS    = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMinAddMultiplyCustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *constantValue  = attrs->GetAttrPointer<float>(0);
    const float *scalingFactor  = attrs->GetAttrPointer<float>(1);

    uint32_t maxTile = UB_BUDGET_BYTES / 24; // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_constantValue(*constantValue);
    tiling.set_scalingFactor(*scalingFactor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dMinAddMultiplyCustom {
public:
    __aicore__ inline KernelConv2dMinAddMultiplyCustom() {}

    __aicore__ inline void Init(GM_ADDR x_gm,
                                GM_ADDR weight_gm,     // unused
                                GM_ADDR conv_bias_gm,  // unused
                                GM_ADDR post_bias_gm,
                                GM_ADDR y_gm,
                                uint32_t totalLength,
                                uint32_t tileLength,
                                float constantValue,
                                float scalingFactor,
                                TPipe* pipe)
    {
        totalLen_ = totalLength;
        tileLen_  = tileLength;
        cap_      = constantValue;
        scale_    = scalingFactor;

        xGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x_gm), totalLen_);
        bGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(post_bias_gm), totalLen_);
        yGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y_gm), totalLen_);

        pipe_ = pipe;
        pipe_->InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_->InitBuffer(inQueueB_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_->InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            CopyIn(base, curLen);
            Compute(curLen);
            CopyOut(base, curLen);

            base += strideElems;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        DataCopy(xLocal, xGlobal_[gmOffset], len);
        DataCopy(bLocal, bGlobal_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
        inQueueB_.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        LocalTensor<float> bLocal = inQueueB_.DeQue<float>();
        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // y = min(x, cap_)
        Mins(yLocal, xLocal, cap_, len);
        // y = y + post_bias
        Add(yLocal, yLocal, bLocal, len);
        // y = y * scale_
        Muls(yLocal, yLocal, scale_, len);

        inQueueX_.FreeTensor(xLocal);
        inQueueB_.FreeTensor(bLocal);
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        DataCopy(yGlobal_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    TPipe* pipe_ {nullptr};
    TQue<QuePosition::VECIN,  BUFFER_NUM> inQueueX_;
    TQue<QuePosition::VECIN,  BUFFER_NUM> inQueueB_;
    TQue<QuePosition::VECOUT, BUFFER_NUM> outQueueY_;

    GlobalTensor<float> xGlobal_;
    GlobalTensor<float> bGlobal_;
    GlobalTensor<float> yGlobal_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_  {0};
    float cap_   {0.0f};
    float scale_ {1.0f};
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(GM_ADDR x,
                                                                     GM_ADDR weight,
                                                                     GM_ADDR conv_bias,
                                                                     GM_ADDR post_bias,
                                                                     GM_ADDR y,
                                                                     GM_ADDR workspace,
                                                                     GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    TPipe pipe;
    KernelConv2dMinAddMultiplyCustom op;
    op.Init(x, weight, conv_bias, post_bias, y,
            tilingData.totalLength, tilingData.tileLength,
            tilingData.constantValue, tilingData.scalingFactor,
            &pipe);
    op.Process();
}

model_src='''
import torch
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

_FN = "conv2d_min_add_multiply_custom"

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, constant_value, bias_shape, scaling_factor):
        super(ModelNew, self).__init__()

    def forward(self, x, weight, conv_bias_opt, post_bias, constant_value, scaling_factor):
        # Do conv2d on PyTorch/NPU, fuse the remaining ops in custom kernel.
        conv_b = conv_bias_opt if conv_bias_opt is not None else None
        x = F.conv2d(x, weight, conv_b, stride=1, padding=0, dilation=1)

        # Expand post_bias to full shape to avoid broadcast inside kernel.
        # Expected original shape: [Cout, 1, 1]
        # Expand to [1, Cout, 1, 1] then to x.shape
        if post_bias.dim() == 3:
            post_bias_ex = post_bias.unsqueeze(0).expand(x.size(0), -1, x.size(2), x.size(3))
        else:
            post_bias_ex = post_bias.expand_as(x)

        cv = float(constant_value.reshape(-1)[0].item())
        sf = float(scaling_factor.reshape(-1)[0].item())

        fn = getattr(custom_ops_lib, _FN)
        # weight/conv_bias are passed for API compatibility, kernel ignores them.
        return fn(x, weight, conv_bias_opt, post_bias_ex, cv, sf)
'''
