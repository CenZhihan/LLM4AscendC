kernel_src="""
#include "kernel_operator.h"

class KernelConvStandard2d {
public:
    __aicore__ inline KernelConvStandard2d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t k, uint32_t r, uint32_t s,
                                uint32_t ho, uint32_t wo)
    {
        n_ = n; c_ = c; h_ = h; w_in_ = w_in;
        k_ = k; r_ = r; s_ = s; ho_ = ho; wo_ = wo;

        totalX_ = static_cast<uint64_t>(n_) * c_ * h_ * w_in_;
        totalW_ = static_cast<uint64_t>(k_) * c_ * r_ * s_;
        totalY_ = static_cast<uint64_t>(n_) * k_ * ho_ * wo_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalX_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, totalW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalY_);

        wBasePtr_ = (__gm__ float*)w;

        // Initialize queues: one row buffer in, one row buffer out
        pipe_.InitBuffer(inQueueX_, 1, wo_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, wo_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || k_ == 0 || c_ == 0 || h_ == 0 || w_in_ == 0) return;
        if (ho_ == 0 || wo_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t tasks = static_cast<uint64_t>(n_) * k_;
        for (uint64_t t = blockIdx; t < tasks; t += blockNum) {
            const uint32_t n_idx = static_cast<uint32_t>(t / k_);
            const uint32_t k_idx = static_cast<uint32_t>(t % k_);
            ComputeOneNK(n_idx, k_idx);
        }
    }

private:
    __aicore__ inline void ComputeOneNK(uint32_t n_idx, uint32_t k_idx)
    {
        const uint64_t wOcStride = static_cast<uint64_t>(c_) * r_ * s_;
        const uint64_t wOcBase = static_cast<uint64_t>(k_idx) * wOcStride;

        for (uint32_t hi = 0; hi < ho_; ++hi) {
            // y row buffer
            AscendC::LocalTensor<float> yRow = outQueueY_.AllocTensor<float>();
            AscendC::Duplicate(yRow, 0.0f, static_cast<int32_t>(wo_));

            for (uint32_t ci = 0; ci < c_; ++ci) {
                for (uint32_t rr = 0; rr < r_; ++rr) {
                    const uint32_t in_h = hi + rr;  // stride=1, dilation=1, pad=0
                    for (uint32_t ss = 0; ss < s_; ++ss) {
                        // Weight index for current (k_idx, ci, rr, ss)
                        const uint64_t wIdx = wOcBase + static_cast<uint64_t>(ci) * (r_ * s_) + static_cast<uint64_t>(rr) * s_ + ss;
                        // Load weight scalar directly from GM
                        float w_val = wBasePtr_[wIdx];

                        // Load input contiguous row segment of length wo_ starting at column = ss
                        const uint32_t start_col = ss;
                        const uint64_t xBaseIdx =
                            (static_cast<uint64_t>(n_idx) * c_ + ci) * h_ * w_in_
                            + static_cast<uint64_t>(in_h) * w_in_
                            + start_col;

                        AscendC::LocalTensor<float> xRow = inQueueX_.AllocTensor<float>();
                        AscendC::DataCopy(xRow, xGm_[xBaseIdx], wo_);

                        // yRow += xRow * w_val
                        AscendC::Axpy(yRow, xRow, w_val, static_cast<int32_t>(wo_));

                        inQueueX_.FreeTensor(xRow);
                    }
                }
            }

            // Store y row to GM
            const uint64_t yBaseIdx =
                (static_cast<uint64_t>(n_idx) * k_ + k_idx) * ho_ * wo_
                + static_cast<uint64_t>(hi) * wo_;
            AscendC::DataCopy(yGm_[yBaseIdx], yRow, wo_);
            outQueueY_.FreeTensor(yRow);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    __gm__ float* wBasePtr_ {nullptr};

    uint32_t n_{0}, c_{0}, h_{0}, w_in_{0};
    uint32_t k_{0}, r_{0}, s_{0};
    uint32_t ho_{0}, wo_{0};

    uint64_t totalX_{0}, totalW_{0}, totalY_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2d op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.k, tiling_data.r, tiling_data.s,
            tiling_data.ho, tiling_data.wo);
    op.Process();
}

#include <torch/library.h>

#include "pytorch_npu_helper.hpp"

at::Tensor conv2d_std_square_asym_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    TORCH_CHECK(x.dim() == 4, \"x must be 4D [N,C,H,W]\");
    TORCH_CHECK(weight.dim() == 4, \"weight must be 4D [K,C,R,S]\");
    TORCH_CHECK(x.scalar_type() == at::kFloat && weight.scalar_type() == at::kFloat,
                \"Only float supported\");

    const int64_t N = x.size(0);
    const int64_t Cx = x.size(1);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);

    const int64_t K = weight.size(0);
    const int64_t Cw = weight.size(1);
    const int64_t R = weight.size(2);
    const int64_t S = weight.size(3);

    TORCH_CHECK(Cx == Cw, \"Input C and Weight C must match\");
    TORCH_CHECK(H >= R && W >= S, \"Kernel must fit into input without padding\");

    const int64_t Ho = H - R + 1;
    const int64_t Wo = W - S + 1;

    at::Tensor y = at::empty({N, K, Ho, Wo}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard2dSquareInputAsymmetricKernelCustom, x, weight, y);
    return y;
}

    m.impl(\"conv_standard_2d_square_input_asymmetric_kernel_custom\", &conv2d_std_square_asym_impl_npu);
}
