static ge::graphStatus TilingFunc(gert::TilingContext *context)

    TripletMarginLossCustomTilingData tiling;
    const gert::Shape *anchorShape = context->GetInputShape(0);
    const std::vector<int64_t> &dims = anchorShape->GetOriginShape().GetDims();
    uint32_t rows = 1;
    uint32_t cols = 1;
    if (dims.size() == 1) {
        rows = 1;
        cols = static_cast<uint32_t>(dims[0]);
    } else if (dims.size() >= 2) {
        rows = static_cast<uint32_t>(dims[dims.size() - 2]);
        cols = static_cast<uint32_t>(dims[dims.size() - 1]);
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *marginPtr = attrs->GetAttrPointer<float>(0);
    const float margin = (marginPtr != nullptr) ? *marginPtr : 1.0f;

    // inner must be 32-byte aligned (for float: 8 elements)
    const uint32_t typeSize = 4;
    const uint32_t innerAligned = AlignUp(cols, 32u / typeSize);
    std::vector<int64_t> meanShape = {1, static_cast<int64_t>(innerAligned)};
    ge::Shape shape(meanShape);

    uint32_t minValue = 0;
    uint32_t maxValue = 0;
    AscendC::GetMeanMaxMinTmpSize(shape, typeSize, false, maxValue, minValue);

    auto platformInfo = context->GetPlatformInfo();
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(platformInfo);
    uint64_t tailSize = 0;
    ascendcPlatform.GetCoreMemSize(platform_ascendc::CoreMemType::UB, tailSize);
    uint32_t tmpSize = static_cast<uint32_t>(tailSize >= maxValue ? maxValue : tailSize);
    if (tmpSize < minValue) tmpSize = minValue;

    context->SetBlockDim(1);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tmpBufferSize(tmpSize);
    tiling.set_margin(margin);

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchorGm, GM_ADDR positiveGm, GM_ADDR negativeGm,
                                GM_ADDR yGm, uint32_t rows, uint32_t cols,
                                uint32_t tmpBufferSize, float margin)
    {
        rows_ = rows;
        cols_ = cols;
        margin_ = margin;
        tmpSize_ = tmpBufferSize;

        aGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(anchorGm), rows_ * cols_);
        pGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(positiveGm), rows_ * cols_);
        nGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(negativeGm), rows_ * cols_);
        yGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(yGm), 1);

        // Buffers: one row per queue
        pipe_.InitBuffer(inQueueA_, 1, cols_ * sizeof(float));
        pipe_.InitBuffer(inQueueP_, 1, cols_ * sizeof(float));
        pipe_.InitBuffer(inQueueN_, 1, cols_ * sizeof(float));
        pipe_.InitBuffer(tmpVec_, 1, cols_ * sizeof(float));
        pipe_.InitBuffer(tmpBufMean_, 1, tmpSize_);
        pipe_.InitBuffer(scalarBuf_, 1, 4 * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) {
            return;
        }

        // Prepare accumulator for mean reduction over rows
        LocalTensor<float> sumLocal = scalarBuf_.AllocTensor<float>();
        float zero = 0.0f;
        Duplicate(sumLocal, zero, 1);

        for (uint32_t i = 0; i < rows_; ++i) {
            // Load rows
            LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            LocalTensor<float> pLocal = inQueueP_.AllocTensor<float>();
            LocalTensor<float> nLocal = inQueueN_.AllocTensor<float>();
            DataCopy(aLocal, aGlobal_[i * cols_], cols_);
            DataCopy(pLocal, pGlobal_[i * cols_], cols_);
            DataCopy(nLocal, nGlobal_[i * cols_], cols_);
            inQueueA_.EnQue(aLocal);
            inQueueP_.EnQue(pLocal);
            inQueueN_.EnQue(nLocal);

            aLocal = inQueueA_.DeQue<float>();
            pLocal = inQueueP_.DeQue<float>();
            nLocal = inQueueN_.DeQue<float>();

            // Compute L2 distance between anchor and positive:
            // ||a - p|| = sqrt(sum((a^2 + p^2 - 2*a*p)))
            LocalTensor<float> tmpVec = tmpVec_.AllocTensor<float>(); // reuse for ab or intermediate
            // tmpVec = a * p  (ab)
            Mul(tmpVec, aLocal, pLocal, cols_);
            // aLocal = a^2
            Mul(aLocal, aLocal, aLocal, cols_);
            // pLocal = p^2
            Mul(pLocal, pLocal, pLocal, cols_);
            // aLocal = a^2 + p^2
            Add(aLocal, aLocal, pLocal, cols_);
            // tmpVec = (-2) * ab
            Muls(tmpVec, tmpVec, -2.0f, cols_);
            // aLocal = a^2 + p^2 - 2ab == (a - p)^2
            Add(aLocal, aLocal, tmpVec, cols_);

            // Mean over last axis (outter=1, inner=cols, n=cols)
            LocalTensor<uint8_t> sharedTmp = tmpBufMean_.AllocTensor<uint8_t>();
            LocalTensor<float> apSum = scalarBuf_[0];
            MeanParams mp;
            mp.outter = 1;
            mp.inner = cols_; // assume cols*sizeof(float) is 32-byte aligned
            mp.n = cols_;
            Mean<float>(apSum, aLocal, sharedTmp, mp);
            // scale mean to sum
            Muls(apSum, apSum, static_cast<float>(cols_), 1);
            // add eps and sqrt
            LocalTensor<float> epsT = scalarBuf_[1];
            float eps = 1.0e-6f;
            Duplicate(epsT, eps, 1);
            Add(apSum, apSum, epsT, 1);
            LocalTensor<float> apDist = scalarBuf_[2];
            Sqrt<float, false>(apDist, apSum, 1);

            tmpVec_.FreeTensor(tmpVec);
            tmpBufMean_.FreeTensor(sharedTmp);
            inQueueP_.FreeTensor(pLocal);

            // Compute L2 distance between anchor (reload) and negative:
            // Reuse pLocal storage to load anchor again for clean aLocal usage if needed
            // nLocal contains original negative row.

            // Reload anchor row to pLocal to preserve aLocal usage
            pLocal = inQueueP_.AllocTensor<float>();
            DataCopy(pLocal, aGlobal_[i * cols_], cols_);
            inQueueP_.EnQue(pLocal);
            pLocal = inQueueP_.DeQue<float>();

            // tmpVec = pLocal * nLocal  (ab for a and n)
            tmpVec = tmpVec_.AllocTensor<float>();
            Mul(tmpVec, pLocal, nLocal, cols_);
            // pLocal = a^2
            Mul(pLocal, pLocal, pLocal, cols_);
            // nLocal = n^2
            Mul(nLocal, nLocal, nLocal, cols_);
            // pLocal = a^2 + n^2
            Add(pLocal, pLocal, nLocal, cols_);
            // tmpVec = (-2) * (a*n)
            Muls(tmpVec, tmpVec, -2.0f, cols_);
            // pLocal = a^2 + n^2 - 2an == (a - n)^2
            Add(pLocal, pLocal, tmpVec, cols_);

            sharedTmp = tmpBufMean_.AllocTensor<uint8_t>();
            LocalTensor<float> anSum = scalarBuf_[3];
            mp.outter = 1;
            mp.inner = cols_;
            mp.n = cols_;
            Mean<float>(anSum, pLocal, sharedTmp, mp);
            Muls(anSum, anSum, static_cast<float>(cols_), 1);
            Duplicate(epsT, eps, 1);
            Add(anSum, anSum, epsT, 1);
            LocalTensor<float> anDist = scalarBuf_[1]; // reuse slot [1]
            Sqrt<float, false>(anDist, anSum, 1);

            tmpVec_.FreeTensor(tmpVec);
            tmpBufMean_.FreeTensor(sharedTmp);

            // delta = apDist - anDist + margin
            LocalTensor<float> negAn = scalarBuf_[3]; // reuse slot
            Muls(negAn, anDist, -1.0f, 1);
            Add(apDist, apDist, negAn, 1);
            LocalTensor<float> marginT = scalarBuf_[3]; // reuse slot
            Duplicate(marginT, margin_, 1);
            Add(apDist, apDist, marginT, 1);

            // hinge = max(delta, 0) = delta - min(delta, 0)
            LocalTensor<float> minPart = scalarBuf_[3]; // reuse slot
            Mins(minPart, apDist, 0.0f, 1);
            Muls(minPart, minPart, -1.0f, 1);
            Add(apDist, apDist, minPart, 1);

            // accumulate
            Add(sumLocal, sumLocal, apDist, 1);

            // Free tensors for this iteration
            inQueueA_.FreeTensor(aLocal);
            inQueueP_.FreeTensor(pLocal);
            inQueueN_.FreeTensor(nLocal);
        }

        // mean reduction over rows
        float invRows = 1.0f / static_cast<float>(rows_);
        Muls(sumLocal, sumLocal, invRows, 1);

        // Write result
        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        DataCopy(yLocal, sumLocal, 1);
        outQueueY_.EnQue<float>(yLocal);

        // CopyOut
        yLocal = outQueueY_.DeQue<float>();
        DataCopy(yGlobal_, yLocal, 1);
        outQueueY_.FreeTensor(yLocal);

        scalarBuf_.FreeTensor(sumLocal);
    }

private:
    TPipe pipe_;
    TQue<QuePosition::VECIN, 1> inQueueA_;
    TQue<QuePosition::VECIN, 1> inQueueP_;
    TQue<QuePosition::VECIN, 1> inQueueN_;
    TQue<QuePosition::VECCALC, 1> tmpVec_;
    TQue<QuePosition::VECCALC, 1> tmpBufMean_;
    TQue<QuePosition::VECCALC, 1> scalarBuf_;
    TQue<QuePosition::VECOUT, 1> outQueueY_;

    GlobalTensor<float> aGlobal_, pGlobal_, nGlobal_, yGlobal_;

    uint32_t rows_{0};
    uint32_t cols_{0};
    uint32_t tmpSize_{0};
    float margin_{1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchorGm, GM_ADDR positiveGm, GM_ADDR negativeGm,
                                                                 GM_ADDR yGm, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchorGm, positiveGm, negativeGm, yGm,
            tiling_data.rows, tiling_data.cols, tiling_data.tmpBufferSize, tiling_data.margin);
    op.Process();
}
