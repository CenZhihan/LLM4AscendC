#include "kernel_operator.h"
#include "mathematics/reduction.h" // depending on environment, ReduceSum may be declared in kernel_operator.h already

class KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom {
public:
    __aicore__ inline KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom() {}

    __aicore__ inline void Init(GM_ADDR y, GM_ADDR scale, GM_ADDR z,
                                uint32_t totalLength,
                                uint32_t n, uint32_t c,
                                uint32_t d, uint32_t h, uint32_t w,
                                uint32_t innerSpatial, uint32_t chunkLen,
                                float clampMin, float clampMax)
    {
        totalLen_ = totalLength;
        n_ = n; c_ = c; d_ = d; h_ = h; w_ = w;
        innerSpatial_ = innerSpatial;
        chunkLen_ = chunkLen;
        clampMin_ = clampMin;
        clampMax_ = clampMax;

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        scaleGm_.SetGlobalBuffer((__gm__ float*)scale, 1);
        zGm_.SetGlobalBuffer((__gm__ float*)z, n_ * c_);

        // Buffers: input chunk, reduce work buffer for ReduceSum, and a single-float output staging buffer.
        pipe_.InitBuffer(inQueueY_, 1, chunkLen_ * sizeof(float));
        pipe_.InitBuffer(workQueue_, 1, chunkLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueZ_, 1, sizeof(float));
        // A tiny queue for reading scale scalar
        pipe_.InitBuffer(scaleQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerSpatial_ == 0) return;

        // Load scale (y.max()) from GM
        float scaleVal = 1.0f;
        {
            AscendC::LocalTensor<float> sLocal = scaleQueue_.AllocTensor<float>();
            AscendC::DataCopy(sLocal, scaleGm_, 1);
            scaleQueue_.FreeTensor(sLocal);
            // There's no direct API to read a scalar from LocalTensor, so we duplicate through a temp buffer
            // As a workaround, move via Duplicate/DataCopy back is not necessary; assume compiler permits reading.
            // Alternatively, reuse ReduceSum on a single element to place it into acc and read back:
            AscendC::LocalTensor<float> dstDummy = outQueueZ_.AllocTensor<float>();
            AscendC::LocalTensor<float> workLocal = workQueue_.AllocTensor<float>();
            AscendC::ReduceSum(dstDummy, sLocal, workLocal, 1);
            scaleVal = AscendC::GetAccVal<float>();
            outQueueZ_.FreeTensor(dstDummy);
            workQueue_.FreeTensor(workLocal);
        }

        // Iterate over N and C, reduce over D*H*W
        for (uint32_t ni = 0; ni < n_; ++ni) {
            for (uint32_t ci = 0; ci < c_; ++ci) {
                const uint32_t base = (ni * c_ + ci) * innerSpatial_;
                float sumVal = 0.0f;

                // Accumulate over innerSpatial_ using ReduceSum on chunks
                for (uint32_t off = 0; off < innerSpatial_; off += chunkLen_) {
                    uint32_t len = innerSpatial_ - off;
                    if (len > chunkLen_) len = chunkLen_;

                    AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
                    AscendC::DataCopy(yLocal, yGm_[base + off], len);

                    AscendC::LocalTensor<float> dstDummy = outQueueZ_.AllocTensor<float>();
                    AscendC::LocalTensor<float> workLocal = workQueue_.AllocTensor<float>();
                    AscendC::ReduceSum(dstDummy, yLocal, workLocal, len);
                    float part = AscendC::GetAccVal<float>();

                    outQueueZ_.FreeTensor(dstDummy);
                    workQueue_.FreeTensor(workLocal);
                    inQueueY_.FreeTensor(yLocal);

                    sumVal += part;
                }

                float avg = (sumVal / static_cast<float>(innerSpatial_)) * scaleVal;
                if (avg < clampMin_) avg = clampMin_;
                if (avg > clampMax_) avg = clampMax_;

                // Write one scalar to GM at (ni, ci, 0, 0, 0) => contiguous index ni*C + ci
                AscendC::LocalTensor<float> outLocal = outQueueZ_.AllocTensor<float>();
                AscendC::Duplicate(outLocal, avg, 1);
                AscendC::DataCopy(zGm_[ni * c_ + ci], outLocal, 1);
                outQueueZ_.FreeTensor(outLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueY_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> workQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> scaleQueue_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueZ_;

    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> scaleGm_;
    AscendC::GlobalTensor<float> zGm_;

    uint32_t totalLen_{0};
    uint32_t n_{0}, c_{0}, d_{0}, h_{0}, w_{0};
    uint32_t innerSpatial_{0};
    uint32_t chunkLen_{128};
    float clampMin_{0.0f};
    float clampMax_{1.0f};
};

extern "C" __global__ __aicore__ void conv_transpose3d_multiply_max_global_avg_pool_clamp_custom(
    GM_ADDR y, GM_ADDR scale, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom op;
    op.Init(y, scale, z,
            tilingData.totalLength,
            tilingData.n, tilingData.c, tilingData.d, tilingData.h, tilingData.w,
            tilingData.innerSpatial, tilingData.chunkLen,
            tilingData.clampMin, tilingData.clampMax);
    if (TILING_KEY_IS(1) || true) {
        op.Process();
    }
}
