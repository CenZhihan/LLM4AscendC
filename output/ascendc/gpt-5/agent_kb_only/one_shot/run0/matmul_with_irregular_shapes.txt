host_tiling_src="""
#include <cstring>
#include "register/tilingdata_base.h"
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Retrieve M, K from A, and N from B
    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);

    if (shapeA->GetOriginShape().GetDimNum() != 2 || shapeB->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const int64_t M = shapeA->GetOriginShape().GetDim(0);
    const int64_t K = shapeA->GetOriginShape().GetDim(1);
    const int64_t Kb = shapeB->GetOriginShape().GetDim(0);
    const int64_t N = shapeB->GetOriginShape().GetDim(1);

    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());

    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);
    tiling.SetDim(1);
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBias(false);

    tiling.SetShape(static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K));
    tiling.SetOrgShape(static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K), static_cast<int32_t>(K));
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Get blockDim from tiler
    int32_t dim = 1, mDim = 1, nDim = 1;
    int retCore = tiling.GetCoreNum(dim, mDim, nDim);
    if (retCore != 0 || dim <= 0) {
        dim = 1;
    }
    context->SetBlockDim(static_cast<uint32_t>(dim));

    // Save tiling data into raw buffer
    auto *raw = context->GetRawTilingData();
    std::memcpy(raw->GetData(), &tilingData, sizeof(optiling::TCubeTiling));
    raw->SetDataSize(sizeof(optiling::TCubeTiling));

    // No workspace required by this kernel
    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;

    return ge::GRAPH_SUCCESS;
}

REGISTER_TILING_DATA_CLASS(MatmulWithIrregularShapesCustom, TCubeTiling)

}  // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }

    const int64_t M = a_shape->GetOriginShape().GetDim(0);
    const int64_t K = a_shape->GetOriginShape().GetDim(1);
    const int64_t Kb = b_shape->GetOriginShape().GetDim(0);
    const int64_t N = b_shape->GetOriginShape().GetDim(1);
    if (K != Kb) {
        return GRAPH_FAILED;
    }

    gert::Shape *c_shape = context->GetOutputShape(0);
    gert::Shape output;
    output.SetDimNum(2);
    output.SetDim(0, M);
    output.SetDim(1, N);
    *c_shape = output;

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Force output to float32 accumulation
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

typedef half A_T;
typedef half B_T;
typedef float C_T;
typedef float BiasT;

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM,
                                                                          GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // Define MatmulType
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, A_T> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, B_T> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

    set_atomic_none();

    // Load tiling from GM
    TCubeTiling tiling;
    auto tilingGMU32 = (__gm__ uint32_t*)tilingGM;
    auto tilingU32 = (uint32_t*)&tiling;
    for (int i = 0; i < sizeof(TCubeTiling) / sizeof(uint32_t); ++i) {
        *tilingU32++ = *tilingGMU32++;
    }

    // Initialize GM tensors
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;

    // Use conservative sizes derived from tiling layout info
    int32_t sizeA = tiling.ALayoutInfoB * tiling.ALayoutInfoS * tiling.ALayoutInfoN * tiling.ALayoutInfoG *
                    tiling.ALayoutInfoD * sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.BLayoutInfoS * tiling.BLayoutInfoN * tiling.BLayoutInfoG *
                    tiling.BLayoutInfoD * sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.CLayoutInfoN * tiling.CLayoutInfoG * tiling.CLayoutInfoS2 *
                    sizeof(C_T);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(cGM), sizeC);

    // Optional shared buffer hints
    tiling.shareMode = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize = 0;

    int offset_a = 0, offset_b = 0, offset_c = 0;

    AscendC::GlobalTensor<A_T> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal[offset_a].GetPhyAddr()),
                         tiling.ALayoutInfoS * tiling.ALayoutInfoN * tiling.ALayoutInfoG * tiling.ALayoutInfoD);
    AscendC::GlobalTensor<B_T> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal[offset_b].GetPhyAddr()),
                         tiling.BLayoutInfoS * tiling.BLayoutInfoN * tiling.BLayoutInfoG * tiling.BLayoutInfoD);
    AscendC::GlobalTensor<C_T> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal[offset_c].GetPhyAddr()),
                         tiling.CLayoutInfoN * tiling.CLayoutInfoG * tiling.CLayoutInfoS2);

    // Create and initialize Matmul instance
    matmul::Matmul<aType, bType, cType, biasType> mm;
    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
    mm.Init(&tiling);

    // Set inputs; no transpose
    mm.SetTensorA(gm_a[0], 0);
    mm.SetTensorB(gm_b[0], 0);
    // Optional: set workspace
    mm.SetWorkspace(workspaceGM, 0);

    // Execute computation
    // Single iteration without batching
    mm.IterateNBatch(1, 1, 1, false, 0, 0, 0);

    // Write back results
    // Matmul high-level APIs handle C write according to tiling.
    // End of kernel.
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor matmul_with_irregular_shapes_custom_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Cast to half for inputs; output is float
    at::Tensor A_half = A.to(at::kHalf);
    at::Tensor B_half = B.to(at::kHalf);
    // Output shape (M, N)
    auto sizesA = A.sizes();
    auto sizesB = B.sizes();
    TORCH_CHECK(sizesA.size() == 2 && sizesB.size() == 2, "Inputs must be 2-D.");
    TORCH_CHECK(sizesA[1] == sizesB[0], "K dimension mismatch.");
    at::Tensor C = at::empty({sizesA[0], sizesB[1]}, A.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnMatmulWithIrregularShapesCustom, A_half, B_half, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_with_irregular_shapes_custom", &matmul_with_irregular_shapes_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_with_irregular_shapes_custom", &matmul_with_irregular_shapes_custom_impl_npu,
