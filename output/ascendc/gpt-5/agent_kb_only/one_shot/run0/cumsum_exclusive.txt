host_operator_src="""

#include "cumsum_exclusive_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t AlignUp(uint32_t x, uint32_t a)
{
    if (a == 0) return x;
    uint32_t r = x % a;
    return r == 0 ? x : (x + a - r);
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model for this kernel (float):
//   inQueueX:    1 * tile
//   outQueueY:   1 * tile
//   tmpQueue:    1 * tile  (dst of CumSum, used for shift)
//   lastRowQueue:1 * tile  (CumSum API requires a tensor handle; config doesn't output)
// Total floats = 4 * tile => bytes = 16 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 8;               // CumSum inner*sizeof(T) must be multiple of 32 bytes
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumExclusiveCustomTilingData tiling;

    // Shape and rank
    const gert::Shape *shape = context->GetInputShape(0);
    const std::vector<int64_t> dims = shape->GetOriginShape().GetDims();
    const int64_t rank = static_cast<int64_t>(dims.size());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    if (dimPtr == nullptr) {
        return ge::GRAPH_FAILED;
    }
    int64_t dim = *dimPtr;
    // normalize dim
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) {
        return ge::GRAPH_FAILED;
    }

    // This kernel supports exclusive cumsum along the last axis only (dim == rank - 1).
    // Fallback/extension can be added if needed.
    if (dim != rank - 1) {
        // Not supported in this implementation.
        return ge::GRAPH_FAILED;
    }

    // Flatten to 2D: [outer, inner], where inner is last axis
    uint64_t outer64 = 1;
    for (int64_t i = 0; i < rank - 1; ++i) {
        outer64 *= static_cast<uint64_t>(dims[i]);
    }
    const uint32_t outer = static_cast<uint32_t>(outer64);
    const uint32_t inner = static_cast<uint32_t>(dims.back());

    // tileLen from UB budget: bytes = 16 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 16;  // elements
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    if (maxTile == 0) maxTile = ALIGN_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (inner < tileLen) {
        const uint32_t aligned = AlignDown(inner, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : inner;  // if inner < ALIGN_ELEMS, we will handle tail in kernel
    }

    // Choose blockDim based on number of rows (outer)
    uint32_t blockDim = outer;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_outer(outer);
    tiling.set_inner(inner);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumsumExclusiveCustom : public OpDef {
public:
    explicit CumsumExclusiveCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int();

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumsumExclusiveCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

// We implement exclusive cumsum along the last axis (inner dimension) for 2D flattened data.
// Tiling processes each row in segments of tileLen elements; for tail (< ALIGN_ELEMS), we use a small scalar loop.
// CumSum high-level API is used to generate inclusive prefix per aligned segment, then shifted to exclusive.

constexpr int32_t BUFFER_NUM = 1;
constexpr uint32_t ALIGN_ELEMS = 8; // CumSum inner must be multiple of 32 bytes for float

class KernelCumsumExclusiveCustom {
public:
    __aicore__ inline KernelCumsumExclusiveCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t inner,
                                uint32_t tileLength, int32_t dim)
    {
        outer_ = outer;
        inner_ = inner;
        tileLen_ = tileLength;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(outer_) * inner_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(outer_) * inner_);

        // UB queues
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(lastRowQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outer_ == 0 || inner_ == 0) return;

        // Only last axis supported in this kernel
        if (dim_ != 1) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < outer_; row += blockNum) {
            ProcessOneRow(row);
        }
    }

private:
    __aicore__ inline void ProcessOneRow(uint32_t row)
    {
        float acc = 0.0f;
        uint64_t base = static_cast<uint64_t>(row) * inner_;

        // Full tiles
        uint32_t processed = 0;
        while (processed + tileLen_ <= inner_) {
            ComputeAlignedTile(base + processed, tileLen_, acc);
            processed += tileLen_;
        }

        // One more aligned chunk if remaining >= ALIGN_ELEMS
        uint32_t remain = inner_ - processed;
        if (remain >= ALIGN_ELEMS) {
            uint32_t aligned = remain - (remain % ALIGN_ELEMS); // AlignDown(remain, ALIGN_ELEMS)
            ComputeAlignedTile(base + processed, aligned, acc);
            processed += aligned;
            remain = inner_ - processed;
        }

        // Final tiny tail (< ALIGN_ELEMS): scalar loop
        if (remain > 0) {
            ComputeTailScalar(base + processed, remain, acc);
        }
    }

    __aicore__ inline void ComputeAlignedTile(uint64_t gmOffset, uint32_t len, float &acc)
    {
        // Copy in
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);

        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> lastRowLocal = lastRowQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Inclusive CumSum for this aligned tile (outer=1, inner=len)
        static constexpr AscendC::CumSumConfig cumSumConfig{true, false, false}; // last axis, not reuse, no lastRow output
        const AscendC::CumSumInfo info{1U, len};
        AscendC::CumSum<float, cumSumConfig>(tmpLocal, lastRowLocal, xLocal, info);

        // sum of this tile = last element of inclusive prefix
        float sumBlock = tmpLocal.GetValue(len - 1U);

        // Convert inclusive to exclusive:
        // y[0] = 0; y[1..len-1] = tmp[0..len-2]
        yLocal.SetValue(0, 0.0f);
        if (len > 1U) {
            AscendC::DataCopy(yLocal[1U], tmpLocal, len - 1U);
        }

        // Add accumulated sum from previous tiles
        AscendC::Adds(yLocal, yLocal, acc, len);

        // Write back
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);

        // Update accumulator
        acc += sumBlock;

        // Free locals
        outQueueY_.FreeTensor(yLocal);
        lastRowQueue_.FreeTensor(lastRowLocal);
        tmpQueue_.FreeTensor(tmpLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeTailScalar(uint64_t gmOffset, uint32_t len, float &acc)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);

        for (uint32_t i = 0; i < len; ++i) {
            yLocal.SetValue(i, acc);
            float xi = xLocal.GetValue(i);
            acc += xi;
        }

        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);

        outQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> lastRowQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_ {0};
    uint32_t inner_ {0};
    uint32_t tileLen_ {ALIGN_ELEMS};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void cumsum_exclusive_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumExclusiveCustom op;
    op.Init(x, y,
            tiling_data.outer, tiling_data.inner,
            tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
"""
