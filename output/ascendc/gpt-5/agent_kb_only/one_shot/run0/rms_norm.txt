host_operator_src="""
#include "rms_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}
} // anonymous namespace

namespace optiling {

// UB model estimate:
// - inQueue: 1 * tileElems
// - outQueue: 1 * tileElems
// - tmp (square buffer): 1 * tileElems
// Total bytes ~= 12 * tileElems
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;   // favor larger DMA bursts
static constexpr uint32_t MAX_BLOCKS = 24;
static constexpr uint32_t MIN_BLOCKS = 8;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    RmsNormCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const uint64_t totalLen64 = xShape->GetOriginShape().GetShapeSize();
    const uint32_t totalLength = static_cast<uint32_t>(totalLen64);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float eps = *(attrs->GetAttrPointer<float>(0));
    const int32_t channels = *(attrs->GetAttrPointer<int32_t>(1));

    // Derive outerSize = totalLength / channels (rows count)
    uint32_t outerSize = (channels > 0) ? static_cast<uint32_t>(totalLength / static_cast<uint32_t>(channels)) : 0u;

    // Determine tileRows from UB budget: 12 bytes per element (3 buffers), elements per row = channels
    uint32_t maxTileElems = UB_BUDGET_BYTES / 12; // elements
    // Convert to rows
    uint32_t maxTileRows = (channels > 0) ? (maxTileElems / static_cast<uint32_t>(channels)) : 0u;
    if (maxTileRows == 0) {
        // Fallback to at least 1 row if channels is too large
        maxTileRows = 1;
    }

    // Align rows upward to favor larger bursts, but ensure at least 1 row and not exceeding outerSize
    uint32_t tileRows = maxTileRows;
    if (outerSize > 0 && tileRows > outerSize) tileRows = outerSize;

    // Try to align rows so that total elements per tile are aligned
    uint32_t tileElems = tileRows * static_cast<uint32_t>(channels);
    uint32_t alignedElems = AlignDown(tileElems, ALIGN_ELEMS);
    if (alignedElems >= static_cast<uint32_t>(channels)) {
        tileRows = alignedElems / static_cast<uint32_t>(channels);
    } else {
        tileRows = (tileRows > 0) ? tileRows : 1u;
    }
    if (tileRows == 0) tileRows = 1;

    // Grid dimension
    uint32_t numTiles = (tileRows == 0) ? 1u : CeilDiv(outerSize, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileRows(tileRows);
    tiling.set_channels(static_cast<uint32_t>(channels));
    tiling.set_outerSize(outerSize);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class RmsNormCustom : public OpDef {
public:
    explicit RmsNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("eps").AttrType(OPTIONAL).Float(1e-5f);
        this->Attr("num_features").AttrType(REQUIRED).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(RmsNormCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelRmsNormCustom {
public:
    __aicore__ inline KernelRmsNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t tileRows,
                                uint32_t channels,
                                uint32_t outerSize,
                                float eps)
    {
        totalLen_ = totalLength;
        tileRows_ = (tileRows == 0) ? 1u : tileRows;
        channels_ = channels;
        outerSize_ = outerSize;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        const uint32_t tileElems = tileRows_ * channels_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileElems * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileElems * sizeof(float)); // squares buffer
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || channels_ == 0 || outerSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride loop over rows
        const uint32_t strideRows = blockNum * tileRows_;
        uint32_t baseRow = blockIdx * tileRows_;
        if (baseRow >= outerSize_) return;

        // Prefetch first tile
        uint32_t curRows = outerSize_ - baseRow;
        if (curRows > tileRows_) curRows = tileRows_;
        CopyIn(baseRow, curRows);

        while (true) {
            const uint32_t nextBaseRow = baseRow + strideRows;
            uint32_t nextRows = 0;
            if (nextBaseRow < outerSize_) {
                nextRows = outerSize_ - nextBaseRow;
                if (nextRows > tileRows_) nextRows = tileRows_;
                CopyIn(nextBaseRow, nextRows);
            }

            Compute(curRows);
            CopyOut(baseRow, curRows);

            if (nextRows == 0) break;
            baseRow = nextBaseRow;
            curRows = nextRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t baseRow, uint32_t rows)
    {
        const uint32_t offsetElems = baseRow * channels_;
        const uint32_t lenElems = rows * channels_;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offsetElems], lenElems);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rows)
    {
        const uint32_t lenElems = rows * channels_;
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> sqLocal = tmpQueue_.AllocTensor<float>();

        // squares = x * x
        AscendC::Mul(sqLocal, xLocal, xLocal, static_cast<int32_t>(lenElems));

        // For each row, compute sum of squares, RMS, and scale
        for (uint32_t r = 0; r < rows; ++r) {
            const uint32_t rowOff = r * channels_;

            AscendC::LocalTensor<float> rowSq = sqLocal[rowOff];
            // ReduceSum over the row (channels_) elements
            AscendC::ReduceSum(rowSq, rowSq, static_cast<int32_t>(channels_));
            float sumsq = AscendC::GetAccVal<float>();

            float mean = sumsq / static_cast<float>(channels_);
            float inv_rms = 1.0f / sqrtf(mean + eps_);

            AscendC::LocalTensor<float> rowX = xLocal[rowOff];
            AscendC::LocalTensor<float> rowY = yLocal[rowOff];
            AscendC::Muls(rowY, rowX, inv_rms, static_cast<int32_t>(channels_));
        }

        tmpQueue_.FreeTensor(sqLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t baseRow, uint32_t rows)
    {
        const uint32_t offsetElems = baseRow * channels_;
        const uint32_t lenElems = rows * channels_;
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offsetElems], yLocal, lenElems);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileRows_{1};
    uint32_t channels_{0};
    uint32_t outerSize_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNormCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileRows,
            tiling_data.channels,
            tiling_data.outerSize,
            tiling_data.eps);
    op.Process();
}
"""

import custom_ops_lib

class ModelNew(torch.nn.Module):

        super(ModelNew, self).__init__()
        self.num_features = int(num_features)
        self.eps = float(eps)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
