#include "conv2d_scaling_min_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core usable UB
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) { return static_cast<uint32_t>((a + b - 1u) / b); }
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);      // [N, Cin, Hin, Win]
    const gert::Shape *wShape = context->GetInputShape(1);      // [Cout, Cin, kH, kW]

    const auto dimsX = xShape->GetStorageShape();
    const auto dimsW = wShape->GetStorageShape();
    if (dimsX.GetShapeSize() != 4 || dimsW.GetShapeSize() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(dimsX.GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(dimsX.GetDim(1));
    const uint32_t Hin = static_cast<uint32_t>(dimsX.GetDim(2));
    const uint32_t Win = static_cast<uint32_t>(dimsX.GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(dimsW.GetDim(0));
    const uint32_t kH   = static_cast<uint32_t>(dimsW.GetDim(2));
    const uint32_t kW   = static_cast<uint32_t>(dimsW.GetDim(3));

    // padding=0, stride=1, dilation=1
    const uint32_t OH = Hin >= kH ? (Hin - kH + 1) : 0;
    const uint32_t OW = Win >= kW ? (Win - kW + 1) : 0;

    const uint64_t patch = static_cast<uint64_t>(Cin) * kH * kW; // elements per single output patch
    // UB usage (floats):
    //   xPatch: patch
    //   wBlock: ocTile * patch
    //   bias:   ocTile
    //   acc:    ocTile
    // total floats = patch + ocTile * (patch + 2)
    // bytes = 4 * (patch + ocTile * (patch + 2))
    uint64_t budgetFloats = UB_BUDGET_BYTES / 4;
    uint32_t ocTile = 1;
    if (budgetFloats > patch) {
        uint64_t denom = (patch + 2);
        uint64_t maxTile = (budgetFloats - patch) / (denom == 0 ? 1 : denom);
        if (maxTile == 0) maxTile = 1;
        if (maxTile > Cout) maxTile = Cout;
        ocTile = static_cast<uint32_t>(maxTile);
    } else {
        ocTile = 1;
    }
    if (ocTile == 0) ocTile = 1;

    const uint64_t totalPixels = static_cast<uint64_t>(N) * OH * OW;
    uint32_t blockDim = static_cast<uint32_t>(totalPixels);
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u; // cap

    context->SetBlockDim(blockDim);

    Conv2dScalingMinCustomTilingData tiling;
    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Hin(Hin);
    tiling.set_Win(Win);
    tiling.set_Cout(Cout);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_OH(OH);
    tiling.set_OW(OW);
    tiling.set_ocTile(ocTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    const gert::Shape *xShape = context->GetInputShape(0); // [N, Cin, Hin, Win]
    const gert::Shape *wShape = context->GetInputShape(1); // [Cout, Cin, kH, kW]
    if (xShape->GetStorageShape().GetShapeSize() != 4 || wShape->GetStorageShape().GetShapeSize() != 4) {
        return GRAPH_FAILED;
    }
    const int64_t N   = xShape->GetStorageShape().GetDim(0);
    const int64_t Hin = xShape->GetStorageShape().GetDim(2);
    const int64_t Win = xShape->GetStorageShape().GetDim(3);
    const int64_t kH  = wShape->GetStorageShape().GetDim(2);
    const int64_t kW  = wShape->GetStorageShape().GetDim(3);

    const int64_t OH = Hin - kH + 1;
    const int64_t OW = Win - kW + 1;

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelConv2dScalingMinCustom {
public:
    __aicore__ inline KernelConv2dScalingMinCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scale, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Hin, uint32_t Win,
                                uint32_t Cout, uint32_t kH, uint32_t kW,
                                uint32_t OH, uint32_t OW, uint32_t ocTile)
    {
        N_ = N; Cin_ = Cin; Hin_ = Hin; Win_ = Win;
        Cout_ = Cout; kH_ = kH; kW_ = kW; OH_ = OH; OW_ = OW;
        ocTile_ = ocTile;
        patchSize_ = Cin_ * kH_ * kW_;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x),
                             static_cast<uint64_t>(N_) * Cin_ * Hin_ * Win_);
        wGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w),
                             static_cast<uint64_t>(Cout_) * Cin_ * kH_ * kW_);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y),
                             static_cast<uint64_t>(N_) * 1 * OH_ * OW_);
        scaleGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(scale), 1);

        hasBias_ = (b != nullptr);
        if (hasBias_) {
            bGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b), Cout_);
        }

        pipe_.InitBuffer(patchQ_, 1, patchSize_ * sizeof(float));
        pipe_.InitBuffer(wQ_,     1, static_cast<uint32_t>(ocTile_) * patchSize_ * sizeof(float));
        if (hasBias_) {
            pipe_.InitBuffer(bQ_,  1, static_cast<uint32_t>(ocTile_) * sizeof(float));
        }
        pipe_.InitBuffer(outQ_,   1, sizeof(float));
        pipe_.InitBuffer(tmpQ_,   1, sizeof(float)); // for loading scale

        // Load scale scalar from GM
        LocalTensor<float> sLocal = tmpQ_.AllocTensor<float>();
        DataCopy(sLocal, scaleGm_, 1);
        scale_ = sLocal.GetValue(0);
        tmpQ_.FreeTensor(sLocal);
    }

    __aicore__ inline void Process()
    {
        if (OH_ == 0 || OW_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        const uint64_t totalPixels = static_cast<uint64_t>(N_) * OH_ * OW_;

        for (uint64_t pix = blockIdx; pix < totalPixels; pix += blockNum) {
            uint64_t tmp = pix;
            uint32_t ow = static_cast<uint32_t>(tmp % OW_); tmp /= OW_;
            uint32_t oh = static_cast<uint32_t>(tmp % OH_); tmp /= OH_;
            uint32_t n  = static_cast<uint32_t>(tmp);

            // Load input patch for this (n, oh, ow): size = Cin * kH * kW
            LocalTensor<float> patchLocal = patchQ_.AllocTensor<float>();
            // Fill patchLocal using contiguous row copies for each (c, kh)
            for (uint32_t c = 0; c < Cin_; ++c) {
                for (uint32_t kh = 0; kh < kH_; ++c, ++kh) { // prevent compiler warnings with two vars
                }
            }
            // Corrected loop (avoid shadowing):
            for (uint32_t c0 = 0; c0 < Cin_; ++c0) {
                for (uint32_t kh0 = 0; kh0 < kH_; ++kh0) {
                    const uint64_t inBase = (((static_cast<uint64_t>(n) * Cin_ + c0) * Hin_ + (oh + kh0)) * Win_ + ow);
                    const uint32_t dstBase = (c0 * kH_ + kh0) * kW_;
                    DataCopy(patchLocal[dstBase], xGm_[inBase], kW_);
                }
            }
            patchQ_.EnQue(patchLocal);
            LocalTensor<float> patchUse = patchQ_.DeQue<float>();

            float accMin = 3.4028235e+38f; // +INF for float
            for (uint32_t ocStart = 0; ocStart < Cout_; ocStart += ocTile_) {
                const uint32_t curTile = (ocStart + ocTile_ <= Cout_) ? ocTile_ : (Cout_ - ocStart);

                // Load weight block: [curTile, Cin, kH, kW] contiguous
                LocalTensor<float> wLocal = wQ_.AllocTensor<float>();
                const uint64_t wOff = static_cast<uint64_t>(ocStart) * patchSize_;
                DataCopy(wLocal, wGm_[wOff], static_cast<uint32_t>(curTile) * patchSize_);
                wQ_.EnQue(wLocal);
                LocalTensor<float> wUse = wQ_.DeQue<float>();

                LocalTensor<float> bUse;
                if (hasBias_) {
                    LocalTensor<float> bLocal = bQ_.AllocTensor<float>();
                    DataCopy(bLocal, bGm_[ocStart], curTile);
                    bQ_.EnQue(bLocal);
                    bUse = bQ_.DeQue<float>();
                }

                for (uint32_t i = 0; i < curTile; ++i) {
                    float acc = hasBias_ ? bUse.GetValue(i) : 0.0f;
                    const uint32_t wBase = i * patchSize_;
                    for (uint32_t p = 0; p < patchSize_; ++p) {
                        acc = acc + wUse.GetValue(wBase + p) * patchUse.GetValue(p);
                    }
                    acc = acc * scale_;
                    accMin = (acc < accMin) ? acc : accMin;
                }

                if (hasBias_) {
                    bQ_.FreeTensor(bUse);
                }
                wQ_.FreeTensor(wUse);
            }

            // Write out accMin to y[n, 0, oh, ow]
            LocalTensor<float> yLocal = outQ_.AllocTensor<float>();
            yLocal.SetValue(0, accMin);
            const uint64_t yOff = (((static_cast<uint64_t>(n) * 1) + 0) * OH_ + oh) * OW_ + ow;
            DataCopy(yGm_[yOff], yLocal, 1);
            outQ_.FreeTensor(yLocal);

            patchQ_.FreeTensor(patchUse);
        }
    }

private:
    // Dimensions and tiling
    uint32_t N_, Cin_, Hin_, Win_;
    uint32_t Cout_, kH_, kW_;
    uint32_t OH_, OW_;
    uint32_t ocTile_;
    uint32_t patchSize_;

    // GM tensors
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> scaleGm_;
    GlobalTensor<float> yGm_;

    // UB management
    TPipe pipe_;
    TQue<QuePosition::VECIN, 1> patchQ_;
    TQue<QuePosition::VECIN, 1> wQ_;
    TQue<QuePosition::VECIN, 1> bQ_;
    TQue<QuePosition::VECOUT, 1> outQ_;
    TQue<QuePosition::VECCALC, 1> tmpQ_;

    // flags and scalars
    bool hasBias_{false};
    float scale_{1.0f};
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                GM_ADDR scale, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv2dScalingMinCustom op;
    op.Init(x, weight, bias, scale, y,
            tilingData.N, tilingData.Cin, tilingData.Hin, tilingData.Win,
            tilingData.Cout, tilingData.kH, tilingData.kW,
            tilingData.OH, tilingData.OW, tilingData.ocTile);
    op.Process();
}

#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv2d_scaling_min_custom_impl_npu(const at::Tensor& x,
                                              const at::Tensor& weight,
                                              c10::optional<at::Tensor> bias_opt,
                                              const at::Tensor& scale)
{
    TORCH_CHECK(x.dim() == 4, "x must be 4D NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be 4D [Cout, Cin, kH, kW]");
    TORCH_CHECK(scale.numel() == 1, "scale must be a scalar tensor");

    const int64_t N   = x.size(0);
    const int64_t Hin = x.size(2);
    const int64_t Win = x.size(3);
    const int64_t kH  = weight.size(2);
    const int64_t kW  = weight.size(3);
    const int64_t OH  = Hin - kH + 1;
    const int64_t OW  = Win - kW + 1;
    TORCH_CHECK(OH >= 0 && OW >= 0, "Invalid kernel size relative to input size");

    at::Tensor y = at::empty({N, 1, OH, OW}, x.options());

    if (bias_opt.has_value()) {
        EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, bias_opt.value(), scale, y);
    } else {
        at::Tensor empty_bias;
        EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, empty_bias, scale, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_scaling_min_custom", &conv2d_scaling_min_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
