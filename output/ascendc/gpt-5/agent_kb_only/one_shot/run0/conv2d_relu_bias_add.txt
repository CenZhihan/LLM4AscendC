kernel_src="""

#include "kernel_operator.h"

class KernelConv2dReluBiasAddCustom {
public:
    __aicore__ inline KernelConv2dReluBiasAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR cb, GM_ADDR pb, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Hi, uint32_t Wi, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Hi_ = Hi; Wi_ = Wi; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * Cin_ * Hi_ * Wi_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, Cout_ * Cin_ * Kh_ * Kw_);
        cbGm_.SetGlobalBuffer((__gm__ float*)cb, Cout_);
        pbGm_.SetGlobalBuffer((__gm__ float*)pb, Cout_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * Cout_ * Ho_ * Wo_);

        // Allocate tiny scalar buffers for streaming compute
        pipe_.InitBuffer(xScalarQ_, 2, sizeof(float));
        pipe_.InitBuffer(wScalarQ_, 2, sizeof(float));
        pipe_.InitBuffer(tmpScalarQ_, 2, sizeof(float));
        pipe_.InitBuffer(accScalarQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (Ho_ == 0 || Wo_ == 0 || Cout_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalPos = static_cast<uint64_t>(N_) * static_cast<uint64_t>(Ho_) * static_cast<uint64_t>(Wo_);
        for (uint64_t idx = blockIdx; idx < totalPos; idx += blockNum) {
            // decode n, oh, ow
            const uint64_t oneHW = static_cast<uint64_t>(Ho_) * static_cast<uint64_t>(Wo_);
            const uint32_t n  = static_cast<uint32_t>(idx / oneHW);
            const uint64_t rem= idx % oneHW;
            const uint32_t oh = static_cast<uint32_t>(rem / Wo_);
            const uint32_t ow = static_cast<uint32_t>(rem % Wo_);

            for (uint32_t co = 0; co < Cout_; ++co) {
                AscendC::LocalTensor<float> acc = accScalarQ_.AllocTensor<float>();
                AscendC::Duplicate(acc, 0.0f, 1);

                // Dot product over Cin * Kh * Kw
                for (uint32_t ci = 0; ci < Cin_; ++ci) {
                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        const uint32_t ih = oh + kh;
                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                            const uint32_t iw = ow + kw;

                            // x index: ((n * Cin + ci) * Hi + ih) * Wi + iw
                            const uint64_t xoff = (((static_cast<uint64_t>(n) * Cin_ + ci) * Hi_ + ih) * Wi_ + iw);

                            // w index: (((co * Cin + ci) * Kh + kh) * Kw + kw)
                            const uint64_t woff = ((((static_cast<uint64_t>(co) * Cin_ + ci) * Kh_ + kh) * Kw_ + kw));

                            AscendC::LocalTensor<float> xloc = xScalarQ_.AllocTensor<float>();
                            AscendC::LocalTensor<float> wloc = wScalarQ_.AllocTensor<float>();
                            AscendC::LocalTensor<float> tmp  = tmpScalarQ_.AllocTensor<float>();

                            AscendC::DataCopy(xloc, xGm_[xoff], 1);
                            AscendC::DataCopy(wloc, wGm_[woff], 1);

                            AscendC::Mul(tmp, xloc, wloc, 1);
                            AscendC::Add(acc, acc, tmp, 1);

                            tmpScalarQ_.FreeTensor(tmp);
                            wScalarQ_.FreeTensor(wloc);
                            xScalarQ_.FreeTensor(xloc);
                        }
                    }
                }

                // Add conv bias
                {
                    AscendC::LocalTensor<float> b = wScalarQ_.AllocTensor<float>();
                    AscendC::DataCopy(b, cbGm_[co], 1);
                    AscendC::Add(acc, acc, b, 1);
                    wScalarQ_.FreeTensor(b);
                }

                // ReLU
                AscendC::Relu(acc, acc, 1);

                // Add post bias [Cout,1,1] -> [Cout]
                {
                    AscendC::LocalTensor<float> pb = wScalarQ_.AllocTensor<float>();
                    AscendC::DataCopy(pb, pbGm_[co], 1);
                    AscendC::Add(acc, acc, pb, 1);
                    wScalarQ_.FreeTensor(pb);
                }

                // Store y at ((n * Cout + co) * Ho + oh) * Wo + ow
                const uint64_t yoff = (((static_cast<uint64_t>(n) * Cout_ + co) * Ho_ + oh) * Wo_ + ow);
                AscendC::DataCopy(yGm_[yoff], acc, 1);

                accScalarQ_.FreeTensor(acc);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> xScalarQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> wScalarQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpScalarQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accScalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> cbGm_;
    AscendC::GlobalTensor<float> pbGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Hi_{0}, Wi_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
};

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x, GM_ADDR weight,
                                                                   GM_ADDR conv_bias, GM_ADDR post_bias,
                                                                   GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConv2dReluBiasAddCustom op;
    op.Init(x, weight, conv_bias, post_bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.Hi, tiling_data.Wi, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo);
    op.Process();
}
