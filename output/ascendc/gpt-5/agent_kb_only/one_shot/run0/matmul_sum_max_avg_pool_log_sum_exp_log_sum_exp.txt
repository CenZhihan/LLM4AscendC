kernel_src="""

#include "kernel_operator.h"
#include "lib/ascendc_math.h"

using namespace AscendC;

class KernelMatmulSumMaxAvgPoolLogSumExpLogSumExp {
public:
    __aicore__ inline KernelMatmulSumMaxAvgPoolLogSumExpLogSumExp() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                GM_ADDR workspace,
                                uint32_t N, uint32_t K, uint32_t O,
                                uint32_t tileKSeg, uint32_t tileOSeg)
    {
        N_ = N; K_ = K; O_ = O;
        tileK_ = tileKSeg; if (tileK_ == 0) tileK_ = K_;
        tileO_ = tileOSeg; if (tileO_ == 0) tileO_ = 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(O_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_); // y is [N,1], stride 1

        // Workspace layout: [c (K floats)] + [beta (1 float)]
        cGm_.SetGlobalBuffer((__gm__ float*)workspace, K_);
        betaGm_.SetGlobalBuffer((__gm__ float*)(workspace + K_ * sizeof(float)), 1);

        // Buffers for phases (allocated conservatively; total under UB budget)
        // Phase A (build c, beta)
        pipe_.InitBuffer(wRowBuf_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(cTileBuf_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(workBuf_, 1, tileK_ * sizeof(float));

        // Phase B (per-row dot): buffers sized by K
        pipe_.InitBuffer(xRowBuf_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(cFullBuf_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(redBuf_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(yScalarBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || K_ == 0) return;

        // Step 1: Build c[k] = sum_{o} w[o, k] and beta = sum_j b[j]
        BuildCAndBeta();

        // Step 2: For each row i: y[i] = dot(x[i, :], c[:]) + beta
        // Load c full once per row (simple and robust).
        for (uint32_t i = 0; i < N_; ++i) {
            ComputeRow(i);
        }
    }

private:
    __aicore__ inline void BuildCAndBeta()
    {
        // Accumulate c in tiles over K to limit UB usage.
        for (uint32_t k0 = 0; k0 < K_; k0 += tileK_) {
            const uint32_t curK = (k0 + tileK_ <= K_) ? tileK_ : (K_ - k0);

            LocalTensor<float> cTile = cTileBuf_.AllocTensor<float>();
            // zero init cTile
            float zero = 0.0f;
            Duplicate(cTile, zero, curK);

            // Sum W over O dimension in stripes of tileO_
            for (uint32_t o0 = 0; o0 < O_; o0 += tileO_) {
                const uint32_t curO = (o0 + tileO_ <= O_) ? tileO_ : (O_ - o0);
                for (uint32_t r = 0; r < curO; ++r) {
                    // Load one row segment W[o0 + r, k0 : k0+curK]
                    const uint64_t gmOffset = (static_cast<uint64_t>(o0 + r) * K_) + k0;
                    LocalTensor<float> wRow = wRowBuf_.AllocTensor<float>();
                    DataCopy(wRow, wGm_[gmOffset], curK);

                    // cTile += wRow
                    Add(cTile, cTile, wRow, curK);

                    wRowBuf_.FreeTensor(wRow);
                }
            }

            // Store cTile segment to GM workspace
            DataCopy(cGm_[k0], cTile, curK);
            cTileBuf_.FreeTensor(cTile);
        }

        // Sum beta = sum_j b[j]
        float betaVal = 0.0f;
        // Process b in chunks to avoid large buffers; reuse wRowBuf_
        const uint32_t bSeg = 1024;
        for (uint32_t o0 = 0; o0 < O_; o0 += bSeg) {
            const uint32_t curO = (o0 + bSeg <= O_) ? bSeg : (O_ - o0);

            LocalTensor<float> bTile = wRowBuf_.AllocTensor<float>();
            DataCopy(bTile, bGm_[o0], curO);

            // ReduceSum across curO elements
            LocalTensor<float> tmp = workBuf_.AllocTensor<float>();
            // dst can reuse tmp buffer; length parameter as curO
            ReduceSum(tmp, bTile, tmp, curO);
            float part = GetAccVal<float>();
            betaVal += part;

            workBuf_.FreeTensor(tmp);
            wRowBuf_.FreeTensor(bTile);
        }

        // Store beta to workspace
        LocalTensor<float> ySc = yScalarBuf_.AllocTensor<float>();
        Duplicate(ySc, betaVal, 1);
        DataCopy(betaGm_, ySc, 1);
        yScalarBuf_.FreeTensor(ySc);
    }

    __aicore__ inline void ComputeRow(uint32_t rowIdx)
    {
        // Load x row
        LocalTensor<float> xRow = xRowBuf_.AllocTensor<float>();
        const uint64_t xOffset = static_cast<uint64_t>(rowIdx) * K_;
        DataCopy(xRow, xGm_[xOffset], K_);

        // Load c full
        LocalTensor<float> cFull = cFullBuf_.AllocTensor<float>();
        DataCopy(cFull, cGm_, K_);

        // tmp = xRow * cFull
        LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>();
        Mul(tmp, xRow, cFull, K_);

        // Sum over K to scalar
        LocalTensor<float> work = redBuf_.AllocTensor<float>();
        ReduceSum(work, tmp, work, K_);
        float dotVal = GetAccVal<float>();

        // Load beta and add
        LocalTensor<float> betaLocal = yScalarBuf_.AllocTensor<float>();
        DataCopy(betaLocal, betaGm_, 1);
        // betaLocal[0] is beta; y = dot + beta
        LocalTensor<float> yLocal = yScalarBuf_.AllocTensor<float>();
        Duplicate(yLocal, dotVal, 1);
        Add(yLocal, yLocal, betaLocal, 1);

        // Store to y GM
        DataCopy(yGm_[rowIdx], yLocal, 1);

        // Free tensors
        yScalarBuf_.FreeTensor(yLocal);
        yScalarBuf_.FreeTensor(betaLocal);
        redBuf_.FreeTensor(work);
        tmpBuf_.FreeTensor(tmp);
        cFullBuf_.FreeTensor(cFull);
        xRowBuf_.FreeTensor(xRow);
    }

private:
    TPipe pipe_;
    // GM tensors
    GlobalTensor<float> xGm_, wGm_, bGm_, yGm_;
    GlobalTensor<float> cGm_, betaGm_;

    // Local buffers
    TQue<TPosition::VECIN, 1>  wRowBuf_;
    TQue<TPosition::VECCALC, 1> cTileBuf_;
    TQue<TPosition::VECCALC, 1> workBuf_;

    TQue<TPosition::VECIN, 1>  xRowBuf_;
    TQue<TPosition::VECIN, 1>  cFullBuf_;
    TQue<TPosition::VECCALC, 1> tmpBuf_;
    TQue<TPosition::VECCALC, 1> redBuf_;
    TQue<TPosition::VECOUT, 1> yScalarBuf_;

    uint32_t N_{0}, K_{0}, O_{0};
    uint32_t tileK_{0}, tileO_{0};
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelMatmulSumMaxAvgPoolLogSumExpLogSumExp op;
    op.Init(x, w, b, y, workspace,
            tiling_data.batchN, tiling_data.inK, tiling_data.outO,
            tiling_data.tileKSeg, tiling_data.tileOSeg);
    op.Process();
}
