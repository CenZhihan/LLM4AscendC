#include "kernel_operator.h"

class KernelConvStandard3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR xGM, GM_ADDR wGM, GM_ADDR yGM,
                                uint32_t n, uint32_t cIn, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t cOut, uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t outD, uint32_t outH, uint32_t outW)
    {
        n_ = n; cIn_ = cIn; d_ = d; h_ = h; w_ = w;
        cOut_ = cOut; kD_ = kD; kH_ = kH; kW_ = kW;
        outD_ = outD; outH_ = outH; outW_ = outW;

        const uint64_t xSize = static_cast<uint64_t>(n_) * cIn_ * d_ * h_ * w_;
        const uint64_t wSize = static_cast<uint64_t>(cOut_) * cIn_ * kD_ * kH_ * kW_;
        const uint64_t ySize = static_cast<uint64_t>(n_) * cOut_ * outD_ * outH_ * outW_;

        xGm_.SetGlobalBuffer((__gm__ float*)xGM, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)wGM, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)yGM, ySize);

        // Prepare UB buffers
        const uint32_t weightElems = cOut_ * cIn_ * kD_ * kH_ * kW_;
        pipe_.InitBuffer(wBuf_, 1, weightElems * sizeof(float));
        pipe_.InitBuffer(segBuf_, 1, kW_ * sizeof(float));     // small 1x kW segment buffer (kW=3)
        pipe_.InitBuffer(accBuf_, 1, cOut_ * sizeof(float));   // accumulators for one output position
        pipe_.InitBuffer(scalarBuf_, 1, 1 * sizeof(float));    // scalar staging for GM write
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cIn_ == 0 || cOut_ == 0 || outD_ == 0 || outH_ == 0 || outW_ == 0) {
            return;
        }

        const uint64_t totalPositions = static_cast<uint64_t>(n_) * outD_ * outH_ * outW_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Load all weights once into UB
        AscendC::LocalTensor<float> wLocal = wBuf_.AllocTensor<float>();
        const uint32_t weightElems = cOut_ * cIn_ * kD_ * kH_ * kW_;
        AscendC::DataCopy(wLocal, wGm_, weightElems);

        // Allocate reusable local buffers
        AscendC::LocalTensor<float> segLocal = segBuf_.AllocTensor<float>();   // length kW (3)
        AscendC::LocalTensor<float> accLocal = accBuf_.AllocTensor<float>();   // length cOut
        AscendC::LocalTensor<float> oneLocal = scalarBuf_.AllocTensor<float>(); // length 1

        // Precompute strides for indexing
        const uint64_t xStrideC = static_cast<uint64_t>(d_) * h_ * w_;
        const uint64_t xStrideD = static_cast<uint64_t>(h_) * w_;
        const uint64_t xStrideH = static_cast<uint64_t>(w_);
        const uint64_t xStrideW = 1ULL;

        const uint64_t yStrideC = static_cast<uint64_t>(outD_) * outH_ * outW_;
        const uint64_t yStrideD = static_cast<uint64_t>(outH_) * outW_;
        const uint64_t yStrideH = static_cast<uint64_t>(outW_);
        const uint64_t yStrideW = 1ULL;

        const uint32_t kSize = cIn_ * kD_ * kH_ * kW_; // 81

        // Grid-stride over spatial positions
        for (uint64_t p = blockIdx; p < totalPositions; p += blockNum) {
            // Map linear position to (n, od, oh, ow)
            uint64_t tmp = p;
            uint32_t ow = static_cast<uint32_t>(tmp % outW_);
            tmp /= outW_;
            uint32_t oh = static_cast<uint32_t>(tmp % outH_);
            tmp /= outH_;
            uint32_t od = static_cast<uint32_t>(tmp % outD_);
            uint32_t n  = static_cast<uint32_t>(tmp / outD_);

            // Zero accumulators
            for (uint32_t co = 0; co < cOut_; ++co) {
                accLocal.SetValue(co, 0.0f);
            }

            // Convolution accumulation: stride=1, padding=0, dilation=1
            uint32_t kIndexBase = 0; // track flattened kernel index over [ci,kz,ky,kx]
            for (uint32_t ci = 0; ci < cIn_; ++ci) {
                for (uint32_t kz = 0; kz < kD_; ++kz) {
                    const uint32_t id = od + kz;  // within input depth
                    for (uint32_t ky = 0; ky < kH_; ++ky) {
                        const uint32_t ih = oh + ky; // within input height

                        // Copy a contiguous kW segment from input into segLocal
                        const uint32_t iw = ow;  // base input w for kx=0
                        const uint64_t xBaseOff =
                            static_cast<uint64_t>(n) * (cIn_ * xStrideC) +
                            static_cast<uint64_t>(ci) * xStrideC +
                            static_cast<uint64_t>(id) * xStrideD +
                            static_cast<uint64_t>(ih) * xStrideH +
                            static_cast<uint64_t>(iw) * xStrideW;

                        AscendC::DataCopy(segLocal, xGm_[xBaseOff], kW_);

                        // For each output channel, accumulate dot(segLocal, corresponding weight slice)
                        for (uint32_t co = 0; co < cOut_; ++co) {
                            float acc = accLocal.GetValue(co);
                            // weight offset for this (co, kIndexBase)
                            const uint32_t wOff = co * kSize + kIndexBase;
                            // kx loop: kW contiguous
                            for (uint32_t kx = 0; kx < kW_; ++kx) {
                                const float xv = segLocal.GetValue(kx);
                                const float wv = wLocal.GetValue(wOff + kx);
                                acc += xv * wv;
                            }
                            accLocal.SetValue(co, acc);
                        }

                        kIndexBase += kW_;
                    }
                }
            }

            // Write out y[n, co, od, oh, ow] one by one
            const uint64_t yBaseSpatial =
                static_cast<uint64_t>(n) * (cOut_ * yStrideC) +
                static_cast<uint64_t>(od) * yStrideD +
                static_cast<uint64_t>(oh) * yStrideH +
                static_cast<uint64_t>(ow) * yStrideW;

            for (uint32_t co = 0; co < cOut_; ++co) {
                const float val = accLocal.GetValue(co);
                oneLocal.SetValue(0, val);
                const uint64_t yOff = yBaseSpatial + static_cast<uint64_t>(co) * yStrideC;
                AscendC::DataCopy(yGm_[yOff], oneLocal, 1);
            }
        }

        // Free local buffers
        scalarBuf_.FreeTensor(oneLocal);
        accBuf_.FreeTensor(accLocal);
        segBuf_.FreeTensor(segLocal);
        wBuf_.FreeTensor(wLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> wBuf_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> segBuf_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> accBuf_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> scalarBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, d_{0}, h_{0}, w_{0};
    uint32_t cOut_{0}, kD_{0}, kH_{0}, kW_{0};
    uint32_t outD_{0}, outH_{0}, outW_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);
    KernelConvStandard3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            t.n, t.cIn, t.d, t.h, t.w,
            t.cOut, t.kD, t.kH, t.kW,
            t.outD, t.outH, t.outW);
    op.Process();
}
