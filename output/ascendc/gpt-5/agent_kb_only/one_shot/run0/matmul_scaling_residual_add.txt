namespace optiling {

static inline int64_t GetDimOr(const gert::Shape* s, size_t idx, int64_t defv) {
    return (s->GetOriginShape().GetDimNum() > idx) ? s->GetOriginShape().GetDim(idx) : defv;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Shapes:
    //   x: [M, K]
    //   w: [N, K]  (will be transposed inside kernel)
    //   b: [N]
    //   scaling: [1] scalar
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    int64_t M = GetDimOr(x_shape, 0, 1);
    int64_t K = GetDimOr(x_shape, 1, 1);
    int64_t N = GetDimOr(w_shape, 0, 1);

    if (M <= 0 || N <= 0 || K <= 0) {
        return ge::GRAPH_FAILED;
    }

    // Build Matmul tiling using MultiCoreMatmulTiling (NORMAL layout).
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);
    tiling.SetDim(1); // let tiler decide actual core num

    // A: half, B: half (transposed in kernel), C: float, Bias: float
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);

    tiling.SetShape(static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K));
    tiling.SetSingleShape(static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K));
    tiling.SetOrgShape(static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K));
    tiling.SetBias(true);
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Set blockDim from tiler suggestion.
    uint32_t blockDim = static_cast<uint32_t>(tiling.GetCoreNum());
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    // Save TCubeTiling into raw tiling buffer
    auto *rawBuf = context->GetRawTilingData();
    std::memcpy(rawBuf->GetData(), &tilingData, sizeof(tilingData));
    rawBuf->SetDataSize(sizeof(tilingData));

    // Workspace (if any) — here we set to 0
    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // [M, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K]
    gert::Shape *y_shape = context->GetOutputShape(0);

    if (x_shape->GetOriginShape().GetDimNum() < 2 || w_shape->GetOriginShape().GetDimNum() < 2) {
        return GRAPH_FAILED;
    }
    int64_t M = x_shape->GetOriginShape().GetDim(0);
    int64_t N = w_shape->GetOriginShape().GetDim(0);

    gert::Shape out;
    out.GetStorageShape().Reset(2);
    out.GetOriginShape().Reset(2);
    out.GetStorageShape().SetDim(0, M);
    out.GetStorageShape().SetDim(1, N);
    out.GetOriginShape().SetDim(0, M);
    out.GetOriginShape().SetDim(1, N);
    *y_shape = out;

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output is float
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = half;
using B_T = half;
using C_T = float;
using BiasT = float;

constexpr int32_t VEC_BUFFER_NUM = 2;
constexpr uint32_t VEC_TILE_ELEMS = 32768; // 128KB UB worth of float

extern \"C\" __global__ __aicore__ void matmul_scaling_residual_add_custom(
    GM_ADDR xGM, GM_ADDR wGM, GM_ADDR bGM, GM_ADDR scalingGM, GM_ADDR yGM,
    GM_ADDR tilingGM, GM_ADDR workspaceGM)
{
    // 1) Load TCubeTiling from GM
    TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // 2) Prepare GM tensors
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;
    AscendC::GlobalTensor<BiasT> biasGlobal;

    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(C_T);
    int32_t sizeBias = tiling.CLayoutInfoB * tiling.singleCoreN * sizeof(BiasT);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(xGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(wGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(yGM), sizeC);
    biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ BiasT*>(bGM), sizeBias);

    int offset_a = 0, offset_b = 0, offset_c = 0, offset_bias = 0;
    AscendC::GlobalTensor<A_T> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal[offset_a].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreK);
    AscendC::GlobalTensor<B_T> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal[offset_b].GetPhyAddr()),
                         tiling.singleCoreK * tiling.singleCoreN);
    AscendC::GlobalTensor<C_T> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal[offset_c].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreN);
    AscendC::GlobalTensor<BiasT> gm_bias;
    gm_bias.SetGlobalBuffer(const_cast<__gm__ BiasT*>(biasGlobal[offset_bias].GetPhyAddr()),
                            tiling.singleCoreN);

    // 3) Matmul (NORMAL layout): y = x @ w^T + b
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false,
                                                           BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BiasT>,
        MM_CFG> mm1;

    AscendC::TPipe cubePipe;
    REGIST_MATMUL_OBJ(&cubePipe, GetSysWorkSpacePtr(), mm1);
    mm1.Init(&tiling);
    mm1.SetTensorA(gm_a, false);
    mm1.SetTensorB(gm_b, true);
    mm1.SetWorkspace(workspaceGM, 0);
    mm1.SetBias(gm_bias);
    mm1.IterateBatch(gm_c, 1, 1, false);

    // 4) Scaling with residual: y_out = y * scaling + y = y * (1 + scaling)
    // scaling is a scalar float tensor at scalingGM
    float s = *reinterpret_cast<__gm__ float*>(scalingGM);
    float factor = 1.0f + s;

    const uint32_t totalLen = static_cast<uint32_t>(
        tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN);

    if (totalLen == 0) {
        return;
    }

    AscendC::TPipe vecPipe;
    AscendC::TQue<AscendC::TPosition::VECIN, VEC_BUFFER_NUM> inQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, VEC_BUFFER_NUM> outQ;

    vecPipe.InitBuffer(inQ, VEC_BUFFER_NUM, VEC_TILE_ELEMS * sizeof(C_T));
    vecPipe.InitBuffer(outQ, VEC_BUFFER_NUM, VEC_TILE_ELEMS * sizeof(C_T));

    const uint32_t tileLen = VEC_TILE_ELEMS;
    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    const uint32_t strideElems = blockNum * tileLen;
    uint32_t base = blockIdx * tileLen;
    if (base >= totalLen) {
        return;
    }

    auto CopyIn = [&] (uint32_t off, uint32_t len) __aicore__ {
        AscendC::LocalTensor<C_T> xLocal = inQ.AllocTensor<C_T>();
        AscendC::DataCopy(xLocal, gm_c[off], len);
        inQ.EnQue(xLocal);
    };

    auto Compute = [&] (uint32_t len) __aicore__ {
        AscendC::LocalTensor<C_T> xLocal = inQ.DeQue<C_T>();
        AscendC::LocalTensor<C_T> yLocal = outQ.AllocTensor<C_T>();
        AscendC::Muls(yLocal, xLocal, factor, len);
        outQ.EnQue(yLocal);
        inQ.FreeTensor(xLocal);
    };

    auto CopyOut = [&] (uint32_t off, uint32_t len) __aicore__ {
        AscendC::LocalTensor<C_T> yLocal = outQ.DeQue<C_T>();
        AscendC::DataCopy(gm_c[off], yLocal, len);
        outQ.FreeTensor(yLocal);
    };

    // Prefetch first
    uint32_t curLen = totalLen - base;
    if (curLen > tileLen) curLen = tileLen;
    CopyIn(base, curLen);

    while (true) {
        const uint32_t nextBase = base + strideElems;
        uint32_t nextLen = 0;
        if (nextBase < totalLen) {
            nextLen = totalLen - nextBase;
            if (nextLen > tileLen) nextLen = tileLen;
            CopyIn(nextBase, nextLen);
        }

        Compute(curLen);
        CopyOut(base, curLen);

        if (nextLen == 0) break;
        base = nextBase;
        curLen = nextLen;
    }
}
"""

#include <torch/library.h>

static inline at::Tensor to_contig_dtype_device(const at::Tensor& t, at::ScalarType dt) {
    at::Tensor x = t;
    if (x.scalar_type() != dt) x = x.to(dt);
    if (!x.is_contiguous()) x = x.contiguous();
    return x;
}

at::Tensor matmul_scaling_residual_add_custom_npu(const at::Tensor& xIn,
                                                  const at::Tensor& wIn,
                                                  const at::Tensor& bIn,
                                                  const at::Tensor& scalingIn)
{
    // Ensure dtypes/layouts match kernel contract:
    // x, w: float16; b: float32; scaling: float32; y: float32
    at::Tensor x = to_contig_dtype_device(xIn, at::kHalf);
    at::Tensor w = to_contig_dtype_device(wIn, at::kHalf);
    at::Tensor b = to_contig_dtype_device(bIn, at::kFloat);
    at::Tensor scaling = to_contig_dtype_device(scalingIn, at::kFloat);

    // Output shape: [M, N] where M=x.size(0), N=w.size(0)
    at::Tensor y = at::empty({x.size(0), w.size(0)}, x.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnMatmulScalingResidualAddCustom, x, w, b, scaling, y);
    return y;
}

    m.impl("matmul_scaling_residual_add_custom", &matmul_scaling_residual_add_custom_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
