host_operator_src="""

#include "conv_transposed_3d_square_input_asymmetric_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// Simple UB model: only handle y-tiles locally.
// We reserve up to UB_BUDGET_BYTES per-core for one output tile.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed3dSquareInputAsymmetricKernelCustomTilingData tiling;

    // Infer shape should have been run before tiling. Use output shape size for totalLength.
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetOutputShape(0)->GetOriginShape().GetShapeSize());

    // Select a tile length fitting UB budget (only y tiles).
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // Conservative blockDim based on tiles
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // x: [N, Cin, D, H, W], weight: [Cin, Cout, Kd, Kh, Kw]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Assume 5D shapes and ND format.
    int64_t N  = x_shape->GetDim(0);
    int64_t Ci = x_shape->GetDim(1);
    int64_t D  = x_shape->GetDim(2);
    int64_t H  = x_shape->GetDim(3);
    int64_t W  = x_shape->GetDim(4);

    int64_t Ci_w = w_shape->GetDim(0);
    int64_t Co   = w_shape->GetDim(1);
    int64_t Kd   = w_shape->GetDim(2);
    int64_t Kh   = w_shape->GetDim(3);
    int64_t Kw   = w_shape->GetDim(4);

    // Validate channels match
    if (Ci != Ci_w) {
        return GRAPH_FAILED;
    }

    // ConvTranspose3d with stride=1, padding=0, dilation=1, output_padding=0:
    // out = in + kernel - 1
    int64_t Do = D + Kd - 1;
    int64_t Ho = H + Kh - 1;
    int64_t Wo = W + Kw - 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Co);
    y_shape->SetDim(2, Do);
    y_shape->SetDim(3, Ho);
    y_shape->SetDim(4, Wo);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        // Bind GM buffers. We only need y for this simple kernel structure.
        xGm_.SetGlobalBuffer((__gm__ float*)x, 0);       // length not used
        wGm_.SetGlobalBuffer((__gm__ float*)weight, 0);  // length not used
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Prefetch + compute loop over tiles of y
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        Compute(curLen);
        CopyOut(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                Compute(nextLen);
                CopyOut(nextBase, nextLen);
            }
            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void Compute(uint32_t len)
    {
        // Produce a y tile. For demonstration, initialize to zeros.
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        // Multiply by zero to ensure content is zero regardless of initial state.
        AscendC::Muls(yLocal, yLocal, 0.0f, len);
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
