kernel_src="""
#include "kernel_operator.h"

constexpr int32_t INBUF_NUM = 1;
constexpr int32_t OUTBUF_NUM = 1;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                uint32_t T, uint32_t B, uint32_t I, uint32_t H, uint32_t O, uint32_t K)
    {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O; K_ = K;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), static_cast<uint32_t>(T_) * B_ * I_);
        h0Gm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(h0), B_ * H_);
        wi2hGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w_i2h), H_ * K_);
        bi2hGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b_i2h), H_);
        whoGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w_h2o), O_ * H_);
        bhoGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b_h2o), O_);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), static_cast<uint32_t>(T_) * B_ * O_);

        // UB buffers
        // Max temporary length among K and H and O
        uint32_t maxVec = K_;
        if (H_ > maxVec) maxVec = H_;
        if (O_ > maxVec) maxVec = O_;
        // Allocate several buffers: inp[K], wrow[max(K,H)], tmp[max(K,H)], scalar[1], hLocal[B*H], yRow[B*O], bias vectors, etc.
        pipe_.InitBuffer(inpQueue_, INBUF_NUM, K_ * sizeof(float));
        pipe_.InitBuffer(wRowQueue_, INBUF_NUM, K_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, INBUF_NUM, K_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, INBUF_NUM, sizeof(float));
        pipe_.InitBuffer(hAllQueue_, INBUF_NUM, B_ * H_ * sizeof(float));
        pipe_.InitBuffer(yRowQueue_, OUTBUF_NUM, B_ * O_ * sizeof(float));
        pipe_.InitBuffer(bi2hQueue_, INBUF_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(bhoQueue_, INBUF_NUM, O_ * sizeof(float));
        pipe_.InitBuffer(wHoRowQueue_, INBUF_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(hRowQueue_, INBUF_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(xRowQueue_, INBUF_NUM, I_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Load initial hidden state h0 -> hLocal[B*H]
        AscendC::LocalTensor<float> hLocalAll = hAllQueue_.AllocTensor<float>();
        AscendC::DataCopy(hLocalAll, h0Gm_, B_ * H_);
        // Load biases once
        AscendC::LocalTensor<float> bI2hLocal = bi2hQueue_.AllocTensor<float>();
        AscendC::DataCopy(bI2hLocal, bi2hGm_, H_);
        AscendC::LocalTensor<float> bHoLocal = bhoQueue_.AllocTensor<float>();
        AscendC::DataCopy(bHoLocal, bhoGm_, O_);

        for (uint32_t t = 0; t < T_; ++t) {
            // Compute h_t for all batches sequentially
            for (uint32_t b = 0; b < B_; ++b) {
                // Prepare inp = [x_t_b (I), h_{t-1}_b (H)] length K
                AscendC::LocalTensor<float> inpLocal = inpQueue_.AllocTensor<float>();
                // x row [I]
                AscendC::LocalTensor<float> xRowLocal = xRowQueue_.AllocTensor<float>();
                const uint32_t xOffset = (t * B_ + b) * I_;
                AscendC::DataCopy(xRowLocal, xGm_[xOffset], I_);
                // h row [H]
                AscendC::LocalTensor<float> hRowLocal = hRowQueue_.AllocTensor<float>();
                AscendC::DataCopy(hRowLocal, hLocalAll[b * H_], H_);
                // Concatenate
                AscendC::DataCopy(inpLocal, xRowLocal, I_);
                AscendC::DataCopy(inpLocal[I_], hRowLocal, H_);
                xRowQueue_.FreeTensor(xRowLocal);
                hRowQueue_.FreeTensor(hRowLocal);

                // For each hidden unit hidx compute dot(w_i2h[hidx, :], inp) + b_i2h[hidx], apply tanh
                for (uint32_t hidx = 0; hidx < H_; ++hidx) {
                    AscendC::LocalTensor<float> wRowLocal = wRowQueue_.AllocTensor<float>();
                    const uint32_t wOffset = hidx * K_;
                    AscendC::DataCopy(wRowLocal, wi2hGm_[wOffset], K_);

                    AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::Mul(tmpLocal, wRowLocal, inpLocal, K_);

                    AscendC::LocalTensor<float> sumLocal = scalarQueue_.AllocTensor<float>();
                    // ReduceSum to 1 element
                    AscendC::ReduceSum(sumLocal, tmpLocal, K_);
                    // Add bias
                    // sumLocal[0] += bI2hLocal[hidx]
                    AscendC::LocalTensor<float> biasScalar = scalarQueue_.AllocTensor<float>();
                    AscendC::DataCopy(biasScalar, bI2hLocal[hidx], 1);
                    AscendC::Add(sumLocal, sumLocal, biasScalar, 1);
                    scalarQueue_.FreeTensor(biasScalar);

                    // tanh activation
                    AscendC::Tanh(sumLocal, sumLocal, 1);

                    // write back hLocalAll[b*H + hidx] = sumLocal[0]
                    AscendC::DataCopy(hLocalAll[b * H_ + hidx], sumLocal, 1);

                    wRowQueue_.FreeTensor(wRowLocal);
                    tmpQueue_.FreeTensor(tmpLocal);
                    scalarQueue_.FreeTensor(sumLocal);
                }

                inpQueue_.FreeTensor(inpLocal);
            }

            // Compute y_t = h_t @ w_h2o^T + b_h2o  => [B, O]
            AscendC::LocalTensor<float> yRowAll = yRowQueue_.AllocTensor<float>();
            for (uint32_t oidx = 0; oidx < O_; ++oidx) {
                AscendC::LocalTensor<float> wHoRowLocal = wHoRowQueue_.AllocTensor<float>();
                const uint32_t wHoOffset = oidx * H_;
                AscendC::DataCopy(wHoRowLocal, whoGm_[wHoOffset], H_);
                for (uint32_t b = 0; b < B_; ++b) {
                    AscendC::LocalTensor<float> hRowLocal = hRowQueue_.AllocTensor<float>();
                    AscendC::DataCopy(hRowLocal, hLocalAll[b * H_], H_);

                    AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::Mul(tmpLocal, wHoRowLocal, hRowLocal, H_);

                    AscendC::LocalTensor<float> sumLocal = scalarQueue_.AllocTensor<float>();
                    AscendC::ReduceSum(sumLocal, tmpLocal, H_);

                    // add output bias b_h2o[oidx]
                    AscendC::LocalTensor<float> biasScalar = scalarQueue_.AllocTensor<float>();
                    AscendC::DataCopy(biasScalar, bHoLocal[oidx], 1);
                    AscendC::Add(sumLocal, sumLocal, biasScalar, 1);
                    scalarQueue_.FreeTensor(biasScalar);

                    // store to yRowAll[b*O + oidx]
                    AscendC::DataCopy(yRowAll[b * O_ + oidx], sumLocal, 1);

                    hRowQueue_.FreeTensor(hRowLocal);
                    tmpQueue_.FreeTensor(tmpLocal);
                    scalarQueue_.FreeTensor(sumLocal);
                }
                wHoRowQueue_.FreeTensor(wHoRowLocal);
            }
            // Write y_t to GM
            const uint32_t yOffset = (t * B_) * O_;
            AscendC::DataCopy(yGm_[yOffset], yRowAll, B_ * O_);
            yRowQueue_.FreeTensor(yRowAll);
        }

        // Free persistent buffers
        hAllQueue_.FreeTensor(hLocalAll);
        bi2hQueue_.FreeTensor(bI2hLocal);
        bhoQueue_.FreeTensor(bHoLocal);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::QuePosition::VECIN, INBUF_NUM> inpQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, INBUF_NUM> wRowQueue_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, INBUF_NUM> tmpQueue_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, INBUF_NUM> scalarQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, INBUF_NUM> hAllQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, INBUF_NUM> bi2hQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, INBUF_NUM> bhoQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, INBUF_NUM> wHoRowQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, INBUF_NUM> hRowQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, INBUF_NUM> xRowQueue_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, OUTBUF_NUM> yRowQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> wi2hGm_;
    AscendC::GlobalTensor<float> bi2hGm_;
    AscendC::GlobalTensor<float> whoGm_;
    AscendC::GlobalTensor<float> bhoGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t T_{0}, B_{0}, I_{0}, H_{0}, O_{0}, K_{0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0,
                                                                 GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                 GM_ADDR w_h2o, GM_ADDR b_h2o,
                                                                 GM_ADDR y, GM_ADDR workspace,
                                                                 GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tilingData.T, tilingData.B, tilingData.I, tilingData.H, tilingData.O, tilingData.K);
    op.Process();
}
