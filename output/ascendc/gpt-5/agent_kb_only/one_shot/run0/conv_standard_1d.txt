#include "conv_standard_1d_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard1dCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);   // [N, Cin, L]
    const gert::Shape *wShape = context->GetInputShape(1);   // [Cout, Cin, K]
    if (xShape->GetDimNum() != 3 || wShape->GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N    = static_cast<uint32_t>(xShape->GetDim(0));
    const uint32_t Cin  = static_cast<uint32_t>(xShape->GetDim(1));
    const uint32_t L    = static_cast<uint32_t>(xShape->GetDim(2));
    const uint32_t Cout = static_cast<uint32_t>(wShape->GetDim(0));
    const uint32_t K    = static_cast<uint32_t>(wShape->GetDim(2));
    if (K == 0 || L < K) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t outL = L - K + 1;

    // UB model per core and per tile (float32):
    //   inQueueX:     1 * tileLen
    //   outQueueY:    1 * tileLen
    //   tmpVecA:      1 * tileLen      (broadcasted weight vector)
    //   tmpVecB:      1 * tileLen      (x * weight)
    //   weightBuf:    K * 1            (small, kept per ci)
    // Total floats ~= 4 * tileLen + K
    // UB budget per-core (conservative):
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t BYTES_PER_ELEM = 4;
    static constexpr uint32_t ALIGN_ELEMS    = 1024;     // favor large DMA bursts
    static constexpr uint32_t HARD_CAP_ELEMS = 65536;    // avoid too-large per-iter latency

    uint32_t maxTileByUb = 0;
    if (UB_BUDGET_BYTES > K * BYTES_PER_ELEM) {
        maxTileByUb = (UB_BUDGET_BYTES - K * BYTES_PER_ELEM) / (BYTES_PER_ELEM * 4u);
    }
    if (maxTileByUb == 0) maxTileByUb = ALIGN_ELEMS;
    maxTileByUb = AlignDown(maxTileByUb, ALIGN_ELEMS);
    if (maxTileByUb > HARD_CAP_ELEMS) maxTileByUb = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTileByUb;
    if (tileLen > outL) {
        uint32_t aligned = AlignDown(outL, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : outL;
    }
    if (tileLen == 0) tileLen = outL;

    // Blocks cover (N * Cout) output channel-batch pairs in grid-stride
    uint32_t pairCount = N * Cout;
    uint32_t blockDim = pairCount;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;  // 910B has 32 cores

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_l(L);
    tiling.set_cout(Cout);
    tiling.set_k(K);
    tiling.set_outL(outL);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);   // [N, Cin, L]
    const gert::Shape *wShape = context->GetInputShape(1);   // [Cout, Cin, K]
    if (xShape->GetDimNum() != 3 || wShape->GetDimNum() != 3) {
        return GRAPH_FAILED;
    }
    const int64_t N    = xShape->GetDim(0);
    const int64_t CinX = xShape->GetDim(1);
    const int64_t L    = xShape->GetDim(2);
    const int64_t Cout = wShape->GetDim(0);
    const int64_t CinW = wShape->GetDim(1);
    const int64_t K    = wShape->GetDim(2);
    if (CinX != CinW || K <= 0 || L < K) {
        return GRAPH_FAILED;
    }
    const int64_t outL = L - K + 1;
    gert::Shape *yShape = context->GetOutputShape(0);
    yShape->SetDimNum(3);
    yShape->SetDim(0, N);
    yShape->SetDim(1, Cout);
    yShape->SetDim(2, outL);
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelConvStandard1dCustom {
public:
    __aicore__ inline KernelConvStandard1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t L,
                                uint32_t Cout, uint32_t K, uint32_t outL,
                                uint32_t tileLen)
    {
        N_ = N; Cin_ = Cin; L_ = L; Cout_ = Cout; K_ = K; outL_ = outL; tileLen_ = tileLen;

        const uint32_t xSize = N_ * Cin_ * L_;
        const uint32_t wSize = Cout_ * Cin_ * K_;
        const uint32_t ySize = N_ * Cout_ * outL_;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), xSize);
        wGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w), wSize);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), ySize);

        pipe_.InitBuffer(inQueueX_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileLen_ * sizeof(float));
        // tmp vectors: scalar broadcast and mul accumulation
        pipe_.InitBuffer(tmpQueueA_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueB_, 1, tileLen_ * sizeof(float));
        // weight buffer per (co, ci): K elements
        pipe_.InitBuffer(tmpQueueW_, 1, K_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || K_ == 0 || outL_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t pairCount = N_ * Cout_;

        for (uint32_t pair = blockIdx; pair < pairCount; pair += blockNum) {
            const uint32_t n  = pair / Cout_;
            const uint32_t co = pair % Cout_;
            ComputeOnePair(n, co);
        }
    }

private:
    __aicore__ inline void ComputeOnePair(uint32_t n, uint32_t co)
    {
        for (uint32_t t0 = 0; t0 < outL_; t0 += tileLen_) {
            uint32_t cur = outL_ - t0;
            if (cur > tileLen_) cur = tileLen_;

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            float zeroVal = 0.0f;
            AscendC::Duplicate<float>(yLocal, zeroVal, cur);

            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                // Load weights for (co, ci, :)
                const uint32_t wOff = ((co * Cin_ + ci) * K_);
                AscendC::LocalTensor<float> wLocal = tmpQueueW_.AllocTensor<float>();
                AscendC::DataCopy(wLocal, wGm_[wOff], K_);

                for (uint32_t k = 0; k < K_; ++k) {
                    // broadcast weight scalar w[co, ci, k] to vector length cur
                    AscendC::LocalTensor<float> scalarVec = tmpQueueA_.AllocTensor<float>();
                    const uint32_t dstShape[1] = { cur };
                    const uint32_t srcShape[1] = { 1 };
                    // view starting at k (length >=1); broadcast will use srcShape
                    AscendC::LocalTensor<float> wScalarView = wLocal[k];
                    AscendC::BroadCast<float, 1, 0>(scalarVec, wScalarView, dstShape, srcShape);

                    // Load input slice x[n, ci, t0 + k : t0 + k + cur]
                    const uint32_t xOffBase = (n * Cin_ + ci) * L_;
                    const uint32_t xOff     = xOffBase + t0 + k;
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[xOff], cur);

                    // tmp = xLocal * scalarVec
                    AscendC::LocalTensor<float> tmpMul = tmpQueueB_.AllocTensor<float>();
                    AscendC::Mul(tmpMul, xLocal, scalarVec, cur);

                    // yLocal += tmpMul
                    AscendC::Add(yLocal, yLocal, tmpMul, cur);

                    // free temporaries for this k
                    tmpQueueB_.FreeTensor(tmpMul);
                    inQueueX_.FreeTensor(xLocal);
                    tmpQueueA_.FreeTensor(scalarVec);
                }
                tmpQueueW_.FreeTensor(wLocal);
            }

            // write back
            const uint32_t yOff = (n * Cout_ + co) * outL_ + t0;
            AscendC::DataCopy(yGm_[yOff], yLocal, cur);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    uint32_t N_{0}, Cin_{0}, L_{0}, Cout_{0}, K_{0}, outL_{0}, tileLen_{0};

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN,   1> inQueueX_;
    AscendC::TQue<AscendC::QuePosition::VECOUT,  1> outQueueY_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueueA_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueueB_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueueW_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
};

extern "C" __global__ __aicore__ void conv_standard_1d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvStandard1dCustom op;
    op.Init(x, weight, y,
            tilingData.n, tilingData.cin, tilingData.l,
            tilingData.cout, tilingData.k, tilingData.outL,
            tilingData.tileLen);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv1d_standard_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    const auto N = x.size(0);
    const auto Cin = x.size(1);
    const auto L = x.size(2);
    const auto Cout = weight.size(0);
    const auto K = weight.size(2);
    TORCH_CHECK(weight.size(1) == Cin, \"weight Cin must match x Cin\");
    TORCH_CHECK(L >= K, \"input length must be >= kernel size\");
    at::Tensor y = at::empty({N, Cout, L - K + 1}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard1dCustom, x, weight, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv_standard_1d_custom\", &conv1d_standard_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"conv_standard_1d_custom\", &conv1d_standard_impl_npu, \"1D standard convolution (ConvStandard1dCustom on NPU)\");
