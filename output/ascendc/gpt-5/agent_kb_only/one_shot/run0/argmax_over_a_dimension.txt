static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    // Input shape
    const gert::Shape *in_shape = context->GetInputShape(0);
    const int64_t rank = static_cast<int64_t>(in_shape->GetOriginShape().GetDimNum());
    if (rank <= 0 || rank > 3) {
        // This sample supports up to 3D (common for the provided example).
        return ge::GRAPH_FAILED;
    }

    // Get dim attribute (normalized later, but Python side will permute to last axis; here it's used for infer)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dimAttr;

    // Gather dims
    int64_t dims_arr[3] = {1, 1, 1};
    for (int64_t i = 0; i < rank; ++i) {
        dims_arr[i] = in_shape->GetOriginShape().GetDim(i);
    }

    // Normalize dim to [0, rank)
    if (dim < 0) dim += static_cast<int32_t>(rank);
    if (dim < 0 || dim >= rank) {
        return ge::GRAPH_FAILED;
    }

    // For compute, we will perform Argmax along the last axis on device side.
    // Python binding permutes tensor so that the requested dim is moved to the last axis.
    // So here, inner = last dimension, outter = product of other dimensions.
    const int32_t inner = static_cast<int32_t>(dims_arr[rank - 1]);
    int32_t outter = 1;
    for (int64_t i = 0; i < rank - 1; ++i) outter *= static_cast<int32_t>(dims_arr[i]);

    // TopK tiling with k=1, largest=true, isInitIndex=true, mode TOPK_NORMAL
    const bool IS_LARGEST = true;
    const bool IS_INITINDEX = true;
    const bool IS_REUSESOURCE = false;
    const uint32_t dtypesize = 4; // float

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());

    uint32_t maxsize = 0;
    uint32_t minsize = 0;

    // Fill topk tiling
    AscendC::TopKTilingFunc(ascendcPlatform, inner, outter, 1, dtypesize, IS_INITINDEX,
                            AscendC::TopKMode::TOPK_NORMAL, IS_LARGEST, tiling.topkTilingData);
    AscendC::GetTopKMaxMinTmpSize(ascendcPlatform, inner, outter, IS_REUSESOURCE,
                                  IS_INITINDEX, AscendC::TopKMode::TOPK_NORMAL,
                                  IS_LARGEST, dtypesize, maxsize, minsize);

    tiling.set_dim(dim);
    tiling.set_dims0(static_cast<uint32_t>(dims_arr[0]));
    tiling.set_dims1(static_cast<uint32_t>(dims_arr[1]));
    tiling.set_dims2(static_cast<uint32_t>(dims_arr[2]));
    tiling.set_dimsNum(static_cast<uint32_t>(rank));
    tiling.set_inner(inner);
    tiling.set_outter(outter);
    tiling.set_k(1);
    tiling.set_islargest(IS_LARGEST);
    tiling.set_isinitindex(IS_INITINDEX);
    tiling.set_tmpsize(minsize);

    // Choose a conservative block dim
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No additional workspace reserved at op level
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

host_operator_src="""
#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const int32_t rank = static_cast<int32_t>(x_shape->GetOriginShape().GetDimNum());

    int32_t dim = 0;
    {
        const gert::RuntimeAttrs *attrs = context->GetAttrs();
        const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
        dim = *dimAttr;
    }
    int32_t nd = dim < 0 ? dim + rank : dim;
    if (nd < 0 || nd >= rank) return GRAPH_FAILED;

    // Remove the dimension 'nd'
    std::vector<int64_t> out_dims;
    out_dims.reserve(rank > 0 ? (rank - 1) : 0);
    for (int32_t i = 0; i < rank; ++i) {
        if (i == nd) continue;
        out_dims.push_back(x_shape->GetOriginShape().GetDim(i));
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(static_cast<uint64_t>(out_dims.size()));
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

    // Argmax returns indices, use int64 to match PyTorch

kernel_src="""
#include "kernel_operator.h"

// This kernel performs Argmax along the last axis.
// Python binding permutes input so that the requested axis becomes the last axis.

extern \"C\" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    // Using TopK high-level API with k=1 to obtain indices of the maximum values along the last axis.
    // Tiling data provides inner (last axis length), outter (product of remaining axes),
    // and TopK tiling information.
    // The example TopK usage in docs shows how to pass tiling parameters into a kernel.

    // Define a kernel TopK operator (template and API provided by math library).
    // Here we only need indices; values are discarded.
    KernelTopK<float, true, true, false, false, AscendC::TopKMode::TOPK_NORMAL> op;

    // Init: srcVal = x, srcIdx (unused because isInitIndex=true), finishLocal uses workspace,
    // dstVal can reuse workspace (dummy), dstIdx = y.
    // Parameters: k=1, islargest=true, tmpsize, outter, inner, n (equal to inner for last-axis reduce), topkTilingData.
    // Note: This usage follows the TopK kernel example pattern in the documentation.
    op.Init(x,
            /*srcIdx*/ workspace,
            /*finishLocal*/ workspace,
            /*dstVal*/ workspace,
            /*dstIdx*/ y,
            tilingData.k,
            tilingData.islargest,
            tilingData.tmpsize,
            tilingData.outter,
            tilingData.inner,
            /*n*/ tilingData.inner,
            tilingData.topkTilingData);

    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include \"pytorch_npu_helper.hpp\"

// Permute input so that 'dim' becomes the last axis, then run kernel.
// Output indices shape equals input shape with 'dim' removed.
static at::Tensor permute_dim_last(const at::Tensor& x, int64_t dim) {
    const int64_t rank = x.dim();
    if (dim < 0) dim += rank;
    TORCH_CHECK(dim >= 0 && dim < rank, \"dim out of range\");
    std::vector<int64_t> perm;
    perm.reserve(rank);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        perm.push_back(i);
    }
    perm.push_back(dim);
    return x.permute(perm).contiguous();
}

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    // Move dim to the last axis for contiguous row-wise TopK
    at::Tensor x_perm = permute_dim_last(self, dim);
    // Output shape is input with 'dim' removed
    std::vector<int64_t> out_sizes;
    out_sizes.reserve(self.dim() - 1);
    for (int64_t i = 0; i < self.dim(); ++i) {
        if (i == dim || i == dim - self.dim()) continue;
        out_sizes.push_back(self.size(i));
    }
    at::Tensor result = at::empty(out_sizes, self.options().dtype(at::kLong));

    // Call custom kernel: argmax_over_a_dimension_custom(x_perm) with attribute 'dim' consumed on host
    // (The host tiling expects Argmax along last axis; attr dim is used for shape inference.)
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, x_perm, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"argmax_over_a_dimension_custom\", &argmax_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"argmax_over_a_dimension_custom\",
          &argmax_over_a_dimension_impl_npu,
          \"Argmax over a specified dimension (ArgmaxOverADimensionCustom on NPU)\");
}
