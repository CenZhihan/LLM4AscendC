    const uint32_t Do = (D >= Kd) ? (D - Kd + 1) : 0;
    const uint32_t Ho = (H >= Kh) ? (H - Kh + 1) : 0;
    const uint32_t Wo = (W >= Kw) ? (W - Kw + 1) : 0;

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int MAX_CO = 256;

class KernelConv3dSoftmaxMaxPoolMaxPoolCustom {
public:
    __aicore__ inline KernelConv3dSoftmaxMaxPoolMaxPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t d, uint32_t h, uint32_t w_,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t do_, uint32_t ho, uint32_t wo,
                                uint32_t d2, uint32_t h2, uint32_t w2,
                                uint32_t totalPos)
    {
        x_ = reinterpret_cast<__gm__ const float*>(x);
        w_ = reinterpret_cast<__gm__ const float*>(w);
        b_ = reinterpret_cast<__gm__ const float*>(b);
        y_ = reinterpret_cast<__gm__ float*>(y);

        N_ = n; Cin_ = cin; Cout_ = cout;
        D_ = d; H_ = h; W_ = w_;
        Kd_ = kd; Kh_ = kh; Kw_ = kw;
        Do_ = do_; Ho_ = ho; Wo_ = wo;
        D2_ = d2; H2_ = h2; W2_ = w2;
        totalPos_ = totalPos;
    }

    __aicore__ inline void Process()
    {
        if (Cout_ > MAX_CO) {
            return; // safeguard for demo implementation
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t pos = blockIdx; pos < totalPos_; pos += blockNum) {
            // decode (n, d2, h2, w2)
            uint32_t tmp = pos;
            const uint32_t w2i = tmp % W2_; tmp /= W2_;
            const uint32_t h2i = tmp % H2_; tmp /= H2_;
            const uint32_t d2i = tmp % D2_; tmp /= D2_;
            const uint32_t ni  = tmp;

            // best values across 4x4x4 cube for each channel after softmax
            float best[MAX_CO];
            for (uint32_t co = 0; co < Cout_; ++co) best[co] = -1e30f;

            for (uint32_t od = 0; od < 4; ++od) {
                for (uint32_t oh = 0; oh < 4; ++oh) {
                    for (uint32_t ow = 0; ow < 4; ++ow) {
                        const uint32_t di = 4 * d2i + od;
                        const uint32_t hi = 4 * h2i + oh;
                        const uint32_t wi = 4 * w2i + ow;
                        // logits vector for this spatial position
                        float logits[MAX_CO];
                        for (uint32_t co = 0; co < Cout_; ++co) {
                            float acc = b_[co];
                            // conv3d accumulation (stride=1, pad=0, dil=1)
                            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                                for (uint32_t z = 0; z < Kd_; ++z) {
                                    for (uint32_t y = 0; y < Kh_; ++y) {
                                        for (uint32_t x = 0; x < Kw_; ++x) {
                                            const uint32_t in_d = di + z;
                                            const uint32_t in_h = hi + y;
                                            const uint32_t in_w = wi + x;
                                            // x index: (((n*Cin + ci)*D + in_d)*H + in_h)*W + in_w
                                            uint64_t x_idx = (((static_cast<uint64_t>(ni) * Cin_ + ci) * D_ + in_d) * H_ + in_h) * W_ + in_w;
                                            // w index: ((((co)*Cin + ci)*Kd + z)*Kh + y)*Kw + x
                                            uint64_t w_idx = ((((static_cast<uint64_t>(co) * Cin_ + ci) * Kd_ + z) * Kh_ + y) * Kw_ + x);
                                            acc += x_[x_idx] * w_[w_idx];
                                        }
                                    }
                                }
                            }
                            logits[co] = acc;
                        }

                        // softmax across channel axis
                        float m = logits[0];
                        for (uint32_t co = 1; co < Cout_; ++co) m = (logits[co] > m) ? logits[co] : m;
                        float s = 0.f;
                        for (uint32_t co = 0; co < Cout_; ++co) {
                            s += expf(logits[co] - m);
                        }
                        const float invs = 1.f / s;
                        for (uint32_t co = 0; co < Cout_; ++co) {
                            float p = expf(logits[co] - m) * invs;
                            if (p > best[co]) best[co] = p;
                        }
                    }
                }
            }

            // write out best (N, Cout, D2, H2, W2)
            for (uint32_t co = 0; co < Cout_; ++co) {
                uint64_t y_idx = ((((static_cast<uint64_t>(ni) * Cout_ + co) * D2_ + d2i) * H2_ + h2i) * W2_ + w2i);
                y_[y_idx] = best[co];
            }
        }
    }

private:
    __gm__ const float *x_ {nullptr};
    __gm__ const float *w_ {nullptr};
    __gm__ const float *b_ {nullptr};
    __gm__ float *y_ {nullptr};

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t D_{0}, H_{0}, W_{0};
    uint32_t Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Do_{0}, Ho_{0}, Wo_{0};
    uint32_t D2_{0}, H2_{0}, W2_{0};
    uint32_t totalPos_{0};
};

extern "C" __global__ __aicore__ void conv3d_softmax_max_pool_max_pool_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dSoftmaxMaxPoolMaxPoolCustom op;
    op.Init(
        x, weight, bias, y,
        tiling_data.n, tiling_data.cin, tiling_data.cout,
        tiling_data.d, tiling_data.h, tiling_data.w,
        tiling_data.kd, tiling_data.kh, tiling_data.kw,
        tiling_data.do_, tiling_data.ho, tiling_data.wo,
        tiling_data.d2, tiling_data.h2, tiling_data.w2,
        tiling_data.totalPos
    );
    op.Process();
}
