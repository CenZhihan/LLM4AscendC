#include "frobenius_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // safe UB per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // DMA-friendly alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap per-iter for latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    FrobeniusNormCustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // UB budget model:
    // - First pass: inQueueX (2 * tile), tmpSquare (1 * tile), tmpScalar (32B)
    // - Second pass: inQueueX (2 * tile), outQueueY (2 * tile)
    // Approx worst-case floats: 5 * tile -> bytes: 20 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // Choose blockDim conservatively based on tiles
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // Workspace: one float accumulator (32-byte aligned by framework)
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 32; // reserve 32 bytes for one float scalar (alignment)
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFrobeniusNormCustom {
public:
    __aicore__ inline KernelFrobeniusNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace,
                                uint32_t totalLength, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        // Bind GM buffers
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        accGm_.SetGlobalBuffer((__gm__ float*)workspace, 1);

        // Initialize global accumulator to 0 before UB allocation as per API constraints.
        AscendC::InitGlobalMemory(accGm_, 1, 0.0f);

        // Init UB queues
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpSquareQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpScalarQueue_, 1, 32); // 32-byte aligned scalar buffer
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        // Pass 1: accumulate sum of squares into GM scalar with atomic add
        AccumulateSumSquares();

        // Ensure all cores finished atomic accumulation before using the result
        AscendC::SyncAll();

        // Load accumulated sum, compute inverse norm scalar
        LoadAndComputeInvNorm();

        // Pass 2: normalize each tile with the computed inverse norm
        NormalizeTiles();
    }

private:
    __aicore__ inline void AccumulateSumSquares()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            // Compute sum of squares for current tile and atomic add to GM scalar
            ComputeTileSumSquares(curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

    __aicore__ inline void LoadAndComputeInvNorm()
    {
        // Read accumulated sum from GM into a 1-element local tensor
        AscendC::LocalTensor<float> sumLocal = tmpScalarQueue_.AllocTensor<float>();
        AscendC::DataCopy(sumLocal, accGm_, 1);

        // Compute norm = sqrt(sum), then invNorm_ = 1 / max(norm, eps)
        // Use UB vector ops on scalar-sized tensor
        AscendC::LocalTensor<float> normLocal = tmpScalarQueue_.AllocTensor<float>();
        AscendC::Sqrt(normLocal, sumLocal, 1);

        // Avoid divide-by-zero by clamping norm >= 1e-12
        AscendC::LocalTensor<float> epsLocal = tmpScalarQueue_.AllocTensor<float>();
        AscendC::Duplicate(epsLocal, 1.0e-12f, 1);
        AscendC::Max(normLocal, normLocal, epsLocal, 1);

        AscendC::LocalTensor<float> invLocal = tmpScalarQueue_.AllocTensor<float>();
        AscendC::Reciprocal(invLocal, normLocal, 1);

        // Extract the scalar inverse into a C++ variable (treated as immediate for Muls/Duplicate)
        // Note: the framework aligns the buffer; here we read first element logically.
        invNorm_ = ((volatile float*)(invLocal.GetPhyAddr()))[0];

        tmpScalarQueue_.FreeTensor(sumLocal);
        tmpScalarQueue_.FreeTensor(normLocal);
        tmpScalarQueue_.FreeTensor(epsLocal);
        tmpScalarQueue_.FreeTensor(invLocal);
    }

    __aicore__ inline void NormalizeTiles()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            ComputeNormalize(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeTileSumSquares(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> sqLocal = tmpSquareQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = tmpScalarQueue_.AllocTensor<float>();

        // square
        AscendC::Mul(sqLocal, xLocal, xLocal, len);

        // reduce sum into scalar buffer (32-byte aligned)
        AscendC::ReduceSum(sumLocal, sqLocal, len);

        // Atomic add the partial sum to GM accumulator
        AscendC::SetAtomicAdd<float>();
        AscendC::DataCopy(accGm_, sumLocal, 1);
        AscendC::SetAtomicNone();

        tmpSquareQueue_.FreeTensor(sqLocal);
        tmpScalarQueue_.FreeTensor(sumLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeNormalize(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // y = x * invNorm_
        AscendC::Muls(yLocal, xLocal, invNorm_, len);

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSquareQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpScalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> accGm_;  // workspace accumulator

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    float invNorm_{0.0f};
};

extern "C" __global__ __aicore__ void frobenius_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelFrobeniusNormCustom op;
    op.Init(x, y, workspace, tilingData.totalLength, tilingData.tileLength);
    op.Process();
}
"""
