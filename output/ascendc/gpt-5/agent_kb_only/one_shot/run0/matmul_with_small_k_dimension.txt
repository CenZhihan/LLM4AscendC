host_operator_src="""

#include "register/op_def_registry.h"
#include "matmul_with_small_k_dimension_custom_tiling.h"

namespace optiling {

static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);
    if (aShape == nullptr || bShape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    // Expect A:[M,K], B:[K,N]
    const auto dimsA = aShape->GetOriginShape().GetDims();
    const auto dimsB = bShape->GetOriginShape().GetDims();
    if (dimsA.size() != 2 || dimsB.size() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(dimsA[0]);
    const uint32_t K = static_cast<uint32_t>(dimsA[1]);
    const uint32_t N = static_cast<uint32_t>(dimsB[1]);

    // Conservative blockDim for huge M,N when K is small
    uint32_t blockDim = 16;
    uint64_t tiles = static_cast<uint64_t>(CeilDiv(static_cast<uint64_t>(M) * static_cast<uint64_t>(N), 1024ULL));
    if (tiles < 8) blockDim = 8;
    if (tiles > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    MatmulWithSmallKDimensionCustomTilingData tiling;
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    context->SetTilingKey(1);

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);
    if (aShape == nullptr || bShape == nullptr) {
        return GRAPH_FAILED;
    }
    const auto dimsA = aShape->GetOriginShape().GetDims();
    const auto dimsB = bShape->GetOriginShape().GetDims();
    if (dimsA.size() != 2 || dimsB.size() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = dimsA[0];
    const int64_t N = dimsB[1];
    gert::Shape *cShape = context->GetOutputShape(0);
    cShape->SetDimNum(2);
    cShape->SetDim(0, M);
    cShape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // A and B are float16, C is float
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MatmulWithSmallKDimensionCustom : public OpDef {
public:
    explicit MatmulWithSmallKDimensionCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulWithSmallKDimensionCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // Define Matmul types for NORMAL layout: A:[M,K], B:[K,N], C:[M,N]
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::Normal> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, true,  LayoutMode::Normal> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::Normal> cType;

    set_atomic_none();

    // Load tiling prepared by GE/CANN tiling system for Matmul
    TCubeTiling tiling;
    auto tilingSrc = (__gm__ uint32_t*)tilingGM;
    auto tilingDst = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tilingSrc, ++tilingDst) {
        *tilingDst = *tilingSrc;
    }

    // Initialize GM tensors (sizes are managed by tiling within the cube library)
    AscendC::GlobalTensor<half> aGlobal;
    AscendC::GlobalTensor<half> bGlobal;
    AscendC::GlobalTensor<float> cGlobal;

    // The Matmul library reads sizes and strides from tiling
    // We set buffer pointers; element counts are provided by the library during execution.
    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(aGM), 0);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(bGM), 0);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cGM), 0);

    // Sharing configuration (conservative defaults)
    tiling.shareMode = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize = 0;

    // Register and initialize Matmul object
    matmul::Matmul<aType, bType, cType> mm;
    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
    mm.Init(&tiling);

    // Bind GM tensors and workspace
    mm.SetTensorA(aGlobal[0], false);  // A not transposed
    mm.SetTensorB(bGlobal[0], false);  // B not transposed
    mm.SetWorkspace(workspaceGM, 0);

    // Run single iteration as NORMAL layout single-batch GEMM
    // batchLoop=1, batchA=1, batchB=1
    mm.IterateNBatch(1, 1, 1, false);
}
"""

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor matmul_smallk_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, \"Inputs must be 2D (M,K) and (K,N)\");
    TORCH_CHECK(A.size(1) == B.size(0), \"Incompatible shapes for matmul\");
    // Convert inputs to half for higher throughput and to match the kernel
    at::Tensor Ah = A.to(at::kHalf).contiguous();
    at::Tensor Bh = B.to(at::kHalf).contiguous();
    // Accumulate into float
    at::Tensor C = at::empty({A.size(0), B.size(1)}, A.options().dtype(at::kFloat));
    EXEC_NPU_CMD(aclnnMatmulWithSmallKDimensionCustom, Ah, Bh, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"matmul_with_small_k_dimension_custom\", &matmul_smallk_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"matmul_with_small_k_dimension_custom\", &matmul_smallk_impl_npu, \"Matmul small-K (MatmulWithSmallKDimensionCustom on NPU)\");
}
