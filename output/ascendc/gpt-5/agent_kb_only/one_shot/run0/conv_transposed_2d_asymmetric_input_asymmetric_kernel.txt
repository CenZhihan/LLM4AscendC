kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

// Implements deconvolution (conv_transpose2d) for stride=(1,1), padding=(0,0), dilation=(1,1), groups=1.
// Weight layout: [Cin, Cout, kH, kW]. Input: [N, Cin, Hin, Win]. Output: [N, Cout, Hin + kH - 1, Win + kW - 1].
class KernelConvTransposed2dAsymInAsymKernelCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymInAsymKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Hin, uint32_t Win,
                                uint32_t Cout, uint32_t kH, uint32_t kW,
                                uint32_t Hout, uint32_t Wout)
    {
        N_ = N; Cin_ = Cin; Hin_ = Hin; Win_ = Win;
        Cout_ = Cout; kH_ = kH; kW_ = kW; Hout_ = Hout; Wout_ = Wout;

        const uint32_t xSize = N_ * Cin_ * Hin_ * Win_;
        const uint32_t ySize = N_ * Cout_ * Hout_ * Wout_;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), xSize);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), ySize);

        // We'll access weight via raw GM pointer for scalars.
        wGmPtr_ = reinterpret_cast<__gm__ float*>(w);

        // Queues: one row of input (Win), one row of output (Wout), one temp row (Win)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, Win_ * sizeof(float));
        pipe_.InitBuffer(outQueueYrow_, BUFFER_NUM, Wout_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, Win_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over pairs (n, cout)
        const uint64_t totalPairs = static_cast<uint64_t>(N_) * static_cast<uint64_t>(Cout_);
        for (uint64_t pair = blockIdx; pair < totalPairs; pair += blockNum) {
            const uint32_t n = static_cast<uint32_t>(pair / Cout_);
            const uint32_t cout = static_cast<uint32_t>(pair % Cout_);
            ComputeOneNC(n, cout);
        }
    }

private:
    __aicore__ inline void ComputeOneNC(uint32_t n, uint32_t cout)
    {
        // For each output row h_out, compute y[n, cout, h_out, :]
        for (uint32_t h_out = 0; h_out < Hout_; ++h_out) {
            // Allocate y row and zero it
            AscendC::LocalTensor<float> yRowLocal = outQueueYrow_.AllocTensor<float>();
            AscendC::Duplicate(yRowLocal, 0.0f, static_cast<int32_t>(Wout_));

            // Accumulate over input channels and kernel spatial offsets
            for (uint32_t cin = 0; cin < Cin_; ++cin) {
                for (uint32_t i = 0; i < kH_; ++i) {
                    // corresponding input row index
                    // y_row h_out receives contribution from x_row (h_out - i) if within [0, Hin_)
                    int32_t x_row = static_cast<int32_t>(h_out) - static_cast<int32_t>(i);
                    if (x_row < 0 || x_row >= static_cast<int32_t>(Hin_)) {
                        continue;
                    }

                    // Load input row x[n, cin, x_row, :]
                    AscendC::LocalTensor<float> xRowLocal = inQueueX_.AllocTensor<float>();
                    const uint64_t xBase = ((static_cast<uint64_t>(n) * Cin_ + cin) * Hin_ + static_cast<uint32_t>(x_row)) * Win_;
                    AscendC::DataCopy(xRowLocal, xGm_[xBase], Win_);
                    inQueueX_.EnQue(xRowLocal);
                    xRowLocal = inQueueX_.DeQue<float>();

                    // For each kernel column offset j, perform y_row[j : j+Win_] += w * x_row
                    for (uint32_t j = 0; j < kW_; ++j) {
                        // weight index: [cin, cout, i, j]
                        const uint64_t wOff = (((static_cast<uint64_t>(cin) * Cout_ + cout) * kH_ + i) * kW_) + j;
                        float wScalar = wGmPtr_[wOff];

                        // temp = w * x_row
                        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                        AscendC::Muls(tmpLocal, xRowLocal, wScalar, Win_);

                        // yRowLocal[j ... j+Win_-1] += tmpLocal[0...Win_-1]
                        AscendC::Add(yRowLocal[j], yRowLocal[j], tmpLocal, Win_);

                        tmpQueue_.FreeTensor(tmpLocal);
                    }

                    inQueueX_.FreeTensor(xRowLocal);
                } // i
            } // cin

            // Write y row back
            const uint64_t yBase = ((static_cast<uint64_t>(n) * Cout_ + cout) * Hout_ + h_out) * Wout_;
            AscendC::DataCopy(yGm_[yBase], yRowLocal, Wout_);
            outQueueYrow_.FreeTensor(yRowLocal);
        } // h_out
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM>  inQueueX_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueYrow_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    __gm__ float* wGmPtr_{nullptr};

    uint32_t N_{0}, Cin_{0}, Hin_{0}, Win_{0};
    uint32_t Cout_{0}, kH_{0}, kW_{0}, Hout_{0}, Wout_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvTransposed2dAsymInAsymKernelCustom op;
    op.Init(x, weight, y,
            tilingData.N, tilingData.Cin, tilingData.Hin, tilingData.Win,
            tilingData.Cout, tilingData.kH, tilingData.kW,
            tilingData.Hout, tilingData.Wout);
    op.Process();
}
