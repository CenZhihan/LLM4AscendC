host_operator_src="""

#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"

// Matmul tiling API
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"

namespace optiling {

static inline uint32_t GetU32(const gert::Shape* s, int idx)
{
    return static_cast<uint32_t>(s->GetOriginShape().GetDim(idx));
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Read shapes
    const gert::Shape *aShape = context->GetInputShape(0); // A: [M, K]
    const gert::Shape *bShape = context->GetInputShape(1); // B: [K, N]
    if (aShape->GetOriginShape().GetDimNum() != 2 || bShape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = GetU32(aShape, 0);
    const uint32_t K_a = GetU32(aShape, 1);
    const uint32_t K_b = GetU32(bShape, 0);
    const uint32_t N = GetU32(bShape, 1);
    if (K_a != K_b) {
        return ge::GRAPH_FAILED;
    }

    // Build single-core Matmul tiling (NORMAL layout).
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling tiling(ascendcPlatform);
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);
    tiling.SetShape(M, N, K_a);
    tiling.SetOrgShape(M, N, K_a);
    tiling.SetBias(false);
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret == -1) {
        return ge::GRAPH_FAILED;
    }

    // For single-core tiling, run on 1 core for simplicity/stability.
    context->SetBlockDim(1);

    // Save tiling bytes into RawTilingData buffer
    auto *raw = context->GetRawTilingData();
    const size_t need = sizeof(optiling::TCubeTiling);
    if (raw->GetCapacity() < need) {
        return ge::GRAPH_FAILED;
    }
    std::memcpy(raw->GetData(), &tilingData, need);
    raw->SetDataSize(need);

    // No extra workspace required
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *aShape = context->GetInputShape(0); // [M, K]
    const gert::Shape *bShape = context->GetInputShape(1); // [K, N]
    if (aShape->GetOriginShape().GetDimNum() != 2 || bShape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = aShape->GetOriginShape().GetDim(0);
    const int64_t K_a = aShape->GetOriginShape().GetDim(1);
    const int64_t K_b = bShape->GetOriginShape().GetDim(0);
    const int64_t N = bShape->GetOriginShape().GetDim(1);
    if (K_a != K_b) return GRAPH_FAILED;

    gert::Shape *cShape = context->GetOutputShape(0); // [M, N]
    gert::Shape newShape;
    newShape.SetDimNum(2);
    newShape.SetDim(0, M);
    newShape.SetDim(1, N);
    cShape->SetOriginShape(newShape);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output dtype is float (FP32) for improved accumulation precision.
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class StandardMatrixMultiplicationCustom : public OpDef {
public:
    explicit StandardMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(StandardMatrixMultiplicationCustom);
} // namespace ops
"""

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

extern "C" __global__ __aicore__ void standard_matrix_multiplication_custom(
    GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // Define MatmulType for NORMAL layout: A(half), B(half), C(float)
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    // Load tiling from GM
    TCubeTiling tiling;
    auto tempTilingGM = reinterpret_cast<__gm__ uint32_t*>(tilingGM);
    auto tempTiling   = reinterpret_cast<uint32_t*>(&tiling);
    for (int i = 0; i < static_cast<int>(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // Initialize GM tensors using single-core shapes from tiling
    AscendC::GlobalTensor<half>  aGlobal;
    AscendC::GlobalTensor<half>  bGlobal;
    AscendC::GlobalTensor<float> cGlobal;

    int32_t sizeA = tiling.singleCoreM * tiling.singleCoreK; // elements
    int32_t sizeB = tiling.singleCoreK * tiling.singleCoreN; // elements
    int32_t sizeC = tiling.singleCoreM * tiling.singleCoreN; // elements
    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cGM), sizeC);

    // Setup per-call GM views
    int offset_a = 0, offset_b = 0, offset_c = 0;
    AscendC::GlobalTensor<half>  gm_a;
    AscendC::GlobalTensor<half>  gm_b;
    AscendC::GlobalTensor<float> gm_c;
    gm_a.SetGlobalBuffer(const_cast<__gm__ half*>(aGlobal[offset_a].GetPhyAddr()), tiling.singleCoreM * tiling.singleCoreK);
    gm_b.SetGlobalBuffer(const_cast<__gm__ half*>(bGlobal[offset_b].GetPhyAddr()), tiling.singleCoreK * tiling.singleCoreN);
    gm_c.SetGlobalBuffer(const_cast<__gm__ float*>(cGlobal[offset_c].GetPhyAddr()), tiling.singleCoreM * tiling.singleCoreN);

    // Create and register Matmul instance
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm;
    TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);

    mm.Init(&tiling);
    mm.SetTensorA(gm_a, 0); // not transposed
    mm.SetTensorB(gm_b, 0); // not transposed
    mm.SetWorkspace(workspaceGM, 0);
    // No bias

    // Execute single-batch Matmul
    mm.IterateBatch(gm_c, 1, 1, false);
}
"""

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor standard_mm_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Validate dims
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, \"Inputs must be 2D tensors\");
    TORCH_CHECK(A.size(1) == B.size(0), \"Incompatible matmul shapes\");

    // Cast inputs to half for cube matmul
    at::Tensor A_half = A.dtype() == at::kHalf ? A : A.to(at::kHalf);
    at::Tensor B_half = B.dtype() == at::kHalf ? B : B.to(at::kHalf);

    // Output uses float (FP32) accumulation
    at::Tensor C_float = at::empty({A.size(0), B.size(1)}, A.options().dtype(at::kFloat));

    // Execute custom kernel
    EXEC_NPU_CMD(aclnnStandardMatrixMultiplicationCustom, A_half, B_half, C_float);

    // Cast back to original dtype of A if needed (PyTorch matmul usually returns same dtype)
    if (A.dtype() != at::kFloat) {
        return C_float.to(A.dtype());
    }
    return C_float;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"standard_matrix_multiplication_custom\", &standard_mm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"standard_matrix_multiplication_custom\", &standard_mm_impl_npu, \"Standard MatMul (StandardMatrixMultiplicationCustom on NPU)\");
}
"""
