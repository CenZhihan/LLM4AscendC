host_operator_src="""
#include <cstring>
#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Infer problem sizes
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);

    if (aShape->GetOriginShape().GetDimNum() != 2 || bShape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const int64_t M = aShape->GetOriginShape().GetDim(0);
    const int64_t K = aShape->GetOriginShape().GetDim(1);
    const int64_t Kb = bShape->GetOriginShape().GetDim(0);
    const int64_t N = bShape->GetOriginShape().GetDim(1);
    if (K != Kb || M <= 0 || N <= 0 || K <= 0) {
        return ge::GRAPH_FAILED;
    }

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);

    // Set types
    tiling.SetDim(1); // minimal requirement; runtime will set blockDim below using GetCoreNum.
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBias(false);

    // Shapes
    tiling.SetShape(static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K));
    tiling.SetOrgShape(static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K));

    // Normal layout batch info: single multiply
    tiling.SetBatchInfoForNormal(1, 1, static_cast<int32_t>(M), static_cast<int32_t>(N), static_cast<int32_t>(K));

    // Use all available buffer by default
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Set blockDim from tiling
    uint32_t blockDim = static_cast<uint32_t>(tiling.GetCoreNum());
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    // Save TCubeTiling into raw buffer
    auto *rawBuf = context->GetRawTilingData()->GetData();
    const size_t sz = sizeof(optiling::TCubeTiling);
    std::memcpy(rawBuf, &tilingData, sz);
    context->GetRawTilingData()->SetDataSize(sz);

    // No workspace required by this kernel
    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);
    if (aShape->GetOriginShape().GetDimNum() != 2 || bShape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = aShape->GetOriginShape().GetDim(0);
    const int64_t N = bShape->GetOriginShape().GetDim(1);
    gert::Shape *cShape = context->GetOutputShape(0);
    std::vector<int64_t> out = {M, N};
    cShape->SetDimNum(2);
    cShape->SetOriginShape(out);
    cShape->SetStride({});
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    // follow PyTorch matmul policy, output type aligns with inputs
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SquareMatrixMultiplicationCustom : public OpDef {
public:
    explicit SquareMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SquareMatrixMultiplicationCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = float;
using B_T = float;
using C_T = float;
using BiasT = float;

extern \"C\" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // Define MatmulType for NORMAL layout
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

    // Load tiling data from GM into local TCubeTiling
    TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // Initialize GM tensors
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;

    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * (int32_t)sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * (int32_t)sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * (int32_t)sizeof(C_T);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(cGM), sizeC);

    // Optional sharing parameters (conservative defaults)
    tiling.shareMode = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize = 0;

    int offset_a = 0, offset_b = 0, offset_c = 0;

    AscendC::GlobalTensor<A_T> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal[offset_a].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreK);

    AscendC::GlobalTensor<B_T> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal[offset_b].GetPhyAddr()),
                         tiling.singleCoreK * tiling.singleCoreN);

    AscendC::GlobalTensor<C_T> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal[offset_c].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreN);

    // Create Matmul instance for NORMAL layout
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;

    TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);

    mm1.Init(&tiling);
    // No transpose in this kernel
    uint32_t isTransposeAIn = 0;
    uint32_t isTransposeBIn = 0;
    mm1.SetTensorA(gm_a, isTransposeAIn);
    mm1.SetTensorB(gm_b, isTransposeBIn);

    // Workspace (0 size by default)
    mm1.SetWorkspace(workspaceGM, 0);

    // Single-batch matmul for NORMAL layout
    int32_t batchA = 1;
    int32_t batchB = 1;
    mm1.IterateBatch(gm_c, batchA, batchB, false);
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include \"pytorch_npu_helper.hpp\"

at::Tensor square_mm_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Output tensor
    auto sizesA = A.sizes();
    auto sizesB = B.sizes();
    TORCH_CHECK(sizesA.size() == 2 && sizesB.size() == 2, \"Inputs must be 2D\");
    TORCH_CHECK(sizesA[1] == sizesB[0], \"Incompatible shapes for matmul\");
    at::Tensor C = at::empty({sizesA[0], sizesB[1]}, A.options());

    EXEC_NPU_CMD(aclnnSquareMatrixMultiplicationCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"square_matrix_multiplication_custom\", &square_mm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"square_matrix_multiplication_custom\", &square_mm_impl_npu, \"Square Matrix Multiplication (SquareMatrixMultiplicationCustom on NPU)\");
}
