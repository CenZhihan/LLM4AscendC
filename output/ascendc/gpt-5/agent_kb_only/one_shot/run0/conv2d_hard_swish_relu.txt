host_operator_src="""

#include "conv2d_hard_swish_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
} // namespace

namespace optiling {

// UB model for this kernel (per core):
//   xRow: 1 * tileW
//   yRow: 1 * tileW
//   tmp1: 1 * tileW
//   tmp2: 1 * tileW
//   weights (per oc): inC * kH * kW
// Total floats ~ 4 * tileW + inC*kH*kW
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MIN_TILEW = 64;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dHardSwishReluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);      // [B, C, H, W]
    const gert::Shape *w_shape = context->GetInputShape(1);      // [OC, IC, kH, kW]

    const uint32_t B  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t C  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t OC = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t IC = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t kH = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t kW = static_cast<uint32_t>(w_shape->GetDim(3));

    // stride=1, padding=0 as per model
    const uint32_t outH = H >= kH ? (H - kH + 1) : 0;
    const uint32_t outW = W >= kW ? (W - kW + 1) : 0;

    // Compute tileW based on UB budget
    const uint32_t weightFloats = IC * kH * kW;
    const uint32_t weightBytes = weightFloats * sizeof(float);

    uint32_t usableBytes = (UB_BUDGET_BYTES > weightBytes) ? (UB_BUDGET_BYTES - weightBytes) : (32 * 1024);
    // 4 * tileW floats
    uint32_t maxTileW = usableBytes / (4u * sizeof(float));
    if (maxTileW < MIN_TILEW) maxTileW = MIN_TILEW;
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW == 0) maxTileW = ALIGN_ELEMS;

    uint32_t tileW = maxTileW;
    if (outW > 0 && outW < tileW) {
        const uint32_t aligned = AlignDown(outW, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : outW;
    }
    if (tileW == 0) tileW = MIN_TILEW;

    // blockDim based on number of output rows
    const uint32_t totalRows = (outH == 0 || outW == 0) ? 1u : (B * OC * outH);
    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_batch(B);
    tiling.set_inC(C);
    tiling.set_outC(OC);
    tiling.set_inH(H);
    tiling.set_inW(W);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConv2dHardSwishReluCustom {
public:
    __aicore__ inline KernelConv2dHardSwishReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t B, uint32_t C, uint32_t OC,
                                uint32_t H, uint32_t W, uint32_t kH, uint32_t kW,
                                uint32_t outH, uint32_t outW, uint32_t tileW)
    {
        B_ = B; C_ = C; OC_ = OC;
        H_ = H; W_ = W; kH_ = kH; kW_ = kW;
        outH_ = outH; outW_ = outW;
        tileW_ = tileW;

        xSize_ = B_ * C_ * H_ * W_;
        wSize_ = OC_ * C_ * kH_ * kW_;
        ySize_ = B_ * OC_ * outH_ * outW_;
        bSize_ = OC_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, wSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize_);

        // Raw pointers for rare scalar reads (bias/weight)
        wRaw_ = reinterpret_cast<__gm__ float*>(weight);
        bRaw_ = reinterpret_cast<__gm__ float*>(bias); // can be nullptr when no bias

        pipe_.InitBuffer(inQueueXRow_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueYRow_, BUFFER_NUM, tileW_ * sizeof(float));
        // tmpQueue: tmp1 + tmp2 + reserve (we reuse tmp1/tmp2 across ops)
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, 2 * tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outH_ == 0 || outW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalRows = B_ * OC_ * outH_;
        for (uint32_t row = blockIdx; row < totalRows; row += blockNum) {
            // Decode (b, oc, oh)
            uint32_t tmp = row;
            const uint32_t rowsPerBatch = OC_ * outH_;
            const uint32_t b  = tmp / rowsPerBatch;
            tmp = tmp % rowsPerBatch;
            const uint32_t oc = tmp / outH_;
            const uint32_t oh = tmp % outH_;
            ComputeRow(b, oc, oh);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t b, uint32_t oc, uint32_t oh)
    {
        // Allocate reusable local tensors
        AscendC::LocalTensor<float> yLocal = outQueueYRow_.AllocTensor<float>();
        AscendC::LocalTensor<float> xLocal = inQueueXRow_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp1Local = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp2Local = tmpQueue_.AllocTensor<float>();

        // Bias scalar (if provided)
        float biasScalar = 0.f;
        const bool hasBias = (bRaw_ != nullptr);
        if (hasBias) {
            // Safe load via GM pointer (one scalar)
            biasScalar = bRaw_[oc];
        }

        // Iterate over width tiles
        for (uint32_t ow = 0; ow < outW_; ow += tileW_) {
            const uint32_t len = (ow + tileW_ <= outW_) ? tileW_ : (outW_ - ow);
            bool initialized = false;

            // Accumulate convolution
            for (uint32_t ic = 0; ic < C_; ++ic) {
                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        const uint32_t ih = oh + kh;
                        const uint32_t iw = ow + kw;

                        // Copy a contiguous input row slice of length 'len'
                        const uint32_t xBase = (((b * C_ + ic) * H_ + ih) * W_ + iw);
                        AscendC::DataCopy(xLocal, xGm_[xBase], len);

                        // Weight scalar load (one per (oc,ic,kh,kw))
                        const uint32_t wOff = ((((oc * C_) + ic) * kH_ + kh) * kW_ + kw);
                        const float wScalar = wRaw_[wOff];

                        if (!initialized) {
                            // yLocal = xLocal * wScalar
                            AscendC::Muls(yLocal, xLocal, wScalar, len);
                            initialized = true;
                        } else {
                            // tmp2Local = xLocal * wScalar; yLocal += tmp2Local
                            AscendC::Muls(tmp2Local, xLocal, wScalar, len);
                            AscendC::Add(yLocal, yLocal, tmp2Local, len);
                        }
                    }
                }
            }

            // Optional bias add
            if (hasBias) {
                AscendC::Adds(yLocal, yLocal, biasScalar, len);
            }

            // Fused HardSwish then ReLU
            // t = clamp(y + 3, 0, 6) / 6; y = y * t; y = max(y, 0)
            AscendC::Adds(tmp1Local, yLocal, 3.0f, len);
            AscendC::Mins(tmp1Local, tmp1Local, 6.0f, len);
            AscendC::Maxs(tmp1Local, tmp1Local, 0.0f, len);
            AscendC::Muls(tmp1Local, tmp1Local, (float)(1.0f / 6.0f), len);
            AscendC::Mul(yLocal, yLocal, tmp1Local, len);
            AscendC::Maxs(yLocal, yLocal, 0.0f, len);

            // Write back to GM: y[b, oc, oh, ow:ow+len]
            const uint32_t yBase = (((b * OC_ + oc) * outH_ + oh) * outW_ + ow);
            AscendC::DataCopy(yGm_[yBase], yLocal, len);
        }

        // Free local tensors
        outQueueYRow_.FreeTensor(yLocal);
        inQueueXRow_.FreeTensor(xLocal);
        tmpQueue_.FreeTensor(tmp1Local);
        tmpQueue_.FreeTensor(tmp2Local);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueXRow_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueYRow_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    __gm__ float* wRaw_ {nullptr};
    __gm__ float* bRaw_ {nullptr};

    uint32_t B_{0}, C_{0}, OC_{0};
    uint32_t H_{0}, W_{0}, kH_{0}, kW_{0};
    uint32_t outH_{0}, outW_{0}, tileW_{0};
    // Aliases for concise use
    uint32_t B_, C_, OC_;
    uint32_t H_, W_, kH_, kW_;
    uint32_t outH_, outW_, tileW_;

    uint32_t xSize_{0};
    uint32_t wSize_{0};
    uint32_t ySize_{0};
    uint32_t bSize_{0};
};

extern "C" __global__ __aicore__ void conv2d_hard_swish_relu_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    KernelConv2dHardSwishReluCustom op;
    op.Init(x, weight, conv_bias, y,
            tilingData.batch(),
            tilingData.inC(),
            tilingData.outC(),
            tilingData.inH(),
            tilingData.inW(),
            tilingData.kH(),
            tilingData.kW(),
            tilingData.outH(),
            tilingData.outW(),
            tilingData.tileW());
    op.Process();
}
