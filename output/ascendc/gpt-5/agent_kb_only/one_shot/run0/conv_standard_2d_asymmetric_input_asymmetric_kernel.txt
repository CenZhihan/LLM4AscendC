static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0U) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1U) / b;
}

} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dAsymmetricInputAsymmetricKernelCustomTilingData td;

    const gert::Shape* x_shape = context->GetInputShape(0);    // [N, Cin, H, W]
    const gert::Shape* w_shape = context->GetInputShape(1);    // [Cout, Cin, Kh, Kw]

    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Kh   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw   = static_cast<uint32_t>(w_shape->GetDim(3));

    // Fixed params from the model:
    const uint32_t strideH = 1, strideW = 1;
    const uint32_t padTop = 0, padBottom = 0, padLeft = 0, padRight = 0;
    const uint32_t dilH = 1, dilW = 1;

    const uint32_t Ho = (H + padTop + padBottom - dilH * (Kh - 1) - 1) / strideH + 1;
    const uint32_t Wo = (W + padLeft + padRight - dilW * (Kw - 1) - 1) / strideW + 1;

    // UB budget and tiling choices. Vectorize along output width.
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t ELE_SIZE = sizeof(float);
    static constexpr uint32_t ALIGN_ELEMS = 256;

    // For each k-iteration we load xVec of length tileOw. Keep headroom for queues and yVec.
    // Allocate 3 buffers of size tileOw (in/out/tmp). Approx bytes: 3 * tileOw * 4.
    uint32_t maxTileOw = UB_BUDGET_BYTES / (3U * ELE_SIZE);
    maxTileOw = AlignDown(maxTileOw, ALIGN_ELEMS);
    if (maxTileOw < ALIGN_ELEMS) maxTileOw = ALIGN_ELEMS;
    if (maxTileOw > Wo) maxTileOw = AlignDown(Wo, ALIGN_ELEMS) > 0 ? AlignDown(Wo, ALIGN_ELEMS) : Wo;

    const uint32_t tileOw = (maxTileOw == 0U) ? 1U : maxTileOw;

    // ocTile: number of output channels computed per inner loop. Use 1 for simplicity/robustness.
    const uint32_t ocTile = 1U;

    // Total tiles to process: N * Ho * ceil(Wo / tileOw) * ceil(Cout / ocTile)
    const uint32_t tilesPerRow = CeilDiv(Wo, tileOw);
    const uint32_t ocTiles = CeilDiv(Cout, ocTile);
    const uint64_t totalTiles64 = static_cast<uint64_t>(N) * Ho * tilesPerRow * ocTiles;
    uint32_t blockDim = static_cast<uint32_t>(totalTiles64 > 24U ? 24U : (totalTiles64 < 1U ? 1U : totalTiles64));

    context->SetBlockDim(blockDim);

    td.set_N(N);
    td.set_Cin(Cin);
    td.set_Cout(Cout);
    td.set_H(H);
    td.set_W(W);
    td.set_Kh(Kh);
    td.set_Kw(Kw);
    td.set_Ho(Ho);
    td.set_Wo(Wo);
    td.set_tileOw(tileOw);
    td.set_blockDim(blockDim);
    td.set_ocTile(ocTile);

    td.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(td.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_QUEUE_NUM = 2;
constexpr int32_t OUT_QUEUE_NUM = 2;

class KernelConv2DWidthVec {
public:
    __aicore__ inline KernelConv2DWidthVec() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W,
                                uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t tileOw, uint32_t ocTile)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;
        tileOw_ = tileOw;
        ocTile_ = ocTile;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, Cout_ * Cin_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * Cout_ * Ho_ * Wo_);

        const uint32_t vecBytes = tileOw_ * sizeof(float);
        pipe_.InitBuffer(inQueueXVec_, IN_QUEUE_NUM, vecBytes);
        pipe_.InitBuffer(outQueueYVec_, OUT_QUEUE_NUM, vecBytes);
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float)); // scalar scratch
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0U || Cin_ == 0U || Cout_ == 0U || Ho_ == 0U || Wo_ == 0U || tileOw_ == 0U) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tilesPerRow = (Wo_ + tileOw_ - 1U) / tileOw_;
        const uint32_t ocTiles = (Cout_ + ocTile_ - 1U) / ocTile_;

        const uint64_t totalTiles = static_cast<uint64_t>(N_) * Ho_ * tilesPerRow * ocTiles;

        for (uint64_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            // Decode tile index into (n, oh, owTile, ocTileIdx)
            uint64_t tmp = tileId;
            const uint32_t ocTileIdx = static_cast<uint32_t>(tmp % ocTiles); tmp /= ocTiles;
            const uint32_t owTileIdx = static_cast<uint32_t>(tmp % tilesPerRow); tmp /= tilesPerRow;
            const uint32_t oh = static_cast<uint32_t>(tmp % Ho_); tmp /= Ho_;
            const uint32_t n = static_cast<uint32_t>(tmp % N_);

            const uint32_t owStart = owTileIdx * tileOw_;
            uint32_t curOw = tileOw_;
            if (owStart + curOw > Wo_) curOw = Wo_ - owStart;

            const uint32_t ocStart = ocTileIdx * ocTile_;
            uint32_t curOc = ocTile_;
            if (ocStart + curOc > Cout_) curOc = Cout_ - ocStart;

            ComputeTile(n, oh, owStart, curOw, ocStart, curOc);
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * Cin_ + c) * H_ + h) * W_ + w;
    }

    __aicore__ inline uint64_t OffsetW(uint32_t oc, uint32_t c, uint32_t kh, uint32_t kw) const
    {
        return ((((uint64_t)oc * Cin_ + c) * Kh_ + kh) * Kw_ + kw);
    }

    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t oc, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * Cout_ + oc) * Ho_ + h) * Wo_ + w;
    }

    __aicore__ inline float LoadWeightScalar(uint64_t wOff)
    {
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(tmpLocal, wGm_[wOff], 1);
        tmpQueue_.EnQue(tmpLocal);
        AscendC::LocalTensor<float> tmpLocal2 = tmpQueue_.DeQue<float>();
        // Assume LocalTensor supports indexing to read the scalar from UB.
        float val = tmpLocal2.GetValue(0);
        tmpQueue_.FreeTensor(tmpLocal2);
        return val;
    }

    __aicore__ inline void ComputeTile(uint32_t n, uint32_t oh, uint32_t owStart, uint32_t curOw,
                                       uint32_t ocStart, uint32_t curOc)
    {
        // Currently compute one oc at a time (curOc==1 typical).
        for (uint32_t oc = 0; oc < curOc; ++oc) {
            const uint32_t ocIdx = ocStart + oc;

            bool firstK = true;
            AscendC::LocalTensor<float> yVecLocal; // will hold accumulated result

            for (uint32_t ic = 0; ic < Cin_; ++ic) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const uint32_t ih = oh + kh;
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const uint32_t iwBase = owStart + kw;

                        // Load x vector of length curOw: x[n, ic, ih, iwBase:iwBase+curOw)
                        AscendC::LocalTensor<float> xVecLocal = inQueueXVec_.AllocTensor<float>();
                        AscendC::DataCopy(xVecLocal, xGm_[OffsetX(n, ic, ih, iwBase)], curOw);
                        inQueueXVec_.EnQue(xVecLocal);

                        AscendC::LocalTensor<float> xVec = inQueueXVec_.DeQue<float>();

                        const float wScalar = LoadWeightScalar(OffsetW(ocIdx, ic, kh, kw));

                        if (firstK) {
                            // yVec = wScalar * xVec
                            AscendC::LocalTensor<float> yAlloc = outQueueYVec_.AllocTensor<float>();
                            AscendC::Muls(yAlloc, xVec, wScalar, curOw);
                            yVecLocal = yAlloc;
                            outQueueYVec_.EnQue<float>(yVecLocal);
                            firstK = false;
                        } else {
                            // yVec += wScalar * xVec
                            // Dequeue current yVec, perform AXPY, re-enqueue.
                            AscendC::LocalTensor<float> yCur = outQueueYVec_.DeQue<float>();
                            AscendC::Axpy(yCur, xVec, wScalar, curOw);
                            outQueueYVec_.EnQue<float>(yCur);
                        }

                        inQueueXVec_.FreeTensor(xVec);
                    }
                }
            }

            // Write out yVec to global memory y[n, ocIdx, oh, owStart:owStart+curOw)
            AscendC::LocalTensor<float> yOut = outQueueYVec_.DeQue<float>();
            AscendC::DataCopy(yGm_[OffsetY(n, ocIdx, oh, owStart)], yOut, curOw);
            outQueueYVec_.FreeTensor(yOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_QUEUE_NUM> inQueueXVec_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_QUEUE_NUM> outQueueYVec_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t tileOw_{0}, ocTile_{1};
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2DWidthVec op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W,
            tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo,
            tiling_data.tileOw, tiling_data.ocTile);
    op.Process();
}
