#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline int64_t SafeDivCeil(int64_t a, int64_t b) {
    return (b == 0) ? 0 : ((a + b - 1) / b);
}

} // anonymous namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;

    // Read epsilon attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps_ptr = attrs->GetAttrPointer<float>(0);
    const float epsilon = (eps_ptr != nullptr) ? *eps_ptr : 1.0e-5f;

    // x shape and gamma shape
    const gert::Shape *x_shape = context->GetInputShape(0);   // ND
    const gert::Shape *g_shape = context->GetInputShape(1);   // 1D flattened gamma

    const uint64_t totalLen = x_shape->GetOriginShape().GetShapeSize();
    const uint64_t hLen = g_shape->GetOriginShape().GetShapeSize();

    if (hLen == 0 || totalLen % hLen != 0) {
        // Fallback guard
        context->SetBlockDim(8);
        tiling.set_epsilon(epsilon);
        // Build a minimal valid tiling by treating the entire tensor as S=totalLen/H, H=hLen
        std::vector<int64_t> shapeVec = {1, static_cast<int64_t>(totalLen / (hLen == 0 ? 1 : hLen)),
                                         static_cast<int64_t>(hLen), static_cast<int64_t>(hLen)};
        ge::Shape srcShape(shapeVec);
        uint32_t maxTmp = 0U, minTmp = 0U;
        AscendC::GetLayerNormMaxMinTmpSize(srcShape, sizeof(float), false, maxTmp, minTmp);
        AscendC::GetLayerNormNDTillingInfo(srcShape, minTmp, sizeof(float), false,
                                           tiling.layernormTilingData);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    const uint64_t sLenU64 = totalLen / hLen;
    const uint32_t bLen = 1U;
    const uint32_t sLen = static_cast<uint32_t>(sLenU64);
    const uint32_t hLenU32 = static_cast<uint32_t>(hLen);

    // Build {B, S, storageHLength, originHLength}
    std::vector<int64_t> shapeVec = {static_cast<int64_t>(bLen),
                                     static_cast<int64_t>(sLen),
                                     static_cast<int64_t>(hLenU32),
                                     static_cast<int64_t>(hLenU32)};
    ge::Shape srcShape(shapeVec);

    // Obtain temporary buffer range and choose minimum to ensure correctness.
    uint32_t maxTmp = 0U;
    uint32_t minTmp = 0U;
    AscendC::GetLayerNormMaxMinTmpSize(srcShape, sizeof(float), false, maxTmp, minTmp);

    // Obtain tiling parameters
    AscendC::GetLayerNormNDTillingInfo(srcShape, minTmp, sizeof(float), false,
                                       tiling.layernormTilingData);

    // Pass epsilon
    tiling.set_epsilon(epsilon);

    // Set a safe default block dim
    context->SetBlockDim(8);

    // Save tiling to buffer
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No additional workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

template <typename dataType, bool isReuseSource = false>
class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR gammaGm, GM_ADDR betaGm,
                                GM_ADDR yGm, float epsilon,
                                const LayerNormTiling &lnTiling)
    {
        // Save tiling parameters
        this->bLength_ = lnTiling.bLength;
        this->sLength_ = lnTiling.sLength;
        this->hLength_ = lnTiling.hLength;
        this->tiling_ = lnTiling;

        bshLength_ = bLength_ * sLength_ * hLength_;
        bsLength_ = bLength_ * sLength_;
        epsilon_ = epsilon;

        // GM bindings
        xGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(xGm), bshLength_);
        gammaGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(gammaGm), hLength_);
        betaGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(betaGm), hLength_);
        yGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(yGm), bshLength_);

        // UB queues (single-shot for demo; internal API uses tiling for performance)
        pipe_.InitBuffer(inQueueX_, 1, sizeof(dataType) * bshLength_);
        pipe_.InitBuffer(inQueueGamma_, 1, sizeof(dataType) * hLength_);
        pipe_.InitBuffer(inQueueBeta_, 1, sizeof(dataType) * hLength_);
        pipe_.InitBuffer(outQueueY_, 1, sizeof(dataType) * bshLength_);

        // Mean/variance are intermediate only, not returned to GM
        pipe_.InitBuffer(tmpQueueMean_, 1, sizeof(dataType) * bsLength_);
        pipe_.InitBuffer(tmpQueueVar_, 1, sizeof(dataType) * bsLength_);
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<dataType> xLocal = inQueueX_.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> gammaLocal = inQueueGamma_.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> betaLocal = inQueueBeta_.AllocTensor<dataType>();

        AscendC::DataCopy(xLocal, xGlobal_, bshLength_);
        AscendC::DataCopy(gammaLocal, gammaGlobal_, hLength_);
        AscendC::DataCopy(betaLocal, betaGlobal_, hLength_);

        inQueueX_.EnQue(xLocal);
        inQueueGamma_.EnQue(gammaLocal);
        inQueueBeta_.EnQue(betaLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<dataType> xLocal = inQueueX_.DeQue<dataType>();
        AscendC::LocalTensor<dataType> gammaLocal = inQueueGamma_.DeQue<dataType>();
        AscendC::LocalTensor<dataType> betaLocal = inQueueBeta_.DeQue<dataType>();
        AscendC::LocalTensor<dataType> yLocal = outQueueY_.AllocTensor<dataType>();

        // Intermediate mean and variance (not written back to GM)
        AscendC::LocalTensor<dataType> meanLocal = tmpQueueMean_.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> varLocal = tmpQueueVar_.AllocTensor<dataType>();

        AscendC::LayerNorm<dataType, isReuseSource>(
            yLocal, meanLocal, varLocal,
            xLocal, gammaLocal, betaLocal,
            static_cast<dataType>(epsilon_), tiling_);

        outQueueY_.EnQue<dataType>(yLocal);

        // Free locals
        tmpQueueMean_.FreeTensor(meanLocal);
        tmpQueueVar_.FreeTensor(varLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueGamma_.FreeTensor(gammaLocal);
        inQueueBeta_.FreeTensor(betaLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<dataType> yLocal = outQueueY_.DeQue<dataType>();
        AscendC::DataCopy(yGlobal_, yLocal, bshLength_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueGamma_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueBeta_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueueMean_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueueVar_;

    AscendC::GlobalTensor<dataType> xGlobal_;
    AscendC::GlobalTensor<dataType> gammaGlobal_;
    AscendC::GlobalTensor<dataType> betaGlobal_;
    AscendC::GlobalTensor<dataType> yGlobal_;

    uint32_t bLength_{0};
    uint32_t sLength_{0};
    uint32_t hLength_{0};
    uint32_t bshLength_{0};
    uint32_t bsLength_{0};
    dataType epsilon_{static_cast<dataType>(1.0e-5f)};
    LayerNormTiling tiling_;
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta,
                                                        GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelLayerNormCustom<float, false> op;
    op.Init(x, gamma, beta, y, tilingData.epsilon, tilingData.layernormTilingData);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        # Replicate nn.LayerNorm affine parameters
        self.weight = nn.Parameter(torch.ones(normalized_shape, dtype=torch.float))
        self.bias = nn.Parameter(torch.zeros(normalized_shape, dtype=torch.float))
        self.eps = 1.0e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias, self.eps)
