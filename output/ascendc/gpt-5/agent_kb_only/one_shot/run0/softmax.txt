host_operator_src="""

#include <vector>
#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"
#include "tiling/tiling_api.h"

namespace {

static inline std::vector<int64_t> GetOriginDims(const gert::Shape* s) {
    std::vector<int64_t> dims;
    if (s == nullptr) return dims;
    auto os = s->GetOriginShape();
    for (size_t i = 0; i < os.GetDimNum(); ++i) {
        dims.push_back(os.GetDim(i));
    }
    return dims;
}

} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    // Build ge::Shape from input
    const gert::Shape* in_shape = context->GetInputShape(0);
    std::vector<int64_t> shape_vec = GetOriginDims(in_shape);
    if (shape_vec.empty()) {
        // fallback to size (treat as 1D)
        uint64_t sz = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
        shape_vec = {static_cast<int64_t>(sz), 1};
    }
    ge::Shape srcShape(shape_vec);

    // Choose minimum required local workspace for correctness
    const uint32_t dataTypeSize = sizeof(float);
    // isReuseSource must match kernel template argument below (we use false).
    const uint32_t localWorkSpaceSize = AscendC::GetSoftMaxMinTmpSize(srcShape, dataTypeSize, false);

    // Query tiling info for SoftMax and save to tiling data
    AscendC::SoftMaxTilingFunc(srcShape, dataTypeSize, localWorkSpaceSize, tiling.softmaxTilingData);

    // Set a conservative block dim: parallelize by batch dimension if available
    uint32_t blockDim = 8;
    if (shape_vec.size() >= 2) {
        uint32_t m = static_cast<uint32_t>(shape_vec[0]);
        blockDim = m;
        if (blockDim < 1u) blockDim = 1u;
        if (blockDim > 24u) blockDim = 24u;
    }
    context->SetBlockDim(blockDim);

    // Save tiling buffer
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    context->SetTilingKey(1);

    // Request system workspace for high-level APIs
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    uint32_t sysWorkspaceSize = ascendcPlatform.GetLibApiWorkSpaceSize();
    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = sysWorkspaceSize;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        // axis retained for extensibility; default to 1 to match the model.
        this->Attr("axis").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace,
                                const SoftMaxTiling& softmaxTiling)
    {
        // Set system workspace for high-level APIs if needed by library
        AscendC::SetSysWorkSpace(workspace);

        // Configure GM tensors. The srcSize in tiling is the total element count.
        xGM_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), softmaxTiling.srcSize);
        yGM_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), softmaxTiling.srcSize);

        tiling_ = softmaxTiling;
    }

    __aicore__ inline void Process()
    {
        // Use SoftMax high-level API with tiling.
        // isReuseSource must align with host tiling GetSoftMaxMinTmpSize(..., false).
        AscendC::SoftMax<float, false>(yGM_, xGM_, tiling_);
    }

private:
    AscendC::GlobalTensor<float> xGM_;
    AscendC::GlobalTensor<float> yGM_;
    SoftMaxTiling tiling_;
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, workspace, tilingData.softmaxTilingData);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}
