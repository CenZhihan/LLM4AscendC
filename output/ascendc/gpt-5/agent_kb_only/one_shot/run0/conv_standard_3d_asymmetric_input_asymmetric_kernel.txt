kernel_src="""

#include "kernel_operator.h"

using AscendC::GlobalTensor;
using AscendC::LocalTensor;

class KernelConv3dStd {
public:
    __aicore__ inline KernelConv3dStd() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w_in,
                                uint32_t o, uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t od, uint32_t oh, uint32_t ow)
    {
        n_ = n; c_ = c; d_ = d; h_ = h; wIn_ = w_in;
        o_ = o; kd_ = kd; kh_ = kh; kw_ = kw;
        od_ = od; oh_ = oh; ow_ = ow;

        xLen_ = (uint64_t)n_ * c_ * d_ * h_ * wIn_;
        wLen_ = (uint64_t)o_ * c_ * kd_ * kh_ * kw_;
        yLen_ = (uint64_t)n_ * o_ * od_ * oh_ * ow_;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), xLen_);
        wGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w), wLen_);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), yLen_);

        // Queues for small slices and scalars
        pipe_.InitBuffer(xSliceQ_, 1, kw_ * sizeof(float));
        pipe_.InitBuffer(wSliceQ_, 1, kw_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, kw_ * sizeof(float));
        pipe_.InitBuffer(accQ_, 1, sizeof(float));
        pipe_.InitBuffer(sumQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (yLen_ == 0 || xLen_ == 0 || wLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint64_t total_pos = (uint64_t)n_ * od_ * oh_ * ow_;
        for (uint64_t pos = blockIdx; pos < total_pos; pos += blockNum) {
            // Decode n, od, oh, ow from linear index (exclude O)
            uint32_t tmp = static_cast<uint32_t>(pos);
            uint32_t ow_idx = tmp % ow_;            tmp /= ow_;
            uint32_t oh_idx = tmp % oh_;            tmp /= oh_;
            uint32_t od_idx = tmp % od_;            tmp /= od_;
            uint32_t n_idx  = tmp;

            // For each output channel
            for (uint32_t o_idx = 0; o_idx < o_; ++o_idx) {
                // Accumulator as 1-element local tensor
                LocalTensor<float> accLocal = accQ_.AllocTensor<float>();
                AscendC::Duplicate(accLocal, 0.0f, 1);

                // Accumulate over input channels and kernel window
                for (uint32_t c_idx = 0; c_idx < c_; ++c_idx) {
                    for (uint32_t kd_idx = 0; kd_idx < kd_; ++kd_idx) {
                        for (uint32_t kh_idx = 0; kh_idx < kh_; ++kh_idx) {
                            // x slice: length kw_, contiguous on W
                            uint64_t xOffset = ((((uint64_t)n_idx * c_ + c_idx) * d_ + (od_idx + kd_idx)) * h_ + (oh_idx + kh_idx)) * wIn_ + ow_idx;
                            LocalTensor<float> xSliceLocal = xSliceQ_.AllocTensor<float>();
                            AscendC::DataCopy(xSliceLocal, xGm_[xOffset], kw_);

                            // weight slice: [o, c, kd, kh, :]
                            uint64_t wOffset = ((((uint64_t)o_idx * c_ + c_idx) * kd_ + kd_idx) * kh_ + kh_idx) * kw_;
                            LocalTensor<float> wSliceLocal = wSliceQ_.AllocTensor<float>();
                            AscendC::DataCopy(wSliceLocal, wGm_[wOffset], kw_);

                            // tmp = xSlice * wSlice
                            LocalTensor<float> tmpLocal = tmpQ_.AllocTensor<float>();
                            AscendC::Mul(tmpLocal, xSliceLocal, wSliceLocal, kw_);

                            // sum = ReduceSum(tmpLocal)
                            LocalTensor<float> sumLocal = sumQ_.AllocTensor<float>();
                            AscendC::ReduceSum(sumLocal, tmpLocal, kw_);

                            // accLocal += sumLocal
                            AscendC::Add(accLocal, accLocal, sumLocal, 1);

                            // Free locals
                            sumQ_.FreeTensor(sumLocal);
                            tmpQ_.FreeTensor(tmpLocal);
                            wSliceQ_.FreeTensor(wSliceLocal);
                            xSliceQ_.FreeTensor(xSliceLocal);
                        }
                    }
                }

                // Write result y[n, o, od, oh, ow]
                uint64_t yOffset = ((((uint64_t)n_idx * o_ + o_idx) * od_ + od_idx) * oh_ + oh_idx) * ow_ + ow_idx;
                AscendC::DataCopy(yGm_[yOffset], accLocal, 1);
                accQ_.FreeTensor(accLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xSliceQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wSliceQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQ_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, d_{0}, h_{0}, wIn_{0};
    uint32_t o_{0}, kd_{0}, kh_{0}, kw_{0};
    uint32_t od_{0}, oh_{0}, ow_{0};
    uint64_t xLen_{0}, wLen_{0}, yLen_{0};
};

extern "C" __global__ __aicore__
void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dStd op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.o, tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.od, tiling_data.oh, tiling_data.ow);
    op.Process();
}
