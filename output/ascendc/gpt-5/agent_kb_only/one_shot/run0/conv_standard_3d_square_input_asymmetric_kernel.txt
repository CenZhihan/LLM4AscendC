kernel_src="""
#include "kernel_operator.h"

constexpr int32_t ONE_BUF = 1;

class KernelConvStandard3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t W, uint32_t H, uint32_t D,
                                uint32_t Cout, uint32_t Cin, uint32_t Kw, uint32_t Kh, uint32_t Kd,
                                uint32_t Wout, uint32_t Hout, uint32_t Dout,
                                uint32_t Ktotal, uint32_t tileOutC)
    {
        N_ = N; C_ = C; W_ = W; H_ = H; D_ = D;
        Cout_ = Cout; Cin_ = Cin; Kw_ = Kw; Kh_ = Kh; Kd_ = Kd;
        Wout_ = Wout; Hout_ = Hout; Dout_ = Dout;
        Ktotal_ = Ktotal;
        tileOutC_ = (tileOutC == 0U) ? 1U : tileOutC;

        const uint32_t xSize = N_ * C_ * W_ * H_ * D_;
        const uint32_t wSize = Cout_ * Cin_ * Kw_ * Kh_ * Kd_;
        const uint32_t ySize = N_ * Cout_ * Wout_ * Hout_ * Dout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // UB buffer = Ktotal + tileOutC * (Ktotal + 1)
        const uint32_t ubElems = Ktotal_ + tileOutC_ * (Ktotal_ + 1U);
        pipe_.InitBuffer(tmpQueue_, ONE_BUF, ubElems * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0U || Wout_ == 0U || Hout_ == 0U || Dout_ == 0U || Cout_ == 0U || Ktotal_ == 0U) {
            return;
        }

        const uint32_t totalPos = N_ * Wout_ * Hout_ * Dout_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t pos = blockIdx; pos < totalPos; pos += blockNum) {
            // Decode pos -> (n, ow, oh, od)
            uint32_t rem = pos;
            const uint32_t plane = Hout_ * Dout_;
            const uint32_t n = rem / (Wout_ * plane);
            rem = rem % (Wout_ * plane);
            const uint32_t ow = rem / plane;
            rem = rem % plane;
            const uint32_t oh = rem / Dout_;
            const uint32_t od = rem % Dout_;

            // Allocate patch buffer [Ktotal_]
            AscendC::LocalTensor<float> patchLocal = tmpQueue_.AllocTensor<float>();
            patchLocal.SetSize(Ktotal_);

            // Fill patchLocal by segments of length Kd_ using contiguous DMA from xGm_
            uint32_t dstBase = 0U;
            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kw = 0; kw < Kw_; ++kw) {
                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        const uint32_t srcW = ow + kw;
                        const uint32_t srcH = oh + kh;
                        const uint32_t srcD = od;
                        const uint32_t gmOff = OffsetX(n, ci, srcW, srcH, srcD);
                        const uint32_t locOff = ((ci * Kw_ + kw) * Kh_ + kh) * Kd_;
                        AscendC::DataCopy(patchLocal[locOff], xGm_[gmOff], Kd_);
                        dstBase = locOff + Kd_;
                    }
                }
            }

            // Iterate output channels by tiles
            for (uint32_t coStart = 0; coStart < Cout_; coStart += tileOutC_) {
                const uint32_t curTile = (coStart + tileOutC_ <= Cout_) ? tileOutC_ : (Cout_ - coStart);

                // Allocate weight tile [curTile * Ktotal_]
                AscendC::LocalTensor<float> wLocal = tmpQueue_.AllocTensor<float>();
                wLocal.SetSize(curTile * Ktotal_);
                const uint32_t wGmOff = coStart * Ktotal_;
                AscendC::DataCopy(wLocal, wGm_[wGmOff], curTile * Ktotal_);

                // Allocate y tile [curTile]
                AscendC::LocalTensor<float> yLocal = tmpQueue_.AllocTensor<float>();
                yLocal.SetSize(curTile);

                // Compute dot for each output channel in tile
                for (uint32_t r = 0; r < curTile; ++r) {
                    float acc = 0.0f;
                    const uint32_t baseOff = r * Ktotal_;
                    for (uint32_t k = 0; k < Ktotal_; ++k) {
                        const float wv = wLocal.GetValue(baseOff + k);
                        const float xv = patchLocal.GetValue(k);
                        acc += wv * xv;
                    }
                    yLocal.SetValue(r, acc);
                }

                // Write results to GM (scatter over Cout axis; use scalar copies)
                for (uint32_t r = 0; r < curTile; ++r) {
                    const uint32_t co = coStart + r;
                    const uint32_t yOff = OffsetY(n, co, ow, oh, od);
                    AscendC::DataCopy(yGm_[yOff], yLocal[r], 1);
                }

                // Free tiles
                // (No explicit FreeTensor required for tmpQueue_ in current pattern)
            }

            // Free patch
            // (No explicit FreeTensor required for tmpQueue_ in current pattern)
        }
    }

private:
    __aicore__ inline uint32_t OffsetX(uint32_t n, uint32_t c, uint32_t w, uint32_t h, uint32_t d) const
    {
        // Layout: [N, C, W, H, D], D is fastest
        return (((n * C_ + c) * W_ + w) * H_ + h) * D_ + d;
    }

    __aicore__ inline uint32_t OffsetY(uint32_t n, uint32_t co, uint32_t w, uint32_t h, uint32_t d) const
    {
        // Layout: [N, Cout, Wout, Hout, Dout], D is fastest
        return (((n * Cout_ + co) * Wout_ + w) * Hout_ + h) * Dout_ + d;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, ONE_BUF> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, W_{0}, H_{0}, D_{0};
    uint32_t Cout_{0}, Cin_{0}, Kw_{0}, Kh_{0}, Kd_{0};
    uint32_t Wout_{0}, Hout_{0}, Dout_{0};
    uint32_t Ktotal_{0}, tileOutC_{1};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvStandard3dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tilingData.N, tilingData.C, tilingData.W, tilingData.H, tilingData.D,
            tilingData.Cout, tilingData.Cin, tilingData.Kw, tilingData.Kh, tilingData.Kd,
            tilingData.Wout, tilingData.Hout, tilingData.Dout,
            tilingData.Ktotal, tilingData.tileOutC);
    op.Process();
}
