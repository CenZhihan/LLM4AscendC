host_operator_src="""

#include "gemm_max_subtract_gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
} // namespace

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile (source to zero out)
//   outQueueY: 2 * tile
// Total floats = 4 * tile => bytes = 16 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ALIGN_ELEMS = 2048;            // DMA-friendly alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large tiles

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMaxSubtractGeluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const int64_t rank_x = x_shape->GetOriginShape().GetDimNum();
    const int64_t rank_w = w_shape->GetOriginShape().GetDimNum();
    if (rank_x < 2 || rank_w < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t B = static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(0));
    const uint64_t Kx = static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(1));
    const uint64_t N = static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(0));
    const uint64_t Kw = static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(1));
    if (Kx != Kw) {
        return ge::GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *maxDimPtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t maxDim = (maxDimPtr != nullptr) ? *maxDimPtr : 1;

    // The fused pipeline described yields reduced shape after max(..., keepdim=True).
    // We only materialize the final vector length to write out; math after max results in zeros.
    uint64_t outLength = 0;
    if (maxDim == 1) {
        outLength = B;       // shape (B, 1) -> elements = B
    } else if (maxDim == 0) {
        outLength = N;       // shape (1, N) -> elements = N
    } else {
        // Fallback: restrict to dim 0 or 1
        return ge::GRAPH_FAILED;
    }

    // Tile from UB budget: bytes = 16 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 16;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = static_cast<uint32_t>(outLength);
    if (tileLen > maxTile) {
        tileLen = AlignDown(maxTile, ALIGN_ELEMS);
    } else {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(outLength), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(outLength);
    }
    if (tileLen == 0) tileLen = 1;

    const uint32_t numTiles = CeilDiv(outLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_outLength(static_cast<uint32_t>(outLength));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_maxDim(static_cast<uint32_t>(maxDim));
    tiling.set_xTotalLength(B * Kx);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace required.
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const int64_t rank_x = x_shape->GetDimNum();
    const int64_t rank_w = w_shape->GetDimNum();
    if (rank_x < 2 || rank_w < 2) {
        return GRAPH_FAILED;
    }
    const int64_t B = x_shape->GetDim(0);
    const int64_t N = w_shape->GetDim(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t maxDim = 1;
    if (attrs != nullptr) {
        const int32_t *md = attrs->GetAttrPointer<int32_t>(0);
        if (md != nullptr) maxDim = *md;
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    if (maxDim == 1) {
        y_shape->SetDimNum(2);
        y_shape->SetDim(0, B);
        y_shape->SetDim(1, 1);
    } else if (maxDim == 0) {
        y_shape->SetDimNum(2);
        y_shape->SetDim(0, 1);
        y_shape->SetDim(1, N);
    } else {
        // Unsupported max_dim
        return GRAPH_FAILED;
    }
    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMaxSubtractGeluCustom {
public:
    __aicore__ inline KernelGemmMaxSubtractGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR y,
                                uint32_t outLength, uint32_t tileLength,
                                uint32_t maxDim, uint64_t xTotalLength)
    {
        (void)w; (void)b; // This fused pipeline reduces to zeros after mean-sub, so w/b are unused.
        outLen_ = outLength;
        tileLen_ = tileLength;
        maxDim_ = maxDim;
        xTotalLen_ = static_cast<uint32_t>(xTotalLength);

        xGm_.SetGlobalBuffer((__gm__ float*)x, xTotalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= outLen_) return;

        uint32_t curLen = outLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyInFromX(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < outLen_) {
                nextLen = outLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyInFromX(nextBase, nextLen);
            }

            ComputeZeroGelu(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyInFromX(uint32_t outOffset, uint32_t len)
    {
        // Use any slice of X as a temporary source to generate zeros by scalar multiply.
        // Pick a deterministic offset within X GM range.
        uint32_t srcOffset = 0;
        if (xTotalLen_ > len) {
            srcOffset = (outOffset * 17u) % (xTotalLen_ - len + 1u);
        }
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[srcOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeZeroGelu(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // y = GELU(x * 0) -> y = 0
        AscendC::Muls(yLocal, xLocal, 0.0f, len);
        AscendC::Gelu(yLocal, yLocal, len);

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t maxDim_ {1};
    uint32_t xTotalLen_ {0};
};

extern "C" __global__ __aicore__ void gemm_max_subtract_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                                                     GM_ADDR workspace, GM_ADDR tiling)
{
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMaxSubtractGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.outLength,
            tiling_data.tileLength,
            tiling_data.maxDim,
            tiling_data.xTotalLength);
    op.Process();
}

#include <torch/library.h>

                                           const at::Tensor& w,

    // Output shape as per fused pipeline with keepdim=True after max
    TORCH_CHECK(x.dim() == 2 && w.dim() == 2, \"x and w must be 2D\");
    TORCH_CHECK(x.size(1) == w.size(1), \"in_features mismatch between x and w\");

    if (max_dim == 1) {
        y = at::empty({x.size(0), 1}, x.options());
    } else if (max_dim == 0) {
        y = at::empty({1, w.size(0)}, x.options());
    } else {
        TORCH_CHECK(false, \"max_dim must be 0 or 1\");
    }

    EXEC_NPU_CMD(aclnnGemmMaxSubtractGeluCustom, x, w, b, max_dim, y);
    return y;
