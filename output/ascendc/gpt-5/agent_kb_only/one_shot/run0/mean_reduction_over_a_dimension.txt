host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1) / b);
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0) ? x : (x / a) * a;
}

// UB model per tile (float):
//   inQueue slice: 1 * tile
//   acc buffer:    1 * tile
//   outQueue:      1 * tile
// total = 3 * tile * 4 bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core budget
static constexpr uint32_t BYTES_PER_EL = 4;
static constexpr uint32_t BUFFERS_PER_TILE = 3;          // in + acc + out
static constexpr uint32_t ALIGN_ELEMS = 256;             // DMA-friendly

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    // Read input shape
    const gert::StorageShape *stor_shape = context->GetInputShape(0);
    const int32_t rank = static_cast<int32_t>(stor_shape->GetStorageShape().GetDimNum());
    if (rank <= 0 || rank > 8) {
        return ge::GRAPH_FAILED;
    }

    // Read attribute dim, normalize negative index
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = *(attrs->GetAttrPointer<int32_t>(0));
    int32_t axis = dim < 0 ? dim + rank : dim;
    if (axis < 0 || axis >= rank) {
        return ge::GRAPH_FAILED;
    }

    // Collect dims
    uint32_t dims[8] = {1,1,1,1,1,1,1,1};
    uint64_t totalLen = 1;
    for (int32_t i = 0; i < rank; ++i) {
        int64_t d = stor_shape->GetStorageShape().GetDim(i);
        if (d <= 0) {
            return ge::GRAPH_FAILED;
        }
        dims[i] = static_cast<uint32_t>(d);
        totalLen *= static_cast<uint64_t>(dims[i]);
    }

    // Compute outer, reduceLen, trailing
    uint64_t outer = 1;
    for (int32_t i = 0; i < axis; ++i) {
        outer *= static_cast<uint64_t>(dims[i]);
    }
    const uint64_t reduceLen = static_cast<uint64_t>(dims[axis]);
    uint64_t trailing = 1;
    for (int32_t i = axis + 1; i < rank; ++i) {
        trailing *= static_cast<uint64_t>(dims[i]);
    }
    const uint64_t outLen = (reduceLen == 0) ? 0 : (totalLen / reduceLen);

    // Choose tileLen along trailing dimension
    uint64_t budget_elems = UB_BUDGET_BYTES / (BUFFERS_PER_TILE * BYTES_PER_EL);
    uint32_t maxTile = static_cast<uint32_t>(budget_elems);
    if (maxTile == 0) maxTile = 1;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = 1;

    uint32_t tileLen = static_cast<uint32_t>(trailing <= static_cast<uint64_t>(maxTile) ? trailing : maxTile);
    if (tileLen > 1) {
        tileLen = AlignDown(tileLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = static_cast<uint32_t>(trailing > 0 ? 1 : 0);
    }
    if (trailing > 0 && tileLen == 0) tileLen = 1;

    // Parallelize across groups (outer)
    const uint64_t groups = outer;
    uint32_t blockDim = (groups == 0) ? 1u : (groups < 8u ? static_cast<uint32_t>(groups) : 8u);
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    // Fill tiling
    tiling.set_rank(static_cast<uint32_t>(rank));
    tiling.set_reduceAxis(axis);
    tiling.set_dims0(dims[0]);
    tiling.set_dims1(dims[1]);
    tiling.set_dims2(dims[2]);
    tiling.set_dims3(dims[3]);
    tiling.set_dims4(dims[4]);
    tiling.set_dims5(dims[5]);
    tiling.set_dims6(dims[6]);
    tiling.set_dims7(dims[7]);
    tiling.set_totalLen(static_cast<uint32_t>(totalLen));
    tiling.set_outLen(static_cast<uint32_t>(outLen));
    tiling.set_outer(static_cast<uint32_t>(outer));
    tiling.set_reduceLen(static_cast<uint32_t>(reduceLen));
    tiling.set_trailing(static_cast<uint32_t>(trailing));
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    // Save
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *in_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = *(attrs->GetAttrPointer<int32_t>(0));

    // Build output shape by removing axis
    int32_t rank = static_cast<int32_t>(in_shape->GetStorageShape().GetDimNum());
    int32_t axis = dim < 0 ? dim + rank : dim;
    if (axis < 0 || axis >= rank) {
        return ge::GRAPH_FAILED;
    }
    std::vector<int64_t> out_dims;
    out_dims.reserve(rank > 0 ? rank - 1 : 0);
    for (int32_t i = 0; i < rank; ++i) {
        if (i == axis) continue;
        out_dims.push_back(in_shape->GetStorageShape().GetDim(i));
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    if (out_dims.empty()) {
        // reduce to scalar -> 1-element tensor
        out_dims.push_back(1);
    }
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLen, uint32_t outLen,
                                uint32_t outer, uint32_t reduceLen,
                                uint32_t trailing, uint32_t tileLen)
    {
        totalLen_ = totalLen;
        outLen_ = outLen;
        outer_ = outer;
        reduceLen_ = reduceLen;
        trailing_ = trailing;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outLen_);

        // One-buffer queues; accumulator stays in VECCALC buffer
        pipe_.InitBuffer(inQueueX_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(calcBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || outLen_ == 0) return;
        if (trailing_ == 0 || reduceLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t groupSpan = static_cast<uint64_t>(reduceLen_) * static_cast<uint64_t>(trailing_);
        const float invReduce = 1.0f / static_cast<float>(reduceLen_);

        // Grid-stride over groups (prefix combinations)
        for (uint64_t g = blockIdx; g < outer_; g += blockNum)
        {
            const uint64_t groupBase = g * groupSpan;

            for (uint64_t t = 0; t < static_cast<uint64_t>(trailing_); t += tileLen_)
            {
                uint32_t len = static_cast<uint32_t>(static_cast<uint64_t>(trailing_) - t);
                if (len > tileLen_) len = tileLen_;

                // Accumulator buffer
                AscendC::LocalTensor<float> accLocal = calcBuf_.AllocTensor<float>();
                float zero = 0.0f;
                AscendC::Duplicate<float>(accLocal, zero, len);

                // Accumulate over reduce axis
                for (uint32_t b = 0; b < reduceLen_; ++b)
                {
                    const uint64_t off = groupBase + static_cast<uint64_t>(b) * static_cast<uint64_t>(trailing_) + t;
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[off], len);
                    if (b == 0) {
                        AscendC::DataCopy(accLocal, xLocal, len);
                    } else {
                        AscendC::Add(accLocal, accLocal, xLocal, len);
                    }
                    inQueueX_.FreeTensor(xLocal);
                }

                // Divide by reduceLen to get mean
                AscendC::Muls(accLocal, accLocal, invReduce, len);

                // Write out
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, accLocal, len);
                const uint64_t yOff = g * static_cast<uint64_t>(trailing_) + t;
                AscendC::DataCopy(yGm_[yOff], yLocal, len);

                outQueueY_.FreeTensor(yLocal);
                calcBuf_.FreeTensor(accLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t outLen_{0};
    uint32_t outer_{0};
    uint32_t reduceLen_{0};
    uint32_t trailing_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(
        x, y,
        tiling_data.totalLen,
        tiling_data.outLen,
        tiling_data.outer,
        tiling_data.reduceLen,
        tiling_data.trailing,
        tiling_data.tileLen
    );
    op.Process();
}
