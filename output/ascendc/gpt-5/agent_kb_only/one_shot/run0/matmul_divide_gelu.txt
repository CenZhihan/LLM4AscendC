#include "matmul_divide_gelu_custom_tiling.h"
#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"

// Matmul tiling headers
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Get input shapes
    const gert::Shape *x_shape = context->GetInputShape(0); // [B, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K]
    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const int64_t B = (x_shape->GetDimNum() >= 1) ? x_shape->GetDim(0) : 0;
    const int64_t K = (x_shape->GetDimNum() >= 2) ? x_shape->GetDim(1) : 0;
    const int64_t N = (w_shape->GetDimNum() >= 1) ? w_shape->GetDim(0) : 0;

    if (B <= 0 || K <= 0 || N <= 0) {
        return ge::GRAPH_FAILED;
    }

    // Build multi-core matmul tiling
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling mmTiling(ascendcPlatform);
    mmTiling.SetDim(1);  // conservative default
    mmTiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                      matmul_tiling::DataType::DT_FLOAT16);
    mmTiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                      matmul_tiling::DataType::DT_FLOAT16);
    mmTiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                      matmul_tiling::DataType::DT_FLOAT);
    mmTiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                         matmul_tiling::DataType::DT_FLOAT);
    mmTiling.SetShape(static_cast<uint32_t>(B), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    mmTiling.SetSingleShape(static_cast<uint32_t>(B), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    mmTiling.SetOrgShape(static_cast<uint32_t>(B), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    mmTiling.SetBias(true);
    mmTiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = mmTiling.GetTiling(tilingData);
    if (ret == -1) {
        return ge::GRAPH_FAILED;
    }

    // Set block dim from multi-core tiling result if available, fallback to 1
    uint32_t blockDim = 1;
    if (mmTiling.GetCoreNum() > 0) {
        blockDim = mmTiling.GetCoreNum();
    }
    context->SetBlockDim(blockDim);

    // Compute post-processing tile for UB-based divide+GELU
    // UB budget model: double-buffer in/out queues, no extra temp for float GELU
    // UB usage ~= BUFFER_NUM * (in + out) * 4B * tile
    static constexpr uint32_t BUFFER_NUM = 2;
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
    static constexpr uint32_t ALIGN_ELEMS = 2048;
    static constexpr uint32_t HARD_CAP_ELEMS = 65536;

    const uint32_t singleM = tilingData.singleCoreM;
    const uint32_t singleN = tilingData.singleCoreN;
    const uint64_t coreElems64 = static_cast<uint64_t>(singleM) * static_cast<uint64_t>(singleN);
    const uint32_t coreElems = (coreElems64 > 0xFFFFFFFFull) ? 0xFFFFFFFFu : static_cast<uint32_t>(coreElems64);

    uint32_t maxTileByUb = UB_BUDGET_BYTES / (BUFFER_NUM * 2u * sizeof(float));
    maxTileByUb = AlignDown(maxTileByUb, ALIGN_ELEMS);
    if (maxTileByUb < ALIGN_ELEMS) maxTileByUb = ALIGN_ELEMS;
    if (maxTileByUb > HARD_CAP_ELEMS) maxTileByUb = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t postTile = maxTileByUb;
    if (coreElems < postTile) {
        const uint32_t aligned = AlignDown(coreElems, ALIGN_ELEMS);
        postTile = (aligned > 0) ? aligned : coreElems;
    }

    // Get divisor attribute (optional, default 1.0)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float divisorAttr = 1.0f;
    if (attrs != nullptr && attrs->GetAttrsNum() > 0) {
        const float *divPtr = attrs->GetAttrPointer<float>(0);
        if (divPtr != nullptr) {
            divisorAttr = *divPtr;
        }
    }

    // Pack tiling buffer: [TCubeTiling | postTile(uint32_t) | divisor(float)]
    uint8_t *raw = reinterpret_cast<uint8_t *>(context->GetRawTilingData()->GetData());
    size_t cap = context->GetRawTilingData()->GetCapacity();

    const size_t need = sizeof(optiling::TCubeTiling) + sizeof(uint32_t) + sizeof(float);
    if (cap < need) {
        return ge::GRAPH_FAILED;
    }

    // Write TCubeTiling
    ::memcpy(raw, &tilingData, sizeof(optiling::TCubeTiling));
    size_t offset = sizeof(optiling::TCubeTiling);

    // Write postTile
    ::memcpy(raw + offset, &postTile, sizeof(uint32_t));
    offset += sizeof(uint32_t);

    // Write divisor attribute
    ::memcpy(raw + offset, &divisorAttr, sizeof(float));
    offset += sizeof(float);

    context->GetRawTilingData()->SetDataSize(offset);

    // Workspace not required for this fused kernel beyond matmul internal
    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // [B, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K]
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    if (x_shape->GetDimNum() != 2 || w_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;

    // y: [B, N]
    std::vector<int64_t> out_dims = { x_shape->GetDim(0), w_shape->GetDim(0) };
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, out_dims[0]);
    y_shape->SetDim(1, out_dims[1]);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR xGM, GM_ADDR wGM, GM_ADDR bGM,
                                                                 GM_ADDR divisorGM, GM_ADDR yGM,
                                                                 GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // Define Matmul types: NORMAL layout, A: [B,K], B: [K,N] (isTransposeB=true for weight [N,K])
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, half, true,  LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    set_atomic_none();

    // Load TCubeTiling from tilingGM
    optiling::TCubeTiling tiling;
    auto gm_tiling_u32 = reinterpret_cast<__gm__ uint32_t*>(tilingGM);
    auto local_tiling_u32 = reinterpret_cast<uint32_t*>(&tiling);
    for (int i = 0; i < static_cast<int>(sizeof(optiling::TCubeTiling) / sizeof(uint32_t)); ++i) {
        local_tiling_u32[i] = gm_tiling_u32[i];
    }

    // Read appended extra fields: postTile (uint32_t) and divisor (float)
    uint8_t* baseByte = reinterpret_cast<uint8_t*>(tilingGM);
    uint32_t* postTilePtr = reinterpret_cast<uint32_t*>(baseByte + sizeof(optiling::TCubeTiling));
    float* divisorAttrPtr = reinterpret_cast<float*>(baseByte + sizeof(optiling::TCubeTiling) + sizeof(uint32_t));

    uint32_t postTile = *reinterpret_cast<__gm__ uint32_t*>(postTilePtr);
    float divisorAttr = *reinterpret_cast<__gm__ float*>(divisorAttrPtr);
    if (divisorAttr == 0.0f) {
        divisorAttr = 1.0f; // avoid div-by-zero
    }
    const float invDiv = 1.0f / divisorAttr;

    // Initialize GM tensors
    AscendC::GlobalTensor<half>  aGlobal;
    AscendC::GlobalTensor<half>  bGlobalW;
    AscendC::GlobalTensor<float> cGlobal;
    AscendC::GlobalTensor<float> biasGlobal;

    // For NORMAL layout example
    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(half);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(half);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(float);
    int32_t sizeBias = tiling.CLayoutInfoB * tiling.singleCoreN * sizeof(float);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(xGM), sizeA);
    bGlobalW.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(wGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(yGM), sizeC);
    biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bGM), sizeBias);

    // Wrap GM slices according to NORMAL sample
    AscendC::GlobalTensor<aType::DataType> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ half*>(aGlobal[0].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreK);

    AscendC::GlobalTensor<bType::DataType> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ half*>(bGlobalW[0].GetPhyAddr()),
                         tiling.singleCoreK * tiling.singleCoreN);

    AscendC::GlobalTensor<cType::DataType> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ float*>(cGlobal[0].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreN);

    AscendC::GlobalTensor<biasType::DataType> gm_bias;
    gm_bias.SetGlobalBuffer(const_cast<__gm__ float*>(biasGlobal[0].GetPhyAddr()),
                            tiling.singleCoreN);

    // Create Matmul instance
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;
    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);

    mm1.Init(&tiling);
    mm1.SetTensorA(gm_a, false);
    mm1.SetTensorB(gm_b, true);
    mm1.SetWorkspace(workspaceGM, 0);
    if (tiling.isBias) {
        mm1.SetBias(gm_bias);
    }

    // Compute C = X @ W^T + b
    mm1.IterateBatch(gm_c, 1, 1, false);

    // Post-process: y = GELU( (C) / divisor )
    // UB streaming over gm_c region
    const uint32_t regionElems = tiling.singleCoreM * tiling.singleCoreN;
    if (regionElems == 0 || postTile == 0) {
        return;
    }

    // Init UB queues for vector ops
    constexpr int32_t BUFFER_NUM = 2;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQ;
    pipe.InitBuffer(inQ, BUFFER_NUM, postTile * sizeof(float));
    pipe.InitBuffer(outQ, BUFFER_NUM, postTile * sizeof(float));

    uint32_t offset = 0;
    while (offset < regionElems) {
        uint32_t cur = regionElems - offset;
        if (cur > postTile) cur = postTile;

        // Copy in
        AscendC::LocalTensor<float> inLocal = inQ.AllocTensor<float>();
        AscendC::DataCopy(inLocal, gm_c[offset], cur);
        inQ.EnQue(inLocal);

        // Compute: scale then GELU
        AscendC::LocalTensor<float> xLocal = inQ.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQ.AllocTensor<float>();

        // xLocal = xLocal * invDiv
        AscendC::Muls(xLocal, xLocal, invDiv, cur);
        // yLocal = GELU(xLocal)
        AscendC::Gelu(yLocal, xLocal, cur);

        outQ.EnQue(yLocal);
        inQ.FreeTensor(xLocal);

        // Copy out
        AscendC::LocalTensor<float> outLocal = outQ.DeQue<float>();
        AscendC::DataCopy(gm_c[offset], outLocal, cur);
        outQ.FreeTensor(outLocal);

        offset += cur;
    }
}

#include <torch/library.h>

static at::Tensor to_half_if_needed(const at::Tensor& t) {
    if (t.scalar_type() == at::kHalf) return t;
    return t.to(at::kHalf);
}

static at::Tensor to_float_if_needed(const at::Tensor& t) {
    if (t.scalar_type() == at::kFloat) return t;
    return t.to(at::kFloat);
}

at::Tensor matmul_divide_gelu_impl_npu(const at::Tensor& x,
                                       const at::Tensor& w,
                                       const at::Tensor& b,
                                       const at::Tensor& divisor_tensor) {
    // Ensure dtypes as expected by kernel: x,w -> float16, b -> float32, result -> float32
    at::Tensor x_h = to_half_if_needed(x);
    at::Tensor w_h = to_half_if_needed(w);
    at::Tensor b_f = to_float_if_needed(b);

    // Result tensor
    auto out_sizes = at::IntArrayRef({x.size(0), w.size(0)});
    at::Tensor y = at::empty(out_sizes, x.options().dtype(at::kFloat));

    // Get divisor as attribute
    double d_host = divisor_tensor.item<double>();

    // Execute custom op: inputs (x,w,b,divisor_tensor), attr (divisor), output (y)
    EXEC_NPU_CMD(aclnnMatmulDivideGeluCustom, x_h, w_h, b_f, divisor_tensor, static_cast<float>(d_host), y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_divide_gelu_custom", &matmul_divide_gelu_impl_npu);
