#include "cumprod_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t SafeU32(int64_t v) { return static_cast<uint32_t>(v < 0 ? 0 : v); }
static inline uint32_t CeilDivU32(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumprodCustomTilingData tiling;

    // Read dim attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dimPtr;

    // Get input shape dims
    const gert::Shape *inShape = context->GetInputShape(0);
    const gert::Shape &origin = inShape->GetOriginShape();
    const int64_t dimNum = origin.GetDimNum();

    if (dim < 0) dim += static_cast<int32_t>(dimNum);
    if (dim < 0) dim = 0;
    if (dim >= dimNum) dim = static_cast<int32_t>(dimNum - 1);

    uint64_t outer64 = 1, axisLen64 = 1, tail64 = 1;
    for (int64_t i = 0; i < dimNum; ++i) {
        int64_t d = origin.GetDim(i);
        if (i < dim) outer64 *= static_cast<uint64_t>(d);
        else if (i == dim) axisLen64 = static_cast<uint64_t>(d);
        else tail64 *= static_cast<uint64_t>(d);
    }
    uint64_t totalSeq64 = outer64 * tail64;

    uint32_t outer = SafeU32(outer64);
    uint32_t axisLen = SafeU32(axisLen64);
    uint32_t tail = SafeU32(tail64);
    uint32_t totalSeq = SafeU32(totalSeq64);

    // Choose a reasonable blockDim based on number of independent sequences
    uint32_t blockDim = totalSeq == 0 ? 1u : totalSeq;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_outer(outer);
    tiling.set_axisLen(axisLen);
    tiling.set_tail(tail);
    tiling.set_totalSeq(totalSeq);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

class KernelCumprodCustom {
public:
    __aicore__ inline KernelCumprodCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t axisLen, uint32_t tail,
                                uint32_t totalSeq)
    {
        outer_ = outer;
        axisLen_ = axisLen;
        tail_ = tail;
        totalSeq_ = totalSeq;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x),
                             static_cast<uint32_t>(outer_ * axisLen_ * tail_));
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y),
                             static_cast<uint32_t>(outer_ * axisLen_ * tail_));

        // 2 floats in UB: carry and current value
        pipe_.InitBuffer(tmpQueue_, 1, 2 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (axisLen_ == 0 || totalSeq_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        AscendC::LocalTensor<float> tmp2 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> carryLocal = tmp2[0];
        AscendC::LocalTensor<float> valLocal = tmp2[1];

        for (uint32_t seq = blockIdx; seq < totalSeq_; seq += blockNum) {
            // Map sequence id to (o, t)
            const uint32_t o = tail_ == 0 ? 0 : (seq / tail_);
            const uint32_t t = tail_ == 0 ? 0 : (seq % tail_);

            // Iterate along axis and compute cumulative product
            for (uint32_t k = 0; k < axisLen_; ++k) {
                const uint32_t idx = ((o * axisLen_) + k) * tail_ + t;

                // Load current element into valLocal (length 1)
                AscendC::DataCopy(valLocal, xGm_[idx], 1);

                if (k == 0) {
                    // carry = x[idx]
                    AscendC::DataCopy(carryLocal, valLocal, 1);
                } else {
                    // carry = carry * x[idx]
                    AscendC::Mul(carryLocal, carryLocal, valLocal, 1);
                }

                // Store cumulative product
                AscendC::DataCopy(yGm_[idx], carryLocal, 1);
                AscendC::PipeBarrier<PIPE_V>();
            }
        }

        tmpQueue_.FreeTensor(tmp2);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_{0};
    uint32_t axisLen_{0};
    uint32_t tail_{0};
    uint32_t totalSeq_{0};
};

extern "C" __global__ __aicore__ void cumprod_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelCumprodCustom op;
    op.Init(x, y, tilingData.outer, tilingData.axisLen, tilingData.tail, tilingData.totalSeq);
    op.Process();
}
"""
