host_operator_src="""

#include "max_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB usage model (per core):
//   Two temporary column buffers for pairwise reduction: 2 * tileColLen * 4 bytes
//   Plus small bookkeeping; reserve headroom.
// Budget 192 KB conservatively.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;             // favor DMA alignment
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // cap per-chunk length

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxReductionOverADimensionCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const gert::Shape &orig = in_shape->GetOriginShape();
    const int64_t dimNum = orig.GetDimNum();

    // Attribute: dim (can be negative)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);

    // Resolve dimension: negative means from the end
    int32_t reduceDim = *dimAttr;
    int32_t rd = reduceDim;
    if (rd < 0) rd += static_cast<int32_t>(dimNum);
    if (rd < 0) rd = 0;
    if (rd >= dimNum) rd = static_cast<int32_t>(dimNum) - 1;

    // We will perform max reduction along the last axis inside kernel for best access.
    // If caller permutes to last axis, rd should equal last dim. For shape inference,
    // we still compute output by removing rd from input shape.
    uint64_t totalLength = static_cast<uint64_t>(orig.GetShapeSize());
    uint32_t lastAxisLen = static_cast<uint32_t>(orig.GetDim(dimNum - 1));
    uint32_t outerSize = static_cast<uint32_t>(totalLength / lastAxisLen);

    // Compute tileColLen from UB budget: 2 buffers * 4 bytes * tileColLen <= UB_BUDGET_BYTES
    uint32_t maxTile = UB_BUDGET_BYTES / (2u * 4u);
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = ALIGN_ELEMS;

    uint32_t tileColLen = (lastAxisLen < maxTile) ? AlignDown(lastAxisLen, ALIGN_ELEMS) : maxTile;
    if (tileColLen == 0) tileColLen = lastAxisLen;

    // blockDim based on number of rows to reduce (outerSize)
    uint32_t blockDim = (outerSize == 0) ? 1u : outerSize;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_axisLen(lastAxisLen);
    tiling.set_outerSize(outerSize);
    tiling.set_tileColLen(tileColLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(rd); // for shape inference only

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);

    const gert::Shape &orig = x_shape->GetOriginShape();
    const int64_t dimNum = orig.GetDimNum();

    int32_t rd = *dimAttr;
    if (rd < 0) rd += static_cast<int32_t>(dimNum);
    if (rd < 0) rd = 0;
    if (rd >= dimNum) rd = static_cast<int32_t>(dimNum) - 1;

    std::vector<int64_t> outDims;
    outDims.reserve(static_cast<size_t>(dimNum > 0 ? dimNum - 1 : 0));
    for (int64_t i = 0; i < dimNum; ++i) {
        if (i == rd) continue;
        outDims.push_back(orig.GetDim(i));
    }
    y_shape->SetDimNum(static_cast<int64_t>(outDims.size()));
    for (size_t i = 0; i < outDims.size(); ++i) {
        y_shape->SetDim(static_cast<int64_t>(i), outDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMaxReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t axisLen,
                                uint32_t outerSize,
                                uint32_t tileColLen)
    {
        totalLen_   = totalLength;
        axisLen_    = axisLen;     // length of last axis to reduce
        outerSize_  = outerSize;   // number of rows (product of other dims)
        tileColLen_ = tileColLen;  // column chunk for reduction

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outerSize_);

        // Two temp buffers for pairwise reduction, and small queues for IO
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileColLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileColLen_ * sizeof(float)); // ping-pong buffers
        pipe_.InitBuffer(accQueue_, 1, 1 * sizeof(float));           // accumulator per row
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || axisLen_ == 0 || outerSize_ == 0) { return; }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes rows in a grid-stride loop
        for (uint32_t row = blockIdx; row < outerSize_; row += blockNum) {
            ReduceOneRow(row);
        }
    }

private:
    // Reduce a single row (contiguous segment of length axisLen_) to one float
    __aicore__ inline void ReduceOneRow(uint32_t rowIdx)
    {
        const uint32_t rowBase = rowIdx * axisLen_;

        // accumulator buffer length 1
        AscendC::LocalTensor<float> accLocal = accQueue_.AllocTensor<float>();
        bool hasInit = false;

        // iterate over the row in column chunks
        for (uint32_t colBase = 0; colBase < axisLen_; colBase += tileColLen_) {
            uint32_t n = axisLen_ - colBase;
            if (n > tileColLen_) n = tileColLen_;

            // Load chunk to UB
            AscendC::LocalTensor<float> bufA = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(bufA, xGm_[rowBase + colBase], n);

            // Prepare second buffer for ping-pong
            AscendC::LocalTensor<float> bufB = tmpQueue_.AllocTensor<float>();

            // Pairwise reduce bufA (n) -> scalar in either bufA[0] or bufB[0]
            // We'll toggle between A and B as src/dst
            int32_t curLen = static_cast<int32_t>(n);
            bool srcIsA = true;

            while (curLen > 1) {
                int32_t half = curLen >> 1; // floor(curLen / 2)

                AscendC::LocalTensor<float> src = srcIsA ? bufA : bufB;
                AscendC::LocalTensor<float> dst = srcIsA ? bufB : bufA;

                // dst[i] = max(src[i], src[i + half]) for i in [0, half-1]
                if (half > 0) {
                    AscendC::Max(dst, src, src[half], half);
                }

                // If odd, merge the tail element into dst[0]
                if (curLen & 1) {
                    // dst[0] = max(dst[0], src[curLen - 1])
                    AscendC::Max(dst, dst, src[curLen - 1], 1);
                    curLen = half + 0; // effective next length is half
                } else {
                    curLen = half;
                }

                srcIsA = !srcIsA;
            }

            // Fetch the scalar result
            AscendC::LocalTensor<float> chunkMax = srcIsA ? bufA : bufB;

            if (!hasInit) {
                // acc = chunkMax
                AscendC::DataCopy(accLocal, chunkMax, 1);
                hasInit = true;
            } else {
                // acc = max(acc, chunkMax)
                AscendC::Max(accLocal, accLocal, chunkMax, 1);
            }

            tmpQueue_.FreeTensor(bufB);
            inQueueX_.FreeTensor(bufA);
        }

        // Write out the result for this row
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, accLocal, 1);
        AscendC::DataCopy(yGm_[rowIdx], yLocal, 1);

        outQueueY_.FreeTensor(yLocal);
        accQueue_.FreeTensor(accLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_  {0};
    uint32_t axisLen_   {0};
    uint32_t outerSize_ {0};
    uint32_t tileColLen_{0};
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.axisLen,
            tiling_data.outerSize,
            tiling_data.tileColLen);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor max_reduce_dim_impl_npu(const at::Tensor& self, int64_t dim) {
    // Expect caller to permute so that reduce axis is last for optimal access.
    // We still pass dim attribute for shape inference; commonly -1 when already permuted.
    at::Tensor result;
    // Output shape is input with 'dim' removed; create empty like by using sizes after reduction on CPU
    // but EXEC_NPU_CMD can infer and allocate internally via result = at::empty(...)
    // Here, we directly create an empty with the proper shape:
    auto sizes = self.sizes().vec();
    int64_t d = dim;
    if (d < 0) d += self.dim();
    TORCH_CHECK(d >= 0 && d < self.dim(), "dim out of range");
    sizes.erase(sizes.begin() + d);
    result = at::empty(sizes, self.options());
    EXEC_NPU_CMD(aclnnMaxReductionOverADimensionCustom, self, static_cast<int>(dim), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_reduction_over_a_dimension_custom", &max_reduce_dim_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_reduction_over_a_dimension_custom", &max_reduce_dim_impl_npu,
          "Max reduce over a dimension (MaxReductionOverADimensionCustom on NPU)");

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'max_reduction_over_a_dimension_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, dim: int):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        dim = self.dim
        rank = x.dim()
        if dim < 0:
            dim = dim + rank
        # Move target dim to the last axis to match kernel's optimal access pattern.
        if dim != rank - 1:
            perm = [i for i in range(rank) if i != dim] + [dim]
            x = x.permute(perm).contiguous()
            # After permutation, reduce last axis => pass dim=-1 to binding
            return getattr(custom_ops_lib, _FN)(x, -1)
        else:
            return getattr(custom_ops_lib, _FN)(x, -1)
'''
