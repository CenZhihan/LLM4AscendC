host_operator_src="""
#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"
#include "tiling/tiling_api.h"
#include "log_softmax_custom_tiling.h"

namespace {
static inline uint32_t ProductExceptLast(const std::vector<int64_t>& dims) {
    if (dims.empty()) return 0U;
    uint64_t prod = 1;
    for (size_t i = 0; i + 1 < dims.size(); ++i) {
        prod *= static_cast<uint64_t>(dims[i]);
    }
    return static_cast<uint32_t>(prod);
}
} // namespace

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    LogSoftmaxCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    const std::vector<int64_t>& odims = in_shape->GetOriginShape().GetDims();
    if (odims.empty()) {
        return ge::GRAPH_FAILED;
    }

    // Read optional attribute dim (default -1 means last axis).
    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    int dim = -1;
    if (attrs != nullptr && attrs->GetNumElements() > 0) {
        const int* pDim = attrs->GetAttrPointer<int>(0);
        if (pDim != nullptr) dim = *pDim;
    }
    const int64_t ndim = static_cast<int64_t>(odims.size());
    if (dim < 0) dim += static_cast<int>(ndim);
    if (dim != static_cast<int>(ndim - 1)) {
        // This custom kernel supports reduction along the last axis only.
        return ge::GRAPH_FAILED;
    }

    const uint32_t m = ProductExceptLast(odims);
    const uint32_t k = static_cast<uint32_t>(odims.back());

    // Build shape [m, k] for tiling API.
    std::vector<int64_t> srcDims = {static_cast<int64_t>(m), static_cast<int64_t>(k)};
    ge::Shape shape(srcDims);

    // Determine tmp buffer size using LogSoftMax tiling API and available UB.
    const uint32_t dtypesize = 4; // float32
    uint32_t maxTmp = AscendC::GetLogSoftMaxMaxTmpSize(shape, dtypesize, false);
    auto platformInfo = context->GetPlatformInfo();
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(platformInfo);
    uint64_t ubRemain = 0;
    ascendcPlatform.GetCoreMemSize(platform_ascendc::CoreMemType::UB, ubRemain);
    uint32_t tmpSize = static_cast<uint32_t>(ubRemain >= maxTmp ? maxTmp : ubRemain);

    // Fill LogSoftMax Tiling Data
    AscendC::LogSoftMaxTilingFunc(shape, dtypesize, tmpSize, tiling.logSoftmaxTilingData);

    // Set block dim conservatively based on m (number of rows).
    uint32_t blockDim = m == 0 ? 1U : (m < 8U ? m : 8U);
    if (blockDim == 0) blockDim = 1U;
    context->SetBlockDim(blockDim);

    tiling.set_m(m);
    tiling.set_k(k);
    tiling.set_tmpBufferSize(tmpSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t m, uint32_t k,
                                uint32_t tmpBufferSize,
                                const optiling::LogSoftMaxTiling& lsmTiling)
    {
        m_ = m;
        k_ = k;
        total_ = m_ * k_;
        tmpSize_ = tmpBufferSize;
        lsmTiling_ = lsmTiling;

        xGm_.SetGlobalBuffer((__gm__ float*)x, total_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, total_);

        // Local buffers (note: demonstration purpose; assumes UB is sufficient for full tensor).
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, total_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, total_ * sizeof(float));

        // Max/Sum buffers are broadcast to 32-byte data blocks:
        // For float32, 32 bytes == 8 elements per block.
        const uint32_t brc_elems = 32U / sizeof(float);
        pipe_.InitBuffer(maxQueue_, BUFFER_NUM, m_ * brc_elems * sizeof(float));
        pipe_.InitBuffer(sumQueue_, BUFFER_NUM, m_ * brc_elems * sizeof(float));

        pipe_.InitBuffer(tmpQueue_, 1, tmpSize_);
    }

    __aicore__ inline void Process()
    {
        if (total_ == 0) return;
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_, total_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> srcLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> dstLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = maxQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tmpLocal = tmpQueue_.AllocTensor<uint8_t>();

        AscendC::SoftMaxShapeInfo softmaxInfo = {m_, k_, m_, k_};

        // LogSoftMax: dst, sumTensor, maxTensor, src, sharedTmpBuffer, tiling, shapeInfo
        AscendC::LogSoftMax<float, false>(dstLocal, sumLocal, maxLocal, srcLocal, tmpLocal, lsmTiling_, softmaxInfo);

        outQueueY_.EnQue<float>(dstLocal);
        inQueueX_.FreeTensor(srcLocal);
        maxQueue_.FreeTensor(maxLocal);
        sumQueue_.FreeTensor(sumLocal);
        tmpQueue_.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_, yLocal, total_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> maxQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> sumQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t m_ {0};
    uint32_t k_ {0};
    uint32_t total_ {0};
    uint32_t tmpSize_ {0};
    optiling::LogSoftMaxTiling lsmTiling_;
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tilingData.m, tilingData.k, tilingData.tmpBufferSize, tilingData.logSoftmaxTilingData);
    op.Process();
}
"""
