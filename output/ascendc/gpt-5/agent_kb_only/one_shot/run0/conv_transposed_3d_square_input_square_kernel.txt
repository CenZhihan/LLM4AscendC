kernel_src="""

#include "kernel_operator.h"

class KernelConvTransposed3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W, uint32_t K,
                                uint32_t Dout, uint32_t Hout, uint32_t Wout)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        D_ = D; H_ = H; W_ = W; K_ = K;
        Dout_ = Dout; Hout_ = Hout; Wout_ = Wout;

        const uint64_t sizeX = static_cast<uint64_t>(N_) * Cin_ * D_ * H_ * W_;
        const uint64_t sizeW = static_cast<uint64_t>(Cin_) * Cout_ * K_ * K_ * K_;
        const uint64_t sizeY = static_cast<uint64_t>(N_) * Cout_ * Dout_ * Hout_ * Wout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, sizeX);
        wGm_.SetGlobalBuffer((__gm__ float*)w, sizeW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, sizeY);

        // Buffers: one x-line (W), one y-line (W), one tmp-line (W), one scalar weight (1)
        pipe_.InitBuffer(qXLine_, 1, W_ * sizeof(float));
        pipe_.InitBuffer(qYLine_, 1, W_ * sizeof(float));
        pipe_.InitBuffer(qTmpLine_, 1, W_ * sizeof(float));
        pipe_.InitBuffer(qWScalar_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t tasks = static_cast<uint64_t>(N_) * Cout_;
        for (uint64_t task = blockIdx; task < tasks; task += blockNum) {
            const uint32_t n = static_cast<uint32_t>(task / Cout_);
            const uint32_t co = static_cast<uint32_t>(task % Cout_);
            ComputeSingleNC(n, co);
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t z, uint32_t y, uint32_t x) const
    {
        return (((((uint64_t)n * Cin_ + c) * D_ + z) * H_ + y) * W_ + x);
    }
    __aicore__ inline uint64_t OffsetW(uint32_t ci, uint32_t co, uint32_t kz, uint32_t ky, uint32_t kx) const
    {
        return (((((uint64_t)ci * Cout_ + co) * K_ + kz) * K_ + ky) * K_ + kx);
    }
    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t c, uint32_t z, uint32_t y, uint32_t x) const
    {
        return (((((uint64_t)n * Cout_ + c) * Dout_ + z) * Hout_ + y) * Wout_ + x);
    }

    __aicore__ inline void ComputeSingleNC(uint32_t n, uint32_t co)
    {
        // Stage 1: initialize output with first kernel-slice: ci=0, kz=0..K-1, ky=0..K-1, kx=0..K-1
        // General accumulation: for each (ci,kz,ky,kx), for every input line, y_line(dx) += x_line * w
        for (uint32_t ci = 0; ci < Cin_; ++ci) {
            for (uint32_t kz = 0; kz < K_; ++kz) {
                for (uint32_t z = 0; z < D_; ++z) {
                    const uint32_t dz = kz;
                    const uint32_t zOut = z + dz;
                    for (uint32_t ky = 0; ky < K_; ++ky) {
                        for (uint32_t y = 0; y < H_; ++y) {
                            const uint32_t dy = ky;
                            // Load input line once
                            AscendC::LocalTensor<float> xLine = qXLine_.AllocTensor<float>();
                            AscendC::DataCopy(xLine, xGm_[OffsetX(n, ci, z, y, 0)], W_);
                            qXLine_.EnQue(xLine);
                            AscendC::LocalTensor<float> xLineDeque = qXLine_.DeQue<float>();

                            for (uint32_t kx = 0; kx < K_; ++kx) {
                                // Load scalar weight
                                AscendC::LocalTensor<float> wScalarLocal = qWScalar_.AllocTensor<float>();
                                AscendC::DataCopy(wScalarLocal, wGm_[OffsetW(ci, co, kz, ky, kx)], 1);
                                qWScalar_.EnQue(wScalarLocal);
                                AscendC::LocalTensor<float> wScalarDeque = qWScalar_.DeQue<float>();

                                // Read scalar value from local UB
                                float wVal = wScalarDeque.GetValue(0);

                                // Compute tmp = xLine * wVal
                                AscendC::LocalTensor<float> tmpLine = qTmpLine_.AllocTensor<float>();
                                AscendC::Axpy(tmpLine, xLineDeque, wVal, W_); // dst = wVal * xLine

                                const uint32_t dx = kx;
                                const uint64_t yBase = OffsetY(n, co, zOut, y + dy, dx);

                                if (ci == 0 && kz == 0 && ky == 0) {
                                    // First contribution initializes y at this (dx) region
                                    AscendC::DataCopy(yGm_[yBase], tmpLine, W_);
                                } else {
                                    // Accumulate: y = y + tmp
                                    AscendC::LocalTensor<float> yLine = qYLine_.AllocTensor<float>();
                                    AscendC::DataCopy(yLine, yGm_[yBase], W_);
                                    AscendC::Add(yLine, yLine, tmpLine, W_);
                                    AscendC::DataCopy(yGm_[yBase], yLine, W_);
                                    qYLine_.FreeTensor(yLine);
                                }

                                qTmpLine_.FreeTensor(tmpLine);
                                qWScalar_.FreeTensor(wScalarDeque);
                                qWScalar_.FreeTensor(wScalarLocal);
                            }

                            qXLine_.FreeTensor(xLineDeque);
                            qXLine_.FreeTensor(xLine);
                        }
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> qXLine_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> qYLine_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> qTmpLine_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> qWScalar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t D_{0}, H_{0}, W_{0}, K_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.D, tiling_data.H, tiling_data.W, tiling_data.K,
            tiling_data.Dout, tiling_data.Hout, tiling_data.Wout);
    op.Process();
}
