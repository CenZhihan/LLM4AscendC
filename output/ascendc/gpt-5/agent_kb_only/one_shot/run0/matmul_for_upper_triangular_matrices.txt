host_operator_src="""

#include "register/op_def_registry.h"
#include "matmul_for_upper_triangular_matrices_custom_tiling.h"

// Matmul tiling headers
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"
#include "tiling/op_tiling.h"

namespace {
static inline uint32_t GetSquareN(const gert::Shape* s) {
    // Expect (N, N)
    if (s->GetStorageShape().GetDimNum() == 2) {
        return static_cast<uint32_t>(s->GetStorageShape().GetDim(0));
    }
    // Fallback to flat size sqrt
    uint64_t sz = s->GetOriginShape().GetShapeSize();
    uint32_t n = 0;
    for (uint32_t k = 1; (uint64_t)k * (uint64_t)k <= sz; ++k) {
        if ((uint64_t)k * (uint64_t)k == sz) n = k;
    }
    return n;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *ashape = context->GetInputShape(0);
    const gert::Shape *bshape = context->GetInputShape(1);
    if (ashape == nullptr || bshape == nullptr) {
        return ge::INTERNAL_ERROR;
    }

    const uint32_t N = GetSquareN(ashape);

    // Configure single-core Matmul tiling
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling tiling(ascendcPlatform);
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBias(false);
    tiling.SetShape(N, N, N);
    tiling.SetOrgShape(N, N, N);
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::INTERNAL_ERROR;
    }

    // Run on single core to apply TRIU masking easily after GEMM
    const uint32_t blockDim = 1U;
    context->SetBlockDim(blockDim);

    // Save TCubeTiling bytes directly into RawTilingData for kernel use
    auto *raw = context->GetRawTilingData();
    const uint8_t *src = reinterpret_cast<const uint8_t *>(&tilingData);
    uint8_t *dst = reinterpret_cast<uint8_t *>(raw->GetData());
    const size_t bytes = sizeof(optiling::TCubeTiling);
    if (raw->GetCapacity() < bytes) {
        return ge::INTERNAL_ERROR;
    }
    for (size_t i = 0; i < bytes; ++i) {
        dst[i] = src[i];
    }
    raw->SetDataSize(bytes);

    // Provide a minimal typed tiling for debugging (not used in kernel)
    MatmulForUpperTriangularMatricesCustomTilingData tdata;
    tdata.set_N(N);
    tdata.set_blockDim(blockDim);
    tdata.SaveToBuffer(context->GetTilingData()->GetData(), context->GetTilingData()->GetCapacity());
    context->GetTilingData()->SetDataSize(tdata.GetDataSize());

    // No workspace
    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *ashape = context->GetInputShape(0);
    const gert::Shape *bshape = context->GetInputShape(1);
    if (ashape == nullptr || bshape == nullptr) {
        return ge::INFER_STATUS_FAILED;
    }
    // Output shape equals matmul(A, B): (N, N)
    gert::Shape *yshape = context->GetOutputShape(0);
    *yshape = *ashape;
    return GRAPH_SUCCESS;
}

{
    // Output accumulates in float for better precision
    context->SetOutputDataType(0, ge::DT_FLOAT);

}

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = half;
using B_T = half;
using C_T = float;

extern "C" __global__ __aicore__ void matmul_for_upper_triangular_matrices_custom(GM_ADDR aGM,
                                                                                  GM_ADDR bGM,
                                                                                  GM_ADDR yGM,
                                                                                  GM_ADDR workspaceGM,
                                                                                  GM_ADDR tilingGM)
{
    // Define Matmul types (NORMAL layout)
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T> biasType;

    // Load TCubeTiling from raw tiling buffer
    optiling::TCubeTiling tiling;
    auto tempTilingGM = reinterpret_cast<__gm__ uint32_t*>(tilingGM);
    auto tempTiling   = reinterpret_cast<uint32_t*>(&tiling);
    for (int i = 0; i < static_cast<int>(sizeof(optiling::TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // Initialize GM tensors
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;

    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(C_T);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(yGM), sizeC);

    // Share buffer config
    tiling.shareMode  = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize  = 0;

    int offset_a = 0, offset_b = 0, offset_c = 0;

    AscendC::GlobalTensor<A_T> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal[offset_a].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreK);

    AscendC::GlobalTensor<B_T> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal[offset_b].GetPhyAddr()),
                         tiling.singleCoreK * tiling.singleCoreN);

    AscendC::GlobalTensor<C_T> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal[offset_c].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreN);

    // Create Matmul instance
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;

    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);
    mm1.Init(&tiling);
    mm1.SetTensorA(gm_a, false);  // A not transposed
    mm1.SetTensorB(gm_b, false);  // B not transposed (bType 'true' signifies column-major in interface)
    mm1.SetWorkspace(workspaceGM, 0);

    // Compute single-batch GEMM: C = A * B
    const int32_t batchA = 1;
    const int32_t batchB = 1;
    mm1.IterateBatch(gm_c, batchA, batchB, false);

    // Post-process: keep upper triangle of C (TRIU), zero out lower triangle in-place.
    // Assumption: ND contiguous [row-major] layout.
    const uint32_t N = static_cast<uint32_t>(tiling.singleCoreN);
    if (N > 1) {
        // Allocate a small UB buffer for zero chunks
        const uint32_t CHUNK_ELEMS = 2048;  // 8KB for float
        AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue;
        pipe.InitBuffer(tmpQueue, 1, CHUNK_ELEMS * sizeof(C_T));
        AscendC::LocalTensor<C_T> zeroBuf = tmpQueue.AllocTensor<C_T>();
        AscendC::Dup(zeroBuf, static_cast<C_T>(0.0f), CHUNK_ELEMS);

        for (uint32_t r = 1; r < N; ++r) {
            uint32_t lower = r;  // number of elements to zero in row r: columns [0 .. r-1]
            if (lower == 0) continue;
            uint64_t base = static_cast<uint64_t>(r) * static_cast<uint64_t>(N);
            uint64_t offset = base;  // start of row r

            while (lower > 0) {
                uint32_t len = (lower > CHUNK_ELEMS) ? CHUNK_ELEMS : lower;
                AscendC::DataCopy(gm_c[static_cast<uint32_t>(offset)], zeroBuf, len);
                offset += len;
                lower  -= len;
            }
        }

        tmpQueue.FreeTensor(zeroBuf);
    }
}

at::Tensor matmul_upper_tri_impl_npu(const at::Tensor& A, const at::Tensor& B) {

    at::Tensor Y = at::empty({A.size(0), B.size(1)}, A.options().dtype(at::kFloat));
    EXEC_NPU_CMD(aclnnMatmulForUpperTriangularMatricesCustom, A, B, Y);
    return Y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_for_upper_triangular_matrices_custom", &matmul_upper_tri_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
