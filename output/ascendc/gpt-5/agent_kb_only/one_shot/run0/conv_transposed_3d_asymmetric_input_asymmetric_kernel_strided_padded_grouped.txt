class KernelConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Din, uint32_t Hin, uint32_t Win,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t padD, uint32_t padH, uint32_t padW,
                                uint32_t outPadD, uint32_t outPadH, uint32_t outPadW,
                                uint32_t dilD, uint32_t dilH, uint32_t dilW,
                                uint32_t groups,
                                uint32_t Do, uint32_t Ho, uint32_t Wo,
                                uint32_t totalOutElems, uint32_t tileOutElems)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Din_ = Din; Hin_ = Hin; Win_ = Win;
        kD_ = kD; kH_ = kH; kW_ = kW;
        sD_ = strideD; sH_ = strideH; sW_ = strideW;
        pD_ = padD; pH_ = padH; pW_ = padW;
        oPD_ = outPadD; oPH_ = outPadH; oPW_ = outPadW;
        dD_ = dilD; dH_ = dilH; dW_ = dilW;
        groups_ = groups;
        Do_ = Do; Ho_ = Ho; Wo_ = Wo;
        totalOut_ = totalOutElems;
        tileOut_ = tileOutElems;
        CoutPerGroup_ = Cout_ / groups_;
        CinPerGroup_ = Cin_ / groups_;

        const uint64_t xSize = (uint64_t)N_*Cin_*Din_*Hin_*Win_;
        const uint64_t wSize = (uint64_t)Cin_*CoutPerGroup_*kD_*kH_*kW_;
        const uint64_t ySize = (uint64_t)N_*Cout_*Do_*Ho_*Wo_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(inQW_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(outQY_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(calcQ_, 2, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileOut_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint32_t strideElems = blockNum * tileOut_;
        uint32_t base = blockIdx * tileOut_;
        while (base < totalOut_) {
            const uint32_t cur = ((totalOut_ - base) > tileOut_) ? tileOut_ : (totalOut_ - base);
            ComputeTile(base, cur);
            base += strideElems;
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        return (((((uint64_t)n * Cin_ + c) * Din_ + d) * Hin_ + h) * Win_ + w);
    }
    __aicore__ inline uint64_t OffsetW(uint32_t ci, uint32_t co_g, uint32_t kd, uint32_t kh, uint32_t kw) const
    {
        return ((((((uint64_t)ci * CoutPerGroup_) + co_g) * kD_ + kd) * kH_ + kh) * kW_ + kw);
    }
    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        return (((((uint64_t)n * Cout_ + c) * Do_ + d) * Ho_ + h) * Wo_ + w);
    }

    __aicore__ inline void DecodeOutIndex(uint32_t lin, uint32_t &n, uint32_t &co, uint32_t &od, uint32_t &oh, uint32_t &ow) const
    {
        uint32_t t = lin;
        ow = t % Wo_; t /= Wo_;
        oh = t % Ho_; t /= Ho_;
        od = t % Do_; t /= Do_;
        co = t % Cout_; t /= Cout_;
        n  = t;
    }

    __aicore__ inline void ComputeTile(uint32_t base, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t n, co, od, oh, ow;
            DecodeOutIndex(base + i, n, co, od, oh, ow);
            const uint32_t g = co / CoutPerGroup_;
            const uint32_t co_g = co % CoutPerGroup_;
            const uint32_t cinStart = g * CinPerGroup_;
            const uint32_t cinEnd = cinStart + CinPerGroup_;

            bool hasAcc = false;
            AscendC::LocalTensor<float> sumLocal = calcQ_.AllocTensor<float>();

            for (uint32_t kd = 0; kd < kD_; ++kd) {
                int64_t id_num = (int64_t)od + (int64_t)pD_ - (int64_t)kd * (int64_t)dD_;
                if (id_num < 0) continue;
                if ((id_num % (int64_t)sD_) != 0) continue;
                uint32_t id = (uint32_t)(id_num / (int64_t)sD_);
                if (id >= Din_) continue;

                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    int64_t ih_num = (int64_t)oh + (int64_t)pH_ - (int64_t)kh * (int64_t)dH_;
                    if (ih_num < 0) continue;
                    if ((ih_num % (int64_t)sH_) != 0) continue;
                    uint32_t ih = (uint32_t)(ih_num / (int64_t)sH_);
                    if (ih >= Hin_) continue;

                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        int64_t iw_num = (int64_t)ow + (int64_t)pW_ - (int64_t)kw * (int64_t)dW_;
                        if (iw_num < 0) continue;
                        if ((iw_num % (int64_t)sW_) != 0) continue;
                        uint32_t iw = (uint32_t)(iw_num / (int64_t)sW_);
                        if (iw >= Win_) continue;

                        for (uint32_t ci = cinStart; ci < cinEnd; ++ci) {
                            // Load x(ci, id, ih, iw)
                            AscendC::LocalTensor<float> xLocal = inQX_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[OffsetX(n, ci, id, ih, iw)], 1);

                            // Load w(ci, co_g, kd, kh, kw)
                            AscendC::LocalTensor<float> wLocal = inQW_.AllocTensor<float>();
                            AscendC::DataCopy(wLocal, wGm_[OffsetW(ci, co_g, kd, kh, kw)], 1);

                            // prod = x * w
                            AscendC::LocalTensor<float> prodLocal = calcQ_.AllocTensor<float>();
                            AscendC::Mul(prodLocal, xLocal, wLocal, 1);

                            if (!hasAcc) {
                                AscendC::DataCopy(sumLocal, prodLocal, 1);
                                hasAcc = true;
                            } else {
                                AscendC::Add(sumLocal, sumLocal, prodLocal, 1);
                            }

                            calcQ_.FreeTensor(prodLocal);
                            inQW_.FreeTensor(wLocal);
                            inQX_.FreeTensor(xLocal);
                        }
                    }
                }
            }

            if (!hasAcc) {
                // Fallback: write zero
                AscendC::LocalTensor<float> yZero = outQY_.AllocTensor<float>();
                // Use sumLocal as source, make it zero by subtracting itself (ensure it's initialized)
                AscendC::Sub(yZero, sumLocal, sumLocal, 1);
                AscendC::DataCopy(yGm_[OffsetY(n, co, od, oh, ow)], yZero, 1);
                outQY_.FreeTensor(yZero);
            } else {
                AscendC::DataCopy(yGm_[OffsetY(n, co, od, oh, ow)], sumLocal, 1);
            }

            calcQ_.FreeTensor(sumLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQX_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQW_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQY_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 2> calcQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, Cin_, Cout_;
    uint32_t Din_, Hin_, Win_;
    uint32_t kD_, kH_, kW_;
    uint32_t sD_, sH_, sW_;
    uint32_t pD_, pH_, pW_;
    uint32_t oPD_, oPH_, oPW_;
    uint32_t dD_, dH_, dW_;
    uint32_t groups_;

    uint32_t totalOut_, tileOut_;
    uint32_t CoutPerGroup_, CinPerGroup_;
};
