host_operator_src="""

#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"
#include "tiling/platform_ascendc.h"
#include "tiling/matmul_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    auto *shapeA = context->GetInputShape(0);
    auto *shapeB = context->GetInputShape(1);

    // Expect A: [M, K], B: [N, K] (we will transpose B in kernel to [K, N])
    const uint32_t M = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(shapeB->GetOriginShape().GetDim(0));

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);

    // Choose a conservative default number of cores. Runtime may adjust internally.
    tiling.SetDim(24);

    // A and B are float16, C is float32
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBias(false);

    // Set shapes: singleCore computation and original shapes (M, N, K)
    tiling.SetShape(M, N, K);
    tiling.SetSingleShape(M, N, K);
    tiling.SetOrgShape(M, N, K);

    // Configure optional params for NORMAL layout
    matmul_tiling::MatmulConfigParams cfg(0 /* Norm template */);
    tiling.SetMatmulConfigParams(cfg);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Set runtime block dim from tiling
    context->SetBlockDim(tiling.GetCoreNum());

    // Save tiling to raw buffer
    auto *raw = context->GetRawTilingData();
    auto *buf = raw->GetData();
    const size_t sz = sizeof(tilingData);
    if (raw->GetCapacity() < sz) {
        return ge::GRAPH_FAILED;
    }
    ::memcpy(buf, &tilingData, sz);
    raw->SetDataSize(sz);

    // No workspace required by default
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *A_shape = context->GetInputShape(0);
    const gert::Shape *B_shape = context->GetInputShape(1);
    gert::Shape *C_shape = context->GetOutputShape(0);

    // A: [M, K], B: [N, K], output should be [M, N]
    int64_t M = A_shape->GetOriginShape().GetDim(0);
    int64_t N = B_shape->GetOriginShape().GetDim(0);

    std::vector<int64_t> out_dims = {M, N};
    C_shape->SetDimNum(2);
    C_shape->SetDims(out_dims);

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output is float32 by definition
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

class MatmulWithTransposedBCustom : public OpDef {
public:
    explicit MatmulWithTransposedBCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape)
            .SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulWithTransposedBCustom);

} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = half;
using B_T = half;
using C_T = float;
using BiasT = float;

typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false,
LayoutMode::NORMAL> aType;
typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true,
LayoutMode::NORMAL> bType;
typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false,
LayoutMode::NORMAL> cType;
typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

extern "C" __global__ __aicore__ void matmul_with_transposed_b_custom(GM_ADDR aGM, GM_ADDR bGM,
                                                                       GM_ADDR cGM, GM_ADDR workspaceGM,
                                                                       GM_ADDR tilingGM)
{
    // Load TCubeTiling
    TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // Initialize GM tensors
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;

    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(C_T);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(cGM), sizeC);

    // Share buffer settings
    tiling.shareMode = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize = 0;

    // Create and run Matmul
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;
    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);

    mm1.Init(&tiling);
    // A not transposed, B transposed
    mm1.SetTensorA(aGlobal, 0);
    mm1.SetTensorB(bGlobal, 1);
    mm1.SetWorkspace(workspaceGM, 0);

    // Single batch computation
    mm1.IterateBatch(cGlobal, 1, 1, false);
}
"""

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_contig_dtype(const at::Tensor& x, at::ScalarType dt) {
    if (x.scalar_type() == dt && x.is_contiguous()) return x;
    return x.to(dt).contiguous();
}

at::Tensor matmul_with_transposed_b_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Expect A: [M, K] and B: [N, K]; kernel will use B^T implicitly.
    // Cast A, B to float16 for better performance; output is float32.
    at::Tensor A_half = to_contig_dtype(A, at::kHalf);
    at::Tensor B_half = to_contig_dtype(B, at::kHalf);

    // Output shape: [M, N]
    auto M = A.size(-2);
    auto N = B.size(-2);
    at::Tensor C = at::empty({M, N}, A.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnMatmulWithTransposedBCustom, A_half, B_half, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_with_transposed_b_custom", &matmul_with_transposed_b_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_with_transposed_b_custom", &matmul_with_transposed_b_impl_npu,
          "Matmul with transposed B (MatmulWithTransposedBCustom on NPU)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'matmul_with_transposed_b_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        # A: [M, K], B: [N, K]; kernel performs A @ B.T internally.
        return fn(A, B)
