#include "conv_transposed_1d_dilated_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : ((x / a) * a);
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model:
//   yLocal: cOutChunk * tileT
//   xSeg  : 1 * tileT
//   tmp   : 1 * tileT
//   wScalar buffer (1 element) used transiently
// bytes ~= 4 * (cOutChunk + 2) * tileT
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_COUT_CHUNK = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed1dDilatedCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 3 || w_shape->GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t Lin = static_cast<uint32_t>(x_shape->GetDim(2));

    const uint32_t W_Cin = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Cout  = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t K     = static_cast<uint32_t>(w_shape->GetDim(2));

    if (W_Cin != Cin) {
        return ge::GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stride_p   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *padding_p  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilation_p = attrs->GetAttrPointer<int32_t>(2);

    const uint32_t stride   = static_cast<uint32_t>(*stride_p);
    const uint32_t padding  = static_cast<uint32_t>(*padding_p);
    const uint32_t dilation = static_cast<uint32_t>(*dilation_p);

    // Lout = (Lin - 1) * stride - 2*padding + dilation*(K - 1) + 1
    int64_t Lout_calc = static_cast<int64_t>(Lin - 1) * static_cast<int64_t>(stride)
                      - static_cast<int64_t>(2 * padding)
                      + static_cast<int64_t>(dilation) * static_cast<int64_t>(K - 1)
                      + 1;
    if (Lout_calc <= 0) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t Lout = static_cast<uint32_t>(Lout_calc);

    uint32_t cOutChunk = (Cout < MAX_COUT_CHUNK) ? Cout : MAX_COUT_CHUNK;

    // tileT from UB budget: bytes = 4 * (cOutChunk + 2) * tileT
    uint32_t maxTileT = UB_BUDGET_BYTES / (4 * (cOutChunk + 2));
    maxTileT = AlignDown(maxTileT, ALIGN_ELEMS);
    if (maxTileT < ALIGN_ELEMS) maxTileT = ALIGN_ELEMS;

    uint32_t tileT = maxTileT;
    if (Lout < tileT) {
        const uint32_t aligned = AlignDown(Lout, ALIGN_ELEMS);
        tileT = (aligned > 0) ? aligned : Lout;
    }

    const uint32_t tilesPerN = (tileT == 0) ? 1u : CeilDiv(Lout, tileT);
    uint32_t blockDim = N * tilesPerN;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_lin(Lin);
    tiling.set_lout(Lout);
    tiling.set_ksize(K);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_dilation(dilation);
    tiling.set_tileT(tileT);
    tiling.set_cOutChunk(cOutChunk);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    const gert::Shape *w_shape = context->GetInputShape(1);

    const int64_t Cout  = w_shape->GetDim(1);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvTransposed1dDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed1dDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t lout, uint32_t ksize,
                                uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t tileT, uint32_t cOutChunk)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        lin_ = lin; lout_ = lout; ksize_ = ksize;
        stride_ = stride; padding_ = padding; dilation_ = dilation;
        tileT_ = tileT; cChunk_ = cOutChunk;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cin_ * lin_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, cin_ * cout_ * ksize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * cout_ * lout_);

        // Buffers:
        // yLocal: cChunk_ * tileT_
        // xSeg  : tileT_
        // tmp   : tileT_
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, cChunk_ * tileT_ * sizeof(float));
        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileT_ * sizeof(float));
        pipe_.InitBuffer(tQueue_, BUFFER_NUM, tileT_ * sizeof(float));
        pipe_.InitBuffer(wScalarQueue_, BUFFER_NUM, sizeof(float)); // scalar weight buffer
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cout_ == 0 || lin_ == 0 || lout_ == 0 || tileT_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // total tiles across N: tilesPerN for each n
        const uint32_t tilesPerN = (lout_ + tileT_ - 1) / tileT_;
        const uint32_t totalTiles = n_ * tilesPerN;

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t nIdx = tileId / tilesPerN;
            const uint32_t tIdx = tileId % tilesPerN;
            const uint32_t t0 = tIdx * tileT_;
            uint32_t curT = lout_ - t0;
            if (curT > tileT_) curT = tileT_;

            ComputeTile(nIdx, t0, curT);
        }
    }

private:
    __aicore__ inline void ZeroY(AscendC::LocalTensor<float> &yLocal, uint32_t elems)
    {
        AscendC::Duplicate(yLocal, 0.0f, elems);
    }

    __aicore__ inline void ComputeTile(uint32_t nIdx, uint32_t t0, uint32_t curT)
    {
        AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> xSeg   = xQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp    = tQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> wScal  = wScalarQueue_.AllocTensor<float>(); // length 1

        const uint32_t yLocalTileStride = tileT_; // stride between channel slices

        for (uint32_t cBase = 0; cBase < cout_; cBase += cChunk_) {
            const uint32_t curChunk = (cBase + cChunk_ <= cout_) ? cChunk_ : (cout_ - cBase);
            ZeroY(yLocal, curChunk * curT);

            for (uint32_t ci = 0; ci < cin_; ++ci) {
                for (uint32_t k = 0; k < ksize_; ++k) {
                    // offset for x index: idx_in = t + padding - k*dilation when stride == 1
                    // For general stride, we only accumulate when (t + padding - k*dilation) % stride == 0,
                    // which we do implicitly by contiguous copy only when stride==1.
                    if (stride_ != 1) {
                        // Not supported in this optimized path; skip to avoid incorrect results.
                        continue;
                    }
                    // Valid t range: 0 <= t + offset < lin_, where offset = padding - k * dilation
                    const int32_t offset = static_cast<int32_t>(padding_) - static_cast<int32_t>(k * dilation_);
                    int32_t t_valid_start = static_cast<int32_t>(t0);
                    int32_t t_valid_end   = static_cast<int32_t>(t0 + curT);
                    const int32_t lower = -offset;
                    const int32_t upper = static_cast<int32_t>(lin_) - offset;
                    if (t_valid_start < lower) t_valid_start = lower;
                    if (t_valid_end > upper)   t_valid_end   = upper;
                    if (t_valid_end <= t_valid_start) {
                        continue;
                    }
                    const uint32_t len = static_cast<uint32_t>(t_valid_end - t_valid_start);
                    const uint32_t tOff = static_cast<uint32_t>(t_valid_start - static_cast<int32_t>(t0));
                    const uint32_t xStart = static_cast<uint32_t>(t_valid_start + offset);

                    // Load x segment: x[nIdx, ci, xStart : xStart + len]
                    const uint32_t xBase = ((nIdx * cin_ + ci) * lin_) + xStart;
                    AscendC::DataCopy(xSeg, xGm_[xBase], len);

                    // Accumulate for each cout in chunk
                    for (uint32_t cc = 0; cc < curChunk; ++cc) {
                        const uint32_t co = cBase + cc;
                        // weight index: w[ci, co, k]
                        const uint32_t wIdx = ((ci * cout_ + co) * ksize_) + k;
                        AscendC::DataCopy(wScal, wGm_[wIdx], 1);

                        // tmp = xSeg * wScal[0]
                        AscendC::Muls(tmp, xSeg, wScal.GetValue(0), len);

                        // yLocal slice for channel 'cc' starting at tOff
                        AscendC::LocalTensor<float> ySlice = yLocal[yLocalTileStride * cc + tOff];
                        AscendC::Add(ySlice, ySlice, tmp, len);
                    }
                }
            }

            // Store y for this chunk: y[nIdx, cBase.., t0:t0+curT]
            for (uint32_t cc = 0; cc < curChunk; ++cc) {
                const uint32_t co = cBase + cc;
                const uint32_t yBase = ((nIdx * cout_ + co) * lout_) + t0;
                AscendC::LocalTensor<float> ySlice = yLocal[yLocalTileStride * cc];
                AscendC::DataCopy(yGm_[yBase], ySlice, curT);
            }
        }

        wScalarQueue_.FreeTensor(wScal);
        tQueue_.FreeTensor(tmp);
        xQueue_.FreeTensor(xSeg);
        yQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  xQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> wScalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t lin_{0}, lout_{0}, ksize_{0};
    uint32_t stride_{1}, padding_{0}, dilation_{1};
    uint32_t tileT_{0}, cChunk_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    AscendC::InitSocState();
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1dDilatedCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.lout, tiling_data.ksize,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation,
            tiling_data.tileT, tiling_data.cOutChunk);
    op.Process();
}
