host_operator_src="""

#include "argmin_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // avoid too-large per-iter latency
// Buffers per inner tile: x (float) + min (float) + idx (int32) + mask (uint8) + kvec (int32)
static constexpr uint32_t BYTES_PER_ELEM = 4 + 4 + 4 + 1 + 4; // 17 bytes, round down by alignment later

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgminOverADimensionCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const auto &origin = shape->GetOriginShape();
    const int64_t rank = origin.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int64_t dim = static_cast<int64_t>(*dimPtr);
    int64_t ndims = rank;
    if (dim < 0) dim += ndims;
    if (dim < 0 || dim >= ndims) {
        return ge::GRAPH_FAILED;
    }

    uint64_t outerCount = 1;
    uint64_t innerCount = 1;
    uint64_t reduceLen = static_cast<uint64_t>(origin.GetDim(dim));

    for (int64_t i = 0; i < dim; ++i) {
        outerCount *= static_cast<uint64_t>(origin.GetDim(i));
    }
    for (int64_t i = dim + 1; i < ndims; ++i) {
        innerCount *= static_cast<uint64_t>(origin.GetDim(i));
    }

    uint32_t maxTile = UB_BUDGET_BYTES / BYTES_PER_ELEM;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileInner = static_cast<uint32_t>(maxTile);
    if (innerCount < tileInner) {
        uint32_t innerAligned = AlignDown(static_cast<uint32_t>(innerCount), ALIGN_ELEMS);
        tileInner = (innerAligned > 0) ? innerAligned : static_cast<uint32_t>(innerCount);
    }
    if (tileInner == 0) tileInner = 1;

    uint32_t blockDim = static_cast<uint32_t>(outerCount);
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_outerCount(static_cast<uint32_t>(outerCount));
    tiling.set_innerCount(static_cast<uint32_t>(innerCount));
    tiling.set_reduceLen(static_cast<uint32_t>(reduceLen));
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin = x_shape->GetOriginShape();
    const int64_t rank = origin.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int64_t dim = static_cast<int64_t>(*dimPtr);
    if (dim < 0) dim += rank;

    std::vector<int64_t> out_dims;
    out_dims.reserve((rank > 0) ? (rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out_dims.emplace_back(origin.GetDim(i));
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(static_cast<uint32_t>(out_dims.size()));
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT32);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t IN_BUFFER_NUM = 2;

class KernelArgminOverADimensionCustom {
public:
    __aicore__ inline KernelArgminOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerCount, uint32_t innerCount,
                                uint32_t reduceLen, uint32_t tileInner)
    {
        outerCount_ = outerCount;
        innerCount_ = innerCount;
        reduceLen_ = reduceLen;
        tileInner_ = tileInner;

        totalIn_ = static_cast<uint64_t>(outerCount_) * reduceLen_ * innerCount_;
        totalOut_ = static_cast<uint64_t>(outerCount_) * innerCount_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalIn_);
        yGm_.SetGlobalBuffer((__gm__ int32_t*)y, totalOut_);

        pipe_.InitBuffer(inQueueX_, IN_BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(minQueue_, 1, tileInner_ * sizeof(float));
        pipe_.InitBuffer(idxQueue_, 1, tileInner_ * sizeof(int32_t));
        pipe_.InitBuffer(maskQueue_, 1, tileInner_ * sizeof(uint8_t));
        pipe_.InitBuffer(tmpIntQueue_, 1, tileInner_ * sizeof(int32_t));
    }

    __aicore__ inline void Process()
    {
        if (outerCount_ == 0 || innerCount_ == 0 || reduceLen_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t o = blockIdx; o < outerCount_; o += blockNum) {
            const uint64_t baseO = static_cast<uint64_t>(o) * reduceLen_ * innerCount_;

            for (uint32_t innerStart = 0; innerStart < innerCount_; innerStart += tileInner_) {
                uint32_t curInner = innerCount_ - innerStart;
                if (curInner > tileInner_) curInner = tileInner_;

                // Allocate persistent min/idx buffers for this inner tile
                LocalTensor<float> minLocal = minQueue_.AllocTensor<float>();
                LocalTensor<int32_t> idxLocal = idxQueue_.AllocTensor<int32_t>();
                // Initialize min to +inf and idx to 0
                AscendC::Dup(minLocal, static_cast<float>(INFINITY), curInner);
                AscendC::Dup(idxLocal, 0, curInner);

                // Double-buffered streaming over reduce dimension
                uint32_t k = 0;
                // Prefetch first slice
                if (k < reduceLen_) {
                    CopyInSlice(baseO, k, innerStart, curInner);
                    k++;
                }
                for (; k <= reduceLen_; ++k) {
                    // Prefetch next slice if any
                    if (k < reduceLen_) {
                        CopyInSlice(baseO, k, innerStart, curInner);
                    }

                    // Compute on current slice
                    ComputeUpdate(minLocal, idxLocal, /*kIdx*/ (k - 1), curInner);

                    if (k >= reduceLen_) break;
                }

                // Write out indices for this inner tile
                const uint64_t yOff = static_cast<uint64_t>(o) * innerCount_ + innerStart;
                LocalTensor<int32_t> idxOut = idxQueue_.DeQue<int32_t>();
                AscendC::DataCopy(yGm_[yOff], idxOut, curInner);
                idxQueue_.FreeTensor(idxOut);

                // Free min buffer
                LocalTensor<float> minBuf = minQueue_.DeQue<float>();
                minQueue_.FreeTensor(minBuf);
            }
        }
    }

private:
    __aicore__ inline void CopyInSlice(uint64_t baseO, uint32_t k, uint32_t innerStart, uint32_t len)
    {
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        const uint64_t gOff = baseO + static_cast<uint64_t>(k) * innerCount_ + innerStart;
        AscendC::DataCopy(xLocal, xGm_[gOff], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeUpdate(LocalTensor<float>& minLocal,
                                         LocalTensor<int32_t>& idxLocal,
                                         uint32_t kIdx, uint32_t len)
    {
        LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        // mask where x < min
        LocalTensor<uint8_t> maskLocal = maskQueue_.AllocTensor<uint8_t>();
        AscendC::Less(maskLocal, xLocal, minLocal, len);

        // minLocal = min(minLocal, xLocal)
        AscendC::Min(minLocal, minLocal, xLocal, len);

        // Prepare k vector
        LocalTensor<int32_t> kVec = tmpIntQueue_.AllocTensor<int32_t>();
        AscendC::Dup(kVec, static_cast<int32_t>(kIdx), len);

        // idxLocal = mask ? k : idxLocal
        AscendC::Select(idxLocal, maskLocal, kVec, idxLocal, len);

        // Free temps
        tmpIntQueue_.FreeTensor(kVec);
        maskQueue_.FreeTensor(maskLocal);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> minQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> idxQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maskQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpIntQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int32_t> yGm_;

    uint32_t outerCount_{0};
    uint32_t innerCount_{0};
    uint32_t reduceLen_{0};
    uint32_t tileInner_{0};

    uint64_t totalIn_{0};
    uint64_t totalOut_{0};
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgminOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.outerCount,
            tiling_data.innerCount,
            tiling_data.reduceLen,
            tiling_data.tileInner);
    op.Process();
}

#include <torch/library.h>

#include "pytorch_npu_helper.hpp"

at::Tensor argmin_over_a_dimension_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    int64_t ndims = self.dim();
    int64_t dimNorm = dim >= 0 ? dim : (dim + ndims);

    std::vector<int64_t> out_sizes = self.sizes().vec();
    if (dimNorm < 0 || dimNorm >= ndims) {
        TORCH_CHECK(false, "Invalid dim for argmin_over_a_dimension_custom");
    }
    out_sizes.erase(out_sizes.begin() + dimNorm);

    auto options = self.options().dtype(at::kInt);
    at::Tensor out_i32 = at::empty(out_sizes, options);

    EXEC_NPU_CMD(aclnnArgminOverADimensionCustom, self, static_cast<int>(dimNorm), out_i32);

    return out_i32.to(at::kLong);
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmin_over_a_dimension_custom", &argmin_over_a_dimension_custom_impl_npu);
