#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR lin_b, GM_ADDR b,
                                GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t rows, uint32_t inFeat, uint32_t outFeat,
                                uint32_t tileK, uint32_t groupSize, uint32_t numGroups, float epsilon)
    {
        rows_ = rows;
        inFeat_ = inFeat;
        outFeat_ = outFeat;
        tileK_ = tileK;
        groupSize_ = groupSize;
        numGroups_ = numGroups;
        eps_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)rows_ * inFeat_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)outFeat_ * inFeat_);
        linbGm_.SetGlobalBuffer((__gm__ float*)lin_b, outFeat_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, outFeat_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, outFeat_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, outFeat_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)rows_ * outFeat_);

        // UB buffers
        pipe_.InitBuffer(xTileQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(wTileQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(accQ_, BUFFER_NUM, groupSize_ * sizeof(float));
        pipe_.InitBuffer(valQ_, BUFFER_NUM, groupSize_ * sizeof(float));
        pipe_.InitBuffer(gammaQ_, BUFFER_NUM, groupSize_ * sizeof(float));
        pipe_.InitBuffer(betaQ_, BUFFER_NUM, groupSize_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, BUFFER_NUM, 4 * sizeof(float)); // room for a few scalars
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || inFeat_ == 0 || outFeat_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        for (uint32_t n = blockIdx; n < rows_; n += blockNum) {
            // Iterate groups on output channels
            for (uint32_t g = 0; g < numGroups_; ++g) {
                const uint32_t cStart = g * groupSize_;

                // acc vector init to 0
                AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
                AscendC::Dup(acc, 0.0f, groupSize_);

                // K loop
                for (uint32_t kBase = 0; kBase < inFeat_; kBase += tileK_) {
                    uint32_t curK = inFeat_ - kBase;
                    if (curK > tileK_) curK = tileK_;

                    AscendC::LocalTensor<float> xTile = xTileQ_.AllocTensor<float>();
                    AscendC::DataCopy(xTile, xGm_[n * inFeat_ + kBase], curK);

                    // For each channel within the group, accumulate dot
                    for (uint32_t c = 0; c < groupSize_; ++c) {
                        const uint32_t cAbs = cStart + c;

                        AscendC::LocalTensor<float> wTile = wTileQ_.AllocTensor<float>();
                        AscendC::DataCopy(wTile, wGm_[cAbs * inFeat_ + kBase], curK);

                        AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                        AscendC::Mul(tmp, xTile, wTile, curK);
                        // ReduceSum tmp -> scalar
                        AscendC::LocalTensor<float> scal = scalarQ_.AllocTensor<float>();
                        AscendC::ReduceSum(scal, tmp, curK);

                        // acc[c] += scal[0]
                        // We'll create a temporary 1-len vector and add to acc[c]
                        // Create a view offset for acc[c]
                        // Since there's no direct scalar add to element, broadcast and add a single element region
                        // Use Muls of scalar then Add with vector slice length 1
                        AscendC::LocalTensor<float> oneElem = scal; // reuse scal as one-element
                        // Write into valQ_ temporarily a 1-element tensor at position c to add
                        AscendC::LocalTensor<float> val = valQ_.AllocTensor<float>();
                        // Use Dup to zero, then place the scalar at position c
                        AscendC::Dup(val, 0.0f, groupSize_);
                        // Expand scalar to len=1 and place at index c
                        // We cannot place at index directly; instead we'll Add with mask using adds for slice
                        // Alternative: Load acc into val, then Add with scalar to a one-length slice and write back.
                        // Simpler: Read acc[c] to scalar, add, write back:
                        AscendC::LocalTensor<float> accScalar = scalarQ_.AllocTensor<float>(); // reuse space
                        // Copy acc[c] to accScalar[0]
                        AscendC::DataCopy(accScalar, acc[c], 1);
                        AscendC::Add(accScalar, accScalar, oneElem, 1);
                        AscendC::DataCopy(acc[c], accScalar, 1);

                        scalarQ_.FreeTensor(accScalar);
                        valQ_.FreeTensor(val);
                        tmpQ_.FreeTensor(tmp);
                        wTileQ_.FreeTensor(wTile);
                        scalarQ_.FreeTensor(scal);
                    }

                    xTileQ_.FreeTensor(xTile);
                }

                // Add biases: lin_bias + bias
                AscendC::LocalTensor<float> vals = valQ_.AllocTensor<float>();
                AscendC::DataCopy(vals, acc, groupSize_);

                AscendC::LocalTensor<float> linb = gammaQ_.AllocTensor<float>(); // reuse as temp for bias
                AscendC::LocalTensor<float> bb = betaQ_.AllocTensor<float>();    // reuse as temp for bias
                AscendC::DataCopy(linb, linbGm_[cStart], groupSize_);
                AscendC::DataCopy(bb, bGm_[cStart], groupSize_);
                AscendC::Add(vals, vals, linb, groupSize_);
                AscendC::Add(vals, vals, bb, groupSize_);

                // HardTanh clamp to [-1, 1]
                AscendC::Maxs(vals, vals, -1.0f, groupSize_);
                AscendC::Mins(vals, vals, 1.0f, groupSize_);

                // Mish: x * tanh(softplus(x)), softplus = ln(1 + exp(x))
                AscendC::LocalTensor<float> expv = accQ_.AllocTensor<float>(); // reuse accQ_ buffer for exp
                AscendC::Exp(expv, vals, groupSize_);
                AscendC::Add(expv, expv, 1.0f, groupSize_); // expv = exp(vals) + 1
                AscendC::Ln(expv, expv, groupSize_);        // expv = softplus
                AscendC::Tanh(expv, expv, groupSize_);      // expv = tanh(softplus)
                AscendC::Mul(vals, vals, expv, groupSize_); // vals = mish(vals)

                // GroupNorm over this group: mean and var over channels within the group for this sample
                // mean
                AscendC::LocalTensor<float> meanT = scalarQ_.AllocTensor<float>();
                AscendC::ReduceSum(meanT, vals, groupSize_);
                AscendC::LocalTensor<float> meanVec = accQ_.AllocTensor<float>();
                AscendC::Dup(meanVec, 0.0f, groupSize_);
                AscendC::LocalTensor<float> invGS = scalarQ_.AllocTensor<float>();
                // inv group size
                float invG = 1.0f / static_cast<float>(groupSize_);
                AscendC::Dup(invGS, invG, 1);
                AscendC::Muls(meanT, meanT, invG, 1);
                // broadcast mean to vector
                AscendC::Dup(meanVec, meanT.GetValue(0), groupSize_);

                // variance
                AscendC::Sub(expv, vals, meanVec, groupSize_);    // reuse expv as (vals - mean)
                AscendC::Mul(expv, expv, expv, groupSize_);       // square
                AscendC::LocalTensor<float> varT = scalarQ_.AllocTensor<float>();
                AscendC::ReduceSum(varT, expv, groupSize_);
                AscendC::Muls(varT, varT, invG, 1);               // var = E[(x-mean)^2]

                // rsqrt(var + eps)
                AscendC::LocalTensor<float> rstd = scalarQ_.AllocTensor<float>();
                // rstd = 1 / sqrt(var + eps)
                float eps_local = eps_;
                AscendC::Add(varT, varT, eps_local, 1);
                AscendC::Rsqrt(rstd, varT, 1);

                // normalization: (vals - mean) * rstd
                AscendC::Sub(vals, vals, meanVec, groupSize_);
                AscendC::Muls(vals, vals, rstd.GetValue(0), groupSize_);

                // scale + shift: gamma, beta
                AscendC::DataCopy(linb, gammaGm_[cStart], groupSize_); // linb reused as gamma
                AscendC::DataCopy(bb, betaGm_[cStart], groupSize_);    // bb reused as beta
                AscendC::Mul(vals, vals, linb, groupSize_);
                AscendC::Add(vals, vals, bb, groupSize_);

                // Store to GM
                AscendC::DataCopy(yGm_[n * outFeat_ + cStart], vals, groupSize_);

                // Free locals
                scalarQ_.FreeTensor(rstd);
                scalarQ_.FreeTensor(varT);
                scalarQ_.FreeTensor(invGS);
                accQ_.FreeTensor(meanVec);
                scalarQ_.FreeTensor(meanT);
                accQ_.FreeTensor(expv);
                betaQ_.FreeTensor(bb);
                gammaQ_.FreeTensor(linb);
                valQ_.FreeTensor(vals);
                accQ_.FreeTensor(acc);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> accQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> valQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> gammaQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> betaQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> linbGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t inFeat_ {0};
    uint32_t outFeat_ {0};
    uint32_t tileK_ {0};
    uint32_t groupSize_ {0};
    uint32_t numGroups_ {0};
    float eps_ {1e-5f};
};
