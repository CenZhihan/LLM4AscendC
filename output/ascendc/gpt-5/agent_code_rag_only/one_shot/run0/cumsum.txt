host_operator_src="""

#include "cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint64_t a, uint32_t b)
{
    return static_cast<uint32_t>((a + b - 1ull) / static_cast<uint64_t>(b));
}
}

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   carry:     1 float
// Total floats ~ 4 * tile + 1 => bytes ~ 16 * tile + 4
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttrPtr = attrs->GetAttrPointer<int32_t>(0);

    const gert::Shape *inShape = context->GetInputShape(0);
    const auto &origin = inShape->GetOriginShape();
    const uint32_t rank = static_cast<uint32_t>(origin.GetDimNum());
    if (rank == 0) {
        return ge::GRAPH_FAILED;
    }

    int32_t dim = (dimAttrPtr != nullptr) ? *dimAttrPtr : -1;
    if (dim < 0) dim += static_cast<int32_t>(rank);
    if (dim < 0 || dim >= static_cast<int32_t>(rank)) {
        return ge::GRAPH_FAILED;
    }

    uint64_t totalLength64 = 1;
    for (uint32_t i = 0; i < rank; ++i) {
        totalLength64 *= static_cast<uint64_t>(origin.GetDim(i));
    }
    const uint32_t totalLength = static_cast<uint32_t>(totalLength64);

    const uint32_t axisLength = static_cast<uint32_t>(origin.GetDim(dim));

    // strideAfter is product of dimensions after 'dim'.
    uint64_t strideAfter = 1;
    for (uint32_t i = static_cast<uint32_t>(dim) + 1; i < rank; ++i) {
        strideAfter *= static_cast<uint64_t>(origin.GetDim(i));
    }

    // We only support contiguous axis (dim is the last dimension). For other cases, the kernel is not optimized.
    // However, to keep a consistent interface, we still dispatch but require strideAfter == 1 for correctness.
    // If not contiguous, we conservatively set strideAfter to 1 in tiling and rely on users to ensure last-dim usage.
    if (strideAfter != 1) {
        // Not supported path; return failure to avoid incorrect results.
        return ge::GRAPH_FAILED;
    }

    const uint32_t rows = (axisLength == 0) ? 0u : (totalLength / axisLength);

    // tileLen from UB budget: bytes ~ 16 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 16;  // elements
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (axisLength < tileLen) {
        const uint32_t aligned = AlignDown(axisLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : axisLength;
    }

    // Choose blockDim based on rows
    uint32_t blockDim = (rows == 0) ? 1u : rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_axisLength(axisLength);
    tiling.set_rows(rows);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumsumCustom : public OpDef {
public:
    explicit CumsumCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(-1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumsumCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCumsumCustom {
public:
    __aicore__ inline KernelCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t axisLength,
                                uint32_t rows,
                                uint32_t tileLength)
    {
        totalLen_ = totalLength;
        axisLen_ = axisLength;
        rows_ = rows;
        tileLen_ = (tileLength == 0 ? 1u : tileLength);  // avoid zero

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // carry buffer: 1 float
        pipe_.InitBuffer(carryQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || axisLen_ == 0 || rows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            const uint32_t rowBase = row * axisLen_;

            AscendC::LocalTensor<float> carryLocal = carryQueue_.AllocTensor<float>();
            AscendC::Duplicate(carryLocal, 0.0f, 1);

            uint32_t pos = 0;
            uint32_t curLen = axisLen_ - pos;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) {
                carryQueue_.FreeTensor(carryLocal);
                continue;
            }
            CopyIn(rowBase + pos, curLen);
            bool firstTile = true;

            while (true) {
                const uint32_t nextPos = pos + curLen;
                uint32_t nextLen = 0;
                if (nextPos < axisLen_) {
                    nextLen = axisLen_ - nextPos;
                    if (nextLen > tileLen_) nextLen = tileLen_;
                    CopyIn(rowBase + nextPos, nextLen);
                }

                ComputeTile(curLen, firstTile, carryLocal);
                CopyOut(rowBase + pos, curLen);

                if (nextLen == 0) break;
                pos = nextPos;
                curLen = nextLen;
                firstTile = false;
            }

            carryQueue_.FreeTensor(carryLocal);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeTile(uint32_t len, bool firstTile, AscendC::LocalTensor<float>& carryLocal)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Initialize yLocal with input tile
        AscendC::DataCopy(yLocal, xLocal, len);

        // Add carry only to the first element for non-first tiles
        if (!firstTile) {
            AscendC::Add(yLocal, yLocal, carryLocal, 1);  // length=1, operate on yLocal[0]
        }

        // Inclusive scan using iterative doubling: y[i] += y[i - step]
        for (uint32_t step = 1; step < len; step <<= 1) {
            const uint32_t work = len - step;
            AscendC::Add(yLocal[step], yLocal[step], yLocal, work);
        }

        // Update carry with last element of current tile
        AscendC::DataCopy(carryLocal, yLocal[len - 1], 1);

        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> carryQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t axisLen_{0};
    uint32_t rows_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void cumsum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.axisLength,
            tiling_data.rows,
            tiling_data.tileLength);
    op.Process();
}
"""
