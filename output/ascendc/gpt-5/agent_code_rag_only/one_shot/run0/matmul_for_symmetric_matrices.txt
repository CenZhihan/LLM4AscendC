host_operator_src="""
#include "matmul_for_symmetric_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}
}

namespace optiling {

// UB budget and alignment assumptions for float32
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per core on 910B
static constexpr uint32_t ALIGN_ELEMS = 256;            // encourage large DMA bursts
static constexpr uint32_t MIN_MTILE = 32;
static constexpr uint32_t MIN_NTILE = 32;
static constexpr uint32_t MAX_BLOCK = 24;
static constexpr uint32_t MIN_BLOCK = 8;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Read input shapes: A [M, K], B [K, N]
    const gert::Shape *A_shape = context->GetInputShape(0);
    const gert::Shape *B_shape = context->GetInputShape(1);
    if (A_shape->GetOriginShape().GetDimNum() != 2 || B_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(A_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(A_shape->GetOriginShape().GetDim(1));
    const uint32_t Kb = static_cast<uint32_t>(B_shape->GetOriginShape().GetDim(0));
    const uint32_t N = static_cast<uint32_t>(B_shape->GetOriginShape().GetDim(1));
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    // Choose tile sizes under UB budget for f32:
    // Buffers: 
    //   bRow: nTile
    //   tmpVec: nTile
    //   cTile: mTile * nTile
    // Total floats ~ (mTile*nTile + 2*nTile)
    // bytes ~ 4 * (mTile*nTile + 2*nTile)
    // We target mTile=128, nTile=64 first, fall back if exceeding UB.
    uint32_t mTile = 128;
    uint32_t nTile = 64;

    auto FitsUB = [](uint32_t mT, uint32_t nT) -> bool {
        const uint64_t totalFloats = static_cast<uint64_t>(mT) * nT + 2ull * nT + 16ull; // some headroom
        const uint64_t totalBytes = totalFloats * 4ull;
        return totalBytes <= UB_BUDGET_BYTES;
    };

    if (!FitsUB(mTile, nTile)) {
        // Try alternative tilings
        const uint32_t candMT[] = {128, 96, 64, 64, 48, 32};
        const uint32_t candNT[] = {64, 96, 128, 48, 32, 32};
        bool found = false;
        for (size_t i = 0; i < sizeof(candMT)/sizeof(candMT[0]) && !found; ++i) {
            for (size_t j = 0; j < sizeof(candNT)/sizeof(candNT[0]) && !found; ++j) {
                if (FitsUB(candMT[i], candNT[j])) {
                    mTile = candMT[i];
                    nTile = candNT[j];
                    found = true;
                    break;
                }
            }
        }
        if (!found) {
            // fallback tiny tiles
            mTile = MIN_MTILE;
            nTile = MIN_NTILE;
        }
    }

    // Align tiles to preferred burst length but not to exceed dims
    mTile = AlignDown(mTile, ALIGN_ELEMS / 2);
    if (mTile < MIN_MTILE) mTile = MIN_MTILE;
    nTile = AlignDown(nTile, ALIGN_ELEMS / 2);
    if (nTile < MIN_NTILE) nTile = MIN_NTILE;

    const uint32_t tilesM = CeilDiv(M, mTile);
    const uint32_t tilesN = CeilDiv(N, nTile);
    uint32_t totalTiles = tilesM * tilesN;

    uint32_t blockDim = totalTiles;
    if (blockDim < MIN_BLOCK) blockDim = MIN_BLOCK;
    if (blockDim > MAX_BLOCK) blockDim = MAX_BLOCK;
    context->SetBlockDim(blockDim);

    MatmulForSymmetricMatricesCustomTilingData tiling;
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_mTile(mTile);
    tiling.set_nTile(nTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0; // no extra workspace
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *A_shape = context->GetInputShape(0);
    const gert::Shape *B_shape = context->GetInputShape(1);

    // A:[M,K], B:[K,N] -> C:[M,N]
    const int64_t M = A_shape->GetOriginShape().GetDim(0);
    const int64_t N = B_shape->GetOriginShape().GetDim(1);

    gert::Shape *C_shape = context->GetOutputShape(0);
    C_shape->SetDimNum(2);
    C_shape->SetDim(0, M);
    C_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType A_dtype = context->GetInputDataType(0);
    const ge::DataType B_dtype = context->GetInputDataType(1);
    if (A_dtype != B_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, A_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulForSymmetricMatricesCustom {
public:
    __aicore__ inline KernelMatmulForSymmetricMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t mTile, uint32_t nTile)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        mTile_ = mTile;
        nTile_ = nTile;

        aGm_.SetGlobalBuffer((__gm__ float*)a, (uint64_t)M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, (uint64_t)M_ * N_);

        // UB buffers:
        // - C tile buffer: mTile * nTile
        // - bRow buffer: nTile
        // - temp vector buffer: nTile
        // - one scalar buffer for A element
        pipe_.InitBuffer(cTileQueue_, 1, mTile_ * nTile_ * sizeof(float));
        pipe_.InitBuffer(bRowQueue_, BUFFER_NUM, nTile_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQueue_, 1, nTile_ * sizeof(float));
        pipe_.InitBuffer(aScalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) {
            return;
        }

        const uint32_t tilesM = CeilDiv(M_, mTile_);
        const uint32_t tilesN = CeilDiv(N_, nTile_);
        const uint32_t totalTiles = tilesM * tilesN;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tileIndex = blockIdx; tileIndex < totalTiles; tileIndex += blockNum) {
            const uint32_t tileRow = tileIndex / tilesN;
            const uint32_t tileCol = tileIndex % tilesN;

            const uint32_t rowBase = tileRow * mTile_;
            const uint32_t colBase = tileCol * nTile_;

            const uint32_t curM = (rowBase + mTile_ <= M_) ? mTile_ : (M_ - rowBase);
            const uint32_t curN = (colBase + nTile_ <= N_) ? nTile_ : (N_ - colBase);

            ComputeTile(rowBase, colBase, curM, curN);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
        return (a + b - 1u) / b;
    }

    __aicore__ inline void ZeroLocal(AscendC::LocalTensor<float>& t, uint32_t len)
    {
        // Multiply by 0 to clear buffer
        AscendC::Muls(t, t, 0.0f, len);
    }

    __aicore__ inline void ComputeTile(uint32_t rowBase, uint32_t colBase,
                                       uint32_t curM, uint32_t curN)
    {
        // Allocate C tile
        AscendC::LocalTensor<float> cTile = cTileQueue_.AllocTensor<float>();
        const uint32_t cTileSize = mTile_ * nTile_;
        ZeroLocal(cTile, cTileSize);

        // Iterate K dimension, stream B rows and A scalars
        for (uint32_t kk = 0; kk < K_; ++kk) {
            // Load bRow = B[kk, colBase:colBase+curN]
            AscendC::LocalTensor<float> bRow = bRowQueue_.AllocTensor<float>();
            AscendC::DataCopy(bRow, bGm_[static_cast<uint64_t>(kk) * N_ + colBase], curN);
            bRowQueue_.EnQue(bRow);

            // For each row i in this tile
            AscendC::LocalTensor<float> bRowLocal = bRowQueue_.DeQue<float>(); // deque for reuse
            for (uint32_t i = 0; i < curM; ++i) {
                // Load scalar aVal = A[rowBase + i, kk]
                AscendC::LocalTensor<float> aScalar = aScalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(aScalar, aGm_[static_cast<uint64_t>(rowBase + i) * K_ + kk], 1);
                // Get scalar value
                float aVal = aScalar.GetValue(0);
                aScalarQueue_.FreeTensor(aScalar);

                // tempVec = bRow * aVal
                AscendC::LocalTensor<float> tmp = tmpVecQueue_.AllocTensor<float>();
                AscendC::Muls(tmp, bRowLocal, aVal, curN);

                // cRow += tempVec
                AscendC::LocalTensor<float> cRow = cTile[i * nTile_];
                AscendC::Add(cRow, cRow, tmp, curN);

                tmpVecQueue_.FreeTensor(tmp);
            }
            // bRow reuse finished
            bRowQueue_.FreeTensor(bRowLocal);
        }

        // Store C tile rows back to GM
        for (uint32_t i = 0; i < curM; ++i) {
            AscendC::LocalTensor<float> cRow = cTile[i * nTile_];
            AscendC::DataCopy(cGm_[static_cast<uint64_t>(rowBase + i) * N_ + colBase], cRow, curN);
        }

        cTileQueue_.FreeTensor(cTile);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVecQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> aScalarQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> cTileQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t mTile_{64};
    uint32_t nTile_{64};
};

extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForSymmetricMatricesCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.mTile, tiling_data.nTile);
    op.Process();
}
