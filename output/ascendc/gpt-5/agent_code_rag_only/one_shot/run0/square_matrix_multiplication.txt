host_tiling_src="""
#include "register/tilingdata_base.h"

namespace optiling {

BEGIN_TILING_DATA_DEF(SquareMatrixMultiplicationCustomTilingData)
  TILING_DATA_FIELD_DEF(TCubeTiling, cubeTilingData);
  TILING_DATA_FIELD_DEF(uint32_t, blockDim);
END_TILING_DATA_DEF;

REGISTER_TILING_DATA_CLASS(SquareMatrixMultiplicationCustom, SquareMatrixMultiplicationCustomTilingData)

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SquareMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape* aShape = context->GetInputShape(0);
    const gert::Shape* bShape = context->GetInputShape(1);

    // Expect A: [M, K], B: [K, N]
    const uint32_t M = static_cast<uint32_t>(aShape->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(aShape->GetDim(1));
    const uint32_t N = static_cast<uint32_t>(bShape->GetDim(1));

    // Fill cube tiling basic dims
    tiling.cubeTilingData.M = M;
    tiling.cubeTilingData.N = N;
    tiling.cubeTilingData.K = K;
    tiling.cubeTilingData.BatchNum = 1;  // single matmul

    // Conservative block dim selection
    // At least 8, at most 24 for stability
    uint32_t blockDim = 16;
    const uint64_t workTiles = static_cast<uint64_t>(CeilDiv(M, 16)) * static_cast<uint64_t>(CeilDiv(N, 16));
    if (workTiles < 8) blockDim = 8;
    if (workTiles > 24) blockDim = 24;

    tiling.set_blockDim(blockDim);
    context->SetBlockDim(blockDim);

    // Save tiling buffer
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace needed
    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    // A: [M, K], B: [K, N] => C: [M, N]
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, a_shape->GetDim(0));
    c_shape->SetDim(1, b_shape->GetDim(1));
    return GRAPH_SUCCESS;
}

    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                                                          GM_ADDR workspace, GM_ADDR tiling)
{
    if (g_coreType == AscendC::AIV) {
        return;
    }

    GET_TILING_DATA(tilingData, tiling);

    // Define GM ND MatMul types: A/B in FP16, C in FP32
    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, true,  LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> matmulKernel;
    AscendC::TPipe pipe;

    // Share full on-chip memory to matmul object
    tilingData.cubeTilingData.shareMode   = 0;
    tilingData.cubeTilingData.shareL1Size = FULL_L1_SIZE;
    tilingData.cubeTilingData.shareL0CSize = FULL_L0C_SIZE;
    tilingData.cubeTilingData.shareUbSize = 0;

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulKernel.matmulObj, &tilingData.cubeTilingData);

    // Init and process a single-batch matmul
    matmulKernel.Init(a, b, nullptr, c, workspace, tilingData.cubeTilingData);
    const int batchNum = 1;
    matmulKernel.Process<false>(&pipe, batchNum, batchNum);
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor square_matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, \"Inputs must be 2D matrices\");
    TORCH_CHECK(A.size(1) == B.size(0), \"Incompatible matmul dimensions\");

    // Convert inputs to half for MMA, output accumulate to float
    at::Tensor A_half = A.to(at::kHalf);
    at::Tensor B_half = B.to(at::kHalf);
    at::Tensor C = at::empty({A.size(0), B.size(1)}, A.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnSquareMatrixMultiplicationCustom, A_half, B_half, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"square_matrix_multiplication_custom\", &square_matmul_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"square_matrix_multiplication_custom\", &square_matmul_impl_npu,
          \"Matrix multiplication (SquareMatrixMultiplicationCustom on NPU)\");
}
"""
