kernel_src="""

#include "kernel_operator.h"

class KernelConvTransposed3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Di, uint32_t Hi, uint32_t Wi,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Do, uint32_t Ho, uint32_t Wo)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Di_ = Di; Hi_ = Hi; Wi_ = Wi;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        Do_ = Do; Ho_ = Ho; Wo_ = Wo;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Cin_ * Di_ * Hi_ * Wi_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cin_ * Cout_ * Kd_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Cout_ * Do_ * Ho_ * Wo_);

        pipe_.InitBuffer(xRowQueue_, 1, Wi_ * sizeof(float));
        pipe_.InitBuffer(yRowQueue_, 1, Wo_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, Wi_ * sizeof(float));
        pipe_.InitBuffer(wSmallQueue_, 1, Kw_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0) return;

        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blk = AscendC::GetBlockIdx();

        for (uint32_t n = 0; n < N_; ++n) {
            for (uint32_t oc = blk; oc < Cout_; oc += blockNum) {

                for (uint32_t od = 0; od < Do_; ++od) {
                    for (uint32_t oh = 0; oh < Ho_; ++oh) {
                        // Allocate and zero y row (length Wo_)
                        AscendC::LocalTensor<float> yRow = yRowQueue_.AllocTensor<float>();
                        AscendC::Dup(yRow, 0.0f, Wo_);

                        // Accumulate over input channels and kernel windows
                        for (uint32_t ic = 0; ic < Cin_; ++ic) {
                            for (uint32_t kd = 0; kd < Kd_; ++kd) {
                                // corresponding input depth index
                                int32_t id = static_cast<int32_t>(od) - static_cast<int32_t>(kd);
                                if (id < 0 || id >= static_cast<int32_t>(Di_)) continue;

                                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                                    int32_t ih = static_cast<int32_t>(oh) - static_cast<int32_t>(kh);
                                    if (ih < 0 || ih >= static_cast<int32_t>(Hi_)) continue;

                                    // Load weights for this (ic, oc, kd, kh, :)
                                    const uint64_t wOff = ((((uint64_t)ic * Cout_ + oc) * Kd_ + kd) * Kh_ + kh) * Kw_;
                                    AscendC::LocalTensor<float> wSmall = wSmallQueue_.AllocTensor<float>();
                                    AscendC::DataCopy(wSmall, wGm_[wOff], Kw_);

                                    // Load x row: length Wi_
                                    const uint64_t xOff = ((((uint64_t)n * Cin_ + ic) * Di_ + (uint32_t)id) * Hi_ + (uint32_t)ih) * Wi_;
                                    AscendC::LocalTensor<float> xRow = xRowQueue_.AllocTensor<float>();
                                    AscendC::DataCopy(xRow, xGm_[xOff], Wi_);

                                    // For each kw, shift-add into yRow
                                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                        // Read weight scalar
                                        float coeff = wSmall.GetValue(kw);

                                        // tmp = xRow * coeff
                                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                                        AscendC::Muls(tmp, xRow, coeff, Wi_);

                                        // yRow[kw:kw+Wi] += tmp
                                        AscendC::Add(yRow[kw], yRow[kw], tmp, Wi_);

                                        tmpQueue_.FreeTensor(tmp);
                                    }

                                    xRowQueue_.FreeTensor(xRow);
                                    wSmallQueue_.FreeTensor(wSmall);
                                }
                            }
                        }

                        // Write y row to GM
                        const uint64_t yOff = ((((uint64_t)n * Cout_ + oc) * Do_ + od) * Ho_ + oh) * Wo_;
                        AscendC::DataCopy(yGm_[yOff], yRow, Wo_);
                        yRowQueue_.FreeTensor(yRow);
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wSmallQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Di_{0}, Hi_{0}, Wi_{0};
    uint32_t Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Do_{0}, Ho_{0}, Wo_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.Di, tiling_data.Hi, tiling_data.Wi,
            tiling_data.Kd, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Do, tiling_data.Ho, tiling_data.Wo);
    op.Process();
}

#include <torch/library.h>

static inline at::Tensor make_output_tensor_like_transpose3d(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, Cin, Di, Hi, Wi], w: [Cin, Cout, Kd, Kh, Kw]
    auto sizes_x = x.sizes();
    auto sizes_w = w.sizes();
    int64_t N = sizes_x[0];
    int64_t Cin = sizes_x[1];
    int64_t Di = sizes_x[2];
    int64_t Hi = sizes_x[3];
    int64_t Wi = sizes_x[4];

    int64_t Cin_w = sizes_w[0];
    int64_t Cout = sizes_w[1];
    int64_t Kd = sizes_w[2];
    int64_t Kh = sizes_w[3];
    int64_t Kw = sizes_w[4];

    TORCH_CHECK(Cin == Cin_w, "Input and weight Cin mismatch");
    int64_t Do = Di + Kd - 1;
    int64_t Ho = Hi + Kh - 1;
    int64_t Wo = Wi + Kw - 1;

    return at::empty({N, Cout, Do, Ho, Wo}, x.options());
}

    EXEC_NPU_CMD(aclnnConvTransposed3dSquareInputSquareKernelCustom, x, w, y);
    return y;
}
