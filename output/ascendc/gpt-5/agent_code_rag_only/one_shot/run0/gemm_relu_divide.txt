host_operator_src="""
#include "gemm_relu_divide_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model for one output element accumulation over K in chunks:
//   xChunk: 1 * tileK
//   wChunk: 1 * tileK
//   tmpMul: 1 * tileK
//   scalars: a few elements (negligible)
// Total floats ~ 3 * tileK => bytes ~ 12 * tileK
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ALIGN_ELEMS = 256;             // favor large burst
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // latency guard
static constexpr uint32_t MIN_BLOCKS = 8;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmReluDivideCustomTilingData tiling;

    // x: [M, K], w: [N, K], b: [N], divisor: [1]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetOriginShape().GetDimNum() != 2 ||
        w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    if (K != Kw) {
        return ge::GRAPH_FAILED;
    }

    // tileK from UB budget: 12 * tileK bytes
    uint32_t maxTileK = UB_BUDGET_BYTES / 12;
    maxTileK = AlignDown(maxTileK, ALIGN_ELEMS);
    if (maxTileK < ALIGN_ELEMS) maxTileK = ALIGN_ELEMS;
    if (maxTileK > HARD_CAP_ELEMS) maxTileK = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileK = maxTileK;
    if (K < tileK) {
        const uint32_t aligned = AlignDown(K, ALIGN_ELEMS);
        tileK = (aligned > 0) ? aligned : K;
    }
    if (tileK == 0) tileK = 1;

    const uint64_t totalOut = static_cast<uint64_t>(M) * static_cast<uint64_t>(N);
    uint32_t blockDim = (totalOut == 0) ? 1u : static_cast<uint32_t>(totalOut);
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // [M, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K]
    if (x_shape->GetOriginShape().GetDimNum() != 2 ||
        w_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = x_shape->GetOriginShape().GetDim(0);
    const int64_t N = w_shape->GetOriginShape().GetDim(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    gert::Shape new_shape;
    new_shape.SetDimNum(2);
    new_shape.SetDim(0, M);
    new_shape.SetDim(1, N);
    y_shape->SetOriginShape(new_shape);
    y_shape->SetStorageShape(new_shape);

}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUFFER_NUM = 2;
constexpr int32_t OUT_BUFFER_NUM = 2;
constexpr int32_t TMP_BUFFER_NUM = 1;
constexpr int32_t SCAL_BUFFER_NUM = 4;

class KernelGemmReluDivideCustom {
public:
    __aicore__ inline KernelGemmReluDivideCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K; tileK_ = tileK;
        totalOut_ = static_cast<uint64_t>(M_) * static_cast<uint64_t>(N_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        divGm_.SetGlobalBuffer((__gm__ float*)divisor, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        pipe_.InitBuffer(inQueueX_, IN_BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, IN_BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMP_BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(scalQueue_, SCAL_BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUFFER_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t idx = blockIdx; idx < totalOut_; idx += blockNum) {
            uint32_t row = static_cast<uint32_t>(idx / N_);
            uint32_t col = static_cast<uint32_t>(idx - static_cast<uint64_t>(row) * N_);
            ComputeOne(row, col);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t row, uint32_t col)
    {
        // acc = 0
        AscendC::LocalTensor<float> acc = scalQueue_.AllocTensor<float>();
        AscendC::Duplicate(acc, 0.0f, 1);

        uint32_t kBase = 0;
        while (kBase < K_) {
            uint32_t len = K_ - kBase;
            if (len > tileK_) len = tileK_;

            // Load x row chunk and w row (for col) chunk
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[static_cast<uint64_t>(row) * K_ + kBase], len);
            AscendC::DataCopy(wLocal, wGm_[static_cast<uint64_t>(col) * K_ + kBase], len);
            inQueueX_.EnQue(xLocal);
            inQueueW_.EnQue(wLocal);

            xLocal = inQueueX_.DeQue<float>();
            wLocal = inQueueW_.DeQue<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(tmp, xLocal, wLocal, len); // elementwise product

            // Tree reduction into tmp[0]
            uint32_t n = len;
            while (n > 1) {
                uint32_t half = n >> 1;
                uint32_t rem = n & 1u;
                if (half > 0) {
                    AscendC::Add(tmp, tmp, tmp[half], half);
                }
                if (rem) {
                    AscendC::Add(tmp, tmp, tmp[(half << 1)], 1);
                }
                n = half + rem;
            }
            // accumulate chunk sum into acc[0]
            AscendC::Add(acc, acc, tmp, 1);

            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xLocal);
            inQueueW_.FreeTensor(wLocal);
            kBase += len;
        }

        // Add bias
        AscendC::LocalTensor<float> bias = scalQueue_.AllocTensor<float>();
        AscendC::DataCopy(bias, bGm_[col], 1);
        AscendC::Add(acc, acc, bias, 1);
        scalQueue_.FreeTensor(bias);

        // ReLU
        AscendC::Maxs(acc, acc, 0.0f, 1);

        // Divide by divisor
        AscendC::LocalTensor<float> div = scalQueue_.AllocTensor<float>();
        AscendC::DataCopy(div, divGm_[0], 1);
        AscendC::Div(acc, acc, div, 1);
        scalQueue_.FreeTensor(div);

        // Store to y
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, acc, 1);
        outQueueY_.EnQue(yLocal);

        AscendC::LocalTensor<float> yL = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[static_cast<uint64_t>(row) * N_ + col], yL, 1);
        outQueueY_.FreeTensor(yL);

        scalQueue_.FreeTensor(acc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, SCAL_BUFFER_NUM> scalQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> divGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileK_{0};
    uint64_t totalOut_{0};
};

extern "C" __global__ __aicore__ void gemm_relu_divide_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor,
                                                              GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmReluDivideCustom op;
    op.Init(x, w, b, divisor, y,
            tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileK);
    op.Process();
}

at::Tensor gemm_relu_divide_impl_npu(const at::Tensor& x,

                                     const at::Tensor& b,
                                     const at::Tensor& divisor) {
    TORCH_CHECK(x.dtype() == at::kFloat && w.dtype() == at::kFloat &&

    TORCH_CHECK(w.size(1) == K, "w shape mismatch with x.");
    auto N = w.size(0);
    TORCH_CHECK(b.size(0) == N, "b size mismatch with w.");

    at::Tensor result = at::empty({M, N}, x.options());
    EXEC_NPU_CMD(aclnnGemmReluDivideCustom, x, w, b, divisor, result);
    return result;
