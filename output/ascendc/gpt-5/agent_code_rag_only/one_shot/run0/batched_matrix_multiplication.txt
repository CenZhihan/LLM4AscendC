host_operator_src="""

#include "batched_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
}

namespace optiling {

// UB model per-core during compute (float32):
//   cVec: tileN
//   bVec: tileN
//   aSeg: tileK
// Total bytes ~= 4 * (2*tileN + tileK)
// Keep well below a conservative per-core UB share
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // 192KB
static constexpr uint32_t SAFE_BYTES      = 160 * 1024;   // ~160KB target
static constexpr uint32_t ALIGN_N = 256;                  // favor larger DMA along N
static constexpr uint32_t ALIGN_K = 64;                   // reasonable align along K
static constexpr uint32_t MAX_BLOCKS = 24;
static constexpr uint32_t MIN_BLOCKS = 8;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    BatchedMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);

    // Expect A: [B, M, K], B: [B, K, N]
    const uint32_t dimsA = static_cast<uint32_t>(shapeA->GetOriginShape().GetDimNum());
    const uint32_t dimsB = static_cast<uint32_t>(shapeB->GetOriginShape().GetDimNum());
    if (dimsA != 3 || dimsB != 3) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t batch = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(0));
    const uint32_t m     = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(1));
    const uint32_t k     = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(2));
    const uint32_t n     = static_cast<uint32_t>(shapeB->GetOriginShape().GetDim(2));

    // Initial caps
    uint32_t tileN = (n >= 1024u) ? 1024u : n;
    uint32_t tileK = (k >= 128u) ? 128u : k;

    // Align down
    tileN = (tileN >= ALIGN_N) ? AlignDown(tileN, ALIGN_N) : tileN;
    tileK = (tileK >= ALIGN_K) ? AlignDown(tileK, ALIGN_K) : tileK;

    if (tileN == 0) tileN = n;
    if (tileK == 0) tileK = k;

    // Ensure UB budget: 4 * (2*tileN + tileK) <= SAFE_BYTES
    auto Fits = [&](uint32_t tn, uint32_t tk) -> bool {
        uint64_t bytes = 4ull * (2ull * tn + tk);
        return bytes <= SAFE_BYTES;
    };

    while (!Fits(tileN, tileK)) {
        if (tileN > ALIGN_N) {
            tileN = AlignDown(tileN - ALIGN_N, ALIGN_N);
            if (tileN == 0) tileN = (n < ALIGN_N) ? n : ALIGN_N;
        } else if (tileK > ALIGN_K) {
            tileK = AlignDown(tileK - ALIGN_K, ALIGN_K);
            if (tileK == 0) tileK = (k < ALIGN_K) ? k : ALIGN_K;
        } else {
            // fallback minimal safe
            tileN = (n < 64u) ? n : 64u;
            tileK = (k < 32u) ? k : 32u;
            break;
        }
    }

    // BlockDim based on number of rows (B * M)
    const uint32_t totalRows = (batch == 0u || m == 0u) ? 1u : (batch * m);
    uint32_t blockDim = totalRows;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_m(m);
    tiling.set_k(k);
    tiling.set_n(n);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t ABUF_NUM = 1;
constexpr int32_t BBUF_NUM = 1;
constexpr int32_t CBUF_NUM = 1;

class KernelBatchedMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelBatchedMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t batch, uint32_t m, uint32_t k, uint32_t n,
                                uint32_t tileK, uint32_t tileN)
    {
        batch_ = batch; m_ = m; k_ = k; n_ = n;
        tileK_ = (tileK == 0 ? 1u : tileK);
        tileN_ = (tileN == 0 ? 1u : tileN);

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint32_t>(batch_ * m_ * k_));
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint32_t>(batch_ * k_ * n_));
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint32_t>(batch_ * m_ * n_));

        pipe_.InitBuffer(aQueue_, ABUF_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, BBUF_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(cQueue_, CBUF_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || m_ == 0 || k_ == 0 || n_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalRows = batch_ * m_;
        if (blockIdx >= totalRows) return;

        for (uint32_t rowLinear = blockIdx; rowLinear < totalRows; rowLinear += blockNum) {
            const uint32_t bIdx = rowLinear / m_;
            const uint32_t mIdx = rowLinear - bIdx * m_;

            for (uint32_t nStart = 0; nStart < n_; nStart += tileN_) {
                uint32_t nn = n_ - nStart;
                if (nn > tileN_) nn = tileN_;

                // Allocate C tile and zero initialize via multiply-by-zero
                AscendC::LocalTensor<float> cLocal = cQueue_.AllocTensor<float>();
                AscendC::Muls(cLocal, cLocal, 0.0f, nn);

                for (uint32_t kStart = 0; kStart < k_; kStart += tileK_) {
                    uint32_t kk = k_ - kStart;
                    if (kk > tileK_) kk = tileK_;

                    // Load continuous A segment for this row
                    AscendC::LocalTensor<float> aLocal = aQueue_.AllocTensor<float>();
                    uint32_t aOff = ((bIdx * m_ + mIdx) * k_) + kStart;
                    AscendC::DataCopy(aLocal, aGm_[aOff], kk);
                    aQueue_.EnQue(aLocal);
                    aLocal = aQueue_.DeQue<float>();

                    // For each scalar in A segment, accumulate a_i * B_i_row into C vector
                    for (uint32_t t = 0; t < kk; ++t) {
                        float aVal = aLocal.GetValue(t);

                        AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();
                        const uint32_t bRow = kStart + t;
                        const uint32_t bOff = ((bIdx * k_) + bRow) * n_ + nStart;
                        AscendC::DataCopy(bLocal, bGm_[bOff], nn);

                        AscendC::Muls(bLocal, bLocal, aVal, nn);
                        AscendC::Add(cLocal, cLocal, bLocal, nn);

                        bQueue_.FreeTensor(bLocal);
                    }

                    aQueue_.FreeTensor(aLocal);
                }

                // Store C tile
                const uint32_t cOff = ((bIdx * m_) + mIdx) * n_ + nStart;
                AscendC::DataCopy(cGm_[cOff], cLocal, nn);
                cQueue_.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, ABUF_NUM>  aQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BBUF_NUM>  bQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, CBUF_NUM> cQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t batch_{0}, m_{0}, k_{0}, n_{0};
    uint32_t tileK_{0}, tileN_{0};
};

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBatchedMatrixMultiplicationCustom op;
    op.Init(A, B, C,
            tiling_data.batch, tiling_data.m, tiling_data.k, tiling_data.n,
            tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
