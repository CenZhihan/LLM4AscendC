host_operator_src="""
#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (per core):
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   calcQueue: 3 * tile (gamma, beta, temp)
// Total floats = 7 * tile => bytes = 28 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // favor decent DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;

    const uint64_t totalLength =
        context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    const uint64_t normSize =
        context->GetInputShape(1)->GetOriginShape().GetShapeSize(); // gamma size defines normalized_shape product

    if (normSize == 0) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t rows = totalLength / normSize;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    const float eps = (epsPtr != nullptr) ? *epsPtr : 1e-5f;

    uint32_t maxTile = UB_BUDGET_BYTES / 28;  // elements per tile
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = static_cast<uint32_t>(maxTile);
    if (normSize < tileLen) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(normSize), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(normSize);
    }

    uint32_t blockDim = (rows == 0) ? 1u : CeilDiv(rows, 1);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_rowSize(static_cast<uint32_t>(normSize));
    tiling.set_tileLength(tileLen);
    tiling.set_rows(static_cast<uint32_t>(rows));
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;

    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowSize,
                                uint32_t tileLength, float eps)
    {
        totalLen_ = totalLength;
        rowSize_  = rowSize;
        tileLen_  = tileLength;
        eps_      = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, rowSize_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, rowSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(calcQueue_, 3,        tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowSize_ == 0 || tileLen_ == 0) { return; }

        const uint32_t rows = totalLen_ / rowSize_;
        if (rows == 0) { return; }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows; row += blockNum) {
            const uint32_t base = row * rowSize_;

            float sumVal = 0.0f;
            float sumSqVal = 0.0f;

            // Pass 1: accumulate sum and sum of squares over chunks
            for (uint32_t off = 0; off < rowSize_; off += tileLen_) {
                uint32_t curLen = rowSize_ - off;
                if (curLen > tileLen_) curLen = tileLen_;

                CopyInX(base + off, curLen);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tmpBuf = calcQueue_.AllocTensor<float>();

                AscendC::ReduceSum(tmpBuf, xLocal, curLen);
                float pSum = tmpBuf.GetValue(0);
                sumVal += pSum;

                // square in-place
                AscendC::Mul(xLocal, xLocal, xLocal, curLen);
                AscendC::ReduceSum(tmpBuf, xLocal, curLen);
                float pSq = tmpBuf.GetValue(0);
                sumSqVal += pSq;

                calcQueue_.FreeTensor(tmpBuf);
                inQueueX_.FreeTensor(xLocal);
            }

            const float N = static_cast<float>(rowSize_);
            const float mean = sumVal / N;
            const float ex2 = sumSqVal / N;
            float var = ex2 - mean * mean;
            if (var < 0.0f) var = 0.0f;  // numerical guard
            const float invStd = 1.0f / AscendC::Sqrt(var + eps_);

            // Pass 2: normalize, scale and shift
            for (uint32_t off = 0; off < rowSize_; off += tileLen_) {
                uint32_t curLen = rowSize_ - off;
                if (curLen > tileLen_) curLen = tileLen_;

                CopyInX(base + off, curLen);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();

                AscendC::LocalTensor<float> gLocal = calcQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> bLocal = calcQueue_.AllocTensor<float>();

                AscendC::DataCopy(gLocal, gammaGm_[off], curLen);
                AscendC::DataCopy(bLocal, betaGm_[off],  curLen);

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                AscendC::Adds(yLocal, xLocal, -mean, curLen);     // y = x - mean
                AscendC::Muls(yLocal, yLocal, invStd, curLen);    // y *= invStd
                AscendC::Mul(yLocal, yLocal, gLocal, curLen);     // y *= gamma
                AscendC::Add(yLocal, yLocal, bLocal, curLen);     // y += beta

                outQueueY_.EnQue(yLocal);

                calcQueue_.FreeTensor(gLocal);
                calcQueue_.FreeTensor(bLocal);
                inQueueX_.FreeTensor(xLocal);

                CopyOut(base + off, curLen);
            }
        }
    }

private:
    __aicore__ inline void CopyInX(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3>         calcQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;

    uint32_t totalLen_{0};
    uint32_t rowSize_{0};
    uint32_t tileLen_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength,
            tiling_data.rowSize,
            tiling_data.tileLength,
            tiling_data.eps);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape, eps: float = 1e-5):
        super(ModelNew, self).__init__()
        # Learnable affine parameters
        self.weight = torch.nn.Parameter(torch.ones(normalized_shape))
        self.bias = torch.nn.Parameter(torch.zeros(normalized_shape))
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias, self.eps)
