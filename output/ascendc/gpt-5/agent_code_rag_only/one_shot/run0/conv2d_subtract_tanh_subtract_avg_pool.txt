kernel_src="""

#include "kernel_operator.h"

using AscendC::GetBlockIdx;
using AscendC::GetBlockNum;

class KernelConv2dSubtractTanhSubtractAvgPoolCustom {
public:
    __aicore__ inline KernelConv2dSubtractTanhSubtractAvgPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t H, uint32_t W,
                                uint32_t Co, uint32_t K, uint32_t P,
                                uint32_t OH, uint32_t OW, uint32_t Hpo, uint32_t Wpo,
                                uint32_t hasBias,
                                float subtract1, float subtract2)
    {
        n_ = N; ci_ = Ci; h_ = H; w_ = W;
        co_ = Co; k_ = K; p_ = P;
        oh_ = OH; ow_ = OW; hpo_ = Hpo; wpo_ = Wpo;
        hasBias_ = hasBias;
        sub1_ = subtract1;
        sub2_ = subtract2;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N) * Ci * H * W);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Co) * Ci * K * K);
        if (b != nullptr) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, Co);
        } else {
            bGm_.SetGlobalBuffer((__gm__ float*)nullptr, 0);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N) * Co * Hpo * Wpo);

        // UB buffers for one (n, oc):
        // yAcc: OH*OW, yPool: Hpo*Wpo, tmpRow: OW
        pipe_.InitBuffer(bufAcc_, 1, oh_ * ow_ * sizeof(float));
        pipe_.InitBuffer(bufPool_, 1, hpo_ * wpo_ * sizeof(float));
        pipe_.InitBuffer(bufRow_, 1, ow_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t totalTasks = n_ * co_;
        const uint32_t blkNum = (uint32_t)GetBlockNum();
        const uint32_t blkIdx = (uint32_t)GetBlockIdx();

        for (uint32_t task = blkIdx; task < totalTasks; task += blkNum) {
            const uint32_t n = task / co_;
            const uint32_t oc = task % co_;
            ComputeOneNC(n, oc);
        }
    }

private:
    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const {
        return (((uint64_t)n * ci_ + c) * this->h_ + h) * this->w_ + w;
    }
    __aicore__ inline uint64_t WOffset(uint32_t oc, uint32_t c, uint32_t ky, uint32_t kx) const {
        return (((uint64_t)oc * ci_ + c) * k_ + ky) * k_ + kx;
    }
    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t oc, uint32_t ph, uint32_t pw) const {
        return (((uint64_t)n * co_ + oc) * hpo_ + ph) * wpo_ + pw;
    }

    __aicore__ inline void ZeroVector(AscendC::LocalTensor<float>& v, uint32_t len) {
        // simple scalar fill
        for (uint32_t i = 0; i < len; ++i) {
            v.SetValue(i, 0.0f);
        }
    }

    __aicore__ inline void FillScalar(AscendC::LocalTensor<float>& v, float val, uint32_t len) {
        for (uint32_t i = 0; i < len; ++i) {
            v.SetValue(i, val);
        }
    }

    __aicore__ inline void ComputeOneNC(uint32_t n, uint32_t oc)
    {
        AscendC::LocalTensor<float> yAcc = bufAcc_.AllocTensor<float>();
        AscendC::LocalTensor<float> yPool = bufPool_.AllocTensor<float>();
        AscendC::LocalTensor<float> row = bufRow_.AllocTensor<float>();

        if (hasBias_) {
            float bval;
            AscendC::DataCopy(&bval, wGm_[0], 0); // placeholder to ensure symbol usage
            bval = bGm_[oc];
            FillScalar(yAcc, bval, oh_ * ow_);
        } else {
            ZeroVector(yAcc, oh_ * ow_);
        }

        // Convolution accumulate: for all input channels and kernel positions
        for (uint32_t ic = 0; ic < ci_; ++ic) {
            for (uint32_t ky = 0; ky < k_; ++ky) {
                for (uint32_t kx = 0; kx < k_; ++kx) {
                    const float wval = wGm_[WOffset(oc, ic, ky, kx)];
                    // for each output row
                    for (uint32_t oh = 0; oh < oh_; ++oh) {
                        // Load X row slice of length OW: X[n, ic, oh+ky, kx : kx+OW]
                        const uint64_t xoff = XOffset(n, ic, oh + ky, kx);
                        AscendC::DataCopy(row, xGm_[xoff], ow_);
                        // yAcc row pointer base
                        const uint32_t yoff = oh * ow_;
                        // yAcc[yoff + j] += row[j] * wval
                        for (uint32_t j = 0; j < ow_; ++j) {
                            const float xv = row.GetValue(j);
                            const float accum = yAcc.GetValue(yoff + j) + xv * wval;
                            yAcc.SetValue(yoff + j, accum);
                        }
                    }
                }
            }
        }

        // Fused: subtract1, tanh, subtract2 in-place on yAcc
        if (sub1_ != 0.0f) {
            for (uint32_t i = 0; i < oh_ * ow_; ++i) {
                yAcc.SetValue(i, yAcc.GetValue(i) - sub1_);
            }
        }
        // tanh
        for (uint32_t i = 0; i < oh_ * ow_; ++i) {
            float v = yAcc.GetValue(i);
            // approximate tanh using built-in if available; else fallback approximation
            // Using AscendC::Tanh if available would be: AscendC::Tanh(yAcc, yAcc, oh_*ow_);
            // Here do scalar approximation for portability
            float e2x = AscendC::Exp(2.0f * v);
            float tv = (e2x - 1.0f) / (e2x + 1.0f);
            yAcc.SetValue(i, tv);
        }
        if (sub2_ != 0.0f) {
            for (uint32_t i = 0; i < oh_ * ow_; ++i) {
                yAcc.SetValue(i, yAcc.GetValue(i) - sub2_);
            }
        }

        // Average pooling with kernel P and stride P (no padding)
        // Output dims hpo_ x wpo_
        const float invPoolArea = 1.0f / (float)(p_ * p_);
        for (uint32_t ph = 0; ph < hpo_; ++ph) {
            for (uint32_t pw = 0; pw < wpo_; ++pw) {
                float sum = 0.0f;
                const uint32_t hStart = ph * p_;
                const uint32_t wStart = pw * p_;
                for (uint32_t kh = 0; kh < p_; ++kh) {
                    const uint32_t base = (hStart + kh) * ow_;
                    for (uint32_t kw = 0; kw < p_; ++kw) {
                        sum += yAcc.GetValue(base + (wStart + kw));
                    }
                }
                yPool.SetValue(ph * wpo_ + pw, sum * invPoolArea);
            }
        }

        // Write back Y: [n, oc, hpo, wpo]
        const uint64_t yoff = YOffset(n, oc, 0, 0);
        AscendC::DataCopy(yGm_[yoff], yPool, hpo_ * wpo_);

        bufRow_.FreeTensor(row);
        bufPool_.FreeTensor(yPool);
        bufAcc_.FreeTensor(yAcc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufAcc_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufPool_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufRow_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, ci_{0}, h_{0}, w_{0};
    uint32_t co_{0}, k_{0}, p_{0};
    uint32_t oh_{0}, ow_{0}, hpo_{0}, wpo_{0};
    uint32_t hasBias_{0};
    float sub1_{0.0f}, sub2_{0.0f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv2dSubtractTanhSubtractAvgPoolCustom op;
    op.Init(
        x, weight, bias, y,
        tilingData.nBatch, tilingData.cIn, tilingData.h, tilingData.w,
        tilingData.cOut, tilingData.ksize, tilingData.pool,
        tilingData.oh, tilingData.ow, tilingData.hpo, tilingData.wpo,
        tilingData.hasBias,
        tilingData.subtract1, tilingData.subtract2
    );
    op.Process();
}
