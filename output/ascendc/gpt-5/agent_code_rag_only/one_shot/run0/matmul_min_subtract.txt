static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMinSubtractCustomTilingData tiling;
    // Shapes:
    // x: [M, K], w: [N, K], b: [N] (row-major, linear: y = x @ w^T + b)
    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);

    const uint64_t M = (x_shape->GetOriginShape().GetDimNum() == 2)
                         ? static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(0)) : 1ull;
    const uint64_t Kx = (x_shape->GetOriginShape().GetDimNum() == 2)
                         ? static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(1)) : 0ull;
    const uint64_t N = (w_shape->GetOriginShape().GetDimNum() == 2)
                         ? static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(0)) : 0ull;
    const uint64_t Kw = (w_shape->GetOriginShape().GetDimNum() == 2)
                         ? static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(1)) : 0ull;

    // Basic validation (optional)
    if (Kx != Kw) {
        // Fallback conservative tiling even if mismatch; runtime should error earlier.
    }

    const uint64_t totalLength64 = M * N;
    const uint32_t totalLength = static_cast<uint32_t>(totalLength64);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float capValue = 0.0f;
    if (attrs && attrs->GetAttrNum() >= 1) {
        const float *capPtr = attrs->GetAttrPointer<float>(0);
        if (capPtr != nullptr) {
            capValue = *capPtr;
        }
    }

    // UB budget in bytes for post-processing: 2 queues * tile + tmp dup
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t ALIGN_ELEMS = 2048;
    static constexpr uint32_t HARD_CAP_ELEMS = 32768;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float) / 4; // in-queue + out-queue + tmp ~= 4x
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = (tileLen == 0u) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_capValue(capValue);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

// A minimal GEMM + Bias using Matmul API wrapper (pattern from examples), then clamp and subtract.
template <typename aType, typename bType, typename cType, typename biasType, bool isAtrans, bool isBtrans>
class MatmulKernelWrapper {
public:
    __aicore__ inline MatmulKernelWrapper() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR bias, GM_ADDR c, const TCubeTiling& t)
    {
        tiling = t;
        aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ aType*>(a), tiling.M * tiling.Ka);
        bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ bType*>(b), tiling.Kb * tiling.N);
        cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ cType*>(c), tiling.M * tiling.N);
        biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ biasType*>(bias), tiling.N);

        int32_t offsetA = 0, offsetB = 0, offsetC = 0, offsetBias = 0;
        CalcOffset(AscendC::GetBlockIdx(), offsetA, offsetB, offsetC, offsetBias);
        aGlobal = aGlobal[offsetA];
        bGlobal = bGlobal[offsetB];
        cGlobal = cGlobal[offsetC];
        biasGlobal = biasGlobal[offsetBias];
    }

    template <bool setTmpSpace = false>
    __aicore__ inline void Process(AscendC::TPipe* pipe)
    {
        if constexpr (setTmpSpace) {
            AscendC::TBuf<> tmpMMFormatUb;
            AscendC::LocalTensor<uint8_t> mmformatUb;
            pipe->InitBuffer(tmpMMFormatUb, AscendC::TOTAL_VEC_LOCAL_SIZE);
            mmformatUb = tmpMMFormatUb.Get<uint8_t>(AscendC::TOTAL_VEC_LOCAL_SIZE);
            matmulObj.SetLocalWorkspace(mmformatUb);
        }
        matmulObj.SetTensorA(aGlobal, isAtrans);
        matmulObj.SetTensorB(bGlobal, isBtrans);
        matmulObj.SetBias(biasGlobal);
        matmulObj.IterateAll(cGlobal);
        matmulObj.End();
    }

    AscendC::Matmul<
        AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, aType, isAtrans>,
        AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, bType, isBtrans>,
        AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, cType>,
        AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, biasType>
    > matmulObj;

    AscendC::GlobalTensor<aType> aGlobal;
    AscendC::GlobalTensor<bType> bGlobal;
    AscendC::GlobalTensor<cType> cGlobal;
    AscendC::GlobalTensor<biasType> biasGlobal;
    TCubeTiling tiling;

private:
    __aicore__ inline uint32_t Ceiling(uint32_t a, uint32_t b)
    {
        return (a + b - 1) / b;
    }

    __aicore__ inline void CalcOffset(
        int32_t blockIdx, int32_t& offsetA, int32_t& offsetB, int32_t& offsetC, int32_t& offsetBias)
    {
        const TCubeTiling& t = this->tiling;
        auto mSingleBlocks = Ceiling(t.M, t.singleCoreM);
        auto mCoreIndx = blockIdx % mSingleBlocks;
        auto nCoreIndx = blockIdx / mSingleBlocks;

        offsetA = mCoreIndx * t.Ka * t.singleCoreM;
        if (isAtrans) {
            offsetA = mCoreIndx * t.singleCoreM;
        }
        offsetB = nCoreIndx * t.singleCoreN;
        if (isBtrans) {
            offsetB = nCoreIndx * t.Kb * t.singleCoreN;
        }
        offsetC = mCoreIndx * t.N * t.singleCoreM + nCoreIndx * t.singleCoreN;
        offsetBias = nCoreIndx * t.singleCoreN;

        int tailM = t.M - mCoreIndx * t.singleCoreM;
        tailM = tailM < t.singleCoreM ? tailM : t.singleCoreM;
        int tailN = t.N - nCoreIndx * t.singleCoreN;
        tailN = tailN < t.singleCoreN ? tailN : t.singleCoreN;
        if (tailM < t.singleCoreM || tailN < t.singleCoreN) {
            matmulObj.SetTail(tailM, tailN);
        }
    }
};

class PostClampSubtract {
public:
    __aicore__ inline PostClampSubtract() {}

    __aicore__ inline void Init(GM_ADDR y, uint32_t totalLength, uint32_t tileLength, float capValue)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        cap_ = capValue;

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueue_.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> inLocal = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> outLocal = outQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        AscendC::Duplicate(tLocal, cap_, len);
        AscendC::Minimum(outLocal, inLocal, tLocal, len);
        AscendC::Sub(outLocal, outLocal, tLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        outQueue_.EnQue(outLocal);
        inQueue_.FreeTensor(inLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> outLocal = outQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], outLocal, len);
        outQueue_.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    float cap_{0.0f};
};

extern "C" __global__ __aicore__ void matmul_min_subtract_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR c, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
#if defined(ASCENDC_CPU_DEBUG)
    if (g_coreType == AscendC::AIV) {
        return;
    }
#endif
    // 1) Matmul + Bias: y = x @ w^T + b
    const bool isAtrans = false;
    const bool isBtrans = true;

    TCubeTiling mmTiling;
    CopyTiling(&mmTiling, tilingGm);

    MatmulKernelWrapper<float, float, float, float, isAtrans, isBtrans> mmKernel;
    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mmKernel.matmulObj, &mmTiling);
    mmKernel.Init(x, w, b, y, mmTiling);
#ifdef CUSTOM_ASCEND310P
    mmKernel.Process<true>(&pipe);
#else
    mmKernel.Process<false>(&pipe);
#endif

    // 2) Post-process: clamp to cap and subtract cap
    // Tiling data for post pass is provided by host in a simple struct
    GET_TILING_DATA(tiling_data, tilingGm);
    const uint32_t totalLen = tiling_data.totalLength;
    const uint32_t tileLen = tiling_data.tileLength;
    const float capValue = tiling_data.capValue;

    PostClampSubtract post;
    post.Init(y, totalLen, tileLen, capValue);
    post.Process();
}
