host_operator_src="""

#include "conv2d_relu_hard_swish_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDivU32(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dReluHardSwishCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect x: [N, Cin, H, W], w: [Cout, Cin, K, K]
    const auto &x_origin = x_shape->GetOriginShape();
    const auto &w_origin = w_shape->GetOriginShape();

    uint32_t N = static_cast<uint32_t>(x_origin.GetDim(0));
    uint32_t Cin = static_cast<uint32_t>(x_origin.GetDim(1));
    uint32_t H = static_cast<uint32_t>(x_origin.GetDim(2));
    uint32_t W = static_cast<uint32_t>(x_origin.GetDim(3));

    uint32_t Cout = static_cast<uint32_t>(w_origin.GetDim(0));
    uint32_t K = static_cast<uint32_t>(w_origin.GetDim(2));
    // stride=1, pad=0, dil=1
    uint32_t Ho = H >= K ? (H - K + 1u) : 0u;
    uint32_t Wo = W >= K ? (W - K + 1u) : 0u;

    // Choose blockDim primarily across Cout to improve parallelism.
    uint32_t blockDim = Cout;
    if (blockDim == 0u) {
        blockDim = 1u;
    } else if (blockDim > 24u) {
        blockDim = 24u; // conservative for 910/910B
    }
    context->SetBlockDim(blockDim);

    uint32_t coPerBlock = CeilDivU32(Cout, blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_K(K);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_coPerBlock(coPerBlock);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const auto &x_origin = x_shape->GetOriginShape();
    const auto &w_origin = w_shape->GetOriginShape();

    uint64_t N = x_origin.GetDim(0);
    uint64_t Cin = x_origin.GetDim(1);
    uint64_t H = x_origin.GetDim(2);
    uint64_t W = x_origin.GetDim(3);

    (void)Cin; // Cin consistency is checked in framework

    uint64_t Cout = w_origin.GetDim(0);
    uint64_t K = w_origin.GetDim(2);

    uint64_t Ho = (H >= K) ? (H - K + 1ULL) : 0ULL;
    uint64_t Wo = (W >= K) ? (W - K + 1ULL) : 0ULL;

kernel_src="""

#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(GM_ADDR x,
                                                                    GM_ADDR weight,
                                                                    GM_ADDR bias,
                                                                    GM_ADDR y,
                                                                    GM_ADDR workspace,
                                                                    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    // Read tiling params
    const uint32_t N  = tiling_data.N;
    const uint32_t Cin = tiling_data.Cin;
    const uint32_t Cout = tiling_data.Cout;
    const uint32_t H = tiling_data.H;
    const uint32_t W = tiling_data.W;
    const uint32_t K = tiling_data.K;
    const uint32_t Ho = tiling_data.Ho;
    const uint32_t Wo = tiling_data.Wo;
    const uint32_t coPerBlock = tiling_data.coPerBlock;

    __gm__ const float* x_gm = reinterpret_cast<__gm__ const float*>(x);
    __gm__ const float* w_gm = reinterpret_cast<__gm__ const float*>(weight);
    __gm__ const float* b_gm = reinterpret_cast<__gm__ const float*>(bias);
    __gm__ float* y_gm       = reinterpret_cast<__gm__ float*>(y);

    const uint32_t blkIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
    const uint32_t coStart = blkIdx * coPerBlock;
    if (coStart >= Cout) {
        return;
    }
    const uint32_t coEnd = (coStart + coPerBlock) > Cout ? Cout : (coStart + coPerBlock);

    // Strides for NCHW layout
    const uint64_t x_stride_n = static_cast<uint64_t>(Cin) * H * W;
    const uint64_t x_stride_c = static_cast<uint64_t>(H) * W;
    const uint64_t x_stride_h = static_cast<uint64_t>(W);
    const uint64_t x_stride_w = 1;

    const uint64_t w_stride_co = static_cast<uint64_t>(Cin) * K * K;
    const uint64_t w_stride_ci = static_cast<uint64_t>(K) * K;
    const uint64_t w_stride_kh = static_cast<uint64_t>(K);
    const uint64_t w_stride_kw = 1;

    const uint64_t y_stride_n = static_cast<uint64_t>(Cout) * Ho * Wo;
    const uint64_t y_stride_c = static_cast<uint64_t>(Ho) * Wo;
    const uint64_t y_stride_h = static_cast<uint64_t>(Wo);
    const uint64_t y_stride_w = 1;

    const bool has_bias = (bias != nullptr);

    // Naive direct convolution with fused ReLU + HardSwish
    for (uint32_t n = 0; n < N; ++n) {
        const uint64_t x_base_n = n * x_stride_n;
        const uint64_t y_base_n = n * y_stride_n;

        for (uint32_t co = coStart; co < coEnd; ++co) {
            const uint64_t w_base_co = static_cast<uint64_t>(co) * w_stride_co;
            const float bval = has_bias ? b_gm[co] : 0.0f;

            for (uint32_t oh = 0; oh < Ho; ++oh) {
                for (uint32_t ow = 0; ow < Wo; ++ow) {
                    float acc = 0.0f;

                    #pragma unroll 1
                    for (uint32_t ci = 0; ci < Cin; ++ci) {
                        const uint64_t x_base_ci = x_base_n + static_cast<uint64_t>(ci) * x_stride_c;

                        #pragma unroll 1
                        for (uint32_t kh = 0; kh < K; ++kh) {
                            const uint64_t ih = static_cast<uint64_t>(oh + kh);
                            const uint64_t w_base_kh = w_base_co + static_cast<uint64_t>(ci) * w_stride_ci
                                                     + static_cast<uint64_t>(kh) * w_stride_kh;

                            #pragma unroll 1
                            for (uint32_t kw = 0; kw < K; ++kw) {
                                const uint64_t iw = static_cast<uint64_t>(ow + kw);

                                const float xv = x_gm[x_base_ci + ih * x_stride_h + iw * x_stride_w];
                                const float wv = w_gm[w_base_kh + static_cast<uint64_t>(kw) * w_stride_kw];

                                acc += xv * wv;
                            }
                        }
                    }

                    acc += bval;

                    // ReLU
                    acc = acc > 0.0f ? acc : 0.0f;

                    // HardSwish: x * clamp((x + 3) / 6, 0, 1)
                    float t = (acc + 3.0f) * (1.0f / 6.0f);
                    t = (t > 0.0f) ? t : 0.0f;
                    t = (t < 1.0f) ? t : 1.0f;
                    acc = acc * t;

                    y_gm[y_base_n + static_cast<uint64_t>(co) * y_stride_c
                         + static_cast<uint64_t>(oh) * y_stride_h
                         + static_cast<uint64_t>(ow) * y_stride_w] = acc;
                }
            }
        }
    }
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor conv2d_relu_hard_swish_impl_npu(const at::Tensor& x,
                                                  const at::Tensor& weight,
                                                  c10::optional<at::Tensor> bias_opt) {
    // Expect float32 NCHW and weight [Cout, Cin, K, K]
    // Output shape: [N, Cout, H-K+1, W-K+1] with stride=1, pad=0, dil=1
    TORCH_CHECK(x.dim() == 4, "x must be 4D NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be 4D [Cout, Cin, K, K]");
    const auto N = x.size(0);
    const auto H = x.size(2);
    const auto W = x.size(3);
    const auto K = weight.size(2);

    const auto Ho = H - K + 1;
    const auto Wo = W - K + 1;
    TORCH_CHECK(Ho > 0 && Wo > 0, "Invalid shapes for stride=1, pad=0");

    auto result = at::empty({N, weight.size(0), Ho, Wo}, x.options());

    if (bias_opt.has_value()) {
        EXEC_NPU_CMD(aclnnConv2dReluHardSwishCustom, x, weight, bias_opt.value(), result);
    } else {
        EXEC_NPU_CMD(aclnnConv2dReluHardSwishCustom, x, weight, result);
    }
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_relu_hard_swish_custom", &conv2d_relu_hard_swish_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_relu_hard_swish_custom", &conv2d_relu_hard_swish_impl_npu,
          "Fused Conv2D + ReLU + HardSwish (Conv2dReluHardSwishCustom on NPU)");
}
