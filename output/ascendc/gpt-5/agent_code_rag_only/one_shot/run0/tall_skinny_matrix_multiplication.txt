host_operator_src="""
#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1U) / b; }
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Obtain input shapes: A:[M,K], B:[K,N]
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    const int64_t a_rank = a_shape->GetOriginShape().GetDimNum();
    const int64_t b_rank = b_shape->GetOriginShape().GetDimNum();
    if (a_rank != 2 || b_rank != 2) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const uint32_t M = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    const uint32_t Kb = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(0));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(1));
    if (K != Kb) {
        return ge::GRAPH_PARAM_INVALID;
    }

    // Conservative blockDim for stability, favoring tall-skinny cases.
    uint32_t tilesM = (M == 0U) ? 1U : CeilDiv(M, 128U);
    uint32_t blockDim = tilesM;
    if (blockDim < 8U) blockDim = 8U;
    if (blockDim > 24U) blockDim = 24U;
    context->SetBlockDim(blockDim);

    // Fill basic cube tiling data
    optiling::TCubeTiling cube;
    cube.Reset();
    cube.M = M;
    cube.N = N;
    cube.K = K;
    cube.BatchNum = 1;
    cube.IsTransA = 0;
    cube.IsTransB = 0;
    cube.shareMode = 0;                 // share L1/L0C
    cube.shareL1Size = FULL_L1_SIZE;    // use full L1
    cube.shareL0CSize = FULL_L0C_SIZE;  // use full L0C
    cube.shareUbSize = 0;               // no UB share

    // Save tiling to buffer
    auto *raw = context->GetRawTilingData();
    cube.SaveToBuffer(raw->GetData(), raw->GetCapacity());
    raw->SetDataSize(cube.GetDataSize());

    // No workspace required
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_PARAM_INVALID;
    }
    const int64_t M = a_shape->GetOriginShape().GetDim(0);
    const int64_t K = a_shape->GetOriginShape().GetDim(1);
    const int64_t Kb = b_shape->GetOriginShape().GetDim(0);
    const int64_t N = b_shape->GetOriginShape().GetDim(1);
    if (K != Kb) {
        return GRAPH_PARAM_INVALID;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    gert::Shape out({M, N});
    *c_shape = out;
    return GRAPH_SUCCESS;
}

static graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output uses float accumulation by design
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class TallSkinnyMatrixMultiplicationCustom : public OpDef {
public:
    explicit TallSkinnyMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape)
            .SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(TallSkinnyMatrixMultiplicationCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

// Tall & skinny GEMM: C = A * B
// A: [M,K] (GM, ND, half), B: [K,N] (GM, ND, half), C: [M,N] (GM, ND, float)
extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    // Prepare tiling from framework
    GET_TILING_DATA(tilingData, tiling);

    // Define MatMul types
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::BSNGD> A_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, /*transB=*/true, LayoutMode::BSNGD> B_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD> C_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> BIAS_TYPE;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> mm;
    AscendC::TPipe pipe;

    // Favor shared L1/L0C to maximize reuse in tall/skinny cases
    tilingData.cubeTilingData.shareMode = 0;
    tilingData.cubeTilingData.shareL1Size = FULL_L1_SIZE;
    tilingData.cubeTilingData.shareL0CSize = FULL_L0C_SIZE;
    tilingData.cubeTilingData.shareUbSize = 0;

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm.matmulObj, &tilingData.cubeTilingData);

    // BatchNum is 1 for a single GEMM
    const int batchNum = 1;

    // No bias
    mm.Init(a, b, nullptr, c, workspace, tilingData.cubeTilingData);

    // Execute
    mm.Process<false>(&pipe, batchNum, batchNum);
}
"""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor tall_skinny_mm_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Result dtype is float (accumulation in higher precision)
    std::vector<int64_t> out_sizes = {A.size(0), B.size(1)};
    at::Tensor C = at::empty(out_sizes, A.options().dtype(at::kFloat));
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("tall_skinny_matrix_multiplication_custom", &tall_skinny_mm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("tall_skinny_matrix_multiplication_custom", &tall_skinny_mm_impl_npu,
          "Tall & Skinny Matrix Multiplication (TallSkinnyMatrixMultiplicationCustom on NPU)");
