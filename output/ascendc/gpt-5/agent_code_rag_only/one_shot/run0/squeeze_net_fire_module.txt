namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_TILE = 8; // small alignment for H/W tiles

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SqueezeNetFireModuleCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    // Expect NCHW
    uint32_t n  = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t c  = static_cast<uint32_t>(xShape->GetDim(1));
    uint32_t h  = static_cast<uint32_t>(xShape->GetDim(2));
    uint32_t w  = static_cast<uint32_t>(xShape->GetDim(3));

    const gert::Shape *wSqShape  = context->GetInputShape(1); // [C_s, C_in, 1, 1]
    uint32_t cSqueeze = static_cast<uint32_t>(wSqShape->GetDim(0));
    uint32_t cIn      = static_cast<uint32_t>(wSqShape->GetDim(1));

    const gert::Shape *wE1Shape = context->GetInputShape(3); // [C_e1, C_s, 1, 1]
    uint32_t cExpand1 = static_cast<uint32_t>(wE1Shape->GetDim(0));

    const gert::Shape *wE3Shape = context->GetInputShape(5); // [C_e3, C_s, 3, 3]
    uint32_t cExpand3 = static_cast<uint32_t>(wE3Shape->GetDim(0));

    // Choose tile size for spatial domain, considering we will compute s (1x1 conv) locally
    // and for expand3 we may recompute s at 9 neighbor points per pixel.
    // Keep tiles modest to limit register/UB usage.
    uint32_t tileW = 32;
    uint32_t tileH = 32;

    // Very conservative memory bound: we will allocate a few local scalars and small vectors.
    // Keep tiles within H/W.
    if (tileW > w) tileW = w;
    if (tileH > h) tileH = h;
    tileW = AlignDown(tileW, ALIGN_TILE);
    tileH = AlignDown(tileH, ALIGN_TILE);
    if (tileW == 0) tileW = w;
    if (tileH == 0) tileH = h;

    // Compute blockDim based on number of tiles
    const uint32_t tilesPerImg = CeilDiv(h, tileH) * CeilDiv(w, tileW);
    uint32_t blockDim = tilesPerImg * n;
#ifdef CUSTOM_ASCEND310P
    if (blockDim > 32u) blockDim = 32u;
#else
    if (blockDim > 24u) blockDim = 24u;
#endif
    if (blockDim == 0u) blockDim = 1u;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_cIn(cIn);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_cSqueeze(cSqueeze);
    tiling.set_cExpand1(cExpand1);
    tiling.set_cExpand3(cExpand3);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t QNUM_SMALL = 1;

class KernelSqueezeNetFireModuleCustom {
public:
    __aicore__ inline KernelSqueezeNetFireModuleCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wSq, GM_ADDR bSq,
                                GM_ADDR wE1, GM_ADDR bE1,
                                GM_ADDR wE3, GM_ADDR bE3,
                                GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t w,
                                uint32_t cS, uint32_t cE1, uint32_t cE3,
                                uint32_t tileH, uint32_t tileW)
    {
        n_ = n; cIn_ = cIn; h_ = h; w_ = w;
        cS_ = cS; cE1_ = cE1; cE3_ = cE3;
        tileH_ = tileH; tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cIn_ * h_ * w_);
        wSqGm_.SetGlobalBuffer((__gm__ float*)wSq, cS_ * cIn_);
        bSqGm_.SetGlobalBuffer((__gm__ float*)bSq, cS_);

        wE1Gm_.SetGlobalBuffer((__gm__ float*)wE1, cE1_ * cS_);
        bE1Gm_.SetGlobalBuffer((__gm__ float*)bE1, cE1_);

        wE3Gm_.SetGlobalBuffer((__gm__ float*)wE3, cE3_ * cS_ * 3 * 3);
        bE3Gm_.SetGlobalBuffer((__gm__ float*)bE3, cE3_);

        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * (cE1_ + cE3_) * h_ * w_);

        pipe_.InitBuffer(tmpQ_, QNUM_SMALL, (cS_ * 9 + 16) * sizeof(float));
        pipe_.InitBuffer(sVecQ_, QNUM_SMALL, cS_ * sizeof(float));
        pipe_.InitBuffer(oneValQ_, QNUM_SMALL, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t tilesW = CeilDiv(w_, tileW_);
        const uint32_t tilesH = CeilDiv(h_, tileH_);
        const uint32_t tilesPerImg = tilesW * tilesH;
        const uint32_t totalTiles = tilesPerImg * n_;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            uint32_t nIdx, tileId;
            DecomposeTile(t, tilesPerImg, nIdx, tileId);
            uint32_t th = tileId / tilesW;
            uint32_t tw = tileId % tilesW;

            uint32_t h0 = th * tileH_;
            uint32_t w0 = tw * tileW_;
            uint32_t curH = Min(tileH_, h_ - h0);
            uint32_t curW = Min(tileW_, w_ - w0);

            ComputeTile(nIdx, h0, w0, curH, curW);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
    __aicore__ inline uint32_t Min(uint32_t a, uint32_t b) { return (a < b) ? a : b; }

    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * cIn_ + c) * h_ + h) * w_ + w;
    }
    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t cTot) const
    {
        return (((uint64_t)n * cTot + c) * h_ + h) * w_ + w;
    }
    __aicore__ inline uint64_t OffsetWSq(uint32_t cs, uint32_t ci) const { return (uint64_t)cs * cIn_ + ci; }
    __aicore__ inline uint64_t OffsetWE1(uint32_t ce, uint32_t cs) const { return (uint64_t)ce * cS_ + cs; }
    __aicore__ inline uint64_t OffsetWE3(uint32_t ce, uint32_t cs, uint32_t kh, uint32_t kw) const
    {
        return (((uint64_t)ce * cS_ + cs) * 3 + kh) * 3 + kw;
    }

    __aicore__ inline float ReadScalarGM(const AscendC::GlobalTensor<float>& gm, uint64_t off)
    {
        AscendC::LocalTensor<float> t = oneValQ_.AllocTensor<float>();
        AscendC::DataCopy(t, gm[off], 1);
        float v = t.GetValue(0);
        oneValQ_.FreeTensor(t);
        return v;
    }

    __aicore__ inline void WriteScalarGM(AscendC::GlobalTensor<float>& gm, uint64_t off, float v)
    {
        AscendC::LocalTensor<float> t = oneValQ_.AllocTensor<float>();
        t.SetValue(0, v);
        AscendC::DataCopy(gm[off], t, 1);
        oneValQ_.FreeTensor(t);
    }

    __aicore__ inline void ComputeSVector(uint32_t nIdx, uint32_t hi, uint32_t wi, AscendC::LocalTensor<float>& sVec)
    {
        // s = ReLU(b_squeeze + sum_ci w_squeeze[:,ci]*x[ci])
        // initialize s with bias
        for (uint32_t cs = 0; cs < cS_; ++cs) {
            float b = ReadScalarGM(bSqGm_, cs);
            sVec.SetValue(cs, b);
        }

        for (uint32_t ci = 0; ci < cIn_; ++ci) {
            float xval = ReadScalarGM(xGm_, OffsetX(nIdx, ci, hi, wi));
            // accumulate for each cs
            for (uint32_t cs = 0; cs < cS_; ++cs) {
                float w = ReadScalarGM(wSqGm_, OffsetWSq(cs, ci));
                float acc = sVec.GetValue(cs);
                acc = AscendC::fadd(acc, AscendC::fmul(w, xval));
                sVec.SetValue(cs, acc);
            }
        }
        // ReLU
        for (uint32_t cs = 0; cs < cS_; ++cs) {
            float v = sVec.GetValue(cs);
            if (v < 0.f) v = 0.f;
            sVec.SetValue(cs, v);
        }
    }

    __aicore__ inline void ComputeE1Write(uint32_t nIdx, uint32_t hi, uint32_t wi, const AscendC::LocalTensor<float>& sVec)
    {
        // e1 = ReLU(b_expand1 + sum_cs w_expand1* s)
        const uint32_t cTot = cE1_ + cE3_;
        for (uint32_t ce = 0; ce < cE1_; ++ce) {
            float acc = ReadScalarGM(bE1Gm_, ce);
            for (uint32_t cs = 0; cs < cS_; ++cs) {
                float w = ReadScalarGM(wE1Gm_, OffsetWE1(ce, cs));
                float s = sVec.GetValue(cs);
                acc = AscendC::fadd(acc, AscendC::fmul(w, s));
            }
            if (acc < 0.f) acc = 0.f;
            WriteScalarGM(yGm_, OffsetY(nIdx, ce, hi, wi, cTot), acc);
        }
    }

    __aicore__ inline void ComputeE3Write(uint32_t nIdx, uint32_t hi, uint32_t wi)
    {
        // Precompute 9 neighbor s vectors (with zero-padding outside image)
        AscendC::LocalTensor<float> s9 = tmpQ_.AllocTensor<float>(); // length cS_*9
        for (uint32_t k = 0; k < 9 * cS_; ++k) {
            s9.SetValue(k, 0.f);
        }

        for (int kh = -1; kh <= 1; ++kh) {
            int hh = static_cast<int>(hi) + kh;
            if (hh < 0 || hh >= static_cast<int>(h_)) {
                continue;
            }
            for (int kw = -1; kw <= 1; ++kw) {
                int ww = static_cast<int>(wi) + kw;
                if (ww < 0 || ww >= static_cast<int>(w_)) {
                    continue;
                }
                AscendC::LocalTensor<float> sVec = sVecQ_.AllocTensor<float>();
                ComputeSVector(nIdx, static_cast<uint32_t>(hh), static_cast<uint32_t>(ww), sVec);
                uint32_t kidx = (kh + 1) * 3 + (kw + 1);
                for (uint32_t cs = 0; cs < cS_; ++cs) {
                    s9.SetValue(kidx * cS_ + cs, sVec.GetValue(cs));
                }
                sVecQ_.FreeTensor(sVec);
            }
        }

        // e3 = ReLU(b_expand3 + sum_{kh,kw,cs} W3 * s9)
        const uint32_t cTot = cE1_ + cE3_;
        for (uint32_t ce = 0; ce < cE3_; ++ce) {
            float acc = ReadScalarGM(bE3Gm_, ce);
            for (uint32_t kh = 0; kh < 3; ++kh) {
                for (uint32_t kw = 0; kw < 3; ++kw) {
                    uint32_t kidx = kh * 3 + kw;
                    for (uint32_t cs = 0; cs < cS_; ++cs) {
                        float w = ReadScalarGM(wE3Gm_, OffsetWE3(ce, cs, kh, kw));
                        float s = s9.GetValue(kidx * cS_ + cs);
                        acc = AscendC::fadd(acc, AscendC::fmul(w, s));
                    }
                }
            }
            if (acc < 0.f) acc = 0.f;
            // write to y at channel (cE1_ + ce)
            WriteScalarGM(yGm_, OffsetY(nIdx, cE1_ + ce, hi, wi, cTot), acc);
        }

        tmpQ_.FreeTensor(s9);
    }

    __aicore__ inline void ComputeTile(uint32_t nIdx, uint32_t h0, uint32_t w0, uint32_t th, uint32_t tw)
    {
        for (uint32_t ih = 0; ih < th; ++ih) {
            uint32_t hi = h0 + ih;
            for (uint32_t iw = 0; iw < tw; ++iw) {
                uint32_t wi = w0 + iw;

                AscendC::LocalTensor<float> sVec = sVecQ_.AllocTensor<float>();
                ComputeSVector(nIdx, hi, wi, sVec);
                ComputeE1Write(nIdx, hi, wi, sVec);
                sVecQ_.FreeTensor(sVec);

                ComputeE3Write(nIdx, hi, wi);
            }
        }
    }

    __aicore__ inline void DecomposeTile(uint32_t t, uint32_t tilesPerImg, uint32_t &nIdx, uint32_t &tileId)
    {
        nIdx = t / tilesPerImg;
        tileId = t - nIdx * tilesPerImg;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM_SMALL> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM_SMALL> sVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM_SMALL> oneValQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wSqGm_;
    AscendC::GlobalTensor<float> bSqGm_;
    AscendC::GlobalTensor<float> wE1Gm_;
    AscendC::GlobalTensor<float> bE1Gm_;
    AscendC::GlobalTensor<float> wE3Gm_;
    AscendC::GlobalTensor<float> bE3Gm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{1}, cIn_{1}, h_{1}, w_{1};
    uint32_t cS_{1}, cE1_{1}, cE3_{1};
    uint32_t tileH_{1}, tileW_{1};
};

extern "C" __global__ __aicore__ void squeeze_net_fire_module_custom(
    GM_ADDR x, GM_ADDR w_squeeze, GM_ADDR b_squeeze,
    GM_ADDR w_expand1, GM_ADDR b_expand1,
    GM_ADDR w_expand3, GM_ADDR b_expand3,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelSqueezeNetFireModuleCustom op;
    op.Init(x, w_squeeze, b_squeeze, w_expand1, b_expand1, w_expand3, b_expand3, y,
            tiling_data.n, tiling_data.cIn, tiling_data.h, tiling_data.w,
            tiling_data.cSqueeze, tiling_data.cExpand1, tiling_data.cExpand3,
            tiling_data.tileH, tiling_data.tileW);
    op.Process();
}
