host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
// We use 5 tiles of float in UB for double-buffering input/output and one temp buffer.
static constexpr uint32_t BYTES_PER_ELEM = 4;
static constexpr uint32_t BUFFERS_PER_TILE = 5;          // in:2, out:2, tmp:1
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *axisPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t axis = (axisPtr != nullptr) ? *axisPtr : 1;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape &oShape = xShape->GetOriginShape();
    const int64_t rank = oShape.GetDimNum();

    // Normalize axis in [-rank, rank)
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }
    int32_t normAxis = axis;
    if (normAxis < 0) normAxis += static_cast<int32_t>(rank);
    if (normAxis < 0) normAxis = 0;
    if (normAxis >= rank) normAxis = static_cast<int32_t>(rank) - 1;

    uint64_t M = 1;
    uint64_t K = 1;
    for (int64_t i = 0; i < normAxis; ++i) {
        M *= static_cast<uint64_t>(oShape.GetDim(i));
    }
    for (int64_t i = normAxis; i < rank; ++i) {
        K *= static_cast<uint64_t>(oShape.GetDim(i));
    }

    // tileCols from UB budget: bytes = BUFFERS_PER_TILE * tileCols * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (BUFFERS_PER_TILE * BYTES_PER_ELEM);  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileCols = static_cast<uint32_t>(maxTile);
    if (K < tileCols) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(K), ALIGN_ELEMS);
        tileCols = (aligned > 0) ? aligned : static_cast<uint32_t>(K);
    }

    // Adaptive blockDim based on number of rows; keep conservative for stability.
    uint32_t mRows = static_cast<uint32_t>(M);
    uint32_t blockDim = mRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_mRows(mRows);
    tiling.set_nCols(static_cast<uint32_t>(K));
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);
    tiling.set_axis(normAxis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("axis").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops
"""

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t mRows, uint32_t nCols, uint32_t tileCols)
    {
        mRows_ = mRows;
        nCols_ = nCols;
        tileCols_ = tileCols;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(mRows_) * nCols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(mRows_) * nCols_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileCols_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (mRows_ == 0 || nCols_ == 0 || tileCols_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < mRows_; row += blockNum) {
            const uint64_t rowBase = static_cast<uint64_t>(row) * nCols_;
            float rowMax = -3.402823466e+38F; // -FLT_MAX
            float rowSum = 0.0f;

            // Pass 1: compute rowMax and rowSum with log-sum-exp trick
            for (uint32_t c0 = 0; c0 < nCols_; c0 += tileCols_) {
                uint32_t len = nCols_ - c0;
                if (len > tileCols_) len = tileCols_;

                CopyIn(rowBase + c0, len);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();

                // tileMax = reduce_max(xLocal)
                float tileMax = AscendC::ReduceMax(xLocal, len);
                float newMax = (rowMax > tileMax) ? rowMax : tileMax;

                // tmp = xLocal - newMax
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                AscendC::Adds(tmp, xLocal, -newMax, len);
                // tmp = exp(tmp)
                AscendC::Exp(tmp, tmp, len);
                // tileSum = reduce_sum(tmp)
                float tileSum = AscendC::ReduceSum(tmp, len);

                // update running sum and max: s' = s*exp(m - m') + tileSum; m' = newMax
                float corr = AscendC::Exp2((rowMax - newMax) * 1.4426950408889634f); // exp(x) via exp2(x*log2(e))
                // In case Exp2 is not available, Exp should be fine as well; keep Exp as primary
                corr = AscendC::Exp(rowMax - newMax);
                rowSum = rowSum * corr + tileSum;
                rowMax = newMax;

                tmpQueue_.FreeTensor(tmp);
                inQueueX_.FreeTensor(xLocal);
            }

            // Avoid division by zero
            float invSum = 0.0f;
            if (rowSum > 0.0f) {
                invSum = 1.0f / rowSum;
            }

            // Pass 2: write outputs
            for (uint32_t c0 = 0; c0 < nCols_; c0 += tileCols_) {
                uint32_t len = nCols_ - c0;
                if (len > tileCols_) len = tileCols_;

                CopyIn(rowBase + c0, len);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // y = exp(x - rowMax) * invSum
                AscendC::Adds(yLocal, xLocal, -rowMax, len);
                AscendC::Exp(yLocal, yLocal, len);
                AscendC::Muls(yLocal, yLocal, invSum, len);

                outQueueY_.EnQue(yLocal);
                inQueueX_.FreeTensor(xLocal);

                CopyOut(rowBase + c0, len);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t mRows_{0};
    uint32_t nCols_{0};
    uint32_t tileCols_{0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.mRows, tiling_data.nCols, tiling_data.tileCols);
    op.Process();
}
"""
