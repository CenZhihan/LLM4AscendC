host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) { return static_cast<uint32_t>((a + b - 1ull) / b); }
} // namespace

namespace optiling {

// UB model for this kernel per tile of 'inner':
//   yLocal (accumulator): 1 * tile
//   xLocal (current slice): 1 * tile
//   potential buffering: ~1 * tile
// Total floats budget ~ 3 * tile => bytes = 12 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor large DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // cap per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    const gert::Shape &orig = inShape->GetOriginShape();
    const int64_t rank = static_cast<int64_t>(orig.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = 0;
    if (attrs != nullptr) {
        const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
        dim = *dimPtr;
    }
    // normalize dim
    int64_t nd = dim;
    if (nd < 0) nd += rank;
    if (nd < 0) nd = 0;
    if (nd >= rank) nd = rank - 1;

    uint64_t outer = 1;
    uint64_t inner = 1;
    uint64_t reduce = 1;
    for (int64_t i = 0; i < rank; ++i) {
        const uint64_t d = static_cast<uint64_t>(orig.GetDim(i));
        if (i < nd) outer *= d;
        else if (i == nd) reduce = d;
        else inner *= d;
    }
    if (reduce == 0) {
        // avoid div by zero; degenerate case
        reduce = 1;
    }

    // compute tileInner from UB budget: bytes = 12 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileInner = static_cast<uint32_t>(maxTile);
    if (inner < tileInner) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(inner), ALIGN_ELEMS);
        tileInner = (aligned > 0) ? aligned : static_cast<uint32_t>(inner);
    }

    const uint32_t tilesPerRow = (tileInner == 0) ? 1u : CeilDiv(inner, tileInner);
    uint64_t totalWorks64 = outer * static_cast<uint64_t>(tilesPerRow);
    uint32_t totalWorks = (totalWorks64 > 0xFFFFFFFFull) ? 0xFFFFFFFFu : static_cast<uint32_t>(totalWorks64);

    uint32_t blockDim = totalWorks;
    if (blockDim < 8u) blockDim = (totalWorks == 0u) ? 1u : ((totalWorks < 8u) ? totalWorks : 8u);
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_outer(static_cast<uint32_t>(outer));
    tiling.set_reduce(static_cast<uint32_t>(reduce));
    tiling.set_inner(static_cast<uint32_t>(inner));
    tiling.set_tileInner(static_cast<uint32_t>(tileInner));
    tiling.set_tilesPerRow(static_cast<uint32_t>(tilesPerRow));
    tiling.set_blockDim(static_cast<uint32_t>(blockDim));
    tiling.set_dim(static_cast<int32_t>(nd));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    int64_t dim = 0;
    if (attrs != nullptr) {
        const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
        dim = *dimPtr;
    }
    const gert::Shape &orig = x_shape->GetOriginShape();
    const int64_t rank = static_cast<int64_t>(orig.GetDimNum());
    int64_t nd = dim;
    if (nd < 0) nd += rank;
    if (nd < 0) nd = 0;
    if (nd >= rank) nd = rank - 1;

    // Build output shape with 'dim' removed
    std::vector<int64_t> outDims;
    outDims.reserve(rank > 0 ? (rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == nd) continue;
        outDims.push_back(orig.GetDim(i));
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    // Assign dims
    y_shape->SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) {
        y_shape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUFFER_NUM = 2;
constexpr int32_t OUT_BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t reduce, uint32_t inner,
                                uint32_t tileInner, uint32_t tilesPerRow)
    {
        outer_ = outer;
        reduce_ = reduce;
        inner_ = inner;
        tileInner_ = tileInner;
        tilesPerRow_ = tilesPerRow;
        invReduce_ = (reduce_ == 0u) ? 0.0f : (1.0f / static_cast<float>(reduce_));

        const uint32_t xTotal = outer_ * reduce_ * inner_;
        const uint32_t yTotal = outer_ * inner_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xTotal);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yTotal);

        pipe_.InitBuffer(inQueueX_, IN_BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUFFER_NUM, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (inner_ == 0 || outer_ == 0 || reduce_ == 0 || tileInner_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalWorks = static_cast<uint64_t>(outer_) * static_cast<uint64_t>(tilesPerRow_);
        if (totalWorks == 0) return;

        for (uint64_t work = blockIdx; work < totalWorks; work += blockNum) {
            const uint32_t o = static_cast<uint32_t>(work / tilesPerRow_);
            const uint32_t tileIdx = static_cast<uint32_t>(work % tilesPerRow_);
            const uint32_t innerOffset = tileIdx * tileInner_;
            uint32_t len = inner_ - innerOffset;
            if (len > tileInner_) len = tileInner_;

            // yLocal accumulator
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Duplicate(yLocal, 0.0f, len);

            // Sum over 'reduce_' slices
            const uint64_t outerBase = static_cast<uint64_t>(o) * static_cast<uint64_t>(reduce_) * static_cast<uint64_t>(inner_);
            for (uint32_t r = 0; r < reduce_; ++r) {
                const uint64_t gmOffset = outerBase + static_cast<uint64_t>(r) * static_cast<uint64_t>(inner_) + innerOffset;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xTmp = inQueueX_.DeQue<float>();
                AscendC::Add(yLocal, yLocal, xTmp, len);
                inQueueX_.FreeTensor(xTmp);
            }

            AscendC::Muls(yLocal, yLocal, invReduce_, len);

            outQueueY_.EnQue<float>(yLocal);

            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            const uint64_t yOffset = static_cast<uint64_t>(o) * static_cast<uint64_t>(inner_) + innerOffset;
            AscendC::DataCopy(yGm_[yOffset], yOut, len);
            outQueueY_.FreeTensor(yOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_{0};
    uint32_t reduce_{0};
    uint32_t inner_{0};
    uint32_t tileInner_{0};
    uint32_t tilesPerRow_{0};
    float invReduce_{0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.outer, tiling_data.reduce, tiling_data.inner,
            tiling_data.tileInner, tiling_data.tilesPerRow);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor mean_reduction_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    // Normalize negative dim on Python side and allocate correct output
    int64_t rank = self.dim();
    int64_t nd = dim;
    if (nd < 0) nd += rank;
    TORCH_CHECK(nd >= 0 && nd < rank, \"dim out of range\");
    std::vector<int64_t> outSizes;
    outSizes.reserve(rank - 1);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == nd) continue;
        outSizes.push_back(self.size(i));
    }
    at::Tensor result = at::empty(outSizes, self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
