host_operator_src="""

#include "matmul_with_transposed_both_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1U) / b; }
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Input A: shape (K, M), Input B: shape (N, K)
    // We compute C = A^T * B^T => shape (M, N)
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t K_a = static_cast<uint64_t>(a_shape->GetDim(0));
    const uint64_t M = static_cast<uint64_t>(a_shape->GetDim(1));
    const uint64_t N = static_cast<uint64_t>(b_shape->GetDim(0));
    const uint64_t K_b = static_cast<uint64_t>(b_shape->GetDim(1));
    if (K_a != K_b) {
        return ge::GRAPH_FAILED;
    }

    // Basic tiling: split M and N across cores. Simple, conservative choices.
    // Aim for per-core tiles that are multiples of 16 for better alignment.
    const uint32_t ALIGN_M = 16;
    const uint32_t ALIGN_N = 16;

    const uint32_t M_u32 = static_cast<uint32_t>(M);
    const uint32_t N_u32 = static_cast<uint32_t>(N);
    const uint32_t K_u32 = static_cast<uint32_t>(K_a);

    // Favorite per-core target tile sizes
    uint32_t targetTileM = 128;
    uint32_t targetTileN = 128;
    if (targetTileM > M_u32) targetTileM = AlignDown(M_u32, ALIGN_M) ? AlignDown(M_u32, ALIGN_M) : M_u32;
    if (targetTileN > N_u32) targetTileN = AlignDown(N_u32, ALIGN_N) ? AlignDown(N_u32, ALIGN_N) : N_u32;

    const uint32_t mTiles = (targetTileM == 0) ? 1U : CeilDiv(M_u32, targetTileM);
    const uint32_t nTiles = (targetTileN == 0) ? 1U : CeilDiv(N_u32, targetTileN);
    uint32_t blockDim = mTiles * nTiles;
    if (blockDim < 1U) blockDim = 1U;
    if (blockDim > 24U) blockDim = 24U; // conservative upper bound

    // Recompute singleCoreM/N to reflect number of blocks we actually launch.
    // Try to keep aspect ratio: use up to blockDim tiles by increasing N tiles first.
    uint32_t usedMTiles = mTiles;
    uint32_t usedNTiles = nTiles;
    while (usedMTiles * usedNTiles > blockDim) {
        if (usedNTiles > 1) {
            --usedNTiles;
        } else if (usedMTiles > 1) {
            --usedMTiles;
        } else {
            break;
        }
    }

    uint32_t singleCoreM = CeilDiv(M_u32, usedMTiles);
    uint32_t singleCoreN = CeilDiv(N_u32, usedNTiles);
    singleCoreM = AlignDown(singleCoreM, ALIGN_M) ? AlignDown(singleCoreM, ALIGN_M) : singleCoreM;
    singleCoreN = AlignDown(singleCoreN, ALIGN_N) ? AlignDown(singleCoreN, ALIGN_N) : singleCoreN;
    if (singleCoreM == 0) singleCoreM = M_u32;
    if (singleCoreN == 0) singleCoreN = N_u32;

    context->SetBlockDim(blockDim);

    MatmulWithTransposedBothCustomTilingData tiling;
    tiling.set_M(M_u32);
    tiling.set_N(N_u32);
    tiling.set_K(K_u32);
    tiling.set_Ka(K_u32);  // Leading K for A
    tiling.set_Kb(K_u32);  // Leading K for B
    tiling.set_singleCoreM(singleCoreM);
    tiling.set_singleCoreN(singleCoreN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0); // (K, M)
    const gert::Shape *b_shape = context->GetInputShape(1); // (N, K)
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = a_shape->GetDim(1);
    const int64_t N = b_shape->GetDim(0);
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

using AscendC::MatmulType;
using AscendC::Matmul;
using AscendC::TPosition;
using AscendC::CubeFormat;

template <typename aType, typename bType, typename cType>
class KernelMatmulTransposedBoth {
public:
    __aicore__ inline KernelMatmulTransposedBoth() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, const auto &t)
    {
        // tiling data fields:
        // t.M: rows of output (M), t.N: cols of output (N), t.K: reduction dim (K)
        // Inputs are provided as:
        //   A of shape (K, M) row-major in GM
        //   B of shape (N, K) row-major in GM
        // We compute C = A^T * B^T with shape (M, N).
        tiling_ = t;

        aGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ aType*>(a), static_cast<uint32_t>(tiling_.K) * static_cast<uint32_t>(tiling_.M));
        bGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ bType*>(b), static_cast<uint32_t>(tiling_.N) * static_cast<uint32_t>(tiling_.K));
        cGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ cType*>(c), static_cast<uint32_t>(tiling_.M) * static_cast<uint32_t>(tiling_.N));

        // Compute per-core GM offsets based on a simple 2D split across (M, N)
        int32_t offsetA = 0, offsetB = 0, offsetC = 0;
        CalcOffset(GetBlockIdx(), offsetA, offsetB, offsetC);

        aGlobal_ = aGlobal_[offsetA];
        bGlobal_ = bGlobal_[offsetB];
        cGlobal_ = cGlobal_[offsetC];

        // Optional local workspace for matmul formatting to improve performance.
        AscendC::TPipe pipe;
        pipe.InitBuffer(tmpUb_, AscendC::TOTAL_VEC_LOCAL_SIZE);
        mmLocal_ = tmpUb_.Get<uint8_t>(AscendC::TOTAL_VEC_LOCAL_SIZE);
        matmulObj_.SetLocalWorkspace(mmLocal_);
    }

    __aicore__ inline void Process()
    {
        // Define Matmul types: both A and B marked as transposed true.
        matmulObj_.SetTensorA(aGlobal_);
        matmulObj_.SetTensorB(bGlobal_);
        matmulObj_.IterateAll(cGlobal_);
        matmulObj_.End();
    }

private:
    __aicore__ inline void CalcOffset(int32_t blockIdx, int32_t &offsetA, int32_t &offsetB, int32_t &offsetC)
    {
        // Divide cores along M (rows) and N (cols) using singleCoreM and singleCoreN.
        const int32_t mBlocks = AscendC::CeilDiv(static_cast<int32_t>(tiling_.M), static_cast<int32_t>(tiling_.singleCoreM));
        const int32_t mCoreIdx = blockIdx % mBlocks;
        const int32_t nCoreIdx = blockIdx / mBlocks;

        // A is (K, M) row-major; advancing 'i' (M dimension) by singleCoreM shifts GM by singleCoreM.
        offsetA = mCoreIdx * static_cast<int32_t>(tiling_.singleCoreM);

        // B is (N, K) row-major; advancing 'j' (N dimension) by singleCoreN shifts GM by singleCoreN * K.
        offsetB = nCoreIdx * static_cast<int32_t>(tiling_.singleCoreN) * static_cast<int32_t>(tiling_.K);

        // C is (M, N) row-major; offset by (mCoreIdx * singleCoreM) rows and (nCoreIdx * singleCoreN) cols.
        offsetC = mCoreIdx * static_cast<int32_t>(tiling_.singleCoreM) * static_cast<int32_t>(tiling_.N)
                + nCoreIdx * static_cast<int32_t>(tiling_.singleCoreN);
    }

private:
    // A and B are marked as transposed in the MatmulType so that the hardware performs A^T * B^T
    Matmul<MatmulType<TPosition::GM, CubeFormat::ND, aType, true>,
           MatmulType<TPosition::GM, CubeFormat::ND, bType, true>,
           MatmulType<TPosition::GM, CubeFormat::ND, cType>> matmulObj_;

    AscendC::GlobalTensor<aType> aGlobal_;
    AscendC::GlobalTensor<bType> bGlobal_;
    AscendC::GlobalTensor<cType> cGlobal_;

    AscendC::TBuf<> tmpUb_;
    AscendC::LocalTensor<uint8_t> mmLocal_;

    // Tiling structure mirrored from host tiling definition
    struct {
        uint32_t M;
        uint32_t N;
        uint32_t K;
        uint32_t Ka;
        uint32_t Kb;
        uint32_t singleCoreM;
        uint32_t singleCoreN;
        uint32_t blockDim;
    } tiling_;
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm)
{
    // Load tiling data from GM
    GET_TILING_DATA(tiling_data, tilingGm);

    KernelMatmulTransposedBoth<float, float, float> kernel;
    kernel.Init(a, b, c, tiling_data);
    kernel.Process();
}
