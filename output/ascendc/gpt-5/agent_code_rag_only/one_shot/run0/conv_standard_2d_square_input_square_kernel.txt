host_operator_src="""

#include "conv_standard_2d_square_input_square_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : ((x / a) * a);
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
} // namespace

namespace optiling {

// Conservative UB budget for buffers (bytes)
static constexpr uint32_t UB_BUDGET_BYTES = 96 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect x: [N, Cin, H, W], weight: [Cout, Cin, KH, KW]
    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t KH   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t KW   = static_cast<uint32_t>(w_shape->GetDim(3));

    // Only support stride=1, pad=0, dilation=1 in this custom kernel.
    const uint32_t OH = (H >= KH) ? (H - KH + 1) : 0;
    const uint32_t OW = (W >= KW) ? (W - KW + 1) : 0;

    // UB model: for each tile (a stripe along width)
    //   yBuf: tileOW floats
    //   tmpVec: tileOW floats
    //   weights per oc: Cin * KH * KW floats (loaded once per (n, oc) pair)
    // Bytes per tile ~ 4 * (2 * tileOW + Cin*KH*KW)
    uint32_t weightElems = Cin * KH * KW;
    uint32_t maxTileOW = 0;
    if (UB_BUDGET_BYTES > 4u * weightElems) {
        maxTileOW = (UB_BUDGET_BYTES / 4u - weightElems) / 2u;
    }
    if (maxTileOW == 0) maxTileOW = 1u;
    maxTileOW = AlignDown(maxTileOW, ALIGN_ELEMS);
    if (maxTileOW == 0) maxTileOW = 1u;

    uint32_t tileOW = maxTileOW;
    if (OW > 0 && OW < tileOW) {
        tileOW = AlignDown(OW, ALIGN_ELEMS);
        if (tileOW == 0) tileOW = OW;
    }

    // Blocks distribute (n, oc) pairs in a grid-stride fashion.
    const uint64_t totalPairs = static_cast<uint64_t>(N) * static_cast<uint64_t>(Cout);
    uint32_t blockDim = static_cast<uint32_t>(totalPairs);
    if (blockDim == 0) blockDim = 1u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    optiling::ConvStandard2dSquareInputSquareKernelCustomTilingData tiling;
    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Cout(Cout);
    tiling.set_KH(KH);
    tiling.set_KW(KW);
    tiling.set_OH(OH);
    tiling.set_OW(OW);
    tiling.set_tileOW(tileOW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelConvStandard2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t KH, uint32_t KW,
                                uint32_t OH, uint32_t OW, uint32_t tileOW)
    {
        N_ = N; Cin_ = Cin; H_ = H; W_ = W;
        Cout_ = Cout; KH_ = KH; KW_ = KW;
        OH_ = OH; OW_ = OW; tileOW_ = tileOW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_)*Cin_*H_*W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Cout_)*Cin_*KH_*KW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_)*Cout_*OH_*OW_);

        // Buffers: weights per oc (Cin*KH*KW), yBuf (tileOW), tmpVec (tileOW)
        uint32_t weightElems = Cin_ * KH_ * KW_;
        pipe_.InitBuffer(wBufQ_, 1, weightElems * sizeof(float));
        pipe_.InitBuffer(yBufQ_, 1, tileOW_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tileOW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cout_ == 0 || OH_ == 0 || OW_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());
        const uint64_t totalPairs = static_cast<uint64_t>(N_) * static_cast<uint64_t>(Cout_);

        for (uint64_t pair = blockIdx; pair < totalPairs; pair += blockNum) {
            uint32_t n = static_cast<uint32_t>(pair / Cout_);
            uint32_t oc = static_cast<uint32_t>(pair % Cout_);

            // Load weights for this oc: [Cin, KH, KW]
            LocalTensor<float> wLocal = wBufQ_.AllocTensor<float>();
            const uint64_t wBase = static_cast<uint64_t>(oc) * Cin_ * KH_ * KW_;
            DataCopy(wLocal, wGm_[wBase], Cin_ * KH_ * KW_);

            // Iterate over output rows
            for (uint32_t oh = 0; oh < OH_; ++oh) {
                // Tile along output width
                for (uint32_t owBase = 0; owBase < OW_; owBase += tileOW_) {
                    uint32_t curTile = OW_ - owBase;
                    if (curTile > tileOW_) curTile = tileOW_;

                    LocalTensor<float> yLocal = yBufQ_.AllocTensor<float>();
                    // zero initialize
                    Duplicate(yLocal, 0.0f, curTile);

                    // Accumulate over input channels and kernel
                    for (uint32_t ic = 0; ic < Cin; ++ic) { // use Cin variable not member? fix
                    }

                    // Corrected loop using member Cin_
                    for (uint32_t ic2 = 0; ic2 < Cin_; ++ic2) {
                        // For each kernel row
                        for (uint32_t ky = 0; ky < KH_; ++ky) {
                            const uint32_t in_row = oh + ky; // stride=1, no pad
                            // For each kernel col
                            for (uint32_t kx = 0; kx < KW_; ++kx) {
                                // Load input segment directly from GM (length curTile)
                                const uint32_t in_col = owBase + kx;
                                const uint64_t xBase = (((static_cast<uint64_t>(n) * Cin_ + ic2) * H_ + in_row) * W_ + in_col);
                                LocalTensor<float> tmpVec = tmpQ_.AllocTensor<float>();
                                DataCopy(tmpVec, xGm_[xBase], curTile);

                                // Fetch scalar weight
                                const uint32_t wIdx = (ic2 * KH_ + ky) * KW_ + kx;
                                float wScalar = wLocal.GetValue(wIdx);

                                // tmpVec *= wScalar; yLocal += tmpVec;
                                Muls(tmpVec, tmpVec, wScalar, curTile);
                                Add(yLocal, yLocal, tmpVec, curTile);

                                tmpQ_.FreeTensor(tmpVec);
                            }
                        }
                    }

                    // Store y
                    const uint64_t yBase = (((static_cast<uint64_t>(n) * Cout_ + oc) * OH_ + oh) * OW_ + owBase);
                    DataCopy(yGm_[yBase], yLocal, curTile);

                    yBufQ_.FreeTensor(yLocal);
                }
            }

            wBufQ_.FreeTensor(wLocal);
        }
    }

private:
    // Queues and buffers
    TPipe pipe_;
    TQue<TPosition::VECCALC, 1> wBufQ_;
    TQue<TPosition::VECCALC, 1> yBufQ_;
    TQue<TPosition::VECCALC, 1> tmpQ_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    // Params
    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, KH_{0}, KW_{0};
    uint32_t OH_{0}, OW_{0}, tileOW_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.H, tiling_data.W,
            tiling_data.Cout, tiling_data.KH, tiling_data.KW,
            tiling_data.OH, tiling_data.OW, tiling_data.tileOW);
    op.Process();
}
