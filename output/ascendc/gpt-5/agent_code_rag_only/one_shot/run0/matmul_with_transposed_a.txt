host_operator_src="""

#include "matmul_with_transposed_a_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (per-core):
//   yTile: 1 * tileN
//   bTile: 2 * tileN  (double-buffered in)
 //  tmpTile: 1 * tileN
//   aScalar: 2 * 1     (double-buffered scalar)
// Total floats ~ 4 * tileN + small scalars => bytes ~ 16 * tileN + overhead.
// Use conservative UB budget per-core.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;              // reasonable DMA alignment
static constexpr uint32_t HARD_CAP_N_ELEMS = 4096;        // bound tileN to keep latency low

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedACustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    const uint32_t K = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    const uint32_t M = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(1));

    // Compute tileN from UB budget: bytes ~= 16 * tileN
    uint32_t maxTileN = UB_BUDGET_BYTES / 16u;
    maxTileN = AlignDown(maxTileN, ALIGN_ELEMS);
    if (maxTileN < ALIGN_ELEMS) maxTileN = ALIGN_ELEMS;
    if (maxTileN > HARD_CAP_N_ELEMS) maxTileN = AlignDown(HARD_CAP_N_ELEMS, ALIGN_ELEMS);

    uint32_t tileN = maxTileN;
    if (N < tileN) {
        const uint32_t aligned = AlignDown(N, ALIGN_ELEMS);
        tileN = (aligned > 0) ? aligned : N;
    }

    // Grid: tiles along N for each row m
    const uint32_t numNTiles = (tileN == 0) ? 1u : CeilDiv(N, tileN);
    const uint32_t totalTiles = (M == 0 || numNTiles == 0) ? 1u : (M * numNTiles);

    // Conservative blockDim in [8, 24]
    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulWithTransposedACustom {
public:
    __aicore__ inline KernelMatmulWithTransposedACustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR Y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, K_ * M_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, K_ * N_);
        yGm_.SetGlobalBuffer((__gm__ float*)Y, M_ * N_);

        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(aScalarQueue_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0 || tileN_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numNTiles = (N_ + tileN_ - 1u) / tileN_;
        const uint32_t totalTiles = M_ * numNTiles;

        uint32_t tileId = blockIdx;
        while (tileId < totalTiles) {
            const uint32_t mIdx = tileId / numNTiles;
            const uint32_t nTileIdx = tileId % numNTiles;

            const uint32_t nBase = nTileIdx * tileN_;
            uint32_t curLenN = N_ - nBase;
            if (curLenN > tileN_) curLenN = tileN_;

            // Prepare output accumulator
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Dup(yLocal, 0.0f, curLenN);

            // Prefetch first k
            if (K_ > 0) {
                CopyInScalar(mIdx, 0);
                CopyInBRow(0, nBase, curLenN);
            }

            for (uint32_t k = 0; k < K_; ++k) {
                const uint32_t nextK = k + 1;
                if (nextK < K_) {
                    CopyInScalar(mIdx, nextK);
                    CopyInBRow(nextK, nBase, curLenN);
                }

                // Compute y += a_scalar * b_row_vector
                AscendC::LocalTensor<float> aLocal = aScalarQueue_.DeQue<float>();
                AscendC::LocalTensor<float> bLocal = inQueueB_.DeQue<float>();
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

                float aVal = AscendC::GetValue(aLocal, 0);
                AscendC::Muls(tLocal, bLocal, aVal, curLenN);
                AscendC::Add(yLocal, yLocal, tLocal, curLenN);

                tmpQueue_.FreeTensor(tLocal);
                inQueueB_.FreeTensor(bLocal);
                aScalarQueue_.FreeTensor(aLocal);
            }

            // Enqueue and write back
            outQueueY_.EnQue(yLocal);
            CopyOutY(mIdx, nBase, curLenN);
            outQueueY_.FreeTensor(yLocal);

            tileId += blockNum;
        }
    }

private:
    __aicore__ inline void CopyInBRow(uint32_t k, uint32_t nBase, uint32_t len)
    {
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        // B is [K, N], contiguous along N
        AscendC::DataCopy(bLocal, bGm_[k * N_ + nBase], len);
        inQueueB_.EnQue(bLocal);
    }

    __aicore__ inline void CopyInScalar(uint32_t mIdx, uint32_t k)
    {
        AscendC::LocalTensor<float> aLocal = aScalarQueue_.AllocTensor<float>();
        // A is [K, M], scalar at offset k*M + mIdx
        AscendC::DataCopy(aLocal, aGm_[k * M_ + mIdx], 1);
        aScalarQueue_.EnQue(aLocal);
    }

    __aicore__ inline void CopyOutY(uint32_t mIdx, uint32_t nBase, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        // Y is [M, N], contiguous along N
        AscendC::DataCopy(yGm_[mIdx * N_ + nBase], yLocal, len);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aScalarQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_a_custom(GM_ADDR A, GM_ADDR B, GM_ADDR Y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedACustom op;
    op.Init(A, B, Y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileN);
    op.Process();
}
