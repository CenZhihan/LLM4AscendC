host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
// Buffers per tile:
//   curSlice: 4 * tile
//   maxSlice: 4 * tile
//   arg32Slice: 4 * tile
//   y64Slice: 8 * tile
// Total bytes: 20 * tile
static constexpr uint32_t BYTES_PER_TILE_ELEMS = 20;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::OriginShape &orig = xShape->GetOriginShape();
    const int64_t rank = orig.GetDimNum();
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }

    int64_t axis = *dimAttr;
    if (axis < 0) axis += rank;
    if (axis < 0 || axis >= rank) {
        return ge::GRAPH_FAILED;
    }

    uint64_t outerLen = 1;
    for (int64_t i = 0; i < axis; ++i) {
        outerLen *= static_cast<uint64_t>(orig.GetDim(i));
    }
    uint64_t axisLen = static_cast<uint64_t>(orig.GetDim(axis));
    uint64_t innerLen = 1;
    for (int64_t i = axis + 1; i < rank; ++i) {
        innerLen *= static_cast<uint64_t>(orig.GetDim(i));
    }

    const uint64_t totalOutLength = outerLen * innerLen;

    uint32_t maxTile = UB_BUDGET_BYTES / BYTES_PER_TILE_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOutLength < tileLen) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(totalOutLength), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(totalOutLength);
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutLength(static_cast<uint32_t>(totalOutLength));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_outerLen(static_cast<uint32_t>(outerLen));
    tiling.set_innerLen(static_cast<uint32_t>(innerLen));
    tiling.set_axisLen(static_cast<uint32_t>(axisLen));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::OriginShape &orig = x_shape->GetOriginShape();
    const int64_t rank = orig.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);

    int64_t axis = *dimAttr;
    if (axis < 0) axis += rank;
    if (axis < 0 || axis >= rank) return GRAPH_FAILED;

    std::vector<int64_t> outDims;
    outDims.reserve(rank > 0 ? (rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == axis) continue;
        outDims.push_back(orig.GetDim(i));
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(static_cast<int64_t>(outDims.size()));
    for (size_t i = 0; i < outDims.size(); ++i) {
        y_shape->SetDim(static_cast<int64_t>(i), outDims[i]);
    }
    return GRAPH_SUCCESS;
}

{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutLength,
                                uint32_t tileLength,
                                uint32_t outerLen,
                                uint32_t innerLen,
                                uint32_t axisLen)
    {
        totalOutLen_ = totalOutLength;
        tileLen_ = tileLength;
        outerLen_ = outerLen;
        innerLen_ = innerLen;
        axisLen_ = axisLen;

        const uint32_t totalInLen = outerLen_ * axisLen_ * innerLen_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInLen);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalOutLen_);

        pipe_.InitBuffer(inQueueCur_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(calcFQueue_, 1, tileLen_ * sizeof(float));      // max buffer
        pipe_.InitBuffer(calcI32Queue_, 1, tileLen_ * sizeof(int32_t));  // arg index buffer
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(int64_t));
        pipe_.InitBuffer(tmpMaskQueue_, 1, tileLen_ * sizeof(uint8_t));
        pipe_.InitBuffer(tmpI32Queue_, 1, tileLen_ * sizeof(int32_t));   // for j fill
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0 || axisLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t outBase = blockIdx * tileLen_;
        if (outBase >= totalOutLen_) return;

        uint32_t curLen = totalOutLen_ - outBase;
        if (curLen > tileLen_) curLen = tileLen_;

        // Initialize max and arg with slice j=0
        AscendC::LocalTensor<float> maxLocal = calcFQueue_.AllocTensor<float>();
        CopySliceToLocal(maxLocal, outBase, curLen, 0);
        AscendC::LocalTensor<int32_t> argLocal32 = calcI32Queue_.AllocTensor<int32_t>();
        AscendC::Dup(argLocal32, (int32_t)0, curLen);

        // Reduction across axis
        for (uint32_t j = 1; j < axisLen_; ++j) {
            AscendC::LocalTensor<float> curLocal = inQueueCur_.AllocTensor<float>();
            CopySliceToLocal(curLocal, outBase, curLen, j);

            AscendC::LocalTensor<uint8_t> maskLocal = tmpMaskQueue_.AllocTensor<uint8_t>();
            AscendC::Greater(maskLocal, curLocal, maxLocal, curLen);

            // Update max with select
            AscendC::Select(maxLocal, curLocal, maxLocal, maskLocal, curLen);

            AscendC::LocalTensor<int32_t> jLocal = tmpI32Queue_.AllocTensor<int32_t>();
            AscendC::Dup(jLocal, (int32_t)j, curLen);
            AscendC::Select(argLocal32, jLocal, argLocal32, maskLocal, curLen);

            tmpI32Queue_.FreeTensor(jLocal);
            tmpMaskQueue_.FreeTensor(maskLocal);
            inQueueCur_.FreeTensor(curLocal);
        }

        // Cast indices to int64 and write out
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.AllocTensor<int64_t>();
        AscendC::Cast(yLocal, argLocal32, curLen);
        outQueueY_.EnQue(yLocal);
        CopyOut(outBase, curLen);
        outQueueY_.FreeTensor(yLocal);

        calcI32Queue_.FreeTensor(argLocal32);
        calcFQueue_.FreeTensor(maxLocal);

        // Grid stride loop
        while (true) {
            const uint32_t nextBase = outBase + strideElems;
            if (nextBase >= totalOutLen_) break;

            outBase = nextBase;
            curLen = totalOutLen_ - outBase;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> maxLocal2 = calcFQueue_.AllocTensor<float>();
            CopySliceToLocal(maxLocal2, outBase, curLen, 0);
            AscendC::LocalTensor<int32_t> argLocal32_2 = calcI32Queue_.AllocTensor<int32_t>();
            AscendC::Dup(argLocal32_2, (int32_t)0, curLen);

            for (uint32_t j = 1; j < axisLen_; ++j) {
                AscendC::LocalTensor<float> curLocal = inQueueCur_.AllocTensor<float>();
                CopySliceToLocal(curLocal, outBase, curLen, j);

                AscendC::LocalTensor<uint8_t> maskLocal = tmpMaskQueue_.AllocTensor<uint8_t>();
                AscendC::Greater(maskLocal, curLocal, maxLocal2, curLen);

                AscendC::Select(maxLocal2, curLocal, maxLocal2, maskLocal, curLen);

                AscendC::LocalTensor<int32_t> jLocal = tmpI32Queue_.AllocTensor<int32_t>();
                AscendC::Dup(jLocal, (int32_t)j, curLen);
                AscendC::Select(argLocal32_2, jLocal, argLocal32_2, maskLocal, curLen);

                tmpI32Queue_.FreeTensor(jLocal);
                tmpMaskQueue_.FreeTensor(maskLocal);
                inQueueCur_.FreeTensor(curLocal);
            }

            AscendC::LocalTensor<int64_t> yLocal2 = outQueueY_.AllocTensor<int64_t>();
            AscendC::Cast(yLocal2, argLocal32_2, curLen);
            outQueueY_.EnQue(yLocal2);
            CopyOut(outBase, curLen);
            outQueueY_.FreeTensor(yLocal2);

            calcI32Queue_.FreeTensor(argLocal32_2);
            calcFQueue_.FreeTensor(maxLocal2);
        }
    }

private:
    __aicore__ inline void CopyOut(uint32_t gmOffsetOut, uint32_t len)
    {
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.DeQue<int64_t>();
        AscendC::DataCopy(yGm_[gmOffsetOut], yLocal, len);
    }

    __aicore__ inline void CopySliceToLocal(AscendC::LocalTensor<float>& dst, uint32_t outBase, uint32_t len, uint32_t j)
    {
        uint32_t pos = 0;
        while (pos < len) {
            const uint32_t outIdx = outBase + pos;
            const uint32_t innerOffset = outIdx % innerLen_;
            const uint32_t outerIdx = outIdx / innerLen_;
            uint32_t runLen = innerLen_ - innerOffset;
            const uint32_t remaining = len - pos;
            if (runLen > remaining) runLen = remaining;

            const uint32_t gmOffset = outerIdx * axisLen_ * innerLen_ + j * innerLen_ + innerOffset;
            AscendC::DataCopy(dst[pos], xGm_[gmOffset], runLen);
            pos += runLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueCur_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcFQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcI32Queue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMaskQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpI32Queue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalOutLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t outerLen_ {0};
    uint32_t innerLen_ {0};
    uint32_t axisLen_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalOutLength,
            tiling_data.tileLength,
            tiling_data.outerLen,
            tiling_data.innerLen,
            tiling_data.axisLen);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline std::vector<int64_t> compute_output_sizes(const at::Tensor& self, int64_t dim) {
    const int64_t rank = self.dim();
    if (dim < 0) dim += rank;
    TORCH_CHECK(dim >= 0 && dim < rank, "argmax_over_a_dimension_custom: dim out of range");
    std::vector<int64_t> sizes;
    sizes.reserve(rank > 0 ? (rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        sizes.push_back(self.size(i));
    }
    return sizes;
}

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    auto out_sizes = compute_output_sizes(self, dim);
    at::Tensor result = at::empty(out_sizes, self.options().dtype(at::kLong));
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
