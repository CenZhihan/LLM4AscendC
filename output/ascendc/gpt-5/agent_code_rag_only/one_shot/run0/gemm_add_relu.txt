#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    const auto &x_origin = x_shape->GetOriginShape();
    const auto &w_origin = w_shape->GetOriginShape();
    (void)b_shape; // bias only validates length N

    // x: [M, K], w: [N, K] (as in torch.nn.functional.linear)
    uint32_t M = static_cast<uint32_t>(x_origin.GetDim(0));
    uint32_t K = static_cast<uint32_t>(x_origin.GetDim(1));
    uint32_t N = static_cast<uint32_t>(w_origin.GetDim(0));
    // Optionally ensure K matches
    // uint32_t K_w = static_cast<uint32_t>(w_origin.GetDim(1));

    // Simple block partition along M dimension
    uint32_t blockDim = (M < 24u) ? M : 24u;
    if (blockDim == 0u) blockDim = 1u;
    context->SetBlockDim(blockDim);
    uint32_t rowsPerBlock = CeilDiv(M, blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const auto &xo = x_shape->GetOriginShape();
    const auto &wo = w_shape->GetOriginShape();

    // x: [M, K], w: [N, K] -> y: [M, N]
    int64_t M = xo.GetDim(0);
    int64_t N = wo.GetDim(0);

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, M);
    y_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

class KernelGemmAddReluCustom {
public:
    __aicore__ inline KernelGemmAddReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t rowsPerBlock)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        rowsPerBlock_ = rowsPerBlock;

        xGm_ = reinterpret_cast<__gm__ float*>(x);
        wGm_ = reinterpret_cast<__gm__ float*>(w);
        bGm_ = reinterpret_cast<__gm__ float*>(bias);
        yGm_ = reinterpret_cast<__gm__ float*>(y);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) {
            return;
        }

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t rowStart = blockIdx * rowsPerBlock_;
        if (rowStart >= M_) return;
        uint32_t rowEnd = rowStart + rowsPerBlock_;
        if (rowEnd > M_) rowEnd = M_;

        // Compute Y[row, :] = ReLU( X[row, :] @ W^T + bias )
        for (uint32_t m = rowStart; m < rowEnd; ++m) {
            const uint32_t xRowBase = m * K_;
            const uint32_t yRowBase = m * N_;

            for (uint32_t n = 0; n < N_; ++n) {
                float acc = bGm_[n];
                const uint32_t wRowBase = n * K_;
                // Dot product over K
                for (uint32_t k = 0; k < K_; ++k) {
                    float xv = xGm_[xRowBase + k];
                    float wv = wGm_[wRowBase + k];
                    acc = AscendC::FusedMulAdd(xv, wv, acc); // acc += xv * wv
                }
                // ReLU
                if (acc < 0.0f) acc = 0.0f;
                yGm_[yRowBase + n] = acc;
            }
        }
    }

private:
    __gm__ float *xGm_{nullptr};
    __gm__ float *wGm_{nullptr};
    __gm__ float *bGm_{nullptr};
    __gm__ float *yGm_{nullptr};

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t rowsPerBlock_{1};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddReluCustom op;
    op.Init(x, w, bias, y, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.rowsPerBlock);
    op.Process();
}
