#include "matmul_with_transposed_b_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

// UB budget per core (bytes). Keep conservative headroom for queues/metadata.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

// We operate on float (4 bytes)
static constexpr uint32_t ELEM_SIZE = 4;

// Select tiling with the following UB usage model per core (floats):
//   B_tile (transposed): tileN * tileK
//   y_row vector: tileN
//   tmp vector: tileN
//   a_row scalar buffer (len tileK): tileK
// Total floats ~ tileN*tileK + 2*tileN + tileK
// Keep within ~85% of UB budget to be safe.
static void ChooseTiles(uint32_t M, uint32_t N, uint32_t K,
                        uint32_t &tileM, uint32_t &tileN, uint32_t &tileK)
{
    // Initial candidates
    tileM = (M >= 16) ? 16u : (M > 0 ? M : 1u);
    tileN = (N >= 256) ? 256u : (N > 0 ? N : 1u);
    tileK = (K >= 128) ? 128u : (K > 0 ? K : 1u);

    // Align to vector-friendly sizes
    auto align = [](uint32_t x, uint32_t a) { return (x / a) * a; };
    if (tileN >= 16) tileN = align(tileN, 16);
    if (tileK >= 16) tileK = align(tileK, 16);

    const uint32_t budgetFloats = static_cast<uint32_t>(UB_BUDGET_BYTES / ELEM_SIZE * 85 / 100);
    auto within = [&](uint32_t tN, uint32_t tK) {
        uint64_t use = static_cast<uint64_t>(tN) * tK + 2ull * tN + tK;
        return use <= budgetFloats;
    };

    // Shrink to fit budget
    while (!within(tileN, tileK)) {
        if (tileN >= tileK && tileN > 16) tileN = AlignDown(tileN / 2u, 16);
        else if (tileK > 16) tileK = AlignDown(tileK / 2u, 16);
        else break;
        if (tileN == 0) tileN = 16;
        if (tileK == 0) tileK = 16;
    }

    // Ensure not exceeding actual dimensions
    if (tileM > M) tileM = M;
    if (tileN > N) tileN = N;
    if (tileK > K) tileK = (K >= 16) ? AlignDown(K, 16) : K;
    if (tileM == 0) tileM = 1;
    if (tileN == 0) tileN = 1;
    if (tileK == 0) tileK = 1;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedBCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    // A: (M,K), B: (N,K) -> C: (M,N) with B transposed inside kernel
    const uint32_t M = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(0));
    // Optional: sanity K matches B's K
    (void)N;

    uint32_t tileM, tileN, tileK;
    ChooseTiles(M, N, K, tileM, tileN, tileK);

    // Determine blockDim from number of tiles in (M,N) grid
    const uint32_t tilesM = (tileM == 0) ? 1u : CeilDiv(M, tileM);
    const uint32_t tilesN = (tileN == 0) ? 1u : CeilDiv(N, tileN);
    uint32_t totalTiles = tilesM * tilesN;
    if (totalTiles == 0) totalTiles = 1;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulWithTransposedBCustom {
public:
    __aicore__ inline KernelMatmulWithTransposedBCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN; tileK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, (uint64_t)M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, (uint64_t)N_ * K_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, (uint64_t)M_ * N_);

        // Buffers:
        // - bTile (transposed): [tileK x tileN] contiguous
        // - yRow vector: [tileN]
        // - tmp vector: [tileN]
        // - aRow chunk: [tileK]
        pipe_.InitBuffer(bTileQueue_, BUFFER_NUM, tileN_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(yRowQueue_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQueue_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(aRowQueue_, BUFFER_NUM, tileK_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t tilesM = (tileM_ == 0) ? 1u : CeilDiv(M_, tileM_);
        const uint32_t tilesN = (tileN_ == 0) ? 1u : CeilDiv(N_, tileN_);
        const uint32_t totalTiles = tilesM * tilesN;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t tm = tileId / tilesN;
            const uint32_t tn = tileId % tilesN;

            const uint32_t m0 = tm * tileM_;
            const uint32_t n0 = tn * tileN_;
            const uint32_t curM = Min(tileM_, M_ - m0);
            const uint32_t curN = Min(tileN_, N_ - n0);
            ComputeTile(m0, n0, curM, curN);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) const { return (a + b - 1u) / b; }
    __aicore__ inline uint32_t Min(uint32_t a, uint32_t b) const { return a < b ? a : b; }

    // Load a transposed tile of B: GM[B(n0:n0+curN, k0:k0+curK)] -> UB bTile [curK x curN] row-major
    __aicore__ inline void LoadBTileTransposed(uint32_t n0, uint32_t k0, uint32_t curN, uint32_t curK)
    {
        AscendC::LocalTensor<float> bTile = bTileQueue_.AllocTensor<float>();
        // Fill bTile by element (kk, jj) mapping to GM[(n0+jj)*K_ + (k0+kk)]
        for (uint32_t kk = 0; kk < curK; ++kk) {
            for (uint32_t jj = 0; jj < curN; ++jj) {
                const uint64_t gmOff = (static_cast<uint64_t>(n0 + jj) * K_) + (k0 + kk);
                AscendC::DataCopy(bTile[kk * curN + jj], bGm_[gmOff], 1);
            }
        }
        bTileQueue_.EnQue(bTile);
    }

    __aicore__ inline void ComputeTile(uint32_t m0, uint32_t n0, uint32_t curM, uint32_t curN)
    {
        // For each output row in this tile
        for (uint32_t ii = 0; ii < curM; ++ii) {
            AscendC::LocalTensor<float> yRow = yRowQueue_.AllocTensor<float>();
            AscendC::Dup(yRow, 0.0f, curN);

            // Iterate over K in chunks
            for (uint32_t k0 = 0; k0 < K_; k0 += tileK_) {
                const uint32_t curK = Min(tileK_, K_ - k0);

                // Load B tile transposed for this (n0, k0) window
                LoadBTileTransposed(n0, k0, curN, curK);
                AscendC::LocalTensor<float> bTile = bTileQueue_.DeQue<float>();

                // Load A row chunk for this row: A[m0+ii, k0:k0+curK]
                AscendC::LocalTensor<float> aRow = aRowQueue_.AllocTensor<float>();
                const uint64_t aOff = (static_cast<uint64_t>(m0 + ii) * K_) + k0;
                AscendC::DataCopy(aRow, aGm_[aOff], curK);

                // Accumulate yRow += sum_{kk} aRow[kk] * bTile_row(kk, :)
                for (uint32_t kk = 0; kk < curK; ++kk) {
                    float aScalar = AscendC::GetValue(aRow, kk);
                    AscendC::LocalTensor<float> tmp = tmpVecQueue_.AllocTensor<float>();
                    AscendC::Muls(tmp, bTile[kk * curN], aScalar, curN);
                    AscendC::Add(yRow, yRow, tmp, curN);
                    tmpVecQueue_.FreeTensor(tmp);
                }

                aRowQueue_.FreeTensor(aRow);
                bTileQueue_.FreeTensor(bTile);
            }

            // Store yRow to GM: C[m0+ii, n0:n0+curN]
            const uint64_t cOff = (static_cast<uint64_t>(m0 + ii) * N_) + n0;
            AscendC::DataCopy(cGm_[cOff], yRow, curN);
            yRowQueue_.FreeTensor(yRow);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bTileQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpVecQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> aRowQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileM_{1}, tileN_{1}, tileK_{1};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_b_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedBCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
