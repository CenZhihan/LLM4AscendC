namespace {
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MIN_BLOCKS = 8;
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwiseSeparable2dCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    if (x_shape->GetDimNum() != 4) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t C = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const gert::Shape* pw_shape = context->GetInputShape(2);
    const uint32_t Cout = static_cast<uint32_t>(pw_shape->GetOriginShape().GetDim(0)); // [Cout, Cin, 1, 1]

    // UB usage per tile approximately:
    //   acc: Cout * tileW
    //   tmp: 1 * tileW
    //   row buffers: 3 * (tileW + 2)
    // total floats ~ (Cout + 4) * tileW + 6
    // bytes ~ 4 * ((Cout + 4) * tileW + 6)
    uint32_t maxTileW = UB_BUDGET_BYTES / (4u * (Cout + 4u));
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;

    if (maxTileW > W) {
        uint32_t aligned = AlignDown(W, ALIGN_ELEMS);
        if (aligned > 0) maxTileW = aligned;
        else maxTileW = W;
    }

    // Conservative blockDim heuristic
    uint32_t tilesPerRow = CeilDiv(W, maxTileW);
    uint32_t workUnits = N * H * tilesPerRow;
    uint32_t blockDim = workUnits;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cIn(C);
    tiling.set_cOut(Cout);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_tileW(maxTileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwiseSeparable2dCustom {
public:
    __aicore__ inline KernelConvDepthwiseSeparable2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wdw, GM_ADDR wpw, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t h, uint32_t w, uint32_t tileW)
    {
        n_ = n;
        cIn_ = cIn;
        cOut_ = cOut;
        h_ = h;
        w_ = w;
        tileW_ = tileW;
        accTileElems_ = cOut_ * tileW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * cIn_ * h_ * w_);
        wdwGm_.SetGlobalBuffer((__gm__ float*)wdw, (uint64_t)cIn_ * 1 * 3 * 3);
        wpwGm_.SetGlobalBuffer((__gm__ float*)wpw, (uint64_t)cOut_ * cIn_ * 1 * 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * cOut_ * h_ * w_);

        // UB buffers
        // acc buffer: cOut * tileW
        pipe_.InitBuffer(accQueue_, 1, accTileElems_ * sizeof(float));
        // tmp vec for depthwise result per channel: tileW
        pipe_.InitBuffer(tmpQueue_, 2, tileW_ * sizeof(float));
        // row buffers: 3 * (tileW + 2)
        pipe_.InitBuffer(rowQueue_, 3, (tileW_ + 2) * sizeof(float));
        // small buffer for 3x3 dw weights (9 floats)
        pipe_.InitBuffer(dw9Queue_, 1, 9 * sizeof(float));
        // pw vector for one input channel: cOut
        pipe_.InitBuffer(pwVecQueue_, 1, cOut_ * sizeof(float));
        // helper multiply buffer: tileW
        pipe_.InitBuffer(mulQueue_, 1, tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cIn_ == 0 || cOut_ == 0 || h_ == 0 || w_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        // We iterate over a flattened work list of tiles: (n, h_row, col_tile)
        const uint32_t tilesPerRow = CeilDiv(w_, tileW_);
        const uint32_t rowsTotal = n_ * h_;
        const uint32_t workTotal = rowsTotal * tilesPerRow;

        for (uint32_t workId = blockIdx; workId < workTotal; workId += blockNum) {
            // decode workId
            uint32_t rowId = workId / tilesPerRow;      // [0, n*h)
            uint32_t tileId = workId % tilesPerRow;     // [0, tilesPerRow)
            uint32_t nIdx = rowId / h_;
            uint32_t hIdx = rowId % h_;
            uint32_t wBase = tileId * tileW_;
            uint32_t curW = w_ - wBase;
            if (curW > tileW_) curW = tileW_;

            ComputeTile(nIdx, hIdx, wBase, curW);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) const
    {
        return (a + b - 1u) / b;
    }

    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * cIn_ + c) * h_ + h) * w_ + w;
    }

    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * cOut_ + c) * h_ + h) * w_ + w;
    }

    __aicore__ inline void LoadRow(LocalTensor<float>& rowBuf, uint32_t nIdx, uint32_t cIdx, int32_t hRow, uint32_t wBase, uint32_t curW)
    {
        // rowBuf length: tileW_ + 2
        // Fill with zeros if hRow out of bounds
        if (hRow < 0 || hRow >= static_cast<int32_t>(h_)) {
            Duplicate(rowBuf, 0.0f, curW + 2);
            return;
        }

        // Central region copy with halos
        bool hasLeft = (wBase > 0);
        bool hasRight = (wBase + curW < w_);

        if (hasLeft && hasRight) {
            DataCopy(rowBuf, xGm_[XOffset(nIdx, cIdx, static_cast<uint32_t>(hRow), wBase - 1)], curW + 2);
        } else if (!hasLeft && hasRight) {
            // left pad
            Duplicate(rowBuf, 0.0f, 1);
            DataCopy(rowBuf[1], xGm_[XOffset(nIdx, cIdx, static_cast<uint32_t>(hRow), wBase)], curW + 1);
        } else if (hasLeft && !hasRight) {
            // right pad
            DataCopy(rowBuf, xGm_[XOffset(nIdx, cIdx, static_cast<uint32_t>(hRow), (wBase == 0) ? 0 : (wBase - 1))], curW + 1);
            Duplicate(rowBuf[curW + 1], 0.0f, 1);
        } else {
            // both edges: tile == full row and W == curW
            if (w_ == curW) {
                Duplicate(rowBuf, 0.0f, 1);
                DataCopy(rowBuf[1], xGm_[XOffset(nIdx, cIdx, static_cast<uint32_t>(hRow), 0)], curW);
                Duplicate(rowBuf[1 + curW], 0.0f, 1);
            } else {
                // fallback
                Duplicate(rowBuf, 0.0f, curW + 2);
            }
        }
    }

    __aicore__ inline void Depthwise3x3(LocalTensor<float>& outTmp, LocalTensor<float>& row0, LocalTensor<float>& row1, LocalTensor<float>& row2,
                                        LocalTensor<float>& mulBuf, float k00, float k01, float k02, float k10, float k11, float k12, float k20, float k21, float k22,
                                        uint32_t curW)
    {
        // outTmp = 0
        Duplicate(outTmp, 0.0f, curW);

        // row0 neighbors
        Muls(mulBuf, row0[0], k00, curW);
        Add(outTmp, outTmp, mulBuf, curW);
        Muls(mulBuf, row0[1], k01, curW);
        Add(outTmp, outTmp, mulBuf, curW);
        Muls(mulBuf, row0[2], k02, curW);
        Add(outTmp, outTmp, mulBuf, curW);

        // row1 neighbors
        Muls(mulBuf, row1[0], k10, curW);
        Add(outTmp, outTmp, mulBuf, curW);
        Muls(mulBuf, row1[1], k11, curW);
        Add(outTmp, outTmp, mulBuf, curW);
        Muls(mulBuf, row1[2], k12, curW);
        Add(outTmp, outTmp, mulBuf, curW);

        // row2 neighbors
        Muls(mulBuf, row2[0], k20, curW);
        Add(outTmp, outTmp, mulBuf, curW);
        Muls(mulBuf, row2[1], k21, curW);
        Add(outTmp, outTmp, mulBuf, curW);
        Muls(mulBuf, row2[2], k22, curW);
        Add(outTmp, outTmp, mulBuf, curW);
    }

    __aicore__ inline void ComputeTile(uint32_t nIdx, uint32_t hIdx, uint32_t wBase, uint32_t curW)
    {
        // Allocate UB buffers
        LocalTensor<float> acc = accQueue_.AllocTensor<float>();
        // zero accumulators: length curW * cOut_
        if (curW * cOut_ > 0) {
            Duplicate(acc, 0.0f, curW * cOut_);
        }

        // shared temporary buffers
        LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        LocalTensor<float> mulBuf = mulQueue_.AllocTensor<float>();

        // row buffers (each length tileW_ + 2)
        LocalTensor<float> row0 = rowQueue_.AllocTensor<float>();
        LocalTensor<float> row1 = rowQueue_.AllocTensor<float>();
        LocalTensor<float> row2 = rowQueue_.AllocTensor<float>();

        // depthwise 3x3 weights local (9 floats)
        LocalTensor<float> dw9 = dw9Queue_.AllocTensor<float>();

        // pointwise weights for one input channel (length cOut_)
        LocalTensor<float> pwVec = pwVecQueue_.AllocTensor<float>();

        for (uint32_t c = 0; c < cIn_; ++c) {
            // Load dw kernel weights for this channel: layout [C,1,3,3] contiguous as 9 floats per c
            DataCopy(dw9, wdwGm_[static_cast<uint64_t>(c) * 9u], 9);

            // Extract them into scalars
            float k00 = dw9.GetValue(0);
            float k01 = dw9.GetValue(1);
            float k02 = dw9.GetValue(2);
            float k10 = dw9.GetValue(3);
            float k11 = dw9.GetValue(4);
            float k12 = dw9.GetValue(5);
            float k20 = dw9.GetValue(6);
            float k21 = dw9.GetValue(7);
            float k22 = dw9.GetValue(8);

            // Load 3 rows with padding
            LoadRow(row0, nIdx, c, static_cast<int32_t>(hIdx) - 1, wBase, curW);
            LoadRow(row1, nIdx, c, static_cast<int32_t>(hIdx),     wBase, curW);
            LoadRow(row2, nIdx, c, static_cast<int32_t>(hIdx) + 1, wBase, curW);

            // Compute depthwise conv into tmp[curW]
            Depthwise3x3(tmp, row0, row1, row2, mulBuf, k00, k01, k02, k10, k11, k12, k20, k21, k22, curW);

            // Load pw weights for this input channel (vector length cOut_)
            // Layout [Cout, Cin, 1, 1]: contiguous Cin changes fastest for each Cout
            // Offset: c + Cin * oc for each oc; We want vector [oc=0..Cout-1] => stride Cin
            // To simplify, we arrange the stored format is [Cout, Cin, 1, 1] contiguous on Cout-major with Cin-minor.
            // Thus the contiguous vector per channel c is at offset cOut_*c if layout is [Cin-major]? If layout is [Cout, Cin], memory is Cout*Cin with Cout fastest? In NCHW, the first dim is fastest? Here we assume standard [Cout, Cin, 1, 1] contiguous with Cin as the second dimension (Cin minor stride = 1).
            // So weights for fixed c are at base offset = c, stride = Cin across Cout. We gather by scalar loads.
            // For simplicity, we DataCopy cOut scalars one-by-one.
            for (uint32_t oc = 0; oc < cOut_; ++oc) {
                // gather weight[oc, c, 0, 0] => offset oc * cIn_ + c
                uint64_t pwOff = static_cast<uint64_t>(oc) * cIn_ + c;
                // load 1 float into first element of pwVec (reuse buffer)
                DataCopy(pwVec, wpwGm_[pwOff], 1);
                float alpha = pwVec.GetValue(0);

                // acc row pointer
                LocalTensor<float> accRow = acc[oc * tileW_];
                // y += tmp * alpha
                Muls(mulBuf, tmp, alpha, curW);
                Add(accRow, accRow, mulBuf, curW);
            }
        }

        // Write out acc to Y
        for (uint32_t oc = 0; oc < cOut_; ++oc) {
            LocalTensor<float> accRow = acc[oc * tileW_];
            DataCopy(yGm_[YOffset(nIdx, oc, hIdx, wBase)], accRow, curW);
        }

        // Free UB tensors
        pwVecQueue_.FreeTensor(pwVec);
        dw9Queue_.FreeTensor(dw9);
        rowQueue_.FreeTensor(row2);
        rowQueue_.FreeTensor(row1);
        rowQueue_.FreeTensor(row0);
        mulQueue_.FreeTensor(mulBuf);
        tmpQueue_.FreeTensor(tmp);
        accQueue_.FreeTensor(acc);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, 1> dw9Queue_;
    TQue<TPosition::VECCALC, 1> mulQueue_;
    TQue<TPosition::VECCALC, 2> tmpQueue_;
    TQue<TPosition::VECCALC, 3> rowQueue_;
    TQue<TPosition::VECCALC, 1> pwVecQueue_;
    TQue<TPosition::VECOUT, 1> accQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wdwGm_;
    GlobalTensor<float> wpwGm_;
    GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, cOut_{0}, h_{0}, w_{0};
    uint32_t tileW_{0};
    uint32_t accTileElems_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_separable_2d_custom(GM_ADDR x, GM_ADDR w_depthwise, GM_ADDR w_pointwise, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwiseSeparable2dCustom op;
    op.Init(x, w_depthwise, w_pointwise, y,
            tiling_data.n, tiling_data.cIn, tiling_data.cOut,
            tiling_data.h, tiling_data.w, tiling_data.tileW);
    op.Process();
}
