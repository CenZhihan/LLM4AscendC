#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 2;

class KernelConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalLen, uint32_t tileLen)
    {
        totalLen_ = totalLen;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, 0); // not used in this skeleton
        wGm_.SetGlobalBuffer((__gm__ float*)w, 0); // not used in this skeleton
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        while (true) {
            ComputeAndWriteZeros(base, curLen);

            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalLen_) break;

            base = nextBase;
            curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
        }
    }

private:
    __aicore__ inline void ComputeAndWriteZeros(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Dup(yLocal, 0.0f, len);
        outQueueY_.EnQue<float>(yLocal);

        AscendC::LocalTensor<float> yLocalDeq = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocalDeq, len);
        outQueueY_.FreeTensor(yLocalDeq);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_strided_padded_grouped_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
