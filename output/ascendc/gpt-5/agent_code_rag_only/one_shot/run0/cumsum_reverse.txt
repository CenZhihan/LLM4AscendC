host_operator_src="""

#include "cumsum_reverse_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

} // namespace

namespace optiling {

// UB model per core for this kernel:
//   inQueue: 2 * tile
//   outQueue: 2 * tile
// Total floats ~ 4 * tile => bytes = 16 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumReverseCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const size_t dimNum = in_shape->GetOriginShape().GetDimNum();

    // Fetch 'dim' attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    if (dimAttr == nullptr) {
        return ge::GRAPH_FAILED;
    }
    int64_t dim = *dimAttr;
    if (dim < 0) dim += static_cast<int64_t>(dimNum);
    if (dim < 0 || dim >= static_cast<int64_t>(dimNum)) {
        return ge::GRAPH_FAILED;
    }

    // Compute sizes
    uint64_t totalLength = in_shape->GetOriginShape().GetShapeSize();
    uint64_t rowLength = static_cast<uint64_t>(in_shape->GetOriginShape().GetDim(dim));
    uint64_t rows = (rowLength == 0) ? 0 : (totalLength / rowLength);

    // Choose tile length along the cumsum axis
    uint32_t maxTile = UB_BUDGET_BYTES / 16;  // elements
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = static_cast<uint32_t>(rowLength);
    if (tileLen > maxTile) {
        tileLen = maxTile;
    } else {
        // try align for better DMA if possible
        uint32_t aligned = AlignDown(static_cast<uint32_t>(rowLength), ALIGN_ELEMS);
        if (aligned > 0) tileLen = aligned;
    }

    // BlockDim based on rows
    uint32_t blockDim = (rows == 0) ? 1u : ((rows > 24u) ? 24u : static_cast<uint32_t>(rows));
    if (blockDim == 0) blockDim = 1u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_rowLength(static_cast<uint32_t>(rowLength));
    tiling.set_rows(static_cast<uint32_t>(rows));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumsumReverseCustom : public OpDef {
public:
    explicit CumsumReverseCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int();

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumsumReverseCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCumsumReverseCustom {
public:
    __aicore__ inline KernelCumsumReverseCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t rowLength,
                                uint32_t rows,
                                uint32_t tileLength)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        rows_ = rows;
        tileLen_ = (tileLength == 0) ? rowLen_ : tileLength;
        if (tileLen_ > rowLen_) tileLen_ = rowLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || rows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t r = blockIdx; r < rows_; r += blockNum) {
            // Each row is contiguous segment of length rowLen_.
            const uint32_t rowBase = r * rowLen_;

            // Process from right to left (reverse cumsum)
            float carry = 0.0f;
            uint32_t processed = 0;

            while (processed < rowLen_) {
                // Determine length of the next tile taken from the right
                uint32_t remain = rowLen_ - processed;
                uint32_t curLen = (remain > tileLen_) ? tileLen_ : remain;

                // GM offset for the current tile start (left index of the tile)
                // We take tile [start : start + curLen) where start = rowLen_ - processed - curLen
                const uint32_t start = rowLen_ - processed - curLen;

                CopyIn(rowBase + start, curLen);

                // Prefetching pattern: could overlap, but keep simple within row loop
                Compute(curLen, carry);

                CopyOut(rowBase + start, curLen);

                processed += curLen;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueue_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len, float &carryInOut)
    {
        AscendC::LocalTensor<float> xLocal = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

        // Reverse inclusive scan within this tile, accumulating with carry
        // y[i] = x[i] + sum_{k=i+1..endOfRow} x[k]
        float acc = carryInOut;
        // Iterate from right to left within the tile
        for (int32_t i = static_cast<int32_t>(len) - 1; i >= 0; --i) {
            // Direct element-wise access in local buffer
            float xv = xLocal.GetValue(i);
            acc = acc + xv;
            yLocal.SetValue(i, acc);
        }
        carryInOut = acc;

        outQueue_.EnQue<float>(yLocal);
        inQueue_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t rows_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverseCustom op;
    op.Init(x,
            y,
            tiling_data.totalLength,
            tiling_data.rowLength,
            tiling_data.rows,
            tiling_data.tileLength);
    op.Process();
}
"""
