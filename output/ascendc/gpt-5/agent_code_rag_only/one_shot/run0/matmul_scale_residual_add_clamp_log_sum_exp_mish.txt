#include "kernel_operator.h"

constexpr int32_t QNUM = 1;

class KernelMatmulScaleResidualAddClampLogSumExpMishCustom {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t kTile, uint32_t nTile,
                                float scaling, float clampMin, float clampMax)
    {
        M_ = M; N_ = N; K_ = K;
        kTile_ = (kTile == 0) ? K : kTile;
        nTile_ = (nTile == 0) ? 1 : nTile;
        scaleFactor_ = 1.0f + scaling;
        clampMin_ = clampMin;
        clampMax_ = clampMax;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M_); // [M,1] flattened

        // Initialize UB buffers
        pipe_.InitBuffer(aQ_, QNUM, kTile_ * sizeof(float));
        pipe_.InitBuffer(wQ_, QNUM, kTile_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, QNUM, kTile_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, QNUM, 16 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each core processes rows with grid-stride loop
        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            float lse_val = RowComputeLSE(m);
            // out = lse * mish(lse) = lse * (lse * tanh(ln(1 + exp(lse))))
            float e = fast_exp(lse_val);
            float sp = fast_log1p(e);
            float t = fast_tanh(sp);
            float out = lse_val * (lse_val * t);
            StoreOutput(m, out);
        }
    }

private:
    // Fast scalar math helpers using device-friendly approximations via vector ops (1 element)
    __aicore__ inline float fast_exp(float x)
    {
        AscendC::LocalTensor<float> vx = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> vy = scalarQ_.AllocTensor<float>();
        // duplicate scalar into 1-len tensor and compute
        vx.SetValue(0, x);
        AscendC::Exp(vy, vx, 1);
        float r = vy.GetValue(0);
        scalarQ_.FreeTensor(vy);
        scalarQ_.FreeTensor(vx);
        return r;
    }

    __aicore__ inline float fast_log(float x)
    {
        AscendC::LocalTensor<float> vx = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> vy = scalarQ_.AllocTensor<float>();
        vx.SetValue(0, x);
        AscendC::Ln(vy, vx, 1);
        float r = vy.GetValue(0);
        scalarQ_.FreeTensor(vy);
        scalarQ_.FreeTensor(vx);
        return r;
    }

    __aicore__ inline float fast_log1p(float x)
    {
        AscendC::LocalTensor<float> vx = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> vt = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> vy = scalarQ_.AllocTensor<float>();
        vx.SetValue(0, x);
        AscendC::Adds(vt, vx, 1.0f, 1);
        AscendC::Ln(vy, vt, 1);
        float r = vy.GetValue(0);
        scalarQ_.FreeTensor(vy);
        scalarQ_.FreeTensor(vt);
        scalarQ_.FreeTensor(vx);
        return r;
    }

    __aicore__ inline float fast_tanh(float x)
    {
        AscendC::LocalTensor<float> vx = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> vy = scalarQ_.AllocTensor<float>();
        vx.SetValue(0, x);
        AscendC::Tanh(vy, vx, 1);
        float r = vy.GetValue(0);
        scalarQ_.FreeTensor(vy);
        scalarQ_.FreeTensor(vx);
        return r;
    }

    __aicore__ inline float RowComputeLSE(uint32_t m)
    {
        // Streaming across N dimension, one output feature at a time, updating running LSE
        float m_run = -AscendC::INFINITY_FLOAT;  // running max
        float s_run = 0.0f;                      // running exp-sum in max-centered domain
        bool first = true;

        for (uint32_t n = 0; n < N_; ++n) {
            float dot = DotRowCol(m, n);
            // y = (x @ w^T + b) * (1+scaling)
            float val = (dot + bGm_[n]) * scaleFactor_;
            // clamp
            if (val < clampMin_) val = clampMin_;
            if (val > clampMax_) val = clampMax_;
            // update running log-sum-exp state
            if (first) {
                m_run = val;
                s_run = 1.0f;
                first = false;
            } else {
                if (val > m_run) {
                    float t = fast_exp(m_run - val);
                    s_run = s_run * t + 1.0f;
                    m_run = val;
                } else {
                    float t = fast_exp(val - m_run);
                    s_run = s_run + t;
                }
            }
        }
        // final lse = log(s_run) + m_run
        float lse = fast_log(s_run) + m_run;
        return lse;
    }

    __aicore__ inline float DotRowCol(uint32_t m, uint32_t n)
    {
        // Compute dot(X[m,:], W[n,:]) by tiling K with kTile_
        float sum = 0.0f;
        for (uint32_t k0 = 0; k0 < K_; k0 += kTile_) {
            uint32_t curKt = AscendC::min(kTile_, K_ - k0);

            AscendC::LocalTensor<float> aLocal = aQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> wLocal = wQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> pLocal = tmpQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> sLocal = scalarQ_.AllocTensor<float>();

            // Load a_sub and w_sub
            const uint64_t xOff = static_cast<uint64_t>(m) * K_ + k0;
            const uint64_t wOff = static_cast<uint64_t>(n) * K_ + k0;
            AscendC::DataCopy(aLocal, xGm_[xOff], curKt);
            AscendC::DataCopy(wLocal, wGm_[wOff], curKt);

            // Multiply and reduce
            AscendC::Mul(pLocal, aLocal, wLocal, curKt);
            AscendC::ReduceSum(sLocal, pLocal, curKt);

            float part = sLocal.GetValue(0);
            sum += part;

            scalarQ_.FreeTensor(sLocal);
            tmpQ_.FreeTensor(pLocal);
            wQ_.FreeTensor(wLocal);
            aQ_.FreeTensor(aLocal);
        }
        return sum;
    }

    __aicore__ inline void StoreOutput(uint32_t m, float value)
    {
        AscendC::LocalTensor<float> vLocal = scalarQ_.AllocTensor<float>();
        vLocal.SetValue(0, value);
        AscendC::DataCopy(yGm_[m], vLocal, 1);
        scalarQ_.FreeTensor(vLocal);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> aQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> wQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;  // [M*K]
    AscendC::GlobalTensor<float> wGm_;  // [N*K]
    AscendC::GlobalTensor<float> bGm_;  // [N]
    AscendC::GlobalTensor<float> yGm_;  // [M]

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t kTile_{0}, nTile_{1};

    float scaleFactor_{1.0f};
    float clampMin_{-1e9f};
    float clampMax_{1e9f};
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScaleResidualAddClampLogSumExpMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.kTile, tiling_data.nTile,
            tiling_data.scaling, tiling_data.clampMin, tiling_data.clampMax);
    op.Process();
}
