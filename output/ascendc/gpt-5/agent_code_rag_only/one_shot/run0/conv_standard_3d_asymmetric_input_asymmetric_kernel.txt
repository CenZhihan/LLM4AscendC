namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (a == 0) ? x : (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// UB budget comment:
// We keep simple conservative memory model per tile:
//   accRow:       1 * tileLen
//   tmpVec:       1 * tileLen
//   wRow:         Kw
// Total floats ~= (tileCo not simultaneously stored)  -> 2*tileLen + Kw
// To keep room for queues and alignment, we compute tileLen from broader formula for safety:
//   floats ~= tileCo*tileLen + 2*tileLen + 2*Kw  (when software double buffering happens)
// Use UB_BUDGET_BYTES ~ 192KB
namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t MAX_TILE_CO = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard3dAsymmetricInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect shapes:
    // x: [N, C, D, H, W]
    // w: [Co, C, Kd, Kh, Kw]
    const uint32_t N  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t C  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t D  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetDim(3));
    const uint32_t W  = static_cast<uint32_t>(x_shape->GetDim(4));

    const uint32_t Co = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Cw = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t Kd = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kh = static_cast<uint32_t>(w_shape->GetDim(3));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetDim(4));

    if (C != Cw) {
        return ge::GRAPH_FAILED;
    }

    // stride = 1, padding = 0, dilation = 1
    const uint32_t Do = (D >= Kd) ? (D - Kd + 1u) : 0u;
    const uint32_t Ho = (H >= Kh) ? (H - Kh + 1u) : 0u;
    const uint32_t Wo = (W >= Kw) ? (W - Kw + 1u) : 0u;

    if (Do == 0 || Ho == 0 || Wo == 0) {
        // Degenerate, still proceed with minimal tiling to avoid div by zero
    }

    // Choose tileCo conservatively
    const uint32_t tileCo = (Co >= MAX_TILE_CO) ? MAX_TILE_CO : ((Co >= 8u) ? 8u : Co);

    // Compute tileLenW by UB budget: bytes ~ 4 * (tileCo*tileLen + 3*tileLen + 2*Kw)
    uint32_t budgetFloats = UB_BUDGET_BYTES / 4u;
    uint32_t maxTileLen = 0u;
    if (tileCo + 3u > 0u) {
        maxTileLen = (budgetFloats > (2u * Kw)) ? (budgetFloats - 2u * Kw) / (tileCo + 3u) : ALIGN_ELEMS;
    }
    if (maxTileLen < ALIGN_ELEMS) maxTileLen = ALIGN_ELEMS;
    if (Wo > 0 && maxTileLen > Wo) maxTileLen = Wo;
    maxTileLen = AlignDown(maxTileLen, ALIGN_ELEMS);

    uint32_t tileLenW = (maxTileLen == 0u) ? 1u : maxTileLen;
    if (tileLenW == 0u) tileLenW = 1u;

    const uint32_t segmentsPerLine = (Wo == 0u) ? 1u : CeilDiv(Wo, tileLenW);
    const uint64_t totalTiles64 = static_cast<uint64_t>(N) * static_cast<uint64_t>(Do) *

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_D(D);
    tiling.set_H(H);
    tiling.set_W(W);

    tiling.set_Co(Co);
    tiling.set_Kd(Kd);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);

    tiling.set_Do(Do);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);

    tiling.set_tileLenW(tileLenW);
    tiling.set_tileCo(tileCo);
    tiling.set_segmentsPerLine(segmentsPerLine);
    tiling.set_blockDim(blockDim);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t INBUF_NUM = 1;
constexpr int32_t WBUF_NUM = 1;
constexpr int32_t TMPBUF_NUM = 2;

class KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t D, uint32_t H, uint32_t W,
                                uint32_t Co, uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Do, uint32_t Ho, uint32_t Wo,
                                uint32_t tileLenW, uint32_t tileCo, uint32_t segmentsPerLine)
    {
        N_ = N; C_ = C; D_ = D; H_ = H; W_ = W;
        Co_ = Co; Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        Do_ = Do; Ho_ = Ho; Wo_ = Wo;
        tileLenW_ = tileLenW;
        tileCo_ = tileCo;
        segmentsPerLine_ = segmentsPerLine;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_*C_*D_*H_*W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Co_*C_*Kd_*Kh_*Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_*Co_*Do_*Ho_*Wo_);

        // Buffers sized by tileLenW (worst case)
        pipe_.InitBuffer(xVecQ_, INBUF_NUM, tileLenW_ * sizeof(float));
        pipe_.InitBuffer(wRowQ_, WBUF_NUM, Kw_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, TMPBUF_NUM, tileLenW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (Do_ == 0 || Ho_ == 0 || Wo_ == 0) {
            return;
        }

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint64_t tilesPerLine = (uint64_t)segmentsPerLine_;
        const uint64_t totalTiles = (uint64_t)N_ * (uint64_t)Do_ * (uint64_t)Ho_ * tilesPerLine;

        for (uint64_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            // Decompose tileId -> (lineIdx, segIdx)
            const uint64_t lineIdx = tileId / tilesPerLine;
            const uint32_t segIdx = (uint32_t)(tileId % tilesPerLine);

            const uint32_t n  = (uint32_t)(lineIdx / ((uint64_t)Do_ * (uint64_t)Ho_));
            const uint64_t rem = lineIdx % ((uint64_t)Do_ * (uint64_t)Ho_);
            const uint32_t od = (uint32_t)(rem / Ho_);
            const uint32_t oh = (uint32_t)(rem % Ho_);

            const uint32_t ow_start = segIdx * tileLenW_;
            if (ow_start >= Wo_) { continue; }
            uint32_t workLen = Wo_ - ow_start;
            if (workLen > tileLenW_) workLen = tileLenW_;

            ComputeTile(n, od, oh, ow_start, workLen);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t n, uint32_t od, uint32_t oh, uint32_t ow_start, uint32_t lenW)
    {
        // strides for x [N,C,D,H,W]
        const uint64_t xStrideW = 1ull;
        const uint64_t xStrideH = (uint64_t)W_;
        const uint64_t xStrideD = (uint64_t)H_ * (uint64_t)W_;
        const uint64_t xStrideC = (uint64_t)D_ * (uint64_t)H_ * (uint64_t)W_;
        const uint64_t xStrideN = (uint64_t)C_ * (uint64_t)D_ * (uint64_t)H_ * (uint64_t)W_;

        // strides for w [Co,C,Kd,Kh,Kw]
        const uint64_t wStrideKw = 1ull;
        const uint64_t wStrideKh = (uint64_t)Kw_;
               const uint64_t wStrideKd = (uint64_t)Kh_ * (uint64_t)Kw_;
        const uint64_t wStrideC  = (uint64_t)Kd_ * (uint64_t)Kh_ * (uint64_t)Kw_;
        const uint64_t wStrideCo = (uint64_t)C_ * (uint64_t)Kd_ * (uint64_t)Kh_ * (uint64_t)Kw_;

        // strides for y [N,Co,Do,Ho,Wo]
        const uint64_t yStrideW  = 1ull;
        const uint64_t yStrideH  = (uint64_t)Wo_;
        const uint64_t yStrideD  = (uint64_t)Ho_ * (uint64_t)Wo_;
        const uint64_t yStrideCo = (uint64_t)Do_ * (uint64_t)Ho_ * (uint64_t)Wo_;
        const uint64_t yStrideN  = (uint64_t)Co_ * (uint64_t)Do_ * (uint64_t)Ho_ * (uint64_t)Wo_;

        // Iterate output channels one by one (conservative UB use)
        for (uint32_t co = 0; co < Co_; ++co) {
            // accRow for the current [co, n, od, oh, ow:ow+lenW)
            AscendC::LocalTensor<float> accRow = tmpQ_.AllocTensor<float>();
            AscendC::Duplicate(accRow, 0.0f, lenW);

            for (uint32_t c = 0; c < C_; ++c) {
                for (uint32_t kd = 0; kd < Kd_; ++kd) {
                    const uint32_t id = od + kd;
                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        const uint32_t ih = oh + kh;

                        // Load weight row w[co, c, kd, kh, 0:Kw]
                        AscendC::LocalTensor<float> wRow = wRowQ_.AllocTensor<float>();
                        const uint64_t wBase =
                            co * wStrideCo + c * wStrideC + kd * wStrideKd + kh * wStrideKh;
                        AscendC::DataCopy(wRow, wGm_[wBase], Kw_);

                        for (uint32_t r = 0; r < Kw_; ++r) {
                            // Load x vector of length lenW corresponding to this r shift
                            AscendC::LocalTensor<float> xVec = xVecQ_.AllocTensor<float>();
                            const uint32_t iw = ow_start + r;
                            const uint64_t xBase =
                                n * xStrideN + c * xStrideC + (uint64_t)id * xStrideD +
                                (uint64_t)ih * xStrideH + (uint64_t)iw * xStrideW;
                            AscendC::DataCopy(xVec, xGm_[xBase], lenW);

                            // tmp = xVec * wRow[r]; accumulate into accRow
                            AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                            // Read scalar weight from local tensor
                            float wr = wRow.GetValue(r);
                            AscendC::Muls(tmp, xVec, wr, lenW);
                            AscendC::Add(accRow, accRow, tmp, lenW);

                            tmpQ_.FreeTensor(tmp);
                            xVecQ_.FreeTensor(xVec);
                        }

                        wRowQ_.FreeTensor(wRow);
                    }
                }
            }

            // Write accRow to y
            const uint64_t yBase =
                n * yStrideN + (uint64_t)co * yStrideCo + (uint64_t)od * yStrideD +
                (uint64_t)oh * yStrideH + (uint64_t)ow_start * yStrideW;
            AscendC::DataCopy(yGm_[yBase], accRow, lenW);

            tmpQ_.FreeTensor(accRow);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INBUF_NUM> xVecQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, WBUF_NUM>  wRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMPBUF_NUM> tmpQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, D_{0}, H_{0}, W_{0};
    uint32_t Co_{0}, Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Do_{0}, Ho_{0}, Wo_{0};
    uint32_t tileLenW_{0}, tileCo_{0}, segmentsPerLine_{1};
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(
        x, weight, y,
        tiling_data.N, tiling_data.C, tiling_data.D, tiling_data.H, tiling_data.W,
        tiling_data.Co, tiling_data.Kd, tiling_data.Kh, tiling_data.Kw,
        tiling_data.Do, tiling_data.Ho, tiling_data.Wo,
        tiling_data.tileLenW, tiling_data.tileCo, tiling_data.segmentsPerLine
    );
    op.Process();
}
