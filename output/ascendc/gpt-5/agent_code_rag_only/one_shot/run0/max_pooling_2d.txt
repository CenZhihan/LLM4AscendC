namespace {
static inline uint32_t FloorDiv(int64_t a, int64_t b)
{
    // assumes b > 0
    return static_cast<uint32_t>(a / b);
}

static inline uint32_t ComputeOutDim(uint32_t in, uint32_t kernel, uint32_t stride, uint32_t pad)
{
    // floor((in + 2*pad - kernel) / stride) + 1
    int64_t numer = static_cast<int64_t>(in) + 2LL * static_cast<int64_t>(pad) - static_cast<int64_t>(kernel);
    if (numer < 0) numer = 0; // safe-guard
    return static_cast<uint32_t>(numer / static_cast<int64_t>(stride) + 1LL);
}

static inline uint32_t Clamp(uint32_t v, uint32_t lo, uint32_t hi)
{
    if (v < lo) return lo;
    if (v > hi) return hi;
    return v;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling2dCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    if (in_shape->GetStorageShape().GetDimNum() != 4 && in_shape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernel_size = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding = attrs->GetAttrPointer<int32_t>(2);

    const uint32_t N = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(0));
    const uint32_t C = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(1));
    const uint32_t H = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(2));
    const uint32_t W = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(3));

    const uint32_t k = static_cast<uint32_t>(*kernel_size);
    const uint32_t s = static_cast<uint32_t>(*stride);
    const uint32_t p = static_cast<uint32_t>(*padding);

    const uint32_t outH = ComputeOutDim(H, k, s, p);
    const uint32_t outW = ComputeOutDim(W, k, s, p);

    const uint32_t totalInElems = N * C * H * W;
    const uint32_t totalOutElems = N * C * outH * outW;

    uint32_t blockDim = totalOutElems / 65536 + 1;
    blockDim = Clamp(blockDim, 8u, 24u);
    if (blockDim > totalOutElems) blockDim = totalOutElems > 0 ? totalOutElems : 1u;
    context->SetBlockDim(blockDim);

    tiling.set_c(C);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.set_kH(k);
    tiling.set_kW(k);
    tiling.set_sH(s);
    tiling.set_sW(s);
    tiling.set_pH(p);
    tiling.set_pW(p);

    tiling.set_totalInElems(totalInElems);
    tiling.set_totalOutElems(totalOutElems);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t TMP_BUFFER_NUM = 2;

class KernelMaxPooling2dCustom {
public:
    __aicore__ inline KernelMaxPooling2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t outH, uint32_t outW,
                                uint32_t kH, uint32_t kW,
                                uint32_t sH, uint32_t sW,
                                uint32_t pH, uint32_t pW,
                                uint32_t totalIn, uint32_t totalOut)
    {
        n_ = n; c_ = c; h_ = h; w_ = w;
        outH_ = outH; outW_ = outW;
        kH_ = kH; kW_ = kW; sH_ = sH; sW_ = sW; pH_ = pH; pW_ = pW;
        totalIn_ = totalIn; totalOut_ = totalOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalIn_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut_);

        pipe_.InitBuffer(tmpQueue_, TMP_BUFFER_NUM, sizeof(float)); // buffers of 1 float for scalar ops
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride loop over output elements
        for (uint32_t outIndex = blockIdx; outIndex < totalOut_; outIndex += blockNum) {
            ComputeOne(outIndex);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t outIndex)
    {
        // Decode flat index to (n, c, oh, ow)
        uint32_t hwOut = outH_ * outW_;
        uint32_t chwOut = c_ * hwOut;

        uint32_t n = outIndex / chwOut;
        uint32_t rem1 = outIndex - n * chwOut;
        uint32_t c = rem1 / hwOut;
        uint32_t rem2 = rem1 - c * hwOut;
        uint32_t oh = rem2 / outW_;
        uint32_t ow = rem2 - oh * outW_;

        // top-left input corner (with padding)
        int32_t ih0 = static_cast<int32_t>(oh) * static_cast<int32_t>(sH_) - static_cast<int32_t>(pH_);
        int32_t iw0 = static_cast<int32_t>(ow) * static_cast<int32_t>(sW_) - static_cast<int32_t>(pW_);

        // Prepare local buffers for scalar max accumulation
        AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
        bool initialized = false;

        // Iterate window
        for (uint32_t kh = 0; kh < kH_; ++kh) {
            int32_t ih = ih0 + static_cast<int32_t>(kh);
            if (ih < 0 || ih >= static_cast<int32_t>(h_)) continue;

            for (uint32_t kw = 0; kw < kW_; ++kw) {
                int32_t iw = iw0 + static_cast<int32_t>(kw);
                if (iw < 0 || iw >= static_cast<int32_t>(w_)) continue;

                // Flatten input index
                uint32_t inIndex = (((n * c_) + c) * h_ + static_cast<uint32_t>(ih)) * w_ + static_cast<uint32_t>(iw);

                // Load current scalar to local
                AscendC::LocalTensor<float> curLocal = tmpQueue_.AllocTensor<float>();
                AscendC::DataCopy(curLocal, xGm_[inIndex], 1);

                if (!initialized) {
                    AscendC::DataCopy(maxLocal, curLocal, 1);
                    initialized = true;
                } else {
                    AscendC::Max(maxLocal, maxLocal, curLocal, 1);
                }
                tmpQueue_.FreeTensor(curLocal);
            }
        }

        // If window fully padded (rare), set 0.0 as fallback
        if (!initialized) {
            // Write 0.0f to maxLocal by multiply-add trick
            AscendC::Muls(maxLocal, maxLocal, 0.0f, 1);
        }

        // Store result
        AscendC::DataCopy(yGm_[outIndex], maxLocal, 1);
        tmpQueue_.FreeTensor(maxLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t kH_{0}, kW_{0}, sH_{0}, sW_{0}, pH_{0}, pW_{0};
    uint32_t totalIn_{0}, totalOut_{0};
};

extern "C" __global__ __aicore__ void max_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling2dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.outH, tiling_data.outW,
            tiling_data.kH, tiling_data.kW,
            tiling_data.sH, tiling_data.sW,
            tiling_data.pH, tiling_data.pW,
            tiling_data.totalInElems, tiling_data.totalOutElems);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline std::vector<int64_t> infer_maxpool2d_out_shape(const at::Tensor& x,
                                                             int64_t kernel_size,
                                                             int64_t stride,
                                                             int64_t padding)
{
    TORCH_CHECK(x.dim() == 4, "MaxPooling2dCustom expects 4D NCHW tensor");
    int64_t N = x.size(0);
    int64_t C = x.size(1);
    int64_t H = x.size(2);
    int64_t W = x.size(3);
    auto out_dim = [](int64_t in, int64_t k, int64_t s, int64_t p) -> int64_t {
        int64_t numer = in + 2 * p - k;
        if (numer < 0) numer = 0;
        return numer / s + 1;
    };
    int64_t outH = out_dim(H, kernel_size, stride, padding);
    int64_t outW = out_dim(W, kernel_size, stride, padding);
    return {N, C, outH, outW};
}

at::Tensor max_pooling_2d_custom_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding) {
    auto out_sizes = infer_maxpool2d_out_shape(self, kernel_size, stride, padding);
    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnMaxPooling2dCustom, self, (int32_t)kernel_size, (int32_t)stride, (int32_t)padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_2d_custom", &max_pooling_2d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_2d_custom", &max_pooling_2d_custom_impl_npu, "MaxPooling2d (MaxPooling2dCustom on NPU)");
}
