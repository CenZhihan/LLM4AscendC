host_operator_src="""

#include "matmul_divide_gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// UB model for post-epilogue:
//   inQueue: 2 * tile
//   outQueue: 2 * tile
//   tmpQueue: 2 * tile (for GELU intermediate)
// Total floats = 6 * tile => bytes = 24 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulDivideGeluCustomTilingData tiling;

    // Shapes:
    // x: [B, I] -> m=B, k=I
    // w: [O, I] -> n=O, k=I (F.linear uses w^T internally)
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t m = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t k = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t n = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));

    const uint32_t totalLen = m * n;

    // Elementwise post-epilogue tileLen from UB budget: bytes ~= 24 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 24;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLen < tileLen) {
        const uint32_t aligned = AlignDown(totalLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLen;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_totalLen(totalLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No additional workspace required
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // y shape is [B, O] from x[B, I] and w[O, I]
    const gert::Shape *x_shape = context->GetInputShape(0);

    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t B = x_shape->GetOriginShape().GetDim(0);
    const int64_t O = w_shape->GetOriginShape().GetDim(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, B);
    y_shape->SetDim(1, O);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulDivideGeluCustom {
public:
    __aicore__ inline KernelMatmulDivideGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y,
                                uint32_t totalLen, uint32_t tileLen)
    {
        totalLen_ = totalLen;
        tileLen_ = tileLen;

        xGm_ = (__gm__ float*)x;
        wGm_ = (__gm__ float*)w;
        bGm_ = (__gm__ float*)b;
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        divisorGm_ = (__gm__ float*)divisor;

        pipe_.InitBuffer(inQueueC_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue0_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void ProcessMatmulAndEpilogue(uint32_t m, uint32_t n, uint32_t k)
    {
        // Phase 1: GEMM C = X @ W^T + b
        // Use Matmul library kernel (assumes float ND, B transposed).
        TCubeTiling cubeTiling;
        // For simplicity, rely on default internal tiling by library if available.
        // If tiling memory is required by runtime, it will be passed via workspace by REGIST_MATMUL_OBJ in host.
        AscendC::TPipe pipe;
        using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
        using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true, LayoutMode::NORMAL>;
        using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
        using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
        BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> batchMatmulKernel;
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), batchMatmulKernel.matmulObj, &cubeTiling);
        // reinterpret yGm_ base as GM_ADDR to comply
        batchMatmulKernel.Init((GM_ADDR)xGm_, (GM_ADDR)wGm_, (GM_ADDR)bGm_, (GM_ADDR)yGm_.GetPtr(), GetSysWorkSpacePtr(), cubeTiling);
        batchMatmulKernel.Process(&pipe, 3, 3);

        // Load scalar divisor
        float d = 1.0f;
        // Safe scalar load
        d = *divisorGm_;
        // Avoid division by zero
        if (d == 0.0f) d = 1.0f;
        invDiv_ = 1.0f / d;

        // Phase 2: Epilogue on y in-place: y = GELU(y / d)
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> cLocal = inQueueC_.AllocTensor<float>();
        AscendC::DataCopy(cLocal, yGm_[gmOffset], len);
        inQueueC_.EnQue<float>(cLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> cLocal = inQueueC_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> t0 = tmpQueue0_.AllocTensor<float>();
        AscendC::LocalTensor<float> t1 = tmpQueue1_.AllocTensor<float>();

        // s = c / d  => t0 = c * invDiv_
        AscendC::Muls(t0, cLocal, invDiv_, len);

        // GELU(s) = 0.5 * s * (1 + erf(s / sqrt(2)))
        // t1 = s / sqrt(2)
        AscendC::Muls(t1, t0, INV_SQRT2, len);
        AscendC::Erf(t1, t1, len);               // t1 = erf(t1)
        AscendC::Adds(t1, t1, 1.0f, len);        // t1 = 1 + erf(...)
        AscendC::Muls(t1, t1, 0.5f, len);        // t1 = 0.5 * (1 + erf(...))
        AscendC::Mul(yLocal, t0, t1, len);       // y = s * t1

        tmpQueue1_.FreeTensor(t1);
        tmpQueue0_.FreeTensor(t0);
        outQueueY_.EnQue<float>(yLocal);
        inQueueC_.FreeTensor(cLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueC_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue0_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;

    AscendC::GlobalTensor<float> yGm_;
    __gm__ float* xGm_ {nullptr};
    __gm__ float* wGm_ {nullptr};
    __gm__ float* bGm_ {nullptr};
    __gm__ float* divisorGm_ {nullptr};

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    float invDiv_ {1.0f};
    static constexpr float INV_SQRT2 = 0.7071067811865476f;
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor,
                                                                GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling_data, tilingGm);
    const uint32_t m = tiling_data.m;
    const uint32_t n = tiling_data.n;
    const uint32_t k = tiling_data.k;

    KernelMatmulDivideGeluCustom op;
    op.Init(x, w, b, divisor, y, tiling_data.totalLen, tiling_data.tileLen);
    op.ProcessMatmulAndEpilogue(m, n, k);
}
