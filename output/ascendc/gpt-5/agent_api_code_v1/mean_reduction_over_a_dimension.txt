host_operator_src="""
#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0U) ? 0U : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0U) ? 0U : (a + b - 1U) / b;
}
}

namespace optiling {

// Memory model:
//   xtile: 2 * tileInner
//   acc:   1 * tileInner
// Total floats = 3 * tileInner => bytes = 12 * tileInner
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    const int64_t rank = static_cast<int64_t>(inShape->GetOriginShape().GetDimNum());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int64_t dim = static_cast<int64_t>(*dimPtr);

    if (rank <= 0) {
        return ge::GRAPH_SUCCESS;
    }
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) {
        return ge::GRAPH_PARAM_INVALID;
    }

    // Compute outer, axisLen, inner
    uint64_t outer_u64 = 1, inner_u64 = 1, axisLen_u64 = 1;
    for (int64_t i = 0; i < dim; ++i) {
        outer_u64 *= static_cast<uint64_t>(inShape->GetOriginShape().GetDim(i));
    }
    axisLen_u64 = static_cast<uint64_t>(inShape->GetOriginShape().GetDim(dim));
    for (int64_t i = dim + 1; i < rank; ++i) {
        inner_u64 *= static_cast<uint64_t>(inShape->GetOriginShape().GetDim(i));
    }

    const uint32_t outer = static_cast<uint32_t>(outer_u64);
    const uint32_t axisLen = static_cast<uint32_t>(axisLen_u64);
    const uint32_t inner = static_cast<uint32_t>(inner_u64);

    // Determine tileInner from UB budget
    uint32_t maxTile = (UB_BUDGET_BYTES / 12); // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileInner = maxTile;
    if (inner < tileInner) {
        uint32_t aligned = AlignDown(inner, ALIGN_ELEMS);
        tileInner = (aligned > 0) ? aligned : inner;
    }
    if (tileInner == 0) tileInner = 1;

    // Choose blockDim using parallelism on 'outer'
    uint32_t blockDim = (outer == 0) ? 1U : outer;
    if (blockDim < 1U) blockDim = 1U;
    if (blockDim > 24U) blockDim = 24U;
    context->SetBlockDim(blockDim);

    tiling.set_outer(outer);
    tiling.set_axisLen(axisLen);
    tiling.set_inner(inner);
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const int64_t rank = static_cast<int64_t>(x_shape->GetOriginShape().GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int64_t dim = static_cast<int64_t>(*dimPtr);
    if (dim < 0) dim += rank;

    gert::Shape *y_shape = context->GetOutputShape(0);
    if (rank <= 0) {
        y_shape->SetDimNum(0);
        return GRAPH_SUCCESS;
    }

    // Build reduced shape with dim removed
    std::vector<int64_t> outDims;
    outDims.reserve((size_t)(rank - 1));
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        outDims.push_back(x_shape->GetOriginShape().GetDim(i));
    }
    if (outDims.size() == 0) {
        // Reduce to scalar: represent as shape [] (0-dim)
        y_shape->SetDimNum(0);
    } else {
        y_shape->SetDimNum(static_cast<uint32_t>(outDims.size()));
        for (size_t i = 0; i < outDims.size(); ++i) {
            y_shape->SetDim(i, outDims[i]);
        }
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM_IN = 2;
constexpr int32_t BUFFER_NUM_OUT = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t axisLen, uint32_t inner,
                                uint32_t tileInner)
    {
        outer_ = outer;
        axisLen_ = axisLen;
        inner_ = inner;
        tileInner_ = tileInner;
        inTotal_ = outer_ * axisLen_ * inner_;
        outTotal_ = outer_ * inner_;
        invAxisLen_ = (axisLen_ > 0U) ? (1.0f / static_cast<float>(axisLen_)) : 0.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, inTotal_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outTotal_);

        pipe_.InitBuffer(inQ_, BUFFER_NUM_IN, tileInner_ * sizeof(float));
        pipe_.InitBuffer(outQ_, BUFFER_NUM_OUT, tileInner_ * sizeof(float));
        pipe_.InitBuffer(accQ_, 1, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outer_ == 0U || inner_ == 0U || axisLen_ == 0U || tileInner_ == 0U) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t o = blockIdx; o < outer_; o += blockNum) {
            // Loop over inner dimension by tiles
            for (uint32_t t = 0; t < inner_; t += tileInner_) {
                const uint32_t curLen = (t + tileInner_ <= inner_) ? tileInner_ : (inner_ - t);
                // Allocate local buffers
                AscendC::LocalTensor<float> xtile = inQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();

                // Initialize accumulator with first axis slice
                uint32_t base = o * axisLen_ * inner_ + t;
                AscendC::DataCopy(xtile, xGm_[base], curLen);
                AscendC::Adds(acc, xtile, 0.0f, curLen);

                // Accumulate remaining axis slices
                for (uint32_t k = 1; k < axisLen_; ++k) {
                    const uint32_t offset = base + k * inner_;
                    AscendC::DataCopy(xtile, xGm_[offset], curLen);
                    AscendC::Add(acc, acc, xtile, curLen);
                }

                // Compute mean
                AscendC::Muls(acc, acc, invAxisLen_, curLen);

                // Write result
                AscendC::LocalTensor<float> ytile = outQ_.AllocTensor<float>();
                AscendC::Adds(ytile, acc, 0.0f, curLen); // copy acc -> ytile

                AscendC::DataCopy(yGm_[o * inner_ + t], ytile, curLen);

                // Free locals
                outQ_.FreeTensor(ytile);
                accQ_.FreeTensor(acc);
                inQ_.FreeTensor(xtile);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM_IN> inQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM_OUT> outQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_{0};
    uint32_t axisLen_{0};
    uint32_t inner_{0};
    uint32_t tileInner_{0};
    uint32_t inTotal_{0};
    uint32_t outTotal_{0};
    float invAxisLen_{0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.outer, tiling_data.axisLen, tiling_data.inner,
            tiling_data.tileInner);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor mean_reduction_over_a_dimension_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    int64_t ndim = self.dim();
    if (dim < 0) dim += ndim;

    std::vector<int64_t> out_sizes;
    out_sizes.reserve((size_t)std::max<int64_t>(0, ndim - 1));
    for (int64_t i = 0; i < ndim; ++i) {
        if (i == dim) continue;
        out_sizes.push_back(self.size(i));
    }

    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, (int32_t)dim, result);
    return result;
}

    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_over_a_dimension_custom_impl_npu);
