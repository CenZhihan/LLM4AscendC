host_operator_src="""
#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
} // namespace

namespace optiling {

// UB model with fp16:
//   xVec: tileM
//   yTile: tileM * colBlock
//   wBlk: colBlock * cIn
// Total elements = tileM * (1 + colBlock) + colBlock * cIn
// Total bytes = 2 * [ tileM * (1 + colBlock) + colBlock * cIn ]
static constexpr uint32_t UB_BUDGET_BYTES = 180 * 1024; // leave margin
static constexpr uint32_t ALIGN_M = 128;                // DMA-friendly
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect x: [N, Cin, H, W], weight: [Cout, Cin, 1, 1]
    const uint32_t N    = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H    = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W    = static_cast<uint32_t>(x_shape->GetDim(3));
    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));

    const uint64_t Sbatch64 = static_cast<uint64_t>(H) * static_cast<uint64_t>(W);
    const uint32_t Sbatch = static_cast<uint32_t>(Sbatch64);

    // Choose colBlock (number of output channels processed together)
    // Try larger first within UB; cap at 128, 64, 32, then fallback to Cout.
    const uint32_t candidates[] = {128u, 64u, 32u, 16u, 8u, 4u, 2u, 1u};
    uint32_t colBlock = (Cout < 128u) ? Cout : 128u;
    for (size_t i = 0; i < sizeof(candidates)/sizeof(candidates[0]); ++i) {
        uint32_t cand = (Cout < candidates[i]) ? Cout : candidates[i];
        if (cand == 0) continue;
        // Memory elements count for fp16
        uint64_t elems_needed = static_cast<uint64_t>(cand) * static_cast<uint64_t>(Cin); // wBlk
        // We'll compute tileM based on budget left: UB/(2) bytes /2 bytes per elem -> elements budget
        // bytes = 2 * [ tileM * (1 + cand) + cand * Cin ] <= UB_BUDGET_BYTES
        // => tileM <= (UB/2 - 2 * cand * Cin) / (2 * (1 + cand)) == (UB/2bytes - cand*Cin) / (1+cand)
        int64_t budget_elems = static_cast<int64_t>(UB_BUDGET_BYTES / 2) - static_cast<int64_t>(elems_needed);
        if (budget_elems <= static_cast<int64_t>(ALIGN_M)) {
            continue;
        }
        uint32_t tileM = static_cast<uint32_t>(budget_elems / (1 + cand));
        tileM = AlignDown(tileM, ALIGN_M);
        if (tileM >= ALIGN_M) {
            colBlock = cand;
            break;
        }
    }
    if (colBlock == 0) colBlock = (Cout > 0) ? 1u : 0u;

    // Compute tileM with chosen colBlock
    uint64_t elems_needed = static_cast<uint64_t>(colBlock) * static_cast<uint64_t>(Cin);
    int64_t budget_elems = static_cast<int64_t>(UB_BUDGET_BYTES / 2) - static_cast<int64_t>(elems_needed);
    uint32_t tileM = 0;
    if (budget_elems > 0) {
        tileM = static_cast<uint32_t>(budget_elems / (1 + colBlock));
        tileM = AlignDown(tileM, ALIGN_M);
        if (tileM == 0) tileM = (budget_elems > 0) ? static_cast<uint32_t>(budget_elems / (1 + colBlock)) : 0u;
    }
    if (tileM == 0) tileM = (Sbatch < ALIGN_M) ? Sbatch : ALIGN_M;
    if (tileM > Sbatch) tileM = Sbatch;

    const uint32_t tilesPerBatch = (tileM == 0) ? 1u : CeilDiv(Sbatch, tileM);
    uint32_t totalTasks = N * tilesPerBatch;
    uint32_t blockDim = totalTasks;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cIn(Cin);
    tiling.set_cOut(Cout);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_tileM(tileM);
    tiling.set_colBlock(colBlock);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

using DType = half;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t h, uint32_t wdim,
                                uint32_t tileM, uint32_t colBlock)
    {
        n_ = n; cIn_ = cIn; cOut_ = cOut; h_ = h; w_ = wdim;
        tileM_ = tileM; colBlock_ = colBlock;

        const uint64_t xSize = static_cast<uint64_t>(n_) * cIn_ * h_ * w_;
        const uint64_t ySize = static_cast<uint64_t>(n_) * cOut_ * h_ * w_;
        const uint64_t wSize = static_cast<uint64_t>(cOut_) * cIn_;
        xGm_.SetGlobalBuffer((__gm__ DType*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ DType*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ DType*)y, ySize);

        // UB buffers
        pipe_.InitBuffer(xQueue_, 1, tileM_ * sizeof(DType));                 // x vector
        pipe_.InitBuffer(yQueue_, 1, tileM_ * colBlock_ * sizeof(DType));     // y tile (columns contiguous)
        pipe_.InitBuffer(wQueue_, 1, colBlock_ * cIn_ * sizeof(DType));       // weight block (colBlock x cIn)
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * sizeof(DType));               // temp vector
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cIn_ == 0 || cOut_ == 0 || h_ == 0 || w_ == 0 || tileM_ == 0) {
            return;
        }
        const uint32_t Sbatch = h_ * w_;
        const uint32_t tilesPerBatch = (tileM_ == 0) ? 1u : (Sbatch + tileM_ - 1u) / tileM_;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t startIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalTasks = n_ * tilesPerBatch;

        for (uint32_t taskId = startIdx; taskId < totalTasks; taskId += blockNum) {
            const uint32_t nIdx = taskId / tilesPerBatch;
            const uint32_t tIdx = taskId % tilesPerBatch;
            const uint32_t sStart = tIdx * tileM_;
            uint32_t len = Sbatch - sStart;
            if (len > tileM_) len = tileM_;

            // For output channels in blocks
            for (uint32_t co0 = 0; co0 < cOut_; co0 += colBlock_) {
                uint32_t cols = cOut_ - co0;
                if (cols > colBlock_) cols = colBlock_;

                // Load weight block (rows = cols, cols = cIn_) contiguous per row
                AscendC::LocalTensor<DType> wLocal = wQueue_.AllocTensor<DType>();
                for (uint32_t j = 0; j < cols; ++j) {
                    // each row: W[co0 + j, 0..cIn_-1]
                    const uint64_t wRowBase = static_cast<uint64_t>(co0 + j) * cIn_;
                    AscendC::DataCopy(wLocal[j * cIn_], wGm_[wRowBase], cIn_);
                }

                // Initialize y tile to zeros
                AscendC::LocalTensor<DType> yLocal = yQueue_.AllocTensor<DType>();
                AscendC::Duplicate<DType>(yLocal, static_cast<DType>(0.0f), static_cast<uint64_t>(cols) * tileM_);

                // Accumulate over input channels
                for (uint32_t k = 0; k < cIn_; ++k) {
                    // Load x vector for this input channel k
                    const uint64_t xBase = (static_cast<uint64_t>(nIdx) * cIn_ + k) * Sbatch + sStart;
                    AscendC::LocalTensor<DType> xLocal = xQueue_.AllocTensor<DType>();
                    AscendC::DataCopy(xLocal, xGm_[xBase], len);

                    // Prepare temp buffer
                    AscendC::LocalTensor<DType> tLocal = tmpQueue_.AllocTensor<DType>();

                    // For each output channel column in this block
                    for (uint32_t j = 0; j < cols; ++j) {
                        // Read scalar weight wLocal[j * cIn_ + k]
                        // Broadcast scalar to vector and mul-add
                        DType wval = wLocal.GetValue(j * cIn_ + k);
                        AscendC::Duplicate<DType>(tLocal, wval, len);        // t = wval (vector)
                        AscendC::Mul(tLocal, tLocal, xLocal, len);           // t = wval * x
                        // yCol offset = j * tileM_
                        AscendC::Add(yLocal[j * tileM_], yLocal[j * tileM_], tLocal, len);
                    }

                    tmpQueue_.FreeTensor(tLocal);
                    xQueue_.FreeTensor(xLocal);
                }

                // Store y columns back to GM
                for (uint32_t j = 0; j < cols; ++j) {
                    const uint64_t yBase = (static_cast<uint64_t>(nIdx) * cOut_ + (co0 + j)) * Sbatch + sStart;
                    AscendC::DataCopy(yGm_[yBase], yLocal[j * tileM_], len);
                }

                yQueue_.FreeTensor(yLocal);
                wQueue_.FreeTensor(wLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<DType> xGm_;
    AscendC::GlobalTensor<DType> wGm_;
    AscendC::GlobalTensor<DType> yGm_;

    uint32_t n_ {0}, cIn_ {0}, cOut_ {0}, h_ {0}, w_ {0};
    uint32_t tileM_ {0}, colBlock_ {0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling, tilingGm);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling.n, tiling.cIn, tiling.cOut,
            tiling.h, tiling.w,
            tiling.tileM, tiling.colBlock);
    op.Process();
}
