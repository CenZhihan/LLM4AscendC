host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <cstdint>
#include <vector>

namespace {

static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

} // anonymous namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // per-core UB budget (conservative)
static constexpr uint32_t ALIGN_ELEMS = 1024;             // favor larger DMA bursts
// Buffers per tile: maxVals(float)*tile + curVals(float)*tile + idx(int32)*tile = 12 * tile bytes
static constexpr uint32_t BYTES_PER_ROW = 12;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> &dims = x_shape->GetOriginShape().GetDims();
    const uint32_t rank = static_cast<uint32_t>(dims.size());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimAttr != nullptr) ? *dimAttr : 0;
    // Normalize dim
    if (dim < 0) dim += static_cast<int32_t>(rank);
    if (dim < 0 || dim >= static_cast<int32_t>(rank)) {
        return ge::GRAPH_FAILED;
    }

    uint64_t outerSize = 1;
    for (int32_t i = 0; i < dim; ++i) {
        outerSize *= static_cast<uint64_t>(dims[i]);
    }
    uint64_t reduceSize = static_cast<uint64_t>(dims[dim]);
    uint64_t innerSize = 1;
    for (uint32_t i = dim + 1; i < rank; ++i) {
        innerSize *= static_cast<uint64_t>(dims[i]);
    }
    if (innerSize == 0 || reduceSize == 0) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t totalRows = outerSize * innerSize;

    // Determine tileRows respecting UB budget and InnerSize alignment
    uint32_t maxTileRows = static_cast<uint32_t>(UB_BUDGET_BYTES / BYTES_PER_ROW);
    if (maxTileRows == 0) maxTileRows = 1;
    maxTileRows = AlignDown(maxTileRows, ALIGN_ELEMS);
    if (maxTileRows < ALIGN_ELEMS) {
        // still allow smaller tiles if innerSize is small
        maxTileRows = std::min<uint32_t>(static_cast<uint32_t>(innerSize), static_cast<uint32_t>(UB_BUDGET_BYTES / BYTES_PER_ROW));
        if (maxTileRows == 0) maxTileRows = 1;
    }
    uint32_t tileRows = std::min<uint64_t>(maxTileRows, innerSize);
    // try align to 256 for decent bandwidth
    tileRows = AlignDown(tileRows, 256);
    if (tileRows == 0) tileRows = std::min<uint64_t>(innerSize, 256ULL);
    if (tileRows == 0) tileRows = 1;

    uint32_t numTilesPerOuter = CeilDiv(static_cast<uint64_t>(innerSize), static_cast<uint64_t>(tileRows));
    // Parallelize across outer slices primarily
    uint64_t workUnits = outerSize * numTilesPerOuter; // each work unit: one outer x one inner-tile
    uint32_t blockDim = static_cast<uint32_t>(std::min<uint64_t>(workUnits, 24ULL));
    if (blockDim < 8u && workUnits >= 8u) blockDim = 8u;
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_outerSize(static_cast<uint32_t>(outerSize));
    tiling.set_innerSize(static_cast<uint32_t>(innerSize));
    tiling.set_reduceSize(static_cast<uint32_t>(reduceSize));
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0; // no workspace
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> &x_dims = x_shape->GetOriginShape().GetDims();
    const uint32_t rank = static_cast<uint32_t>(x_dims.size());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimAttr != nullptr) ? *dimAttr : 0;
    if (dim < 0) dim += static_cast<int32_t>(rank);
    if (dim < 0 || dim >= static_cast<int32_t>(rank)) {
        return GRAPH_FAILED;
    }

    std::vector<int64_t> y_dims;
    y_dims.reserve(rank > 0 ? rank - 1 : 0);
    for (uint32_t i = 0; i < rank; ++i) {
        if (static_cast<int32_t>(i) == dim) continue;
        y_dims.push_back(x_dims[i]);
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(static_cast<uint32_t>(y_dims.size()));
    for (uint32_t i = 0; i < y_dims.size(); ++i) {
        y_shape->SetDim(i, y_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Argmax indices as int32 for kernel; Python binding will cast to int64
    context->SetOutputDataType(0, ge::DT_INT32);
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outerSize, uint32_t innerSize,
                                uint32_t reduceSize, uint32_t tileRows, int32_t dim)
    {
        outerSize_  = outerSize;
        innerSize_  = innerSize;
        reduceSize_ = reduceSize;
        tileRows_   = tileRows;
        dim_        = dim;

        // total elements for x
        uint64_t totalElems = static_cast<uint64_t>(outerSize_) * reduceSize_ * innerSize_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElems);
        // total rows for y
        uint64_t totalRows = static_cast<uint64_t>(outerSize_) * innerSize_;
        yGm_.SetGlobalBuffer((__gm__ int32_t*)y, totalRows);

        pipe_.InitBuffer(inQueueCur_, BUFFER_NUM, tileRows_ * sizeof(float));
        pipe_.InitBuffer(inQueueMax_, BUFFER_NUM, tileRows_ * sizeof(float));
        pipe_.InitBuffer(idxQueue_,    BUFFER_NUM, tileRows_ * sizeof(int32_t));
    }

    __aicore__ inline void Process()
    {
        if (innerSize_ == 0 || reduceSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Distribute over (outer, innerTile) pairs: each outer has ceil(innerSize/tileRows) tiles
        const uint32_t tilesPerOuter = (innerSize_ + tileRows_ - 1u) / tileRows_;
        const uint64_t totalUnits = static_cast<uint64_t>(outerSize_) * tilesPerOuter;

        for (uint64_t unit = blockIdx; unit < totalUnits; unit += blockNum) {
            uint32_t outer = static_cast<uint32_t>(unit / tilesPerOuter);
            uint32_t innerTileIdx = static_cast<uint32_t>(unit % tilesPerOuter);
            uint32_t innerStart = innerTileIdx * tileRows_;
            uint32_t curRows = innerSize_ - innerStart;
            if (curRows > tileRows_) curRows = tileRows_;
            DoArgmaxOneTile(outer, innerStart, curRows);
        }
    }

private:
    __aicore__ inline __gm__ float* XPtr(uint64_t offset) {
        return (__gm__ float*)(xGm_.GetPhyAddr()) + offset;
    }
    __aicore__ inline __gm__ int32_t* YPtr(uint64_t offset) {
        return (__gm__ int32_t*)(yGm_.GetPhyAddr()) + offset;
    }

    __aicore__ inline void LoadSliceTo(LocalTensor<float>& buf, uint32_t outer, uint32_t k, uint32_t innerStart, uint32_t len)
    {
        // x offset = ((outer * reduceSize) + k) * innerSize + innerStart
        uint64_t base = (static_cast<uint64_t>(outer) * reduceSize_ + k) * innerSize_ + innerStart;
        AscendC::DataCopy(buf, xGm_[base], len);
    }

    __aicore__ inline void InitIdx(LocalTensor<int32_t>& idxBuf, uint32_t len)
    {
        // Set all indices to 0
        // Prefer vector duplicate if available; fallback to simple loop
        AscendC::Duplicate(idxBuf, 0, len);
    }

    __aicore__ inline void DoArgmaxOneTile(uint32_t outer, uint32_t innerStart, uint32_t curRows)
    {
        if (curRows == 0) return;

        // Allocate local buffers
        LocalTensor<float> maxVals = inQueueMax_.AllocTensor<float>();
        LocalTensor<int32_t> maxIdx = idxQueue_.AllocTensor<int32_t>();

        // Load k=0 slice as initial maxima, initialize idx=0
        LoadSliceTo(maxVals, outer, 0, innerStart, curRows);
        InitIdx(maxIdx, curRows);

        // Iterate over k = 1..reduceSize_-1
        for (uint32_t k = 1; k < reduceSize_; ++k) {
            LocalTensor<float> curVals = inQueueCur_.AllocTensor<float>();
            LoadSliceTo(curVals, outer, k, innerStart, curRows);

            // Scalar compare-update loop in UB
            for (uint32_t i = 0; i < curRows; ++i) {
                float cv = curVals.GetValue(i);
                float mv = maxVals.GetValue(i);
                if (cv > mv) {
                    maxVals.SetValue(i, cv);
                    maxIdx.SetValue(i, static_cast<int32_t>(k));
                }
            }

            inQueueCur_.FreeTensor(curVals);
        }

        // Write out indices to GM: y offset = outer * innerSize + innerStart
        uint64_t yBase = static_cast<uint64_t>(outer) * innerSize_ + innerStart;
        AscendC::DataCopy(yGm_[yBase], maxIdx, curRows);

        // Free local buffers
        idxQueue_.FreeTensor(maxIdx);
        inQueueMax_.FreeTensor(maxVals);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueCur_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMax_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> idxQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int32_t> yGm_;

    uint32_t outerSize_{0};
    uint32_t innerSize_{0};
    uint32_t reduceSize_{0};
    uint32_t tileRows_{0};
    int32_t dim_{0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.outerSize, tiling_data.innerSize, tiling_data.reduceSize,
            tiling_data.tileRows, tiling_data.dim);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include <vector>
#include "pytorch_npu_helper.hpp"

static inline std::vector<int64_t> compute_argmax_output_shape(const at::Tensor& x, int64_t dim) {
    int64_t rank = x.dim();
    if (dim < 0) dim += rank;
    std::vector<int64_t> outShape;
    outShape.reserve(rank > 0 ? (size_t)rank - 1 : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        outShape.push_back(x.size(i));
    }
    return outShape;
}

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    auto outShape = compute_argmax_output_shape(self, dim);
    at::Tensor result_i32 = at::empty(outShape, self.options().dtype(at::kInt));
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, static_cast<int>(dim), result_i32);
    // Cast to int64 to match PyTorch argmax dtype
    return result_i32.to(at::kLong);
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu,
          "Argmax over a specified dimension (ArgmaxOverADimensionCustom on NPU)");
}
