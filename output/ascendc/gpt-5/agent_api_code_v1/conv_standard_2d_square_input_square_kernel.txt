host_operator_src="""
#include "conv_standard_2d_square_input_square_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// We store only output tile in UB; computation uses 1-element temporaries.
static constexpr uint32_t UB_BUDGET_BYTES = 128 * 1024; // per-core conservative
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Kh   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw   = static_cast<uint32_t>(w_shape->GetDim(3));
    // Only groups=1, stride=(1,1), padding=(0,0), dilation=(1,1) supported in this custom kernel.

    if (Cin != static_cast<uint32_t>(w_shape->GetDim(1))) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t outH = (H >= Kh) ? (H - Kh + 1u) : 0u;
    const uint32_t outW = (W >= Kw) ? (W - Kw + 1u) : 0u;

    ConvStandard2dSquareInputSquareKernelCustomTilingData tiling;

    const uint64_t totalOut64 = static_cast<uint64_t>(N) * Cout * outH * outW;
    const uint32_t totalOut   = static_cast<uint32_t>(totalOut64);

    // UB tile length allocation: store only y tile
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOut < tileLen) {
        const uint32_t aligned = AlignDown(totalOut, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOut;
    }

    const uint32_t numTiles = (tileLen == 0u) ? 1u : CeilDiv(totalOut, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim == 0u) blockDim = 1u;
    if (blockDim > 32u) blockDim = 32u;
    context->SetBlockDim(blockDim);

    tiling.set_totalOut(totalOut);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_H(H);
    tiling.set_W(W);

    tiling.set_Cout(Cout);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);

    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t outH, uint32_t outW,
                                uint32_t totalOut, uint32_t tileLength)
    {
        N_ = N; Cin_ = Cin; H_ = H; W_ = W;
        Cout_ = Cout; Kh_ = Kh; Kw_ = Kw;
        outH_ = outH; outW_ = outW;
        totalOut_ = totalOut;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Cout_) * Cin_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * Cout_ * outH_ * outW_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // scalar buffers for single-element loads and temporaries
        pipe_.InitBuffer(inQueueX1_, 1, sizeof(float));
        pipe_.InitBuffer(inQueueW1_, 1, sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOut_) return;

        while (true) {
            uint32_t curLen = totalOut_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            ComputeTile(base, curLen);
            CopyOut(base, curLen);

            base += strideElems;
            if (base >= totalOut_) break;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t base, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        // Initialize outputs to zero
        AscendC::Duplicate(yLocal, 0.0f, len);

        for (uint32_t i = 0; i < len; ++i) {
            const uint32_t idx = base + i;

            uint32_t tmp = idx;
            const uint32_t ow = tmp % outW_; tmp /= outW_;
            const uint32_t oh = tmp % outH_; tmp /= outH_;
            const uint32_t oc = tmp % Cout_; tmp /= Cout_;
            const uint32_t n  = tmp;

            // acc buffer
            AscendC::LocalTensor<float> acc = tmpQueue_.AllocTensor<float>();
            AscendC::Duplicate(acc, 0.0f, 1);

            bool first_term = true;
            AscendC::LocalTensor<float> mulTmp = tmpQueue_.AllocTensor<float>();

            for (uint32_t ic = 0; ic < Cin_; ++ic) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const uint32_t ih = oh + kh;
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const uint32_t iw = ow + kw;

                        const uint64_t xOff =
                            (((static_cast<uint64_t>(n) * Cin_ + ic) * H_ + ih) * W_ + iw);
                        const uint64_t wOff =
                            ((((static_cast<uint64_t>(oc) * Cin_ + ic) * Kh_ + kh) * Kw_ + kw));

                        AscendC::LocalTensor<float> x1 = inQueueX1_.AllocTensor<float>();
                        AscendC::LocalTensor<float> w1 = inQueueW1_.AllocTensor<float>();

                        AscendC::DataCopy(x1, xGm_[xOff], 1);
                        AscendC::DataCopy(w1, wGm_[wOff], 1);

                        if (first_term) {
                            // acc = x1 * w1
                            AscendC::Muls(acc, x1, w1, 1);
                            first_term = false;
                        } else {
                            // mulTmp = x1 * w1; acc += mulTmp
                            AscendC::Muls(mulTmp, x1, w1, 1);
                            AscendC::Add(acc, acc, mulTmp, 1);
                        }

                        inQueueX1_.FreeTensor(x1);
                        inQueueW1_.FreeTensor(w1);
                    }
                }
            }

            // store accumulated result into yLocal[i]
            AscendC::DataCopy(yLocal[i], acc, 1);

            tmpQueue_.FreeTensor(mulTmp);
            tmpQueue_.FreeTensor(acc);
        }

        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX1_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, Kh_{0}, Kw_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t totalOut_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputSquareKernelCustom op;
    op.Init(
        x, weight, y,
        tiling_data.N, tiling_data.Cin, tiling_data.H, tiling_data.W,
        tiling_data.Cout, tiling_data.Kh, tiling_data.Kw,
        tiling_data.outH, tiling_data.outW,
        tiling_data.totalOut, tiling_data.tileLength
    );
    op.Process();
}
