#include "conv_transposed_1d_dilated_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// Conservative UB budget. We need buffers for:
//   yTile: tileLen
//   xSeg: up to tileLen
// plus some safety margin => approx 3 * tileLen floats => bytes ~ 12 * tileLen
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed1dDilatedCustomTilingData tiling;

    // x: (N, Cin, Lin)
    const auto* x_shape = context->GetInputShape(0);
    const auto* w_shape = context->GetInputShape(1);
    const uint32_t N   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t Lin = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));

    const uint32_t WinCin = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t Cout   = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    const uint32_t K      = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));

    // Attributes
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stride   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *padding  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilation = attrs->GetAttrPointer<int32_t>(2);

    // Validate compatible shapes (basic check)
    if (WinCin != Cin) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Lout = static_cast<uint32_t>((static_cast<int64_t>(Lin) - 1) * (*stride) - 2 * (*padding) + (*dilation) * (static_cast<int64_t>(K) - 1) + 1);

    // Derive tile length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 12; // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (Lout < tileLen) {
        const uint32_t aligned = AlignDown(Lout, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : Lout;
    }

    // Blocks schedule across N*Cout pairs
    const uint32_t tasks = N * Cout;
    uint32_t blockDim = tasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_lin(Lin);
    tiling.set_lout(Lout);
    tiling.set_k(K);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_stride(*stride);
    tiling.set_padding(*padding);
    tiling.set_dilation(*dilation);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM_X = 1;
constexpr int32_t BUFFER_NUM_Y = 1;
constexpr int32_t BUFFER_NUM_W = 1;

class KernelConvTransposed1dDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed1dDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t lout, uint32_t ksize,
                                uint32_t tileLen,
                                int32_t stride, int32_t padding, int32_t dilation)
    {
        n_ = n; cin_ = cin; cout_ = cout; lin_ = lin; lout_ = lout; k_ = ksize;
        tileLen_ = tileLen;
        stride_ = stride; padding_ = padding; dilation_ = dilation;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cin_ * lin_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, cin_ * cout_ * k_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * cout_ * lout_);

        // Raw pointers (for scalar reads if needed)
        xPtr_ = reinterpret_cast<__gm__ float*>(x);
        wPtr_ = reinterpret_cast<__gm__ float*>(weight);
        yPtr_ = reinterpret_cast<__gm__ float*>(y);

        pipe_.InitBuffer(xSegQueue_, BUFFER_NUM_X, tileLen_ * sizeof(float));
        pipe_.InitBuffer(yTileQueue_, BUFFER_NUM_Y, tileLen_ * sizeof(float));
        pipe_.InitBuffer(wScalarQueue_, BUFFER_NUM_W, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (lout_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalPairs = n_ * cout_;

        for (uint32_t pair = blockIdx; pair < totalPairs; pair += blockNum) {
            const uint32_t nIdx = pair / cout_;
            const uint32_t oc   = pair % cout_;

            // Iterate over L_out tiles
            for (uint32_t base = 0; base < lout_; base += tileLen_) {
                uint32_t curLen = lout_ - base;
                if (curLen > tileLen_) curLen = tileLen_;

                // Allocate y tile and zero-fill
                AscendC::LocalTensor<float> yLocal = yTileQueue_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, curLen);

                // Accumulate over input channels and kernel taps
                for (uint32_t ci = 0; ci < cin_; ++ci) {
                    for (uint32_t kk = 0; kk < k_; ++kk) {
                        // Weight index: ((ci * cout + oc) * k) + kk
                        const uint32_t wOff = (ci * cout_ + oc) * k_ + kk;
                        // Load scalar weight (assume direct GM read is permitted)
                        float wVal = wPtr_[wOff];

                        // For stride == 1 fast path: contiguous over tile per kk
                        // in_idx = base + padding - kk*dilation + i (i in [0, curLen))
                        int64_t start_in = static_cast<int64_t>(base) + static_cast<int64_t>(padding_) - static_cast<int64_t>(kk) * static_cast<int64_t>(dilation_);
                        if (stride_ == 1) {
                            int64_t l0 = 0;
                            if (start_in < 0) l0 = -start_in;
                            int64_t l1 = curLen;
                            int64_t maxEnd = static_cast<int64_t>(lin_) - start_in;
                            if (l1 > maxEnd) l1 = maxEnd;
                            if (l1 > l0) {
                                uint32_t seg = static_cast<uint32_t>(l1 - l0);
                                uint32_t xOff = static_cast<uint32_t>(((nIdx * cin_ + ci) * lin_) + (start_in + l0));
                                AscendC::LocalTensor<float> xLocal = xSegQueue_.AllocTensor<float>();
                                AscendC::DataCopy(xLocal, xGm_[xOff], seg);
                                // Scale by weight scalar and accumulate into yLocal at offset l0
                                AscendC::Muls(xLocal, xLocal, wVal, seg);
                                AscendC::LocalTensor<float> ySub = yLocal[l0];
                                AscendC::Add(ySub, ySub, xLocal, seg);
                                xSegQueue_.FreeTensor(xLocal);
                            }
                        } else {
                            // Generic stride: only positions where (start_in + i) % stride == 0 contribute
                            // We process in small runs by stepping i by 1 and gathering segments of valid positions spaced by stride.
                            // To keep UB operations contiguous, we perform scalar fallback on segments of length 1 (simple and correct, less efficient).
                            for (uint32_t i = 0; i < curLen; ++i) {
                                int64_t pos = start_in + static_cast<int64_t>(i);
                                if (pos >= 0 && pos < static_cast<int64_t>(lin_) && ((pos % stride_) == 0)) {
                                    uint32_t in_idx = static_cast<uint32_t>(pos / stride_);
                                    uint32_t xOff = (nIdx * cin_ + ci) * lin_ + in_idx;
                                    AscendC::LocalTensor<float> xLocal = xSegQueue_.AllocTensor<float>();
                                    AscendC::DataCopy(xLocal, xGm_[xOff], 1);
                                    AscendC::Muls(xLocal, xLocal, wVal, 1);
                                    AscendC::LocalTensor<float> ySub = yLocal[i];
                                    AscendC::Add(ySub, ySub, xLocal, 1);
                                    xSegQueue_.FreeTensor(xLocal);
                                }
                            }
                        }
                    }
                }

                // Store y tile back
                uint32_t yOff = (nIdx * cout_ + oc) * lout_ + base;
                AscendC::DataCopy(yGm_[yOff], yLocal, curLen);
                yTileQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM_X> xSegQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM_Y> yTileQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM_W> wScalarQueue_; // reserved, possibly unused

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    __gm__ float* xPtr_{nullptr};
    __gm__ float* wPtr_{nullptr};
    __gm__ float* yPtr_{nullptr};

    uint32_t n_{0}, cin_{0}, cout_{0}, lin_{0}, lout_{0}, k_{0};
    uint32_t tileLen_{0};
    int32_t stride_{1}, padding_{0}, dilation_{1};
};

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1dDilatedCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.lout, tiling_data.k,
            tiling_data.tileLength,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation);
    op.Process();
}
