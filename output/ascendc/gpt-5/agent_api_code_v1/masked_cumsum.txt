host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 1 * tile
//   inQueueM: 1 * tile
//   outQueueY: 1 * tile
// Total floats = 3 * tile => bytes = 12 * tile
// We target full-row processing for performance. Use 512KB per-core budget typical on 910B.
static constexpr uint32_t UB_BUDGET_BYTES = 512 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;             // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 65536;         // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int32_t dim = static_cast<int32_t>(*dimPtr);

    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin = x_shape->GetOriginShape();

    int32_t rank = static_cast<int32_t>(origin.GetDimNum());
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }
    // Normalize dim
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) {
        return ge::GRAPH_FAILED;
    }

    uint64_t totalSizeU64 = origin.GetShapeSize();
    uint32_t totalSize = static_cast<uint32_t>(totalSizeU64);

    uint32_t axisLen = static_cast<uint32_t>(origin.GetDim(dim));
    // Products for outer and trailing
    uint64_t outer64 = 1;
    for (int i = 0; i < dim; ++i) {
        outer64 *= static_cast<uint64_t>(origin.GetDim(i));
    }
    uint64_t trailing64 = 1;
    for (int i = dim + 1; i < rank; ++i) {
        trailing64 *= static_cast<uint64_t>(origin.GetDim(i));
    }
    // We currently support cumsum along the last physical dimension for contiguous row processing.
    // Enforce trailing == 1
    if (trailing64 != 1) {
        return ge::GRAPH_FAILED;
    }
    uint32_t outer = static_cast<uint32_t>(outer64);

    // tileAxisLen from UB budget: bytes = 12 * tileAxisLen
    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileAxisLen = maxTile;
    if (axisLen < tileAxisLen) {
        const uint32_t aligned = AlignDown(axisLen, ALIGN_ELEMS);
        tileAxisLen = (aligned > 0) ? aligned : axisLen;
    }

    // Prefer full-row tiles; ensure tileAxisLen >= axisLen when possible
    if (tileAxisLen < axisLen) {
        // keep as is; kernel handles full rows only; return failure if axis exceeds tile
        // You may tune UB_BUDGET_BYTES if needed.
        // For now we still set tileAxisLen, but kernel expects tileAxisLen >= axisLen.
    }

    // Blocks based on number of rows
    uint32_t blockDim = outer;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_outer(outer);
    tiling.set_axisLen(axisLen);
    tiling.set_tileAxisLen(tileAxisLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *m_shape = context->GetInputShape(1);
    // Basic validation: masks must match x
    if (x_shape->GetOriginShape().GetShapeSize() != m_shape->GetOriginShape().GetShapeSize()) {
        return GRAPH_FAILED;
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    const ge::DataType m_dtype = context->GetInputDataType(1);
    if (x_dtype != ge::DT_FLOAT || m_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR m, GM_ADDR y,
                                uint32_t outer, uint32_t axisLen,
                                uint32_t tileAxisLen, int32_t dim)
    {
        outer_ = outer;
        axisLen_ = axisLen;
        tileAxisLen_ = tileAxisLen;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, outer_ * axisLen_);
        mGm_.SetGlobalBuffer((__gm__ float*)m, outer_ * axisLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outer_ * axisLen_);

        // one-tile buffers (we aim to process a whole row)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileAxisLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueM_, BUFFER_NUM, tileAxisLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileAxisLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outer_ == 0 || axisLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block handles a grid-stride of rows
        for (uint32_t row = blockIdx; row < outer_; row += blockNum) {
            const uint32_t base = row * axisLen_;

            // For performance and simplicity we expect tileAxisLen_ >= axisLen_
            uint32_t curLen = axisLen_;
            if (curLen > tileAxisLen_) curLen = tileAxisLen_;

            CopyIn(base, curLen);
            Compute(curLen);
            CopyOut(base, curLen);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> mLocal = inQueueM_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        AscendC::DataCopy(mLocal, mGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
        inQueueM_.EnQue(mLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> mLocal = inQueueM_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // y = x * mask
        AscendC::Mul(yLocal, xLocal, mLocal, len);

        // In-place inclusive prefix sum using iterative doubling
        // For each offset k, y[k:] += y[:-k]
        for (uint32_t offset = 1; offset < len; offset <<= 1) {
            AscendC::Add(yLocal[offset], yLocal[offset], yLocal, len - offset);
            AscendC::PipeBarrier<PIPE_V>();
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueM_.FreeTensor(mLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueM_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> mGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_ {0};
    uint32_t axisLen_ {0};
    uint32_t tileAxisLen_ {0};
    int32_t dim_ { -1 };
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.outer, tiling_data.axisLen,
            tiling_data.tileAxisLen, tiling_data.dim);
    op.Process();
}

#include <torch/library.h>

#include "pytorch_npu_helper.hpp"

at::Tensor masked_cumsum_impl_npu(const at::Tensor& self, const at::Tensor& mask, int64_t dim) {
    // Ensure mask is float on the same device for kernel simplicity.
    at::Tensor maskf = mask.to(self.options().dtype());
    at::Tensor result = at::empty_like(self);
    EXEC_NPU_CMD(aclnnMaskedCumsumCustom, self, maskf, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("masked_cumsum_custom", &masked_cumsum_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("masked_cumsum_custom", &masked_cumsum_impl_npu, "Masked Cumsum (MaskedCumsumCustom on NPU)");
}
