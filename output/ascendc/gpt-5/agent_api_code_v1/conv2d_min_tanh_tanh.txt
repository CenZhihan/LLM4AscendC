static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t MAX_BLOCKS = 24;
static constexpr uint32_t MIN_BLOCKS = 8;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMinTanhTanhCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const uint32_t nSize = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t cIn   = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t hIn   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t wIn   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t cOut  = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t kH    = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t kW    = static_cast<uint32_t>(w_shape->GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const bool *hasBias = attrs->GetAttrPointer<bool>(0);
    const int32_t stride   = attrs->GetAttrValue<int32_t>(1, 1);
    const int32_t padding  = attrs->GetAttrValue<int32_t>(2, 0);
    const int32_t dilation = attrs->GetAttrValue<int32_t>(3, 1);

    // Restricted to stride=1, padding=0, dilation=1 per model, but keep checks
    const uint32_t effKH = (kH - 1) * dilation + 1;
    const uint32_t effKW = (kW - 1) * dilation + 1;

    const uint32_t hOut = static_cast<uint32_t>(
        (static_cast<int64_t>(hIn) + 2 * padding - effKH) / stride + 1);
    const uint32_t wOut = static_cast<uint32_t>(
        (static_cast<int64_t>(wIn) + 2 * padding - effKW) / stride + 1);

    // Tile heuristic:
    // UB usage approx = inputTile(cIn*(tileH + kH - 1)*(tileW + kW - 1)) +
    //                   weightsOneOutChannel(cIn*kH*kW) +
    //                   minBuf(tileH*tileW)
    // Aim to fit within ~80% of UB budget.
    const float targetBytes = UB_BUDGET_BYTES * 0.80f;
    uint32_t baseTileW = 16;
    if (wOut < baseTileW) baseTileW = wOut;
    if (baseTileW == 0) baseTileW = 1;

    // Solve for tileH given baseTileW
    uint32_t tileH = 8;
    uint32_t tileW = baseTileW;

    auto FitsUB = [&](uint32_t th, uint32_t tw) -> bool {
        uint64_t inputTileElems = static_cast<uint64_t>(cIn) * (th + kH - 1) * (tw + kW - 1);
        uint64_t weightElems = static_cast<uint64_t>(cIn) * kH * kW;
        uint64_t minElems = static_cast<uint64_t>(th) * tw;
        uint64_t totalBytes = (inputTileElems + weightElems + minElems) * 4ULL;
        return totalBytes <= static_cast<uint64_t>(targetBytes);
    };

    // Try increase tileH up to 64 while fitting
    for (uint32_t tryH = 8; tryH <= 64; tryH += 8) {
        if (FitsUB(tryH, tileW)) tileH = tryH;
        else break;
    }

    // Adjust tileW lightly if too wide
    while (!FitsUB(tileH, tileW) && tileW > 8) {
        tileW -= 4;
    }
    if (tileW == 0) tileW = 1;

    // Bound tile sizes by output dims
    if (tileH > hOut) tileH = hOut;
    if (tileW > wOut) tileW = wOut;

    const uint32_t tilesH = (hOut == 0) ? 1u : CeilDiv(hOut, tileH);
    const uint32_t tilesW = (wOut == 0) ? 1u : CeilDiv(wOut, tileW);
    const uint32_t tilesPerN = tilesH * tilesW;
    const uint32_t totalTiles = (nSize == 0) ? 1u : nSize * tilesPerN;

    uint32_t blockDim = totalTiles;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_nSize(nSize);
    tiling.set_cIn(cIn);
    tiling.set_hIn(hIn);
    tiling.set_wIn(wIn);
    tiling.set_cOut(cOut);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_hOut(hOut);
    tiling.set_wOut(wOut);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);
    tiling.set_useBias((*hasBias) ? 1u : 0u);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;

class KernelConv2dMinTanhTanhCustom {
public:
    __aicore__ inline KernelConv2dMinTanhTanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t nSize, uint32_t cIn, uint32_t hIn, uint32_t wIn,
                                uint32_t cOut, uint32_t kH, uint32_t kW,
                                uint32_t hOut, uint32_t wOut,
                                uint32_t tileH, uint32_t tileW,
                                uint32_t useBias)
    {
        nSize_ = nSize; cIn_ = cIn; hIn_ = hIn; wIn_ = wIn;
        cOut_ = cOut; kH_ = kH; kW_ = kW;
        hOut_ = hOut; wOut_ = wOut;
        tileH_ = (tileH == 0 ? 1u : tileH); tileW_ = (tileW == 0 ? 1u : tileW);
        useBias_ = (useBias != 0);

        uint32_t xLen = nSize_ * cIn_ * hIn_ * wIn_;
        uint32_t wLen = cOut_ * cIn_ * kH_ * kW_;
        uint32_t yLen = nSize_ * 1 * hOut_ * wOut_;
        uint32_t bLen = cOut_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xLen);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yLen);
        bGm_.SetGlobalBuffer((__gm__ float*)b, bLen);

        // Local buffers: input tile, weight one-out-channel, and min accumulation tile
        uint32_t maxInTileElems = cIn_ * (tileH_ + kH_ - 1) * (tileW_ + kW_ - 1);
        uint32_t weightElems = cIn_ * kH_ * kW_;
        uint32_t minElems = tileH_ * tileW_;

        pipe_.InitBuffer(inTileQueue_, BUFFER_NUM, maxInTileElems * sizeof(float));
        pipe_.InitBuffer(weightQueue_, BUFFER_NUM, weightElems * sizeof(float));
        pipe_.InitBuffer(minQueue_, BUFFER_NUM, minElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, minElems * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (nSize_ == 0 || hOut_ == 0 || wOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        const uint32_t tilesH = CeilDiv(hOut_, tileH_);
        const uint32_t tilesW = CeilDiv(wOut_, tileW_);
        const uint32_t tilesPerN = tilesH * tilesW;
        const uint32_t totalTiles = nSize_ * tilesPerN;

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            uint32_t n = tileId / tilesPerN;
            uint32_t rem = tileId % tilesPerN;
            uint32_t th = rem / tilesW;
            uint32_t tw = rem % tilesW;

            uint32_t hBase = th * tileH_;
            uint32_t wBase = tw * tileW_;

            uint32_t curTileH = ((hBase + tileH_) <= hOut_) ? tileH_ : (hOut_ - hBase);
            uint32_t curTileW = ((wBase + tileW_) <= wOut_) ? tileW_ : (wOut_ - wBase);

            ComputeTile(n, hBase, wBase, curTileH, curTileW);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
        return (a + b - 1u) / b;
    }

    __aicore__ inline void CopyInputTile(LocalTensor<float>& inLocal,
                                         uint32_t n, uint32_t hBase, uint32_t wBase,
                                         uint32_t curTileH, uint32_t curTileW)
    {
        // Copy input tile: shape [cIn, curTileH + kH - 1, curTileW + kW - 1]
        uint32_t extH = curTileH + kH_ - 1;
        uint32_t extW = curTileW + kW_ - 1;
        uint32_t planeSize = extH * extW;

        for (uint32_t ci = 0; ci < cIn_; ++ci) {
            for (uint32_t hh = 0; hh < extH; ++hh) {
                uint32_t srcH = hBase + hh;
                uint32_t srcW = wBase;
                uint32_t gmOffset = (((n * cIn_ + ci) * hIn_ + srcH) * wIn_) + srcW;
                uint32_t localOffset = ci * planeSize + hh * extW;
                // Copy a contiguous row of length extW
                DataCopy(inLocal[localOffset], xGm_[gmOffset], extW);
            }
        }
    }

    __aicore__ inline void ComputeTile(uint32_t n, uint32_t hBase, uint32_t wBase,
                                       uint32_t curTileH, uint32_t curTileW)
    {
        // Allocate local buffers
        LocalTensor<float> inLocal = inTileQueue_.AllocTensor<float>();
        LocalTensor<float> wLocal  = weightQueue_.AllocTensor<float>();
        LocalTensor<float> minLocal = minQueue_.AllocTensor<float>();
        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Prepare input tile
        CopyInputTile(inLocal, n, hBase, wBase, curTileH, curTileW);

        // Initialize minLocal with large value
        const float largeVal = 1.0e30f;
        uint32_t minSize = curTileH * curTileW;
        for (uint32_t i = 0; i < minSize; ++i) {
            minLocal.SetValue(i, largeVal);
        }

        // For each output channel, compute convolution and update min
        const uint32_t weightPerCo = cIn_ * kH_ * kW_;
        for (uint32_t co = 0; co < cOut_; ++co) {
            // Load weights for this output channel
            uint32_t wOffset = co * weightPerCo;
            DataCopy(wLocal, wGm_[wOffset], weightPerCo);

            // Bias
            float biasVal = 0.0f;
            if (useBias_) {
                biasVal = bGm_[co];
            }

            // Compute conv for each position in the tile
            uint32_t extH = curTileH + kH_ - 1;
            uint32_t extW = curTileW + kW_ - 1;
            uint32_t inPlaneSize = extH * extW;

            for (uint32_t th = 0; th < curTileH; ++th) {
                for (uint32_t tw = 0; tw < curTileW; ++tw) {
                    float acc = biasVal;
                    // Sum over input channels and kernel window
                    for (uint32_t ci = 0; ci < cIn_; ++ci) {
                        uint32_t inBase = ci * inPlaneSize + th * extW + tw;
                        uint32_t wBase = ci * (kH_ * kW_);
                        for (uint32_t kh = 0; kh < kH_; ++kh) {
                            for (uint32_t kw = 0; kw < kW_; ++kw) {
                                float xVal = inLocal.GetValue(inBase + kh * extW + kw);
                                float wVal = wLocal.GetValue(wBase + kh * kW_ + kw);
                                acc += xVal * wVal;
                            }
                        }
                    }
                    uint32_t pos = th * curTileW + tw;
                    float mv = minLocal.GetValue(pos);
                    if (acc < mv) mv = acc;
                    minLocal.SetValue(pos, mv);
                }
            }
        }

        // Apply tanh twice: yLocal = tanh(tanh(minLocal))
        // First tanh
        Tanh(yLocal, minLocal, minSize);
        // Second tanh into minLocal (reuse buffer to reduce copy)
        Tanh(minLocal, yLocal, minSize);

        // Write results to output GM: contiguous rows of curTileW for curTileH
        for (uint32_t th = 0; th < curTileH; ++th) {
            uint32_t gmOffset = (((n * 1u + 0u) * hOut_ + (hBase + th)) * wOut_) + wBase;
            uint32_t localOffset = th * curTileW;
            DataCopy(yGm_[gmOffset], minLocal[localOffset], curTileW);
        }

        // Free local buffers
        outQueueY_.FreeTensor(yLocal);
        minQueue_.FreeTensor(minLocal);
        weightQueue_.FreeTensor(wLocal);
        inTileQueue_.FreeTensor(inLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, BUFFER_NUM> inTileQueue_;
    TQue<TPosition::VECCALC, BUFFER_NUM> weightQueue_;
    TQue<TPosition::VECCALC, BUFFER_NUM> minQueue_;
    TQue<TPosition::VECOUT,  BUFFER_NUM> outQueueY_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t nSize_{0}, cIn_{0}, hIn_{0}, wIn_{0};
    uint32_t cOut_{0}, kH_{0}, kW_{0};
    uint32_t hOut_{0}, wOut_{0};
    uint32_t tileH_{1}, tileW_{1};
    bool useBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanhCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.nSize, tiling_data.cIn, tiling_data.hIn, tiling_data.wIn,
            tiling_data.cOut, tiling_data.kH, tiling_data.kW,
            tiling_data.hOut, tiling_data.wOut,
            tiling_data.tileH, tiling_data.tileW,
            tiling_data.useBias);
    op.Process();
}
