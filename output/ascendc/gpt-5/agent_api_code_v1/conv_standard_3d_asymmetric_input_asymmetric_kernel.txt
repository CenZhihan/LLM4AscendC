kernel_src="""
#include "kernel_operator.h"

// Naive 3D convolution for stride=1, padding=0, dilation=1, groups=1.
// Processes tiles of output elements in flattened order for simplicity.

class KernelConv3DNaive {
public:
    __aicore__ inline KernelConv3DNaive() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint64_t totalOut, uint32_t tileOut)
    {
        n_ = n; cIn_ = cIn; cOut_ = cOut;
        inD_ = inD; inH_ = inH; inW_ = inW;
        kD_ = kD; kH_ = kH; kW_ = kW;
        outD_ = outD; outH_ = outH; outW_ = outW;
        totalOut_ = totalOut; tileOut_ = tileOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_)*cIn_*inD_*inH_*inW_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cOut_)*cIn_*kD_*kH_*kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut_);

        // Local line buffers for Kw elements (width line for a given ci,kd,kh)
        const uint32_t lineBytes = kW_ * sizeof(float);
        pipe_.InitBuffer(xLineQ_, 1, lineBytes);
        pipe_.InitBuffer(wLineQ_, 1, lineBytes);
        pipe_.InitBuffer(outScalQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * tileOut_;
        uint64_t base = static_cast<uint64_t>(blockIdx) * tileOut_;
        if (base >= totalOut_) return;

        while (base < totalOut_) {
            uint32_t curLen = static_cast<uint32_t>(totalOut_ - base);
            if (curLen > tileOut_) curLen = tileOut_;

            for (uint32_t i = 0; i < curLen; ++i) {
                const uint64_t outLinear = base + i;

                // Unravel linear index into (n, oc, od, oh, ow) with NCDHW ordering on output (N, C_out, outD, outH, outW)
                uint64_t t = outLinear;
                const uint32_t ow = static_cast<uint32_t>(t % outW_); t /= outW_;
                const uint32_t oh = static_cast<uint32_t>(t % outH_); t /= outH_;
                const uint32_t od = static_cast<uint32_t>(t % outD_); t /= outD_;
                const uint32_t oc = static_cast<uint32_t>(t % cOut_); t /= cOut_;
                const uint32_t nb = static_cast<uint32_t>(t);

                float acc = 0.0f;

                for (uint32_t ci = 0; ci < cIn_; ++ci) {
                    for (uint32_t kd = 0; kd < kD_; ++kd) {
                        for (uint32_t kh = 0; kh < kH_; ++kh) {
                            // Copy weight line [oc, ci, kd, kh, 0:Kw]
                            AscendC::LocalTensor<float> wLine = wLineQ_.AllocTensor<float>();
                            const uint64_t wOffset = ((((uint64_t)oc * cIn_ + ci) * kD_ + kd) * kH_ + kh) * kW_;
                            AscendC::DataCopy(wLine, wGm_[wOffset], kW_);

                            // Copy input line [nb, ci, od+kd, oh+kh, ow : ow+Kw]
                            AscendC::LocalTensor<float> xLine = xLineQ_.AllocTensor<float>();
                            const uint32_t id = od + kd;
                            const uint32_t ih = oh + kh;
                            const uint32_t iw = ow;
                            const uint64_t xBase = ((((uint64_t)nb * cIn_ + ci) * inD_ + id) * inH_ + ih) * inW_ + iw;
                            AscendC::DataCopy(xLine, xGm_[xBase], kW_);

                            // Dot product over width line
                            float partial = 0.0f;
                            for (uint32_t j = 0; j < kW_; ++j) {
                                // Assume LocalTensor provides element access for illustrative purposes
                                float xv = xLine.GetValue(j);
                                float wv = wLine.GetValue(j);
                                partial += xv * wv;
                            }
                            acc += partial;

                            xLineQ_.FreeTensor(xLine);
                            wLineQ_.FreeTensor(wLine);
                        }
                    }
                }

                // Store result
                AscendC::LocalTensor<float> outLoc = outScalQ_.AllocTensor<float>();
                outLoc.SetValue(0, acc);
                AscendC::DataCopy(yGm_[outLinear], outLoc, 1);
                outScalQ_.FreeTensor(outLoc);
            }

            base += strideElems;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xLineQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wLineQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outScalQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_;
    uint32_t cIn_;
    uint32_t cOut_;
    uint32_t inD_, inH_, inW_;
    uint32_t kD_, kH_, kW_;
    uint32_t outD_, outH_, outW_;
    uint64_t totalOut_;
    uint32_t tileOut_;
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3DNaive op;
    op.Init(
        x, weight, y,
        tiling_data.n, tiling_data.cIn, tiling_data.cOut,
        tiling_data.inD, tiling_data.inH, tiling_data.inW,
        tiling_data.kD, tiling_data.kH, tiling_data.kW,
        tiling_data.outD, tiling_data.outH, tiling_data.outW,
        tiling_data.totalOut, tiling_data.tileOut
    );
    op.Process();
}
