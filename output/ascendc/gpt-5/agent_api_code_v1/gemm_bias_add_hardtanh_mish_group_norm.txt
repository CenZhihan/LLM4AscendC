#include "kernel_operator.h"
using namespace AscendC;

constexpr int32_t INQ_NUM = 1;
constexpr int32_t OUTQ_NUM = 1;
constexpr int32_t CALC_NUM = 4;

struct PostTiling {
    uint32_t batch;
    uint32_t inFeat;
    uint32_t outFeat;
    uint32_t numGroups;
    uint32_t groupSize;
    uint32_t tileCols;
    uint32_t blockDim;
    float eps;
};

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR lin_bias,
                                GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y, GM_ADDR tiling)
    {
        // Load tiling
        // The tiling buffer is a packed PostTiling struct prepared by host
        postTilingLocal_ = tmpCalcQ_.AllocTensor<uint8_t>();
        // Copy tiling struct
        GlobalTensor<uint8_t> tilingGm;
        tilingGm.SetGlobalBuffer((__gm__ uint8_t*)tiling, sizeof(PostTiling));
        DataCopy(postTilingLocal_, tilingGm, sizeof(PostTiling));
        PostTiling *pt = reinterpret_cast<PostTiling *>(postTilingLocal_.GetPtr());

        batch_ = pt->batch;
        inFeat_ = pt->inFeat;
        outFeat_ = pt->outFeat;
        numGroups_ = pt->numGroups;
        groupSize_ = pt->groupSize;
        tileCols_ = pt->tileCols;
        eps_ = pt->eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(batch_) * inFeat_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(outFeat_) * inFeat_);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeat_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeat_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeat_);
        betaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeat_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(batch_) * outFeat_);

        // Allocate small working buffers for per-group processing
        // One group slice length = groupSize_
        pipe_.InitBuffer(inQ_, INQ_NUM, groupSize_ * sizeof(float));
        pipe_.InitBuffer(outQ_, OUTQ_NUM, groupSize_ * sizeof(float));
        pipe_.InitBuffer(tmpCalcQ_, CALC_NUM, groupSize_ * sizeof(float));

        // Allocate local space for gamma/beta group slices
        pipe_.InitBuffer(paramQ_, 2, groupSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // GEMM: y = x @ weight^T + lin_bias
        // NOTE: For demonstration, we implement a simple row-wise GEMV using inner loops and vector ops.
        // This is not an optimal GEMM implementation.
        for (uint32_t b = GetBlockIdx(); b < batch_; b += GetBlockNum()) {
            // Initialize y row with linear bias
            for (uint32_t g = 0; g < numGroups_; ++g) {
                const uint32_t cStart = g * groupSize_;
                // Load bias lin_bias segment to y row as init accumulator
                LocalTensor<float> yLocal = outQ_.AllocTensor<float>();
                DataCopy(yLocal, linBiasGm_[cStart], groupSize_);
                // Accumulator over K
                for (uint32_t k = 0; k < inFeat_; ++k) {
                    // load x[b,k] scalar
                    float x_bk = *xGm_.GetData() + static_cast<uint64_t>(b) * inFeat_ + k;
                    // load weight group column k for all channels in this group
                    LocalTensor<float> wLocal = tmpCalcQ_.AllocTensor<float>();
                    // weight layout: [C, K] contiguous by K
                    DataCopy(wLocal, wGm_[static_cast<uint64_t>(cStart) * inFeat_ + k], groupSize_, inFeat_); // stride load
                    // yLocal += wLocal * x_bk
                    Muls(wLocal, wLocal, x_bk, groupSize_);
                    Add(yLocal, yLocal, wLocal, groupSize_);
                    tmpCalcQ_.FreeTensor(wLocal);
                }
                outQ_.EnQue<float>(yLocal);
                // write to GM row
                LocalTensor<float> yW = outQ_.DeQue<float>();
                DataCopy(yGm_[static_cast<uint64_t>(b) * outFeat_ + cStart], yW, groupSize_);
                outQ_.FreeTensor(yW);
            }
        }

        // Post-processing per row and per group
        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());
        for (uint32_t b = blockIdx; b < batch_; b += blockNum) {
            for (uint32_t g = 0; g < numGroups_; ++g) {
                const uint32_t cStart = g * groupSize_;
                // Copy y row group slice
                LocalTensor<float> yLocal = inQ_.AllocTensor<float>();
                DataCopy(yLocal, yGm_[static_cast<uint64_t>(b) * outFeat_ + cStart], groupSize_);
                inQ_.EnQue(yLocal);

                // Load second bias, gamma, beta for this group
                LocalTensor<float> biasLocal = paramQ_.AllocTensor<float>();
                LocalTensor<float> gammaLocal = paramQ_.AllocTensor<float>();
                LocalTensor<float> betaLocal  = paramQ_.AllocTensor<float>();
                DataCopy(biasLocal, biasGm_[cStart], groupSize_);
                DataCopy(gammaLocal, gammaGm_[cStart], groupSize_);
                DataCopy(betaLocal,  betaGm_[cStart], groupSize_);

                // y = y + bias
                yLocal = inQ_.DeQue<float>();
                Add(yLocal, yLocal, biasLocal, groupSize_);

                // hardtanh: clamp to [-1,1]
                Maxs(yLocal, yLocal, -1.0f, groupSize_);
                Mins(yLocal, yLocal,  1.0f, groupSize_);

                // mish(x) = x * tanh(ln(1 + exp(x)))
                LocalTensor<float> tmp1 = tmpCalcQ_.AllocTensor<float>();
                LocalTensor<float> tmp2 = tmpCalcQ_.AllocTensor<float>();
                Exp(tmp1, yLocal, groupSize_);
                Adds(tmp1, tmp1, 1.0f, groupSize_);
                Ln(tmp1, tmp1, groupSize_);
                Tanh(tmp2, tmp1, groupSize_);
                Mul(yLocal, yLocal, tmp2, groupSize_);
                tmpCalcQ_.FreeTensor(tmp1);
                tmpCalcQ_.FreeTensor(tmp2);

                // GroupNorm over this group slice (length groupSize_)
                // Compute mean and variance
                LocalTensor<float> sqLocal = tmpCalcQ_.AllocTensor<float>();
                Mul(sqLocal, yLocal, yLocal, groupSize_);
                LocalTensor<float> sumBuf = tmpCalcQ_.AllocTensor<float>();
                LocalTensor<float> sumSqBuf = tmpCalcQ_.AllocTensor<float>();
                ReduceSum(sumBuf, yLocal, groupSize_);     // sum
                ReduceSum(sumSqBuf, sqLocal, groupSize_);  // sum of squares
                float sumVal = sumBuf.GetValue(0);
                float sumSqVal = sumSqBuf.GetValue(0);
                tmpCalcQ_.FreeTensor(sumBuf);
                tmpCalcQ_.FreeTensor(sumSqBuf);
                tmpCalcQ_.FreeTensor(sqLocal);

                const float mean = sumVal / static_cast<float>(groupSize_);
                const float meanSq = sumSqVal / static_cast<float>(groupSize_);
                const float var = meanSq - mean * mean;
                const float invStd = rsqrtf(var + eps_);

                // y = (y - mean) * invStd
                Adds(yLocal, yLocal, -mean, groupSize_);
                Muls(yLocal, yLocal, invStd, groupSize_);

                // Affine: y = y * gamma + beta
                Mul(yLocal, yLocal, gammaLocal, groupSize_);
                Add(yLocal, yLocal, betaLocal, groupSize_);

                // Write back
                DataCopy(yGm_[static_cast<uint64_t>(b) * outFeat_ + cStart], yLocal, groupSize_);

                // Free tensors
                inQ_.FreeTensor(yLocal);
                paramQ_.FreeTensor(biasLocal);
                paramQ_.FreeTensor(gammaLocal);
                paramQ_.FreeTensor(betaLocal);
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, INQ_NUM> inQ_;
    TQue<TPosition::VECOUT, OUTQ_NUM> outQ_;
    TQue<TPosition::VECCALC, CALC_NUM> tmpCalcQ_;
    TQue<TPosition::VECCALC, 2> paramQ_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> linBiasGm_;
    GlobalTensor<float> biasGm_;
    GlobalTensor<float> gammaGm_;
    GlobalTensor<float> betaGm_;
    GlobalTensor<float> yGm_;

    LocalTensor<uint8_t> postTilingLocal_;

    uint32_t batch_{0};
    uint32_t inFeat_{0};
    uint32_t outFeat_{0};
    uint32_t numGroups_{0};
    uint32_t groupSize_{0};
    uint32_t tileCols_{0};
    float eps_{1.0e-5f};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(GM_ADDR x, GM_ADDR weight,
                                                                                     GM_ADDR lin_bias, GM_ADDR bias,
                                                                                     GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                                                                     GM_ADDR y, GM_ADDR workspace,
                                                                                     GM_ADDR tiling)
{
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y, tiling);
    op.Process();
}
