#include "kernel_operator.h"

constexpr int32_t TMP_BUFFERS = 16;

class KernelConv2dSubtractHardSwishMaxPoolMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractHardSwishMaxPoolMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutLength, uint32_t tileLength,
                                float subtractValue, uint32_t poolKernel,
                                uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW,
                                uint32_t channels, uint32_t batch)
    {
        totalOutLen_ = totalOutLength;
        tileLen_ = tileLength;
        subtractValue_ = subtractValue;
        poolK_ = poolKernel;

        inH_ = inH;
        inW_ = inW;
        outH_ = outH;
        outW_ = outW;
        C_ = channels;
        N_ = batch;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * C_ * inH_ * inW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * C_ * outH_ * outW_);

        pipe_.InitBuffer(tmpQueue_, TMP_BUFFERS, TMP_BUFFERS * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutLen_) return;

        uint32_t curLen = totalOutLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalOutLen_) {
                nextLen = totalOutLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
            }

            ComputeRange(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void ComputeRange(uint32_t start, uint32_t len)
    {
        for (uint32_t idx = 0; idx < len; ++idx) {
            const uint32_t outIndex = start + idx;
            if (outIndex >= totalOutLen_) break;
            ComputeOne(outIndex);
        }
    }

    __aicore__ inline void ComputeOne(uint32_t outIndex)
    {
        // Decode linear index to (n, c, hp, wp)
        const uint32_t perN = C_ * outH_ * outW_;
        const uint32_t n = outIndex / perN;
        const uint32_t rem1 = outIndex - n * perN;
        const uint32_t c = rem1 / (outH_ * outW_);
        const uint32_t rem2 = rem1 - c * (outH_ * outW_);
        const uint32_t hp = rem2 / outW_;
        const uint32_t wp = rem2 - hp * outW_;

        const uint32_t hStart = hp * poolK_;
        const uint32_t wStart = wp * poolK_;

        bool hasInit = false;

        AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> valLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp1 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp2 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp3 = tmpQueue_.AllocTensor<float>();

        for (uint32_t kh = 0; kh < poolK_; ++kh) {
            const uint32_t h = hStart + kh;
            for (uint32_t kw = 0; kw < poolK_; ++kw) {
                const uint32_t w = wStart + kw;
                const uint64_t inIndex = (((uint64_t)n * C_ + c) * inH_ + h) * inW_ + w;

                // Load one element x
                AscendC::DataCopy(valLocal, xGm_[inIndex], 1);

                // y1 = x - subtractValue_
                AscendC::Adds(valLocal, valLocal, -subtractValue_, 1);

                // HardSwish: v = valLocal * clamp(valLocal + 3, 0, 6) / 6
                AscendC::Adds(tmp1, valLocal, 3.0f, 1);     // tmp1 = val + 3
                AscendC::Maxs(tmp1, tmp1, 0.0f, 1);         // clamp lower
                AscendC::Mins(tmp1, tmp1, 6.0f, 1);         // clamp upper
                AscendC::Mul(tmp2, valLocal, tmp1, 1);      // tmp2 = val * clamp
                AscendC::Muls(tmp2, tmp2, (1.0f / 6.0f), 1);// tmp2 = hswish(val)

                if (!hasInit) {
                    AscendC::DataCopy(maxLocal, tmp2, 1);
                    hasInit = true;
                } else {
                    AscendC::Maximum(maxLocal, maxLocal, tmp2, 1);
                }
            }
        }

        // Mish on maxLocal:
        // mish(x) = x * tanh(ln(1+e^x)) = x * (a^2 + 2a)/(a^2 + 2a + 2), a = exp(x)
        AscendC::LocalTensor<float> aLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> a2Local = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> numLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> denLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> ratioLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> outLocal = tmpQueue_.AllocTensor<float>();

        AscendC::Exp(aLocal, maxLocal, 1);                 // a = exp(x)
        AscendC::Mul(a2Local, aLocal, aLocal, 1);          // a^2
        AscendC::Muls(tmp1, aLocal, 2.0f, 1);              // 2a
        AscendC::Add(numLocal, a2Local, tmp1, 1);          // a^2 + 2a
        AscendC::Adds(denLocal, numLocal, 2.0f, 1);        // a^2 + 2a + 2
        AscendC::Div(ratioLocal, numLocal, denLocal, 1);   // ratio
        AscendC::Mul(outLocal, maxLocal, ratioLocal, 1);   // mish(max)

        // Store to output
        AscendC::DataCopy(yGm_[outIndex], outLocal, 1);

        // Free temps
        tmpQueue_.FreeTensor(outLocal);
        tmpQueue_.FreeTensor(ratioLocal);
        tmpQueue_.FreeTensor(denLocal);
        tmpQueue_.FreeTensor(numLocal);
        tmpQueue_.FreeTensor(a2Local);
        tmpQueue_.FreeTensor(aLocal);
        tmpQueue_.FreeTensor(tmp3);
        tmpQueue_.FreeTensor(tmp2);
        tmpQueue_.FreeTensor(tmp1);
        tmpQueue_.FreeTensor(valLocal);
        tmpQueue_.FreeTensor(maxLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUFFERS> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLen_{0};
    uint32_t tileLen_{0};
    float subtractValue_{0.5f};
    uint32_t poolK_{2};

    uint32_t inH_{0}, inW_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t C_{0}, N_{0};
};

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
