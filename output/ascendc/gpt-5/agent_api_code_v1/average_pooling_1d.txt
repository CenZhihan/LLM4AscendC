#include "average_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling1dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    // Expect shape (N, C, L)
    uint32_t n = 1, c = 1, lin = 1;
    if (x_shape->GetDimNum() == 3) {
        n = static_cast<uint32_t>(x_shape->GetDim(0));
        c = static_cast<uint32_t>(x_shape->GetDim(1));
        lin = static_cast<uint32_t>(x_shape->GetDim(2));
    } else {
        // Fallback: flatten
        n = 1;
        c = 1;
        lin = static_cast<uint32_t>(x_shape->GetShapeSize());
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size_ptr = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride_ptr = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding_ptr = attrs->GetAttrPointer<int64_t>(2);
    const bool *count_include_pad_ptr = attrs->GetAttrPointer<bool>(3);

    const uint32_t k = static_cast<uint32_t>(*kernel_size_ptr);
    const uint32_t s = static_cast<uint32_t>(*stride_ptr);
    const uint32_t p = static_cast<uint32_t>(*padding_ptr);
    const uint32_t countIncludePad = *count_include_pad_ptr ? 1u : 0u;

    // Output length (ceil_mode = false)
    const uint32_t lout = (lin + 2u * p >= k) ? (static_cast<uint32_t>((lin + 2u * p - k) / s) + 1u) : 0u;

    // UB requirement: xBuf = (tileOut-1)*s + k, yBuf = tileOut
    // bytes = 4 * ( (s+1)*tileOut + (k - s) )
    uint32_t maxTileOut = 1;
    if (UB_BUDGET_BYTES > 4u * (k >= s ? (k - s) : 0u)) {
        const uint32_t numer = UB_BUDGET_BYTES / 4u - (k >= s ? (k - s) : 0u);
        const uint32_t denom = (s + 1u);
        maxTileOut = numer / denom;
    }
    if (maxTileOut > HARD_CAP_ELEMS) maxTileOut = HARD_CAP_ELEMS;
    maxTileOut = AlignDown(maxTileOut, ALIGN_ELEMS);
    if (maxTileOut < ALIGN_ELEMS) maxTileOut = ALIGN_ELEMS;
    if (lout < maxTileOut) {
        uint32_t aligned = AlignDown(lout, ALIGN_ELEMS);
        maxTileOut = (aligned > 0) ? aligned : lout;
    }
    if (lout == 0) maxTileOut = 0;

    // Block dim: balance over N*C and tiles
    const uint32_t groups = n * c;
    uint32_t tilesPerGroup = (maxTileOut == 0) ? 1u : CeilDiv(lout, maxTileOut);
    uint32_t totalTiles = groups * tilesPerGroup;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim > totalTiles) blockDim = totalTiles == 0 ? 1u : totalTiles;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_lin(lin);
    tiling.set_lout(lout);
    tiling.set_kernelSize(k);
    tiling.set_stride(s);
    tiling.set_padding(p);
    tiling.set_tileOut(maxTileOut);
    tiling.set_blockDim(blockDim);
    tiling.set_countIncludePad(countIncludePad);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size_ptr = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride_ptr = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding_ptr = attrs->GetAttrPointer<int64_t>(2);

    int64_t k = *kernel_size_ptr;
    int64_t s = *stride_ptr;
    int64_t p = *padding_ptr;

    if (x_shape->GetDimNum() == 3) {
        int64_t n = x_shape->GetDim(0);
        int64_t c = x_shape->GetDim(1);
        int64_t lin = x_shape->GetDim(2);
        int64_t lout = (lin + 2 * p >= k) ? ((lin + 2 * p - k) / s + 1) : 0;
        y_shape->SetDimNum(3);
        y_shape->SetDim(0, n);
        y_shape->SetDim(1, c);
        y_shape->SetDim(2, lout);
    } else {
        // Fallback: keep same rank, replace last dim
        *y_shape = *x_shape;
    }
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;

class KernelAveragePooling1dCustom {
public:
    __aicore__ inline KernelAveragePooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t lin, uint32_t lout,
                                uint32_t k, uint32_t s, uint32_t p,
                                uint32_t tileOut, uint32_t countIncludePad)
    {
        n_ = n; c_ = c; lin_ = lin; lout_ = lout;
        k_ = k; s_ = s; p_ = p;
        tileOut_ = tileOut;
        countIncludePad_ = (countIncludePad != 0);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * c_ * lin_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * c_ * lout_);

        // Worst-case x window length per tile: (tileOut-1)*s + k
        uint32_t xSeg = (tileOut_ > 0) ? ((tileOut_ - 1u) * s_ + k_) : 0u;
        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, xSeg * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0u || c_ == 0u || lin_ == 0u || lout_ == 0u || k_ == 0u || tileOut_ == 0u) {
            return;
        }

        const uint32_t groups = n_ * c_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t g = blockIdx; g < groups; g += blockNum) {
            // Base offsets in GM
            const uint64_t xBase = static_cast<uint64_t>(g) * lin_;
            const uint64_t yBase = static_cast<uint64_t>(g) * lout_;

            for (uint32_t j0 = 0; j0 < lout_; j0 += tileOut_) {
                const uint32_t curOut = AscendC::min(tileOut_, lout_ - j0);

                // Compute required input slice for this tile
                const int32_t a0 = static_cast<int32_t>(j0) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);
                const int32_t lastJ = static_cast<int32_t>(j0 + curOut - 1u);
                const int32_t bLast = lastJ * static_cast<int32_t>(s_) - static_cast<int32_t>(p_) + static_cast<int32_t>(k_) - 1;

                const int32_t gmStartI = AscendC::max(0, a0);
                const int32_t gmEndI = AscendC::min(static_cast<int32_t>(lin_) - 1, bLast);
                const uint32_t gmCopyLen = (gmEndI >= gmStartI) ? static_cast<uint32_t>(gmEndI - gmStartI + 1) : 0u;

                AscendC::LocalTensor<float> xLocal;
                if (gmCopyLen > 0) {
                    xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[xBase + static_cast<uint64_t>(gmStartI)], gmCopyLen);
                    inQueueX_.EnQue<float>(xLocal);
                }

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // Compute each output in the tile
                // y[j] = avg over window [a, a + k - 1] with zero padding
                // denom = k or valid_count depending on countIncludePad_
                float *yPtr = reinterpret_cast<float*>(yLocal.GetBuf());
                for (uint32_t t = 0; t < curOut; ++t) {
                    const int32_t j = static_cast<int32_t>(j0 + t);
                    const int32_t a = j * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);

                    float acc = 0.0f;
                    uint32_t valid = 0u;

                    for (uint32_t u = 0; u < k_; ++u) {
                        const int32_t idx = a + static_cast<int32_t>(u);
                        if (idx >= 0 && idx < static_cast<int32_t>(lin_) && gmCopyLen > 0 &&
                            idx >= gmStartI && idx <= gmEndI) {
                            const uint32_t li = static_cast<uint32_t>(idx - gmStartI);
                            acc += xLocal.GetValue(li);
                            valid += 1u;
                        } else {
                            // padding => add 0, valid not incremented
                        }
                    }

                    const float denom = countIncludePad_ ? static_cast<float>(k_) : static_cast<float>(valid == 0u ? 1u : valid);
                    yPtr[t] = acc / denom;
                }

                // Pop and free xLocal if enqueued
                if (gmCopyLen > 0) {
                    AscendC::LocalTensor<float> xLocalDeq = inQueueX_.DeQue<float>();
                    inQueueX_.FreeTensor<float>(xLocalDeq);
                }

                outQueueY_.EnQue<float>(yLocal);

                // Copy y out
                AscendC::LocalTensor<float> yLocalDeq = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[yBase + static_cast<uint64_t>(j0)], yLocalDeq, curOut);
                outQueueY_.FreeTensor<float>(yLocalDeq);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM>  inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, lin_{0}, lout_{0};
    uint32_t k_{0}, s_{0}, p_{0};
    uint32_t tileOut_{0};
    bool countIncludePad_{true};
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling1dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.lin, tiling_data.lout,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.tileOut, tiling_data.countIncludePad);
    op.Process();
}
