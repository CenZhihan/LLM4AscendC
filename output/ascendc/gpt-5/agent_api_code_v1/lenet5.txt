static inline uint32_t AlignDown(uint32_t x, uint32_t a)

    return (a + b - 1u) / b;

    Lenet5CustomTilingData tiling;

kernel_src="""

#include "kernel_operator.h"

// This kernel implements the full LeNet-5 forward for a batch of images.
// Fixed geometry:
//   Input: (N, 1, 32, 32)
//   Conv1: (6, 1, 5, 5), stride=1, no pad => (6, 28, 28), ReLU + MaxPool2d(2,2) => (6, 14, 14)
//   Conv2: (16, 6, 5, 5), stride=1, no pad => (16, 10, 10), ReLU + MaxPool2d(2,2) => (16, 5, 5)
//   Flatten => 400
//   FC1: (120, 400) + ReLU
//   FC2: (84, 120) + ReLU
//   FC3: (C, 84) => logits

constexpr int32_t IMG_C = 1;
constexpr int32_t IMG_H = 32;
constexpr int32_t IMG_W = 32;

constexpr int32_t CONV1_OUT_C = 6;
constexpr int32_t CONV1_K = 5;
constexpr int32_t CONV1_OUT_H = IMG_H - CONV1_K + 1; // 28
constexpr int32_t CONV1_OUT_W = IMG_W - CONV1_K + 1; // 28
constexpr int32_t POOL1_H = CONV1_OUT_H / 2;          // 14
constexpr int32_t POOL1_W = CONV1_OUT_W / 2;          // 14

constexpr int32_t CONV2_OUT_C = 16;
constexpr int32_t CONV2_K = 5;
constexpr int32_t CONV2_OUT_H = POOL1_H - CONV2_K + 1; // 10
constexpr int32_t CONV2_OUT_W = POOL1_W - CONV2_K + 1; // 10
constexpr int32_t POOL2_H = CONV2_OUT_H / 2;           // 5
constexpr int32_t POOL2_W = CONV2_OUT_W / 2;           // 5

constexpr int32_t FLAT_LEN = CONV2_OUT_C * POOL2_H * POOL2_W; // 16*5*5 = 400

class KernelLenet5Custom {
public:
    __aicore__ inline KernelLenet5Custom() {}

    __aicore__ inline void Init(
        GM_ADDR x,
        GM_ADDR conv1_w, GM_ADDR conv1_b,
        GM_ADDR conv2_w, GM_ADDR conv2_b,
        GM_ADDR fc1_w, GM_ADDR fc1_b,
        GM_ADDR fc2_w, GM_ADDR fc2_b,
        GM_ADDR fc3_w, GM_ADDR fc3_b,
        GM_ADDR y,
        uint32_t batch, uint32_t tileBatch, uint32_t numClasses)
    {
        batch_ = batch;
        tileBatch_ = tileBatch;
        numClasses_ = numClasses;

        // Bind global buffers
        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_ * IMG_C * IMG_H * IMG_W);
        conv1WGm_.SetGlobalBuffer((__gm__ float*)conv1_w, CONV1_OUT_C * IMG_C * CONV1_K * CONV1_K);
        conv1BGm_.SetGlobalBuffer((__gm__ float*)conv1_b, CONV1_OUT_C);
        conv2WGm_.SetGlobalBuffer((__gm__ float*)conv2_w, CONV2_OUT_C * CONV1_OUT_C * CONV2_K * CONV2_K);
        conv2BGm_.SetGlobalBuffer((__gm__ float*)conv2_b, CONV2_OUT_C);
        fc1WGm_.SetGlobalBuffer((__gm__ float*)fc1_w, 120 * FLAT_LEN);
        fc1BGm_.SetGlobalBuffer((__gm__ float*)fc1_b, 120);
        fc2WGm_.SetGlobalBuffer((__gm__ float*)fc2_w, 84 * 120);
        fc2BGm_.SetGlobalBuffer((__gm__ float*)fc2_b, 84);
        fc3WGm_.SetGlobalBuffer((__gm__ float*)fc3_w, numClasses_ * 84);
        fc3BGm_.SetGlobalBuffer((__gm__ float*)fc3_b, numClasses_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch_ * numClasses_);

        // Init pipe queues
        // Scratch for feature maps and intermediate vectors
        uint32_t scratchFloats = IMG_C*IMG_H*IMG_W
                               + CONV1_OUT_C*CONV1_OUT_H*CONV1_OUT_W
                               + CONV1_OUT_C*POOL1_H*POOL1_W
                               + CONV2_OUT_C*CONV2_OUT_H*CONV2_OUT_W
                               + CONV2_OUT_C*POOL2_H*POOL2_W
                               + 120 + 84 + numClasses_;
        pipe_.InitBuffer(scratchQ_, 1, scratchFloats * sizeof(float));

        // Small constant buffers for conv weights and biases
        pipe_.InitBuffer(conv1WQ_, 1, CONV1_OUT_C * IMG_C * CONV1_K * CONV1_K * sizeof(float)); // 150
        pipe_.InitBuffer(conv1BQ_, 1, CONV1_OUT_C * sizeof(float)); // 6
        pipe_.InitBuffer(conv2WQ_, 1, CONV2_OUT_C * CONV1_OUT_C * CONV2_K * CONV2_K * sizeof(float)); // 2400
        pipe_.InitBuffer(conv2BQ_, 1, CONV2_OUT_C * sizeof(float)); // 16

        // FC row buffers (max row length = FLAT_LEN=400)
        pipe_.InitBuffer(fcRowQ_, 1, FLAT_LEN * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0) return;

        // Preload conv weights and biases to local buffers (reused across batch)
        AscendC::LocalTensor<float> conv1WLocal = conv1WQ_.AllocTensor<float>();
        AscendC::DataCopy(conv1WLocal, conv1WGm_[0], CONV1_OUT_C * IMG_C * CONV1_K * CONV1_K);
        conv1WQ_.EnQue(conv1WLocal);

        AscendC::LocalTensor<float> conv1BLocal = conv1BQ_.AllocTensor<float>();
        AscendC::DataCopy(conv1BLocal, conv1BGm_[0], CONV1_OUT_C);
        conv1BQ_.EnQue(conv1BLocal);

        AscendC::LocalTensor<float> conv2WLocal = conv2WQ_.AllocTensor<float>();
        AscendC::DataCopy(conv2WLocal, conv2WGm_[0], CONV2_OUT_C * CONV1_OUT_C * CONV2_K * CONV2_K);
        conv2WQ_.EnQue(conv2WLocal);

        AscendC::LocalTensor<float> conv2BLocal = conv2BQ_.AllocTensor<float>();
        AscendC::DataCopy(conv2BLocal, conv2BGm_[0], CONV2_OUT_C);
        conv2BQ_.EnQue(conv2BLocal);

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride on batches
        uint32_t strideB = blockNum * tileBatch_;
        uint32_t baseB = blockIdx * tileBatch_;
        if (baseB >= batch_) return;

        while (baseB < batch_) {
            uint32_t curB = tileBatch_;
            if (baseB + curB > batch_) curB = batch_ - baseB;

            // We process tileBatch images sequentially (tileBatch usually 1)
            for (uint32_t b = 0; b < curB; ++b) {
                const uint32_t imgIdx = baseB + b;
                ComputeOne(imgIdx);
            }

            baseB += strideB;
        }

        // Free preloaded buffers
        conv1BQ_.DeQue<float>();
        conv1BQ_.FreeTensor(conv1BLocal);
        conv1WQ_.DeQue<float>();
        conv1WQ_.FreeTensor(conv1WLocal);
        conv2BQ_.DeQue<float>();
        conv2BQ_.FreeTensor(conv2BLocal);
        conv2WQ_.DeQue<float>();
        conv2WQ_.FreeTensor(conv2WLocal);
    }

private:
    __aicore__ inline void ComputeOne(uint32_t imgIdx)
    {
        // Prepare local scratch partitions
        AscendC::LocalTensor<float> scratch = scratchQ_.AllocTensor<float>();
        float* buf = (float*)scratch.GetBuffer(); // Assume LocalTensor provides buffer pointer

        float* xImg      = buf;                                                               // 0 .. 1023
        float* conv1Out  = xImg + IMG_C*IMG_H*IMG_W;                                          // next 4704
        float* pool1Out  = conv1Out + CONV1_OUT_C*CONV1_OUT_H*CONV1_OUT_W;                    // next 1176
        float* conv2Out  = pool1Out + CONV1_OUT_C*POOL1_H*POOL1_W;                            // next 1600
        float* pool2Out  = conv2Out + CONV2_OUT_C*CONV2_OUT_H*CONV2_OUT_W;                    // next 400
        float* fc1Out    = pool2Out + CONV2_OUT_C*POOL2_H*POOL2_W;                            // next 120
        float* fc2Out    = fc1Out + 120;                                                      // next 84
        float* logits    = fc2Out + 84;                                                       // next numClasses_

        // Bring image to UB
        const uint32_t imgOffset = imgIdx * IMG_C * IMG_H * IMG_W;
        AscendC::DataCopy(scratch, xGm_[imgOffset], IMG_C * IMG_H * IMG_W); // copy into beginning of scratch

        // Fetch preloaded conv weights and biases
        AscendC::LocalTensor<float> conv1WLocal = conv1WQ_.DeQue<float>();
        AscendC::LocalTensor<float> conv1BLocal = conv1BQ_.DeQue<float>();
        AscendC::LocalTensor<float> conv2WLocal = conv2WQ_.DeQue<float>();
        AscendC::LocalTensor<float> conv2BLocal = conv2BQ_.DeQue<float>();

        // Conv1 with ReLU
        for (int oc = 0; oc < CONV1_OUT_C; ++oc) {
            float bias = conv1BLocal[oc];
            for (int oy = 0; oy < CONV1_OUT_H; ++oy) {
                for (int ox = 0; ox < CONV1_OUT_W; ++ox) {
                    float acc = 0.0f;
                    // Single input channel
                    for (int ky = 0; ky < CONV1_K; ++ky) {
                        for (int kx = 0; kx < CONV1_K; ++kx) {
                            int inY = oy + ky;
                            int inX = ox + kx;
                            float xval = xImg[inY * IMG_W + inX];
                            int kOffset = oc * (IMG_C * CONV1_K * CONV1_K) + ky * CONV1_K + kx;
                            float wval = conv1WLocal[kOffset];
                            acc += xval * wval;
                        }
                    }
                    acc += bias;
                    if (acc < 0.0f) acc = 0.0f; // ReLU
                    conv1Out[oc * (CONV1_OUT_H * CONV1_OUT_W) + oy * CONV1_OUT_W + ox] = acc;
                }
            }
        }

        // MaxPool 2x2 stride 2 -> (6, 14, 14)
        for (int c = 0; c < CONV1_OUT_C; ++c) {
            for (int py = 0; py < POOL1_H; ++py) {
                for (int px = 0; px < POOL1_W; ++px) {
                    int y0 = py * 2;
                    int x0 = px * 2;
                    float m = -1e30f;
                    for (int dy = 0; dy < 2; ++dy) {
                        for (int dx = 0; dx < 2; ++dx) {
                            float v = conv1Out[c * (CONV1_OUT_H * CONV1_OUT_W)
                                              + (y0 + dy) * CONV1_OUT_W + (x0 + dx)];
                            if (v > m) m = v;
                        }
                    }
                    pool1Out[c * (POOL1_H * POOL1_W) + py * POOL1_W + px] = m;
                }
            }
        }

        // Conv2 with ReLU: input (6,14,14), kernel (16,6,5,5), out (16,10,10)
        for (int oc = 0; oc < CONV2_OUT_C; ++oc) {
            float bias = conv2BLocal[oc];
            for (int oy = 0; oy < CONV2_OUT_H; ++oy) {
                for (int ox = 0; ox < CONV2_OUT_W; ++ox) {
                    float acc = 0.0f;
                    for (int ic = 0; ic < CONV1_OUT_C; ++ic) {
                        for (int ky = 0; ky < CONV2_K; ++ky) {
                            for (int kx = 0; kx < CONV2_K; ++kx) {
                                int inY = oy + ky;
                                int inX = ox + kx;
                                float xval = pool1Out[ic * (POOL1_H * POOL1_W) + inY * POOL1_W + inX];
                                int kOffset = oc * (CONV1_OUT_C * CONV2_K * CONV2_K)
                                            + ic * (CONV2_K * CONV2_K)
                                            + ky * CONV2_K + kx;
                                float wval = conv2WLocal[kOffset];
                                acc += xval * wval;
                            }
                        }
                    }
                    acc += bias;
                    if (acc < 0.0f) acc = 0.0f; // ReLU
                    conv2Out[oc * (CONV2_OUT_H * CONV2_OUT_W) + oy * CONV2_OUT_W + ox] = acc;
                }
            }
        }

        // MaxPool 2x2 stride 2 -> (16, 5, 5)
        for (int c = 0; c < CONV2_OUT_C; ++c) {
            for (int py = 0; py < POOL2_H; ++py) {
                for (int px = 0; px < POOL2_W; ++px) {
                    int y0 = py * 2;
                    int x0 = px * 2;
                    float m = -1e30f;
                    for (int dy = 0; dy < 2; ++dy) {
                        for (int dx = 0; dx < 2; ++dx) {
                            float v = conv2Out[c * (CONV2_OUT_H * CONV2_OUT_W)
                                               + (y0 + dy) * CONV2_OUT_W + (x0 + dx)];
                            if (v > m) m = v;
                        }
                    }
                    pool2Out[c * (POOL2_H * POOL2_W) + py * POOL2_W + px] = m;
                }
            }
        }

        // Flatten to vector of length 400 (reuse pool2Out buffer)
        float* flat = pool2Out; // length = FLAT_LEN

        // FC1: (120, 400) with bias and ReLU
        for (int o = 0; o < 120; ++o) {
            AscendC::LocalTensor<float> fcRow = fcRowQ_.AllocTensor<float>();
            // Copy one row of weights (length = FLAT_LEN)
            AscendC::DataCopy(fcRow, fc1WGm_[o * FLAT_LEN], FLAT_LEN);
            // Dot product
            float acc = 0.0f;
            for (int k = 0; k < FLAT_LEN; ++k) {
                acc += fcRow[k] * flat[k];
            }
            // Bias
            AscendC::LocalTensor<float> tmpBias = fcRowQ_.AllocTensor<float>();
            AscendC::DataCopy(tmpBias, fc1BGm_[o], 1);
            acc += tmpBias[0];
            // ReLU
            if (acc < 0.0f) acc = 0.0f;
            fc1Out[o] = acc;
            // Free temp buffers
            fcRowQ_.FreeTensor(tmpBias);
            fcRowQ_.FreeTensor(fcRow);
        }

        // FC2: (84, 120) with bias and ReLU
        for (int o = 0; o < 84; ++o) {
            AscendC::LocalTensor<float> fcRow = fcRowQ_.AllocTensor<float>();
            AscendC::DataCopy(fcRow, fc2WGm_[o * 120], 120);
            float acc = 0.0f;
            for (int k = 0; k < 120; ++k) {
                acc += fcRow[k] * fc1Out[k];
            }
            AscendC::LocalTensor<float> tmpBias = fcRowQ_.AllocTensor<float>();
            AscendC::DataCopy(tmpBias, fc2BGm_[o], 1);
            acc += tmpBias[0];
            if (acc < 0.0f) acc = 0.0f;
            fc2Out[o] = acc;
            fcRowQ_.FreeTensor(tmpBias);
            fcRowQ_.FreeTensor(fcRow);
        }

        // FC3: (numClasses, 84) with bias, no activation
        for (uint32_t o = 0; o < numClasses_; ++o) {
            AscendC::LocalTensor<float> fcRow = fcRowQ_.AllocTensor<float>();
            AscendC::DataCopy(fcRow, fc3WGm_[o * 84], 84);
            float acc = 0.0f;
            for (int k = 0; k < 84; ++k) {
                acc += fcRow[k] * fc2Out[k];
            }
            AscendC::LocalTensor<float> tmpBias = fcRowQ_.AllocTensor<float>();
            AscendC::DataCopy(tmpBias, fc3BGm_[o], 1);
            acc += tmpBias[0];
            logits[o] = acc;
            fcRowQ_.FreeTensor(tmpBias);
            fcRowQ_.FreeTensor(fcRow);
        }

        // Write logits to GM
        const uint32_t yOffset = imgIdx * numClasses_;
        AscendC::LocalTensor<float> outLocal = scratchQ_.AllocTensor<float>();
        // Copy logits to outLocal start (reuse scratch)
        for (uint32_t i = 0; i < numClasses_; ++i) {
            outLocal[i] = logits[i];
        }
        AscendC::DataCopy(yGm_[yOffset], outLocal, numClasses_);
        scratchQ_.FreeTensor(outLocal);

        // Restore preloaded queues state
        conv1WQ_.EnQue(conv1WLocal);
        conv1BQ_.EnQue(conv1BLocal);
        conv2WQ_.EnQue(conv2WLocal);
        conv2BQ_.EnQue(conv2BLocal);

        scratchQ_.FreeTensor(scratch);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scratchQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> conv1WQ_;
