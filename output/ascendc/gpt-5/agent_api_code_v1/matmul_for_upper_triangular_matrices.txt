host_operator_src="""
#include "matmul_for_upper_triangular_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// UB budgeting:
//   - ySegment: tileN floats
//   - bSegment: tileN floats
// Total floats: 2 * tileN => bytes: 8 * tileN
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core UB budget (conservative)
static constexpr uint32_t ALIGN_ELEMS = 256;            // favor large DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 16384;       // cap to avoid long per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForUpperTriangularMatricesCustomTilingData tiling;

    const auto *in0_shape = context->GetInputShape(0);
    if (in0_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }
    const size_t rank = in0_shape->GetOriginShape().GetDimNum();
    if (rank != 2) {
        return ge::GRAPH_FAILED;
    }
    const int64_t dim0 = in0_shape->GetOriginShape().GetDim(0);
    const int64_t dim1 = in0_shape->GetOriginShape().GetDim(1);
    if (dim0 != dim1 || dim0 <= 0) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(dim0);

    uint32_t maxTile = UB_BUDGET_BYTES / 8; // elements for ySegment and bSegment
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileN = (N < maxTile) ? ((AlignDown(N, ALIGN_ELEMS) > 0) ? AlignDown(N, ALIGN_ELEMS) : N) : maxTile;

    // Choose blockDim based on N (each block processes a subset of rows in grid-stride)
    uint32_t blockDim = N;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0; // no workspace required
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return GRAPH_FAILED;
    }
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    if (a_shape->GetOriginShape().GetDim(0) != a_shape->GetOriginShape().GetDim(1)) {
        return GRAPH_FAILED;
    }
    if (b_shape->GetOriginShape().GetDim(0) != b_shape->GetOriginShape().GetDim(1)) {
        return GRAPH_FAILED;
    }
    if (a_shape->GetOriginShape().GetDim(0) != b_shape->GetOriginShape().GetDim(0)) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    *c_shape = *a_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != ge::DT_FLOAT || b_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"
#include "matmul_for_upper_triangular_matrices_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulForUpperTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForUpperTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileN)
    {
        N_ = N;
        tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(N_) * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(N_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(N_) * N_);

        // Buffers for:
        //  - y segment accumulator (tileN)
        //  - b segment slice (tileN)
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, BUFFER_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        for (uint32_t i = blockIdx; i < N_; i += blockNum) {
            // Zero lower triangle for row i: columns [0, i)
            ZeroLowerTriangleRow(i);

            // Compute upper triangle for row i in segments
            for (uint32_t jBase = i; jBase < N_; jBase += tileN_) {
                uint32_t L = N_ - jBase;
                if (L > tileN_) L = tileN_;

                AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, L);

                // Accumulate yLocal[j - jBase] += A[i, k] * B[k, j] for k in [i..N-1], j in [max(jBase, k)..jBase+L-1]
                for (uint32_t k = i; k < N_; ++k) {
                    uint32_t startJ = (k > jBase) ? k : jBase;
                    if (startJ >= jBase + L) {
                        continue;
                    }
                    uint32_t segLen = (jBase + L) - startJ;

                    // Load contiguous B segment: B[k, startJ : startJ + segLen)
                    AscendC::LocalTensor<float> bSeg = bQueue_.AllocTensor<float>();
                    AscendC::DataCopy(bSeg, bGm_[static_cast<uint64_t>(k) * N_ + startJ], segLen);

                    // Load A[i, k] into a scalar by copying to a one-element local tensor
                    float aVal = 0.0f;
                    {
                        AscendC::LocalTensor<float> tmpA = bQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmpA, aGm_[static_cast<uint64_t>(i) * N_ + k], 1);
                        // Move from local tensor to scalar via vector op: scale bSeg by tmpA[0]
                        // To get the scalar, use Dup on a small vector then Read back via operations chain avoidance,
                        // but here we directly use tmpA[0] in Muls if supported as variable.
                        // Fallback: copy to scalar by a vector add with zero and retrieve via intrinsic is not required;
                        // we will rely on Muls supporting a float variable.
                        aVal = *((float*)tmpA.GetAddr()); // Access underlying address to read scalar
                        bQueue_.FreeTensor(tmpA);
                    }

                    // Scale bSeg by scalar aVal and accumulate into yLocal at offset (startJ - jBase)
                    AscendC::Muls(bSeg, bSeg, aVal, segLen);
                    AscendC::Add(yLocal[startJ - jBase], yLocal[startJ - jBase], bSeg, segLen);

                    bQueue_.FreeTensor(bSeg);
                }

                // Store yLocal to C row segment
                AscendC::DataCopy(cGm_[static_cast<uint64_t>(i) * N_ + jBase], yLocal, L);
                yQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline void ZeroLowerTriangleRow(uint32_t i)
    {
        uint32_t remaining = i;
        uint32_t jBase = 0;
        while (remaining > 0) {
            uint32_t L = remaining;
            if (L > tileN_) L = tileN_;
            AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
            AscendC::Dup(yLocal, 0.0f, L);
            AscendC::DataCopy(cGm_[static_cast<uint64_t>(i) * N_ + jBase], yLocal, L);
            yQueue_.FreeTensor(yLocal);
            jBase += L;
            remaining -= L;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t tileN_{0};
};

extern "C" __global__ __aicore__ void matmul_for_upper_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForUpperTriangularMatricesCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.tileN);
    op.Process();
}
