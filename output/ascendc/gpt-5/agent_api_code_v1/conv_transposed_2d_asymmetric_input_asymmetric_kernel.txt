    const int32_t strideW = context->GetAttrs()->GetAttrValue<int32_t>(1);

    const int32_t dilW    = context->GetAttrs()->GetAttrValue<int32_t>(7);
    const int32_t groups  = context->GetAttrs()->GetAttrValue<int32_t>(8);
    if (groups != 1) {
        return GRAPH_FAILED;

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t TMP_BUFFERS = 6; // enough for x, w, tmp products, and accumulator

class KernelConvTransposed2dAsymInAsymKernelCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymInAsymKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t Hi, uint32_t Wi,
                                uint32_t Co, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo, uint32_t tileLen)
    {
        N_ = N; Ci_ = Ci; Hi_ = Hi; Wi_ = Wi;
        Co_ = Co; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;
        tileLen_ = tileLen;

        const uint64_t xSize = static_cast<uint64_t>(N_) * Ci_ * Hi_ * Wi_;
        const uint64_t wSize = static_cast<uint64_t>(Ci_) * Co_ * Kh_ * Kw_;
        const uint64_t ySize = static_cast<uint64_t>(N_) * Co_ * Ho_ * Wo_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint32_t)xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint32_t)wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint32_t)ySize);

        pipe_.InitBuffer(tmpQueue_, TMP_BUFFERS, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Ci_ == 0 || Co_ == 0 || Kh_ == 0 || Kw_ == 0 || Ho_ == 0 || Wo_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());
        const uint64_t totalOut = static_cast<uint64_t>(N_) * Co_ * Ho_ * Wo_;

        for (uint64_t yIndex = blockIdx; yIndex < totalOut; yIndex += blockNum) {
            ComputeSingleOutput(yIndex);
        }
    }

private:
    __aicore__ inline void ComputeSingleOutput(uint64_t yIndex)
    {
        // Decode yIndex -> (n, co, ho, wo)
        uint64_t t = yIndex;
        const uint32_t wo = static_cast<uint32_t>(t % Wo_); t /= Wo_;
        const uint32_t ho = static_cast<uint32_t>(t % Ho_); t /= Ho_;
        const uint32_t co = static_cast<uint32_t>(t % Co_); t /= Co_;
        const uint32_t n  = static_cast<uint32_t>(t);

        bool hasInit = false;
        LocalTensor<float> yAcc; // accumulation buffer (length 1)

        for (uint32_t ci = 0; ci < Ci_; ++ci) {
            for (uint32_t kh = 0; kh < Kh_; ++kh) {
                // h_in = ho - kh
                int32_t h_in = static_cast<int32_t>(ho) - static_cast<int32_t>(kh);
                if (h_in < 0 || h_in >= static_cast<int32_t>(Hi_)) continue;

                for (uint32_t kw = 0; kw < Kw_; ++kw) {
                    int32_t w_in = static_cast<int32_t>(wo) - static_cast<int32_t>(kw);
                    if (w_in < 0 || w_in >= static_cast<int32_t>(Wi_)) continue;

                    const uint64_t xOff = (((static_cast<uint64_t>(n) * Ci_ + ci) * Hi_ + static_cast<uint32_t>(h_in)) * Wi_ + static_cast<uint32_t>(w_in));
                    const uint64_t wOff = (((static_cast<uint64_t>(ci) * Co_ + co) * Kh_ + kh) * Kw_ + kw);

                    LocalTensor<float> xLocal = tmpQueue_.AllocTensor<float>();
                    LocalTensor<float> wLocal = tmpQueue_.AllocTensor<float>();
                    DataCopy(xLocal, xGm_[xOff], 1);
                    DataCopy(wLocal, wGm_[wOff], 1);

                    if (!hasInit) {
                        yAcc = tmpQueue_.AllocTensor<float>();
                        Mul(yAcc, xLocal, wLocal, 1);
                        hasInit = true;
                    } else {
                        LocalTensor<float> prodLocal = tmpQueue_.AllocTensor<float>();
                        Mul(prodLocal, xLocal, wLocal, 1);
                        Add(yAcc, yAcc, prodLocal, 1);
                        tmpQueue_.FreeTensor(prodLocal);
                    }

                    tmpQueue_.FreeTensor(wLocal);
                    tmpQueue_.FreeTensor(xLocal);
                }
            }
        }

        // Write back
        const uint64_t yOff = yIndex;
        DataCopy(yGm_[yOff], yAcc, 1);
        tmpQueue_.FreeTensor(yAcc);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, TMP_BUFFERS> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t N_{0}, Ci_{0}, Hi_{0}, Wi_{0};
    uint32_t Co_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t tileLen_{1};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling_data, tilingGm);
    KernelConvTransposed2dAsymInAsymKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Ci, tiling_data.Hi, tiling_data.Wi,
            tiling_data.Co, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo, tiling_data.tileLen);
    op.Process();
}
