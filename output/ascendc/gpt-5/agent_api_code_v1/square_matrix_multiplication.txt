#include "square_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
}  // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
// UB usage per tile (floats):
//   A_tile: tileM * tileK
//   B_tile: tileK * tileN
//   C_tile: tileM * tileN
//   tmp_y:  tileN
//   tmp_m:  tileN
// Total floats = tileM*tileK + tileK*tileN + tileM*tileN + 2*tileN
// We'll use defaults tileM=64, tileN=64, tileK=256 if fits.
static void ChooseTiles(uint32_t N, uint32_t &tileM, uint32_t &tileN, uint32_t &tileK)
{
    uint32_t candM = 64;
    uint32_t candN = 64;
    uint32_t candK = 256;

    // If N is small, shrink tiles to N-aligned
    if (N < candM) candM = (N >= 16) ? AlignDown(N, 16) : N;
    if (N < candN) candN = (N >= 16) ? AlignDown(N, 16) : N;
    if (N < candK) candK = (N >= 16) ? AlignDown(N, 16) : N;

    // Check UB budget in bytes
    uint64_t floats = (uint64_t)candM * candK + (uint64_t)candK * candN + (uint64_t)candM * candN + 2ull * candN;
    uint64_t bytes = floats * 4ull;
    if (bytes > UB_BUDGET_BYTES) {
        // Reduce K aggressively first
        while (bytes > UB_BUDGET_BYTES && candK > 64) {
            candK = AlignDown(candK / 2, 16);
            floats = (uint64_t)candM * candK + (uint64_t)candK * candN + (uint64_t)candM * candN + 2ull * candN;
            bytes = floats * 4ull;
        }
        // Then reduce M/N together
        while (bytes > UB_BUDGET_BYTES && (candM > 32 || candN > 32)) {
            if (candM > 32) candM = AlignDown(candM / 2, 16);
            if (candN > 32) candN = AlignDown(candN / 2, 16);
            floats = (uint64_t)candM * candK + (uint64_t)candK * candN + (uint64_t)candM * candN + 2ull * candN;
            bytes = floats * 4ull;
        }
    }

    if (candM == 0) candM = 16;
    if (candN == 0) candN = 16;
    if (candK == 0) candK = 16;

    tileM = candM;
    tileN = candN;
    tileK = candK;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *ashape = context->GetInputShape(0);
    const gert::Shape *bshape = context->GetInputShape(1);
    if (ashape->GetDimNum() != 2 || bshape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const int64_t a0 = ashape->GetDim(0);
    const int64_t a1 = ashape->GetDim(1);
    const int64_t b0 = bshape->GetDim(0);
    const int64_t b1 = bshape->GetDim(1);
    if (a0 <= 0 || a1 <= 0 || b0 <= 0 || b1 <= 0) {
        return ge::GRAPH_FAILED;
    }
    if (!(a0 == a1 && b0 == b1 && a1 == b0)) {  // ensure square and compatible
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(a0);

    uint32_t tileM = 64, tileN = 64, tileK = 256;
    ChooseTiles(N, tileM, tileN, tileK);

    const uint32_t rowTiles = CeilDiv(N, tileM);
    const uint32_t colTiles = CeilDiv(N, tileN);
    const uint32_t totalTiles = rowTiles * colTiles;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    SquareMatrixMultiplicationCustomTilingData tiling;
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext *ctx)
{
    const gert::Shape *ashape = ctx->GetInputShape(0);
    const gert::Shape *bshape = ctx->GetInputShape(1);
    gert::Shape *cshape = ctx->GetOutputShape(0);

    // Expect (N,N) x (N,N) -> (N,N)
    if (ashape->GetDimNum() != 2 || bshape->GetDimNum() != 2) return GRAPH_FAILED;
    if (ashape->GetDim(1) != bshape->GetDim(0) || ashape->GetDim(0) != ashape->GetDim(1) || bshape->GetDim(0) != bshape->GetDim(1)) {
        return GRAPH_FAILED;

    cshape->SetDimNum(2);
    cshape->SetDim(0, ashape->GetDim(0));
    cshape->SetDim(1, bshape->GetDim(1));
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelSquareMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelSquareMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t N, uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        N_ = N;
        tM_ = tileM;
        tN_ = tileN;
        tK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)a, static_cast<uint32_t>(N_) * static_cast<uint32_t>(N_));
        bGm_.SetGlobalBuffer((__gm__ float*)b, static_cast<uint32_t>(N_) * static_cast<uint32_t>(N_));
        cGm_.SetGlobalBuffer((__gm__ float*)c, static_cast<uint32_t>(N_) * static_cast<uint32_t>(N_));

        // UB buffers
        pipe_.InitBuffer(aTileQ_, BUFFER_NUM, tM_ * tK_ * sizeof(float));
        pipe_.InitBuffer(bTileQ_, BUFFER_NUM, tK_ * tN_ * sizeof(float));
        pipe_.InitBuffer(cTileQ_, BUFFER_NUM, tM_ * tN_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 2, tN_ * sizeof(float)); // yVec and mulVec
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t rowTiles = CeilDiv(N_, tM_);
        const uint32_t colTiles = CeilDiv(N_, tN_);
        const uint32_t totalTiles = rowTiles * colTiles;

        for (uint32_t tileIndex = blockIdx; tileIndex < totalTiles; tileIndex += blockNum) {
            uint32_t tileRow = tileIndex / colTiles;
            uint32_t tileCol = tileIndex % colTiles;

            uint32_t row0 = tileRow * tM_;
            uint32_t col0 = tileCol * tN_;
            uint32_t tm = N_ - row0;
            if (tm > tM_) tm = tM_;
            uint32_t tn = N_ - col0;
            if (tn > tN_) tn = tN_;

            // Allocate C tile in UB
            AscendC::LocalTensor<float> cTile = cTileQ_.AllocTensor<float>();

            bool firstK = true;
            for (uint32_t kBase = 0; kBase < N_; kBase += tK_) {
                uint32_t tk = N_ - kBase;
                if (tk > tK_) tk = tK_;

                // Load A tile [tm x tk] row by row
                AscendC::LocalTensor<float> aTile = aTileQ_.AllocTensor<float>();
                for (uint32_t i = 0; i < tm; ++i) {
                    uint32_t gmOff = (row0 + i) * N_ + kBase;
                    AscendC::DataCopy(aTile[i * tK_], aGm_[gmOff], tk);
                }

                // Load B tile [tk x tn] row by row
                AscendC::LocalTensor<float> bTile = bTileQ_.AllocTensor<float>();
                for (uint32_t rk = 0; rk < tk; ++rk) {
                    uint32_t gmOff = (kBase + rk) * N_ + col0;
                    AscendC::DataCopy(bTile[rk * tN_], bGm_[gmOff], tn);
                }

                // Compute C partial sums
                for (uint32_t i = 0; i < tm; ++i) {
                    AscendC::LocalTensor<float> yVec = tmpQ_.AllocTensor<float>();
                    AscendC::LocalTensor<float> mulVec = tmpQ_.AllocTensor<float>();

                    // Accumulate over tk dimension
                    for (uint32_t rk = 0; rk < tk; ++rk) {
                        // bRow vector
                        AscendC::LocalTensor<float> bRow = bTile[rk * tN_];
                        // scalar aVal from aTile
                        float aVal = aTile.GetValue(i * tK_ + rk);
                        // mulVec = bRow * aVal
                        AscendC::Muls(mulVec, bRow, aVal, tn);
                        if (rk == 0 && firstK) {
                            // initialize yVec from mulVec
                            AscendC::DataCopy(yVec, mulVec, tn);
                        } else if (rk == 0 && !firstK) {
                            // initialize yVec as zeros then will add to existing cTile row later
                            AscendC::Muls(yVec, mulVec, 1.0f, tn);
                        } else {
                            AscendC::Add(yVec, yVec, mulVec, tn);
                        }
                    }

                    // Write back to cTile
                    AscendC::LocalTensor<float> cRow = cTile[i * tN_];
                    if (firstK) {
                        AscendC::DataCopy(cRow, yVec, tn);
                    } else {
                        AscendC::Add(cRow, cRow, yVec, tn);
                    }

                    tmpQ_.FreeTensor(mulVec);
                    tmpQ_.FreeTensor(yVec);
                }

                aTileQ_.FreeTensor(aTile);
                bTileQ_.FreeTensor(bTile);
                firstK = false;
            }

            // Store C tile to GM row by row
            for (uint32_t i = 0; i < tm; ++i) {
                uint32_t gmOff = (row0 + i) * N_ + col0;
                AscendC::LocalTensor<float> cRow = cTile[i * tN_];
                AscendC::DataCopy(cGm_[gmOff], cRow, tn);
            }

            cTileQ_.FreeTensor(cTile);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> aTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> cTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_ {0};
    uint32_t tM_ {0};
    uint32_t tN_ {0};
    uint32_t tK_ {0};
};

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);
    KernelSquareMatrixMultiplicationCustom op;
    op.Init(A, B, C, t.N, t.tileM, t.tileN, t.tileK);
    op.Process();
}
