host_operator_src="""

#include "conv2d_mish_mish_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (float):
//   yLocal:   1 * tileOW
//   xRow:     1 * tileOW
//   tmp0:     1 * tileOW
//   tmp1:     1 * tileOW
// Total floats ~ 4 * tileOW => bytes ~ 16 * tileOW. Add headroom -> use 20 * tileOW.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // favor DMA alignment
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // limit per-iter

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMishMishCustomTilingData tiling;

    // Input shapes: x [N, CIn, H, W], weight [COut, CIn, K, K], conv_bias [COut] optional
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const bool hasBias = (context->NumInputs() >= 3) && (context->GetInputShape(2)->GetShapeSize() > 0);

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t CIn = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t COut = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t K    = static_cast<uint32_t>(w_shape->GetDim(2)); // assuming square kernel

    // Stride=1, Padding=0, Dilation=1 per problem statement
    const uint32_t OH = (H >= K) ? (H - K + 1) : 0;
    const uint32_t OW = (W >= K) ? (W - K + 1) : 0;

    // Choose tile size along OW
    uint32_t maxTileOW = UB_BUDGET_BYTES / 20;  // element budget
    maxTileOW = AlignDown(maxTileOW, ALIGN_ELEMS);
    if (maxTileOW < ALIGN_ELEMS) maxTileOW = ALIGN_ELEMS;
    if (maxTileOW > HARD_CAP_ELEMS) maxTileOW = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileOW = maxTileOW;
    if (OW < tileOW) {
        const uint32_t aligned = AlignDown(OW, ALIGN_ELEMS);
        tileOW = (aligned > 0) ? aligned : OW;
    }

    // blockDim based on number of (N, COut, OH) rows to process
    const uint64_t rows64 = static_cast<uint64_t>(N) * static_cast<uint64_t>(COut) * static_cast<uint64_t>(OH);
    uint32_t rows = rows64 > 0xFFFFFFFFull ? 0xFFFFFFFFu : static_cast<uint32_t>(rows64);
    uint32_t blockDim = rows == 0 ? 1u : rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_CIn(CIn);
    tiling.set_COut(COut);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_K(K);
    tiling.set_OH(OH);
    tiling.set_OW(OW);
    tiling.set_tileOW(tileOW);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1u : 0u);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConv2dMishMishCustom {
public:
    __aicore__ inline KernelConv2dMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t CIn, uint32_t COut,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t OH, uint32_t OW, uint32_t tileOW,
                                bool hasBias)
    {
        N_ = N; CIn_ = CIn; COut_ = COut; H_ = H; W_ = W; K_ = K; OH_ = OH; OW_ = OW;
        tileOW_ = tileOW; hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * CIn_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)COut_ * CIn_ * K_ * K_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, COut_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * COut_ * OH_ * OW_);

        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileOW_ * sizeof(float));
        pipe_.InitBuffer(xRowQueue_, BUFFER_NUM, tileOW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileOW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (OH_ == 0 || OW_ == 0 || tileOW_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint64_t totalRows = (uint64_t)N_ * (uint64_t)COut_ * (uint64_t)OH_;
        if (totalRows == 0) return;

        // Grid-stride over rows: (n, oc, oh)
        for (uint64_t row = blockIdx; row < totalRows; row += blockNum) {
            uint32_t n, oc, oh;
            UnflattenRow(row, n, oc, oh);

            for (uint32_t owBase = 0; owBase < OW_; owBase += tileOW_) {
                uint32_t len = OW_ - owBase;
                if (len > tileOW_) len = tileOW_;

                // yLocal = 0
                AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, len);

                // Accumulate over input channels and kernel
                for (uint32_t ic = 0; ic < CIn_; ++ic) {
                    for (uint32_t kh = 0; kh < K_; ++kh) {
                        const uint32_t ih = oh + kh;
                        for (uint32_t kw = 0; kw < K_; ++kw) {
                            const uint32_t iwBase = owBase + kw;

                            // Load input row slice: x[n, ic, ih, iwBase : iwBase + len]
                            AscendC::LocalTensor<float> xRow = xRowQueue_.AllocTensor<float>();
                            const uint64_t xOffset = OffsetX(n, ic, ih, iwBase);
                            AscendC::DataCopy(xRow, xGm_[xOffset], len);

                            // Load weight scalar w[oc, ic, kh, kw] into a local scalar then replicate
                            AscendC::LocalTensor<float> tmp0 = tmpQueue_.AllocTensor<float>();
                            AscendC::LocalTensor<float> tmp1 = tmpQueue_.AllocTensor<float>();

                            const uint64_t wOffset = OffsetW(oc, ic, kh, kw);
                            AscendC::DataCopy(tmp0, wGm_[wOffset], 1); // tmp0[0] holds weight
                            // Broadcast weight value in tmp0[0] into tmp1[0:len]
                            AscendC::Dup(tmp1, tmp0[0], len); // assume Dup can take scalar from tmp0[0]

                            // Multiply and accumulate: yLocal += xRow * wVec
                            AscendC::Mul(tmp0, xRow, tmp1, len);
                            AscendC::Add(yLocal, yLocal, tmp0, len);

                            tmpQueue_.FreeTensor(tmp1);
                            tmpQueue_.FreeTensor(tmp0);
                            xRowQueue_.FreeTensor(xRow);
                        }
                    }
                }

                // Add bias if present: yLocal += bias[oc]
                if (hasBias_) {
                    AscendC::LocalTensor<float> tmp0 = tmpQueue_.AllocTensor<float>();
                    AscendC::LocalTensor<float> tmp1 = tmpQueue_.AllocTensor<float>();
                    AscendC::DataCopy(tmp0, bGm_[oc], 1); // tmp0[0] = bias
                    AscendC::Dup(tmp1, tmp0[0], len);
                    AscendC::Add(yLocal, yLocal, tmp1, len);
                    tmpQueue_.FreeTensor(tmp1);
                    tmpQueue_.FreeTensor(tmp0);
                }

                // Apply Mish twice in-place on yLocal using two temporaries
                ApplyMishInplace(yLocal, len);
                ApplyMishInplace(yLocal, len);

                // Store back
                const uint64_t yOffset = OffsetY(n, oc, oh, owBase);
                AscendC::DataCopy(yGm_[yOffset], yLocal, len);
                yQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline void UnflattenRow(uint64_t row, uint32_t &n, uint32_t &oc, uint32_t &oh) const
    {
        const uint64_t perOC = (uint64_t)OH_;
        const uint64_t perN  = (uint64_t)COut_ * perOC;
        n  = (uint32_t)(row / perN);
        uint64_t rem = row % perN;
        oc = (uint32_t)(rem / perOC);
        oh = (uint32_t)(rem % perOC);
    }

    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t ic, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * CIn_ + ic) * H_ + h) * W_ + w;
    }

    __aicore__ inline uint64_t OffsetW(uint32_t oc, uint32_t ic, uint32_t kh, uint32_t kw) const
    {
        return (((uint64_t)oc * CIn_ + ic) * K_ + kh) * K_ + kw;
    }

    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t oc, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * COut_ + oc) * OH_ + h) * OW_ + w;
    }

    // Mish: y = x * tanh(softplus(x)), with tanh(softplus(x)) = (a^2 - 1) / (a^2 + 1), where a = 1 + exp(x)
    __aicore__ inline void ApplyMishInplace(AscendC::LocalTensor<float>& x, uint32_t len)
    {
        AscendC::LocalTensor<float> tmp0 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp1 = tmpQueue_.AllocTensor<float>();

        // tmp0 = exp(x)
        AscendC::Exp(tmp0, x, len);
        // tmp0 = 1 + exp(x)
        AscendC::Adds(tmp0, tmp0, 1.0f, len);
        // tmp1 = (1 + exp(x))^2
        AscendC::Mul(tmp1, tmp0, tmp0, len);
        // tmp0 = den = a2 + 1
        AscendC::Adds(tmp0, tmp1, 1.0f, len);
        // tmp1 = num = a2 - 1
        AscendC::Adds(tmp1, tmp1, -1.0f, len);
        // tmp1 = t = num / den
        AscendC::Div(tmp1, tmp1, tmp0, len);
        // x = x * t
        AscendC::Mul(x, x, tmp1, len);

        tmpQueue_.FreeTensor(tmp1);
        tmpQueue_.FreeTensor(tmp0);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, CIn_{0}, COut_{0}, H_{0}, W_{0}, K_{0}, OH_{0}, OW_{0}, tileOW_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMishMishCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.N, tiling_data.CIn, tiling_data.COut,
            tiling_data.H, tiling_data.W, tiling_data.K,
            tiling_data.OH, tiling_data.OW, tiling_data.tileOW,
            tiling_data.hasBias != 0);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor conv2d_mish_mish_impl_npu(const at::Tensor& x,
                                            const at::Tensor& weight,
                                            c10::optional<at::Tensor> bias_opt)
{
    const auto N = x.size(0);
    const auto COut = weight.size(0);
    const auto H = x.size(2);
    const auto W = x.size(3);
    const auto K = weight.size(2);
    const auto OH = H - K + 1;
    const auto OW = W - K + 1;
    at::Tensor y = at::empty({N, COut, OH, OW}, x.options());

    if (bias_opt.has_value()) {
        EXEC_NPU_CMD(aclnnConv2dMishMishCustom, x, weight, bias_opt.value(), y);
    } else {
        at::Tensor empty_bias;
        EXEC_NPU_CMD(aclnnConv2dMishMishCustom, x, weight, empty_bias, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_mish_mish_custom", &conv2d_mish_mish_impl_npu);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_mish_mish_custom", &conv2d_mish_mish_impl_npu,
