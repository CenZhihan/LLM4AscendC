host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (conservative):
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue: 1 * tile (scratch/reduce buffer upper bound)
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap per-iter work

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    const gert::Shape &shape = context->GetInputShape(0)->GetOriginShape();
    const uint64_t rank = shape.GetDimNum();
    uint64_t totalLength64 = shape.GetShapeSize();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t axis = dimPtr ? *dimPtr : 1;

    // Normalize axis
    int64_t nAxis = axis;
    if (nAxis < 0) nAxis += static_cast<int64_t>(rank);
    if (nAxis < 0 || static_cast<uint64_t>(nAxis) >= rank) {
        return ge::GRAPH_FAILED;
    }

    uint64_t lastDim64 = static_cast<uint64_t>(shape.GetDim(static_cast<uint64_t>(nAxis)));
    if (lastDim64 == 0) {
        // empty tensor, trivial tiling
        tiling.set_totalLength(0);
        tiling.set_lastDim(0);
        tiling.set_tileLength(0);
        tiling.set_blockDim(1);
        context->SetBlockDim(1);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    const uint32_t totalLength = static_cast<uint32_t>(totalLength64);
    const uint32_t lastDim = static_cast<uint32_t>(lastDim64);
    const uint32_t outer = (lastDim == 0) ? 0u : (totalLength / lastDim);

    // tileLen from UB budget: bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (lastDim < tileLen) {
        const uint32_t aligned = AlignDown(lastDim, ALIGN_ELEMS);
        tileLen = (aligned > 0u) ? aligned : lastDim;
    }
    if (tileLen == 0u) tileLen = lastDim; // degenerate tiny row

    // Adaptive blockDim based on number of rows; conservative bounds for stability.
    uint32_t blockDim = outer;
    if (blockDim < 8u) blockDim = (outer == 0u) ? 1u : outer;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim == 0u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_lastDim(lastDim);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;

    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t lastDim,
                                uint32_t tileLength)
    {
        totalLen_ = totalLength;
        lastDim_ = lastDim;
        tileLen_ = tileLength;
        outerRows_ = (lastDim_ == 0u) ? 0u : (totalLen_ / lastDim_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Buffers: in (2*tile), out (2*tile), tmp (1*tile) - conservative
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        // A tiny scalar buffer for reductions
        pipe_.InitBuffer(scalarQueue_, 1, 4 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || lastDim_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < outerRows_; row += blockNum) {
            const uint32_t rowBase = row * lastDim_;

            // Pass 1: compute row max
            float rowMax = -3.402823e+38f; // -FLT_MAX
            for (uint32_t col = 0; col < lastDim_; col += tileLen_) {
                uint32_t len = lastDim_ - col;
                if (len > tileLen_) len = tileLen_;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + col], len);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> red = scalarQueue_.AllocTensor<float>();
                AscendC::ReduceMax(red, xLocal, len);
                float v = red.GetValue(0);
                if (v > rowMax) rowMax = v;
                scalarQueue_.FreeTensor(red);
                inQueueX_.FreeTensor(xLocal);
            }

            // Pass 2: compute sum of exp(x - rowMax)
            float sumExp = 0.0f;
            for (uint32_t col = 0; col < lastDim_; col += tileLen_) {
                uint32_t len = lastDim_ - col;
                if (len > tileLen_) len = tileLen_;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + col], len);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();
                // xLocal = xLocal - rowMax
                AscendC::Adds(xLocal, xLocal, -rowMax, len);
                // exp in-place
                AscendC::Exp(xLocal, xLocal, len);
                AscendC::LocalTensor<float> red = scalarQueue_.AllocTensor<float>();
                AscendC::ReduceSum(red, xLocal, len);
                float partial = red.GetValue(0);
                sumExp += partial;
                scalarQueue_.FreeTensor(red);
                inQueueX_.FreeTensor(xLocal);
            }

            // log(sumExp)
            float logSumExp = logf(sumExp);

            // Pass 3: write y = x - rowMax - logSumExp
            for (uint32_t col = 0; col < lastDim_; col += tileLen_) {
                uint32_t len = lastDim_ - col;
                if (len > tileLen_) len = tileLen_;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + col], len);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();
                // yLocal = xLocal - rowMax
                AscendC::Adds(yLocal, xLocal, -rowMax, len);
                // yLocal = yLocal - logSumExp
                AscendC::Adds(yLocal, yLocal, -logSumExp, len);

                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xLocal);

                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[rowBase + col], yLocal, len);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t lastDim_{0};
    uint32_t tileLen_{0};
    uint32_t outerRows_{0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.lastDim, tiling_data.tileLength);
    op.Process();
}
"""
