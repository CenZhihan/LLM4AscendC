host_operator_src="""

#include "matmul_with_irregular_shapes_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative UB budget
static constexpr uint32_t ALIGN_BLOCK = 16;              // favor vec alignment

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithIrregularShapesCustomTilingData tiling;

    // Input A: (M, K), Input B: (K, N)
    const auto *shapeA = context->GetInputShape(0);
    const auto *shapeB = context->GetInputShape(1);

    if (shapeA->GetDimNum() != 2 || shapeB->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(shapeA->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(shapeA->GetDim(1));
    const uint32_t N = static_cast<uint32_t>(shapeB->GetDim(1));

    // Choose initial tile sizes
    uint32_t tileM = (M >= 128) ? 128u : AlignDown(M, ALIGN_BLOCK);
    if (tileM == 0) tileM = M;
    uint32_t tileN = (N >= 128) ? 128u : AlignDown(N, ALIGN_BLOCK);
    if (tileN == 0) tileN = N;
    uint32_t tileK = (K >= 64) ? 64u : AlignDown(K, ALIGN_BLOCK);
    if (tileK == 0) tileK = K;

    // Ensure memory fit: bytes = 2*(tileM*tileK + tileK*tileN) + 4*(tileM*tileN) + margin
    auto MemBytes = [](uint32_t tm, uint32_t tn, uint32_t tk) -> uint64_t {
        uint64_t aBytes = static_cast<uint64_t>(tm) * tk * 2u;
        uint64_t bBytes = static_cast<uint64_t>(tk) * tn * 2u;
        uint64_t cBytes = static_cast<uint64_t>(tm) * tn * 4u;
        uint64_t tmpBytes = static_cast<uint64_t>(tn) * 4u; // one temp row in float
        return aBytes + bBytes + cBytes + tmpBytes + 1024u; // margin
    };

    while (MemBytes(tileM, tileN, tileK) > UB_BUDGET_BYTES) {
        if (tileN > ALIGN_BLOCK) tileN = AlignDown(tileN / 2, ALIGN_BLOCK);
        else if (tileM > ALIGN_BLOCK) tileM = AlignDown(tileM / 2, ALIGN_BLOCK);
        else if (tileK > ALIGN_BLOCK) tileK = AlignDown(tileK / 2, ALIGN_BLOCK);
        else break;
        if (tileM == 0) tileM = 1;
        if (tileN == 0) tileN = 1;
        if (tileK == 0) tileK = 1;
    }

    const uint32_t numTilesM = CeilDiv(M, tileM);
    const uint32_t numTilesN = CeilDiv(N, tileN);
    const uint32_t totalTiles = numTilesM * numTilesN;

    // Choose blockDim conservatively
    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_numTilesM(numTilesM);
    tiling.set_numTilesN(numTilesN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0; // no workspace needed
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0); // (M, K)
    const gert::Shape *b_shape = context->GetInputShape(1); // (K, N)
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = a_shape->GetDim(0);
    const int64_t K = a_shape->GetDim(1);
    const int64_t Kb = b_shape->GetDim(0);
    const int64_t N = b_shape->GetDim(1);
    if (K != Kb) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

{

}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulWithIrregularShapesCustom {
public:
    __aicore__ inline KernelMatmulWithIrregularShapesCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                uint32_t numTilesM, uint32_t numTilesN)
    {
        M_ = M; N_ = N; K_ = K;
        tM_ = tileM; tN_ = tileN; tK_ = tileK;
        tilesM_ = numTilesM; tilesN_ = numTilesN;

        aGm_.SetGlobalBuffer((__gm__ half*)a, static_cast<uint64_t>(M_) * K_);
        bGm_.SetGlobalBuffer((__gm__ half*)b, static_cast<uint64_t>(K_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, static_cast<uint64_t>(M_) * N_);

        pipe_.InitBuffer(aQueue_, BUFFER_NUM, static_cast<uint32_t>(tM_ * tK_ * sizeof(half)));
        pipe_.InitBuffer(bQueue_, BUFFER_NUM, static_cast<uint32_t>(tK_ * tN_ * sizeof(half)));
        pipe_.InitBuffer(cAccQueue_, BUFFER_NUM, static_cast<uint32_t>(tM_ * tN_ * sizeof(float)));
        pipe_.InitBuffer(tmpRowQueue_, BUFFER_NUM, static_cast<uint32_t>(tN_ * sizeof(float)));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t totalTiles = tilesM_ * tilesN_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t tileRow = tileId / tilesN_;
            const uint32_t tileCol = tileId % tilesN_;
            const uint32_t m0 = tileRow * tM_;
            const uint32_t n0 = tileCol * tN_;

            uint32_t curTM = M_ - m0; if (curTM > tM_) curTM = tM_;
            uint32_t curTN = N_ - n0; if (curTN > tN_) curTN = tN_;

            // Initialize local accumulator C tile to zeros
            AscendC::LocalTensor<float> cAcc = cAccQueue_.AllocTensor<float>();
            AscendC::Dup(cAcc, 0.0f, curTM * curTN);

            // Iterate over K dimension in tiles
            for (uint32_t k0 = 0; k0 < K_; k0 += tK_) {
                uint32_t curTK = K_ - k0; if (curTK > tK_) curTK = tK_;

                // Load A_sub (curTM x curTK) row by row
                AscendC::LocalTensor<half> aSub = aQueue_.AllocTensor<half>();
                for (uint32_t r = 0; r < curTM; ++r) {
                    AscendC::DataCopy(aSub[r * curTK], aGm_[(m0 + r) * K_ + k0], curTK);
                }

                // Load B_sub (curTK x curTN) row by row
                AscendC::LocalTensor<half> bSub = bQueue_.AllocTensor<half>();
                for (uint32_t r = 0; r < curTK; ++r) {
                    AscendC::DataCopy(bSub[r * curTN], bGm_[(k0 + r) * N_ + n0], curTN);
                }

                // Compute C_acc += A_sub * B_sub using outer-product
                // For each kk in current TK
                for (uint32_t kk = 0; kk < curTK; ++kk) {
                    // Convert B_row(kk) from half to float once
                    AscendC::LocalTensor<float> bRowF = tmpRowQueue_.AllocTensor<float>();
                    AscendC::Cast(bRowF, bSub[kk * curTN], AscendC::DT_FLOAT, curTN);

                    // For each row i of A_sub and C_acc
                    for (uint32_t i = 0; i < curTM; ++i) {
                        // Load scalar a(i, kk)
                        half a_h = aSub[i * curTK + kk];
                        float a_f = static_cast<float>(a_h);

                        // yRow = C_acc[i, :]
                        AscendC::LocalTensor<float> yRow = cAcc[i * curTN];

                        // tmp = a_f * bRowF
                        AscendC::Muls(bRowF, bRowF, a_f, curTN);
                        // yRow += tmp
                        AscendC::Add(yRow, yRow, bRowF, curTN);
                        // Restore bRowF by dividing back is unnecessary as we will overwrite bRowF next kk
                    }

                    tmpRowQueue_.FreeTensor(bRowF);
                }

                // Free A and B sub-tiles
                aQueue_.FreeTensor(aSub);
                bQueue_.FreeTensor(bSub);
            }

            // Store C_acc to GM: (curTM x curTN)
            for (uint32_t r = 0; r < curTM; ++r) {
                AscendC::DataCopy(cGm_[(m0 + r) * N_ + n0], cAcc[r * curTN], curTN);
            }
            cAccQueue_.FreeTensor(cAcc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> cAccQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpRowQueue_;

    AscendC::GlobalTensor<half> aGm_;
    AscendC::GlobalTensor<half> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tM_{0}, tN_{0}, tK_{0};
    uint32_t tilesM_{0}, tilesN_{0};
};

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithIrregularShapesCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.numTilesM, tiling_data.numTilesN);
    op.Process();
}
