host_operator_src="""
#include "conv2d_relu_bias_add_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative per-core UB share
static constexpr uint32_t ALIGN_ELEMS_W = 16;             // align to multiples of width-friendly segment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;         // avoid very large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : ((x / a) * a);
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dReluBiasAddCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin = x_shape->GetOriginShape();
    if (origin.GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t n = static_cast<uint32_t>(origin.GetDim(0));
    const uint32_t c = static_cast<uint32_t>(origin.GetDim(1));
    const uint32_t h = static_cast<uint32_t>(origin.GetDim(2));
    const uint32_t w = static_cast<uint32_t>(origin.GetDim(3));

    const uint32_t hw = h * w;
    const uint32_t totalLength = n * c * hw;

    // UB model: inQueueX 2*tile + outQueueY 2*tile + small tmp
    // bytes ~= 8 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 8u;
    if (maxTile > HARD_CAP_ELEMS) {
        maxTile = HARD_CAP_ELEMS;
    }

    // Prefer channel-aligned tiles to simplify bias broadcast; choose H*W if fits, else align to W
    uint32_t tileLen = hw;
    if (tileLen > maxTile) {
        uint32_t aligned = AlignDown(maxTile, (w > 0 ? w : ALIGN_ELEMS_W));
        tileLen = (aligned > 0) ? aligned : (w > 0 ? w : 1);
    }

    if (tileLen == 0) {
        tileLen = 1;
    }

    const uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_hw(hw);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dReluBiasAddCustom {
public:
    __aicore__ inline KernelConv2dReluBiasAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t hw)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        n_ = n; c_ = c; h_ = h; w_ = w; hw_ = hw;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, c_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // tmp for bias scalar staging and small helpers
        pipe_.InitBuffer(tmpQueue_, 1, 1024); // small scratch
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(base, curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // ReLU
        AscendC::Maxs(yLocal, xLocal, 0.0f, len);

        // Add channel bias broadcast: process contiguous chunks within a channel map
        uint32_t processed = 0;
        while (processed < len) {
            uint32_t globalIndex = gmOffset + processed;
            uint32_t spatialOffset = globalIndex % hw_;
            uint32_t channelIndex = (globalIndex / hw_) % c_;
            uint32_t chunk = len - processed;
            uint32_t remainInChannel = hw_ - spatialOffset;
            if (chunk > remainInChannel) chunk = remainInChannel;

            // load bias[channelIndex] to local scalar
            AscendC::LocalTensor<float> biasTmp = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(biasTmp, bGm_[channelIndex], 1);
            // read scalar value from local tensor
            float bval = biasTmp.GetValue(0);
            tmpQueue_.FreeTensor(biasTmp);

            AscendC::Adds(yLocal[processed], yLocal[processed], bval, chunk);

            processed += chunk;
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> bGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t n_{0}, c_{0}, h_{0}, w_{0}, hw_{0};
};

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluBiasAddCustom op;
    op.Init(x, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w, tiling_data.hw);
    op.Process();
}
