kernel_src="""

#include "kernel_operator.h"
using namespace AscendC;

class KernelConv2dSubtractTanhSubtractAvgPoolCustom {
public:
    __aicore__ inline KernelConv2dSubtractTanhSubtractAvgPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t K,
                                uint32_t H, uint32_t W,
                                uint32_t R, uint32_t S,
                                uint32_t OH, uint32_t OW,
                                uint32_t OHP, uint32_t OWP,
                                uint32_t totalPooled,
                                uint32_t tilePooled,
                                uint32_t poolK,
                                float sub1, float sub2)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N*C*H*W);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)K*C*R*S);
        if (b != nullptr) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)K);
            hasBias_ = true;
        } else {
            hasBias_ = false;
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N*K*OHP*OWP);

        N_ = N; C_ = C; K_ = K;
        H_ = H; W_ = W;
        R_ = R; S_ = S;
        OH_ = OH; OW_ = OW;
        OHP_ = OHP; OWP_ = OWP;
        totalPooled_ = totalPooled;
        tilePooled_ = tilePooled;
        poolK_ = poolK;
        sub1_ = sub1;
        sub2_ = sub2;

        // Init small UB buffers
        pipe_.InitBuffer(xRowQ_, 1, S_ * sizeof(float));
        pipe_.InitBuffer(wRowQ_, 1, S_ * sizeof(float));
        pipe_.InitBuffer(tmpInQ_, 1, sizeof(float));
        pipe_.InitBuffer(tmpOutQ_, 1, sizeof(float));
        pipe_.InitBuffer(outScalarQ_, 1, sizeof(float));
        // Optionally bias scalar cache
        if (hasBias_) {
            pipe_.InitBuffer(biasQ_, 1, sizeof(float));
        }
    }

    __aicore__ inline void Process()
    {
        if (totalPooled_ == 0) return;

        const uint32_t blockNum = (uint32_t)GetBlockNum();
        const uint32_t blockIdx = (uint32_t)GetBlockIdx();

        uint32_t strideElems = blockNum * tilePooled_;
        uint32_t base = blockIdx * tilePooled_;
        while (base < totalPooled_) {
            uint32_t cur = totalPooled_ - base;
            if (cur > tilePooled_) cur = tilePooled_;
            for (uint32_t i = 0; i < cur; ++i) {
                uint32_t linear = base + i;
                ComputeOne(linear);
            }
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t linearIndex)
    {
        // Decode linear index to (n, k, poh, pow)
        uint32_t pow = linearIndex % OWP_;
        uint32_t tmp = linearIndex / OWP_;
        uint32_t poh = tmp % OHP_;
        tmp = tmp / OHP_;
        uint32_t k = tmp % K_;
        uint32_t n = tmp / K_;

        float poolSum = 0.0f;
        const uint32_t poolArea = poolK_ * poolK_;

        // Optional bias load
        float biasVal = 0.0f;
        if (hasBias_) {
            LocalTensor<float> bLoc = biasQ_.AllocTensor<float>();
            DataCopy(bLoc, bGm_[k], 1);
            // Barrier is implicit in DataCopy sync
            biasVal = bLoc.GetValue(0);
            biasQ_.FreeTensor(bLoc);
        }

        for (uint32_t pi = 0; pi < poolK_; ++pi) {
            for (uint32_t pj = 0; pj < poolK_; ++pj) {
                const uint32_t oh = poh * poolK_ + pi;
                const uint32_t ow = pow * poolK_ + pj;

                float acc = biasVal;

                for (uint32_t c = 0; c < C_; ++c) {
                    for (uint32_t kh = 0; kh < R_; ++kh) {
                        // Load input row x[n,c,oh+kh, ow:ow+S]
                        LocalTensor<float> xRow = xRowQ_.AllocTensor<float>();
                        const uint64_t xOff = (((uint64_t)n*C_ + c)*H_ + (oh + kh))*(uint64_t)W_ + ow;
                        DataCopy(xRow, xGm_[xOff], S_);

                        // Load weight row w[k,c,kh, 0:S]
                        LocalTensor<float> wRow = wRowQ_.AllocTensor<float>();
                        const uint64_t wOff = (((uint64_t)k*C_ + c)*R_ + kh)*(uint64_t)S_;
                        DataCopy(wRow, wGm_[wOff], S_);

                        // Dot product of length S_
                        for (uint32_t kw = 0; kw < S_; ++kw) {
                            float xv = xRow.GetValue(kw);
                            float wv = wRow.GetValue(kw);
                            acc += xv * wv;
                        }

                        wRowQ_.FreeTensor(wRow);
                        xRowQ_.FreeTensor(xRow);
                    }
                }

                // Apply: subtract1 -> tanh -> subtract2
                // Use vector Tanh on a single element buffer
                LocalTensor<float> tIn = tmpInQ_.AllocTensor<float>();
                LocalTensor<float> tOut = tmpOutQ_.AllocTensor<float>();
                tIn.SetValue(0, acc - sub1_);
                AscendC::Tanh(tOut, tIn, 1);
                float act = tOut.GetValue(0) - sub2_;
                tmpOutQ_.FreeTensor(tOut);
                tmpInQ_.FreeTensor(tIn);

                poolSum += act;
            }
        }

        float avg = poolSum / (float)poolArea;

        // Store to output y[n,k,poh,pow]
        const uint64_t yOff = (((uint64_t)n*K_ + k)*OHP_ + poh)*(uint64_t)OWP_ + pow;
        LocalTensor<float> yLoc = outScalarQ_.AllocTensor<float>();
        yLoc.SetValue(0, avg);
        DataCopy(yGm_[yOff], yLoc, 1);
        outScalarQ_.FreeTensor(yLoc);
    }

private:
    // Global tensors
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    // Queues and pipe
    TPipe pipe_;
    TQue<TPosition::VECCALC, 1> xRowQ_;
    TQue<TPosition::VECCALC, 1> wRowQ_;
    TQue<TPosition::VECCALC, 1> tmpInQ_;
    TQue<TPosition::VECCALC, 1> tmpOutQ_;
    TQue<TPosition::VECOUT, 1> outScalarQ_;
    TQue<TPosition::VECCALC, 1> biasQ_;

    // Params
    uint32_t N_, C_, K_;
    uint32_t H_, W_;
    uint32_t R_, S_;
    uint32_t OH_, OW_;
    uint32_t OHP_, OWP_;
    uint32_t totalPooled_, tilePooled_;
    uint32_t poolK_;
    float sub1_, sub2_;
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConv2dSubtractTanhSubtractAvgPoolCustom op;
    op.Init(
        x, w, b, y,
        tiling_data.N, tiling_data.C, tiling_data.K,
        tiling_data.H, tiling_data.W,
        tiling_data.R, tiling_data.S,
        tiling_data.OH, tiling_data.OW,
        tiling_data.OHP, tiling_data.OWP,
        tiling_data.totalPooled, tiling_data.tilePooled,
        tiling_data.poolK,
        tiling_data.subtract1Value, tiling_data.subtract2Value
    );
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor ensure_bias_or_zero(const at::Tensor& maybe_bias, int64_t K, const at::Tensor& like_tensor) {
    if (maybe_bias.defined() && maybe_bias.numel() == K) {
        return maybe_bias;
    }
    // Create zeros bias on same device/dtype
    return at::zeros({K}, like_tensor.options());
}

at::Tensor conv2d_sub_tanh_sub_avgpool_npu(const at::Tensor& x,
                                           const at::Tensor& weight,
                                           const c10::optional<at::Tensor>& bias_opt,
                                           double subtract1_value,
                                           double subtract2_value,
                                           int64_t kernel_size_pool)
{
    TORCH_CHECK(x.dim() == 4, "x must be 4D NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be 4D [K,C,R,S]");
    const int64_t K = weight.size(0);
    at::Tensor bias = ensure_bias_or_zero(bias_opt.has_value() ? bias_opt.value() : at::Tensor(), K, x);

    // Infer output shape
    const int64_t N = x.size(0);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);
    const int64_t R = weight.size(2);
    const int64_t S = weight.size(3);
    TORCH_CHECK(H >= R && W >= S, "Input smaller than kernel");
    const int64_t OH = H - R + 1;
    const int64_t OW = W - S + 1;
    TORCH_CHECK(kernel_size_pool > 0, "kernel_size_pool must be > 0");
    TORCH_CHECK(OH >= kernel_size_pool && OW >= kernel_size_pool, "Pool kernel larger than conv output");
    const int64_t OHP = OH / kernel_size_pool;
    const int64_t OWP = OW / kernel_size_pool;

    at::Tensor y = at::empty({N, K, OHP, OWP}, x.options());

    EXEC_NPU_CMD(aclnnConv2dSubtractTanhSubtractAvgPoolCustom,
                 x, weight, bias,
                 static_cast<float>(subtract1_value),
                 static_cast<float>(subtract2_value),
                 static_cast<int>(kernel_size_pool),
                 y);
    return y;
}
