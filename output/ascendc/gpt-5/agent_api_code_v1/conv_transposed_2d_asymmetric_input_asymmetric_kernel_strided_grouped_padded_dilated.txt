kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvTransposed2D {
public:
    __aicore__ inline KernelConvTransposed2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t hin, uint32_t win,
                                uint32_t hout, uint32_t wout,
                                uint32_t groups, uint32_t cinPerG, uint32_t coutPerG,
                                uint32_t kH, uint32_t kW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilH, uint32_t dilW,
                                uint32_t tileWOut)
    {
        n_ = n; cin_ = cin; cout_ = cout; hin_ = hin; win_ = win;
        hout_ = hout; wout_ = wout;
        groups_ = groups; cinPerG_ = cinPerG; coutPerG_ = coutPerG;
        kH_ = kH; kW_ = kW;
        strideH_ = strideH; strideW_ = strideW; padH_ = padH; padW_ = padW;
        dilH_ = dilH; dilW_ = dilW;
        tileW_ = tileWOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cin_ * hin_ * win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, cin_ * coutPerG_ * kH_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * cout_ * hout_ * wout_);

        // Buffers: y-tile (Hout * tileW), input row (Win), weight small buf (kH*kW)
        pipe_.InitBuffer(yTileQ_, BUFFER_NUM, hout_ * tileW_ * sizeof(float));
        pipe_.InitBuffer(xRowQ_, BUFFER_NUM, win_ * sizeof(float));
        pipe_.InitBuffer(wSmallQ_, BUFFER_NUM, kH_ * kW_ * sizeof(float));
        pipe_.InitBuffer(tmpScalarQ_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        uint32_t linearTasks = n_ * cout_;
        if (linearTasks == 0) return;
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t task = blockIdx; task < linearTasks; task += blockNum) {
            const uint32_t nIdx = task / cout_;
            const uint32_t coIdx = task % cout_;
            const uint32_t gIdx = coIdx / coutPerG_;
            const uint32_t coRel = coIdx % coutPerG_;

            // For each width stripe
            for (uint32_t wStart = 0; wStart < wout_; wStart += tileW_) {
                uint32_t curTileW = wout_ - wStart;
                if (curTileW > tileW_) curTileW = tileW_;

                // Allocate and zero the y-tile
                AscendC::LocalTensor<float> yTile = yTileQ_.AllocTensor<float>();
                const uint32_t tileElems = hout_ * curTileW;
                AscendC::Dup<float>(yTile, 0.0f, tileElems);

                // Accumulate contributions from all input channels in this group
                for (uint32_t ciRel = 0; ciRel < cinPerG_; ++ciRel) {
                    const uint32_t ciIdx = gIdx * cinPerG_ + ciRel;

                    // Load weights for this (ciRel, coRel)
                    {
                        AscendC::LocalTensor<float> wSmall = wSmallQ_.AllocTensor<float>();
                        const uint32_t wOff = ((ciRel * coutPerG_ + coRel) * kH_ * kW_);
                        AscendC::DataCopy(wSmall, wGm_[wOff], kH_ * kW_);
                        wSmallQ_.EnQue<float>(wSmall);
                    }

                    // For each input row: load into UB row buffer
                    for (uint32_t hi = 0; hi < hin_; ++hi) {
                        AscendC::LocalTensor<float> xRow = xRowQ_.AllocTensor<float>();
                        // x offset: ((nIdx*cin + ciIdx)*hin + hi)*win
                        const uint32_t xOff = (((nIdx * cin_ + ciIdx) * hin_ + hi) * win_);
                        AscendC::DataCopy(xRow, xGm_[xOff], win_);
                        xRowQ_.EnQue<float>(xRow);

                        // Dequeue weights buffer for use in inner loop
                        AscendC::LocalTensor<float> wSmall = wSmallQ_.DeQue<float>();

                        // Iterate across input width
                        for (uint32_t wi = 0; wi < win_; ++wi) {
                            // Load scalar val = xRow[wi]
                            AscendC::LocalTensor<float> tmpVal = tmpScalarQ_.AllocTensor<float>();
                            AscendC::DataCopy(tmpVal, xRow[wi], 1);

                            // For each kernel tap
                            for (uint32_t kh = 0; kh < kH_; ++kh) {
                                // h_out position
                                int32_t hOut = (int32_t)hi * (int32_t)strideH_ - (int32_t)padH_ + (int32_t)kh * (int32_t)dilH_;
                                if (hOut < 0 || (uint32_t)hOut >= hout_) continue;

                                for (uint32_t kw = 0; kw < kW_; ++kw) {
                                    int32_t wOut = (int32_t)wi * (int32_t)strideW_ - (int32_t)padW_ + (int32_t)kw * (int32_t)dilW_;
                                    if (wOut < (int32_t)wStart || wOut >= (int32_t)(wStart + curTileW)) continue;

                                    const uint32_t relW = (uint32_t)wOut - wStart;
                                    const uint32_t yIdx = (uint32_t)hOut * curTileW + relW;
                                    const uint32_t wIdx = kh * kW_ + kw;

                                    // yTile[yIdx] += tmpVal[0] * wSmall[wIdx]
                                    // Load y element
                                    AscendC::LocalTensor<float> tmpY = tmpScalarQ_.AllocTensor<float>();
                                    AscendC::DataCopy(tmpY, yTile[yIdx], 1);

                                    AscendC::LocalTensor<float> tmpW = tmpScalarQ_.AllocTensor<float>();
                                    AscendC::DataCopy(tmpW, wSmall[wIdx], 1);

                                    // tmpY = tmpY + tmpVal * tmpW
                                    AscendC::Muls(tmpVal, tmpVal, tmpW, 1);
                                    AscendC::Add(tmpY, tmpY, tmpVal, 1);

                                    // Store back y element
                                    AscendC::DataCopy(yTile[yIdx], tmpY, 1);

                                    tmpScalarQ_.FreeTensor(tmpW);
                                    tmpScalarQ_.FreeTensor(tmpY);
                                }
                            }
                            tmpScalarQ_.FreeTensor(tmpVal);
                        }

                        wSmallQ_.FreeTensor(wSmall);
                        xRowQ_.FreeTensor(xRow);
                    }
                }

                // Write yTile to output GM at [nIdx, coIdx, :, wStart : wStart+curTileW]
                for (uint32_t ho = 0; ho < hout_; ++ho) {
                    const uint32_t yOff = (((nIdx * cout_ + coIdx) * hout_ + ho) * wout_) + wStart;
                    AscendC::DataCopy(yGm_[yOff], yTile[ho * curTileW], curTileW);
                }

                yTileQ_.FreeTensor(yTile);
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xRowQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wSmallQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpScalarQ_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t hin_{0}, win_{0};
    uint32_t hout_{0}, wout_{0};
    uint32_t groups_{0}, cinPerG_{0}, coutPerG_{0};
    uint32_t kH_{0}, kW_{0};
    uint32_t strideH_{0}, strideW_{0}, padH_{0}, padW_{0}, dilH_{0}, dilW_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2D op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.hin, tiling_data.win,
            tiling_data.hout, tiling_data.wout,
            tiling_data.groups, tiling_data.cinPerGroup, tiling_data.coutPerGroup,
            tiling_data.kH, tiling_data.kW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilH, tiling_data.dilW,
            tiling_data.tileWOut);
    op.Process();
}
