#include "gemm_multiply_leakyrelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model for post-fuse:
//   inQueue: 2 * tile
//   outQueue: 2 * tile
//   tmpQueue: 1 * tile
// bytes = (2 + 2 + 1) * tile * sizeof(float) = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMultiplyLeakyreluCustomTilingData tiling;

    // Shapes:
    // x: [M, K]
    // w: [N, K]
    // b: [N]
    // multiplier: scalar tensor [1]
    // negative_slope: scalar tensor [1]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 2 ||
        w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t Kx = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    uint32_t Kw = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    if (Kx != Kw) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t totalLength = M * N;

    // Choose a conservative tile length for post stage
    uint32_t maxTile = UB_BUDGET_BYTES / 20; // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(Kx);
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0; // let matmul object manage its internal workspace if any
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 2 ||
        w_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = x_shape->GetOriginShape().GetDim(0);

kernel_src="""
#include "kernel_operator.h"

// Fused GEMM + scalar multiply + LeakyReLU
// Inputs:
//   x: [M, K] float
//   w: [N, K] float  (GEMM uses transB=true => treats as [K, N])
//   b: [N] float
//   multiplier: scalar tensor [1] float
//   negative_slope: scalar tensor [1] float
// Output:
//   y: [M, N] float
//
// Tiling: post-processing iterates on flattened length M*N in tiles of tileLength.

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMultiplyLeakyreluCustom {
public:
    __aicore__ inline KernelGemmMultiplyLeakyreluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR multiplier, GM_ADDR negative_slope,
                                GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t totalLength, uint32_t tileLength)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M_ * N_);

        // Scalar params
        multGm_.SetGlobalBuffer((__gm__ float*)multiplier, 1);
        nsGm_.SetGlobalBuffer((__gm__ float*)negative_slope, 1);

        // Fetch scalar multiplier and negative slope
        {
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(tmp, multGm_, 1);
            float val = tmp.GetValue(0);
            multiplier_ = val;
            tmpQueue_.FreeTensor(tmp);
        }
        {
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(tmp, nsGm_, 1);
            float val = tmp.GetValue(0);
            negativeSlope_ = val;
            tmpQueue_.FreeTensor(tmp);
        }
        negCorr_ = negativeSlope_ - 1.0f;

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void RunGemm()
    {
        // Use built-in Matmul object to compute: y = x @ w^T + b
        // A_TYPE: x [M, K], GM ND, transA=false
        // B_TYPE: w [N, K], GM ND, transB=true (so it's treated as [K, N])
        // C_TYPE: y [M, N], GM ND
        // BIAS_TYPE: b [N], GM ND
        using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
        using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true, LayoutMode::NORMAL>;
        using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
        using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

        // The matmul object and tiling are managed by runtime. Register and invoke.
        REGIST_MATMUL_OBJ(&pipe_, GetSysWorkSpacePtr(), matmulObj_, &cubeTiling_);
        // Initialize and process GEMM. The following pseudo-calls mirror CANN samples.
        matmulObj_.SetTensorA<A_TYPE>(xGm_.GetPhyAddr(), M_, K_);
        matmulObj_.SetTensorB<B_TYPE>(wGm_.GetPhyAddr(), N_, K_);
        matmulObj_.SetTensorC<C_TYPE>(yGm_.GetPhyAddr(), M_, N_);
        matmulObj_.SetBias<BIAS_TYPE>(bGm_.GetPhyAddr(), N_);
        matmulObj_.Run(&pipe_);
    }

    __aicore__ inline void PostProcess()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }
            Compute(curLen);
            CopyOut(base, curLen);
            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> inLocal = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(inLocal, yGm_[gmOffset], len);
        inQueue_.EnQue<float>(inLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> inLocal = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> outLocal = outQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Multiply by scalar multiplier
        AscendC::Muls(outLocal, inLocal, multiplier_, len);

        // LeakyReLU: y = x + (negativeSlope - 1) * min(x, 0)
        // Use outLocal as x after multiply
        AscendC::Mins(tLocal, outLocal, 0.0f, len);
        AscendC::Muls(tLocal, tLocal, negCorr_, len);
        AscendC::Add(outLocal, outLocal, tLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        outQueue_.EnQue<float>(outLocal);
        inQueue_.FreeTensor(inLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> outLocal = outQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], outLocal, len);
        outQueue_.FreeTensor(outLocal);
    }

public:
    // public so kernel entry can set cubeTiling_ if needed
    AscendC::TCubeTiling cubeTiling_;

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> multGm_;
    AscendC::GlobalTensor<float> nsGm_;

    // Matmul object placeholder - managed by REGIST_MATMUL_OBJ
    decltype(GetSysWorkSpacePtr()) matmulObj_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};

    float multiplier_{1.0f};
    float negativeSlope_{0.01f};
    float negCorr_{-0.99f};
};

extern "C" __global__ __aicore__ void gemm_multiply_leakyrelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                                                      GM_ADDR multiplier, GM_ADDR negative_slope,
                                                                      GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMultiplyLeakyreluCustom op;
    // totalLength = M * N
    op.Init(x, w, b, multiplier, negative_slope, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.totalLength, tiling_data.tileLength);
    // GEMM stage
    op.RunGemm();
    // fused scalar multiply + LeakyReLU post-processing
    op.PostProcess();
}
