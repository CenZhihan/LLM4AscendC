host_operator_src="""

#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t AlignUp(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : ((x + a - 1u) / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// We implement a simple tiled GEMM using UB buffers:
// - xRow buffer: chunkN floats
// - wRow buffer: chunkN floats
// - outRow buffer: tileK floats
// Memory per core ~ (2 * chunkN + tileK) * 4 bytes.
// Keep chunkN conservative for stability on 910B.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t DEFAULT_CHUNKN  = 1024;      // elements
static constexpr uint32_t MIN_CHUNKN      = 256;
static constexpr uint32_t ALIGN_ELEMS     = 256;       // DMA friendly
static constexpr uint32_t DEFAULT_TILEM   = 8;
static constexpr uint32_t DEFAULT_TILEK   = 16;
static constexpr uint32_t MAX_BLOCKS      = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;

    // Shapes:
    // x: [M, N]
    // w: [K, N]
    // bias: [K]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t K = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));

    uint32_t tileM = DEFAULT_TILEM;
    uint32_t tileK = DEFAULT_TILEK;

    // Derive chunkN based on UB budget: (2 * chunkN + tileK) * 4 bytes <= budget
    uint32_t chunkN = DEFAULT_CHUNKN;
    uint64_t needed = (2ull * chunkN + tileK) * 4ull;
    if (needed > UB_BUDGET_BYTES) {
        uint64_t allowed = (UB_BUDGET_BYTES / 4ull);
        if (allowed > tileK) {
            chunkN = static_cast<uint32_t>((allowed - tileK) / 2ull);
        } else {
            chunkN = MIN_CHUNKN;
        }
        if (chunkN < MIN_CHUNKN) chunkN = MIN_CHUNKN;
    }

    // Cap chunkN by N and align down for DMA friendliness
    if (chunkN > N) {
        chunkN = N;
    }
    chunkN = AlignDown(chunkN, ALIGN_ELEMS);
    if (chunkN == 0) chunkN = (N == 0 ? 0 : std::min<uint32_t>(N, MIN_CHUNKN));

    // Boundaries for small problems
    if (M < tileM) tileM = M == 0 ? 1u : M;
    if (K < tileK) tileK = K == 0 ? 1u : K;

    // Blocks distribute across M tiles
    const uint32_t numMTiles = (tileM == 0 ? 1u : CeilDiv(M, tileM));
    uint32_t blockDim = numMTiles;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_chunkN(chunkN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // [M,N]
    const gert::Shape *w_shape = context->GetInputShape(1); // [K,N]
    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = x_shape->GetOriginShape().GetDim(0);
    const int64_t N = x_shape->GetOriginShape().GetDim(1);
    const int64_t K = w_shape->GetOriginShape().GetDim(0);
    // require N to match
    if (N != w_shape->GetOriginShape().GetDim(1)) {
        return GRAPH_FAILED;
    }
    gert::Shape *y_shape = context->GetOutputShape(0);

}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t INBUF_NUM = 1;
constexpr int32_t WBUF_NUM = 1;
constexpr int32_t OUTBUF_NUM = 1;

class KernelGemmAddReluCustom {
public:
    __aicore__ inline KernelGemmAddReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileK, uint32_t chunkN)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM == 0 ? 1u : tileM;
        tileK_ = tileK == 0 ? 1u : tileK;
        chunkN_ = chunkN == 0 ? 1u : chunkN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<size_t>(M_) * N_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<size_t>(K_) * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, K_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<size_t>(M_) * K_);

        // Initialize UB queues
        pipe_.InitBuffer(xRowQ_, INBUF_NUM, chunkN_ * sizeof(float));
        pipe_.InitBuffer(wRowQ_, WBUF_NUM, chunkN_ * sizeof(float));
        pipe_.InitBuffer(outRowQ_, OUTBUF_NUM, tileK_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numMTiles = CeilDiv(M_, tileM_);
        // grid-stride loop over tiles of the M dimension
        for (uint32_t mt = blockIdx; mt < numMTiles; mt += (blockNum == 0 ? 1u : blockNum)) {
            const uint32_t mStart = mt * tileM_;
            const uint32_t mLen = MinU32(tileM_, M_ - mStart);

            for (uint32_t kStart = 0; kStart < K_; kStart += tileK_) {
                const uint32_t kCur = MinU32(tileK_, K_ - kStart);

                for (uint32_t moff = 0; moff < mLen; ++moff) {
                    // Allocate and zero outRow buffer
                    AscendC::LocalTensor<float> outRow = outRowQ_.AllocTensor<float>();
                    // zero initialize via scalar loop
                    for (uint32_t kk = 0; kk < kCur; ++kk) {
                        outRow.SetValue(kk, 0.0f);
                    }

                    // Iterate N in chunks
                    for (uint32_t nStart = 0; nStart < N_; nStart += chunkN_) {
                        const uint32_t nCur = MinU32(chunkN_, N_ - nStart);

                        // Load x row segment for this m
                        AscendC::LocalTensor<float> xRow = xRowQ_.AllocTensor<float>();
                        const uint32_t xOff = (mStart + moff) * N_ + nStart;
                        AscendC::DataCopy(xRow, xGm_[xOff], nCur);

                        for (uint32_t koff = 0; koff < kCur; ++koff) {
                            // Load w row segment for this k
                            AscendC::LocalTensor<float> wRow = wRowQ_.AllocTensor<float>();
                            const uint32_t wOff = (kStart + koff) * N_ + nStart;
                            AscendC::DataCopy(wRow, wGm_[wOff], nCur);

                            // Dot accumulate into outRow[koff]
                            float sum = 0.0f;
                            for (uint32_t i = 0; i < nCur; ++i) {
                                sum += xRow.GetValue(i) * wRow.GetValue(i);
                            }
                            // outRow[koff] += sum
                            float prev = outRow.GetValue(koff);
                            outRow.SetValue(koff, prev + sum);

                            wRowQ_.FreeTensor(wRow);
                        }
                        xRowQ_.FreeTensor(xRow);
                    }

                    // Add bias and ReLU; then store y row segment
                    for (uint32_t koff = 0; koff < kCur; ++koff) {
                        float v = outRow.GetValue(koff) + bGm_.GetValue(kStart + koff);
                        if (v < 0.0f) v = 0.0f;
                        outRow.SetValue(koff, v);
                    }

                    // Store into GM
                    const uint32_t yOff = (mStart + moff) * K_ + kStart;
                    AscendC::DataCopy(yGm_[yOff], outRow, kCur);

                    outRowQ_.FreeTensor(outRow);
                } // moff
            } // kStart
        } // mt
    }

private:
    __aicore__ static inline uint32_t MinU32(uint32_t a, uint32_t b) { return a < b ? a : b; }
    __aicore__ static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INBUF_NUM> xRowQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, WBUF_NUM>  wRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, OUTBUF_NUM> outRowQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {0};
    uint32_t tileK_ {0};
    uint32_t chunkN_ {0};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                                            GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddReluCustom op;
    op.Init(x, w, bias, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileK, tiling_data.chunkN);
    op.Process();
}

#include <torch/library.h>

at::Tensor gemm_add_relu_impl_npu(const at::Tensor& x, const at::Tensor& w, const at::Tensor& bias) {
    // x: [M,N], w: [K,N], bias: [K] -> y: [M,K]
    TORCH_CHECK(x.scalar_type() == at::kFloat, "x must be float32");
    TORCH_CHECK(w.scalar_type() == at::kFloat, "w must be float32");
    TORCH_CHECK(bias.scalar_type() == at::kFloat, "bias must be float32");
    TORCH_CHECK(x.dim() == 2 && w.dim() == 2 && bias.dim() == 1, "invalid dims");
    TORCH_CHECK(x.size(1) == w.size(1), "in_features mismatch");
    TORCH_CHECK(w.size(0) == bias.size(0), "out_features vs bias mismatch");

    at::Tensor y = at::empty({x.size(0), w.size(0)}, x.options());
    EXEC_NPU_CMD(aclnnGemmAddReluCustom, x, w, bias, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("gemm_add_relu_custom", &gemm_add_relu_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("gemm_add_relu_custom", &gemm_add_relu_impl_npu, "GEMM + Bias + ReLU (GemmAddReluCustom on NPU)");
