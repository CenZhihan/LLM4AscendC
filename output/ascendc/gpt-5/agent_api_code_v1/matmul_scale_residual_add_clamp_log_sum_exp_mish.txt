#include "kernel_operator.h"

// Optional: if the MatMul API headers are available in the environment.
using AMatType = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
using BMatType = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL>;
using CMatType = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
using BiasType = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;
constexpr int32_t TMP_BUF_NUM = 2;

class KernelMatmulScaleResidualAddClampLogSumExpMishCustom {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling,
                                GM_ADDR clamp_min, GM_ADDR clamp_max,
                                GM_ADDR y, GM_ADDR gemmOut,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileN)
    {
        M_ = M; K_ = K; N_ = N; tileN_ = tileN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * 1);
        gemmGm_.SetGlobalBuffer((__gm__ float*)gemmOut, static_cast<uint64_t>(M_) * N_);

        // Load runtime scalars
        scale_ = LoadScalar(scaling);
        clampLo_ = LoadScalar(clamp_min);
        clampHi_ = LoadScalar(clamp_max);
        factor_ = 1.0f + scale_;

        // Queues for post-processing tiles (per-row chunks over N)
        pipe_.InitBuffer(inTileQ_, IN_BUF_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpTileQ_, TMP_BUF_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(redTileQ_, TMP_BUF_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(outTileQ_, OUT_BUF_NUM, sizeof(float));  // final scalar per row
    }

    __aicore__ inline void GemmCompute()
    {
        // Use Matmul API to compute GEMM into gemmGm_ with bias add
        AMatType A;
        BMatType B;
        CMatType C;
        BiasType Bias;

        // Matmul kernel wrapper
        BatchMatmulKernel<AMatType, BMatType, CMatType, BiasType> batchMatmulKernel;
        AscendC::TPipe mmPipe;

        // Minimal tiling settings (share all on L0C/L1)
        TCubeTiling tiling;
        tiling.shareMode = 0;
        tiling.shareL1Size = FULL_L1_SIZE;
        tiling.shareL0CSize = FULL_L0C_SIZE;
        tiling.shareUbSize = 0;
        REGIST_MATMUL_OBJ(&mmPipe, GetSysWorkSpacePtr(), batchMatmulKernel.matmulObj, &tiling);

        // Run single-batch MatMul: (M,K) x (N,K)^T -> (M,N)
        batchMatmulKernel.Init((GM_ADDR)xGm_.GetPhyAddr(), (GM_ADDR)wGm_.GetPhyAddr(), (GM_ADDR)bGm_.GetPhyAddr(),
                               (GM_ADDR)gemmGm_.GetPhyAddr(), GetSysWorkSpacePtr(), tiling);
        batchMatmulKernel.Process<true>(&mmPipe, 1, 1);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) { return; }

        // 1) GEMM to workspace
        GemmCompute();

        // 2) Post-process per row: scale+residual, clamp, logsumexp across dim=1, mish, final y = x * mish(x)
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        for (uint32_t row = blockIdx; row < M_; row += blockNum) {
            PostProcessRow(row);
        }
    }

private:
    __aicore__ inline float LoadScalar(GM_ADDR gmScalar)
    {
        AscendC::GlobalTensor<float> tmp;
        tmp.SetGlobalBuffer((__gm__ float*)gmScalar, 1);
        AscendC::LocalTensor<float> loc = tmpTileQ_.AllocTensor<float>();
        AscendC::DataCopy(loc, tmp, 1);
        float val = 0.0f;
        // Read back the scalar value
        // Assuming LocalTensor supports direct element access via GetValue(0)
        val = loc.GetValue(0);
        tmpTileQ_.FreeTensor(loc);
        return val;
    }

    __aicore__ inline void PostProcessRow(uint32_t row)
    {
        float runningMax = -1.0e38f;
        float runningSumExp = 0.0f;

        uint64_t rowBase = static_cast<uint64_t>(row) * N_;
        for (uint32_t c0 = 0; c0 < N_; c0 += tileN_) {
            const uint32_t len = (c0 + tileN_ <= N_) ? tileN_ : (N_ - c0);

            // Load a tile of GEMM outputs for this row
            AscendC::LocalTensor<float> vec = inTileQ_.AllocTensor<float>();
            AscendC::DataCopy(vec, gemmGm_[rowBase + c0], len);

            // y = y * factor = y * (1 + s)
            AscendC::Muls(vec, vec, factor_, len);
            // clamp
            AscendC::Maximums(vec, vec, clampLo_, len);
            AscendC::Minimums(vec, vec, clampHi_, len);

            // Compute max over this tile
            AscendC::LocalTensor<float> tileMaxBuf = tmpTileQ_.AllocTensor<float>();   // 1-element dst
            AscendC::LocalTensor<float> redBuf = redTileQ_.AllocTensor<float>();       // workspace buffer
            AscendC::ReduceMax(tileMaxBuf, vec, redBuf, len);
            float tileMax = tileMaxBuf.GetValue(0);

            float newMax = (tileMax > runningMax) ? tileMax : runningMax;

            // Compute exp(y - newMax) and sum
            AscendC::LocalTensor<float> tmpSub = redTileQ_.AllocTensor<float>();
            // tmpSub = vec - newMax
            AscendC::Adds(tmpSub, vec, -newMax, len);
            AscendC::Exp(tmpSub, tmpSub, len);

            AscendC::LocalTensor<float> tileSumBuf = tmpTileQ_.AllocTensor<float>();   // 1-element dst for sum
            AscendC::ReduceSum(tileSumBuf, tmpSub, redBuf, len);
            float tileSum = tileSumBuf.GetValue(0);

            // Update running sum using stable log-sum-exp merge rule
            if (newMax == runningMax) {
                runningSumExp += tileSum;
            } else {
                float scalePrev = AscendC::FastExp(runningMax - newMax);
                runningSumExp = runningSumExp * scalePrev + tileSum;
                runningMax = newMax;
            }

            // Free temporaries
            tmpTileQ_.FreeTensor(tileMaxBuf);
            tmpTileQ_.FreeTensor(tileSumBuf);
            redTileQ_.FreeTensor(tmpSub);
            redTileQ_.FreeTensor(redBuf);
            inTileQ_.FreeTensor(vec);
        }

        // logsumexp = log(runningSumExp) + runningMax
        AscendC::LocalTensor<float> scalarVec = outTileQ_.AllocTensor<float>();
        scalarVec.SetValue(0, runningSumExp);
        AscendC::Log(scalarVec, scalarVec, 1);
        float logSumExp = scalarVec.GetValue(0) + runningMax;

        // mish(x) = x * tanh(softplus(x)) ; with softplus(x)=log(1+exp(x))
        // Compute on length-1
        AscendC::LocalTensor<float> tmp1 = tmpTileQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp2 = tmpTileQ_.AllocTensor<float>();
        tmp1.SetValue(0, logSumExp);
        AscendC::Exp(tmp2, tmp1, 1);                    // exp(x)
        AscendC::Adds(tmp2, tmp2, 1.0f, 1);             // 1 + exp(x)
        AscendC::Log(tmp2, tmp2, 1);                    // softplus(x)
        AscendC::Tanh(tmp2, tmp2, 1);                   // tanh(softplus(x))
        AscendC::Muls(tmp2, tmp2, logSumExp, 1);        // mish(x)
        // y = x * mish(x)
        AscendC::Muls(scalarVec, tmp2, logSumExp, 1);
        float finalVal = scalarVec.GetValue(0);

        // Store result
        yGm_[static_cast<uint64_t>(row) * 1 + 0] = finalVal;

        outTileQ_.FreeTensor(scalarVec);
        tmpTileQ_.FreeTensor(tmp1);
        tmpTileQ_.FreeTensor(tmp2);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> tmpTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> redTileQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outTileQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> gemmGm_;

    uint32_t M_{0}, K_{0}, N_{0}, tileN_{0};
    uint32_t M_, K_, N_, tileN_;
    float scale_{0.0f}, clampLo_{-1e9f}, clampHi_{1e9f}, factor_{1.0f};
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
    GM_ADDR scaling, GM_ADDR clamp_min, GM_ADDR clamp_max,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    // Workspace holds GEMM output matrix of shape (M,N)
    KernelMatmulScaleResidualAddClampLogSumExpMishCustom op;
    op.Init(x, weight, bias, scaling, clamp_min, clamp_max,
            y, workspace,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileN);
    op.Process();
}
