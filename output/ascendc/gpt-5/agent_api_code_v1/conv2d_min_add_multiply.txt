static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;

static inline uint32_t AlignUp(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : ((x + a - 1u) / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t XBUF_NUM = 1;
constexpr int32_t TMPBUF_NUM = 1;
constexpr int32_t ACCBUF_NUM = 1;

class KernelConv2dMinAddMultiplyCustom {
public:
    __aicore__ inline KernelConv2dMinAddMultiplyCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR cb, GM_ADDR pb, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t Co,
                                uint32_t Hi, uint32_t Wi,
                                uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t tileW, uint32_t tileCo, uint32_t hasBias,
                                float constantValue, float scalingFactor)
    {
        N_ = N; Ci_ = Ci; Co_ = Co; Hi_ = Hi; Wi_ = Wi;
        Kh_ = Kh; Kw_ = Kw; Ho_ = Ho; Wo_ = Wo;
        tileW_ = tileW; tileCo_ = tileCo; hasBias_ = hasBias != 0;
        cval_ = constantValue; sf_ = scalingFactor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * Ci_ * Hi_ * Wi_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Co_) * Ci_ * Kh_ * Kw_);
        if (hasBias_) {
            cbGm_.SetGlobalBuffer((__gm__ float*)cb, Co_);
        }
        pbGm_.SetGlobalBuffer((__gm__ float*)pb, Co_); // [Co,1,1] flattened

        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * Co_ * Ho_ * Wo_);

        pipe_.InitBuffer(xQueue_, XBUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMPBUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, ACCBUF_NUM, tileCo_ * tileW_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, tileCo_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, 1, tileCo_ * sizeof(float));
        pipe_.InitBuffer(pbQueue_, 1, tileCo_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (Ho_ == 0 || Wo_ == 0 || Co_ == 0 || Ci_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t totalRows = static_cast<uint64_t>(N_) * Ho_;

        for (uint64_t rowIdx = blockIdx; rowIdx < totalRows; rowIdx += blockNum) {
            const uint32_t n = static_cast<uint32_t>(rowIdx / Ho_);
            const uint32_t ho = static_cast<uint32_t>(rowIdx % Ho_);

            for (uint32_t co0 = 0; co0 < Co_; co0 += tileCo_) {
                const uint32_t curCo = AscendC::min(tileCo_, Co_ - co0);

                for (uint32_t wo0 = 0; wo0 < Wo_; wo0 += tileW_) {
                    const uint32_t curW = AscendC::min(tileW_, Wo_ - wo0);

                    // Accumulator buffer for [curCo, curW]
                    AscendC::LocalTensor<float> yAcc = accQueue_.AllocTensor<float>();
                    // Zero-init each co slice
                    for (uint32_t j = 0; j < curCo; ++j) {
                        AscendC::LocalTensor<float> ySlice = yAcc[j * tileW_];
                        AscendC::Muls(ySlice, ySlice, 0.0f, curW);
                    }

                    // Convolution: stride=1, pad=0, dilation=1
                    for (uint32_t ci = 0; ci < Ci_; ++ci) {
                        for (uint32_t kh = 0; kh < Kh_; ++kh) {
                            const uint32_t hsrc = ho + kh;
                            for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                const uint32_t wsrc0 = wo0 + kw;

                                // Load input row vector x[n, ci, hsrc, wsrc0:wsrc0+curW-1]
                                AscendC::LocalTensor<float> xVec = xQueue_.AllocTensor<float>();
                                const uint64_t xOff = (((static_cast<uint64_t>(n) * Ci_ + ci) * Hi_ + hsrc) * Wi_) + wsrc0;
                                AscendC::DataCopy(xVec, xGm_[xOff], curW);
                                xQueue_.EnQue(xVec);

                                // Load weight vector for this (ci,kh,kw) across curCo (strided in GM)
                                AscendC::LocalTensor<float> wVec = wQueue_.AllocTensor<float>();
                                AscendC::DataCopyExtParams param;
                                param.blockCount = curCo;
                                param.blockLen = sizeof(float);
                                const uint32_t strideElems = Ci_ * Kh_ * Kw_;
                                param.srcStride = (strideElems - 1u) * sizeof(float);
                                param.dstStride = 0;
                                const uint64_t wBase = ((((uint64_t)co0) * Ci_ + ci) * Kh_ + kh) * Kw_ + kw;
                                AscendC::DataCopy(wVec, wGm_[wBase], param);
                                wQueue_.EnQue(wVec);

                                // Accumulate: for each co in tile
                                AscendC::LocalTensor<float> xVecUse = xQueue_.DeQue<float>();
                                AscendC::LocalTensor<float> wVecUse = wQueue_.DeQue<float>();
                                AscendC::LocalTensor<float> tmpVec = tmpQueue_.AllocTensor<float>();

                                for (uint32_t j = 0; j < curCo; ++j) {
                                    // tmp = xVec * wVal
                                    float wv = wVecUse.GetValue(j);
                                    AscendC::Muls(tmpVec, xVecUse, wv, curW);

                                    // yAcc[j] += tmp
                                    AscendC::LocalTensor<float> ySlice = yAcc[j * tileW_];
                                    AscendC::Add(ySlice, ySlice, tmpVec, curW);
                                }

                                tmpQueue_.FreeTensor(tmpVec);
                                wQueue_.FreeTensor(wVecUse);
                                xQueue_.FreeTensor(xVecUse);
                            }
                        }
                    }

                    // Epilogue: +conv_bias (optional), min(constant), +post_bias, *scale
                    // Load conv_bias vector for co tile
                    AscendC::LocalTensor<float> bVec;
                    if (hasBias_) {
                        bVec = bQueue_.AllocTensor<float>();
                        AscendC::DataCopy(bVec, cbGm_[co0], curCo);
                        bQueue_.EnQue(bVec);
                    }
                    // Load post_bias vector for co tile (assumed [Co,1,1] -> flattened [Co])
                    AscendC::LocalTensor<float> pbVec = pbQueue_.AllocTensor<float>();
                    AscendC::DataCopy(pbVec, pbGm_[co0], curCo);
                    pbQueue_.EnQue(pbVec);

                    AscendC::LocalTensor<float> pbUse = pbQueue_.DeQue<float>();
                    AscendC::LocalTensor<float> bUse;
                    if (hasBias_) {
                        bUse = bQueue_.DeQue<float>();
                    }

                    for (uint32_t j = 0; j < curCo; ++j) {
                        AscendC::LocalTensor<float> ySlice = yAcc[j * tileW_];
                        if (hasBias_) {
                            float bv = bUse.GetValue(j);
                            AscendC::Adds(ySlice, ySlice, bv, curW);
                        }
                        // min with constant
                        AscendC::Mins(ySlice, ySlice, cval_, curW);
                        // add post bias
                        float pbv = pbUse.GetValue(j);
                        AscendC::Adds(ySlice, ySlice, pbv, curW);
                        // multiply scale
                        AscendC::Muls(ySlice, ySlice, sf_, curW);
                    }

                    if (hasBias_) {
                        bQueue_.FreeTensor(bUse);
                    }
                    pbQueue_.FreeTensor(pbUse);

                    // Store results to GM
                    for (uint32_t j = 0; j < curCo; ++j) {
                        AscendC::LocalTensor<float> ySlice = yAcc[j * tileW_];
                        const uint32_t co = co0 + j;
                        const uint64_t yOff = (((static_cast<uint64_t>(n) * Co_ + co) * Ho_ + ho) * Wo_) + wo0;
                        AscendC::DataCopy(yGm_[yOff], ySlice, curW);
                    }

                    accQueue_.FreeTensor(yAcc);
                } // wo0
            } // co0
        } // rowIdx
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, XBUF_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMPBUF_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, ACCBUF_NUM> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> pbQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> cbGm_;
    AscendC::GlobalTensor<float> pbGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Ci_{0}, Co_{0}, Hi_{0}, Wi_{0}, Kh_{0}, Kw_{0}, Ho_{0}, Wo_{0};
    uint32_t tileW_{0}, tileCo_{0};
    bool hasBias_{false};
    float cval_{0.0f};
    float sf_{1.0f};
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR conv_bias, GM_ADDR post_bias, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinAddMultiplyCustom op;
    op.Init(x, w, conv_bias, post_bias, y,
            tiling_data.N, tiling_data.Ci, tiling_data.Co,
            tiling_data.Hi, tiling_data.Wi,
            tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo,
            tiling_data.tileW, tiling_data.tileCo,
            tiling_data.hasBias,
            tiling_data.constantValue, tiling_data.scalingFactor);
    op.Process();
}
