kernel_src="""

#include "kernel_operator.h"

class KernelConvDepthwise2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t wi,
                                uint32_t kh, uint32_t kw,
                                uint32_t outH, uint32_t outW)
    {
        n_ = n; c_ = c; h_ = h; w_ = wi;
        kh_ = kh; kw_ = kw; outH_ = outH; outW_ = outW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_*c_*h_*w_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(c_*kh_*kw_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_*c_*outH_*outW_));

        // UB buffers: 3 input rows + 1 output row + 1 tmp row + small weight buffer
        pipe_.InitBuffer(inRowQ_, 3, w_ * sizeof(float));
        pipe_.InitBuffer(outRowQ_, 1, outW_ * sizeof(float));
        pipe_.InitBuffer(tmpRowQ_, 1, outW_ * sizeof(float));
        pipe_.InitBuffer(wLocalQ_, 1, 16 * sizeof(float)); // enough for 3x3 weights
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || h_ == 0 || w_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t tasks = n_ * c_;
        if (blockIdx >= tasks) return;

        for (uint32_t task = blockIdx; task < tasks; task += blockNum) {
            const uint32_t nIdx = task / c_;
            const uint32_t cIdx = task % c_;

            const uint32_t xBase = ((nIdx * c_ + cIdx) * h_ * w_);
            const uint32_t yBase = ((nIdx * c_ + cIdx) * outH_ * outW_);
            const uint32_t wBase = (cIdx * kh_ * kw_);

            // Load 3x3 weights for this channel into UB
            AscendC::LocalTensor<float> wLoc = wLocalQ_.AllocTensor<float>();
            AscendC::DataCopy(wLoc, wGm_[wBase], kh_ * kw_);
            wLocalQ_.EnQue(wLoc);
            AscendC::LocalTensor<float> wLocal = wLocalQ_.DeQue<float>();

            // Extract 9 weights as scalars
            float w00 = wLocal.GetValue(0);
            float w01 = wLocal.GetValue(1);
            float w02 = wLocal.GetValue(2);
            float w10 = wLocal.GetValue(3);
            float w11 = wLocal.GetValue(4);
            float w12 = wLocal.GetValue(5);
            float w20 = wLocal.GetValue(6);
            float w21 = wLocal.GetValue(7);
            float w22 = wLocal.GetValue(8);

            // Process each output row
            for (uint32_t r = 0; r < outH_; ++r) {
                // Bring the 3 needed input rows to UB
                for (uint32_t dy = 0; dy < kh_; ++dy) {
                    const uint32_t inRowOffset = xBase + (r + dy) * w_;
                    AscendC::LocalTensor<float> inRow = inRowQ_.AllocTensor<float>();
                    AscendC::DataCopy(inRow, xGm_[inRowOffset], w_);
                    inRowQ_.EnQue(inRow);
                }

                // Dequeue input rows for compute
                AscendC::LocalTensor<float> row0 = inRowQ_.DeQue<float>();
                AscendC::LocalTensor<float> row1 = inRowQ_.DeQue<float>();
                AscendC::LocalTensor<float> row2 = inRowQ_.DeQue<float>();

                AscendC::LocalTensor<float> yRow = outRowQ_.AllocTensor<float>();
                AscendC::Dup(yRow, 0.0f, outW_);

                AscendC::LocalTensor<float> tRow = tmpRowQ_.AllocTensor<float>();

                // Row 0 contributions
                AscendC::Muls(tRow, row0[0], w00, outW_); AscendC::Add(yRow, yRow, tRow, outW_);
                AscendC::Muls(tRow, row0[1], w01, outW_); AscendC::Add(yRow, yRow, tRow, outW_);
                AscendC::Muls(tRow, row0[2], w02, outW_); AscendC::Add(yRow, yRow, tRow, outW_);

                // Row 1 contributions
                AscendC::Muls(tRow, row1[0], w10, outW_); AscendC::Add(yRow, yRow, tRow, outW_);
                AscendC::Muls(tRow, row1[1], w11, outW_); AscendC::Add(yRow, yRow, tRow, outW_);
                AscendC::Muls(tRow, row1[2], w12, outW_); AscendC::Add(yRow, yRow, tRow, outW_);

                // Row 2 contributions
                AscendC::Muls(tRow, row2[0], w20, outW_); AscendC::Add(yRow, yRow, tRow, outW_);
                AscendC::Muls(tRow, row2[1], w21, outW_); AscendC::Add(yRow, yRow, tRow, outW_);
                AscendC::Muls(tRow, row2[2], w22, outW_); AscendC::Add(yRow, yRow, tRow, outW_);

                // Write back output row
                const uint32_t yRowOffset = yBase + r * outW_;
                AscendC::DataCopy(yGm_[yRowOffset], yRow, outW_);

                // Free UB resources
                tmpRowQ_.FreeTensor(tRow);
                outRowQ_.FreeTensor(yRow);
                inRowQ_.FreeTensor(row0);
                inRowQ_.FreeTensor(row1);
                inRowQ_.FreeTensor(row2);
            }

            wLocalQ_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 3> inRowQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wLocalQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t kh_{0}, kw_{0};
    uint32_t outH_{0}, outW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputSquareKernelCustom op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kh, tiling_data.kw,
            tiling_data.outH, tiling_data.outW);
    op.Process();
}
