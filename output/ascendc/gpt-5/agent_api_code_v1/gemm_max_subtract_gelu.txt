namespace {
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // DMA/buffer alignment
static constexpr uint32_t MIN_BLOCKS = 8u;
static constexpr uint32_t MAX_BLOCKS = 24u;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
static inline uint32_t MinU32(uint32_t a, uint32_t b)
{
    return (a < b) ? a : b;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMaxSubtractGeluCustomTilingData tiling;

    // Shapes
    const gert::Shape *x_shape = context->GetInputShape(0); // [B, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K]
    const gert::Shape *b_shape = context->GetInputShape(2); // [N]

    const auto &x_dims = x_shape->GetOriginShape().GetDims();
    const auto &w_dims = w_shape->GetOriginShape().GetDims();
    if (x_dims.size() < 2 || w_dims.size() < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t m = static_cast<uint32_t>(x_dims[x_dims.size() - 2]); // batch
    const uint32_t k = static_cast<uint32_t>(x_dims[x_dims.size() - 1]);
    const uint32_t n = static_cast<uint32_t>(w_dims[w_dims.size() - 2]); // out_features
    const uint32_t k2 = static_cast<uint32_t>(w_dims[w_dims.size() - 1]);
    if (k != k2) {
        return ge::GRAPH_FAILED;
    }

    // Attributes
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t maxDim = 1;
    if (attrs && attrs->GetAttrNum() > 0) {
        const int64_t *maxDimPtr = attrs->GetAttrPointer<int64_t>(0);
        if (maxDimPtr) maxDim = *maxDimPtr;
    }
    // approximate: 0=ERF, 1=TANH (default TANH)
    uint32_t approxMode = 1u;
    if (attrs && attrs->GetAttrNum() > 1) {
        const char* approxStr = attrs->GetAttrPointer<char>(1);
        // keep default if not provided
        (void)approxStr;
    }

    // UB budget: local buffers per iteration
    // We'll use: xChunk(tileK) + wChunk(tileK) + mulTmp(tileK) + a few scalars
    // => bytes ~ 3 * tileK * 4 + overhead. Keep margin.
    uint32_t maxTileK = (UB_BUDGET_BYTES / 16); // simple conservative
    maxTileK = AlignDown(maxTileK, ALIGN_ELEMS);
    if (maxTileK < ALIGN_ELEMS) maxTileK = ALIGN_ELEMS;
    uint32_t tileK = MinU32(k, maxTileK);
    tileK = AlignDown(tileK, ALIGN_ELEMS);
    if (tileK == 0u) tileK = MinU32(k, ALIGN_ELEMS);

    // Blocks based on m rows
    uint32_t blockDim = (m == 0u) ? MIN_BLOCKS : CeilDiv(m, 1u);
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);
    tiling.set_maxDim(static_cast<uint32_t>(maxDim));
    tiling.set_approxMode(approxMode);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // [B, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K]
    const auto &x_dims = x_shape->GetOriginShape().GetDims();
    const auto &w_dims = w_shape->GetOriginShape().GetDims();
    if (x_dims.size() < 2 || w_dims.size() < 2) {
        return GRAPH_FAILED;
    }
    int64_t max_dim = 1;
    auto *attrs = context->GetAttrs();
    if (attrs && attrs->GetAttrNum() > 0) {
        const int64_t *maxDimPtr = attrs->GetAttrPointer<int64_t>(0);
        if (maxDimPtr) max_dim = *maxDimPtr;
    }

    // Linear: [B, K] x [N, K]^T -> [B, N]
    std::vector<int64_t> out_dims;
    if (max_dim == 1) {
        out_dims = {static_cast<int64_t>(x_dims[x_dims.size()-2]), 1};
    } else if (max_dim == 0) {
        out_dims = {1, static_cast<int64_t>(w_dims[w_dims.size()-2])};
    } else {
        // Fallback: keepdim true, reduce last dim by default
        out_dims = {static_cast<int64_t>(x_dims[x_dims.size()-2]), 1};
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMaxSubtractGeluCustom {
public:
    __aicore__ inline KernelGemmMaxSubtractGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileK, uint32_t maxDim, uint32_t approxMode)
    {
        m_ = m; n_ = n; k_ = k;
        tileK_ = tileK > 0 ? tileK : 256;
        maxDim_ = maxDim;
        approxMode_ = approxMode;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(m_) * k_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(n_) * k_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n_);

        // Output shape depends on maxDim; default to (m,1)
        outCols_ = 1;
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(m_) * outCols_);

        // Queues/buffers
        pipe_.InitBuffer(xInQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(wInQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(vecTmpQ_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, 4, sizeof(float)); // scalars: acc, sum, max, tmp
        pipe_.InitBuffer(outQ_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || n_ == 0 || k_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < m_; row += blockNum) {
            ComputeRow(row);
        }
    }

private:
    __aicore__ inline void LoadVector(LocalTensor<float>& dst, AscendC::GlobalTensor<float>& src, uint64_t gmOffset, uint32_t len)
    {
        AscendC::DataCopy(dst, src[gmOffset], len);
    }

    __aicore__ inline void ComputeRow(uint32_t row)
    {
        // max across N
        AscendC::LocalTensor<float> maxLocal = scalarQ_.AllocTensor<float>();
        AscendC::Duplicate(maxLocal, -3.4e38f, 1);

        for (uint32_t col = 0; col < n_; ++col) {
            // acc = 0
            AscendC::LocalTensor<float> accLocal = scalarQ_.AllocTensor<float>();
            AscendC::Duplicate(accLocal, 0.0f, 1);

            // Dot product over K in tiles
            for (uint32_t k0 = 0; k0 < k_; k0 += tileK_) {
                uint32_t curK = k_ - k0;
                if (curK > tileK_) curK = tileK_;

                AscendC::LocalTensor<float> xChunk = xInQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> wChunk = wInQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> mulTmp = vecTmpQ_.AllocTensor<float>();

                // Load x[row, k0:k0+curK]
                const uint64_t xOff = static_cast<uint64_t>(row) * k_ + k0;
                LoadVector(xChunk, xGm_, xOff, curK);

                // Load w[col, k0:k0+curK]
                const uint64_t wOff = static_cast<uint64_t>(col) * k_ + k0;
                LoadVector(wChunk, wGm_, wOff, curK);

                // mulTmp = xChunk * wChunk
                AscendC::Mul(mulTmp, xChunk, wChunk, curK);

                // sumLocal = reduce_sum(mulTmp)
                AscendC::LocalTensor<float> sumLocal = scalarQ_.AllocTensor<float>();
                AscendC::ReduceSum(sumLocal, mulTmp, curK);

                // accLocal += sumLocal
                AscendC::Add(accLocal, accLocal, sumLocal, 1);

                // free temps
                scalarQ_.FreeTensor(sumLocal);
                vecTmpQ_.FreeTensor(mulTmp);
                wInQ_.FreeTensor(wChunk);
                xInQ_.FreeTensor(xChunk);
            }

            // accLocal += b[col]
            AscendC::LocalTensor<float> bLocal = scalarQ_.AllocTensor<float>();
            AscendC::DataCopy(bLocal, bGm_[col], 1);
            AscendC::Add(accLocal, accLocal, bLocal, 1);
            scalarQ_.FreeTensor(bLocal);

            // maxLocal = max(maxLocal, accLocal)
            AscendC::Max(maxLocal, maxLocal, accLocal, 1);

            scalarQ_.FreeTensor(accLocal);
        }

        // After max along dim=maxDim (default 1), we keepdim => scalar for this row
        // mean along dim=1 of this scalar is itself; x - mean = 0
        // For generality, compute yTmp = maxLocal - meanLocal (meanLocal == maxLocal here)
        AscendC::LocalTensor<float> yTmp = scalarQ_.AllocTensor<float>();
        AscendC::Sub(yTmp, maxLocal, maxLocal, 1);

        // GELU (approximate by tanh)
        AscendC::LocalTensor<float> t1 = scalarQ_.AllocTensor<float>(); // x^2
        AscendC::LocalTensor<float> t2 = scalarQ_.AllocTensor<float>(); // x^3
        AscendC::LocalTensor<float> t3 = scalarQ_.AllocTensor<float>(); // inner
        AscendC::LocalTensor<float> t4 = scalarQ_.AllocTensor<float>(); // tanh
        AscendC::Mul(t1, yTmp, yTmp, 1);                 // x^2
        AscendC::Mul(t2, t1, yTmp, 1);                   // x^3
        AscendC::Muls(t2, t2, 0.044715f, 1);             // 0.044715*x^3
        AscendC::Add(t3, yTmp, t2, 1);                   // x + 0.044715*x^3
        AscendC::Muls(t3, t3, 0.7978845608f, 1);         // sqrt(2/pi)*(...)
        AscendC::Tanh(t4, t3, 1);                        // tanh(...)
        AscendC::Muls(t4, t4, 0.5f, 1);                  // 0.5*tanh
        AscendC::LocalTensor<float> oneLocal = scalarQ_.AllocTensor<float>();
        AscendC::Duplicate(oneLocal, 1.0f, 1);
        AscendC::Add(t4, t4, oneLocal, 1);               // 0.5*tanh + 1
        AscendC::Mul(t4, t4, yTmp, 1);                   // y * (0.5*tanh + 1)
        AscendC::Muls(t4, t4, 0.5f, 1);                  // 0.5 * y * (1 + tanh(...))

        // Store output y[row, 0]
        AscendC::LocalTensor<float> yOut = outQ_.AllocTensor<float>();
        AscendC::DataCopy(yOut, t4, 1);
        const uint64_t yOff = static_cast<uint64_t>(row) * outCols_;
        AscendC::DataCopy(yGm_[yOff], yOut, 1);
        outQ_.FreeTensor(yOut);

        // free scalars
        scalarQ_.FreeTensor(oneLocal);
        scalarQ_.FreeTensor(t4);
        scalarQ_.FreeTensor(t3);
        scalarQ_.FreeTensor(t2);
        scalarQ_.FreeTensor(t1);
        scalarQ_.FreeTensor(yTmp);
        scalarQ_.FreeTensor(maxLocal);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xInQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wInQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> vecTmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> scalarQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t m_{0}, n_{0}, k_{0};
    uint32_t m_{0u}, n_{0u}, k_{0u}; // avoid name clash
    uint32_t m_, n_, k_;
    uint32_t tileK_{256};
    uint32_t maxDim_{1};
    uint32_t approxMode_{1}; // 1 = TANH
    uint32_t outCols_{1};
};

extern "C" __global__ __aicore__ void gemm_max_subtract_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMaxSubtractGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileK, tiling_data.maxDim, tiling_data.approxMode);
    op.Process();
}
