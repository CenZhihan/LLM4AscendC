static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForLowerTriangularMatricesCustomTilingData tiling;

    const auto *ashape = context->GetInputShape(0);
    const auto *bshape = context->GetInputShape(1);
    if (ashape->GetOriginShape().GetDimNum() != 2 || bshape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const uint64_t a0 = static_cast<uint64_t>(ashape->GetOriginShape().GetDim(0));
    const uint64_t a1 = static_cast<uint64_t>(ashape->GetOriginShape().GetDim(1));
    const uint64_t b0 = static_cast<uint64_t>(bshape->GetOriginShape().GetDim(0));
    const uint64_t b1 = static_cast<uint64_t>(bshape->GetOriginShape().GetDim(1));

    if (a0 != a1 || b0 != b1 || a1 != b0) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(a0);

    // Choose a conservative block dim between 8 and 24 depending on N
    uint32_t blockDim = (N < 8u) ? N : 8u;
    if (blockDim > 24u) blockDim = 24u;

    uint32_t rowsPerBlock = CeilDiv(N, blockDim);
    if (rowsPerBlock == 0u) rowsPerBlock = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

class KernelMatmulForLowerTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForLowerTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t N, uint32_t rowsPerBlock)
    {
        N_ = N;
        rowsPerBlock_ = rowsPerBlock;

        aGm_.SetGlobalBuffer((__gm__ float*)a, static_cast<uint64_t>(N_) * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, static_cast<uint64_t>(N_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, static_cast<uint64_t>(N_) * N_);
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over blocks of rows
        for (uint32_t baseRow = blockIdx * rowsPerBlock_;
             baseRow < N_;
             baseRow += blockNum * rowsPerBlock_) {

            const uint32_t rowEnd = AscendC::Min(baseRow + rowsPerBlock_, N_);
            for (uint32_t i = baseRow; i < rowEnd; ++i) {
                // Compute lower triangular part: j in [0..i]
                for (uint32_t j = 0; j <= i; ++j) {
                    float acc = 0.0f;
                    // Due to lower-triangular structure of A and B:
                    // sum over k from j..i of A[i,k] * B[k,j]
                    for (uint32_t k = j; k <= i; ++k) {
                        acc += aGm_[static_cast<uint64_t>(i) * N_ + k] *
                               bGm_[static_cast<uint64_t>(k) * N_ + j];
                    }
                    cGm_[static_cast<uint64_t>(i) * N_ + j] = acc;
                }
                // Explicitly zero the upper triangular part for row i
                for (uint32_t j = i + 1; j < N_; ++j) {
                    cGm_[static_cast<uint64_t>(i) * N_ + j] = 0.0f;
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t rowsPerBlock_{1};
};

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                                                                   GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling_data, tilingGm);
    KernelMatmulForLowerTriangularMatricesCustom op;
    op.Init(a, b, c, tiling_data.N, tiling_data.rowsPerBlock);
    op.Process();
}
