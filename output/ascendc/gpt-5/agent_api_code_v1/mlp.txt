namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MlpCustomTilingData tiling;

    // Inputs: x [B, IN], w_packed [w_total], b_packed [b_total]
    const gert::Shape *x_shape = context->GetInputShape(0);
    if (x_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t batch = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t inSize = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *input_size = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *h1 = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *h2 = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *outSize = attrs->GetAttrPointer<int32_t>(3);

    // Basic validation
    if (static_cast<uint32_t>(*input_size) != inSize) {
        // Fallback if mismatch
        // still proceed but keep inSize from tensor
    }

    // Determine blockDim: split rows over cores
    uint32_t blockDim = batch;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // Workspace: y1 + y2
    size_t *workspace = context->GetWorkspaceSizes(1);
    const uint64_t y1_elems = static_cast<uint64_t>(batch) * static_cast<uint64_t>(*h1);
    const uint64_t y2_elems = static_cast<uint64_t>(batch) * static_cast<uint64_t>(*h2);
    workspace[0] = static_cast<size_t>((y1_elems + y2_elems) * sizeof(float));

    tiling.set_batch(batch);
    tiling.set_inSize(inSize);
    tiling.set_h1(static_cast<uint32_t>(*h1));
    tiling.set_h2(static_cast<uint32_t>(*h2));
    tiling.set_outSize(static_cast<uint32_t>(*outSize));
    tiling.set_blockDim(blockDim);
    tiling.set_reserved(0);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *outSize = attrs->GetAttrPointer<int32_t>(3);

    gert::Shape *y_shape = context->GetOutputShape(0);
    if (x_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    std::vector<int64_t> y_dims(2);
    y_dims[0] = x_shape->GetOriginShape().GetDim(0);
    y_dims[1] = static_cast<int64_t>(*outSize);
    y_shape->SetOriginShape(ge::Shape(y_dims));
    y_shape->SetStorageShape(ge::Shape(y_dims));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUF_NUM = 1;

class KernelMlpCustom {
public:
    __aicore__ inline KernelMlpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batch, uint32_t inSize, uint32_t h1, uint32_t h2, uint32_t outSize,
                                GM_ADDR workspace)
    {
        batch_ = batch;
        inSize_ = inSize;
        h1_ = h1;
        h2_ = h2;
        outSize_ = outSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(batch_) * inSize_);
        wPackedGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(h1_) * inSize_ +
                                              static_cast<uint64_t>(h2_) * h1_ +
                                              static_cast<uint64_t>(outSize_) * h2_);
        bPackedGm_.SetGlobalBuffer((__gm__ float*)b, static_cast<uint64_t>(h1_) + h2_ + outSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(batch_) * outSize_);

        // Workspace layout: y1 [batch*h1] then y2 [batch*h2]
        y1Gm_.SetGlobalBuffer((__gm__ float*)workspace, static_cast<uint64_t>(batch_) * h1_);
        y2Gm_.SetGlobalBuffer(((__gm__ float*)workspace) + static_cast<uint64_t>(batch_) * h1_, static_cast<uint64_t>(batch_) * h2_);

        pipe_.InitBuffer(xBufQ_, BUF_NUM, sizeof(float));
        pipe_.InitBuffer(wBufQ_, BUF_NUM, sizeof(float));
        pipe_.InitBuffer(tmpQ_, BUF_NUM, sizeof(float));
        pipe_.InitBuffer(accQ_, BUF_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || inSize_ == 0 || h1_ == 0 || h2_ == 0 || outSize_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < batch_; row += blockNum) {
            Stage1(row);
            Stage2(row);
            Stage3(row);
        }
    }

private:
    __aicore__ inline void ZeroLocal(AscendC::LocalTensor<float>& t)
    {
        AscendC::Dup(t, 0.0f, 1);
    }

    __aicore__ inline void CopyScalarIn(AscendC::LocalTensor<float>& l, AscendC::GlobalTensor<float>& g, uint64_t offset)
    {
        AscendC::DataCopy(l, g[offset], 1);
    }

    __aicore__ inline void CopyScalarOut(AscendC::GlobalTensor<float>& g, uint64_t offset, AscendC::LocalTensor<float>& l)
    {
        AscendC::DataCopy(g[offset], l, 1);
    }

    __aicore__ inline void Stage1(uint32_t row)
    {
        // y1[row, oc] = ReLU( sum_k x[row,k] * w1[oc,k] + b1[oc] )
        const uint64_t xRowBase = static_cast<uint64_t>(row) * inSize_;
        const uint64_t w1Base = 0;
        const uint64_t b1Base = 0;
        const uint64_t y1RowBase = static_cast<uint64_t>(row) * h1_;

        for (uint32_t oc = 0; oc < h1_; ++oc) {
            AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
            ZeroLocal(acc);

            const uint64_t wRowBase = w1Base + static_cast<uint64_t>(oc) * inSize_;
            for (uint32_t k = 0; k < inSize_; ++k) {
                AscendC::LocalTensor<float> xLocal = xBufQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> wLocal = wBufQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> prod = tmpQ_.AllocTensor<float>();

                CopyScalarIn(xLocal, xGm_, xRowBase + k);
                CopyScalarIn(wLocal, wPackedGm_, wRowBase + k);
                AscendC::Mul(prod, xLocal, wLocal, 1);
                AscendC::Add(acc, acc, prod, 1);

                tmpQ_.FreeTensor(prod);
                wBufQ_.FreeTensor(wLocal);
                xBufQ_.FreeTensor(xLocal);
            }

            // Add bias
            AscendC::LocalTensor<float> bias = tmpQ_.AllocTensor<float>();
            CopyScalarIn(bias, bPackedGm_, b1Base + oc);
            AscendC::Add(acc, acc, bias, 1);
            tmpQ_.FreeTensor(bias);

            // ReLU
            AscendC::Maxs(acc, acc, 0.0f, 1);

            // Store to y1
            CopyScalarOut(y1Gm_, y1RowBase + oc, acc);
            accQ_.FreeTensor(acc);
        }
    }

    __aicore__ inline void Stage2(uint32_t row)
    {
        // y2[row, oc] = ReLU( sum_k y1[row,k] * w2[oc,k] + b2[oc] )
        const uint64_t y1RowBase = static_cast<uint64_t>(row) * h1_;
        const uint64_t w2Base = static_cast<uint64_t>(h1_) * inSize_;
        const uint64_t b2Base = static_cast<uint64_t>(h1_);
        const uint64_t y2RowBase = static_cast<uint64_t>(row) * h2_;

        for (uint32_t oc = 0; oc < h2_; ++oc) {
            AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
            ZeroLocal(acc);

            const uint64_t wRowBase = w2Base + static_cast<uint64_t>(oc) * h1_;
            for (uint32_t k = 0; k < h1_; ++k) {
                AscendC::LocalTensor<float> xLocal = xBufQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> wLocal = wBufQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> prod = tmpQ_.AllocTensor<float>();

                CopyScalarIn(xLocal, y1Gm_, y1RowBase + k);
                CopyScalarIn(wLocal, wPackedGm_, wRowBase + k);
                AscendC::Mul(prod, xLocal, wLocal, 1);
                AscendC::Add(acc, acc, prod, 1);

                tmpQ_.FreeTensor(prod);
                wBufQ_.FreeTensor(wLocal);
                xBufQ_.FreeTensor(xLocal);
            }

            // Add bias
            AscendC::LocalTensor<float> bias = tmpQ_.AllocTensor<float>();
            CopyScalarIn(bias, bPackedGm_, b2Base + oc);
            AscendC::Add(acc, acc, bias, 1);
            tmpQ_.FreeTensor(bias);

            // ReLU
            AscendC::Maxs(acc, acc, 0.0f, 1);

            // Store to y2
            CopyScalarOut(y2Gm_, y2RowBase + oc, acc);
            accQ_.FreeTensor(acc);
        }
    }

    __aicore__ inline void Stage3(uint32_t row)
    {
        // y[row, oc] = sum_k y2[row,k] * w3[oc,k] + b3[oc]
        const uint64_t y2RowBase = static_cast<uint64_t>(row) * h2_;
        const uint64_t w3Base = static_cast<uint64_t>(h1_) * inSize_ + static_cast<uint64_t>(h2_) * h1_;
        const uint64_t b3Base = static_cast<uint64_t>(h1_) + static_cast<uint64_t>(h2_);
        const uint64_t yRowBase = static_cast<uint64_t>(row) * outSize_;

        for (uint32_t oc = 0; oc < outSize_; ++oc) {
            AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
            ZeroLocal(acc);

            const uint64_t wRowBase = w3Base + static_cast<uint64_t>(oc) * h2_;
            for (uint32_t k = 0; k < h2_; ++k) {
                AscendC::LocalTensor<float> xLocal = xBufQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> wLocal = wBufQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> prod = tmpQ_.AllocTensor<float>();

                CopyScalarIn(xLocal, y2Gm_, y2RowBase + k);
                CopyScalarIn(wLocal, wPackedGm_, wRowBase + k);
                AscendC::Mul(prod, xLocal, wLocal, 1);
                AscendC::Add(acc, acc, prod, 1);

                tmpQ_.FreeTensor(prod);
                wBufQ_.FreeTensor(wLocal);
                xBufQ_.FreeTensor(xLocal);
            }

            // Add bias
            AscendC::LocalTensor<float> bias = tmpQ_.AllocTensor<float>();
            CopyScalarIn(bias, bPackedGm_, b3Base + oc);
            AscendC::Add(acc, acc, bias, 1);
            tmpQ_.FreeTensor(bias);

            // No activation
            CopyScalarOut(yGm_, yRowBase + oc, acc);
            accQ_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_NUM> xBufQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_NUM> wBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_NUM> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_NUM> accQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wPackedGm_;
    AscendC::GlobalTensor<float> bPackedGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> y1Gm_;
    AscendC::GlobalTensor<float> y2Gm_;

    uint32_t batch_{0};
    uint32_t inSize_{0};
    uint32_t h1_{0};
    uint32_t h2_{0};
    uint32_t outSize_{0};
};

extern "C" __global__ __aicore__ void mlp_custom(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling, tilingGm);

    KernelMlpCustom op;
    op.Init(x, w_packed, b_packed, y,
            tiling.batch, tiling.inSize, tiling.h1, tiling.h2, tiling.outSize,
            workspace);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor mlp_impl_npu(const at::Tensor& x,
                        const at::Tensor& w_packed,
                        const at::Tensor& b_packed,
                        int64_t input_size,
                        int64_t layer1_size,
                        int64_t layer2_size,
                        int64_t output_size)
{
    // Ensure contiguous float32 for kernel
    at::Tensor x_c = x.contiguous().to(at::kFloat);
    at::Tensor w_c = w_packed.contiguous().to(at::kFloat);
    at::Tensor b_c = b_packed.contiguous().to(at::kFloat);
    std::vector<int64_t> y_sizes = {x_c.size(0), output_size};
    at::Tensor y = at::empty(y_sizes, x_c.options());

    EXEC_NPU_CMD(aclnnMlpCustom, x_c, w_c, b_c,
                 static_cast<int>(input_size),
                 static_cast<int>(layer1_size),
                 static_cast<int>(layer2_size),
                 static_cast<int>(output_size),
                 y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mlp_custom", &mlp_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("mlp_custom", &mlp_impl_npu, "Three-layer MLP fused (MlpCustom on NPU)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'mlp_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, layer_sizes, output_size):
        super(ModelNew, self).__init__()
        self.input_size = int(input_size)
        self.h1 = int(layer_sizes[0])
        self.h2 = int(layer_sizes[1])
        self.output_size = int(output_size)

    def forward(self, x, w_packed, b_packed):

        return fn(x, w_packed, b_packed, self.input_size, self.h1, self.h2, self.output_size)
