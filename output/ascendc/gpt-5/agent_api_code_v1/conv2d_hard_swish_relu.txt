    const uint32_t sW = static_cast<uint32_t>(*strideW);

    const uint32_t pL = static_cast<uint32_t>(*padLeft);

    if (blockDim == 0) blockDim = 1u;

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dHardSwishReluCustom {
public:
    __aicore__ inline KernelConv2dHardSwishReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t Hi, uint32_t Wi,
                                uint32_t Co, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo, uint32_t tileW)
    {
        N_ = N; Ci_ = Ci; Hi_ = Hi; Wi_ = Wi;
        Co_ = Co; Kh_ = Kh; Kw_ = Kw; Ho_ = Ho; Wo_ = Wo; tileW_ = tileW;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), N_ * Ci_ * Hi_ * Wi_);
        wGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w), Co_ * Ci_ * Kh_ * Kw_);
        if (b != nullptr) {
            bGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b), Co_);
            hasBias_ = true;
        } else {
            hasBias_ = false;
        }
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), N_ * Co_ * Ho_ * Wo_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(biasVecQueue_, 1, tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Co_ == 0 || Ho_ == 0 || Wo_ == 0) return;
        const uint32_t numTiles = N_ * Ho_; // each tile is one output row (Wo)
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tileId = blockIdx; tileId < numTiles; tileId += blockNum) {
            uint32_t n = tileId / Ho_;
            uint32_t ho = tileId % Ho_;

            // For each output channel
            for (uint32_t oc = 0; oc < Co_; ++oc) {
                // Allocate output accumulator vector
                LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
                bool firstTerm = true;

                // Accumulate over Ci, Kh, Kw
                for (uint32_t ci = 0; ci < Ci_; ++ci) {
                    for (uint32_t ky = 0; ky < Kh_; ++ky) {
                        for (uint32_t kx = 0; kx < Kw_; ++kx) {
                            // Load input row slice of length Wo_ starting at x = kx
                            LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                            // GM offset for input: ((n*Ci+ci)*Hi + (ho+ky)) * Wi + kx, then length Wo_
                            uint32_t xOffset = ((((n * Ci_ + ci) * Hi_ + (ho + ky)) * Wi_) + kx);
                            DataCopy(xLocal, xGm_[xOffset], Wo_);

                            // Multiply by weight scalar and accumulate
                            float wScalar = GetWeightScalar(oc, ci, ky, kx);
                            Muls(xLocal, xLocal, wScalar, Wo_);

                            if (firstTerm) {
                                DataCopy(yLocal, xLocal, Wo_);
                                firstTerm = false;
                            } else {
                                Add(yLocal, yLocal, xLocal, Wo_);
                            }

                            inQueue_.FreeTensor(xLocal);
                        }
                    }
                }

                // Add bias if present: broadcast bias[oc] over the row
                if (hasBias_) {
                    float bScalar = GetBiasScalar(oc);
                    Adds(yLocal, yLocal, bScalar, Wo_);
                }

                // HardSwish: y = y * relu6(y + 3) / 6
                LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                // tLocal = yLocal + 3
                DataCopy(tLocal, yLocal, Wo_);
                Adds(tLocal, tLocal, 3.0f, Wo_);
                // tLocal = min(max(tLocal, 0), 6)
                Maxs(tLocal, tLocal, 0.0f, Wo_);
                Mins(tLocal, tLocal, 6.0f, Wo_);
                // tLocal = tLocal / 6
                Muls(tLocal, tLocal, 1.0f / 6.0f, Wo_);
                // yLocal = yLocal * tLocal
                Mul(yLocal, yLocal, tLocal, Wo_);
                // ReLU (redundant but kept for functional parity)
                Maxs(yLocal, yLocal, 0.0f, Wo_);

                // Store to GM at y offset: ((n*Co+oc)*Ho + ho) * Wo
                uint32_t yOffset = ((((n * Co_ + oc) * Ho_) + ho) * Wo_);
                DataCopy(yGm_[yOffset], yLocal, Wo_);

                tmpQueue_.FreeTensor(tLocal);
                outQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline float GetWeightScalar(uint32_t oc, uint32_t ci, uint32_t ky, uint32_t kx)
    {
        // Linear index into weight [Co, Ci, Kh, Kw]
        uint32_t idx = (((oc * Ci_ + ci) * Kh_ + ky) * Kw_ + kx);
        // Fetch scalar from GM into a 1-element local tensor
        LocalTensor<float> t = tmpQueue_.AllocTensor<float>();
        DataCopy(t, wGm_[idx], 1);
        // NOTE: Adds/Muls accept scalar literals; here we rely on compiler to load from SRF.
        // We read back via LocalTensor read intrinsic.
        float v = t.GetValue(0);
        tmpQueue_.FreeTensor(t);
        return v;
    }

    __aicore__ inline float GetBiasScalar(uint32_t oc)
    {
        if (!hasBias_) return 0.0f;
        LocalTensor<float> t = tmpQueue_.AllocTensor<float>();
        DataCopy(t, bGm_[oc], 1);
        float v = t.GetValue(0);
        tmpQueue_.FreeTensor(t);
        return v;
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueue_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueue_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;
    TQue<TPosition::VECCALC, 1> biasVecQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t N_{0}, Ci_{0}, Hi_{0}, Wi_{0};
    uint32_t Co_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t tileW_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_hard_swish_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dHardSwishReluCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Ci, tiling_data.Hi, tiling_data.Wi,
            tiling_data.Co, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo, tiling_data.tileW);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor conv2d_hard_swish_relu_impl_npu(const at::Tensor& x,
                                                  const at::Tensor& weight,
                                                  c10::optional<at::Tensor> bias_opt) {
    // Compute output shape with stride=1, pad=0
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, \"x and weight must be 4D\");
    TORCH_CHECK(x.size(1) == weight.size(1), \"in_channels mismatch\");

    const int64_t N = x.size(0);
    const int64_t Ci = x.size(1);
    const int64_t Hi = x.size(2);
    const int64_t Wi = x.size(3);
    const int64_t Co = weight.size(0);
    const int64_t Kh = weight.size(2);
    const int64_t Kw = weight.size(3);

    const int64_t Ho = Hi - Kh + 1;
    const int64_t Wo = Wi - Kw + 1;
    TORCH_CHECK(Ho > 0 && Wo > 0, \"Invalid output spatial size\");

    at::Tensor result = at::empty({N, Co, Ho, Wo}, x.options());

    at::Tensor bias;
    if (bias_opt.has_value() && bias_opt.value().defined()) {
        bias = bias_opt.value();
        TORCH_CHECK(bias.dim() == 1 && bias.size(0) == Co, \"bias shape must be [out_channels]\");
    } else {
        bias = at::zeros({Co}, weight.options());
    }

    EXEC_NPU_CMD(aclnnConv2dHardSwishReluCustom, x, weight, bias, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv2d_hard_swish_relu_custom\", &conv2d_hard_swish_relu_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
