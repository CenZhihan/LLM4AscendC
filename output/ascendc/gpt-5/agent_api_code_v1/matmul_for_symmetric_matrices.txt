host_operator_src="""
#include "matmul_for_symmetric_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Inputs: A (N,N), B (N,N); Output: C (N,N)
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const int64_t m = a_shape->GetDim(0);
    const int64_t kA = a_shape->GetDim(1);
    const int64_t kB = b_shape->GetDim(0);
    const int64_t n = b_shape->GetDim(1);
    if (m <= 0 || n <= 0 || kA <= 0 || kB <= 0 || kA != kB || m != n) {
        // Require square matrices (N,N) and inner K match; symmetry is a contract from model.
        return ge::GRAPH_FAILED;
    }

    MatmulForSymmetricMatricesCustomTilingData tiling;
    // Minimal cube tiling settings; let kernel register Matmul object with default full-share.
    auto &cube = tiling.mutable_cubeTilingData();
    cube.shareMode = 0;               // enable shared mode
    cube.shareL1Size = FULL_L1_SIZE;  // use full L1
    cube.shareL0CSize = FULL_L0C_SIZE;// use full L0C
    cube.shareUbSize = 0;             // no dedicated UB
    cube.BatchNum = 1;                // single matmul

    // Choose a conservative blockDim to distribute rows; kernel uses builder scheduling.
    uint32_t blockDim = 8u;
    const uint32_t idealBlocks = static_cast<uint32_t>(m);
    if (idealBlocks < blockDim) blockDim = idealBlocks;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // No workspace required
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;

    // Save tiling
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    // Expect square matrices; but output shape is (A0, B1) like standard matmul
    const int64_t m = a_shape->GetDim(0);
    const int64_t n = b_shape->GetDim(1);
    gert::Shape *out = context->GetOutputShape(0);
    out->SetDimNum(2);
    out->SetDim(0, m);
    out->SetDim(1, n);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != ge::DT_FLOAT || b_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

class MatmulForSymmetricMatricesCustom : public OpDef {
public:
    explicit MatmulForSymmetricMatricesCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulForSymmetricMatricesCustom);

} // namespace ops

kernel_src="""
#include "kernel_operator.h"

// This kernel uses AscendC Matmul builder. We exploit symmetry by declaring B as transposed,
// so the kernel will read B rows (which are equal to columns for symmetric matrices), enabling
// more contiguous accesses internally while keeping mathematical equivalence.
extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                                                           GM_ADDR workspace, GM_ADDR tiling)
{
    // prepare tiling
    GET_TILING_DATA(tilingData, tiling);

    // define matmul kernel: inputs FP32, output FP32, no bias.
    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL>; // transB=true
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> matmulKernel;
    AscendC::TPipe pipe;

    // Share full local resources to matmul
    tilingData.cubeTilingData.shareMode = 0;
    tilingData.cubeTilingData.shareL1Size = FULL_L1_SIZE;
    tilingData.cubeTilingData.shareL0CSize = FULL_L0C_SIZE;
    tilingData.cubeTilingData.shareUbSize = 0;

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulKernel.matmulObj, &tilingData.cubeTilingData);

    // init matmul kernel
    matmulKernel.Init(A, B, nullptr, C, workspace, tilingData.cubeTilingData);

    // process single matmul (batch=1)
    matmulKernel.Process<false>(&pipe, 1, 1);
}
"""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor matmul_for_symmetric_matrices_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "Inputs must be rank-2 tensors");
    TORCH_CHECK(A.size(0) == A.size(1), "A must be square (symmetric by contract)");
    TORCH_CHECK(B.size(0) == B.size(1), "B must be square (symmetric by contract)");
    TORCH_CHECK(A.size(1) == B.size(0), "Inner dimensions must match");

    at::Tensor C = at::empty({A.size(0), B.size(1)}, A.options());
    EXEC_NPU_CMD(aclnnMatmulForSymmetricMatricesCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_for_symmetric_matrices_custom", &matmul_for_symmetric_matrices_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_for_symmetric_matrices_custom",
          &matmul_for_symmetric_matrices_impl_npu,
          "MatMul for symmetric matrices (MatmulForSymmetricMatricesCustom on NPU)");
}
