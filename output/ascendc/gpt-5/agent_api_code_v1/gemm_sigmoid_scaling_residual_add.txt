    // UB model for epilogue:
    //   inQueue: 2 * tile
    //   outQueue: 2 * tile
    //   tmpQueue: 2 * tile
    // total floats ~ 6 * tile => 24 * tile bytes
    static constexpr uint32_t UB_BYTES = 192 * 1024;
    static constexpr uint32_t ALIGN_ELEMS = 2048;
    uint32_t maxTile = UB_BYTES / 24;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    const uint64_t totalLen64 = static_cast<uint64_t>(M) * static_cast<uint64_t>(N);
    uint32_t totalLen = static_cast<uint32_t>(totalLen64);
    if (totalLen64 > 0xffffffffULL) {
        // large shapes fallback; cap at UINT32_MAX boundary
        totalLen = 0xffffffffu;
    }

    uint32_t tileLen = maxTile;
    if (totalLen < tileLen) {
        const uint32_t aligned = AlignDown(totalLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLen;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

kernel_src="""

#include "kernel_operator.h"

// Fused op: y = sigmoid(x @ w^T + b) * s + (x @ w^T + b)
// DataType: float32
// A: [M, K] GM ND, not transposed
// B: [N, K] GM ND, need transposed=true for MatmulType because we compute A * B^T effectively
// Bias: [N]
// C: [M, N]

using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL>;
using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

template <typename aType, typename bType, typename cType, typename biasType>
class MatmulTileKernel {
public:
    __aicore__ inline MatmulTileKernel() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR bias, GM_ADDR c, GM_ADDR workspace, const TCubeTiling &tiling)
    {
        tiling_ = tiling;

        aGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ aType *>(a), tiling_.M * tiling_.Ka);
        bGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ bType *>(b), tiling_.Kb * tiling_.N);
        cGlobal_.SetGlobalBuffer(reinterpret_cast__gm__<cType *>(c), tiling_.M * tiling_.N);
        biasGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ biasType *>(bias), tiling_.N);

        // Compute per-core GM offsets
        CalcOffset(AscendC::GetBlockIdx(), tiling_, offA_, offB_, offC_, offBias_);

        aGlobal_ = aGlobal_[offA_];
        bGlobal_ = bGlobal_[offB_];
        cGlobal_ = cGlobal_[offC_];
        biasGlobal_ = biasGlobal_[offBias_];

        (void)workspace;
    }

    __aicore__ inline void RunMatmul(AscendC::TPipe *pipe)
    {
        matmulObj_.SetTensorA(aGlobal_);
        matmulObj_.SetTensorB(bGlobal_);
        matmulObj_.SetBias(biasGlobal_);
        matmulObj_.IterateAll(cGlobal_);
        matmulObj_.End();
    }

    __aicore__ inline void GetCoreSpan(uint32_t &mStart, uint32_t &mLen, uint32_t &nStart, uint32_t &nLen) const
    {
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t mBlocks = AscendC::Ceiling(tiling_.M, tiling_.singleCoreM);
        const uint32_t mCore = blockIdx % mBlocks;
        const uint32_t nCore = blockIdx / mBlocks;

        mStart = mCore * tiling_.singleCoreM;
        nStart = nCore * tiling_.singleCoreN;
        mLen = tiling_.singleCoreM;
        nLen = tiling_.singleCoreN;

        if (mStart + mLen > static_cast<uint32_t>(tiling_.M)) {
            mLen = static_cast<uint32_t>(tiling_.M) - mStart;
        }
        if (nStart + nLen > static_cast<uint32_t>(tiling_.N)) {
            nLen = static_cast<uint32_t>(tiling_.N) - nStart;
        }
    }

    __aicore__ inline AscendC::GlobalTensor<cType> &CGlobal() { return cGlobal_; }
    __aicore__ inline const TCubeTiling &Tiling() const { return tiling_; }

private:
    __aicore__ inline void CalcOffset(int32_t blockIdx, const TCubeTiling &tiling, int32_t &offsetA, int32_t &offsetB,
                                      int32_t &offsetC, int32_t &offsetBias)
    {
        auto mSingleBlocks = AscendC::Ceiling(tiling.M, tiling.singleCoreM);
        auto mCoreIndx = blockIdx % mSingleBlocks;
        auto nCoreIndx = blockIdx / mSingleBlocks;

        offsetA = mCoreIndx * tiling.Ka * tiling.singleCoreM;
        offsetB = nCoreIndx * tiling.singleCoreN;
        offsetC = mCoreIndx * tiling.N * tiling.singleCoreM + nCoreIndx * tiling.singleCoreN;
        offsetBias = nCoreIndx * tiling.singleCoreN;
    }

private:
    TCubeTiling tiling_;
    AscendC::GlobalTensor<aType> aGlobal_;
    AscendC::GlobalTensor<bType> bGlobal_;
    AscendC::GlobalTensor<cType> cGlobal_;
    AscendC::GlobalTensor<biasType> biasGlobal_;
    int32_t offA_{0}, offB_{0}, offC_{0}, offBias_{0};

public:
    Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> matmulObj_;
};

constexpr int32_t VEC_INOUT_BUFFERS = 2;
constexpr uint32_t VEC_ALIGN = 2048;

class EpilogueSigmoidScaleResAdd {
public:
    __aicore__ inline EpilogueSigmoidScaleResAdd() {}

    __aicore__ inline void Init(GM_ADDR scaling, float &scaleOut)
    {
        AscendC::GlobalTensor<float> sGm;
        sGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(scaling), 1);
        // Read scaling scalar once from GM
        scaleOut = sGm.GetValue(0);
    }

    __aicore__ inline void ProcessTile(AscendC::TPipe &pipe,
                                       AscendC::GlobalTensor<float> &yGm,
                                       uint32_t gmOffset, uint32_t len,
                                       float scale)
    {
        if (len == 0) return;

        // Allocate buffers
        if (!vecInit_) {
            pipe.InitBuffer(inQ_, VEC_INOUT_BUFFERS, tileLen_ * sizeof(float));
            pipe.InitBuffer(outQ_, VEC_INOUT_BUFFERS, tileLen_ * sizeof(float));
            pipe.InitBuffer(tmpQ_, 2, tileLen_ * sizeof(float));
            vecInit_ = true;
        }

        uint32_t done = 0;
        while (done < len) {
            const uint32_t cur = AscendC::min(tileLen_, len - done);
            // Load
            AscendC::LocalTensor<float> in = inQ_.AllocTensor<float>();
            AscendC::DataCopy(in, yGm[gmOffset + done], cur);
            inQ_.EnQue(in);

            // Compute
            AscendC::LocalTensor<float> x = inQ_.DeQue<float>();
            AscendC::LocalTensor<float> y = outQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> t0 = tmpQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> t1 = tmpQ_.AllocTensor<float>();

            // t0 = -x
            AscendC::Muls(t0, x, -1.0f, cur);
            // t0 = exp(-x)
            AscendC::Exp(t0, t0, cur);
            // t0 = t0 + 1.0
            AscendC::Adds(t0, t0, 1.0f, cur);
            // t0 = 1.0 / t0  => sigmoid(x)
            AscendC::Reciprocal(t0, t0, cur);
            // t0 = t0 * scale
            AscendC::Muls(t0, t0, scale, cur);
            // y = t0 + x  => sigmoid(x)*s + x
            AscendC::Add(y, t0, x, cur);

            // Store
            tmpQ_.FreeTensor(t1);
            tmpQ_.FreeTensor(t0);
            inQ_.FreeTensor(x);
            outQ_.EnQue(y);

            AscendC::LocalTensor<float> yOut = outQ_.DeQue<float>();
            AscendC::DataCopy(yGm[gmOffset + done], yOut, cur);
            outQ_.FreeTensor(yOut);

            done += cur;
        }
    }

    __aicore__ inline void SetTileLen(uint32_t tileLen)
    {
        tileLen_ = tileLen;
        if (tileLen_ < VEC_ALIGN) tileLen_ = VEC_ALIGN;
        tileLen_ = (tileLen_ / VEC_ALIGN) * VEC_ALIGN;
    }

private:
    AscendC::TQue<AscendC::TPosition::VECIN,  VEC_INOUT_BUFFERS> inQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, VEC_INOUT_BUFFERS> outQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQ_;
    uint32_t tileLen_{32768};
    bool vecInit_{false};
};

extern "C" __global__ __aicore__ void gemm_sigmoid_scaling_residual_add_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
#if defined(ASCENDC_CPU_DEBUG)
    if (g_coreType == AscendC::AIV) {
        return;
    }
#endif
    // Prepare tiling for Matmul
    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    AscendC::TPipe pipe;

    // Register Matmul object with workspace and tiling
    MatmulTileKernel<float, float, float, float> mm;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm.matmulObj_, &tiling);
    // Init Matmul tiles and GM tensors
    mm.Init(x, w, b, y, workspace, tiling);
    // Run Matmul with bias into y
    mm.RunMatmul(&pipe);

    // Epilogue: y = sigmoid(y) * scaling + y (in-place)
    EpilogueSigmoidScaleResAdd ep;
    float scale = 1.0f;
    ep.Init(scaling, scale);

    // tileLen for vector stage (conservative)
    const uint32_t vecTile = 32768;
    ep.SetTileLen(vecTile);

    AscendC::GlobalTensor<float> &yGm = mm.CGlobal();
    // Determine this core's submatrix span
    uint32_t mStart=0, nStart=0, mLen=0, nLen=0;
    mm.GetCoreSpan(mStart, mLen, nStart, nLen);

    if (mLen == 0 || nLen == 0) {
        return;
    }

    // Process row by row within the core-assigned tile
    const uint32_t rowStride = static_cast<uint32_t>(tiling.N);
    for (uint32_t r = 0; r < mLen; ++r) {
        const uint32_t rowOffset = r * rowStride + nStart;
        ep.ProcessTile(pipe, yGm, rowOffset, nLen, scale);
    }
}
