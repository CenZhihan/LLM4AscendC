    const uint32_t H  = static_cast<uint32_t>(xO.GetDim(3));
    const uint32_t W  = static_cast<uint32_t>(xO.GetDim(4));

    const uint32_t Cout = static_cast<uint32_t>(wO.GetDim(0));
    const uint32_t KD   = static_cast<uint32_t>(wO.GetDim(2));
    const uint32_t KH   = static_cast<uint32_t>(wO.GetDim(3));
    const uint32_t KW   = static_cast<uint32_t>(wO.GetDim(4));

    if (Cin != static_cast<uint32_t>(wO.GetDim(1))) {
        return ge::GRAPH_FAILED;

    const uint32_t OW = (W >= KW) ? (W - KW + 1u) : 0u;

    uint32_t pairs = (N == 0 || Cout == 0) ? 1u : (N * Cout);
    uint32_t blockDim = pairs;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    tiling.set_Cin(Cin);
    tiling.set_D(D);
    tiling.set_H(H);

    tiling.set_KD(KD);
    tiling.set_KH(KH);
    tiling.set_KW(KW);

    tiling.set_OH(OH);
    tiling.set_OW(OW);

kernel_src="""
#include "kernel_operator.h"

class KernelConvStandard3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t D, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t KD, uint32_t KH, uint32_t KW,
                                uint32_t OD, uint32_t OH, uint32_t OW)
    {
        N_ = N; Cin_ = Cin; D_ = D; H_ = H; W_ = W;
        Cout_ = Cout; KD_ = KD; KH_ = KH; KW_ = KW;
        OD_ = OD; OH_ = OH; OW_ = OW;

        const uint64_t xSize = (uint64_t)N_ * Cin_ * D_ * H_ * W_;
        const uint64_t wSize = (uint64_t)Cout_ * Cin_ * KD_ * KH_ * KW_;
        const uint64_t ySize = (uint64_t)N_ * Cout_ * OD_ * OH_ * OW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // Buffers
        pipe_.InitBuffer(inLineQ_, 1, OW_ * sizeof(float));
        pipe_.InitBuffer(outRowQ_, 1, OW_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, OW_ * sizeof(float));
        pipe_.InitBuffer(wLocalQ_, 1, (Cin_ * KD_ * KH_ * KW_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cout_ == 0 || OD_ == 0 || OH_ == 0 || OW_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalPairs = N_ * Cout_;

        for (uint32_t idx = blockIdx; idx < totalPairs; idx += blockNum) {
            uint32_t n = idx / Cout_;
            uint32_t co = idx % Cout_;

            // Load weight for (co, :, :, :, :)
            const uint32_t wLen = Cin_ * KD_ * KH_ * KW_;
            AscendC::LocalTensor<float> wLocal = wLocalQ_.AllocTensor<float>();
            const uint64_t wBase = (uint64_t)co * wLen;
            AscendC::DataCopy(wLocal, wGm_[wBase], wLen);
            wLocalQ_.EnQue(wLocal); // keep resident

            // For each (od, oh), compute OW-length row
            for (uint32_t od = 0; od < OD_; ++od) {
                for (uint32_t oh = 0; oh < OH_; ++oh) {
                    AscendC::LocalTensor<float> yRow = outRowQ_.AllocTensor<float>();
                    bool yRowInit = false;

                    for (uint32_t c = 0; c < Cin_; ++c) {
                        for (uint32_t kd = 0; kd < KD_; ++kd) {
                            for (uint32_t kh = 0; kh < KH_; ++kh) {
                                for (uint32_t kw = 0; kw < KW_; ++kw) {
                                    // Load input line of length OW_ starting at w = kw
                                    const uint64_t xBase =
                                        ((((uint64_t)n * Cin_ + c) * D_ + (od + kd)) * H_ + (oh + kh)) * W_ + kw;

                                    AscendC::LocalTensor<float> xLine = inLineQ_.AllocTensor<float>();
                                    AscendC::DataCopy(xLine, xGm_[xBase], OW_);
                                    inLineQ_.EnQue(xLine);

                                    // Weight scalar index for this (c,kd,kh,kw)
                                    const uint32_t wIdx = (((c * KD_ + kd) * KH_ + kh) * KW_ + kw);
                                    // Read scalar from local weights
                                    float wScalar = wLocal.GetValue(wIdx);

                                    if (!yRowInit) {
                                        AscendC::Muls(yRow, xLine, wScalar, OW_);
                                        yRowInit = true;
                                    } else {
                                        AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                                        AscendC::Muls(tmp, xLine, wScalar, OW_);
                                        AscendC::Add(yRow, yRow, tmp, OW_);
                                        tmpQ_.FreeTensor(tmp);
                                    }

                                    AscendC::LocalTensor<float> xLineFree = inLineQ_.DeQue<float>();
                                    inLineQ_.FreeTensor(xLineFree);
                                }
                            }
                        }
                    }

                    // Store y row
                    const uint64_t yBase =
                        ((((uint64_t)n * Cout_ + co) * OD_ + od) * OH_ + oh) * OW_;
                    AscendC::DataCopy(yGm_[yBase], yRow, OW_);
                    outRowQ_.FreeTensor(yRow);
                }
            }

            AscendC::LocalTensor<float> wLocalFree = wLocalQ_.DeQue<float>();
            wLocalQ_.FreeTensor(wLocalFree);
        }
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, 1> inLineQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wLocalQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, D_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, KD_{0}, KH_{0}, KW_{0};
    uint32_t OD_{0}, OH_{0}, OW_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvStandard3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.Cout, tiling_data.KD, tiling_data.KH, tiling_data.KW,
            tiling_data.OD, tiling_data.OH, tiling_data.OW);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv3d_standard_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    TORCH_CHECK(x.dim() == 5 && weight.dim() == 5, "Expect 5D input and weight");
    TORCH_CHECK(x.scalar_type() == at::kFloat && weight.scalar_type() == at::kFloat, "Only float supported");

    // x: [N, Cin, D, H, W], weight: [Cout, Cin, KD, KH, KW]
    const int64_t N  = x.size(0);
    const int64_t Cin= x.size(1);
    const int64_t D  = x.size(2);
    const int64_t H  = x.size(3);
    const int64_t W  = x.size(4);

    const int64_t Cout = weight.size(0);
    const int64_t KD   = weight.size(2);
    const int64_t KH   = weight.size(3);
    const int64_t KW   = weight.size(4);

    TORCH_CHECK(Cin == weight.size(1), "Cin mismatch between input and weight");
    TORCH_CHECK(KD > 0 && KH > 0 && KW > 0, "Invalid kernel size");
    TORCH_CHECK(D >= KD && H >= KH && W >= KW, "Kernel larger than input with no padding");

    const int64_t OD = D - KD + 1;
    const int64_t OH = H - KH + 1;
    const int64_t OW = W - KW + 1;

    at::Tensor y = at::empty({N, Cout, OD, OH, OW}, x.options());

    EXEC_NPU_CMD(aclnnConvStandard3dSquareInputSquareKernelCustom, x, weight, y);
    return y;
}

    m.impl("conv_standard_3d_square_input_square_kernel_custom", &conv3d_standard_impl_npu);
