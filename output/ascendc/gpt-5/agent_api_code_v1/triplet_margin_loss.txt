host_operator_src="""
#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core UB budget
// Buffers per tile:
//   inQueueA: 1 * tile
//   inQueueP: 1 * tile
//   inQueueN: 1 * tile
//   diff1:    1 * tile
//   diff2:    1 * tile
//   reduceWs: 1 * tile
// Total floats ~= 6 * tile -> bytes = 24 * tile
static constexpr uint32_t BYTES_PER_ELEM = 4;
static constexpr uint32_t ALIGN_ELEMS = 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *p_shape = context->GetInputShape(1);
    const gert::Shape *n_shape = context->GetInputShape(2);

    // Expect [N, D] for all
    uint32_t dimNum = static_cast<uint32_t>(a_shape->GetDimNum());
    if (dimNum == 0) {
        return ge::GRAPH_FAILED;
    }
    uint32_t nRows = 1;
    uint32_t rowSize = 1;
    if (dimNum == 1) {
        nRows = 1;
        rowSize = static_cast<uint32_t>(a_shape->GetDim(0));
    } else {
        // collapse last dimension as row size
        rowSize = static_cast<uint32_t>(a_shape->GetDim(dimNum - 1));
        for (uint32_t i = 0; i + 1 < dimNum; ++i) {
            nRows *= static_cast<uint32_t>(a_shape->GetDim(i));
        }
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // Compute tile length from UB budget: 24 * tileLen bytes
    uint32_t maxTile = UB_BUDGET_BYTES / (24);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > rowSize) {
        uint32_t aligned = AlignDown(rowSize, ALIGN_ELEMS);
        maxTile = (aligned > 0) ? aligned : rowSize;
    }
    uint32_t tileLen = maxTile;
    if (tileLen == 0) tileLen = rowSize;

    // For simplicity and correctness of reduction, use a single core
    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_nRows(nRows);
    tiling.set_rowSize(rowSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace needed for blockDim == 1
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float EPSILON_DEFAULT = 1e-12f;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y,
                                uint32_t nRows, uint32_t rowSize, uint32_t tileLen, float margin)
    {
        nRows_ = nRows;
        rowSize_ = rowSize;
        tileLen_ = tileLen;
        margin_ = margin;

        aGm_.SetGlobalBuffer((__gm__ float*)anchor, (uint64_t)nRows_ * rowSize_);
        pGm_.SetGlobalBuffer((__gm__ float*)positive, (uint64_t)nRows_ * rowSize_);
        nGm_.SetGlobalBuffer((__gm__ float*)negative, (uint64_t)nRows_ * rowSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        pipe_.InitBuffer(inA_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inP_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inN_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(diff1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(diff2_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(reduceWs_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarBuf_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (nRows_ == 0 || rowSize_ == 0 || tileLen_ == 0) {
            yGm_.SetValue(0, 0.0f);
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        float totalLoss = 0.0f;
        uint32_t processedRows = 0;

        for (uint32_t row = blockIdx; row < nRows_; row += blockNum) {
            float sumAp = 0.0f;
            float sumAn = 0.0f;
            uint64_t base = static_cast<uint64_t>(row) * rowSize_;

            for (uint32_t off = 0; off < rowSize_; off += tileLen_) {
                uint32_t len = rowSize_ - off;
                if (len > tileLen_) len = tileLen_;

                AscendC::LocalTensor<float> aLocal = inA_.AllocTensor<float>();
                AscendC::LocalTensor<float> pLocal = inP_.AllocTensor<float>();
                AscendC::LocalTensor<float> nLocal = inN_.AllocTensor<float>();
                AscendC::DataCopy(aLocal, aGm_[base + off], len);
                AscendC::DataCopy(pLocal, pGm_[base + off], len);
                AscendC::DataCopy(nLocal, nGm_[base + off], len);

                AscendC::LocalTensor<float> d1 = diff1_.AllocTensor<float>();
                AscendC::LocalTensor<float> d2 = diff2_.AllocTensor<float>();
                AscendC::Sub(d1, aLocal, pLocal, len); // a - p
                AscendC::Sub(d2, aLocal, nLocal, len); // a - n
                AscendC::Mul(d1, d1, d1, len);         // (a-p)^2
                AscendC::Mul(d2, d2, d2, len);         // (a-n)^2

                AscendC::LocalTensor<float> ws = reduceWs_.AllocTensor<float>();
                AscendC::LocalTensor<float> sc1 = scalarBuf_.AllocTensor<float>();
                AscendC::ReduceSum(sc1, d1, ws, static_cast<int32_t>(len));
                sumAp += sc1(0);
                scalarBuf_.FreeTensor(sc1);

                AscendC::LocalTensor<float> sc2 = scalarBuf_.AllocTensor<float>();
                AscendC::ReduceSum(sc2, d2, ws, static_cast<int32_t>(len));
                sumAn += sc2(0);
                scalarBuf_.FreeTensor(sc2);

                reduceWs_.FreeTensor(ws);
                diff2_.FreeTensor(d2);
                diff1_.FreeTensor(d1);
                inN_.FreeTensor(nLocal);
                inP_.FreeTensor(pLocal);
                inA_.FreeTensor(aLocal);
            }

            float ap = sqrtf(sumAp + EPSILON_DEFAULT);
            float an = sqrtf(sumAn + EPSILON_DEFAULT);
            float val = ap - an + margin_;
            if (val < 0.0f) val = 0.0f;
            totalLoss += val;
            processedRows += 1;
        }

        // blockDim is set to 1 in tiling for correctness; write result directly
        if (processedRows == 0) {
            yGm_.SetValue(0, 0.0f);
        } else {
            float meanLoss = totalLoss / static_cast<float>(processedRows);
            yGm_.SetValue(0, meanLoss);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inP_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inN_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> diff1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> diff2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> reduceWs_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scalarBuf_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> pGm_;
    AscendC::GlobalTensor<float> nGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t nRows_ {0};
    uint32_t rowSize_ {0};
    uint32_t tileLen_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                 GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y,
            tiling_data.nRows, tiling_data.rowSize, tiling_data.tileLength, tiling_data.margin);
    op.Process();
}

#include <torch/library.h>

at::Tensor triplet_margin_loss_impl_npu(const at::Tensor& anchor,
                                        const at::Tensor& positive,
                                        const at::Tensor& negative,
                                        const at::Tensor& margin_tensor) {
    double m = 0.0;
    if (margin_tensor.defined() && margin_tensor.numel() > 0) {
        at::Tensor flat = margin_tensor.reshape({-1});
        m = flat[0].item<double>();
    }
    // Allocate scalar as 1-element and squeeze to return 0-dim
    at::Tensor result = at::empty({1}, anchor.options());

    return result.squeeze();

    m.impl("triplet_margin_loss_custom", &triplet_margin_loss_impl_npu);
}
