namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t MIN_BLOCKS = 8;
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t a0 = static_cast<uint64_t>(a_shape->GetOriginShape().GetDim(0));
    const uint64_t a1 = static_cast<uint64_t>(a_shape->GetOriginShape().GetDim(1));
    const uint64_t b0 = static_cast<uint64_t>(b_shape->GetOriginShape().GetDim(0));
    const uint64_t b1 = static_cast<uint64_t>(b_shape->GetOriginShape().GetDim(1));

    if (a1 != b0) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(a0);
    const uint32_t K = static_cast<uint32_t>(a1);
    const uint32_t N = static_cast<uint32_t>(b1);

    // Choose small column tile (nTile) to suit tall-skinny (small N) and reduce UB pressure.
    uint32_t nTile = (N < 16u) ? N : 16u;
    if (nTile == 0u) nTile = 1u;

    // Choose kStep as modest segment for K streaming.
    uint32_t kStep = (K < 64u) ? K : 64u;
    if (kStep == 0u) kStep = 1u;

    // UB usage:
    //  - B_block: kStep * nTile floats
    //  - A_row_segment: kStep floats
    //  - yRow + tmp: 2 * nTile floats
    // Total bytes = 4 * (kStep * nTile + kStep + 2 * nTile)
    const uint32_t baseBytesPerRowTile = 4u * (kStep * nTile + kStep + 2u * nTile);

    // Remaining UB for accumulating multiple rows at once (but kernel computes per-row across kStep, stores/accumulates in GM)
    // We process rows one-by-one in kernel to keep logic simple; mTile controls how many rows per core per iteration.
    // Keep mTile moderate to increase parallelism; however UB is not the constraint since we don't keep all rows in UB.
    // Use mTile primarily to guide block scheduling granularity.
    uint32_t tentative_mTile = 512u;  // heuristic for tall-skinny
    if (tentative_mTile > M) tentative_mTile = M;
    if (tentative_mTile == 0u) tentative_mTile = 1u;

    // Set blockDim based on number of row tiles
    uint32_t numRowTiles = CeilDiv(M, tentative_mTile);
    uint32_t blockDim = numRowTiles;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_mTile(tentative_mTile);
    tiling.set_nTile(nTile);
    tiling.set_kStep(kStep);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t mTile, uint32_t nTile, uint32_t kStep)
    {
        M_ = M; N_ = N; K_ = K;
        mTile_ = (mTile == 0u) ? 1u : mTile;
        nTile_ = (nTile == 0u) ? 1u : nTile;
        kStep_ = (kStep == 0u) ? 1u : kStep;

        aGm_.SetGlobalBuffer((__gm__ float*)a, (uint64_t)M_ * (uint64_t)K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)K_ * (uint64_t)N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, (uint64_t)M_ * (uint64_t)N_);

        // UB buffers
        // B_block: kStep * nTile
        pipe_.InitBuffer(bBlockQueue_, BUFFER_NUM, (uint64_t)kStep_ * (uint64_t)nTile_ * sizeof(float));
        // A_row_segment: kStep
        pipe_.InitBuffer(aSegQueue_, BUFFER_NUM, (uint64_t)kStep_ * sizeof(float));
        // yRow: nTile
        pipe_.InitBuffer(yRowQueue_, BUFFER_NUM, (uint64_t)nTile_ * sizeof(float));
        // tmp: nTile
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, (uint64_t)nTile_ * sizeof(float));
        // cPrev: nTile
        pipe_.InitBuffer(cPrevQueue_, BUFFER_NUM, (uint64_t)nTile_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0u || N_ == 0u || K_ == 0u) { return; }

        const uint32_t blkNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blkIdx = (uint32_t)AscendC::GetBlockIdx();

        // Grid-stride over row tiles
        const uint64_t rowStride = (uint64_t)blkNum * (uint64_t)mTile_;
        uint64_t rowBase = (uint64_t)blkIdx * (uint64_t)mTile_;

        for (; rowBase < M_; rowBase += rowStride) {
            uint32_t rowsThisTile = (uint32_t)AscendC::min((uint64_t)mTile_, (uint64_t)M_ - rowBase);

            // Iterate over column tiles
            for (uint32_t nStart = 0; nStart < N_; nStart += nTile_) {
                uint32_t nSegLen = AscendC::min(nTile_, N_ - nStart);

                // Iterate over K in steps
                for (uint32_t kBase = 0; kBase < K_; kBase += kStep_) {
                    uint32_t kSegLen = AscendC::min(kStep_, K_ - kBase);

                    // Load B_block(kSegLen x nSegLen) into UB as row-major rows of length nSegLen
                    AscendC::LocalTensor<float> bBlock = bBlockQueue_.AllocTensor<float>();
                    for (uint32_t t = 0; t < kSegLen; ++t) {
                        uint64_t bOffset = ((uint64_t)(kBase + t) * (uint64_t)N_) + (uint64_t)nStart;
                        AscendC::DataCopy(bBlock[t * nSegLen], bGm_[bOffset], nSegLen);
                    }
                    bBlockQueue_.EnQue(bBlock);

                    // For each row in this tile
                    for (uint32_t r = 0; r < rowsThisTile; ++r) {
                        const uint64_t aRowBase = ((uint64_t)(rowBase + r) * (uint64_t)K_) + (uint64_t)kBase;
                        // Load A_row_segment (kSegLen)
                        AscendC::LocalTensor<float> aSeg = aSegQueue_.AllocTensor<float>();
                        AscendC::DataCopy(aSeg, aGm_[aRowBase], kSegLen);
                        aSegQueue_.EnQue(aSeg);

                        // Compute yRow = sum_{t=0..kSegLen-1} (aSeg[t] * bBlock[t, :])
                        AscendC::LocalTensor<float> yRow = yRowQueue_.AllocTensor<float>();
                        // Initialize with t=0
                        AscendC::LocalTensor<float> bRow0 = bBlockQueue_.Get<float>()[0 * nSegLen];
                        float a0;
                        // Load first scalar aSeg[0] into tmp one-element buffer via DataCopy to local tmp then read
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, aSeg, 1u);
                        // Assuming we can read back scalar from local tensor by vector to scalar conversion
                        a0 = *((float*)tmp.GetPhyAddr()); // direct access for illustration
                        AscendC::Muls(yRow, bRow0, a0, nSegLen);

                        // Accumulate for t=1..kSegLen-1
                        for (uint32_t t = 1; t < kSegLen; ++t) {
                            AscendC::LocalTensor<float> bRow = bBlockQueue_.Get<float>()[t * nSegLen];
                            // Copy aSeg[t] into tmp[0]
                            AscendC::DataCopy(tmp, aSeg[t], 1u);
                            float aval = *((float*)tmp.GetPhyAddr());
                            AscendC::LocalTensor<float> tmpVec = tmpQueue_.Get<float>();
                            AscendC::Muls(tmpVec, bRow, aval, nSegLen);
                            AscendC::Add(yRow, yRow, tmpVec, nSegLen);
                        }

                        // Accumulate into C: if first kBase, write; else add to existing
                        const uint64_t cRowBase = ((uint64_t)(rowBase + r) * (uint64_t)N_) + (uint64_t)nStart;
                        if (kBase == 0u) {
                            AscendC::DataCopy(cGm_[cRowBase], yRow, nSegLen);
                        } else {
                            AscendC::LocalTensor<float> cPrev = cPrevQueue_.AllocTensor<float>();
                            AscendC::DataCopy(cPrev, cGm_[cRowBase], nSegLen);
                            AscendC::Add(cPrev, cPrev, yRow, nSegLen);
                            AscendC::DataCopy(cGm_[cRowBase], cPrev, nSegLen);
                            cPrevQueue_.FreeTensor(cPrev);
                        }

                        // Free buffers for this row
                        tmpQueue_.FreeTensor(tmp);
                        yRowQueue_.FreeTensor(yRow);
                        aSegQueue_.FreeTensor(aSeg);
                    }

                    // Free B_block
                    bBlockQueue_.FreeTensor(bBlock);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bBlockQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aSegQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> cPrevQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t mTile_{1}, nTile_{1}, kStep_{1};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.mTile, tiling_data.nTile, tiling_data.kStep);
    op.Process();
}
