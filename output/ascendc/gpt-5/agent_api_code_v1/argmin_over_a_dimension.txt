#include "argmin_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}

static inline int64_t SafeProd(const std::vector<int64_t>& dims, size_t begin, size_t end)
{
    int64_t prod = 1;
    for (size_t i = begin; i < end; ++i) prod *= dims[i];
    return prod < 0 ? 0 : prod;
}

} // anonymous namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgminOverADimensionCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const size_t rank = shape->GetOriginShape().GetDimNum();
    std::vector<int64_t> dims(rank);
    for (size_t i = 0; i < rank; ++i) {
        dims[i] = shape->GetOriginShape().GetDim(i);
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dimAttr;
    if (dim < 0) dim += static_cast<int32_t>(rank);
    if (dim < 0 || dim >= static_cast<int32_t>(rank)) {
        return ge::GRAPH_FAILED;
    }

    int64_t outer64 = SafeProd(dims, 0, static_cast<size_t>(dim));
    int64_t reduce64 = dims[static_cast<size_t>(dim)];
    int64_t inner64 = SafeProd(dims, static_cast<size_t>(dim + 1), rank);

    if (outer64 <= 0) outer64 = 1;
    if (inner64 <= 0) inner64 = 1;

    uint32_t outer = static_cast<uint32_t>(outer64);
    uint32_t reduce = static_cast<uint32_t>(reduce64);
    uint32_t inner = static_cast<uint32_t>(inner64);

    uint32_t blockDim = outer;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_outerSize(outer);
    tiling.set_reduceSize(reduce);
    tiling.set_innerSize(inner);
    tiling.set_blockDim(blockDim);
    tiling.set_dimCanonical(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    std::vector<int64_t> out_dims;
    out_dims.reserve(x_shape->GetDimNum() > 0 ? x_shape->GetDimNum() - 1 : 0);

    int32_t dim = 0;
    if (context->GetAttrs() != nullptr) {
        const int32_t* p = context->GetAttrs()->GetAttrPointer<int32_t>(0);
        dim = p ? *p : 0;
    }
    int32_t rank = static_cast<int32_t>(x_shape->GetDimNum());
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    for (int32_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out_dims.push_back(x_shape->GetDim(i));
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelArgminOverADimensionCustom {
public:
    __aicore__ inline KernelArgminOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerSize, uint32_t reduceSize, uint32_t innerSize)
    {
        xPtr_ = reinterpret_cast<__gm__ float*>(x);
        yPtr_ = reinterpret_cast<__gm__ int64_t*>(y);
        outer_ = outerSize;
        reduce_ = reduceSize;
        inner_ = innerSize;
    }

    __aicore__ inline void Process()
    {
        if (outer_ == 0 || inner_ == 0 || reduce_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        for (uint32_t o = blockIdx; o < outer_; o += blockNum) {
            uint64_t base_outer_in = static_cast<uint64_t>(o) * static_cast<uint64_t>(reduce_) * static_cast<uint64_t>(inner_);
            uint64_t base_outer_out = static_cast<uint64_t>(o) * static_cast<uint64_t>(inner_);
            for (uint32_t i = 0; i < inner_; ++i) {
                uint64_t base = base_outer_in + static_cast<uint64_t>(i);
                float minv = xPtr_[base];
                int64_t arg = 0;
                for (uint32_t r = 1; r < reduce_; ++r) {
                    float v = xPtr_[base + static_cast<uint64_t>(r) * static_cast<uint64_t>(inner_)];
                    if (v < minv) {
                        minv = v;
                        arg = static_cast<int64_t>(r);
                    }
                }
                yPtr_[base_outer_out + static_cast<uint64_t>(i)] = arg;
            }
        }
    }

private:
    __gm__ float* xPtr_ {nullptr};
    __gm__ int64_t* yPtr_ {nullptr};
    uint32_t outer_ {0};
    uint32_t reduce_ {0};
    uint32_t inner_ {0};
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgminOverADimensionCustom op;
    op.Init(x, y, tiling_data.outerSize, tiling_data.reduceSize, tiling_data.innerSize);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline std::vector<int64_t> compute_reduced_sizes(const at::Tensor& self, int64_t dim) {
    auto sizes = self.sizes().vec();
    int64_t rank = static_cast<int64_t>(sizes.size());
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) {
        TORCH_CHECK(false, \"Invalid dim for argmin_over_a_dimension_custom\");
    }
    sizes.erase(sizes.begin() + dim);
    return sizes;
}

at::Tensor argmin_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    auto out_sizes = compute_reduced_sizes(self, dim);
    at::Tensor result = at::empty(out_sizes, self.options().dtype(at::kLong));
    EXEC_NPU_CMD(aclnnArgminOverADimensionCustom, self, static_cast<int>(dim), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmin_over_a_dimension_custom", &argmin_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmin_over_a_dimension_custom", &argmin_over_a_dimension_impl_npu, "ArgMin over a dimension (ArgminOverADimensionCustom on NPU)");
