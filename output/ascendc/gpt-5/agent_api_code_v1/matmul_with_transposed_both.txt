static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedBothCustomTilingData tiling;

    // Expect A shape: (K, M), B shape: (N, K)
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const uint64_t a0 = static_cast<uint64_t>(a_shape->GetOriginShape().GetDim(0));
    const uint64_t a1 = static_cast<uint64_t>(a_shape->GetOriginShape().GetDim(1));
    const uint64_t b0 = static_cast<uint64_t>(b_shape->GetOriginShape().GetDim(0));
    const uint64_t b1 = static_cast<uint64_t>(b_shape->GetOriginShape().GetDim(1));

    const uint32_t K = static_cast<uint32_t>(a0);
    const uint32_t M = static_cast<uint32_t>(a1);
    const uint32_t N = static_cast<uint32_t>(b0);
    (void)b1; // b1 should equal K

    // Conservative tile choices to limit UB usage
    uint32_t tileM = 128;
    uint32_t tileN = 128;
    uint32_t tileK = 64;

    if (M < tileM) tileM = (M == 0) ? 1u : M;
    if (N < tileN) tileN = (N == 0) ? 1u : N;
    if (K < tileK) tileK = (K == 0) ? 1u : K;

    const uint32_t mTiles = (tileM == 0) ? 1u : CeilDiv(M, tileM);
    const uint32_t nTiles = (tileN == 0) ? 1u : CeilDiv(N, tileN);
    uint32_t blockDim = mTiles * nTiles;
    if (blockDim == 0) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelMatmulWithTransposedBothCustom {
public:
    __aicore__ inline KernelMatmulWithTransposedBothCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tM_ = tileM; tN_ = tileN; tK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(K_) * static_cast<uint64_t>(M_));
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(N_) * static_cast<uint64_t>(K_));
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(M_) * static_cast<uint64_t>(N_));

        // Allocate minimal UB buffers (1-element scalars and vectors up to max row size)
        // Accumulator tile buffer: D_sub (nLen x mLen), but to keep UB minimal, we compute per (nn, mm) using 1-element accumulation
        pipe_.InitBuffer(aRowBuf_, 1, tM_ * sizeof(float)); // Row of A length tM
        pipe_.InitBuffer(accBuf_, 1, sizeof(float));        // Accumulator scalar
        pipe_.InitBuffer(tmpBuf_, 1, sizeof(float));        // Temp scalar multiply
        pipe_.InitBuffer(scalarBufA_, 1, sizeof(float));    // One element from A
        pipe_.InitBuffer(scalarBufB_, 1, sizeof(float));    // One element from B
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t mTiles = CeilDiv(M_, tM_);
        const uint32_t nTiles = CeilDiv(N_, tN_);
        const uint32_t totalTiles = mTiles * nTiles;
        if (totalTiles == 0) return;

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t mTileIdx = tileId % mTiles;
            const uint32_t nTileIdx = tileId / mTiles;

            const uint32_t mStart = mTileIdx * tM_;
            const uint32_t nStart = nTileIdx * tN_;
            const uint32_t mLen = MinUint(tM_, M_ - mStart);
            const uint32_t nLen = MinUint(tN_, N_ - nStart);

            // Compute C[mStart:mStart+mLen, nStart:nStart+nLen] = A^T * B^T
            // i.e., C(m, n) = sum_k A(k, m) * B(n, k)

            for (uint32_t mm = 0; mm < mLen; ++mm) {
                for (uint32_t nn = 0; nn < nLen; ++nn) {
                    AscendC::LocalTensor<float> acc = accBuf_.AllocTensor<float>();
                    AscendC::Dup(acc, 0.0f, 1);

                    // Accumulate over K in tiles for locality (still scalar loads per step)
                    for (uint32_t kBase = 0; kBase < K_; kBase += tK_) {
                        const uint32_t kLen = MinUint(tK_, K_ - kBase);

                        // Iterate k
                        for (uint32_t kk = 0; kk < kLen; ++kk) {
                            const uint32_t kIdx = kBase + kk;

                            // Load scalar A(kIdx, mStart + mm)
                            AscendC::LocalTensor<float> aScalar = scalarBufA_.AllocTensor<float>();
                            AscendC::DataCopy(aScalar, aGm_[static_cast<uint64_t>(kIdx) * M_ + (mStart + mm)], 1);

                            // Load scalar B(nStart + nn, kIdx)
                            AscendC::LocalTensor<float> bScalar = scalarBufB_.AllocTensor<float>();
                            AscendC::DataCopy(bScalar, bGm_[static_cast<uint64_t>(nStart + nn) * K_ + kIdx], 1);

                            // temp = aScalar * bScalar
                            AscendC::LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>();
                            AscendC::Mul(tmp, aScalar, bScalar, 1);

                            // acc += temp
                            AscendC::Add(acc, acc, tmp, 1);

                            tmpBuf_.FreeTensor(tmp);
                            scalarBufB_.FreeTensor(bScalar);
                            scalarBufA_.FreeTensor(aScalar);
                        }
                    }

                    // Write C(mStart+mm, nStart+nn)
                    AscendC::DataCopy(cGm_[static_cast<uint64_t>(mStart + mm) * N_ + (nStart + nn)], acc, 1);
                    accBuf_.FreeTensor(acc);
                }
            }
        }
    }

private:
    __aicore__ inline static uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
    __aicore__ inline static uint32_t MinUint(uint32_t a, uint32_t b) { return a < b ? a : b; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> aRowBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBufA_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBufB_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0}, N_ {0}, K_ {0};
    uint32_t tM_ {0}, tN_ {0}, tK_ {0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedBothCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
