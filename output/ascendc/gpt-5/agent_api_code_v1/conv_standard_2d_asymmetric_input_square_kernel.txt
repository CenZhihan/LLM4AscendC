static ge::graphStatus TilingFunc(gert::TilingContext *context)

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect NCHW for x, [Cout, Cin, Kh, Kw] for weight
    const uint32_t n   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t cIn = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t h   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t w   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t cOut = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t kH   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t kW   = static_cast<uint32_t>(w_shape->GetDim(3));

    // Fixed stride=(1,1), padding=(0,0), dilation=(1,1), groups=1
    const uint32_t outH = (h >= kH) ? (h - kH + 1) : 0u;
    const uint32_t outW = (w >= kW) ? (w - kW + 1) : 0u;

    // To simplify kernel logic, we tile along width by 1 element.
    const uint32_t tileW = 1u;

    // Choose blockDim to parallelize across output channels conservatively.
    uint32_t blockDim = (cOut > 0u) ? cOut : 1u;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;
    context->SetBlockDim(blockDim);

    tiling.set_cIn(cIn);
    tiling.set_cOut(cOut);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUF_NUM = 1;

class KernelConvStandard2dAsymSq {
public:
    __aicore__ inline KernelConvStandard2dAsymSq() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t h, uint32_t wdim,
                                uint32_t kH, uint32_t kW,
                                uint32_t outH, uint32_t outW,
                                uint32_t tileW)
    {
        n_ = n; cIn_ = cIn; cOut_ = cOut;
        h_ = h; w_ = wdim; kH_ = kH; kW_ = kW;
        outH_ = outH; outW_ = outW;
        tileW_ = tileW; // host sets tileW_=1

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * cIn_ * h_ * w_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(cOut_ * cIn_ * kH_ * kW_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_ * cOut_ * outH_ * outW_));

        pipe_.InitBuffer(inX_, BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(inW_, BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpMul_, BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(accY_, BUF_NUM, tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cIn_ == 0 || cOut_ == 0 || outH_ == 0 || outW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a subset of output channels
        for (uint32_t oc = blockIdx; oc < cOut_; oc += (blockNum == 0 ? 1u : blockNum)) {
            for (uint32_t n = 0; n < n_; ++n) {
                for (uint32_t oh = 0; oh < outH_; ++oh) {
                    // process width in tiles of tileW_ (tileW_==1)
                    for (uint32_t owBase = 0; owBase < outW_; owBase += tileW_) {
                        const uint32_t curTileW = (owBase + tileW_ <= outW_) ? tileW_ : (outW_ - owBase);

                        bool firstAcc = true;
                        // Accumulate over input channels and kernel window
                        for (uint32_t ic = 0; ic < cIn_; ++ic) {
                            for (uint32_t kh = 0; kh < kH_; ++kh) {
                                const uint32_t ih = oh + kh; // stride 1, pad 0
                                for (uint32_t kw = 0; kw < kW_; ++kw) {
                                    // Compute indices
                                    const uint32_t iwBase = owBase + kw;

                                    // Load x tile (length curTileW, contiguous)
                                    AscendC::LocalTensor<float> xLocal = inX_.AllocTensor<float>();
                                    AscendC::DataCopy(xLocal, xGm_[XOffset(n, ic, ih, iwBase)], curTileW);
                                    inX_.EnQue(xLocal);

                                    // Load w tile (length curTileW == 1; weights are scalar per position replicated per tile width 1)
                                    AscendC::LocalTensor<float> wLocal = inW_.AllocTensor<float>();
                                    AscendC::DataCopy(wLocal, wGm_[WOffset(oc, ic, kh, kw)], 1);
                                    inW_.EnQue(wLocal);

                                    AscendC::LocalTensor<float> xLocalUse = inX_.DeQue<float>();
                                    AscendC::LocalTensor<float> wLocalUse = inW_.DeQue<float>();

                                    AscendC::LocalTensor<float> mulLocal = tmpMul_.AllocTensor<float>();
                                    // Elementwise multiply
                                    AscendC::Mul(mulLocal, xLocalUse, wLocalUse, curTileW);

                                    if (firstAcc) {
                                        AscendC::LocalTensor<float> yAcc = accY_.AllocTensor<float>();
                                        AscendC::Muls(yAcc, mulLocal, 1.0f, curTileW); // yAcc = mulLocal
                                        accY_.EnQue(yAcc);
                                        firstAcc = false;
                                    } else {
                                        AscendC::LocalTensor<float> yAcc = accY_.DeQue<float>();
                                        AscendC::Add(yAcc, yAcc, mulLocal, curTileW);
                                        accY_.EnQue(yAcc);
                                    }

                                    tmpMul_.FreeTensor(mulLocal);
                                    inW_.FreeTensor(wLocalUse);
                                    inX_.FreeTensor(xLocalUse);
                                }
                            }
                        }

                        // Store result to y
                        AscendC::LocalTensor<float> yFinal = accY_.DeQue<float>();
                        AscendC::DataCopy(yGm_[YOffset(n, oc, oh, owBase)], yFinal, curTileW);
                        accY_.FreeTensor(yFinal);
                    }
                }
            }
        }
    }

private:
    __aicore__ inline uint32_t XOffset(uint32_t n, uint32_t c, uint32_t ih, uint32_t iw) const
    {
        // NCHW layout
        return (((n * cIn_ + c) * h_ + ih) * w_) + iw;
    }

    __aicore__ inline uint32_t WOffset(uint32_t oc, uint32_t ic, uint32_t kh, uint32_t kw) const
    {
        // [Cout, Cin, Kh, Kw]
        return (((oc * cIn_ + ic) * kH_ + kh) * kW_) + kw;
    }

    __aicore__ inline uint32_t YOffset(uint32_t n, uint32_t oc, uint32_t oh, uint32_t ow) const
    {
        // [N, Cout, outH, outW]
        return (((n * cOut_ + oc) * outH_ + oh) * outW_) + ow;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUF_NUM> inX_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUF_NUM> inW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_NUM> tmpMul_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUF_NUM> accY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, cOut_{0};
    uint32_t h_{0}, w_{0};
    uint32_t kH_{0}, kW_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t tileW_{1};
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    if (g_coreType == AscendC::AIV) {
        return;
    }
    GET_TILING_DATA(tiling, tilingGm);

    KernelConvStandard2dAsymSq op;
    op.Init(x, weight, y,
            tiling.n, tiling.cIn, tiling.cOut,
            tiling.h, tiling.w,
            tiling.kH, tiling.kW,
            tiling.outH, tiling.outW,
            tiling.tileW);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv2d_standard_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    TORCH_CHECK(x.dim() == 4, "x must be 4D NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be 4D [Cout, Cin, Kh, Kw]");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 supported");
    TORCH_CHECK(weight.scalar_type() == at::kFloat, "only float32 supported");

    const auto N = x.size(0);
    const auto Cin = x.size(1);
    const auto H = x.size(2);
    const auto W = x.size(3);
    const auto Cout = weight.size(0);
    const auto Kh = weight.size(2);
    const auto Kw = weight.size(3);

    TORCH_CHECK(weight.size(1) == Cin, "Cin mismatch between x and weight");
    TORCH_CHECK(Kh > 0 && Kw > 0, "kernel dims must be positive");
    TORCH_CHECK(H >= Kh && W >= Kw, "input smaller than kernel without padding");

    const auto Oh = H - Kh + 1;
    const auto Ow = W - Kw + 1;

    at::Tensor y = at::empty({N, Cout, Oh, Ow}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard2dAsymmetricInputSquareKernelCustom, x, weight, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_standard_2d_asymmetric_input_square_kernel_custom", &conv2d_standard_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_standard_2d_asymmetric_input_square_kernel_custom", &conv2d_standard_impl_npu,
          "Conv2D (stride=1,pad=0) via ConvStandard2dAsymmetricInputSquareKernelCustom on NPU");
