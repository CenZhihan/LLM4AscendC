#include "conv2d_divide_leaky_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 160 * 1024;  // per-core conservative UB share
static constexpr uint32_t ALIGN_ELEMS = 256;             // alignment for better DMA efficiency
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dDivideLeakyReluCustomTilingData tiling;

    // Inputs:
    // x: [N, Cin, H, W]
    // weight: [Cout, Cin, K, K]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t WCin = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    const uint32_t K    = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    // Assume K == w_shape->[3]
    (void)WCin;

    // Validations for this simplified kernel: stride=1, pad=0, dil=1
    if (H < K || W < K) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t Ho = H - K + 1;
    const uint32_t Wo = W - K + 1;

    // Detect bias existence
    bool hasBias = false;
    if (context->GetInputsSize() >= 3) {
        const gert::Shape *b_shape = context->GetInputShape(2);
        if (b_shape != nullptr && b_shape->GetOriginShape().GetShapeSize() > 0) {
            hasBias = true;
        }
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *divisorAttr = attrs->GetAttrPointer<float>(0);
    float divisor = (divisorAttr != nullptr) ? *divisorAttr : 2.0f;

    // UB requirements: two vectors of tileW for tmp and acc
    // bytes = 2 * tileW * 4 (+ small slack)
    uint32_t maxTileW = UB_BUDGET_BYTES / 8; // elements
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;

    uint32_t tileW = maxTileW;
    if (Wo < tileW) {
        const uint32_t aligned = AlignDown(Wo, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : Wo;
    }

    // BlockDim: parallelize over output channels
    uint32_t blockDim = Cout;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    // Fill tiling
    tiling.set_n(N);
    tiling.set_cIn(Cin);
    tiling.set_cOut(Cout);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_k(K);
    tiling.set_ho(Ho);
    tiling.set_wo(Wo);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1u : 0u);
    tiling.set_divisor(divisor);
    tiling.set_negativeSlope(0.01f);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dDivideLeakyReluCustom {
public:
    __aicore__ inline KernelConv2dDivideLeakyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t h, uint32_t wIn, uint32_t k,
                                uint32_t ho, uint32_t wo, uint32_t tileW,
                                uint32_t hasBias, float divisor, float negativeSlope)
    {
        n_ = n; cIn_ = cIn; cOut_ = cOut; h_ = h; w_ = wIn; k_ = k;
        ho_ = ho; wo_ = wo; tileW_ = tileW; hasBias_ = (hasBias != 0);
        invDiv_ = (divisor != 0.0f) ? (1.0f / divisor) : 0.5f;
        negativeSlope_ = negativeSlope;
        negCorr_ = negativeSlope_ - 1.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cIn_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, cOut_ * cIn_ * k_ * k_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, cOut_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * cOut_ * ho_ * wo_);

        pipe_.InitBuffer(tmpXQ_, 2, tileW_ * sizeof(float));
        pipe_.InitBuffer(accQ_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpCalcQ_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(wScalarQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cIn_ == 0 || cOut_ == 0 || ho_ == 0 || wo_ == 0 || tileW_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t oc = blockIdx; oc < cOut_; oc += blockNum) {
            float biasVal = 0.0f;
            if (hasBias_) {
                // load bias[oc]
                AscendC::LocalTensor<float> bLoc = wScalarQ_.AllocTensor<float>();
                const uint64_t bOff = oc;
                AscendC::DataCopy(bLoc, bGm_[bOff], 1);
                biasVal = bLoc.GetValue(0);
                wScalarQ_.FreeTensor(bLoc);
            }

            for (uint32_t n = 0; n < n_; ++n) {
                for (uint32_t oy = 0; oy < ho_; ++oy) {
                    for (uint32_t ox = 0; ox < wo_; ox += tileW_) {
                        uint32_t curW = wo_ - ox;
                        if (curW > tileW_) curW = tileW_;

                        AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
                        AscendC::Dup<float>(acc, 0.0f, curW);

                        // Accumulate convolution for this [n, oc, oy, ox:ox+curW]
                        for (uint32_t ic = 0; ic < cIn_; ++ic) {
                            for (uint32_t ky = 0; ky < k_; ++ky) {
                                for (uint32_t kx = 0; kx < k_; ++kx) {
                                    // Load weight scalar w[oc, ic, ky, kx]
                                    AscendC::LocalTensor<float> wLoc = wScalarQ_.AllocTensor<float>();
                                    const uint64_t wIndex = (((uint64_t)oc * cIn_ + ic) * k_ + ky) * k_ + kx;
                                    AscendC::DataCopy(wLoc, wGm_[wIndex], 1);
                                    const float wVal = wLoc.GetValue(0);
                                    wScalarQ_.FreeTensor(wLoc);

                                    // Load input slice x[n, ic, oy+ky, ox+kx : ox+kx+curW]
                                    AscendC::LocalTensor<float> xSlice = tmpXQ_.AllocTensor<float>();
                                    const uint64_t xBase = (((uint64_t)n * cIn_ + ic) * h_ + (oy + ky)) * w_ + (ox + kx);
                                    AscendC::DataCopy(xSlice, xGm_[xBase], curW);

                                    // acc += xSlice * wVal
                                    AscendC::Muls(xSlice, xSlice, wVal, curW);
                                    AscendC::Add(acc, acc, xSlice, curW);

                                    tmpXQ_.FreeTensor(xSlice);
                                }
                            }
                        }

                        // Add bias if present
                        if (hasBias_) {
                            AscendC::Adds(acc, acc, biasVal, curW);
                        }

                        // Divide by divisor: multiply by invDiv_
                        AscendC::Muls(acc, acc, invDiv_, curW);

                        // LeakyReLU: y = x + (negativeSlope - 1) * min(x, 0)
                        AscendC::LocalTensor<float> tLocal = tmpCalcQ_.AllocTensor<float>();
                        AscendC::Mins(tLocal, acc, 0.0f, curW);
                        AscendC::Muls(tLocal, tLocal, negCorr_, curW);
                        AscendC::Add(acc, acc, tLocal, curW);
                        tmpCalcQ_.FreeTensor(tLocal);

                        // Store to y[n, oc, oy, ox:ox+curW]
                        const uint64_t yBase = (((uint64_t)n * cOut_ + oc) * ho_ + oy) * wo_ + ox;
                        AscendC::DataCopy(yGm_[yBase], acc, curW);

                        accQ_.FreeTensor(acc);
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> tmpXQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpCalcQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wScalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, cOut_{0}, h_{0}, w_{0}, k_{0};
    uint32_t ho_{0}, wo_{0}, tileW_{0};
    bool hasBias_{false};
    float invDiv_{0.5f};
    float negativeSlope_{0.01f};
    float negCorr_{-0.99f};
};

extern "C" __global__ __aicore__ void conv2d_divide_leaky_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                       GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dDivideLeakyReluCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.n, tiling_data.cIn, tiling_data.cOut,
            tiling_data.h, tiling_data.w, tiling_data.k,
            tiling_data.ho, tiling_data.wo, tiling_data.tileW,
            tiling_data.hasBias, tiling_data.divisor, tiling_data.negativeSlope);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor allocate_output_like_conv2d(const at::Tensor& x,
                                                     const at::Tensor& weight)
{
    // x: [N, Cin, H, W], weight: [Cout, Cin, K, K]
    auto sizes_x = x.sizes();
    auto sizes_w = weight.sizes();
    int64_t N = sizes_x[0];
    int64_t H = sizes_x[2];
    int64_t W = sizes_x[3];
    int64_t Cout = sizes_w[0];
    int64_t K = sizes_w[2];
    int64_t Ho = H - K + 1;
    int64_t Wo = W - K + 1;
    return at::empty({N, Cout, Ho, Wo}, x.options());
}

at::Tensor conv2d_divide_leaky_relu_impl_npu(const at::Tensor& x,
                                             const at::Tensor& weight,
                                             c10::optional<at::Tensor> bias,
                                             double divisor)
{
    at::Tensor y = allocate_output_like_conv2d(x, weight);
    // NPU custom op: (x, weight, bias(optional), divisor, y)
    // divisor is expected to be 2.0 for best performance in this kernel
    EXEC_NPU_CMD(aclnnConv2dDivideLeakyReluCustom, x, weight, bias, static_cast<float>(divisor), y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
