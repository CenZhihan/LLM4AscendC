host_operator_src="""

#include "cumsum_exclusive_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 384 * 1024; // target safe UB budget
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumExclusiveCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    const std::vector<int64_t>& dims = in_shape->GetOriginShape().GetDims();
    uint32_t totalLength = static_cast<uint32_t>(in_shape->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim_ptr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dim_ptr;
    if (dim < 0) {
        dim += static_cast<int32_t>(dims.size());
    }
    if (dim < 0 || dim >= static_cast<int32_t>(dims.size())) {
        return ge::GRAPH_PARAM_INVALID;
    }

    uint32_t axisLength = static_cast<uint32_t>(dims[dim]);
    uint32_t outerSize = (axisLength == 0) ? 0 : (totalLength / axisLength);

    // We choose tileLength to cover the full axis in one go for each row (to avoid cross-tile carry).
    // Budget check: 2 * tileLen * sizeof(float) <= UB_BUDGET_BYTES
    uint32_t maxTileByUb = UB_BUDGET_BYTES / (2u * sizeof(float));
    maxTileByUb = AlignDown(maxTileByUb, ALIGN_ELEMS);
    if (maxTileByUb > HARD_CAP_ELEMS) maxTileByUb = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLength = axisLength;
    if (tileLength > maxTileByUb) {
        // Fallback to aligned max under UB; kernel assumes axis fits in a single tile for performance.
        tileLength = maxTileByUb;
    }
    if (tileLength == 0 && axisLength > 0) {
        tileLength = axisLength; // minimal safeguard
    }

    // BlockDim scheduling across outer rows.
    uint32_t numRows = outerSize;
    uint32_t blockDim = (numRows == 0) ? 1u : numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_axisLength(axisLength);
    tiling.set_outerSize(outerSize);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumsumExclusiveCustom : public OpDef {
public:
    explicit CumsumExclusiveCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumsumExclusiveCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUFFERS = 1;
constexpr int32_t OUT_BUFFERS = 1;

class KernelCumsumExclusiveCustom {
public:
    __aicore__ inline KernelCumsumExclusiveCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t axisLength,
                                uint32_t outerSize,
                                uint32_t tileLength,
                                int32_t dim)
    {
        totalLen_ = totalLength;
        axisLen_ = axisLength;
        outerSize_ = outerSize;
        tileLen_ = tileLength;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, IN_BUFFERS, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUFFERS, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || axisLen_ == 0) {
            return;
        }

        // This kernel assumes contiguous axis (last dimension) for optimal path.
        // General dim is provided for future extension; current model uses last dim.
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Iterate over rows (outerSize) in a block-stride pattern
        for (uint32_t row = blockIdx; row < outerSize_; row += blockNum) {
            const uint32_t base = row * axisLen_;
            // We assume tileLen_ >= axisLen_ (host tiling enforces), so single tile per row.
            const uint32_t len = axisLen_;

            // Copy in one row
            CopyIn(base, len);

            // Compute exclusive cumsum on the row
            ComputeExclusive(len);

            // Copy out result
            CopyOut(base, len);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeExclusive(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // y = 0
        AscendC::Muls(yLocal, xLocal, 0.0f, len);

        if (len > 1) {
            // y[1..] = x[0..len-2]
            AscendC::Add(yLocal[1], yLocal[1], xLocal[0], len - 1);

            // Inclusive scan on y to realize exclusive prefix sum
            for (uint32_t offset = 1; offset < len; offset <<= 1) {
                const uint32_t span = len - offset;
                if (span == 0) break;
                AscendC::Add(yLocal[offset], yLocal[offset], yLocal[0], span);
            }
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFERS> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFERS> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t axisLen_ {0};
    uint32_t outerSize_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ { -1 };
};

extern "C" __global__ __aicore__ void cumsum_exclusive_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumExclusiveCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.axisLength,
            tiling_data.outerSize,
            tiling_data.tileLength,
            tiling_data.dim);
    op.Process();
}
