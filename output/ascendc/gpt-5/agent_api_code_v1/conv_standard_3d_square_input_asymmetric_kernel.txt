kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t W, uint32_t H, uint32_t D,
                                uint32_t Kw, uint32_t Kh, uint32_t Kd,
                                uint32_t Wout, uint32_t Hout, uint32_t Dout)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        W_ = W; H_ = H; D_ = D;
        Kw_ = Kw; Kh_ = Kh; Kd_ = Kd;
        Wout_ = Wout; Hout_ = Hout; Dout_ = Dout;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * cin_ * W_ * H_ * D_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)cout_ * cin_ * Kw_ * Kh_ * Kd_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * cout_ * Wout_ * Hout_ * Dout_);

        // Allocate UB buffers: three vectors of length Dout_ and one filter buffer per output channel if needed
        pipe_.InitBuffer(xVecQ_, 1, Dout_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQ_, 1, Dout_ * sizeof(float));
        pipe_.InitBuffer(accVecQ_, 1, Dout_ * sizeof(float));
        pipe_.InitBuffer(wScalarQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cout_ == 0 || Wout_ == 0 || Hout_ == 0 || Dout_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t pairCount = (uint64_t)n_ * (uint64_t)cout_;

        for (uint64_t pair = blockIdx; pair < pairCount; pair += blockNum) {
            const uint32_t nIdx  = static_cast<uint32_t>(pair / cout_);
            const uint32_t coIdx = static_cast<uint32_t>(pair % cout_);

            // For each (w0, h0), compute y[n,co,w0,h0, :]
            for (uint32_t w0 = 0; w0 < Wout_; ++w0) {
                for (uint32_t h0 = 0; h0 < Hout_; ++h0) {
                    AscendC::LocalTensor<float> acc = accVecQ_.AllocTensor<float>();
                    // Initialize accumulator lazily in first multiply
                    bool accInitialized = false;

                    // Iterate over input channels and kernel spatial
                    for (uint32_t ci = 0; ci < cin_; ++ci) {
                        for (uint32_t kh = 0; kh < Kh_; ++kh) {
                            for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                // Base GM offset for x vector start at d=0; we will add kd later
                                // x layout [N,C,W,H,D] (D is fastest)
                                const uint64_t baseXNoD =
                                    (((((uint64_t)nIdx * cin_ + ci) * W_) + (w0 + kw)) * H_ + (h0 + kh)) * (uint64_t)D_;

                                for (uint32_t kd = 0; kd < Kd_; ++kd) {
                                    // Load input segment of length Dout_ starting at d = kd
                                    AscendC::LocalTensor<float> xVec = xVecQ_.AllocTensor<float>();
                                    AscendC::DataCopy(xVec, xGm_[baseXNoD + kd], Dout_);

                                    // Load scalar weight w[co,ci,kw,kh,kd]
                                    const uint64_t wIdx =
                                        (((((uint64_t)coIdx * cin_ + ci) * Kw_) + kw) * Kh_ + kh) * (uint64_t)Kd_ + kd;
                                    AscendC::LocalTensor<float> wScBuf = wScalarQ_.AllocTensor<float>();
                                    AscendC::DataCopy(wScBuf, wGm_[wIdx], 1);
                                    // Assuming API supports reading scalar value from LocalTensor
                                    float wScalar = wScBuf.GetValue(0);

                                    AscendC::LocalTensor<float> tmp = tmpVecQ_.AllocTensor<float>();
                                    AscendC::Muls(tmp, xVec, wScalar, Dout_);

                                    if (!accInitialized) {
                                        // acc = tmp
                                        AscendC::DataCopy(acc, tmp, Dout_);
                                        accInitialized = true;
                                    } else {
                                        AscendC::Add(acc, acc, tmp, Dout_);
                                    }

                                    tmpVecQ_.FreeTensor(tmp);
                                    wScalarQ_.FreeTensor(wScBuf);
                                    xVecQ_.FreeTensor(xVec);
                                }
                            }
                        }
                    }

                    // Write back y[n, co, w0, h0, :]
                    const uint64_t yBase =
                        (((((uint64_t)nIdx * cout_ + coIdx) * Wout_) + w0) * Hout_ + h0) * (uint64_t)Dout_;
                    AscendC::DataCopy(yGm_[yBase], acc, Dout_);
                    accVecQ_.FreeTensor(acc);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  xVecQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  wScalarQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVecQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> accVecQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t W_{0}, H_{0}, D_{0};
    uint32_t Kw_{0}, Kh_{0}, Kd_{0};
    uint32_t Wout_{0}, Hout_{0}, Dout_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvStandard3dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.w, tiling_data.h, tiling_data.d,
            tiling_data.kw, tiling_data.kh, tiling_data.kd,
            tiling_data.wOut, tiling_data.hOut, tiling_data.dOut);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv3d_custom_layout_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    // x: [N,C,W,H,D], weight: [Cout,Cin,Kw,Kh,Kd]
    // y: [N,Cout,Wout,Hout,Dout]
    // Output size derivation on host side for allocation
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto W  = x.size(2);
    auto H  = x.size(3);
    auto D  = x.size(4);

    auto Cout = weight.size(0);
    auto CinW = weight.size(1);
    auto Kw   = weight.size(2);
    auto Kh   = weight.size(3);
    auto Kd   = weight.size(4);

    TORCH_CHECK(Cin == CinW, "Input Cin must match weight Cin");

    auto Wout = W - Kw + 1;
    auto Hout = H - Kh + 1;
    auto Dout = D - Kd + 1;
    TORCH_CHECK(Wout >= 0 && Hout >= 0 && Dout >= 0, "Invalid kernel larger than input");

    at::Tensor y = at::empty({N, Cout, Wout, Hout, Dout}, x.options());

    EXEC_NPU_CMD(aclnnConvStandard3dSquareInputAsymmetricKernelCustom, x, weight, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_standard_3d_square_input_asymmetric_kernel_custom", &conv3d_custom_layout_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_standard_3d_square_input_asymmetric_kernel_custom",
          &conv3d_custom_layout_impl_npu,
