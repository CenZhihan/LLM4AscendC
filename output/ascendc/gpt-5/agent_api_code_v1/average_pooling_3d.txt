kernel_src = """
#include "kernel_operator.h"
using namespace AscendC;
class KernelAveragePooling3d {
public:
    __aicore__ inline KernelAveragePooling3d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, AveragePooling3dCustomTilingData* tilingData) {
        tiling = *tilingData;
        xGm.SetGlobalBuffer((__gm__ float*)x);
        yGm.SetGlobalBuffer((__gm__ float*)y);
        uint32_t total_out = tiling.n * tiling.c * tiling.out_d * tiling.out_h * tiling.out_w;
        uint32_t items_per_block = (total_out + tiling.block_dim - 1) / tiling.block_dim;
        start_idx = items_per_block * GetBlockIdx();
        end_idx = items_per_block * (GetBlockIdx() + 1);
        if (end_idx > total_out) end_idx = total_out;
    }
    __aicore__ inline void Process() {
        for (uint32_t i = start_idx; i < end_idx; i++) {
            uint32_t temp = i;
            uint32_t ow = temp % tiling.out_w; temp /= tiling.out_w;
            uint32_t oh = temp % tiling.out_h; temp /= tiling.out_h;
            uint32_t od = temp % tiling.out_d; temp /= tiling.out_d;
            uint32_t ic = temp % tiling.c; temp /= tiling.c;
            uint32_t in = temp;
            int32_t id_start = (int32_t)od * (int32_t)tiling.s - (int32_t)tiling.p;
            int32_t ih_start = (int32_t)oh * (int32_t)tiling.s - (int32_t)tiling.p;
            int32_t iw_start = (int32_t)ow * (int32_t)tiling.s - (int32_t)tiling.p;
            float sum = 0.0f;
            for (uint32_t kd = 0; kd < tiling.k; kd++) {
                for (uint32_t kh = 0; kh < tiling.k; kh++) {
                    for (uint32_t kw = 0; kw < tiling.k; kw++) {
                        int32_t curr_d = id_start + (int32_t)kd;
                        int32_t curr_h = ih_start + (int32_t)kh;
                        int32_t curr_w = iw_start + (int32_t)kw;
                        if (curr_d >= 0 && curr_d < (int32_t)tiling.d && curr_h >= 0 && curr_h < (int32_t)tiling.h && curr_w >= 0 && curr_w < (int32_t)tiling.w) {
                            uint32_t idx = (((in * tiling.c + ic) * tiling.d + (uint32_t)curr_d) * tiling.h + (uint32_t)curr_h) * tiling.w + (uint32_t)curr_w;
                            sum += xGm.GetValue(idx);
                        }
                    }
                }
            }
            yGm.SetValue(i, sum / (float)(tiling.k * tiling.k * tiling.k));
        }
    }
private:
    GlobalTensor<float> xGm, yGm;
    AveragePooling3dCustomTilingData tiling;
    uint32_t start_idx, end_idx;
};
extern "C" __global__ __aicore__ void average_pooling3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelAveragePooling3d op;
    op.Init(x, y, &tilingData);
    op.Process();
}

class KernelAveragePooling3d {
public:
    __aicore__ inline KernelAveragePooling3d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, AveragePooling3dCustomTilingData* tilingData) {
        tiling = *tilingData;

            uint32_t oh = temp % tiling.out_h; temp /= tiling.out_h;
