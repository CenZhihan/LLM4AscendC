host_operator_src="""

#include "gemm_relu_divide_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmReluDivideCustomTilingData tiling;

    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);
    const auto *b_shape = context->GetInputShape(2);

    // Expect shapes: x: [M, K], w: [N, K], b: [N]
    uint32_t dims_x = x_shape->GetOriginShape().GetDimNum();
    uint32_t dims_w = w_shape->GetOriginShape().GetDimNum();
    (void)b_shape;

    uint32_t M = 1, Kx = 1, N = 1, Kw = 1;
    if (dims_x >= 2) {
        M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
        Kx = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    } else {
        M = 1;
        Kx = static_cast<uint32_t>(x_shape->GetOriginShape().GetShapeSize());
    }
    if (dims_w >= 2) {
        N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
        Kw = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    } else {
        N = 1;
        Kw = static_cast<uint32_t>(w_shape->GetOriginShape().GetShapeSize());
    }

    // Basic checks (best-effort)
    if (Kx != Kw) {
        // fallback values; runtime should ensure correctness
        Kw = Kx;
    }

    const uint32_t totalLength = M * N;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float divisor = 1.0f;
    if (attrs != nullptr && attrs->GetNum() > 0) {
        const float *divPtr = attrs->GetAttrPointer<float>(0);
        if (divPtr) divisor = *divPtr;
    }

    // Post-process tile length from UB budget; 2 in/out buffers -> 2 * tile * 4 bytes each
    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // BlockDim heuristic based on tiles
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(Kx);
    tiling.set_divisor(divisor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    // No extra workspace for this fused kernel
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

// Post-processing: ReLU and divide over GM tensor y.
class KernelPostReluDivide {
public:
    __aicore__ inline KernelPostReluDivide() {}

    __aicore__ inline void Init(GM_ADDR y, uint32_t totalLength, uint32_t tileLength, float divisor)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        invDiv_ = (divisor != 0.0f) ? (1.0f / divisor) : 0.0f;

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    static constexpr int32_t BUFFER_NUM = 2;

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocalIn = inQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> yLocalOut = outQueueY_.AllocTensor<float>();

        // y = max(y, 0)
        AscendC::Maxs(yLocalOut, yLocalIn, 0.0f, len);
        // y = y / divisor  -> multiply by invDiv_
        if (invDiv_ != 1.0f) {
            AscendC::Muls(yLocalOut, yLocalOut, invDiv_, len);
        }

        outQueueY_.EnQue(yLocalOut);
        inQueueY_.FreeTensor(yLocalIn);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    float invDiv_ {1.0f};
};

extern "C" __global__ __aicore__ void gemm_relu_divide_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    // Prepare tiling (user-defined for post process)
    GET_TILING_DATA(tilingData, tiling);

    // --- Matmul phase: y = x @ w^T + b ---
    // A: [M, K] (half, not transposed)
    // B: [N, K] (half, transposed in kernel)
    // C: [M, N] (float)
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::BSNGD> A_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, true,  LayoutMode::BSNGD> B_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD> C_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> BIAS_TYPE;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> batchMatmulKernel;
    AscendC::TPipe pipe;

    // Share resources for matmul
    // If the underlying tiler provides cube tiling data, it will be used via REGIST_MATMUL_OBJ.
    // Otherwise, these hints keep full resources to matmul.
    tilingData.shareMode = 0;
    tilingData.shareL1Size = FULL_L1_SIZE;
    tilingData.shareL0CSize = FULL_L0C_SIZE;
    tilingData.shareUbSize = 0;

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), batchMatmulKernel.matmulObj, &tilingData);

    // Initialize and execute GEMM with bias add
    batchMatmulKernel.Init(x, w, b, y, workspace, tilingData);
    // Use M as batch rows, simple heuristic
    int batchNum = (int)tilingData.M;
    if (batchNum <= 0) batchNum = 1;
    batchMatmulKernel.Process<true>(&pipe, batchNum, batchNum);

    // --- Post phase: ReLU and divide by scalar ---
    KernelPostReluDivide post;
    post.Init(y, tilingData.totalLength, tilingData.tileLength, tilingData.divisor);
    post.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor gemm_relu_divide_impl_npu(const at::Tensor& x,
                                     const at::Tensor& w,
                                     const at::Tensor& b,
                                     const at::Tensor& divisor_t) {
    // Cast x, w to half for cube matmul, keep bias and output as float
    at::Tensor x_half = x.to(at::kHalf);
    at::Tensor w_half = w.to(at::kHalf);
    at::Tensor b_f32 = b.to(at::kFloat);
    at::Tensor y = at::empty({x.size(0), w.size(0)}, x.options().dtype(at::kFloat));

    // Extract scalar divisor
    double divisor = divisor_t.item<double>();

    EXEC_NPU_CMD(aclnnGemmReluDivideCustom, x_half, w_half, b_f32, static_cast<float>(divisor), y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("gemm_relu_divide_custom", &gemm_relu_divide_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("gemm_relu_divide_custom", &gemm_relu_divide_impl_npu,
          "Fused GEMM + ReLU + Divide (GemmReluDivideCustom on NPU)");
