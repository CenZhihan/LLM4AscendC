#include "conv2d_scaling_min_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dScalingMinCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect NCHW for x, and OIHW for weight
    const uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t C_in = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t C_out = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(w_shape->GetDim(2));
    // stride=1, pad=0, dilation=1
    const uint32_t H_out = (H >= K) ? (H - K + 1) : 0;
    const uint32_t W_out = (W >= K) ? (W - K + 1) : 0;

    const uint32_t totalPos = N * H_out * W_out;

    // Simple tiling: process positions per tile
    // Budget small to be safe on UB, position-wise compute is scalar-heavy anyway
    static constexpr uint32_t UB_BUDGET_BYTES = 128 * 1024;
    static constexpr uint32_t BYTES_PER_POS = 4; // output scalar per pos in UB temp if needed
    uint32_t maxTilePos = UB_BUDGET_BYTES / BYTES_PER_POS;
    // align to 32 for better DMA
    maxTilePos = AlignDown(maxTilePos, 32);
    if (maxTilePos < 32) maxTilePos = 32;

    uint32_t tilePos = maxTilePos;
    if (totalPos < tilePos) {
        tilePos = (totalPos == 0) ? 0u : totalPos;
    }

    uint32_t numTiles = (tilePos == 0) ? 1u : CeilDiv(totalPos, tilePos);
    uint32_t blockDim = numTiles;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    uint32_t hasBias = 0u;
    if (context->GetInputNum() > 2) {
        const gert::Shape *b_shape = context->GetInputShape(2);
        if (b_shape != nullptr && b_shape->GetShapeSize() > 0) {
            hasBias = 1u;
        }
    }

    tiling.set_nSize(N);
    tiling.set_cIn(C_in);
    tiling.set_cOut(C_out);
    tiling.set_hSize(H);
    tiling.set_wSize(W);
    tiling.set_kSize(K);
    tiling.set_hOut(H_out);
    tiling.set_wOut(W_out);
    tiling.set_totalPos(totalPos);
    tiling.set_tilePos(tilePos);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dScalingMinCustom {
public:
    __aicore__ inline KernelConv2dScalingMinCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scale, GM_ADDR y,
                                uint32_t N, uint32_t C_in, uint32_t C_out,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t H_out, uint32_t W_out,
                                uint32_t totalPos, uint32_t tilePos,
                                uint32_t hasBias)
    {
        N_ = N; C_in_ = C_in; C_out_ = C_out;
        H_ = H; W_ = W; K_ = K;
        H_out_ = H_out; W_out_ = W_out;
        totalPos_ = totalPos; tilePos_ = tilePos; hasBias_ = (hasBias != 0);

        // Global tensor sizes
        const uint32_t xSize = N_ * C_in_ * H_ * W_;
        const uint32_t wSize = C_out_ * C_in_ * K_ * K_;
        const uint32_t bSize = hasBias_ ? C_out_ : 0;
        const uint32_t ySize = N_ * H_out_ * W_out_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, bSize);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);
        scaleGm_.SetGlobalBuffer((__gm__ float*)scale, 1);

        // Small local buffer for the scale scalar
        pipe_.InitBuffer(scaleQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalPos_ == 0) return;

        // Load scale
        float s = LoadScale();

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint32_t stride = blockNum * tilePos_;
        uint32_t base = blockIdx * tilePos_;
        if (base >= totalPos_) return;

        while (base < totalPos_) {
            const uint32_t cur = (base + tilePos_ <= totalPos_) ? tilePos_ : (totalPos_ - base);
            ComputeTile(base, cur, s);
            base += stride;
        }
    }

private:
    __aicore__ inline float LoadScale()
    {
        AscendC::LocalTensor<float> sLocal = scaleQueue_.AllocTensor<float>();
        AscendC::DataCopy(sLocal, scaleGm_[0], 1);
        scaleQueue_.EnQue<float>(sLocal);
        AscendC::LocalTensor<float> sLocal2 = scaleQueue_.DeQue<float>();
        float s = sLocal2.GetValue(0);
        scaleQueue_.FreeTensor(sLocal2);
        return s;
    }

    __aicore__ inline void ComputeTile(uint32_t base, uint32_t len, float scale)
    {
        // Iterate over positions: pos = n * (H_out * W_out) + h * W_out + w
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t pos = base + i;
            uint32_t hwArea = H_out_ * W_out_;
            uint32_t n = pos / hwArea;
            uint32_t rem = pos - n * hwArea;
            uint32_t h = rem / W_out_;
            uint32_t w = rem - h * W_out_;
            float minVal = 0.0f;

            // For out channel 0, compute to initialize minVal
            if (C_out_ > 0) {
                float accum0 = ConvPoint(n, 0, h, w);
                if (hasBias_) {
                    accum0 += bGm_[0];
                }
                accum0 *= scale;
                minVal = accum0;

                // Remaining channels
                for (uint32_t co = 1; co < C_out_; ++co) {
                    float accum = ConvPoint(n, co, h, w);
                    if (hasBias_) {
                        accum += bGm_[co];
                    }
                    accum *= scale;
                    // min across channels
                    minVal = (accum < minVal) ? accum : minVal;
                }
            }
            // Store to y (N, 1, H_out, W_out) flattened as (N * H_out * W_out)
            yGm_[pos] = minVal;
        }
    }

    __aicore__ inline float ConvPoint(uint32_t n, uint32_t co, uint32_t hOut, uint32_t wOut)
    {
        // sum over ci, kh, kw: x[n, ci, hOut+kh, wOut+kw] * w[co, ci, kh, kw]
        float sum = 0.0f;
        const uint32_t xNStep = C_in_ * H_ * W_;
        const uint32_t xCStep = H_ * W_;
        const uint32_t wOStep = C_in_ * K_ * K_;
        const uint32_t wCStep = K_ * K_;

        const uint32_t xNBase = n * xNStep;

        for (uint32_t ci = 0; ci < C_in_; ++ci) {
            const uint32_t xCBase = xNBase + ci * xCStep;
            const uint32_t wOCBase = co * wOStep + ci * wCStep;
            for (uint32_t kh = 0; kh < K_; ++kh) {
                const uint32_t xHBase = xCBase + (hOut + kh) * W_;
                const uint32_t wHBase = wOCBase + kh * K_;
                for (uint32_t kw = 0; kw < K_; ++kw) {
                    float xv = xGm_[xHBase + (wOut + kw)];
                    float wv = wGm_[wHBase + kw];
                    sum += xv * wv;
                }
            }
        }
        return sum;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scaleQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> scaleGm_;

    uint32_t N_{0}, C_in_{0}, C_out_{0};
    uint32_t H_{0}, W_{0}, K_{0};
    uint32_t H_out_{0}, W_out_{0};
    uint32_t totalPos_{0}, tilePos_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scale,
                                                                GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dScalingMinCustom op;
    op.Init(x, weight, bias, scale, y,
            tiling_data.nSize, tiling_data.cIn, tiling_data.cOut,
            tiling_data.hSize, tiling_data.wSize, tiling_data.kSize,
            tiling_data.hOut, tiling_data.wOut,
            tiling_data.totalPos, tiling_data.tilePos,
            tiling_data.hasBias);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor make_output_tensor_like_conv_min(const at::Tensor& x, const at::Tensor& weight)
{
    // x: [N, C_in, H, W], w: [C_out, C_in, K, K]; stride=1, pad=0, dil=1
    auto sizes = x.sizes();
    int64_t N = sizes[0], H = sizes[2], W = sizes[3];
    int64_t K = weight.size(2);
    int64_t H_out = H - K + 1;
    int64_t W_out = W - K + 1;
    return at::empty({N, 1, H_out, W_out}, x.options());
}

at::Tensor conv2d_scaling_min_impl_npu(const at::Tensor& x, const at::Tensor& weight,
                                       c10::optional<at::Tensor> bias_opt,
                                       const at::Tensor& scale)
{
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, "x and weight must be 4D tensors (NCHW, OIHW)");
    at::Tensor y = make_output_tensor_like_conv_min(x, weight);
    if (bias_opt.has_value()) {
        at::Tensor bias = bias_opt.value();
        EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, bias, scale, y);
    } else {
        // pass an empty tensor for optional bias
        at::Tensor empty_bias;
        EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, empty_bias, scale, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_scaling_min_custom", &conv2d_scaling_min_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_scaling_min_custom", &conv2d_scaling_min_impl_npu,
          "Conv2d + scalar scaling + channelwise min (Conv2dScalingMinCustom on NPU)");
