host_operator_src="""

#include "frobenius_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // namespace

namespace optiling {

// UB model for this kernel:
//   inQueueX:    2 * tile
//   outQueueY:   2 * tile
//   tmpSq:       1 * tile
//   tmpWs:       1 * tile
//   smallCalc:   ~32
// Total floats ~= 6 * tile => bytes ~= 24 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS     = 2048;         // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS  = 16384;        // keep latency bounded

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    FrobeniusNormCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    uint32_t maxTile = UB_BUDGET_BYTES / 24;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // Choose a reasonable blockDim; all blocks will participate in normalization phase.
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // Workspace: 2 floats (invNorm and ready flag), keep aligned
    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 64; // bytes
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class FrobeniusNormCustom : public OpDef {
public:
    explicit FrobeniusNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(FrobeniusNormCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFrobeniusNormCustom {
public:
    __aicore__ inline KernelFrobeniusNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace,
                                uint32_t totalLength, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        // Workspace layout:
        //   workspace[0] = invNorm
        //   workspace[1] = readyFlag (0.0f or 1.0f)
        wsGm_.SetGlobalBuffer((__gm__ float*)workspace, 2);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpSqQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpWsQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(smallQueue_, 1, 32U * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        if (blockIdx == 0) {
            // Initialize ready flag to 0
            WriteScalarToGM(1u, 0.0f);
            // Compute 1 / ||x||_F and publish it to workspace[0], then set ready flag to 1
            float invNorm = ComputeInvNorm();
            WriteScalarToGM(0u, invNorm);
            // Publish ready flag
            WriteScalarToGM(1u, 1.0f);
        } else {
            // Wait until ready flag is set by block 0
            WaitReadyFlag();
        }

        // All blocks read the final invNorm and proceed with normalization
        float invNorm = ReadScalarFromGM(0u);

        // Grid-stride tiling for normalization
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            ComputeNormalize(curLen, invNorm);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

    __aicore__ inline void ComputeNormalize(uint32_t len, float invNorm)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::Muls(yLocal, xLocal, invNorm, len);

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline float ComputeInvNorm()
    {
        float sumAll = 0.0f;

        // Iterate across the entire tensor with this block alone (blockIdx==0)
        uint32_t base = 0;
        while (base < totalLen_) {
            uint32_t len = totalLen_ - base;
            if (len > tileLen_) len = tileLen_;

            // Copy in tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], len);
            inQueueX_.EnQue(xLocal);

            // Compute squares
            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> sqLocal = tmpSqQueue_.AllocTensor<float>();
            AscendC::Mul(sqLocal, xLocal, xLocal, len);

            // Reduce sum of squares into a scalar
            AscendC::LocalTensor<float> wsLocal = tmpWsQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> sumLocal = smallQueue_.AllocTensor<float>();
            ReduceSum(sumLocal, sqLocal, wsLocal, static_cast<int32_t>(len));

            sumAll += sumLocal.GetValue(0);

            // Free temporaries
            smallQueue_.FreeTensor(sumLocal);
            tmpWsQueue_.FreeTensor(wsLocal);
            tmpSqQueue_.FreeTensor(sqLocal);
            inQueueX_.FreeTensor(xLocal);

            base += len;
        }

        // Compute invNorm = 1.0 / sqrt(sumAll)
        AscendC::LocalTensor<float> sqx = smallQueue_.AllocTensor<float>();
        Duplicate(sqx, sumAll, AscendC::DEFAULT_REPEAT_STRIDE);
        AscendC::PipeBarrier<PIPE_V>();
        Sqrt(sqx, sqx, AscendC::DEFAULT_REPEAT_STRIDE);
        float norm = sqx.GetValue(0);
        smallQueue_.FreeTensor(sqx);

        // Avoid division by zero; if norm==0, keep invNorm as 0 to avoid NaNs
        float invNorm = (norm > 0.0f) ? (1.0f / norm) : 0.0f;
        return invNorm;
    }

    __aicore__ inline void WaitReadyFlag()
    {
        while (true) {
            float flag = ReadScalarFromGM(1u);
            if (flag >= 1.0f) {
                break;
            }
        }
    }

    __aicore__ inline void WriteScalarToGM(uint32_t wsIndex, float value)
    {
        AscendC::LocalTensor<float> tmp = smallQueue_.AllocTensor<float>();
        Duplicate(tmp, value, AscendC::DEFAULT_REPEAT_STRIDE);
        AscendC::DataCopy(wsGm_[wsIndex], tmp, 1);
        smallQueue_.FreeTensor(tmp);
    }

    __aicore__ inline float ReadScalarFromGM(uint32_t wsIndex)
    {
        AscendC::LocalTensor<float> tmp = smallQueue_.AllocTensor<float>();
        AscendC::DataCopy(tmp, wsGm_[wsIndex], 1);
        float v = tmp.GetValue(0);
        smallQueue_.FreeTensor(tmp);
        return v;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>   inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM>  outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>          tmpSqQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>          tmpWsQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>          smallQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wsGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void frobenius_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFrobeniusNormCustom op;
    op.Init(x, y, workspace, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
