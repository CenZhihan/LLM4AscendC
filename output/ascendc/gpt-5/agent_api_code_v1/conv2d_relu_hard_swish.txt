#include "conv2d_relu_hard_swish_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dReluHardSwishCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);     // N, C, H, W
    const gert::Shape *w_shape = context->GetInputShape(1);     // Cout, Cin, Kh, Kw

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t CinW = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t Kh   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw   = static_cast<uint32_t>(w_shape->GetDim(3));

    // sanity check Cin matches
    (void)CinW;

    // stride=1, pad=0, dil=1
    const uint32_t Ho = (H >= Kh) ? (H - Kh + 1) : 0;
    const uint32_t Wo = (W >= Kw) ? (W - Kw + 1) : 0;

    const uint32_t totalOutElems = N * Cout * Ho * Wo;

    // Optional bias present if input size >= 3 and bias shape dims == Cout
    uint32_t hasBias = 0;
    if (context->GetInputNum() >= 3) {
        const gert::Shape *b_shape = context->GetInputShape(2);
        if (b_shape != nullptr && b_shape->GetShapeSize() == Cout) {
            hasBias = 1;
        }
    }

    // A lightweight tileLength used for grid-stride; not used for UB buffering in this naive kernel.
    uint32_t tileLen = 2048;
    if (totalOutElems < tileLen) tileLen = totalOutElems > 0 ? totalOutElems : 1;
    tileLen = AlignDown(tileLen, 1);

    // Choose blockDim based on tiles
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutElems(totalOutElems);
    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_hasBias(hasBias);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
using namespace AscendC;

// Naive direct conv2d (stride 1, pad 0, dil 1) fused with ReLU + HardSwish.
// Shapes targeted: N=128, Cin=8, Cout=64, H=W=128, K=3 but code uses general dims from tiling.

class KernelConv2dReluHardSwishCustom {
public:
    __aicore__ inline KernelConv2dReluHardSwishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W,
                                uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t hasBias,
                                uint32_t totalOutElems,
                                uint32_t tileLen)
    {
        x_gm_ = (__gm__ float*)x;
        w_gm_ = (__gm__ float*)w;
        b_gm_ = (__gm__ float*)b;
        y_gm_ = (__gm__ float*)y;

        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W;
        Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;
        hasBias_ = hasBias;

        totalOutElems_ = totalOutElems;
        tileLen_ = tileLen;
    }

    __aicore__ inline void Process()
    {
        if (totalOutElems_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = (tileLen_ == 0) ? blockNum : (blockNum * tileLen_);
        uint32_t base = (tileLen_ == 0) ? blockIdx : (blockIdx * tileLen_);

        if (base >= totalOutElems_) return;

        while (base < totalOutElems_) {
            uint32_t cur = base;
            uint32_t end = (tileLen_ == 0) ? (base + 1) : (base + tileLen_);
            if (end > totalOutElems_) end = totalOutElems_;

            for (; cur < end; ++cur) {
                // Decode flat output index: [n, oc, ho, wo]
                uint32_t t = cur;
                const uint32_t wo = t % Wo_; t /= Wo_;
                const uint32_t ho = t % Ho_; t /= Ho_;
                const uint32_t oc = t % Cout_; t /= Cout_;
                const uint32_t n  = t;

                float acc = 0.0f;

                // Bias
                if (hasBias_) {
                    acc = b_gm_[oc];
                }

                // Convolution sum
                // for ic in [0..Cin), kh in [0..Kh), kw in [0..Kw)
                for (uint32_t ic = 0; ic < Cin_; ++ic) {
                    // base pointer offsets
                    const uint64_t in_base = (((uint64_t)n * Cin_ + ic) * H_) * W_;
                    const uint64_t w_base  = ((((uint64_t)oc * Cin_ + ic) * Kh_) * Kw_);

                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        const uint32_t ih = ho + kh; // no pad, stride=1
                        const uint64_t in_row = in_base + (uint64_t)ih * W_;
                        const uint64_t w_row  = w_base + (uint64_t)kh * Kw_;

                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                            const uint32_t iw = wo + kw;
                            const float in_val = x_gm_[in_row + iw];
                            const float w_val  = w_gm_[w_row + kw];
                            acc += in_val * w_val;
                        }
                    }
                }

                // ReLU
                float r = (acc > 0.0f) ? acc : 0.0f;

                // HardSwish: r * clamp((r + 3)/6, 0, 1)
                float t0 = (r + 3.0f) * (1.0f / 6.0f);
                if (t0 < 0.0f) t0 = 0.0f;
                if (t0 > 1.0f) t0 = 1.0f;
                float yv = r * t0;

                // Store
                y_gm_[cur] = yv;
            }

            if (strideElems == 0) break;
            base += strideElems;
        }
    }

private:
    __gm__ float* x_gm_{nullptr};
    __gm__ float* w_gm_{nullptr};
    __gm__ float* b_gm_{nullptr};
    __gm__ float* y_gm_{nullptr};

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0};
    uint32_t Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t hasBias_{0};

    uint32_t totalOutElems_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluHardSwishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W,
            tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo,
            tiling_data.hasBias,
            tiling_data.totalOutElems, tiling_data.tileLength);
    op.Process();
}

#include <torch/library.h>

at::Tensor conv2d_relu_hard_swish_impl_npu(const at::Tensor& x,
                                           const at::Tensor& weight,
                                           c10::optional<at::Tensor> bias_opt)
{
    // Output shape: N, Cout, Ho, Wo with stride=1, pad=0, dil=1
    int64_t N = x.size(0);
    int64_t H = x.size(2);
    int64_t W = x.size(3);
    int64_t Cout = weight.size(0);
    int64_t Kh = weight.size(2);
    int64_t Kw = weight.size(3);
    int64_t Ho = H - Kh + 1;
    int64_t Wo = W - Kw + 1;

    at::Tensor y = at::empty({N, Cout, Ho, Wo}, x.options());

    if (bias_opt.has_value()) {
        const at::Tensor& b = bias_opt.value();
        EXEC_NPU_CMD(aclnnConv2dReluHardSwishCustom, x, weight, b, y);
    } else {
        at::Tensor b_undefined;
        EXEC_NPU_CMD(aclnnConv2dReluHardSwishCustom, x, weight, b_undefined, y);
    }
    return y;
}
