static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core UB share
static constexpr uint32_t ALIGN_ELEMS = 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    VanillaRnnHiddenCustomTilingData tiling;

    // x: [T, B, I]
    const gert::Shape *x_shape = context->GetInputShape(0);
    if (x_shape->GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t T = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t B = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t I = static_cast<uint32_t>(x_shape->GetDim(2));

    const gert::Shape *h0_shape = context->GetInputShape(1);
    if (h0_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t H = static_cast<uint32_t>(h0_shape->GetDim(1));

    // w_i2h: [H, K], expect K = I + H
    const gert::Shape *w_i2h_shape = context->GetInputShape(2);
    if (w_i2h_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t WiH = static_cast<uint32_t>(w_i2h_shape->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(w_i2h_shape->GetDim(1));
    if (WiH != H) {
        return ge::GRAPH_FAILED;
    }
    // w_h2o: [O, H]
    const gert::Shape *w_h2o_shape = context->GetInputShape(4);
    if (w_h2o_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t O = static_cast<uint32_t>(w_h2o_shape->GetDim(0));
    const uint32_t Wh2 = static_cast<uint32_t>(w_h2o_shape->GetDim(1));
    if (Wh2 != H) {
        return ge::GRAPH_FAILED;
    }

    // Choose blockDim across batch
    uint32_t blockDim = (B > 0) ? B : 1u;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim == 0) blockDim = 1u;

    const uint32_t bPerCore = CeilDiv(B, blockDim);

    // UB tile for tanh: need in/out = 2 * tile * sizeof(float)
    uint32_t maxTile = AlignDown(UB_BUDGET_BYTES / (2u * sizeof(float)), ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    tiling.set_B(B);
    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_blockDim(blockDim);
    tiling.set_bPerCore(bPerCore);
    tiling.set_tileVecLen(tileVecLen);

                        context->GetRawTilingData()->GetCapacity());

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0,
                                GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o,
                                GM_ADDR y, GM_ADDR workspace,
                                uint32_t T, uint32_t B, uint32_t I,
                                uint32_t H, uint32_t O, uint32_t K,
                                uint32_t bPerCore, uint32_t tileVecLen)
    {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O; K_ = K;
        bPerCore_ = bPerCore;
        tileVecLen_ = tileVecLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<int64_t>(T_) * B_ * I_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, static_cast<int64_t>(B_) * H_);
        wi2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, static_cast<int64_t>(H_) * K_);
        bi2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);
        whoGm_.SetGlobalBuffer((__gm__ float*)w_h2o, static_cast<int64_t>(O_) * H_);
        bhoGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<int64_t>(T_) * B_ * O_);

        workspaceGm_ = reinterpret_cast<__gm__ float*>(workspace);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileVecLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileVecLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileVecLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Determine this core's batch slice
        uint32_t bStart = (B_ * blockIdx) / blockNum;
        uint32_t bEnd = (B_ * (blockIdx + 1u)) / blockNum;
        if (bEnd > B_) bEnd = B_;
        if (bEnd <= bStart) {
            return;
        }
        uint32_t m = bEnd - bStart;  // rows per GEMM

        // Workspace partition per core
        int64_t perCoreElems = static_cast<int64_t>(bPerCore_) * (K_ + H_);
        __gm__ float* aWorkBase = workspaceGm_ + static_cast<int64_t>(blockIdx) * perCoreElems;
        __gm__ float* hBufBase  = aWorkBase + static_cast<int64_t>(bPerCore_) * K_;

        // Initialize h buffer with h0 slice
        {
            int64_t h0Off = static_cast<int64_t>(bStart) * H_;
            int64_t copyElems = static_cast<int64_t>(m) * H_;
            // Copy GM -> GM via UB tiling
            CopyGmToGm(hBufBase, h0Gm_.GetPhyAddr() + h0Off, copyElems);
        }

        // Register matmul objects
        REGIST_MATMUL_OBJ(&pipe_, GetSysWorkSpacePtr(), mm_i2h_, (TCubeTiling*)nullptr);
        REGIST_MATMUL_OBJ(&pipe_, GetSysWorkSpacePtr(), mm_h2o_, (TCubeTiling*)nullptr);

        // Configure constant B and biases
        mm_i2h_.SetOrgShape(m, H_, K_);
        mm_i2h_.SetTail(m, H_, K_);
        // A comes from aWorkBase per step; B is w_i2h (transB=true)
        mm_i2h_.SetTensorB(wi2hGm_[0], false);
        mm_i2h_.SetBias(bi2hGm_);

        mm_h2o_.SetOrgShape(m, O_, H_);
        mm_h2o_.SetTail(m, O_, H_);
        mm_h2o_.SetTensorB(whoGm_[0], false);
        mm_h2o_.SetBias(bhoGm_);

        for (uint32_t t = 0; t < T_; ++t) {
            // Build A_work [m, K] by concatenating x[t, bStart:bEnd, :] and hBufBase
            BuildAWork(aWorkBase, hBufBase, t, bStart, m);

            // Compute pre-activation into hBufBase: [m, H] = [m,K] x [K,H] + b
            mm_i2h_.SetTensorA(aWorkBase[0], false);
            mm_i2h_.IterateAll(hBufBase[0]);

            // h = tanh(pre) in place on hBufBase (m*H elements)
            TanhInplace(hBufBase, static_cast<uint32_t>(m * H_));

            // y_t = h @ w_h2o^T + b_h2o -> write to y[t, bStart:bEnd, :]
            int64_t yOff = (static_cast<int64_t>(t) * B_ + bStart) * O_;
            mm_h2o_.SetTensorA(hBufBase[0], false);
            mm_h2o_.IterateAll(yGm_[yOff]);
        }

        mm_i2h_.End();
        mm_h2o_.End();
    }

private:
    // Build A_work rows by rows: A[b, 0:I] = x[t, bStart+b, :], A[b, I:I+H] = hBuf[b, :]
    __aicore__ inline void BuildAWork(__gm__ float* aWorkBase, __gm__ float* hBufBase,
                                      uint32_t t, uint32_t bStart, uint32_t m)
    {
        for (uint32_t rb = 0; rb < m; ++rb) {
            int64_t aRowOff = static_cast<int64_t>(rb) * K_;
            // Copy x row -> aWork row [0:I]
            int64_t xOff = (static_cast<int64_t>(t) * B_ + static_cast<int64_t>(bStart + rb)) * I_;
            CopyGmToGm(aWorkBase + aRowOff + 0, xGm_.GetPhyAddr() + xOff, I_);

            // Copy h row -> aWork row [I:I+H]
            int64_t hOff = static_cast<int64_t>(rb) * H_;
            CopyGmToGm(aWorkBase + aRowOff + I_, hBufBase + hOff, H_);
        }
    }

    // GM->GM copy via UB, length in elements
    __aicore__ inline void CopyGmToGm(__gm__ float* dst, __gm__ float* src, int64_t elems)
    {
        int64_t offset = 0;
        while (offset < elems) {
            uint32_t chunk = tileVecLen_;
            if (static_cast<int64_t>(chunk) > (elems - offset)) {
                chunk = static_cast<uint32_t>(elems - offset);
            }
            AscendC::LocalTensor<float> inLocal = inQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> outLocal = outQueue_.AllocTensor<float>();
            AscendC::DataCopy(inLocal, src + offset, chunk);
            AscendC::DataCopy(outLocal, inLocal, chunk);
            AscendC::DataCopy(dst + offset, outLocal, chunk);
            outQueue_.FreeTensor(outLocal);
            inQueue_.FreeTensor(inLocal);
            offset += chunk;
        }
    }

    // In-place tanh on GM buffer
    __aicore__ inline void TanhInplace(__gm__ float* buf, uint32_t elems)
    {
        uint32_t processed = 0;
        while (processed < elems) {
            uint32_t chunk = tileVecLen_;
            uint32_t remain = elems - processed;
            if (chunk > remain) chunk = remain;

            AscendC::LocalTensor<float> inLocal = inQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> outLocal = outQueue_.AllocTensor<float>();

            AscendC::DataCopy(inLocal, buf + processed, chunk);
            AscendC::Tanh<float, false>(outLocal, inLocal, chunk);
            AscendC::DataCopy(buf + processed, outLocal, chunk);

            outQueue_.FreeTensor(outLocal);
            inQueue_.FreeTensor(inLocal);

            processed += chunk;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> wi2hGm_;
    AscendC::GlobalTensor<float> bi2hGm_;
    AscendC::GlobalTensor<float> whoGm_;
    AscendC::GlobalTensor<float> bhoGm_;
    AscendC::GlobalTensor<float> yGm_;
    __gm__ float* workspaceGm_ {nullptr};

    // Matmul types: A from GM ND float; B from GM ND float with transpose
    using A_TYPE_I2H = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
    using B_TYPE_I2H = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true>;
    using C_TYPE_I2H = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
    using BIAS_TYPE_I2H = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
    AscendC::Matmul<A_TYPE_I2H, B_TYPE_I2H, C_TYPE_I2H, BIAS_TYPE_I2H> mm_i2h_;

    using A_TYPE_H2O = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
    using B_TYPE_H2O = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true>;
    using C_TYPE_H2O = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
    using BIAS_TYPE_H2O = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
    AscendC::Matmul<A_TYPE_H2O, B_TYPE_H2O, C_TYPE_H2O, BIAS_TYPE_H2O> mm_h2o_;

    uint32_t T_{0}, B_{0}, I_{0}, H_{0}, O_{0}, K_{0};
    uint32_t bPerCore_{0};
    uint32_t tileVecLen_{0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                 GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                                                 GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y, workspace,
            tiling_data.T, tiling_data.B, tiling_data.I,
            tiling_data.H, tiling_data.O, tiling_data.K,
            tiling_data.bPerCore, tiling_data.tileVecLen);
    op.Process();
}
