namespace optiling {

// UB model for this kernel:
//   inRow0: 1 * (tileW + 2)
//   inRow1: 1 * (tileW + 2)
//   inRow2: 1 * (tileW + 2)
//   outQueueY: 2 * tileW
//   tmpQueue: 1 * tileW
//   wBuf: 9
// Total floats ~= 5 * tileW + (6 + 9) => approx 5 * tileW + 15 => bytes ~ 4 * (5 * tileW + 15)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // favor DMA burst alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap tile width size

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect NCHW
    const uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t C = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t KH = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t KW = static_cast<uint32_t>(w_shape->GetDim(3));

    const uint32_t strideH = 1;
    const uint32_t strideW = 1;
    const uint32_t dilationH = 1;
    const uint32_t dilationW = 1;
    const uint32_t padH = 0;
    const uint32_t padW = 0;

    // Output shape for depthwise conv (groups=C), padding=0, stride=1, dilation=1
    const uint32_t HOut = (H + 2 * padH - dilationH * (KH - 1) - 1) / strideH + 1;
    const uint32_t WOut = (W + 2 * padW - dilationW * (KW - 1) - 1) / strideW + 1;

    // tileW from UB budget: bytes ~= 4 * (5 * tileW + 15)
    uint32_t maxTileW = (UB_BUDGET_BYTES / 4u);
    if (maxTileW > 15u) {
        maxTileW = (maxTileW - 15u) / 5u;
    } else {
        maxTileW = ALIGN_ELEMS;
    }
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;
    if (maxTileW > HARD_CAP_ELEMS) maxTileW = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileW = maxTileW;
    if (WOut < tileW) {
        const uint32_t aligned = AlignDown(WOut, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : WOut;

    // Grid across N*C tasks
    const uint32_t tasks = N * C;
    uint32_t blockDim = tasks;

    tiling.set_w(W);

    tiling.set_kw(KW);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);

    tiling.set_dilationW(dilationW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_hOut(HOut);
    tiling.set_wOut(WOut);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t kh, uint32_t kw,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t padH, uint32_t padW,
                                uint32_t hOut, uint32_t wOut, uint32_t tileW)
    {
        n_ = n; c_ = c; h_ = h; w_ = w_in;
        kh_ = kh; kw_ = kw;
        strideH_ = strideH; strideW_ = strideW;
        dilationH_ = dilationH; dilationW_ = dilationW;
        padH_ = padH; padW_ = padW;
        hOut_ = hOut; wOut_ = wOut;
        tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * c_ * h_ * w_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(c_ * kh_ * kw_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_ * c_ * hOut_ * wOut_));

        // Buffers: three input rows (tileW + 2) each, output tileW, tmp tileW, weights 9
        pipe_.InitBuffer(inRow0_, 1, (tileW_ + 2) * sizeof(float));
        pipe_.InitBuffer(inRow1_, 1, (tileW_ + 2) * sizeof(float));
        pipe_.InitBuffer(inRow2_, 1, (tileW_ + 2) * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(wBufQueue_, 1, 9 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || hOut_ == 0 || wOut_ == 0 || tileW_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tasks = n_ * c_;
        for (uint32_t task = blockIdx; task < tasks; task += blockNum) {
            const uint32_t nIdx = task / c_;
            const uint32_t cIdx = task % c_;

            LoadWeights(cIdx);

            for (uint32_t ho = 0; ho < hOut_; ++ho) {
                // Input base row index (no padding, stride 1, dilation 1)
                const uint32_t h0 = ho;
                const uint32_t h1 = ho + 1;
                const uint32_t h2 = ho + 2;

                const uint64_t inBase0 = IndexInput(nIdx, cIdx, h0, 0);
                const uint64_t inBase1 = IndexInput(nIdx, cIdx, h1, 0);
                const uint64_t inBase2 = IndexInput(nIdx, cIdx, h2, 0);

                const uint64_t outBase = IndexOutput(nIdx, cIdx, ho, 0);

                // Tile across width
                uint32_t wStart = 0;
                while (wStart < wOut_) {
                    uint32_t curTileW = wOut_ - wStart;
                    if (curTileW > tileW_) curTileW = tileW_;

                    CopyInRows(inBase0 + wStart, inBase1 + wStart, inBase2 + wStart, curTileW);
                    Compute(curTileW);
                    CopyOut(outBase + wStart, curTileW);

                    wStart += curTileW;
                }
            }
        }
    }

private:
    __aicore__ inline uint64_t IndexInput(uint32_t nIdx, uint32_t cIdx, uint32_t hIdx, uint32_t wIdx) const
    {
        // NCHW layout
        return static_cast<uint64_t>(nIdx) * (static_cast<uint64_t>(c_) * h_ * w_) +
               static_cast<uint64_t>(cIdx) * (static_cast<uint64_t>(h_) * w_) +
               static_cast<uint64_t>(hIdx) * w_ +
               static_cast<uint64_t>(wIdx);
    }

    __aicore__ inline uint64_t IndexOutput(uint32_t nIdx, uint32_t cIdx, uint32_t hIdx, uint32_t wIdx) const
    {
        // NCHW layout
        return static_cast<uint64_t>(nIdx) * (static_cast<uint64_t>(c_) * hOut_ * wOut_) +
               static_cast<uint64_t>(cIdx) * (static_cast<uint64_t>(hOut_) * wOut_) +
               static_cast<uint64_t>(hIdx) * wOut_ +
               static_cast<uint64_t>(wIdx);
    }

    __aicore__ inline void LoadWeights(uint32_t cIdx)
    {
        AscendC::LocalTensor<float> wLocal = wBufQueue_.AllocTensor<float>();
        // depthwise: weight for channel cIdx has shape [1, KH, KW] flattened to 9
        const uint64_t wOffset = static_cast<uint64_t>(cIdx) * (kh_ * kw_);
        AscendC::DataCopy(wLocal, wGm_[wOffset], kh_ * kw_);
        // Extract scalars
        w00_ = wLocal.GetValue(0);
        w01_ = wLocal.GetValue(1);
        w02_ = wLocal.GetValue(2);
        w10_ = wLocal.GetValue(3);
        w11_ = wLocal.GetValue(4);
        w12_ = wLocal.GetValue(5);
        w20_ = wLocal.GetValue(6);
        w21_ = wLocal.GetValue(7);
        w22_ = wLocal.GetValue(8);
        wBufQueue_.FreeTensor(wLocal);
    }

    __aicore__ inline void CopyInRows(uint64_t gmOffset0, uint64_t gmOffset1, uint64_t gmOffset2, uint32_t curTileW)
    {
        AscendC::LocalTensor<float> r0 = inRow0_.AllocTensor<float>();
        AscendC::LocalTensor<float> r1 = inRow1_.AllocTensor<float>();
        AscendC::LocalTensor<float> r2 = inRow2_.AllocTensor<float>();

        // Need WOut segment plus right halo of 2 for 3x3 window
        const uint32_t loadLen = curTileW + 2;

        AscendC::DataCopy(r0, xGm_[gmOffset0], loadLen);
        AscendC::DataCopy(r1, xGm_[gmOffset1], loadLen);
        AscendC::DataCopy(r2, xGm_[gmOffset2], loadLen);

        inRow0_.EnQue(r0);
        inRow1_.EnQue(r1);
        inRow2_.EnQue(r2);
    }

    __aicore__ inline void Compute(uint32_t curTileW)
    {
        AscendC::LocalTensor<float> r0 = inRow0_.DeQue<float>();
        AscendC::LocalTensor<float> r1 = inRow1_.DeQue<float>();
        AscendC::LocalTensor<float> r2 = inRow2_.DeQue<float>();

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // y = r0[0:W]*w00 + r0[1:W+1]*w01 + r0[2:W+2]*w02
        AscendC::Muls(yLocal, r0, w00_, curTileW);
        AscendC::Muls(tLocal, r0[1], w01_, curTileW);
        AscendC::Add(yLocal, yLocal, tLocal, curTileW);
        AscendC::Muls(tLocal, r0[2], w02_, curTileW);
        AscendC::Add(yLocal, yLocal, tLocal, curTileW);

        // + r1[*]*w1*
        AscendC::Muls(tLocal, r1, w10_, curTileW);
        AscendC::Add(yLocal, yLocal, tLocal, curTileW);
        AscendC::Muls(tLocal, r1[1], w11_, curTileW);
        AscendC::Add(yLocal, yLocal, tLocal, curTileW);
        AscendC::Muls(tLocal, r1[2], w12_, curTileW);
        AscendC::Add(yLocal, yLocal, tLocal, curTileW);

        // + r2[*]*w2*
        AscendC::Muls(tLocal, r2, w20_, curTileW);
        AscendC::Add(yLocal, yLocal, tLocal, curTileW);
        AscendC::Muls(tLocal, r2[1], w21_, curTileW);
        AscendC::Add(yLocal, yLocal, tLocal, curTileW);
        AscendC::Muls(tLocal, r2[2], w22_, curTileW);
        AscendC::Add(yLocal, yLocal, tLocal, curTileW);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue(yLocal);

        inRow0_.FreeTensor(r0);
        inRow1_.FreeTensor(r1);
        inRow2_.FreeTensor(r2);
    }

    __aicore__ inline void CopyOut(uint64_t gmOffset, uint32_t curTileW)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, curTileW);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inRow0_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inRow1_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inRow2_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wBufQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t kh_{0}, kw_{0};
    uint32_t strideH_{1}, strideW_{1};
    uint32_t dilationH_{1}, dilationW_{1};
    uint32_t padH_{0}, padW_{0};
    uint32_t hOut_{0}, wOut_{0}, tileW_{0};

    float w00_{0.f}, w01_{0.f}, w02_{0.f};
    float w10_{0.f}, w11_{0.f}, w12_{0.f};
    float w20_{0.f}, w21_{0.f}, w22_{0.f};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymmetricInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kh, tiling_data.kw,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.hOut, tiling_data.wOut, tiling_data.tileW);
    op.Process();
}
