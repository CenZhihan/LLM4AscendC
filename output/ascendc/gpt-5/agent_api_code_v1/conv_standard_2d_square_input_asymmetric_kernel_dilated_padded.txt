#include \"kernel_operator.h\"

using AscendC::DataCopy;

class KernelConv2DStdDilatedPadded {
public:
    __aicore__ inline KernelConv2DStdDilatedPadded() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t cout, uint32_t kh, uint32_t kw,
                                uint32_t out_h, uint32_t out_w,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_top, uint32_t pad_bottom,
                                uint32_t pad_left, uint32_t pad_right,
                                uint32_t dilation_h, uint32_t dilation_w,
                                uint32_t tile_w)
    {
        n_ = n; c_ = c; h_ = h; wIn_ = w_in;
        cout_ = cout; kh_ = kh; kw_ = kw;
        outH_ = out_h; outW_ = out_w;
        strideH_ = stride_h; strideW_ = stride_w;
        padTop_ = pad_top; padBottom_ = pad_bottom;
        padLeft_ = pad_left; padRight_ = pad_right;
        dilH_ = dilation_h; dilW_ = dilation_w;
        tileW_ = tile_w;

        // K size
        K_ = c_ * kh_ * kw_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * c_ * h_ * wIn_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)cout_ * K_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * cout_ * outH_ * outW_);

        // Buffers: acc, tmp, v (each tileW_), and weights (K_)
        pipe_.InitBuffer(accQ_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(vQ_,   1, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQ_,   1, K_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outH_ == 0 || outW_ == 0 || cout_ == 0) return;

        const uint32_t tilesPerRow = CeilDiv(outW_, tileW_);
        const uint32_t totalTiles = n_ * outH_ * tilesPerRow;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tile = blockIdx; tile < totalTiles; tile += blockNum) {
            // Map tile to (n, oh, tIdx)
            uint32_t tmp = tile;
            const uint32_t n = tmp / (outH_ * tilesPerRow);
            tmp = tmp % (outH_ * tilesPerRow);
            const uint32_t oh = tmp / tilesPerRow;
            const uint32_t tIdx = tmp % tilesPerRow;

            const uint32_t ow_start = tIdx * tileW_;
            const uint32_t curT = (ow_start + tileW_ <= outW_) ? tileW_ : (outW_ - ow_start);

            for (uint32_t oc = 0; oc < cout_; ++oc) {
                // Load weights for this oc
                AscendC::LocalTensor<float> wLocal = wQ_.AllocTensor<float>();
                const uint64_t wOffset = (uint64_t)oc * K_;
                DataCopy(wLocal, wGm_[wOffset], K_);
                wQ_.EnQue(wLocal);

                // Prepare accumulator
                AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
                AscendC::Dup(acc, 0.0f, curT);

                // Iterate over K = c * kh * kw
                AscendC::LocalTensor<float> wLocalUse = wQ_.DeQue<float>();

                for (uint32_t ci = 0; ci < c_; ++ci) {
                    for (uint32_t kh = 0; kh < kh_; ++kh) {
                        const int32_t in_h = static_cast<int32_t>(oh * strideH_) - static_cast<int32_t>(padTop_) +
                                             static_cast<int32_t>(kh * dilH_);
                        bool h_valid = (in_h >= 0) && (in_h < static_cast<int32_t>(h_));
                        for (uint32_t kw = 0; kw < kw_; ++kw) {
                            const uint32_t kIndex = ((ci * kh_) + kh) * kw_ + kw;

                            // Prepare v vector for this k
                            AscendC::LocalTensor<float> v = vQ_.AllocTensor<float>();
                            if (!h_valid) {
                                AscendC::Dup(v, 0.0f, curT);
                            } else {
                                const int32_t base_w_in = static_cast<int32_t>(ow_start * strideW_) -
                                                          static_cast<int32_t>(padLeft_) +
                                                          static_cast<int32_t>(kw * dilW_);
                                // left zeros
                                int32_t leftInv = 0;
                                if (base_w_in < 0) leftInv = -base_w_in;
                                // right zeros
                                int32_t rightInv = 0;
                                const int32_t last_w_in = base_w_in + static_cast<int32_t>(curT) - 1;
                                if (last_w_in >= static_cast<int32_t>(wIn_)) {
                                    rightInv = last_w_in - static_cast<int32_t>(wIn_) + 1;
                                }
                                int32_t midLen = static_cast<int32_t>(curT) - leftInv - rightInv;
                                if (midLen <= 0) {
                                    AscendC::Dup(v, 0.0f, curT);
                                } else {
                                    // Zero fill then copy mid segment
                                    AscendC::Dup(v, 0.0f, curT);
                                    const uint64_t gmOffset = GetXOffset(n, ci, (uint32_t)in_h,
                                                                         (uint32_t)(base_w_in + leftInv));
                                    DataCopy(v[leftInv], xGm_[gmOffset], (uint32_t)midLen);
                                }
                            }

                            // tmp = v * wScalar; acc += tmp
                            AscendC::LocalTensor<float> tmpv = tmpQ_.AllocTensor<float>();
                            float wScalar = wLocalUse.GetValue(kIndex);
                            AscendC::Muls(tmpv, v, wScalar, curT);
                            AscendC::Add(acc, acc, tmpv, curT);

                            tmpQ_.FreeTensor(tmpv);
                            vQ_.FreeTensor(v);
                        }
                    }
                }

                // Write back acc to y
                const uint64_t yOffset = GetYOffset(n, oc, oh, ow_start);
                DataCopy(yGm_[yOffset], acc, curT);

                accQ_.FreeTensor(acc);
                wQ_.FreeTensor(wLocalUse);
            }
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) const { return (a + b - 1u) / b; }

    __aicore__ inline uint64_t GetXOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * c_ + c) * h_ + h) * wIn_ + w;
    }

    __aicore__ inline uint64_t GetYOffset(uint32_t n, uint32_t oc, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * cout_ + oc) * outH_ + h) * outW_ + w;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECIN,   1> vQ_;
    AscendC::TQue<AscendC::TPosition::VECIN,   1> wQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, wIn_{0};
    uint32_t cout_{0}, kh_{0}, kw_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t strideH_{1}, strideW_{1};
    uint32_t padTop_{0}, padBottom_{0}, padLeft_{0}, padRight_{0};
    uint32_t dilH_{1}, dilW_{1};
    uint32_t tileW_{0};
    uint32_t K_{0};
};

extern \"C\" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2DStdDilatedPadded op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.cout, tiling_data.kh, tiling_data.kw,
            tiling_data.out_h, tiling_data.out_w,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_top, tiling_data.pad_bottom,
            tiling_data.pad_left, tiling_data.pad_right,
            tiling_data.dilation_h, tiling_data.dilation_w,
            tiling_data.tile_w);
    op.Process();
}
