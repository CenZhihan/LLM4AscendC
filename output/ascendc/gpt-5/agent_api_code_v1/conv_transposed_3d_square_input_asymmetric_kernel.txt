#include "kernel_operator.h"

class KernelConvTransposed3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t outD, uint32_t outH, uint32_t outW)
    {
        n_ = n; cIn_ = cIn; cOut_ = cOut;
        inD_ = inD; inH_ = inH; inW_ = inW;
        kD_ = kD; kH_ = kH; kW_ = kW;
        outD_ = outD; outH_ = outH; outW_ = outW;

        // total elements
        uint64_t xElems = (uint64_t)n_ * cIn_ * inD_ * inH_ * inW_;
        uint64_t wElems = (uint64_t)cIn_ * cOut_ * kD_ * kH_ * kW_;
        uint64_t yElems = (uint64_t)n_ * cOut_ * outD_ * outH_ * outW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint32_t)xElems);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint32_t)wElems);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint32_t)yElems);

        // UB buffers
        // wLocal: cIn*kD*kH*kW
        // xRow: inW
        // yRow: outW
        // tmp: inW
        pipe_.InitBuffer(wQueue_, 1, (uint32_t)(cIn_ * kD_ * kH_ * kW_ * sizeof(float)));
        pipe_.InitBuffer(xQueue_, 1, (uint32_t)(inW_ * sizeof(float)));
        pipe_.InitBuffer(yQueue_, 1, (uint32_t)(outW_ * sizeof(float)));
        pipe_.InitBuffer(tmpQueue_, 1, (uint32_t)(inW_ * sizeof(float)));
    }

    __aicore__ inline void Process()
    {
        const uint32_t totalTasks = n_ * cOut_ * outD_;
        if (totalTasks == 0) {
            return;
        }
        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t t = blockIdx; t < totalTasks; t += blockNum) {
            uint32_t t1 = t / outD_;
            uint32_t oz = t - t1 * outD_;
            uint32_t co = t1 % cOut_;
            uint32_t n = t1 / cOut_;

            ComputeTask(n, co, oz);
        }
    }

private:
    __aicore__ inline void LoadWeightsForCo(uint32_t co, AscendC::LocalTensor<float>& wLocal)
    {
        // weight layout: [C_in, C_out, kD, kH, kW]
        const uint32_t KTot = kD_ * kH_ * kW_;
        const uint64_t stride_kw = 1ull;
        const uint64_t stride_kh = (uint64_t)kW_;
        const uint64_t stride_kd = (uint64_t)kH_ * kW_;
        const uint64_t stride_co = (uint64_t)kD_ * kH_ * kW_;
        const uint64_t stride_ci = (uint64_t)cOut_ * kD_ * kH_ * kW_;

        for (uint32_t ci = 0; ci < cIn_; ++ci) {
            uint64_t base = (uint64_t)ci * stride_ci + (uint64_t)co * stride_co; // start of [kD,kH,kW] for this (ci,co)
            AscendC::DataCopy(wLocal[ci * KTot], wGm_[(uint32_t)base], KTot);
        }
    }

    __aicore__ inline void ComputeTask(uint32_t n, uint32_t co, uint32_t oz)
    {
        // Prepare weights for this output channel
        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        LoadWeightsForCo(co, wLocal);

        // Strides for x and y
        const uint64_t x_stride_w = 1ull;
        const uint64_t x_stride_h = (uint64_t)inW_;
        const uint64_t x_stride_d = (uint64_t)inH_ * inW_;
        const uint64_t x_stride_c = (uint64_t)inD_ * inH_ * inW_;
        const uint64_t x_stride_n = (uint64_t)cIn_ * inD_ * inH_ * inW_;

        const uint64_t y_stride_w = 1ull;
        const uint64_t y_stride_h = (uint64_t)outW_;
        const uint64_t y_stride_d = (uint64_t)outH_ * outW_;
        const uint64_t y_stride_c = (uint64_t)outD_ * outH_ * outW_;
        const uint64_t y_stride_n = (uint64_t)cOut_ * outD_ * outH_ * outW_;

        // For each output row oy
        for (uint32_t oy = 0; oy < outH_; ++oy) {
            AscendC::LocalTensor<float> yRow = yQueue_.AllocTensor<float>();
            AscendC::Duplicate(yRow, 0.0f, (int32_t)outW_);

            for (uint32_t ci = 0; ci < cIn_; ++ci) {
                for (uint32_t kd = 0; kd < kD_; ++kd) {
                    // input z index
                    int64_t iz = (int64_t)oz - (int64_t)kd;
                    if (iz < 0 || iz >= (int64_t)inD_) continue;

                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        int64_t iy = (int64_t)oy - (int64_t)kh;
                        if (iy < 0 || iy >= (int64_t)inH_) continue;

                        // Load x row [ix=0..inW-1]
                        uint64_t xBase = (uint64_t)n * x_stride_n
                                       + (uint64_t)ci * x_stride_c
                                       + (uint64_t)iz * x_stride_d
                                       + (uint64_t)iy * x_stride_h;
                        AscendC::LocalTensor<float> xVec = xQueue_.AllocTensor<float>();
                        AscendC::DataCopy(xVec, xGm_[(uint32_t)xBase], inW_);

                        // For each kernel width kx, accumulate into yRow[kx : kx+inW-1]
                        const uint32_t KTot = kD_ * kH_ * kW_;
                        const uint32_t wBaseLocal = ci * KTot + kd * (kH_ * kW_) + kh * kW_;

                        for (uint32_t kx = 0; kx < kW_; ++kx) {
                            float w_scalar = wLocal[wBaseLocal + kx];
                            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                            AscendC::Muls(tmp, xVec, w_scalar, inW_);

                            AscendC::LocalTensor<float> ySub = yRow[kx];
                            AscendC::Add(ySub, ySub, tmp, inW_);

                            tmpQueue_.FreeTensor(tmp);
                        }

                        xQueue_.FreeTensor(xVec);
                    }
                }
            }

            // Store y row to GM
            uint64_t yBase = (uint64_t)n * y_stride_n
                           + (uint64_t)co * y_stride_c
                           + (uint64_t)oz * y_stride_d
                           + (uint64_t)oy * y_stride_h;
            AscendC::DataCopy(yGm_[(uint32_t)yBase], yRow, outW_);
            yQueue_.FreeTensor(yRow);
        }

        wQueue_.FreeTensor(wLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, cOut_{0};
    uint32_t inD_{0}, inH_{0}, inW_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};
    uint32_t outD_{0}, outH_{0}, outW_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputAsymmetricKernelCustom op;
    op.Init(
        x, weight, y,
        tiling_data.n, tiling_data.cIn, tiling_data.cOut,
        tiling_data.inD, tiling_data.inH, tiling_data.inW,
        tiling_data.kD, tiling_data.kH, tiling_data.kW,
        tiling_data.outD, tiling_data.outH, tiling_data.outW
    );
    op.Process();
}
