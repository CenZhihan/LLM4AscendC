kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Din, uint32_t Hin, uint32_t Win,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t Dout, uint32_t Hout, uint32_t Wout,
                                uint64_t totalRows, uint32_t rowLength)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Din_ = Din; Hin_ = Hin; Win_ = Win;
        kD_ = kD; kH_ = kH; kW_ = kW;
        Dout_ = Dout; Hout_ = Hout; Wout_ = Wout;
        totalRows_ = totalRows; rowLen_ = rowLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_)*Cin_*Din_*Hin_*Win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Cin_)*Cout_*kD_*kH_*kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_)*Cout_*Dout_*Hout_*Wout_);

        // Buffers:
        // yRow: length Wout
        // xRow: length Win
        // tmp: length Win
        pipe_.InitBuffer(yRowQueue_, BUFFER_NUM, rowLen_ * sizeof(float));
        pipe_.InitBuffer(xRowQueue_, BUFFER_NUM, Win_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, Win_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint64_t row = blockIdx;
        while (row < totalRows_) {
            ComputeRow(row);
            row += blockNum;
        }
    }

private:
    __aicore__ inline void ZeroInit(AscendC::LocalTensor<float> &yRow, uint32_t len)
    {
        // yRow = 0
        AscendC::Muls(yRow, yRow, 0.0f, len);
    }

    __aicore__ inline void ComputeRow(uint64_t rowIndex)
    {
        // Map rowIndex to n, cout, oz, oy
        uint64_t t = rowIndex;
        const uint32_t oy = static_cast<uint32_t>(t % Hout_); t /= Hout_;
        const uint32_t oz = static_cast<uint32_t>(t % Dout_); t /= Dout_;
        const uint32_t cout = static_cast<uint32_t>(t % Cout_); t /= Cout_;
        const uint32_t n = static_cast<uint32_t>(t % N_);

        // Allocate y row
        AscendC::LocalTensor<float> yRow = yRowQueue_.AllocTensor<float>();
        ZeroInit(yRow, Wout_);

        // Precompute bases
        const uint64_t xNCBase = (static_cast<uint64_t>(n) * Cin_) * (static_cast<uint64_t>(Din_) * Hin_ * Win_);
        const uint64_t yRowBase = ((((static_cast<uint64_t>(n) * Cout_ + cout) * Dout_ + oz) * Hout_) + oy) * Wout_;

        // Iterate over input channels and kernel
        for (uint32_t cin = 0; cin < Cin_; ++cin) {
            for (uint32_t kd = 0; kd < kD_; ++kd) {
                // iz = oz - kd
                int32_t iz = static_cast<int32_t>(oz) - static_cast<int32_t>(kd);
                if (iz < 0 || iz >= static_cast<int32_t>(Din_)) continue;

                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    int32_t iy = static_cast<int32_t>(oy) - static_cast<int32_t>(kh);
                    if (iy < 0 || iy >= static_cast<int32_t>(Hin_)) continue;

                    // Load x row once per (cin, kd, kh)
                    const uint64_t xRowBase = ((((xNCBase + cin * (uint64_t)Din_ + (uint64_t)iz) * Hin_) + (uint64_t)iy) * Win_);
                    // Copy x row
                    AscendC::LocalTensor<float> xRow = xRowQueue_.AllocTensor<float>();
                    AscendC::DataCopy(xRow, xGm_[xRowBase], Win_);

                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        // Weight scalar
                        const uint64_t wIndex =
                            (((((uint64_t)cin) * Cout_ + (uint64_t)cout) * kD_ + (uint64_t)kd) * kH_ + (uint64_t)kh) * kW_ + (uint64_t)kw;
                        float wScalar = 0.0f;

                        // Load one scalar from GM weight to tmp buffer to fetch value
                        // We reuse tmp buffer's first element to hold scalar by copying length 1
                        AscendC::LocalTensor<float> tmpVec = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmpVec, wGm_[wIndex], 1);
                        wScalar = tmpVec.GetValue(0); // Access helper
                        // Multiply xRow by scalar into tmpVec
                        AscendC::Muls(tmpVec, xRow, wScalar, Win_);

                        // Accumulate into yRow shifted by kw: yRow[kw + i] += tmpVec[i] for i in [0..Win-1]
                        for (uint32_t i = 0; i < Win_; ++i) {
                            // Bounds guaranteed: kw + i < Wout_ because Wout_ = Win_ + kW_ - 1
                            float acc = yRow.GetValue(kw + i);
                            float addv = tmpVec.GetValue(i);
                            yRow.SetValue(kw + i, acc + addv);
                        }

                        tmpQueue_.FreeTensor(tmpVec);
                    }

                    xRowQueue_.FreeTensor(xRow);
                }
            }
        }

        // Store y row to GM
        AscendC::DataCopy(yGm_[yRowBase], yRow, Wout_);
        yRowQueue_.EnQue<float>(yRow);
        yRowQueue_.DeQue<float>();
        yRowQueue_.FreeTensor(yRow);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Din_{0}, Hin_{0}, Win_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};
    uint64_t totalRows_{0};
    uint32_t rowLen_{0};
};

// Helper methods for LocalTensor element access
namespace AscendC {
template <>
__aicore__ inline float LocalTensor<float>::GetValue(uint32_t idx) const {
    return this->GetPointer()[idx];
}
template <>
__aicore__ inline void LocalTensor<float>::SetValue(uint32_t idx, float v) {
    this->GetPointer()[idx] = v;
}
}

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvTransposed3dAsymmetricInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.Din, tiling_data.Hin, tiling_data.Win,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.Dout, tiling_data.Hout, tiling_data.Wout,
            tiling_data.totalRows, tiling_data.rowLength);
    op.Process();
}
