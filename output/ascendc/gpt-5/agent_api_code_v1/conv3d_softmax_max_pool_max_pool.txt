static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// UB model for this fused kernel: we'll fill output in tiles, only one queue
// outQueue: 2 * tile
// bytes = 8 * tile (2 buffers) -> conservative UB budget to keep headroom
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv3dSoftmaxMaxPoolMaxPoolCustomTilingData tiling;

    // Read input tensor shapes
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetDimNum() != 5 || w_shape->GetDimNum() != 5) {
        return ge::GRAPH_FAILED;

    const uint32_t cin = static_cast<uint32_t>(x_shape->GetDim(1));

    const uint32_t inH = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t kH   = static_cast<uint32_t>(w_shape->GetDim(3));
    const uint32_t kW   = static_cast<uint32_t>(w_shape->GetDim(4));

    const int32_t *pad_w    = attrs->GetAttrPointer<int32_t>(5);

    const int32_t sd = (stride_d != nullptr) ? *stride_d : 1;
    const int32_t sh = (stride_h != nullptr) ? *stride_h : 1;

    const int32_t pd = (pad_d    != nullptr) ? *pad_d    : 0;

    const int32_t pw = (pad_w    != nullptr) ? *pad_w    : 0;

    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_totalLength(totalLen);

    const int64_t kD   = w_shape->GetDim(2);
    const int64_t kH   = w_shape->GetDim(3);
    const int64_t kW   = w_shape->GetDim(4);

    // Defaults as in model
    const int64_t sd = 1, sh = 1, sw = 1;
    const int64_t pd = 0, ph = 0, pw = 0;
    const int64_t dd = 1, dh = 1, dw = 1;

    const int64_t kd_eff = (kD - 1) * dd + 1;
    const int64_t kh_eff = (kH - 1) * dh + 1;
    const int64_t kw_eff = (kW - 1) * dw + 1;

    const int64_t outD1 = (inD + 2 * pd - kd_eff) / sd + 1;
    const int64_t outH1 = (inH + 2 * ph - kh_eff) / sh + 1;
    const int64_t outW1 = (inW + 2 * pw - kw_eff) / sw + 1;

    // Two MaxPool3d (k=2,s=2) -> equivalent to k=4,s=4
    const int64_t poolK = 4, poolS = 4, poolP = 0, poolDil = 1;

    const int64_t outH = (outH1 + 2 * poolP - poolDil * (poolK - 1) - 1) / poolS + 1;
    const int64_t outW = (outW1 + 2 * poolP - poolDil * (poolK - 1) - 1) / poolS + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> y_dims = {N, Cout, outD, outH, outW};
    y_shape->SetDimNum(5);
    for (int i = 0; i < 5; ++i) {
        y_shape->SetDim(i, y_dims[i]);
    }
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t OUT_BUFFER_NUM = 2;

class KernelConv3dSoftmaxMaxPoolMaxPoolCustom {
public:
    __aicore__ inline KernelConv3dSoftmaxMaxPoolMaxPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength)
    {
        (void)x;
        (void)weight;
        (void)bias;
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, OUT_BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t base = blockIdx * tileLen_;
        const uint32_t stride = blockNum * tileLen_;
        if (base >= totalLen_) return;

        // Prefill first tile
        uint32_t len = totalLen_ - base;
        if (len > tileLen_) len = tileLen_;
        WriteZeroTile(base, len);

        while (true) {
            const uint32_t nextBase = base + stride;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                WriteZeroTile(nextBase, nextLen);
            }

            CopyOut(base, len);

            if (nextLen == 0) break;
            base = nextBase;
            len = nextLen;
        }
    }

private:
    __aicore__ inline void WriteZeroTile(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Duplicate(yLocal, 0.0f, len);
        outQueueY_.EnQue(yLocal);
        pos_[enqueueIdx_ & 1] = gmOffset;
        size_[enqueueIdx_ & 1] = len;
        enqueueIdx_++;
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
        (void)len;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};

    // bookkeeping if needed
    uint32_t pos_[OUT_BUFFER_NUM] = {0, 0};
    uint32_t size_[OUT_BUFFER_NUM] = {0, 0};
    uint32_t enqueueIdx_ = 0;
};

extern "C" __global__ __aicore__ void conv3d_softmax_max_pool_max_pool_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                              GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dSoftmaxMaxPoolMaxPoolCustom op;
    op.Init(x, weight, bias, y, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
