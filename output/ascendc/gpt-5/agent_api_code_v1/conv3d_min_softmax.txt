static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : (a + b - 1u) / b;
}

// This kernel computes softmax along the last dimension of the input tensor.
// In the Python model, inputs are permuted to [N, H, W, C] (C is last).
// Tiling is done over rows (N*H*W), each row length is lastDim = C.
static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv3dMinSoftmaxCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto rank = in_shape->GetOriginShape().GetDimNum();
    if (rank == 0) {
        return ge::GRAPH_FAILED;
    }
    uint32_t lastDim = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(rank - 1));
    uint32_t totalLength = static_cast<uint32_t>(in_shape->GetOriginShape().GetShapeSize());

    // UB budgeting:
    // - inQueue: 2 * (tileRows * lastDim) floats
    // - outQueue: 2 * (tileRows * lastDim) floats
    // - tmpQueue: 1 * (lastDim) floats
    // Total bytes ~= 16 * tileRows * lastDim + 4 * lastDim
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t MIN_TILE_ROWS = 1;
    static constexpr uint32_t MAX_BLOCKS = 24;
    static constexpr uint32_t MIN_BLOCKS = 8;

    uint32_t denom = 16u * sizeof(float) * (lastDim == 0 ? 1u : lastDim);
    uint32_t maxTileRows = (denom == 0) ? 1u : (UB_BUDGET_BYTES / denom);
    if (maxTileRows < MIN_TILE_ROWS) maxTileRows = MIN_TILE_ROWS;

    const uint32_t totalRows = (lastDim == 0) ? 0u : (totalLength / lastDim);
    uint32_t tileRows = maxTileRows;
    if (totalRows > 0 && tileRows > totalRows) tileRows = totalRows;

    const uint32_t numTiles = (tileRows == 0) ? 1u : CeilDiv(totalRows, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_lastDim(lastDim);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace required for this operator
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv3dMinSoftmaxCustom {
public:
    __aicore__ inline KernelConv3dMinSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t lastDim,
                                uint32_t tileRows)
    {
        totalLen_ = totalLength;
        lastDim_ = lastDim;
        tileRows_ = tileRows;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        const uint32_t tileLenElems = tileRows_ * lastDim_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLenElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLenElems * sizeof(float));
        // tmp buffer holds one row's working vector (lastDim elements)
        pipe_.InitBuffer(tmpQueue_, 1, lastDim_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || lastDim_ == 0) return;

        const uint32_t totalRows = totalLen_ / lastDim_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideRows = blockNum * tileRows_;
        uint32_t baseRow = blockIdx * tileRows_;
        if (baseRow >= totalRows) return;

        // Prefetch first tile
        uint32_t curRows = totalRows - baseRow;
        if (curRows > tileRows_) curRows = tileRows_;
        CopyIn(baseRow, curRows);

        while (true) {
            // Prefetch next tile if any
            const uint32_t nextBaseRow = baseRow + strideRows;
            uint32_t nextRows = 0;
            if (nextBaseRow < totalRows) {
                nextRows = totalRows - nextBaseRow;
                if (nextRows > tileRows_) nextRows = tileRows_;
                CopyIn(nextBaseRow, nextRows);
            }

            Compute(curRows);
            CopyOut(baseRow, curRows);

            if (nextRows == 0) break;
            baseRow = nextBaseRow;
            curRows = nextRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t baseRow, uint32_t rows)
    {
        const uint32_t len = rows * lastDim_;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[baseRow * lastDim_], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> workRow = tmpQueue_.AllocTensor<float>();

        // Process each row independently: stable softmax over lastDim_
        for (uint32_t r = 0; r < rows; ++r) {
            const uint32_t rowOffset = r * lastDim_;

            // Compute row max
            float maxVal = -3.402823466e+38F; // -FLT_MAX
            for (uint32_t j = 0; j < lastDim_; ++j) {
                float v = xLocal.GetValue(rowOffset + j);
                if (v > maxVal) maxVal = v;
            }

            // Subtract max, exp
            AscendC::Duplicate(workRow, -maxVal, lastDim_);
            AscendC::Add(xLocal[rowOffset], xLocal[rowOffset], workRow, lastDim_);
            AscendC::Exp(xLocal[rowOffset], xLocal[rowOffset], lastDim_);

            // Compute sum of exponentials
            float sumVal = 0.0f;
            for (uint32_t j = 0; j < lastDim_; ++j) {
                sumVal += xLocal.GetValue(rowOffset + j);
            }

            // Normalize
            AscendC::Duplicate(workRow, sumVal, lastDim_);
            AscendC::Div(xLocal[rowOffset], xLocal[rowOffset], workRow, lastDim_);
        }

        // Write to yLocal and free
        AscendC::DataCopy(yLocal, xLocal, rows * lastDim_);
        tmpQueue_.FreeTensor(workRow);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t baseRow, uint32_t rows)
    {
        const uint32_t len = rows * lastDim_;
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[baseRow * lastDim_], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t lastDim_ {0};
    uint32_t tileRows_ {1};
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMinSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.lastDim, tiling_data.tileRows);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'conv3d_min_softmax_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, x, weight, bias):
        # Compute conv3d as in original model
        y = torch.nn.functional.conv3d(x, weight, bias, stride=1, padding=0, dilation=1, groups=1)
        # Reduce along depth (dim=2)
        y_min = torch.min(y, dim=2)[0]  # shape [N, C, H, W]
        # Move channels to last dim for contiguous rows suitable for kernel
        y_nhwc = y_min.permute(0, 2, 3, 1).contiguous()  # [N, H, W, C]
        # Custom kernel: softmax along last dim
        fn = getattr(custom_ops_lib, _FN)
        y_softmax_nhwc = fn(y_nhwc)
        # Restore original layout [N, C, H, W]
        y_softmax = y_softmax_nhwc.permute(0, 3, 1, 2).contiguous()
        return y_softmax
