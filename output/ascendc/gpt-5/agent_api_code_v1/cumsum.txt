host_operator_src="""

#include "cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}

} // anonymous

namespace optiling {

// UB model for this kernel:
//   Single UB buffer that holds rowsPerTile * innerLen elements
// Bytes per element = 4 (float)
// UB budget is conservative per-core share
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // 192KB
static constexpr uint32_t ALIGN_ROWS = 1;                // rows per tile alignment (keep >=1)
static constexpr uint32_t MIN_BLOCKS = 8;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();

    const uint32_t rank = static_cast<uint32_t>(origin.GetDimNum());
    uint64_t totalElemsU64 = origin.GetShapeSize();
    if (totalElemsU64 == 0) {
        totalElemsU64 = 1;
    }

    int32_t dim = -1;
    if (const gert::RuntimeAttrs *attrs = context->GetAttrs()) {
        const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
        if (dimAttr) dim = *dimAttr;
    }

    int32_t axis = dim;
    if (axis < 0) axis += static_cast<int32_t>(rank);
    if (axis < 0) axis = static_cast<int32_t>(rank) - 1;
    if (axis >= static_cast<int32_t>(rank)) axis = static_cast<int32_t>(rank) - 1;

    // We target last-axis cumsum for performance. Compute innerLen as last dimension.
    // If axis != last, we still compute with last axis parameters (user model uses last axis).
    uint32_t innerLen = (rank > 0) ? static_cast<uint32_t>(origin.GetDim(rank - 1)) : static_cast<uint32_t>(totalElemsU64);
    uint32_t totalRows = (innerLen == 0) ? 0u : static_cast<uint32_t>(totalElemsU64 / innerLen);

    // rowsPerTile from UB budget: bytes = 4 * (rowsPerTile * innerLen)
    uint32_t maxRowsPerTile = (innerLen == 0) ? 0u : (UB_BUDGET_BYTES / 4u) / innerLen;
    if (maxRowsPerTile == 0) maxRowsPerTile = 1u;
    maxRowsPerTile = AlignDown(maxRowsPerTile, ALIGN_ROWS);
    if (maxRowsPerTile == 0) maxRowsPerTile = 1u;
    if (maxRowsPerTile > totalRows && totalRows > 0) {
        maxRowsPerTile = totalRows;
    }

    const uint32_t rowsPerTile = maxRowsPerTile;

    const uint32_t numTiles = (rowsPerTile == 0) ? 1u : CeilDiv(totalRows, rowsPerTile);
    uint32_t blockDim = numTiles;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_innerLen(innerLen);
    tiling.set_rowsPerTile(rowsPerTile);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(axis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumsumCustom : public OpDef {
public:
    explicit CumsumCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(-1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumsumCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCumsumCustom {
public:
    __aicore__ inline KernelCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t innerLen, uint32_t rowsPerTile)
    {
        totalRows_ = totalRows;
        innerLen_ = innerLen;
        rowsPerTile_ = rowsPerTile;
        totalElems_ = totalRows_ * innerLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElems_);

        // single UB buffer: rowsPerTile * innerLen elements
        const uint32_t tileElems = rowsPerTile_ * innerLen_;
        pipe_.InitBuffer(bufQueue_, BUFFER_NUM, tileElems * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || innerLen_ == 0 || rowsPerTile_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideRows = blockNum * rowsPerTile_;
        uint32_t baseRow = blockIdx * rowsPerTile_;
        if (baseRow >= totalRows_) return;

        // Prefetch first tile
        uint32_t curRows = totalRows_ - baseRow;
        if (curRows > rowsPerTile_) curRows = rowsPerTile_;
        CopyIn(baseRow, curRows);

        while (true) {
            const uint32_t nextBaseRow = baseRow + strideRows;
            uint32_t nextRows = 0;
            if (nextBaseRow < totalRows_) {
                nextRows = totalRows_ - nextBaseRow;
                if (nextRows > rowsPerTile_) nextRows = rowsPerTile_;
                CopyIn(nextBaseRow, nextRows);
            }

            Compute(curRows);
            CopyOut(baseRow, curRows);

            if (nextRows == 0) break;
            baseRow = nextBaseRow;
            curRows = nextRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t baseRow, uint32_t rows)
    {
        const uint32_t gmOffset = baseRow * innerLen_;
        const uint32_t lenElems = rows * innerLen_;
        AscendC::LocalTensor<float> buf = bufQueue_.AllocTensor<float>();
        AscendC::DataCopy(buf, xGm_[gmOffset], lenElems);
        bufQueue_.EnQue(buf);
    }

    __aicore__ inline void Compute(uint32_t rows)
    {
        AscendC::LocalTensor<float> yLocal = bufQueue_.DeQue<float>();

        for (uint32_t r = 0; r < rows; ++r) {
            const uint32_t rowBase = r * innerLen_;
            // Inclusive scan using iterative doubling:
            // For offset = 1,2,4,... while offset < innerLen:
            //   y[rowBase + offset : rowBase + innerLen] += y[rowBase : rowBase + innerLen - offset]
            for (uint32_t offset = 1; offset < innerLen_; offset <<= 1) {
                const uint32_t len = innerLen_ - offset;
                AscendC::Add(yLocal[rowBase + offset], yLocal[rowBase + offset], yLocal[rowBase], len);
            }
        }

        bufQueue_.FreeTensor(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t baseRow, uint32_t rows)
    {
        const uint32_t gmOffset = baseRow * innerLen_;
        const uint32_t lenElems = rows * innerLen_;
        AscendC::LocalTensor<float> yLocal = bufQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, lenElems);
        bufQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bufQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t innerLen_ {0};
    uint32_t rowsPerTile_ {0};
    uint32_t totalElems_ {0};
};

extern "C" __global__ __aicore__ void cumsum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);
    KernelCumsumCustom op;
    op.Init(x, y, t.totalRows, t.innerLen, t.rowsPerTile);
    op.Process();
}
"""
