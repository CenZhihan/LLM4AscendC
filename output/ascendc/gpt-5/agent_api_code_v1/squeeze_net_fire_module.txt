static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0U) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1U) / b;
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SqueezeNetFireModuleCustomTilingData tiling;

    // Input 0: x [N, C_in, H, W]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto x_dims = x_shape->GetStorageShape();
    if (x_dims.GetDimNum() < 4) {
        return ge::GRAPH_PARAM_INVALID;
    }
    const uint32_t N = static_cast<uint32_t>(x_dims.GetDim(0));
    const uint32_t C_in = static_cast<uint32_t>(x_dims.GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_dims.GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_dims.GetDim(3));

    // Input 1: w_squeeze [C_sq, C_in, 1, 1]
    const gert::Shape *w_sq_shape = context->GetInputShape(1);
    const auto wsq_dims = w_sq_shape->GetStorageShape();
    const uint32_t C_sq = static_cast<uint32_t>(wsq_dims.GetDim(0));

    // Input 3: w_expand1 [C_e1, C_sq, 1, 1]
    const gert::Shape *w_e1_shape = context->GetInputShape(3);
    const auto we1_dims = w_e1_shape->GetStorageShape();
    const uint32_t C_e1 = static_cast<uint32_t>(we1_dims.GetDim(0));

    // Input 5: w_expand3 [C_e3, C_sq, 3, 3]
    const gert::Shape *w_e3_shape = context->GetInputShape(5);
    const auto we3_dims = w_e3_shape->GetStorageShape();
    const uint32_t C_e3 = static_cast<uint32_t>(we3_dims.GetDim(0));

    // UB tiling along W dimension: choose a moderate width to fit common UB limits
    // We will process [N * H * tilesAlongW] tiles
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    // For 1x1 steps, we need buffers for a few vectors; expand3 uses scalar inner loop fallback
    // Pick tileW conservatively based on C_in and C_sq
    uint32_t bytesPerElem = 4;
    uint32_t denom = (C_in + C_sq + (C_e1 > C_e3 ? C_e1 : C_e3));
    if (denom == 0) denom = 1;
    uint32_t tileW = UB_BUDGET_BYTES / (bytesPerElem * denom);
    if (tileW < 64U) tileW = 64U;
    if (tileW > W) tileW = W;
    tileW = AlignDown(tileW, 16U);

    const uint32_t tilesAlongW = (tileW == 0U) ? 1U : CeilDiv(W, tileW);
    uint32_t blockDim = N * H * tilesAlongW;
    if (blockDim < 8U) blockDim = 8U;
    if (blockDim > 24U) blockDim = 24U;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cIn(C_in);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_cSq(C_sq);
    tiling.set_cE1(C_e1);
    tiling.set_cE3(C_e3);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t s_size = static_cast<size_t>(N) * C_sq * H * W * sizeof(float);

kernel_src="""
#include "kernel_operator.h"

struct TilingParams {
    uint32_t n;
    uint32_t cIn;
    uint32_t h;
    uint32_t w;
    uint32_t cSq;
    uint32_t cE1;
    uint32_t cE3;
    uint32_t tileW;
    uint32_t blockDim;
};

class KernelSqueezeNetFireModuleCustom {
public:
    __aicore__ inline KernelSqueezeNetFireModuleCustom() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR w_sq, GM_ADDR b_sq,
                                GM_ADDR w_e1, GM_ADDR b_e1,
                                GM_ADDR w_e3, GM_ADDR b_e3,
                                GM_ADDR y,
                                GM_ADDR workspace,
                                const TilingParams &tp) {
        n_ = tp.n; cin_ = tp.cIn; h_ = tp.h; w_ = tp.w;
        cs_ = tp.cSq; ce1_ = tp.cE1; ce3_ = tp.cE3; tileW_ = tp.tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * cin_ * h_ * w_);
        wSqGm_.SetGlobalBuffer((__gm__ float*)w_sq, (uint64_t)cs_ * cin_);
        bSqGm_.SetGlobalBuffer((__gm__ float*)b_sq, (uint64_t)cs_);
        wE1Gm_.SetGlobalBuffer((__gm__ float*)w_e1, (uint64_t)ce1_ * cs_);
        bE1Gm_.SetGlobalBuffer((__gm__ float*)b_e1, (uint64_t)ce1_);
        wE3Gm_.SetGlobalBuffer((__gm__ float*)w_e3, (uint64_t)ce3_ * cs_ * 3 * 3);
        bE3Gm_.SetGlobalBuffer((__gm__ float*)b_e3, (uint64_t)ce3_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * (ce1_ + ce3_) * h_ * w_);

        // workspace to hold squeeze output: [N, C_sq, H, W]
        sGm_.SetGlobalBuffer((__gm__ float*)workspace, (uint64_t)n_ * cs_ * h_ * w_);

        // Simple local buffers for vector segments along W
        pipe_.InitBuffer(bufIn_, 4, tileW_ * sizeof(float));   // reused per channel
        pipe_.InitBuffer(bufAcc_, 4, tileW_ * sizeof(float));  // accumulation/output vector
    }

    __aicore__ inline void Process() {
        if (n_ == 0 || h_ == 0 || w_ == 0) return;

        const uint32_t tilesAlongW = (tileW_ == 0) ? 1U : (w_ + tileW_ - 1U) / tileW_;
        const uint32_t totalTiles = n_ * h_ * tilesAlongW;
        const uint32_t blkNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blkIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t tileId = blkIdx; tileId < totalTiles; tileId += blkNum) {
            // decode tileId -> (n, h, wStart)
            uint32_t t = tileId;
            const uint32_t wTileIdx = t % tilesAlongW;
            t /= tilesAlongW;
            const uint32_t hh = t % h_;
            t /= h_;
            const uint32_t nn = t;
            const uint32_t wStart = wTileIdx * tileW_;
            uint32_t len = w_ - wStart;
            if (len > tileW_) len = tileW_;
            if (len == 0) continue;

            // Step 1: squeeze 1x1 -> s [N, C_sq, H, W], ReLU
            RunSqueeze(nn, hh, wStart, len);

            // Step 2: expand 1x1 -> y[0:ce1, :, :]
            RunExpand1(nn, hh, wStart, len);

            // Step 3: expand 3x3 (pad=1) -> y[ce1:ce1+ce3, :, :]
            RunExpand3(nn, hh, wStart, len);
        }
    }

private:
    __aicore__ inline uint64_t Offset4D(uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                        uint32_t C, uint32_t H, uint32_t W) const {
        return ((uint64_t)n * C + c) * H * W + (uint64_t)h * W + w;
    }

    // Copy a contiguous [len] from GM x[n, c, h, wStart:wStart+len) to local buffer
    __aicore__ inline void LoadVecFromGM(AscendC::LocalTensor<float>& dst,
                                         AscendC::GlobalTensor<float>& src,
                                         uint64_t baseOff, uint32_t len) {
        AscendC::DataCopy(dst, src[baseOff], len);
    }

    // Copy local buffer to GM y[n, c, h, wStart:wStart+len)
    __aicore__ inline void StoreVecToGM(AscendC::GlobalTensor<float>& dst,
                                        AscendC::LocalTensor<float>& src,
                                        uint64_t baseOff, uint32_t len) {
        AscendC::DataCopy(dst[baseOff], src, len);
    }

    __aicore__ inline void VecAdds(AscendC::LocalTensor<float>& dst, float scalar, uint32_t len) {
        // dst = dst + scalar
        AscendC::Adds(dst, dst, scalar, len);
    }

    __aicore__ inline void VecMaxZero(AscendC::LocalTensor<float>& dst, uint32_t len) {
        AscendC::Maxs(dst, dst, 0.0f, len);
    }

    // dst = a * scalar + dst
    __aicore__ inline void FmaAcc(AscendC::LocalTensor<float>& dst,
                                  AscendC::LocalTensor<float>& a,
                                  float scalar, uint32_t len) {
        // tmp = a * scalar
        AscendC::LocalTensor<float> tmp = bufIn_.AllocTensor<float>();
        AscendC::Muls(tmp, a, scalar, len);
        AscendC::Add(dst, dst, tmp, len);
        bufIn_.FreeTensor(tmp);
    }

    __aicore__ inline void RunSqueeze(uint32_t nn, uint32_t hh, uint32_t wStart, uint32_t len)
    {
        // For each out channel in squeeze
        for (uint32_t oc = 0; oc < cs_; ++oc) {
            AscendC::LocalTensor<float> acc = bufAcc_.AllocTensor<float>();
            // Initialize acc with bias
            // Fill acc with bias by first zeroing and adding bias
            AscendC::Zeros(acc, len);
            VecAdds(acc, GetBias(bSqGm_, oc), len);

            // Accumulate over input channels
            for (uint32_t ic = 0; ic < cin_; ++ic) {
                AscendC::LocalTensor<float> xin = bufIn_.AllocTensor<float>();
                uint64_t xoff = Offset4D(nn, ic, hh, wStart, cin_, h_, w_);
                LoadVecFromGM(xin, xGm_, xoff, len);
                float wval = GetWeight1x1(wSqGm_, oc, ic, cin_);
                FmaAcc(acc, xin, wval, len);
                bufIn_.FreeTensor(xin);
            }
            // ReLU
            VecMaxZero(acc, len);
            // Store to sGm
            uint64_t soff = Offset4D(nn, oc, hh, wStart, cs_, h_, w_);
            StoreVecToGM(sGm_, acc, soff, len);
            bufAcc_.FreeTensor(acc);
        }
    }

    __aicore__ inline void RunExpand1(uint32_t nn, uint32_t hh, uint32_t wStart, uint32_t len)
    {
        for (uint32_t oc = 0; oc < ce1_; ++oc) {
            AscendC::LocalTensor<float> acc = bufAcc_.AllocTensor<float>();
            AscendC::Zeros(acc, len);
            VecAdds(acc, GetBias(bE1Gm_, oc), len);

            for (uint32_t ic = 0; ic < cs_; ++ic) {
                AscendC::LocalTensor<float> sin = bufIn_.AllocTensor<float>();
                uint64_t soff = Offset4D(nn, ic, hh, wStart, cs_, h_, w_);
                LoadVecFromGM(sin, sGm_, soff, len);
                float wval = GetWeight1x1(wE1Gm_, oc, ic, cs_);
                FmaAcc(acc, sin, wval, len);
                bufIn_.FreeTensor(sin);
            }
            VecMaxZero(acc, len);
            // y out channel oc at [nn, oc, hh, wStart:...]
            uint64_t yoff = Offset4D(nn, oc, hh, wStart, ce1_ + ce3_, h_, w_);
            StoreVecToGM(yGm_, acc, yoff, len);
            bufAcc_.FreeTensor(acc);
        }
    }

    __aicore__ inline void RunExpand3(uint32_t nn, uint32_t hh, uint32_t wStart, uint32_t len)
    {
        // We use a vector buffer for output and accumulate per ic,kh,kw by partial loads with edges handling
        for (uint32_t oc = 0; oc < ce3_; ++oc) {
            AscendC::LocalTensor<float> acc = bufAcc_.AllocTensor<float>();
            AscendC::Zeros(acc, len);
            VecAdds(acc, GetBias(bE3Gm_, oc), len);

            for (uint32_t ic = 0; ic < cs_; ++ic) {
                // For each kernel position
                for (int kh = 0; kh < 3; ++kh) {
                    int32_t ih = (int32_t)hh + kh - 1;
                    if (ih < 0 || ih >= (int32_t)h_) {
                        continue; // padding with zeros
                    }
                    for (int kw = 0; kw < 3; ++kw) {
                        int32_t dx = kw - 1;
                        // Compute valid subrange
                        int32_t gStart = (int32_t)wStart + dx;
                        int32_t gEnd = gStart + (int32_t)len;
                        int32_t vStart = 0;
                        int32_t vLen = len;

                        if (gStart < 0) {
                            vStart = -gStart;
                            gStart = 0;
                        }
                        if (gEnd > (int32_t)w_) {
                            vLen = vLen - (gEnd - (int32_t)w_);
                            gEnd = (int32_t)w_;
                        }
                        if (vLen <= 0) {
                            continue;
                        }

                        // Load valid slice from sGm into a local buffer of length vLen
                        AscendC::LocalTensor<float> seg = bufIn_.AllocTensor<float>();
                        uint64_t baseOff = Offset4D(nn, ic, (uint32_t)ih, (uint32_t)gStart, cs_, h_, w_);
                        LoadVecFromGM(seg, sGm_, baseOff, (uint32_t)vLen);

                        // Create an aligned buffer 'tmp' of length len, zero-initialized, and place seg at offset vStart
                        AscendC::LocalTensor<float> tmp = bufIn_.AllocTensor<float>();
                        AscendC::Zeros(tmp, len);
                        // Place seg into tmp[vStart : vStart+vLen)
                        // We emulate by accumulating tmp_slice += seg with mask via BinaryRepeatParams stride==1
                        // Here we fallback to a simple approach: copy seg to tmp at offset using DataCopy between local tensors
                        AscendC::DataCopy(tmp[vStart], seg, (uint32_t)vLen);

                        float wval = GetWeight3x3(wE3Gm_, oc, ic, kh, kw, cs_);
                        FmaAcc(acc, tmp, wval, len);

                        bufIn_.FreeTensor(seg);
                        bufIn_.FreeTensor(tmp);
                    }
                }
            }
            VecMaxZero(acc, len);
            // Store to yGm at channel offset ce1_
            uint64_t yoff = Offset4D(nn, ce1_ + oc, hh, wStart, ce1_ + ce3_, h_, w_);
            StoreVecToGM(yGm_, acc, yoff, len);
            bufAcc_.FreeTensor(acc);
        }
    }

    __aicore__ inline float GetBias(AscendC::GlobalTensor<float>& b, uint32_t oc) const {
        // bias layout [C]
        // We fetch via scalar GM pointer cast. For performance, could DataCopy to local; kept simple.
        __gm__ float* ptr = b.GetRawPtr();
        return ptr[oc];
    }

    __aicore__ inline float GetWeight1x1(AscendC::GlobalTensor<float>& w,
                                         uint32_t oc, uint32_t ic, uint32_t C_in) const {
        // layout [C_out, C_in, 1, 1]
        __gm__ float* ptr = w.GetRawPtr();
        uint64_t idx = (uint64_t)oc * C_in + ic;
        return ptr[idx];
    }

    __aicore__ inline float GetWeight3x3(AscendC::GlobalTensor<float>& w,
                                         uint32_t oc, uint32_t ic, uint32_t kh, uint32_t kw,
                                         uint32_t C_in) const {
        // layout [C_out, C_in, 3, 3]
        __gm__ float* ptr = w.GetRawPtr();
        uint64_t idx = ((((uint64_t)oc * C_in) + ic) * 3 + kh) * 3 + kw;
        return ptr[idx];
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 4>  bufIn_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 4> bufAcc_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wSqGm_;
    AscendC::GlobalTensor<float> bSqGm_;
    AscendC::GlobalTensor<float> wE1Gm_;
    AscendC::GlobalTensor<float> bE1Gm_;
    AscendC::GlobalTensor<float> wE3Gm_;
    AscendC::GlobalTensor<float> bE3Gm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> sGm_; // workspace

    uint32_t n_{0}, cin_{0}, h_{0}, w_{0};
    uint32_t cs_{0}, ce1_{0}, ce3_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void squeeze_net_fire_module_custom(
    GM_ADDR x,
    GM_ADDR w_squeeze, GM_ADDR b_squeeze,
    GM_ADDR w_expand1, GM_ADDR b_expand1,
    GM_ADDR w_expand3, GM_ADDR b_expand3,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{

    tp.n = tiling_data.n;

    tp.cE3 = tiling_data.cE3;

    op.Init(x, w_squeeze, b_squeeze, w_expand1, b_expand1, w_expand3, b_expand3, y, workspace, tp);

}

static inline at::Tensor make_output_like_fire(const at::Tensor& x,
                                               const at::Tensor& w_expand1,
                                               const at::Tensor& w_expand3) {
    // x: [N, C_in, H, W]

    return at::empty(out_sizes, x.options());
}

at::Tensor squeeze_net_fire_module_impl_npu(const at::Tensor& x,
