}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithSmallKDimensionCustomTilingData tiling;

    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);

    // Expect A: [M, K], B: [K, N]
    if (shapeA->GetOriginShape().GetDimNum() != 2 || shapeB->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const int64_t M64 = shapeA->GetOriginShape().GetDim(0);
    const int64_t K64 = shapeA->GetOriginShape().GetDim(1);
    const int64_t Kb64 = shapeB->GetOriginShape().GetDim(0);
    const int64_t N64 = shapeB->GetOriginShape().GetDim(1);

    if (K64 != Kb64 || M64 <= 0 || N64 <= 0 || K64 <= 0) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(M64);
    const uint32_t N = static_cast<uint32_t>(N64);
    const uint32_t K = static_cast<uint32_t>(K64);

    // Choose a conservative blockDim to parallelize across M rows.
    uint32_t blockDim = M;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No additional workspace needed.
    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *A_shape = context->GetInputShape(0);
    const gert::Shape *B_shape = context->GetInputShape(1);
    if (A_shape->GetOriginShape().GetDimNum() != 2 || B_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = A_shape->GetOriginShape().GetDim(0);
    const int64_t K = A_shape->GetOriginShape().GetDim(1);
    const int64_t Kb = B_shape->GetOriginShape().GetDim(0);
    const int64_t N = B_shape->GetOriginShape().GetDim(1);

    C_shape->SetDimNum(2);
    C_shape->SetDim(0, M);
    C_shape->SetDim(1, N);

kernel_src="""

#include "kernel_operator.h"

class KernelMatmulWithSmallKDimensionCustom {
public:
    __aicore__ inline KernelMatmulWithSmallKDimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K)
    {
        A_gm_ = reinterpret_cast<__gm__ const float*>(A);
        B_gm_ = reinterpret_cast<__gm__ const float*>(B);
        C_gm_ = reinterpret_cast<__gm__ float*>(C);
        M_ = M; N_ = N; K_ = K;
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // 1D block distribution along M dimension (rows)
        for (uint32_t i = blockIdx; i < M_; i += blockNum) {
            __gm__ const float* Arow = A_gm_ + static_cast<uint64_t>(i) * K_;
            __gm__ float* Crow = C_gm_ + static_cast<uint64_t>(i) * N_;

            for (uint32_t j = 0; j < N_; ++j) {
                float acc = 0.0f;
                // Small-K accumulation
                for (uint32_t p = 0; p < K_; ++p) {
                    const float a = Arow[p];
                    const float b = B_gm_[static_cast<uint64_t>(p) * N_ + j];
                    acc += a * b;
                }
                Crow[j] = acc;
            }
        }
    }

private:
    __gm__ const float* A_gm_ {nullptr};
    __gm__ const float* B_gm_ {nullptr};
    __gm__ float* C_gm_ {nullptr};
    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
};

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                                                           GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithSmallKDimensionCustom op;
    op.Init(A, B, C, tiling_data.M, tiling_data.N, tiling_data.K);
    op.Process();
}

    TORCH_CHECK(A.size(1) == B.size(0), "Incompatible shapes for matmul: K mismatch");
