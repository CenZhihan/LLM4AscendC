#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
static inline uint32_t FloorDiv(int64_t a, int64_t b) {
    if (a >= 0) return static_cast<uint32_t>(a / b);
    // Not expected here; kept for completeness.
    int64_t q = a / b;
    int64_t r = a % b;
    if ((r != 0) && ((r > 0) != (b > 0))) --q;
    return static_cast<uint32_t>(q);
}
static inline uint32_t OutDim(uint32_t in, uint32_t k, uint32_t s, uint32_t p, bool ceil_mode) {
    if (s == 0) s = k;
    int64_t numer = static_cast<int64_t>(in) + 2 * static_cast<int64_t>(p) - static_cast<int64_t>(k);
    if (ceil_mode) {
        int64_t v = (numer + static_cast<int64_t>(s) - 1) / static_cast<int64_t>(s);
        int64_t outv = v + 1;
        if (outv < 0) outv = 0;
        return static_cast<uint32_t>(outv);
    } else {
        int64_t v = numer / static_cast<int64_t>(s);
        int64_t outv = v + 1;
        if (outv < 0) outv = 0;
        return static_cast<uint32_t>(outv);
    }
}

    const int32_t *stride_ptr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding_ptr = attrs->GetAttrPointer<int32_t>(2);
    const bool *ceil_mode_ptr = attrs->GetAttrPointer<bool>(3);

    uint32_t k = static_cast<uint32_t>(*kernel_size);
    uint32_t s = (*stride_ptr <= 0) ? k : static_cast<uint32_t>(*stride_ptr);
    uint32_t p = static_cast<uint32_t>(*padding_ptr);

    uint32_t pD = p, pH = p, pW = p;

    uint32_t Ho = OutDim(Hi, kH, sH, pH, ceil_mode);

    uint32_t numTiles = CeilDiv(totalOut, tileElems);

    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    tiling.set_sH(sH);
    tiling.set_sW(sW);
    tiling.set_pD(pD);
    tiling.set_pH(pH);
    tiling.set_pW(pW);

kernel_src="""

#include "kernel_operator.h"

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t od, uint32_t oh, uint32_t ow,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t sD, uint32_t sH, uint32_t sW,
                                uint32_t pD, uint32_t pH, uint32_t pW,
                                uint32_t outElems, uint32_t tileElems)
    {
        n_ = n; c_ = c; d_ = d; h_ = h; w_ = w;
        od_ = od; oh_ = oh; ow_ = ow;
        kD_ = kD; kH_ = kH; kW_ = kW;
        sD_ = sD; sH_ = sH; sW_ = sW;
        pD_ = pD; pH_ = pH; pW_ = pW;
        outElems_ = outElems;
        tileElems_ = tileElems;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * c_ * d_ * h_ * w_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * c_ * od_ * oh_ * ow_);

        // Two single-float buffers: current max and element buffer.
        pipe_.InitBuffer(tmpQ_, 2, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outElems_ == 0 || tileElems_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileElems_;
        uint64_t base = static_cast<uint64_t>(blockIdx) * tileElems_;
        if (base >= outElems_) return;

        uint32_t curLen = static_cast<uint32_t>(outElems_ - base);
        if (curLen > tileElems_) curLen = tileElems_;

        while (true) {
            // Compute the current tile [base, base+curLen)
            TileCompute(base, curLen);

            const uint64_t nextBase = base + static_cast<uint64_t>(strideElems);
            uint32_t nextLen = 0;
            if (nextBase < outElems_) {
                nextLen = static_cast<uint32_t>(outElems_ - nextBase);
                if (nextLen > tileElems_) nextLen = tileElems_;
            }
            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void TileCompute(uint64_t base, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            uint64_t outIndex = base + i;
            // Decode outIndex to (n,c,od,oh,ow)
            uint32_t ow = static_cast<uint32_t>(outIndex % ow_);
            uint64_t t1 = outIndex / ow_;
            uint32_t oh = static_cast<uint32_t>(t1 % oh_);
            uint64_t t2 = t1 / oh_;
            uint32_t od = static_cast<uint32_t>(t2 % od_);
            uint64_t t3 = t2 / od_;
            uint32_t c = static_cast<uint32_t>(t3 % c_);
            uint32_t n = static_cast<uint32_t>(t3 / c_);

            // Compute input window ranges
            int64_t dstart = static_cast<int64_t>(od) * sD_ - static_cast<int64_t>(pD_);
            int64_t hstart = static_cast<int64_t>(oh) * sH_ - static_cast<int64_t>(pH_);
            int64_t wstart = static_cast<int64_t>(ow) * sW_ - static_cast<int64_t>(pW_);

            uint32_t dend = static_cast<uint32_t>(AscendC::Min(static_cast<int64_t>(d_), dstart + static_cast<int64_t>(kD_) > 0 ? dstart + static_cast<int64_t>(kD_) : 0));
            uint32_t hend = static_cast<uint32_t>(AscendC::Min(static_cast<int64_t>(h_), hstart + static_cast<int64_t>(kH_) > 0 ? hstart + static_cast<int64_t>(kH_) : 0));
            uint32_t wend = static_cast<uint32_t>(AscendC::Min(static_cast<int64_t>(w_), wstart + static_cast<int64_t>(kW_) > 0 ? wstart + static_cast<int64_t>(kW_) : 0));

            uint32_t dbeg = static_cast<uint32_t>(dstart > 0 ? dstart : 0);
            uint32_t hbeg = static_cast<uint32_t>(hstart > 0 ? hstart : 0);
            uint32_t wbeg = static_cast<uint32_t>(wstart > 0 ? wstart : 0);

            bool hasAny = false;

            AscendC::LocalTensor<float> curMax = tmpQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> elem = tmpQ_.AllocTensor<float>();

            for (uint32_t dd = dbeg; dd < dend; ++dd) {
                for (uint32_t hh = hbeg; hh < hend; ++hh) {
                    for (uint32_t ww = wbeg; ww < wend; ++ww) {
                        uint64_t inIndex =
                            (((static_cast<uint64_t>(n) * c_ + c) * d_ + dd) * h_ + hh) * w_ + ww;

                        if (!hasAny) {
                            AscendC::DataCopy(curMax, xGm_[inIndex], 1);
                            hasAny = true;
                        } else {
                            AscendC::DataCopy(elem, xGm_[inIndex], 1);
                            AscendC::Max(curMax, curMax, elem, 1);
                        }
                    }
                }
            }

            // If window had no valid elements (shouldn't happen with standard params), write 0
            if (!hasAny) {
                // Fill zero in curMax by copying zero from y (cannot); alternatively, set by multiply zero.
                // Workaround: copy from out GM (same location) after we ensure y is initialized; here we just write zero via DataCopy from a temporary not set.
                // Instead, just skip and rely on default 0 since result tensor is zero-initialized by host.
                // However, to be explicit, we can initialize elem with 0 by a no-op multiply.
                // We'll copy elem (which is undefined) multiplied by 0 into curMax:
                AscendC::Muls(curMax, curMax, 0.0f, 1);
            }

            uint64_t outOffset =
                (((static_cast<uint64_t>(n) * c_ + c) * od_ + od) * oh_ + oh) * ow_ + ow;
            AscendC::DataCopy(yGm_[outOffset], curMax, 1);

            tmpQ_.FreeTensor(elem);
            tmpQ_.FreeTensor(curMax);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, d_{0}, h_{0}, w_{0};
    uint32_t od_{0}, oh_{0}, ow_{0};
    uint32_t kD_{1}, kH_{1}, kW_{1};
    uint32_t sD_{1}, sH_{1}, sW_{1};
    uint32_t pD_{0}, pH_{0}, pW_{0};

    uint32_t outElems_{0};
    uint32_t tileElems_{1};
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.od, tiling_data.oh, tiling_data.ow,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.sD, tiling_data.sH, tiling_data.sW,
            tiling_data.pD, tiling_data.pH, tiling_data.pW,
            tiling_data.outElems, tiling_data.tileElems);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static std::array<int64_t, 3> compute_output_3d(int64_t D, int64_t H, int64_t W,
                                                int64_t k, int64_t s, int64_t p,
                                                bool ceil_mode) {
    auto calc = [&](int64_t in) -> int64_t {
        int64_t numer = in + 2 * p - k;
        if (ceil_mode) {
            int64_t v = (numer + s - 1) / s;
            int64_t outv = v + 1;
            return outv > 0 ? outv : 0;
        } else {
            int64_t v = numer / s;
            int64_t outv = v + 1;
            return outv > 0 ? outv : 0;
        }
    };
    return {calc(D), calc(H), calc(W)};
}

at::Tensor max_pooling_3d_impl_npu(const at::Tensor& self, int64_t kernel_size,
                                   int64_t stride, int64_t padding, bool ceil_mode=false) {
    TORCH_CHECK(self.dim() == 5, "Expected 5D input (N, C, D, H, W)");
    int64_t s = (stride <= 0) ? kernel_size : stride;
    auto sizes = self.sym_sizes();
    int64_t N = sizes[0].expect_int();
    int64_t C = sizes[1].expect_int();
    int64_t D = sizes[2].expect_int();
    int64_t H = sizes[3].expect_int();
    int64_t W = sizes[4].expect_int();
    auto outDHW = compute_output_3d(D, H, W, kernel_size, s, padding, ceil_mode);
    at::Tensor result = at::empty({N, C, outDHW[0], outDHW[1], outDHW[2]}, self.options());
    EXEC_NPU_CMD(aclnnMaxPooling3dCustom, self, static_cast<int>(kernel_size),
                 static_cast<int>(s), static_cast<int>(padding), ceil_mode, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_3d_custom", &max_pooling_3d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_3d_custom", &max_pooling_3d_impl_npu,
          "MaxPool3D (MaxPooling3dCustom on NPU)",
          py::arg("x"), py::arg("kernel_size"), py::arg("stride"),
