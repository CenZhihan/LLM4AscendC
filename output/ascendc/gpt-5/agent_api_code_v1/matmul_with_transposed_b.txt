host_operator_src="""

#include "matmul_with_transposed_b_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
}

namespace optiling {

// UB budget per core in bytes (conservative)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// Align elements for DMA efficiency
static constexpr uint32_t ALIGN_ELEMS = 256;
// Hard caps to keep per-iter latency in check
static constexpr uint32_t KSTEP_CAP = 2048;
static constexpr uint32_t TILEM_CANDIDATE = 8;
static constexpr uint32_t TILEN_CANDIDATE = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedBCustomTilingData td;

    const gert::Shape *a_shape = context->GetInputShape(0); // A: (M, K)
    const gert::Shape *b_shape = context->GetInputShape(1); // B: (N, K)
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(a_shape->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(a_shape->GetDim(1));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetDim(0));
    const uint32_t Kb = static_cast<uint32_t>(b_shape->GetDim(1));
    if (Kb != K) {
        return ge::GRAPH_FAILED;
    }

    // Choose tile sizes
    uint32_t tileM = TILEM_CANDIDATE;
    uint32_t tileN = TILEN_CANDIDATE;

    // Compute maximum kStep fitting UB: aRow(kStep) + bSub(tileN*kStep) + yRow(tileN)
    // Elements budget ~= UB_BUDGET_BYTES / 4
    uint32_t budgetElems = UB_BUDGET_BYTES / 4;
    // Leave headroom for pipeline overhead
    if (budgetElems > 4096) budgetElems -= 4096;

    // kStep <= (budgetElems - tileN) / (tileN + 1)
    uint32_t kStepMax = 0;
    if (budgetElems > tileN) {
        kStepMax = (budgetElems - tileN) / (tileN + 1);
    }
    if (kStepMax == 0) kStepMax = ALIGN_ELEMS;
    if (kStepMax > KSTEP_CAP) kStepMax = KSTEP_CAP;
    uint32_t kStep = kStepMax;
    if (kStep > K) kStep = K;
    kStep = AlignDown(kStep, ALIGN_ELEMS);
    if (kStep == 0) kStep = (K < ALIGN_ELEMS) ? K : ALIGN_ELEMS;

    // BlockDim along M tiles
    uint32_t numMTiles = CeilDiv(M, tileM);
    uint32_t blockDim = numMTiles;
    if (blockDim < 4u) blockDim = 4u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    td.set_M(M);
    td.set_N(N);
    td.set_K(K);
    td.set_tileM(tileM);
    td.set_tileN(tileN);
    td.set_kStep(kStep);
    td.set_blockDim(blockDim);

    td.SaveToBuffer(context->GetRawTilingData()->GetData(),
                    context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(td.GetDataSize());

    // No workspace
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0); // (M, K)
    const gert::Shape *b_shape = context->GetInputShape(1); // (N, K)
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    if (a_shape->GetDim(1) != b_shape->GetDim(1)) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0); // (M, N)
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, a_shape->GetDim(0));
    c_shape->SetDim(1, b_shape->GetDim(0));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulWithTransposedBCustom {
public:
    __aicore__ inline KernelMatmulWithTransposedBCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t kStep)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;
        kStep_ = kStep;

        aGm_.SetGlobalBuffer((__gm__ float*)a, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_ * K_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, M_ * N_);

        // Buffers:
        // aRow: kStep floats
        // bTile: tileN * kStep floats
        // yRow: tileN floats
        pipe_.InitBuffer(aRowQ_, BUFFER_NUM, kStep_ * sizeof(float));
        pipe_.InitBuffer(bTileQ_, BUFFER_NUM, tileN_ * kStep_ * sizeof(float));
        pipe_.InitBuffer(yRowQ_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, BUFFER_NUM, kStep_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over tiles of rows
        uint32_t mStart = blockIdx * tileM_;
        const uint32_t mStride = blockNum * tileM_;
        for (; mStart < M_; mStart += mStride) {
            const uint32_t mTile = AscendC::min(tileM_, M_ - mStart);
            // Iterate over N in columns of tileN_
            for (uint32_t nStart = 0; nStart < N_; nStart += tileN_) {
                const uint32_t nTile = AscendC::min(tileN_, N_ - nStart);

                // Process each row in this tile independently to control UB usage
                for (uint32_t mi = 0; mi < mTile; ++mi) {
                    const uint32_t mRow = mStart + mi;

                    // Allocate and zero yRow
                    AscendC::LocalTensor<float> yRow = yRowQ_.AllocTensor<float>();
                    AscendC::Dup(yRow, 0.0f, nTile);

                    // Walk through K in steps
                    for (uint32_t kBase = 0; kBase < K_; kBase += kStep_) {
                        const uint32_t curK = AscendC::min(kStep_, K_ - kBase);

                        // Load aRow (length curK)
                        AscendC::LocalTensor<float> aRow = aRowQ_.AllocTensor<float>();
                        AscendC::DataCopy(aRow, aGm_[mRow * K_ + kBase], curK);

                        // Load bTile: nTile rows x curK each, contiguous per row in B
                        AscendC::LocalTensor<float> bTile = bTileQ_.AllocTensor<float>();
                        for (uint32_t nj = 0; nj < nTile; ++nj) {
                            const uint32_t nCol = nStart + nj;
                            AscendC::DataCopy(bTile[nj * kStep_], bGm_[nCol * K_ + kBase], curK);
                        }

                        // Compute partial dots: for each column in nTile, accumulate dot(aRow, bTile[nj, :])
                        // Use a temp buffer for elementwise product
                        AscendC::LocalTensor<float> prod = tmpQ_.AllocTensor<float>();
                        for (uint32_t nj = 0; nj < nTile; ++nj) {
                            AscendC::Mul(prod, aRow, bTile[nj * kStep_], curK);
                            float partial = AscendC::ReduceSum(prod, curK);
                            // yRow[nj] += partial
                            AscendC::Add(yRow[nj], yRow[nj], partial);
                        }

                        // Free temporaries of this k-chunk
                        tmpQ_.FreeTensor(prod);
                        bTileQ_.FreeTensor(bTile);
                        aRowQ_.FreeTensor(aRow);
                    } // end kBase

                    // Write back yRow to C
                    AscendC::DataCopy(cGm_[mRow * N_ + nStart], yRow, nTile);
                    yRowQ_.FreeTensor(yRow);
                } // end each mi
            } // end nStart
        } // end mStart
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aRowQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bTileQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileM_{0}, tileN_{0}, kStep_{0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_b_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                                                      GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedBCustom op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.kStep);
    op.Process();
}
