    const uint32_t sW = static_cast<uint32_t>(*stride_w);
    const uint32_t pH = static_cast<uint32_t>(*pad_h);
    const uint32_t pW = static_cast<uint32_t>(*pad_w);
    const uint32_t dH = static_cast<uint32_t>(*dil_h);
    const uint32_t dW = static_cast<uint32_t>(*dil_w);
    const uint32_t G  = static_cast<uint32_t>(*groups);

    // Output sizes (floor division)
    const uint32_t Ho = (H + 2U * pH - dH * (kH - 1U) - 1U) / sH + 1U;

#include "kernel_operator.h"

class KernelConvStandard2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t K, uint32_t kH, uint32_t kW,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t sH, uint32_t sW,
                                uint32_t pH, uint32_t pW,
                                uint32_t dH, uint32_t dW,
                                uint32_t /*groups*/,
                                uint32_t tileH, uint32_t tileW, uint32_t tileK)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        K_ = K; kH_ = kH; kW_ = kW;
        Ho_ = Ho; Wo_ = Wo;
        sH_ = sH; sW_ = sW;
        pH_ = pH; pW_ = pW;
        dH_ = dH; dW_ = dW;
        tH_ = tileH; tW_ = tileW; tK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, K_ * C_ * kH_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * K_ * Ho_ * Wo_);

        // Max UB sizes
        const uint32_t inHNeedMax = (tH_ > 0 ? (tH_ - 1U) * sH_ + 1U : 1U) + (kH_ - 1U) * dH_;
        const uint32_t inWNeedMax = (tW_ > 0 ? (tW_ - 1U) * sW_ + 1U : 1U) + (kW_ - 1U) * dW_;

        inNeedHMax_ = inHNeedMax;
        inNeedWMax_ = inWNeedMax;

        const uint32_t inTileMaxElems  = C_ * inHNeedMax * inWNeedMax;
        const uint32_t wTileMaxElems   = tK_ * C_ * kH_ * kW_;
        const uint32_t outTileMaxElems = tK_ * tH_ * tW_;

        pipe_.InitBuffer(inQueue_, 1, inTileMaxElems * sizeof(float));
        pipe_.InitBuffer(wQueue_,  1, wTileMaxElems * sizeof(float));
        pipe_.InitBuffer(outQueue_,1, outTileMaxElems * sizeof(float));
    }

    __aicore__ inline void Process(uint32_t tilesN, uint32_t tilesH, uint32_t tilesW, uint32_t tilesK)
    {
        const uint32_t totalTiles = tilesN * tilesK * tilesH * tilesW;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t tileId = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (; tileId < totalTiles; tileId += blockNum) {
            // Decode tile indices: [n][kTile][hTile][wTile]
            uint32_t rem = tileId;
            const uint32_t nIdx = rem / (tilesK * tilesH * tilesW);
            rem = rem % (tilesK * tilesH * tilesW);
            const uint32_t kTile = rem / (tilesH * tilesW);
            rem = rem % (tilesH * tilesW);
            const uint32_t hTile = rem / tilesW;
            const uint32_t wTile = rem % tilesW;

            const uint32_t hBase = hTile * tH_;
            const uint32_t wBase = wTile * tW_;
            const uint32_t kBase = kTile * tK_;

            const uint32_t hLen = AscendC::Min(tH_, Ho_ - hBase);
            const uint32_t wLen = AscendC::Min(tW_, Wo_ - wBase);
            const uint32_t kLen = AscendC::Min(tK_, K_  - kBase);

            // Compute required input patch span for this tile
            const uint32_t inStartH = hBase * sH_ >= pH_ ? (hBase * sH_ - pH_) : 0U; // assuming pad=0 usually
            const uint32_t inStartW = wBase * sW_ >= pW_ ? (wBase * sW_ - pW_) : 0U;
            const uint32_t inNeedH  = (hLen > 0 ? (hLen - 1U) * sH_ + 1U : 1U) + (kH_ - 1U) * dH_;
            const uint32_t inNeedW  = (wLen > 0 ? (wLen - 1U) * sW_ + 1U : 1U) + (kW_ - 1U) * dW_;

            // Load input tile [C, inNeedH, inNeedW]
            AscendC::LocalTensor<float> inLocal = inQueue_.AllocTensor<float>();
            // per-channel, row-by-row copy
            for (uint32_t c = 0; c < C_; ++c) {
                for (uint32_t ih = 0; ih < inNeedH; ++ih) {
                    const uint32_t gH = inStartH + ih;
                    const uint32_t gmBase = (((nIdx * C_ + c) * H_) + gH) * W_ + inStartW;
                    const uint32_t locBase = (c * inNeedH + ih) * inNeedW;
                    AscendC::DataCopy(inLocal[locBase], xGm_[gmBase], inNeedW);
                }
            }
            inQueue_.EnQue(inLocal);

            // Load weight tile [kLen, C, kH, kW]
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            for (uint32_t kk = 0; kk < kLen; ++kk) {
                const uint32_t kIdx = kBase + kk;
                const uint32_t gmBase = (kIdx * C_) * (kH_ * kW_);
                const uint32_t locBase = kk * (C_ * kH_ * kW_);
                AscendC::DataCopy(wLocal[locBase], wGm_[gmBase], C_ * kH_ * kW_);
            }
            wQueue_.EnQue(wLocal);

            // Prepare output local buffer [kLen, hLen, wLen]
            AscendC::LocalTensor<float> outLocal = outQueue_.AllocTensor<float>();
            const uint32_t outTileElems = kLen * hLen * wLen;
            // Zero initialize
            for (uint32_t i = 0; i < outTileElems; ++i) {
                outLocal.SetValue(i, 0.0f);
            }

            // Compute convolution
            // Index helpers:
            // inLocal index: ((c * inNeedH + (hi * sH + r*dH)) * inNeedW + (wi * sW + s*dW))
            // wLocal index : (((kk * C + c) * kH + r) * kW + s)
            for (uint32_t kk = 0; kk < kLen; ++kk) {
                for (uint32_t c = 0; c < C_; ++c) {
                    for (uint32_t r = 0; r < kH_; ++r) {
                        const uint32_t rOff = r * dH_;
                        for (uint32_t s = 0; s < kW_; ++s) {
                            const uint32_t sOff = s * dW_;
                            const uint32_t wIdxBase = (((kk * C_ + c) * kH_ + r) * kW_) + s;
                            const float wVal = wLocal.GetValue(wIdxBase);
                            for (uint32_t hi = 0; hi < hLen; ++hi) {
                                const uint32_t inRow = hi * sH_ + rOff;
                                const uint32_t outRowBase = ((kk * hLen) + hi) * wLen;
                                for (uint32_t wi = 0; wi < wLen; ++wi) {
                                    const uint32_t inCol = wi * sW_ + sOff;
                                    const uint32_t inIdx = ((c * inNeedH) + inRow) * inNeedW + inCol;
                                    const float xVal = inLocal.GetValue(inIdx);
                                    const uint32_t outIdx = outRowBase + wi;
                                    const float acc = outLocal.GetValue(outIdx) + xVal * wVal;
                                    outLocal.SetValue(outIdx, acc);
                                }
                            }
                        }
                    }
                }
            }

            // Store output
            for (uint32_t kk = 0; kk < kLen; ++kk) {
                const uint32_t kIdx = kBase + kk;
                for (uint32_t hi = 0; hi < hLen; ++hi) {
                    const uint32_t gmBase = (((nIdx * K_ + kIdx) * Ho_) + (hBase + hi)) * Wo_ + wBase;
                    const uint32_t locBase = ((kk * hLen) + hi) * wLen;
                    AscendC::DataCopy(yGm_[gmBase], outLocal[locBase], wLen);
                }
            }

            outQueue_.FreeTensor(outLocal);
            wQueue_.FreeTensor(wLocal);
            inQueue_.FreeTensor(inLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT,   1> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t K_{0}, kH_{0}, kW_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t sH_{1}, sW_{1}, pH_{0}, pW_{0}, dH_{1}, dW_{1};
    uint32_t tH_{0}, tW_{0}, tK_{0};

    // cached maxima for input tile needs
    uint32_t inNeedHMax_{0}, inNeedWMax_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{

    op.Init(x, weight, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,

            tiling_data.dilH, tiling_data.dilW,
            tiling_data.groups,
            tiling_data.tileH, tiling_data.tileW, tiling_data.tileK);

    op.Process(tiling_data.tilesN, tiling_data.tilesH, tiling_data.tilesW, tiling_data.tilesK);
}
