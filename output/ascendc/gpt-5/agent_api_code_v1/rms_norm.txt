host_operator_src="""

#include "rms_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   denomVec: 1 * tile
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    RmsNormCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const gert::Shape *orig = &shape->GetOriginShape();

    const uint64_t totalLength64 = orig->GetShapeSize();
    const uint32_t nBatches = static_cast<uint32_t>(orig->GetDim(0));
    const uint32_t channels = (orig->GetDimNum() > 1) ? static_cast<uint32_t>(orig->GetDim(1)) : 1u;
    const uint32_t innerLength = static_cast<uint32_t>(totalLength64 / (static_cast<uint64_t>(nBatches) * static_cast<uint64_t>(channels)));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    const float eps = (epsPtr != nullptr) ? *epsPtr : 1e-5f;

    // tileInner from UB budget: bytes = 20 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileInner = maxTile;
    if (innerLength < tileInner) {
        const uint32_t aligned = AlignDown(innerLength, ALIGN_ELEMS);
        tileInner = (aligned > 0) ? aligned : innerLength;
    }

    const uint32_t tilesPerN = (tileInner == 0) ? 1u : CeilDiv(innerLength, tileInner);
    const uint32_t totalTasks = nBatches * tilesPerN;

    uint32_t blockDim = totalTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_nBatches(nBatches);
    tiling.set_channels(channels);
    tiling.set_innerLength(innerLength);
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

OP_ADD(RmsNormCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelRmsNormCustom {
public:
    __aicore__ inline KernelRmsNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t nBatches, uint32_t channels,
                                uint32_t innerLength, uint32_t tileInner,
                                float eps)
    {
        nBatches_ = nBatches;
        channels_ = channels;
        innerLen_ = innerLength;
        tileInner_ = tileInner;
        eps_ = eps;
        invChannels_ = (channels_ > 0) ? (1.0f / static_cast<float>(channels_)) : 0.0f;

        totalLen_ = static_cast<uint64_t>(nBatches_) * static_cast<uint64_t>(channels_) * static_cast<uint64_t>(innerLen_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(totalLen_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(totalLen_));

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(denomQueue_, 1, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileInner_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tilesPerN = (tileInner_ == 0) ? 1u : CeilDiv(innerLen_, tileInner_);
        const uint32_t totalTasks = nBatches_ * tilesPerN;

        for (uint32_t task = blockIdx; task < totalTasks; task += blockNum) {
            const uint32_t n = task / tilesPerN;
            const uint32_t tIdx = task % tilesPerN;
            const uint32_t start = tIdx * tileInner_;
            uint32_t curLen = innerLen_ - start;
            if (curLen > tileInner_) curLen = tileInner_;
            ComputeTile(n, start, curLen);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) const
    {
        return (a + b - 1u) / b;
    }

    __aicore__ inline uint64_t Offset(uint32_t n, uint32_t c, uint32_t innerStart) const
    {
        // Offset in elements: ((n * C + c) * innerLen) + innerStart
        return (static_cast<uint64_t>(n) * static_cast<uint64_t>(channels_) + static_cast<uint64_t>(c)) *
                   static_cast<uint64_t>(innerLen_) +
               static_cast<uint64_t>(innerStart);
    }

    __aicore__ inline void ComputeTile(uint32_t n, uint32_t innerStart, uint32_t len)
    {
        // denomLocal will store running sum of squares, then transformed into sqrt(mean + eps)
        AscendC::LocalTensor<float> denomLocal = denomQueue_.AllocTensor<float>();
        AscendC::Dup(denomLocal, 0.0f, len);

        // First pass: accumulate sum of squares across channels
        for (uint32_t c = 0; c < channels_; ++c) {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[Offset(n, c, innerStart)], len);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> xCalc = inQueueX_.DeQue<float>();
            // denomLocal += xCalc * xCalc
            AscendC::LocalTensor<float> tmpLocal = outQueueY_.AllocTensor<float>(); // reuse as temp to save extra buffer
            AscendC::Mul(tmpLocal, xCalc, xCalc, len);
            AscendC::Add(denomLocal, denomLocal, tmpLocal, len);
            outQueueY_.FreeTensor(tmpLocal);
            inQueueX_.FreeTensor(xCalc);
        }

        // denomLocal = sqrt(denomLocal / channels_ + eps_)
        AscendC::Muls(denomLocal, denomLocal, invChannels_, len);
        AscendC::Adds(denomLocal, denomLocal, eps_, len);
        AscendC::Sqrt(denomLocal, denomLocal, len);

        // Second pass: normalize and store
        for (uint32_t c = 0; c < channels_; ++c) {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[Offset(n, c, innerStart)], len);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> xCalc = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Div(yLocal, xCalc, denomLocal, len);
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xCalc);

            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[Offset(n, c, innerStart)], yOut, len);
            outQueueY_.FreeTensor(yOut);
        }

        denomQueue_.FreeTensor(denomLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> denomQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t nBatches_ {0};
    uint32_t channels_ {0};
    uint32_t innerLen_ {0};
    uint32_t tileInner_ {0};
    float eps_ {1e-5f};
    float invChannels_ {0.0f};

    uint64_t totalLen_ {0};
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNormCustom op;
    op.Init(x, y,
            tiling_data.nBatches, tiling_data.channels,
            tiling_data.innerLength, tiling_data.tileInner,
            tiling_data.eps);
    op.Process();
}
"""
