host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
// Double-buffer for in/out: inQueue(2) + outQueue(2) => 4 * tileLen floats in UB
// bytes = 4 * tileLen * sizeof(float) = 16 * tileLen
static constexpr uint32_t BYTES_PER_ELEM = 4;
static constexpr uint32_t ALIGN_ELEMS = 256;   // align for better vectorization
static constexpr uint32_t MAX_BLOCKS = 24;
static constexpr uint32_t MIN_BLOCKS = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape* inShape = context->GetInputShape(0);
    const auto &shapeVec = inShape->GetOriginShape().GetDimSizes();
    int64_t rank = static_cast<int64_t>(shapeVec.size());

    // Get dim attribute (default 1)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttrPtr = attrs->GetAttrPointer<int32_t>(0);
    int64_t dim = static_cast<int64_t>(*dimAttrPtr);
    if (dim < 0 && rank > 0) {
        dim += rank;
    }
    // For this kernel we target 2D tensor and dim == 1.
    // Gracefully clamp for safety.
    if (rank < 2) {
        // Treat as 1D vector softmax
        // N = 1, lastDim = totalLength
    }

    uint64_t totalLength = inShape->GetOriginShape().GetShapeSize();
    uint32_t lastDim = 1;
    uint32_t rows = 1;
    if (rank >= 2 && dim == 1) {
        rows = static_cast<uint32_t>(shapeVec[0]);
        lastDim = static_cast<uint32_t>(shapeVec[1]);
    } else {
        // Fallback: flatten all but axis into "rows", axis length into "lastDim"
        if (rank == 0) {
            lastDim = 1;
            rows = 1;
        } else {
            uint64_t ld = 1, rs = 1;
            for (int64_t i = 0; i < rank; ++i) {
                if (i == dim) {
                    ld = static_cast<uint64_t>(shapeVec[i]);
                } else {
                    rs *= static_cast<uint64_t>(shapeVec[i]);
                }
            }
            lastDim = static_cast<uint32_t>(ld);
            rows = static_cast<uint32_t>(rs);
        }
    }

    // Decide chunk tile length from UB budget:
    uint32_t maxTileElems = UB_BUDGET_BYTES / (4u * BYTES_PER_ELEM); // (inQueue2 + outQueue2) = 4 tiles => 16*tile bytes
    // UB_BUDGET_BYTES / 16 bytes per element
    maxTileElems = UB_BUDGET_BYTES / 16u; // elements
    maxTileElems = AlignDown(maxTileElems, ALIGN_ELEMS);
    if (maxTileElems < ALIGN_ELEMS) maxTileElems = ALIGN_ELEMS;

    uint32_t tileLen = maxTileElems;
    if (lastDim < tileLen) {
        // allow smaller chunk if lastDim smaller
        uint32_t aligned = AlignDown(lastDim, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : lastDim;
    }

    // BlockDim across rows
    uint32_t blockDim = rows;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_lastDim(lastDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t lastDim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        lastDim_ = lastDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || lastDim_ == 0) return;

        const uint32_t rows = totalLen_ / lastDim_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows; row += blockNum) {
            const uint32_t rowStart = row * lastDim_;

            // Pass 1: compute row max
            float rowMax = -3.402823466e+38F; // -FLT_MAX
            for (uint32_t base = 0; base < lastDim_; base += tileLen_) {
                uint32_t cur = lastDim_ - base;
                if (cur > tileLen_) cur = tileLen_;

                AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowStart + base], cur);
                inQueue_.EnQue(xLocal);

                AscendC::LocalTensor<float> xBuf = inQueue_.DeQue<float>();
                // Scalar reduction on local buffer
                for (uint32_t i = 0; i < cur; ++i) {
                    float v = xBuf.GetValue(i);
                    rowMax = (v > rowMax) ? v : rowMax;
                }
                inQueue_.FreeTensor(xBuf);
            }

            // Pass 2: compute row sum of exp(x - rowMax)
            float rowSum = 0.0f;
            for (uint32_t base = 0; base < lastDim_; base += tileLen_) {
                uint32_t cur = lastDim_ - base;
                if (cur > tileLen_) cur = tileLen_;

                AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowStart + base], cur);
                inQueue_.EnQue(xLocal);

                AscendC::LocalTensor<float> xBuf = inQueue_.DeQue<float>();
                AscendC::Adds(xBuf, xBuf, -rowMax, cur);
                AscendC::Exp(xBuf, xBuf, cur);

                // Accumulate sum from local buffer
                for (uint32_t i = 0; i < cur; ++i) {
                    rowSum += xBuf.GetValue(i);
                }
                inQueue_.FreeTensor(xBuf);
            }

            // Prevent division by zero
            float invSum = (rowSum > 0.0f) ? (1.0f / rowSum) : 0.0f;

            // Pass 3: write normalized outputs
            for (uint32_t base = 0; base < lastDim_; base += tileLen_) {
                uint32_t cur = lastDim_ - base;
                if (cur > tileLen_) cur = tileLen_;

                AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowStart + base], cur);
                inQueue_.EnQue(xLocal);

                AscendC::LocalTensor<float> xBuf = inQueue_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

                AscendC::Adds(xBuf, xBuf, -rowMax, cur);
                AscendC::Exp(xBuf, xBuf, cur);
                AscendC::Muls(yLocal, xBuf, invSum, cur);

                outQueue_.EnQue<float>(yLocal);
                inQueue_.FreeTensor(xBuf);

                AscendC::LocalTensor<float> yBuf = outQueue_.DeQue<float>();
                AscendC::DataCopy(yGm_[rowStart + base], yBuf, cur);
                outQueue_.FreeTensor(yBuf);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t lastDim_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.lastDim);
    op.Process();
}
"""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor softmax_impl_npu(const at::Tensor& self, int64_t dim) {
    at::Tensor input = self;
    // Ensure contiguous for kernel simplicity
    if (!input.is_contiguous()) {
        input = input.contiguous();
    }
    at::Tensor result = at::empty_like(input);
    EXEC_NPU_CMD(aclnnSoftmaxCustom, input, static_cast<int>(dim), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("softmax_custom", &softmax_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("softmax_custom", &softmax_impl_npu, "Softmax activation (SoftmaxCustom on NPU)");
}
