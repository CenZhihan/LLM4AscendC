#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

__aicore__ inline void ReduceSumInplace(AscendC::LocalTensor<float>& vec, uint32_t len)
{
    uint32_t cur = len;
    while (cur > 1) {
        // Add upper part onto lower part
        uint32_t half = cur >> 1;
        uint32_t rest = cur - half;
        AscendC::Add<float>(vec, vec, vec[half], rest);
        cur = half;
    }
}

class KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom {
public:
    __aicore__ inline KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t tileK, uint32_t rowsTile)
    {
        B_ = batchSize;
        I_ = inFeatures;
        O_ = outFeatures;
        tileK_ = tileK;
        rowsTile_ = rowsTile;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(B_) * I_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(O_) * I_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, B_);

        // Buffers:
        // vecKQueue_: 4 buffers each of tileK_ floats (wSum, wRow, x, prod)
        pipe_.InitBuffer(vecKQueue_, 4, tileK_ * sizeof(float));
        // accQueue_: 1 buffer of rowsTile_ floats
        pipe_.InitBuffer(accQueue_, 1, rowsTile_ * sizeof(float));
        // scalarQueue_: 1 buffer of 1 float
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (B_ == 0 || I_ == 0) {
            return;
        }

        // Prepare local buffers
        AscendC::LocalTensor<float> wSumLocal = vecKQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> wRowLocal = vecKQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> xLocal    = vecKQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> prodLocal = vecKQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> accLocal  = accQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> bScalar   = scalarQueue_.AllocTensor<float>();

        // Compute bSum once: bScalar[0] = sum(b)
        ComputeBSum(bScalar, prodLocal);

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Process rows in tiles of rowsTile_
        uint32_t baseRow = blockIdx * rowsTile_;
        while (baseRow < B_) {
            const uint32_t rowsCount = (baseRow + rowsTile_ <= B_) ? rowsTile_ : (B_ - baseRow);

            // We'll accumulate into accLocal[0:rowsCount]
            // First K-chunk determines initialization of accLocal
            bool firstChunk = true;

            for (uint32_t k0 = 0; k0 < I_; k0 += tileK_) {
                const uint32_t curK = (k0 + tileK_ <= I_) ? tileK_ : (I_ - k0);

                // Compute w_sum_chunk into wSumLocal[0:curK]
                ComputeWSumChunk(k0, curK, wSumLocal, wRowLocal);

                // For each row in this tile: compute partial dot and accumulate
                for (uint32_t r = 0; r < rowsCount; ++r) {
                    const uint32_t rowIdx = baseRow + r;
                    const uint64_t xOffset = static_cast<uint64_t>(rowIdx) * I_ + k0;
                    AscendC::DataCopy(xLocal, xGm_[xOffset], curK);

                    // prodLocal = xLocal * wSumLocal
                    AscendC::Mul(prodLocal, xLocal, wSumLocal, curK);
                    // Reduce to scalar in prodLocal[0]
                    ReduceSumInplace(prodLocal, curK);

                    if (firstChunk) {
                        // accLocal[r] = prodLocal[0]
                        AscendC::DataCopy(accLocal[r], prodLocal, 1);
                    } else {
                        // accLocal[r] += prodLocal[0]
                        AscendC::Add<float>(accLocal[r], accLocal[r], prodLocal, 1);
                    }
                }

                firstChunk = false;
            }

            // Add bSum to each row accumulator, then write to GM y
            for (uint32_t r = 0; r < rowsCount; ++r) {
                const uint32_t rowIdx = baseRow + r;
                AscendC::Add<float>(accLocal[r], accLocal[r], bScalar, 1);
                AscendC::DataCopy(yGm_[rowIdx], accLocal[r], 1);
            }

            // Next group of rows
            baseRow += blockNum * rowsTile_;
        }

        // Free local tensors
        scalarQueue_.FreeTensor(bScalar);
        accQueue_.FreeTensor(accLocal);
        vecKQueue_.FreeTensor(prodLocal);
        vecKQueue_.FreeTensor(xLocal);
        vecKQueue_.FreeTensor(wRowLocal);
        vecKQueue_.FreeTensor(wSumLocal);
    }

private:
    // Compute bScalar[0] = sum(b[0:O_]) using prodLocal as temp buffer (length at least 1)
    __aicore__ inline void ComputeBSum(AscendC::LocalTensor<float>& bScalar,
                                       AscendC::LocalTensor<float>& tmpLocal)
    {
        bool first = true;
        // Use tile size tileK_ for b chunks to re-use DMA pattern
        for (uint32_t o0 = 0; o0 < O_; o0 += tileK_) {
            const uint32_t cur = (o0 + tileK_ <= O_) ? tileK_ : (O_ - o0);
            // Temporarily use tmpLocal as a vector buffer
            AscendC::DataCopy(tmpLocal, bGm_[o0], cur);
            // Reduce to scalar in tmpLocal[0]
            ReduceSumInplace(tmpLocal, cur);
            if (first) {
                AscendC::DataCopy(bScalar, tmpLocal, 1);
                first = false;
            } else {
                AscendC::Add<float>(bScalar, bScalar, tmpLocal, 1);
            }
        }
    }

    // Compute wSumLocal[0:curK] = sum_{o}(w[o, k0:k0+curK])
    __aicore__ inline void ComputeWSumChunk(uint32_t k0, uint32_t curK,
                                            AscendC::LocalTensor<float>& wSumLocal,
                                            AscendC::LocalTensor<float>& wRowLocal)
    {
        bool first = true;
        for (uint32_t o = 0; o < O_; ++o) {
            const uint64_t wOffset = static_cast<uint64_t>(o) * I_ + k0;
            AscendC::DataCopy(wRowLocal, wGm_[wOffset], curK);
            if (first) {
                AscendC::DataCopy(wSumLocal, wRowLocal, curK);
                first = false;
            } else {
                AscendC::Add<float>(wSumLocal, wSumLocal, wRowLocal, curK);
            }
        }
        // Now wSumLocal holds sum over O for the columns [k0, k0+curK)
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> vecKQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t B_{0};
    uint32_t I_{0};
    uint32_t O_{0};
    uint32_t tileK_{0};
    uint32_t rowsTile_{1};
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom op;
    op.Init(x, w, b, y,
            tiling_data.batchSize,
            tiling_data.inFeatures,
            tiling_data.outFeatures,
            tiling_data.tileK,
            tiling_data.rowsTile);
    op.Process();
}
