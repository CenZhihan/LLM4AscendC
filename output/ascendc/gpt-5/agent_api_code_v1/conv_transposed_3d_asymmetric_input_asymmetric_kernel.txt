#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Din, uint32_t Hin, uint32_t Win,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Dout, uint32_t Hout, uint32_t Wout,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t padD, uint32_t padH, uint32_t padW,
                                uint32_t outPadD, uint32_t outPadH, uint32_t outPadW,
                                uint32_t dilD, uint32_t dilH, uint32_t dilW,
                                uint32_t tileW)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Din_ = Din; Hin_ = Hin; Win_ = Win;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        Dout_ = Dout; Hout_ = Hout; Wout_ = Wout;

        strideD_ = strideD; strideH_ = strideH; strideW_ = strideW;
        padD_ = padD; padH_ = padH; padW_ = padW;
        outPadD_ = outPadD; outPadH_ = outPadH; outPadW_ = outPadW;
        dilD_ = dilD; dilH_ = dilH; dilW_ = dilW;

        tileW_ = tileW;
        if (tileW_ == 0) tileW_ = Wout_;

        xTotalLen_ = N_ * Cin_ * Din_ * Hin_ * Win_;
        wTotalLen_ = Cin_ * Cout_ * Kd_ * Kh_ * Kw_;
        yTotalLen_ = N_ * Cout_ * Dout_ * Hout_ * Wout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xTotalLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wTotalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yTotalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalSlices = static_cast<uint64_t>(N_) * Cout_ * Dout_ * Hout_;
        if (totalSlices == 0) return;

        for (uint64_t s = blockIdx; s < totalSlices; s += blockNum) {
            // Decompose slice index into (n, oc, od, oh)
            uint64_t tmp = s;
            uint32_t oh = static_cast<uint32_t>(tmp % Hout_);
            tmp /= Hout_;
            uint32_t od = static_cast<uint32_t>(tmp % Dout_);
            tmp /= Dout_;
            uint32_t oc = static_cast<uint32_t>(tmp % Cout_);
            uint32_t n  = static_cast<uint32_t>(tmp / Cout_);

            ProcessSlice(n, oc, od, oh);
        }
    }

private:
    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t ic, uint32_t id, uint32_t ih, uint32_t iw) const
    {
        return (((static_cast<uint64_t>(n) * Cin_ + ic) * Din_ + id) * Hin_ + ih) * Win_ + iw;
    }
    __aicore__ inline uint64_t WOffset(uint32_t ic, uint32_t oc, uint32_t kz, uint32_t ky, uint32_t kx) const
    {
        return ((((static_cast<uint64_t>(ic) * Cout_ + oc) * Kd_ + kz) * Kh_ + ky) * Kw_ + kx);
    }
    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t oc, uint32_t od, uint32_t oh, uint32_t ow) const
    {
        return ((((static_cast<uint64_t>(n) * Cout_ + oc) * Dout_ + od) * Hout_ + oh) * Wout_ + ow);
    }

    __aicore__ inline void ProcessSlice(uint32_t n, uint32_t oc, uint32_t od, uint32_t oh)
    {
        for (uint32_t owBase = 0; owBase < Wout_; owBase += tileW_) {
            uint32_t curTileW = Wout_ - owBase;
            if (curTileW > tileW_) curTileW = tileW_;

            // yLocal tile init to zeros
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Dup(yLocal, 0.0f, curTileW);

            // Accumulate contributions from all ic, kz, ky, kx
            for (uint32_t ic = 0; ic < Cin_; ++ic) {
                for (uint32_t kz = 0; kz < Kd_; ++kz) {
                    // Compute input depth index considering dilation
                    int32_t id = static_cast<int32_t>(od) - static_cast<int32_t>(kz * dilD_);
                    if (id < 0 || id >= static_cast<int32_t>(Din_)) continue;

                    for (uint32_t ky = 0; ky < Kh_; ++ky) {
                        int32_t ih = static_cast<int32_t>(oh) - static_cast<int32_t>(ky * dilH_);
                        if (ih < 0 || ih >= static_cast<int32_t>(Hin_)) continue;

                        for (uint32_t kx = 0; kx < Kw_; ++kx) {
                            // Read weight scalar
                            AscendC::LocalTensor<float> wLocal = tmpQueue_.AllocTensor<float>();
                            AscendC::DataCopy(wLocal, wGm_[WOffset(ic, oc, kz, ky, kx)], 1);
                            float wScalar = wLocal.GetValue(0);
                            tmpQueue_.FreeTensor(wLocal);

                            // Determine overlapping output range [startOw, endOw) within tile
                            int32_t startOwGlobal = static_cast<int32_t>(kx * dilW_);
                            int32_t endOwGlobal = startOwGlobal + static_cast<int32_t>(Win_);
                            // Intersect with current tile [owBase, owBase + curTileW)
                            int32_t localStart = AscendC::Max(startOwGlobal, static_cast<int32_t>(owBase)) - static_cast<int32_t>(owBase);
                            int32_t localEnd = AscendC::Min(endOwGlobal, static_cast<int32_t>(owBase + curTileW)) - static_cast<int32_t>(owBase);
                            int32_t segLen = localEnd - localStart;
                            if (segLen <= 0) continue;

                            uint32_t lStart = static_cast<uint32_t>(localStart);
                            uint32_t len = static_cast<uint32_t>(segLen);

                            // Input start index along width
                            int32_t iwStart = (static_cast<int32_t>(owBase) + localStart) - static_cast<int32_t>(kx * dilW_);
                            // Copy input segment x[n, ic, id, ih, iwStart : iwStart + len]
                            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[XOffset(n, ic, static_cast<uint32_t>(id), static_cast<uint32_t>(ih), static_cast<uint32_t>(iwStart))], len);

                            // Scale and accumulate into yLocal segment
                            AscendC::Muls(xLocal, xLocal, wScalar, len);
                            AscendC::LocalTensor<float> ySeg = yLocal[lStart];
                            AscendC::Add(ySeg, ySeg, xLocal, len);

                            inQueueX_.FreeTensor(xLocal);
                        }
                    }
                }
            }

            // Write back yLocal tile to GM
            AscendC::DataCopy(yGm_[YOffset(n, oc, od, oh, owBase)], yLocal, curTileW);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Din_{0}, Hin_{0}, Win_{0};
    uint32_t Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};
    uint32_t strideD_{1}, strideH_{1}, strideW_{1};
    uint32_t padD_{0}, padH_{0}, padW_{0};
    uint32_t outPadD_{0}, outPadH_{0}, outPadW_{0};
    uint32_t dilD_{1}, dilH_{1}, dilW_{1};
    uint32_t tileW_{0};

    uint64_t xTotalLen_{0};
    uint64_t wTotalLen_{0};
    uint64_t yTotalLen_{0};
};
