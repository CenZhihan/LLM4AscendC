#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom {
public:
    __aicore__ inline KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                float negativeSlope)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        negativeSlope_ = negativeSlope;
        negCorr_ = negativeSlope_ - 1.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * (uint64_t)K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)N_ * (uint64_t)K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * 1u);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, K_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, K_ * sizeof(float));
        pipe_.InitBuffer(scoresQueue_, BUFFER_NUM, N_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, K_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < M_; row += (blockNum == 0 ? 1u : blockNum)) {
            // Load x[row, :]
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[row * K_], K_);
            inQueueX_.EnQue<float>(xLocal);

            AscendC::LocalTensor<float> scoresLocal = scoresQueue_.AllocTensor<float>();

            // Compute s_j = dot(x_i, w_j) + b_j for j in [0, N)
            AscendC::LocalTensor<float> prodLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
            AscendC::LocalTensor<float> scalarLocal = outQueueY_.AllocTensor<float>(); // reuse scalar buffer

            AscendC::LocalTensor<float> xLocalUse = inQueueX_.DeQue<float>();
            for (uint32_t j = 0; j < N_; ++j) {
                // Load w[j, :]
                AscendC::DataCopy(wLocal, wGm_[j * K_], K_);
                // prodLocal = xLocal * wLocal
                AscendC::Mul(prodLocal, xLocalUse, wLocal, K_);
                // scalarLocal[0] = ReduceSum(prodLocal)
                AscendC::ReduceSum(scalarLocal, prodLocal, K_);
                // Add bias b[j]
                AscendC::Add(scalarLocal, scalarLocal, bGm_[j], 1);
                // Store into scoresLocal[j]
                AscendC::DataCopy(scoresLocal[j], scalarLocal, 1);
            }
            inQueueX_.FreeTensor(xLocalUse);
            inQueueW_.FreeTensor(wLocal);
            tmpQueue_.FreeTensor(prodLocal);
            outQueueY_.FreeTensor(scalarLocal);

            // Compute lse = logsumexp(scoresLocal[0:N]) along N
            AscendC::LocalTensor<float> maxLocal = outQueueY_.AllocTensor<float>();
            AscendC::ReduceMax(maxLocal, scoresLocal, N_);
            // center = scoresLocal - maxLocal
            AscendC::LocalTensor<float> centerLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(centerLocal, scoresLocal, maxLocal, N_);
            // exp(center)
            AscendC::Exp(centerLocal, centerLocal, N_);
            // sum = ReduceSum(exp(center))
            AscendC::LocalTensor<float> sumLocal = outQueueY_.AllocTensor<float>();
            AscendC::ReduceSum(sumLocal, centerLocal, N_);
            // lse = ln(sum) + max
            AscendC::Ln(sumLocal, sumLocal, 1);
            AscendC::Add(sumLocal, sumLocal, maxLocal, 1);

            // Apply two LeakyReLU and two GELU on 1-element vector
            AscendC::LeakyRelu(sumLocal, sumLocal, negativeSlope_, 1);
            AscendC::LeakyRelu(sumLocal, sumLocal, negativeSlope_, 1);
            AscendC::Gelu(sumLocal, sumLocal, 1);
            AscendC::Gelu(sumLocal, sumLocal, 1);

            // Store result y[row, 0]
            AscendC::DataCopy(yGm_[row * 1u], sumLocal, 1);

            outQueueY_.FreeTensor(sumLocal);
            tmpQueue_.FreeTensor(centerLocal);
            outQueueY_.FreeTensor(maxLocal);
            scoresQueue_.FreeTensor(scoresLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scoresQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    float negativeSlope_{0.01f};
    float negCorr_{-0.99f};
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom op;
    op.Init(x, w, b, y, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.negativeSlope);
    op.Process();
}
