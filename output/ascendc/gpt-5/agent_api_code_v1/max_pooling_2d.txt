host_operator_src="""
#include "max_pooling_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t CeilDivU32(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static inline int64_t FloorDivI64(int64_t a, int64_t b)
{
    if (b == 0) return 0;
    int64_t q = a / b;
    int64_t r = a % b;
    if ((r != 0) && ((r > 0) != (b > 0))) --q;
    return q;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling2dCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    if (in_shape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }
    const int64_t N = in_shape->GetOriginShape().GetDim(0);
    const int64_t C = in_shape->GetOriginShape().GetDim(1);
    const int64_t H = in_shape->GetOriginShape().GetDim(2);
    const int64_t W = in_shape->GetOriginShape().GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernel_size = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding = attrs->GetAttrPointer<int32_t>(2);

    const int64_t kH = static_cast<int64_t>(*kernel_size);
    const int64_t kW = static_cast<int64_t>(*kernel_size);
    const int64_t sH = static_cast<int64_t>(*stride);
    const int64_t sW = static_cast<int64_t>(*stride);
    const int64_t pH = static_cast<int64_t>(*padding);
    const int64_t pW = static_cast<int64_t>(*padding);

    // PyTorch-style output shape: floor((H + 2*pad - k)/stride) + 1
    const int64_t outH = static_cast<int64_t>(FloorDivI64((H + 2 * pH - kH), sH) + 1);
    const int64_t outW = static_cast<int64_t>(FloorDivI64((W + 2 * pW - kW), sW) + 1);
    if (outH <= 0 || outW <= 0) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t uN = static_cast<uint32_t>(N);
    const uint32_t uC = static_cast<uint32_t>(C);
    const uint32_t uH = static_cast<uint32_t>(H);
    const uint32_t uW = static_cast<uint32_t>(W);
    const uint32_t uOutH = static_cast<uint32_t>(outH);
    const uint32_t uOutW = static_cast<uint32_t>(outW);
    const uint32_t totalOut = uN * uC * uOutH * uOutW;

    // Choose blockDim conservatively based on total work
    uint32_t blockDim = (totalOut < 24u) ? totalOut : 24u;
    if (blockDim == 0u) blockDim = 1u;
    context->SetBlockDim(blockDim);

    // Each block processes tiles of outputs in a grid-stride loop
    const uint32_t tileOut = CeilDivU32(totalOut, blockDim);

    tiling.set_N(uN);
    tiling.set_C(uC);
    tiling.set_H(uH);
    tiling.set_W(uW);
    tiling.set_outH(uOutH);
    tiling.set_outW(uOutW);
    tiling.set_kH(static_cast<uint32_t>(kH));
    tiling.set_kW(static_cast<uint32_t>(kW));
    tiling.set_sH(static_cast<uint32_t>(sH));
    tiling.set_sW(static_cast<uint32_t>(sW));
    tiling.set_pH(static_cast<uint32_t>(pH));
    tiling.set_pW(static_cast<uint32_t>(pW));
    tiling.set_totalOut(totalOut);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    const int32_t *kernel_size = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding = attrs->GetAttrPointer<int32_t>(2);

    const int64_t N = x_shape->GetOriginShape().GetDim(0);
    const int64_t C = x_shape->GetOriginShape().GetDim(1);
    const int64_t H = x_shape->GetOriginShape().GetDim(2);
    const int64_t W = x_shape->GetOriginShape().GetDim(3);

    const int64_t kH = static_cast<int64_t>(*kernel_size);
    const int64_t kW = static_cast<int64_t>(*kernel_size);
    const int64_t sH = static_cast<int64_t>(*stride);
    const int64_t sW = static_cast<int64_t>(*stride);
    const int64_t pH = static_cast<int64_t>(*padding);
    const int64_t pW = static_cast<int64_t>(*padding);

    const int64_t outH = (H + 2 * pH - kH) / sH + 1;
    const int64_t outW = (W + 2 * pW - kW) / sW + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, outH);
    y_shape->SetDim(3, outW);
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

class KernelMaxPooling2dCustom {
public:
    __aicore__ inline KernelMaxPooling2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t outH, uint32_t outW,
                                uint32_t kH, uint32_t kW,
                                uint32_t sH, uint32_t sW,
                                uint32_t pH, uint32_t pW,
                                uint32_t totalOut, uint32_t tileOut)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        outH_ = outH; outW_ = outW;
        kH_ = kH; kW_ = kW; sH_ = sH; sW_ = sW; pH_ = pH; pW_ = pW;
        totalOut_ = totalOut; tileOut_ = tileOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * C_ * H_ * W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * C_ * outH_ * outW_);

        // Two scalar buffers in UB for reduction
        pipe_.InitBuffer(tmpQueue_, 2, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0u) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t base = blockIdx * tileOut_;
        const uint32_t stride = blockNum * tileOut_;

        while (base < totalOut_) {
            uint32_t remaining = totalOut_ - base;
            uint32_t count = (remaining < tileOut_) ? remaining : tileOut_;
            for (uint32_t i = 0; i < count; ++i) {
                const uint32_t outIndex = base + i;

                // Decode linear outIndex -> (n, c, oh, ow)
                uint32_t t = outIndex;
                const uint32_t ow = t % outW_; t /= outW_;
                const uint32_t oh = t % outH_; t /= outH_;
                const uint32_t c = t % C_;     t /= C_;
                const uint32_t n = t;

                bool inited = false;
                AscendC::LocalTensor<float> m = tmpQueue_.AllocTensor<float>();

                // Iterate over pooling window
                const int32_t hstart = static_cast<int32_t>(oh) * static_cast<int32_t>(sH_) - static_cast<int32_t>(pH_);
                const int32_t wstart = static_cast<int32_t>(ow) * static_cast<int32_t>(sW_) - static_cast<int32_t>(pW_);
                const int32_t hend = hstart + static_cast<int32_t>(kH_);
                const int32_t wend = wstart + static_cast<int32_t>(kW_);

                for (int32_t ih = hstart; ih < hend; ++ih) {
                    if (ih < 0 || ih >= static_cast<int32_t>(H_)) continue;
                    for (int32_t iw = wstart; iw < wend; ++iw) {
                        if (iw < 0 || iw >= static_cast<int32_t>(W_)) continue;

                        AscendC::LocalTensor<float> v = tmpQueue_.AllocTensor<float>();
                        const uint64_t xOff = Offset(n, c, static_cast<uint32_t>(ih), static_cast<uint32_t>(iw));
                        AscendC::DataCopy(v, xGm_[xOff], 1);

                        if (!inited) {
                            AscendC::DataCopy(m, v, 1);
                            inited = true;
                        } else {
                            AscendC::Max(m, m, v, 1);
                        }
                        tmpQueue_.FreeTensor(v);
                    }
                }

                // Write result if initialized
                if (inited) {
                    const uint64_t yOff = outIndex;
                    AscendC::DataCopy(yGm_[yOff], m, 1);
                } else {
                    // Fallback case (unlikely): set 0
                    AscendC::LocalTensor<float> v0 = tmpQueue_.AllocTensor<float>();
                    // v0 = v0 * 0 -> 0
                    AscendC::Muls(v0, v0, 0.0f, 1);
                    AscendC::DataCopy(yGm_[outIndex], v0, 1);
                    tmpQueue_.FreeTensor(v0);
                }
                tmpQueue_.FreeTensor(m);
            }
            base += stride;
        }
    }

private:
    __aicore__ inline uint64_t Offset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        // linear offset into xGm_ (NCHW)
        return (((static_cast<uint64_t>(n) * C_ + c) * H_ + h) * W_ + w);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t kH_{0}, kW_{0}, sH_{0}, sW_{0}, pH_{0}, pW_{0};
    uint32_t totalOut_{0}, tileOut_{0};
};

extern "C" __global__ __aicore__ void max_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling2dCustom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.outH, tiling_data.outW,
            tiling_data.kH, tiling_data.kW,
            tiling_data.sH, tiling_data.sW,
            tiling_data.pH, tiling_data.pW,
            tiling_data.totalOut, tiling_data.tileOut);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor max_pooling2d_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding) {
    at::Tensor result = at::empty({0}, self.options());
    result = at::empty({self.size(0), self.size(1),
                        (self.size(2) + 2 * padding - kernel_size) / stride + 1,
                        (self.size(3) + 2 * padding - kernel_size) / stride + 1},
                        self.options());
    EXEC_NPU_CMD(aclnnMaxPooling2dCustom, self, (int)kernel_size, (int)stride, (int)padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_2d_custom", &max_pooling2d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_2d_custom", &max_pooling2d_impl_npu, "MaxPooling2D (MaxPooling2dCustom on NPU)");
