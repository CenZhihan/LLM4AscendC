#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative UB budget
static constexpr uint32_t ALIGN_ELEMS = 256;             // DMA friendly alignment
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // cap per-iteration to avoid long latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    // Expect (N, C, L)
    const int64_t n64 = inShape->GetOriginShape().GetDim(0);
    const int64_t c64 = inShape->GetOriginShape().GetDim(1);
    const int64_t l64 = inShape->GetOriginShape().GetDim(2);

    const uint32_t N = static_cast<uint32_t>(n64);
    const uint32_t C = static_cast<uint32_t>(c64);
    const uint32_t L = static_cast<uint32_t>(l64);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kPtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *sPtr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *pPtr = attrs->GetAttrPointer<int32_t>(2);
    const bool    *rPtr = attrs->GetAttrPointer<bool>(3);

    const uint32_t K = static_cast<uint32_t>(*kPtr);
    const uint32_t S = static_cast<uint32_t>((*sPtr) > 0 ? *sPtr : *kPtr);
    const uint32_t P = static_cast<uint32_t>((*pPtr) > 0 ? *pPtr : 0u);
    (void)rPtr; // return_indices not supported in this kernel; forward-only

    // ceil_mode = false, follow PyTorch default
    const int64_t numer = static_cast<int64_t>(L) + 2 * static_cast<int64_t>(P) - static_cast<int64_t>(K);
    const uint32_t Lout = (numer >= 0) ? static_cast<uint32_t>(numer / static_cast<int64_t>(S) + 1) : 0u;

    // Determine blockDim based on groups N*C
    uint32_t groups = (N == 0 || C == 0) ? 0u : (N * C);
    uint32_t blockDim = (groups == 0) ? 1u : (groups < 32u ? groups : 32u);
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    // Tile over output positions per (n,c) group
    // UB usage per tile: segLen + 2 * tileOut (floats)
    // segLen = tileOut * S + (K - S)
    uint32_t tileOut = (Lout == 0) ? 0u : HARD_CAP_ELEMS;
    if (tileOut > Lout) tileOut = Lout;
    if (tileOut > HARD_CAP_ELEMS) tileOut = HARD_CAP_ELEMS;
    tileOut = AlignDown(tileOut, ALIGN_ELEMS);
    if (tileOut == 0 && Lout > 0) tileOut = (Lout < ALIGN_ELEMS ? Lout : ALIGN_ELEMS);

    auto FitsUB = [&](uint32_t tOut)->bool {
        if (tOut == 0) return true;
        uint64_t segLen = static_cast<uint64_t>(tOut) * static_cast<uint64_t>(S) + static_cast<uint64_t>(K > S ? (K - S) : 0u);
        uint64_t floats = segLen + 2ull * static_cast<uint64_t>(tOut);
        uint64_t bytes = floats * 4ull;
        return bytes <= UB_BUDGET_BYTES;
    };

    while (tileOut > 0 && !FitsUB(tileOut)) {
        tileOut = AlignDown(tileOut - ALIGN_ELEMS, ALIGN_ELEMS);
    }
    if (tileOut == 0 && Lout > 0) {
        // fallback minimal tileOut that fits
        uint32_t t = 1;
        while (!FitsUB(t) && t < Lout) ++t;
        tileOut = t;
    }
    if (tileOut > Lout) tileOut = Lout;

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_lIn(L);
    tiling.set_lOut(Lout);
    tiling.set_k(K);
    tiling.set_s(S);
    tiling.set_p(P);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUFFERS = 1;
constexpr int32_t OUT_BUFFERS = 2;
constexpr int32_t TMP_BUFFERS = 1;

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t lIn, uint32_t lOut,
                                uint32_t k, uint32_t s, uint32_t p, uint32_t tileOut)
    {
        n_ = n; c_ = c; lIn_ = lIn; lOut_ = lOut;
        k_ = k; s_ = s; p_ = p; tileOut_ = tileOut;
        groups_ = (n_ == 0 || c_ == 0) ? 0u : (n_ * c_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * c_ * lIn_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_ * c_ * lOut_));

        // segment length per tile
        segLen_ = tileOut_ == 0 ? 0u : (tileOut_ * s_ + (k_ > s_ ? (k_ - s_) : 0u));

        pipe_.InitBuffer(inQueue_, IN_BUFFERS, segLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUT_BUFFERS, tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMP_BUFFERS, tileOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (groups_ == 0 || lOut_ == 0 || k_ == 0 || tileOut_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t g = blockIdx; g < groups_; g += blockNum) {
            // Iterate over output positions in tiles
            for (uint32_t outStart = 0; outStart < lOut_; outStart += tileOut_) {
                uint32_t curOut = lOut_ - outStart;
                if (curOut > tileOut_) curOut = tileOut_;

                // Compute padded index range for this tile
                const uint32_t beginPad = outStart * s_;
                const uint32_t endPad   = beginPad + (curOut - 1u) * s_ + (k_ - 1u);
                const uint32_t segLen   = endPad - beginPad + 1u;

                // Prepare local input segment filled with -INF
                AscendC::LocalTensor<float> segLocal = inQueue_.AllocTensor<float>();
                AscendC::Dup(segLocal, -AscendC::INFINITY<float>(), segLen);

                // Intersection with real input [0, lIn_-1] shifted by padding P
                // padded j = i + p_, j in [beginPad, endPad] => i in [beginPad - p_, endPad - p_]
                int64_t iBegin = static_cast<int64_t>(beginPad) - static_cast<int64_t>(p_);
                int64_t iEnd   = static_cast<int64_t>(endPad)   - static_cast<int64_t>(p_);
                if (iBegin < 0) iBegin = 0;
                if (iEnd > static_cast<int64_t>(lIn_ - 1u)) iEnd = static_cast<int64_t>(lIn_ - 1u);

                if (iBegin <= iEnd) {
                    const uint32_t realLen = static_cast<uint32_t>(iEnd - iBegin + 1);
                    const uint32_t dstOff  = static_cast<uint32_t>((iBegin + static_cast<int64_t>(p_)) - static_cast<int64_t>(beginPad));
                    const uint32_t xBase   = g * lIn_;
                    AscendC::DataCopy(segLocal[dstOff], xGm_[xBase + static_cast<uint32_t>(iBegin)], realLen);
                }

                // Compute max pooling for stride == 1 using vector max of shifted slices
                AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
                if (s_ == 1u) {
                    AscendC::DataCopy(yLocal, segLocal, curOut);
                    for (uint32_t offs = 1; offs < k_; ++offs) {
                        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmpLocal, segLocal[offs], curOut);
                        AscendC::Max(yLocal, yLocal, tmpLocal, curOut);
                        tmpQueue_.FreeTensor(tmpLocal);
                    }
                } else {
                    // Basic slow path for general stride: compute per-output by combining slices
                    // Note: For performance-sensitive use, prefer stride == 1.
                    for (uint32_t t = 0; t < curOut; ++t) {
                        // Initialize with first element in window
                        float maxv;
                        AscendC::DataCopy(reinterpret_cast<AscendC::LocalTensor<float>&>(yLocal[t]), segLocal[t * s_], 1);
                        // yLocal[t] holds initial
                        // Iterate remaining k-1 elements
                        for (uint32_t kk = 1; kk < k_; ++kk) {
                            AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                            AscendC::DataCopy(tmpLocal, segLocal[t * s_ + kk], 1);
                            AscendC::Max(reinterpret_cast<AscendC::LocalTensor<float>&>(yLocal[t]),
                                         reinterpret_cast<AscendC::LocalTensor<float>&>(yLocal[t]),
                                         tmpLocal, 1);
                            tmpQueue_.FreeTensor(tmpLocal);
                        }
                    }
                }

                // Write back
                const uint32_t yBase = g * lOut_ + outStart;
                outQueue_.EnQue<float>(yLocal);
                AscendC::LocalTensor<float> yLocalDeq = outQueue_.DeQue<float>();
                AscendC::DataCopy(yGm_[yBase], yLocalDeq, curOut);
                outQueue_.FreeTensor(yLocalDeq);

                inQueue_.FreeTensor(segLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFERS> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFERS> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUFFERS> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, lIn_{0}, lOut_{0};
    uint32_t k_{0}, s_{1}, p_{0}, tileOut_{0};
    uint32_t groups_{0};
    uint32_t segLen_{0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.lIn, tiling_data.lOut,
            tiling_data.k, tiling_data.s, tiling_data.p, tiling_data.tileOut);
    op.Process();
}

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0, return_indices: bool = False):

        self.kernel_size = int(kernel_size)
        self.stride = int(stride) if stride is not None else None
