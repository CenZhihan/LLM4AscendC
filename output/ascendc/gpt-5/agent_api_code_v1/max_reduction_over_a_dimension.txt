host_operator_src="""

#include "max_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
}

namespace optiling {

// UB model for this kernel:
//   accBuffer: 1 * tileRight
//   bufBuffer: 1 * tileRight
// Total floats = 2 * tileRight => bytes = 8 * tileRight
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // limit per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxReductionOverADimensionCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    if (dimPtr == nullptr) {
        return ge::GRAPH_FAILED;
    }
    int64_t dimAttr = static_cast<int64_t>(*dimPtr);

    const gert::Shape *inShape = context->GetInputShape(0);
    // Fetch rank and dims
    int64_t rank = static_cast<int64_t>(inShape->GetStorageShape().GetDimNum());
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }
    int64_t axis = dimAttr;
    if (axis < 0) axis += rank;
    if (axis < 0 || axis >= rank) {
        return ge::GRAPH_FAILED;
    }

    // Compute left, k, right
    uint64_t leftU = 1, kU = 1, rightU = 1;
    for (int64_t i = 0; i < axis; ++i) {
        leftU *= static_cast<uint64_t>(inShape->GetStorageShape().GetDim(i));
    }
    kU = static_cast<uint64_t>(inShape->GetStorageShape().GetDim(axis));
    for (int64_t i = axis + 1; i < rank; ++i) {
        rightU *= static_cast<uint64_t>(inShape->GetStorageShape().GetDim(i));
    }

    if (kU == 0 || rightU == 0 || leftU == 0) {
        // Degenerate case, still set minimal config
        tiling.set_left(static_cast<uint32_t>(leftU));
        tiling.set_k(static_cast<uint32_t>(kU));
        tiling.set_right(static_cast<uint32_t>(rightU));
        tiling.set_tileRight(0);
        tiling.set_blockDim(1);
        context->SetBlockDim(1);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    // tileRight from UB budget: bytes = 8 * tileRight
    uint32_t maxTile = UB_BUDGET_BYTES / 8;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t right = static_cast<uint32_t>(rightU);
    uint32_t tileRight = maxTile;
    if (right < tileRight) {
        const uint32_t aligned = AlignDown(right, ALIGN_ELEMS);
        tileRight = (aligned > 0) ? aligned : right;
    }

    // Adaptive blockDim based on left; 1D grid over 'left' dimension
    uint32_t blockDim = static_cast<uint32_t>(leftU);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_left(static_cast<uint32_t>(leftU));
    tiling.set_k(static_cast<uint32_t>(kU));
    tiling.set_right(static_cast<uint32_t>(rightU));
    tiling.set_tileRight(tileRight);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs == nullptr) {
        return GRAPH_FAILED;
    }
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    if (dimPtr == nullptr) {
        return GRAPH_FAILED;
    }
    int64_t dimAttr = static_cast<int64_t>(*dimPtr);

    int64_t rank = static_cast<int64_t>(x_shape->GetStorageShape().GetDimNum());
    if (rank <= 0) {
        return GRAPH_FAILED;
    }
    int64_t axis = dimAttr;
    if (axis < 0) axis += rank;
    if (axis < 0 || axis >= rank) {
        return GRAPH_FAILED;
    }

    // Build output dims by removing the reduced axis
    std::vector<int64_t> outDims;
    outDims.reserve(static_cast<size_t>(rank - 1));
    for (int64_t i = 0; i < rank; ++i) {
        if (i == axis) continue;
        outDims.push_back(static_cast<int64_t>(x_shape->GetStorageShape().GetDim(i)));
    }

    gert::Shape out;
    out.SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) {
        out.SetDim(i, outDims[i]);
    }
    y_shape->SetStorageShape(out);
    y_shape->SetOriginShape(out);
    return GRAPH_SUCCESS;
}

    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);

}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMaxReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMaxReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t left, uint32_t k, uint32_t right,
                                uint32_t tileRight)
    {
        left_ = left;
        k_ = k;
        right_ = right;
        tileRight_ = tileRight;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(left_) * k_ * right_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(left_) * right_);

        // Two local buffers: acc and buf
        pipe_.InitBuffer(accQueue_, BUFFER_NUM, tileRight_ * sizeof(float));
        pipe_.InitBuffer(bufQueue_, BUFFER_NUM, tileRight_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (left_ == 0 || k_ == 0 || right_ == 0 || tileRight_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t l = blockIdx; l < left_; l += blockNum) {
            uint64_t lBaseIn = static_cast<uint64_t>(l) * k_ * right_;
            uint64_t lBaseOut = static_cast<uint64_t>(l) * right_;

            for (uint32_t r0 = 0; r0 < right_; r0 += tileRight_) {
                uint32_t curR = right_ - r0;
                if (curR > tileRight_) curR = tileRight_;

                // acc = slice at k=0
                AscendC::LocalTensor<float> accLocal = accQueue_.AllocTensor<float>();
                AscendC::DataCopy(accLocal, xGm_[lBaseIn + r0], curR);

                // iterate remaining k slices
                for (uint32_t kk = 1; kk < k_; ++kk) {
                    AscendC::LocalTensor<float> bufLocal = bufQueue_.AllocTensor<float>();
                    AscendC::DataCopy(bufLocal, xGm_[lBaseIn + static_cast<uint64_t>(kk) * right_ + r0], curR);

                    // acc = max(acc, buf)
                    AscendC::Max(accLocal, accLocal, bufLocal, curR);

                    bufQueue_.FreeTensor(bufLocal);
                }

                // write result
                AscendC::DataCopy(yGm_[lBaseOut + r0], accLocal, curR);
                accQueue_.FreeTensor(accLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bufQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t left_ {0};
    uint32_t k_ {0};
    uint32_t right_ {0};
    uint32_t tileRight_ {0};
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReductionOverADimensionCustom op;
    op.Init(x, y, tiling_data.left, tiling_data.k, tiling_data.right, tiling_data.tileRight);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor max_reduction_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    // Normalize dim to positive within rank
    int64_t rank = self.dim();
    int64_t axis = dim;
    if (axis < 0) axis += rank;

    // Output shape is input with 'axis' removed
    std::vector<int64_t> outSizes;
    outSizes.reserve(static_cast<size_t>(rank - 1));
    for (int64_t i = 0; i < rank; ++i) {
        if (i == axis) continue;
        outSizes.push_back(self.size(i));
    }
    at::Tensor result = at::empty(outSizes, self.options());

    EXEC_NPU_CMD(aclnnMaxReductionOverADimensionCustom, self, static_cast<int>(dim), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_reduction_over_a_dimension_custom", &max_reduction_over_a_dimension_impl_npu);
