host_operator_src="""

#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model for this kernel (per-core):
//   xTile:      1 * tile
//   yTile:      1 * tile
//   gammaTile:  1 * tile
//   betaTile:   1 * tile
//   tmpTile:    1 * tile   (VECCALC tmp/workspace)
//   scalarBuf:  a few floats
// Total floats ~ 5 * tile => bytes ~ 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS     = 2048;
static constexpr uint32_t HARD_CAP_ELEMS  = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t normSize =
        static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize()); // gamma defines normalized extent

    if (normSize == 0) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t rows = (totalLength + normSize - 1u) / normSize; // tolerate non-divisible but expect divisible

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilonAttr = attrs->GetAttrPointer<float>(0);
    const float epsilon = (epsilonAttr != nullptr) ? *epsilonAttr : 1e-5f;

    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (normSize < tileLen) {
        const uint32_t aligned = AlignDown(normSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : normSize;
    }

    uint32_t blockDim = rows;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_normSize(normSize);
    tiling.set_rows(rows);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_epsilon(epsilon);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // no extra workspace
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t normSize, uint32_t rows,
                                uint32_t tileLength, float epsilon)
    {
        totalLen_ = totalLength;
        normSize_ = normSize;
        rows_ = rows;
        tileLen_ = tileLength;
        epsilon_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, normSize_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, normSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(gammaQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(betaQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(calcQueue_, 1, tileLen_ * sizeof(float));   // workspace/tmp per tile
        pipe_.InitBuffer(scalarQueue_, 2, sizeof(float));            // single-element buffers
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || normSize_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            const uint32_t rowStart = row * normSize_;
            if (rowStart >= totalLen_) break;

            // Pass 1: compute sum for mean
            float sumRow = 0.0f;
            uint32_t processed = 0;
            while (processed < normSize_) {
                const uint32_t len = ((normSize_ - processed) > tileLen_) ? tileLen_ : (normSize_ - processed);
                CopyInX(rowStart + processed, len);
                LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                LocalTensor<float> scalar = scalarQueue_.AllocTensor<float>();
                LocalTensor<float> workspace = calcQueue_.AllocTensor<float>();

                ReduceSum(scalar, xLocal, workspace, static_cast<int32_t>(len));
                sumRow += scalar.GetValue(0);

                calcQueue_.FreeTensor(workspace);
                scalarQueue_.FreeTensor(scalar);
                inQueueX_.FreeTensor(xLocal);
                processed += len;
            }
            const float mean = sumRow / static_cast<float>(normSize_);

            // Pass 2: compute variance
            float ssRow = 0.0f;
            processed = 0;
            while (processed < normSize_) {
                const uint32_t len = ((normSize_ - processed) > tileLen_) ? tileLen_ : (normSize_ - processed);
                CopyInX(rowStart + processed, len);
                LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                // t = x - mean
                const float negMean = -mean;
                Adds(xLocal, xLocal, negMean, static_cast<int32_t>(len));
                // t2 = t * t
                LocalTensor<float> t2Local = calcQueue_.AllocTensor<float>();
                Mul(t2Local, xLocal, xLocal, static_cast<int32_t>(len));

                LocalTensor<float> scalar = scalarQueue_.AllocTensor<float>();
                LocalTensor<float> workspace = calcQueue_.AllocTensor<float>();
                ReduceSum(scalar, t2Local, workspace, static_cast<int32_t>(len));
                ssRow += scalar.GetValue(0);

                calcQueue_.FreeTensor(workspace);
                scalarQueue_.FreeTensor(scalar);
                calcQueue_.FreeTensor(t2Local);
                inQueueX_.FreeTensor(xLocal);
                processed += len;
            }
            const float var = ssRow / static_cast<float>(normSize_);
            const float denom = 1.0f / sqrtf(var + epsilon_);

            // Pass 3: normalize and write out
            processed = 0;
            while (processed < normSize_) {
                const uint32_t len = ((normSize_ - processed) > tileLen_) ? tileLen_ : (normSize_ - processed);
                // Load chunks
                CopyInX(rowStart + processed, len);
                CopyInGamma(processed, len);
                CopyInBeta(processed, len);

                LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                LocalTensor<float> gLocal = gammaQueue_.DeQue<float>();
                LocalTensor<float> bLocal = betaQueue_.DeQue<float>();
                LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // y = ((x - mean) * denom) * gamma + beta
                const float negMean = -mean;
                Adds(xLocal, xLocal, negMean, static_cast<int32_t>(len));
                Muls(yLocal, xLocal, denom, static_cast<int32_t>(len));
                Mul(yLocal, yLocal, gLocal, static_cast<int32_t>(len));
                Add(yLocal, yLocal, bLocal, static_cast<int32_t>(len));

                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xLocal);
                gammaQueue_.FreeTensor(gLocal);
                betaQueue_.FreeTensor(bLocal);

                CopyOut(rowStart + processed, len);
                processed += len;
            }
        }
    }

private:
    __aicore__ inline void CopyInX(uint32_t gmOffset, uint32_t len)
    {
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        DataCopy(xLocal, xGm_[gmOffset], static_cast<int32_t>(len));
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInGamma(uint32_t gmOffset, uint32_t len)
    {
        LocalTensor<float> gLocal = gammaQueue_.AllocTensor<float>();
        DataCopy(gLocal, gammaGm_[gmOffset], static_cast<int32_t>(len));
        gammaQueue_.EnQue(gLocal);
    }

    __aicore__ inline void CopyInBeta(uint32_t gmOffset, uint32_t len)
    {
        LocalTensor<float> bLocal = betaQueue_.AllocTensor<float>();
        DataCopy(bLocal, betaGm_[gmOffset], static_cast<int32_t>(len));
        betaQueue_.EnQue(bLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        DataCopy(yGm_[gmOffset], yLocal, static_cast<int32_t>(len));
        outQueueY_.FreeTensor(yLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECIN, 1> gammaQueue_;
    TQue<TPosition::VECIN, 1> betaQueue_;
    TQue<TPosition::VECCALC, 1> calcQueue_;
    TQue<TPosition::VECCALC, 2> scalarQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> gammaGm_;
    GlobalTensor<float> betaGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t normSize_ {0};
    uint32_t rows_ {0};
    uint32_t tileLen_ {0};
    float epsilon_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength, tiling_data.normSize, tiling_data.rows,
            tiling_data.tileLength, tiling_data.epsilon);
    op.Process();
}
