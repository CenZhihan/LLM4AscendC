#include "conv_depthwise_2d_square_input_asymmetric_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetOriginShape().GetDimNum() != 4 || w_shape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t C = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const uint32_t kH = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    const uint32_t kW = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(3));

    // Expect asymmetric vertical kernel: kW == 1
    if (kW != 1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t stride = 1;
    const uint32_t padding = 0;
    const uint32_t dilation = 1;
    const uint32_t eff_kH = (kH - 1) * dilation + 1;

    if (H < eff_kH) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Hout = (H + 2 * padding - eff_kH) / stride + 1;
    const uint32_t Wout = (W + 2 * 0 - 1) / 1 + 1; // kW=1, stride=1, padding=0

    // UB usage: we need buffers for xRow, acc, tmp, plus weights (kH)
    // bytes = 3 * tileW * sizeof(float) + kH * sizeof(float)
    uint32_t maxTile = (UB_BUDGET_BYTES > kH * 4u) ? (UB_BUDGET_BYTES - kH * 4u) / (3u * 4u) : 1u;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = ALIGN_ELEMS;

    uint32_t tileW = maxTile;
    if (Wout < tileW) {
        const uint32_t aligned = AlignDown(Wout, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : Wout;
    }

    const uint32_t pairs = N * C;
    uint32_t blockDim = pairs;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim > pairs) blockDim = pairs;

    context->SetBlockDim(blockDim);

    tiling.set_nSize(N);
    tiling.set_cSize(C);
    tiling.set_hSize(H);
    tiling.set_wSize(W);
    tiling.set_kH(kH);
    tiling.set_hOut(Hout);
    tiling.set_wOut(Wout);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 1;
constexpr int32_t TMP_BUF_NUM = 1;
constexpr int32_t W_BUF_NUM = 1;

class KernelConvDepthwise2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t nSize, uint32_t cSize,
                                uint32_t hSize, uint32_t wSize,
                                uint32_t kH, uint32_t hOut, uint32_t wOut,
                                uint32_t tileW)
    {
        nSize_ = nSize;
        cSize_ = cSize;
        hSize_ = hSize;
        wSize_ = wSize;
        kH_ = kH;
        hOut_ = hOut;
        wOut_ = wOut;
        tileW_ = tileW;

        const uint32_t xLen = nSize_ * cSize_ * hSize_ * wSize_;
        const uint32_t yLen = nSize_ * cSize_ * hOut_ * wOut_;
        const uint32_t wLen = cSize_ * kH_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xLen);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yLen);

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMP_BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, W_BUF_NUM, kH_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (nSize_ == 0 || cSize_ == 0 || hOut_ == 0 || wOut_ == 0 || kH_ == 0) {
            return;
        }

        const uint32_t totalPairs = nSize_ * cSize_;
        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        for (uint32_t pair = blockIdx; pair < totalPairs; pair += blockNum) {
            const uint32_t n = pair / cSize_;
            const uint32_t c = pair % cSize_;

            const uint32_t xBase = ((n * cSize_ + c) * hSize_) * wSize_;
            const uint32_t yBase = ((n * cSize_ + c) * hOut_) * wOut_;
            const uint32_t wBase = c * kH_;

            // Load weights for this channel
            LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            DataCopy(wLocal, wGm_[wBase], kH_);
            wQueue_.EnQue(wLocal);
            wLocal = wQueue_.DeQue<float>();

            for (uint32_t ho = 0; ho < hOut_; ++ho) {
                for (uint32_t wb = 0; wb < wOut_; wb += tileW_) {
                    uint32_t curW = wOut_ - wb;
                    if (curW > tileW_) curW = tileW_;

                    LocalTensor<float> acc = outQueueY_.AllocTensor<float>();
                    Dup(acc, 0.0f, curW);

                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        const uint32_t xOff = xBase + (ho + kh) * wSize_ + wb;

                        LocalTensor<float> xRow = inQueueX_.AllocTensor<float>();
                        DataCopy(xRow, xGm_[xOff], curW);
                        inQueueX_.EnQue(xRow);
                        xRow = inQueueX_.DeQue<float>();

                        LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        // Broadcast multiply by scalar weight (per kh)
                        float wval = wLocal.GetValue(kh);
                        Muls(tmp, xRow, wval, curW);
                        Add(acc, acc, tmp, curW);

                        tmpQueue_.FreeTensor(tmp);
                        inQueueX_.FreeTensor(xRow);
                    }

                    const uint32_t yOff = yBase + ho * wOut_ + wb;
                    outQueueY_.EnQue(acc);
                    LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
                    DataCopy(yGm_[yOff], yLocal, curW);
                    outQueueY_.FreeTensor(yLocal);
                }
            }

            wQueue_.FreeTensor(wLocal);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    TQue<TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;
    TQue<TPosition::VECCALC, TMP_BUF_NUM> tmpQueue_;
    TQue<TPosition::VECCALC, W_BUF_NUM> wQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t nSize_{0};
    uint32_t cSize_{0};
    uint32_t hSize_{0};
    uint32_t wSize_{0};
    uint32_t kH_{0};
    uint32_t hOut_{0};
    uint32_t wOut_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.nSize, tiling_data.cSize,
            tiling_data.hSize, tiling_data.wSize,
            tiling_data.kH, tiling_data.hOut, tiling_data.wOut,
            tiling_data.tileW);
    op.Process();
}
