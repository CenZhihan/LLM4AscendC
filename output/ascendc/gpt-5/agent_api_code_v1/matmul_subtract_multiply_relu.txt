#include "matmul_subtract_multiply_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// UB model for post-processing pass on C:
//   in/out queues double-buffering: 2 * tile + 2 * tile
// Total floats ~= 4 * tile => bytes ~= 16 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap per-iter latency

} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulSubtractMultiplyReluCustomTilingData tiling;

    // Shapes
    const gert::Shape *x_shape = context->GetInputShape(0); // [M, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K]
    const gert::Shape *b_shape = context->GetInputShape(2); // [N]

    if (x_shape->GetOriginShape().GetDimNum() != 2 ||
        w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t M64 = static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(0));
    const uint64_t Kx64 = static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(1));
    const uint64_t N64 = static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(0));
    const uint64_t Kw64 = static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(1));

    if (Kx64 != Kw64) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(M64);
    const uint32_t K = static_cast<uint32_t>(Kx64);
    const uint32_t N = static_cast<uint32_t>(N64);

    // attributes
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *subtractValuePtr = attrs->GetAttrPointer<float>(0);
    const float *multiplyValuePtr = attrs->GetAttrPointer<float>(1);
    const float subtractValue = (subtractValuePtr != nullptr) ? *subtractValuePtr : 0.0f;
    const float multiplyValue = (multiplyValuePtr != nullptr) ? *multiplyValuePtr : 1.0f;

    // Fill cubeTilingData minimally for single-batch GEMM
    tiling.mutable_cubeTilingData()->M = M;
    tiling.mutable_cubeTilingData()->N = N;
    tiling.mutable_cubeTilingData()->K = K;
    tiling.mutable_cubeTilingData()->BatchNum = 1;

    // Post-process tiling
    const uint64_t totalLen64 = static_cast<uint64_t>(M) * static_cast<uint64_t>(N);
    const uint32_t totalLength = static_cast<uint32_t>(totalLen64);
    uint32_t maxTile = UB_BUDGET_BYTES / 16;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_subtractValue(subtractValue);
    tiling.set_multiplyValue(multiplyValue);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t POST_BUFFER_NUM = 2;

class KernelMatmulPostFuser {
public:
    __aicore__ inline KernelMatmulPostFuser() {}

    __aicore__ inline void Init(GM_ADDR y, uint32_t totalLength, uint32_t tileLength, float subVal, float mulVal)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        subVal_ = subVal;
        mulVal_ = mulVal;

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueY_, POST_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, POST_BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.DeQue<float>();

        // y = relu((y - subVal) * mulVal)
        AscendC::Adds(yLocal, yLocal, -subVal_, len);
        AscendC::Muls(yLocal, yLocal, mulVal_, len);
        AscendC::Maxs(yLocal, yLocal, 0.0f, len);

        outQueueY_.EnQue<float>(yLocal);
        inQueueY_.FreeTensor(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, POST_BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, POST_BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    float subVal_ {0.0f};
    float mulVal_ {1.0f};
};

extern "C" __global__ __aicore__ void matmul_subtract_multiply_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                                                            GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    // Prepare tiling
    GET_TILING_DATA(tilingData, tiling);

    // Define Matmul types: A = x [M,K] (not transposed), B = w [N,K] (transposed for GEMM), C = y [M,N], Bias = b [N]
    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> batchMatmulKernel;
    AscendC::TPipe pipe;

    // Optional share settings
    tilingData.cubeTilingData.shareMode = 0;
    tilingData.cubeTilingData.shareL1Size = FULL_L1_SIZE;
    tilingData.cubeTilingData.shareL0CSize = FULL_L0C_SIZE;
    tilingData.cubeTilingData.shareUbSize = 0;

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), batchMatmulKernel.matmulObj, &tilingData.cubeTilingData);

    // Init and run GEMM with bias add
    const int batchNum = (tilingData.cubeTilingData.BatchNum > 0) ? tilingData.cubeTilingData.BatchNum : 1;
    batchMatmulKernel.Init(x, w, b, y, workspace, tilingData.cubeTilingData);
    batchMatmulKernel.Process<true>(&pipe, batchNum, batchNum);

    // Post-processing: (C - subVal) * mulVal, then ReLU
    KernelMatmulPostFuser post;
    const uint32_t totalLen = tilingData.totalLength;
    const uint32_t tileLen = tilingData.tileLength;
    post.Init(y, totalLen, tileLen, tilingData.subtractValue, tilingData.multiplyValue);
    post.Process();
}
