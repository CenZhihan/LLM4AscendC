host_operator_src="""
#include "cumprod_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}

} // anonymous

namespace optiling {

// UB model for this kernel (per core):
//   yLocal: 1 * tile
//   tmpLocal: 1 * tile
//   carry: 1 element
// Total floats ~= 2 * tile + 1 => bytes ~= 8 * tile + 4
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t MAX_TILE_ELEMS = (UB_BUDGET_BYTES - 4) / 8; // floor((UB-4)/8) elements for 2 tiles

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumprodCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const int64_t rank = static_cast<int64_t>(in_shape->GetOriginShape().GetDimNum());
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }

    // Read attr dim (may be negative)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int64_t dim = static_cast<int64_t>(*dimPtr);
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    // Compute sizes
    uint64_t totalLength64 = 1;
    for (int64_t i = 0; i < rank; ++i) {
        totalLength64 *= static_cast<uint64_t>(in_shape->GetOriginShape().GetDim(i));
    }
    if (totalLength64 == 0) {
        // Empty tensor
        tiling.set_innerLen(0);
        tiling.set_outerNum(0);
        tiling.set_tileLen(0);
        tiling.set_blockDim(1);
        tiling.set_dim(static_cast<int32_t>(dim));
        context->SetBlockDim(1);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *wk = context->GetWorkspaceSizes(1);
        wk[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    uint32_t innerLen = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(dim));
    // This kernel is optimized for last-axis cumprod. When dim is not last,
    // the access pattern may be non-contiguous and is not supported by this kernel currently.
    // We still compute innerLen as last dimension to match the kernel expectation.
    const uint32_t lastAxisLen = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(rank - 1));
    // If dim != last, we conservatively fallback to treat last axis; user graph should pass last-axis for performance.
    innerLen = lastAxisLen;

    uint64_t outerNum64 = totalLength64 / static_cast<uint64_t>(innerLen);
    uint32_t outerNum = static_cast<uint32_t>(outerNum64);

    uint32_t maxTile = MAX_TILE_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = (innerLen < maxTile) ? AlignDown(innerLen, ALIGN_ELEMS) : maxTile;
    if (tileLen == 0) tileLen = innerLen; // small inner, possibly < ALIGN_ELEMS

    // Choose blockDim based on number of rows (outerNum)
    uint32_t blockDim = (outerNum == 0u) ? 1u : outerNum;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_innerLen(innerLen);
    tiling.set_outerNum(outerNum);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumprodCustom : public OpDef {
public:
    explicit CumprodCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(-1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(CumprodCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

__aicore__ inline uint32_t umin_u32(uint32_t a, uint32_t b) { return (a < b) ? a : b; }

class KernelCumprodCustom {
public:
    __aicore__ inline KernelCumprodCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t innerLen, uint32_t outerNum,
                                uint32_t tileLen)
    {
        innerLen_ = innerLen;
        outerNum_ = outerNum;
        tileLen_ = (tileLen == 0u) ? innerLen_ : tileLen_;

        const uint32_t totalLen = innerLen_ * outerNum_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen);

        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileLen_ * sizeof(float));    // yLocal
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileLen_ * sizeof(float));  // tmpLocal
        pipe_.InitBuffer(carryQueue_, BUFFER_NUM, sizeof(float));           // carry (1 element)
    }

    __aicore__ inline void Process()
    {
        if (innerLen_ == 0 || outerNum_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < outerNum_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void HillisSteelePrefix(AscendC::LocalTensor<float> &yLocal,
                                              AscendC::LocalTensor<float> &tmpLocal,
                                              uint32_t len)
    {
        // yLocal initially contains x segment.
        // After this function, yLocal becomes in-segment inclusive prefix product.
        uint32_t step = 1;
        while (step < len) {
            // tmp[0:step] = 1.0f
            if (step > 0) {
                AscendC::Duplicate(tmpLocal, 1.0f, step);
            }
            // tmp[step:len] = yLocal[0:len-step]
            const uint32_t remain = len - step;
            if (remain > 0) {
                AscendC::DataCopy(tmpLocal[step], yLocal, remain);
            }
            // yLocal = yLocal * tmpLocal
            AscendC::Mul(yLocal, yLocal, tmpLocal, len);
            step <<= 1;
        }
    }

    __aicore__ inline void ExpandScalarToVector(AscendC::LocalTensor<float> &dstVec,
                                                const AscendC::LocalTensor<float> &srcScalar,
                                                uint32_t len)
    {
        // Assumes srcScalar has at least 1 element. Replicate srcScalar[0] into dstVec[0:len].
        AscendC::DataCopy(dstVec, srcScalar, 1);
        uint32_t filled = 1;
        while (filled < len) {
            const uint32_t copyLen = umin_u32(filled, len - filled);
            AscendC::DataCopy(dstVec[filled], dstVec, copyLen);
            filled += copyLen;
        }
    }

    __aicore__ inline void UpdateCarry(AscendC::LocalTensor<float> &carryLocal,
                                       const AscendC::LocalTensor<float> &yLocal,
                                       uint32_t len)
    {
        // carry = last element of yLocal
        AscendC::DataCopy(carryLocal, yLocal[len - 1], 1);
    }

    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        const uint32_t rowBase = rowIdx * innerLen_;

        AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> carryLocal = carryQueue_.AllocTensor<float>();

        // Initialize carry to 1.0f
        AscendC::Duplicate(carryLocal, 1.0f, 1);

        uint32_t processed = 0;
        while (processed < innerLen_) {
            const uint32_t len = umin_u32(tileLen_, innerLen_ - processed);

            // Load segment
            AscendC::DataCopy(yLocal, xGm_[rowBase + processed], len);

            // Compute in-segment prefix product
            HillisSteelePrefix(yLocal, tmpLocal, len);

            // Multiply by previous carry
            ExpandScalarToVector(tmpLocal, carryLocal, len);
            AscendC::Mul(yLocal, yLocal, tmpLocal, len);

            // Store segment result
            AscendC::DataCopy(yGm_[rowBase + processed], yLocal, len);

            // Update carry to last element of this segment
            UpdateCarry(carryLocal, yLocal, len);

            processed += len;
        }

        carryQueue_.FreeTensor(carryLocal);
        tmpQueue_.FreeTensor(tmpLocal);
        yQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> carryQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t innerLen_{0};
    uint32_t outerNum_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void cumprod_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);
    KernelCumprodCustom op;
    op.Init(x, y, t.innerLen, t.outerNum, t.tileLen);
    op.Process();
}
"""
