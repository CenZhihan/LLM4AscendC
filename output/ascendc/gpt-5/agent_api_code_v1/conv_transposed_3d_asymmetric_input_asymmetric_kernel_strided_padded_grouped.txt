using namespace AscendC;

class KernelConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                const optiling::ConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustomTilingData *t)
    {
        totalOutLen_ = t->totalOutLen();
        tileLen_ = t->tileLen();

        N_ = t->N();
        Cin_ = t->Cin();
        Cout_ = t->Cout();

        Din_ = t->Din();
        Hin_ = t->Hin();
        Win_ = t->Win();

        Dout_ = t->Dout();
        Hout_ = t->Hout();
        Wout_ = t->Wout();

        kD_ = t->kD();
        kH_ = t->kH();
        kW_ = t->kW();

        sD_ = t->strideD();
        sH_ = t->strideH();
        sW_ = t->strideW();

        pD_ = t->padD();
        pH_ = t->padH();
        pW_ = t->padW();

        // outPad not directly used in computation here since shape already accounts for it
        oD_ = t->outPadD();
        oH_ = t->outPadH();
        oW_ = t->outPadW();

        dD_ = t->dilD();
        dH_ = t->dilH();
        dW_ = t->dilW();

        groups_ = t->groups();

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Cin_ * Din_ * Hin_ * Win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cin_ * (Cout_ / groups_) * kD_ * kH_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Cout_ * Dout_ * Hout_ * Wout_);

        pipe_.InitBuffer(tmpQueue_, 8, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        uint64_t strideElems = (uint64_t)blockNum * (uint64_t)tileLen_;
        uint64_t base = (uint64_t)blockIdx * (uint64_t)tileLen_;

        for (uint64_t outBase = base; outBase < totalOutLen_; outBase += strideElems) {
            uint32_t curLen = (uint32_t)((outBase + tileLen_ <= totalOutLen_) ? tileLen_ : (totalOutLen_ - outBase));
            for (uint32_t i = 0; i < curLen; ++i) {
                ComputeSingle(outBase + i);
            }
        }
    }

private:
    __aicore__ inline void UnravelOutIndex(uint64_t idx, uint32_t &n, uint32_t &co, uint32_t &od, uint32_t &oh, uint32_t &ow)
    {
        uint64_t t = idx;
        ow = (uint32_t)(t % Wout_); t /= Wout_;
        oh = (uint32_t)(t % Hout_); t /= Hout_;
        od = (uint32_t)(t % Dout_); t /= Dout_;
        co = (uint32_t)(t % Cout_); t /= Cout_;
        n  = (uint32_t)t;
    }

    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t ci, uint32_t id, uint32_t ih, uint32_t iw)
    {
        // (((n*Cin + ci)*D + id)*H + ih)*W + iw
        uint64_t off = iw;
        off += (uint64_t)ih * Win_;
        off += (uint64_t)id * (uint64_t)Hin_ * (uint64_t)Win_;
        off += (uint64_t)ci * (uint64_t)Din_ * (uint64_t)Hin_ * (uint64_t)Win_;
        off += (uint64_t)n * (uint64_t)Cin_ * (uint64_t)Din_ * (uint64_t)Hin_ * (uint64_t)Win_;
        return off;
    }

    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t co, uint32_t od, uint32_t oh, uint32_t ow)
    {
        // (((n*Cout + co)*Dout + od)*Hout + oh)*Wout + ow
        uint64_t off = ow;
        off += (uint64_t)oh * Wout_;
        off += (uint64_t)od * (uint64_t)Hout_ * (uint64_t)Wout_;
        off += (uint64_t)co * (uint64_t)Dout_ * (uint64_t)Hout_ * (uint64_t)Wout_;
        off += (uint64_t)n  * (uint64_t)Cout_ * (uint64_t)Dout_ * (uint64_t)Hout_ * (uint64_t)Wout_;
        return off;
    }

    __aicore__ inline uint64_t OffsetW(uint32_t ci, uint32_t co_g, uint32_t kd, uint32_t kh, uint32_t kw)
    {
        // (((ci*CoutPerGroup + co_g)*kD + kd)*kH + kh)*kW + kw
        const uint32_t CoutPerGroup = Cout_ / groups_;
        uint64_t off = kw;
        off += (uint64_t)kh * kW_;
        off += (uint64_t)kd * (uint64_t)kH_ * (uint64_t)kW_;
        off += (uint64_t)(ci * CoutPerGroup + co_g) * (uint64_t)kD_ * (uint64_t)kH_ * (uint64_t)kW_;
        return off;
    }

    __aicore__ inline void ComputeSingle(uint64_t outIdx)
    {
        uint32_t n, co, od, oh, ow;
        UnravelOutIndex(outIdx, n, co, od, oh, ow);

        const uint32_t CoutPerGroup = Cout_ / groups_;
        const uint32_t CinPerGroup  = Cin_ / groups_;
        const uint32_t g = co / CoutPerGroup;
        const uint32_t co_g = co - g * CoutPerGroup;
        const uint32_t ciStart = g * CinPerGroup;
        const uint32_t ciEnd   = ciStart + CinPerGroup;

        LocalTensor<float> sum = tmpQueue_.AllocTensor<float>();
        Duplicate(sum, 0.0f, 1);

        LocalTensor<float> xVal = tmpQueue_.AllocTensor<float>();
        LocalTensor<float> wVal = tmpQueue_.AllocTensor<float>();
        LocalTensor<float> prod = tmpQueue_.AllocTensor<float>();

        for (uint32_t ci = ciStart; ci < ciEnd; ++ci) {
            for (uint32_t kd = 0; kd < kD_; ++kd) {
                // id must satisfy: id * sD_ - pD_ + kd*dD_ == od
                int64_t idp = (int64_t)od + (int64_t)pD_ - (int64_t)kd * (int64_t)dD_;
                if (idp < 0) continue;
                if ((uint32_t)idp % sD_ != 0) continue;
                uint32_t id = (uint32_t)idp / sD_;
                if (id >= Din_) continue;

                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    int64_t ihp = (int64_t)oh + (int64_t)pH_ - (int64_t)kh * (int64_t)dH_;
                    if (ihp < 0) continue;
                    if ((uint32_t)ihp % sH_ != 0) continue;
                    uint32_t ih = (uint32_t)ihp / sH_;
                    if (ih >= Hin_) continue;

                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        int64_t iwp = (int64_t)ow + (int64_t)pW_ - (int64_t)kw * (int64_t)dW_;
                        if (iwp < 0) continue;
                        if ((uint32_t)iwp % sW_ != 0) continue;
                        uint32_t iw = (uint32_t)iwp / sW_;
                        if (iw >= Win_) continue;

                        uint64_t xOff = OffsetX(n, ci, id, ih, iw);
                        uint64_t wOff = OffsetW(ci, co_g, kd, kh, kw);

                        DataCopy(xVal, xGm_[xOff], 1);
                        DataCopy(wVal, wGm_[wOff], 1);
                        Mul(prod, xVal, wVal, 1);
                        Add(sum, sum, prod, 1);
                    }
                }
            }
        }

        uint64_t yOff = OffsetY(n, co, od, oh, ow);
        DataCopy(yGm_[yOff], sum, 1);

        tmpQueue_.FreeTensor(prod);
        tmpQueue_.FreeTensor(wVal);
        tmpQueue_.FreeTensor(xVal);
        tmpQueue_.FreeTensor(sum);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, 8> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalOutLen_ {0};
    uint32_t tileLen_ {0};

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Din_{0}, Hin_{0}, Win_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};
    uint32_t sD_{0}, sH_{0}, sW_{0};
    uint32_t pD_{0}, pH_{0}, pW_{0};
    uint32_t oD_{0}, oH_{0}, oW_{0};
    uint32_t dD_{0}, dH_{0}, dW_{0};
    uint32_t groups_{1};
};

static inline std::array<int64_t, 3> conv_transpose3d_out_dims(
    int64_t Din, int64_t Hin, int64_t Win,
    int64_t kD, int64_t kH, int64_t kW,
    int64_t sD, int64_t sH, int64_t sW,
    int64_t pD, int64_t pH, int64_t pW,
    int64_t oD, int64_t oH, int64_t oW,
    int64_t dD, int64_t dH, int64_t dW)
{

    int64_t Wout = (Win - 1) * sW - 2 * pW + dW * (kW - 1) + oW + 1;
    return {Dout, Hout, Wout};
}
