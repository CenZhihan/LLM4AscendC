host_operator_src="""

#include "cumsum_reverse_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumReverseCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &os = in_shape->GetOriginShape();
    const int64_t dims = static_cast<int64_t>(os.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    if (dimAttr == nullptr) {
        return ge::GRAPH_FAILED;
    }
    int64_t axis = *dimAttr;
    if (axis < 0) axis += dims;
    if (axis < 0 || axis >= dims) {
        return ge::GRAPH_FAILED;
    }

    uint64_t totalLength = static_cast<uint64_t>(os.GetShapeSize());
    uint64_t rowLen = static_cast<uint64_t>(os.GetDim(axis));

    uint64_t outer = 1;
    for (int64_t i = 0; i < axis; ++i) outer *= static_cast<uint64_t>(os.GetDim(i));
    uint64_t inner = 1;
    for (int64_t i = axis + 1; i < dims; ++i) inner *= static_cast<uint64_t>(os.GetDim(i));

    // This implementation assumes axis is the innermost dimension (inner == 1) for contiguous rows.
    // If not, we still set rows but kernel may perform with reduced efficiency or assert.
    uint64_t rows = outer * inner;

    // tileLen from UB budget: bytes = 4 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 4;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLen < tileLen) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(rowLen), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(rowLen);
    }
    if (tileLen == 0) tileLen = 1;

    // Blocks based on number of rows
    uint32_t blockDim = static_cast<uint32_t>(rows);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_rowLen(static_cast<uint32_t>(rowLen));
    tiling.set_rows(static_cast<uint32_t>(rows));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCumsumReverseCustom {
public:
    __aicore__ inline KernelCumsumReverseCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t rowLen, uint32_t rows)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        rowLen_ = rowLen;
        rows_ = rows;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(calcQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t r = blockIdx; r < rows_; r += blockNum) {
            uint32_t rowBase = r * rowLen_;
            uint32_t remain = rowLen_;
            uint32_t right = rowBase + rowLen_;
            float carry = 0.0f;

            while (remain > 0) {
                uint32_t clen = (remain > tileLen_) ? tileLen_ : remain;
                uint32_t start = right - clen;

                // Copy in
                AscendC::LocalTensor<float> buf = calcQueue_.AllocTensor<float>();
                AscendC::DataCopy(buf, xGm_[start], clen);

                // Compute reverse inclusive scan with carry
                // buf[i] = sum_{j=i..end} x[j], processed right-to-left
                float acc = carry;
                for (int32_t i = static_cast<int32_t>(clen) - 1; i >= 0; --i) {
                    float v = buf.GetValue(i);
                    acc += v;
                    buf.SetValue(i, acc);
                }
                carry = buf.GetValue(0);

                // Copy out
                AscendC::DataCopy(yGm_[start], buf, clen);
                calcQueue_.FreeTensor(buf);

                right = start;
                remain -= clen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> calcQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t rows_ {0};
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverseCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.rowLen,
            tiling_data.rows);
    op.Process();
}
"""
