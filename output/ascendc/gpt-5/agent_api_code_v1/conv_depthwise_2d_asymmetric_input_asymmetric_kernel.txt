static inline int64_t FloorDiv(int64_t a, int64_t b) { return a / b; }

static inline int64_t ConvOutDim(int64_t in, int64_t pad, int64_t dilation, int64_t k, int64_t stride)
{
    // PyTorch conv2d output formula
    return FloorDiv(in + 2 * pad - dilation * (k - 1) - 1, stride) + 1;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

// Kernel computes depthwise conv2d for NCHW with weight shape [C,1,Kh,Kw] and groups=C.
// It tiles over output positions along width for contiguous segments per (n,c,oh).
class KernelConvDepthwise2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t kh, uint32_t kw,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t dil_h, uint32_t dil_w,
                                uint32_t oh, uint32_t ow,
                                uint32_t totalOut, uint32_t tileLen)
    {
        n_ = n; c_ = c; h_ = h; w_in_ = w_in;
        kh_ = kh; kw_ = kw;
        sh_ = stride_h; sw_ = stride_w;
        ph_ = pad_h; pw_ = pad_w;
        dh_ = dil_h; dw_ = dil_w;
        oh_ = oh; ow_ = ow;
        totalOut_ = totalOut;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * c_ * h_ * w_in_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(c_) * kh_ * kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * c_ * oh_ * ow_);

        // UB buffers sized by tileLen_
        pipe_.InitBuffer(inQueueVec_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueVec_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        // buffer for filter vector (kh*kw)
        uint32_t fsize = kh_ * kw_;
        if (fsize < 1) fsize = 1;
        pipe_.InitBuffer(filtQueue_, 1, fsize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;

        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOut_) return;

        uint32_t curLen = totalOut_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        // Process tiles in a grid-stride loop
        while (true) {
            // Compute the tile [base, base+curLen)
            ComputeTile(base, curLen);
            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalOut_) break;
            base = nextBase;
            curLen = totalOut_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t base, uint32_t len)
    {
        uint32_t idx = 0;
        while (idx < len) {
            // Decode first element in this segment
            uint32_t g = base + idx;
            uint32_t ow_idx = g % ow_;
            uint32_t tmp = g / ow_;
            uint32_t oh_idx = tmp % oh_;
            tmp /= oh_;
            uint32_t c_idx = tmp % c_;
            uint32_t n_idx = tmp / c_;

            // Maximum contiguous segment along width (same n,c,oh)
            uint32_t segAvail = ow_ - ow_idx;
            uint32_t segLen = len - idx;
            if (segLen > segAvail) segLen = segAvail;

            // Allocate output local buffer for this segment and zero-initialize by first accumulation
            AscendC::LocalTensor<float> yLocal = outQueueVec_.AllocTensor<float>();
            // We'll compute yLocal = sum_k (xSeg * w_scalar)
            bool firstAcc = true;

            // For each filter tap
            for (uint32_t ky = 0; ky < kh_; ++ky) {
                // Compute input row index
                int32_t ih = static_cast<int32_t>(oh_idx) * static_cast<int32_t>(sh_) - static_cast<int32_t>(ph_) +
                             static_cast<int32_t>(ky) * static_cast<int32_t>(dh_);
                // If this row is entirely outside input vertically, skip (implicitly zero)
                bool rowValid = (ih >= 0) && (ih < static_cast<int32_t>(h_));

                for (uint32_t kx = 0; kx < kw_; ++kx) {
                    // Weight scalar index: [c, 1, ky, kx]
                    uint32_t wOff = c_idx * (kh_ * kw_) + ky * kw_ + kx;

                    // Load weight scalar from GM into a 1-length local tensor and read value
                    AscendC::LocalTensor<float> wLocal = filtQueue_.AllocTensor<float>();
                    // copy 1 value
                    AscendC::DataCopy(wLocal, wGm_[wOff], 1);
                    filtQueue_.EnQue(wLocal);
                    AscendC::LocalTensor<float> wLocalDeq = filtQueue_.DeQue<float>();
                    // Assume we can read element 0 into scalar
                    float wVal = wLocalDeq.GetValue(0);  // API to read single scalar from local tensor
                    filtQueue_.FreeTensor(wLocalDeq);

                    // Prepare x segment for this ky,kx, length segLen
                    AscendC::LocalTensor<float> xSeg = inQueueVec_.AllocTensor<float>();
                    // Initialize xSeg to 0 by clearing using Muls with 0 if needed.
                    // We'll overwrite only valid parts below.
                    AscendC::Muls(xSeg, xSeg, 0.0f, segLen);

                    if (rowValid) {
                        // starting input col for first ow in segment
                        int32_t iw0 = static_cast<int32_t>(ow_idx) * static_cast<int32_t>(sw_) -
                                      static_cast<int32_t>(pw_) +
                                      static_cast<int32_t>(kx) * static_cast<int32_t>(dw_);
                        // We support arbitrary stride_w by gathering contiguous span and then slicing if sw_==1.
                        // Here we copy per-element safely to handle general stride/dilation.
                        for (uint32_t t = 0; t < segLen; ++t) {
                            int32_t iw = iw0 + static_cast<int32_t>(t) * static_cast<int32_t>(sw_);
                            if (iw >= 0 && iw < static_cast<int32_t>(w_in_)) {
                                // Global offset for x[n,c,ih,iw]
                                uint64_t off = (((static_cast<uint64_t>(n_idx) * c_ + c_idx) * h_ + static_cast<uint32_t>(ih)) * w_in_ + static_cast<uint32_t>(iw));
                                // Load single element into a temp tensor of length 1, then place into xSeg[t]
                                AscendC::LocalTensor<float> tmp1 = tmpQueue_.AllocTensor<float>();
                                AscendC::DataCopy(tmp1, xGm_[off], 1);
                                // write into xSeg[t]
                                xSeg.SetValue(t, tmp1.GetValue(0));
                                tmpQueue_.FreeTensor(tmp1);
                            } else {
                                // OOB horizontally -> zero
                                xSeg.SetValue(t, 0.0f);
                            }
                        }
                    }

                    // Accumulate: y += xSeg * wVal
                    if (firstAcc) {
                        // yLocal = xSeg * wVal
                        AscendC::Muls(yLocal, xSeg, wVal, segLen);
                        firstAcc = false;
                    } else {
                        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                        AscendC::Muls(tLocal, xSeg, wVal, segLen);
                        AscendC::Add(yLocal, yLocal, tLocal, segLen);
                        tmpQueue_.FreeTensor(tLocal);
                    }
                    inQueueVec_.FreeTensor(xSeg);
                }
            }

            // Store yLocal to GM: y[n,c,oh, ow_idx : ow_idx+segLen]
            uint64_t yoff = (((static_cast<uint64_t>(n_idx) * c_ + c_idx) * oh_ + oh_idx) * ow_ + ow_idx);
            AscendC::DataCopy(yGm_[yoff], yLocal, segLen);
            outQueueVec_.FreeTensor(yLocal);

            idx += segLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueVec_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueVec_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> filtQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{1}, c_{1}, h_{1}, w_in_{1};
    uint32_t kh_{1}, kw_{1};
    uint32_t sh_{1}, sw_{1}, ph_{0}, pw_{0}, dh_{1}, dw_{1};
    uint32_t oh_{1}, ow_{1};
    uint32_t totalOut_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kh, tiling_data.kw,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.dil_h, tiling_data.dil_w,
            tiling_data.oh, tiling_data.ow,
            tiling_data.totalOut, tiling_data.tileLen);
    op.Process();
}
