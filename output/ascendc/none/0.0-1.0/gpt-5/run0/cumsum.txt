host_tiling_src = r"""#ifndef CUMSUM_CUSTOM_TILING_H_
#define CUMSUM_CUSTOM_TILING_H_

#include <cstdint>
#include <vector>
#include <algorithm>
#include <numeric>

struct CumsumCustomTilingConfig {
    int32_t rows;
    int32_t cols;
    int32_t stride;
    int32_t dim;
};

inline CumsumCustomTilingConfig CumsumCustomTiling(const std::vector<int64_t>& shape, int32_t dim) {
    CumsumCustomTilingConfig cfg{};
    // Normalize dim
    int32_t ndims = static_cast<int32_t>(shape.size());
    int32_t d = dim;
    if (d < 0) {
        d += ndims;
    }
    if (d < 0 || d >= ndims) {
        // Fallback to last dimension
        d = ndims - 1;
    }
    cfg.dim = d;

    // Compute rows and cols for a simple 2D view by flattening all except the target dim
    int64_t rows64 = 1;
    for (int32_t i = 0; i < d; ++i) {
        rows64 *= shape[i];
    }
    int64_t cols64 = shape[d];
    int64_t stride64 = 1;
    for (int32_t i = d + 1; i < ndims; ++i) {
        stride64 *= shape[i];
    }

    cfg.rows = static_cast<int32_t>(rows64);
    cfg.cols = static_cast<int32_t>(cols64);
    cfg.stride = static_cast<int32_t>(stride64);

    return cfg;
}

#endif  // CUMSUM_CUSTOM_TILING_H_
"""

kernel_src = r"""// AscendC kernel for cumulative sum along an arbitrary dimension flattened into rows/cols.
// This is a simplified demonstration kernel; real kernels should use AscendC intrinsics and UB/GM management.

extern "C" __attribute__((visibility("default")))
void cumsum_custom(const void* x_ptr, void* y_ptr,
                   int dim, int rows, int cols, int stride) {
    // Assume dtype is float32 and contiguous layout for simplicity.
    const float* x = reinterpret_cast<const float*>(x_ptr);
    float* y = reinterpret_cast<float*>(y_ptr);

    // The logical view is rows x cols x stride, where stride is the product of trailing dims.
    // We perform a cumsum along 'cols' (target dim), for every row and inner stride.
    // Total elements = rows * cols * stride.
    // For each (row, s) slice of length 'cols', compute prefix sum across cols.

    // Note: This is a host-like reference implementation. Device implementations would use AscendC parallelism.

    int64_t total_rows = static_cast<int64_t>(rows);
    int64_t total_cols = static_cast<int64_t>(cols);
    int64_t inner = static_cast<int64_t>(stride);

    for (int64_t r = 0; r < total_rows; ++r) {
        for (int64_t s = 0; s < inner; ++s) {
            int64_t base = (r * total_cols * inner) + s;
            float acc = 0.0f;
            for (int64_t c = 0; c < total_cols; ++c) {
                int64_t idx = base + (c * inner);
                acc += x[idx];
                y[idx] = acc;
            }
        }
    }
}
"""
