#include "ca6k_00056_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00056CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t channels =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));
    const uint32_t channelSize = totalLength / channels;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);

    // UB budget: 3 tiles (inX, outY, tmp) each of size tileLength * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (channelSize < tileLen) {
        uint32_t aligned = AlignDown(channelSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : channelSize;
    }

    // blockDim = channels (assuming <=24)
    uint32_t blockDim = channels;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_channelSize(channelSize);
    tiling.set_channels(channels);
    tiling.set_tileLength(tileLen);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00056Custom {
public:
    __aicore__ inline KernelCa6k00056Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t channelSize,
                                uint32_t channels, uint32_t tileLength, float eps) {
        totalLen_ = totalLength;
        channelSize_ = channelSize;
        channels_ = channels;
        tileLen_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, channels_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, channels_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(accumQueue_, 1, sizeof(float)); // for sum/mean/var scalars
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= channels_) return;

        const uint32_t c = blockIdx;
        const float w = weightGm_.GetValue(c);
        const float b = biasGm_.GetValue(c);

        // First pass: compute mean and variance for channel c
        float sum = 0.0f;
        float sumSq = 0.0f;
        uint32_t remaining = channelSize_;
        uint32_t offset = c * channelSize_;

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // Compute sigmoid: 1/(1+exp(-x))
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> one = tmpQueue_.AllocTensor<float>(); // will reuse later
            AscendC::Neg(tmp, xLocal, curLen);
            AscendC::Exp(tmp, tmp, curLen);
            AscendC::Adds(tmp, tmp, 1.0f, curLen);
            AscendC::Reciprocal(tmp, tmp, curLen);

            // Temporarily store sigmoid in xLocal buffer (overwrite)
            AscendC::Copy(xLocal, tmp, curLen); // now xLocal holds sigmoid

            // Accumulate sum and sumSq
            AscendC::LocalTensor<float> sumLocal = accumQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> sumSqLocal = accumQueue_.AllocTensor<float>();
            // Compute sum of tile
            AscendC::ReduceAdd(sumLocal, xLocal, curLen);
            float tileSum = sumLocal.GetValue(0);
            // Compute sum of squares: multiply pointwise then reduce
            AscendC::Mul(tmp, xLocal, xLocal, curLen);
            AscendC::ReduceAdd(sumSqLocal, tmp, curLen);
            float tileSumSq = sumSqLocal.GetValue(0);

            sum += tileSum;
            sumSq += tileSumSq;

            accumQueue_.FreeTensor(sumLocal);
            accumQueue_.FreeTensor(sumSqLocal);
            tmpQueue_.FreeTensor(tmp);
            tmpQueue_.FreeTensor(one); // free placeholder
            inQueueX_.FreeTensor(xLocal);

            offset += curLen;
            remaining -= curLen;
        }

        float mean = sum / (float)channelSize_;
        float var = sumSq / (float)channelSize_ - mean * mean;
        float rstd = AscendC::Rsqrt(var + eps_); // 1/sqrt(var+eps)

        // Second pass: compute output
        offset = c * channelSize_;
        remaining = channelSize_;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // Load original x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // Compute sigmoid again
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Neg(tmp, xLocal, curLen);
            AscendC::Exp(tmp, tmp, curLen);
            AscendC::Adds(tmp, tmp, 1.0f, curLen);
            AscendC::Reciprocal(tmp, tmp, curLen);

            // Normalize: (sigmoid - mean) * rstd * w + b
            AscendC::Sub(tmp, tmp, mean, curLen);
            AscendC::Muls(tmp, tmp, rstd, curLen);
            AscendC::Muls(tmp, tmp, w, curLen);
            AscendC::Adds(tmp, tmp, b, curLen);

            // Output
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Copy(yLocal, tmp, curLen);
            outQueueY_.EnQue(yLocal);

            // CopyOut
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);

            inQueueX_.FreeTensor(xLocal);
            tmpQueue_.FreeTensor(tmp);

            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> accumQueue_; // used for scalar accumulation

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t channelSize_ {0};
    uint32_t channels_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_00056_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00056Custom op;
    op.Init(x, weight, bias, y, tiling_data.totalLength, tiling_data.channelSize,
            tiling_data.channels, tiling_data.tileLength, tiling_data.eps);
    op.Process();
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00056_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, input_features, eps):
        super().__init__()
        # Store affine parameters as buffers
        self.weight = torch.nn.Parameter(torch.ones(input_features, device='npu:0'))
        self.bias = torch.nn.Parameter(torch.zeros(input_features, device='npu:0'))
        self.eps = eps

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        # weight and bias need to be reshaped? The op expects 1D tensors per channel
        w = self.weight.view(-1)
        b = self.bias.view(-1)
        return fn(x, w, b, self.eps)
'''
