#include "ca6k_00057_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;
static constexpr uint32_t MAX_BLOCK_DIM = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00057CustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalLength = static_cast<uint32_t>(shape.GetShapeSize());

    // shape is ND: [N, C, D, H, W]
    const uint32_t numBatch = static_cast<uint32_t>(shape.GetDim(0));
    const uint32_t numChannels = static_cast<uint32_t>(shape.GetDim(1));
    const uint32_t spatialSize =
        static_cast<uint32_t>(shape.GetDim(2)) *
        static_cast<uint32_t>(shape.GetDim(3)) *
        static_cast<uint32_t>(shape.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);

    // Each block processes one channel (spatialSize elements) at a time.
    // tileLength = spatialSize (aligned)
    uint32_t tileLen = AlignDown(spatialSize, ALIGN_ELEMS);
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;
    // For small spatial size, ensure tileLen <= spatialSize
    if (tileLen > spatialSize) tileLen = spatialSize;

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_numBatch(numBatch);
    tiling.set_numChannels(numChannels);
    tiling.set_spatialSize(spatialSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00057Custom {
public:
    __aicore__ inline KernelCa6k00057Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t numBatch,
                                uint32_t numChannels, uint32_t spatialSize,
                                uint32_t tileLength, float eps)
    {
        totalLen_ = totalLength;
        numBatch_ = numBatch;
        numChannels_ = numChannels;
        spatialSize_ = spatialSize;
        tileLen_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Only need one buffer for the tile data (spatialSize floats)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes one channel per iteration.
        // A channel occupies spatialSize consecutive elements.
        // We stride by blockNum * spatialSize (since each block takes one channel)
        const uint32_t step = blockNum * spatialSize_;
        uint32_t base = blockIdx * spatialSize_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            // Current channel offset (within its batch and channel)
            // base is the start of one channel's spatial region after relu
            // We need to compute the channel index to get correct global offset for stats?
            // Actually we only need to process the spatial block.
            uint32_t curLen = totalLen_ - base;
            if (curLen > spatialSize_) curLen = spatialSize_;
            // Align to actual tileLen (which is <= spatialSize and aligned)
            uint32_t len = (curLen < tileLen_) ? curLen : tileLen_;

            // Load tile, apply relu, compute sum and sumSq, then normalize
            ProcessOneTile(base, len);

            base += step;
        }
    }

private:
    __aicore__ inline void ProcessOneTile(uint32_t gmOffset, uint32_t len)
    {
        // Allocate local tensor for tile
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);

        // Apply ReLU: x = max(x, 0)
        AscendC::Max(xLocal, xLocal, 0.0f, len);

        // Compute sum and sumSq over the tile
        float sum = 0.0f;
        float sumSq = 0.0f;
        for (uint32_t i = 0; i < len; i++) {
            float val = xLocal.GetValue(i);
            sum += val;
            sumSq += val * val;
        }

        // Compute mean and variance
        float mean = sum / static_cast<float>(spatialSize_);
        float var = sumSq / static_cast<float>(spatialSize_) - mean * mean;
        float invStd = 1.0f / sqrtf(var + eps_);

        // Normalize: y = (x - mean) * invStd
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Adds(yLocal, xLocal, -mean, len);
        AscendC::Muls(yLocal, yLocal, invStd, len);

        // Write back to global
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);

        // Free tensors
        outQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t numBatch_ {0};
    uint32_t numChannels_ {0};
    uint32_t spatialSize_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_00057_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00057Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.numBatch,
            tiling_data.numChannels, tiling_data.spatialSize,
            tiling_data.tileLength, tiling_data.eps);
    op.Process();
}
