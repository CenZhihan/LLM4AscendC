host_operator_src = """
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;              // DMA alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;          // per-tile cap

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const uint32_t batchSize = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t dim = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t totalLength = batchSize * dim;

    // read optional dim attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    // const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0); // not used for ND shape

    // tile length from UB budget: we need 4 buffers: 2*input tile + 1*output tile + 1*temp (exp) + 1*sum accumulator
    // conservative: 4 * tileSize * sizeof(float) <= UB_BUDGET_BYTES
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (dim < tileLen) {
        uint32_t aligned = AlignDown(dim, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : dim;
    }

    // block dimension: distribute batch rows
    uint32_t blockDim = batchSize;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_batchSize(batchSize);
    tiling.set_dim(dim);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src = """
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t batchSize,
                                uint32_t dim, uint32_t tileLength,
                                uint32_t blockDim) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        batchSize_ = batchSize;
        dim_ = dim;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sumQueue_, 1, sizeof(float)); // scalar for sum
        pipe_.InitBuffer(maxQueue_, 1, sizeof(float)); // scalar for max
    }

    __aicore__ inline void Process() {
        if (dim_ == 0 || batchSize_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (batchSize_ + blockDim_ - 1) / blockDim_;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > batchSize_) endRow = batchSize_;

        for (uint32_t row = startRow; row < endRow; ++row) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row) {
        uint32_t rowOffset = row * dim_;

        // ----- Pass 1: find max -----
        float maxVal = -FLT_MAX;
        uint32_t tiles = (dim_ + tileLen_ - 1) / tileLen_;
        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t offset = rowOffset + t * tileLen_;
            uint32_t len = tileLen_;
            if (offset + len > rowOffset + dim_) len = rowOffset + dim_ - offset;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            // compute tile max
            AscendC::LocalTensor<float> maxLocal = maxQueue_.AllocTensor<float>();
            AscendC::ReduceMax(maxLocal, xLocal, len);
            float tileMax = maxLocal.GetValue(0);
            maxQueue_.FreeTensor(maxLocal);
            inQueueX_.FreeTensor(xLocal);
            if (tileMax > maxVal) maxVal = tileMax;
        }

        // ----- Pass 2: compute sum of exp(x - max) -----
        float sumVal = 0.0f;
        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t offset = rowOffset + t * tileLen_;
            uint32_t len = tileLen_;
            if (offset + len > rowOffset + dim_) len = rowOffset + dim_ - offset;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            // exp(x - max)
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(tmp, xLocal, maxVal, len);
            AscendC::Exp(tmp, tmp, len);
            // reduce sum
            AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumLocal, tmp, len);
            sumVal += sumLocal.GetValue(0);
            sumQueue_.FreeTensor(sumLocal);
            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xLocal);
        }

        float logSum = logf(sumVal);
        float negLogSum = -logSum; // for final addition: x - max - logSum = x + ( -max - logSum)

        // ----- Pass 3: write output -----
        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t offset = rowOffset + t * tileLen_;
            uint32_t len = tileLen_;
            if (offset + len > rowOffset + dim_) len = rowOffset + dim_ - offset;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // y = x - max - logSum = x + ( -max - logSum)
            AscendC::Adds(yLocal, xLocal, negLogSum, len);
            // Actually need to subtract max: x - max - logSum = x + negLogSum - max? Wait:
            // x - max - logSum = x + (-max - logSum). So we have negLogSum already includes -logSum, but we still need -max.
            // Let's compute: y = x - max + negLogSum? Simpler: first subtract max, then add negLogSum.
            // We'll do two ops:
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(tmp, xLocal, maxVal, len);
            AscendC::Add(yLocal, tmp, negLogSum, len);
            tmpQueue_.FreeTensor(tmp);
            AscendC::DataCopy(yGm_[offset], yLocal, len);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maxQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t dim_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.batchSize,
            tiling_data.dim, tiling_data.tileLength, tiling_data.blockDim);
    op.Process();
}
"""
