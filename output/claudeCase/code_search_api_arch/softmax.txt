host_operator_src = '''
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SoftmaxCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    const uint32_t totalRows = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t rowSize = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t totalLength = totalRows * rowSize;

    // calculate tileSize from UB budget: we need 4 buffers: input, output, exp, temp each of tileSize floats = 4*tileSize*4=16*tileSize
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float)); // 4 buffers
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileSize = maxTile;
    if (rowSize < tileSize) {
        tileSize = AlignDown(rowSize, ALIGN_ELEMS);
        if (tileSize == 0) tileSize = rowSize;
    }

    // adaptive blockDim based on number of rows
    uint32_t blockDim = std::min(totalRows, 24u);
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    uint32_t blockRows = CeilDiv(totalRows, blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_rowSize(rowSize);
    tiling.set_tileSize(tileSize);
    tiling.set_blockRows(blockRows);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops
'''

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
        GET_TILING_DATA(td, tiling);
        totalRows_ = td.totalRows;
        rowSize_ = td.rowSize;
        tileSize_ = td.tileSize;
        blockRows_ = td.blockRows;
        numTiles_ = (rowSize_ + tileSize_ - 1) / tileSize_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * rowSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * rowSize_);
        pipe_.InitBuffer(inBuf_, tileSize_ * sizeof(float));
        pipe_.InitBuffer(outBuf_, tileSize_ * sizeof(float));
        pipe_.InitBuffer(expBuf_, tileSize_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, tileSize_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowStart = blockIdx * blockRows_;
        uint32_t rowEnd = (rowStart + blockRows_ > totalRows_) ? totalRows_ : (rowStart + blockRows_);
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            ProcessRow(r);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx) {
        uint32_t offset = rowIdx * rowSize_;

        // First pass: compute max over row
        float rowMax = -__FLT_MAX__;
        for (uint32_t t = 0; t < numTiles_; ++t) {
            uint32_t tileOff = t * tileSize_;
            uint32_t curLen = (tileOff + tileSize_ <= rowSize_) ? tileSize_ : (rowSize_ - tileOff);
            AscendC::DataCopy(inBuf_, xGm_[offset + tileOff], curLen);
            float localMax;
            AscendC::ReduceMax(localMax, inBuf_, curLen);
            if (localMax > rowMax) rowMax = localMax;
        }

        // Second pass: compute sum of exp(x - max)
        float rowSum = 0.0f;
        for (uint32_t t = 0; t < numTiles_; ++t) {
            uint32_t tileOff = t * tileSize_;
            uint32_t curLen = (tileOff + tileSize_ <= rowSize_) ? tileSize_ : (rowSize_ - tileOff);
            AscendC::DataCopy(inBuf_, xGm_[offset + tileOff], curLen);
            AscendC::Sub(tmpBuf_, inBuf_, rowMax, curLen);
            AscendC::Exp(expBuf_, tmpBuf_, curLen);
            float tileSum;
            AscendC::ReduceAdd(tileSum, expBuf_, curLen);
            rowSum += tileSum;
        }

        // Third pass: compute output = exp(x - max) / sum
        float invSum = 1.0f / rowSum;
        for (uint32_t t = 0; t < numTiles_; ++t) {
            uint32_t tileOff = t * tileSize_;
            uint32_t curLen = (tileOff + tileSize_ <= rowSize_) ? tileSize_ : (rowSize_ - tileOff);
            AscendC::DataCopy(inBuf_, xGm_[offset + tileOff], curLen);
            AscendC::Sub(tmpBuf_, inBuf_, rowMax, curLen);
            AscendC::Exp(expBuf_, tmpBuf_, curLen);
            AscendC::Mul(outBuf_, expBuf_, invSum, curLen);
            AscendC::DataCopy(yGm_[offset + tileOff], outBuf_, curLen);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::LocalTensor<float> inBuf_;
    AscendC::LocalTensor<float> outBuf_;
    AscendC::LocalTensor<float> expBuf_;
    AscendC::LocalTensor<float> tmpBuf_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalRows_ {0};
    uint32_t rowSize_ {0};
    uint32_t tileSize_ {0};
    uint32_t blockRows_ {0};
    uint32_t numTiles_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}
'''
