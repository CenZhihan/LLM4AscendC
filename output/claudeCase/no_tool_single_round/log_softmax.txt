host_operator_src="""
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model:
//   inQueueX: 1 * tile
//   outQueueY: 1 * tile
//   tmpQueue: 1 * tile
//   tmpQueue2: 1 * tile
// Total floats = 4 * tile => bytes = 16 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 12288;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : 1;

    // For simplicity, assume the last dimension is the one to reduce over.
    // Get number of columns from shape.
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t numCols = static_cast<uint32_t>(shape.GetDim(shape.GetDimNum() - 1));

    // tileLen from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 16;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (numCols < tileLen) {
        uint32_t aligned = AlignDown(numCols, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : numCols;
    }

    uint32_t numRows = totalLength / numCols;
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_numCols(numCols);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"
#include <math.h>

constexpr int32_t BUFFER_NUM = 1;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t numCols)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        numCols_ = numCols;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t numRows = totalLen_ / numCols_;
        uint32_t rowsPerBlock = (numRows + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = (startRow + rowsPerBlock > numRows) ? numRows : startRow + rowsPerBlock;

        for (uint32_t r = startRow; r < endRow; ++r) {
            uint32_t rowOffset = r * numCols_;

            // ---- first pass: find row max ----
            float rowMax = -3.402823e+38f;
            uint32_t tileStart = 0;
            while (tileStart < numCols_) {
                uint32_t curTileLen = (tileLen_ < numCols_ - tileStart) ? tileLen_ : (numCols_ - tileStart);
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curTileLen);
                AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceMax(maxLocal, xLocal, curTileLen);
                float localMax = maxLocal(0);
                if (localMax > rowMax) rowMax = localMax;
                tmpQueue_.FreeTensor(maxLocal);
                inQueueX_.FreeTensor(xLocal);
                tileStart += curTileLen;
            }

            // ---- second pass: compute sum of exp(x - rowMax) ----
            float sumExp = 0.0f;
            tileStart = 0;
            while (tileStart < numCols_) {
                uint32_t curTileLen = (tileLen_ < numCols_ - tileStart) ? tileLen_ : (numCols_ - tileStart);
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curTileLen);
                AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
                AscendC::Sub(expLocal, xLocal, rowMax, curTileLen);
                AscendC::Exp(expLocal, expLocal, curTileLen);
                AscendC::LocalTensor<float> sumLocal = tmpQueue2_.AllocTensor<float>();
                AscendC::ReduceSum(sumLocal, expLocal, curTileLen);
                sumExp += sumLocal(0);
                tmpQueue2_.FreeTensor(sumLocal);
                tmpQueue_.FreeTensor(expLocal);
                inQueueX_.FreeTensor(xLocal);
                tileStart += curTileLen;
            }

            float logSum = rowMax + logf(sumExp);

            // ---- third pass: output y = x - logSum ----
            tileStart = 0;
            while (tileStart < numCols_) {
                uint32_t curTileLen = (tileLen_ < numCols_ - tileStart) ? tileLen_ : (numCols_ - tileStart);
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curTileLen);
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Sub(yLocal, xLocal, logSum, curTileLen);
                AscendC::DataCopy(yGm_[rowOffset + tileStart], yLocal, curTileLen);
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
                tileStart += curTileLen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numCols_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.numCols);
    op.Process();
}
