host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 1 * tile
//   outQueueY: 1 * tile
//   tmpQueue: 1 * tile
// Total floats = 3 * tile => bytes = 12 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const gert::Shape& inputShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(inputShape.GetShapeSize());
    uint32_t batchSize = static_cast<uint32_t>(inputShape.GetDim(0));
    uint32_t featSize = static_cast<uint32_t>(inputShape.GetDim(1));

    // tileLen from UB budget: bytes = 12 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (featSize < tileLen) {
        tileLen = AlignDown(featSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = featSize;
    }

    // blockDim based on batchSize, keep conservative
    uint32_t blockDim = batchSize;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batchSize(batchSize);
    tiling.set_featSize(featSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

            .Format({ge::FORMAT_ND})

            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t featSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        featSize_ = featSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride over rows
        uint32_t rowStart = blockIdx;
        while (rowStart < batchSize_) {
            ProcessRow(rowStart);
            rowStart += blockNum;
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        uint32_t rowOffset = rowIdx * featSize_;

        // ---------- Pass 1: find max over row ----------
        float rowMax = -INFINITY;
        for (uint32_t col = 0; col < featSize_; col += tileLen_) {
            uint32_t curLen = (featSize_ - col < tileLen_) ? (featSize_ - col) : tileLen_;
            // load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + col], curLen);
            inQueueX_.EnQue(xLocal);
            // dequeue and compute local max
            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Max(tmp, xLocal, AscendC::LocalTensor<float>(tmp)); // placeholder, we'll do scalar
            // Actually we need to compute max per element and reduce. Use a scalar loop.
            // Simpler: use a for loop over elements
            for (uint32_t i = 0; i < curLen; i++) {
                float val = xLocal.GetValue(i);
                if (val > rowMax) rowMax = val;
            }
            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xLocal);
        }

        // ---------- Pass 2: compute sum of exp(x - rowMax) ----------
        float rowSum = 0.0f;
        for (uint32_t col = 0; col < featSize_; col += tileLen_) {
            uint32_t curLen = (featSize_ - col < tileLen_) ? (featSize_ - col) : tileLen_;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + col], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            // compute exp(x - rowMax) and accumulate
            for (uint32_t i = 0; i < curLen; i++) {
                float val = xLocal.GetValue(i) - rowMax;
                float e = expf(val);
                rowSum += e;
                // store exp in tmp for later? We'll recompute in norm pass.
            }
            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xLocal);
        }

        // ---------- Pass 3: norm and write output ----------
        for (uint32_t col = 0; col < featSize_; col += tileLen_) {
            uint32_t curLen = (featSize_ - col < tileLen_) ? (featSize_ - col) : tileLen_;
            // load input
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + col], curLen);
            inQueueX_.EnQue(xLocal);
            // allocate output buffer
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // compute exp(x - rowMax) / rowSum
            xLocal = inQueueX_.DeQue<float>();
            for (uint32_t i = 0; i < curLen; i++) {
                float val = (xLocal.GetValue(i) - rowMax);
                float e = expf(val);
                yLocal.SetValue(i, e / rowSum);
            }
            outQueueY_.EnQue(yLocal);
            // copy out
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[rowOffset + col], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t featSize_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.batchSize, tiling_data.featSize);
    op.Process();
}
"""
