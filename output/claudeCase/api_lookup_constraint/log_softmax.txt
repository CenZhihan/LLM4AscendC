host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int *dimAttr = attrs->GetAttrPointer<int>(0);
    uint32_t dim = static_cast<uint32_t>(*dimAttr);
    uint32_t batchSize = totalLength / dim;

    // tileLen from UB budget: bytes = 20 * tileLen (we use 5 tensors: x, y, tmp, tmp2, tmp3)
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (tileLen > dim) {
        uint32_t aligned = AlignDown(dim, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : dim;
    }

    context->SetBlockDim(batchSize);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_dim(dim);
    tiling.set_blockDim(batchSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

kernel_src="""

#include "kernel_operator.h"
#include <cfloat>

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t dim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t batchSize = totalLen_ / dim_;
        uint32_t rowIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        if (rowIdx >= batchSize) return;

        uint32_t rowBase = rowIdx * dim_;
        uint32_t numTiles = (dim_ + tileLen_ - 1) / tileLen_;

        // Pass 1: compute global max for this row
        float globalMax = -FLT_MAX;
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = rowBase + t * tileLen_;
            uint32_t curLen = (dim_ - t * tileLen_ > tileLen_) ? tileLen_ : (dim_ - t * tileLen_);
            CopyIn(offset, curLen);
            float localMax;
            ComputeMax(curLen, localMax);
            if (localMax > globalMax) globalMax = localMax;
        }

        // Pass 2: compute sum of exp(x - max)
        float globalSum = 0.0f;
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = rowBase + t * tileLen_;
            uint32_t curLen = (dim_ - t * tileLen_ > tileLen_) ? tileLen_ : (dim_ - t * tileLen_);
            CopyIn(offset, curLen);
            float localSum;
            ComputeExpSum(curLen, globalMax, localSum);
            globalSum += localSum;
        }

        // Pass 3: compute log_softmax output
        float logSum = logf(globalSum);
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = rowBase + t * tileLen_;
            uint32_t curLen = (dim_ - t * tileLen_ > tileLen_) ? tileLen_ : (dim_ - t * tileLen_);
            CopyIn(offset, curLen);
            ComputeOutput(curLen, globalMax, logSum);
            CopyOut(offset, curLen);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeMax(uint32_t len, float &localMax)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::ReduceMax(xLocal, localMax, len);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeExpSum(uint32_t len, float maxVal, float &localSum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        // tmp = x - maxVal
        AscendC::Subs(tmpLocal, xLocal, maxVal, len);
        // x (reuse) = exp(tmp)
        AscendC::Exp(xLocal, tmpLocal, len);
        AscendC::ReduceSum(xLocal, localSum, len);
        tmpQueue_.FreeTensor(tmpLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeOutput(uint32_t len, float maxVal, float logSum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        // tmp = x - maxVal
        AscendC::Subs(tmpLocal, xLocal, maxVal, len);
        // y = tmp - logSum
        AscendC::Subs(yLocal, tmpLocal, logSum, len);
        outQueueY_.EnQue(yLocal);
        tmpQueue_.FreeTensor(tmpLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t dim_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
"""
