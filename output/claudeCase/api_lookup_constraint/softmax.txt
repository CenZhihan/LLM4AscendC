host_operator_src = '''
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 8192;         // max tile length

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    // assume input is 2D, rowLength = dim[1]
    const uint32_t rowLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));
    const uint32_t numRows = rowLength ? (totalLength / rowLength) : 0;

    // blockDim: one block per row, capped at 24 for stability
    uint32_t blockDim = numRows;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;

    // tileLength from UB budget: we need 3 * tileLength floats (in, out, tmp)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        const uint32_t aligned = AlignDown(rowLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : rowLength;
    }

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops
'''

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -1e38f;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t tileLength, uint32_t blockDim)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        numRows_ = rowLen_ ? (totalLen_ / rowLen_) : 0;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        // Each block processes rows in a grid-stride manner
        uint32_t startRow = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (startRow >= numRows_) return;

        for (uint32_t row = startRow; row < numRows_; row += blockDim_) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t rowBase = row * rowLen_;

        // Pass 1: compute row max
        float rowMax = NEG_INF;
        for (uint32_t offset = 0; offset < rowLen_; offset += tileLen_) {
            uint32_t len = (offset + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - offset);
            uint32_t gmOffset = rowBase + offset;
            CopyIn(gmOffset, len);
            float tileMax = ComputePassMax(len);
            if (tileMax > rowMax) rowMax = tileMax;
        }

        // Pass 2: compute row sum of exp(x - rowMax)
        float rowSum = 0.0f;
        for (uint32_t offset = 0; offset < rowLen_; offset += tileLen_) {
            uint32_t len = (offset + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - offset);
            uint32_t gmOffset = rowBase + offset;
            CopyIn(gmOffset, len);
            float tileSum = ComputePassExpSum(len, rowMax);
            rowSum += tileSum;
        }

        // Pass 3: normalize and write back
        for (uint32_t offset = 0; offset < rowLen_; offset += tileLen_) {
            uint32_t len = (offset + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - offset);
            uint32_t gmOffset = rowBase + offset;
            CopyIn(gmOffset, len);
            ComputePassNormalize(len, rowMax, rowSum);
            CopyOut(gmOffset, len);
        }
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    /// Pass 1: compute local maximum of tile
    __aicore__ inline float ComputePassMax(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        // reduce max using VecReduce
        AscendC::LocalTensor<float> maxLocal = AscendC::TensorUpcast(
            AscendC::TensorType<float>(), AscendC::VecReduceMax<float>(xLocal, len));
        float res = maxLocal.GetValue(0);
        tmpQueue_.FreeTensor(tmp);
        inQueueX_.FreeTensor(xLocal);
        return res;
    }

    /// Pass 2: compute sum of exp(x - rowMax)
    __aicore__ inline float ComputePassExpSum(uint32_t len, float rowMax)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        // tmp = x - rowMax
        AscendC::Adds(tmp, xLocal, -rowMax, len);
        // tmp = exp(tmp)
        AscendC::Exp(tmp, tmp, len);
        // reduce sum
        AscendC::LocalTensor<float> sumLocal = AscendC::TensorUpcast(
            AscendC::TensorType<float>(), AscendC::VecReduceSum<float>(tmp, len));
        float res = sumLocal.GetValue(0);
        tmpQueue_.FreeTensor(tmp);
        inQueueX_.FreeTensor(xLocal);
        return res;
    }

    /// Pass 3: compute y[i] = exp(x[i] - rowMax) / rowSum and write back
    __aicore__ inline void ComputePassNormalize(uint32_t len, float rowMax, float rowSum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        // tmp = x - rowMax
        AscendC::Adds(tmp, xLocal, -rowMax, len);
        // tmp = exp(tmp)
        AscendC::Exp(tmp, tmp, len);
        // y = tmp / rowSum
        AscendC::Muls(yLocal, tmp, 1.0f / rowSum, len);
        tmpQueue_.FreeTensor(tmp);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.rowLength,
            tiling_data.tileLength,
            tiling_data.blockDim);
    op.Process();
}
'''
