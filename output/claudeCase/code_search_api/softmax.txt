host_operator_src = '''
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SoftmaxCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(x_shape->GetOriginShape().GetShapeSize());
    uint32_t rowLength = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));

    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }

    uint32_t lines = totalLength / rowLength;
    uint32_t blockDim = lines;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops
'''

kernel_src = '''
#include "kernel_operator.h"
#include "softmax_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t tileLength) {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t lines = totalLen_ / rowLen_;

        // Each block processes lines in grid-stride fashion
        for (uint32_t lineIdx = blockIdx; lineIdx < lines; lineIdx += blockNum) {
            uint32_t lineBase = lineIdx * rowLen_;
            ProcessLine(lineBase);
        }
    }

private:
    __aicore__ inline void ProcessLine(uint32_t lineBase) {
        const uint32_t lineEnd = lineBase + rowLen_;

        // ---- First pass: compute max and sum ----
        float globalMax = -std::numeric_limits<float>::infinity();
        float globalSum = 0.0f;

        // Prefetch first tile
        uint32_t curBase = lineBase;
        uint32_t curLen = (tileLen_ < rowLen_) ? tileLen_ : rowLen_;
        CopyIn(curBase, curLen);

        while (true) {
            uint32_t nextBase = curBase + tileLen_;
            uint32_t nextLen = 0;
            bool hasNext = (nextBase < lineEnd);
            if (hasNext) {
                nextLen = (nextBase + tileLen_ <= lineEnd) ? tileLen_ : (lineEnd - nextBase);
                CopyIn(nextBase, nextLen);
            }

            // Compute reduce on current tile
            ComputeReduce(curLen, globalMax, globalSum);

            if (!hasNext) break;
            curBase = nextBase;
            curLen = nextLen;
        }

        // Store global values for the second pass
        globalMax_ = globalMax;
        globalSum_ = globalSum;

        // ---- Second pass: normalized output ----
        // Prefetch first tile again
        curBase = lineBase;
        curLen = (tileLen_ < rowLen_) ? tileLen_ : rowLen_;
        CopyIn(curBase, curLen);

        while (true) {
            uint32_t nextBase = curBase + tileLen_;
            uint32_t nextLen = 0;
            bool hasNext = (nextBase < lineEnd);
            if (hasNext) {
                nextLen = (nextBase + tileLen_ <= lineEnd) ? tileLen_ : (lineEnd - nextBase);
                CopyIn(nextBase, nextLen);
            }

            // Compute normalized output for current tile
            ComputeNorm(curLen, curBase);

            if (!hasNext) break;
            curBase = nextBase;
            curLen = nextLen;
        }
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeReduce(uint32_t len, float &globalMax, float &globalSum) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

        // local max
        AscendC::LocalTensor<float> tileMax = AscendC::Scalar<float>();
        AscendC::ReduceMax(xLocal, tileMax, len);
        float tileMaxVal = tileMax.GetValue(0);

        // exp(x - tileMax)
        AscendC::Sub(tmp, xLocal, tileMaxVal, len);
        AscendC::Exp(tmp, tmp, len);

        // local sum
        AscendC::LocalTensor<float> tileSum = AscendC::Scalar<float>();
        AscendC::ReduceSum(tmp, tileSum, len);
        float tileSumVal = tileSum.GetValue(0);

        // merge with global
        float oldMax = globalMax;
        float newMax = (tileMaxVal > globalMax) ? tileMaxVal : globalMax;
        float newSum;
        if (tileMaxVal > oldMax) {
            newSum = globalSum * expf(oldMax - newMax) + tileSumVal;
        } else {
            newSum = globalSum + tileSumVal * expf(tileMaxVal - newMax);
        }
        globalMax = newMax;
        globalSum = newSum;

        // free temporaries
        tileMax.Free();
        tileSum.Free();
        tmpQueue_.FreeTensor(tmp);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeNorm(uint32_t len, uint32_t gmOffset) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // y = exp(x - globalMax_) / globalSum_
        AscendC::Sub(yLocal, xLocal, globalMax_, len);
        AscendC::Exp(yLocal, yLocal, len);
        AscendC::Div(yLocal, yLocal, globalSum_, len);

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);

        // CopyOut
        yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    float globalMax_ {0.0f};
    float globalSum_ {0.0f};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLength, tiling_data.tileLength);
    op.Process();
}
'''
