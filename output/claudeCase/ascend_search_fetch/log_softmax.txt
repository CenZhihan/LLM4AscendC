host_operator_src = """
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t batchSize = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t featureSize = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t totalLength = batchSize * featureSize;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = *attrs->GetAttrPointer<int32_t>(0);

    // Tile entire row as one tile (assumes featureSize fits in UB, else need more sophisticated tiling)
    uint32_t tileLen = featureSize;
    // Cap to UB budget for safety (4 floats per element: xLocal, yLocal, ones, tmp)
    uint32_t maxElem = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxElem = AlignDown(maxElem, ALIGN_ELEMS);
    if (tileLen > maxElem) tileLen = maxElem;
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;

    uint32_t blockDim = batchSize;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_batchSize(batchSize);
    tiling.set_featureSize(featureSize);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src = """
#include "kernel_operator.h"
#include "reduce.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t batchSize,
                                uint32_t featureSize, uint32_t blockDim) {
        totalLen_ = totalLength;
        batchSize_ = batchSize;
        featureSize_ = featureSize;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, featureSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, featureSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, featureSize_ * sizeof(float) * 3); // ones, maxT, sumT
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || featureSize_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Each block handles multiple rows with grid-stride
        uint32_t row = blockIdx;
        while (row < batchSize_) {
            uint32_t base = row * featureSize_;
            CopyIn(base, featureSize_);
            Compute(featureSize_);
            CopyOut(base, featureSize_);
            row += blockNum;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Temporary tensors from tmpQueue
        AscendC::LocalTensor<float> ones = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxT = tmpQueue_.AllocTensor<float>(); // scalar: length 1
        AscendC::LocalTensor<float> sumT = tmpQueue_.AllocTensor<float>(); // scalar: length 1

        // Create ones tensor
        AscendC::Duplicate(ones, 1.0f, len);

        // 1) Compute max along the row
        AscendC::ReduceMax(maxT, xLocal, len, 0);   // maxT[0] = max(x)
        float maxVal = maxT.GetValue(0);

        // 2) Compute sum(exp(x - max))
        // yLocal = x - max
        AscendC::Muls(yLocal, ones, maxVal, len);   // fill with maxVal
        AscendC::Sub(yLocal, xLocal, yLocal, len);  // x - max
        AscendC::Exp(yLocal, yLocal, len);           // exp(x - max)
        AscendC::ReduceSum(sumT, yLocal, len, 0);    // sum(exp)
        float sumVal = sumT.GetValue(0);
        float logSum = logf(sumVal);

        // 3) y = (x - max) - logSum = x - max - logSum
        // Reuse yLocal for result: first restore x - max
        AscendC::Muls(ones, ones, maxVal, len);
        AscendC::Sub(yLocal, xLocal, ones, len);    // x - max
        float negLogSum = -logSum;
        AscendC::Muls(ones, ones, 1.0f, len);       // reset ones to 1
        AscendC::Muls(ones, ones, negLogSum, len);   // ones = -logSum
        AscendC::Add(yLocal, yLocal, ones, len);     // y = x - max - logSum

        tmpQueue_.FreeTensor(sumT);
        tmpQueue_.FreeTensor(maxT);
        tmpQueue_.FreeTensor(ones);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t featureSize_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.batchSize,
            tiling_data.featureSize, tiling_data.blockDim);
    op.Process();
}
"""
