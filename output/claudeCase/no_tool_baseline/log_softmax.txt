host_operator_src = """

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/types.h"
#include "graph/ge_error_codes.h"
#include "graph/utils/op_desc_utils.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    int64_t *dims = const_cast<int64_t*>(x_shape->GetDims());
    int32_t rank = x_shape->GetDimNum();
    uint32_t rows = (rank >= 2) ? static_cast<uint32_t>(dims[0]) : 1;
    uint32_t cols = (rank >= 2) ? static_cast<uint32_t>(dims[1]) : static_cast<uint32_t>(dims[0]);
    uint32_t totalLength = rows * cols;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim_ptr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dim_ptr != nullptr) ? *dim_ptr : 1;

    uint32_t maxTile = UB_BUDGET_BYTES / (8 * sizeof(float)); // 6144
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (cols < tileLen) {
        uint32_t alignedCols = AlignDown(cols, ALIGN_ELEMS);
        tileLen = (alignedCols > 0) ? alignedCols : cols;
    }

    uint32_t blockDim = rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")

            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t tileLength,
                                int32_t dim)
    {
        rows_ = rows;
        cols_ = cols;
        tileLen_ = tileLength;
        dim_ = dim; // for completeness, not used in this simple 2D version
        xGm_.SetGlobalBuffer((__gm__ float*)x, rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, rows_ * cols_);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(copyQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(expQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        if (rows_ == 0 || cols_ == 0) return;

        uint32_t rowsPerBlock = CeilDiv(rows_, blockNum);
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowStart >= rows_) return;
        if (rowEnd > rows_) rowEnd = rows_;

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            uint32_t globalRowOffset = row * cols_;
            // First pass: compute max across row
            float rowMax = -__builtin_huge_valf();
            uint32_t tiles = CeilDiv(cols_, tileLen_);
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t colOffset = t * tileLen_;
                uint32_t curLen = (colOffset + tileLen_ <= cols_) ? tileLen_ : cols_ - colOffset;
                if (curLen == 0) continue;
                // Load tile from global
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[globalRowOffset + colOffset], curLen);
                // Compute local max (manual loop)
                float localMax = -__builtin_huge_valf();
                for (uint32_t i = 0; i < curLen; ++i) {
                    float val = xLocal(i);
                    if (val > localMax) localMax = val;
                }
                if (localMax > rowMax) rowMax = localMax;
                // Copy to copyQueue
                AscendC::LocalTensor<float> copyLocal = copyQueue_.AllocTensor<float>();
                AscendC::DataCopy(copyLocal, xLocal, curLen);
                copyQueue_.EnQue(copyLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            // Second pass: compute sum of exp(x - rowMax)
            float rowSum = 0.0f;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t colOffset = t * tileLen_;
                uint32_t curLen = (colOffset + tileLen_ <= cols_) ? tileLen_ : cols_ - colOffset;
                if (curLen == 0) continue;
                AscendC::LocalTensor<float> copyLocal = copyQueue_.DeQue<float>();
                // Allocate exp tensor
                AscendC::LocalTensor<float> expLocal = expQueue_.AllocTensor<float>();
                // Compute exp(x - rowMax)
                AscendC::Sub(expLocal, copyLocal, rowMax, curLen);
                AscendC::Exp(expLocal, expLocal, curLen);
                // Compute sum (manual loop)
                float localSum = 0.0f;
                for (uint32_t i = 0; i < curLen; ++i) {
                    localSum += expLocal(i);
                }
                rowSum += localSum;
                expQueue_.FreeTensor(expLocal);
                // Re-enqueue copyLocal for third pass
                copyQueue_.EnQue(copyLocal);
            }

            // Third pass: compute output
            float logSum = AscendC::Log(rowSum);
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t colOffset = t * tileLen_;
                uint32_t curLen = (colOffset + tileLen_ <= cols_) ? tileLen_ : cols_ - colOffset;
                if (curLen == 0) continue;
                AscendC::LocalTensor<float> copyLocal = copyQueue_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                // y = x - rowMax - logSum
                AscendC::Sub(yLocal, copyLocal, rowMax, curLen);
                AscendC::Sub(yLocal, yLocal, logSum, curLen);
                // Write to global
                AscendC::DataCopy(yGm_[globalRowOffset + colOffset], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
                copyQueue_.FreeTensor(copyLocal);
            }

            // Ensure completion before next row
            AscendC::PipeBarrier<AscendC::PIPE_V>();
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> copyQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> expQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t rows_{0};
    uint32_t cols_{0};
    uint32_t tileLen_{0};
    int32_t dim_{1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.rows, tiling_data.cols, tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
"""
