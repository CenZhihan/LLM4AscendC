host_operator_src = """

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rows = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t rowLength = static_cast<uint32_t>(shape.GetDim(1));

    // UB budget: each pass uses one local tensor for tile, double buffered -> 2*tileLen floats.
    // Also temporary scalars negligible. Budget = 8 * tileLen bytes (2 floats per tile?) Actually 2 tiles * 4 bytes = 8* tileLen? Wait: 2 tiles * tileLen floats * 4 = 8 * tileLen. Use factor 8.
    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = rowLength;
    if (tileLen > maxTile) tileLen = AlignDown(maxTile, ALIGN_ELEMS);
    else {
        uint32_t aligned = AlignDown(tileLen, ALIGN_ELEMS);
        if (aligned > 0) tileLen = aligned;
    }

    // blockDim: number of blocks processing rows in parallel
    uint32_t blockDim = rows;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rows(rows);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops
"""

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rows, uint32_t rowLength,
                                uint32_t tileLength, uint32_t blockDim)
    {
        totalLen_ = totalLength;
        rows_ = rows;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // For intermediate sum and max we use scalar registers, no local buffer needed.
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || rowLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = blockDim_;  // actual launched blocks

        // Distribute rows across blocks (block0 handles first set, etc.)
        for (uint32_t r = blockIdx; r < rows_; r += blockNum) {
            processRow(r);
        }
    }

private:
    __aicore__ inline void processRow(uint32_t rowIdx)
    {
        uint32_t base = rowIdx * rowLen_;
        float rowMax = -FLT_MAX;
        float rowSum = 0.0f;

        // Pass 1: compute max
        for (uint32_t offset = 0; offset < rowLen_; offset += tileLen_) {
            uint32_t chunkLen = (offset + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - offset);
            copyInTile(base + offset, chunkLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            // Compute local max using AscendC::ReduceMax? Use vector max and scalar
            float localMax = 0.0f;
            AscendC::ReduceMax(localMax, xLocal, chunkLen);
            if (localMax > rowMax) rowMax = localMax;
            inQueueX_.FreeTensor(xLocal);
        }

        // Pass 2: compute sum of exp(x - max)
        for (uint32_t offset = 0; offset < rowLen_; offset += tileLen_) {
            uint32_t chunkLen = (offset + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - offset);
            copyInTile(base + offset, chunkLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tmp = outQueueY_.AllocTensor<float>(); // reuse output queue for temp
            // Compute exp(x - rowMax)
            AscendC::Sub(tmp, xLocal, rowMax, chunkLen);
            AscendC::Exp(tmp, tmp, chunkLen);
            float partialSum = 0.0f;
            AscendC::ReduceSum(partialSum, tmp, chunkLen);
            rowSum += partialSum;
            outQueueY_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xLocal);
        }

        // Pass 3: compute y = exp(x - rowMax) / rowSum
        for (uint32_t offset = 0; offset < rowLen_; offset += tileLen_) {
            uint32_t chunkLen = (offset + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - offset);
            copyInTile(base + offset, chunkLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // Compute exp(x - max)
            AscendC::Sub(yLocal, xLocal, rowMax, chunkLen);
            AscendC::Exp(yLocal, yLocal, chunkLen);
            // Divide by sum
            float invSum = 1.0f / rowSum;
            AscendC::Muls(yLocal, yLocal, invSum, chunkLen);
            // Copy out
            copyOut(yLocal, base + offset, chunkLen);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

    __aicore__ inline void copyInTile(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void copyOut(AscendC::LocalTensor<float> &yLocal, uint32_t gmOffset, uint32_t len)
    {
        // DataCopy from local to global
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rows_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.rows,
            tiling_data.rowLength,
            tiling_data.tileLength,
            tiling_data.blockDim);
    op.Process();
}
"""
