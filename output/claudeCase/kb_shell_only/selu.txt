host_operator_src = """
#include "selu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile => 2 * tile floats
//   outQueueY: 2 * tile => 2 * tile floats
//   temp: 6 * tile => 6 * tile floats
// Total floats = 10 * tile => bytes = 40 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SeluCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scale = attrs->GetAttrPointer<float>(0);
    const float *alpha = attrs->GetAttrPointer<float>(1);

    // tileLen from UB budget: bytes = 40 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 40;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // Adaptive blockDim based on number of tiles; keep conservative for stability.
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_scale(*scale);
    tiling.set_alpha(*alpha);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SeluCustom : public OpDef {
public:
    explicit SeluCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("scale").AttrType(OPTIONAL).Float(1.0507009873554804934193349852946);
        this->Attr("alpha").AttrType(OPTIONAL).Float(1.6732632423543772848170429916717);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SeluCustom);
} // namespace ops

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSeluCustom {
public:
    __aicore__ inline KernelSeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                float scale, float alpha)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        scale_ = scale;
        alpha_ = alpha;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // Allocate temporary tensors from pipe (non-queued)
        // We'll allocate them on the fly but need to account for memory budget.
        // For simplicity, we pre-allocate a big chunk and reuse offsets.
        // However, the tiling has already reserved enough UB; we can use AllocTensor.
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Allocate temporary tensors from pipe (they are part of UB budget)
        AscendC::LocalTensor<float> expLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
        AscendC::LocalTensor<float> tmpLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
        AscendC::LocalTensor<float> maskLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
        AscendC::LocalTensor<float> selLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));

        // Compute t = exp(x) - 1
        AscendC::Exp(expLocal, xLocal, len);
        AscendC::Sub(tmpLocal, expLocal, 1.0f, len);
        // tmp = alpha * (exp(x) - 1)
        AscendC::Muls(tmpLocal, tmpLocal, alpha_, len);

        // Compute mask = (x > 0) ? 1.0 : 0.0
        AscendC::VCmp(maskLocal, xLocal, 0.0f, AscendC::CMP_GT, len);

        // Select: sel = mask ? x : tmp
        AscendC::Select(selLocal, maskLocal, xLocal, tmpLocal, len);

        // y = scale * sel
        AscendC::Muls(yLocal, selLocal, scale_, len);

        // Free temporaries
        pipe_.FreeTensor(expLocal);
        pipe_.FreeTensor(tmpLocal);
        pipe_.FreeTensor(maskLocal);
        pipe_.FreeTensor(selLocal);

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    float scale_ {1.0507009873554804934193349852946f};
    float alpha_ {1.6732632423543772848170429916717f};
};

extern "C" __global__ __aicore__ void selu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSeluCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.scale, tiling_data.alpha);
    op.Process();
}
"""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor selu_impl_npu(const at::Tensor& self, double scale, double alpha) {
    at::Tensor result = at::empty_like(self);
    EXEC_NPU_CMD(aclnnSeluCustom, self, scale, alpha, result);
    return result;
}

// We expose a function that matches the expected signature: (Tensor) -> Tensor
// For simplicity, we use a wrapper that uses default constants.
at::Tensor selu_custom_npu(const at::Tensor& self) {
    // Use standard SELU constants
    constexpr double default_scale = 1.0507009873554804934193349852946;
    constexpr double default_alpha = 1.6732632423543772848170429916717;
    return selu_impl_npu(self, default_scale, default_alpha);
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("selu_custom", &selu_custom_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("selu_custom", &selu_custom_npu, "SELU activation (SeluCustom on NPU)");
