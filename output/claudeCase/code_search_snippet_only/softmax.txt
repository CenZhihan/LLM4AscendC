host_operator_src = '''
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t MAX_TILE_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    uint32_t totalRows = static_cast<uint32_t>(inShape->GetDim(0));
    uint32_t rowLen = static_cast<uint32_t>(inShape->GetDim(1));
    uint32_t totalLength = totalRows * rowLen;

    // UB budget: we need 3 buffers per tile (input, output, temp)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLen < tileLen) {
        uint32_t aligned = AlignDown(rowLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : rowLen;
    }

    uint32_t numTilesPerRow = CeilDiv(rowLen, tileLen);
    // blockDim based on number of rows; cap at 24
    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_rowLen(rowLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops
'''

kernel_src = '''
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;  // double buffer for ping-pong

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLen,
                                uint32_t tileLen, uint32_t blockDim)
    {
        totalRows_ = totalRows;
        rowLen_ = rowLen;
        tileLen_ = tileLen;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * rowLen_);

        // Allocate local buffers: 2 input queues, 2 output queues, 1 temp queue
        // For softmax we also need a queue for exp temporary? Use tmpQueue for that.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = blockDim_;  // use the same blockDim as from tiling

        // Each block processes a set of complete rows
        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        // For each row, we need three passes:
        // Pass 1: compute global max for the row
        // Pass 2: compute sum of exp(x - max)
        // Pass 3: write output = exp(x - max) / sum

        uint32_t rowOffset = row * rowLen_;
        uint32_t numTiles = (rowLen_ + tileLen_ - 1) / tileLen_;
        float rowMax = -FLT_MAX;

        // ---- Pass 1: find max ----
        for (uint32_t t = 0; t < numTiles; t++) {
            uint32_t tileStart = t * tileLen_;
            uint32_t curLen = (tileStart + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - tileStart);
            if (curLen == 0) break;

            // Load tile into local buffer
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curLen);
            inQueueX_.EnQue(xLocal);

            // Dequeue to process
            xLocal = inQueueX_.DeQue<float>();

            // Compute local max (scalar loop – simplest)
            float localMax = -FLT_MAX;
            for (uint32_t i = 0; i < curLen; i++) {
                float val = xLocal.GetValue(i);
                if (val > localMax) localMax = val;
            }
            if (localMax > rowMax) rowMax = localMax;

            inQueueX_.FreeTensor(xLocal);
        }

        // ---- Pass 2: compute sum of exp(x - rowMax) ----
        float rowSum = 0.0f;
        for (uint32_t t = 0; t < numTiles; t++) {
            uint32_t tileStart = t * tileLen_;
            uint32_t curLen = (tileStart + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - tileStart);
            if (curLen == 0) break;

            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // Compute exp(x - rowMax) and accumulate sum
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(tmp, xLocal, rowMax, curLen);   // tmp = x - rowMax
            AscendC::Exp(tmp, tmp, curLen);               // tmp = exp(x - rowMax)

            // Sum elements in this tile (scalar loop)
            float tileSum = 0.0f;
            for (uint32_t i = 0; i < curLen; i++) {
                tileSum += tmp.GetValue(i);
            }
            rowSum += tileSum;

            inQueueX_.FreeTensor(xLocal);
            tmpQueue_.FreeTensor(tmp);
        }

        // ---- Pass 3: write output = exp(x - rowMax) / rowSum ----
        for (uint32_t t = 0; t < numTiles; t++) {
            uint32_t tileStart = t * tileLen_;
            uint32_t curLen = (tileStart + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - tileStart);
            if (curLen == 0) break;

            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // exp(x - rowMax) / rowSum
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

            AscendC::Sub(tmp, xLocal, rowMax, curLen);
            AscendC::Exp(tmp, tmp, curLen);
            AscendC::Muls(yLocal, tmp, 1.0f / rowSum, curLen);

            // Write output
            AscendC::DataCopy(yGm_[rowOffset + tileStart], yLocal, curLen);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            tmpQueue_.FreeTensor(tmp);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.rowLen, tiling_data.tileLen, tiling_data.blockDim);
    op.Process();
}
'''

python_bind_src = '''
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor softmax_impl_npu(const at::Tensor& self) {
    at::Tensor result = at::empty_like(self);
    // The custom operator "SoftmaxCustom" is registered via OP_ADD.
    // EXEC_NPU_CMD dispatches it using the aclnn interface generated from the op def.
    EXEC_NPU_CMD(aclnnSoftmaxCustom, self, 1, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("softmax_custom", &softmax_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("softmax_custom", &softmax_impl_npu, "Softmax activation (SoftmaxCustom on NPU)");
}
