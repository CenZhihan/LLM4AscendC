host_operator_src = """
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: inQueueX: 2 * tileLength, outQueueY: 2 * tileLength, tmpQueue: 1 * tileLength (for scalar reduce)
// We use two passes: first pass needs xLocal and a scalar, second pass needs xLocal and exp, third pass needs xLocal.
// Conservative: at most 3 * tileLength floats = 12 * tileLength bytes.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0);
    const uint32_t totalLength = static_cast<uint32_t>(xShape->GetOriginShape().GetShapeSize());
    // Assume dim=1 => rowLength = shape[1]
    uint32_t rowLength = (xShape->GetOriginShape().GetDimNum() >= 2) ?
                         static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1)) :
                         totalLength;

    // tileLength from UB budget: bytes = 3 * tileLength * 4 = 12 * tileLength
    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        tileLen = AlignDown(rowLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowLength;
    }

    // Block dimension based on number of tiles per row and total tiles
    const uint32_t tilesPerRow = (tileLen == 0) ? 1u : CeilDiv(rowLength, tileLen);
    const uint32_t totalTiles = (tilesPerRow == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_rowLength(rowLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

        this->AICore().SetTiling(optiling::TilingFunc);

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t rowLength, uint32_t blockDim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        rowLen_ = rowLength;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float)); // for scalar reduce result
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();

        // Each block processes a set of complete rows (grid‑stride over rows)
        const uint32_t rows = totalLen_ / rowLen_;
        for (uint32_t r = blockIdx; r < rows; r += blockNum) {
            ProcessRow(r);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        uint32_t rowBase = rowIdx * rowLen_;
        uint32_t remaining = rowLen_;
        uint32_t offset = 0;

        // --- Pass 1: compute global max over row ---
        float rowMax = -__builtin_huge_valf();
        uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
        // First tile: load and compute max, then loop over remaining tiles
        {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowBase], curLen);
            inQueueX_.EnQue(xLocal);
        }
        // Process first tile and prefetch next
        {
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            // compute local max
            AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceMax(maxLocal, xLocal, curLen);
            float tileMax = maxLocal.GetValue(0);
            tmpQueue_.FreeTensor(maxLocal);
            rowMax = (tileMax > rowMax) ? tileMax : rowMax;
            inQueueX_.FreeTensor(xLocal);
        }
        offset += curLen;
        remaining -= curLen;

        while (remaining > 0) {
            curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // load next tile
            {
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + offset], curLen);
                inQueueX_.EnQue(xLocal);
            }
            // process current tile
            {
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceMax(maxLocal, xLocal, curLen);
                float tileMax = maxLocal.GetValue(0);
                tmpQueue_.FreeTensor(maxLocal);
                rowMax = (tileMax > rowMax) ? tileMax : rowMax;
                inQueueX_.FreeTensor(xLocal);
            }
            offset += curLen;
            remaining -= curLen;
        }

        // --- Pass 2: compute sum of exp(x - max) over row ---
        float rowSum = 0.0f;
        offset = 0;
        remaining = rowLen_;
        // First tile
        {
            curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowBase + offset], curLen);
            inQueueX_.EnQue(xLocal);
        }
        {
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::Subs(xLocal, xLocal, rowMax, curLen);
            AscendC::Exp(xLocal, xLocal, curLen);
            AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumLocal, xLocal, curLen);
            rowSum += sumLocal.GetValue(0);
            tmpQueue_.FreeTensor(sumLocal);
            inQueueX_.FreeTensor(xLocal);
        }
        offset += curLen;
        remaining -= curLen;
        while (remaining > 0) {
            curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            {
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + offset], curLen);
                inQueueX_.EnQue(xLocal);
            }
            {
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::Subs(xLocal, xLocal, rowMax, curLen);
                AscendC::Exp(xLocal, xLocal, curLen);
                AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceSum(sumLocal, xLocal, curLen);
                rowSum += sumLocal.GetValue(0);
                tmpQueue_.FreeTensor(sumLocal);
                inQueueX_.FreeTensor(xLocal);
            }
            offset += curLen;
            remaining -= curLen;
        }

        float logSum = AscendC::Log(rowSum);

        // --- Pass 3: compute output = x - max - logSum and store ---
        offset = 0;
        remaining = rowLen_;
        // First tile
        {
            curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowBase + offset], curLen);
            inQueueX_.EnQue(xLocal);
        }
        {
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // y = x - max - logSum
            AscendC::Subs(yLocal, xLocal, rowMax, curLen);
            AscendC::Subs(yLocal, yLocal, logSum, curLen);
            // Store to global
            AscendC::DataCopy(yGm_[rowBase + offset], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
        offset += curLen;
        remaining -= curLen;
        while (remaining > 0) {
            curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            {
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + offset], curLen);
                inQueueX_.EnQue(xLocal);
            }
            {
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Subs(yLocal, xLocal, rowMax, curLen);
                AscendC::Subs(yLocal, yLocal, logSum, curLen);
                AscendC::DataCopy(yGm_[rowBase + offset], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
            }
            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t rowLen_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.rowLength, tiling_data.blockDim);
    op.Process();
}
"""
