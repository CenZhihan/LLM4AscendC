host_operator_src = '''

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t numRows = static_cast<uint32_t>(shape.GetDim(0));
    const uint32_t rowLength = static_cast<uint32_t>(shape.GetDim(1));
    const uint32_t totalLength = numRows * rowLength;

    // tileLen from UB budget: bytes = 5 * tileLen * sizeof(float) = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        uint32_t aligned = AlignDown(rowLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : rowLength;
    }

    const uint32_t numTilesPerRow = CeilDiv(rowLength, tileLen);
    const uint32_t numTilesTotal = numRows * numTilesPerRow;
    uint32_t blockDim = numTilesTotal;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_numRows(numRows);
    tiling.set_rowLength(rowLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops
'''

kernel_src = '''

#include "kernel_operator.h"
#include <cfloat>

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t tileLength,
                                uint32_t numRows,
                                uint32_t rowLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        numRows_ = numRows;
        rowLen_ = rowLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes consecutive rows
        const uint32_t rowsPerBlock = (numRows_ + blockNum - 1u) / blockNum;
        const uint32_t startRow = blockIdx * rowsPerBlock;
        const uint32_t endRow = (startRow + rowsPerBlock > numRows_) ? numRows_ : startRow + rowsPerBlock;

        for (uint32_t r = startRow; r < endRow; ++r) {
            ProcessRow(r);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        const uint32_t base = row * rowLen_;
        const uint32_t numTiles = (rowLen_ + tileLen_ - 1u) / tileLen_;

        // ---- first pass: find row maximum ----
        float rowMax = -FLT_MAX;
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = base + t * tileLen_;
            uint32_t len = (t == numTiles - 1) ? (rowLen_ - t * tileLen_) : tileLen_;
            float localMax = ComputeTileMax(offset, len);
            if (localMax > rowMax) rowMax = localMax;
        }

        // ---- second pass: compute exp sum ----
        float rowSum = 0.0f;
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = base + t * tileLen_;
            uint32_t len = (t == numTiles - 1) ? (rowLen_ - t * tileLen_) : tileLen_;
            float localSum = ComputeTileExpSum(offset, len, rowMax);
            rowSum += localSum;
        }

        // ---- third pass: normalize and write output ----
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = base + t * tileLen_;
            uint32_t len = (t == numTiles - 1) ? (rowLen_ - t * tileLen_) : tileLen_;
            NormalizeTile(offset, len, rowMax, rowSum);
        }
    }

    __aicore__ inline float ComputeTileMax(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        // Use ReduceMax to find max in vector
        AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
        AscendC::ReduceMax(maxLocal, xLocal, len);
        float maxVal = maxLocal(0);
        tmpQueue_.FreeTensor(maxLocal);
        inQueueX_.FreeTensor(xLocal);
        return maxVal;
    }

    __aicore__ inline float ComputeTileExpSum(uint32_t gmOffset, uint32_t len, float rowMax)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
        // compute exp(x - rowMax)
        AscendC::Sub(expLocal, xLocal, rowMax, len);
        AscendC::Exp(expLocal, expLocal, len);
        // reduce sum
        AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>(); // reuse previous? careful
        AscendC::ReduceSum(sumLocal, expLocal, len);
        float sumVal = sumLocal(0);
        tmpQueue_.FreeTensor(expLocal);
        tmpQueue_.FreeTensor(sumLocal);
        inQueueX_.FreeTensor(xLocal);
        return sumVal;
    }

    __aicore__ inline void NormalizeTile(uint32_t gmOffset, uint32_t len, float rowMax, float rowSum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
        // y = exp(x - max) / sum
        AscendC::Sub(tLocal, xLocal, rowMax, len);
        AscendC::Exp(tLocal, tLocal, len);
        AscendC::Div(yLocal, tLocal, rowSum, len);
        // write output
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numRows_ {0};
    uint32_t rowLen_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.numRows, tiling_data.rowLength);
    op.Process();
}
'''
