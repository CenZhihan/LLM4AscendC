host_operator_src = """
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (per row we need two passes):
//   Pass1: inQueueX: 2*tile, tmpQueue: 1*tile  -> 3 tiles
//   Pass2: inQueueX: 2*tile, outQueueY: 2*tile, tmpQueue: 1*tile -> 5 tiles
//   Largest is 5*tile floats -> 20*tile bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_TILE_ELEMS = 8192;  // 32 KB per tile

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;

    const auto &originShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(originShape.GetShapeSize());
    uint32_t ndim = originShape.GetDimNum();
    int32_t dim = 1;  // default
    if (context->GetAttrs()->GetAttrNum() > 0) {
        const int32_t *dimPtr = context->GetAttrs()->GetAttrPointer<int32_t>(0);
        if (dimPtr != nullptr) dim = *dimPtr;
    }
    // Handle negative dim
    if (dim < 0) dim += ndim;
    // Compute rowLength: product of dimensions except the reduction dim
    uint32_t rowLength = 1;
    uint32_t numRows = 1;
    for (uint32_t i = 0; i < ndim; ++i) {
        uint32_t s = static_cast<uint32_t>(originShape.GetDim(i));
        if (static_cast<int32_t>(i) == dim) {
            // reduction dimension: treat as row length
            rowLength = s;
        } else {
            numRows *= s;
        }
    }
    // Compute tileLength for reduction dimension
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = (rowLength < maxTile) ? rowLength : maxTile;
    // Align tileLen to reduce boundary waste
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = (rowLength < ALIGN_ELEMS) ? rowLength : ALIGN_ELEMS;

    // Block dim: we can parallelize per row, but we have many rows.
    // Use simple: blockDim = min(numRows, 24) for load balancing.
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);
    tiling.set_rowLength(rowLength);
    tiling.set_numRows(numRows);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Attr("dim").AttrType(OPTIONAL).Int(1);
        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, int32_t dim,
                                uint32_t rowLength, uint32_t numRows) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        dim_ = dim;
        rowLen_ = rowLength;
        numRows_ = numRows;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        // additional buffers for reduction scalars
        pipe_.InitBuffer(redBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Each block handles a subset of rows
        uint32_t rowsPerBlock = CeilDiv(numRows_, blockNum);
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = (rowStart + rowsPerBlock > numRows_) ? numRows_ : rowStart + rowsPerBlock;

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            uint32_t rowBase = row * rowLen_;
            ProcessRow(rowBase);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowBase) {
        // Pass 1: compute global max
        float globalMax = -__f32_inf();
        uint32_t offset = 0;
        while (offset < rowLen_) {
            uint32_t curLen = (rowLen_ - offset > tileLen_) ? tileLen_ : rowLen_ - offset;
            LoadTile(rowBase + offset, curLen);

            // Compute local max of this tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceMax(tLocal, xLocal, curLen);
            float localMax = tLocal.GetValue(0);
            if (localMax > globalMax) globalMax = localMax;

            tmpQueue_.FreeTensor(tLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
        }

        // Pass 2: compute sum of exp(x - max) and final output
        float globalSum = 0.0f;
        offset = 0;
        // We will write output in place; need to re-read x from global
        while (offset < rowLen_) {
            uint32_t curLen = (rowLen_ - offset > tileLen_) ? tileLen_ : rowLen_ - offset;
            LoadTile(rowBase + offset, curLen);

            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            // t = x - globalMax
            AscendC::Subs(tLocal, xLocal, globalMax, curLen);
            // t = exp(t)
            AscendC::Exp(tLocal, tLocal, curLen);
            // reduce sum for this tile
            AscendC::ReduceSum(redBuf_, tLocal, curLen);
            float tileSum = redBuf_.GetValue(0);
            globalSum += tileSum;

            // Store exp(x - max) back to tLocal for later multiplication? Wait, we need log_softmax = x - max - log(globalSum)
            // We cannot compute final output until we know globalSum.
            // So we need to keep the computed exp values? Instead, we can compute final output in a third pass.
            // Two-pass approach: Pass1: globalMax, Pass2: globalSum, Pass3: final output.
            // But to save bandwidth, we can compute exp and accumulate sum, then in a third pass compute final output.
            // For simplicity, we'll do three passes (as many kernels do).
            // We'll free temporary tensors and not output yet.
            // Store the exp values? No, we can re-read x from global in pass3.
            // So in pass2, just compute sum and store x unchanged? Actually we need to compute exp to get sum, but we can discard it.
            // So we can do all three passes:
            // Pass1: max
            // Pass2: sum (reads x again)
            // Pass3: output (reads x again) using stored max and sum.
            // This is inefficient but acceptable for demonstration.

            tmpQueue_.FreeTensor(tLocal);
            outQueueY_.FreeTensor(yLocal); // not used yet
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
        }

        // Pass 3: compute final output
        float logSum = logf(globalSum);
        offset = 0;
        while (offset < rowLen_) {
            uint32_t curLen = (rowLen_ - offset > tileLen_) ? tileLen_ : rowLen_ - offset;
            LoadTile(rowBase + offset, curLen);

            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            // t = x - max - log(sum)
            AscendC::Subs(tLocal, xLocal, globalMax, curLen);
            AscendC::Subs(yLocal, tLocal, logSum, curLen);
            // Write back to global
            AscendC::DataCopy(yGm_[rowBase + offset], yLocal, curLen);

            tmpQueue_.FreeTensor(tLocal);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
        }
    }

    __aicore__ inline void LoadTile(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline CeilDiv(uint32_t a, uint32_t b) {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> redBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ {1};
    uint32_t rowLen_ {0};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.blockDim, tiling_data.dim,
            tiling_data.rowLength, tiling_data.numRows);
    op.Process();
}
"""
