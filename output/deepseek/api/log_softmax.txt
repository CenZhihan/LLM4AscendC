host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue: 1 * tile
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::Shape *inputShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t ndim = static_cast<uint32_t>(inputShape->GetDimNum());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    // Support negative dim (wrap around)
    int32_t normalizedDim = (dim < 0) ? (ndim + dim) : dim;
    if (normalizedDim < 0 || normalizedDim >= static_cast<int32_t>(ndim)) {
        normalizedDim = ndim - 1; // fallback to last dim
    }
    uint32_t rowLength = 1;
    for (uint32_t i = 0; i < ndim; ++i) {
        if (i == static_cast<uint32_t>(normalizedDim)) {
            rowLength = static_cast<uint32_t>(inputShape->GetDim(i));
        }
    }
    // totalLength must be multiple of rowLength
    // uint32_t batchSize = totalLength / rowLength; // not needed directly

    // Tile length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        const uint32_t aligned = AlignDown(rowLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : rowLength;
    }

    // Clamp tileLen to rowLength (cannot exceed one row)
    if (tileLen > rowLength) tileLen = rowLength;

    // blockDim: number of tiles across rows (each row processed by one block? but we can have multiple rows per block)
    // Use number of tiles across all elements as a base.
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_rowLength(rowLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        // default dim=1 to match original model
        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include "lib/math.hpp"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t rowLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        rowLen_ = rowLength;
        if (rowLen_ == 0) rowLen_ = 1; // avoid division by zero
        batchSize_ = totalLen_ / rowLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || rowLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        const uint32_t rowsPerBlock = (batchSize_ + blockNum - 1) / blockNum;
        const uint32_t startRow = blockIdx * rowsPerBlock;
        const uint32_t endRow = (blockIdx + 1) * rowsPerBlock;
        if (startRow >= batchSize_) return;
        const uint32_t lastRow = (endRow > batchSize_) ? batchSize_ : endRow;

        for (uint32_t row = startRow; row < lastRow; ++row) {
            const uint32_t rowOffset = row * rowLen_;
            ProcessRow(rowOffset);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowOffset)
    {
        // Phase 1: compute global max across the row
        float rowMax = -FLT_MAX;
        for (uint32_t tileStart = 0; tileStart < rowLen_; tileStart += tileLen_) {
            uint32_t curLen = (tileStart + tileLen_ > rowLen_) ? (rowLen_ - tileStart) : tileLen_;
            // Copy tile from global
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curLen);
            // Reduce to max
            AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>(); // single element
            AscendC::LocalTensor<float> wsTmp = tmpQueue_.AllocTensor<float>();   // workspace
            AscendC::ReduceMax(maxLocal, xLocal, wsTmp, curLen);
            float tileMax = maxLocal(0);
            if (tileMax > rowMax) rowMax = tileMax;
            tmpQueue_.FreeTensor(wsTmp);
            tmpQueue_.FreeTensor(maxLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        // Phase 2: compute sum of exp(x - rowMax) across the row
        float rowSum = 0.0f;
        for (uint32_t tileStart = 0; tileStart < rowLen_; tileStart += tileLen_) {
            uint32_t curLen = (tileStart + tileLen_ > rowLen_) ? (rowLen_ - tileStart) : tileLen_;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curLen);
            // Subtract rowMax
            AscendC::Subs(xLocal, xLocal, rowMax, curLen);
            // Compute exp
            AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Exp(expLocal, xLocal, curLen);
            // Reduce to sum
            AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>(); // single element
            AscendC::LocalTensor<float> wsTmp = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumLocal, expLocal, wsTmp, curLen);
            rowSum += sumLocal(0);
            tmpQueue_.FreeTensor(wsTmp);
            tmpQueue_.FreeTensor(sumLocal);
            tmpQueue_.FreeTensor(expLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        // Phase 3: compute output y = x - rowMax - log(rowSum)
        float logSum = logf(rowSum);  // using standard math library (NLC)
        for (uint32_t tileStart = 0; tileStart < rowLen_; tileStart += tileLen_) {
            uint32_t curLen = (tileStart + tileLen_ > rowLen_) ? (rowLen_ - tileStart) : tileLen_;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curLen);
            // y = x - rowMax - logSum
            AscendC::Subs(xLocal, xLocal, rowMax, curLen);  // reuse xLocal
            AscendC::Subs(xLocal, xLocal, logSum, curLen);
            // Write output
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, xLocal, curLen);  // copy to output buffer
            AscendC::DataCopy(yGm_[rowOffset + tileStart], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t batchSize_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.rowLength);
    op.Process();
}
"""
