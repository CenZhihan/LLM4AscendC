host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget conservative: 192KB
// Each tile uses: inQueueX: 2 tiles (for double buffering), outQueueY: 2 tiles, tmpQueue: 1 tile
// Each tile holds tileLength floats (4 bytes each) => overhead = 5 * tileLength * 4 = 20*tileLen.
// Additionally we need a few scalars.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Get shape dims: assume 2D, dim=1 softmax
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t rows = static_cast<uint32_t>(shape.GetDim(0));
    const uint32_t cols = static_cast<uint32_t>(shape.GetDim(1));
    const uint32_t rowLength = cols;

    // Compute tileLen: we need to process each row in multiple tiles if needed
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements per tile based on UB
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        tileLen = AlignDown(rowLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowLength;
    } else {
        // Ensure tileLen divides rowLength? Not required, but for simplicity align to ALIGN_ELEMS.
        tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    }

    // Adaptive blockDim based on number of rows
    uint32_t blockDim = rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t tileLength, uint32_t blockDim)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Buffers per tile: inQueueX (double buffer for read), outQueueY (double buffer for write)
        // tmpQueue for per-tile reduction temporary arrays
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(redBuf_, 1, tileLen_ * sizeof(float)); // for reduction operations
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t numRows = totalLen_ / rowLen_;

        // Each block processes rows with stride blockDim
        for (uint32_t r = blockIdx; r < numRows; r += blockDim_) {
            uint32_t rowStart = r * rowLen_;
            ProcessRow(rowStart);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowStart)
    {
        uint32_t remaining = rowLen_;
        uint32_t offset = 0;

        // --- First pass: compute global max and sum for this row (online combination) ---
        float globalMax = -__builtin_huge_valf();
        float globalSum = 0.0f;

        // iterate tiles
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // copy tile from global to local
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowStart + offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();

            // Compute tile max
            AscendC::LocalTensor<float> tMax = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceMax(tMax, xLocal, curLen);  // reduces into a scalar? Actually ReduceMax outputs a tensor of size 1? We'll use a scalar approach.
            // For simplicity, we'll use AscendC vector reduce max using intrinsic
            float tileMax = 0.0f;
            AscendC::LocalTensor<float> redScalar = redBuf_.AllocTensor<float>();
            AscendC::ReduceMax(redScalar, xLocal, curLen);
            // redScalar[0] contains max
            tileMax = redScalar(0);
            redBuf_.FreeTensor(redScalar);

            // Compute tile sum = sum(exp(x - tileMax))
            AscendC::LocalTensor<float> tExp = tmpQueue_.AllocTensor<float>();
            AscendC::Subs(tExp, xLocal, tileMax, curLen);
            AscendC::Exp(tExp, tExp, curLen);
            AscendC::ReduceSum(redScalar, tExp, curLen);
            float tileSum = redScalar(0);
            redBuf_.FreeTensor(redScalar);

            // Combine with global max/sum
            float newMax = (tileMax > globalMax) ? tileMax : globalMax;
            float ratioOld = (globalMax == newMax) ? 1.0f : expf(globalMax - newMax);
            float ratioNew = (tileMax == newMax) ? 1.0f : expf(tileMax - newMax);
            globalSum = globalSum * ratioOld + tileSum * ratioNew;
            globalMax = newMax;

            tmpQueue_.FreeTensor(tExp);
            inQueueX_.FreeTensor(xLocal);

            remaining -= curLen;
            offset += curLen;
        }

        // --- Second pass: compute output y = exp(x - globalMax) / globalSum ---
        remaining = rowLen_;
        offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

            // copy x tile from global (double buffer)
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowStart + offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();

            // compute y
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tExp = tmpQueue_.AllocTensor<float>();
            AscendC::Subs(tExp, xLocal, globalMax, curLen);
            AscendC::Exp(tExp, tExp, curLen);
            AscendC::Div(yLocal, tExp, globalSum, curLen);

            // copy out
            AscendC::DataCopy(yGm_[rowStart + offset], yLocal, curLen);

            tmpQueue_.FreeTensor(tExp);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);

            remaining -= curLen;
            offset += curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> redBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLength, tiling_data.tileLength, tiling_data.blockDim);
    op.Process();
}
