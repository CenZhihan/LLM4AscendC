host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape &shape = context->GetInputShape(0)->GetOriginShape();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t dim = attrs->GetAttrPointer<int32_t>(0) ? *attrs->GetAttrPointer<int32_t>(0) : 1;

    uint32_t rank = shape.GetDimNum();
    uint32_t outerSize = 1;
    uint32_t innerSize = (dim >= 0) ? shape.GetDim(dim) : shape.GetDim(rank + dim);
    for (uint32_t i = 0; i < rank; i++) {
        if (i != (uint32_t)dim) {
            outerSize *= shape.GetDim(i);
        }
    }
    uint32_t totalLength = outerSize * innerSize;

    // Per-chunk UB usage: input, output, tmp (3 * tileInner * sizeof(float))
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileInner = (innerSize < maxTile) ? AlignDown(innerSize, ALIGN_ELEMS) : maxTile;
    if (tileInner == 0) tileInner = innerSize; // fallback

    // Block dim: one row per block if possible
    uint32_t blockDim = (outerSize  > 24) ? 24 : (outerSize > 8 ? outerSize : 8);
    // Ensure we don't over-subscribe
    if (blockDim > outerSize) blockDim = outerSize;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outerSize,
                                uint32_t innerSize, uint32_t tileInner,
                                uint32_t blockDim)
    {
        totalLen_ = totalLength;
        outerSize_ = outerSize;
        innerSize_ = innerSize;
        tileInner_ = tileInner;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // One buffer each for input chunk, output chunk, temporary (exp etc.)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileInner_ * sizeof(float));
        // Scalars for per-row max and sum
        pipe_.InitBuffer(scalarQueue_, 2, sizeof(float)); // two scalars
    }

    __aicore__ inline void Process()
    {
        if (outerSize_ == 0 || innerSize_ == 0) return;

        const uint32_t blockId = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Each block handles rows in a strided fashion
        for (uint32_t rowBase = blockId; rowBase < outerSize_; rowBase += blockNum) {
            computeRow(rowBase);
        }
    }

private:
    __aicore__ inline void computeRow(uint32_t rowIdx)
    {
        // Phase 1: compute max across all chunks
        float globalMax = -FLT_MAX;
        // Phase 2: compute sum of exp(x - globalMax)
        float globalSum = 0.0f;

        // ----- Phase 1 -----
        uint32_t remaining = innerSize_;
        uint32_t offset = rowIdx * innerSize_;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileInner_) ? tileInner_ : remaining;
            curLen = (curLen / 8) * 8; // ensure 8-aligned for vector ops

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            // Compute max of this chunk using a scalar reduction
            AscendC::LocalTensor<float> scalar = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceMax(xLocal, scalar, curLen);
            float chunkMax = *((float*)scalar.GetAddr()); // get scalar via address
            scalarQueue_.FreeTensor(scalar);
            if (chunkMax > globalMax) globalMax = chunkMax;

            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }

        // ----- Phase 2: sum of exp(x - globalMax) -----
        remaining = innerSize_;
        offset = rowIdx * innerSize_;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileInner_) ? tileInner_ : remaining;
            curLen = (curLen / 8) * 8;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> expTmp = tmpQueue_.AllocTensor<float>();
            // max subtracted for numerical stability
            AscendC::Subs(expTmp, xLocal, globalMax, curLen);
            AscendC::Exp(expTmp, expTmp, curLen);

            AscendC::LocalTensor<float> scalar = scalarQueue_.AllocTensor<float>();
            float sumChunk = 0.0f;
            AscendC::ReduceSum(expTmp, scalar, curLen);
            sumChunk = *((float*)scalar.GetAddr());
            scalarQueue_.FreeTensor(scalar);
            globalSum += sumChunk;

            tmpQueue_.FreeTensor(expTmp);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }

        // compute logSum = log(globalSum)
        float logSum;
        AscendC::LocalTensor<float> scalar = scalarQueue_.AllocTensor<float>();
        *((float*)scalar.GetAddr()) = globalSum;
        AscendC::Log(scalar, scalar, 1);
        logSum = *((float*)scalar.GetAddr());
        scalarQueue_.FreeTensor(scalar);

        // ----- Phase 3: output y = x - logSum -----
        remaining = innerSize_;
        offset = rowIdx * innerSize_;
        uint32_t outOffset = offset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileInner_) ? tileInner_ : remaining;
            curLen = (curLen / 8) * 8;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            xLocal = inQueueX_.DeQue<float>();
            AscendC::Subs(yLocal, xLocal, logSum, curLen);
            AscendC::DataCopy(yGm_[outOffset], yLocal, curLen);

            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            outOffset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> scalarQueue_; // for scalar values

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outerSize_ {0};
    uint32_t innerSize_ {0};
    uint32_t tileInner_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.outerSize,
            tiling_data.innerSize, tiling_data.tileInner, tiling_data.blockDim);
    op.Process();
}
"""
