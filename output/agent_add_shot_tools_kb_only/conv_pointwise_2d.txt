kernel_src="""
#include "kernel_operator.h"

// 简化：使用 float 类型
using dataType = float;
constexpr int32_t BUFFER_NUM = 2; // 每个队列的张量数量

class KernelConvPW2D {
public:
    __aicore__ inline KernelConvPW2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t w, uint32_t tileNum)
    {
        this->n = n;
        this->cin = cin;
        this->cout = cout;
        this->h = h;
        this->w = w;
        this->tileNum = tileNum;

        spatial = h * w;
        // 每个 batch 内的每个通道平面大小
        tileLen = (spatial / (tileNum * BUFFER_NUM));
        if (tileLen == 0) {
            tileLen = spatial; // 很小的情况不分块
            this->tileNum = 1;
        }

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ dataType*>(x), (uint64_t)n * cin * spatial);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ dataType*>(weight), (uint64_t)cout * cin);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ dataType*>(y), (uint64_t)n * cout * spatial);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, tileLen * sizeof(dataType));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileLen * sizeof(dataType));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, tileLen * sizeof(dataType));
        pipe.InitBuffer(inQueueW, 1, cin * sizeof(dataType)); // 每次加载一个输出通道的权重向量
    }

    __aicore__ inline void Process()
    {
        for (uint32_t bn = 0; bn < n; ++bn) {
            for (uint32_t co = 0; co < cout; ++co) {
                // 加载当前输出通道的 cin 权重
                AscendC::LocalTensor<dataType> wLocal = inQueueW.AllocTensor<dataType>();
                AscendC::DataCopy(wLocal, wGm[co * cin], cin);
                inQueueW.EnQue(wLocal);
                wLocal = inQueueW.DeQue<dataType>();

                uint32_t rounds = (spatial + tileLen - 1) / tileLen;
                for (uint32_t t = 0; t < rounds; ++t) {
                    uint32_t offset = t * tileLen;
                    uint32_t curr = (offset + tileLen <= spatial) ? tileLen : (spatial - offset);

                    AscendC::LocalTensor<dataType> yLocal = outQueueY.AllocTensor<dataType>();

                    // 使用第一输入通道初始化 yLocal = x * w
                    {
                        AscendC::LocalTensor<dataType> xLocal = inQueueX.AllocTensor<dataType>();
                        // x 索引: ((bn * cin + 0) * spatial + offset)
                        AscendC::DataCopy(xLocal, xGm[((uint64_t)bn * cin + 0) * spatial + offset], curr);
                        inQueueX.EnQue(xLocal);
                        xLocal = inQueueX.DeQue<dataType>();

                        // 取标量权重 w[co, 0]
                        dataType alpha0 = 0;
                        // 假设 LocalTensor 支持按索引读出标量（设备侧编译器会内联）
                        alpha0 = wLocal.GetValue(0);

                        AscendC::Muls(yLocal, xLocal, alpha0, curr);

                        inQueueX.FreeTensor(xLocal);
                    }

                    // 累加后续输入通道
                    for (uint32_t ci = 1; ci < cin; ++ci) {
                        AscendC::LocalTensor<dataType> xLocal = inQueueX.AllocTensor<dataType>();
                        AscendC::DataCopy(xLocal, xGm[((uint64_t)bn * cin + ci) * spatial + offset], curr);
                        inQueueX.EnQue(xLocal);
                        xLocal = inQueueX.DeQue<dataType>();

                        dataType alpha = wLocal.GetValue(ci);

                        AscendC::LocalTensor<dataType> tmpLocal = tmpQueue.AllocTensor<dataType>();
                        AscendC::Muls(tmpLocal, xLocal, alpha, curr);
                        AscendC::Add(yLocal, yLocal, tmpLocal, curr);

                        tmpQueue.FreeTensor(tmpLocal);
                        inQueueX.FreeTensor(xLocal);
                    }

                    // 写回输出: ((bn * cout + co) * spatial + offset)
                    AscendC::DataCopy(yGm[((uint64_t)bn * cout + co) * spatial + offset], yLocal, curr);
                    outQueueY.FreeTensor(yLocal);
                }

                inQueueW.FreeTensor(wLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueW;

    AscendC::GlobalTensor<dataType> xGm;
    AscendC::GlobalTensor<dataType> wGm;
    AscendC::GlobalTensor<dataType> yGm;

    uint32_t n, cin, cout, h, w, tileNum;
    uint32_t spatial;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPW2D op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w, tiling_data.tileNum);
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv_pointwise_2d_custom_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    // x: [N, Cin, H, W], weight: [Cout, Cin, 1, 1]
    auto sizes = x.sizes();
    auto N = sizes[0];
    auto H = sizes[2];
    auto W = sizes[3];
    auto Cout = weight.size(0);

    at::Tensor y = at::empty({N, Cout, H, W}, x.options());
    EXEC_NPU_CMD(aclnnConvPointwise2dCustom, x, weight, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_pointwise_2d_custom", &conv_pointwise_2d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_pointwise_2d_custom", &conv_pointwise_2d_custom_impl_npu, "1x1 conv2d (pointwise)");

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用 AscendC 自定义核实现的 1x1 点卷积（无偏置）。
    """
    def __init__(self, in_channels: int, out_channels: int):
        super().__init__()
        # 权重形状与 nn.Conv2d 的 1x1 权重一致: [out_channels, in_channels, 1, 1]
        self.weight = torch.nn.Parameter(torch.randn(out_channels, in_channels, 1, 1) * (1.0 / (in_channels ** 0.5)))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_pointwise_2d_custom(x, self.weight)
