    gert::Shape* y_shape = context->GetOutputShape(0);

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Y = float;

constexpr int32_t IN_Q_NUM = 1;
constexpr int32_t OUT_Q_NUM = 1;
constexpr int32_t WORK_Q_NUM = 1;

class KernelMaxReduceLastDim {
public:
    __aicore__ inline KernelMaxReduceLastDim() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalOuter, uint32_t inner, uint32_t tileRows)
    {
        this->inner = inner;
        this->tileRows = tileRows;

        // 将行数在Block之间划分
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (totalOuter + blockNum - 1) / blockNum;
        uint32_t startRow = rowsPerBlock * blockIdx;
        uint32_t endRow = startRow + rowsPerBlock;
        endRow = endRow > totalOuter ? totalOuter : endRow;
        this->rowsThisBlock = (endRow > startRow) ? (endRow - startRow) : 0;
        this->startRow = startRow;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + static_cast<uint64_t>(startRow) * inner, static_cast<uint64_t>(rowsThisBlock) * inner);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + startRow, rowsThisBlock);

        // UB队列：输入为一行(inner)，输出为标量(1)，工作空间分配固定最小大小
        pipe.InitBuffer(inQueueX, IN_Q_NUM, inner * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, OUT_Q_NUM, sizeof(DTYPE_Y));
        // ReduceMax最小工作空间，示例中给出64字节（32 * half）；这里使用float，给定32 * sizeof(float)
        pipe.InitBuffer(workQueue, WORK_Q_NUM, 32 * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (rowsThisBlock == 0) {
            return;
        }
        // 每次处理1行
        for (uint32_t r = 0; r < rowsThisBlock; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // 拷贝一整行
        AscendC::DataCopy(xLocal, xGm[r * inner], inner);
        inQueueX.EnQue<DTYPE_X>(xLocal);
    }

    __aicore__ inline void Compute(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> workLocal = workQueue.AllocTensor<DTYPE_Y>();

        // 对一行做ReduceMax，输出一个标量
        AscendC::ReduceMax<DTYPE_Y>(yLocal, xLocal, workLocal, inner, true);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
        workQueue.FreeTensor(workLocal);
    }

    __aicore__ inline void CopyOut(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        // 写出该行的最大值（标量）
        AscendC::DataCopy(yGm[r], yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_Q_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_Q_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, WORK_Q_NUM> workQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t inner;
    uint32_t tileRows;
    uint32_t rowsThisBlock;
    uint32_t startRow;
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReduceLastDim op;
    op.Init(x, y, tiling_data.totalOuter, tiling_data.inner, tiling_data.tileRows);
    op.Process();
}
"""

#include <torch/library.h>

#include <torch/extension.h>

at::Tensor max_reduction_over_a_dimension_custom_impl_npu(const at::Tensor& x, int64_t dim) {
    // 仅支持沿最后一维进行约简
    TORCH_CHECK(dim == x.dim() - 1 || dim == -1, \"Only reduction along the last dimension is supported by this custom kernel\");

    // 构造输出形状：去掉最后一维
    std::vector<int64_t> sizes = x.sizes().vec();
    if (dim < 0) dim += x.dim();
    sizes.erase(sizes.begin() + dim);

    at::Tensor result = at::empty(sizes, x.options());

    // 调用自定义算子
    EXEC_NPU_CMD(aclnnMaxReductionOverADimensionCustom, x, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"max_reduction_over_a_dimension_custom\", &max_reduction_over_a_dimension_custom_impl_npu);
