kernel_src="""
#include "kernel_operator.h"

// 本自定义核完成：LeakyReLU 后接 GELU 的逐元素计算。
// 说明：模型中的 conv2d 与逐通道乘法已在 PyTorch 端完成，核内仅对输入张量执行激活复合（LeakyReLU -> GELU）。

constexpr int32_t BUFFER_NUM = 2; // 每个队列的张量缓存数

class KernelConvMulLeakyGelu {
public:
    __aicore__ inline KernelConvMulLeakyGelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        // 基于块并行的均分切片
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // 均分到每个 Block 的元素数量（向下取整），尾块自然略去尾巴以保持示例简洁
        this->blockLength = totalLength / blockNum;
        this->tileNum = tileNum;
        // 为了降低 UB 压力，将 block 切为 tileNum * BUFFER_NUM 片
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * blockIdx, this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * blockIdx, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        // 取入队数据并分配输出本地张量
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // 先进行 LeakyReLU：zLocal = LeakyRelu(xLocal, alpha=0.01)
        // 参考基础 API 4.1.3.7 LeakyRelu：接口支持 calCount 个元素计算
        const DTYPE_Z alpha = static_cast<DTYPE_Z>(0.01f);
        AscendC::LeakyRelu(zLocal, xLocal, alpha, static_cast<int32_t>(this->tileLength));

        // 再进行 GELU：就地或非就地均可。高阶 API Gelu 支持张量空间复用
        // 这里直接以 zLocal 作为输入输出
        AscendC::Gelu(zLocal, zLocal, this->tileLength);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t blockLength = 0;
    uint32_t tileNum = 0;
    uint32_t tileLength = 0;
};

extern "C" __global__ __aicore__ void conv2d_multiply_leaky_relu_gelu_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvMulLeakyGelu op;
    op.Init(x, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版：保留 PyTorch conv2d 与逐通道乘法，将后续的 LeakyReLU + GELU 融合为 AscendC 自定义算子。
    """
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.multiplier = torch.nn.Parameter(torch.randn(multiplier_shape)) 
        self.leaky_relu_slope = 0.01

    def forward(self, x):
        # 先进行卷积与逐通道乘法（按原模型语义）
        x = self.conv(x)
        x = x * self.multiplier
        # 使用自定义核执行 LeakyReLU + GELU 融合
        x = custom_ops_lib.conv2d_multiply_leaky_relu_gelu_custom(x)
        return x
