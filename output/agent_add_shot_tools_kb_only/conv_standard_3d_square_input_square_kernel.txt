kernel_src="""
#include "kernel_operator.h"

// 说明：该内核演示实现假定 stride=1, padding=0, dilation=1，且在核内采用 1x1x1 点卷积思路（K=1）的计算路径。
// 若 K>1 的情形可在宿主侧通过其它方式展开，或在后续版本扩展。

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1;

class KernelConv3DPointwise {
public:
    __aicore__ inline KernelConv3DPointwise() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR z,
                                uint32_t totalSpatial, uint32_t cin,
                                uint32_t cout, uint32_t channelStride, uint32_t tileNum)
    {
        this->totalSpatial   = totalSpatial;    // N * D * H * W
        this->cin            = cin;
        this->cout           = cout;
        this->channelStride  = channelStride;   // D * H * W
        this->tileNum        = tileNum;

        this->blockSpatial   = totalSpatial / AscendC::GetBlockNum();
        this->tileLen        = (this->blockSpatial / tileNum / BUFFER_NUM);

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(x) + this->blockSpatial * AscendC::GetBlockIdx() * this->cin,
                            this->blockSpatial * this->cin);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(w), this->cout * this->cin); // 假设 K=1
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(z) + this->blockSpatial * AscendC::GetBlockIdx() * this->cout,
                            this->blockSpatial * this->cout);

        // UB: 输入 tile 为 [cin, tileLen]，权重为 [cout, cin]，输出向量为 [tileLen]
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->cin * this->tileLen * sizeof(DTYPE));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, this->cout * this->cin * sizeof(DTYPE));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLen * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // 拷贝权重到UB一次（K=1）
        AscendC::LocalTensor<DTYPE> wLocal = inQueueW.AllocTensor<DTYPE>();
        AscendC::DataCopy(wLocal, wGm, this->cout * this->cin);
        inQueueW.EnQue(wLocal);

        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }

        // 释放权重
        AscendC::LocalTensor<DTYPE> wLocalDeq = inQueueW.DeQue<DTYPE>();
        inQueueW.FreeTensor(wLocalDeq);
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        // 为当前 tile 准备输入：按空间维展开的起始位置
        uint32_t tileStart = progress * this->tileLen;  // 基于当前 block 的偏移
        // 把 [cin, tileLen] 连续复制到 UB：逐通道复制每个通道的连续 tileLen 片段
        AscendC::LocalTensor<DTYPE> xTile = inQueueX.AllocTensor<DTYPE>();
        uint32_t nIdx, posWithin;
        nIdx = tileStart / this->channelStride;
        posWithin = tileStart % this->channelStride;

        for (uint32_t c = 0; c < this->cin; ++c) {
            // 全局偏移：((nIdx * Cin + c) * channelStride + posWithin)
            uint32_t goff = ((nIdx * this->cin + c) * this->channelStride + posWithin);
            AscendC::DataCopy(xTile[c * this->tileLen], xGm[goff], this->tileLen);
        }
        inQueueX.EnQue(xTile);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        // 从队列取出输入与权重
        AscendC::LocalTensor<DTYPE> xTile = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> wLocal = inQueueW.DeQue<DTYPE>();
        inQueueW.EnQue(wLocal); // 权重保持在队列中

        // 逐输出通道计算 z(vec) = sum_{ci} xTile[ci,:] * w[co,ci]
        for (uint32_t co = 0; co < this->cout; ++co) {
            AscendC::LocalTensor<DTYPE> zVec = outQueueZ.AllocTensor<DTYPE>();

            // ci=0：z = x * scalar
            DTYPE w0 = wLocal[co * this->cin + 0];
            AscendC::Muls(zVec, xTile[0 * this->tileLen], w0, this->tileLen);

            // ci >= 1：z += x * scalar
            for (uint32_t ci = 1; ci < this->cin; ++ci) {
                DTYPE wsc = wLocal[co * this->cin + ci];
                AscendC::Axpy(zVec, xTile[ci * this->tileLen], wsc, this->tileLen);
            }

            // 将该输出通道的结果入队，后续 CopyOut 写回
            outQueueZ.EnQue<DTYPE>(zVec);
        }

        inQueueX.FreeTensor(xTile);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        // 写回 cout 个输出通道的向量
        uint32_t tileStart = progress * this->tileLen;
        uint32_t nIdx = tileStart / this->channelStride;
        uint32_t posWithin = tileStart % this->channelStride;

        for (uint32_t co = 0; co < this->cout; ++co) {
            AscendC::LocalTensor<DTYPE> zVec = outQueueZ.DeQue<DTYPE>();
            uint32_t goff = ((nIdx * this->cout + co) * this->channelStride + posWithin);
            AscendC::DataCopy(zGm[goff], zVec, this->tileLen);
            outQueueZ.FreeTensor(zVec);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> zGm;

    uint32_t totalSpatial;
    uint32_t cin;
    uint32_t cout;
    uint32_t channelStride;
    uint32_t tileNum;

    uint32_t blockSpatial;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv3DPointwise op;
    op.Init(x, weight, z,
            tilingData.totalSpatial,
            tilingData.cin,
            tilingData.cout,
            tilingData.channelStride,
            tilingData.tileNum);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}
