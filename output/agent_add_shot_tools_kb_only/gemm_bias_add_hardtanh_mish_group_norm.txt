kernel_src="""
#include "kernel_operator.h"

#define DTYPE_X float
#define DTYPE_W float
#define DTYPE_B float
#define DTYPE_Y float

constexpr int32_t BUFFER_NUM = 1;
constexpr float HARDTANH_MIN = -1.0f;
constexpr float HARDTANH_MAX = 1.0f;
constexpr float GN_EPS = 1e-5f;

struct FusedTiling {
    uint32_t batch;
    uint32_t inFeat;
    uint32_t outFeat;
    uint32_t numGroups;
    uint32_t groupSize;
};

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b0, GM_ADDR b1, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                const FusedTiling &t)
    {
        this->B = t.batch;
        this->K = t.inFeat;
        this->N = t.outFeat;
        this->G = t.numGroups;
        this->groupSize = t.groupSize;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint64_t>(B) * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, static_cast<uint64_t>(N) * K);
        b0Gm.SetGlobalBuffer((__gm__ DTYPE_B *)b0, N);
        b1Gm.SetGlobalBuffer((__gm__ DTYPE_B *)b1, N);
        gammaGm.SetGlobalBuffer((__gm__ DTYPE_B *)gamma, N);
        betaGm.SetGlobalBuffer((__gm__ DTYPE_B *)beta, N);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, static_cast<uint64_t>(B) * N);

        // Local buffers (per group tile)
        pipe.InitBuffer(inQueueXRow, BUFFER_NUM, this->K * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueYTile, BUFFER_NUM, this->groupSize * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueBias0Tile, BUFFER_NUM, this->groupSize * sizeof(DTYPE_B));
        pipe.InitBuffer(inQueueBias1Tile, BUFFER_NUM, this->groupSize * sizeof(DTYPE_B));
        pipe.InitBuffer(inQueueGammaTile, BUFFER_NUM, this->groupSize * sizeof(DTYPE_B));
        pipe.InitBuffer(inQueueBetaTile, BUFFER_NUM, this->groupSize * sizeof(DTYPE_B));
    }

    __aicore__ inline void Process()
    {
        // multi-core by rows
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerCore = B / blockNum;
        uint32_t remain = B % blockNum;
        uint32_t startRow = blockIdx * rowsPerCore + (blockIdx < remain ? blockIdx : remain);
        uint32_t myRows = rowsPerCore + (blockIdx < remain ? 1 : 0);

        for (uint32_t r = 0; r < myRows; ++r) {
            uint32_t rowIdx = startRow + r;
            // load x row
            AscendC::LocalTensor<DTYPE_X> xRowLocal = inQueueXRow.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xRowLocal, xGm[rowIdx * K], K);
            inQueueXRow.EnQue(xRowLocal);

            // iterate over groups (tile over N dimension)
            for (uint32_t g = 0; g < G; ++g) {
                uint32_t n0 = g * groupSize;

                // y tile buffer
                AscendC::LocalTensor<DTYPE_Y> yTileLocal = outQueueYTile.AllocTensor<DTYPE_Y>();
                // bias tiles
                AscendC::LocalTensor<DTYPE_B> b0TileLocal = inQueueBias0Tile.AllocTensor<DTYPE_B>();
                AscendC::LocalTensor<DTYPE_B> b1TileLocal = inQueueBias1Tile.AllocTensor<DTYPE_B>();
                // gn params tiles
                AscendC::LocalTensor<DTYPE_B> gammaTileLocal = inQueueGammaTile.AllocTensor<DTYPE_B>();
                AscendC::LocalTensor<DTYPE_B> betaTileLocal = inQueueBetaTile.AllocTensor<DTYPE_B>();

                AscendC::DataCopy(b0TileLocal, b0Gm[n0], groupSize);
                AscendC::DataCopy(b1TileLocal, b1Gm[n0], groupSize);
                AscendC::DataCopy(gammaTileLocal, gammaGm[n0], groupSize);
                AscendC::DataCopy(betaTileLocal, betaGm[n0], groupSize);

                // init yTile = bias0 + bias1
                AscendC::Add(yTileLocal, b0TileLocal, b1TileLocal, groupSize);

                // MatVec accumulation: yTile += W_tile * xRow
                // loop over K dimension in chunks
                const uint32_t TILE_K = 256; // chunk for K
                for (uint32_t k0 = 0; k0 < K; k0 += TILE_K) {
                    uint32_t curK = (k0 + TILE_K <= K) ? TILE_K : (K - k0);
                    // slice x chunk
                    AscendC::LocalTensor<DTYPE_X> xChunk = inQueueXRow.DeQue<DTYPE_X>();
                    AscendC::LocalTensor<DTYPE_X> xChunkLocal = pipe.AllocTmpBuffer<DTYPE_X>(curK);
                    AscendC::DataCopy(xChunkLocal, xChunk[k0], curK);
                    inQueueXRow.EnQue(xChunk); // keep row in queue for reuse

                    // for each row within tile (groupSize), accumulate dot with weight row segment
                    for (uint32_t t = 0; t < groupSize; ++t) {
                        // load weight row segment [n0 + t, k0 : k0 + curK]
                        AscendC::LocalTensor<DTYPE_W> wSegLocal = pipe.AllocTmpBuffer<DTYPE_W>(curK);
                        AscendC::DataCopy(wSegLocal, wGm[(n0 + t) * K + k0], curK);

                        // tmp = wSeg * xChunk
                        AscendC::LocalTensor<DTYPE_Y> tmpVec = pipe.AllocTmpBuffer<DTYPE_Y>(curK);
                        AscendC::Mul(tmpVec, wSegLocal, xChunkLocal, curK);

                        // reduce sum to scalar
                        AscendC::LocalTensor<DTYPE_Y> sumScalar = pipe.AllocTmpBuffer<DTYPE_Y>(1);
                        AscendC::ReduceSum(sumScalar, tmpVec, curK);

                        // add scalar to yTile[t]
                        // implement yTile[t] += sumScalar[0] via Adds on a 1-length window
                        AscendC::LocalTensor<DTYPE_Y> yElem = pipe.AllocTmpBuffer<DTYPE_Y>(1);
                        AscendC::DataCopy(yElem, yTileLocal[t], 1);
                        AscendC::Add(yElem, yElem, sumScalar, 1);
                        AscendC::DataCopy(yTileLocal[t], yElem, 1);

                        pipe.FreeTmpBuffer(sumScalar);
                        pipe.FreeTmpBuffer(tmpVec);
                        pipe.FreeTmpBuffer(wSegLocal);
                        pipe.FreeTmpBuffer(yElem);
                    }
                    pipe.FreeTmpBuffer(xChunkLocal);
                }

                // Hardtanh clamp to [-1, 1]
                AscendC::ClampMin(yTileLocal, yTileLocal, HARDTANH_MIN, groupSize);
                AscendC::ClampMax(yTileLocal, yTileLocal, HARDTANH_MAX, groupSize);

                // Mish: y = y * tanh(ln(1 + exp(y)))
                AscendC::LocalTensor<DTYPE_Y> tmpExp = pipe.AllocTmpBuffer<DTYPE_Y>(groupSize);
                AscendC::Exp(tmpExp, yTileLocal, groupSize);
                AscendC::LocalTensor<DTYPE_Y> tmpSp = pipe.AllocTmpBuffer<DTYPE_Y>(groupSize);
                // tmpSp = 1 + exp(y)
                AscendC::Adds(tmpSp, tmpExp, 1.0f, groupSize);
                AscendC::LocalTensor<DTYPE_Y> tmpLn = pipe.AllocTmpBuffer<DTYPE_Y>(groupSize);
                AscendC::Ln(tmpLn, tmpSp, groupSize);
                AscendC::LocalTensor<DTYPE_Y> tmpTanh = pipe.AllocTmpBuffer<DTYPE_Y>(groupSize);
                AscendC::Tanh(tmpTanh, tmpLn, groupSize);
                AscendC::Mul(yTileLocal, yTileLocal, tmpTanh, groupSize);
                pipe.FreeTmpBuffer(tmpExp);
                pipe.FreeTmpBuffer(tmpSp);
                pipe.FreeTmpBuffer(tmpLn);
                pipe.FreeTmpBuffer(tmpTanh);

                // GroupNorm per group: normalize yTileLocal using mean/var over group elements
                AscendC::LocalTensor<DTYPE_Y> meanScalar = pipe.AllocTmpBuffer<DTYPE_Y>(1);
                AscendC::ReduceSum(meanScalar, yTileLocal, groupSize);
                AscendC::Muls(meanScalar, meanScalar, (1.0f / static_cast<float>(groupSize)), 1);

                // y_centered = y - mean
                AscendC::LocalTensor<DTYPE_Y> yCentered = pipe.AllocTmpBuffer<DTYPE_Y>(groupSize);
                AscendC::Adds(yCentered, yTileLocal, -meanScalar.GetValue(0), groupSize);

                // var = mean((y_centered)^2)
                AscendC::LocalTensor<DTYPE_Y> ySq = pipe.AllocTmpBuffer<DTYPE_Y>(groupSize);
                AscendC::Mul(ySq, yCentered, yCentered, groupSize);
                AscendC::LocalTensor<DTYPE_Y> varScalar = pipe.AllocTmpBuffer<DTYPE_Y>(1);
                AscendC::ReduceSum(varScalar, ySq, groupSize);
                AscendC::Muls(varScalar, varScalar, (1.0f / static_cast<float>(groupSize)), 1);
                AscendC::Adds(varScalar, varScalar, GN_EPS, 1);

                // rstd = 1/sqrt(var + eps)
                AscendC::LocalTensor<DTYPE_Y> rstdScalar = pipe.AllocTmpBuffer<DTYPE_Y>(1);
                AscendC::Rsqrt(rstdScalar, varScalar, 1);

                // y_norm = (y - mean) * rstd
                AscendC::Muls(yCentered, yCentered, rstdScalar.GetValue(0), groupSize);

                // scale and shift: y = y_norm * gamma + beta
                AscendC::Mul(yCentered, yCentered, gammaTileLocal, groupSize);
                AscendC::Add(yTileLocal, yCentered, betaTileLocal, groupSize);

                // write back
                AscendC::DataCopy(yGm[rowIdx * N + n0], yTileLocal, groupSize);

                outQueueYTile.EnQue(yTileLocal);
                inQueueBias0Tile.EnQue(b0TileLocal);
                inQueueBias1Tile.EnQue(b1TileLocal);
                inQueueGammaTile.EnQue(gammaTileLocal);
                inQueueBetaTile.EnQue(betaTileLocal);

                outQueueYTile.FreeTensor(yTileLocal);
                inQueueBias0Tile.FreeTensor(b0TileLocal);
                inQueueBias1Tile.FreeTensor(b1TileLocal);
                inQueueGammaTile.FreeTensor(gammaTileLocal);
                inQueueBetaTile.FreeTensor(betaTileLocal);

                pipe.FreeTmpBuffer(meanScalar);
                pipe.FreeTmpBuffer(yCentered);
                pipe.FreeTmpBuffer(ySq);
                pipe.FreeTmpBuffer(varScalar);
                pipe.FreeTmpBuffer(rstdScalar);
            }

            AscendC::LocalTensor<DTYPE_X> xRow = inQueueXRow.DeQue<DTYPE_X>();
            inQueueXRow.FreeTensor(xRow);
        }
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueXRow;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueYTile;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueBias0Tile;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueBias1Tile;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueGammaTile;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueBetaTile;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> b0Gm;
    AscendC::GlobalTensor<DTYPE_B> b1Gm;
    AscendC::GlobalTensor<DTYPE_B> gammaGm;
    AscendC::GlobalTensor<DTYPE_B> betaGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t B;
    uint32_t K;
    uint32_t N;
    uint32_t G;
    uint32_t groupSize;
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias0, GM_ADDR bias1, GM_ADDR gamma, GM_ADDR beta,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    FusedTiling t;
    t.batch = tilingData.batch();
    t.inFeat = tilingData.inFeat();
    t.outFeat = tilingData.outFeat();
    t.numGroups = tilingData.numGroups();
    t.groupSize = tilingData.groupSize();

    KernelFused op;
    op.Init(x, weight, bias0, bias1, gamma, beta, y, t);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, bias_shape, num_groups):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.num_groups = num_groups

        # Keep the same parameters as original model to ensure weight sharing when transforming
        self.weight = torch.nn.Parameter(torch.empty(out_features, in_features))
        self.bias0 = torch.nn.Parameter(torch.empty(out_features))  # gemm bias
        self.bias1 = torch.nn.Parameter(torch.randn(bias_shape))    # extra bias add
        self.gamma = torch.nn.Parameter(torch.ones(out_features))   # groupnorm weight
        self.beta = torch.nn.Parameter(torch.zeros(out_features))   # groupnorm bias

        # Initialize like nn.Linear and nn.GroupNorm defaults
        torch.nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        fan_in = in_features
        bound = 1 / math.sqrt(fan_in)
        torch.nn.init.uniform_(self.bias0, -bound, bound)
        # bias1 is already randomized above

    def forward(self, x):
        # Fused kernel call: y = GroupNorm(Mish(Hardtanh(x @ W^T + bias0) ), gamma, beta) + and extra bias1 are fused inside
        return custom_ops_lib.gemm_bias_add_hardtanh_mish_group_norm_custom(
            x, self.weight, self.bias0, self.bias1, self.gamma, self.beta, self.num_groups
        )
