#include "downsample_bilinear_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    DownsampleBilinearCustomTilingData tiling;

    const gert::StorageShape* in_shape = context->GetInputShape(0);
    // Expect ND: [N, C, H, W]
    uint32_t n = in_shape->GetStorageShape().GetDim(0);
    uint32_t c = in_shape->GetStorageShape().GetDim(1);
    uint32_t inH = in_shape->GetStorageShape().GetDim(2);
    uint32_t inW = in_shape->GetStorageShape().GetDim(3);

    // Fixed target size as per model: (60, 80)
    const uint32_t outH = 60;
    const uint32_t outW = 80;

    uint32_t plane = n * c;
    // Simple block strategy
    uint32_t blockDim = plane < 8 ? plane : 8;
    blockDim = blockDim == 0 ? 1 : blockDim;
    context->SetBlockDim(blockDim);
    uint32_t planePerBlock = (plane + blockDim - 1) / blockDim;

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_plane(plane);
    tiling.set_planePerBlock(planePerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);
    // Keep N, C, but set H=60, W=80
    std::vector<int64_t> dims = in_shape->GetStorageShape().GetDims();
    if (dims.size() < 4) {
        return GRAPH_FAILED;
    }
    dims[2] = 60;
    dims[3] = 80;
    gert::Shape ret;
    ret.SetDimNum(dims.size());
    for (size_t i = 0; i < dims.size(); ++i) {
        ret.SetDim(i, dims[i]);
    }
    *out_shape = ret;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

class KernelDownsampleBilinear {
public:
    __aicore__ inline KernelDownsampleBilinear() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW, uint32_t plane,
                                uint32_t planePerBlock)
    {
        this->n = n;
        this->c = c;
        this->inH = inH;
        this->inW = inW;
        this->outH = outH;
        this->outW = outW;
        this->plane = plane;

        // Determine plane range for this block
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startPlane = blockIdx * planePerBlock;
        if (startPlane > plane) startPlane = plane;
        uint32_t endPlane = startPlane + planePerBlock;
        if (endPlane > plane) endPlane = plane;
        this->startPlane = startPlane;
        this->planeCount = (endPlane > startPlane) ? (endPlane - startPlane) : 0;

        xGm.SetGlobalBuffer((__gm__ float*)x + (uint64_t)startPlane * (uint64_t)inH * (uint64_t)inW);
        yGm.SetGlobalBuffer((__gm__ float*)y + (uint64_t)startPlane * (uint64_t)outH * (uint64_t)outW);

        // Allocate buffers: two input rows (inW) and one output row (outW)
        pipe.InitBuffer(inRowQueue0, 1, inW * sizeof(float));
        pipe.InitBuffer(inRowQueue1, 1, inW * sizeof(float));
        pipe.InitBuffer(outRowQueue, 1, outW * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (planeCount == 0) {
            return;
        }
        const float scaleH = static_cast<float>(inH) / static_cast<float>(outH);
        const float scaleW = static_cast<float>(inW) / static_cast<float>(outW);

        for (uint32_t p = 0; p < planeCount; ++p) {
            for (uint32_t oh = 0; oh < outH; ++oh) {
                float inY = (static_cast<float>(oh) + 0.5f) * scaleH - 0.5f;
                int32_t y0 = static_cast<int32_t>(floorf(inY));
                y0 = y0 < 0 ? 0 : (y0 >= (int32_t)inH ? (int32_t)inH - 1 : y0);
                int32_t y1 = y0 + 1; if (y1 >= (int32_t)inH) y1 = (int32_t)inH - 1;
                float wy1 = inY - static_cast<float>(y0);
                float wy0 = 1.0f - wy1;

                // Load two source rows for this plane
                AscendC::LocalTensor<float> row0 = inRowQueue0.AllocTensor<float>();
                AscendC::LocalTensor<float> row1 = inRowQueue1.AllocTensor<float>();
                uint64_t planeInStride = (uint64_t)inH * (uint64_t)inW;
                AscendC::DataCopy(row0, xGm[p * planeInStride + (uint64_t)y0 * (uint64_t)inW], inW);
                AscendC::DataCopy(row1, xGm[p * planeInStride + (uint64_t)y1 * (uint64_t)inW], inW);

                AscendC::LocalTensor<float> outRow = outRowQueue.AllocTensor<float>();
                // Compute bilinear along width
                for (uint32_t ow = 0; ow < outW; ++ow) {
                    float inX = (static_cast<float>(ow) + 0.5f) * scaleW - 0.5f;
                    int32_t x0 = static_cast<int32_t>(floorf(inX));
                    x0 = x0 < 0 ? 0 : (x0 >= (int32_t)inW ? (int32_t)inW - 1 : x0);
                    int32_t x1 = x0 + 1; if (x1 >= (int32_t)inW) x1 = (int32_t)inW - 1;
                    float wx1 = inX - static_cast<float>(x0);
                    float wx0 = 1.0f - wx1;

                    float v00 = row0[x0];
                    float v01 = row0[x1];
                    float v10 = row1[x0];
                    float v11 = row1[x1];

                    float top = v00 * wx0 + v01 * wx1;
                    float bot = v10 * wx0 + v11 * wx1;
                    float val = top * wy0 + bot * wy1;
                    outRow[ow] = val;
                }

                // Store output row
                uint64_t planeOutStride = (uint64_t)outH * (uint64_t)outW;
                AscendC::DataCopy(yGm[p * planeOutStride + (uint64_t)oh * (uint64_t)outW], outRow, outW);

                inRowQueue0.FreeTensor(row0);
                inRowQueue1.FreeTensor(row1);
                outRowQueue.FreeTensor(outRow);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inRowQueue0;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inRowQueue1;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outRowQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n, c, inH, inW, outH, outW, plane;
    uint32_t startPlane;
    uint32_t planeCount;
};

extern "C" __global__ __aicore__ void downsample_bilinear_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelDownsampleBilinear op;
    op.Init(x, y,
            tilingData.n(), tilingData.c(),
            tilingData.inH(), tilingData.inW(),
            tilingData.outH(), tilingData.outW(),
            tilingData.plane(), tilingData.planePerBlock());
    op.Process();
}
"""
