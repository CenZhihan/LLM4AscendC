host_operator_src="""
#include "square_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"
#include <cmath>
#include <cstring>

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto *in0 = context->GetInputShape(0);
    auto *in1 = context->GetInputShape(1);
    if (in0 == nullptr || in1 == nullptr) {
        return ge::GRAPH_FAILED;
    }
    const gert::Shape &sA = in0->GetOriginShape();
    const gert::Shape &sB = in1->GetOriginShape();
    if (sA.GetDimNum() != 2 || sB.GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    int64_t M = sA.GetDim(0);
    int64_t K = sA.GetDim(1);
    int64_t KB = sB.GetDim(0);
    int64_t N = sB.GetDim(1);
    if (K != KB || M <= 0 || N <= 0) {
        return ge::GRAPH_FAILED;
    }

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);
    tiling.SetDim(1);
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBias(false);
    tiling.SetShape(static_cast<int>(M), static_cast<int>(N), static_cast<int>(K));
    tiling.SetSingleShape(static_cast<int>(M), static_cast<int>(N), static_cast<int>(K));
    tiling.SetOrgShape(static_cast<int>(M), static_cast<int>(N), static_cast<int>(K));
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tcube;
    int ret = tiling.GetTiling(tcube);
    if (ret == -1) {
        return ge::GRAPH_FAILED;
    }
    context->SetBlockDim(tiling.GetCoreNum());

    auto *raw = context->GetRawTilingData();
    if (raw == nullptr) {
        return ge::GRAPH_FAILED;
    }
    size_t need = sizeof(optiling::TCubeTiling);
    if (raw->GetCapacity() < need) {
        return ge::GRAPH_FAILED;
    }
    std::memcpy(raw->GetData(), &tcube, need);
    raw->SetDataSize(need);

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return GRAPH_FAILED;
    }
    const gert::Shape &sa = a_shape->GetOriginShape();
    const gert::Shape &sb = b_shape->GetOriginShape();
    if (sa.GetDimNum() != 2 || sb.GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    if (sa.GetDim(1) != sb.GetDim(0)) {
        return GRAPH_FAILED;
    }
    gert::Shape* out_shape = context->GetOutputShape(0);
    out_shape->SetDimNum(2);
    out_shape->SetDim(0, sa.GetDim(0));
    out_shape->SetDim(1, sb.GetDim(1));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class SquareMatrixMultiplicationCustom : public OpDef {
public:
    explicit SquareMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SquareMatrixMultiplicationCustom);
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR AGM, GM_ADDR BGM, GM_ADDR CGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    set_atomic_none();

    // Load TCubeTiling from GM
    TCubeTiling tiling;
    auto gmPtr = (__gm__ uint32_t*)tilingGM;
    auto localPtr = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++gmPtr, ++localPtr) {
        *localPtr = *gmPtr;
    }

    // Define Matmul types for NORMAL layout: A/B in FP16, C in FP32
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half,  false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half,  true,  LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    // Prepare GM tensors
    AscendC::GlobalTensor<half> aGm;
    AscendC::GlobalTensor<half> bGm;
    AscendC::GlobalTensor<float> cGm;

    // Estimate sizes from tiling (fields exist in TCubeTiling)
    int32_t sizeA = tiling.ALayoutInfoB * tiling.ALayoutInfoS * tiling.ALayoutInfoN * tiling.ALayoutInfoG * tiling.ALayoutInfoD;
    int32_t sizeB = tiling.BLayoutInfoB * tiling.BLayoutInfoS * tiling.BLayoutInfoN * tiling.BLayoutInfoG * tiling.BLayoutInfoD;
    // For C, use N and M related layout infos
    int32_t sizeC = tiling.CLayoutInfoB * tiling.CLayoutInfoN * tiling.CLayoutInfoG * tiling.CLayoutInfoS2;

    aGm.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(AGM), sizeA);
    bGm.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(BGM), sizeB);
    cGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(CGM), sizeC);

    // Create Matmul instance
    matmul::Matmul<aType, bType, cType, biasType> mm;
    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
    mm.Init(&tiling);

    // Set input/output tensors
    mm.SetTensorA(aGm[0], false);
    mm.SetTensorB(bGm[0], false);
    // Set output (if required by interface)
#ifdef MATMUL_HAS_SETTENSORC
    mm.SetTensorC(cGm[0]);
#endif
    mm.SetWorkspace(workspaceGM, 0);

    // Execute single matmul as a single-batch BMM
    mm.IterateNBatch(1, 1, 1, false);
}
