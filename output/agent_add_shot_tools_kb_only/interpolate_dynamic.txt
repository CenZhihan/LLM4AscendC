kernel_src="""
#include "kernel_operator.h"

// 简化实现：NCHW，float32，双线性插值，align_corners=false

class KernelInterpolateDynamic {
public:
    __aicore__ inline KernelInterpolateDynamic() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR size, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t hin, uint32_t win,
                                uint32_t hout, uint32_t wout)
    {
        (void)size; // size通过host tiling传递给hout/wout，kernel不直接读取size张量
        this->n = n;
        this->c = c;
        this->hin = hin;
        this->win = win;
        this->hout = hout;
        this->wout = wout;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n * c * hin * win);
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n * c * hout * wout);

        pipe.InitBuffer(tmpQueue, 1, 8 * sizeof(float)); // 临时1元素缓冲复用
    }

    __aicore__ inline void Process()
    {
        uint32_t nc_total = n * c;
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t block_idx = AscendC::GetBlockIdx();
        uint32_t nc_per_block = (nc_total + blocks - 1) / blocks;
        uint32_t nc_start = block_idx * nc_per_block;
        uint32_t nc_end = (nc_start + nc_per_block) > nc_total ? nc_total : (nc_start + nc_per_block);

        float scale_h = static_cast<float>(hout) / static_cast<float>(hin);
        float scale_w = static_cast<float>(wout) / static_cast<float>(win);

        for (uint32_t nc_idx = nc_start; nc_idx < nc_end; ++nc_idx) {
            uint32_t n_idx = nc_idx / c;
            uint32_t c_idx = nc_idx % c;

            uint64_t in_base_nc  = ((uint64_t)n_idx * c + c_idx) * (uint64_t)hin * win;
            uint64_t out_base_nc = ((uint64_t)n_idx * c + c_idx) * (uint64_t)hout * wout;

            for (uint32_t oh = 0; oh < hout; ++oh) {
                float in_h_f = ((static_cast<float>(oh) + 0.5f) / scale_h) - 0.5f;
                int32_t h0 = static_cast<int32_t>(AscendC::Floor(in_h_f));
                float wy = in_h_f - static_cast<float>(h0);
                if (h0 < 0) { h0 = 0; wy = 0.0f; }
                if (h0 >= static_cast<int32_t>(hin - 1)) { h0 = hin - 1; wy = 0.0f; }
                int32_t h1 = h0 + 1;
                if (h1 >= static_cast<int32_t>(hin)) { h1 = hin - 1; }

                uint64_t row0_base = in_base_nc + (uint64_t)h0 * win;
                uint64_t row1_base = in_base_nc + (uint64_t)h1 * win;

                for (uint32_t ow = 0; ow < wout; ++ow) {
                    float in_w_f = ((static_cast<float>(ow) + 0.5f) / scale_w) - 0.5f;
                    int32_t w0 = static_cast<int32_t>(AscendC::Floor(in_w_f));
                    float wx = in_w_f - static_cast<float>(w0);
                    if (w0 < 0) { w0 = 0; wx = 0.0f; }
                    if (w0 >= static_cast<int32_t>(win - 1)) { w0 = win - 1; wx = 0.0f; }
                    int32_t w1 = w0 + 1;
                    if (w1 >= static_cast<int32_t>(win)) { w1 = win - 1; }

                    // 读取四邻域像素到UB (每次1元素)
                    AscendC::LocalTensor<float> p00 = tmpQueue.AllocTensor<float>();
                    AscendC::LocalTensor<float> p01 = tmpQueue.AllocTensor<float>();
                    AscendC::LocalTensor<float> p10 = tmpQueue.AllocTensor<float>();
                    AscendC::LocalTensor<float> p11 = tmpQueue.AllocTensor<float>();
                    AscendC::DataCopy(p00, xGm[row0_base + (uint64_t)w0], 1);
                    AscendC::DataCopy(p01, xGm[row0_base + (uint64_t)w1], 1);
                    AscendC::DataCopy(p10, xGm[row1_base + (uint64_t)w0], 1);
                    AscendC::DataCopy(p11, xGm[row1_base + (uint64_t)w1], 1);

                    // 计算 v0 = p00 + (p01 - p00) * wx
                    AscendC::LocalTensor<float> tmpA = tmpQueue.AllocTensor<float>();
                    AscendC::LocalTensor<float> v0   = tmpQueue.AllocTensor<float>();
                    AscendC::Sub(tmpA, p01, p00, 1);
                    AscendC::Muls(tmpA, tmpA, wx, 1);
                    AscendC::Add(v0, p00, tmpA, 1);

                    // 计算 v1 = p10 + (p11 - p10) * wx
                    AscendC::LocalTensor<float> tmpB = tmpQueue.AllocTensor<float>();
                    AscendC::LocalTensor<float> v1   = tmpQueue.AllocTensor<float>();
                    AscendC::Sub(tmpB, p11, p10, 1);
                    AscendC::Muls(tmpB, tmpB, wx, 1);
                    AscendC::Add(v1, p10, tmpB, 1);

                    // out = v0 + (v1 - v0) * wy
                    AscendC::LocalTensor<float> tmpC = tmpQueue.AllocTensor<float>();
                    AscendC::LocalTensor<float> outV = tmpQueue.AllocTensor<float>();
                    AscendC::Sub(tmpC, v1, v0, 1);
                    AscendC::Muls(tmpC, tmpC, wy, 1);
                    AscendC::Add(outV, v0, tmpC, 1);

                    // 写回输出
                    AscendC::DataCopy(yGm[out_base_nc + (uint64_t)oh * wout + ow], outV, 1);

                    // 释放临时tensor
                    tmpQueue.FreeTensor(outV);
                    tmpQueue.FreeTensor(tmpC);
                    tmpQueue.FreeTensor(v1);
                    tmpQueue.FreeTensor(tmpB);
                    tmpQueue.FreeTensor(v0);
                    tmpQueue.FreeTensor(tmpA);
                    tmpQueue.FreeTensor(p11);
                    tmpQueue.FreeTensor(p10);
                    tmpQueue.FreeTensor(p01);
                    tmpQueue.FreeTensor(p00);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n, c, hin, win, hout, wout;
};

extern "C" __global__ __aicore__ void interpolate_dynamic_custom(GM_ADDR x, GM_ADDR size, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelInterpolateDynamic op;
    op.Init(x, size, y,
            tiling_data.n(),
            tiling_data.c(),
            tiling_data.hin(),
            tiling_data.win(),
            tiling_data.hout(),
            tiling_data.wout());
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor interpolate_dynamic_custom_impl_npu(const at::Tensor& self, const at::Tensor& size) {
    TORCH_CHECK(size.dim() == 1 && size.size(0) == 2, \\"size must be a 1D tensor with length 2\\");
    // 从size读取目标高宽
    auto size_cpu = size.to(torch::kCPU);
    int64_t hout = size_cpu[0].item<int32_t>();
    int64_t wout = size_cpu[1].item<int32_t>();

    auto out_sizes = self.sizes().vec();
    TORCH_CHECK(self.dim() == 4, \\"expect 4D NCHW\\");
    out_sizes[2] = hout;
    out_sizes[3] = wout;

    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnInterpolateDynamicCustom, self, size, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\\"interpolate_dynamic_custom\\", &interpolate_dynamic_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\\"interpolate_dynamic_custom\\", &interpolate_dynamic_custom_impl_npu, \\"bilinear interpolate (align_corners=False)\\");
}
