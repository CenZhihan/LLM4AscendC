host_operator_src="""
#include "register/op_def_registry.h"
#include "lib/matmul_intf.h"
#include <cstring>

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 获取输入形状：A(M,K), B(K,N)
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }
    auto a_origin = a_shape->GetOriginShape();
    auto b_origin = b_shape->GetOriginShape();
    if (a_origin.GetDimNum() != 2 || b_origin.GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    uint32_t M = static_cast<uint32_t>(a_origin.GetDim(0));
    uint32_t K = static_cast<uint32_t>(a_origin.GetDim(1));
    uint32_t Kb = static_cast<uint32_t>(b_origin.GetDim(0));
    uint32_t N = static_cast<uint32_t>(b_origin.GetDim(1));
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    context->SetBlockDim(8); // 简单设定 block 数

    // 构造 TCubeTiling（NORMAL 模式），按整矩阵一次 GEMM
    TCubeTiling tiling;
    std::memset(&tiling, 0, sizeof(TCubeTiling));
    tiling.singleCoreM = M;
    tiling.singleCoreK = K;
    tiling.singleCoreN = N;
    tiling.ALayoutInfoB = 1;
    tiling.BLayoutInfoB = 1;
    tiling.CLayoutInfoB = 1;
    tiling.isBias = 0;
    tiling.shareMode = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize = 0;

    // 写入原始 tiling 缓冲
    auto rawTiling = context->GetRawTilingData();
    if (rawTiling->GetCapacity() < sizeof(TCubeTiling)) {
        return ge::GRAPH_FAILED;
    }
    std::memcpy(rawTiling->GetData(), &tiling, sizeof(TCubeTiling));
    rawTiling->SetDataSize(sizeof(TCubeTiling));

    // 无需 workspace
    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return GRAPH_FAILED;
    }
    auto a_origin = a_shape->GetOriginShape();
    auto b_origin = b_shape->GetOriginShape();
    if (a_origin.GetDimNum() != 2 || b_origin.GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    if (a_origin.GetDim(1) != b_origin.GetDim(0)) {
        return GRAPH_FAILED;
    }
    gert::Shape* c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, a_origin.GetDim(0));
    c_shape->SetDim(1, b_origin.GetDim(1));
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {

public:
    explicit MatmulWithSmallKDimensionCustom(const char* name) : OpDef(name)

        this->Input("a")

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);

    }

OP_ADD(MatmulWithSmallKDimensionCustom);
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // 将 TCubeTiling 从 GM 复制到本地
    TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // GM 张量封装（float）
    AscendC::GlobalTensor<float> aGlobal;
    AscendC::GlobalTensor<float> bGlobal;
    AscendC::GlobalTensor<float> cGlobal;

    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(float);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(float);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(float);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cGM), sizeC);

    // Matmul 类型（NORMAL 布局，float）
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    // 设置 GM 子张量（单次计算完整矩阵）
    AscendC::GlobalTensor<float> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ float*>(aGlobal.GetPhyAddr()), tiling.singleCoreM * tiling.singleCoreK);
    AscendC::GlobalTensor<float> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ float*>(bGlobal.GetPhyAddr()), tiling.singleCoreK * tiling.singleCoreN);
    AscendC::GlobalTensor<float> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ float*>(cGlobal.GetPhyAddr()), tiling.singleCoreM * tiling.singleCoreN);

    // Matmul 实例
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;

    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);
    mm1.Init(&tiling);
    mm1.SetTensorA(gm_a, false);
    mm1.SetTensorB(gm_b, false);
    mm1.SetWorkspace(workspaceGM, 0);

    // 执行计算（单批次）
    mm1.IterateBatch(gm_c, 1, 1, false);
}
"""
