kernel_src="""
#include "kernel_operator.h"

// 本kernel对输入x执行: y = scale * Hardtanh(mish(x) + addValue), 其中 mish(x)=x * tanh(log(1+exp(x)))
// 需要用到的API: Exp, Ln, Tanh(高阶API，需临时缓冲), Mul, Adds, Maxs, Mins, Muls, DataCopy

constexpr int32_t BUFFER_NUM = 2;

class KernelFusedPost {
public:
    __aicore__ inline KernelFusedPost() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, uint32_t tmpBufferSize, float addValue, float scaleValue)
    {
        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        ASSERT(tileNum != 0 && "tile num can not be zero!");
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        this->addValue = addValue;
        this->scaleValue = scaleValue;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // 输入/输出队列
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));

        // Tanh高阶API需要的临时缓冲(sharedTmpBuffer)
        pipe.InitBuffer(tmpU8Queue, 1, tmpBufferSize);

        // 计算中间结果缓冲（例如tanh结果）
        pipe.InitBuffer(tmpCalcQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        // 预分配一次sharedTmpBuffer
        sharedTmpBuffer = tmpU8Queue.AllocTensor<uint8_t>();

        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }

        tmpU8Queue.FreeTensor(sharedTmpBuffer);
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_X> tmpLocal = tmpCalcQueue.AllocTensor<DTYPE_X>();

        // 1) t0 = exp(x)
        AscendC::Exp(yLocal, xLocal, this->tileLength); // yLocal = exp(x)

        // 2) t0 = t0 + 1
        AscendC::Adds(yLocal, yLocal, (DTYPE_Z)1.0f, this->tileLength); // yLocal = exp(x) + 1

        // 3) t0 = ln(t0)  -> softplus(x)
        AscendC::Ln(yLocal, yLocal, this->tileLength); // yLocal = log(1 + exp(x))

        // 4) t1 = tanh(t0)
        AscendC::Tanh(tmpLocal, yLocal, sharedTmpBuffer, this->tileLength); // tmpLocal = tanh(softplus(x))

        // 5) y = x * t1 -> mish(x)
        AscendC::Mul(yLocal, xLocal, tmpLocal, this->tileLength); // yLocal = x * tanh(softplus(x))

        // 6) y = y + addValue
        AscendC::Adds(yLocal, yLocal, (DTYPE_Z)this->addValue, this->tileLength);

        // 7) y = hardtanh(y, -1, 1) -> 先下限再上限
        AscendC::Maxs(yLocal, yLocal, (DTYPE_Z)(-1.0f), this->tileLength);
        AscendC::Mins(yLocal, yLocal, (DTYPE_Z)(1.0f), this->tileLength);

        // 8) y = y * scaleValue
        AscendC::Muls(yLocal, yLocal, (DTYPE_Z)this->scaleValue, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        tmpCalcQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpCalcQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpU8Queue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    AscendC::LocalTensor<uint8_t> sharedTmpBuffer;

    float addValue;
    float scaleValue;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose2d_mish_add_hardtanh_scaling_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelFusedPost op;
    op.Init(x, y, tilingData.totalLength, tilingData.tileNum, tilingData.tmpBufferSize, tilingData.addValue, tilingData.scaleValue);
    op.Process();
}

import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版：保留PyTorch的ConvTranspose2d，实现后续 mish -> add -> hardtanh -> scaling 的融合AscendC算子
    """
