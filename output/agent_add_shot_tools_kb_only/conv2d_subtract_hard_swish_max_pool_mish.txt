kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 1;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR subtract, GM_ADDR y, uint32_t n, uint32_t c, uint32_t h, uint32_t w)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->ho = h / 2;
        this->wo = w / 2;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x);
        subGm.SetGlobalBuffer((__gm__ DTYPE*)subtract);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y);

        // 行缓冲区、临时缓冲区和输出行缓冲区
        pipe.InitBuffer(inRow0Q, BUFFER_NUM, w * sizeof(DTYPE));
        pipe.InitBuffer(inRow1Q, BUFFER_NUM, w * sizeof(DTYPE));
        pipe.InitBuffer(tmpQ0, BUFFER_NUM, w * sizeof(DTYPE));
        pipe.InitBuffer(tmpQ1, BUFFER_NUM, w * sizeof(DTYPE));

        pipe.InitBuffer(outRowQ, BUFFER_NUM, (w/2) * sizeof(DTYPE));
        pipe.InitBuffer(halfQ0, BUFFER_NUM, (w/2) * sizeof(DTYPE));
        pipe.InitBuffer(halfQ1, BUFFER_NUM, (w/2) * sizeof(DTYPE));
        pipe.InitBuffer(halfQ2, BUFFER_NUM, (w/2) * sizeof(DTYPE));

        // 读取减常量（shape 为 [1]）
        AscendC::LocalTensor<DTYPE> subLocal = tmpQ0.AllocTensor<DTYPE>();
        AscendC::DataCopy(subLocal, subGm, 1);
        subtractScalar = subLocal.GetValue(0);
        tmpQ0.FreeTensor(subLocal);
    }

    __aicore__ inline void Process()
    {
        // 遍历 N、C、Ho 行
        for (uint32_t ni = 0; ni < n; ++ni) {
            for (uint32_t ci = 0; ci < c; ++ci) {
                for (uint32_t hoi = 0; hoi < ho; ++hoi) {
                    CopyIn(ni, ci, hoi);
                    Compute(ni, ci, hoi);
                    CopyOut(ni, ci, hoi);
                }
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t ni, uint32_t ci, uint32_t hoi)
    {
        const uint64_t baseInput = (((uint64_t)ni * c + ci) * h + (hoi * 2)) * w;
        AscendC::LocalTensor<DTYPE> row0 = inRow0Q.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> row1 = inRow1Q.AllocTensor<DTYPE>();
        AscendC::DataCopy(row0, xGm[baseInput], w);
        AscendC::DataCopy(row1, xGm[baseInput + w], w);
        inRow0Q.EnQue(row0);
        inRow1Q.EnQue(row1);
    }

    // 矢量化实现：减常量 -> HardSwish -> 2x2 s=2 MaxPool -> Mish
    __aicore__ inline void Compute(uint32_t, uint32_t, uint32_t)
    {
        AscendC::LocalTensor<DTYPE> row0 = inRow0Q.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> row1 = inRow1Q.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> tmp0 = tmpQ0.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> tmp1 = tmpQ1.AllocTensor<DTYPE>();

        // x = x - subtractScalar
        AscendC::Duplicate(tmp0, subtractScalar, w);
        AscendC::Sub(row0, row0, tmp0, w);
        AscendC::Sub(row1, row1, tmp0, w);

        // HardSwish: x * relu6(x + 3) / 6
        // 处理 row0
        AscendC::Duplicate(tmp0, 3.0f, w);
        AscendC::Add(tmp1, row0, tmp0, w);               // x + 3
        AscendC::Duplicate(tmp0, 0.0f, w);
        AscendC::Max(tmp1, tmp1, tmp0, w);               // max(x+3, 0)
        AscendC::Duplicate(tmp0, 6.0f, w);
        AscendC::Min(tmp1, tmp1, tmp0, w);               // min(..., 6)
        AscendC::Mul(row0, row0, tmp1, w);               // x * relu6(x+3)
        AscendC::Duplicate(tmp0, 6.0f, w);
        AscendC::Div(row0, row0, tmp0, w);               // / 6

        // 处理 row1
        AscendC::Duplicate(tmp0, 3.0f, w);
        AscendC::Add(tmp1, row1, tmp0, w);
        AscendC::Duplicate(tmp0, 0.0f, w);
        AscendC::Max(tmp1, tmp1, tmp0, w);
        AscendC::Duplicate(tmp0, 6.0f, w);
        AscendC::Min(tmp1, tmp1, tmp0, w);
        AscendC::Mul(row1, row1, tmp1, w);
        AscendC::Duplicate(tmp0, 6.0f, w);
        AscendC::Div(row1, row1, tmp0, w);

        // 2x2 s=2 MaxPool 沿宽度方向合并
        AscendC::LocalTensor<DTYPE> outRow = outRowQ.AllocTensor<DTYPE>();
        const uint32_t halfW = w / 2;
        for (uint32_t i = 0; i < halfW; ++i) {
            DTYPE a0 = row0.GetValue(2 * i);
            DTYPE a1 = row0.GetValue(2 * i + 1);
            DTYPE b0 = row1.GetValue(2 * i);
            DTYPE b1 = row1.GetValue(2 * i + 1);
            DTYPE m0 = a0 > a1 ? a0 : a1;
            DTYPE m1 = b0 > b1 ? b0 : b1;
            DTYPE m  = m0 > m1 ? m0 : m1;
            outRow.SetValue(i, m);
        }

        // Mish: x * tanh(softplus(x)), softplus(x)=ln(1+exp(x))
        AscendC::LocalTensor<DTYPE> half0 = halfQ0.AllocTensor<DTYPE>();  // 保存原始 pooled 行
        AscendC::LocalTensor<DTYPE> half1 = halfQ1.AllocTensor<DTYPE>();  // 中间 buffer
        AscendC::LocalTensor<DTYPE> half2 = halfQ2.AllocTensor<DTYPE>();  // 中间 buffer

        AscendC::DataCopy(half0, outRow, halfW);                         // half0 = pooled

        AscendC::Exp(half1, half0, halfW);                                // half1 = exp(x)
        AscendC::Duplicate(half2, 1.0f, halfW);
        AscendC::Add(half1, half1, half2, halfW);                         // half1 = exp(x) + 1
        AscendC::Ln(half1, half1, halfW);                                 // half1 = softplus(x)
        AscendC::Add(half2, half1, half1, halfW);                         // half2 = 2 * softplus
        AscendC::Exp(half2, half2, halfW);                                // half2 = a = exp(2u)

        AscendC::LocalTensor<DTYPE> aCopy = halfQ1.AllocTensor<DTYPE>();  // 复用一个半宽缓存保存 a 拷贝
        AscendC::DataCopy(aCopy, half2, halfW);                           // aCopy = a

        AscendC::Duplicate(half1, 1.0f, halfW);
        AscendC::Sub(half2, half2, half1, halfW);                         // half2 = a - 1 (numerator)
        AscendC::Add(aCopy, aCopy, half1, halfW);                         // aCopy = a + 1 (denominator)

        AscendC::Div(half2, half2, aCopy, halfW);                         // half2 = tanh(u)
        AscendC::Mul(outRow, half0, half2, halfW);                        // outRow = x * tanh(u)

        // 回收临时块
        halfQ1.FreeTensor(aCopy);
        halfQ2.FreeTensor(half2);
        halfQ1.FreeTensor(half1);
        halfQ0.FreeTensor(half0);

        // 入队/回收输入
        outRowQ.EnQue(outRow);
        tmpQ1.FreeTensor(tmp1);
        tmpQ0.FreeTensor(tmp0);
        inRow1Q.FreeTensor(row1);
        inRow0Q.FreeTensor(row0);
    }

    __aicore__ inline void CopyOut(uint32_t ni, uint32_t ci, uint32_t hoi)
    {
        AscendC::LocalTensor<DTYPE> outRow = outRowQ.DeQue<DTYPE>();
        const uint64_t baseOut = (((uint64_t)ni * c + ci) * ho + hoi) * wo;
        AscendC::DataCopy(yGm[baseOut], outRow, wo);
        outRowQ.FreeTensor(outRow);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inRow0Q, inRow1Q;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQ0, tmpQ1;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outRowQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> halfQ0, halfQ1, halfQ2;

    AscendC::GlobalTensor<DTYPE> xGm, subGm, yGm;

    uint32_t n = 0, c = 0, h = 0, w = 0, ho = 0, wo = 0;
    DTYPE subtractScalar = 0.0f;
};

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(GM_ADDR x, GM_ADDR subtract, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, subtract, y, tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w);
    op.Process();
}

import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    使用自定义AscendC核将卷积后的 subtract + HardSwish + MaxPool2d(kernel=2,stride=2) + Mish 融合加速
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract_value, pool_kernel_size):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract_value = float(subtract_value)
        self.pool_kernel_size = int(pool_kernel_size)
        assert self.pool_kernel_size == 2, "该自定义核当前仅支持 MaxPool2d(kernel=2, stride=2)"

    def forward(self, x):
        x = self.conv(x)
        sub = torch.tensor(self.subtract_value, dtype=x.dtype, device=x.device)
        x = custom_ops_lib.conv2d_subtract_hard_swish_max_pool_mish_custom(x, sub)
