kernel_src="""
#include "kernel_operator.h"

using dtype = float;

class KernelConv3D {
public:
    __aicore__ inline KernelConv3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t outD, uint32_t outH, uint32_t outW)
    {
        this->n = n; this->cin = cin; this->cout = cout;
        this->inD = inD; this->inH = inH; this->inW = inW;
        this->kD = kD; this->kH = kH; this->kW = kW;
        this->outD = outD; this->outH = outH; this->outW = outW;

        uint64_t xSize = (uint64_t)n * cin * inD * inH * inW;
        uint64_t wSize = (uint64_t)cout * cin * kD * kH * kW;
        uint64_t ySize = (uint64_t)n * cout * outD * outH * outW;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ dtype *>(x), xSize);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ dtype *>(w), wSize);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ dtype *>(y), ySize);

        tileW = outW;

        pipe.InitBuffer(inQueueX, 1, tileW * sizeof(dtype));
        pipe.InitBuffer(tmpQueue, 1, 16); // for one scalar weight
        pipe.InitBuffer(outQueueY, 1, tileW * sizeof(dtype));
    }

    __aicore__ inline void Process()
    {
        int32_t blockIdx = AscendC::GetBlockIdx();
        int32_t blockNum = AscendC::GetBlockNum();

        uint32_t ocPerBlock = (cout + blockNum - 1) / blockNum;
        uint32_t ocBegin = blockIdx * ocPerBlock;
        uint32_t ocEnd = ocBegin + ocPerBlock;
        if (ocEnd > cout) ocEnd = cout;

        for (uint32_t oc = ocBegin; oc < ocEnd; ++oc) {
            for (uint32_t bn = 0; bn < n; ++bn) {
                for (uint32_t od = 0; od < outD; ++od) {
                    for (uint32_t oh = 0; oh < outH; ++oh) {
                        AscendC::LocalTensor<dtype> yLocal = outQueueY.AllocTensor<dtype>();
                        AscendC::Duplicate<dtype>(yLocal, static_cast<dtype>(0), tileW);

                        for (uint32_t ic = 0; ic < cin; ++ic) {
                            for (uint32_t kd = 0; kd < kD; ++kd) {
                                for (uint32_t kh = 0; kh < kH; ++kh) {
                                    for (uint32_t kw = 0; kw < kW; ++kw) {
                                        // Load input line of length outW
                                        AscendC::LocalTensor<dtype> xLocal = inQueueX.AllocTensor<dtype>();
                                        uint64_t xoff = ((((uint64_t)bn * cin + ic) * inD + (od + kd)) * inH + (oh + kh)) * inW + kw;
                                        AscendC::DataCopy(xLocal, xGm[xoff], tileW);
                                        inQueueX.EnQue(xLocal);
                                        AscendC::LocalTensor<dtype> xUse = inQueueX.DeQue<dtype>();

                                        // Load weight scalar
                                        AscendC::LocalTensor<dtype> wLocal = tmpQueue.AllocTensor<dtype>();
                                        uint64_t woff = ((((uint64_t)oc * cin + ic) * kD + kd) * kH + kh) * kW + kw;
                                        AscendC::DataCopy(wLocal, wGm[woff], 1);
                                        dtype alpha = wLocal.GetValue(0);
                                        tmpQueue.FreeTensor(wLocal);

                                        // y += alpha * x
                                        AscendC::Axpy(yLocal, xUse, alpha, tileW);

                                        inQueueX.FreeTensor(xUse);
                                    }
                                }
                            }
                        }

                        outQueueY.EnQue<dtype>(yLocal);
                        AscendC::LocalTensor<dtype> yOut = outQueueY.DeQue<dtype>();
                        uint64_t yoff = ((((uint64_t)bn * cout + oc) * outD + od) * outH + oh) * outW;
                        AscendC::DataCopy(yGm[yoff], yOut, tileW);
                        outQueueY.FreeTensor(yOut);
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<dtype> xGm;
    AscendC::GlobalTensor<dtype> wGm;
    AscendC::GlobalTensor<dtype> yGm;

    uint32_t n, cin, cout;
    uint32_t inD, inH, inW;
    uint32_t kD, kH, kW;
    uint32_t outD, outH, outW;
    uint32_t tileW;
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3D op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.inD, tiling_data.inH, tiling_data.inW,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW);
    op.Process();
}
