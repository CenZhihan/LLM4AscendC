kernel_src="""
#include "kernel_operator.h"

class KernelCatChannels {
public:
    __aicore__ inline KernelCatChannels() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t n, uint32_t c1hw, uint32_t c2hw, uint32_t tileSize)
    {
        this->n = n;
        this->c1hw = c1hw;
        this->c2hw = c2hw;
        this->tileSize = tileSize;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t nPerBlock = (n + blockNum - 1) / blockNum;
        startN = blockIdx * nPerBlock;
        endN = startN + nPerBlock;
        if (endN > n) { endN = n; }

        uint32_t xOffset = startN * c1hw;
        uint32_t yOffset = startN * c2hw;
        uint32_t zOffset = startN * (c1hw + c2hw);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + xOffset, (endN > startN) ? ((endN - startN) * c1hw) : 0);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + yOffset, (endN > startN) ? ((endN - startN) * c2hw) : 0);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + zOffset, (endN > startN) ? ((endN - startN) * (c1hw + c2hw)) : 0);

        pipe.InitBuffer(bufQueue, 1, this->tileSize * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        if (startN >= endN) { return; }
        for (uint32_t bn = startN; bn < endN; ++bn) {
            uint32_t localIdx = bn - startN;
            uint32_t xBase = localIdx * c1hw;
            uint32_t yBase = localIdx * c2hw;
            uint32_t zBase = localIdx * (c1hw + c2hw);

            // 复制 x -> z 前半段
            CopyChunk(xGm, xBase, zGm, zBase, c1hw);

            // 复制 y -> z 后半段
            CopyChunk(yGm, yBase, zGm, zBase + c1hw, c2hw);
        }
    }

private:
    __aicore__ inline void CopyChunk(AscendC::GlobalTensor<DTYPE_X>& src, uint32_t srcBase,
                                     AscendC::GlobalTensor<DTYPE_Z>& dst, uint32_t dstBase,
                                     uint32_t length)
    {
        uint32_t offset = 0;
        while (offset < length) {
            uint32_t curr = (length - offset) > tileSize ? tileSize : (length - offset);
            AscendC::LocalTensor<DTYPE_X> buf = bufQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(buf, src[srcBase + offset], curr);
            AscendC::DataCopy(dst[dstBase + offset], buf, curr);
            bufQueue.FreeTensor(buf);
            offset += curr;
        }
    }

private:
    uint32_t n;
    uint32_t c1hw;
    uint32_t c2hw;
    uint32_t tileSize;

    uint32_t startN;
    uint32_t endN;

    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> bufQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
};

extern "C" __global__ __aicore__ void densenet121_dense_block_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelCatChannels op;
    op.Init(x, y, z, tilingData.n, tilingData.c1hw, tilingData.c2hw, tilingData.tileSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        super().__init__()
        layers = []
        for i in range(num_layers):
            layers.append(self._make_layer(num_input_features + i * growth_rate, growth_rate))
        self.layers = torch.nn.ModuleList(layers)

    def _make_layer(self, in_features: int, growth_rate: int):
        return torch.nn.Sequential(
            torch.nn.BatchNorm2d(in_features),
            torch.nn.ReLU(inplace=True),
            torch.nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
            torch.nn.Dropout(0.0)
        )

    def forward(self, x):
        for layer in self.layers:
            new_feature = layer(x)
            x = custom_ops_lib.densenet121_dense_block_custom(x, new_feature)
        return x
'''
