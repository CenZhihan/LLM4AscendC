#include "regnet_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    // Assume NCHW
    uint32_t N = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(0));
    uint32_t C = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(1));
    uint32_t H = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(2));
    uint32_t W = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(3));

    uint32_t NC = N * C;
    uint32_t HW = H * W;
    uint32_t blockDim = std::min<uint32_t>(NC, 8);
    if (blockDim == 0) {
        blockDim = 1;
    }
    context->SetBlockDim(blockDim);
    uint32_t groupsPerBlock = (NC + blockDim - 1) / blockDim;

    RegnetCustomTilingData tiling;
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_NC(NC);
    tiling.set_HW(HW);
    tiling.set_groupsPerBlock(groupsPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    int64_t n = in_shape->GetOriginShape().GetDim(0);
    int64_t c = in_shape->GetOriginShape().GetDim(1);
    gert::Shape* out_shape = context->GetOutputShape(0);

    // Set output shape to [N, C]
    std::vector<int64_t> dims = {n, c};
    out_shape->SetDimNum(static_cast<uint32_t>(dims.size()));
    for (size_t i = 0; i < dims.size(); ++i) {
        out_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)

    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

// 全局平均池化 (GAP): 输入 [N, C, H, W]，输出 [N, C]，每个 (n,c) 对应 H*W 上的均值
constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t TMP_BUF_NUM = 1;
constexpr int32_t OUT_BUF_BYTES = 32; // 按框架要求对齐

class KernelRegnetGap {
public:
    __aicore__ inline KernelRegnetGap() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t NC, uint32_t HW, uint32_t groupsPerBlock)
    {
        this->N = N;
        this->C = C;
        this->H = H;
        this->W = W;
        this->NC = NC;
        this->HW = HW;
        this->groupsPerBlock = groupsPerBlock;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint64_t xOffset = static_cast<uint64_t>(blockIdx) * static_cast<uint64_t>(groupsPerBlock) * static_cast<uint64_t>(HW);
        uint64_t yOffset = static_cast<uint64_t>(blockIdx) * static_cast<uint64_t>(groupsPerBlock);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + xOffset, groupsPerBlock * HW);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + yOffset, groupsPerBlock);

        // 初始化本地缓冲队列
        pipe.InitBuffer(inQueueX, IN_BUF_NUM, this->HW * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQue, TMP_BUF_NUM, this->HW * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, 1, OUT_BUF_BYTES);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < groupsPerBlock; ++i) {
            uint64_t globalGroupIdx = static_cast<uint64_t>(AscendC::GetBlockIdx()) * static_cast<uint64_t>(groupsPerBlock) + i;
            if (globalGroupIdx >= NC) {
                break;
            }
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // 每个 (n,c) 的数据是连续的 HW 段
        AscendC::DataCopy(xLocal, xGm[progress * this->HW], this->HW);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> workLocal = tmpQue.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // ReduceSum 计算 HW 个元素的和，然后通过 GetAccVal 读取累加器的结果
        AscendC::ReduceSum(yLocal, xLocal, workLocal, this->HW);
        DTYPE_Y sumVal = AscendC::GetAccVal<DTYPE_Y>();
        DTYPE_Y meanVal = sumVal / static_cast<DTYPE_Y>(this->HW);

        // 将标量 meanVal 写入本地缓冲的第一个元素
        AscendC::Duplicate(yLocal, meanVal, 1);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
        tmpQue.FreeTensor(workLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress], yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> tmpQue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, C, H, W;
    uint32_t NC, HW;
    uint32_t groupsPerBlock;
};

extern "C" __global__ __aicore__ void regnet_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelRegnetGap op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.NC, tiling_data.HW, tiling_data.groupsPerBlock);
    op.Process();
}

at::Tensor regnet_custom_impl_npu(const at::Tensor& x) {

    EXEC_NPU_CMD(aclnnRegnetCustom, x, y);

    m.impl("regnet_custom", &regnet_custom_impl_npu);

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_channels, stages, block_widths, output_classes):
        super().__init__()
        self.stages = stages
        self.block_widths = block_widths

        layers = []
        current_channels = input_channels
        for i in range(stages):
            layers.append(self._make_stage(current_channels, block_widths[i]))
            current_channels = block_widths[i]
        self.feature_extractor = torch.nn.Sequential(*layers)
        self.fc = torch.nn.Linear(block_widths[-1], output_classes)

    def _make_stage(self, in_channels, out_channels):
        return torch.nn.Sequential(
            torch.nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1),
            torch.nn.BatchNorm2d(out_channels),
            torch.nn.ReLU(),
            torch.nn.Conv2d(out_channels, out_channels, kernel_size=3, padding=1),
            torch.nn.BatchNorm2d(out_channels),
            torch.nn.ReLU(),
            torch.nn.MaxPool2d(kernel_size=2, stride=2)
        )

    def forward(self, x):
        x = self.feature_extractor(x)
        # 使用自定义AscendC内核实现全局平均池化 (N, C, H, W) -> (N, C)
        x = custom_ops_lib.regnet_custom(x)
        x = self.fc(x)
        return x
'''
