kernel_src="""
#include "kernel_operator.h"

class KernelMaxPool3D {
public:
    __aicore__ inline KernelMaxPool3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t ksize, uint32_t stride, uint32_t padding,
                                uint32_t outD, uint32_t outH, uint32_t outW)
    {
        this->n = n; this->c = c; this->d = d; this->h = h; this->w = w;
        this->ksize = ksize; this->stride = stride; this->padding = padding;
        this->outD = outD; this->outH = outH; this->outW = outW;

        inSize = (uint64_t)n * c * d * h * w;
        outSize = (uint64_t)n * c * outD * outH * outW;

        xGm.SetGlobalBuffer((__gm__ float *)x, inSize);
        yGm.SetGlobalBuffer((__gm__ float *)y, outSize);

        // Minimal temporary buffers (single element buffers used repeatedly)
        pipe.InitBuffer(tmpQueueA, 1, 1 * sizeof(float));
        pipe.InitBuffer(tmpQueueB, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Iterate over N, C, and output spatial dimensions
        for (uint32_t ni = 0; ni < n; ++ni) {
            for (uint32_t ci = 0; ci < c; ++ci) {
                for (uint32_t od = 0; od < outD; ++od) {
                    for (uint32_t oh = 0; oh < outH; ++oh) {
                        for (uint32_t ow = 0; ow < outW; ++ow) {
                            ComputeOne(ni, ci, od, oh, ow);
                        }
                    }
                }
            }
        }
    }

private:
    __aicore__ inline uint64_t InOffset(uint32_t ni, uint32_t ci, uint32_t id, uint32_t ih, uint32_t iw) const
    {
        return (((uint64_t)ni * c + ci) * d + id) * h * w + (uint64_t)ih * w + iw;
    }
    __aicore__ inline uint64_t OutOffset(uint32_t ni, uint32_t ci, uint32_t od, uint32_t oh, uint32_t ow) const
    {
        return (((uint64_t)ni * c + ci) * outD + od) * outH * outW + (uint64_t)oh * outW + ow;
    }

    __aicore__ inline void ComputeOne(uint32_t ni, uint32_t ci, uint32_t od, uint32_t oh, uint32_t ow)
    {
        // max accumulator as LocalTensor (size 1), init to a very small number
        AscendC::LocalTensor<float> maxLocal = tmpQueueA.AllocTensor<float>();
        float negInf = -3.4e38f;
        AscendC::Duplicate(maxLocal, negInf, 1);

        int32_t idStart = (int32_t)od * (int32_t)stride - (int32_t)padding;
        int32_t ihStart = (int32_t)oh * (int32_t)stride - (int32_t)padding;
        int32_t iwStart = (int32_t)ow * (int32_t)stride - (int32_t)padding;

        for (uint32_t kd = 0; kd < ksize; ++kd) {
            int32_t id = idStart + (int32_t)kd;
            if (id < 0 || id >= (int32_t)d) continue;
            for (uint32_t kh = 0; kh < ksize; ++kh) {
                int32_t ih = ihStart + (int32_t)kh;
                if (ih < 0 || ih >= (int32_t)h) continue;
                for (uint32_t kw = 0; kw < ksize; ++kw) {
                    int32_t iw = iwStart + (int32_t)kw;
                    if (iw < 0 || iw >= (int32_t)w) continue;

                    uint64_t inOff = InOffset(ni, ci, (uint32_t)id, (uint32_t)ih, (uint32_t)iw);
                    AscendC::LocalTensor<float> cur = tmpQueueB.AllocTensor<float>();
                    AscendC::DataCopy(cur, xGm[inOff], 1);

                    // element-wise maximum on size-1 tensors
                    AscendC::Maximum(maxLocal, maxLocal, cur, 1);

                    tmpQueueB.FreeTensor(cur);
                }
            }
        }

        uint64_t outOff = OutOffset(ni, ci, od, oh, ow);
        AscendC::DataCopy(yGm[outOff], maxLocal, 1);
        tmpQueueA.FreeTensor(maxLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueueA;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueueB;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n, c, d, h, w;
    uint32_t ksize, stride, padding;
    uint32_t outD, outH, outW;
    uint64_t inSize;
    uint64_t outSize;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    KernelMaxPool3D op;
    op.Init(x, y,
            tiling_data.n(), tiling_data.c(), tiling_data.d(), tiling_data.h(), tiling_data.w(),
            tiling_data.ksize(), tiling_data.stride(), tiling_data.padding(),
            tiling_data.outD(), tiling_data.outH(), tiling_data.outW());
    op.Process();
}
