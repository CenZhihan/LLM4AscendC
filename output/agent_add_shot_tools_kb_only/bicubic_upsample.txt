inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t ROW_TILE = 16;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    BicubicUpsampleCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    // Expect 4D: [N, C, H, W]
    uint32_t n  = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(0));
    uint32_t c  = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(1));
    uint32_t hIn = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(2));
    uint32_t wIn = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(3));

    // Fixed output size per model requirement: (256, 256)
    uint32_t hOut = 256;
    uint32_t wOut = 256;

    uint32_t totalPlanes = n * c;
    uint32_t planesPerCore = CeilDiv(totalPlanes, BLOCK_DIM);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_hIn(hIn);
    tiling.set_wIn(wIn);
    tiling.set_hOut(hOut);
    tiling.set_wOut(wOut);
    tiling.set_planesPerCore(planesPerCore);
    tiling.set_rowTile(ROW_TILE);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0; // no workspace
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    // Output: [N, C, 256, 256]
    gert::Shape* y_shape = context->GetOutputShape(0);
    std::vector<int64_t> out_dims(4);
    out_dims[0] = x_shape->GetOriginShape().GetDim(0);
    out_dims[1] = x_shape->GetOriginShape().GetDim(1);
    out_dims[2] = 256;
    out_dims[3] = 256;
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using DTYPE = float;
constexpr int32_t IN_QUEUE_NUM = 1;
constexpr int32_t OUT_QUEUE_NUM = 1;

__aicore__ inline int32_t clamp_int(int32_t v, int32_t low, int32_t high) {
    return v < low ? low : (v > high ? high : v);
}

__aicore__ inline float cubic_interp(float p0, float p1, float p2, float p3, float t) {
    // Catmull-Rom spline
    float a0 = -0.5f * p0 + 1.5f * p1 - 1.5f * p2 + 0.5f * p3;
    float a1 = p0 - 2.5f * p1 + 2.0f * p2 - 0.5f * p3;
    float a2 = -0.5f * p0 + 0.5f * p2;
    float a3 = p1;
    return ((a0 * t + a1) * t + a2) * t + a3;
}

class KernelBicubicUpsample {
public:
    __aicore__ inline KernelBicubicUpsample() {}

    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR zGm, uint32_t n, uint32_t c,
                                uint32_t hIn, uint32_t wIn, uint32_t hOut, uint32_t wOut,
                                uint32_t planesPerCore, uint32_t rowTile)
    {
        this->n = n;
        this->c = c;
        this->hIn = hIn;
        this->wIn = wIn;
        this->hOut = hOut;
        this->wOut = wOut;
        this->planesPerCore = planesPerCore;
        this->rowTile = rowTile;

        inPlaneSize = hIn * wIn;
        outPlaneSize = hOut * wOut;
        totalPlanes = n * c;

        xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(xGm), n * c * inPlaneSize);
        zGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(zGm), n * c * outPlaneSize);

        // Local buffer: one input plane, and one output tile (rowTile * wOut)
        pipe.InitBuffer(inQueueX, IN_QUEUE_NUM, inPlaneSize * sizeof(DTYPE));
        pipe.InitBuffer(outQueueZ, OUT_QUEUE_NUM, rowTile * wOut * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        uint32_t coreIdx = AscendC::GetBlockIdx();
        uint32_t coreNum = AscendC::GetBlockNum();

        uint32_t startPlane = coreIdx * planesPerCore;
        uint32_t endPlane = startPlane + planesPerCore;
        if (endPlane > totalPlanes) endPlane = totalPlanes;

        for (uint32_t plane = startPlane; plane < endPlane; ++plane) {
            uint32_t inBase  = plane * inPlaneSize;
            uint32_t outBase = plane * outPlaneSize;

            // Copy one input plane into local
            AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
            AscendC::DataCopy(xLocal, xGlobal[inBase], inPlaneSize);
            inQueueX.EnQue(xLocal);

            // Compute output plane in row tiles
            float scaleY = (hOut > 1 && hIn > 1) ? static_cast<float>(hIn - 1) / static_cast<float>(hOut - 1) : 0.f;
            float scaleX = (wOut > 1 && wIn > 1) ? static_cast<float>(wIn - 1) / static_cast<float>(wOut - 1) : 0.f;

            for (uint32_t oyStart = 0; oyStart < hOut; oyStart += rowTile) {
                uint32_t rows = (oyStart + rowTile <= hOut) ? rowTile : (hOut - oyStart);

                AscendC::LocalTensor<DTYPE> zTileLocal = outQueueZ.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> xTileLocal = inQueueX.DeQue<DTYPE>();

                // Compute rows in [oyStart, oyStart + rows)
                for (uint32_t ry = 0; ry < rows; ++ry) {
                    uint32_t oy = oyStart + ry;
                    float fy = oy * scaleY;
                    int32_t y1 = static_cast<int32_t>(floorf(fy));
                    float ty = fy - y1;
                    int32_t y0 = y1 - 1;
                    int32_t y2 = y1 + 1;
                    int32_t y3 = y1 + 2;
                    y0 = clamp_int(y0, 0, static_cast<int32_t>(hIn) - 1);
                    y1 = clamp_int(y1, 0, static_cast<int32_t>(hIn) - 1);
                    y2 = clamp_int(y2, 0, static_cast<int32_t>(hIn) - 1);
                    y3 = clamp_int(y3, 0, static_cast<int32_t>(hIn) - 1);

                    for (uint32_t ox = 0; ox < wOut; ++ox) {
                        float fx = ox * scaleX;
                        int32_t x1 = static_cast<int32_t>(floorf(fx));
                        float tx = fx - x1;
                        int32_t x0 = x1 - 1;
                        int32_t x2 = x1 + 1;
                        int32_t x3 = x1 + 2;
                        x0 = clamp_int(x0, 0, static_cast<int32_t>(wIn) - 1);
                        x1 = clamp_int(x1, 0, static_cast<int32_t>(wIn) - 1);
                        x2 = clamp_int(x2, 0, static_cast<int32_t>(wIn) - 1);
                        x3 = clamp_int(x3, 0, static_cast<int32_t>(wIn) - 1);

                        // Load 4x4 neighborhood
                        float r0c0 = xTileLocal[y0 * wIn + x0];
                        float r0c1 = xTileLocal[y0 * wIn + x1];
                        float r0c2 = xTileLocal[y0 * wIn + x2];
                        float r0c3 = xTileLocal[y0 * wIn + x3];

                        float r1c0 = xTileLocal[y1 * wIn + x0];
                        float r1c1 = xTileLocal[y1 * wIn + x1];
                        float r1c2 = xTileLocal[y1 * wIn + x2];
                        float r1c3 = xTileLocal[y1 * wIn + x3];

                        float r2c0 = xTileLocal[y2 * wIn + x0];
                        float r2c1 = xTileLocal[y2 * wIn + x1];
                        float r2c2 = xTileLocal[y2 * wIn + x2];
                        float r2c3 = xTileLocal[y2 * wIn + x3];

                        float r3c0 = xTileLocal[y3 * wIn + x0];
                        float r3c1 = xTileLocal[y3 * wIn + x1];
                        float r3c2 = xTileLocal[y3 * wIn + x2];
                        float r3c3 = xTileLocal[y3 * wIn + x3];

                        // Horizontal cubic for each row
                        float v0 = cubic_interp(r0c0, r0c1, r0c2, r0c3, tx);
                        float v1 = cubic_interp(r1c0, r1c1, r1c2, r1c3, tx);
                        float v2 = cubic_interp(r2c0, r2c1, r2c2, r2c3, tx);
                        float v3 = cubic_interp(r3c0, r3c1, r3c2, r3c3, tx);

                        // Vertical cubic
                        float val = cubic_interp(v0, v1, v2, v3, ty);
                        zTileLocal[ry * wOut + ox] = val;
                    }
                }

                // Enqueue and write out
                outQueueZ.EnQue<DTYPE>(zTileLocal);
                inQueueX.FreeTensor(xTileLocal);

                AscendC::LocalTensor<DTYPE> zTileOut = outQueueZ.DeQue<DTYPE>();
                // Copy tile to corresponding output position
                AscendC::DataCopy(zGlobal[outBase + oyStart * wOut], zTileOut, rows * wOut);
                outQueueZ.FreeTensor(zTileOut);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, IN_QUEUE_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, OUT_QUEUE_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE> xGlobal;
    AscendC::GlobalTensor<DTYPE> zGlobal;

    uint32_t n, c, hIn, wIn, hOut, wOut;
    uint32_t planesPerCore, rowTile;
    uint32_t inPlaneSize, outPlaneSize, totalPlanes;
};

extern "C" __global__ __aicore__ void bicubic_upsample_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBicubicUpsample op;
    op.Init(x, z,
            tiling_data.n(), tiling_data.c(),
            tiling_data.hIn(), tiling_data.wIn(),
            tiling_data.hOut(), tiling_data.wOut(),
            tiling_data.planesPerCore(), tiling_data.rowTile());
    op.Process();
}
