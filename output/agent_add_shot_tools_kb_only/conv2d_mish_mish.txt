kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// mish(x) = x * tanh(softplus(x)), softplus(x) = ln(1 + exp(x))
// This kernel applies mish twice: y = mish(mish(x))
class KernelMishTwice {
public:
    __aicore__ inline KernelMishTwice() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x) + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(y) + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));

        // temp buffers
        pipe.InitBuffer(tmp0, 1, this->tileLength * sizeof(float)); // t / t2
        pipe.InitBuffer(tmp1, 1, this->tileLength * sizeof(float)); // ones
        pipe.InitBuffer(tmp2, 1, this->tileLength * sizeof(float)); // sp
        pipe.InitBuffer(tmp3, 1, this->tileLength * sizeof(float)); // tanh(sp)
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        AscendC::LocalTensor<float> tLocal   = tmp0.AllocTensor<float>();
        AscendC::LocalTensor<float> oneLocal = tmp1.AllocTensor<float>();
        AscendC::LocalTensor<float> spLocal  = tmp2.AllocTensor<float>();
        AscendC::LocalTensor<float> thLocal  = tmp3.AllocTensor<float>();

        // softplus(x): tLocal = exp(x)
        AscendC::Exp(tLocal, xLocal, this->tileLength);
        // oneLocal = 1.0
        AscendC::Duplicate(oneLocal, 1.0f, this->tileLength);
        // tLocal = tLocal + 1
        AscendC::Add(tLocal, tLocal, oneLocal, this->tileLength);
        // spLocal = ln(tLocal)
        AscendC::Ln(spLocal, tLocal, this->tileLength);
        // thLocal = tanh(spLocal)
        AscendC::Tanh(thLocal, spLocal, this->tileLength);
        // yLocal = xLocal * thLocal  (first mish)
        AscendC::Mul(yLocal, xLocal, thLocal, this->tileLength);

        // softplus(yLocal): reuse tLocal
        AscendC::Exp(tLocal, yLocal, this->tileLength);
        AscendC::Add(tLocal, tLocal, oneLocal, this->tileLength);
        AscendC::Ln(spLocal, tLocal, this->tileLength);
        AscendC::Tanh(thLocal, spLocal, this->tileLength);
        // yLocal = yLocal * thLocal  (second mish)
        AscendC::Mul(yLocal, yLocal, thLocal, this->tileLength);

        outQueueY.EnQue<float>(yLocal);

        inQueueX.FreeTensor(xLocal);
        tmp0.FreeTensor(tLocal);
        tmp1.FreeTensor(oneLocal);
        tmp2.FreeTensor(spLocal);
        tmp3.FreeTensor(thLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp0, tmp1, tmp2, tmp3;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelMishTwice op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Conv2d followed by fused Mish(Mish(.)) custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        x = self.conv(x)
        x = custom_ops_lib.conv2d_mish_mish_custom(x)
        return x
