kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelGemmGroupNormMinBiasAdd {
public:
    __aicore__ inline KernelGemmGroupNormMinBiasAdd() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z, uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t reduceWorkSize)
    {
        this->n = n;
        this->c = c;
        this->hw = h * w;
        this->rowSize = c * hw;

        xGm.SetGlobalBuffer((__gm__ float *)x, n * rowSize);
        biasGm.SetGlobalBuffer((__gm__ float *)bias, c);
        zGm.SetGlobalBuffer((__gm__ float *)z, n * rowSize);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, rowSize * sizeof(float));
        pipe.InitBuffer(inQueueBias, BUFFER_NUM, c * sizeof(float));
        pipe.InitBuffer(workQueue, BUFFER_NUM, reduceWorkSize * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, rowSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Load bias once to local
        AscendC::LocalTensor<float> biasLocal = inQueueBias.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, biasGm, c);
        inQueueBias.EnQue(biasLocal);

        for (uint32_t i = 0; i < n; ++i) {
            CopyInRow(i);
            ComputeRow(i);
            CopyOutRow(i);
        }
        // free bias tensor
        AscendC::LocalTensor<float> biasLocalRet = inQueueBias.DeQue<float>();
        inQueueBias.FreeTensor(biasLocalRet);
    }

private:
    __aicore__ inline void CopyInRow(uint32_t idxN)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[idxN * rowSize], rowSize);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRow(uint32_t idxN)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
        AscendC::LocalTensor<float> workLocal = workQueue.AllocTensor<float>();
        AscendC::LocalTensor<float> biasLocal = inQueueBias.DeQue<float>();

        // ReduceMin along channel dimension for H*W==1 case (rowSize == c)
        // If H*W > 1, treat as flattened row and still compute min across all c*hw elements.
        AscendC::LocalTensor<float> minBuf = workLocal; // reuse workLocal to hold result temporarily
        // ReduceMin writes result to the head of dst buffer; using size 1
        AscendC::ReduceMin<float>(minBuf, xLocal, workLocal, rowSize, true);

        // Duplicate min value across channel dimension and add bias vector
        // Assume Duplicate API exists to fill tensor with scalar.
        for (uint32_t offset = 0; offset < rowSize; offset += c) {
            AscendC::LocalTensor<float> tmpLocal = outQueueZ.AllocTensor<float>(); // temporary for scalar broadcast
            // broadcast min scalar to length c
            AscendC::Duplicate(tmpLocal, minBuf.GetValue(0), c);
            // z = broadcast_min + bias
            AscendC::Add(zLocal[offset], tmpLocal, biasLocal, c);
            outQueueZ.FreeTensor(tmpLocal);
        }

        inQueueX.FreeTensor(xLocal);
        workQueue.FreeTensor(workLocal);
        inQueueBias.EnQue(biasLocal);
        outQueueZ.EnQue<float>(zLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t idxN)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm[idxN * rowSize], zLocal, rowSize);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueBias;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> workQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> biasGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t n;
    uint32_t c;
    uint32_t hw;
    uint32_t rowSize;
};

extern "C" __global__ __aicore__ void gemm_group_norm_min_bias_add_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelGemmGroupNormMinBiasAdd op;
    op.Init(x, bias, z, tilingData.n, tilingData.c, tilingData.h, tilingData.w, tilingData.reduceWorkSize);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep GEMM + GroupNorm on framework, fuse min-over-channel and bias add into a custom kernel.
    Input/Output shape: [N, C]
    """
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        x = self.gemm(x)                              # [N, C]
        x = self.group_norm(x)                        # [N, C]
        bias_vec = self.bias.view(-1)                 # [C]
        # Custom kernel computes min over channel (dim=1, keepdim=True) broadcast-added with bias
        # Output shape: [N, C]
        out = custom_ops_lib.gemm_group_norm_min_bias_add_custom(x, bias_vec)
        return out
