kernel_src="""
#include "kernel_operator.h"

using DTYPE_LOGITS = float;
using DTYPE_MASK = uint8_t;
using DTYPE_OUT = float;

constexpr int32_t BUFFER_NUM = 2; // tensors per queue
constexpr DTYPE_OUT NEG_LARGE = -1e9f;

class KernelSparseAttentionMask {
public:
    __aicore__ inline KernelSparseAttentionMask() {}
    __aicore__ inline void Init(GM_ADDR logits, GM_ADDR mask, GM_ADDR out,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        logitsGm.SetGlobalBuffer((__gm__ DTYPE_LOGITS *)logits + this->blockLength * AscendC::GetBlockIdx(),
                                 this->blockLength);
        maskGm.SetGlobalBuffer((__gm__ DTYPE_MASK *)mask + this->blockLength * AscendC::GetBlockIdx(),
                               this->blockLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_OUT *)out + this->blockLength * AscendC::GetBlockIdx(),
                              this->blockLength);

        pipe.InitBuffer(inQueueLogits, BUFFER_NUM, this->tileLength * sizeof(DTYPE_LOGITS));
        pipe.InitBuffer(inQueueMask, BUFFER_NUM, this->tileLength * sizeof(DTYPE_MASK));
        pipe.InitBuffer(outQueueOut, BUFFER_NUM, this->tileLength * sizeof(DTYPE_OUT));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_LOGITS> logitsLocal = inQueueLogits.AllocTensor<DTYPE_LOGITS>();
        AscendC::LocalTensor<DTYPE_MASK> maskLocal = inQueueMask.AllocTensor<DTYPE_MASK>();
        AscendC::DataCopy(logitsLocal, logitsGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(maskLocal, maskGm[progress * this->tileLength], this->tileLength);
        inQueueLogits.EnQue(logitsLocal);
        inQueueMask.EnQue(maskLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_LOGITS> logitsLocal = inQueueLogits.DeQue<DTYPE_LOGITS>();
        AscendC::LocalTensor<DTYPE_MASK> maskLocal = inQueueMask.DeQue<DTYPE_MASK>();
        AscendC::LocalTensor<DTYPE_OUT> outLocal = outQueueOut.AllocTensor<DTYPE_OUT>();

        // cast mask uint8 -> float into outLocal (temp buffer)
        AscendC::Cast(outLocal, maskLocal, this->tileLength);
        // outLocal = mask_float * NEG_LARGE
        AscendC::Muls(outLocal, outLocal, NEG_LARGE, this->tileLength);
        // outLocal += logitsLocal
        AscendC::Add(outLocal, outLocal, logitsLocal, this->tileLength);

        outQueueOut.EnQue<DTYPE_OUT>(outLocal);
        inQueueLogits.FreeTensor(logitsLocal);
        inQueueMask.FreeTensor(maskLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_OUT> outLocal = outQueueOut.DeQue<DTYPE_OUT>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueLogits;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut;

    AscendC::GlobalTensor<DTYPE_LOGITS> logitsGm;
    AscendC::GlobalTensor<DTYPE_MASK> maskGm;
    AscendC::GlobalTensor<DTYPE_OUT> outGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void sparse_attention_custom(GM_ADDR logits, GM_ADDR mask, GM_ADDR out,
                                                              GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSparseAttentionMask op;
    op.Init(logits, mask, out, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import math
import torch
import torch_npu
import torch.nn as nn
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized sparse attention: offloads mask application to a custom AscendC kernel,
    while keeping projections and matmuls in PyTorch.
    """
    def __init__(self, d_model=256, num_heads=4, window_size=32, max_seq_len=4096):
        super().__init__()
        self.num_heads = num_heads
        self.head_dim = d_model // num_heads
        self.window_size = window_size
        self.max_seq_len = max_seq_len

        self.q_proj = nn.Linear(d_model, d_model, bias=False)
        self.k_proj = nn.Linear(d_model, d_model, bias=False)
        self.v_proj = nn.Linear(d_model, d_model, bias=False)
        self.out_proj = nn.Linear(d_model, d_model, bias=False)

        # Precompute sparse mask (True -> masked)
        mask = torch.ones(max_seq_len, max_seq_len, dtype=torch.bool)
        for i in range(max_seq_len):
            start = max(0, i - window_size)
            end = min(max_seq_len, i + window_size)
            mask[i, start:end] = False
        self.register_buffer("attn_mask", mask, persistent=False)

    def forward(self, x):
        """
        x: (B, S, D)
        returns: (B, S, D)
        """
        B, S, D = x.shape
        H = self.num_heads
        Dh = self.head_dim

        # Projections
        q = self.q_proj(x).view(B, S, H, Dh).transpose(1, 2)   # (B, H, S, Dh)
        k = self.k_proj(x).view(B, S, H, Dh).transpose(1, 2)   # (B, H, S, Dh)
        v = self.v_proj(x).view(B, S, H, Dh).transpose(1, 2)   # (B, H, S, Dh)

        # Scaled dot-product attention scores
        logits = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(Dh)  # (B, H, S, S)

        # Prepare mask (broadcast to (B, H, S, S)) and move to NPU dtype (uint8)
        base_mask = self.attn_mask[:S, :S]            # (S, S), True means masked
        mask_bhss = base_mask.unsqueeze(0).unsqueeze(0).expand(B, H, S, S).contiguous()
        mask_u8 = mask_bhss.to(torch.uint8)

        # Apply mask using custom AscendC kernel: out = logits + mask * (-1e9)
        masked_logits = custom_ops_lib.sparse_attention_custom(logits, mask_u8)

        # Softmax and value aggregation
        attn = torch.softmax(masked_logits, dim=-1)
        ctx = torch.matmul(attn, v)                   # (B, H, S, Dh)

        # Merge heads
        ctx = ctx.transpose(1, 2).contiguous().view(B, S, D)
        out = self.out_proj(ctx)
        return out
'''
