kernel_src="""
#include "kernel_operator.h"

using dataType = float;

class KernelConv2dGeluGlobalAvgPool {
public:
    __aicore__ inline KernelConv2dGeluGlobalAvgPool() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t n, uint32_t c, uint32_t h, uint32_t w)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->hw = h * w;
        this->totalLen = n * c * h * w;
        this->outLen = n * c;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(x), totalLen);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(y), outLen);

        // Buffers
        pipe.InitBuffer(inQueueX, 1, totalLen * sizeof(dataType));
        pipe.InitBuffer(tmpQueueGelu, 1, totalLen * sizeof(dataType));
        // output mean buffer (no extra padding copy out)
        pipe.InitBuffer(outQueueY, 1, outLen * sizeof(dataType));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<dataType> xLocal = inQueueX.AllocTensor<dataType>();
        AscendC::DataCopy(xLocal, xGm, totalLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<dataType> xLocal = inQueueX.DeQue<dataType>();
        AscendC::LocalTensor<dataType> geluLocal = tmpQueueGelu.AllocTensor<dataType>();
        // Apply GELU element-wise
        AscendC::Gelu(geluLocal, xLocal, totalLen);

        // Global Average Pool over H*W for each (N, C)
        // Treat geluLocal as [N*C, H*W], reduce mean across the last dimension.
        AscendC::LocalTensor<dataType> meanLocal = outQueueY.AllocTensor<dataType>();
        AscendC::MeanParams meanParams;
        meanParams.outter = outLen;   // N * C
        meanParams.inner  = hw;       // H * W
        meanParams.n      = hw;       // reduce over H*W
        AscendC::Mean<dataType, dataType>(meanLocal, geluLocal, meanParams);

        outQueueY.EnQue<dataType>(meanLocal);
        inQueueX.FreeTensor(xLocal);
        tmpQueueGelu.FreeTensor(geluLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<dataType> meanLocal = outQueueY.DeQue<dataType>();
        AscendC::DataCopy(yGm, meanLocal, outLen);
        outQueueY.FreeTensor(meanLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueueGelu;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<dataType> xGm;
    AscendC::GlobalTensor<dataType> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t h;
    uint32_t w;
    uint32_t hw;
    uint32_t totalLen;
    uint32_t outLen;
};

extern "C" __global__ __aicore__ void conv2d_gelu_global_avg_pool_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dGeluGlobalAvgPool op;
    op.Init(x, y, tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep PyTorch conv, fuse GELU + GlobalAvgPool into a custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        # PyTorch conv
        x = self.conv(x)
        # Fused GELU + GlobalAvgPool (H, W -> 1,1) + squeeze to [N, C]
        x = custom_ops_lib.conv2d_gelu_global_avg_pool_custom(x)
        return x
