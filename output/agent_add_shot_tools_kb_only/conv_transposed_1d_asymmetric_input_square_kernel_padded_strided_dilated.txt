kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelConvTransposed1D {
public:
    __aicore__ inline KernelConvTransposed1D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR z,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Lin, uint32_t Lout, uint32_t K,
                                uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t tileOutLen, uint32_t tileNum)
    {
        this->N = N;
        this->Cin = Cin;
        this->Cout = Cout;
        this->Lin = Lin;
        this->Lout = Lout;
        this->K = K;
        this->stride = stride;
        this->padding = padding;
        this->dilation = dilation;
        this->tileOutLen = tileOutLen;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x, (uint64_t)N * Cin * Lin);
        wGm.SetGlobalBuffer((__gm__ DTYPE *)w, (uint64_t)Cin * Cout * K);
        zGm.SetGlobalBuffer((__gm__ DTYPE *)z, (uint64_t)N * Cout * Lout);

        // 队列与缓冲设置：
        // - z向量：按Cout分块（例如64）
        // - w向量：同Cout分块
        // 简化设置：每次处理一个向量
        const uint32_t coutChunk = GetCoutChunk();
        pipe.InitBuffer(qZVec, 1, coutChunk * sizeof(DTYPE));
        pipe.InitBuffer(qWVec, 1, coutChunk * sizeof(DTYPE));
        pipe.InitBuffer(qXScalar, 1, sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        const uint32_t coutChunk = GetCoutChunk();
        for (uint32_t n = 0; n < N; ++n) {
            // 输出长度分块
            for (uint32_t tileIdx = 0; tileIdx < tileNum; ++tileIdx) {
                uint32_t oStart = tileIdx * tileOutLen;
                uint32_t curTile = (oStart + tileOutLen <= Lout) ? tileOutLen : (Lout - oStart);
                // 对输入步长和核进行散射累加
                for (uint32_t ci = 0; ci < Cin; ++ci) {
                    for (uint32_t t = 0; t < Lin; ++t) {
                        // 加载输入标量 x[n, ci, t]
                        AscendC::LocalTensor<DTYPE> xLocal = qXScalar.AllocTensor<DTYPE>();
                        uint64_t xOff = ((uint64_t)n * Cin + ci) * Lin + t;
                        AscendC::DataCopy(xLocal, xGm[xOff], 1);
                        // 读取标量
                        DTYPE x_val = xLocal.GetValue(0);
                        qXScalar.FreeTensor(xLocal);

                        // 遍历核元素 r
                        for (uint32_t r = 0; r < K; ++r) {
                            // 计算输出位置
                            int64_t o = (int64_t)t * (int64_t)stride - (int64_t)padding + (int64_t)r * (int64_t)dilation;
                            if (o < (int64_t)oStart || o >= (int64_t)(oStart + curTile)) {
                                continue;
                            }
                            uint32_t oIdx = (uint32_t)o;

                            // 沿Cout方向分块
                            for (uint32_t co0 = 0; co0 < Cout; co0 += coutChunk) {
                                uint32_t curChunk = (co0 + coutChunk <= Cout) ? coutChunk : (Cout - co0);

                                // 读取 z 向量 z[n, co0:co0+chunk, o]
                                AscendC::LocalTensor<DTYPE> zVec = qZVec.AllocTensor<DTYPE>();
                                uint64_t zBase = ((uint64_t)n * Cout + co0) * Lout + oIdx;
                                // z 在内存上 co是快变，Lout 紧随其后，故取向量需跨步为Lout，但简化为按co连续访问：
                                // 实际布局: ((n * Cout + co) * Lout + o). 因为co连续，DataCopy可以线性读取 'curChunk' 个元素。
                                AscendC::DataCopy(zVec, zGm[zBase], curChunk);

                                // 读取 w 向量 w[ci, co0:co0+chunk, r]
                                AscendC::LocalTensor<DTYPE> wVec = qWVec.AllocTensor<DTYPE>();
                                uint64_t wBase = ((uint64_t)ci * Cout + co0) * K + r;
                                AscendC::DataCopy(wVec, wGm[wBase], curChunk);

                                // zVec += x_val * wVec
                                AscendC::Axpy(zVec, wVec, x_val, curChunk);

                                // 回写 z 向量
                                AscendC::DataCopy(zGm[zBase], zVec, curChunk);

                                qWVec.FreeTensor(wVec);
                                qZVec.FreeTensor(zVec);
                            } // co0
                        } // r
                    } // t
                } // ci
            } // tileIdx
        } // n
    }

private:
    __aicore__ inline uint32_t GetCoutChunk() const
    {
        // 简化设定一个固定chunk，避免UB溢出
        const uint32_t kChunk = 64;
        return (Cout < kChunk) ? Cout : kChunk;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> qWVec, qXScalar, qZVec;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> zGm;

    uint32_t N, Cin, Cout, Lin, Lout, K;
    uint32_t stride, padding, dilation;
    uint32_t tileOutLen, tileNum;
};

extern "C" __global__ __aicore__ void conv_transposed_1d_asymmetric_input_square_kernel_padded_strided_dilated_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    KernelConvTransposed1D op;
    op.Init(x, w, z,
            tilingData.N, tilingData.Cin, tilingData.Cout,
            tilingData.Lin, tilingData.Lout, tilingData.K,
            tilingData.stride, tilingData.padding, tilingData.dilation,
            tilingData.tileOutLen, tilingData.tileNum);
    op.Process();
}
