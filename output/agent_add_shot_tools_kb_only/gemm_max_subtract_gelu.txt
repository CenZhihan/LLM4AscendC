kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMaxSubtractGelu {
public:
    __aicore__ inline KernelGemmMaxSubtractGelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t rows, uint32_t cols, uint32_t tileNum)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileNum = tileNum;

        // 输出为 (rows, 1)，每个block按行均分
        this->rowsPerBlock = (rows + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        // 实际最后一个block可能会越界，计算有效行数
        uint32_t startRow = this->rowsPerBlock * AscendC::GetBlockIdx();
        this->validRows = (startRow + this->rowsPerBlock <= rows) ? this->rowsPerBlock : (rows > startRow ? rows - startRow : 0);
        if (this->validRows == 0) {
            this->rowsPerBlock = 0;
            this->tileLength = 0;
        } else {
            // 简单按tileNum和BUFFER_NUM切分
            uint32_t pieces = (this->tileNum * BUFFER_NUM) > 0 ? (this->tileNum * BUFFER_NUM) : 1;
            this->tileLength = (this->validRows + pieces - 1) / pieces;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)startRow * cols, (uint64_t)this->validRows * cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + startRow, this->validRows);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (this->validRows == 0 || this->tileLength == 0) {
            return;
        }
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        // 由于 x_max 在维度1上保持keepdim为True，随后再做mean仍为自身，减法后恒为0
        // 直接生成0并执行GELU（GELU(0)=0），保证语义一致
        DTYPE_Y zero(0);
        AscendC::Duplicate(yLocal, zero, this->tileLength);
        AscendC::Gelu(yLocal, yLocal, this->tileLength);
        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        uint32_t offset = progress * this->tileLength;
        uint32_t remain = (offset + this->tileLength <= this->validRows) ? this->tileLength : (this->validRows > offset ? this->validRows - offset : 0);
        if (remain > 0) {
            AscendC::DataCopy(yGm[offset], yLocal, remain);
        }
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows = 0;
    uint32_t cols = 0;
    uint32_t tileNum = 0;

    uint32_t rowsPerBlock = 0;
    uint32_t validRows = 0;
    uint32_t tileLength = 0;
};

extern "C" __global__ __aicore__ void gemm_max_subtract_gelu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMaxSubtractGelu op;
    op.Init(x, y, tiling_data.rows, tiling_data.cols, tiling_data.tileNum);
    op.Process();
}
