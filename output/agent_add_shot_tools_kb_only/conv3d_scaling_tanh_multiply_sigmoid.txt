kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // double buffer

class KernelConv3dScalingTanhMultiplySigmoid {
public:
    __aicore__ inline KernelConv3dScalingTanhMultiplySigmoid() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale_nc, GM_ADDR bias_nc, GM_ADDR y,
                                uint32_t totalLength, uint32_t innerSize, uint32_t pairCount)
    {
        this->innerSize = innerSize;
        this->pairCount = pairCount;
        this->blockLength = (AscendC::GetBlockNum() == 0) ? totalLength : (totalLength / AscendC::GetBlockNum());
        this->pairsPerBlock = (innerSize == 0) ? 0 : (this->blockLength / innerSize);

        // Global buffer binding
        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x) + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(y) + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        scaleNcGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(scale_nc), this->pairCount);
        biasNcGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(bias_nc), this->pairCount);

        // Queues for tiles (each tile = one (N,C) pair of size innerSize)
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->innerSize * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->innerSize * sizeof(float));

        // Queues for scalar loads (scale and bias per (N,C) pair)
        pipe.InitBuffer(inQueueScale, BUFFER_NUM, sizeof(float));
        pipe.InitBuffer(inQueueBias,  BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Each iteration processes one (N,C) pair segment of length innerSize
        for (uint32_t i = 0; i < this->pairsPerBlock; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress)
    {
        // Load X tile
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->innerSize], this->innerSize);

        // Load corresponding scale and bias scalars for the (N,C) pair
        AscendC::LocalTensor<float> scaleLocal = inQueueScale.AllocTensor<float>();
        AscendC::LocalTensor<float> biasLocal  = inQueueBias.AllocTensor<float>();

        uint64_t globalPairStart = (static_cast<uint64_t>(this->blockLength) * AscendC::GetBlockIdx()) / this->innerSize;
        uint64_t pairIndex = globalPairStart + progress;

        AscendC::DataCopy(scaleLocal, scaleNcGm[pairIndex], 1);
        AscendC::DataCopy(biasLocal,  biasNcGm[pairIndex],  1);

        inQueueX.EnQue(xLocal);
        inQueueScale.EnQue(scaleLocal);
        inQueueBias.EnQue(biasLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progress*/)
    {
        AscendC::LocalTensor<float> xLocal     = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> scaleLocal = inQueueScale.DeQue<float>();
        AscendC::LocalTensor<float> biasLocal  = inQueueBias.DeQue<float>();
        AscendC::LocalTensor<float> yLocal     = outQueueY.AllocTensor<float>();

        // Read scalar values from local tensors
        // Assuming LocalTensor supports direct pointer access via GetValue-like interface.
        // If not, toolchain will map the first element to a scalar register as parameter.
        float s = *reinterpret_cast<float *>(scaleLocal.GetPhyAddr());
        float b = *reinterpret_cast<float *>(biasLocal.GetPhyAddr());

        // u = x * s
        AscendC::LocalTensor<float> uLocal = tmpVec.AllocTensor<float>();
        AscendC::Muls(uLocal, xLocal, s);

        // Compute tanh(u) = (exp(2u)-1)/(exp(2u)+1)
        AscendC::LocalTensor<float> twouLocal = tmpVec.AllocTensor<float>();
        AscendC::Muls(twouLocal, uLocal, 2.0f);

        AscendC::LocalTensor<float> e2uLocal = tmpVec.AllocTensor<float>();
        AscendC::Exp(e2uLocal, twouLocal);

        AscendC::LocalTensor<float> numLocal = tmpVec.AllocTensor<float>();
        AscendC::LocalTensor<float> denLocal = tmpVec.AllocTensor<float>();
        // numerator = e2u - 1
        AscendC::Sub(numLocal, e2uLocal, 1.0f);
        // denominator = e2u + 1
        AscendC::Add(denLocal, e2uLocal, 1.0f);

        AscendC::LocalTensor<float> tanhLocal = tmpVec.AllocTensor<float>();
        AscendC::Div(tanhLocal, numLocal, denLocal);

        // v = tanh(u) * b
        AscendC::LocalTensor<float> vLocal = tmpVec.AllocTensor<float>();
        AscendC::Muls(vLocal, tanhLocal, b);

        // sigmoid(v) = 1 / (1 + exp(-v))
        AscendC::LocalTensor<float> negLocal = tmpVec.AllocTensor<float>();
        AscendC::Muls(negLocal, vLocal, -1.0f);

        AscendC::LocalTensor<float> expNegLocal = tmpVec.AllocTensor<float>();
        AscendC::Exp(expNegLocal, negLocal);

        AscendC::LocalTensor<float> onePlusLocal = tmpVec.AllocTensor<float>();
        AscendC::Add(onePlusLocal, expNegLocal, 1.0f);

        AscendC::Reciprocal(yLocal, onePlusLocal);

        outQueueY.EnQue<float>(yLocal);

        // Free inputs and temporaries
        inQueueX.FreeTensor(xLocal);
        inQueueScale.FreeTensor(scaleLocal);
        inQueueBias.FreeTensor(biasLocal);

        tmpVec.FreeTensor(uLocal);
        tmpVec.FreeTensor(twouLocal);
        tmpVec.FreeTensor(e2uLocal);
        tmpVec.FreeTensor(numLocal);
        tmpVec.FreeTensor(denLocal);
        tmpVec.FreeTensor(tanhLocal);
        tmpVec.FreeTensor(vLocal);
        tmpVec.FreeTensor(negLocal);
        tmpVec.FreeTensor(expNegLocal);
        tmpVec.FreeTensor(onePlusLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->innerSize], yLocal, this->innerSize);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> inQueueScale;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> inQueueBias;
    AscendC::TQue<AscendC::TPosition::VECCALC, 12> tmpVec; // temporary vectors

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<float> scaleNcGm;
    AscendC::GlobalTensor<float> biasNcGm;

    uint32_t innerSize;
    uint32_t pairCount;
    uint32_t blockLength;
    uint32_t pairsPerBlock;
};

extern "C" __global__ __aicore__ void conv3d_scaling_tanh_multiply_sigmoid_custom(GM_ADDR x,
                                                                                   GM_ADDR scale_nc,
                                                                                   GM_ADDR bias_nc,
                                                                                   GM_ADDR y,
                                                                                   GM_ADDR workspace,
                                                                                   GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dScalingTanhMultiplySigmoid op;
    op.Init(x, scale_nc, bias_nc, y,
            tiling_data.totalLength,
            tiling_data.innerSize,
            tiling_data.pairCount);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused kernel after Conv3d: y = sigmoid( bias * tanh( scale * x ) )
    scale_nc and bias_nc are prepared as [N, C] to avoid large broadcast.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape):
        super().__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        # Parameters keep original shapes (C,1,1,1)
        self.scaling_factor = torch.nn.Parameter(torch.randn(bias_shape))
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        x = self.conv(x)  # [N, C, D, H, W]
        n, c, d, h, w = x.shape
        # Prepare per-(N,C) scalars: [N, C]
        scale_nc = self.scaling_factor.view(1, c).expand(n, c).contiguous()
        bias_nc  = self.bias.view(1, c).expand(n, c).contiguous()
        # Call fused custom kernel
        y = custom_ops_lib.conv3d_scaling_tanh_multiply_sigmoid_custom(x, scale_nc, bias_nc)
        return y
