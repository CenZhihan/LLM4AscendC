kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 2; // queue buffers

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR s1, GM_ADDR s2, GM_ADDR ksz, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t poolK, uint32_t ncNumPerBlock)
    {
        this->n_ = n;
        this->c_ = c;
        this->h_ = h;
        this->w_ = w;
        this->poolK_ = poolK;
        // Partition over N*C by block
        uint32_t total_nc = n_ * c_;
        uint32_t block_num = AscendC::GetBlockNum();
        uint32_t per = (total_nc + block_num - 1) / block_num;
        this->ncBegin_ = per * AscendC::GetBlockIdx();
        this->ncEnd_ = this->ncBegin_ + per;
        if (this->ncEnd_ > total_nc) this->ncEnd_ = total_nc;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, (uint64_t)n_ * c_ * h_ * w_);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, (uint64_t)n_ * c_ * (h_ / poolK_) * (w_ / poolK_));
        s1Gm.SetGlobalBuffer((__gm__ DTYPE*)s1, 1);
        s2Gm.SetGlobalBuffer((__gm__ DTYPE*)s2, 1);
        kGm.SetGlobalBuffer((__gm__ int32_t*)ksz, 1);

        // Buffer for rows and intermediates of width w
        pipe.InitBuffer(inRow0Q, BUFFER_NUM, w_ * sizeof(DTYPE));
        pipe.InitBuffer(inRow1Q, BUFFER_NUM, w_ * sizeof(DTYPE));
        pipe.InitBuffer(tmpQ, BUFFER_NUM, w_ * sizeof(DTYPE));      // for intermediate per-row ops
        pipe.InitBuffer(tmpQ2, BUFFER_NUM, w_ * sizeof(DTYPE));     // second intermediate
        pipe.InitBuffer(poolRowQ, BUFFER_NUM, (w_ / poolK_) * sizeof(DTYPE)); // pooled row
    }

    __aicore__ inline void Process()
    {
        // Load scalar subtract values and kernel size
        DTYPE subtract1 = 0.f, subtract2 = 0.f;
        int32_t ksize = 2;
        {
            AscendC::LocalTensor<DTYPE> s1Local = tmpQ.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> s2Local = tmpQ.AllocTensor<DTYPE>();
            AscendC::DataCopy(s1Local, s1Gm, 1);
            AscendC::DataCopy(s2Local, s2Gm, 1);
            subtract1 = s1Local.GetValue(0);
            subtract2 = s2Local.GetValue(0);
            tmpQ.FreeTensor(s1Local);
            tmpQ.FreeTensor(s2Local);
            AscendC::LocalTensor<int32_t> kLocal = inRow0Q.AllocTensor<int32_t>();
            AscendC::DataCopy(kLocal, kGm, 1);
            ksize = kLocal.GetValue(0);
            inRow0Q.FreeTensor(kLocal);
        }
        if (ksize != 2) {
            // Only ksize=2 supported in this sample
            ksize = 2;
        }
        this->poolK_ = ksize;

        // Prepare constant vectors for scalar ops
        AscendC::LocalTensor<DTYPE> s1Vec = tmpQ.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> s2Vec = tmpQ2.AllocTensor<DTYPE>();
        AscendC::Duplicate(s1Vec, subtract1, w_);
        AscendC::Duplicate(s2Vec, subtract2, w_);

        DTYPE invKK = static_cast<DTYPE>(1.0f / (ksize * ksize));
        AscendC::LocalTensor<DTYPE> invVec = tmpQ.AllocTensor<DTYPE>();
        AscendC::Duplicate(invVec, invKK, (uint32_t)(w_ / poolK_));

        for (uint32_t nc = ncBegin_; nc < ncEnd_; ++nc) {
            uint32_t nIdx = nc / c_;
            uint32_t cIdx = nc % c_;
            uint64_t baseIn = ((uint64_t)nIdx * c_ + cIdx) * h_ * w_;
            uint64_t baseOut = ((uint64_t)nIdx * c_ + cIdx) * (h_ / poolK_) * (w_ / poolK_);
            for (uint32_t hh = 0; hh < h_; hh += poolK_) {
                // Copy two rows
                AscendC::LocalTensor<DTYPE> row0 = inRow0Q.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> row1 = inRow1Q.AllocTensor<DTYPE>();
                AscendC::DataCopy(row0, xGm[baseIn + (uint64_t)hh * w_], w_);
                AscendC::DataCopy(row1, xGm[baseIn + (uint64_t)(hh + 1) * w_], w_);
                inRow0Q.EnQue(row0);
                inRow1Q.EnQue(row1);

                // DeQue row0, process: (row - s1) -> tanh -> (result - s2)
                AscendC::LocalTensor<DTYPE> r0 = inRow0Q.DeQue<DTYPE>();
                AscendC::LocalTensor<DTYPE> r0m = tmpQ.AllocTensor<DTYPE>();
                AscendC::Sub(r0m, r0, s1Vec, w_);
                AscendC::LocalTensor<DTYPE> r0t = tmpQ2.AllocTensor<DTYPE>();
                AscendC::Tanh(r0t, r0m, w_);
                AscendC::LocalTensor<DTYPE> r0f = tmpQ.AllocTensor<DTYPE>();
                AscendC::Sub(r0f, r0t, s2Vec, w_);
                // Pair reduce sum horizontally: W -> W/2
                AscendC::LocalTensor<DTYPE> r0p = poolRowQ.AllocTensor<DTYPE>();
                AscendC::PairReduceSum(r0p, r0f, (uint32_t)(w_ / 2));
                tmpQ2.FreeTensor(r0t);
                tmpQ.FreeTensor(r0m);
                tmpQ.FreeTensor(r0f);
                inRow0Q.FreeTensor(r0);

                // Process row1 similarly
                AscendC::LocalTensor<DTYPE> r1 = inRow1Q.DeQue<DTYPE>();
                AscendC::LocalTensor<DTYPE> r1m = tmpQ.AllocTensor<DTYPE>();
                AscendC::Sub(r1m, r1, s1Vec, w_);
                AscendC::LocalTensor<DTYPE> r1t = tmpQ2.AllocTensor<DTYPE>();
                AscendC::Tanh(r1t, r1m, w_);
                AscendC::LocalTensor<DTYPE> r1f = tmpQ.AllocTensor<DTYPE>();
                AscendC::Sub(r1f, r1t, s2Vec, w_);
                AscendC::LocalTensor<DTYPE> r1p = poolRowQ.AllocTensor<DTYPE>();
                AscendC::PairReduceSum(r1p, r1f, (uint32_t)(w_ / 2));
                tmpQ2.FreeTensor(r1t);
                tmpQ.FreeTensor(r1m);
                tmpQ.FreeTensor(r1f);
                inRow1Q.FreeTensor(r1);

                // Sum the two pooled rows and scale by 1/(k*k)
                AscendC::LocalTensor<DTYPE> pooled = poolRowQ.AllocTensor<DTYPE>();
                AscendC::Add(pooled, r0p, r1p, (uint32_t)(w_ / 2));
                poolRowQ.FreeTensor(r0p);
                poolRowQ.FreeTensor(r1p);
                AscendC::LocalTensor<DTYPE> pooledScaled = poolRowQ.AllocTensor<DTYPE>();
                AscendC::Mul(pooledScaled, pooled, invVec, (uint32_t)(w_ / poolK_));
                poolRowQ.FreeTensor(pooled);

                // Write out pooled row
                AscendC::DataCopy(yGm[baseOut + (uint64_t)(hh / poolK_) * (w_ / poolK_)], pooledScaled, (uint32_t)(w_ / poolK_));
                poolRowQ.FreeTensor(pooledScaled);
            }
        }

        tmpQ.FreeTensor(s1Vec);
        tmpQ2.FreeTensor(s2Vec);
        tmpQ.FreeTensor(invVec);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inRow0Q, inRow1Q;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQ, tmpQ2;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> poolRowQ;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;
    AscendC::GlobalTensor<DTYPE> s1Gm;
    AscendC::GlobalTensor<DTYPE> s2Gm;
    AscendC::GlobalTensor<int32_t> kGm;

    uint32_t n_, c_, h_, w_;
    uint32_t poolK_;
    uint32_t ncBegin_, ncEnd_;
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(GM_ADDR x, GM_ADDR subtract1, GM_ADDR subtract2, GM_ADDR pool_ksize, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, subtract1, subtract2, pool_ksize, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.pool_ksize, tiling_data.ncNumPerBlock);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep Conv2d in PyTorch, fuse (x - subtract1) -> tanh -> (x - subtract2) -> AvgPool2d into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super().__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = float(subtract1_value)
        self.subtract2_value = float(subtract2_value)
        # assume square kernel for pooling
        self.kernel_size_pool = int(kernel_size_pool)

    def forward(self, x):
        x = self.conv(x)
        # Wrap scalars as 0D/1D tensors for device execution
        s1 = torch.tensor([self.subtract1_value], dtype=x.dtype, device=x.device)
        s2 = torch.tensor([self.subtract2_value], dtype=x.dtype, device=x.device)
        k  = torch.tensor([self.kernel_size_pool], dtype=torch.int32, device=x.device)
        y = custom_ops_lib.conv2d_subtract_tanh_subtract_avg_pool_custom(x, s1, s2, k)
        return y
