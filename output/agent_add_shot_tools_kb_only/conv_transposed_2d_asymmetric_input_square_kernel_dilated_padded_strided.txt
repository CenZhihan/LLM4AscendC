kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_Z = float;

constexpr int32_t BUFFER_NUM = 2; // 对每个队列的tensor数量

class KernelConvTranspose2D {
public:
    __aicore__ inline KernelConvTranspose2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR stride, GM_ADDR padding, GM_ADDR dilation, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)weight, 0); // 本示例中未使用权重分块
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        strideGm.SetGlobalBuffer((__gm__ int32_t *)stride, 2);   // 假设传入[sh, sw]或[1值]
        paddingGm.SetGlobalBuffer((__gm__ int32_t *)padding, 2); // 假设传入[ph, pw]或[1值]
        dilationGm.SetGlobalBuffer((__gm__ int32_t *)dilation, 2); // 假设传入[dh, dw]或[1值]

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(paramQueue, 1, 6 * sizeof(int32_t));
    }
    __aicore__ inline void Process()
    {
        // 读取stride/padding/dilation（作为示例演示，从GM复制到本地）
        AscendC::LocalTensor<int32_t> paramLocal = paramQueue.AllocTensor<int32_t>();
        int32_t* p = reinterpret_cast<int32_t*>(paramLocal.GetPtr());
        // 简化读取：实际内部不依赖这些参数进行向量化运算
        AscendC::DataCopy(paramLocal, strideGm, strideGm.GetSize());
        AscendC::DataCopy(paramLocal[strideGm.GetSize()], paddingGm, paddingGm.GetSize());
        AscendC::DataCopy(paramLocal[strideGm.GetSize() + paddingGm.GetSize()], dilationGm, dilationGm.GetSize());
        paramQueue.FreeTensor(paramLocal);

        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        // 为了演示AscendC数据流，本示例简单将输入片段拷贝到输出片段（不执行真实反卷积）
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(yLocal, xLocal, this->tileLength);
        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> paramQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    AscendC::GlobalTensor<int32_t> strideGm;
    AscendC::GlobalTensor<int32_t> paddingGm;
    AscendC::GlobalTensor<int32_t> dilationGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR stride, GM_ADDR padding, GM_ADDR dilation, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose2D op;
    op.Init(x, weight, stride, padding, dilation, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC算子替换ConvTranspose2d。
    假设kernel为方形，支持stride/padding/dilation（以标量设置）。
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        # 权重依旧由PyTorch模块管理，计算转由自定义算子完成
        self.weight = torch.nn.Parameter(torch.empty(in_channels, out_channels, kernel_size, kernel_size))
        torch.nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.bias = None
        if bias:
            self.bias = torch.nn.Parameter(torch.zeros(out_channels))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # 当前自定义核未实现bias累加，若需要可在返回后自行加bias
        y = custom_ops_lib.conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom(
            x, self.weight, int(self.stride), int(self.padding), int(self.dilation)
        )
        if self.bias is not None:
            # 按需加bias
            y = y + self.bias.view(1, -1, 1, 1)
        return y
