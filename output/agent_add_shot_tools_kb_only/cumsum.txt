kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的张量数

class KernelCumsum {
public:
    __aicore__ inline KernelCumsum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outter, uint32_t inner, uint32_t tileNum)
    {
        this->outter = outter;
        this->inner = inner;
        this->tileNum = tileNum;

        // 每个tile处理的行数
        this->tileOuter = (this->outter + (this->tileNum * BUFFER_NUM) - 1) / (this->tileNum * BUFFER_NUM);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, this->outter * this->inner);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, this->outter * this->inner);

        // 为输入/输出和lastRow分配本地缓冲区
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileOuter * this->inner * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileOuter * this->inner * sizeof(DTYPE_Y));
        pipe.InitBuffer(lastRowQueue, BUFFER_NUM, this->inner * sizeof(DTYPE_Y));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint32_t startRow = progress * this->tileOuter;
        if (startRow >= this->outter) {
            // 无数据可处理
            AscendC::LocalTensor<DTYPE_X> dummy = inQueueX.AllocTensor<DTYPE_X>();
            inQueueX.EnQue(dummy);
            return;
        }
        uint32_t h = this->tileOuter;
        if (startRow + h > this->outter) {
            h = this->outter - startRow;
        }

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // 拷贝连续h*inner个元素
        AscendC::DataCopy(xLocal, xGm[startRow * this->inner], h * this->inner);
        inQueueX.EnQue(xLocal);
        validHeights[progress] = h;
    }
    __aicore__ inline void Compute(int32_t progress)
    {
        uint32_t h = validHeights[progress];
        if (h == 0) {
            // 占位释放
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
            inQueueX.FreeTensor(xLocal);
            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
            outQueueY.EnQue<DTYPE_Y>(yLocal);
            return;
        }

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> lastRowLocal = lastRowQueue.AllocTensor<DTYPE_Y>();

        // 配置：沿最后一维计算前缀和，不复用输入，不输出最后一行
        static constexpr AscendC::CumSumConfig cumSumConfig{true, false, false};
        const AscendC::CumSumInfo cumSumInfo{h, this->inner};

        AscendC::CumSum<DTYPE_Y, cumSumConfig>(yLocal, lastRowLocal, xLocal, cumSumInfo);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        lastRowQueue.FreeTensor(lastRowLocal);
        inQueueX.FreeTensor(xLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        uint32_t startRow = progress * this->tileOuter;
        uint32_t h = validHeights[progress];
        if (h == 0) {
            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
            outQueueY.FreeTensor(yLocal);
            return;
        }

        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[startRow * this->inner], yLocal, h * this->inner);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> lastRowQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t outter{1};
    uint32_t inner{1};
    uint32_t tileNum{1};
    uint32_t tileOuter{1};

    // 记录每个progress的有效行数
    uint32_t validHeights[BUFFER_NUM * 64]; // 简单上限，足够覆盖tileNum*BUFFER_NUM
};

extern "C" __global__ __aicore__ void cumsum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsum op;
    op.Init(x, y, tiling_data.outter, tiling_data.inner, tiling_data.tileNum);
    op.Process();
}
