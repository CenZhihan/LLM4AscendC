kernel_src="""
#include "kernel_operator.h"

using DTYPE = half;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue
constexpr int32_t REDUCE_CHUNK = 128; // use PairReduceSum on 128 half elements

class KernelSigmoidSum {
public:
    __aicore__ inline KernelSigmoidSum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLenPerBatch, uint32_t tileNum)
    {
        this->blockLength = totalLenPerBatch;
        this->tileNum = tileNum;
        // Align tile length to 128 to simplify reduction
        uint32_t rawTile = (this->blockLength / this->tileNum / BUFFER_NUM);
        this->tileLength = (rawTile / REDUCE_CHUNK) * REDUCE_CHUNK;
        if (this->tileLength == 0) {
            this->tileLength = REDUCE_CHUNK;
        }

        // One block processes one batch: offset by block idx
        xGm.SetGlobalBuffer((__gm__ DTYPE *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y + AscendC::GetBlockIdx(), 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(tmpQueueA, 1, REDUCE_CHUNK * sizeof(DTYPE));
        pipe.InitBuffer(tmpQueueB, 1, (REDUCE_CHUNK / 2) * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, 1, sizeof(DTYPE)); // final scalar per batch
    }
    __aicore__ inline void Process()
    {
        // Initialize output to zero
        AscendC::LocalTensor<DTYPE> outScalar = outQueueY.AllocTensor<DTYPE>();
        DTYPE zero(0.0f);
        AscendC::Duplicate(outScalar, zero, 1);

        int32_t loopsPerBuffer = this->tileNum * BUFFER_NUM;
        int64_t processed = 0;

        for (int32_t i = 0; i < loopsPerBuffer; i++) {
            if (processed >= (int64_t)this->blockLength) { break; }
            uint32_t curLen = this->tileLength;
            if ((int64_t)curLen > ((int64_t)this->blockLength - processed)) {
                curLen = ((int64_t)this->blockLength - processed) / REDUCE_CHUNK * REDUCE_CHUNK;
                if (curLen == 0) { break; }
            }
            CopyIn(i, processed, curLen);
            ComputeReduceAdd(curLen, outScalar);
            processed += curLen;
        }

        // Write back final scalar
        outQueueY.EnQue(outScalar);
        CopyOut();
        outQueueY.FreeTensor(outScalar);
    }

private:
    __aicore__ inline void CopyIn(int32_t progress, int64_t offset, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[offset], len);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeReduceAdd(uint32_t len, AscendC::LocalTensor<DTYPE>& outScalar)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        // Apply sigmoid in-place: y = sigmoid(x)
        AscendC::Sigmoid(xLocal, xLocal, len);

        // Accumulate sum over this tile in chunks of 128
        uint32_t chunks = len / REDUCE_CHUNK;
        for (uint32_t c = 0; c < chunks; ++c) {
            // Copy 128 elems into tempA
            AscendC::LocalTensor<DTYPE> tempA = tmpQueueA.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> tempB = tmpQueueB.AllocTensor<DTYPE>();
            AscendC::DataCopy(tempA, xLocal[c * REDUCE_CHUNK], REDUCE_CHUNK);

            // Reduce 128 -> 64
            {
                uint64_t mask128 = REDUCE_CHUNK;
                AscendC::PairReduceSum<DTYPE>(tempB, tempA, 1, mask128, 1, 1, 8);
            }
            // 64 -> 32
            {
                uint64_t mask64 = REDUCE_CHUNK / 2;
                AscendC::PairReduceSum<DTYPE>(tempA, tempB, 1, mask64, 1, 1, 8);
            }
            // 32 -> 16
            {
                uint64_t mask32 = REDUCE_CHUNK / 4;
                AscendC::PairReduceSum<DTYPE>(tempB, tempA, 1, mask32, 1, 1, 8);
            }
            // 16 -> 8
            {
                uint64_t mask16 = REDUCE_CHUNK / 8;
                AscendC::PairReduceSum<DTYPE>(tempA, tempB, 1, mask16, 1, 1, 8);
            }
            // 8 -> 4
            {
                uint64_t mask8 = REDUCE_CHUNK / 16;
                AscendC::PairReduceSum<DTYPE>(tempB, tempA, 1, mask8, 1, 1, 8);
            }
            // 4 -> 2
            {
                uint64_t mask4 = REDUCE_CHUNK / 32;
                AscendC::PairReduceSum<DTYPE>(tempA, tempB, 1, mask4, 1, 1, 8);
            }
            // 2 -> 1
            {
                uint64_t mask2 = REDUCE_CHUNK / 64;
                AscendC::PairReduceSum<DTYPE>(tempB, tempA, 1, mask2, 1, 1, 8);
            }

            // tempB[0] holds the sum of this 128-chunk
            // Accumulate into outScalar (length 1)
            AscendC::Add(outScalar, outScalar, tempB, 1);

            tmpQueueB.FreeTensor(tempB);
            tmpQueueA.FreeTensor(tempA);
        }

        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm, yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueueA;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueueB;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_avg_pool_sigmoid_sum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSigmoidSum op;
    op.Init(x, y, tiling_data.totalLenPerBatch, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized: keep Conv2d and AvgPool2d from PyTorch, fuse sigmoid + sum into custom AscendC kernel.
    Input/Output dtype for custom kernel is float16.
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.avg_pool = torch.nn.AvgPool2d(pool_kernel_size)

    def forward(self, x):
        x = self.conv(x)
        x = self.avg_pool(x)
        # fused sigmoid + sum over C,H,W
        x_half = x.to(torch.half)
        y = custom_ops_lib.conv2d_avg_pool_sigmoid_sum_custom(x_half)
        return y
