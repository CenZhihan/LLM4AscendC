kernel_src="""
#include "kernel_operator.h"

// 使用最后一维进行累加（isLastAxis = true）
template <typename T>
class KernelMaskedCumsum {
public:
    __aicore__ inline KernelMaskedCumsum() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y, uint32_t outter, uint32_t inner)
    {
        this->outer = outter;
        this->inner = inner;
        totalSize = outer * inner;

        xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(x), totalSize);
        maskGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(mask), totalSize);
        yGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(y), totalSize);

        pipe.InitBuffer(inQueueX, 1, totalSize * sizeof(T));
        pipe.InitBuffer(inQueueMask, 1, totalSize * sizeof(T));
        pipe.InitBuffer(outQueueY, 1, totalSize * sizeof(T));
        pipe.InitBuffer(lastRowQueue, 1, inner * sizeof(T));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX.AllocTensor<T>();
        AscendC::LocalTensor<T> maskLocal = inQueueMask.AllocTensor<T>();
        AscendC::DataCopy(xLocal, xGlobal, totalSize);
        AscendC::DataCopy(maskLocal, maskGlobal, totalSize);
        inQueueX.EnQue(xLocal);
        inQueueMask.EnQue(maskLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX.DeQue<T>();
        AscendC::LocalTensor<T> maskLocal = inQueueMask.DeQue<T>();
        AscendC::LocalTensor<T> yLocal = outQueueY.AllocTensor<T>();
        AscendC::LocalTensor<T> lastRowLocal = lastRowQueue.AllocTensor<T>();

        // 先做逐元素乘法：x = x * mask
        AscendC::Mul(xLocal, xLocal, maskLocal, totalSize);

        static constexpr AscendC::CumSumConfig cumSumConfig{true, false, true};
        const AscendC::CumSumInfo cumSumInfo{outer, inner};
        AscendC::CumSum<T, cumSumConfig>(yLocal, lastRowLocal, xLocal, cumSumInfo);

        outQueueY.EnQue<T>(yLocal);
        lastRowQueue.EnQue<T>(lastRowLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueMask.FreeTensor(maskLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<T> yLocal = outQueueY.DeQue<T>();
        AscendC::DataCopy(yGlobal, yLocal, totalSize);
        outQueueY.FreeTensor(yLocal);

        AscendC::LocalTensor<T> lastRowLocal = lastRowQueue.DeQue<T>();
        // 不需要把最后一行输出拷回GM，直接释放
        lastRowQueue.FreeTensor(lastRowLocal);
    }

private:
    AscendC::GlobalTensor<T> xGlobal;
    AscendC::GlobalTensor<T> maskGlobal;
    AscendC::GlobalTensor<T> yGlobal;

    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueMask;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> lastRowQueue;

    uint32_t outer{1};
    uint32_t inner{1};
    uint32_t totalSize{0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsum<float> op;
    op.Init(x, mask, y, tiling_data.outter, tiling_data.inner);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.dim = dim  # 当前实现按最后一维计算（dim应为-1或最后一维）

        # 将bool掩码转换为与x相同的数据类型以便在NPU侧进行乘法
