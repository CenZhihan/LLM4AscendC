kernel_src="""
#include "kernel_operator.h"

// 采用按通道块(N*C个block，每block大小为H*W)的tile策略
constexpr int32_t BUFFER_NUM = 1;

class KernelMishBN {
public:
    __aicore__ inline KernelMishBN() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR shift, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t channels, uint32_t hw)
    {
        this->channels = channels;
        this->hw = hw;

        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum; // 期望等于N*C
        this->tileLength = hw;   // 每个tile处理一个通道块(H*W)

        // 设定全局内存缓冲
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // scale/shift为长度为C的向量，不做分块
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_X*)scale, this->channels);
        shiftGm.SetGlobalBuffer((__gm__ DTYPE_X*)shift, this->channels);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(inQueueS, 1, sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, 1, sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < this->tileNum; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress)
    {
        // 输入特征块
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);

        // 通道索引
        uint32_t cIdx = progress % this->channels;

        // 对应通道的scale/shift标量
        AscendC::LocalTensor<DTYPE_X> sLocal = inQueueS.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> bLocal = inQueueB.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(sLocal, scaleGm[cIdx], 1);
        AscendC::DataCopy(bLocal, shiftGm[cIdx], 1);
        inQueueS.EnQue(sLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Mish(x) = x * tanh(softplus(x)) = x * tanh(ln(1 + exp(x)))
        // 1) t = exp(x)
        AscendC::Exp(yLocal, xLocal, this->tileLength);
        // 2) t = t + 1
        AscendC::Adds(yLocal, yLocal, (DTYPE_Z)1.0f, this->tileLength);
        // 3) t = ln(t)
        AscendC::Ln(yLocal, yLocal, this->tileLength);
        // 4) t = tanh(t)
        AscendC::Tanh(yLocal, yLocal, this->tileLength);
        // 5) y = x * t
        AscendC::Mul(yLocal, xLocal, yLocal, this->tileLength);

        // BN仿射：y = y * scale + shift（scale/shift按通道预先计算）
        AscendC::LocalTensor<DTYPE_X> sLocal = inQueueS.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> bLocal = inQueueB.DeQue<DTYPE_X>();
        DTYPE_X scale = sLocal.GetValue(0);
        DTYPE_X bias  = bLocal.GetValue(0);

        AscendC::Muls(yLocal, yLocal, (DTYPE_Z)scale, this->tileLength);
        AscendC::Adds(yLocal, yLocal, (DTYPE_Z)bias,  this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueS.FreeTensor(sLocal);
        inQueueB.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueS; // scale
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueB; // shift

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> scaleGm;
    AscendC::GlobalTensor<DTYPE_X> shiftGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t channels;
    uint32_t hw;
};

extern "C" __global__ __aicore__ void conv2d_activation_batch_norm_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR shift, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelMishBN op;
    op.Init(x, scale, shift, y,
            tilingData.totalLength(),
            tilingData.tileNum(),
            tilingData.channels(),
            tilingData.hw());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import torch.nn as nn
import torch.nn.functional as F

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)

    def forward(self, x):
        # 先执行Conv2d
        x = self.conv(x)

        # 从BatchNorm参数计算通道尺度与偏置：scale = gamma / sqrt(var + eps), shift = beta - mean * scale
        # 此计算在CPU/PyTorch侧完成，避免在自定义核内重复广播
        running_mean = self.bn.running_mean
        running_var = self.bn.running_var
        gamma = self.bn.weight if self.bn.weight is not None else torch.ones_like(running_mean)
        beta = self.bn.bias if self.bn.bias is not None else torch.zeros_like(running_mean)
        eps = self.bn.eps

        scale = gamma / torch.sqrt(running_var + eps)
        shift = beta - running_mean * scale

        # 将scale/shift移动到x所在设备与类型
        scale = scale.to(device=x.device, dtype=x.dtype)
        shift = shift.to(device=x.device, dtype=x.dtype)

        # 自定义AscendC核：实现 Mish(x) 后进行通道仿射(y = y * scale + shift)
        # 期望输入x为[N,C,H,W]，scale/shift为[C]
        y = custom_ops_lib.conv2d_activation_batch_norm_custom(x, scale, shift)
        return y
'''
