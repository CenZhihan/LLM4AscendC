#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    LogSoftmaxCustomTilingData tiling;

    // 计算输入形状 m(非最后一维的乘积) 和 k(最后一维)
    const gert::Shape* inShapePtr = context->GetInputShape(0);
    auto origin = inShapePtr->GetOriginShape();
    int64_t dimNum = origin.GetDimNum();
    int64_t k = origin.GetDim(dimNum - 1);
    int64_t m = 1;
    for (int64_t i = 0; i < dimNum - 1; ++i) {
        m *= origin.GetDim(i);
    }

    // 设置并行 block 数
    const uint32_t BLOCK_DIM = 32;
    context->SetBlockDim(BLOCK_DIM);

    // 数据类型大小
    auto dt = context->GetInputTensor(0)->GetDataType();
    uint32_t dtypesize = (dt == ge::DT_FLOAT16) ? 2 : 4;

    // 计算临时空间和 tiling
    std::vector<int64_t> srcDims = {m, k};
    ge::Shape shape(srcDims);
    bool isReuseSource = false;
    uint32_t tmpSize = AscendC::GetLogSoftMaxMaxTmpSize(shape, dtypesize, isReuseSource);
    AscendC::LogSoftMaxTilingFunc(shape, dtypesize, tmpSize, tiling.logSoftmaxTilingData);

    // 设置 tiling 数据
    tiling.set_totalM(static_cast<uint32_t>(m));
    tiling.set_k(static_cast<uint32_t>(k));
    tiling.set_tileRows(1);  // 每次处理1行，保证UB占用安全
    tiling.set_tmpSize(tmpSize);

    // 保存 tiling 数据
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // 工作空间
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);
    *out_shape = *in_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {

    explicit LogSoftmaxCustom(const char* name) : OpDef(name)

        this->Input("x")
            .ParamType(REQUIRED)

            .Format({ge::FORMAT_ND})

            .Format({ge::FORMAT_ND})

kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1;

class KernelLogSoftmax {
public:
    __aicore__ inline KernelLogSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalM, uint32_t k,
                                uint32_t tileRows, uint32_t tmpSize,
                                const AscendC::LogSoftMaxTiling &logsmTiling) {
        this->totalM = totalM;
        this->k = k;
        this->tileRows = tileRows;
        this->tmpSize = tmpSize;
        this->rowsPerBlock = totalM / AscendC::GetBlockNum();
        this->tileElem = tileRows * k;
        this->brcElemsPerRow = 32 / sizeof(DTYPE); // float: 8

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x + (size_t)rowsPerBlock * k * AscendC::GetBlockIdx(),
                            (size_t)rowsPerBlock * k);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y + (size_t)rowsPerBlock * k * AscendC::GetBlockIdx(),
                            (size_t)rowsPerBlock * k);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileElem * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileElem * sizeof(DTYPE));
        // max/sum 按BRC(32B)对齐，每行占 brcElemsPerRow 个元素
        pipe.InitBuffer(outQueueMax, BUFFER_NUM, this->tileRows * this->brcElemsPerRow * sizeof(DTYPE));
        pipe.InitBuffer(outQueueSum, BUFFER_NUM, this->tileRows * this->brcElemsPerRow * sizeof(DTYPE));
        if (this->tmpSize != 0) {
            pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tmpSize);
        }
        this->logsmTiling = logsmTiling;
    }

    __aicore__ inline void Process() {
        int32_t loopCount = rowsPerBlock / tileRows;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress) {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[(size_t)progress * tileElem], tileElem);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/) {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();

        AscendC::LocalTensor<DTYPE> maxLocal = outQueueMax.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> sumLocal = outQueueSum.AllocTensor<DTYPE>();

        AscendC::LocalTensor<uint8_t> tmpLocal;
        if (this->tmpSize != 0) {
            tmpLocal = tmpQueue.AllocTensor<uint8_t>();
        }

        AscendC::SoftMaxShapeInfo shapeInfo = {tileRows, k, tileRows, k};

        // 调用 LogSoftMax 高级API
        AscendC::LogSoftMax<DTYPE, false>(yLocal, maxLocal, sumLocal, xLocal,
                                           tmpLocal, this->logsmTiling, shapeInfo);

        if (this->tmpSize != 0) {
            tmpQueue.FreeTensor(tmpLocal);
        }
        outQueueY.EnQue<DTYPE>(yLocal);
        inQueueX.FreeTensor(xLocal);
        outQueueMax.FreeTensor(maxLocal);
        outQueueSum.FreeTensor(sumLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress) {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[(size_t)progress * tileElem], yLocal, tileElem);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueMax;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueSum;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t totalM;
    uint32_t k;
    uint32_t tileRows;
    uint32_t tmpSize;
    uint32_t rowsPerBlock;
    uint32_t tileElem;
    uint32_t brcElemsPerRow;
    AscendC::LogSoftMaxTiling logsmTiling;
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelLogSoftmax op;
    op.Init(x, y,
            tilingData.totalM,
            tilingData.k,
            tilingData.tileRows,
            tilingData.tmpSize,
            tilingData.logSoftmaxTilingData);
    op.Process();
}
