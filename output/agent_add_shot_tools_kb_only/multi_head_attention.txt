kernel_src="""
#include "kernel_operator.h"

// 简化的缩放点积注意力：对每个 (b, h) 计算 [S, D] 上的 softmax(QK^T/scale) @ V
// 输入/输出均为 float，shape: [B, H, S, D]
class KernelMHA {
public:
    __aicore__ inline KernelMHA() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o,
                                uint32_t B, uint32_t H, uint32_t S, uint32_t D, float scale)
    {
        this->B = B;
        this->H = H;
        this->S = S;
        this->D = D;
        this->scale = scale;

        totalHeads = B * H;
        headIdx = AscendC::GetBlockIdx(); // 0..B*H-1
        if (totalHeads == 0) {
            totalHeads = 1;
        }
        if (headIdx >= totalHeads) {
            headIdx = totalHeads - 1;
        }

        // 将 GM 指针绑定为 GlobalTensor<float>
        qGm.SetGlobalBuffer((__gm__ float*)q, (uint64_t)B * H * S * D);
        kGm.SetGlobalBuffer((__gm__ float*)k, (uint64_t)B * H * S * D);
        vGm.SetGlobalBuffer((__gm__ float*)v, (uint64_t)B * H * S * D);
        oGm.SetGlobalBuffer((__gm__ float*)o, (uint64_t)B * H * S * D);

        // 为局部计算申请缓冲区（单 buffer）
        pipe.InitBuffer(qBuf, 1, D * sizeof(float));
        pipe.InitBuffer(kBuf, 1, D * sizeof(float));
        pipe.InitBuffer(vBuf, 1, D * sizeof(float));
        pipe.InitBuffer(tmpBuf, 1, D * sizeof(float));   // 临时乘积
        pipe.InitBuffer(outBuf, 1, D * sizeof(float));   // 累加/输出向量
        pipe.InitBuffer(scalarBuf, 1, sizeof(float));    // 用于 ReduceSum 的标量结果
    }

    __aicore__ inline void Process()
    {
        // head 基址偏移（单 head 跨度为 S*D）
        uint64_t headBase = (uint64_t)headIdx * S * D;

        for (uint32_t qi = 0; qi < S; ++qi) {
            // 加载 q[qi, :]
            AscendC::LocalTensor<float> qLocal = qBuf.AllocTensor<float>();
            AscendC::DataCopy(qLocal, qGm[headBase + (uint64_t)qi * D], D);

            // Pass 1: 计算 max_score
            float maxScore = -1.0e20f;
            for (uint32_t kj = 0; kj < S; ++kj) {
                AscendC::LocalTensor<float> kLocal = kBuf.AllocTensor<float>();
                AscendC::DataCopy(kLocal, kGm[headBase + (uint64_t)kj * D], D);

                // tmp = qLocal * kLocal
                AscendC::LocalTensor<float> tmpLocal = tmpBuf.AllocTensor<float>();
                AscendC::Mul(tmpLocal, qLocal, kLocal, D);

                // reduce sum -> 标量
                AscendC::LocalTensor<float> sumLocal = scalarBuf.AllocTensor<float>();
                AscendC::ReduceSum(sumLocal, tmpLocal, D);
                // 获取 ReduceSum 的累加值
                float dotVal = AscendC::GetAccVal(); // ISASI: 获取 ReduceSum 的计算结果
                dotVal = dotVal * this->scale;
                if (dotVal > maxScore) {
                    maxScore = dotVal;
                }

                kBuf.FreeTensor(kLocal);
                tmpBuf.FreeTensor(tmpLocal);
                scalarBuf.FreeTensor(sumLocal);
            }

            // Pass 2: 计算分子向量 numerator = sum_j exp(score_j - max) * v[j,:]
            //         以及分母 sumExp = sum_j exp(score_j - max)
            AscendC::LocalTensor<float> accumLocal = outBuf.AllocTensor<float>();
            AscendC::Duplicate(accumLocal, 0.0f, D);
            float sumExp = 0.0f;

            for (uint32_t kj = 0; kj < S; ++kj) {
                // 重新计算 score_j（q·k_j）
                AscendC::LocalTensor<float> kLocal2 = kBuf.AllocTensor<float>();
                AscendC::DataCopy(kLocal2, kGm[headBase + (uint64_t)kj * D], D);
                AscendC::LocalTensor<float> tmpLocal2 = tmpBuf.AllocTensor<float>();
                AscendC::Mul(tmpLocal2, qLocal, kLocal2, D);

                AscendC::LocalTensor<float> sumLocal2 = scalarBuf.AllocTensor<float>();
                AscendC::ReduceSum(sumLocal2, tmpLocal2, D);
                float dotVal2 = AscendC::GetAccVal();
                dotVal2 = dotVal2 * this->scale;

                float w = FastExp(dotVal2 - maxScore);
                sumExp += w;

                // 加载 v[kj, :] 并累加 w * v[kj,:]
                AscendC::LocalTensor<float> vLocal = vBuf.AllocTensor<float>();
                AscendC::DataCopy(vLocal, vGm[headBase + (uint64_t)kj * D], D);
                // vLocal = vLocal * w
                AscendC::Muls(vLocal, vLocal, w, D);
                // accumLocal += vLocal
                AscendC::Add(accumLocal, accumLocal, vLocal, D);

                kBuf.FreeTensor(kLocal2);
                tmpBuf.FreeTensor(tmpLocal2);
                scalarBuf.FreeTensor(sumLocal2);
                vBuf.FreeTensor(vLocal);
            }

            // 输出 = accumLocal / sumExp
            float invSum = 1.0f / sumExp;
            AscendC::Muls(accumLocal, accumLocal, invSum, D);

            // 写回 o[qi,:]
            AscendC::DataCopy(oGm[headBase + (uint64_t)qi * D], accumLocal, D);

            qBuf.FreeTensor(qLocal);
            outBuf.FreeTensor(accumLocal);
        }
    }

private:
    __aicore__ inline float FastExp(float x) const
    {
        // 简易近似或直接调用 Exp 的标量替代（这里直接调用向量指令不可行，故用近似）
        // 为简化，使用泰勒近似的简化版本；在实际优化中可替换为更稳定实现。
        // 但由于 AscendC 向量 API 有 Exp(zLocal, xLocal, len)，此处主要作为标量计算过渡。
        // 简单而稳定的做法是利用 C 运行时的 expf。
        // 在设备侧可直接返回近似值，保持示例完整性。
        // 这里假定内置支持。
        return __expf(x);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qBuf, kBuf, vBuf;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf, outBuf, scalarBuf;

    AscendC::GlobalTensor<float> qGm;
    AscendC::GlobalTensor<float> kGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> oGm;

    uint32_t B;
    uint32_t H;
    uint32_t S;
    uint32_t D;
    float scale;

    uint32_t totalHeads;
    uint32_t headIdx;
};

extern "C" __global__ __aicore__ void multi_head_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMHA op;
    op.Init(q, k, v, o,
            tiling_data.B, tiling_data.H, tiling_data.S, tiling_data.D, tiling_data.scale);
    op.Process();
}

model_src='''
import torch
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, d_model=512, num_heads=8):
        super().__init__()
        self.d_model = d_model
        self.num_heads = num_heads
        self.head_dim = d_model // num_heads
        assert d_model % num_heads == 0
        self.in_proj_weight = torch.nn.Parameter(torch.empty(3 * d_model, d_model))
        self.in_proj_bias = torch.nn.Parameter(torch.empty(3 * d_model))
        self.out_proj = torch.nn.Linear(d_model, d_model, bias=True)
        # 参数初始化，和 nn.MultiheadAttention 保持一致风格
        torch.nn.init.xavier_uniform_(self.in_proj_weight)
        torch.nn.init.zeros_(self.in_proj_bias)
        torch.nn.init.xavier_uniform_(self.out_proj.weight)
        torch.nn.init.zeros_(self.out_proj.bias)

    def forward(self, x):
        # x: [B, S, d_model], batch_first=True
        B, S, _ = x.shape
        d = self.d_model
        h = self.num_heads
        dh = self.head_dim

        # 拆分 Q/K/V 投影
        w = self.in_proj_weight
        b = self.in_proj_bias
        w_q, w_k, w_v = w[:d, :], w[d:2*d, :], w[2*d:3*d, :]
        b_q, b_k, b_v = b[:d], b[d:2*d], b[2*d:3*d]

        q = F.linear(x, w_q, b_q)  # [B,S,d]
        k = F.linear(x, w_k, b_k)
        v = F.linear(x, w_v, b_v)

        # 变形为 [B,H,S,D]
        q = q.view(B, S, h, dh).transpose(1, 2).contiguous()
        k = k.view(B, S, h, dh).transpose(1, 2).contiguous()
        v = v.view(B, S, h, dh).transpose(1, 2).contiguous()

        # 调用自定义 AscendC 内核完成 scaled dot-product attention
        ctx = custom_ops_lib.multi_head_attention_custom(q, k, v)  # [B,H,S,D]

        # 合并头并做输出投影
        ctx = ctx.transpose(1, 2).contiguous().view(B, S, d)       # [B,S,d]
        out = self.out_proj(ctx)
        return out
'''
