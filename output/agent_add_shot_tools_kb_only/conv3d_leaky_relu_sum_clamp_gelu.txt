kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_B = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor个数

class KernelConv3dLeakyReluSumClampGelu {
public:
    __aicore__ inline KernelConv3dLeakyReluSumClampGelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->totalLength = totalLength;
        this->blockLength = (totalLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        // 起始偏移与实际负责长度（最后一个block可能更短）
        uint32_t start = this->blockLength * AscendC::GetBlockIdx();
        if (start + this->blockLength > totalLength) {
            this->blockLength = totalLength - start;
        }
        this->tileNum = tileNum > 0 ? tileNum : 1;
        // 基于BUFFER_NUM划分tile，保证tileLength>=1
        this->tileLength = (this->blockLength + (this->tileNum * BUFFER_NUM) - 1) / (this->tileNum * BUFFER_NUM);
        if (this->tileLength == 0) this->tileLength = 1;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + start, this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)bias + start, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + start, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        // 计算中间缓冲
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (this->blockLength == 0) {
            return;
        }
        int32_t loopCount = (this->blockLength + this->tileLength - 1) / this->tileLength;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint32_t offset = progress * this->tileLength;
        uint32_t curLen = this->tileLength;
        if (offset + curLen > this->blockLength) {
            curLen = this->blockLength - offset;
        }

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
        AscendC::DataCopy(xLocal, xGm[offset], curLen);
        AscendC::DataCopy(bLocal, bGm[offset], curLen);
        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
        curLens[progress] = curLen;
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        uint32_t curLen = curLens[progress];

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.DeQue<DTYPE_B>();
        AscendC::LocalTensor<DTYPE_Y> tmpLocal = tmpQueue.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // 1) LeakyReLU, negative_slope=0.2
        const DTYPE_Y negSlope = (DTYPE_Y)0.2f;
        AscendC::LeakyRelu(tmpLocal, xLocal, negSlope, (int32_t)curLen);

        // 2) 加上已经广播到同形状的bias
        AscendC::Add(tmpLocal, tmpLocal, bLocal, curLen);

        // 3) clamp到[-1.0, 1.0]
        AscendC::Maxs(tmpLocal, tmpLocal, (DTYPE_Y)(-1.0f), curLen);
        AscendC::Mins(tmpLocal, tmpLocal, (DTYPE_Y)(1.0f), curLen);

        // 4) GELU
        AscendC::Gelu(yLocal, tmpLocal, curLen);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        tmpQueue.FreeTensor(tmpLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        uint32_t offset = progress * this->tileLength;
        uint32_t curLen = curLens[progress];

        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[offset], yLocal, curLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    // 简单记录每次长度以处理尾块
    uint32_t curLens[1024]; // 足够覆盖常见tile次数，实际环境可根据需要调整
};

extern "C" __global__ __aicore__ void conv3d_leaky_relu_sum_clamp_gelu_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dLeakyReluSumClampGelu op;
    op.Init(x, bias, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    先用原生Conv3d完成卷积，再用自定义AscendC内核完成 LeakyReLU(+0.2) -> 加法 -> clamp[-1,1] -> GELU 的融合计算。
    由于内核实现的是与输入同形状的按元素运算，调用前请将 sum_tensor 扩展到与卷积输出相同的形状。
    """
    def __init__(self, in_channels, out_channels, kernel_size, sum_tensor_shape):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        # 仍然保持参数形状为 (C,1,1,1)
        self.sum_tensor = torch.nn.Parameter(torch.randn(sum_tensor_shape))

    def forward(self, x):
        y = self.conv(x)  # [N, C, D, H, W]
        # 将 sum_tensor 扩展到与 y 相同形状，再交由自定义核完成 LeakyReLU->Add->Clamp->GELU
        bias = self.sum_tensor.expand(y.shape[0], y.shape[1], y.shape[2], y.shape[3], y.shape[4])
        # 确保连续内存
        bias = bias.contiguous()
        return custom_ops_lib.conv3d_leaky_relu_sum_clamp_gelu_custom(y, bias)
'''
