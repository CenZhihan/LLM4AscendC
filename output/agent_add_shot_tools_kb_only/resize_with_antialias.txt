namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();
    // Expect NCHW
    uint32_t n = (origin.GetDimNum() > 0) ? origin.GetDim(0) : 1;
    uint32_t c = (origin.GetDimNum() > 1) ? origin.GetDim(1) : 1;
    uint32_t inH = (origin.GetDimNum() > 2) ? origin.GetDim(2) : 1;
    uint32_t inW = (origin.GetDimNum() > 3) ? origin.GetDim(3) : 1;

    // Fixed target size per task requirement
    uint32_t outH = 128;
    uint32_t outW = 128;

    // Each tile processes one output row (length outW) for a (n,c,oh)
    uint32_t tileNum = n * c * outH;

    // Set reasonable block dim
    uint32_t blockDim = 8;
    if (tileNum < blockDim) blockDim = tileNum == 0 ? 1 : tileNum;
    context->SetBlockDim(blockDim);

    ResizeWithAntialiasCustomTilingData tiling;
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_tileNum(tileNum);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    // Keep N, C from input; set H, W to 128, 128
    int64_t n = (x_shape->GetDimNum() > 0) ? x_shape->GetDim(0) : 1;
    int64_t c = (x_shape->GetDimNum() > 1) ? x_shape->GetDim(1) : 1;
    std::vector<int64_t> out_dims = {n, c, 128, 128};
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // tiles buffered

class KernelResizeWithAntialias {
public:
    __aicore__ inline KernelResizeWithAntialias() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW,
                                uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->inH = inH;
        this->inW = inW;
        this->outH = outH;
        this->outW = outW;
        this->tileNum = tileNum;

        uint64_t totalIn = (uint64_t)n * c * inH * inW;
        uint64_t totalOut = (uint64_t)n * c * outH * outW;

        // Split tiles across blocks: each tile is one output row for a (n,c,oh)
        this->tilesPerBlock = tileNum / AscendC::GetBlockNum();
        this->startTile = this->tilesPerBlock * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, totalIn);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y, totalOut);

        // Each tile needs to load two input row segments of length 2*outW (contiguous),
        // and produce one output row of length outW.
        rowSegLen = 2 * outW;
        outRowLen = outW;

        pipe.InitBuffer(inQueueRow0, BUFFER_NUM, rowSegLen * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueRow1, BUFFER_NUM, rowSegLen * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueue0, BUFFER_NUM, outRowLen * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueue1, BUFFER_NUM, outRowLen * sizeof(DTYPE_Z));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, outRowLen * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tilesPerBlock * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void TileNCohFromIndex(uint32_t tileIndex, uint32_t &tn, uint32_t &tc, uint32_t &toh)
    {
        // tileIndex in [0, N*C*outH)
        uint32_t perNC = outH;
        uint32_t nc_index = tileIndex / perNC;
        toh = tileIndex % perNC;
        tc = nc_index % c;
        tn = nc_index / c;
    }

    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint32_t tileIndex = this->startTile + (uint32_t)progress;
        uint32_t tn, tc, toh;
        TileNCohFromIndex(tileIndex, tn, tc, toh);

        uint32_t ih0 = toh * 2;

        uint64_t row0Offset = (((uint64_t)tn * c + tc) * inH + ih0) * inW;
        uint64_t row1Offset = row0Offset + inW;

        AscendC::LocalTensor<DTYPE_X> row0Local = inQueueRow0.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> row1Local = inQueueRow1.AllocTensor<DTYPE_X>();

        AscendC::DataCopy(row0Local, xGm[row0Offset], rowSegLen);
        AscendC::DataCopy(row1Local, xGm[row1Offset], rowSegLen);

        inQueueRow0.EnQue(row0Local);
        inQueueRow1.EnQue(row1Local);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> row0Local = inQueueRow0.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> row1Local = inQueueRow1.DeQue<DTYPE_X>();

        AscendC::LocalTensor<DTYPE_Z> tmp0 = tmpQueue0.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> tmp1 = tmpQueue1.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Horizontal pairwise sum for two rows: sum of adjacent elements
        AscendC::Add(tmp0, row0Local, row0Local[1], outRowLen);
        AscendC::Add(tmp1, row1Local, row1Local[1], outRowLen);

        // Sum the two rows and scale by 0.25 (average of 2x2)
        AscendC::Add(tmp0, tmp0, tmp1, outRowLen);
        AscendC::Muls(yLocal, tmp0, (DTYPE_Z)0.25f, outRowLen);

        outQueueY.EnQue<DTYPE_Z>(yLocal);

        inQueueRow0.FreeTensor(row0Local);
        inQueueRow1.FreeTensor(row1Local);
        tmpQueue0.FreeTensor(tmp0);
        tmpQueue1.FreeTensor(tmp1);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        uint32_t tileIndex = this->startTile + (uint32_t)progress;
        uint32_t tn, tc, toh;
        TileNCohFromIndex(tileIndex, tn, tc, toh);

        uint64_t outOffset = (((uint64_t)tn * c + tc) * outH + toh) * outW;

        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[outOffset], yLocal, outRowLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueRow0;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueRow1;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> tmpQueue0;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> tmpQueue1;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t n, c, inH, inW, outH, outW;
    uint32_t tileNum;
    uint32_t tilesPerBlock;
    uint32_t startTile;
    uint32_t rowSegLen;
    uint32_t outRowLen;
};

extern "C" __global__ __aicore__ void resize_with_antialias_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelResizeWithAntialias op;
    op.Init(x, y,
            tilingData.n(),
            tilingData.c(),
            tilingData.inH(),
            tilingData.inW(),
            tilingData.outH(),
            tilingData.outW(),
            tilingData.tileNum());
    op.Process();
}
"""

at::Tensor resize_with_antialias_custom_impl_npu(const at::Tensor& x) {
    TORCH_CHECK(x.dim() == 4, "Input must be NCHW");
    auto opts = x.options();
    auto sizes = x.sizes();

    int64_t C = sizes[1];
    int64_t outH = 128;
    int64_t outW = 128;
    at::Tensor y = at::empty({N, C, outH, outW}, opts);
    EXEC_NPU_CMD(aclnnResizeWithAntialiasCustom, x, y);
    return y;
}
