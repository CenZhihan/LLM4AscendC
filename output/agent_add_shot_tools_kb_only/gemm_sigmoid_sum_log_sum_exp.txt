kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelGemmSigmoidLSE {
public:
    __aicore__ inline KernelGemmSigmoidLSE() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR w2, GM_ADDR b2, GM_ADDR y,
                                uint32_t batch, uint32_t hidden, uint32_t output, uint32_t tileRows)
    {
        this->batch = batch;
        this->hidden = hidden;
        this->output = output;
        this->tileRows = tileRows;

        uint32_t blkNum = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();
        // 简单均分batch（尾部舍去）
        this->rowsPerBlock = batch / blkNum;
        this->startRow = this->rowsPerBlock * blkIdx;

        aGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(a) + static_cast<uint64_t>(startRow) * hidden,
                            static_cast<uint64_t>(rowsPerBlock) * hidden);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(w2), static_cast<uint64_t>(output) * hidden);
        bGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(b2), output);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(y) + startRow, rowsPerBlock);

        // 队列与缓冲区初始化
        pipe.InitBuffer(inQueueA, 1, hidden * sizeof(DTYPE));          // 单行输入
        pipe.InitBuffer(inQueueW, 1, hidden * sizeof(DTYPE));          // 单行权重
        pipe.InitBuffer(calcQueue, 3, hidden * sizeof(DTYPE));         // sLocal/mulLocal/work
        pipe.InitBuffer(scalarQueue, 2, 8 * sizeof(DTYPE));            // 标量缓存（32字节对齐）
        pipe.InitBuffer(outQueue, 1, 8 * sizeof(DTYPE));               // 输出/临时标量
    }

    __aicore__ inline void Process()
    {
        uint32_t totalRows = rowsPerBlock; // 每核处理行数
        for (uint32_t r = 0; r < totalRows; ++r) {
            // 1) 读取一行 a[r,:] -> xLocal
            AscendC::LocalTensor<DTYPE> xLocal = inQueueA.AllocTensor<DTYPE>();
            AscendC::DataCopy(xLocal, aGm[r * hidden], hidden);
            inQueueA.EnQue(xLocal);

            // 2) Sigmoid 到 sLocal
            AscendC::LocalTensor<DTYPE> sLocal = calcQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> xLocalDeq = inQueueA.DeQue<DTYPE>();
            AscendC::Sigmoid(sLocal, xLocalDeq, hidden);
            inQueueA.FreeTensor(xLocalDeq);

            // 标量buffer：curMaxLocal, sumExpLocal
            AscendC::LocalTensor<DTYPE> curMaxLocal = scalarQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> sumExpLocal = scalarQueue.AllocTensor<DTYPE>();

            bool firstElem = true;

            // 3) 遍历输出维 j，执行 y_j = dot(sLocal, w2[j,:]) + b2[j]，并进行稳定的LogSumExp增量累加
            for (uint32_t j = 0; j < output; ++j) {
                // 3.1 读取权重行 w2[j,:]
                AscendC::LocalTensor<DTYPE> wLocal = inQueueW.AllocTensor<DTYPE>();
                AscendC::DataCopy(wLocal, wGm[j * hidden], hidden);
                inQueueW.EnQue(wLocal);
                AscendC::LocalTensor<DTYPE> wLocalDeq = inQueueW.DeQue<DTYPE>();

                // 3.2 逐元素乘法并ReduceSum得到点积
                AscendC::LocalTensor<DTYPE> mulLocal = calcQueue.AllocTensor<DTYPE>();
                AscendC::Mul(mulLocal, sLocal, wLocalDeq, hidden);

                AscendC::LocalTensor<DTYPE> workLocal = calcQueue.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> yScalar = outQueue.AllocTensor<DTYPE>(); // 标量(32B)
                AscendC::ReduceSum(yScalar, mulLocal, workLocal, hidden);

                // 3.3 加上bias
                AscendC::LocalTensor<DTYPE> bScalar = outQueue.AllocTensor<DTYPE>();
                AscendC::DataCopy(bScalar, bGm[j], 1);
                AscendC::Add(yScalar, yScalar, bScalar, 1);
                outQueue.FreeTensor(bScalar);

                // 3.4 稳定的logsumexp增量更新
                if (firstElem) {
                    // curMax = y0；sumExp = exp(0) = 1
                    AscendC::DataCopy(curMaxLocal, yScalar, 1);
                    AscendC::LocalTensor<DTYPE> zeroTmp = outQueue.AllocTensor<DTYPE>();
                    AscendC::Sub(zeroTmp, yScalar, yScalar, 1); // 0
                    AscendC::Exp(zeroTmp, zeroTmp, 1);          // 1
                    AscendC::DataCopy(sumExpLocal, zeroTmp, 1);
                    outQueue.FreeTensor(zeroTmp);
                    firstElem = false;
                } else {
                    // newMax = max(curMax, y)
                    AscendC::LocalTensor<DTYPE> newMax = outQueue.AllocTensor<DTYPE>();
                    AscendC::Max(newMax, curMaxLocal, yScalar, 1);

                    // factor_old = exp(curMax - newMax)
                    AscendC::LocalTensor<DTYPE> factorOld = outQueue.AllocTensor<DTYPE>();
                    AscendC::Sub(factorOld, curMaxLocal, newMax, 1);
                    AscendC::Exp(factorOld, factorOld, 1);

                    // factor_new = exp(y - newMax)
                    AscendC::LocalTensor<DTYPE> factorNew = outQueue.AllocTensor<DTYPE>();
                    AscendC::Sub(factorNew, yScalar, newMax, 1);
                    AscendC::Exp(factorNew, factorNew, 1);

                    // sumExp = sumExp * factor_old + factor_new
                    AscendC::FusedMulAdd(sumExpLocal, factorOld, factorNew, 1);

                    // curMax = newMax
                    AscendC::DataCopy(curMaxLocal, newMax, 1);

                    outQueue.FreeTensor(newMax);
                    outQueue.FreeTensor(factorOld);
                    outQueue.FreeTensor(factorNew);
                }

                outQueue.FreeTensor(yScalar);
                calcQueue.FreeTensor(mulLocal);
                calcQueue.FreeTensor(workLocal);
                inQueueW.FreeTensor(wLocalDeq);
            }

            // 4) lse = ln(sumExp) + curMax
            AscendC::LocalTensor<DTYPE> lseScalar = outQueue.AllocTensor<DTYPE>();
            AscendC::Ln(lseScalar, sumExpLocal, 1);
            AscendC::Add(lseScalar, lseScalar, curMaxLocal, 1);

            // 5) 写回输出 y[startRow + r]
            AscendC::DataCopy(yGm[r], lseScalar, 1);

            // 释放本行资源
            outQueue.FreeTensor(lseScalar);
            scalarQueue.FreeTensor(curMaxLocal);
            scalarQueue.FreeTensor(sumExpLocal);
            calcQueue.FreeTensor(sLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueA;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 3> calcQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 2> scalarQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue;

    AscendC::GlobalTensor<DTYPE> aGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t batch;
    uint32_t hidden;
    uint32_t output;
    uint32_t tileRows;

    uint32_t startRow;
    uint32_t rowsPerBlock;
};

extern "C" __global__ __aicore__ void gemm_sigmoid_sum_log_sum_exp_custom(GM_ADDR a, GM_ADDR w2, GM_ADDR b2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSigmoidLSE op;
    op.Init(a, w2, b2, y,
            tiling_data.batch, tiling_data.hidden, tiling_data.output, tiling_data.tileRows);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused kernel: Sigmoid after first Linear, then GEMM with second Linear's weights,
    add bias, and LogSumExp over features per sample.
    """
    def __init__(self, input_size, hidden_size, output_size):
        super(ModelNew, self).__init__()
        self.linear1 = torch.nn.Linear(input_size, hidden_size)
        self.linear2 = torch.nn.Linear(hidden_size, output_size)

    def forward(self, x):
        a = self.linear1(x)  # [B, H]
        # fused: sigmoid(a) -> gemm with linear2 -> logsumexp over dim=1
        y = custom_ops_lib.gemm_sigmoid_sum_log_sum_exp_custom(a, self.linear2.weight, self.linear2.bias)
        return y
