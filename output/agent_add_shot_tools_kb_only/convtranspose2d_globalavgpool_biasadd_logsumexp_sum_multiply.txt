kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR b, GM_ADDR y, uint32_t totalLength, uint32_t n, uint32_t c, uint32_t tileNum, float scale)
    {
        // 单核执行（BLOCK_DIM=1），按batch迭代，每次处理C个通道
        this->n = n;
        this->c = c;
        this->tileNum = tileNum;
        this->scale = scale;
        this->tileLength = c;

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x, totalLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE *)b, c);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y, n);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < this->tileNum; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = inQueueB.AllocTensor<DTYPE>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(bLocal, bGm, this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = inQueueB.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yScalarLocal = outQueueY.AllocTensor<DTYPE>(); // 长度1，保存归约结果

        // 逐通道：y = x + bias
        AscendC::Add(xLocal, xLocal, bLocal, this->tileLength);

        // exp
        AscendC::Exp(xLocal, xLocal, this->tileLength);

        // sum over channels -> yScalarLocal[0]
        AscendC::ReduceSum(yScalarLocal, xLocal, this->tileLength);

        // log
        AscendC::Ln(yScalarLocal, yScalarLocal, 1);

        // multiply by scale
        AscendC::Muls(yScalarLocal, yScalarLocal, this->scale, 1);

        outQueueY.EnQue<DTYPE>(yScalarLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE> yScalarLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[progress], yScalarLocal, 1);
        outQueueY.FreeTensor(yScalarLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t tileNum;
    uint32_t tileLength;
    float scale;
};

extern "C" __global__ __aicore__ void convtranspose2d_globalavgpool_biasadd_logsumexp_sum_multiply_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, bias, y,
            tiling_data.totalLength,
            tiling_data.n,
            tiling_data.c,
            tiling_data.tileNum,
            tiling_data.scale);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep ConvTranspose2d on PyTorch, fuse subsequent ops with a custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        # Deconvolution
        x = self.conv_transpose(x)
        # Global average pooling
        x = torch.mean(x, dim=(2, 3), keepdim=True)  # [N, C, 1, 1]
        # Fused: bias add + logsumexp over channel + sum(spatial=1x1) + multiply(10.0)
        y = custom_ops_lib.convtranspose2d_globalavgpool_biasadd_logsumexp_sum_multiply_custom(x, self.bias)
