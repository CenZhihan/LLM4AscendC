kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // 每个队列一个缓冲

class KernelDWConv {
public:
    __aicore__ inline KernelDWConv() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t hi, uint32_t wi,
                                uint32_t kh, uint32_t kw,
                                uint32_t ho, uint32_t wo,
                                uint32_t stride, uint32_t padding,
                                uint32_t tileWo, uint32_t ncPerBlock)
    {
        this->n = n;
        this->c = c;
        this->hi = hi;
        this->wi = wi;
        this->kh = kh;
        this->kw = kw;
        this->ho = ho;
        this->wo = wo;
        this->stride = stride;
        this->padding = padding;
        this->tileWo = tileWo;

        totalNC = n * c;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t start = blockIdx * ncPerBlock;
        uint32_t end = AscendC::Min(start + ncPerBlock, totalNC);
        this->startIdx = start;
        this->endIdx = end;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, static_cast<uint64_t>(n) * c * hi * wi);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, static_cast<uint64_t>(c) * kh * kw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, static_cast<uint64_t>(n) * c * ho * wo);

        // 工作缓冲：输入条带、权重、输出tile
        uint32_t stripeLen = tileWo + kw - 1 + 2 * padding; // 需要考虑padding
        uint32_t inStripeSize = kh * stripeLen;
        uint32_t weightSize = kh * kw;
        uint32_t outTileSize = tileWo;

        pipe.InitBuffer(inStripeQueue, BUFFER_NUM, inStripeSize * sizeof(DTYPE_X));
        pipe.InitBuffer(weightQueue , BUFFER_NUM, weightSize   * sizeof(DTYPE_W));
        pipe.InitBuffer(outTileQueue, BUFFER_NUM, outTileSize  * sizeof(DTYPE_Y));

        this->stripeLen = stripeLen;
        this->inStripeSize = inStripeSize;
        this->weightSize = weightSize;
        this->outTileSize = outTileSize;
    }

    __aicore__ inline void Process()
    {
        for (uint32_t idx = startIdx; idx < endIdx; ++idx) {
            uint32_t nIdx = idx / c;
            uint32_t cIdx = idx % c;

            // 拷贝当前通道的权重 (kh*kw)
            AscendC::LocalTensor<DTYPE_W> wLocal = weightQueue.AllocTensor<DTYPE_W>();
            uint64_t wBase = static_cast<uint64_t>(cIdx) * kh * kw;
            AscendC::DataCopy(wLocal, wGm[wBase], weightSize);
            weightQueue.EnQue(wLocal);

            for (uint32_t oh = 0; oh < ho; ++oh) {
                for (uint32_t owStart = 0; owStart < wo; owStart += tileWo) {
                    uint32_t curTileW = AscendC::Min(tileWo, wo - owStart);

                    // 分配/清零输入条带
                    AscendC::LocalTensor<DTYPE_X> inStripe = inStripeQueue.AllocTensor<DTYPE_X>();
                    AscendC::Duplicate(inStripe, (DTYPE_X)0, inStripeSize);

                    // 计算输入起始行
                    int32_t inRowBase = static_cast<int32_t>(oh) * static_cast<int32_t>(stride) - static_cast<int32_t>(padding);

                    // 为条带按kh逐行填充有效数据
                    for (uint32_t kr = 0; kr < kh; ++kr) {
                        int32_t srcRow = inRowBase + static_cast<int32_t>(kr);
                        if (srcRow < 0 || srcRow >= static_cast<int32_t>(hi)) {
                            continue; // 超出上/下边界留0
                        }
                        int32_t colStart = static_cast<int32_t>(owStart) * static_cast<int32_t>(stride) - static_cast<int32_t>(padding);
                        int32_t needLen = static_cast<int32_t>(curTileW) + static_cast<int32_t>(kw) - 1;

                        int32_t validL = 0;
                        int32_t validR = 0;
                        int32_t srcColStart = colStart;
                        if (srcColStart < 0) { validL = -srcColStart; srcColStart = 0; }
                        int32_t srcColEnd = colStart + needLen - 1;
                        if (srcColEnd >= static_cast<int32_t>(wi)) { validR = srcColEnd - static_cast<int32_t>(wi) + 1; }

                        int32_t copyLen = needLen - validL - validR;
                        if (copyLen > 0) {
                            uint64_t xBase = (((static_cast<uint64_t>(nIdx) * c + cIdx) * hi + static_cast<uint32_t>(srcRow)) * wi + static_cast<uint32_t>(srcColStart));
                            uint64_t dstOff = static_cast<uint64_t>(kr) * stripeLen + static_cast<uint32_t>(validL);
                            AscendC::DataCopy(inStripe[dstOff], xGm[xBase], static_cast<uint32_t>(copyLen));
                        }
                    }

                    // 计算输出tile
                    AscendC::LocalTensor<DTYPE_Y> yLocal = outTileQueue.AllocTensor<DTYPE_Y>();
                    AscendC::Duplicate(yLocal, (DTYPE_Y)0, outTileSize);

                    AscendC::LocalTensor<DTYPE_W> wLocalUse = weightQueue.DeQue<DTYPE_W>();

                    for (uint32_t tw = 0; tw < curTileW; ++tw) {
                        float acc = 0.f;
                        for (uint32_t kr = 0; kr < kh; ++kr) {
                            uint64_t inOffBase = static_cast<uint64_t>(kr) * stripeLen + tw;
                            uint64_t wOffBase  = static_cast<uint64_t>(kr) * kw;
                            for (uint32_t kc = 0; kc < kw; ++kc) {
                                // 逐点累加
                                float a = (float)inStripe.GetValue(inOffBase + kc);
                                float b = (float)wLocalUse.GetValue(wOffBase + kc);
                                acc += a * b;
                            }
                        }
                        yLocal.SetValue(tw, (DTYPE_Y)acc);
                    }

                    // 回收权重局部tensor并重新入队，供后续tile复用
                    weightQueue.EnQue(wLocalUse);

                    // 输出写回
                    uint64_t yBase = (((static_cast<uint64_t>(nIdx) * c + cIdx) * ho + oh) * wo + owStart);
                    AscendC::DataCopy(yGm[yBase], yLocal, curTileW);

                    // 释放局部tensor
                    outTileQueue.FreeTensor(yLocal);
                    inStripeQueue.FreeTensor(inStripe);
                }
            }

            // 释放权重
            AscendC::LocalTensor<DTYPE_W> wLocalDone = weightQueue.DeQue<DTYPE_W>();
            weightQueue.FreeTensor(wLocalDone);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> inStripeQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> weightQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> outTileQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, hi, wi, kh, kw, ho, wo, stride, padding, tileWo;
    uint32_t startIdx, endIdx, totalNC;

    uint32_t stripeLen;
    uint32_t inStripeSize;
    uint32_t weightSize;
    uint32_t outTileSize;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelDWConv op;
    op.Init(x, weight, y,
            tilingData.n,
            tilingData.c,
            tilingData.hi,
            tilingData.wi,
            tilingData.kh,
            tilingData.kw,
            tilingData.ho,
            tilingData.wo,
            tilingData.stride,
            tilingData.padding,
            tilingData.tileWo,
            tilingData.ncPerBlock);
    op.Process();
}
