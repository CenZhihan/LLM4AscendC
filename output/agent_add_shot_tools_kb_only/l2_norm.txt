#include "l2_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    L2NormCustomTilingData tiling;
    const gert::Shape* ishape = context->GetInputShape(0);
    auto origin = ishape->GetOriginShape();
    uint32_t aLength = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t rLength = static_cast<uint32_t>(origin.GetDim(1));
    // 32-byte alignment for float (4B) -> 8 elements
    uint32_t rLengthWithPadding = (rLength + 7) / 8 * 8;

    // set block dim
    const uint32_t BLOCK_DIM = 8;
    context->SetBlockDim(BLOCK_DIM);

    // choose a tile length along R axis, aligned to 64 elements (one repeat for float is 64 elems)
    uint32_t rTile = 4096;
    if (rTile > rLengthWithPadding) rTile = rLengthWithPadding;
    rTile = (rTile + 63) / 64 * 64;

    tiling.set_aLength(aLength);
    tiling.set_rLength(rLength);
    tiling.set_rLengthWithPadding(rLengthWithPadding);
    tiling.set_rTileLength(rTile);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);
    *out_shape = *in_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

typedef float DTYPE_X;
typedef float DTYPE_Y;

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 1;

class KernelL2Norm {
public:
    __aicore__ inline KernelL2Norm() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t aLength, uint32_t rLength, uint32_t rLengthWithPadding, uint32_t rTileLength)
    {
        this->aLength = aLength;
        this->rLength = rLength;
        this->rLengthWithPadding = rLengthWithPadding;
        this->rTileLength = rTileLength;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = (aLength + blockNum - 1) / blockNum;
        this->rowStart = this->rowsPerBlock * blockIdx;
        uint32_t rowEnd = this->rowStart + this->rowsPerBlock;
        if (rowEnd > aLength) rowEnd = aLength;
        this->rowCount = (rowEnd > this->rowStart) ? (rowEnd - this->rowStart) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)this->rowStart * this->rLengthWithPadding,
                            (uint64_t)this->rowCount * this->rLengthWithPadding);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + (uint64_t)this->rowStart * this->rLengthWithPadding,
                            (uint64_t)this->rowCount * this->rLengthWithPadding);

        pipe.InitBuffer(inQueueX, IN_BUF_NUM, this->rTileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, OUT_BUF_NUM, this->rTileLength * sizeof(DTYPE_Y));
        // temp buffer for squares
        pipe.InitBuffer(tmpSqQueue, 1, this->rTileLength * sizeof(float));
        // work buffer for ReduceSum, allocate conservative size
        pipe.InitBuffer(workQueue, 1, 2048);
        // small dst buffer for ReduceSum
        pipe.InitBuffer(dstRedQueue, 1, 64 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (this->rowCount == 0) {
            return;
        }
        for (uint32_t r = 0; r < this->rowCount; ++r) {
            float sumsq = 0.0f;
            // Pass 1: accumulate sum of squares
            for (uint32_t off = 0; off < this->rLengthWithPadding; off += this->rTileLength) {
                uint32_t len = this->rTileLength;
                if (off + len > this->rLengthWithPadding) {
                    len = this->rLengthWithPadding - off;
                }
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[(uint64_t)r * this->rLengthWithPadding + off], len);
                inQueueX.EnQue(xLocal);

                xLocal = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<float> sqLocal = tmpSqQueue.AllocTensor<float>();
                AscendC::Mul(sqLocal, xLocal, xLocal, len);

                AscendC::LocalTensor<uint8_t> workLocal = workQueue.AllocTensor<uint8_t>();
                AscendC::LocalTensor<float> dstRedLocal = dstRedQueue.AllocTensor<float>();
                AscendC::ReduceSum<float>(dstRedLocal, sqLocal, workLocal, len);
                float partial = AscendC::GetAccVal<float>();
                sumsq += partial;

                dstRedQueue.FreeTensor(dstRedLocal);
                workQueue.FreeTensor(workLocal);
                tmpSqQueue.FreeTensor(sqLocal);
                inQueueX.FreeTensor(xLocal);
            }

            float invnorm = 1.0f / sqrtf(sumsq);

            // Pass 2: scale by invnorm
            for (uint32_t off = 0; off < this->rLengthWithPadding; off += this->rTileLength) {
                uint32_t len = this->rTileLength;
                if (off + len > this->rLengthWithPadding) {
                    len = this->rLengthWithPadding - off;
                }
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[(uint64_t)r * this->rLengthWithPadding + off], len);
                inQueueX.EnQue(xLocal);

                xLocal = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
                AscendC::Muls(yLocal, xLocal, invnorm, len);
                outQueueY.EnQue(yLocal);
                inQueueX.FreeTensor(xLocal);

                yLocal = outQueueY.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[(uint64_t)r * this->rLengthWithPadding + off], yLocal, len);
                outQueueY.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpSqQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> workQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> dstRedQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t aLength;
    uint32_t rLength;
    uint32_t rLengthWithPadding;
    uint32_t rTileLength;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowCount;
};

extern "C" __global__ __aicore__ void l2_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelL2Norm op;
    op.Init(x, y,
            tilingData.aLength,
            tilingData.rLength,
            tilingData.rLengthWithPadding,
            tilingData.rTileLength);
    op.Process();
}
