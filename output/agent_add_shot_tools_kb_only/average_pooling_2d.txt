namespace {
inline uint32_t FloorDiv(uint32_t a, uint32_t b) {
    return a / b;
}

namespace optiling {
const uint32_t BLOCK_DIM = 1;
const uint32_t TILE_OW = 128;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // Input: NCHW
    const gert::Shape* in_shape = context->GetInputShape(0);
    auto real_shape = in_shape->GetOriginShape();
    uint32_t n = (real_shape.GetDimNum() > 0) ? (uint32_t)real_shape.GetDim(0) : 1;
    uint32_t c = (real_shape.GetDimNum() > 1) ? (uint32_t)real_shape.GetDim(1) : 1;
    uint32_t h = (real_shape.GetDimNum() > 2) ? (uint32_t)real_shape.GetDim(2) : 1;
    uint32_t w = (real_shape.GetDimNum() > 3) ? (uint32_t)real_shape.GetDim(3) : 1;

    int64_t k_attr = 1;
    int64_t s_attr = -1;
    int64_t p_attr = 0;
    (void)context->GetAttr("kernel_size", k_attr);
    (void)context->GetAttr("stride", s_attr);
    (void)context->GetAttr("padding", p_attr);

    uint32_t k = (uint32_t)k_attr;
    uint32_t s = (s_attr > 0) ? (uint32_t)s_attr : k;
    uint32_t p = (uint32_t)p_attr;

    // Output size (PyTorch style, integer division floors)
    // outH = floor((H + 2P - K)/S) + 1
    // outW = floor((W + 2P - K)/S) + 1
    uint32_t outH = 0, outW = 0;
    if (h + 2 * p >= k && w + 2 * p >= k) {
        outH = FloorDiv(h + 2 * p - k, s) + 1;
        outW = FloorDiv(w + 2 * p - k, s) + 1;
    }

    AveragePooling2dCustomTilingData tiling;
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_k(k);
    tiling.set_s(s);
    tiling.set_p(p);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_tileOw(TILE_OW);

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    auto s = x_shape->GetOriginShape();
    int64_t k_attr = 1, s_attr = -1, p_attr = 0;
    (void)context->GetAttr("kernel_size", k_attr);
    (void)context->GetAttr("stride", s_attr);
    (void)context->GetAttr("padding", p_attr);
    int64_t k = k_attr;
    int64_t stride = (s_attr > 0) ? s_attr : k;
    int64_t pad = p_attr;

    int64_t n = (s.GetDimNum() > 0) ? s.GetDim(0) : 1;
    int64_t c = (s.GetDimNum() > 1) ? s.GetDim(1) : 1;
    int64_t h = (s.GetDimNum() > 2) ? s.GetDim(2) : 1;
    int64_t w = (s.GetDimNum() > 3) ? s.GetDim(3) : 1;
    int64_t outH = (h + 2 * pad - k) >= 0 ? ((h + 2 * pad - k) / stride + 1) : 0;
    int64_t outW = (w + 2 * pad - k) >= 0 ? ((w + 2 * pad - k) / stride + 1) : 0;

    gert::Shape* y_shape = context->GetOutputShape(0);
    gert::Shape y;
    y.SetDimNum(4);
    y.SetDim(0, n);
    y.SetDim(1, c);
    y.SetDim(2, outH);
    y.SetDim(3, outW);
    *y_shape = y;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

// Kernel implements AvgPool2d with attributes (kernel_size, stride, padding).
// For simplicity and best performance in this demo, stride==kernel_size and padding==0 are recommended.

class KernelAveragePool2D {
public:
    __aicore__ inline KernelAveragePool2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t k, uint32_t s, uint32_t p,
                                uint32_t outH, uint32_t outW, uint32_t tileOw) {
        this->N = n; this->C = c; this->H = h; this->W = w;
        this->K = k; this->S = s; this->P = p;
        this->OH = outH; this->OW = outW;
        this->tileOw = tileOw;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N*C*H*W);
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N*C*OH*OW);

        pipe.InitBuffer(outQueue, 1, tileOw * sizeof(float));
        pipe.InitBuffer(tmpQueue, 1, 2 * sizeof(float)); // for 1-element tmp/sum
    }

    __aicore__ inline void Process() {
        const float invK2 = 1.0f / (float)(K * K);
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t c = 0; c < C; ++c) {
                for (uint32_t oh = 0; oh < OH; ++oh) {
                    // process by tiles along width
                    for (uint32_t owStart = 0; owStart < OW; owStart += tileOw) {
                        uint32_t curTile = AscendC::min(tileOw, OW - owStart);
                        AscendC::LocalTensor<float> outLocal = outQueue.AllocTensor<float>();

                        for (uint32_t j = 0; j < curTile; ++j) {
                            // sumLocal = 0
                            AscendC::LocalTensor<float> sumLocal = tmpQueue.AllocTensor<float>();
                            AscendC::Duplicate(sumLocal, 0.0f, 1);

                            uint32_t inBaseH = oh * S;
                            uint32_t inBaseW = (owStart + j) * S;

                            for (uint32_t kh = 0; kh < K; ++kh) {
                                uint32_t ih = inBaseH + kh;
                                if (ih >= H) { continue; }
                                for (uint32_t kw = 0; kw < K; ++kw) {
                                    uint32_t iw = inBaseW + kw;
                                    if (iw >= W) { continue; }

                                    AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();
                                    uint64_t inOff = InOffset(n, c, ih, iw);
                                    AscendC::DataCopy(tmpLocal, xGm[inOff], 1);
                                    AscendC::Add(sumLocal, sumLocal, tmpLocal, 1);
                                    tmpQueue.FreeTensor(tmpLocal);
                                }
                            }

                            AscendC::Muls(sumLocal, sumLocal, invK2);
                            AscendC::DataCopy(outLocal[j], sumLocal, 1);
                            tmpQueue.FreeTensor(sumLocal);
                        }

                        // write back
                        uint64_t outOff = OutOffset(n, c, oh, owStart);
                        AscendC::DataCopy(yGm[outOff], outLocal, curTile);
                        outQueue.FreeTensor(outLocal);
                    }
                }
            }
        }
    }

private:
    __aicore__ inline uint64_t InOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const {
        return (((uint64_t)n * C + c) * H + h) * W + w;
    }
    __aicore__ inline uint64_t OutOffset(uint32_t n, uint32_t c, uint32_t oh, uint32_t ow) const {
        return (((uint64_t)n * C + c) * OH + oh) * OW + ow;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t N, C, H, W;
    uint32_t K, S, P;
    uint32_t OH, OW;
    uint32_t tileOw;
};

extern "C" __global__ __aicore__ void average_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelAveragePool2D op;
    op.Init(x, y,
            tilingData.n(), tilingData.c(), tilingData.h(), tilingData.w(),
            tilingData.k(), tilingData.s(), tilingData.p(),
            tilingData.outH(), tilingData.outW(), tilingData.tileOw());
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0):
        super().__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding

    def forward(self, x: torch.Tensor) -> torch.Tensor:

        return custom_ops_lib.average_pooling_2d_custom(x, self.kernel_size, stride, self.padding)
