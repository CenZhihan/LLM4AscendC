#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/tensor.h"

// AscendC matmul tiling

#include "tiling/matmul_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // Input shapes:
    // x: [M, K], w: [N, K] (original Linear weight is [N, K], but B will be transposed in kernel)
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    // Derive M, N, K
    uint64_t M = x_shape->GetOriginShape().GetDim(0);
    uint64_t K = x_shape->GetOriginShape().GetDim(1);
    uint64_t N = w_shape->GetOriginShape().GetDim(0); // weight is [N, K]

    // Setup tiling for single-core Matmul (BlockDim=1)
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling tiling(ascendcPlatform);
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetShape((int32_t)M, (int32_t)N, (int32_t)K);
    tiling.SetOrgShape((int32_t)M, (int32_t)N, (int32_t)K);
    tiling.EnableBias(true);
    tiling.SetBufferSpace(-1, -1, -1); // use default available spaces

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Prepare raw tiling buffer: [TCubeTiling bytes][uint32 totalLength][uint32 tileNum]
    const uint32_t tileNum = 8; // number of tiles for post-Relu pass
    uint64_t totalLength = M * N; // elements of output

    uint8_t* rawBuf = reinterpret_cast<uint8_t*>(context->GetRawTilingData()->GetData());
    size_t capacity = context->GetRawTilingData()->GetCapacity();
    size_t needSize = sizeof(optiling::TCubeTiling) + sizeof(uint32_t) * 2;
    if (capacity < needSize) {
        return ge::GRAPH_FAILED;
    }

    // Copy TCubeTiling
    memcpy(rawBuf, &tilingData, sizeof(optiling::TCubeTiling));
    // Append totalLength and tileNum
    uint32_t* tail = reinterpret_cast<uint32_t*>(rawBuf + sizeof(optiling::TCubeTiling));
    tail[0] = static_cast<uint32_t>(totalLength);
    tail[1] = tileNum;

    context->GetRawTilingData()->SetDataSize(needSize);

    // Use single block for simplicity
    context->SetBlockDim(1);

    size_t* workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // y shape = [x.shape[0], w.shape[0]]
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;
    }
    gert::Shape* y_shape = context->GetOutputShape(0);

    std::vector<int64_t> out_dims;
    out_dims.push_back(x_shape->GetOriginShape().GetDim(0));
    out_dims.push_back(w_shape->GetOriginShape().GetDim(0));
    y_shape->SetOriginShape(out_dims);
    y_shape->SetShape(out_dims);

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // output dtype follows input x dtype
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = half;
using B_T = half;
using C_T = half;
using BiasT = half;

constexpr int32_t RELU_BUFFER_NUM = 2;

class KernelGemmAddRelu {
public:
    __aicore__ inline KernelGemmAddRelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR z, GM_ADDR tilingGM, GM_ADDR workspace)
    {
        // Load TCubeTiling from GM tiling buffer
        auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
        auto tempTiling = (uint32_t*)&tiling_;
        for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
            *tempTiling = *tempTilingGM;
        }
        // Read appended fields: totalLength and tileNum
        totalLength_ = *(tempTilingGM++);
        tileNum_ = *(tempTilingGM++);

        // Set GM buffers
        AscendC::GlobalTensor<A_T> aGlobal;
        AscendC::GlobalTensor<B_T> bGlobal;
        AscendC::GlobalTensor<C_T> cGlobal;
        AscendC::GlobalTensor<BiasT> biasGlobal;

        int32_t sizeA = tiling_.ALayoutInfoB * tiling_.singleCoreM * tiling_.singleCoreK * sizeof(A_T);
        int32_t sizeB = tiling_.BLayoutInfoB * tiling_.singleCoreK * tiling_.singleCoreN * sizeof(B_T);
        int32_t sizeC = tiling_.CLayoutInfoB * tiling_.singleCoreM * tiling_.singleCoreN * sizeof(C_T);
        int32_t sizeBias = tiling_.CLayoutInfoB * tiling_.singleCoreN * sizeof(BiasT);

        aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(x), sizeA);
        bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(w), sizeB);
        cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(z), sizeC);
        biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ BiasT*>(bias), sizeBias);

        gm_a_.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal.GetPhyAddr()), tiling_.singleCoreM * tiling_.singleCoreK);
        gm_b_.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal.GetPhyAddr()), tiling_.singleCoreK * tiling_.singleCoreN);
        gm_c_.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal.GetPhyAddr()), tiling_.singleCoreM * tiling_.singleCoreN);
        gm_bias_.SetGlobalBuffer(const_cast<__gm__ BiasT*>(biasGlobal.GetPhyAddr()), tiling_.singleCoreN);

        workspace_ = workspace;

        // Init vector buffers for post-Relu
        blockLength_ = totalLength_; // BlockDim=1
        tileLength_ = blockLength_ / tileNum_ / RELU_BUFFER_NUM;
        if (tileLength_ == 0) {
            tileLength_ = blockLength_;
            tileNum_ = 1;
        }
        pipe_.InitBuffer(inQueueC_, RELU_BUFFER_NUM, tileLength_ * sizeof(C_T));
        pipe_.InitBuffer(outQueueC_, RELU_BUFFER_NUM, tileLength_ * sizeof(C_T));
    }

    __aicore__ inline void ComputeGemmAdd()
    {
        // Matmul types for NORMAL layout, B will be transposed logically in SetTensorB
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

        constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
        matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm;
        AscendC::TPipe mmPipe;
        REGIST_MATMUL_OBJ(&mmPipe, GetSysWorkSpacePtr(), mm);

        tiling_.shareMode = 0;
        tiling_.shareL1Size = 512 * 1024;
        tiling_.shareL0CSize = 128 * 1024;
        tiling_.shareUbSize = 0;

        mm.Init(&tiling_);
        mm.SetTensorA(gm_a_, false);   // A not transposed
        mm.SetTensorB(gm_b_, true);    // Treat B as transposed so that W [N,K] becomes [K,N]
        mm.SetWorkspace(workspace_, 0);
        mm.SetBias(gm_bias_);
        mm.IterateBatch(gm_c_, 1, 1, false);
    }

    __aicore__ inline void ComputeRelu()
    {
        int32_t loopCount = tileNum_ * RELU_BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            ReluTile(i);
            CopyOut(i);
        }
        // Handle tail
        uint32_t handled = tileNum_ * RELU_BUFFER_NUM * tileLength_;
        if (handled < blockLength_) {
            uint32_t tail = blockLength_ - handled;
            AscendC::LocalTensor<C_T> cLocalIn = inQueueC_.AllocTensor<C_T>();
            AscendC::LocalTensor<C_T> cLocalOut = outQueueC_.AllocTensor<C_T>();
            AscendC::DataCopy(cLocalIn, gm_c_[handled], tail);
            AscendC::Relu(cLocalOut, cLocalIn, tail);
            AscendC::DataCopy(gm_c_[handled], cLocalOut, tail);
            inQueueC_.FreeTensor(cLocalIn);
            outQueueC_.FreeTensor(cLocalOut);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<C_T> cLocal = inQueueC_.AllocTensor<C_T>();
        AscendC::DataCopy(cLocal, gm_c_[progress * tileLength_], tileLength_);
        inQueueC_.EnQue(cLocal);
    }

    __aicore__ inline void ReluTile(int32_t /*progress*/)
    {
        AscendC::LocalTensor<C_T> cIn = inQueueC_.DeQue<C_T>();
        AscendC::LocalTensor<C_T> cOut = outQueueC_.AllocTensor<C_T>();
        AscendC::Relu(cOut, cIn, tileLength_);
        outQueueC_.EnQue<C_T>(cOut);
        inQueueC_.FreeTensor(cIn);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<C_T> cOut = outQueueC_.DeQue<C_T>();
        AscendC::DataCopy(gm_c_[progress * tileLength_], cOut, tileLength_);
        outQueueC_.FreeTensor(cOut);
    }

private:
    // Matmul tiling
    TCubeTiling tiling_;
    // GM tensors for GEMM
    AscendC::GlobalTensor<A_T> gm_a_;
    AscendC::GlobalTensor<B_T> gm_b_;
    AscendC::GlobalTensor<C_T> gm_c_;
    AscendC::GlobalTensor<BiasT> gm_bias_;
    GM_ADDR workspace_;

    // Post-Relu pipeline
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, RELU_BUFFER_NUM> inQueueC_;
    AscendC::TQue<AscendC::TPosition::VECOUT, RELU_BUFFER_NUM> outQueueC_;

    // Relu tiling info
    uint32_t totalLength_{0};
    uint32_t tileNum_{1};
    uint32_t blockLength_{0};
    uint32_t tileLength_{0};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelGemmAddRelu op;
    op.Init(x, w, bias, z, tiling, workspace);
    op.ComputeGemmAdd();
    op.ComputeRelu();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused GEMM + Bias + ReLU using custom AscendC kernel.
    Expect dtype float16 for best performance.
    """
    def __init__(self, in_features, out_features, bias_shape):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features, bias=False)
        # Bias is a separate parameter as in the original model
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x: torch.Tensor):
        # Cast to fp16 for kernel compatibility and performance
        x_fp16 = x.to(torch.float16)
        w_fp16 = self.gemm.weight.to(torch.float16)  # weight shape [out_features, in_features]
        b_fp16 = self.bias.to(torch.float16)         # bias shape [out_features]
        # Custom op expects (x, w, bias) and performs y = ReLU(x @ w.T + bias)
        return custom_ops_lib.gemm_add_relu_custom(x_fp16, w_fp16, b_fp16)
