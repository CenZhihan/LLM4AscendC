inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1) / b);
}
}

namespace optiling {

const uint32_t BLOCK_DIM = 16;
const uint32_t DEFAULT_TILE_W = 256; // 必须满足32字节对齐(256*4=1024B)

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ConvTransposed3dAsymmetricInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape* xShape = context->GetInputShape(0);
    const gert::Shape* wShape = context->GetInputShape(1);
    if (xShape->GetDimNum() != 5 || wShape->GetDimNum() != 5) {
        return ge::GRAPH_FAILED;
    }
    // x: [N, Cin, Di, Hi, Wi]
    uint32_t N  = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t Cin = static_cast<uint32_t>(xShape->GetDim(1));
    uint32_t Di  = static_cast<uint32_t>(xShape->GetDim(2));
    uint32_t Hi  = static_cast<uint32_t>(xShape->GetDim(3));
    uint32_t Wi  = static_cast<uint32_t>(xShape->GetDim(4));
    // w: [Cin, Cout, Kd, Kh, Kw] （按PyTorch ConvTranspose权重布局）
    uint32_t Kd = static_cast<uint32_t>(wShape->GetDim(2));
    uint32_t Kh = static_cast<uint32_t>(wShape->GetDim(3));
    uint32_t Kw = static_cast<uint32_t>(wShape->GetDim(4));
    uint32_t Cout = static_cast<uint32_t>(wShape->GetDim(1));

    // 仅支持 stride=(1,1,1), padding=(0,0,0), output_padding=(0,0,0), groups=1 的基线实现
    uint32_t Do = Di + Kd - 1;
    uint32_t Ho = Hi + Kh - 1;
    uint32_t Wo = Wi + Kw - 1;

    // 纵向分块: 一条line=(n, co, d, h) 对应连续的 Wo 个元素
    uint64_t totalLines64 = static_cast<uint64_t>(N) * Cout * Do * Ho;
    uint32_t linesPerBlock = CeilDiv(totalLines64, BLOCK_DIM);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_din(Di);
    tiling.set_hin(Hi);
    tiling.set_win(Wi);
    tiling.set_kd(Kd);
    tiling.set_kh(Kh);
    tiling.set_kw(Kw);
    tiling.set_dout(Do);
    tiling.set_hout(Ho);
    tiling.set_wout(Wo);
    tiling.set_tileW(DEFAULT_TILE_W);
    tiling.set_linesPerBlock(linesPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0; // 无额外workspace
    return ge::GRAPH_SUCCESS;
}

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3d {
public:
    __aicore__ inline KernelConvTransposed3d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t din, uint32_t hin, uint32_t win,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout,
                                uint32_t tileW, uint32_t linesPerBlock)
    {
        this->N = n; this->Cin = cin; this->Cout = cout;
        this->Di = din; this->Hi = hin; this->Wi = win;
        this->Kd = kd; this->Kh = kh; this->Kw = kw;
        this->Do = dout; this->Ho = hout; this->Wo = wout;
        this->tileW = tileW;
        this->linesPerBlock = linesPerBlock;

        uint32_t xSize = N * Cin * Di * Hi * Wi;
        uint32_t wSize = Cin * Cout * Kd * Kh * Kw;
        uint32_t ySize = N * Cout * Do * Ho * Wo;

        xGm.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileW * sizeof(float));
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileW * sizeof(float)); // 作为临时缓冲使用
    }

    __aicore__ inline void Process()
    {
        // 基线实现：仅分块清零输出，示例结构，便于替换为真实计算
        // 实际优化请在ComputeTile中完成转置卷积累加
        uint32_t totalLines = N * Cout * Do * Ho;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startLine = blockIdx * linesPerBlock;
        uint32_t endLine = AscendC::Min(startLine + linesPerBlock, totalLines);

        for (uint32_t line = startLine; line < endLine; ++line) {
            uint32_t tmp = line;
            uint32_t co = tmp % Cout; tmp /= Cout;
            uint32_t d  = tmp % Do;   tmp /= Do;
            uint32_t h  = tmp % Ho;   tmp /= Ho;
            uint32_t n  = tmp;

            for (uint32_t wOff = 0; wOff < Wo; wOff += tileW) {
                uint32_t curLen = AscendC::Min(tileW, Wo - wOff);
                ZeroAndStoreTile(n, co, d, h, wOff, curLen);
            }
        }
    }

private:
    __aicore__ inline void ZeroAndStoreTile(uint32_t n, uint32_t co, uint32_t d, uint32_t h, uint32_t wStart, uint32_t len)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
        // 将本地缓冲置零（注意：若未提供对应API，请在此处替换为等效清零方法）
        AscendC::Dup(zLocal, (float)0.0f, tileW);

        // 注：此处可调用ComputeTile(n, co, d, h, wStart, len, zLocal)完成真实计算

        uint64_t yBase = (((((uint64_t)n * Cout + co) * Do + d) * Ho + h) * Wo + wStart);
        AscendC::DataCopy(yGm[yBase], zLocal, len);
        outQueueZ.EnQue<float>(zLocal);
        AscendC::LocalTensor<float> zLocalDeque = outQueueZ.DeQue<float>(); // 取出后立即释放
        outQueueZ.FreeTensor(zLocalDeque);
    }

    // 示例：占位函数，真实实现应在此进行逐通道、逐核大小的累加
    __aicore__ inline void ComputeTile(uint32_t n, uint32_t co, uint32_t d, uint32_t h, uint32_t wStart, uint32_t len, AscendC::LocalTensor<float>& zLocal)
    {
        // 伪代码结构（请根据需求完善）:
        // for ci in [0, Cin):
        //   for kd in [0, Kd):
        //     int32_t di = (int32_t)d + (int32_t)kd - (int32_t)(Kd - 1);
        //     if (di < 0 || di >= (int32_t)Di) continue;
        //     for kh in [0, Kh):
        //       int32_t hi = (int32_t)h + (int32_t)kh - (int32_t)(Kh - 1);

        //         // AscendC::Axpy(zLocal[validLeft], xLocal, alpha, validLen);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inQueueX;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t Do, Ho, Wo;
    uint32_t tileW;
