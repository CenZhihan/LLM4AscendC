namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)

    PowerBroadcastCustomTilingData tiling;

    // x shape
    uint64_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    // exponent shape (broadcast along last dim)
    uint64_t cols = context->GetInputShape(1)->GetOriginShape().GetShapeSize();

        return ge::GRAPH_FAILED;
    }
    uint64_t rows = totalLength / cols;

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_rows(static_cast<uint32_t>(rows));
    tiling.set_cols(static_cast<uint32_t>(cols));
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_E = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2; // x/y 双缓冲

class KernelPowerBroadcast {
public:
    __aicore__ inline KernelPowerBroadcast() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR exp, GM_ADDR y,
                                uint32_t totalLength, uint32_t rows, uint32_t cols, uint32_t tileNum)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileNum = tileNum;

        // 根据block划分行
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlockBase = rows / blockNum;
        uint32_t rowsRemainder = rows % blockNum;
        this->startRow = rowsPerBlockBase * blockIdx + (blockIdx < rowsRemainder ? blockIdx : rowsRemainder);
        this->blockRows = rowsPerBlockBase + (blockIdx < rowsRemainder ? 1 : 0);

        // 设置全局指针（从startRow起始）
        uint32_t blockOffset = this->startRow * cols;
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + blockOffset, this->blockRows * cols);
        eGm.SetGlobalBuffer((__gm__ DTYPE_E*)exp, cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + blockOffset, this->blockRows * cols);

        // 列分块，按tileNum平均切分，队列大小为单次处理的最大段长
        uint32_t base = cols / tileNum;
        uint32_t rem = cols % tileNum;
        this->segMax = base + (rem > 0 ? 1 : 0);
        if (this->segMax == 0) this->segMax = cols;

        // 初始化队列缓冲
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->segMax * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->segMax * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueE, 1, this->segMax * sizeof(DTYPE_E));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t t = 0; t < tileNum; ++t) {
            uint32_t base = cols / tileNum;
            uint32_t rem = cols % tileNum;
            uint32_t curCols = base + (t < rem ? 1u : 0u);
            uint32_t colStart = t * base + (t < rem ? t : rem);
            if (curCols == 0) continue;

            // 加载当前列段的exponent到本地
            AscendC::LocalTensor<DTYPE_E> eLocalAlloc = inQueueE.AllocTensor<DTYPE_E>();
            AscendC::DataCopy(eLocalAlloc, eGm[colStart], curCols);
            inQueueE.EnQue(eLocalAlloc);
            AscendC::LocalTensor<DTYPE_E> eLocal = inQueueE.DeQue<DTYPE_E>();

            // 遍历本block负责的每一行
            for (uint32_t r = 0; r < blockRows; ++r) {
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

                uint32_t rowOffset = r * cols + colStart;
                AscendC::DataCopy(xLocal, xGm[rowOffset], curCols);

                // 使用AscendC Power张量-张量接口（由框架分配临时空间）
                AscendC::Power<DTYPE_X>(yLocal, xLocal, eLocal, curCols);

                AscendC::DataCopy(yGm[rowOffset], yLocal, curCols);

                inQueueX.FreeTensor(xLocal);
                outQueueY.FreeTensor(yLocal);
            }
            inQueueE.FreeTensor(eLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> inQueueE;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_E> eGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tileNum;

    uint32_t startRow;
    uint32_t blockRows;
    uint32_t segMax;
};

extern "C" __global__ __aicore__ void power_broadcast_custom(GM_ADDR x, GM_ADDR exponent, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelPowerBroadcast op;
    op.Init(x, exponent, y,
            tiling_data.totalLength(),
            tiling_data.rows(),
            tiling_data.cols(),
            tiling_data.tileNum());
    op.Process();
}
