kernel_src="""
#include "kernel_operator.h"

// 简化实现：标准1D卷积，支持 stride=1, padding=0, dilation=1, groups=1, 无bias
// 输入: x[N, C_in, L], w[C_out, C_in, K]
// 输出: y[N, C_out, L_out], 其中 L_out = L - K + 1
// 计算方式：在输出长度方向上做tile，每个tile长度为 tileOutLen

class KernelConv1D {
public:
    __aicore__ inline KernelConv1D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t length, uint32_t ksize,
                                uint32_t lout, uint32_t tileOutLen)
    {
        this->N = n;
        this->Cin = cin;
        this->Cout = cout;
        this->L = length;
        this->K = ksize;
        this->Lout = lout;
        this->tileOutLen = (tileOutLen == 0) ? lout : tileOutLen;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y*)w);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y);

        // 队列与本地缓存
        // x切片: tileOutLen
        pipe.InitBuffer(inQueueX, 2, this->tileOutLen * sizeof(DTYPE_X));
        // 权重向量(由标量duplicate得到): tileOutLen
        pipe.InitBuffer(inQueueWVec, 1, this->tileOutLen * sizeof(DTYPE_Y));
        // 权重标量读取: 1元素
        pipe.InitBuffer(inQueueWScalar, 1, sizeof(DTYPE_Y));
        // 输出tile: tileOutLen
        pipe.InitBuffer(outQueueY, 1, this->tileOutLen * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (Lout == 0 || K == 0) {
            return;
        }
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oc = 0; oc < Cout; ++oc) {
                for (uint32_t t0 = 0; t0 < Lout; t0 += tileOutLen) {
                    uint32_t curr = AscendC::Min(tileOutLen, Lout - t0);

                    // 分配输出tile并置零
                    AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
                    DTYPE_Z zeroVal = static_cast<DTYPE_Z>(0);
                    AscendC::Duplicate<DTYPE_Z>(yLocal, zeroVal, curr);

                    // 遍历输入通道与kernel
                    for (uint32_t ic = 0; ic < Cin; ++ic) {
                        // 每个k偏移
                        for (uint32_t kk = 0; kk < K; ++kk) {
                            // 读取输入x切片 x[n, ic, t0+kk : t0+kk+curr]
                            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                            uint64_t xBase = (static_cast<uint64_t>(n) * Cin + ic) * L;
                            uint64_t xOffset = xBase + t0 + kk;
                            AscendC::DataCopy(xLocal, xGm[xOffset], curr);

                            // 读取权重标量 w[oc, ic, kk]
                            AscendC::LocalTensor<DTYPE_Y> wScalarLocal = inQueueWScalar.AllocTensor<DTYPE_Y>();
                            uint64_t wOffset = (static_cast<uint64_t>(oc) * Cin + ic) * K + kk;
                            AscendC::DataCopy(wScalarLocal, wGm[wOffset], 1);
                            DTYPE_Y wScalar = wScalarLocal.GetValue(0);
                            inQueueWScalar.FreeTensor(wScalarLocal);

                            // 将标量扩展为向量
                            AscendC::LocalTensor<DTYPE_Y> wVecLocal = inQueueWVec.AllocTensor<DTYPE_Y>();
                            AscendC::Duplicate<DTYPE_Y>(wVecLocal, wScalar, curr);

                            // 累加: y += x * w
                            AscendC::MulAddDst(yLocal, xLocal, wVecLocal, curr);

                            inQueueX.FreeTensor(xLocal);
                            inQueueWVec.FreeTensor(wVecLocal);
                        }
                    }

                    // 写回输出 y[n, oc, t0 : t0+curr]
                    uint64_t yBase = (static_cast<uint64_t>(n) * Cout + oc) * Lout;
                    AscendC::DataCopy(yGm[yBase + t0], yLocal, curr);
                    outQueueY.FreeTensor(yLocal);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 2> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueWVec;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueWScalar;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N;
    uint32_t Cin;
    uint32_t Cout;
    uint32_t L;
    uint32_t K;
    uint32_t Lout;
    uint32_t tileOutLen;
};

extern "C" __global__ __aicore__ void conv_standard_1d_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv1D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.length, tiling_data.ksize,
            tiling_data.lout, tiling_data.tileOutLen);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC核实现标准1D卷积。
    约束：
    - stride=1, padding=0, dilation=1
    - groups=1
    - 无bias
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super().__init__()
        assert stride == 1 and padding == 0 and dilation == 1, "Only stride=1, padding=0, dilation=1 are supported"
        assert groups == 1, "Only groups=1 is supported"
        assert bias is False, "bias=False is supported"

        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size

        # 权重参数 [C_out, C_in, K]
        w = torch.empty(out_channels, in_channels, kernel_size)
        torch.nn.init.kaiming_uniform_(w, a=math.sqrt(5))
        self.weight = torch.nn.Parameter(w)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_standard_1d_custom(x, self.weight)
