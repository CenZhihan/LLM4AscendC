kernel_src="""
#include "kernel_operator.h"

// 仅示例：按axis=1执行argmin（与给定模型一致），通过TopK(k=1, isLargest=false)获得最小值下标
// 输入：x(float32, 3D)，dim(int32标量，未在kernel中使用，按tiling固定axis=1)
// 输出：idx(int32, 2D [d0, d2])

class KernelArgminAxis1 {
public:
    __aicore__ inline KernelArgminAxis1() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR dim, GM_ADDR idx,
                                int32_t d0, int32_t d1, int32_t d2,
                                uint32_t k, bool isLargest, bool isInitIndex,
                                uint32_t tmpsize_axis1, const AscendC::TopkTiling &topkTilingAxis1)
    {
        // 全局张量绑定
        xGm.SetGlobalBuffer((__gm__ float*)x);
        // dim输入在本示例中未在kernel中使用，按tiling固定axis=1
        dimGm.SetGlobalBuffer((__gm__ int32_t*)dim);
        idxGm.SetGlobalBuffer((__gm__ int32_t*)idx);

        this->D0 = d0;
        this->D1 = d1;
        this->D2 = d2;

        this->k = k; // =1
        this->isLargest = isLargest;     // false
        this->isInitIndex = isInitIndex; // false

        // TopK相关
        this->topkTilingAxis1 = topkTilingAxis1;
        this->tmpSizeAxis1 = tmpsize_axis1;

        // 每行长度(inner)与步长(step)（axis=1）
        this->inner = D1;
        this->step = D2;

        // 每个block处理的行数（outer = d0*d2）
        this->outer = static_cast<uint32_t>(D0) * static_cast<uint32_t>(D2);
        this->rowsPerBlock = (outer + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        this->startRow = AscendC::GetBlockIdx() * rowsPerBlock;
        this->endRow = AscendC::Min(startRow + rowsPerBlock, outer);

        // UB缓冲：一行数据+TopK临时空间+结果缓冲
        maxRowSizeBytes = static_cast<uint32_t>(inner) * sizeof(float);
        pipe.InitBuffer(rowBufQ, 1, maxRowSizeBytes);
        pipe.InitBuffer(tmpBufQ, 1, tmpSizeAxis1);         // TopK tmp
        pipe.InitBuffer(idxBufQ, 1, k * sizeof(int32_t));  // TopK输出idx（k=1）
        pipe.InitBuffer(valBufQ, 1, k * sizeof(float));    // TopK输出val（k=1），仅占位
        pipe.InitBuffer(finBufQ, 1, sizeof(bool));         // finishLocal（未使用）
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = startRow; r < endRow; ++r) {
            // 将全局的第r行（对应[b, j]，b=j//D2, j=j%D2）按axis=1收集到UB连续缓冲
            // base = b*D1*D2 + 0*D2 + j_in_D2
            uint32_t b = r / static_cast<uint32_t>(D2);
            uint32_t j2 = r % static_cast<uint32_t>(D2);
            uint64_t base = static_cast<uint64_t>(b) * static_cast<uint64_t>(D1) * static_cast<uint64_t>(D2)
                            + static_cast<uint64_t>(j2);

            // 分配本地缓冲
            AscendC::LocalTensor<float> rowLocal = rowBufQ.AllocTensor<float>();
            // 将该行的inner元素逐个拷入UB（步长为step=D2）
            for (int32_t t = 0; t < D1; ++t) {
                // 目标位置：rowLocal[t]，源：xGm[base + t*step]
                AscendC::DataCopy(rowLocal[t], xGm[base + static_cast<uint64_t>(t) * static_cast<uint64_t>(step)], 1);
            }

            // TopK(k=1, isLargest=false) -> argmin
            AscendC::LocalTensor<uint8_t> tmpLocal = tmpBufQ.AllocTensor<uint8_t>();
            AscendC::LocalTensor<float> dstValLocal = valBufQ.AllocTensor<float>();
            AscendC::LocalTensor<int32_t> dstIndexLocal = idxBufQ.AllocTensor<int32_t>();
            AscendC::LocalTensor<int32_t> srcIndexStub = idxBufQ.AllocTensor<int32_t>(); // 占位，不使用
            AscendC::LocalTensor<bool> finishStub = finBufQ.AllocTensor<bool>();         // 占位，不使用

            // 使用TopK NORMAL模式（tiling已按inner和outter=1生成）
            AscendC::TopK<float, false, false, false, AscendC::TopKMode::TOPK_NORMAL>(
                dstValLocal, dstIndexLocal, rowLocal, srcIndexStub, finishStub, tmpLocal,
                static_cast<int32_t>(k), topkTilingAxis1, GetTopKInfo(), /*isLargest=*/false);

            // 将idx写回全局（每行输出1个int32）
            AscendC::DataCopy(idxGm[r], dstIndexLocal, k);

            // 释放本地缓冲
            idxBufQ.FreeTensor(dstIndexLocal);
            valBufQ.FreeTensor(dstValLocal);
            tmpBufQ.FreeTensor(tmpLocal);
            finBufQ.FreeTensor(finishStub);
            idxBufQ.FreeTensor(srcIndexStub);
            rowBufQ.FreeTensor(rowLocal);
        }
    }

private:
    // TopKInfo 占位（默认构造）
    __aicore__ inline AscendC::TopKInfo GetTopKInfo() {
        AscendC::TopKInfo info;
        return info;
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::TPosition::VECIN, 1> rowBufQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> tmpBufQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> idxBufQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> valBufQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> finBufQ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<int32_t> dimGm; // 未使用
    AscendC::GlobalTensor<int32_t> idxGm;

    int32_t D0, D1, D2;
    uint32_t inner;
    uint32_t step;
    uint32_t outer;
    uint32_t rowsPerBlock, startRow, endRow;
    uint32_t k;
    bool isLargest;
    bool isInitIndex;

    uint32_t maxRowSizeBytes;

    AscendC::TopkTiling topkTilingAxis1;
    uint32_t tmpSizeAxis1;
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR dim, GM_ADDR idx, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelArgminAxis1 op;
    op.Init(x, dim, idx,
            tilingData.d0, tilingData.d1, tilingData.d2,
            tilingData.k, tilingData.islargest, tilingData.isinitindex,
            tilingData.tmpsize_axis1, tilingData.topkTilingAxis1);
    op.Process();
}
