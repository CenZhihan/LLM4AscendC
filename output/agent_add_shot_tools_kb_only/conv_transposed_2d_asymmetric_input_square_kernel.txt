kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelDeconv2D {
public:
    __aicore__ inline KernelDeconv2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t hin, uint32_t win, uint32_t k, uint32_t tileOc)
    {
        this->n = n;
        this->cin = cin;
        this->cout = cout;
        this->hin = hin;
        this->win = win;
        this->k = k;
        this->hout = hin + k - 1;
        this->wout = win + k - 1;

        // Block partition by output channel
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();
        this->ocStart = bid * tileOc;
        this->ocEnd = this->ocStart + tileOc;
        if (this->ocEnd > cout) this->ocEnd = cout;

        // Set global buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, n * cin * hin * win);
        wGm.SetGlobalBuffer((__gm__ DTYPE*)w, cin * cout * k * k);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, n * cout * hout * wout);

        // Queues: rows and temps
        // xRow: W, yRow: Wout, tmpRow: W
        uint32_t xRowSize = win;
        uint32_t yRowSize = wout;
        pipe.InitBuffer(xRowQ, 1, xRowSize * sizeof(DTYPE));
        pipe.InitBuffer(yRowQ, 1, yRowSize * sizeof(DTYPE));
        pipe.InitBuffer(tmpRowQ, 1, xRowSize * sizeof(DTYPE));
        // weight tile buffer (K*K)
        pipe.InitBuffer(wTileQ, 1, k * k * sizeof(DTYPE));
        // zero row buffer
        pipe.InitBuffer(zeroRowQ, 1, yRowSize * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t b = 0; b < n; ++b) {
            // For each output channel owned by this block, zero initialize y
            for (uint32_t oc = ocStart; oc < ocEnd; ++oc) {
                ZeroOutY(b, oc);
            }
            for (uint32_t ic = 0; ic < cin; ++ic) {
                // Load all K*K weights for this ic across oc range (we reuse per oc)
                for (uint32_t oc = ocStart; oc < ocEnd; ++oc) {
                    LoadWeightTile(ic, oc);
                    // Accumulate contributions: for each kh, kw
                    for (uint32_t kh = 0; kh < k; ++kh) {
                        for (uint32_t r = 0; r < hin; ++r) {
                            // Load one input row x[b, ic, r, :]
                            AscendC::LocalTensor<DTYPE> xRow = xRowQ.AllocTensor<DTYPE>();
                            uint64_t xOff = ((uint64_t)b * cin + ic) * ((uint64_t)hin * win) + (uint64_t)r * win;
                            AscendC::DataCopy(xRow, xGm[xOff], win);
                            xRowQ.EnQue(xRow);

                            for (uint32_t kw = 0; kw < k; ++kw) {
                                // Compute tmpRow = xRow * wVal
                                AscendC::LocalTensor<DTYPE> tmpRow = tmpRowQ.AllocTensor<DTYPE>();
                                AscendC::LocalTensor<DTYPE> xRowUse = xRowQ.DeQue<DTYPE>();
                                // Fetch weight scalar from local tile
                                // Assume contiguous buffer; index = kh * k + kw
                                DTYPE wVal = wTile[kh * k + kw];
                                AscendC::Muls(tmpRow, xRowUse, wVal, win);
                                tmpRowQ.EnQue(tmpRow);
                                xRowQ.EnQue(xRowUse);

                                // Load y segment row and accumulate
                                uint32_t yr = r + kh;
                                uint64_t yOff = ((uint64_t)b * cout + oc) * ((uint64_t)hout * wout) + (uint64_t)yr * wout + kw;
                                AscendC::LocalTensor<DTYPE> yRow = yRowQ.AllocTensor<DTYPE>();
                                AscendC::DataCopy(yRow, yGm[yOff], win);
                                // yRow += tmpRow
                                AscendC::LocalTensor<DTYPE> tmpUse = tmpRowQ.DeQue<DTYPE>();
                                AscendC::Add(yRow, yRow, tmpUse, win);
                                tmpRowQ.FreeTensor(tmpUse);
                                // store back
                                AscendC::DataCopy(yGm[yOff], yRow, win);
                                yRowQ.FreeTensor(yRow);
                            }
                            // Free xRow
                            AscendC::LocalTensor<DTYPE> xFree = xRowQ.DeQue<DTYPE>();
                            xRowQ.FreeTensor(xFree);
                        }
                    }
                }
            }
        }
    }

private:
    __aicore__ inline void LoadWeightTile(uint32_t ic, uint32_t oc)
    {
        // Load K*K weights for (ic, oc)
        AscendC::LocalTensor<DTYPE> wLocal = wTileQ.AllocTensor<DTYPE>();
        uint64_t wOff = (((uint64_t)ic * cout) + oc) * ((uint64_t)k * k);
        AscendC::DataCopy(wLocal, wGm[wOff], k * k);
        // Cache to a simple array for quick access in Process
        for (uint32_t i = 0; i < k * k; ++i) {
            wTile[i] = wLocal[i];
        }
        wTileQ.FreeTensor(wLocal);
    }

    __aicore__ inline void ZeroOutY(uint32_t b, uint32_t oc)
    {
        AscendC::LocalTensor<DTYPE> zeroRow = zeroRowQ.AllocTensor<DTYPE>();
        AscendC::Duplicate(zeroRow, (DTYPE)0, wout);
        for (uint32_t r = 0; r < hout; ++r) {
            uint64_t yOff = ((uint64_t)b * cout + oc) * ((uint64_t)hout * wout) + (uint64_t)r * wout;
            AscendC::DataCopy(yGm[yOff], zeroRow, wout);
        }
        zeroRowQ.FreeTensor(zeroRow);
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::QuePosition::VECIN, 1> xRowQ;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> yRowQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpRowQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> wTileQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> zeroRowQ;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t n, cin, cout, hin, win, k, hout, wout;
    uint32_t ocStart, ocEnd;

    // Simple local cache for one (ic, oc) weight tile
    DTYPE wTile[225]; // support up to 15x15 if needed
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelDeconv2D op;
    op.Init(x, w, y,
            tiling_data.n(),
            tiling_data.cin(),
            tiling_data.cout(),
            tiling_data.hin(),
            tiling_data.win(),
            tiling_data.k(),
            tiling_data.tileOc());
    op.Process();
}
