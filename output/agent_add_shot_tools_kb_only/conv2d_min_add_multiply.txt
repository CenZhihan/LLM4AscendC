kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_BIAS = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2; // 每个队列的双缓冲

class KernelConv2dMinAddMultiply {
public:
    __aicore__ inline KernelConv2dMinAddMultiply() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                float constVal, float scaleVal)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->constVal = constVal;
        this->scaleVal = scaleVal;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_BIAS *)bias + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_BIAS));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueueConst, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueueScale, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // 循环按tile处理，处理不足tileLength的尾块
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            uint32_t offset = i * this->tileLength;
            if (offset >= this->blockLength) {
                break;
            }
            uint32_t curLen = this->tileLength;
            if (offset + curLen > this->blockLength) {
                curLen = this->blockLength - offset;
            }
            CopyIn(i, offset, curLen);
            Compute(i, curLen);
            CopyOut(i, offset, curLen);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress, uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_BIAS> bLocal = inQueueB.AllocTensor<DTYPE_BIAS>();
        AscendC::DataCopy(xLocal, xGm[offset], len);
        AscendC::DataCopy(bLocal, biasGm[offset], len);
        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);

        // 准备常量与缩放因子向量
        AscendC::LocalTensor<float> constLocal = tmpQueueConst.AllocTensor<float>();
        AscendC::LocalTensor<float> scaleLocal = tmpQueueScale.AllocTensor<float>();
        AscendC::Duplicate(constLocal, this->constVal, len);
        AscendC::Duplicate(scaleLocal, this->scaleVal, len);
        tmpQueueConst.EnQue(constLocal);
        tmpQueueScale.EnQue(scaleLocal);
    }

    __aicore__ inline void Compute(int32_t progress, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_BIAS> bLocal = inQueueB.DeQue<DTYPE_BIAS>();
        AscendC::LocalTensor<float> constLocal = tmpQueueConst.DeQue<float>();
        AscendC::LocalTensor<float> scaleLocal = tmpQueueScale.DeQue<float>();

        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // y = min(x, const) + bias
        AscendC::Min(yLocal, xLocal, constLocal, len);
        AscendC::Add(yLocal, yLocal, bLocal, len);
        // y = y * scale
        AscendC::Mul(yLocal, yLocal, scaleLocal, len);

        outQueueY.EnQue<DTYPE_Y>(yLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
        tmpQueueConst.FreeTensor(constLocal);
        tmpQueueScale.FreeTensor(scaleLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress, uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[offset], yLocal, len);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueConst, tmpQueueScale;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_BIAS> biasGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    float constVal;
    float scaleVal;
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinAddMultiply op;
    op.Init(x, bias, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.constVal, tiling_data.scaleVal);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    替换了卷积后的min+add+broadcast+mul为AscendC自定义核
    """
    def __init__(self, in_channels, out_channels, kernel_size, constant_value, bias_shape, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.constant_value = float(constant_value)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.scaling_factor = float(scaling_factor)

    def forward(self, x):
        x = self.conv(x)
        # 将bias广播至与x相同形状，并保持连续以便核函数高效访问
        b = self.bias.view(1, self.bias.shape[0], 1, 1).expand_as(x).contiguous()
        return custom_ops_lib.conv2d_min_add_multiply_custom(x, b, self.constant_value, self.scaling_factor)
