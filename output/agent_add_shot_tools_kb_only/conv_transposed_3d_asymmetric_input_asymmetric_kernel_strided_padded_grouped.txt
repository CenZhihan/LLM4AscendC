kernel_src="""
#include "kernel_operator.h"

using data_t = float;

namespace CustomConvT3D {

struct TilingPack {
  uint32_t n, cin, cout;
  uint32_t din, hin, win;
  uint32_t kd, kh, kw;
  uint32_t sd, sh, sw;
  uint32_t pd, ph, pw;
  uint32_t opd, oph, opw;
  uint32_t groups;
  uint32_t hasBias;
  uint64_t xSize;
  uint64_t wSize;
  uint64_t bSize;
  uint64_t ySize;
};

class KernelConvTransposed3d {
public:
    __aicore__ inline KernelConvTransposed3d() {}

    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR wGm, GM_ADDR bGm, GM_ADDR yGm, const TilingPack &tp)
    {
        tiling = tp;

        xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ data_t*>(xGm), tiling.xSize);
        wGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ data_t*>(wGm), tiling.wSize);
        if (tiling.hasBias) {
            bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ data_t*>(bGm), tiling.bSize);
        }
        yGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ data_t*>(yGm), tiling.ySize);

        // 简化实现：一次性搬运全部数据
        pipe.InitBuffer(inQueueX, 1, tiling.xSize * sizeof(data_t));
        pipe.InitBuffer(inQueueW, 1, tiling.wSize * sizeof(data_t));
        if (tiling.hasBias) {
            pipe.InitBuffer(inQueueB, 1, tiling.bSize * sizeof(data_t));
        }
        pipe.InitBuffer(outQueueY, 1, tiling.ySize * sizeof(data_t));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        xLocal = inQueueX.AllocTensor<data_t>();
        wLocal = inQueueW.AllocTensor<data_t>();
        AscendC::DataCopy(xLocal, xGlobal, tiling.xSize);
        AscendC::DataCopy(wLocal, wGlobal, tiling.wSize);
        if (tiling.hasBias) {
            bLocal = inQueueB.AllocTensor<data_t>();
            AscendC::DataCopy(bLocal, bGlobal, tiling.bSize);
            inQueueB.EnQue(bLocal);
        }
        inQueueX.EnQue(xLocal);
        inQueueW.EnQue(wLocal);

        yLocal = outQueueY.AllocTensor<data_t>();
        // 将输出初始化为0
        // 由于没有专门的填充API，这里采用逐块清零（简单实现）
        // 假定LocalTensor支持线性访问，每次用Add命令清零也可行：y = y + 0
        // 为演示简化，使用循环写0（伪代码式）
        for (uint64_t i = 0; i < tiling.ySize; ++i) {
            yLocal.SetValue(i, (data_t)0);
        }
    }

    __aicore__ inline void Compute()
    {
        // 反卷积计算（NCHW-DHW）
        const uint32_t N = tiling.n;
        const uint32_t Cin = tiling.cin;
        const uint32_t Cout = tiling.cout;
        const uint32_t Din = tiling.din, Hin = tiling.hin, Win = tiling.win;
        const uint32_t Kd = tiling.kd, Kh = tiling.kh, Kw = tiling.kw;
        const uint32_t Sd = tiling.sd, Sh = tiling.sh, Sw = tiling.sw;
        const uint32_t Pd = tiling.pd, Ph = tiling.ph, Pw = tiling.pw;
        const uint32_t OPd = tiling.opd, OPh = tiling.oph, OPw = tiling.opw;
        const uint32_t G = tiling.groups;

        const uint32_t CoutPerG = Cout / G;
        const uint32_t CinPerG = Cin / G;

        const uint32_t Dout = (Din - 1) * Sd - 2 * Pd + Kd + OPd;
        const uint32_t Hout = (Hin - 1) * Sh - 2 * Ph + Kh + OPh;
        const uint32_t Wout = (Win - 1) * Sw - 2 * Pw + Kw + OPw;

        // 辅助lambda完成线性索引
        auto x_idx = [Cin, Din, Hin, Win](uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w)->uint64_t {
            return (((uint64_t)n * Cin + c) * Din + d) * Hin * Win + ((uint64_t)h * Win + w);
        };
        // 权重： [Cin, Cout/G, Kd,Kh,Kw]
        auto w_idx = [CoutPerG, Kd, Kh, Kw](uint32_t ci, uint32_t co_g, uint32_t kd, uint32_t kh, uint32_t kw)->uint64_t {
            return ((((uint64_t)ci) * CoutPerG + co_g) * Kd + kd) * Kh * Kw + ((uint64_t)kh * Kw + kw);
        };
        auto y_idx = [Cout, Dout, Hout, Wout](uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w)->uint64_t {
            return (((uint64_t)n * Cout + c) * Dout + d) * Hout * Wout + ((uint64_t)h * Wout + w);
        };

        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t co = 0; co < Cout; ++co) {
                uint32_t g = co / CoutPerG;
                uint32_t co_g = co % CoutPerG;
                for (uint32_t od = 0; od < Dout; ++od) {
                    // 反算输入d索引
                    for (uint32_t oh = 0; oh < Hout; ++oh) {
                        for (uint32_t ow = 0; ow < Wout; ++ow) {
                            data_t acc = (data_t)0;
                            // 枚举核
                            for (uint32_t kd = 0; kd < Kd; ++kd) {
                                int64_t id_numer = (int64_t)od + (int64_t)Pd - (int64_t)kd;
                                if (id_numer % (int64_t)Sd != 0) continue;
                                int64_t id = id_numer / (int64_t)Sd;
                                if (id < 0 || id >= (int64_t)Din) continue;
                                for (uint32_t kh = 0; kh < Kh; ++kh) {
                                    int64_t ih_numer = (int64_t)oh + (int64_t)Ph - (int64_t)kh;
                                    if (ih_numer % (int64_t)Sh != 0) continue;
                                    int64_t ih = ih_numer / (int64_t)Sh;
                                    if (ih < 0 || ih >= (int64_t)Hin) continue;
                                    for (uint32_t kw = 0; kw < Kw; ++kw) {
                                        int64_t iw_numer = (int64_t)ow + (int64_t)Pw - (int64_t)kw;
                                        if (iw_numer % (int64_t)Sw != 0) continue;
                                        int64_t iw = iw_numer / (int64_t)Sw;
                                        if (iw < 0 || iw >= (int64_t)Win) continue;

                                        // 枚举输入通道（组内）
                                        for (uint32_t ci_g = 0; ci_g < CinPerG; ++ci_g) {
                                            uint32_t ci = g * CinPerG + ci_g;
                                            uint64_t xi = x_idx(n, ci, (uint32_t)id, (uint32_t)ih, (uint32_t)iw);
                                            uint64_t wi = w_idx(ci, co_g, kd, kh, kw);
                                            data_t xv = xLocal.GetValue(xi);
                                            data_t wv = wLocal.GetValue(wi);
                                            acc = acc + xv * wv;
                                        }
                                    }
                                }
                            }
                            if (tiling.hasBias) {
                                acc = acc + bLocal.GetValue(co);
                            }
                            yLocal.SetValue(y_idx(n, co, od, oh, ow), acc);
                        }
                    }
                }
            }
        }
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::DataCopy(yGlobal, yLocal, tiling.ySize);
        outQueueY.EnQue(yLocal);
        // 释放in队列
        inQueueX.FreeTensor(xLocal);
        inQueueW.FreeTensor(wLocal);
        if (tiling.hasBias) {
            inQueueB.FreeTensor(bLocal);
        }
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueB;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<data_t> xGlobal;
    AscendC::GlobalTensor<data_t> wGlobal;
    AscendC::GlobalTensor<data_t> bGlobal;
    AscendC::GlobalTensor<data_t> yGlobal;

    AscendC::LocalTensor<data_t> xLocal;
    AscendC::LocalTensor<data_t> wLocal;
    AscendC::LocalTensor<data_t> bLocal;
    AscendC::LocalTensor<data_t> yLocal;

    TilingPack tiling;
};

} // namespace CustomConvT3D
