kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE;

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmScaleBatchnorm {
public:
    __aicore__ inline KernelGemmScaleBatchnorm() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR gamma, GM_ADDR beta, GM_ADDR eps,
                                GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(tiling_data, tiling);
        this->B = tiling_data.bLength;
        this->N = tiling_data.nLength;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, this->B * this->N);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE*)scale, this->N);
        gammaGm.SetGlobalBuffer((__gm__ DTYPE*)gamma, this->N);
        betaGm.SetGlobalBuffer((__gm__ DTYPE*)beta, this->N);
        epsGm.SetGlobalBuffer((__gm__ DTYPE*)eps, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, this->B * this->N);

        // workspace用于保存中间向量：mean[N], varSum[N], invStd[N]
        workMeanGm.SetGlobalBuffer((__gm__ DTYPE*)workspace, this->N);
        workVarSumGm.SetGlobalBuffer((__gm__ DTYPE*)workspace + this->N, this->N);
        workInvStdGm.SetGlobalBuffer((__gm__ DTYPE*)workspace + 2 * this->N, this->N);

        // 初始化队列
        pipe.InitBuffer(inRowQueue, BUFFER_NUM, this->N * sizeof(DTYPE));
        pipe.InitBuffer(paramQueue, 3, this->N * sizeof(DTYPE));   // scale, gamma, beta
        pipe.InitBuffer(outRowQueue, BUFFER_NUM, this->N * sizeof(DTYPE));
        pipe.InitBuffer(calcQueue, 3, this->N * sizeof(DTYPE));    // mean, varSum, invStd
    }

    __aicore__ inline void Process()
    {
        // 载入参数向量
        AscendC::LocalTensor<DTYPE> scaleLocal = paramQueue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> gammaLocal = paramQueue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> betaLocal  = paramQueue.AllocTensor<DTYPE>();
        AscendC::DataCopy(scaleLocal, scaleGm, this->N);
        AscendC::DataCopy(gammaLocal, gammaGm, this->N);
        AscendC::DataCopy(betaLocal,  betaGm,  this->N);
        paramQueue.EnQue(scaleLocal);
        paramQueue.EnQue(gammaLocal);
        paramQueue.EnQue(betaLocal);

        DTYPE epsVal;
        AscendC::LocalTensor<DTYPE> epsLocal = calcQueue.AllocTensor<DTYPE>();
        AscendC::DataCopy(epsLocal, epsGm, 1);
        epsVal = epsLocal.GetValue(0); // 读取标量
        calcQueue.FreeTensor(epsLocal);

        // 第一遍：计算mean
        AscendC::LocalTensor<DTYPE> meanLocal = calcQueue.AllocTensor<DTYPE>();
        // meanLocal = x_scaled(row0)
        {
            AscendC::LocalTensor<DTYPE> rowLocal = inRowQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(rowLocal, xGm[0], this->N);
            AscendC::LocalTensor<DTYPE> scaleT = paramQueue.DeQue<DTYPE>(); // 获取scale
            AscendC::Mul(meanLocal, rowLocal, scaleT, this->N);
            paramQueue.EnQue(scaleT);
            inRowQueue.FreeTensor(rowLocal);
        }
        for (uint32_t b = 1; b < this->B; ++b) {
            AscendC::LocalTensor<DTYPE> rowLocal = inRowQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> scaledLocal = outRowQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(rowLocal, xGm[b * this->N], this->N);
            AscendC::LocalTensor<DTYPE> scaleT = paramQueue.DeQue<DTYPE>();
            AscendC::Mul(scaledLocal, rowLocal, scaleT, this->N);
            AscendC::Add(meanLocal, meanLocal, scaledLocal, this->N);
            paramQueue.EnQue(scaleT);
            inRowQueue.FreeTensor(rowLocal);
            outRowQueue.FreeTensor(scaledLocal);
        }
        // meanLocal = meanLocal / B
        {
            AscendC::LocalTensor<DTYPE> denomLocal = calcQueue.AllocTensor<DTYPE>();
            // 使用betaLocal作为临时向量拷贝常数1/B
            for (uint32_t i = 0; i < this->N; ++i) {
                denomLocal.SetValue(i, (DTYPE)(1.0f / (DTYPE)this->B));
            }
            AscendC::Mul(meanLocal, meanLocal, denomLocal, this->N);
            calcQueue.FreeTensor(denomLocal);
        }

        // 第二遍：计算方差累积 varSum = sum((x_scaled - mean)^2)
        AscendC::LocalTensor<DTYPE> varSumLocal = calcQueue.AllocTensor<DTYPE>();
        // 初始化为 (row0*scale - mean)^2
        {
            AscendC::LocalTensor<DTYPE> rowLocal = inRowQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> scaledLocal = outRowQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> diffLocal = outRowQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(rowLocal, xGm[0], this->N);
            AscendC::LocalTensor<DTYPE> scaleT = paramQueue.DeQue<DTYPE>();
            AscendC::Mul(scaledLocal, rowLocal, scaleT, this->N);
            AscendC::Sub(diffLocal, scaledLocal, meanLocal, this->N);
            AscendC::Mul(varSumLocal, diffLocal, diffLocal, this->N);
            paramQueue.EnQue(scaleT);
            inRowQueue.FreeTensor(rowLocal);
            outRowQueue.FreeTensor(scaledLocal);
            outRowQueue.FreeTensor(diffLocal);
        }
        for (uint32_t b = 1; b < this->B; ++b) {
            AscendC::LocalTensor<DTYPE> rowLocal = inRowQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> scaledLocal = outRowQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> diffLocal = outRowQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> sqLocal = outRowQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(rowLocal, xGm[b * this->N], this->N);
            AscendC::LocalTensor<DTYPE> scaleT = paramQueue.DeQue<DTYPE>();
            AscendC::Mul(scaledLocal, rowLocal, scaleT, this->N);
            AscendC::Sub(diffLocal, scaledLocal, meanLocal, this->N);
            AscendC::Mul(sqLocal, diffLocal, diffLocal, this->N);
            AscendC::Add(varSumLocal, varSumLocal, sqLocal, this->N);
            paramQueue.EnQue(scaleT);
            inRowQueue.FreeTensor(rowLocal);
            outRowQueue.FreeTensor(scaledLocal);
            outRowQueue.FreeTensor(diffLocal);
            outRowQueue.FreeTensor(sqLocal);
        }
        // 计算 invStd = 1.0 / sqrt(var + eps)
        AscendC::LocalTensor<DTYPE> invStdLocal = calcQueue.AllocTensor<DTYPE>();
        {
            AscendC::LocalTensor<DTYPE> varLocal = calcQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> denomLocal = calcQueue.AllocTensor<DTYPE>();
            for (uint32_t i = 0; i < this->N; ++i) {
                denomLocal.SetValue(i, (DTYPE)(1.0f / (DTYPE)this->B));
            }
            AscendC::Mul(varLocal, varSumLocal, denomLocal, this->N); // var = varSum / B

            // var = var + eps
            AscendC::LocalTensor<DTYPE> epsVec = calcQueue.AllocTensor<DTYPE>();
            for (uint32_t i = 0; i < this->N; ++i) {
                epsVec.SetValue(i, epsVal);
            }
            AscendC::Add(varLocal, varLocal, epsVec, this->N);

            // invStd = 1 / sqrt(var)
            AscendC::LocalTensor<DTYPE> sqrtVarLocal = calcQueue.AllocTensor<DTYPE>();
            AscendC::Sqrt(sqrtVarLocal, varLocal, this->N);
            // 构造常数1向量
            AscendC::LocalTensor<DTYPE> oneVec = calcQueue.AllocTensor<DTYPE>();
            for (uint32_t i = 0; i < this->N; ++i) {
                oneVec.SetValue(i, (DTYPE)1.0f);
            }
            AscendC::Div(invStdLocal, oneVec, sqrtVarLocal, this->N);

            calcQueue.FreeTensor(denomLocal);
            calcQueue.FreeTensor(epsVec);
            calcQueue.FreeTensor(sqrtVarLocal);
            calcQueue.FreeTensor(varLocal);
            calcQueue.FreeTensor(oneVec);
        }

        // 第三遍：输出 y = ((x*scale - mean) * invStd) * gamma + beta
        for (uint32_t b = 0; b < this->B; ++b) {
            AscendC::LocalTensor<DTYPE> rowLocal = inRowQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> scaledLocal = outRowQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> normLocal = outRowQueue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> outLocal = outRowQueue.AllocTensor<DTYPE>();

            AscendC::DataCopy(rowLocal, xGm[b * this->N], this->N);
            AscendC::LocalTensor<DTYPE> scaleT = paramQueue.DeQue<DTYPE>();
            AscendC::LocalTensor<DTYPE> gammaT = paramQueue.DeQue<DTYPE>();
            AscendC::LocalTensor<DTYPE> betaT  = paramQueue.DeQue<DTYPE>();

            AscendC::Mul(scaledLocal, rowLocal, scaleT, this->N);             // x*scale
            AscendC::Sub(normLocal, scaledLocal, meanLocal, this->N);         // x*scale - mean
            AscendC::Mul(normLocal, normLocal, invStdLocal, this->N);         // (x*scale - mean) * invStd
            AscendC::Mul(outLocal, normLocal, gammaT, this->N);               // * gamma
            AscendC::Add(outLocal, outLocal, betaT, this->N);                 // + beta

            AscendC::DataCopy(yGm[b * this->N], outLocal, this->N);

            paramQueue.EnQue(scaleT);
            paramQueue.EnQue(gammaT);
            paramQueue.EnQue(betaT);

            inRowQueue.FreeTensor(rowLocal);
            outRowQueue.FreeTensor(scaledLocal);
            outRowQueue.FreeTensor(normLocal);
            outRowQueue.FreeTensor(outLocal);
        }

        // 释放局部向量
        calcQueue.FreeTensor(meanLocal);
        calcQueue.FreeTensor(varSumLocal);
        calcQueue.FreeTensor(invStdLocal);

        // 释放参数局部向量
        AscendC::LocalTensor<DTYPE> tmp;
        tmp = paramQueue.DeQue<DTYPE>(); paramQueue.FreeTensor(tmp);
        tmp = paramQueue.DeQue<DTYPE>(); paramQueue.FreeTensor(tmp);
        tmp = paramQueue.DeQue<DTYPE>(); paramQueue.FreeTensor(tmp);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inRowQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 3> paramQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outRowQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 3> calcQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> scaleGm;
    AscendC::GlobalTensor<DTYPE> gammaGm;
    AscendC::GlobalTensor<DTYPE> betaGm;
    AscendC::GlobalTensor<DTYPE> epsGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    AscendC::GlobalTensor<DTYPE> workMeanGm;
    AscendC::GlobalTensor<DTYPE> workVarSumGm;
    AscendC::GlobalTensor<DTYPE> workInvStdGm;

    uint32_t B;
    uint32_t N;
};

extern "C" __global__ __aicore__ void gemm_scale_batchnorm_custom(
    GM_ADDR x, GM_ADDR scale, GM_ADDR gamma, GM_ADDR beta, GM_ADDR eps,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelGemmScaleBatchnorm op;
    op.Init(x, scale, gamma, beta, eps, y, workspace, tiling);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom NPU op: scale + batchnorm applied to GEMM output.
    GEMM仍使用nn.Linear，随后调用自定义AscendC核完成缩放与批归一化。
    """
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.scale = torch.nn.Parameter(torch.rand(scale_shape))
        self.bn = torch.nn.BatchNorm1d(out_features, eps=eps, momentum=momentum)

    def forward(self, x):
        y = self.gemm(x)
        eps_tensor = torch.tensor(self.bn.eps, device=y.device, dtype=y.dtype).reshape(1)
        y = custom_ops_lib.gemm_scale_batchnorm_custom(y, self.scale, self.bn.weight, self.bn.bias, eps_tensor)
        return y
