kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelDepthwiseConv2d {
public:
    __aicore__ inline KernelDepthwiseConv2d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t K, uint32_t outH, uint32_t outW,
                                uint32_t tileH, uint32_t tileW)
    {
        this->N = N; this->C = C; this->H = H; this->W = W;
        this->K = K; this->outH = outH; this->outW = outW;
        this->tileH = tileH; this->tileW = tileW;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint32_t)(N*C*H*W));
        wGm.SetGlobalBuffer((__gm__ DTYPE_X*)w, (uint32_t)(C*K*K));
        yGm.SetGlobalBuffer((__gm__ DTYPE_X*)y, (uint32_t)(N*C*outH*outW));

        // Max local buffer sizes for a tile
        uint32_t inPatchMax = (tileH + K - 1) * (tileW + K - 1);
        uint32_t outTileMax = tileH * tileW;
        uint32_t kSize = K * K;

        pipe.InitBuffer(inQueueX, BUFFER_NUM, inPatchMax * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, kSize * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, outTileMax * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blocks = AscendC::GetBlockNum();

        if (C == 0 || N == 0) { return; }

        uint32_t ncTotal = N * C;
        if (blockIdx >= ncTotal) { return; }

        uint32_t n = blockIdx / C;
        uint32_t c = blockIdx % C;

        // Base offsets in GM for this (n,c)
        uint32_t xBase = (n * C + c) * H * W;
        uint32_t yBase = (n * C + c) * outH * outW;
        uint32_t wBase = c * K * K;

        // Preload kernel weights for this channel tile-by-tile (simple approach)
        for (uint32_t oh = 0; oh < outH; oh += tileH) {
            uint32_t ph = (oh + tileH <= outH) ? tileH : (outH - oh);
            for (uint32_t ow = 0; ow < outW; ow += tileW) {
                uint32_t pw = (ow + tileW <= outW) ? tileW : (outW - ow);

                uint32_t inPatchH = ph + K - 1;
                uint32_t inPatchW = pw + K - 1;

                // Local allocations
                AscendC::LocalTensor<DTYPE_X> xPatch = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_X> wLocal = inQueueW.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_X> yTile = outQueueY.AllocTensor<DTYPE_X>();

                // Copy weight kernel [K*K]
                AscendC::DataCopy(wLocal, wGm[wBase], K * K);

                // Copy input patch from GM: shape [inPatchH, inPatchW]
                for (uint32_t r = 0; r < inPatchH; ++r) {
                    uint32_t gmOff = xBase + (oh + r) * W + ow;
                    AscendC::DataCopy(xPatch[r * inPatchW], xGm[gmOff], inPatchW);
                }

                // Compute convolution for the tile: output [ph, pw]
                for (uint32_t i = 0; i < ph; ++i) {
                    for (uint32_t j = 0; j < pw; ++j) {
                        DTYPE_X acc = (DTYPE_X)0.0f;
                        for (uint32_t kh = 0; kh < K; ++k h) {
                            for (uint32_t kw = 0; kw < K; ++kw) {
                                DTYPE_X xv = xPatch[(i + kh) * inPatchW + (j + kw)];
                                DTYPE_X wv = wLocal[kh * K + kw];
                                acc = acc + xv * wv;
                            }
                        }
                        yTile[i * pw + j] = acc;
                    }
                }

                // Copy out tile to GM, row by row
                for (uint32_t r = 0; r < ph; ++r) {
                    uint32_t yOff = yBase + (oh + r) * outW + ow;
                    AscendC::DataCopy(yGm[yOff], &yTile[r * pw], pw);
                }

                outQueueY.EnQue<DTYPE_X>(yTile);
                inQueueW.EnQue(wLocal);
                inQueueX.EnQue(xPatch);
                outQueueY.FreeTensor(yTile);
                inQueueW.FreeTensor(wLocal);
                inQueueX.FreeTensor(xPatch);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> wGm;
    AscendC::GlobalTensor<DTYPE_X> yGm;

    uint32_t N, C, H, W, K, outH, outW;
    uint32_t tileH, tileW;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDepthwiseConv2d op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.k, tiling_data.outH, tiling_data.outW,
            tiling_data.tileH, tiling_data.tileW);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        # keep a depthwise conv to store learned weights/bias; we will use its weights in custom op
        self.conv2d = torch.nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)
        self.stride = stride
        self.padding = padding
        self.kernel_size = kernel_size

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Only support stride=1, padding=0 in this custom kernel demonstration
        assert self.stride == 1 and self.padding == 0, "Custom kernel currently supports stride=1 and padding=0"
        return custom_ops_lib.conv_depthwise_2d_square_input_square_kernel_custom(x, self.conv2d.weight)
