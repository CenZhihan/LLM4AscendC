kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数量
constexpr float EPS = 1e-8f;

class KernelCosineSimilarityLoss {
public:
    __aicore__ inline KernelCosineSimilarityLoss() {}
    __aicore__ inline void Init(GM_ADDR predictions, GM_ADDR targets, GM_ADDR z, uint32_t batch, uint32_t featureLen, uint32_t tileNum)
    {
        this->batch = batch;
        this->featureLen = featureLen;
        this->tileNum = tileNum;
        // 假设batch可整除block数
        this->rowsPerBlock = this->batch / AscendC::GetBlockNum();
        this->tileLen = this->featureLen / this->tileNum;

        uint32_t startRow = this->rowsPerBlock * AscendC::GetBlockIdx();
        uint32_t totalBlockElems = this->rowsPerBlock * this->featureLen;

        predGm.SetGlobalBuffer((__gm__ DTYPE_X*)predictions + (uint64_t)startRow * this->featureLen, totalBlockElems);
        targGm.SetGlobalBuffer((__gm__ DTYPE_Y*)targets + (uint64_t)startRow * this->featureLen, totalBlockElems);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z, 1); // 标量输出

        // 输入/计算/输出队列
        pipe.InitBuffer(inQueuePred, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueTarg, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpVecQue, 1, this->tileLen * sizeof(float));    // 临时向量
        pipe.InitBuffer(workQue, 1, this->tileLen * sizeof(float));      // ReduceSum工作空间
        pipe.InitBuffer(outQueueZ, 1, 32); // 标量输出，分配32字节对齐空间
    }

    __aicore__ inline void Process()
    {
        float invBatch = 1.0f / static_cast<float>(this->batch);
        float blockAcc = 0.0f;

        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            float sum_xy = 0.0f;
            float sum_x2 = 0.0f;
            float sum_y2 = 0.0f;

            for (uint32_t t = 0; t < this->tileNum; ++t) {
                CopyIn(r, t);

                // 取本tile局部向量
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueuePred.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueTarg.DeQue<DTYPE_Y>();

                AscendC::LocalTensor<float> tmpVec = tmpVecQue.AllocTensor<float>();
                AscendC::LocalTensor<float> workVec = workQue.AllocTensor<float>();

                // dot 分片
                AscendC::Mul(tmpVec, xLocal, yLocal, this->tileLen);
                AscendC::ReduceSum(tmpVec, tmpVec, workVec, this->tileLen);
                sum_xy += AscendC::GetAccVal<float>();

                // x^2 分片
                AscendC::Mul(tmpVec, xLocal, xLocal, this->tileLen);
                AscendC::ReduceSum(tmpVec, tmpVec, workVec, this->tileLen);
                sum_x2 += AscendC::GetAccVal<float>();

                // y^2 分片
                AscendC::Mul(tmpVec, yLocal, yLocal, this->tileLen);
                AscendC::ReduceSum(tmpVec, tmpVec, workVec, this->tileLen);
                sum_y2 += AscendC::GetAccVal<float>();

                // 释放本次使用的本地tensor
                tmpVecQue.FreeTensor(tmpVec);
                workQue.FreeTensor(workVec);
                inQueuePred.FreeTensor(xLocal);
                inQueueTarg.FreeTensor(yLocal);
            }

            float nx = AscendC::SqrtScalar(sum_x2);
            float ny = AscendC::SqrtScalar(sum_y2);
            if (nx < EPS) nx = EPS;
            if (ny < EPS) ny = EPS;
            float cos_val = sum_xy / (nx * ny);
            float loss = 1.0f - cos_val;
            blockAcc += (loss * invBatch);
        }

        // 原子加到全局标量z
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
        AscendC::Duplicate(zLocal, blockAcc, 1);
        AscendC::SetAtomicAdd<float>();
        AscendC::DataCopy(outGm, zLocal, 1);
        AscendC::SetAtomicNone();
        outQueueZ.FreeTensor(zLocal);
    }

private:
    __aicore__ inline void CopyIn(uint32_t row, uint32_t tileIdx)
    {
        uint64_t baseOffset = (uint64_t)row * this->featureLen + (uint64_t)tileIdx * this->tileLen;

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueuePred.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueTarg.AllocTensor<DTYPE_Y>();

        AscendC::DataCopy(xLocal, predGm[baseOffset], this->tileLen);
        AscendC::DataCopy(yLocal, targGm[baseOffset], this->tileLen);

        inQueuePred.EnQue(xLocal);
        inQueueTarg.EnQue(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePred, inQueueTarg;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVecQue;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workQue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> predGm;
    AscendC::GlobalTensor<DTYPE_Y> targGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;

    uint32_t batch;
    uint32_t featureLen;
    uint32_t tileNum;
    uint32_t rowsPerBlock;
    uint32_t tileLen;
};

// 标准kernel入口
extern "C" __global__ __aicore__ void cosine_similarity_loss_custom(GM_ADDR predictions, GM_ADDR targets, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCosineSimilarityLoss op;
    op.Init(predictions, targets, z, tiling_data.batch, tiling_data.featureLen, tiling_data.tileNum);
    op.Process();
}
"""
