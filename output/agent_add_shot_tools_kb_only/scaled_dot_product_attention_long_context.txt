#include "scaled_dot_product_attention_long_context_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint64_t ProductExceptLast(const gert::Shape &shape) {
    uint64_t prod = 1;
    auto dims = shape.GetOriginShape().GetDims();
    if (dims.size() == 0) { return 1; }
    for (size_t i = 0; i + 1 < dims.size(); ++i) {
        prod *= static_cast<uint64_t>(dims[i]);
    }
    return prod;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();
    auto dims = origin.GetDims();
    if (dims.size() == 0) {
        return ge::GRAPH_FAILED;
    }
    uint32_t srcK = static_cast<uint32_t>(dims.back());
    uint32_t srcM = static_cast<uint32_t>(ProductExceptLast(*in_shape));

    ScaledDotProductAttentionLongContextCustomTilingData td;
    td.set_srcM(srcM);
    td.set_srcK(srcK);
    // 为了通用性，这里按行处理，每次处理1行
    td.set_tileRows(1);

    context->SetBlockDim(BLOCK_DIM);
    td.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(td.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0; // 无额外workspace
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using dtype_t = float;

// 简化版：对输入scores按行执行softmax（适用于长序列场景），输出probs
// 使用SoftmaxFlashV2完成按行softmax（先计算exp和sum，再归一化）。
class KernelScaledDotProductAttentionLongContextCustom {
public:
    __aicore__ inline KernelScaledDotProductAttentionLongContextCustom() {}

    __aicore__ inline void Init(GM_ADDR scores, GM_ADDR probs, uint32_t srcM, uint32_t srcK, uint32_t tileRows)
    {
        this->srcM = srcM;
        this->srcK = srcK;
        this->rowsPerBlock = srcM / AscendC::GetBlockNum();
        this->rowStart = this->rowsPerBlock * AscendC::GetBlockIdx();
        this->tileRows = tileRows;

        scoresGm.SetGlobalBuffer((__gm__ dtype_t *)scores + (uint64_t)rowStart * srcK, (uint64_t)rowsPerBlock * srcK);
        probsGm.SetGlobalBuffer((__gm__ dtype_t *)probs + (uint64_t)rowStart * srcK, (uint64_t)rowsPerBlock * srcK);

        // 输入行缓冲与输出行缓冲
        uint32_t rowBytes = srcK * sizeof(dtype_t);
        pipe.InitBuffer(inQueueScores, 1, rowBytes);
        pipe.InitBuffer(outQueueProbs, 1, rowBytes);

        // Softmax需要的临时tensor：exp(sum)/max缓冲区
        // 对于float类型，广播后的最后一维长度固定为32字节，即8个float。
        uint32_t brcElems = 32 / sizeof(dtype_t);
        pipe.InitBuffer(sumQueue, 1, brcElems * sizeof(dtype_t) * tileRows);
        pipe.InitBuffer(maxQueue, 1, brcElems * sizeof(dtype_t) * tileRows);
        // expMax只在update=true时使用，这里准备占位（大小与sum/max一致，按需分配）
        pipe.InitBuffer(expMaxQueue, 1, brcElems * sizeof(dtype_t) * tileRows);
        // inSum/inMax作为占位输入（isUpdate=false时不使用）
        pipe.InitBuffer(inSumQueue, 1, brcElems * sizeof(dtype_t) * tileRows);
        pipe.InitBuffer(inMaxQueue, 1, brcElems * sizeof(dtype_t) * tileRows);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsPerBlock; r += tileRows) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowOffset)
    {
        AscendC::LocalTensor<dtype_t> scoresLocal = inQueueScores.AllocTensor<dtype_t>();
        // 每次处理tileRows行（此处tileRows=1）
        AscendC::DataCopy(scoresLocal, scoresGm[rowOffset * srcK], srcK * tileRows);
        inQueueScores.EnQue(scoresLocal);
    }

    __aicore__ inline void Compute(uint32_t rowOffset)
    {
        AscendC::LocalTensor<dtype_t> scoresLocal = inQueueScores.DeQue<dtype_t>();
        AscendC::LocalTensor<dtype_t> probsLocal = outQueueProbs.AllocTensor<dtype_t>();

        // SoftmaxFlashV2需要sum/max缓冲区（广播到32字节）
        AscendC::LocalTensor<dtype_t> sumLocal = sumQueue.AllocTensor<dtype_t>();
        AscendC::LocalTensor<dtype_t> maxLocal = maxQueue.AllocTensor<dtype_t>();
        AscendC::LocalTensor<dtype_t> expMaxLocal = expMaxQueue.AllocTensor<dtype_t>();
        AscendC::LocalTensor<dtype_t> inSumLocal = inSumQueue.AllocTensor<dtype_t>();
        AscendC::LocalTensor<dtype_t> inMaxLocal = inMaxQueue.AllocTensor<dtype_t>();

        // 形状信息：按行（m=tileRows，n=srcK）
        AscendC::SoftMaxShapeInfo shapeInfo = {tileRows, srcK, tileRows, srcK};
        // 在核内重新计算tiling
        auto tiling = AscendC::SoftMaxFlashV2TilingFunc(shapeInfo, sizeof(dtype_t), sizeof(dtype_t), 0, false, false, false, false);

        // 先用SoftmaxFlashV2计算 exp(scores - max) 和 sum
        AscendC::SoftmaxFlashV2<dtype_t, false>(probsLocal, sumLocal, maxLocal, scoresLocal, expMaxLocal, inSumLocal, inMaxLocal, tiling, shapeInfo);

        // 再进行归一化：probs = exp / sum
        AscendC::Div(probsLocal, probsLocal, sumLocal, srcK * tileRows);

        outQueueProbs.EnQue<dtype_t>(probsLocal);

        // 释放本次使用的本地tensor
        sumQueue.FreeTensor(sumLocal);
        maxQueue.FreeTensor(maxLocal);
        expMaxQueue.FreeTensor(expMaxLocal);
        inSumQueue.FreeTensor(inSumLocal);
        inMaxQueue.FreeTensor(inMaxLocal);
        inQueueScores.FreeTensor(scoresLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowOffset)
    {
        AscendC::LocalTensor<dtype_t> probsLocal = outQueueProbs.DeQue<dtype_t>();
        AscendC::DataCopy(probsGm[rowOffset * srcK], probsLocal, srcK * tileRows);
        outQueueProbs.FreeTensor(probsLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueScores;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueProbs;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> sumQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> maxQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> expMaxQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> inSumQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> inMaxQueue;

    AscendC::GlobalTensor<dtype_t> scoresGm;
    AscendC::GlobalTensor<dtype_t> probsGm;

    uint32_t srcM;
    uint32_t srcK;
    uint32_t tileRows;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
};

extern "C" __global__ __aicore__ void scaled_dot_product_attention_long_context_custom(GM_ADDR scores, GM_ADDR probs, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelScaledDotProductAttentionLongContextCustom op;
    op.Init(scores, probs, tiling_data.srcM, tiling_data.srcK, tiling_data.tileRows);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import math

class ModelNew(torch.nn.Module):
    def __init__(self) -> None:
        super().__init__()

    def forward(self, Q, K, V):
        # 先在PyTorch侧计算缩放点积注意力的分数矩阵
        scale = 1.0 / math.sqrt(Q.size(-1))
        scores = torch.matmul(Q, K.transpose(-2, -1)) * scale
        # 使用自定义AscendC核对长序列进行高效softmax
        probs = custom_ops_lib.scaled_dot_product_attention_long_context_custom(scores)
        # 再与V相乘得到输出
        output = torch.matmul(probs, V)
        return output
