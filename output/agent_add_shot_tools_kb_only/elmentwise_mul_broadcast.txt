namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ElmentwiseMulBroadcastCustomTilingData tiling;

    // 读取输入形状（按ND 3维处理）
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::StorageShape *y_shape = context->GetInputShape(1);
    uint32_t xD0 = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(0));
    uint32_t xD1 = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(1));
    uint32_t xD2 = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(2));
    uint32_t yD0 = static_cast<uint32_t>(y_shape->GetStorageShape().GetDim(0));
    uint32_t yD1 = static_cast<uint32_t>(y_shape->GetStorageShape().GetDim(1));
    uint32_t yD2 = static_cast<uint32_t>(y_shape->GetStorageShape().GetDim(2));

    // 广播尺寸计算
    auto brc_dim = [](uint32_t a, uint32_t b) -> uint32_t {
        if (a == b) return a;
        if (a == 1) return b;
        if (b == 1) return a;
        return 0; // 无法广播
    };

    uint32_t outD0 = brc_dim(xD0, yD0);
    uint32_t outD1 = brc_dim(xD1, yD1);
    uint32_t outD2 = brc_dim(xD2, yD2);

    if (outD0 == 0 || outD1 == 0 || outD2 == 0) {
        return ge::GRAPH_FAILED;
    }

    // 简单设定：tileK = kLen（可按需调整为更小块以适配UB）
    uint32_t tileK = outD2;

    // 设置tiling与BLOCK数
    context->SetBlockDim(1);

    tiling.set_outD0(outD0);
    tiling.set_outD1(outD1);
    tiling.set_kLen(outD2);
    tiling.set_tileK(tileK);
    tiling.set_xD0(xD0);
    tiling.set_xD1(xD1);
    tiling.set_xD2(xD2);
    tiling.set_yD0(yD0);
    tiling.set_yD1(yD1);
    tilling.set_yD2(yD2);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMulBroadcast {
public:
    __aicore__ inline KernelMulBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t outD0, uint32_t outD1, uint32_t kLen,
                                uint32_t xD0, uint32_t xD1, uint32_t xD2,
                                uint32_t yD0, uint32_t yD1, uint32_t yD2,
                                uint32_t tileK)
    {
        this->outD0 = outD0;
        this->outD1 = outD1;
        this->kLen  = kLen;
        this->xD0 = xD0; this->xD1 = xD1; this->xD2 = xD2;
        this->yD0 = yD0; this->yD1 = yD1; this->yD2 = yD2;
        this->tileK = tileK;

        xGm.SetGlobalBuffer((__gm__ float *)x);
        yGm.SetGlobalBuffer((__gm__ float *)y);
        zGm.SetGlobalBuffer((__gm__ float *)z);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileK * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < outD0; ++i) {
            for (uint32_t j = 0; j < outD1; ++j) {
                uint32_t xi = (xD0 == 1) ? 0 : i;
                uint32_t xj = (xD1 == 1) ? 0 : j;
                uint32_t yi = (yD0 == 1) ? 0 : i;
                uint32_t yj = (yD1 == 1) ? 0 : j;

                uint32_t xBase = (xi * xD1 + xj) * xD2;
                uint32_t yBase = (yi * yD1 + yj) * yD2;
                uint32_t zBase = (i * outD1 + j) * kLen;

                for (uint32_t k = 0; k < kLen; k += tileK) {
                    uint32_t curLen = (k + tileK <= kLen) ? tileK : (kLen - k);

                    AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                    AscendC::LocalTensor<float> yLocal = inQueueY.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm[xBase + k], curLen);
                    AscendC::DataCopy(yLocal, yGm[yBase + k], curLen);
                    inQueueX.EnQue(xLocal);
                    inQueueY.EnQue(yLocal);

                    AscendC::LocalTensor<float> xLocalDeq = inQueueX.DeQue<float>();
                    AscendC::LocalTensor<float> yLocalDeq = inQueueY.DeQue<float>();
                    AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
                    AscendC::Mul(zLocal, xLocalDeq, yLocalDeq, curLen);
                    outQueueZ.EnQue<float>(zLocal);
                    inQueueX.FreeTensor(xLocalDeq);
                    inQueueY.FreeTensor(yLocalDeq);

                    AscendC::LocalTensor<float> zLocalDeq = outQueueZ.DeQue<float>();
                    AscendC::DataCopy(zGm[zBase + k], zLocalDeq, curLen);
                    outQueueZ.FreeTensor(zLocalDeq);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t outD0, outD1, kLen;
    uint32_t xD0, xD1, xD2;
    uint32_t yD0, yD1, yD2;
    uint32_t tileK;
};

extern "C" __global__ __aicore__ void elmentwise_mul_broadcast_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMulBroadcast op;
    op.Init(x, y, z,
            tiling_data.outD0, tiling_data.outD1, tiling_data.kLen,
            tiling_data.xD0, tiling_data.xD1, tiling_data.xD2,
            tiling_data.yD0, tiling_data.yD1, tiling_data.yD2,
            tiling_data.tileK);
    op.Process();
}
