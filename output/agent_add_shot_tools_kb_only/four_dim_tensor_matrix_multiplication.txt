host_operator_src="""
#include <cstdint>
#include <cstring>
#include <algorithm>
#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"
#include "tiling/tiling_api.h"
#include "four_dim_tensor_matrix_multiplication_custom_tiling.h"

// 依赖 Matmul tiling 高级接口
#include "tiling/matmul_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 读取输入形状
    const gert::Shape* aShape = context->GetInputShape(0);
    const gert::Shape* bShape = context->GetInputShape(1);

    // 假定接口提供如下方法获取各维度（常见形状接口）
    // A: (b, i, j, l), B: (l, k)
    uint32_t b = static_cast<uint32_t>(aShape->GetOriginShape().GetDim(0));
    uint32_t iDim = static_cast<uint32_t>(aShape->GetOriginShape().GetDim(1));
    uint32_t jDim = static_cast<uint32_t>(aShape->GetOriginShape().GetDim(2));
    uint32_t lDim = static_cast<uint32_t>(aShape->GetOriginShape().GetDim(3));
    uint32_t kDim = static_cast<uint32_t>(bShape->GetOriginShape().GetDim(1));

    // 将 (b, i, j, l) 看作 batchA = b*i 的批次；每个批次做 [M=j, K=l] x [K=l, N=k]
    uint32_t batchA = b * iDim;
    int32_t M = static_cast<int32_t>(jDim);
    int32_t K = static_cast<int32_t>(lDim);
    int32_t N = static_cast<int32_t>(kDim);

    // 生成 Matmul 的 tiling 数据（TCubeTiling）
    auto platformInfo = context->GetPlatformInfo();
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(platformInfo);

    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);
    tiling.SetDim(1);
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetShape(M, N, K);
    tiling.SetOrgShape(M, N, K);
    tiling.SetBias(false);
    tiling.SetBufferSpace(-1, -1, -1);
    // NORMAL 布局的多批次信息：batchA 批，batchB=1（右矩阵广播）
    tiling.SetBatchInfoForNormal(static_cast<int32_t>(batchA), 1, M, N, K);

    optiling::TCubeTiling cubeTiling;
    int ret = tiling.GetTiling(cubeTiling);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // 计算并设置 blockDim，简单起见设定为 8 或批次上限
    uint32_t blockDim = std::min<uint32_t>(8U, std::max<uint32_t>(1U, batchA));
    context->SetBlockDim(blockDim);

    // 组装扩展的自定义 tiling 数据（紧随 TCubeTiling 之后）
    FourDimTensorMatrixMultiplicationCustomExtra extra;
    extra.set_b(b);
    extra.set_iDim(iDim);
    extra.set_jDim(jDim);
    extra.set_lDim(lDim);
    extra.set_kDim(kDim);
    extra.set_batchA(batchA);

    // 将 TCubeTiling 和 extra 依次写入 raw tiling buffer
    auto raw = context->GetRawTilingData();
    uint8_t* buf = reinterpret_cast<uint8_t*>(raw->GetData());
    size_t cap = raw->GetCapacity();
    size_t needSize = sizeof(optiling::TCubeTiling) + extra.GetDataSize();

    if (needSize > cap) {
        return ge::GRAPH_FAILED;
    }

    // 写入 TCubeTiling
    std::memcpy(buf, &cubeTiling, sizeof(optiling::TCubeTiling));
    // 写入扩展自定义字段
    extra.SaveToBuffer(buf + sizeof(optiling::TCubeTiling), cap - sizeof(optiling::TCubeTiling));
    raw->SetDataSize(needSize);

    // 无需 workspace
    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;

    // 设置 TilingKey
    context->SetTilingKey(1);
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // A: (b, i, j, l), B: (l, k) => C: (b, i, j, k)
    const gert::Shape* aShape = context->GetInputShape(0);
    const gert::Shape* bShape = context->GetInputShape(1);

    int64_t b = aShape->GetOriginShape().GetDim(0);
    int64_t iDim = aShape->GetOriginShape().GetDim(1);
    int64_t jDim = aShape->GetOriginShape().GetDim(2);
    int64_t kDim = bShape->GetOriginShape().GetDim(1);

    gert::Shape* cShape = context->GetOutputShape(0);
    cShape->SetDimNum(4);
    cShape->SetDim(0, b);
    cShape->SetDim(1, iDim);
    cShape->SetDim(2, jDim);
    cShape->SetDim(3, kDim);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    const auto dtA = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtA);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

// 额外的扩展 tiling 数据布局（与 host 写入顺序一致）
struct ExtraTilingInfo {
    uint32_t b;
    uint32_t iDim;
    uint32_t jDim;  // M
    uint32_t lDim;  // K
    uint32_t kDim;  // N
    uint32_t batchA;
};

extern "C" __global__ __aicore__ void four_dim_tensor_matrix_multiplication_custom(
    GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // 读取 TCubeTiling
    optiling::TCubeTiling tiling;
    auto tilingPtrGM = reinterpret_cast<__gm__ uint32_t*>(tilingGM);
    auto tilingPtrLocal = reinterpret_cast<uint32_t*>(&tiling);
    for (int i = 0; i < (int)(sizeof(optiling::TCubeTiling) / sizeof(uint32_t)); ++i) {
        tilingPtrLocal[i] = tilingPtrGM[i];
    }

    // 读取扩展的自定义 tiling 字段（紧随 TCubeTiling 之后）
    ExtraTilingInfo extra;
    auto extraBase = reinterpret_cast<__gm__ uint8_t*>(tilingGM) + sizeof(optiling::TCubeTiling);
    __gm__ ExtraTilingInfo* extraGM = reinterpret_cast<__gm__ ExtraTilingInfo*>(extraBase);
    extra = *extraGM;

    // A: [batchA, M=jDim, K=lDim], B: [K=lDim, N=kDim], C: [batchA, M=jDim, N=kDim]
    uint32_t batchA = extra.batchA;
    uint32_t M = extra.jDim;
    uint32_t K = extra.lDim;
    uint32_t N = extra.kDim;

    // 包装 GM Tensor
    AscendC::GlobalTensor<float> aGlobal;
    AscendC::GlobalTensor<float> bGlobal;
    AscendC::GlobalTensor<float> cGlobal;
    uint32_t aElems = batchA * M * K;
    uint32_t bElems = K * N;
    uint32_t cElems = batchA * M * N;
    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(aGM), aElems);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bGM), bElems);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cGM), cElems);

    // MatmulType 定义，采用 NORMAL 布局做多批次计算
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    // 创建并注册 Matmul 实例
    matmul::Matmul<aType, bType, cType, biasType> mm;
    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);

    // 初始化 Matmul
    mm.Init(&tiling);

    // 设置张量、工作区与（可选）bias
    mm.SetTensorA(aGlobal[0], false);
    mm.SetTensorB(bGlobal[0], false);
    mm.SetWorkspace(workspaceGM, 0);

    // 执行多批次 Matmul
    // NORMAL 布局下，batchA 为左矩阵批次数，右矩阵 batchB=1（广播）
    // enSequentialWrite = true：输出按连续方式写入
    const uint32_t batchLoop = batchA; // 当前计算的 BMM 次数（简单设为 batchA）
    mm.IterateNBatch(batchLoop, batchA, 1, true);

    // 结果直接写入 cGM（由 Matmul 高级 API 完成）
}
