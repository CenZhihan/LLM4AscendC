#include "sgd_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    SgdCustomTilingData tiling;
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    float momentum = 0.9f;
    float lr = 1e-2f;

    // 若框架侧透传了属性，这里可读取；未读取到则使用默认值
    // 不同版本接口可能差异，以下仅示意
    auto attrs = context->GetAttrs();
    if (attrs != nullptr) {
        (void)attrs->GetFloat("momentum", momentum);
        (void)attrs->GetFloat("lr", lr);
    }

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_momentum(momentum);
    tiling.set_lr(lr);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    context->SetTilingKey(1);
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);
    *out_shape = *in_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

// 基于 Add 示例，按块-分片处理
constexpr int32_t BUFFER_NUM = 2; // 每个队列中 tensor 数

class KernelSgd {
public:
    __aicore__ inline KernelSgd() {}
    __aicore__ inline void Init(GM_ADDR param, GM_ADDR grad, GM_ADDR velocity, GM_ADDR out,
                                uint32_t totalLength, uint32_t tileNum,
                                float momentum, float lr)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->momentum = momentum;
        this->lr = lr;

        paramGm.SetGlobalBuffer((__gm__ DTYPE_X *)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gradGm.SetGlobalBuffer((__gm__ DTYPE_X *)grad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        velGm.SetGlobalBuffer((__gm__ DTYPE_X *)velocity + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_X *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueParam, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueGrad,  BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueVel,   BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueOut,  BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> paramLocal = inQueueParam.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> gradLocal  = inQueueGrad.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> velLocal   = inQueueVel.AllocTensor<DTYPE_X>();

        AscendC::DataCopy(paramLocal, paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gradLocal,  gradGm[progress * this->tileLength],  this->tileLength);
        AscendC::DataCopy(velLocal,   velGm[progress * this->tileLength],   this->tileLength);

        inQueueParam.EnQue(paramLocal);
        inQueueGrad.EnQue(gradLocal);
        inQueueVel.EnQue(velLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> paramLocal = inQueueParam.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> gradLocal  = inQueueGrad.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> velLocal   = inQueueVel.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> outLocal   = outQueueOut.AllocTensor<DTYPE_X>();

        // velocity = momentum * velocity + grad
        AscendC::Muls(velLocal, velLocal, this->momentum, this->tileLength);
        AscendC::Add(velLocal, velLocal, gradLocal, this->tileLength);

        // out = param - lr * velocity
        AscendC::Muls(outLocal, velLocal, this->lr, this->tileLength);
        AscendC::Sub(outLocal, paramLocal, outLocal, this->tileLength);

        outQueueOut.EnQue<DTYPE_X>(outLocal);
        inQueueParam.FreeTensor(paramLocal);
        inQueueGrad.FreeTensor(gradLocal);
        inQueueVel.FreeTensor(velLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> outLocal = outQueueOut.DeQue<DTYPE_X>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueParam, inQueueGrad, inQueueVel;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut;

    AscendC::GlobalTensor<DTYPE_X> paramGm;
    AscendC::GlobalTensor<DTYPE_X> gradGm;
    AscendC::GlobalTensor<DTYPE_X> velGm;
    AscendC::GlobalTensor<DTYPE_X> outGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    float momentum;
    float lr;
};

extern "C" __global__ __aicore__ void sgd_custom(GM_ADDR param, GM_ADDR grad, GM_ADDR velocity, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSgd op;
    op.Init(param, grad, velocity, out, tiling_data.totalLength, tiling_data.tileNum, tiling_data.momentum, tiling_data.lr);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}
