kernel_src="""
#include "kernel_operator.h"

using dataType = float;

class KernelInstanceNorm {
public:
    __aicore__ inline KernelInstanceNorm() {}
    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR gammaGm, GM_ADDR betaGm, GM_ADDR yGm,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t g, float eps)
    {
        this->n_ = n;
        this->c_ = c;
        this->h_ = h;
        this->w_ = w;
        this->g_ = g;
        this->epsilon_ = eps;

        xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ dataType*>(xGm), n_ * c_ * h_ * w_);
        gammaGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ dataType*>(gammaGm), c_);
        betaGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ dataType*>(betaGm), c_);
        yGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ dataType*>(yGm), n_ * c_ * h_ * w_);

        uint32_t bshLength = n_ * c_ * h_ * w_;
        uint32_t hwAlignSize = (sizeof(dataType) * h_ * w_ + ONE_BLK_SIZE - 1) / ONE_BLK_SIZE * ONE_BLK_SIZE / sizeof(dataType);

        pipe.InitBuffer(inQueueX, 1, sizeof(dataType) * n_ * c_ * hwAlignSize);
        pipe.InitBuffer(inQueueGamma, 1, (sizeof(dataType) * c_ + 31) / 32 * 32);
        pipe.InitBuffer(inQueueBeta, 1, (sizeof(dataType) * c_ + 31) / 32 * 32);
        pipe.InitBuffer(outQueueY, 1, sizeof(dataType) * n_ * c_ * hwAlignSize);
        pipe.InitBuffer(meanBuffer, (sizeof(dataType) * g_ * n_ + 31) / 32 * 32);
        pipe.InitBuffer(varBuffer, (sizeof(dataType) * g_ * n_ + 31) / 32 * 32);
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<dataType> xLocal = inQueueX.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> gammaLocal = inQueueGamma.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> betaLocal = inQueueBeta.AllocTensor<dataType>();

        uint32_t hwAlignSize = (sizeof(dataType) * h_ * w_ + ONE_BLK_SIZE - 1) / ONE_BLK_SIZE * ONE_BLK_SIZE / sizeof(dataType);
        AscendC::DataCopyParams copyParamsX{static_cast<uint16_t>(n_ * c_), static_cast<uint16_t>(h_ * w_ * sizeof(dataType)), 0, 0};
        AscendC::DataCopyPadParams padParamsX{true, 0, static_cast<uint8_t>(hwAlignSize - h_ * w_), 0};
        AscendC::DataCopyPad(xLocal, xGlobal, copyParamsX, padParamsX);

        AscendC::DataCopyParams copyParamsGamma{1, static_cast<uint16_t>(c_ * sizeof(dataType)), 0, 0};
        AscendC::DataCopyPadParams padParamsGamma{false, 0, 0, 0};
        AscendC::DataCopyPad(gammaLocal, gammaGlobal, copyParamsGamma, padParamsGamma);
        AscendC::DataCopyPad(betaLocal, betaGlobal, copyParamsGamma, padParamsGamma);

        inQueueX.EnQue(xLocal);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<dataType> xLocal = inQueueX.DeQue<dataType>();
        AscendC::LocalTensor<dataType> gammaLocal = inQueueGamma.DeQue<dataType>();
        AscendC::LocalTensor<dataType> betaLocal = inQueueBeta.DeQue<dataType>();
        AscendC::LocalTensor<dataType> yLocal = outQueueY.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> meanLocal = meanBuffer.Get<dataType>();
        AscendC::LocalTensor<dataType> varianceLocal = varBuffer.Get<dataType>();

        AscendC::PipeBarrier<PIPE_ALL>();

        uint32_t stackBufferSize = 0;
        {
            AscendC::LocalTensor<float> stackBuffer;
            bool ok = AscendC::PopStackBuffer<float, AscendC::TPosition::LCM>(stackBuffer);
            (void)ok;
            stackBufferSize = stackBuffer.GetSize();
        }

        uint32_t inputShape[4] = {n_, c_, h_, w_};
        AscendC::ShapeInfo shapeInfo{ (uint8_t)4, inputShape, (uint8_t)4, inputShape, AscendC::DataFormat::ND };
        AscendC::GroupNormTiling groupNormTiling;

        AscendC::GetGroupNormNDTillingInfo(shapeInfo, stackBufferSize, sizeof(dataType), false, g_, groupNormTiling);

        AscendC::GroupNorm<dataType, false>(yLocal, meanLocal, varianceLocal, xLocal, gammaLocal, betaLocal, (dataType)epsilon_, groupNormTiling);

        outQueueY.EnQue(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueGamma.FreeTensor(gammaLocal);
        inQueueBeta.FreeTensor(betaLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<dataType> yLocal = outQueueY.DeQue<dataType>();
        uint32_t hwAlignSize = (sizeof(dataType) * h_ * w_ + ONE_BLK_SIZE - 1) / ONE_BLK_SIZE * ONE_BLK_SIZE / sizeof(dataType);
        AscendC::DataCopyParams copyParamsY{static_cast<uint16_t>(n_ * c_), static_cast<uint16_t>(h_ * w_ * sizeof(dataType)), 0, 0};
        AscendC::DataCopyPadParams padParamsY{false, 0, static_cast<uint8_t>(hwAlignSize - h_ * w_), 0};
        AscendC::DataCopyPad(yGlobal, yLocal, copyParamsY, padParamsY);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueGamma;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueBeta;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;
    AscendC::TBuf<AscendC::QuePosition::VECCALC> meanBuffer, varBuffer;

    AscendC::GlobalTensor<dataType> xGlobal;
    AscendC::GlobalTensor<dataType> gammaGlobal;
    AscendC::GlobalTensor<dataType> betaGlobal;
    AscendC::GlobalTensor<dataType> yGlobal;

    uint32_t n_;
    uint32_t c_;
    uint32_t h_;
    uint32_t w_;
    uint32_t g_;
    dataType epsilon_;
};

extern "C" __global__ __aicore__ void instance_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelInstanceNorm op;
    op.Init(x, gamma, beta, y,
            tilingData.n(), tilingData.c(), tilingData.h(), tilingData.w(),
            tilingData.g(), tilingData.epsilon());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义 AscendC InstanceNormCustom 算子替换 PyTorch InstanceNorm2d。
    每通道分组，等价于 InstanceNorm2d(affine=False)；可通过 gamma/beta 调整为 affine=True。
    """
    def __init__(self, num_features: int):
        super().__init__()
        # 等价于 affine=True 的参数（若需要可设置为可训练）
        self.register_buffer("gamma", torch.ones(num_features, dtype=torch.float32))
        self.register_buffer("beta", torch.zeros(num_features, dtype=torch.float32))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x: [N, C, H, W]; gamma/beta: [C]
        # 广播到设备
        gamma = self.gamma.to(x.device, dtype=x.dtype)
        beta = self.beta.to(x.device, dtype=x.dtype)
        return custom_ops_lib.instance_norm_custom(x, gamma, beta)
'''
