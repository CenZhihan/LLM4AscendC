#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = half;     // x
using B_T = half;     // w
using C_T = float;    // y

extern "C" __global__ __aicore__ void matmul_batch_norm_bias_add_divide_swish_custom(
    GM_ADDR xGM,
    GM_ADDR wGM,
    GM_ADDR biasLinGM,
    GM_ADDR bnGammaGM,
    GM_ADDR bnBetaGM,
    GM_ADDR bnMeanGM,
    GM_ADDR bnInvStdGM,
    GM_ADDR addBiasGM,   // length N vector
    GM_ADDR divideGM,    // length N vector
    GM_ADDR yGM,
    GM_ADDR workspaceGM,
    GM_ADDR tilingGM)
{
    // 读取TCubeTiling
    TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // 全局Tensor封装
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;
    AscendC::GlobalTensor<C_T> biasGlobal;

    // NORMAL布局总尺寸估算（参考文档示例）
    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(C_T);
    int32_t sizeBias = tiling.CLayoutInfoB * tiling.singleCoreN * sizeof(C_T);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(xGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(wGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(yGM), sizeC);
    biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(biasLinGM), sizeBias);

    // 每核子块视图
    int offset_a = 0, offset_b = 0, offset_c = 0, offset_bias = 0;
    AscendC::GlobalTensor<A_T> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal[offset_a].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreK);
    AscendC::GlobalTensor<B_T> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal[offset_b].GetPhyAddr()),
                         tiling.singleCoreK * tiling.singleCoreN);
    AscendC::GlobalTensor<C_T> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal[offset_c].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreN);
    AscendC::GlobalTensor<C_T> gm_biasLin;
    gm_biasLin.SetGlobalBuffer(const_cast<__gm__ C_T*>(biasGlobal[offset_bias].GetPhyAddr()),
                               tiling.singleCoreN);

    // Matmul实例
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType;  // B转置
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T> biasType;

    AscendC::TPipe pipe;
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);
    mm1.Init(&tiling);
    mm1.SetTensorA(gm_a, 0);  // A不转置
    mm1.SetTensorB(gm_b, 1);  // B转置
    mm1.SetWorkspace(workspaceGM, 0);
    mm1.SetBias(gm_biasLin);
    // 单次BMM
    mm1.IterateBatch(gm_c, 1, 1, false);

    // BN + bias_add + divide + swish 后处理（在gm_c上原地进行）
    // bn参数和后处理向量（每核N维）
    AscendC::GlobalTensor<C_T> gammaGlobal, betaGlobal, meanGlobal, invstdGlobal, addBiasGlobal, divideGlobal;
    gammaGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(bnGammaGM), tiling.CLayoutInfoB * tiling.singleCoreN);
    betaGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(bnBetaGM), tiling.CLayoutInfoB * tiling.singleCoreN);
    meanGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(bnMeanGM), tiling.CLayoutInfoB * tiling.singleCoreN);
    invstdGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(bnInvStdGM), tiling.CLayoutInfoB * tiling.singleCoreN);
    addBiasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(addBiasGM), tiling.CLayoutInfoB * tiling.singleCoreN);
    divideGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(divideGM), tiling.CLayoutInfoB * tiling.singleCoreN);

    AscendC::GlobalTensor<C_T> gm_gamma, gm_beta, gm_mean, gm_invstd, gm_addBias, gm_divide;
    gm_gamma.SetGlobalBuffer(const_cast<__gm__ C_T*>(gammaGlobal[0].GetPhyAddr()), tiling.singleCoreN);
    gm_beta.SetGlobalBuffer(const_cast<__gm__ C_T*>(betaGlobal[0].GetPhyAddr()), tiling.singleCoreN);
    gm_mean.SetGlobalBuffer(const_cast<__gm__ C_T*>(meanGlobal[0].GetPhyAddr()), tiling.singleCoreN);
    gm_invstd.SetGlobalBuffer(const_cast<__gm__ C_T*>(invstdGlobal[0].GetPhyAddr()), tiling.singleCoreN);
    gm_addBias.SetGlobalBuffer(const_cast<__gm__ C_T*>(addBiasGlobal[0].GetPhyAddr()), tiling.singleCoreN);
    gm_divide.SetGlobalBuffer(const_cast<__gm__ C_T*>(divideGlobal[0].GetPhyAddr()), tiling.singleCoreN);

    // 分配UB缓冲：1行数据 + 需要的参数向量 + 临时sigmoid
    const uint32_t N = tiling.singleCoreN;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qParamGamma, qParamBeta, qParamMean, qParamInvStd, qParamAddB, qParamDiv;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qRowIn;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> qRowOut;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qTmpSigmoid;

    pipe.InitBuffer(qParamGamma, 1, N * sizeof(C_T));
    pipe.InitBuffer(qParamBeta,  1, N * sizeof(C_T));
    pipe.InitBuffer(qParamMean,  1, N * sizeof(C_T));
    pipe.InitBuffer(qParamInvStd,1, N * sizeof(C_T));
    pipe.InitBuffer(qParamAddB,  1, N * sizeof(C_T));
    pipe.InitBuffer(qParamDiv,   1, N * sizeof(C_T));
    pipe.InitBuffer(qRowIn,      1, N * sizeof(C_T));
    pipe.InitBuffer(qRowOut,     1, N * sizeof(C_T));
    pipe.InitBuffer(qTmpSigmoid, 1, N * sizeof(C_T));

    // 载入参数向量到UB
    AscendC::LocalTensor<C_T> gammaL = qParamGamma.AllocTensor<C_T>();
    AscendC::LocalTensor<C_T> betaL  = qParamBeta.AllocTensor<C_T>();
    AscendC::LocalTensor<C_T> meanL  = qParamMean.AllocTensor<C_T>();
    AscendC::LocalTensor<C_T> invstdL= qParamInvStd.AllocTensor<C_T>();
    AscendC::LocalTensor<C_T> addBL  = qParamAddB.AllocTensor<C_T>();
    AscendC::LocalTensor<C_T> divL   = qParamDiv.AllocTensor<C_T>();

    AscendC::DataCopy(gammaL, gm_gamma, N);
    AscendC::DataCopy(betaL,  gm_beta,  N);
    AscendC::DataCopy(meanL,  gm_mean,  N);
    AscendC::DataCopy(invstdL,gm_invstd,N);
    AscendC::DataCopy(addBL,  gm_addBias,N);
    AscendC::DataCopy(divL,   gm_divide, N);

    qParamGamma.EnQue(gammaL);
    qParamBeta.EnQue(betaL);
    qParamMean.EnQue(meanL);
    qParamInvStd.EnQue(invstdL);
    qParamAddB.EnQue(addBL);
    qParamDiv.EnQue(divL);

    gammaL = qParamGamma.DeQue<C_T>();
    betaL  = qParamBeta.DeQue<C_T>();
    meanL  = qParamMean.DeQue<C_T>();
    invstdL= qParamInvStd.DeQue<C_T>();
    addBL  = qParamAddB.DeQue<C_T>();
    divL   = qParamDiv.DeQue<C_T>();

    // 按行处理
    for (uint32_t r = 0; r < (uint32_t)tiling.singleCoreM; ++r) {
        AscendC::LocalTensor<C_T> rowL = qRowIn.AllocTensor<C_T>();
        AscendC::DataCopy(rowL, gm_c[r * N], N);

        // row = ((row - mean) * invstd) * gamma + beta
        AscendC::Sub(rowL, rowL, meanL, N);
        AscendC::Mul(rowL, rowL, invstdL, N);
        AscendC::Mul(rowL, rowL, gammaL, N);
        AscendC::Add(rowL, rowL, betaL, N);

        // + add_bias
        AscendC::Add(rowL, rowL, addBL, N);
        // / divide
        AscendC::Div(rowL, rowL, divL, N);

        // swish: row = row * sigmoid(row)
        AscendC::LocalTensor<C_T> sigL = qTmpSigmoid.AllocTensor<C_T>();
        AscendC::Sigmoid(sigL, rowL, N);
        AscendC::Mul(rowL, rowL, sigL, N);
        qTmpSigmoid.FreeTensor(sigL);

        AscendC::LocalTensor<C_T> outL = qRowOut.AllocTensor<C_T>();
        AscendC::DataCopy(outL, rowL, N);
        qRowIn.FreeTensor(rowL);

        AscendC::DataCopy(gm_c[r * N], outL, N);
        qRowOut.FreeTensor(outL);
    }

    // 释放参数局部tensor
    qParamGamma.FreeTensor(gammaL);
    qParamBeta.FreeTensor(betaL);
    qParamMean.FreeTensor(meanL);
    qParamInvStd.FreeTensor(invstdL);
    qParamAddB.FreeTensor(addBL);
    qParamDiv.FreeTensor(divL);
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused: Matmul + BatchNorm (inference) + bias add + divide + swish.
    Expect:
      - x: [M, in_features] float or half
      - matmul.weight: [out_features, in_features]
      - matmul.bias: [out_features]
      - bn.{weight,bias,running_mean,running_var}
      - bias: (1,) broadcast to out_features
      - divide_value: scalar
    """
    def __init__(self, in_features, out_features, bn_eps=1e-5, bn_momentum=0.1, bias_shape=(1,), divide_value=1.0):
        super().__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.bn = torch.nn.BatchNorm1d(out_features, eps=bn_eps, momentum=bn_momentum)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.divide_value = divide_value

    def forward(self, x: torch.Tensor):
        # 设备与dtype准备
        device = x.device
        M = x.size(0)
        N = self.matmul.out_features

        # half输入以提升Matmul性能
        x_half = x.to(torch.float16)
        w_half = self.matmul.weight.to(torch.float16)

        # 线性bias（N）, BN参数（N, float）
        bias_linear = self.matmul.bias.to(torch.float32) if self.matmul.bias is not None else torch.zeros(N, dtype=torch.float32, device=device)
        bn_gamma = self.bn.weight.to(torch.float32)
        bn_beta  = self.bn.bias.to(torch.float32)
        # 使用推理用running统计量
        bn_mean  = self.bn.running_mean.to(torch.float32)
        bn_var   = self.bn.running_var.to(torch.float32)
        # 预先计算 invstd = 1 / sqrt(var + eps)
        bn_invstd = torch.rsqrt(bn_var + self.bn.eps)

        # 广播bias(1,)与divide标量至长度N的向量，简化内核实现
        add_bias_vec = self.bias.expand(N).contiguous().to(torch.float32)
        divide_vec   = torch.full((N,), float(self.divide_value), dtype=torch.float32, device=device)

        # 自定义算子
        y = custom_ops_lib.matmul_batch_norm_bias_add_divide_swish_custom(
            x_half, w_half, bias_linear, bn_gamma, bn_beta, bn_mean, bn_invstd, add_bias_vec, divide_vec
        )
        return y

def get_inputs():
    batch_size = 1024
    in_features = 8192
    return [torch.rand(batch_size, in_features)]

def get_init_inputs():
    batch_size = 1024
    in_features = 8192
    out_features = 8192
    bn_eps = 1e-5
    bn_momentum = 0.1
    bias_shape = (1,)
    divide_value = 1.0
    return [in_features, out_features, bn_eps, bn_momentum, bias_shape, divide_value]
'''
