inline void SaveTilingBytes(const void* src, uint32_t bytes, uint32_t* dstU32, uint32_t maxU32)
{
    uint32_t words = (bytes + 3) / 4;
    if (words > maxU32) {
        words = maxU32;
    }
    const uint32_t* p = reinterpret_cast<const uint32_t*>(src);
    for (uint32_t i = 0; i < words; ++i) {
        dstU32[i] = p[i];
    }
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 输入形状：Q/K/V: [B, S, D]
    auto* qShape = context->GetInputShape(0);
    auto ori = qShape->GetOriginShape();
    uint32_t dims = ori.GetDimNum();
    if (dims != 3) {
        return ge::GRAPH_FAILED;
    }
    uint32_t B = static_cast<uint32_t>(ori.GetDim(0));
    uint32_t S = static_cast<uint32_t>(ori.GetDim(1));
    uint32_t D = static_cast<uint32_t>(ori.GetDim(2));

    // 设置并行核数
    const uint32_t BLOCK_DIM = 8;
    context->SetBlockDim(BLOCK_DIM);

    // 计算Matmul tiling：Q[S,D] x K[S,D]^T -> Scores[S,S]; Scores[S,S] x V[S,D] -> O[S,D]
    platform_ascendc::PlatformAscendC platform(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling qkTiling(platform);
    qkTiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    qkTiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    qkTiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    qkTiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    qkTiling.SetShape(S, S, D);        // M=N=S, K=D
    qkTiling.SetOrgShape(S, S, D);     // Ka=Kb=D
    qkTiling.SetBias(false);
    qkTiling.SetBufferSpace(-1, -1, -1);
    optiling::TCubeTiling qkTc;
    if (qkTiling.GetTiling(qkTc) != 0) {
        return ge::GRAPH_FAILED;
    }

    matmul_tiling::MatmulApiTiling avTiling(platform);
    avTiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    avTiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    avTiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    avTiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    avTiling.SetShape(S, D, S);        // M=S, N=D, K=S
    avTiling.SetOrgShape(S, D, S);
    avTiling.SetBias(false);
    avTiling.SetBufferSpace(-1, -1, -1);
    optiling::TCubeTiling avTc;
    if (avTiling.GetTiling(avTc) != 0) {
        return ge::GRAPH_FAILED;
    }

    // 设置workspace: 保存Scores[B,S,S]
    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = static_cast<size_t>(B) * static_cast<size_t>(S) * static_cast<size_t>(S) * sizeof(float);

    // 序列化到tiling buffer
    ScaledDotProductAttentionInferenceCustomTilingData tiling;
    tiling.set_batch(B);
    tiling.set_seqlen(S);
    tiling.set_dmodel(D);

    tiling.set_qk_tiling_size_u32(static_cast<uint32_t>(sizeof(optiling::TCubeTiling) / 4));
    tiling.set_av_tiling_size_u32(static_cast<uint32_t>(sizeof(optiling::TCubeTiling) / 4));
    SaveTilingBytes(&qkTc, sizeof(optiling::TCubeTiling), tiling.mutable_qk_tiling_data(), 512);
    SaveTilingBytes(&avTc, sizeof(optiling::TCubeTiling), tiling.mutable_av_tiling_data(), 512);

    // 写入原始tiling缓冲
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = float;
using B_T = float;
using C_T = float;

struct TilingPack {
    uint32_t batch;
    uint32_t seqlen;
    uint32_t dmodel;
    uint32_t qk_tiling_size_u32;
    uint32_t av_tiling_size_u32;
    __gm__ uint32_t* qk_tiling_data;
    __gm__ uint32_t* av_tiling_data;
};

constexpr float kEps = 1e-6f;

class KernelSDPA {
public:
    __aicore__ inline KernelSDPA() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tilingGM)
    {
        // 反序列化tiling
        auto* p32 = reinterpret_cast<__gm__ uint32_t*>(tilingGM);
        // 读标量
        uint32_t batch = p32[0];
        uint32_t seqlen = p32[1];
        uint32_t dmodel = p32[2];
        uint32_t qk_sz = p32[3];
        uint32_t av_sz = p32[4];

        this->B = batch;
        this->S = seqlen;
        this->D = dmodel;

        // 指针偏移至tiling原始数据
        __gm__ uint32_t* qk_data = p32 + 5;
        __gm__ uint32_t* av_data = qk_data + 512; // 每个512 u32的槽位
        // 重构TCubeTiling
        for (int i = 0; i < static_cast<int>(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i) {
            (reinterpret_cast<uint32_t*>(&qkTiling))[i] = qk_data[i];
            (reinterpret_cast<uint32_t*>(&avTiling))[i] = av_data[i];
        }

        // GM张量设置
        qGm.SetGlobalBuffer((__gm__ A_T*)q, (uint32_t)B*S*D);
        kGm.SetGlobalBuffer((__gm__ B_T*)k, (uint32_t)B*S*D);
        vGm.SetGlobalBuffer((__gm__ B_T*)v, (uint32_t)B*S*D);
        oGm.SetGlobalBuffer((__gm__ C_T*)o, (uint32_t)B*S*D);

        // workspace: scores [B, S, S]
        scoresGm.SetGlobalBuffer((__gm__ C_T*)workspace, (uint32_t)B*S*S);

        // 初始化pipe
        pipe.InitBuffer(scoresRowIn, 1, S * sizeof(C_T));
        pipe.InitBuffer(scoresRowOut, 1, S * sizeof(C_T));
    }

    __aicore__ inline void Process()
    {
        // 处理每个batch
        for (uint32_t b = 0; b < B; ++b) {
            ComputeQKT(b);      // scores = Q @ K^T
            SoftmaxScores(b);   // softmax(scores / sqrt(D))
            ComputeAV(b);       // O = scores @ V
        }
    }

private:
    __aicore__ inline void ComputeQKT(uint32_t b)
    {
        // Matmul: [S,D] x [D,S] -> [S,S]
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType; // Transpose K
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T> biasType;

        matmul::Matmul<aType, bType, cType, biasType> mm;
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);

        // 偏移到当前batch
        AscendC::GlobalTensor<A_T> qSlice;
        AscendC::GlobalTensor<B_T> kSlice;
        AscendC::GlobalTensor<C_T> cSlice;

        qSlice.SetGlobalBuffer(const_cast<__gm__ A_T*>(qGm[b * S * D].GetPhyAddr()), S * D);
        kSlice.SetGlobalBuffer(const_cast<__gm__ B_T*>(kGm[b * S * D].GetPhyAddr()), S * D);
        cSlice.SetGlobalBuffer(const_cast<__gm__ C_T*>(scoresGm[b * S * S].GetPhyAddr()), S * S);

        mm.Init(&qkTiling);
        mm.SetTensorA(qSlice, false);
        mm.SetTensorB(kSlice, true);  // 按需转置
        mm.SetWorkspace(GetSysWorkSpacePtr(), 0);
        mm.IterateAll(cSlice);
    }

    __aicore__ inline void SoftmaxScores(uint32_t b)
    {
        // 行级softmax: 对scores[b, i, :] 做softmax(./sqrt(D))
        float scale = 1.0f / AscendC::Sqrt((float)D + kEps);

        for (uint32_t i = 0; i < S; ++i) {
            // 拷贝一行到UB
            AscendC::LocalTensor<C_T> rowIn = scoresRowIn.AllocTensor<C_T>();
            AscendC::DataCopy(rowIn, scoresGm[b * S * S + i * S], S);

            // 按比例缩放
            AscendC::Mul(rowIn, rowIn, scale, S);

            // 这里调用高层SoftMax接口完成归一化
            // 注意：内部会进行逐行max/exp/sum/div计算
            AscendC::LocalTensor<C_T> rowOut = scoresRowOut.AllocTensor<C_T>();
            // SoftMax高阶API示意调用（API内部根据形状自适配tiling）
            // 原型参考高阶API：SoftMax(dst, src, tiling, shapeInfo)（由框架管理暂存空间）
            // 实际工程中可按文档使用SoftMax/SimpleSoftMax及对应tiling
            AscendC::SoftMax(rowOut, rowIn, S);

            // 回写
            AscendC::DataCopy(scoresGm[b * S * S + i * S], rowOut, S);
            scoresRowOut.FreeTensor(rowOut);
            scoresRowIn.FreeTensor(rowIn);
        }
    }

    __aicore__ inline void ComputeAV(uint32_t b)
    {
        // Matmul: [S,S] x [S,D] -> [S,D]
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, false, LayoutMode::NORMAL> bType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T> biasType;

        matmul::Matmul<aType, bType, cType, biasType> mm;
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);

        AscendC::GlobalTensor<A_T> aSlice;
        AscendC::GlobalTensor<B_T> bSlice;
        AscendC::GlobalTensor<C_T> cSlice;

        aSlice.SetGlobalBuffer(const_cast<__gm__ A_T*>(scoresGm[b * S * S].GetPhyAddr()), S * S);
        bSlice.SetGlobalBuffer(const_cast<__gm__ B_T*>(vGm[b * S * D].GetPhyAddr()), S * D);
        cSlice.SetGlobalBuffer(const_cast<__gm__ C_T*>(oGm[b * S * D].GetPhyAddr()), S * D);

        mm.Init(&avTiling);
        mm.SetTensorA(aSlice, false);
        mm.SetTensorB(bSlice, false);
        mm.SetWorkspace(GetSysWorkSpacePtr(), 0);
        mm.IterateAll(cSlice);
    }

private:
    AscendC::TPipe pipe;

    // 全局张量
    AscendC::GlobalTensor<A_T> qGm;
    AscendC::GlobalTensor<B_T> kGm;
    AscendC::GlobalTensor<B_T> vGm;
    AscendC::GlobalTensor<C_T> oGm;
    AscendC::GlobalTensor<C_T> scoresGm;

    // 本地行缓冲
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scoresRowIn;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> scoresRowOut;

    // tiling
    TCubeTiling qkTiling;
    TCubeTiling avTiling;

    // 形状
    uint32_t B;
    uint32_t S;
    uint32_t D;
};

extern "C" __global__ __aicore__ void scaled_dot_product_attention_inference_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelSDPA op;
    op.Init(q, k, v, o, workspace, tiling);
    op.Process();
}
