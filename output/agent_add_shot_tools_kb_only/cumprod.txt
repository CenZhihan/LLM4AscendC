#include "cumprod_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CumprodCustomTilingData tiling;

    const gert::StorageShape* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();

    int64_t dim = 0;
    if (auto attrs = context->GetAttrs(); attrs != nullptr && attrs->GetSize() > 0) {
        dim = *(attrs->GetAttrPointer<int64_t>(0));
    }
    int64_t rank = origin.GetDimNum();
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    uint64_t total = origin.GetShapeSize();
    uint32_t inner = static_cast<uint32_t>(origin.GetDim(dim));
    uint32_t outter = static_cast<uint32_t>(total / inner);

    // 简化：固定每次处理1行，保证UB使用稳定
    const uint32_t TILE_ROWS = 1;

    tiling.set_outter(outter);
    tiling.set_inner(inner);
    tiling.set_tileRows(TILE_ROWS);

    // block并行：按行分块
    context->SetBlockDim(32);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {

    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 双缓冲

class KernelCumprod {
public:
    __aicore__ inline KernelCumprod() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outter, uint32_t inner, uint32_t tileRows)
    {
        this->outter = outter;
        this->inner = inner;
        this->tileRows = tileRows;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowStart = static_cast<uint32_t>((static_cast<uint64_t>(outter) * blockIdx) / blockNum);
        uint32_t rowEnd = static_cast<uint32_t>((static_cast<uint64_t>(outter) * (blockIdx + 1)) / blockNum);
        this->rowCount = rowEnd - this->rowStart;

        xGm.SetGlobalBuffer((__gm__ float*)x + static_cast<uint64_t>(rowStart) * inner, static_cast<uint64_t>(rowCount) * inner);
        yGm.SetGlobalBuffer((__gm__ float*)y + static_cast<uint64_t>(rowStart) * inner, static_cast<uint64_t>(rowCount) * inner);

        // 每次处理 tileRows 行
        this->tileElem = tileRows * inner;

        // 输入/输出队列
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileElem * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileElem * sizeof(float));
        // 计算中间缓冲(用于log与cumsum的临时空间)
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileElem * sizeof(float));
        // 存放CumSum的lastRow（即使不输出，也需提供有效缓冲）
        pipe.InitBuffer(lastRowQueue, 1, inner * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rowCount == 0) {
            return;
        }
        uint32_t tiles = (rowCount + tileRows - 1) / tileRows;
        for (uint32_t t = 0; t < tiles; ++t) {
            CopyIn(t);
            Compute(t);
            CopyOut(t);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t t)
    {
        uint32_t rowsThisTile = GetRowsThisTile(t);
        uint32_t elem = rowsThisTile * inner;

        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[t * tileRows * inner], elem);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t t)
    {
        uint32_t rowsThisTile = GetRowsThisTile(t);
        uint32_t elem = rowsThisTile * inner;

        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();    // 存放log与cumsum结果
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        AscendC::LocalTensor<float> lastRowLocal = lastRowQueue.AllocTensor<float>();

        // 逐元素log
        AscendC::Log(tmpLocal, xLocal, elem);

        // 沿最后一维执行累加（log域），配置：最后轴、复用源、无需输出lastRow
        static constexpr AscendC::CumSumConfig cfg{true, true, false};
        const AscendC::CumSumInfo info{rowsThisTile, inner};
        AscendC::CumSum<float, cfg>(tmpLocal, lastRowLocal, tmpLocal, info);

        // 逐元素exp，得到累乘结果
        AscendC::Exp(yLocal, tmpLocal, elem);

        outQueueY.EnQue<float>(yLocal);
        lastRowQueue.EnQue<float>(lastRowLocal);
        inQueueX.FreeTensor(xLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(uint32_t t)
    {
        uint32_t rowsThisTile = GetRowsThisTile(t);
        uint32_t elem = rowsThisTile * inner;

        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[t * tileRows * inner], yLocal, elem);
        outQueueY.FreeTensor(yLocal);
    }

    __aicore__ inline uint32_t GetRowsThisTile(uint32_t t) const
    {
        uint32_t start = t * tileRows;
        uint32_t remain = (start + tileRows <= rowCount) ? tileRows : (rowCount - start);
        return remain;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> lastRowQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t outter;
    uint32_t inner;
    uint32_t tileRows;

    uint32_t rowStart;
    uint32_t rowCount;
    uint32_t tileElem;
};

extern "C" __global__ __aicore__ void cumprod_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumprod op;
    op.Init(x, y, tiling_data.outter, tiling_data.inner, tiling_data.tileRows);
    op.Process();
}
"""
