    auto padding_attr  = context->GetAttr("padding");
    auto dilation_attr = context->GetAttr("dilation");

    int64_t padding  = (padding_attr == nullptr) ? 0 : padding_attr->GetInt();
    int64_t dilation = (dilation_attr == nullptr) ? 1 : dilation_attr->GetInt();

    y_shape->SetDim(1, c);
    y_shape->SetDim(2, hout);
    y_shape->SetDim(3, wout);

    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

#define DTYPE_X float
#define DTYPE_W float
#define DTYPE_B float
#define DTYPE_Y float

constexpr int32_t BUFFER_NUM = 1;

class KernelDepthwiseConv {
public:
    __aicore__ inline KernelDepthwiseConv() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t kh, uint32_t kw, uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t hout, uint32_t wout, uint32_t hasBias, uint32_t tileW)
    {
        this->n = n; this->c = c; this->h = h; this->w = w_in;
        this->kh = kh; this->kw = kw; this->stride = stride; this->padding = padding; this->dilation = dilation;
        this->hout = hout; this->wout = wout; this->hasBias = hasBias; this->tileW = tileW;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, n * c * h * w_in);
        // w shape is (c, 1, kh, 1) -> contiguous size c * kh
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, c * kh);
        if (hasBias) {
            bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, c);
        }
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, n * c * hout * wout);

        pipe.InitBuffer(inQueueW, BUFFER_NUM, kh * sizeof(DTYPE_W));
        pipe.InitBuffer(inQueueBias, BUFFER_NUM, sizeof(DTYPE_B));
        pipe.InitBuffer(inQueueTmpX, BUFFER_NUM, sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileW * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t ni = 0; ni < n; ++ni) {
            for (uint32_t ci = 0; ci < c; ++ci) {
                // Load weights of channel ci
                AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.AllocTensor<DTYPE_W>();
                AscendC::DataCopy(wLocal, wGm[ci * kh], kh);

                // Load bias if present
                DTYPE_B bval = (DTYPE_B)0;
                if (hasBias) {
                    AscendC::LocalTensor<DTYPE_B> bLocal = inQueueBias.AllocTensor<DTYPE_B>();
                    AscendC::DataCopy(bLocal, bGm[ci], 1);
                    bval = bLocal.GetValue(0);
                    inQueueBias.FreeTensor(bLocal);
                }

                for (uint32_t oh = 0; oh < hout; ++oh) {
                    for (uint32_t ow_tile = 0; ow_tile < wout; ow_tile += tileW) {
                        uint32_t curTileW = AscendC::Min(tileW, wout - ow_tile);
                        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
                        // Initialize yLocal if needed
                        for (uint32_t t = 0; t < curTileW; ++t) {
                            yLocal.SetValue(t, (DTYPE_Y)0);
                        }

                        for (uint32_t t = 0; t < curTileW; ++t) {
                            uint32_t ow = ow_tile + t;
                            int32_t w_in_idx = static_cast<int32_t>(ow * stride) - static_cast<int32_t>(padding);
                            DTYPE_Y acc = (DTYPE_Y)bval;

                            if (w_in_idx >= 0 && w_in_idx < static_cast<int32_t>(w)) {
                                for (uint32_t kh_i = 0; kh_i < kh; ++kh_i) {
                                    int32_t h_in_idx = static_cast<int32_t>(oh * stride) - static_cast<int32_t>(padding) + static_cast<int32_t>(kh_i * dilation);
                                    if (h_in_idx >= 0 && h_in_idx < static_cast<int32_t>(h)) {
                                        // x offset: ((n*C + c)*H + h_in_idx)*W + w_in_idx
                                        uint64_t x_offset = (((uint64_t)ni * c + ci) * h + (uint64_t)h_in_idx) * w + (uint64_t)w_in_idx;
                                        AscendC::LocalTensor<DTYPE_X> xTmp = inQueueTmpX.AllocTensor<DTYPE_X>();
                                        AscendC::DataCopy(xTmp, xGm[x_offset], 1);
                                        DTYPE_X xv = xTmp.GetValue(0);
                                        inQueueTmpX.FreeTensor(xTmp);

                                        DTYPE_W wv = wLocal.GetValue(kh_i);
                                        acc = (DTYPE_Y)((float)acc + (float)xv * (float)wv);
                                    }
                                }
                            }
                            yLocal.SetValue(t, acc);
                        }

                        // y offset base for current (ni, ci, oh, ow_tile)
                        uint64_t y_offset = (((uint64_t)ni * c + ci) * hout + (uint64_t)oh) * wout + (uint64_t)ow_tile;
                        AscendC::DataCopy(yGm[y_offset], yLocal, curTileW);
                        outQueueY.FreeTensor(yLocal);
                    }
                }

                inQueueW.FreeTensor(wLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBias;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTmpX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, h, w;
    uint32_t kh, kw;
    uint32_t stride, padding, dilation;
    uint32_t hout, wout;
    uint32_t hasBias;
    uint32_t tileW;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelDepthwiseConv op;
    op.Init(x, w, b, y,
            tiling_data.n(), tiling_data.c(), tiling_data.h(), tiling_data.w(),
            tiling_data.kh(), tiling_data.kw(), tiling_data.stride(), tiling_data.padding(), tiling_data.dilation(),
            tiling_data.hout(), tiling_data.wout(), tiling_data.hasBias(), tiling_data.tileW());
    op.Process();
}

#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv_dw2d_asym_impl_npu(const at::Tensor& x,
                                   const at::Tensor& w,
                                   c10::optional<at::Tensor> b,
                                   int64_t stride,
                                   int64_t padding,
                                   int64_t dilation) {
    auto x_dtype = x.scalar_type();
    TORCH_CHECK(x_dtype == at::kFloat, \"Only float32 is supported\");

    // Infer output shape using PyTorch formula for convenience
    auto N = x.size(0);
    auto C = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);

    auto KH = w.size(2);
    auto KW = w.size(3);
    TORCH_CHECK(KW == 1, \"Kernel width must be 1 for asymmetric kernel\");

    int64_t Hout = (H + 2*padding - dilation*(KH - 1) - 1) / stride + 1;
    int64_t Wout = (W + 2*padding - dilation*(KW - 1) - 1) / stride + 1;

    at::Tensor y = at::empty({N, C, Hout, Wout}, x.options());

    if (b.has_value() && b.value().defined()) {
        EXEC_NPU_CMD(aclnnConvDepthwise2dSquareInputAsymmetricKernelCustom, x, w, b.value(), y, stride, padding, dilation);
    } else {
        // Pass an empty tensor as bias (not used). The host tiling will set hasBias=0.
        at::Tensor empty_bias = at::empty({0}, x.options());
        EXEC_NPU_CMD(aclnnConvDepthwise2dSquareInputAsymmetricKernelCustom, x, w, empty_bias, y, stride, padding, dilation);
    }

    return y;
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        # 保留原始Conv2d用于参数管理（权重/偏置），计算改用自定义AscendC算子
        self.conv2d = torch.nn.Conv2d(in_channels, in_channels, kernel_size=(kernel_size, 1),
                                      stride=stride, padding=padding, dilation=dilation,
                                      groups=in_channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        bias_tensor = self.conv2d.bias if self.conv2d.bias is not None else None
        # stride/padding/dilation是int或tuple，这里只支持对称单值
        s = self.conv2d.stride[0] if isinstance(self.conv2d.stride, tuple) else self.conv2d.stride
        p = self.conv2d.padding[0] if isinstance(self.conv2d.padding, tuple) else self.conv2d.padding
        d = self.conv2d.dilation[0] if isinstance(self.conv2d.dilation, tuple) else self.conv2d.dilation
        return custom_ops_lib.conv_depthwise_2d_square_input_asymmetric_kernel_custom(x, self.conv2d.weight, bias_tensor, s, p, d)
