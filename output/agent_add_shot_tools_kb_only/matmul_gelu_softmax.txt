kernel_src="""
#include "kernel_operator.h"

using AType = half;   // x
using BType = half;   // w
using CType = float;  // y

class KernelMatmulGeluSoftmax {
public:
    __aicore__ inline KernelMatmulGeluSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x_gm, GM_ADDR w_gm, GM_ADDR b_gm, GM_ADDR y_gm,
                                uint32_t m_in, uint32_t k_in, uint32_t n_in)
    {
        m = m_in;
        k = k_in;
        n = n_in;

        xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ AType*>(x_gm), m * k);
        wGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ BType*>(w_gm), k * n);
        bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ CType*>(b_gm), n);
        yGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ CType*>(y_gm), m * n);

        // Queues for GEMM
        pipe.InitBuffer(inQueueA1, 1, m * k * sizeof(AType));
        pipe.InitBuffer(inQueueB1, 1, k * n * sizeof(BType));
        pipe.InitBuffer(outQueueCO1, 1, m * n * sizeof(CType));
        // Additional buffer for UB processing
        pipe.InitBuffer(vecTmp, 1, m * n * sizeof(CType));
        pipe.InitBuffer(biasBuf, 1, n * sizeof(CType));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<AType> aLocal = inQueueA1.AllocTensor<AType>();
        AscendC::LocalTensor<BType> bLocal = inQueueB1.AllocTensor<BType>();
        AscendC::DataCopy(aLocal, xGlobal, m * k);
        AscendC::DataCopy(bLocal, wGlobal, k * n);
        inQueueA1.EnQue(aLocal);
        inQueueB1.EnQue(bLocal);

        // Copy bias (length n) into local once
        AscendC::LocalTensor<CType> bLocalRow = biasBuf.AllocTensor<CType>();
        AscendC::DataCopy(bLocalRow, bGlobal, n);
        biasBuf.EnQue(bLocalRow);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<AType> aLocal = inQueueA1.DeQue<AType>();
        AscendC::LocalTensor<BType> bLocal = inQueueB1.DeQue<BType>();
        AscendC::LocalTensor<CType> cLocal = outQueueCO1.AllocTensor<CType>();

        // GEMM: [m,k] x [k,n] -> [m,n], dst is float
        AscendC::GemmTiling tilling = AscendC::GetGemmTiling<AType>(m, k, n);
        // move result to cLocal
        AscendC::Gemm(cLocal, aLocal, bLocal, m, k, n, tilling, false, /*initValue*/0);

        // Add bias per row
        AscendC::LocalTensor<CType> biasLocal = biasBuf.DeQue<CType>();
        // Row-wise add bias
        // Process by contiguous row tiles
        for (uint32_t row = 0; row < m; ++row) {
            // allocate a temp row buffer
            AscendC::LocalTensor<CType> rowTmp = vecTmp.AllocTensor<CType>();
            // copy one row out
            AscendC::DataCopy(rowTmp, cLocal[row * n], n);
            // z = row + bias
            AscendC::Add(rowTmp, rowTmp, biasLocal, n);
            // write back
            AscendC::DataCopy(cLocal[row * n], rowTmp, n);
            vecTmp.FreeTensor(rowTmp);
        }
        biasBuf.FreeTensor(biasLocal);

        // GELU in place
        AscendC::Gelu(cLocal, cLocal, m * n);

        // SoftMax along last axis (n), on [m,n]
        // Use default config; let API check tiling internally
        AscendC::SoftMax(cLocal, cLocal, /*tiling*/AscendC::SoftMaxTiling{}, /*shape*/{});

        outQueueCO1.EnQue<CType>(cLocal);
        inQueueA1.FreeTensor(aLocal);
        inQueueB1.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<CType> cLocal = outQueueCO1.DeQue<CType>();
        AscendC::DataCopy(yGlobal, cLocal, m * n);
        outQueueCO1.FreeTensor(cLocal);
    }

private:
    uint32_t m, k, n;

    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::A1, 1> inQueueA1;
    AscendC::TQue<AscendC::QuePosition::B1, 1> inQueueB1;
    AscendC::TQue<AscendC::QuePosition::CO1, 1> outQueueCO1;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> vecTmp;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> biasBuf;

    AscendC::GlobalTensor<AType> xGlobal;
    AscendC::GlobalTensor<BType> wGlobal;
    AscendC::GlobalTensor<CType> bGlobal;
    AscendC::GlobalTensor<CType> yGlobal;
};

extern "C" __global__ __aicore__ void matmul_gelu_softmax_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                                                 GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulGeluSoftmax op;
    op.Init(x, w, b, y, tiling_data.m, tiling_data.k, tiling_data.n);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features):
        super(ModelNew, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.linear = torch.nn.Linear(in_features, out_features)

    def forward(self, x):
        # Prepare inputs: weight must be [k,n]; bias is [n]
        w_t = self.linear.weight.t().contiguous()
        # Cast to match kernel expectation: x,w in half; bias in float; output float
        return custom_ops_lib.matmul_gelu_softmax_custom(x.to(torch.float16), w_t.to(torch.float16), self.linear.bias.to(torch.float))

def get_inputs():
    batch_size = 1024
    in_features = 8192
    return [torch.rand(batch_size, in_features)]

def get_init_inputs():
    in_features = 8192
    out_features = 8192
    return [in_features, out_features]
