#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline int64_t div_floor(int64_t a, int64_t b) {
    return a / b;
}
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 输入形状: N, C, L
    const gert::Shape *in_shape = context->GetInputShape(0);
    const gert::Shape &ori = in_shape->GetOriginShape();
    int64_t n = ori.GetDimNum() > 0 ? ori.GetDim(0) : 1;
    int64_t c = ori.GetDimNum() > 1 ? ori.GetDim(1) : 1;
    int64_t l = ori.GetDimNum() > 2 ? ori.GetDim(2) : 1;

    // 为演示按常量参数进行tiling，需与模型初始化参数保持一致
    // 也可在实际工程中通过context读取算子属性
    const int64_t KERNEL_SIZE = 8;
    int64_t STRIDE = 1;
    int64_t PADDING = 4;

    if (STRIDE <= 0) STRIDE = KERNEL_SIZE;
    int64_t lOut = 0;
    if (l + 2 * PADDING >= KERNEL_SIZE) {
        lOut = div_floor((l + 2 * PADDING - KERNEL_SIZE), STRIDE) + 1;
    }

    // 简单tiling：每核处理若干通道(N*C维)，每次在L维做定长tile
    const uint32_t MAX_BLOCKS = 32;
    uint32_t totalChannels = static_cast<uint32_t>(n * c);
    uint32_t blockDim = totalChannels < MAX_BLOCKS ? (totalChannels == 0 ? 1 : totalChannels) : MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    uint32_t channelsPerBlock = (totalChannels + blockDim - 1) / blockDim;

    // 固定tile长度(出端)，输入tile需要考虑stride与kernel_size
    uint32_t tileOutLen = 256;
    if (lOut > 0 && static_cast<int64_t>(tileOutLen) > lOut) {
        tileOutLen = static_cast<uint32_t>(lOut);
    }
    uint32_t tileInLen = static_cast<uint32_t>(tileOutLen * STRIDE + (KERNEL_SIZE - 1));

    MaxPooling1dCustomTilingData tiling;
    tiling.set_n(static_cast<uint32_t>(n));
    tiling.set_c(static_cast<uint32_t>(c));
    tiling.set_lIn(static_cast<uint32_t>(l));
    tiling.set_lOut(static_cast<uint32_t>(lOut));
    tiling.set_kernelSize(static_cast<uint32_t>(KERNEL_SIZE));
    tiling.set_stride(static_cast<uint32_t>(STRIDE));
    tiling.set_padding(static_cast<uint32_t>(PADDING));
    tiling.set_channelsPerBlock(channelsPerBlock);
    tiling.set_tileOutLen(tileOutLen);
    tiling.set_tileInLen(tileInLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // 无额外workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    context->SetTilingKey(1);
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    const gert::Shape& ori = in_shape->GetOriginShape();
    int64_t n = ori.GetDim(0);
    int64_t c = ori.GetDim(1);
    int64_t l = ori.GetDim(2);

    // 与上面TilingFunc保持一致的参数
    const int64_t KERNEL_SIZE = 8;
    int64_t STRIDE = 1;
    int64_t PADDING = 4;
    if (STRIDE <= 0) STRIDE = KERNEL_SIZE;
    int64_t lOut = 0;
    if (l + 2 * PADDING >= KERNEL_SIZE) {
        lOut = (l + 2 * PADDING - KERNEL_SIZE) / STRIDE + 1;
    }

    gert::Shape* out_shape = context->GetOutputShape(0);
    std::vector<int64_t> dims = {n, c, lOut};
    out_shape->SetDimNum(3);
    for (size_t i = 0; i < dims.size(); ++i) {
        out_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include <float.h>

using DTYPE = float;

class KernelMaxPool1d {
public:
    __aicore__ inline KernelMaxPool1d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t lIn, uint32_t lOut,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding,
                                uint32_t channelsPerBlock,
                                uint32_t tileOutLen, uint32_t tileInLen)
    {
        this->n = n;
        this->c = c;
        this->lIn = lIn;
        this->lOut = lOut;
        this->kernelSize = kernelSize;
        this->stride = stride;
        this->padding = padding;
        this->channelsPerBlock = channelsPerBlock;
        this->tileOutLen = tileOutLen;
        this->tileInLen = tileInLen;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(x));
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(y));

        // UB队列
        pipe.InitBuffer(inQueue, 1, this->tileInLen * sizeof(DTYPE));
        pipe.InitBuffer(outQueue, 1, this->tileOutLen * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        uint32_t totalChannels = n * c;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // 使用来自tiling的channelsPerBlock以确保与host一致
        uint32_t startCh = blockIdx * channelsPerBlock;
        uint32_t endCh = startCh + channelsPerBlock;
        if (endCh > totalChannels) endCh = totalChannels;

        for (uint32_t ch = startCh; ch < endCh; ++ch) {
            // 每个通道沿L维做滑窗
            for (uint32_t outStart = 0; outStart < lOut; outStart += tileOutLen) {
                uint32_t curOutLen = (outStart + tileOutLen <= lOut) ? tileOutLen : (lOut - outStart);
                if (curOutLen == 0) break;
                uint32_t curInLen = curOutLen * stride + (kernelSize - 1);

                AscendC::LocalTensor<DTYPE> inLocal = inQueue.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> outLocal = outQueue.AllocTensor<DTYPE>();

                // 以 -INF 填充(对齐 PyTorch MaxPool 的padding行为)
                DTYPE negInf = -FLT_MAX;
                AscendC::Duplicate(inLocal, negInf, curInLen);

                // 计算需要从全局内存拷贝的区间
                int64_t inStartIdx = static_cast<int64_t>(outStart) * static_cast<int64_t>(stride) - static_cast<int64_t>(padding);
                uint32_t copySrcStart = 0;
                uint32_t copyDstStart = 0;
                uint32_t copyLen = 0;

                if (inStartIdx < 0) {
                    // 左侧越界，用-Inf占位
                    int64_t leftPad = -inStartIdx;
                    if (leftPad >= static_cast<int64_t>(curInLen)) {
                        // 全部是padding
                        copyLen = 0;
                    } else {
                        copyDstStart = static_cast<uint32_t>(leftPad);
                        copySrcStart = 0;
                        uint32_t remain = curInLen - copyDstStart;
                        uint32_t avail = (lIn < remain) ? lIn : remain;
                        copyLen = avail;
                    }
                } else {
                    // 起点在合法范围内
                    copySrcStart = static_cast<uint32_t>(inStartIdx);
                    if (copySrcStart < lIn) {
                        uint32_t maxAvailable = lIn - copySrcStart;
                        uint32_t need = curInLen;
                        copyLen = (maxAvailable < need) ? maxAvailable : need;
                        copyDstStart = 0;
                    } else {
                        copyLen = 0;
                    }
                }

                if (copyLen > 0) {
                    // 全局基址: ch * lIn
                    uint32_t base = ch * lIn;
                    AscendC::DataCopy(inLocal[copyDstStart], xGm[base + copySrcStart], copyLen);
                }

                // 计算滑窗最大值
                for (uint32_t i = 0; i < curOutLen; ++i) {
                    // 窗口起点
                    uint32_t wStart = i * stride;
                    DTYPE m = inLocal[wStart];
                    for (uint32_t k = 1; k < kernelSize; ++k) {
                        DTYPE v = inLocal[wStart + k];
                        m = v > m ? v : m;
                    }
                    outLocal[i] = m;
                }

                outQueue.EnQue<DTYPE>(outLocal);

                // 写回到全局内存
                AscendC::LocalTensor<DTYPE> outLocalDeq = outQueue.DeQue<DTYPE>();
                uint32_t outBase = ch * lOut + outStart;
                AscendC::DataCopy(yGm[outBase], outLocalDeq, curOutLen);
                outQueue.FreeTensor(outLocalDeq);

                inQueue.FreeTensor(inLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t n, c, lIn, lOut;
    uint32_t kernelSize, stride, padding;
    uint32_t channelsPerBlock;
    uint32_t tileOutLen, tileInLen;
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelMaxPool1d op;
    op.Init(x, y,
            tilingData.n, tilingData.c,
            tilingData.lIn, tilingData.lOut,
            tilingData.kernelSize, tilingData.stride, tilingData.padding,
            tilingData.channelsPerBlock,
            tilingData.tileOutLen, tilingData.tileInLen);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include \"pytorch_npu_helper.hpp\"
#include <torch/extension.h>

at::Tensor max_pooling_1d_custom_impl_npu(const at::Tensor& x, int64_t kernel_size, int64_t stride, int64_t padding, bool return_indices) {
    TORCH_CHECK(x.dim() == 3, \"Input must be 3D (N, C, L)\");
    int64_t N = x.size(0);
    int64_t C = x.size(1);
    int64_t L = x.size(2);
    if (stride <= 0) stride = kernel_size;

    int64_t L_out = 0;
    if (L + 2 * padding >= kernel_size) {
        L_out = (L + 2 * padding - kernel_size) / stride + 1;
    }
    TORCH_CHECK(L_out >= 0, \"Invalid output length computed for MaxPooling1dCustom\");

    at::Tensor y = at::empty({N, C, L_out}, x.options());
    // 不返回索引
    (void)return_indices;

    EXEC_NPU_CMD(aclnnMaxPooling1dCustom, x, (int)kernel_size, (int)stride, (int)padding, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"max_pooling_1d_custom\", &max_pooling_1d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"max_pooling_1d_custom\", &max_pooling_1d_custom_impl_npu, \"MaxPool1d custom op\");
}
