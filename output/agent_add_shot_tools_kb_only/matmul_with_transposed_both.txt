host_operator_src="""
#include "register/op_def_registry.h"
#include "matmul_with_transposed_both_custom_tiling.h"

#include "lib/matmul_tiling_api.h"
#include "lib/platform_ascendc.h"
#include <cstring>

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // A shape (K, M), B shape (N, K). We need output (M, N) for A^T * B^T.
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }
    auto a_origin = a_shape->GetOriginShape();
    auto b_origin = b_shape->GetOriginShape();
    if (a_origin.GetDimNum() != 2 || b_origin.GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    uint32_t K = static_cast<uint32_t>(a_origin.GetDim(0));
    uint32_t M = static_cast<uint32_t>(a_origin.GetDim(1));
    uint32_t N = static_cast<uint32_t>(b_origin.GetDim(0));
    // Build Matmul tiling (NORMAL layout) with transposed A and B at runtime
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling tiling(ascendcPlatform);
    // Set types and mark transpose for A and B
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT, true);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT, true);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBias(false);
    tiling.SetShape(M, N, K);
    tiling.SetOrgShape(M, N, K);
    tiling.SetBufferSpace(-1, -1, -1);
    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }
    // Set block dim
    context->SetBlockDim(tilingData.blockDim > 0 ? tilingData.blockDim : 8);
    // Save raw tiling bytes
    auto raw = context->GetRawTilingData();
    if (raw->GetCapacity() < sizeof(tilingData)) {
        return ge::GRAPH_FAILED;
    }
    std::memcpy(raw->GetData(), &tilingData, sizeof(tilingData));
    raw->SetDataSize(sizeof(tilingData));
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // A: (K, M), B: (N, K) -> C: (M, N) for A^T @ B^T
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return GRAPH_FAILED;
    }
    auto a_origin = a_shape->GetOriginShape();
    auto b_origin = b_shape->GetOriginShape();
    if (a_origin.GetDimNum() != 2 || b_origin.GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    uint64_t M = static_cast<uint64_t>(a_origin.GetDim(1));
    uint64_t N = static_cast<uint64_t>(b_origin.GetDim(0));
    gert::Shape* c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class MatmulWithTransposedBothCustom : public OpDef {
public:
    explicit MatmulWithTransposedBothCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulWithTransposedBothCustom);
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // Load TCubeTiling from GM (host prepared via matmul tiling API)
    TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // Define Matmul types: NORMAL layout, both A and B allow transpose
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    // Global tensors
    AscendC::GlobalTensor<float> aGlobal;
    AscendC::GlobalTensor<float> bGlobal;
    AscendC::GlobalTensor<float> cGlobal;

    // Total sizes (bytes) for GM buffers: A(K,M), B(N,K), C(M,N)
    // Use base/single shapes from tiling (NORMAL mode uses singleCoreM/N/K)
    int32_t sizeA = tiling.singleCoreM * tiling.singleCoreK * sizeof(float);
    int32_t sizeB = tiling.singleCoreK * tiling.singleCoreN * sizeof(float);
    int32_t sizeC = tiling.singleCoreM * tiling.singleCoreN * sizeof(float);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cGM), sizeC);

    // Slice views passed to Matmul
    AscendC::GlobalTensor<float> gm_a;
    AscendC::GlobalTensor<float> gm_b;
    AscendC::GlobalTensor<float> gm_c;

    gm_a.SetGlobalBuffer(const_cast<__gm__ float*>(aGlobal[0].GetPhyAddr()), tiling.singleCoreM * tiling.singleCoreK);
    gm_b.SetGlobalBuffer(const_cast<__gm__ float*>(bGlobal[0].GetPhyAddr()), tiling.singleCoreK * tiling.singleCoreN);
    gm_c.SetGlobalBuffer(const_cast<__gm__ float*>(cGlobal[0].GetPhyAddr()), tiling.singleCoreM * tiling.singleCoreN);

    // Create Matmul instance with NORMAL configuration
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;
    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);

    mm1.Init(&tiling);
    // Set tensors, enabling transpose on both A and B
    mm1.SetTensorA(gm_a, true);  // A is provided as (K, M), transpose -> (M, K)
    mm1.SetTensorB(gm_b, true);  // B is provided as (N, K), transpose -> (K, N)
    mm1.SetWorkspace(workspaceGM, 0);

    // Single matmul (no batches)
    mm1.IterateBatch(gm_c, 1, 1, false);
}
