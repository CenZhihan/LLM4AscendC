kernel_src="""
#include "kernel_operator.h"

// 该内核实现对输入张量(N, C, H, W)在通道维度上取最小值，输出(N, 1, H, W)
// 注意：模型中已将缩放因子融合到卷积权重/偏置中，本内核仅负责通道最小化

constexpr int32_t BUFFER_NUM = 2; // 每个队列的张量数量

class KernelChannelMin {
public:
    __aicore__ inline KernelChannelMin() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t tileHW, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->hw = h * w;
        this->tileNum = tileNum;
        this->tileHW = tileHW;
        // block 维度按 batch 划分
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->nPerBlock = (n + blockNum - 1) / blockNum;
        this->nStart = blockIdx * this->nPerBlock;
        if (this->nStart > n) this->nStart = n;
        this->nEnd = this->nStart + this->nPerBlock;
        if (this->nEnd > n) this->nEnd = n;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x), static_cast<uint32_t>(n) * c * hw);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(z), static_cast<uint32_t>(n) * hw);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileHW * sizeof(float));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileHW * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileHW * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // 遍历分配到本 block 的 batch
        for (uint32_t nb = nStart; nb < nEnd; ++nb) {
            // 针对本 batch 的 H*W 维度分块
            uint32_t tiles = (hw + tileHW - 1) / tileHW;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t curLen = (t == tiles - 1) ? (hw - t * tileHW) : tileHW;
                CopyComputeReduce(nb, t, curLen);
            }
        }
    }

private:
    __aicore__ inline void CopyComputeReduce(uint32_t nb, uint32_t tileIdx, uint32_t curLen)
    {
        uint32_t baseHW = tileIdx * tileHW;
        // 读取第0个通道块，作为初始最小值
        AscendC::LocalTensor<float> accLocal = outQueueZ.AllocTensor<float>();
        AscendC::LocalTensor<float> xLocal0 = inQueueX.AllocTensor<float>();
        uint32_t off0 = ((nb * c + 0) * hw + baseHW);
        AscendC::DataCopy(xLocal0, xGm[off0], curLen);
        // 将初始块拷贝到累积缓冲
        AscendC::DataCopy(accLocal, xLocal0, curLen);
        inQueueX.FreeTensor(xLocal0);

        // 遍历剩余通道，做逐元素最小
        for (uint32_t cc = 1; cc < c; ++cc) {
            AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
            uint32_t off = ((nb * c + cc) * hw + baseHW);
            AscendC::DataCopy(xLocal, xGm[off], curLen);

            AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();
            // tmpLocal = elementwise min(accLocal, xLocal)
            AscendC::Min(tmpLocal, accLocal, xLocal, curLen);
            // 交换：将tmp写回acc
            AscendC::DataCopy(accLocal, tmpLocal, curLen);

            inQueueX.FreeTensor(xLocal);
            tmpQueue.FreeTensor(tmpLocal);
        }

        // 写回输出 (N,1,H,W) 按 batch 的 H*W 展平存储
        uint32_t outOff = (nb * hw + baseHW);
        outQueueZ.EnQue<float>(accLocal);
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm[outOff], zLocal, curLen);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t n, c, h, w, hw;
    uint32_t tileNum, tileHW;
    uint32_t nPerBlock, nStart, nEnd;
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelChannelMin op;
    op.Init(x, z,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.tileHW, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义 AscendC 内核，将通道最小化替换为 conv2d_scaling_min_custom。
    将缩放因子预先融合到卷积权重与偏置中：y = conv2d(x, w*scale, b*scale)
    然后调用自定义算子在通道维上取最小，保持 keepdim=True 输出为 (N,1,H,W)。
    """
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scale_factor = scale_factor

    def forward(self, x):
        # 将缩放因子融合到卷积权重/偏置
        w = self.conv.weight * self.scale_factor
        b = None
        if self.conv.bias is not None:
            b = self.conv.bias * self.scale_factor
        y = F.conv2d(
            x, w, b,
            stride=self.conv.stride,
            padding=self.conv.padding,
            dilation=self.conv.dilation,
            groups=self.conv.groups
        )
        # 自定义通道最小化 (N, C, H, W) -> (N, 1, H, W)
        z = custom_ops_lib.conv2d_scaling_min_custom(y)
        return z
'''
