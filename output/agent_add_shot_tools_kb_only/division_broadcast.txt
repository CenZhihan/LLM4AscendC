#include "division_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    DivisionBroadcastCustomTilingData tiling;
    // 总长度：N*C*H*W
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    // C 来自 scale 的形状 [C,1,1] => 形状大小即为 C
    uint32_t c = context->GetInputShape(1)->GetOriginShape().GetShapeSize();

    // 将 BlockDim 设置为 N，使每个核心处理一个 batch，保证分块不跨越通道边界
    uint32_t n = context->GetInputShape(0)->GetOriginShape().GetDim(0);
    context->SetBlockDim(n);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(1);
    tiling.set_c(c);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* z_shape = context->GetOutputShape(0);
    *z_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelDivisionBroadcast {
public:
    __aicore__ inline KernelDivisionBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR z, uint32_t totalLength, uint32_t c)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum(); // 每核处理 C*H*W
        this->c = c;
        this->hw = this->blockLength / c; // H*W

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_Y *)scale, this->c);

        // 缓冲区
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->blockLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->blockLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpScaleIn, BUFFER_NUM, this->c * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpScaleBCast, BUFFER_NUM, this->blockLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm, this->blockLength);
        inQueueX.EnQue(xLocal);

        AscendC::LocalTensor<DTYPE_Y> scaleLocal = tmpScaleIn.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(scaleLocal, scaleGm, this->c); // scale: [C,1,1] => 读取 C
        tmpScaleIn.EnQue(scaleLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> scaleLocal = tmpScaleIn.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> scaleBCastLocal = tmpScaleBCast.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // 广播：将 [C,1] 广播到 [C,H*W]，随后做逐元素除法
        const uint32_t srcShape[2] = { this->c, 1 };
        const uint32_t dstShape[2] = { this->c, this->hw };
        AscendC::BroadCast<DTYPE_Y, 2, 1>(scaleBCastLocal, scaleLocal, dstShape, srcShape);

        AscendC::Div(zLocal, xLocal, scaleBCastLocal, this->blockLength);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        tmpScaleBCast.EnQue<DTYPE_Y>(scaleBCastLocal);
        inQueueX.FreeTensor(xLocal);
        tmpScaleIn.FreeTensor(scaleLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm, zLocal, this->blockLength);
        outQueueZ.FreeTensor(zLocal);

        AscendC::LocalTensor<DTYPE_Y> scaleBCastLocal = tmpScaleBCast.DeQue<DTYPE_Y>();
        tmpScaleBCast.FreeTensor(scaleBCastLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> tmpScaleIn;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> tmpScaleBCast;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> scaleGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t blockLength;
    uint32_t c;
    uint32_t hw;
};

extern "C" __global__ __aicore__ void division_broadcast_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelDivisionBroadcast op;
    op.Init(x, scale, z, tilingData.totalLength, tilingData.c);
    op.Process();
}
"""
