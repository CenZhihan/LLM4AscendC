kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_Y = float;

class KernelConvTransposed2D {
public:
    __aicore__ inline KernelConvTransposed2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t wi, uint32_t kh, uint32_t kw,
                                uint32_t tileH) {
        this->N = n;
        this->Cin = cin;
        this->Cout = cout;
        this->H = h;
        this->W = wi;
        this->Kh = kh;
        this->Kw = kw;
        this->Hout = H + Kh - 1;
        this->Wout = W + Kw - 1;
        this->tileH = (tileH == 0 || tileH > H) ? H : tileH;

        // batch 切分
        uint32_t blkNum = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();
        this->nBegin = (N * blkIdx) / blkNum;
        this->nEnd   = (N * (blkIdx + 1)) / blkNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint64_t>(N) * Cin * H * W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, static_cast<uint64_t>(Cin) * Cout * Kh * Kw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, static_cast<uint64_t>(N) * Cout * Hout * Wout);

        // 为输入行片和输出行片申请 UB 空间
        // x 行片: tileH * W
        pipe.InitBuffer(inQueueX, 1, static_cast<uint64_t>(tileH) * W * sizeof(DTYPE_X));
        // 临时输出片: (tileH + Kh - 1) * (W + Kw - 1)
        uint64_t yTileSize = static_cast<uint64_t>(tileH + Kh - 1) * (W + Kw - 1);
        pipe.InitBuffer(outQueueY, 1, yTileSize * sizeof(DTYPE_Y));
        // 临时行缓冲
        pipe.InitBuffer(tmpQueue, 1, W * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process() {
        for (uint32_t n = nBegin; n < nEnd; ++n) {
            for (uint32_t co = 0; co < Cout; ++co) {
                // 沿输入高度做分片
                for (uint32_t hs = 0; hs < H; hs += tileH) {
                    uint32_t curTileH = AscendC::Min(tileH, H - hs);
                    ComputeTile(n, co, hs, curTileH);
                }
            }
        }
    }

private:
    __aicore__ inline void ZeroLocal(AscendC::LocalTensor<DTYPE_Y> &buf, uint64_t len) {
        // 使用 tmp 缓冲清零
        AscendC::LocalTensor<DTYPE_Y> tmp = tmpQueue.AllocTensor<DTYPE_Y>();
        // 初始化一行 0
        for (uint32_t i = 0; i < W; ++i) {
            tmp.SetValue(i, static_cast<DTYPE_Y>(0));
        }
        // 分块复制清零
        uint64_t offset = 0;
        const uint64_t step = W;
        while (offset < len) {
            uint64_t cur = AscendC::Min(step, len - offset);
            AscendC::DataCopy(buf[offset], tmp, cur);
            offset += cur;
        }
        tmpQueue.FreeTensor(tmp);
    }

    __aicore__ inline void ComputeTile(uint32_t n, uint32_t co, uint32_t hStart, uint32_t curTileH) {
        // 准备 UB 缓冲
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        uint64_t xTileElems = static_cast<uint64_t>(curTileH) * W;
        // 拷贝输入片 [N, Cin, H, W] 的每个 Cin 后续按通道累加
        // 先把输出片清零
        uint64_t yTileElems = static_cast<uint64_t>(curTileH + Kh - 1) * (W + Kw - 1);
        ZeroLocal(yLocal, yTileElems);

        for (uint32_t ci = 0; ci < Cin; ++ci) {
            // 逐行拷贝 x 行片（按 ci 通道）
            // 扁平化偏移计算: (((n * Cin + ci) * H + hStart) * W)
            uint64_t xBase = (((uint64_t)n * Cin + ci) * H + hStart) * W;
            AscendC::DataCopy(xLocal, xGm[xBase], xTileElems);

            // 遍历 kernel (Kh, Kw)
            for (uint32_t kh = 0; kh < Kh; ++kh) {
                for (uint32_t kw = 0; kw < Kw; ++kw) {
                    // 权重偏移: ((ci * Cout + co) * Kh + kh) * Kw + kw
                    uint64_t wOff = (((uint64_t)ci * Cout + co) * Kh + kh) * Kw + kw;
                    DTYPE_W wVal = wGm.GetValue(wOff);

                    // 对 tile 内每一行做累加: yLocal[(r+kh, kw : kw+W)] += xLocal[(r, 0:W)] * wVal
                    for (uint32_t r = 0; r < curTileH; ++r) {
                        uint64_t xRowOff = (uint64_t)r * W;
                        uint64_t yRowOff = ((uint64_t)r + kh) * (W + Kw - 1) + kw;

                        // 使用临时行缓冲: tmp = xRow * wVal
                        AscendC::LocalTensor<DTYPE_Y> tmpRow = tmpQueue.AllocTensor<DTYPE_Y>();
                        // 拷贝 x 行到 tmpRow 前 W 段
                        AscendC::DataCopy(reinterpret_cast<AscendC::LocalTensor<DTYPE_Y>&>(tmpRow), reinterpret_cast<AscendC::LocalTensor<DTYPE_Y>&>(xLocal[xRowOff]), W);
                        // 逐元素乘以标量 wVal
                        AscendC::Muls(tmpRow, tmpRow, wVal, W);
                        // yRow += tmpRow
                        AscendC::Add(yLocal[yRowOff], yLocal[yRowOff], tmpRow, W);
                        tmpQueue.FreeTensor(tmpRow);
                    }
                }
            }
        }

        // 将 yLocal 写回对应 y 的位置
        // y 偏移: ((n * Cout + co) * Hout + hStart) * Wout
        uint64_t yBase = (((uint64_t)n * Cout + co) * Hout + hStart) * Wout;
        // 逐行拷贝回 GM
        for (uint32_t r = 0; r < curTileH + Kh - 1; ++r) {
            uint64_t yLocalRowOff = (uint64_t)r * (W + Kw - 1);
            AscendC::DataCopy(yGm[yBase + (uint64_t)r * Wout], yLocal[yLocalRowOff], (uint64_t)(W + Kw - 1));
        }

        outQueueY.FreeTensor(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, Cin, Cout;
    uint32_t H, W, Kh, Kw;
    uint32_t Hout, Wout;
    uint32_t tileH;
    uint32_t nBegin, nEnd;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w, tiling_data.kh, tiling_data.kw,
            tiling_data.tileH);
    op.Process();
}
