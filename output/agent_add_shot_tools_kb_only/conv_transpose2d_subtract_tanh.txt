kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE_X;
typedef float DTYPE_B;
typedef float DTYPE_Y;

constexpr int32_t BUFFER_NUM = 2; // ping-pong

class KernelConvTranspose2dSubtractTanh {
public:
    __aicore__ inline KernelConvTranspose2dSubtractTanh() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y, uint32_t totalLength, uint32_t planeSize, uint32_t channels, uint32_t tileNum)
    {
        this->planeSize = planeSize;
        this->channels = channels;

        // 按平面(N*C)划分到block
        uint32_t totalPlanes = totalLength / planeSize;
        this->blockPlaneStart = (totalPlanes / AscendC::GetBlockNum()) * AscendC::GetBlockIdx();
        this->tileNum = tileNum; // 每个block处理的平面数(保证整除)
        this->tileLength = planeSize;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockPlaneStart * this->planeSize, this->tileNum * this->tileLength);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_B *)bias, this->channels);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockPlaneStart * this->planeSize, this->tileNum * this->tileLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < this->tileNum; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // 计算当前全局平面索引与通道索引
        uint32_t globalPlaneIdx = this->blockPlaneStart + progress;
        uint32_t cIdx = globalPlaneIdx % this->channels;

        // 从GM读取通道bias标量
        // 将bias单元素搬到本地，再读出标量
        DTYPE_B bScalar = 0.f;
        {
            // 1-element local buffer on stack
            AscendC::LocalTensor<DTYPE_B> bLocal;
            // 使用DataCopy到临时局部变量: 通过将本地tensor映射到yLocal的前1个元素区域临时复用，减少额外队列
            // 这里直接使用一个单独变量模拟，从GM按标量读取
            // 为兼容AscendC风格，采用将1个元素搬入临时向量，再取第0个标量的方式
        }

        // 简化：直接使用DataCopy到一个长度为1的局部tensor
        AscendC::TPipe tmpPipe;
        AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpQue;
        tmpPipe.InitBuffer(tmpQue, 1, sizeof(DTYPE_B));
        AscendC::LocalTensor<DTYPE_B> biasLocal = tmpQue.AllocTensor<DTYPE_B>();
        AscendC::DataCopy(biasLocal, biasGm[cIdx], 1);
        // 从本地tensor取出标量
        bScalar = biasLocal.GetValue(0);

        // 构造与xLocal同shape的bias向量并做减法
        AscendC::LocalTensor<DTYPE_Y> tmpBias = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::Duplicate(tmpBias, static_cast<DTYPE_Y>(bScalar), this->tileLength); // 将标量复制成向量
        AscendC::Sub(yLocal, xLocal, tmpBias, this->tileLength);                      // y = x - bias

        // tanh激活：使用框架分配临时空间的变体
        AscendC::Tanh(yLocal, yLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
        outQueueY.FreeTensor(tmpBias);
    }

    __aicore__ inline void CopyOut(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_B> biasGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t planeSize;
    uint32_t channels;
    uint32_t blockPlaneStart;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose2d_subtract_tanh_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose2dSubtractTanh op;
    op.Init(x, bias, y, tiling_data.totalLength, tiling_data.planeSize, tiling_data.channels, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom AscendC op for (x - bias) + tanh after ConvTranspose2d.
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape, stride=2, padding=1, output_padding=1):
        super().__init__()
        # 仍使用PyTorch自带反卷积
        self.conv_transpose = torch.nn.ConvTranspose2d(
            in_channels, out_channels, kernel_size,
            stride=stride, padding=padding, output_padding=output_padding, bias=True
        )
        # 后续减去的通道bias(C,1,1)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        x = self.conv_transpose(x)
        # 使用自定义核完成逐通道减bias并tanh
        y = custom_ops_lib.conv_transpose2d_subtract_tanh_custom(x, self.bias)
        return y
