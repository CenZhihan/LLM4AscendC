kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelMSE {
public:
    __aicore__ inline KernelMSE() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        // Use single block to avoid cross-block reduction
        this->blockLength = (totalLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        // ceil for safety
        uint32_t denom = (this->tileNum * BUFFER_NUM);
        this->tileLength = (this->blockLength + denom - 1) / denom;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        // output is scalar
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        // tmp buffer for ReduceSum needs extra workspace. Allocate 2x tileLength as temporary area.
        pipe.InitBuffer(tmpQue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z) * 2);
        // one small buffer for scalar output
        pipe.InitBuffer(outQueueZ, 1, 32);
    }
    __aicore__ inline void Process()
    {
        uint32_t offset = 0;
        float sumAcc = 0.0f;
        while (offset < this->blockLength) {
            uint32_t curLen = AscendC::Min(this->tileLength, this->blockLength - offset);
            CopyIn(offset, curLen);
            sumAcc += Compute(curLen);
            offset += curLen;
        }
        float meanVal = 0.0f;
        if (this->blockLength > 0) {
            meanVal = sumAcc / static_cast<float>(this->blockLength);
        }
        // write scalar out
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::Dup(zLocal, static_cast<DTYPE_Z>(meanVal), 1);
        AscendC::DataCopy(zGm, zLocal, 1);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[offset], len);
        AscendC::DataCopy(yLocal, yGm[offset], len);
        inQueueX.EnQue(xLocal);
        inQueueY.EnQue(yLocal);
    }
    __aicore__ inline float Compute(uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.DeQue<DTYPE_Y>();
        // tmp buffers for ReduceSum: split the tmpQue buffer into two regions
        AscendC::LocalTensor<DTYPE_Z> tmpA = tmpQue.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> tmpB = tmpQue.AllocTensor<DTYPE_Z>();

        // diff = x - y  ==> x + (-1) * y
        AscendC::Muls(yLocal, yLocal, (DTYPE_Y)(-1.0f), len);
        AscendC::Add(xLocal, xLocal, yLocal, len); // xLocal now holds diff

        // sq = diff * diff, reuse yLocal as output to save buffer
        AscendC::Mul(yLocal, xLocal, xLocal, len);

        // ReduceSum over this tile, use tmpA as dst and tmpB as work
        AscendC::ReduceSum(tmpA, yLocal, tmpB, len);
        float partSum = AscendC::GetAccVal<float>();

        // release queues
        tmpQue.FreeTensor(tmpB);
        tmpQue.FreeTensor(tmpA);
        inQueueX.FreeTensor(xLocal);
        inQueueY.FreeTensor(yLocal);
        return partSum;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void mse_loss_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMSE op;
    op.Init(x, y, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
