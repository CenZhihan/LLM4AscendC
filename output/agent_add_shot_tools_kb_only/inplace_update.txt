#include "inplace_update_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    InplaceUpdateCustomTilingData tiling;

    // 输入0: x [R, C]
    const auto *xShape = context->GetInputShape(0);
    const auto &xOrigin = xShape->GetOriginShape();
    uint32_t dims = xOrigin.GetDimNum();
    if (dims < 2) {
        return ge::GRAPH_FAILED;
    }
    uint32_t xRows = static_cast<uint32_t>(xOrigin.GetDim(0));
    uint32_t cols  = static_cast<uint32_t>(xOrigin.GetDim(xOrigin.GetDimNum() - 1));

    // 输入1: idx [updateRows]
    const auto *idxShape = context->GetInputShape(1);
    const auto &idxOrigin = idxShape->GetOriginShape();
    uint32_t updateRows = static_cast<uint32_t>(idxOrigin.GetShapeSize());

    tiling.set_xRows(xRows);
    tiling.set_cols(cols);
    tiling.set_updateRows(updateRows);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_zTotalLength(xRows * cols);

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelInplaceUpdate {
public:
    __aicore__ inline KernelInplaceUpdate() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR value, GM_ADDR y,
                                uint32_t xRows, uint32_t cols, uint32_t updateRows, uint32_t tileNum)
    {
        this->cols = cols;
        this->xRows = xRows;
        this->blockRows = updateRows / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileRows = this->blockRows / tileNum / BUFFER_NUM;

        // GM buffer 设置
        idxGm.SetGlobalBuffer((__gm__ int32_t*)idx + this->blockRows * AscendC::GetBlockIdx(), this->blockRows);
        valGm.SetGlobalBuffer((__gm__ DTYPE_X*)value + static_cast<uint64_t>(this->blockRows) * this->cols * AscendC::GetBlockIdx(),
                              static_cast<uint64_t>(this->blockRows) * this->cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y, static_cast<uint64_t>(this->xRows) * this->cols);

        // 队列缓冲
        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, this->tileRows * sizeof(int32_t));
        pipe.InitBuffer(inQueueVal, BUFFER_NUM, static_cast<uint64_t>(this->tileRows) * this->cols * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<int32_t> idxLocal = inQueueIdx.AllocTensor<int32_t>();
        AscendC::LocalTensor<DTYPE_X> valLocal  = inQueueVal.AllocTensor<DTYPE_X>();

        AscendC::DataCopy(idxLocal, idxGm[progress * this->tileRows], this->tileRows);
        AscendC::DataCopy(valLocal, valGm[static_cast<uint64_t>(progress) * this->tileRows * this->cols],
                          static_cast<uint64_t>(this->tileRows) * this->cols);

        inQueueIdx.EnQue(idxLocal);
        inQueueVal.EnQue(valLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<int32_t> idxLocal = inQueueIdx.DeQue<int32_t>();
        AscendC::LocalTensor<DTYPE_X>  valLocal = inQueueVal.DeQue<DTYPE_X>();

        // 逐行将 value 写入 y 的对应行
        for (uint32_t r = 0; r < this->tileRows; ++r) {
            // 从本地缓冲读取目标行索引
            int32_t dstRow = idxLocal.GetValue(r);
            // 每次拷贝一整行
            AscendC::DataCopy(yGm[static_cast<uint64_t>(dstRow) * this->cols],
                              valLocal[static_cast<uint64_t>(r) * this->cols],
                              this->cols);
        }

        inQueueVal.FreeTensor(valLocal);
        inQueueIdx.FreeTensor(idxLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inQueueVal;

    AscendC::GlobalTensor<int32_t> idxGm;
    AscendC::GlobalTensor<DTYPE_X> valGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t cols;
    uint32_t xRows;
    uint32_t blockRows;
    uint32_t tileNum;
    uint32_t tileRows;
};

extern "C" __global__ __aicore__ void inplace_update_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR value,
                                                            GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    (void)x; (void)workspace;
    GET_TILING_DATA(tilingData, tiling);
    KernelInplaceUpdate op;
    op.Init(x, idx, value, y,
            tilingData.xRows, tilingData.cols, tilingData.updateRows, tilingData.tileNum);
    op.Process();
}
"""
