const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_OUT = 512; // 每次在长度维度上处理的输出元素数量（可根据平台调优）

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    AveragePooling1dCustomTilingData tiling;

    // 获取输入形状 (N, C, L)
    const auto *inputShape = context->GetInputShape(0);
    const auto &originShape = inputShape->GetOriginShape();
    uint32_t n = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t c = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t inLen = static_cast<uint32_t>(originShape.GetDim(2));

    // 获取属性
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t k = *(attrs->GetAttrPointer<uint32_t>(0)); // kernel_size
    uint32_t s = *(attrs->GetAttrPointer<uint32_t>(1)); // stride
    uint32_t p = *(attrs->GetAttrPointer<uint32_t>(2)); // padding

    // 计算输出长度 (floor((L + 2P - K) / S) + 1)
    uint32_t outLen = 0;
    if (s > 0 && (int64_t)inLen + 2LL*(int64_t)p >= (int64_t)k) {
        outLen = (static_cast<uint32_t>((int64_t)inLen + 2LL*(int64_t)p - (int64_t)k) / s) + 1;
    } else {
        outLen = 0;
    }

    // 设置块数
    context->SetBlockDim(BLOCK_DIM);

    // 设置tiling参数
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inLen(inLen);
    tiling.set_outLen(outLen);
    tiling.set_k(k);
    tiling.set_s(s);
    tiling.set_p(p);
    tiling.set_tileOut(TILE_OUT);

    // 对于stride>=1，单个tile需要的最大输入跨度约为 tileOut*stride + kernel
    uint32_t maxInTile = TILE_OUT * s + k;
    tiling.set_maxInTile(maxInTile);

    // 保存tiling数据
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // 无需workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_QUEUE_NUM = 1;
constexpr int32_t OUT_QUEUE_NUM = 1;

class KernelAveragePooling1d {
public:
    __aicore__ inline KernelAveragePooling1d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t inLen, uint32_t outLen,
                                uint32_t k, uint32_t s, uint32_t p,
                                uint32_t tileOut, uint32_t maxInTile)
    {
        this->n = n;
        this->c = c;
        this->inLen = inLen;
        this->outLen = outLen;
        this->k = k;
        this->s = s;
        this->p = p;
        this->tileOut = tileOut;
        this->maxInTile = maxInTile;

        this->mcTotal = n * c;

        // 基于多Block划分(N*C)维度
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startMc = (mcTotal * blockIdx) / blockNum;
        uint32_t endMc   = (mcTotal * (blockIdx + 1)) / blockNum;
        this->mcThisBlock = endMc - startMc;
        this->mcStart = startMc;

        // 设置全局buffer窗口（以当前block起始MC为基址）
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)startMc * inLen, (uint64_t)mcThisBlock * inLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + (uint64_t)startMc * outLen, (uint64_t)mcThisBlock * outLen);

        // 本地缓冲：输入tile和输出tile
        pipe.InitBuffer(inQueueX, IN_QUEUE_NUM, maxInTile * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, OUT_QUEUE_NUM, tileOut * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // 对当前block覆盖的(N*C)行逐行处理
        for (uint32_t mc = 0; mc < mcThisBlock; ++mc) {
            // 沿长度维(输出长度)分tile
            for (uint32_t t0 = 0; t0 < outLen; t0 += tileOut) {
                uint32_t curTileOut = AscendC::Min<uint32_t>(tileOut, outLen - t0);

                // 计算当前输出tile对应的输入需求区间 [reqStart, reqEnd)
                int64_t reqStart = (int64_t)t0 * (int64_t)s - (int64_t)p;
                int64_t reqEnd   = (int64_t)(t0 + curTileOut - 1) * (int64_t)s - (int64_t)p + (int64_t)k;

                // 本地输入缓冲区大小
                uint32_t reqSpan = (uint32_t)(reqEnd - reqStart);
                if (reqSpan > maxInTile) {
                    reqSpan = maxInTile; // 安全保护
                }

                // 分配本地tensor
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

                // 将输入tile清零（处理填充区域）
                DTYPE_X zero(0.0f);
                AscendC::Duplicate<DTYPE_X>(xLocal, zero, reqSpan);

                // 将输出tile清零（累加池化用）
                DTYPE_Y yzero(0.0f);
                AscendC::Duplicate<DTYPE_Y>(yLocal, yzero, curTileOut);

                // 计算有效输入拷贝范围 [validStart, validEnd)
                int64_t validStart = AscendC::Max<int64_t>(reqStart, 0);
                int64_t validEnd   = AscendC::Min<int64_t>(reqEnd, (int64_t)inLen);

                if (validEnd > validStart) {
                    uint32_t validLen = (uint32_t)(validEnd - validStart);
                    uint32_t localOffset = (uint32_t)(validStart - reqStart); // 将有效输入放到本地buffer中的正确偏移
                    // 从全局拷贝有效数据
                    // xGm以当前block的起始行为0基址，当前行偏移= mc * inLen
                    AscendC::DataCopy(xLocal[localOffset], xGm[(uint64_t)mc * inLen + (uint32_t)validStart], validLen);
                }

                // 累加K个偏移窗口（count_include_pad = True）
                // 对于每个k偏移，将xLocal[k ... k+curTileOut-1]累加到yLocal
                for (uint32_t kk = 0; kk < k; ++kk) {
                    if (kk + curTileOut <= reqSpan) {
                        AscendC::Add(yLocal, yLocal, xLocal[kk], curTileOut);
                    } else {
                        // 边界保护，避免越界（极端情况下reqSpan不足以覆盖kk+curTileOut）
                        uint32_t validAccum = (kk < reqSpan) ? (reqSpan - kk) : 0;
                        if (validAccum > 0) {
                            AscendC::Add(yLocal, yLocal, xLocal[kk], validAccum);
                        }
                    }
                }

                // 计算平均：除以kernel大小（此实现等价于count_include_pad=True）
                float invK = 1.0f / static_cast<float>(k);
                AscendC::Muls(yLocal, yLocal, invK, curTileOut);

                // 写回全局输出
                outQueueY.EnQue<DTYPE_Y>(yLocal);
                AscendC::LocalTensor<DTYPE_Y> yLocalDeque = outQueueY.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[(uint64_t)mc * outLen + t0], yLocalDeque, curTileOut);
                outQueueY.FreeTensor(yLocalDeque);

                inQueueX.FreeTensor(xLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_QUEUE_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_QUEUE_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t inLen;
    uint32_t outLen;
    uint32_t k;
    uint32_t s;
    uint32_t p;
    uint32_t tileOut;
    uint32_t maxInTile;

    uint32_t mcTotal;
    uint32_t mcThisBlock;
    uint32_t mcStart;
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling1d op;
    op.Init(x, y,
            tiling_data.n(),
            tiling_data.c(),
            tiling_data.inLen(),
            tiling_data.outLen(),
            tiling_data.k(),
            tiling_data.s(),
            tiling_data.p(),
            tiling_data.tileOut(),
            tiling_data.maxInTile());
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor average_pooling_1d_custom_impl_npu(const at::Tensor& x, int64_t kernel_size, int64_t stride, int64_t padding) {
    // 计算输出长度
    auto sizes = x.sizes();
    TORCH_CHECK(sizes.size() == 3, \"Input must be (N, C, L)\");
    int64_t N = sizes[0];
    int64_t C = sizes[1];
    int64_t L = sizes[2];
    TORCH_CHECK(stride > 0, \"stride must be positive\");
    TORCH_CHECK(kernel_size > 0, \"kernel_size must be positive\");

    int64_t out_len = 0;
    if ((L + 2 * padding) >= kernel_size) {
        out_len = ( (L + 2 * padding - kernel_size) / stride ) + 1;
    }

    auto options = x.options();
    at::Tensor y = at::empty({N, C, out_len}, options);

    // 调用自定义算子（属性按顺序：kernel_size, stride, padding）
    EXEC_NPU_CMD(aclnnAveragePooling1dCustom, x, static_cast<int>(kernel_size), static_cast<int>(stride), static_cast<int>(padding), y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"average_pooling_1d_custom\", &average_pooling_1d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"average_pooling_1d_custom\", &average_pooling_1d_custom_impl_npu, \"AvgPool1d Custom\");
}
