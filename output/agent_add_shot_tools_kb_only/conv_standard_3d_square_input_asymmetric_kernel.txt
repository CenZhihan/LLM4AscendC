kernel_src="""
#include "kernel_operator.h"

// 简化实现：假设 stride=1, padding=0, dilation=1, groups=1，数据类型为 float
class KernelConv3d {
public:
    __aicore__ inline KernelConv3d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t din, uint32_t hin, uint32_t win,
                                uint32_t cout, uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout)
    {
        this->n = n; this->cin = cin; this->din = din; this->hin = hin; this->win = win;
        this->cout = cout; this->kd = kd; this->kh = kh; this->kw = kw;
        this->dout = dout; this->hout = hout; this->wout = wout;

        uint64_t xSize = (uint64_t)n * cin * din * hin * win;
        uint64_t wSize = (uint64_t)cout * cin * kd * kh * kw;
        uint64_t ySize = (uint64_t)n * cout * dout * hout * wout;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), xSize);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w), wSize);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), ySize);

        // 分配局部缓冲：一个权重向量(K)，一个输入补丁(K)，一个临时向量(K)，一个标量结果
        K = cin * kd * kh * kw;
        pipe.InitBuffer(qWeight, 1, K * sizeof(float));
        pipe.InitBuffer(qPatch, 1, K * sizeof(float));
        pipe.InitBuffer(qTmp,   1, K * sizeof(float));
        pipe.InitBuffer(qOne,   1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // 逐 batch、逐输出通道、逐输出位置的朴素计算
        for (uint32_t nb = 0; nb < n; ++nb) {
            for (uint32_t oc = 0; oc < cout; ++oc) {
                // 拷贝当前输出通道的权重向量到UB
                AscendC::LocalTensor<float> wLocal = qWeight.AllocTensor<float>();
                uint64_t wOffset = ((uint64_t)oc) * K;
                AscendC::DataCopy(wLocal, wGm[wOffset], K);

                for (uint32_t od = 0; od < dout; ++od) {
                    for (uint32_t oh = 0; oh < hout; ++oh) {
                        for (uint32_t ow = 0; ow < wout; ++ow) {
                            // 构造输入补丁向量 (长度K)，按 cin/kd/kh 逐行搬运 kw 个连续数据
                            AscendC::LocalTensor<float> patchLocal = qPatch.AllocTensor<float>();
                            uint32_t idx = 0;
                            for (uint32_t c = 0; c < cin; ++c) {
                                for (uint32_t kd_i = 0; kd_i < kd; ++kd_i) {
                                    uint32_t id = od + kd_i; // stride=1, pad=0
                                    for (uint32_t kh_i = 0; kh_i < kh; ++kh_i) {
                                        uint32_t ih = oh + kh_i;
                                        // 起始全局偏移（按 N,C,D,H,W 连续）
                                        uint64_t base = (((((uint64_t)nb) * cin + c) * din + id) * hin + ih) * win + ow;
                                        // 搬运 kw 个连续元素
                                        AscendC::DataCopy(patchLocal[idx], xGm[base], kw);
                                        idx += kw;
                                    }
                                }
                            }

                            // 逐元素相乘并求和 (tmp = wLocal * patchLocal; sum(tmp))
                            AscendC::LocalTensor<float> tmpLocal = qTmp.AllocTensor<float>();
                            AscendC::Mul(tmpLocal, wLocal, patchLocal, K);

                            AscendC::LocalTensor<float> sumLocal = qOne.AllocTensor<float>();
                            // 假定 ReduceSum(dst, src, length) 接口按元素求和
                            AscendC::ReduceSum(sumLocal, tmpLocal, K);

                            // 将标量结果写回全局输出：y[nb, oc, od, oh, ow]
                            uint64_t yOffset = (((((uint64_t)nb) * cout + oc) * dout + od) * hout + oh) * wout + ow;
                            AscendC::DataCopy(yGm[yOffset], sumLocal, 1);

                            qTmp.FreeTensor(tmpLocal);
                            qPatch.FreeTensor(patchLocal);
                            qOne.FreeTensor(sumLocal);
                        }
                    }
                }
                qWeight.FreeTensor(wLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  qWeight;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  qPatch;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  qTmp;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> qOne;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n, cin, din, hin, win;
    uint32_t cout, kd, kh, kw;
    uint32_t dout, hout, wout;
    uint32_t K;
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv3d op;
    op.Init(x, weight, y,
            tilingData.n, tilingData.cin, tilingData.din, tilingData.hin, tilingData.win,
            tilingData.cout, tilingData.kd, tilingData.kh, tilingData.kw,
            tilingData.dout, tilingData.hout, tilingData.wout);
    op.Process();
}

at::Tensor conv3d_custom_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
