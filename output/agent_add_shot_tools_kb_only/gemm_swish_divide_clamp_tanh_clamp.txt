kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

class KernelGemmSwishDivideClampTanhClamp {
public:
    __aicore__ inline KernelGemmSwishDivideClampTanhClamp() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, float swishBeta, float divideScalar, float clampMin, float clampMax,
                                uint32_t totalLength, uint32_t tileNum)
    {
        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        ASSERT(tileNum != 0 && "tile num can not be zero!");
        this->tileLength = (this->blockLength / tileNum) / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE));

        this->swishBeta = swishBeta;
        this->divideScalar = divideScalar;
        this->clampMin = clampMin;
        this->clampMax = clampMax;
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();

        // Swish: y = x * sigmoid(beta * x)
        AscendC::Swish(yLocal, xLocal, this->tileLength, (DTYPE)swishBeta);

        // Divide by 2.0 -> multiply by 0.5 (divideScalar)
        AscendC::Muls(yLocal, yLocal, (DTYPE)divideScalar, this->tileLength);

        // Clamp to [clampMin, clampMax]
        AscendC::Maxs(yLocal, yLocal, (DTYPE)clampMin, this->tileLength);
        AscendC::Mins(yLocal, yLocal, (DTYPE)clampMax, this->tileLength);

        // Tanh using exp: tanh(a) = (exp(2a) - 1) / (exp(2a) + 1)
        AscendC::LocalTensor<DTYPE> tmpLocal = tmpQueue.AllocTensor<DTYPE>();
        AscendC::Muls(tmpLocal, yLocal, (DTYPE)2.0f, this->tileLength);      // tmp = 2a
        AscendC::Exp(tmpLocal, tmpLocal, this->tileLength);                   // tmp = exp(2a)

        AscendC::Adds(yLocal, tmpLocal, (DTYPE)1.0f, this->tileLength);       // y = tmp + 1  (denominator)
        AscendC::Adds(tmpLocal, tmpLocal, (DTYPE)(-1.0f), this->tileLength);  // tmp = tmp - 1 (numerator)

        AscendC::Div(yLocal, tmpLocal, yLocal, this->tileLength);             // y = numerator / denominator

        // Final clamp to [-1, 1]
        AscendC::Maxs(yLocal, yLocal, (DTYPE)clampMin, this->tileLength);
        AscendC::Mins(yLocal, yLocal, (DTYPE)clampMax, this->tileLength);

        outQueueY.EnQue<DTYPE>(yLocal);
        inQueueX.FreeTensor(xLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    float swishBeta;
    float divideScalar;
    float clampMin;
    float clampMax;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gemm_swish_divide_clamp_tanh_clamp_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelGemmSwishDivideClampTanhClamp op;
    op.Init(x, y,
            tilingData.swishBeta(), tilingData.divideScalar(),
            tilingData.clampMin(), tilingData.clampMax(),
            tilingData.totalLength(), tilingData.tileNum());
    op.Process();
}
