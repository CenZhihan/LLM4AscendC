kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR v, GM_ADDR c, GM_ADDR y,
                                uint32_t batch, uint32_t inFeatures, uint32_t tileK, uint32_t tileNum)
    {
        this->batch = batch;
        this->inFeatures = inFeatures;
        this->tileK = tileK;
        this->tileNum = tileNum;

        // assign block rows
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (batch + blocks - 1) / blocks;
        this->rowStart = rowsPerBlock * blockIdx;
        this->rowEnd = this->rowStart + rowsPerBlock;
        if (this->rowEnd > batch) this->rowEnd = batch;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(x) + static_cast<uint64_t>(this->rowStart) * inFeatures,
                            static_cast<uint64_t>(this->rowEnd - this->rowStart) * inFeatures);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(y) + static_cast<uint64_t>(this->rowStart) * inFeatures,
                            static_cast<uint64_t>(this->rowEnd - this->rowStart) * inFeatures);
        vGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(v), inFeatures);
        cGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(c), 1);

        // init buffers
        uint32_t tileLen = this->tileK;
        pipe.InitBuffer(inQueueX, BUFFER_NUM, tileLen * sizeof(DTYPE));
        pipe.InitBuffer(inQueueV, BUFFER_NUM, tileLen * sizeof(DTYPE));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, tileLen * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileLen * sizeof(DTYPE));
        pipe.InitBuffer(workQ, 1, tileLen * sizeof(DTYPE)); // for ReduceSum work buffer
        pipe.InitBuffer(constQ, BUFFER_NUM, tileLen * sizeof(DTYPE)); // for broadcast const
        pipe.InitBuffer(geluQ, BUFFER_NUM, tileLen * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // read scalar c once
        DTYPE cScalar = 0.0f;
        {
            AscendC::LocalTensor<DTYPE> cLocal = tmpQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(cLocal, cGm, 1);
            cScalar = cLocal.GetValue(0);
            tmpQueue.FreeTensor(cLocal);
        }

        uint32_t rows = this->rowEnd - this->rowStart;
        for (uint32_t r = 0; r < rows; ++r) {
            // Pass 1: compute dot(x_row, v) -> sumDot
            float sumDot = 0.0f;
            uint32_t processed = 0;
            for (uint32_t t = 0; t < this->tileNum; ++t) {
                uint32_t currLen = this->tileK;
                if (processed + currLen > this->inFeatures) {
                    currLen = this->inFeatures - processed;
                }
                AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> vLocal = inQueueV.AllocTensor<DTYPE>();
                AscendC::DataCopy(xLocal, xGm[r * this->inFeatures + processed], currLen);
                AscendC::DataCopy(vLocal, vGm[processed], currLen);

                AscendC::LocalTensor<DTYPE> prodLocal = tmpQueue.AllocTensor<DTYPE>();
                AscendC::Mul(prodLocal, xLocal, vLocal, currLen);

                AscendC::LocalTensor<DTYPE> dstDummy = geluQ.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> workLocal = workQ.AllocTensor<DTYPE>();
                AscendC::ReduceSum(dstDummy, prodLocal, workLocal, currLen);
                float partial = AscendC::GetAccVal<float>();
                sumDot += partial;

                geluQ.FreeTensor(dstDummy);
                tmpQueue.FreeTensor(prodLocal);
                inQueueX.FreeTensor(xLocal);
                inQueueV.FreeTensor(vLocal);
                processed += currLen;
                if (processed >= this->inFeatures) break;
            }

            // GELU on scalar (broadcast later)
            float gIn = sumDot + cScalar;

            // Pass 2: y_row = x_row + GELU(gIn) (broadcast)
            processed = 0;
            for (uint32_t t = 0; t < this->tileNum; ++t) {
                uint32_t currLen = this->tileK;
                if (processed + currLen > this->inFeatures) {
                    currLen = this->inFeatures - processed;
                }
                AscendC::LocalTensor<DTYPE> xLocal2 = inQueueX.AllocTensor<DTYPE>();
                AscendC::DataCopy(xLocal2, xGm[r * this->inFeatures + processed], currLen);

                AscendC::LocalTensor<DTYPE> constLocal = constQ.AllocTensor<DTYPE>();
                AscendC::Duplicate(constLocal, gIn, currLen);

                AscendC::LocalTensor<DTYPE> geluLocal = geluQ.AllocTensor<DTYPE>();
                AscendC::Gelu(geluLocal, constLocal, currLen);

                AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
                AscendC::Add(yLocal, xLocal2, geluLocal, currLen);

                AscendC::DataCopy(yGm[r * this->inFeatures + processed], yLocal, currLen);

                outQueueY.FreeTensor(yLocal);
                geluQ.FreeTensor(geluLocal);
                constQ.FreeTensor(constLocal);
                inQueueX.FreeTensor(xLocal2);

                processed += currLen;
                if (processed >= this->inFeatures) break;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX, inQueueV;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> workQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> constQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> geluQ;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;
    AscendC::GlobalTensor<DTYPE> vGm;
    AscendC::GlobalTensor<DTYPE> cGm;

    uint32_t batch;
    uint32_t inFeatures;
    uint32_t tileK;
    uint32_t tileNum;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void gemm_subtract_global_avg_pool_log_sum_exp_gelu_residual_add_custom(
    GM_ADDR x, GM_ADDR v, GM_ADDR c, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, v, c, y,
            tiling_data.batch,
            tiling_data.inFeatures,
            tiling_data.tileK,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused forward: compute v = mean_j W[j,:], c = mean_j (b_j - s_j), then y = X + GELU(X @ v + c).
    This is mathematically equivalent to the original sequence.
    """
    def __init__(self, in_features, out_features, bias=True):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features, bias=bias)
        self.subtract = torch.nn.Parameter(torch.randn(out_features))

    def forward(self, x):
        # Precompute v and c on-the-fly to keep consistency with current parameters
        # v: column-wise mean of weight over output features
        W = self.gemm.weight    # [out_features, in_features]
        v = W.mean(dim=0)       # [in_features]

        if self.gemm.bias is not None:
            c = (self.gemm.bias - self.subtract).mean()
        else:
            c = (-self.subtract).mean()

        # Ensure tensors are on NPU and correct dtype
        v = v.to(x.device, dtype=x.dtype).contiguous()
        c_tensor = c.to(x.device, dtype=x.dtype).view(1)

        # Custom fused op: y = x + GELU(x @ v + c)
        y = custom_ops_lib.gemm_subtract_global_avg_pool_log_sum_exp_gelu_residual_add_custom(x, v, c_tensor)
        return y
