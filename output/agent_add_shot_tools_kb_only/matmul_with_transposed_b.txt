host_operator_src="""
#include <cstring>
#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"
#include "matmul_with_transposed_b_custom_tiling.h"
#include "platform_ascendc.h"
#include "matmul_tiling.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 获取输入形状
    const auto *shapeA = context->GetInputShape(0);
    const auto *shapeB = context->GetInputShape(1);
    // A: (M, K), B: (N, K) 在算子中按 B^T 使用 => (K, N)
    uint32_t M = static_cast<uint32_t>(shapeA->GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA->GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB->GetDim(0));

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);

    // 多核维度可以让框架根据 tiling 结果给出建议，先设置一个占位值
    tiling.SetDim(1);

    // 配置 A、B、C 类型与位置、格式、转置状态
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT, false);
    // B 在计算中使用转置（输入为 (N, K)，计算需要 (K, N)）
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT, true);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBias(false);

    // 设置形状
    tiling.SetShape(M, N, K);
    tiling.SetSingleShape(M, N, K);
    tiling.SetOrgShape(M, N, K);

    // 设置可用缓冲空间（-1 表示默认使用全部可用空间）
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret == -1) {
        // 获取 tiling 失败，退化为单核
        context->SetBlockDim(1);
    } else {
        // 使用 tiling 建议的多核数量
        context->SetBlockDim(tiling.GetCoreNum());
    }

    // 将 tiling 数据写入 RawTilingData
    auto *raw = context->GetRawTilingData();
    std::memcpy(raw->GetData(), &tilingData, sizeof(optiling::TCubeTiling));
    raw->SetDataSize(sizeof(optiling::TCubeTiling));

    // 不需要 workspace
    size_t *workspace_sizes = context->GetWorkspaceSizes(1);
    workspace_sizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    // A: (M, K), B: (N, K) => C: (M, N)
    gert::Shape* c_shape = context->GetOutputShape(0);
    std::vector<int64_t> out_dims(2);
    out_dims[0] = a_shape->GetDim(0);
    out_dims[1] = b_shape->GetDim(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, out_dims[0]);
    c_shape->SetDim(1, out_dims[1]);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // 输出类型跟随输入 A 类型
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}
}

namespace ops {
class MatmulWithTransposedBCustom : public OpDef {
public:
    explicit MatmulWithTransposedBCustom(const char* name) : OpDef(name)
    {
        this->Input("A")

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().AddConfig("ascend910b");
    }

OP_ADD(MatmulWithTransposedBCustom);
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

extern "C" __global__  __aicore__ void matmul_with_transposed_b_custom(GM_ADDR aGM, GM_ADDR bGM,
                                                                       GM_ADDR cGM, GM_ADDR workspaceGM,
                                                                       GM_ADDR tilingGM)
{
    // 从 GM 中拷贝 TCubeTiling
    optiling::TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < sizeof(optiling::TCubeTiling) / sizeof(uint32_t); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // 定义 MatmulType：A/B/C 为 GM，格式 ND，数据类型 float，B 使用转置
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::Normal> aType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::Normal> bType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::Normal> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    // 初始化 GM Tensor
    // 这里按单核计算形状设置缓冲大小，若为多核 tiling 会在 Init 中处理具体拆分
    AscendC::GlobalTensor<float> aGlobal;
    AscendC::GlobalTensor<float> bGlobal;
    AscendC::GlobalTensor<float> cGlobal;

    // 依据 SetSingleShape(M, N, K) 与 SetOrgShape(M, N, K) 的约定，TCubeTiling 中会保存单核/原始形状
    // 元素数量计算为 M*K、N*K、M*N
    uint32_t singleM = tiling.singleCoreM;
    uint32_t singleN = tiling.singleCoreN;
    uint32_t singleK = tiling.singleCoreK;

    int32_t sizeA = static_cast<int32_t>(singleM * singleK);
    int32_t sizeB = static_cast<int32_t>(singleN * singleK);
    int32_t sizeC = static_cast<int32_t>(singleM * singleN);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cGM), sizeC);

    // 根据 tiling 再次设置 GM 句柄
    AscendC::GlobalTensor<aType::DataType> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ float*>(aGlobal.GetPhyAddr()), sizeA);
    AscendC::GlobalTensor<bType::DataType> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ float*>(bGlobal.GetPhyAddr()), sizeB);
    AscendC::GlobalTensor<cType::DataType> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ float*>(cGlobal.GetPhyAddr()), sizeC);

    // 创建管道与 Matmul 实例
    set_atomic_none();
    matmul::Matmul<aType, bType, cType, biasType> mm;
    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);

    // 初始化并设置输入输出
    mm.Init(&tiling);
    mm.SetTensorA(gm_a[0], false); // A 不转置
    mm.SetTensorB(gm_b[0], true);  // B 在计算中转置
    mm.SetWorkspace(workspaceGM, 0);

    // 无 bias
    // 执行计算：单次批次
    mm.IterateNBatch(1, 1, 1, true);
}
