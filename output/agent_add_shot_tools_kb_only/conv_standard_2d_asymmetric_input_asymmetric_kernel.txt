kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

// 线性索引到(N, Co, Oh, Ow)的映射
__aicore__ inline void LinearToNCHW(uint32_t idx, uint32_t Co, uint32_t Oh, uint32_t Ow,
                                     uint32_t &n, uint32_t &co, uint32_t &oh, uint32_t &ow) {
    uint32_t per_n = Co * Oh * Ow;
    n = idx / per_n;
    uint32_t rem = idx % per_n;
    co = rem / (Oh * Ow);
    rem = rem % (Oh * Ow);
    oh = rem / Ow;
    ow = rem % Ow;
}

class KernelConv2DStd {
public:
    __aicore__ inline KernelConv2DStd() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t Wi,
                                uint32_t Co, uint32_t kH, uint32_t kW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilH, uint32_t dilW,
                                uint32_t Oh, uint32_t Ow,
                                uint32_t tileNum)
    {
        this->N = N; this->C = C; this->H = H; this->W = Wi;
        this->Co = Co; this->kH = kH; this->kW = kW;
        this->strideH = strideH; this->strideW = strideW;
        this->padH = padH; this->padW = padW;
        this->dilH = dilH; this->dilW = dilW;
        this->Oh = Oh; this->Ow = Ow;
        this->tileNum = tileNum;

        totalOut = N * Co * Oh * Ow;
        blockOut = (totalOut + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        startIdx = blockOut * AscendC::GetBlockIdx();
        if (startIdx + blockOut > totalOut) blockOut = totalOut - startIdx;

        // GM绑定
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint32_t)(N * C * H * W));
        wGm.SetGlobalBuffer((__gm__ DTYPE_X*)w, (uint32_t)(Co * C * kH * kW));
        yGm.SetGlobalBuffer((__gm__ DTYPE_X*)y, (uint32_t)(N * Co * Oh * Ow));

        // 局部缓存：权重一个输出通道的卷积核 (C*kH*kW)，输入一行patch (kW)
        pipe.InitBuffer(inQueuePatch, BUFFER_NUM, kW * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, TILE_WRITE_NUM * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        // 简单的逐元素输出计算（stride=1, pad=0, dil=1）场景
        // 每次写出一小段结果
        uint32_t idx = 0;
        while (idx < blockOut) {
            uint32_t chunk = TILE_WRITE_NUM;
            if (idx + chunk > blockOut) {
                chunk = blockOut - idx;
            }
            CopyComputeWrite(idx, chunk);
            idx += chunk;
        }
    }

private:
    static constexpr uint32_t TILE_WRITE_NUM = 256;

    __aicore__ inline uint32_t offsetX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((n * C + c) * H + h) * W + w);
    }

    __aicore__ inline uint32_t offsetW(uint32_t co, uint32_t c, uint32_t r, uint32_t s) const
    {
        return (((co * C + c) * kH + r) * kW + s);
    }

    __aicore__ inline uint32_t offsetY(uint32_t n, uint32_t co, uint32_t h, uint32_t w) const
    {
        return (((n * Co + co) * Oh + h) * Ow + w);
    }

    __aicore__ inline void CopyComputeWrite(uint32_t localStart, uint32_t count)
    {
        AscendC::LocalTensor<DTYPE_X> yLocal = outQueueY.AllocTensor<DTYPE_X>();
        // 填零
        AscendC::Duplicate(yLocal, (DTYPE_X)0, count);

        for (uint32_t i = 0; i < count; ++i) {
            uint32_t gidx = startIdx + localStart + i;
            uint32_t n, co, oh, ow;
            LinearToNCHW(gidx, Co, Oh, Ow, n, co, oh, ow);

            // 一个输出元素的卷积累加
            float acc = 0.f;
            // 输入起点（无padding、无dilation、stride=1）
            int32_t in_h_base = (int32_t)oh * (int32_t)strideH - (int32_t)padH;
            int32_t in_w_base = (int32_t)ow * (int32_t)strideW - (int32_t)padW;

            for (uint32_t c = 0; c < C; ++c) {
                for (uint32_t r = 0; r < kH; ++r) {
                    int32_t h_in = in_h_base + (int32_t)r * (int32_t)dilH;
                    if (h_in < 0 || h_in >= (int32_t)H) continue;

                    // 拷贝一行patch (kW) 到本地
                    AscendC::LocalTensor<DTYPE_X> patchLocal = inQueuePatch.AllocTensor<DTYPE_X>();
                    // 边界检查（无padding=0，越界直接跳过）
                    bool validRow = true;
                    int32_t w_start = in_w_base;
                    int32_t w_end = in_w_base + (int32_t)kW; // 不包含
                    if (w_start < 0 || w_end > (int32_t)W) {
                        validRow = false;
                    }

                    if (validRow) {
                        AscendC::DataCopy(
                            patchLocal,
                            xGm[offsetX(n, c, (uint32_t)h_in, (uint32_t)w_start)],
                            kW);
                        // 与权重对应行做内积
                        for (uint32_t s = 0; s < kW; ++s) {
                            float xv = (float)patchLocal.GetValue(s);
                            float wv = (float)wGm[offsetW(co, c, r, s)];
                            acc += xv * wv;
                        }
                    } else {
                        // 越界补零行为：只跳过（等价于零贡献）
                    }

                    inQueuePatch.FreeTensor(patchLocal);
                }
            }

            // 写回到本地结果缓冲
            yLocal.SetValue(i, (DTYPE_X)acc);
        }

        // 将本地结果写回GM
        AscendC::DataCopy(yGm[offsetYIdx(startIdx + localStart)], yLocal, count);
        outQueueY.EnQue<DTYPE_X>(yLocal);
        AscendC::LocalTensor<DTYPE_X> yLocalDeq = outQueueY.DeQue<DTYPE_X>();
        outQueueY.FreeTensor(yLocalDeq);
    }

    __aicore__ inline uint32_t offsetYIdx(uint32_t gidx) const
    {
        uint32_t n, co, oh, ow;
        LinearToNCHW(gidx, Co, Oh, Ow, n, co, oh, ow);
        return offsetY(n, co, oh, ow);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueuePatch;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> wGm;
    AscendC::GlobalTensor<DTYPE_X> yGm;

    uint32_t N, C, H, W;
    uint32_t Co, kH, kW;
    uint32_t strideH, strideW, padH, padW, dilH, dilW;
    uint32_t Oh, Ow;

    uint32_t tileNum;
    uint32_t totalOut;
    uint32_t blockOut;
    uint32_t startIdx;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv2DStd op;
    op.Init(x, w, y,
            tilingData.n(), tilingData.cIn(), tilingData.h(), tilingData.w(),
            tilingData.cOut(), tilingData.kH(), tilingData.kW(),
            tilingData.strideH(), tilingData.strideW(),
            tilingData.padH(), tilingData.padW(),
            tilingData.dilH(), tilingData.dilW(),
            tilingData.outH(), tilingData.outW(),
            tilingData.tileNum());
    op.Process();
}
