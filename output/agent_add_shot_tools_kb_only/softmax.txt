host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/debug/ge_log.h"
#include <vector>

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    SoftmaxCustomTilingData tiling;

    // 获取输入形状 (假定为2D: [M, K])
    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin_shape = in_shape->GetOriginShape();
    int64_t m = origin_shape.GetDim(0);
    int64_t k = origin_shape.GetDim(1);

    std::vector<int64_t> shapeVec = {m, k};
    ge::Shape srcShape(shapeVec);

    // 计算最小临时空间，并生成SoftMax的tiling信息
    const uint32_t localWorkSpaceSize = AscendC::GetSoftMaxMinTmpSize(srcShape, sizeof(float), false);
    AscendC::SoftMaxTilingFunc(srcShape, sizeof(float), localWorkSpaceSize, tiling.mutable_softmaxTilingData());

    // 保存tiling
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // GM级Workspace无需额外分配
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    context->SetBlockDim(8);
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}
}

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})

kernel_src="""
#include "kernel_operator.h"

class KernelSoftmax {
public:
    __aicore__ inline KernelSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, const SoftMaxTiling &tilingData)
    {
        // 从tiling中取整体形状
        this->M = tilingData.srcM;
        this->K = tilingData.srcK;
        this->tiling = tilingData;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // 行维度在不同block间划分
        this->rowsPerBlock = (M + blockNum - 1) / blockNum;
        this->rowStart = rowsPerBlock * blockIdx;
        this->rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > M) rowEnd = M;
        this->rowsThisBlock = (rowEnd > rowStart) ? (rowEnd - rowStart) : 0;

        // 绑定当前block的数据区间
        xGm.SetGlobalBuffer((__gm__ float *)x + (uint64_t)rowStart * K, (uint64_t)rowsThisBlock * K);
        yGm.SetGlobalBuffer((__gm__ float *)y + (uint64_t)rowStart * K, (uint64_t)rowsThisBlock * K);

        // 每次处理一行，减少UB占用
        pipe.InitBuffer(inQueueX, 1, K * sizeof(float));
        pipe.InitBuffer(outQueueY, 1, K * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsThisBlock; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t r)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[r * K], K);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*r*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        // 当前仅处理一行: [1, K]
        AscendC::SoftMaxShapeInfo shapeInfo = {1u, K, 1u, K};
        AscendC::SoftMax<float>(yLocal, xLocal, tiling, shapeInfo);

        outQueueY.EnQue<float>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t r)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[r * K], yLocal, K);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;
    AscendC::GlobalTensor<float> xGm, yGm;

    uint32_t M = 0;
    uint32_t K = 0;
    uint32_t rowsPerBlock = 0;
    uint32_t rowStart = 0;
    uint32_t rowEnd = 0;
    uint32_t rowsThisBlock = 0;

    SoftMaxTiling tiling;
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelSoftmax op;
    op.Init(x, y, tilingData.softmaxTilingData);
    op.Process();
}
"""
