kernel_src="""
#include "kernel_operator.h"

using fp32 = float;

class KernelConvTransposed2d {
public:
    __aicore__ inline KernelConvTransposed2d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t wi,
                                uint32_t kH, uint32_t kW,
                                uint32_t padH, uint32_t padW)
    {
        this->N = n;
        this->Cin = cin;
        this->Cout = cout;
        this->H = h;
        this->W = wi;
        this->kH = kH;
        this->kW = kW;
        this->padH = padH;
        this->padW = padW;

        xGm.SetGlobalBuffer((__gm__ fp32*)x);
        wGm.SetGlobalBuffer((__gm__ fp32*)w);
        yGm.SetGlobalBuffer((__gm__ fp32*)y);

        // split output channels across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();
        this->ocPerBlock = (Cout + blockNum - 1) / blockNum;
        this->ocStart = blkIdx * this->ocPerBlock;
        this->ocEnd = this->ocStart + this->ocPerBlock;
        if (this->ocEnd > Cout) this->ocEnd = Cout;

        // UB buffers: one feature map (H*W), one kernel (kH*kW), one output map (H*W)
        pipe.InitBuffer(inQX, 1, H * W * sizeof(fp32));
        pipe.InitBuffer(inQW, 1, kH * kW * sizeof(fp32));
        pipe.InitBuffer(outQY, 1, H * W * sizeof(fp32));
    }

    __aicore__ inline void Process()
    {
        // Iterate batch and assigned output channels
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oc = ocStart; oc < ocEnd; ++oc) {
                // Allocate and zero-init output map in UB
                AscendC::LocalTensor<fp32> yLocal = outQY.AllocTensor<fp32>();
                uint32_t ySize = H * W;
                for (uint32_t i = 0; i < ySize; ++i) {
                    yLocal.SetValue(i, 0.0f);
                }

                // Accumulate over input channels
                for (uint32_t ic = 0; ic < Cin; ++ic) {
                    // Load input feature map x[n, ic, :, :]
                    AscendC::LocalTensor<fp32> xLocal = inQX.AllocTensor<fp32>();
                    uint64_t xOffset = ((uint64_t)n * Cin + ic) * (uint64_t)H * (uint64_t)W;
                    AscendC::DataCopy(xLocal, xGm[xOffset], H * W);

                    // Load kernel w[ic, oc, :, :]
                    AscendC::LocalTensor<fp32> wLocal = inQW.AllocTensor<fp32>();
                    uint64_t wOffset = (((uint64_t)ic * Cout) + oc) * (uint64_t)kH * (uint64_t)kW;
                    AscendC::DataCopy(wLocal, wGm[wOffset], kH * kW);

                    // Compute deconvolution for stride=1, padding=(padH,padW)
                    for (uint32_t oh = 0; oh < H; ++oh) {
                        for (uint32_t ow = 0; ow < W; ++ow) {
                            fp32 acc = yLocal.GetValue(oh * W + ow);
                            for (uint32_t kh = 0; kh < kH; ++kh) {
                                int32_t ih = (int32_t)oh - (int32_t)kh + (int32_t)padH;
                                if (ih < 0 || ih >= (int32_t)H) continue;
                                for (uint32_t kw = 0; kw < kW; ++kw) {
                                    int32_t iw = (int32_t)ow - (int32_t)kw + (int32_t)padW;
                                    if (iw < 0 || iw >= (int32_t)W) continue;
                                    fp32 xv = xLocal.GetValue((uint32_t)ih * W + (uint32_t)iw);
                                    fp32 wv = wLocal.GetValue(kh * kW + kw);
                                    acc += xv * wv;
                                }
                            }
                            yLocal.SetValue(oh * W + ow, acc);
                        }
                    }

                    inQW.FreeTensor(wLocal);
                    inQX.FreeTensor(xLocal);
                }

                // Store y[n, oc, :, :]
                uint64_t yOffset = ((uint64_t)n * Cout + oc) * (uint64_t)H * (uint64_t)W;
                AscendC::DataCopy(yGm[yOffset], yLocal, H * W);
                outQY.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQW;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQY;

    AscendC::GlobalTensor<fp32> xGm;
    AscendC::GlobalTensor<fp32> wGm;
    AscendC::GlobalTensor<fp32> yGm;

    uint32_t N, Cin, Cout, H, W, kH, kW, padH, padW;
    uint32_t ocPerBlock, ocStart, ocEnd;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_padded_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvTransposed2d op;
    op.Init(x, w, y,
            tilingData.n(), tilingData.cin(), tilingData.cout(),
            tilingData.h(), tilingData.w(),
            tilingData.kH(), tilingData.kW(),
            tilingData.padH(), tilingData.padW());
    op.Process();
}

    Optimized model using custom AscendC kernel:
    conv_transposed_2d_asymmetric_input_asymmetric_kernel_padded_custom

        assert not bias, "This custom kernel currently does not support bias."

        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.padding = padding
        # Keep a standard layer to hold weight tensor shape and storage
        self.conv_transpose2d = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
