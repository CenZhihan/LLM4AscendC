inline uint32_t ceil_div(uint32_t a, uint32_t b) {

namespace optiling {
const uint32_t BLOCK_DIM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ConvTransposed1dCustomTilingData tiling;

    auto xShape = context->GetInputShape(0)->GetOriginShape().GetDims();
    auto wShape = context->GetInputShape(1)->GetOriginShape().GetDims();
    // x: [N, IC, Lin], w: [IC, OC, K]
    if (xShape.size() != 3 || wShape.size() != 3) {
        return ge::GRAPH_FAILED;
    }
    uint32_t N  = static_cast<uint32_t>(xShape[0]);
    uint32_t IC = static_cast<uint32_t>(xShape[1]);
    uint32_t Lin = static_cast<uint32_t>(xShape[2]);
    uint32_t ICw = static_cast<uint32_t>(wShape[0]);
    uint32_t OC = static_cast<uint32_t>(wShape[1]);
    uint32_t K  = static_cast<uint32_t>(wShape[2]);

    if (IC != ICw) {
        return ge::GRAPH_FAILED;
    }

    uint32_t Lout = Lin + K - 1; // stride=1, padding=0, output_padding=0

    uint32_t tileLen = (Lout < DEFAULT_TILE_LEN) ? Lout : DEFAULT_TILE_LEN;
    uint32_t tileNum = (tileLen == 0) ? 0 : ceil_div(Lout, tileLen);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(N);
    tiling.set_ic(IC);
    tiling.set_oc(OC);
    tiling.set_lin(Lin);
    tiling.set_ksize(K);
    tiling.set_lout(Lout);
    tiling.set_tileLen(tileLen);
    tiling.set_tileNum(tileNum);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 1;

class KernelConvTransposed1D {
public:
    __aicore__ inline KernelConvTransposed1D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t ic, uint32_t oc,
                                uint32_t lin, uint32_t ksize, uint32_t lout,
                                uint32_t tileLen, uint32_t tileNum)
    {
        this->N = n;
        this->IC = ic;
        this->OC = oc;
        this->Lin = lin;
        this->K = ksize;
        this->Lout = lout;
        this->tileLen = tileLen;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint32_t)(N * IC * Lin));
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint32_t)(IC * OC * K));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, (uint32_t)(N * OC * Lout));

        uint32_t maxTmp = tileLen; // 临时缓冲区长度（按需调整）
        pipe.InitBuffer(tmpQue, BUFFER_NUM, maxTmp * sizeof(DTYPE_Y));
        pipe.InitBuffer(xQue, BUFFER_NUM, maxTmp * sizeof(DTYPE_X));
        pipe.InitBuffer(ySegQue, BUFFER_NUM, maxTmp * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (tileLen == 0 || tileNum == 0) {
            return;
        }
        // 逐 tile 处理输出长度维度
        for (uint32_t t = 0; t < tileNum; ++t) {
            uint32_t l0 = t * tileLen;
            uint32_t curTile = AscendC::min(tileLen, Lout - l0);

            for (uint32_t n = 0; n < N; ++n) {
                for (uint32_t oc = 0; oc < OC; ++oc) {
                    // 先把当前 (n, oc) 的输出 tile 清零
                    ZeroOutTile(n, oc, l0, curTile);

                    // 累加各个 (ic, k) 的贡献
                    for (uint32_t ic = 0; ic < IC; ++ic) {
                        for (uint32_t k = 0; k < K; ++k) {
                            int64_t start_in = (int64_t)l0 - (int64_t)k;
                            int64_t end_in = (int64_t)l0 + (int64_t)curTile - (int64_t)k;
                            if (start_in < 0) start_in = 0;
                            if (end_in > (int64_t)Lin) end_in = (int64_t)Lin;
                            uint32_t segLen = (start_in < end_in) ? (uint32_t)(end_in - start_in) : 0;
                            if (segLen == 0) continue;

                            uint32_t out_off = (uint32_t)((start_in + (int64_t)k) - (int64_t)l0);

                            // 加载输入片段 x[n, ic, start_in : start_in+segLen]
                            AscendC::LocalTensor<DTYPE_X> xLocal = xQue.AllocTensor<DTYPE_X>();
                            AscendC::DataCopy(xLocal, xGm[GetXOffset(n, ic, (uint32_t)start_in)], segLen);
                            xQue.EnQue(xLocal);
                            xLocal = xQue.DeQue<DTYPE_X>();

                            // tmp = xLocal * weight_scalar
                            DTYPE_W wScalar = wGm.GetValue(GetWOffset(ic, oc, k));
                            AscendC::LocalTensor<DTYPE_Y> tmpLocal = tmpQue.AllocTensor<DTYPE_Y>();
                            AscendC::Axpy(tmpLocal, xLocal, (DTYPE_Y)wScalar, segLen); // 生成缩放后的向量
                            tmpQue.EnQue(tmpLocal);
                            xQue.FreeTensor(xLocal);
                            tmpLocal = tmpQue.DeQue<DTYPE_Y>();

                            // 从 y 中取出目标片段，做加法再写回
                            AscendC::LocalTensor<DTYPE_Y> ySeg = ySegQue.AllocTensor<DTYPE_Y>();
                            AscendC::DataCopy(ySeg, yGm[GetYOffset(n, oc, l0 + out_off)], segLen);
                            AscendC::Add(ySeg, ySeg, tmpLocal, segLen);
                            AscendC::DataCopy(yGm[GetYOffset(n, oc, l0 + out_off)], ySeg, segLen);
                            ySegQue.EnQue(ySeg);
                            ySegQue.FreeTensor(ySeg);
                            tmpQue.FreeTensor(tmpLocal);
                        }
                    }
                }
            }
        }
    }

private:
    __aicore__ inline uint32_t GetXOffset(uint32_t n, uint32_t c, uint32_t l) const
    {
        return (n * IC + c) * Lin + l;
    }
    __aicore__ inline uint32_t GetWOffset(uint32_t ic, uint32_t oc, uint32_t k) const
    {
        return (ic * OC + oc) * K + k;
    }
    __aicore__ inline uint32_t GetYOffset(uint32_t n, uint32_t c, uint32_t l) const
    {
        return (n * OC + c) * Lout + l;
    }

    __aicore__ inline void ZeroOutTile(uint32_t n, uint32_t oc, uint32_t l0, uint32_t curTile)
    {
        // 使用 Axpy(alpha=0) 生成零向量，并循环写满 y 的当前 tile
        // 选择一个可用的模板源，如 x 的开头一段
        uint32_t pattern = curTile;
        if (pattern > Lin) pattern = (Lin > 0) ? Lin : 1;

        AscendC::LocalTensor<DTYPE_X> srcLocal = xQue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(srcLocal, xGm[0], pattern); // 任意数据，用于生成零向量
        xQue.EnQue(srcLocal);
        srcLocal = xQue.DeQue<DTYPE_X>();

        AscendC::LocalTensor<DTYPE_Y> zeroLocal = tmpQue.AllocTensor<DTYPE_Y>();
        AscendC::Axpy(zeroLocal, srcLocal, (DTYPE_Y)0.0f, pattern); // 0 * src -> 0
        tmpQue.EnQue(zeroLocal);
        xQue.FreeTensor(srcLocal);
        zeroLocal = tmpQue.DeQue<DTYPE_Y>();

        uint32_t written = 0;
        while (written < curTile) {
            uint32_t step = AscendC::min(pattern, curTile - written);
            AscendC::DataCopy(yGm[GetYOffset(n, oc, l0 + written)], zeroLocal, step);
            written += step;
        }
        tmpQue.FreeTensor(zeroLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> ySegQue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, IC, OC;
    uint32_t Lin, K, Lout;
    uint32_t tileLen, tileNum;
};

extern "C" __global__ __aicore__ void conv_transposed_1d_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.ic, tiling_data.oc,
            tiling_data.lin, tiling_data.ksize, tiling_data.lout,
            tiling_data.tileLen, tiling_data.tileNum);
    op.Process();
}
