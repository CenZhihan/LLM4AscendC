kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

extern "C" __global__ __aicore__ void net_vlad_with_ghost_clusters_custom(
    GM_ADDR assignment,  // B x N x K
    GM_ADDR x,           // B x N x D
    GM_ADDR clusters2,   // 1 x D x K
    GM_ADDR vlad,        // B x D x K
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t B = tiling_data.batch();
    const uint32_t N = tiling_data.numFeatures();
    const uint32_t D = tiling_data.featureSize();
    const uint32_t K = tiling_data.numClusters();

    const uint32_t blockIdx = GetBlockIdx();
    if (blockIdx >= B) {
        return;
    }

    __gm__ float* asgGm = (__gm__ float*)assignment + (uint64_t)blockIdx * (uint64_t)N * (uint64_t)K;
    __gm__ float* xGm   = (__gm__ float*)x         + (uint64_t)blockIdx * (uint64_t)N * (uint64_t)D;
    __gm__ float* c2Gm  = (__gm__ float*)clusters2; // length D*K
    __gm__ float* outGm = (__gm__ float*)vlad      + (uint64_t)blockIdx * (uint64_t)D * (uint64_t)K;

    // 1) 初始化 out = 0 (shape D x K，按列主序 d*K + k)
    for (uint32_t dk = 0; dk < D * K; ++dk) {
        outGm[dk] = 0.0f;
    }

    // 2) 计算 vlad_pre = x^T @ assignment = sum_n x[n,:]^T * assignment[n,:]
    // 即对每个样本 n，执行 rank-1 外积累加：out[d,k] += x[n,d] * assignment[n,k]
    for (uint32_t n = 0; n < N; ++n) {
        const uint64_t x_row_base   = (uint64_t)n * (uint64_t)D;
        const uint64_t asg_row_base = (uint64_t)n * (uint64_t)K;

        for (uint32_t k = 0; k < K; ++k) {
            float a = asgGm[asg_row_base + k];
            // 累加到列 k
            uint64_t out_col_base = (uint64_t)k; // out[d*K + k]
            for (uint32_t d = 0; d < D; ++d) {
                float xv = xGm[x_row_base + d];
                outGm[(uint64_t)d * (uint64_t)K + out_col_base] += (xv * a);
            }
        }
    }

    // 3) 计算 a_sum = sum_n assignment[:, :, k]，并执行 out -= a_sum * clusters2
    // clusters2: [1, D, K]，按 [D, K] 展开
    for (uint32_t k = 0; k < K; ++k) {
        float a_sum = 0.0f;
        for (uint32_t n = 0; n < N; ++n) {
            a_sum += asgGm[(uint64_t)n * (uint64_t)K + k];
        }
        for (uint32_t d = 0; d < D; ++d) {
            float c2 = c2Gm[(uint64_t)d * (uint64_t)K + k];
            outGm[(uint64_t)d * (uint64_t)K + k] -= (a_sum * c2);
        }
    }
}
"""

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch as th
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(ModelNew, self).__init__()
        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        import math
        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        self.clusters = nn.Parameter(init_sc * th.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        self.clusters2 = nn.Parameter(init_sc * th.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size

    def forward(self, x, mask=None):
        # x: B x N x D
        B, N, D = x.shape
        x2d = x.reshape(-1, self.feature_size)  # BN x D
        assignment = th.matmul(x2d, self.clusters)               # BN x (K+G)
        assignment = self.batch_norm(assignment)                  # BN x (K+G)
        assignment = F.softmax(assignment, dim=1)                 # BN x (K+G)
        assignment = assignment[:, :self.cluster_size]            # BN x K
        assignment = assignment.view(B, N, self.cluster_size)     # B x N x K

        # 自定义算子：核心NetVLAD累积（完成 assignment^T@x 并减去 a_sum*clusters2）
        vlad = custom_ops_lib.net_vlad_with_ghost_clusters_custom(assignment, x, self.clusters2)  # B x D x K

        # L2 intra norm
        vlad = F.normalize(vlad, dim=1)

        # flattening + L2 norm
        vlad = vlad.reshape(B, self.cluster_size * self.feature_size)  # B x DK
        vlad = F.normalize(vlad, dim=1)
        return vlad
'''
