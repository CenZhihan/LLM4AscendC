#include "trilinear_upsample_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline int64_t GetDim(const gert::Shape* s, int idx) {
    if (s->GetOriginShape().GetDimNum() > 0) {
        return s->GetOriginShape().GetDim(idx);
    }
    return s->GetStorageShape().GetDim(idx);
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    if (in_shape->GetOriginShape().GetDimNum() != 5 && in_shape->GetStorageShape().GetDimNum() != 5) {
        return ge::GRAPH_PARAM_INVALID;
    }

    int64_t N = GetDim(in_shape, 0);
    int64_t C = GetDim(in_shape, 1);
    int64_t D_in = GetDim(in_shape, 2);
    int64_t H_in = GetDim(in_shape, 3);
    int64_t W_in = GetDim(in_shape, 4);

    // align_corners=True, scale_factor=2 => out = (in-1)*2+1
    int64_t D_out = (D_in - 1) * 2 + 1;
    int64_t H_out = (H_in - 1) * 2 + 1;
    int64_t W_out = (W_in - 1) * 2 + 1;

    TrilinearUpsampleCustomTilingData tiling;
    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_c(static_cast<uint32_t>(C));
    tiling.set_d_in(static_cast<uint32_t>(D_in));
    tiling.set_h_in(static_cast<uint32_t>(H_in));
    tiling.set_w_in(static_cast<uint32_t>(W_in));
    tiling.set_d_out(static_cast<uint32_t>(D_out));
    tiling.set_h_out(static_cast<uint32_t>(H_out));
    tiling.set_w_out(static_cast<uint32_t>(W_out));
    uint64_t totalOutLength = static_cast<uint64_t>(N) * static_cast<uint64_t>(C) *
                              static_cast<uint64_t>(D_out) * static_cast<uint64_t>(H_out) *
                              static_cast<uint64_t>(W_out);
    tiling.set_totalOutLength(static_cast<uint32_t>(totalOutLength)); // assume length fits uint32 for demo

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    if (x_shape->GetOriginShape().GetDimNum() != 5 && x_shape->GetStorageShape().GetDimNum() != 5) {
        return GRAPH_PARAM_INVALID;
    }
    int64_t N = x_shape->GetOriginShape().GetDimNum() ? x_shape->GetOriginShape().GetDim(0) : x_shape->GetStorageShape().GetDim(0);
    int64_t C = x_shape->GetOriginShape().GetDimNum() ? x_shape->GetOriginShape().GetDim(1) : x_shape->GetStorageShape().GetDim(1);
    int64_t D_in = x_shape->GetOriginShape().GetDimNum() ? x_shape->GetOriginShape().GetDim(2) : x_shape->GetStorageShape().GetDim(2);
    int64_t H_in = x_shape->GetOriginShape().GetDimNum() ? x_shape->GetOriginShape().GetDim(3) : x_shape->GetStorageShape().GetDim(3);
    int64_t W_in = x_shape->GetOriginShape().GetDimNum() ? x_shape->GetOriginShape().GetDim(4) : x_shape->GetStorageShape().GetDim(4);

    int64_t D_out = (D_in - 1) * 2 + 1;
    int64_t H_out = (H_in - 1) * 2 + 1;
    int64_t W_out = (W_in - 1) * 2 + 1;

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, D_out);
    y_shape->SetDim(3, H_out);
    y_shape->SetDim(4, W_out);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void trilinear_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.n();
    const uint32_t C = tiling_data.c();
    const uint32_t D_in = tiling_data.d_in();
    const uint32_t H_in = tiling_data.h_in();
    const uint32_t W_in = tiling_data.w_in();
    const uint32_t D_out = tiling_data.d_out();
    const uint32_t H_out = tiling_data.h_out();
    const uint32_t W_out = tiling_data.w_out();
    const uint32_t totalOutLength = tiling_data.totalOutLength();

    __gm__ float* xPtr = (__gm__ float*)x;
    __gm__ float* yPtr = (__gm__ float*)y;

    uint32_t blkNum = AscendC::GetBlockNum();
    uint32_t blkIdx = AscendC::GetBlockIdx();

    // 块划分，最后一个块可能小于blockLen
    uint32_t blockLen = (totalOutLength + blkNum - 1) / blkNum;
    uint32_t start = blkIdx * blockLen;
    uint32_t end = start + blockLen;
    if (end > totalOutLength) end = totalOutLength;

    // 输入输出步长（连续内存，W为最内层）
    uint64_t inStrideW = 1;
    uint64_t inStrideH = W_in;
    uint64_t inStrideD = (uint64_t)H_in * (uint64_t)W_in;
    uint64_t inStrideC = (uint64_t)D_in * (uint64_t)H_in * (uint64_t)W_in;
    uint64_t inStrideN = (uint64_t)C * inStrideC;

    uint64_t outStrideW = 1;
    uint64_t outStrideH = W_out;
    uint64_t outStrideD = (uint64_t)H_out * (uint64_t)W_out;
    uint64_t outStrideC = (uint64_t)D_out * (uint64_t)H_out * (uint64_t)W_out;
    uint64_t outStrideN = (uint64_t)C * outStrideC;

    for (uint32_t p = start; p < end; ++p) {
        // 线性索引 -> (n,c,d,h,w)
        uint64_t rem = p;
        uint32_t n = rem / outStrideN; rem -= (uint64_t)n * outStrideN;
        uint32_t c = rem / outStrideC; rem -= (uint64_t)c * outStrideC;
        uint32_t od = rem / outStrideD; rem -= (uint64_t)od * outStrideD;
        uint32_t oh = rem / outStrideH; rem -= (uint64_t)oh * outStrideH;
        uint32_t ow = rem;

        // 对齐角点+倍率2 => 输出坐标是输入格点或其正中点
        uint32_t zd = od >> 1;
        uint32_t zh = oh >> 1;
        uint32_t zw = ow >> 1;
        uint32_t fd = od & 1;
        uint32_t fh = oh & 1;
        uint32_t fw = ow & 1;

        float wd = fd ? 0.5f : 1.0f;
        float wh = fh ? 0.5f : 1.0f;
        float ww = fw ? 0.5f : 1.0f;

        float acc = 0.0f;
        uint32_t dmax = fd ? 1u : 0u;
        uint32_t hmax = fh ? 1u : 0u;
        uint32_t wmax = fw ? 1u : 0u;

        for (uint32_t dz = 0; dz <= dmax; ++dz) {
            for (uint32_t dh = 0; dh <= hmax; ++dh) {
                for (uint32_t dw = 0; dw <= wmax; ++dw) {
                    uint64_t inIdx =
                        (uint64_t)n * inStrideN +
                        (uint64_t)c * inStrideC +
                        (uint64_t)(zd + dz) * inStrideD +
                        (uint64_t)(zh + dh) * inStrideH +
                        (uint64_t)(zw + dw) * inStrideW;
                    float v = xPtr[inIdx];
                    float wgt = (dz ? wd : (1.0f - (fd ? 1.0f : 0.0f)) + (fd ? 0.0f : 0.0f)); // 用wd表示两点平均时每个点0.5，否则1.0
                    // 更清晰：当fd=0时仅dz=0，权重1.0；当fd=1时dz=0/1各0.5
                    wgt = fd ? 0.5f : 1.0f;
                    float wgh = fh ? 0.5f : 1.0f;
                    float wgw = fw ? 0.5f : 1.0f;
                    float weight = 1.0f;
                    weight *= (fd ? 0.5f : 1.0f);
                    weight *= (fh ? 0.5f : 1.0f);
                    weight *= (fw ? 0.5f : 1.0f);
                    acc += v * weight;
                }
            }
        }
        yPtr[p] = acc;
    }
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor trilinear_upsample_custom_impl_npu(const at::Tensor& x) {
    TORCH_CHECK(x.dim() == 5, \"Input must be 5D (N, C, D, H, W)\");
    auto sizes = x.sizes();
    int64_t N = sizes[0];
    int64_t C = sizes[1];
    int64_t D_in = sizes[2];
    int64_t H_in = sizes[3];
    int64_t W_in = sizes[4];
    // align_corners=True, scale_factor=2
    int64_t D_out = (D_in - 1) * 2 + 1;
    int64_t H_out = (H_in - 1) * 2 + 1;
    int64_t W_out = (W_in - 1) * 2 + 1;

    at::Tensor y = at::empty({N, C, D_out, H_out, W_out}, x.options());
    EXEC_NPU_CMD(aclnnTrilinearUpsampleCustom, x, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"trilinear_upsample_custom\", &trilinear_upsample_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"trilinear_upsample_custom\", &trilinear_upsample_custom_impl_npu, \"Trilinear Upsample (scale=2, align_corners=True)\");
}
