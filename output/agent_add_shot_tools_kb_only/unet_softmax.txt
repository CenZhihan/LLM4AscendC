host_operator_src="""
#include "unet_softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/tensor.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    UnetSoftmaxCustomTilingData tiling;

    // 获取输入形状与总长度
    const gert::Shape* in_shape = context->GetInputShape(0);
    ge::Shape originShape = in_shape->GetOriginShape();
    uint32_t totalLength = originShape.GetShapeSize();

    // 设置BlockDim
    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);

    // 计算SoftMax所需的最小workspace并生成tiling
    // dataTypeSize: float = 4
    const uint32_t localWorkSpaceSize = AscendC::GetSoftMaxMinTmpSize(originShape, sizeof(float), false);
    AscendC::SoftMaxTilingFunc(originShape, sizeof(float), localWorkSpaceSize, tiling.softmaxTilingData());

    // 保存tiling数据
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // 申请workspace
    size_t* workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = static_cast<size_t>(localWorkSpaceSize);

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class UnetSoftmaxCustom : public OpDef {
public:
    explicit UnetSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(UnetSoftmaxCustom);
}

kernel_src="""
#include "kernel_operator.h"

class KernelUnetSoftmax {
public:
    __aicore__ inline KernelUnetSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, const optiling::UnetSoftmaxCustomTilingData& tilingData)
    {
        // 根据tiling获取M、K信息
        const AscendC::SoftMaxTiling& smTiling = tilingData.softmaxTilingData();
        this->m = smTiling.srcM;
        this->k = smTiling.srcK;
        this->elemNum = this->m * this->k;

        // 绑定全局内存
        xGm.SetGlobalBuffer((__gm__ float*)x, this->elemNum);
        yGm.SetGlobalBuffer((__gm__ float*)y, this->elemNum);
        this->tiling = smTiling;
    }

    __aicore__ inline void Process()
    {
        // 直接调用SoftMax高阶API，使用API框架分配的临时空间（workspace由host保留）
        // 输入为形状[m, k]，按最后一维进行归一化
        AscendC::SoftMax<float, false, false, false>(yGm, xGm, this->tiling);
    }

private:
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::SoftMaxTiling tiling;
    uint32_t m;
    uint32_t k;
    uint32_t elemNum;
};

extern "C" __global__ __aicore__ void unet_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelUnetSoftmax op;
    op.Init(x, y, tilingData);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib
import torch.nn as nn

class NpuSoftmaxLastDim(nn.Module):
    def __init__(self):
        super().__init__()
    def forward(self, x):
        return custom_ops_lib.unet_softmax_custom(x)

class DoubleConv(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.double_conv = nn.Sequential(
            nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(out_channels),
            NpuSoftmaxLastDim(),
            nn.Conv2d(out_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(out_channels),
            NpuSoftmaxLastDim()
        )

    def forward(self, x):
        return self.double_conv(x)

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, features):
        super(ModelNew, self).__init__()
        self.encoder1 = DoubleConv(in_channels, features)
        self.pool1 = nn.MaxPool2d(kernel_size=2, stride=2)
        self.encoder2 = DoubleConv(features, features * 2)
        self.pool2 = nn.MaxPool2d(kernel_size=2, stride=2)
        self.encoder3 = DoubleConv(features * 2, features * 4)
        self.pool3 = nn.MaxPool2d(kernel_size=2, stride=2)
        self.encoder4 = DoubleConv(features * 4, features * 8)
        self.pool4 = nn.MaxPool2d(kernel_size=2, stride=2)

        self.bottleneck = DoubleConv(features * 8, features * 16)

        self.upconv4 = nn.ConvTranspose2d(features * 16, features * 8, kernel_size=2, stride=2)
        self.decoder4 = DoubleConv(features * 16, features * 8)
        self.upconv3 = nn.ConvTranspose2d(features * 8, features * 4, kernel_size=2, stride=2)
        self.decoder3 = DoubleConv(features * 8, features * 4)
        self.upconv2 = nn.ConvTranspose2d(features * 4, features * 2, kernel_size=2, stride=2)
        self.decoder2 = DoubleConv(features * 4, features * 2)
        self.upconv1 = nn.ConvTranspose2d(features * 2, features, kernel_size=2, stride=2)
        self.decoder1 = DoubleConv(features * 2, features)

        self.final_conv = nn.Conv2d(features, out_channels, kernel_size=1)

    def forward(self, x):
        enc1 = self.encoder1(x)
        enc2 = self.encoder2(self.pool1(enc1))
        enc3 = self.encoder3(self.pool2(enc2))
        enc4 = self.encoder4(self.pool3(enc3))

        bottleneck = self.bottleneck(self.pool4(enc4))

        dec4 = self.upconv4(bottleneck)
        dec4 = torch.cat((dec4, enc4), dim=1)
        dec4 = self.decoder4(dec4)
        dec3 = self.upconv3(dec4)
        dec3 = torch.cat((dec3, enc3), dim=1)
        dec3 = self.decoder3(dec3)
        dec2 = self.upconv2(dec3)
        dec2 = torch.cat((dec2, enc2), dim=1)
        dec2 = self.decoder2(dec2)
        dec1 = self.upconv1(dec2)
        dec1 = torch.cat((dec1, enc1), dim=1)
        dec1 = self.decoder1(dec1)

        return self.final_conv(dec1)
'''
