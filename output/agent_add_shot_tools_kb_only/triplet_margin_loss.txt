kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 1;

class KernelTripletMarginLoss {
public:
    __aicore__ inline KernelTripletMarginLoss() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
                                uint32_t rows, uint32_t cols, uint32_t tileInner, float margin)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileInner = tileInner;
        this->margin = margin;

        // 按block划分样本行
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = this->rows / blockNum;
        this->rowStart = this->rowsPerBlock * blockIdx;
        if (blockIdx == blockNum - 1) {
            this->rowsThisBlock = this->rows - this->rowStart;
        } else {
            this->rowsThisBlock = this->rowsPerBlock;
        }

        // 设置GM指针偏移
        aGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(anchor) + static_cast<size_t>(rowStart) * cols,
                            static_cast<size_t>(rowsThisBlock) * cols);
        pGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(positive) + static_cast<size_t>(rowStart) * cols,
                            static_cast<size_t>(rowsThisBlock) * cols);
        nGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(negative) + static_cast<size_t>(rowStart) * cols,
                            static_cast<size_t>(rowsThisBlock) * cols);
        lossGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(loss) + rowStart, rowsThisBlock);

        // 初始化本地缓冲
        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileInner * sizeof(float));
        pipe.InitBuffer(inQueueP, BUFFER_NUM, this->tileInner * sizeof(float));
        pipe.InitBuffer(inQueueN, BUFFER_NUM, this->tileInner * sizeof(float));
        pipe.InitBuffer(tmpQueueVec, BUFFER_NUM, this->tileInner * sizeof(float)); // 中间向量
        pipe.InitBuffer(tmpQueueWork, BUFFER_NUM, this->tileInner * sizeof(float)); // ReduceSum工作区（示例按tile分配）
        pipe.InitBuffer(outQueueLoss, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsThisBlock; ++r) {
            float sumAP = 0.0f;
            float sumAN = 0.0f;

            for (uint32_t c = 0; c < cols; c += tileInner) {
                // 读取一段tileInner（假设cols能整除tileInner）
                AscendC::LocalTensor<float> aLocal = inQueueA.AllocTensor<float>();
                AscendC::LocalTensor<float> pLocal = inQueueP.AllocTensor<float>();
                AscendC::LocalTensor<float> nLocal = inQueueN.AllocTensor<float>();
                AscendC::DataCopy(aLocal, aGm[r * cols + c], tileInner);
                AscendC::DataCopy(pLocal, pGm[r * cols + c], tileInner);
                AscendC::DataCopy(nLocal, nGm[r * cols + c], tileInner);
                inQueueA.EnQue(aLocal);
                inQueueP.EnQue(pLocal);
                inQueueN.EnQue(nLocal);

                // (a - p)^2
                AscendC::LocalTensor<float> aLoc = inQueueA.DeQue<float>();
                AscendC::LocalTensor<float> pLoc = inQueueP.DeQue<float>();
                AscendC::LocalTensor<float> nLoc = inQueueN.DeQue<float>();
                AscendC::LocalTensor<float> tmpVec = tmpQueueVec.AllocTensor<float>();
                AscendC::LocalTensor<float> workBuf = tmpQueueWork.AllocTensor<float>();

                AscendC::Sub(tmpVec, aLoc, pLoc, tileInner);
                AscendC::Mul(tmpVec, tmpVec, tmpVec, tileInner);
                AscendC::ReduceSum(tmpVec, tmpVec, workBuf, tileInner);
                float partAP = AscendC::GetAccVal<float>();
                sumAP += partAP;

                // (a - n)^2
                AscendC::Sub(tmpVec, aLoc, nLoc, tileInner);
                AscendC::Mul(tmpVec, tmpVec, tmpVec, tileInner);
                AscendC::ReduceSum(tmpVec, tmpVec, workBuf, tileInner);
                float partAN = AscendC::GetAccVal<float>();
                sumAN += partAN;

                tmpQueueVec.FreeTensor(tmpVec);
                tmpQueueWork.FreeTensor(workBuf);
                inQueueA.FreeTensor(aLoc);
                inQueueP.FreeTensor(pLoc);
                inQueueN.FreeTensor(nLoc);
            }

            float distAP = sqrtf(sumAP);
            float distAN = sqrtf(sumAN);
            float l = distAP - distAN + margin;
            if (l < 0.0f) l = 0.0f;

            // 写回每个样本的loss到GM
            AscendC::LocalTensor<float> lossLocal = outQueueLoss.AllocTensor<float>();
            AscendC::Duplicate(lossLocal, l, 1);
            outQueueLoss.EnQue<float>(lossLocal);
            AscendC::LocalTensor<float> lossOut = outQueueLoss.DeQue<float>();
            AscendC::DataCopy(lossGm[r], lossOut, 1);
            outQueueLoss.FreeTensor(lossOut);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueA;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueP;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueN;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueueVec;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueueWork;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueLoss;

    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> pGm;
    AscendC::GlobalTensor<float> nGm;
    AscendC::GlobalTensor<float> lossGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tileInner;
    float margin;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowsThisBlock;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLoss op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.rows, tiling_data.cols, tiling_data.tileInner, tiling_data.margin);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor triplet_margin_loss_custom_impl_npu(const at::Tensor& anchor,
                                               const at::Tensor& positive,
                                               const at::Tensor& negative) {
    // 输出每个样本的loss，形状为 [N]
    auto opts = anchor.options();
    at::Tensor per_sample = at::empty({anchor.size(0)}, opts);
    EXEC_NPU_CMD(aclnnTripletMarginLossCustom, anchor, positive, negative, per_sample);
    // 返回mean作为标量loss
    return per_sample.mean();
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("triplet_margin_loss_custom", &triplet_margin_loss_custom_impl_npu);
}
