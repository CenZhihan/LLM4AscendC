namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    LambCustomTilingData tiling;
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();

    // 从属性读取学习率、epsilon、以及信任比（由前端传入）
    float lr = 0.001f;
    float eps = 1e-6f;
    float trust = 1.0f;
    if (context->GetAttrs() != nullptr) {
        auto attrs = context->GetAttrs();
        if (attrs->Find("lr") != nullptr) {
            lr = attrs->GetFloat("lr");
        }
        if (attrs->Find("eps") != nullptr) {
            eps = attrs->GetFloat("eps");
        }
        if (attrs->Find("trust") != nullptr) {
            trust = attrs->GetFloat("trust");
        }
    }

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_lr(lr);
    tiling.set_eps(eps);
    tiling.set_trust(trust);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // 本算子无需workspace

    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

// 数据类型定义
using DTYPE_PARAM = float;
using DTYPE_M = float;
using DTYPE_V = float;
using DTYPE_Z = float;

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数

class KernelLamb {
public:
    __aicore__ inline KernelLamb() {}
    __aicore__ inline void Init(GM_ADDR param, GM_ADDR m, GM_ADDR v, GM_ADDR out,
                                float lr, float eps, float trust,
                                uint32_t totalLength, uint32_t tileNum)
    {
        ASSERT(GetBlockNum() != 0 && "block dim can not be zero!");
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        ASSERT(tileNum != 0 && "tile num can not be zero!");
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        this->lr = lr;
        this->eps = eps;
        this->trust = trust;

        paramGm.SetGlobalBuffer((__gm__ DTYPE_PARAM*)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mGm.SetGlobalBuffer((__gm__ DTYPE_M*)m + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ DTYPE_V*)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z*)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueParam, BUFFER_NUM, this->tileLength * sizeof(DTYPE_PARAM));
        pipe.InitBuffer(inQueueM, BUFFER_NUM, this->tileLength * sizeof(DTYPE_M));
        pipe.InitBuffer(inQueueV, BUFFER_NUM, this->tileLength * sizeof(DTYPE_V));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_PARAM> paramLocal = inQueueParam.AllocTensor<DTYPE_PARAM>();
        AscendC::LocalTensor<DTYPE_M> mLocal = inQueueM.AllocTensor<DTYPE_M>();
        AscendC::LocalTensor<DTYPE_V> vLocal = inQueueV.AllocTensor<DTYPE_V>();

        AscendC::DataCopy(paramLocal, paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(mLocal, mGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(vLocal, vGm[progress * this->tileLength], this->tileLength);

        inQueueParam.EnQue(paramLocal);
        inQueueM.EnQue(mLocal);
        inQueueV.EnQue(vLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_PARAM> paramLocal = inQueueParam.DeQue<DTYPE_PARAM>();
        AscendC::LocalTensor<DTYPE_M> mLocal = inQueueM.DeQue<DTYPE_M>();
        AscendC::LocalTensor<DTYPE_V> vLocal = inQueueV.DeQue<DTYPE_V>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // denom = sqrt(v) + eps
        AscendC::Sqrt(zLocal, vLocal, this->tileLength);                          // zLocal = sqrt(v)
        AscendC::Adds(zLocal, zLocal, (DTYPE_Z)this->eps, this->tileLength);      // zLocal = sqrt(v) + eps

        // r = m / denom  -> reuse zLocal as r
        AscendC::Div(zLocal, mLocal, zLocal, this->tileLength);                   // zLocal = r

        // scale r by trust and lr: r = r * trust * lr
        AscendC::Muls(zLocal, zLocal, (DTYPE_Z)this->trust, this->tileLength);
        AscendC::Muls(zLocal, zLocal, (DTYPE_Z)this->lr, this->tileLength);

        // out = param - r_scaled
        AscendC::Sub(zLocal, paramLocal, zLocal, this->tileLength);               // zLocal = param - lr*trust*r

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueParam.FreeTensor(paramLocal);
        inQueueM.FreeTensor(mLocal);
        inQueueV.FreeTensor(vLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(outGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueParam;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueM;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueV;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_PARAM> paramGm;
    AscendC::GlobalTensor<DTYPE_M> mGm;
    AscendC::GlobalTensor<DTYPE_V> vGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;

    float lr;
    float eps;
    float trust;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void lamb_custom(GM_ADDR param, GM_ADDR m, GM_ADDR v, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelLamb op;
    op.Init(param, m, v, out,
            tilingData.lr, tilingData.eps, tilingData.trust,
            tilingData.totalLength, tilingData.tileNum);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

// 计算 trust_ratio = ||param||_2 / (||r||_2 + eps)，其中 r = m / (sqrt(v) + eps)
static inline float compute_trust_ratio(const at::Tensor& param, const at::Tensor& m,
                                        const at::Tensor& v, double eps) {
    // 确保在同一设备上计算
    auto denom = v.sqrt().add_(eps);
    auto r = m / denom;
    auto num = param.norm(2.0);
    auto den = r.norm(2.0).add_(eps);
    // 转为 CPU 标量
    float trust = (num / den).item<float>();
    return trust;
}

at::Tensor lamb_custom_impl_npu(const at::Tensor& param, const at::Tensor& m, const at::Tensor& v,
                                double lr, double eps) {
    at::Tensor out = at::empty_like(param);
    float trust = compute_trust_ratio(param, m, v, eps);
    // 通过属性将 lr/eps/trust 传递给后端（与主机侧 TilingFunc 对应）
    EXEC_NPU_CMD(aclnnLambCustom, param, m, v, lr, eps, (double)trust, out);
    return out;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("lamb_custom", &lamb_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("lamb_custom", &lamb_custom_impl_npu, "LAMB fused update");
}
