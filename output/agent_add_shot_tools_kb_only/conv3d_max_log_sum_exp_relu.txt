kernel_src="""
#include "kernel_operator.h"

class KernelLseRelu {
public:
    __aicore__ inline KernelLseRelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t spatialSize, uint32_t channels, uint32_t tileNum)
    {
        this->spatialSize = spatialSize;
        this->channels = channels;
        this->tileNum = tileNum;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint64_t start = (static_cast<uint64_t>(spatialSize) * blockIdx) / blockNum;
        uint64_t end   = (static_cast<uint64_t>(spatialSize) * (blockIdx + 1)) / blockNum;

        this->blockStart = static_cast<uint32_t>(start);
        this->blockLen   = static_cast<uint32_t>(end - start);
        if (this->blockLen == 0) {
            this->tileLen = 0;
        } else {
            this->tileLen = this->blockLen / (this->tileNum == 0 ? 1 : this->tileNum);
            if (this->tileLen == 0) this->tileLen = this->blockLen; // 至少覆盖一片
        }

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), static_cast<uint32_t>(spatialSize * channels));
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), spatialSize);

        // 分配本地缓冲：临时输入、通道最大值、exp 累加和
        pipe.InitBuffer(tmpQueue, 1, this->tileLen * sizeof(float));
        pipe.InitBuffer(maxQueue, 1, this->tileLen * sizeof(float));
        pipe.InitBuffer(sumQueue, 1, this->tileLen * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (tileLen == 0) {
            return;
        }
        for (uint32_t offset = 0; offset < blockLen; offset += tileLen) {
            uint32_t curLen = (offset + tileLen <= blockLen) ? tileLen : (blockLen - offset);
            ComputeOneTile(blockStart + offset, curLen);
        }
    }

private:
    __aicore__ inline void LoadChannelSlice(AscendC::LocalTensor<float>& dst, uint32_t c, uint32_t base, uint32_t len)
    {
        // 从第 c 个通道读取长度为 len 的切片，起始为 (c * spatialSize + base)
        AscendC::DataCopy(dst, xGm[c * spatialSize + base], len);
    }

    __aicore__ inline void ComputeOneTile(uint32_t base, uint32_t len)
    {
        // 第一次遍历：求每个空间位置的通道最大值 m = max_c x_c
        AscendC::LocalTensor<float> maxLocal = maxQueue.AllocTensor<float>();
        // 使用通道 0 初始化
        AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();
        LoadChannelSlice(tmpLocal, 0, base, len);
        AscendC::Copy(maxLocal, tmpLocal, len);

        for (uint32_t c = 1; c < channels; ++c) {
            LoadChannelSlice(tmpLocal, c, base, len);
            AscendC::Max(maxLocal, maxLocal, tmpLocal, len);
        }

        // 第二次遍历：sum_exp = sum_c exp(x_c - m)
        AscendC::LocalTensor<float> sumLocal = sumQueue.AllocTensor<float>();
        // 先用通道 0 初始化 sumLocal = exp(x0 - m)
        LoadChannelSlice(tmpLocal, 0, base, len);
        AscendC::Sub(tmpLocal, tmpLocal, maxLocal, len);
        AscendC::Exp(tmpLocal, tmpLocal, len);
        AscendC::Copy(sumLocal, tmpLocal, len);

        for (uint32_t c = 1; c < channels; ++c) {
            LoadChannelSlice(tmpLocal, c, base, len);
            AscendC::Sub(tmpLocal, tmpLocal, maxLocal, len);
            AscendC::Exp(tmpLocal, tmpLocal, len);
            AscendC::Add(sumLocal, sumLocal, tmpLocal, len);
        }

        // 输出：y = relu( ln(sum_exp) + m )
        AscendC::Ln(tmpLocal, sumLocal, len);           // tmp = ln(sum_exp)
        AscendC::Add(tmpLocal, tmpLocal, maxLocal, len);// tmp = ln(sum_exp) + m
        AscendC::Maxs(tmpLocal, tmpLocal, 0.0f, len);   // relu

        // y 写回（目标形状 [N,1,D,H,W] 的线性空间）
        AscendC::DataCopy(yGm[base], tmpLocal, len);

        // 释放本地张量
        sumQueue.FreeTensor(sumLocal);
        tmpQueue.FreeTensor(tmpLocal);
        maxQueue.FreeTensor(maxLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN,   1> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> maxQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> sumQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t spatialSize = 0;
    uint32_t channels = 0;
    uint32_t tileNum = 0;
    uint32_t tileLen = 0;
    uint32_t blockStart = 0;
    uint32_t blockLen = 0;
};

extern "C" __global__ __aicore__ void conv3d_max_log_sum_exp_relu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLseRelu op;
    op.Init(x, y, tiling_data.spatialSize, tiling_data.channels, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib
import torch.nn as nn

class ModelNew(torch.nn.Module):
    """
    优化版：保留 Conv3d 与 MaxPool3d 使用 PyTorch/NPU 实现，
    将通道维上的 logsumexp + ReLU 融合为自定义 AscendC 算子。
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.max_pool = nn.MaxPool3d(kernel_size=2, stride=2)

    def forward(self, x):
        x = self.conv(x)                      # [N, C, D, H, W]
        x = self.max_pool(x)                  # [N, C, D', H', W']
        x = custom_ops_lib.conv3d_max_log_sum_exp_relu_custom(x)  # [N, 1, D', H', W']
