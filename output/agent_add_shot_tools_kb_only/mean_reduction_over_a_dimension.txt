#include "mean_reduction_over_a_dimension_custom_tiling.h"

namespace optiling {

static inline uint32_t AlignUpToBytes(uint64_t bytes, uint32_t align) {
    return static_cast<uint32_t>((bytes + align - 1) / align * align);
}

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MeanReductionOverADimensionTilingData tiling;

    // Get input shape and dtype
    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin_shape = in_shape->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(origin_shape.GetShapeSize());
    int64_t rank = origin_shape.GetDimNum();
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }

    // Last axis length (we always reduce on the last axis in kernel)
    uint32_t n = static_cast<uint32_t>(origin_shape.GetDim(rank - 1));
    if (n == 0) {
        return ge::GRAPH_FAILED;
    }
    uint32_t outter = totalLength / n;

    // dtype size
    auto dt = context->GetInputTensor(0)->GetDataType();
    uint32_t dtypesize = (dt == ge::DT_FLOAT16) ? 2 : 4;

    // inner is 32-byte aligned in elements
    uint32_t inner = AlignUpToBytes(static_cast<uint64_t>(n) * dtypesize, 32) / dtypesize;

    // tmp size for Sum
    uint32_t maxValue = 0;
    uint32_t minValue = 0;
    AscendC::GetSumMaxMinTmpSize(n, dtypesize, false, maxValue, minValue);

    // fill tiling
    tiling.set_size(minValue);
    tiling.set_outter(outter);
    tiling.set_inner(inner);
    tiling.set_n(n);

    // set launch config
    context->SetBlockDim(8);

    // save tiling
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // no extra workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling


namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    auto origin = x_shape->GetOriginShape();
    int64_t rank = origin.GetDimNum();
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }
    // output is input with last dim reduced (removed)
    std::vector<int64_t> out_dims;
    out_dims.reserve(static_cast<size_t>(rank - 1));
    for (int64_t i = 0; i < rank - 1; ++i) {
        out_dims.push_back(origin.GetDim(i));
    }
    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDim(out_dims.data(), out_dims.size());
    return ge::GRAPH_SUCCESS;
}

{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

template <typename T>
class KernelMeanReduceLastDim
{
public:
    __aicore__ inline KernelMeanReduceLastDim() {}
    __aicore__ inline void Init(__gm__ uint8_t *xGm, __gm__ uint8_t *yGm, uint32_t tmpSize,
                                uint32_t outter, uint32_t inner, uint32_t n)
    {
        height = outter; // number of rows
        width = inner;   // aligned inner length
        nNum = n;        // actual inner length

        xGlobal.SetGlobalBuffer((__gm__ T *)xGm);
        yGlobal.SetGlobalBuffer((__gm__ T *)yGm);

        // Allocate local buffers
        pipe.InitBuffer(inQueueX, 1, static_cast<uint32_t>(height * width * sizeof(T)));
        // Output aligned to ONE_BLK_SIZE bytes
        uint32_t out_aligned_elems = ((height * sizeof(T) + AscendC::ONE_BLK_SIZE - 1) / AscendC::ONE_BLK_SIZE) * (AscendC::ONE_BLK_SIZE / sizeof(T));
        pipe.InitBuffer(outQueueY, 1, out_aligned_elems * sizeof(T));
        pipe.InitBuffer(workQueue, 1, tmpSize);
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX.AllocTensor<T>();
        AscendC::DataCopy(xLocal, xGlobal, height * width);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX.DeQue<T>();
        AscendC::LocalTensor<uint8_t> workLocal = workQueue.AllocTensor<uint8_t>();
        AscendC::LocalTensor<T> yLocal = outQueueY.AllocTensor<T>();

        AscendC::SumParams params{height, width, nNum};

        // Initialize output buffer to 0 (aligned size)
        uint32_t out_aligned_elems = ((height * sizeof(T) + AscendC::ONE_BLK_SIZE - 1) / AscendC::ONE_BLK_SIZE) * (AscendC::ONE_BLK_SIZE / sizeof(T));
        T zero(0);
        AscendC::Duplicate<T>(yLocal, zero, out_aligned_elems);

        // Reduce sum over last dimension
        AscendC::Sum(yLocal, xLocal, params);

        // Compute mean: yLocal = yLocal * (1.0 / nNum)
        T scale = static_cast<T>(1.0f / static_cast<float>(nNum));
        AscendC::Muls(yLocal, yLocal, scale, height);

        outQueueY.EnQue<T>(yLocal);
        workQueue.FreeTensor(workLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<T> yLocal = outQueueY.DeQue<T>();
        // Only copy valid 'height' elements (results per row)
        AscendC::DataCopy(yGlobal, yLocal, height);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> workQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<T> xGlobal, yGlobal;

    uint32_t width = 0;
    uint32_t height = 0;
    uint32_t nNum = 0;
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReduceLastDim<DTYPE_X> op;
    op.Init((__gm__ uint8_t*)x, (__gm__ uint8_t*)y,
            tiling_data.size, tiling_data.outter, tiling_data.inner, tiling_data.n);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that performs mean reduction over a specific dimension using a custom AscendC kernel.
    The custom kernel reduces along the last dimension. We permute input so that the target dim becomes last.
    """
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Normalize dim to non-negative
        d = self.dim if self.dim >= 0 else self.dim + x.dim()
        assert 0 <= d < x.dim(), "Invalid reduction dimension"

        # Move the target dimension to the last position
        dims = list(range(x.dim()))
        new_order = [i for i in dims if i != d] + [d]
        x_perm = x.permute(new_order).contiguous()

        # Call custom kernel that reduces along last dimension
        y = custom_ops_lib.mean_reduction_over_a_dimension_custom(x_perm)

        # The output already has the correct order of remaining axes
        return y
