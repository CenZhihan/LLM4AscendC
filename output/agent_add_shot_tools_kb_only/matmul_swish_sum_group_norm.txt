kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = float;
using B_T = float;
using C_T = float;
using BiasT = float;

constexpr int32_t VEC_BUFFER_NUM = 2; // for post-ops streaming

class KernelMatmulSwishSum {
public:
    __aicore__ inline KernelMatmulSwishSum() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias_linear, GM_ADDR extra_bias,
                                GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGM)
    {
        // Load TCubeTiling from GM
        auto tilingGMU32 = reinterpret_cast<__gm__ uint32_t*>(tilingGM);
        auto tilingLocalU32 = reinterpret_cast<uint32_t*>(&tiling_);
        for (int i = 0; i < static_cast<int>(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i) {
            tilingLocalU32[i] = tilingGMU32[i];
        }

        // Global buffers for A, B, C, bias
        // Set intermediate C to y directly, then do post-ops in-place
        aGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(x),
                                 tiling_.singleCoreM * tiling_.singleCoreK);
        bGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(w),
                                 tiling_.singleCoreK * tiling_.singleCoreN);
        cGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(y),
                                 tiling_.singleCoreM * tiling_.singleCoreN);
        biasLinearGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ BiasT*>(bias_linear),
                                          tiling_.singleCoreN);
        extraBiasGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ BiasT*>(extra_bias),
                                         tiling_.singleCoreN);

        // Initialize queue buffers for post-ops
        postPipe_.InitBuffer(inQueue_,  VEC_BUFFER_NUM, kTileElems_ * sizeof(C_T));
        postPipe_.InitBuffer(outQueue_, VEC_BUFFER_NUM, kTileElems_ * sizeof(C_T));
        postPipe_.InitBuffer(biasQueue_,VEC_BUFFER_NUM, kTileElems_ * sizeof(BiasT));
    }

    __aicore__ inline void ComputeMatmul()
    {
        // Define Matmul types
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

        // Matmul config: NORMAL layout, batch-less-than-L1 for safety
        constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);

        matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm;
        AscendC::TPipe pipe;
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);

        mm.Init(&tiling_);
        mm.SetTensorA(aGlobal_, false);  // A not transposed
        mm.SetTensorB(bGlobal_, true);   // B is transposed layout (weight: [N, K])
        mm.SetWorkspace(nullptr, 0);
        mm.SetBias(biasLinearGlobal_);

        // Execute Matmul (write to cGlobal_)
        // Use IterateAll for full compute under NORMAL layout
        mm.IterateAll(cGlobal_);
    }

    __aicore__ inline void PostProcessSwishAndAdd()
    {
        // After Matmul, cGlobal_ holds [M_sub, N_full] of this core.
        // Stream by a 1D tile along N dimension per row.
        const uint32_t singleM = tiling_.singleCoreM;
        const uint32_t singleN = tiling_.singleCoreN;
        const float swishBeta = 1.0f;

        // Process row by row
        for (uint32_t m = 0; m < singleM; ++m) {
            uint32_t rowBase = m * singleN;
            // Tile across N
            for (uint32_t c = 0; c < singleN; c += kTileElems_) {
                uint32_t len = AscendC::min(kTileElems_, singleN - c);

                // Copy in Matmul result tile
                AscendC::LocalTensor<C_T> inLocal  = inQueue_.AllocTensor<C_T>();
                AscendC::DataCopy(inLocal, cGlobal_[rowBase + c], len);
                inQueue_.EnQue(inLocal);

                // Swish
                AscendC::LocalTensor<C_T> inLocal2 = inQueue_.DeQue<C_T>();
                AscendC::LocalTensor<C_T> outLocal = outQueue_.AllocTensor<C_T>();
                Swish<C_T>(outLocal, inLocal2, len, swishBeta);
                inQueue_.FreeTensor(inLocal2);

                // Load extra bias slice (per-channel, broadcast on batch)
                AscendC::LocalTensor<BiasT> biasLocal = biasQueue_.AllocTensor<BiasT>();
                AscendC::DataCopy(biasLocal, extraBiasGlobal_[c], len);
                biasQueue_.EnQue(biasLocal);
                AscendC::LocalTensor<BiasT> biasLocal2 = biasQueue_.DeQue<BiasT>();

                // z = swish_out + extra_bias
                AscendC::Add(outLocal, outLocal, biasLocal2, len);
                biasQueue_.FreeTensor(biasLocal2);

                // Write back
                AscendC::LocalTensor<C_T> outLocal2 = outQueue_.DeQue<C_T>();
                AscendC::DataCopy(cGlobal_[rowBase + c], outLocal2, len);
                outQueue_.FreeTensor(outLocal2);
            }
        }
    }

private:
    // Post-ops tile length (elements)
    static constexpr uint32_t kTileElems_ = 1024;

    // Tiling
    TCubeTiling tiling_;

    // Global tensors
    AscendC::GlobalTensor<A_T> aGlobal_;
    AscendC::GlobalTensor<B_T> bGlobal_;
    AscendC::GlobalTensor<C_T> cGlobal_;
    AscendC::GlobalTensor<BiasT> biasLinearGlobal_;
    AscendC::GlobalTensor<BiasT> extraBiasGlobal_;

    // Vector pipeline for post-ops
    AscendC::TPipe postPipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  VEC_BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, VEC_BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN,  VEC_BUFFER_NUM> biasQueue_;
};

extern "C" __global__ __aicore__ void matmul_swish_sum_group_norm_custom(GM_ADDR x, GM_ADDR weight,
                                                                          GM_ADDR bias_linear, GM_ADDR extra_bias,
                                                                          GM_ADDR y,
                                                                          GM_ADDR workspace, GM_ADDR tilingGM)
{
    set_atomic_none();
    KernelMatmulSwishSum op;
    op.Init(x, weight, bias_linear, extra_bias, y, workspace, tilingGM);
    op.ComputeMatmul();
    op.PostProcessSwishAndAdd();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized: fused Matmul + Swish + Sum by custom AscendC kernel,
    followed by GroupNorm in PyTorch.
    """
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super().__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features)

    def forward(self, x):
        # fused: (x @ W^T + bias_linear) -> swish -> + extra_bias
        y = custom_ops_lib.matmul_swish_sum_group_norm_custom(
            x, self.matmul.weight, self.matmul.bias, self.bias
        )
        # GroupNorm stays in PyTorch for stability and simplicity
        y = self.group_norm(y)
        return y
