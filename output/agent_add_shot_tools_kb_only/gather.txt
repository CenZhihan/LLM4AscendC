#include "gather_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GatherCustomTilingData tiling;

    const gert::Shape* xShape = context->GetInputShape(0);
    const gert::Shape* idxShape = context->GetInputShape(1);

    // 假设输入为 [N, C]，按 dim=1 聚合
    uint32_t rows = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t srcWidth = static_cast<uint32_t>(xShape->GetDim(1));
    uint32_t idxWidth = static_cast<uint32_t>(idxShape->GetDim(1));

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_rows(rows);
    tiling.set_srcWidth(srcWidth);
    tiling.set_idxWidth(idxWidth);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    context->SetTilingKey(1);
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // 输出形状与 idx 相同
    const gert::Shape* idxShape = context->GetInputShape(1);
    gert::Shape* yShape = context->GetOutputShape(0);
    *yShape = *idxShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // 输出 dtype 与 x 相同
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Y = float;
using ITYPE_IDX = uint32_t;

constexpr int32_t BUFFER_NUM = 2; // 每个队列的张量数

class KernelGather {
public:
    __aicore__ inline KernelGather() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR y,
                                uint32_t rows, uint32_t srcWidth, uint32_t idxWidth, uint32_t tileNum)
    {
        this->rows = rows;
        this->srcWidth = srcWidth;
        this->idxWidth = idxWidth;
        this->tileNum = tileNum;

        this->rowsPerBlock = rows / AscendC::GetBlockNum();
        this->tileLength = this->idxWidth / this->tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->rowsPerBlock * this->srcWidth * AscendC::GetBlockIdx(),
                            this->rowsPerBlock * this->srcWidth);
        idxGm.SetGlobalBuffer((__gm__ ITYPE_IDX*)idx + this->rowsPerBlock * this->idxWidth * AscendC::GetBlockIdx(),
                            this->rowsPerBlock * this->idxWidth);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->rowsPerBlock * this->idxWidth * AscendC::GetBlockIdx(),
                            this->rowsPerBlock * this->idxWidth);

        // 缓冲初始化
        pipe.InitBuffer(inQueueXRow, 1, this->srcWidth * sizeof(DTYPE_X));              // 每次载入一整行 x
        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, this->tileLength * sizeof(ITYPE_IDX));  // 索引分块
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));     // 输出分块
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            // 拷贝一行 x 到 UB
            AscendC::LocalTensor<DTYPE_X> xRowLocal = inQueueXRow.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xRowLocal, xGm[r * this->srcWidth], this->srcWidth);

            int32_t loopCount = this->tileNum * BUFFER_NUM;
            for (int32_t i = 0; i < loopCount; ++i) {
                // 拷贝一段 idx 到 UB（int32）
                AscendC::LocalTensor<ITYPE_IDX> idxLocal = inQueueIdx.AllocTensor<ITYPE_IDX>();
                AscendC::DataCopy(idxLocal, idxGm[r * this->idxWidth + i * this->tileLength], this->tileLength);
                inQueueIdx.EnQue(idxLocal);

                // 计算：将 idx(元素索引)转换为字节偏移，并使用 Gather
                AscendC::LocalTensor<ITYPE_IDX> offsetLocal = inQueueIdx.DeQue<ITYPE_IDX>();
                // 元素大小为4字节，左移2位得到字节偏移（计 this->tileLength 个元素）
                // 使用向量左移标量的基础指令
                AscendC::ShiftLeft(offsetLocal, offsetLocal, 2, this->tileLength);

                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
                // 从 xRowLocal 基址(0)按偏移表聚合 count = tileLength
                AscendC::Gather(yLocal, xRowLocal, offsetLocal, (uint32_t)0, this->tileLength);

                outQueueY.EnQue<DTYPE_Y>(yLocal);
                inQueueIdx.FreeTensor(offsetLocal);

                // 写回对应输出段
                AscendC::LocalTensor<DTYPE_Y> yOutLocal = outQueueY.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[r * this->idxWidth + i * this->tileLength], yOutLocal, this->tileLength);
                outQueueY.FreeTensor(yOutLocal);
            }
            inQueueXRow.FreeTensor(xRowLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueXRow;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<ITYPE_IDX> idxGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows;
    uint32_t srcWidth;
    uint32_t idxWidth;
    uint32_t tileNum;

    uint32_t rowsPerBlock;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gather_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelGather op;
    op.Init(x, idx, y, tilingData.rows, tilingData.srcWidth, tilingData.idxWidth, tilingData.tileNum);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}

at::Tensor gather_custom_impl_npu(const at::Tensor& x, const at::Tensor& idx) {
    at::Tensor idx_i32 = idx.to(at::kInt);
    at::Tensor y = at::empty({idx.size(0), idx.size(1)}, x.options());

    return y;
}

    m.impl("gather_custom", &gather_custom_impl_npu);
}

    m.def("gather_custom", &gather_custom_impl_npu, "gather along dim=1 with indices");
