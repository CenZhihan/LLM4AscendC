kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Z = float;

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数

class KernelReduceProd {
public:
    __aicore__ inline KernelReduceProd() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t reduceLen, uint32_t lastAxis, uint32_t tileNum)
    {
        this->batch = batch;
        this->reduceLen = reduceLen;
        this->lastAxis = lastAxis;
        this->tileNum = tileNum;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        // 按batch维在多核间划分
        this->batchesPerBlock = (batch + blockNum - 1) / blockNum;
        this->blockStartB = blockIdx * this->batchesPerBlock;
        if (this->blockStartB > batch) this->blockStartB = batch;
        uint32_t blockEndB = this->blockStartB + this->batchesPerBlock;
        if (blockEndB > batch) blockEndB = batch;
        this->blockBatch = blockEndB > this->blockStartB ? (blockEndB - this->blockStartB) : 0;

        // 设置全局内存窗口
        uint64_t xOffset = static_cast<uint64_t>(this->blockStartB) * this->reduceLen * this->lastAxis;
        uint64_t yOffset = static_cast<uint64_t>(this->blockStartB) * this->lastAxis;
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + xOffset, this->blockBatch * this->reduceLen * this->lastAxis);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + yOffset, this->blockBatch * this->lastAxis);

        // 计算tileLength，处理尾块
        uint32_t totalTiles = this->tileNum * BUFFER_NUM;
        this->tileLength = (this->lastAxis + totalTiles - 1) / totalTiles;
        if (this->tileLength == 0) this->tileLength = this->lastAxis;

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (this->blockBatch == 0) {
            return;
        }
        for (uint32_t b = 0; b < this->blockBatch; ++b) {
            for (uint32_t progress = 0; ; ++progress) {
                uint32_t nOffset = progress * this->tileLength;
                if (nOffset >= this->lastAxis) break;
                uint32_t curLen = this->lastAxis - nOffset;
                if (curLen > this->tileLength) curLen = this->tileLength;

                // 累乘缓冲
                AscendC::LocalTensor<DTYPE_Z> accLocal = outQueueY.AllocTensor<DTYPE_Z>();
                // 首片：复制到累乘缓冲
                {
                    AscendC::LocalTensor<DTYPE_X> tmpLocal = inQueueX.AllocTensor<DTYPE_X>();
                    uint64_t xSliceOff = (static_cast<uint64_t>(b) * this->reduceLen + 0) * this->lastAxis + nOffset;
                    AscendC::DataCopy(tmpLocal, xGm[xSliceOff], curLen);
                    // 将首片复制到累乘缓冲
                    AscendC::DataCopy(accLocal, tmpLocal, curLen);
                    inQueueX.FreeTensor(tmpLocal);
                }
                // 后续片：逐片相乘
                for (uint32_t m = 1; m < this->reduceLen; ++m) {
                    AscendC::LocalTensor<DTYPE_X> tmpLocal = inQueueX.AllocTensor<DTYPE_X>();
                    uint64_t xSliceOff = (static_cast<uint64_t>(b) * this->reduceLen + m) * this->lastAxis + nOffset;
                    AscendC::DataCopy(tmpLocal, xGm[xSliceOff], curLen);
                    AscendC::Mul(accLocal, accLocal, tmpLocal, curLen);
                    inQueueX.FreeTensor(tmpLocal);
                }
                outQueueY.EnQue<DTYPE_Z>(accLocal);

                // 写回输出
                AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
                uint64_t yOff = static_cast<uint64_t>(b) * this->lastAxis + nOffset;
                AscendC::DataCopy(yGm[yOff], yLocal, curLen);
                outQueueY.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t batch;
    uint32_t reduceLen;
    uint32_t lastAxis;
    uint32_t tileNum;

    uint32_t batchesPerBlock;
    uint32_t blockStartB;
    uint32_t blockBatch;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void product_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelReduceProd op;
    op.Init(x, y,
            tiling_data.batch,
            tiling_data.reduceLen,
            tiling_data.lastAxis,
            tiling_data.tileNum);
    op.Process();
}
