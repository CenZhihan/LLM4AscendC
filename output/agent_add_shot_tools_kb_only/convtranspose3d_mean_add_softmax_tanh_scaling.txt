kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelConvtranspose3dMeanAddSoftmaxTanhScaling {
public:
    __aicore__ inline KernelConvtranspose3dMeanAddSoftmaxTanhScaling() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR y, uint32_t B, uint32_t C, uint32_t H, uint32_t W)
    {
        this->B = B;
        this->C = C;
        this->H = H;
        this->W = W;

        // Each block processes one batch index if possible
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        uint32_t batchPerBlock = (B + blocks - 1) / blocks;
        startB = idx * batchPerBlock;
        endB = AscendC::Min(startB + batchPerBlock, B);

        uint64_t hw = (uint64_t)H * (uint64_t)W;
        uint64_t perBatchSize = (uint64_t)C * hw; // D already reduced to 1

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x + startB * perBatchSize, (endB - startB) * perBatchSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y + startB * perBatchSize, (endB - startB) * perBatchSize);
        biasGm.SetGlobalBuffer((__gm__ DTYPE *)bias, C);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE *)scale, 1);

        // Local buffers
        // Channel vector per (h,w): length C
        pipe.InitBuffer(vecQueue, 1, C * sizeof(DTYPE));
        pipe.InitBuffer(tmpQueue, 1, C * sizeof(DTYPE));
        // For SoftMax sum/max temporary
        elementNumPerBlk = 32 / sizeof(DTYPE); // 8 for float
        pipe.InitBuffer(sumQueue, 1, elementNumPerBlk * sizeof(DTYPE));
        pipe.InitBuffer(maxQueue, 1, elementNumPerBlk * sizeof(DTYPE));
        pipe.InitBuffer(biasQueue, 1, C * sizeof(DTYPE));
        pipe.InitBuffer(scaleQueue, 1, sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // Load bias once per block
        AscendC::LocalTensor<DTYPE> biasLocal = biasQueue.AllocTensor<DTYPE>();
        AscendC::DataCopy(biasLocal, biasGm, C);
        biasQueue.EnQue(biasLocal);

        AscendC::LocalTensor<DTYPE> scaleLocal = scaleQueue.AllocTensor<DTYPE>();
        AscendC::DataCopy(scaleLocal, scaleGm, 1);
        scaleQueue.EnQue(scaleLocal);

        uint64_t hw = (uint64_t)H * (uint64_t)W;
        for (uint32_t b = startB; b < endB; ++b) {
            uint64_t batchOffset = (uint64_t)(b - startB) * (uint64_t)C * hw;
            for (uint32_t h = 0; h < H; ++h) {
                for (uint32_t w = 0; w < W; ++w) {
                    uint64_t pos = (uint64_t)h * W + w;

                    // CopyIn: assemble channel vector for this (b,h,w)
                    AscendC::LocalTensor<DTYPE> vecLocal = vecQueue.AllocTensor<DTYPE>();
                    for (uint32_t c = 0; c < C; ++c) {
                        uint64_t off = batchOffset + (uint64_t)c * hw + pos;
                        AscendC::DataCopy(vecLocal[c], xGm[off], 1);
                    }
                    vecQueue.EnQue(vecLocal);

                    // Compute: bias add -> softmax over channels -> tanh -> scaling
                    AscendC::LocalTensor<DTYPE> inVec = vecQueue.DeQue<DTYPE>();
                    AscendC::LocalTensor<DTYPE> tmpVec = tmpQueue.AllocTensor<DTYPE>();
                    AscendC::LocalTensor<DTYPE> outVec = vecQueue.AllocTensor<DTYPE>(); // reuse vecQueue buffer for output
                    AscendC::LocalTensor<DTYPE> biasLocal2 = biasQueue.DeQue<DTYPE>();
                    AscendC::LocalTensor<DTYPE> scaleLocal2 = scaleQueue.DeQue<DTYPE>();

                    // Bias add
                    AscendC::Add(tmpVec, inVec, biasLocal2, C);

                    // SoftMax over last axis of size C (m=1, n=C)
                    AscendC::LocalTensor<DTYPE> sumTempLocal = sumQueue.AllocTensor<DTYPE>();
                    AscendC::LocalTensor<DTYPE> maxTempLocal = maxQueue.AllocTensor<DTYPE>();
                    AscendC::SoftMaxShapeInfo shapeInfo = {1, C, 1, C};
                    AscendC::SoftMax<DTYPE>(outVec, sumTempLocal, maxTempLocal, tmpVec, softmaxTiling, shapeInfo);

                    // Tanh via exp: tanh(x) = (e^(2x) - e^(-2x)) / (e^(2x) + e^(-2x))
                    AscendC::Mul(tmpVec, outVec, (DTYPE)2.0, C);              // tmp = 2x
                    AscendC::Exp(tmpVec, tmpVec, C);                          // tmp = e^(2x)
                    // e^(-2x) = 1 / e^(2x)
                    AscendC::LocalTensor<DTYPE> invTmp = maxQueue.AllocTensor<DTYPE>(); // reuse
                    AscendC::Reciprocal(invTmp, tmpVec, C);                    // inv = e^(-2x)
                    AscendC::Sub(outVec, tmpVec, invTmp, C);                   // out = e^(2x) - e^(-2x)
                    AscendC::Add(tmpVec, tmpVec, invTmp, C);                   // tmp = e^(2x) + e^(-2x)
                    AscendC::Div(outVec, outVec, tmpVec, C);                   // out = tanh

                    // Scaling
                    // Broadcast scalar scaleLocal2[0]
                    AscendC::Mul(outVec, outVec, scaleLocal2[0], C);

                    // Enqueue and free temps
                    maxQueue.FreeTensor(invTmp);
                    maxQueue.FreeTensor(maxTempLocal);
                    sumQueue.FreeTensor(sumTempLocal);
                    tmpQueue.FreeTensor(tmpVec);
                    scaleQueue.EnQue(scaleLocal2);
                    biasQueue.EnQue(biasLocal2);
                    vecQueue.EnQue(outVec);

                    // CopyOut: scatter back to GM
                    AscendC::LocalTensor<DTYPE> resVec = vecQueue.DeQue<DTYPE>();
                    for (uint32_t c = 0; c < C; ++c) {
                        uint64_t off = batchOffset + (uint64_t)c * hw + pos;
                        AscendC::DataCopy(yGm[off], resVec[c], 1);
                    }
                    vecQueue.FreeTensor(inVec);
                    vecQueue.FreeTensor(resVec);
                }
            }
        }

        // Free cached tensors
        AscendC::LocalTensor<DTYPE> biasTmp = biasQueue.DeQue<DTYPE>();
        biasQueue.FreeTensor(biasTmp);
        AscendC::LocalTensor<DTYPE> scaleTmp = scaleQueue.DeQue<DTYPE>();
        scaleQueue.FreeTensor(scaleTmp);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> vecQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> sumQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> maxQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scaleQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;
    AscendC::GlobalTensor<DTYPE> biasGm;
    AscendC::GlobalTensor<DTYPE> scaleGm;

    uint32_t B, C, H, W;
    uint32_t startB, endB;
    uint32_t elementNumPerBlk = 0;
    AscendC::SoftMaxTiling softmaxTiling; // use default; API will check and compute as needed
};

extern "C" __global__ __aicore__ void convtranspose3d_mean_add_softmax_tanh_scaling_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvtranspose3dMeanAddSoftmaxTanhScaling op;
    op.Init(x, bias, scale, y, tiling_data.batch, tiling_data.channels, tiling_data.height, tiling_data.width);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized Model:
    - Keep ConvTranspose3d in PyTorch
    - Keep mean over depth in PyTorch
    - Fuse bias add + softmax (channel dim) + tanh + scaling into a custom AscendC kernel
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, scaling_factor):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.bias = torch.nn.Parameter(torch.randn(1, out_channels, 1, 1, 1))
        # pass scaling as 1-element tensor to device for kernel
        self.register_buffer("scaling_tensor", torch.tensor([float(scaling_factor)], dtype=torch.float32))

    def forward(self, x):
        x = self.conv_transpose(x)                # (B, C, D, H, W)
        x = x.mean(dim=2, keepdim=True)           # (B, C, 1, H, W)
        # Custom fused kernel: bias add + softmax over channels + tanh + scaling
        return custom_ops_lib.convtranspose3d_mean_add_softmax_tanh_scaling_custom(x, self.bias, self.scaling_tensor)
