    int64_t N = xShape->GetOriginShape().GetDim(0);
    int64_t Cin = xShape->GetOriginShape().GetDim(1);
    int64_t Hin = xShape->GetOriginShape().GetDim(2);
    int64_t Win = xShape->GetOriginShape().GetDim(3);

    int64_t Kw = wShape->GetOriginShape().GetDim(3);
    int64_t CoutPerGroup = wShape->GetOriginShape().GetDim(1);

    int64_t Hout = (Hin - 1) * stride[0] - 2 * padding[0] + (Kh - 1) * dilation[0] + 1;

kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE_X;
typedef float DTYPE_W;
typedef float DTYPE_B;
typedef float DTYPE_Y;

class KernelConvTranspose2d {
public:
    __aicore__ inline KernelConvTranspose2d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t hin, uint32_t win, uint32_t hout, uint32_t wout,
                                uint32_t kh, uint32_t kw,
                                int32_t stride_h, int32_t stride_w,
                                int32_t pad_h, int32_t pad_w,
                                int32_t dil_h, int32_t dil_w,
                                int32_t groups, uint32_t has_bias, uint64_t totalOutput)
    {
        this->N = n; this->Cin = cin; this->Cout = cout;
        this->Hin = hin; this->Win = win; this->Hout = hout; this->Wout = wout;
        this->Kh = kh; this->Kw = kw;
        this->stride_h = stride_h; this->stride_w = stride_w;
        this->pad_h = pad_h; this->pad_w = pad_w;
        this->dil_h = dil_h; this->dil_w = dil_w;
        this->groups = groups; this->has_bias = has_bias;
        this->totalOutput = totalOutput;

        xPtr = (__gm__ DTYPE_X*)x;
        wPtr = (__gm__ DTYPE_W*)w;
        bPtr = (__gm__ DTYPE_B*)b;
        yPtr = (__gm__ DTYPE_Y*)y;

        // per-block range
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();
        uint64_t base = (totalOutput / blocks) * bid;
        uint64_t rem = totalOutput % blocks;
        // Distribute the remainder to the first 'rem' blocks
        uint64_t start = base + ((uint64_t) (bid < rem ? bid : rem));
        uint64_t myLen = (totalOutput / blocks) + (bid < rem ? 1 : 0);
        this->startIdx = start;
        this->endIdx = start + myLen;
    }

    __aicore__ inline void Process()
    {
        // Initialize output slice to zero
        for (uint64_t idx = startIdx; idx < endIdx; ++idx) {
            yPtr[idx] = (DTYPE_Y)0.0f;
        }

        // Compute output elements assigned to this block
        for (uint64_t linear = startIdx; linear < endIdx; ++linear) {
            // Decode linear index into (n, co, oh, ow)
            uint64_t tmp = linear;
            uint64_t ow = tmp % Wout; tmp /= Wout;
            uint64_t oh = tmp % Hout; tmp /= Hout;
            uint64_t co = tmp % Cout; tmp /= Cout;
            uint64_t n = tmp;

            int32_t g = (int32_t)(co / (Cout / groups));
            uint32_t cout_per_g = Cout / groups;
            uint32_t cin_per_g = Cin / groups;
            uint32_t co_in_g = co % cout_per_g;

            float acc = 0.0f;
            // Iterate over kernel and input channels in this group
            for (uint32_t ci_in_g = 0; ci_in_g < cin_per_g; ++ci_in_g) {
                // Real input channel index
                uint32_t ci = g * cin_per_g + ci_in_g;
                for (uint32_t kh = 0; kh < Kh; ++kh) {
                    // Compute corresponding input row index
                    // oh = ih * stride_h - pad_h + kh * dil_h  => ih = (oh + pad_h - kh*dil_h)
                    int32_t ih_nominal = (int32_t)oh + pad_h - (int32_t)kh * dil_h;
                    if (ih_nominal % stride_h != 0) continue;
                    int32_t ih = ih_nominal / stride_h;
                    if (ih < 0 || ih >= (int32_t)Hin) continue;

                    for (uint32_t kw = 0; kw < Kw; ++kw) {
                        int32_t iw_nominal = (int32_t)ow + pad_w - (int32_t)kw * dil_w;
                        if (iw_nominal % stride_w != 0) continue;
                        int32_t iw = iw_nominal / stride_w;
                        if (iw < 0 || iw >= (int32_t)Win) continue;

                        // Load input element x[n, ci, ih, iw]
                        uint64_t x_off = ((((uint64_t)n * Cin + ci) * Hin + (uint64_t)ih) * Win + (uint64_t)iw);
                        DTYPE_X x_val = xPtr[x_off];

                        // Load weight element w[ci, co_in_g, kh, kw]
                        uint64_t w_off = ((((uint64_t)ci) * (Cout / groups) + co_in_g) * Kh + kh) * Kw + kw;
                        DTYPE_W w_val = wPtr[w_off];

                        acc += (float)x_val * (float)w_val;
                    }
                }
            }

            if (has_bias) {
                uint64_t b_off = co;
                acc += (float)bPtr[b_off];
            }

            yPtr[linear] = (DTYPE_Y)acc;
        }
    }

private:
    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_W* wPtr;
    __gm__ DTYPE_B* bPtr;
    __gm__ DTYPE_Y* yPtr;

    uint32_t N, Cin, Cout, Hin, Win, Hout, Wout, Kh, Kw;
    int32_t stride_h, stride_w, pad_h, pad_w, dil_h, dil_w, groups;
    uint32_t has_bias;
    uint64_t totalOutput;
    uint64_t startIdx, endIdx;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose2d op;
    op.Init(
        x, weight, bias, y,
        tiling_data.n, tiling_data.cin, tiling_data.cout,
        tiling_data.hin, tiling_data.win, tiling_data.hout, tiling_data.wout,
        tiling_data.kh, tiling_data.kw,
        tiling_data.stride_h, tiling_data.stride_w,
        tiling_data.pad_h, tiling_data.pad_w,
        tiling_data.dil_h, tiling_data.dil_w,
        tiling_data.groups, tiling_data.has_bias, tiling_data.totalOutput
    );
    op.Process();
}
