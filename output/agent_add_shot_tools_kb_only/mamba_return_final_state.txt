kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMambaReturnFinalState {
public:
    __aicore__ inline KernelMambaReturnFinalState() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t B, uint32_t C, uint32_t H, uint32_t PN, uint32_t tileNum)
    {
        this->B = B;
        this->C = C;
        this->H = H;
        this->PN = PN;
        this->tileNum = tileNum;

        // 分块到各个核心：以 (B*H) 为并行维度
        totalBH = B * H;
        coreNum = AscendC::GetBlockNum();
        coreIdx = AscendC::GetBlockIdx();
        bhPerCore = (totalBH + coreNum - 1) / coreNum;
        bhStart = coreIdx * bhPerCore;
        bhEnd = bhStart + bhPerCore;
        if (bhEnd > totalBH) bhEnd = totalBH;

        // 计算每块的 tile 长度
        tileLen = (PN + tileNum - 1) / tileNum;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x));  // [B, C, H, P, N]
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y));  // [B, H, P, N]

        pipe.InitBuffer(tmpQueue, BUFFER_NUM, tileLen * sizeof(float));
        pipe.InitBuffer(accQueue, BUFFER_NUM, tileLen * sizeof(float));
        pipe.InitBuffer(outQueue, BUFFER_NUM, tileLen * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t bh = bhStart; bh < bhEnd; ++bh) {
            uint32_t b = bh / H;
            uint32_t h = bh % H;

            for (uint32_t t = 0; t < tileNum; ++t) {
                uint32_t offsetPN = t * tileLen;
                if (offsetPN >= PN) break;
                uint32_t curLen = (offsetPN + tileLen <= PN) ? tileLen : (PN - offsetPN);

                // 累加器：读取 c=0 切片初始化
                AscendC::LocalTensor<float> accLocal = accQueue.AllocTensor<float>();
                {
                    uint64_t srcOffset = ((((uint64_t)b) * C + 0) * H + h) * PN + offsetPN;
                    AscendC::DataCopy(accLocal, xGm[srcOffset], curLen);
                }

                // 对后续 c 进行累加
                for (uint32_t c = 1; c < C; ++c) {
                    AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();
                    uint64_t srcOffset = ((((uint64_t)b) * C + c) * H + h) * PN + offsetPN;
                    AscendC::DataCopy(tmpLocal, xGm[srcOffset], curLen);
                    // 矢量加法：accLocal += tmpLocal
                    AscendC::Add(accLocal, accLocal, tmpLocal, curLen);
                    tmpQueue.FreeTensor(tmpLocal);
                }

                // 写回输出 y[b, h, :, :] 当前 tile
                AscendC::LocalTensor<float> outLocal = outQueue.AllocTensor<float>();
                AscendC::DataCopy(outLocal, accLocal, curLen);
                outQueue.EnQue<float>(outLocal);
                accQueue.FreeTensor(accLocal);

                AscendC::LocalTensor<float> outLocal2 = outQueue.DeQue<float>();
                uint64_t dstOffset = ((((uint64_t)b) * H + h) * PN) + offsetPN;
                AscendC::DataCopy(yGm[dstOffset], outLocal2, curLen);
                outQueue.FreeTensor(outLocal2);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> accQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t B, C, H, PN;
    uint32_t tileNum;
    uint32_t tileLen;

    uint32_t totalBH, coreNum, coreIdx, bhPerCore, bhStart, bhEnd;
};

extern "C" __global__ __aicore__ void mamba_return_final_state_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMambaReturnFinalState op;
    op.Init(x, y, tiling_data.B, tiling_data.C, tiling_data.H, tiling_data.PN, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
from einops import rearrange
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, batch_size, seq_length, n_heads, d_head, d_state, block_len=64):
        super(ModelNew, self).__init__()
        assert seq_length % block_len == 0, "Sequence length must be divisible by block length"
        self.batch_size = batch_size
        self.seq_length = seq_length
        self.n_heads = n_heads
        self.d_head = d_head
        self.d_state = d_state
        self.block_len = block_len

        self.A = nn.Parameter(torch.randn(batch_size, seq_length, n_heads))
        self.B = nn.Parameter(torch.randn(batch_size, seq_length, n_heads, d_state))
        self.C = nn.Parameter(torch.randn(batch_size, seq_length, n_heads, d_state))

    def segsum(self, x):
        T = x.size(-1)
        x_cumsum = torch.cumsum(x, dim=-1)
        x_segsum = x_cumsum[..., :, None] - x_cumsum[..., None, :]
        mask = torch.tril(torch.ones(T, T, device=x.device, dtype=torch.bool), diagonal=0)
        x_segsum = x_segsum.masked_fill(~mask, -torch.inf)
        return x_segsum

    def forward(self, X, initial_states=None):
        # 块重排
        X_blocks, A_blocks, B_blocks, C_blocks = [
            rearrange(x, "b (c l) ... -> b c l ...", l=self.block_len).contiguous()
            for x in (X, self.A, self.B, self.C)
        ]

        A_blocks = rearrange(A_blocks, "b c l h -> b h c l").contiguous()
        A_cumsum = torch.cumsum(A_blocks, dim=-1)

        # 对角块输出（保留与原实现一致，便于与上游兼容）
        L = torch.exp(self.segsum(A_blocks))
        _ = torch.einsum("bclhn,bcshn,bhcls,bcshp->bclhp",
                         C_blocks, B_blocks, L, X_blocks)

        # 块内状态
        decay_states = torch.exp((A_cumsum[:, :, :, -1:] - A_cumsum))
        states = torch.einsum("bclhn,bhcl,bclhp->bchpn", B_blocks, decay_states, X_blocks).contiguous()
        # states: [B, C, H, P, N]

        # 跨块递归衰减矩阵（只需要最后一行 z = Z-1）
        if initial_states is None:
            initial_states = torch.zeros_like(states[:, :1])
        states = torch.cat([initial_states, states], dim=1).contiguous()  # [B, C, H, P, N]，此处 C 已包含初始状态

        decay_chunk = torch.exp(self.segsum(F.pad(A_cumsum[:, :, :, -1], (1, 0)))).contiguous()  # [B, H, Z, C]
        # 仅取 z = Z-1 行，得到权重向量 w_last: [B, H, C]
        w_last = decay_chunk[:, :, -1, :].contiguous()

        # 预权重后按 C 维求和：将 w_last 扩展到 [B, C, H, 1, 1] 与 states 相乘
        w_last_bc = w_last.permute(0, 2, 1).unsqueeze(-1).unsqueeze(-1).contiguous()  # [B, C, H, 1, 1]
        weighted_states = states * w_last_bc  # [B, C, H, P, N]

        # 自定义 AscendC Kernel：沿 C 维度归约，输出最终状态 [B, H, P, N]
        final_state = custom_ops_lib.mamba_return_final_state_custom(weighted_states)
        return final_state
'''
