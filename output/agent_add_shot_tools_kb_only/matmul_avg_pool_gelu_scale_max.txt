kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Y = float;
using DTYPE_SCALE = float;
using DTYPE_INT = int32_t;

constexpr int32_t BUFFER_NUM = 2; // 每个队列的张量数量

class KernelMatmulAvgPoolGeluScaleMaxCustom {
public:
    __aicore__ inline KernelMatmulAvgPoolGeluScaleMaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR pool_kernel, GM_ADDR y,
                                uint32_t totalLength, uint32_t batchSize, uint32_t tileNum)
    {
        this->totalLen = totalLength;   // 输入每行长度 F
        this->batch = batchSize;        // 批大小 B
        this->tileNum = tileNum;

        // 计算 block 负责的行区间
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t baseRows = batch / blockNum;
        uint32_t rem = batch % blockNum;
        this->rowStart = baseRows * blockIdx + (blockIdx < rem ? blockIdx : rem);
        this->rowCount = baseRows + (blockIdx < rem ? 1 : 0);

        // 简单按输入长度切分 tile（池化核在运行时读取）
        // 单 tile 输入元素数 = totalLen / tileNum / BUFFER_NUM
        this->tileInLen = (this->totalLen / this->tileNum) / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)rowStart * totalLen,
                            (uint64_t)rowCount * totalLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + (uint64_t)rowStart * totalLen,
                            (uint64_t)rowCount * totalLen);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_SCALE *)scale, 1);
        poolGm.SetGlobalBuffer((__gm__ DTYPE_INT *)pool_kernel, 1);

        // 预留 UB 缓冲区：输入、临时累加/输出
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileInLen * sizeof(DTYPE_X));
        // 最坏情况下池化核为1，输出与输入等长；为安全起见，输出 tile 大小取 tileInLen
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileInLen * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileInLen * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // 读取 scale 与 pool_kernel
        DTYPE_SCALE scaleVal = 1.0f;
        DTYPE_INT poolK = 1;
        AscendC::LocalTensor<DTYPE_SCALE> scaleLocal = tmpQueue.AllocTensor<DTYPE_SCALE>();
        AscendC::DataCopy(scaleLocal, scaleGm, 1);
        scaleVal = scaleLocal.GetValue(0);
        tmpQueue.FreeTensor(scaleLocal);

        AscendC::LocalTensor<DTYPE_INT> poolLocal = tmpQueue.AllocTensor<DTYPE_INT>();
        AscendC::DataCopy(poolLocal, poolGm, 1);
        poolK = poolLocal.GetValue(0);
        tmpQueue.FreeTensor(poolLocal);
        if (poolK <= 0) { poolK = 1; }

        // 对每一行分别处理：AvgPool1d -> GELU -> scale
        for (uint32_t r = 0; r < this->rowCount; ++r) {
            // 该行的有效输出长度（池化后）= totalLen / poolK，余数部分忽略
            uint32_t outLenRow = this->totalLen / static_cast<uint32_t>(poolK);
            uint32_t tilesPerRow = this->tileNum * BUFFER_NUM;
            // 以输出为基准划分 tile
            uint32_t tileOutLen = outLenRow / tilesPerRow;
            if (tileOutLen == 0) {
                tileOutLen = outLenRow; // 兜底：一次处理整行
                tilesPerRow = 1;
            }

            for (uint32_t t = 0; t < tilesPerRow; ++t) {
                // 读取输入分块（对应 tileOutLen 个输出 -> 需要 tileOutLen * poolK 个输入）
                uint32_t inLenThisTile = tileOutLen * static_cast<uint32_t>(poolK);
                uint64_t inOffset = (uint64_t)r * totalLen + (uint64_t)t * inLenThisTile;

                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[inOffset], inLenThisTile);
                inQueueX.EnQue(xLocal);

                // 计算 AvgPool: 对每个窗口求和再除以 poolK
                AscendC::LocalTensor<DTYPE_X> xTile = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> accLocal = tmpQueue.AllocTensor<DTYPE_Y>();

                // accLocal <- 第一个窗口片
                AscendC::DataCopy(accLocal, xTile, tileOutLen);
                for (int k = 1; k < poolK; ++k) {
                    AscendC::LocalTensor<DTYPE_X> seg = xTile[(uint32_t)k * tileOutLen];
                    AscendC::Add(accLocal, accLocal, seg, tileOutLen);
                }
                // 平均
                DTYPE_Y invK = 1.0f / static_cast<DTYPE_Y>(poolK);
                AscendC::Muls(accLocal, accLocal, invK, tileOutLen);

                // GELU
                AscendC::LocalTensor<DTYPE_Y> geluLocal = outQueueY.AllocTensor<DTYPE_Y>();
                AscendC::Gelu(geluLocal, accLocal, tileOutLen);

                // scale
                AscendC::Muls(geluLocal, geluLocal, scaleVal, tileOutLen);

                // 写回（此处仅返回 AvgPool+GELU+Scale 的结果，max 在 Python 侧完成）
                uint64_t outOffset = (uint64_t)r * totalLen + (uint64_t)t * tileOutLen * static_cast<uint32_t>(poolK); // 对齐到原始长度位置
                AscendC::DataCopy(yGm[outOffset], geluLocal, tileOutLen);

                outQueueY.FreeTensor(geluLocal);
                tmpQueue.FreeTensor(accLocal);
                inQueueX.FreeTensor(xTile);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_SCALE> scaleGm;
    AscendC::GlobalTensor<DTYPE_INT> poolGm;

    uint32_t totalLen;
    uint32_t batch;
    uint32_t tileNum;

    uint32_t rowStart;
    uint32_t rowCount;

    uint32_t tileInLen;
};

extern "C" __global__ __aicore__ void matmul_avg_pool_gelu_scale_max_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR pool_kernel, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulAvgPoolGeluScaleMaxCustom op;
    op.Init(x, scale, pool_kernel, y, tiling_data.totalLength, tiling_data.batchSize, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版：Linear 后的 AvgPool1d + GELU + Scale 使用 AscendC 自定义算子融合实现，
    最终的按维度最大值在 Python 侧完成。
    """
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super().__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = scale_factor

    def forward(self, x):
        # 线性层仍由高性能库执行
        y = self.matmul(x)                               # [B, F]
        # 自定义 AscendC 融合：AvgPool1d(kernel) -> GELU -> Scale
        scale_tensor = torch.tensor([self.scale_factor], dtype=y.dtype, device=y.device)
        pool_tensor = torch.tensor([self.pool_kernel_size], dtype=torch.int32, device=y.device)
        y_fused = custom_ops_lib.matmul_avg_pool_gelu_scale_max_custom(y, scale_tensor, pool_tensor)  # 形状与输入保持一致
        # 池化核与步长一致时，y_fused 的有效值位于每个池化窗口的首元素位置；此处直接做最大化归约
        out = torch.max(y_fused, dim=1).values
        return out
