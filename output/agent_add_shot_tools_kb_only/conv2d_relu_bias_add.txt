kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_B = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2; // 每个队列的张量数

class KernelConvReluBiasAdd {
public:
    __aicore__ inline KernelConvReluBiasAdd() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b_conv, GM_ADDR b_add, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t StrideH, uint32_t StrideW,
                                uint32_t PadTop, uint32_t PadLeft,
                                uint32_t OutH, uint32_t OutW,
                                uint32_t totalOutLength, uint32_t tileNum)
    {
        this->N = N; this->Cin = Cin; this->H = H; this->W = W;
        this->Cout = Cout; this->Kh = Kh; this->Kw = Kw;
        this->StrideH = StrideH; this->StrideW = StrideW;
        this->PadTop = PadTop; this->PadLeft = PadLeft;
        this->OutH = OutH; this->OutW = OutW;

        this->totalOutLength = totalOutLength;
        this->blockLength = totalOutLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / (tileNum * BUFFER_NUM);

        // Global buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N * Cin * H * W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)Cout * Cin * Kh * Kw);
        bConvGm.SetGlobalBuffer((__gm__ DTYPE_B*)b_conv, (uint64_t)Cout);
        bAddGm.SetGlobalBuffer((__gm__ DTYPE_B*)b_add, (uint64_t)Cout);
        // 每个block写自己的输出范围
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // Queues
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));

        // 临时队列：单元素局部张量用于累加/取数（演示用，非高效实现）
        pipe.InitBuffer(tmpAccQ, 1, 1 * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpInQ,  1, 1 * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpWQ,   1, 1 * sizeof(DTYPE_W));
        pipe.InitBuffer(tmpBConvQ, 1, 1 * sizeof(DTYPE_B));
        pipe.InitBuffer(tmpBAddQ,  1, 1 * sizeof(DTYPE_B));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    // 计算一个tile的输出
    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        // 线性索引起点
        uint64_t baseIdx = (uint64_t)progress * this->tileLength + 0;

        // 计算 tileLength 个元素
        for (uint32_t t = 0; t < this->tileLength; ++t) {
            uint64_t lin = (uint64_t)AscendC::GetBlockIdx() * this->blockLength + baseIdx + t;
            // 线性索引 -> (n, co, ho, wo)
            uint32_t ow = lin % OutW;
            uint64_t tmp1 = lin / OutW;
            uint32_t oh = tmp1 % OutH;
            uint64_t tmp2 = tmp1 / OutH;
            uint32_t co = tmp2 % Cout;
            uint32_t n  = tmp2 / Cout;

            // 累加器 acc = 0
            AscendC::LocalTensor<DTYPE_Y> accLocal = tmpAccQ.AllocTensor<DTYPE_Y>();
            AscendC::Duplicate(accLocal, static_cast<DTYPE_Y>(0.0f), 1);

            // 卷积核范围循环（极简演示实现）
            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t kr = 0; kr < Kh; ++kr) {
                    for (uint32_t ks = 0; ks < Kw; ++ks) {
                        int32_t ih = static_cast<int32_t>(oh * StrideH + kr) - static_cast<int32_t>(PadTop);
                        int32_t iw = static_cast<int32_t>(ow * StrideW + ks) - static_cast<int32_t>(PadLeft);
                        if (ih < 0 || iw < 0 || ih >= (int32_t)H || iw >= (int32_t)W) {
                            continue;
                        }
                        uint64_t xOff = ((((uint64_t)n * Cin + ci) * H + (uint64_t)ih) * W + (uint64_t)iw);
                        uint64_t wOff = ((((uint64_t)co * Cin + ci) * Kh + kr) * Kw + ks);

                        AscendC::LocalTensor<DTYPE_X> inLocal = tmpInQ.AllocTensor<DTYPE_X>();
                        AscendC::LocalTensor<DTYPE_W> wLocal  = tmpWQ.AllocTensor<DTYPE_W>();
                        AscendC::DataCopy(inLocal, xGm[xOff], 1);
                        AscendC::DataCopy(wLocal,  wGm[wOff], 1);
                        // acc += in * w
                        AscendC::MulAddDst(accLocal, inLocal, wLocal, 1);
                        tmpInQ.FreeTensor(inLocal);
                        tmpWQ.FreeTensor(wLocal);
                    }
                }
            }

            // 加上卷积自带 bias
            AscendC::LocalTensor<DTYPE_B> bConvLocal = tmpBConvQ.AllocTensor<DTYPE_B>();
            AscendC::DataCopy(bConvLocal, bConvGm[co], 1);
            AscendC::Add(accLocal, accLocal, bConvLocal, 1);
            tmpBConvQ.FreeTensor(bConvLocal);

            // ReLU
            AscendC::Relu(accLocal, accLocal, 1);

            // 额外 bias add (按通道广播)
            AscendC::LocalTensor<DTYPE_B> bAddLocal = tmpBAddQ.AllocTensor<DTYPE_B>();
            AscendC::DataCopy(bAddLocal, bAddGm[co], 1);
            AscendC::Add(accLocal, accLocal, bAddLocal, 1);
            tmpBAddQ.FreeTensor(bAddLocal);

            // 写入本地输出 buffer
            // 将accLocal(1)拷贝到yLocal的相对位置t
            AscendC::DataCopy(yLocal[t], accLocal, 1);
            tmpAccQ.FreeTensor(accLocal);
        }
        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;

    // 输出队列
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;

    // 临时单元素队列
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpAccQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpInQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpWQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpBConvQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpBAddQ;

    // 全局张量
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bConvGm;
    AscendC::GlobalTensor<DTYPE_B> bAddGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    // 形状参数
    uint32_t N, Cin, H, W, Cout, Kh, Kw, StrideH, StrideW, PadTop, PadLeft, OutH, OutW;

    // 分块/切片参数
    uint32_t totalOutLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b_conv, GM_ADDR b_add, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvReluBiasAdd op;
    op.Init(x, w, b_conv, b_add, y,
            tiling_data.N(), tiling_data.Cin(), tiling_data.H(), tiling_data.W(),
            tiling_data.Cout(), tiling_data.Kh(), tiling_data.Kw(),
            tiling_data.StrideH(), tiling_data.StrideW(),
            tiling_data.PadTop(), tiling_data.PadLeft(),
            tiling_data.OutH(), tiling_data.OutW(),
            tiling_data.totalOutLength(), tiling_data.tileNum());
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC算子实现 conv2d -> ReLU -> bias add 融合
    """

        super(ModelNew, self).__init__()
