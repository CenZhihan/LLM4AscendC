host_tiling_src="""
#include "register/tilingdata_base.h"
#include "register/op_def_registry.h"
#include "tiling/platform/platform_ascend.h"
#include "tiling/op_tiling.h"
#include "graph/tensor.h"
#include "graph/debug/ge_attr_define.h"
#include "rms_norm_custom_tiling.h"

using namespace AscendC;

namespace optiling {

BEGIN_TILING_DATA_DEF(RmsNormCustomTilingData)
  TILING_DATA_FIELD_DEF(uint32_t, tmpBufSize);
  TILING_DATA_FIELD_DEF_STRUCT(RmsNormTiling, rmsnormTilingData);
END_TILING_DATA_DEF;

REGISTER_TILING_DATA_CLASS(RmsNormCustom, RmsNormCustomTilingData)

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    RmsNormCustomTilingData tilingData;

    // 读取输入原始形状，期望为 [N, H1, H2, C] (NHWC)
    const gert::Tensor* inTensor = context->GetInputTensor(0);
    const gert::Shape& inOriginShape = inTensor->GetOriginShape();

    // 安全获取维度
    int64_t dimNum = inOriginShape.GetDimNum();
    if (dimNum < 2) {
        return ge::GRAPH_FAILED;
    }
    int64_t N = 1;
    int64_t H1 = 1;
    int64_t H2 = 1;
    int64_t C = 1;
    if (dimNum == 2) {
        // 退化情况 [N, C]
        N = inOriginShape.GetDim(0);
        C = inOriginShape.GetDim(1);
    } else if (dimNum == 3) {
        // 退化情况 [N, S, C]
        N = inOriginShape.GetDim(0);
        H1 = inOriginShape.GetDim(1);
        C = inOriginShape.GetDim(2);
    } else {
        // 常规情况 [N, H1, H2, C]
        N = inOriginShape.GetDim(0);
        H1 = inOriginShape.GetDim(1);
        H2 = inOriginShape.GetDim(2);
        C = inOriginShape.GetDim(3);
    }

    int64_t B = N;
    int64_t S = H1 * H2;
    int64_t H = C;

    // 构造 RmsNorm 的 [B, S, H] 形状
    std::vector<int64_t> oriShapeVec = {B, S, H};
    ge::Shape oriSrcShape(oriShapeVec);

    // 按文档要求，H 轴需 32 字节对齐（float=4B -> 8 元素对齐）。若本身满足可直接使用。
    int64_t alignUnit = 32 / sizeof(float); // 8
    int64_t HAlign = ((H + alignUnit - 1) / alignUnit) * alignUnit;
    std::vector<int64_t> srcShapeVec = {B, S, HAlign};
    ge::Shape srcShape(srcShapeVec);

    // 获取临时空间需求并生成 tiling 参数
    uint32_t minValue = 0;
    uint32_t maxValue = 0;
    bool isBasicBlock = false;
    if (!AscendC::GetRmsNormMaxMinTmpSize(srcShape, sizeof(float), maxValue, minValue, isBasicBlock)) {
        return ge::GRAPH_FAILED;
    }
    tilingData.set_tmpBufSize(minValue);

    optiling::RmsNormTiling rmsTiling;
    if (!AscendC::GetRmsNormTilingInfo(srcShape, oriSrcShape, minValue, sizeof(float), rmsTiling, isBasicBlock)) {
        return ge::GRAPH_FAILED;
    }
    tilingData.set_rmsnormTilingData(rmsTiling);

    // 设置 block 数
    const uint32_t BLOCK_DIM = 8;
    context->SetBlockDim(BLOCK_DIM);

    // 保存 tiling 数据
    tilingData.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tilingData.GetDataSize());
    context->SetTilingKey(1);

    // 无需额外 workspace（内核通过 sharedTmpBuffer 从 UB 管理临时空间）
    size_t* workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling


namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

} // namespace ge


namespace ops {

class RmsNormCustom : public OpDef {
public:
    explicit RmsNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})

        this->SetInferShape(ge::InferShape)
            .SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(RmsNormCustom);

} // namespace ops

kernel_src="""
#include "kernel_operator.h"

using dataType = float;

class KernelRmsNormCustom {
public:
    __aicore__ inline KernelRmsNormCustom() {}

    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR yGm, const RmsNormCustomTilingData &customTiling)
    {
        tiling = customTiling.rmsnormTilingData;
        hLength = tiling.hLength;
        bshLength = tiling.bLength * tiling.sLength * tiling.hLength;
        tmpBufSize = customTiling.tmpBufSize;

        inputGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(xGm), bshLength);
        outputGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(yGm), bshLength);

        constexpr uint32_t typeSize = sizeof(dataType);
        pipe.InitBuffer(inQueue, 1, bshLength * typeSize);
        pipe.InitBuffer(outQueue, 1, bshLength * typeSize);
        pipe.InitBuffer(gammaQueue, 1, hLength * typeSize);
        pipe.InitBuffer(tmpQueue, 1, tmpBufSize);
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<dataType> inputLocal = inQueue.AllocTensor<dataType>();
        AscendC::DataCopy(inputLocal, inputGlobal, bshLength);
        inQueue.EnQue(inputLocal);

        // 准备 gamma=1 向量
        AscendC::LocalTensor<dataType> gammaLocal = gammaQueue.AllocTensor<dataType>();
        dataType one = static_cast<dataType>(1.0f);
        AscendC::Duplicate(gammaLocal, one, hLength);
        gammaQueue.EnQue(gammaLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<dataType> inputLocal = inQueue.DeQue<dataType>();
        AscendC::LocalTensor<dataType> gammaLocal = gammaQueue.DeQue<dataType>();
        AscendC::LocalTensor<dataType> outputLocal = outQueue.AllocTensor<dataType>();
        AscendC::LocalTensor<uint8_t> stackBuffer = tmpQueue.AllocTensor<uint8_t>();

        // 固定 epsilon（与原模型默认值一致）
        dataType epsilon = static_cast<dataType>(1e-5f);

        // 使用共享临时空间版本的 RmsNorm 高级 API
        AscendC::RmsNorm<dataType, false>(outputLocal, inputLocal, gammaLocal, stackBuffer, epsilon, tiling);

        inQueue.FreeTensor(inputLocal);
        gammaQueue.FreeTensor(gammaLocal);
        tmpQueue.FreeTensor(stackBuffer);
        outQueue.EnQue(outputLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<dataType> outputLocal = outQueue.DeQue<dataType>();
        AscendC::DataCopy(outputGlobal, outputLocal, bshLength);
        outQueue.FreeTensor(outputLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> gammaQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue;

    AscendC::GlobalTensor<dataType> inputGlobal;
    AscendC::GlobalTensor<dataType> outputGlobal;

    RmsNormTiling tiling;
    uint32_t hLength;
    uint32_t bshLength;
    uint32_t tmpBufSize;
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR tiling)
{
    GET_TILING_DATA(customTilingData, tiling)
    KernelRmsNormCustom op;
    op.Init(x, y, customTilingData);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义 AscendC 算子实现基于通道维度(原 dim=1)的 RMS Normalization。
    算子在 NHWC（最后一维为通道）上执行，内部沿最后一维做 RmsNorm。
    """
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps  # 当前内核使用固定 1e-5，与默认值一致

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # 将输入从 NCHW 转为 NHWC，使通道在最后一维，便于内核沿 H 轴做 RmsNorm
        x_nhwc = x.permute(0, 2, 3, 1).contiguous()
        y_nhwc = custom_ops_lib.rms_norm_custom(x_nhwc)
        # 还原回 NCHW
        y = y_nhwc.permute(0, 3, 1, 2).contiguous()
        return y
