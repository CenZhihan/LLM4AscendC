kernel_src="""
#include "kernel_operator.h"

using DTYPE = half;

constexpr int32_t BUF_DH_NUM = 4;   // temporary Dh vectors
constexpr int32_t BUF_OUT_NUM = 1;  // output Dh vector buffer
constexpr int32_t BUF_SCALAR_NUM = 2; // 1-element temporary buffers
constexpr uint32_t EXP_TMP_BYTES = 512; // temporary buffer for Exp on small vectors

// Dot product using vector instructions and PairReduceSum
__aicore__ inline void DotProductReduce(AscendC::LocalTensor<DTYPE>& tmpProd, uint32_t dh)
{
    // tmpProd holds element-wise product; reduce in-place until length=1
    uint32_t len = dh;
    while (len > 1) {
        uint32_t maskCount = len;
        // Contiguous mask mode
        AscendC::PairReduceSum<DTYPE>(tmpProd, tmpProd, 1, maskCount, 1, 1, 8);
        len = (len >> 1);
    }
}

class KernelWindowedCausalAttention {
public:
    __aicore__ inline KernelWindowedCausalAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y,
                                uint32_t B, uint32_t N, uint32_t H, uint32_t Dh,
                                uint32_t strideN, uint32_t strideB, uint32_t window, float scale_f)
    {
        this->B = B;
        this->N = N;
        this->H = H;
        this->Dh = Dh;
        this->strideN = strideN;
        this->strideB = strideB;
        this->window = window;
        this->scale = static_cast<DTYPE>(scale_f);

        uint32_t bhIdx = AscendC::GetBlockIdx();
        if (bhIdx >= B * H) {
            return;
        }
        this->b = bhIdx / H;
        this->h = bhIdx % H;

        uint64_t baseOffset = static_cast<uint64_t>(b) * strideB + static_cast<uint64_t>(h) * Dh;

        qGm.SetGlobalBuffer((__gm__ DTYPE*)q + baseOffset, static_cast<uint64_t>(N) * Dh);
        kGm.SetGlobalBuffer((__gm__ DTYPE*)k + baseOffset, static_cast<uint64_t>(N) * Dh);
        vGm.SetGlobalBuffer((__gm__ DTYPE*)v + baseOffset, static_cast<uint64_t>(N) * Dh);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + baseOffset, static_cast<uint64_t>(N) * Dh);

        // Init buffers
        pipe.InitBuffer(bufDh, BUF_DH_NUM, Dh * sizeof(DTYPE));          // for q, k, v, tmp
        pipe.InitBuffer(bufOut, BUF_OUT_NUM, Dh * sizeof(DTYPE));         // output vector per token
        pipe.InitBuffer(bufScalar, BUF_SCALAR_NUM, 32);                   // 1-element tensors (32B aligned)
        pipe.InitBuffer(bufExpTmp, 1, EXP_TMP_BYTES);                     // Exp temp buffer
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < N; ++i) {
            ComputeOneToken(i);
        }
    }

private:
    __aicore__ inline void ComputeOneToken(uint32_t idx)
    {
        // Allocate local tensors
        AscendC::LocalTensor<DTYPE> qLocal = bufDh.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> kLocal = bufDh.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> vLocal = bufDh.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> tmpProd = bufDh.AllocTensor<DTYPE>();   // Dh-sized for product & reduction
        AscendC::LocalTensor<DTYPE> yLocal = bufOut.AllocTensor<DTYPE>();   // Dh-sized output accumulator
        AscendC::LocalTensor<DTYPE> sumBuf = bufScalar.AllocTensor<DTYPE>(); // 1-element sum of exp(scores)
        AscendC::LocalTensor<uint8_t> expTmp = bufExpTmp.AllocTensor<uint8_t>(); // temp for Exp

        // Load q[idx]
        AscendC::DataCopy(qLocal, qGm[idx * strideN], Dh);

        // yLocal = 0
        DTYPE zeroVal = DTYPE(0.0f);
        AscendC::Duplicate(yLocal, zeroVal, Dh);

        // sumBuf = 0
        AscendC::Duplicate(sumBuf, zeroVal, 1);

        // window range [start, idx]
        uint32_t start = (idx > window - 1) ? (idx - (window - 1)) : 0;
        uint32_t L = idx - start + 1;

        // First pass: compute unnormalized weights exp(score*scale) and accumulate:
        //   sumExp += e_j
        //   yLocal += e_j * v_j
        for (uint32_t j = 0; j < L; ++j) {
            uint32_t pos = start + j;

            // Load k[pos] and v[pos]
            AscendC::DataCopy(kLocal, kGm[pos * strideN], Dh);
            AscendC::DataCopy(vLocal, vGm[pos * strideN], Dh);

            // tmpProd = qLocal * kLocal
            AscendC::Mul(tmpProd, qLocal, kLocal, Dh);

            // reduce tmpProd to 1 element (dot)
            DotProductReduce(tmpProd, Dh);

            // scale and exp on the single element
            // tmpProd[0] = tmpProd[0] * scale
            AscendC::Muls(tmpProd, tmpProd, this->scale, 1);
            AscendC::Exp<DTYPE, 15, false>(tmpProd, tmpProd, expTmp, 1);  // e_j = exp(score*scale)

            // sumBuf += tmpProd (1 element)
            AscendC::Add(sumBuf, sumBuf, tmpProd, 1);

            // yLocal += e_j * vLocal
            // Multiply vLocal by scalar e_j (tmpProd[0]) and then accumulate
            AscendC::Muls(vLocal, vLocal, tmpProd[0], Dh);
            AscendC::Add(yLocal, yLocal, vLocal, Dh);
        }

        // Normalize: yLocal /= sumExp
        // invSum = 1 / sumBuf[0]
        AscendC::LocalTensor<DTYPE> invBuf = bufScalar.AllocTensor<DTYPE>();
        AscendC::Reciprocal(invBuf, sumBuf, 1);
        AscendC::Muls(yLocal, yLocal, invBuf[0], Dh);

        // Store output y[idx]
        AscendC::DataCopy(yGm[idx * strideN], yLocal, Dh);

        // Free tensors
        bufDh.FreeTensor(qLocal);
        bufDh.FreeTensor(kLocal);
        bufDh.FreeTensor(vLocal);
        bufDh.FreeTensor(tmpProd);
        bufOut.FreeTensor(yLocal);
        bufScalar.FreeTensor(sumBuf);
        bufScalar.FreeTensor(invBuf);
        bufExpTmp.FreeTensor(expTmp);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_DH_NUM> bufDh;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUF_OUT_NUM> bufOut;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_SCALAR_NUM> bufScalar;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufExpTmp;

    AscendC::GlobalTensor<DTYPE> qGm, kGm, vGm, yGm;

    uint32_t B, N, H, Dh;
    uint32_t strideN, strideB, window;
    DTYPE scale;
    uint32_t b, h;
};

extern "C" __global__ __aicore__ void windowed_causal_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelWindowedCausalAttention op;
    op.Init(q, k, v, y,
            tiling_data.B, tiling_data.N, tiling_data.H, tiling_data.Dh,
            tiling_data.strideN, tiling_data.strideB, tiling_data.window, tiling_data.scale);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import math
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Sliding-window causal attention with custom AscendC kernel.
    Inputs/outputs use float16 for better throughput on Ascend.
    """
    def __init__(self, d_model=1024, num_heads=16, window_size=256):
        super().__init__()
        self.q_proj = nn.Linear(d_model, d_model)
        self.k_proj = nn.Linear(d_model, d_model)
        self.v_proj = nn.Linear(d_model, d_model)
        self.out_proj = nn.Linear(d_model, d_model)
        self.num_heads = num_heads
        self.window_size = window_size

    def forward(self, x):
        B, N, D = x.shape
        H = self.num_heads
        d_h = D // H

        # projections
        Q = self.q_proj(x)     # [B, N, D]
        K = self.k_proj(x)
        V = self.v_proj(x)

        # reshape to [B, N, H, d_h] and convert to float16
        Q = Q.view(B, N, H, d_h).to(torch.float16)
        K = K.view(B, N, H, d_h).to(torch.float16)
        V = V.view(B, N, H, d_h).to(torch.float16)

        # custom kernel computes local attention with sliding window
        Y = custom_ops_lib.windowed_causal_attention_custom(Q, K, V, self.window_size)  # [B, N, H, d_h], fp16

        # reshape back and project
        Y = Y.reshape(B, N, D)
        Y = self.out_proj(Y.to(x.dtype))
        return Y
'''
