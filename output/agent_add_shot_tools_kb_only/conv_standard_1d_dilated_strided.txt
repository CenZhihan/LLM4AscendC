    uint32_t K  = static_cast<uint32_t>(GetDim(wShape, 2));

static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* xShape = context->GetInputShape(0);
    const gert::Shape* wShape = context->GetInputShape(1);

    // attrs: stride, dilation
    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    uint32_t stride   = *(attrs->GetAttrPointer<uint32_t>(0));
    uint32_t dilation = *(attrs->GetAttrPointer<uint32_t>(1));

    int64_t N  = xShape->GetOriginShape().GetDim(0);
    int64_t IC = xShape->GetOriginShape().GetDim(1);
    int64_t L  = xShape->GetOriginShape().GetDim(2);
    (void)IC;

    int64_t OC = wShape->GetOriginShape().GetDim(0);
    int64_t K  = wShape->GetOriginShape().GetDim(2);

    int64_t Lout = (L >= (static_cast<int64_t>(dilation) * (K - 1) + 1))
        ? ((L - static_cast<int64_t>(dilation) * (K - 1) - 1) / static_cast<int64_t>(stride) + 1)

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv1d {
public:
    __aicore__ inline KernelConv1d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t N, uint32_t IC, uint32_t OC, uint32_t L,
                                uint32_t K, uint32_t stride, uint32_t dilation,
                                uint32_t Lout, uint32_t hasBias)
    {
        this->totalLength = totalLength;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        this->N = N;
        this->IC = IC;
        this->OC = OC;
        this->L = L;
        this->K = K;
        this->stride = stride;
        this->dilation = dilation;
        this->Lout = Lout;
        this->hasBias = hasBias;

        xGm.SetGlobalBuffer((__gm__ float *)x, N * IC * L);
        wGm.SetGlobalBuffer((__gm__ float *)w, OC * IC * K);
        if (b != nullptr) {
            bGm.SetGlobalBuffer((__gm__ float *)b, OC);
        } else {
            bGm.SetGlobalBuffer((__gm__ float *)b, 0);
        }
        // Output is flattened as length totalLength
        yGm.SetGlobalBuffer((__gm__ float *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // Local buffers for a tile of outputs
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, sizeof(float) * 256); // small scratch
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        // Pseudo-pointer to local data; fill per-element results
        // Note: This code follows generic AscendC style and uses simple scalar accumulation for clarity.
        for (uint32_t t = 0; t < this->tileLength; ++t) {
            uint32_t globalIdx = this->blockLength * AscendC::GetBlockIdx() + progress * this->tileLength + t;
            if (globalIdx >= this->totalLength) {
                break;
            }
            // Map flattened index -> (n, oc, l_out)
            uint32_t tmp = globalIdx;
            uint32_t n  = tmp / (this->OC * this->Lout);
            tmp         = tmp % (this->OC * this->Lout);
            uint32_t oc = tmp / this->Lout;
            uint32_t lo = tmp % this->Lout;

            float acc = 0.f;
            // optional bias
            if (this->hasBias != 0) {
                // Copy bias[oc] into a local scalar
                AscendC::LocalTensor<float> biasLocal = tmpQueue.AllocTensor<float>();
                AscendC::DataCopy(biasLocal, bGm[oc], 1);
                float* biasPtr = reinterpret_cast<float*>(biasLocal.Get<>());
                acc = *biasPtr;
                tmpQueue.FreeTensor(biasLocal);
            }

            // Conv accumulation
            for (uint32_t ic = 0; ic < this->IC; ++ic) {
                for (uint32_t k = 0; k < this->K; ++k) {
                    uint32_t li = lo * this->stride + k * this->dilation;
                    if (li >= this->L) {
                        continue;
                    }

                    // Load x[n, ic, li]
                    AscendC::LocalTensor<float> xLocal = tmpQueue.AllocTensor<float>();
                    // offset = ((n*IC + ic)*L + li)
                    uint32_t xOff = (n * this->IC + ic) * this->L + li;
                    AscendC::DataCopy(xLocal, xGm[xOff], 1);

                    // Load w[oc, ic, k]
                    AscendC::LocalTensor<float> wLocal = tmpQueue.AllocTensor<float>();
                    uint32_t wOff = (oc * this->IC + ic) * this->K + k;
                    AscendC::DataCopy(wLocal, wGm[wOff], 1);

                    float* xv = reinterpret_cast<float*>(xLocal.Get<>());
                    float* wv = reinterpret_cast<float*>(wLocal.Get<>());

                    acc += (*xv) * (*wv);

                    tmpQueue.FreeTensor(wLocal);
                    tmpQueue.FreeTensor(xLocal);
                }
            }

            // Store acc into yLocal[t]
            // Use a temporary 1-element buffer and then vector copy into yLocal[t]
            AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();
            float* outScalarPtr = reinterpret_cast<float*>(tmpLocal.Get<>());
            *outScalarPtr = acc;
            // Copy scalar into the t-th position of yLocal
            AscendC::DataCopy(yLocal[t], tmpLocal, 1);
            tmpQueue.FreeTensor(tmpLocal);
        }

        outQueueY.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t N;
    uint32_t IC;
    uint32_t OC;
    uint32_t L;
    uint32_t K;
    uint32_t stride;
    uint32_t dilation;
    uint32_t Lout;
    uint32_t hasBias;
};

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv1d op;
    op.Init(x, w, b, y,
            tilingData.totalLength, tilingData.tileNum,
            tilingData.N, tilingData.IC, tilingData.OC, tilingData.L,
            tilingData.K, tilingData.stride, tilingData.dilation,
            tilingData.Lout, tilingData.hasBias);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline at::Tensor make_output_like_conv1d(const at::Tensor& x,
                                                 const at::Tensor& w,
                                                 int64_t stride, int64_t dilation) {
    // x: [N, IC, L], w: [OC, IC, K]
    int64_t N = x.size(0);
    int64_t IC = x.size(1);
    int64_t L = x.size(2);
    (void)IC;
    int64_t OC = w.size(0);
    int64_t K = w.size(2);
    int64_t Lout = (L >= (dilation * (K - 1) + 1))
        ? ((L - dilation * (K - 1) - 1) / stride + 1)
        : 0;
    return at::empty({N, OC, Lout}, x.options());
}

at::Tensor conv_standard_1d_dilated_strided_custom_impl_npu(const at::Tensor& x,
                                                            const at::Tensor& w,
                                                            const at::Tensor& b, // may be empty tensor
                                                            int64_t stride, int64_t dilation) {
    at::Tensor y = make_output_like_conv1d(x, w, stride, dilation);
    // hasBias flag
    int64_t hasBias = (b.defined() && b.numel() > 0) ? 1 : 0;
    // EXEC_NPU_CMD will forward attrs and tensors to the custom AICore kernel binding
    EXEC_NPU_CMD(aclnnConvStandard1dDilatedStridedCustom, x, w, b, y, (int)stride, (int)dilation, (int)hasBias);
    return y;
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, dilation: int = 1, bias: bool = False):
        super().__init__()
        self.conv1d = torch.nn.Conv1d(in_channels, out_channels, kernel_size, stride=stride, dilation=dilation, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        bias_tensor = self.conv1d.bias if self.conv1d.bias is not None else torch.empty(0, device=x.device, dtype=x.dtype)
        return custom_ops_lib.conv_standard_1d_dilated_strided_custom(
            x, self.conv1d.weight, bias_tensor, self.conv1d.stride[0], self.conv1d.dilation[0]
        )
