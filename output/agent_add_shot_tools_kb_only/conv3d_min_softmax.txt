kernel_src="""
#include "kernel_operator.h"

// 本自定义核期望输入为 [N, H, W, C]，并在最后一维（C）上执行Softmax。
// 模型中通过先进行Conv3d和沿D维min，再permute到最后一维为C，调用该核。

using DTYPE_X = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxLastDim {
public:
    __aicore__ inline KernelSoftmaxLastDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, uint32_t rowSize)
    {
        this->rowSize = rowSize;          // C
        this->tileNum = tileNum;          // N*H*W
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->rowsPerBlock = this->blockLength / this->rowSize;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // 为每行准备本地缓冲
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->rowSize * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->rowSize * sizeof(DTYPE_Y));

        // SoftmaxFlashV3需要辅助输出/中间张量
        // 这里采用VECCALC位置的本地张量队列，每次分配一行对应空间。
        // mean/expSum/max 各自需要一块本地空间，这里简单以32字节对齐的块大小按行申请。
        // 实际业务可根据API的临时空间查询接口在主机侧保守预留。
        pipe.InitBuffer(auxMeanQueue, BUFFER_NUM, 32);     // 一个数据块（32字节）
        pipe.InitBuffer(auxSumQueue,  BUFFER_NUM, 32);
        pipe.InitBuffer(auxMaxQueue,  BUFFER_NUM, 32);
        // expMax仅当update=true时使用，这里占位初始化
        pipe.InitBuffer(auxExpMaxQueue, BUFFER_NUM, this->rowSize * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        uint32_t rowsToProcess = this->rowsPerBlock;
        for (uint32_t i = 0; i < rowsToProcess; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // 每行连续 rowSize 个元素
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->rowSize], this->rowSize);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*rowIdx*/)
    {
        AscendC::LocalTensor<DTYPE_X> srcLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> dstLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // SoftmaxFlashV3需要提供mean/expSum/max等辅助张量
        AscendC::LocalTensor<float> meanLocal = auxMeanQueue.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal  = auxSumQueue.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal  = auxMaxQueue.AllocTensor<float>();
        AscendC::LocalTensor<DTYPE_X> expMaxLocal = auxExpMaxQueue.AllocTensor<DTYPE_X>();

        // 输入为单行C元素，按[m, n]的m=1, n=rowSize处理
        // 这里使用默认配置调用，未开启update路径
        AscendC::SoftMaxTiling tiling;     // 默认构造
        AscendC::SoftMaxParams params;     // 默认构造

        AscendC::SoftmaxFlashV3<DTYPE_X, float, false, false, false, false>(
            dstLocal, meanLocal, sumLocal, maxLocal,
            srcLocal, expMaxLocal,
            /*inMean*/meanLocal, /*inExpSum*/sumLocal, /*inMax*/maxLocal,
            tiling, params);

        outQueueY.EnQue<DTYPE_Y>(dstLocal);
        inQueueX.FreeTensor(srcLocal);
        auxMeanQueue.FreeTensor(meanLocal);
        auxSumQueue.FreeTensor(sumLocal);
        auxMaxQueue.FreeTensor(maxLocal);
        auxExpMaxQueue.FreeTensor(expMaxLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[rowIdx * this->rowSize], yLocal, this->rowSize);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> auxMeanQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> auxSumQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> auxMaxQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> auxExpMaxQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockLength;
    uint32_t rowsPerBlock;
    uint32_t rowSize;
    uint32_t tileNum;
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxLastDim op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.rowSize);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, dim):
        super().__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.dim = dim  # depth dimension for min

    def forward(self, x):
        # 原始顺序: Conv3d -> min(dim=2) -> softmax(dim=1)
        x = self.conv(x)                         # [N, C, D, H, W]
        x = torch.min(x, dim=self.dim)[0]        # [N, C, H, W]
        # 将通道移至最后一维，便于核在最后一维执行Softmax
        x_last = x.permute(0, 2, 3, 1).contiguous()  # [N, H, W, C]
        y_last = custom_ops_lib.conv3d_min_softmax_custom(x_last)  # softmax on last dim
        y = y_last.permute(0, 3, 1, 2).contiguous()  # [N, C, H, W]
        return y
'''
