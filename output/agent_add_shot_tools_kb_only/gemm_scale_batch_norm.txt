kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

class KernelGemmScaleBatchNorm {
public:
    __aicore__ inline KernelGemmScaleBatchNorm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mul, GM_ADDR add, GM_ADDR y, uint32_t totalLength, uint32_t featureSize, uint32_t tileNumIn)
    {
        this->totalLen = totalLength;       // B * F
        this->F = featureSize;              // feature size
        this->tileNum = tileNumIn;

        // We work in rows; choose BLOCK_DIM = 1 on host so blockLength == totalLen
        this->blockLength = totalLen;
        // rows in this block
        this->rowsInBlock = this->blockLength / this->F;

        // rows per tile
        uint32_t denom = (this->tileNum * BUFFER_NUM);
        uint32_t rowsPerTileBase = (denom > 0) ? (this->rowsInBlock / denom) : this->rowsInBlock;
        if (rowsPerTileBase == 0) rowsPerTileBase = 1;
        this->rowsPerTile = rowsPerTileBase;
        this->tileLength = this->rowsPerTile * this->F;

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x, this->blockLength);
        mulGm.SetGlobalBuffer((__gm__ DTYPE *)mul, this->F);
        addGm.SetGlobalBuffer((__gm__ DTYPE *)add, this->F);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // number of loops needed to cover all rows
        uint32_t loops = (rowsInBlock + rowsPerTile - 1) / rowsPerTile;
        for (uint32_t i = 0; i < loops; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t iter)
    {
        uint32_t startRow = iter * rowsPerTile;
        if (startRow >= rowsInBlock) return;
        uint32_t curRows = rowsPerTile;
        if (startRow + curRows > rowsInBlock) curRows = rowsInBlock - startRow;
        uint32_t curLen = curRows * F;

        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[startRow * F], curLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t iter)
    {
        uint32_t startRow = iter * rowsPerTile;
        if (startRow >= rowsInBlock) return;
        uint32_t curRows = rowsPerTile;
        if (startRow + curRows > rowsInBlock) curRows = rowsInBlock - startRow;
        uint32_t curLen = curRows * F;

        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();

        // Load per-feature affine parameters
        AscendC::LocalTensor<DTYPE> mulLocal = pipe.AllocTensor<AscendC::TPosition::VECIN, DTYPE>(F);
        AscendC::LocalTensor<DTYPE> addLocal = pipe.AllocTensor<AscendC::TPosition::VECIN, DTYPE>(F);
        AscendC::DataCopy(mulLocal, mulGm, F);
        AscendC::DataCopy(addLocal, addGm, F);

        // For each row in the tile
        for (uint32_t r = 0; r < curRows; ++r) {
            AscendC::LocalTensor<DTYPE> xRow = xLocal[r * F];
            AscendC::LocalTensor<DTYPE> yRow = yLocal[r * F];
            AscendC::Mul(yRow, xRow, mulLocal, F);
            AscendC::Add(yRow, yRow, addLocal, F);
        }

        outQueueY.EnQue<DTYPE>(yLocal);
        inQueueX.FreeTensor(xLocal);
        pipe.FreeTensor(mulLocal);
        pipe.FreeTensor(addLocal);
    }

    __aicore__ inline void CopyOut(uint32_t iter)
    {
        uint32_t startRow = iter * rowsPerTile;
        if (startRow >= rowsInBlock) return;
        uint32_t curRows = rowsPerTile;
        if (startRow + curRows > rowsInBlock) curRows = rowsInBlock - startRow;
        uint32_t curLen = curRows * F;

        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[startRow * F], yLocal, curLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> mulGm;
    AscendC::GlobalTensor<DTYPE> addGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t totalLen;
    uint32_t F;
    uint32_t tileNum;
    uint32_t blockLength;
    uint32_t rowsInBlock;
    uint32_t rowsPerTile;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gemm_scale_batch_norm_custom(GM_ADDR x, GM_ADDR mul, GM_ADDR add, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScaleBatchNorm op;
    op.Init(x, mul, add, y, tiling_data.totalLength, tiling_data.featureSize, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized version: keep GEMM in PyTorch, fuse scale and BatchNorm into a single custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.scale = torch.nn.Parameter(torch.randn(scale_shape))
        self.bn = torch.nn.BatchNorm1d(out_features, eps=eps, momentum=momentum)

    def forward(self, x):
        # GEMM (PyTorch)
        y = self.gemm(x)
        # Precompute fused per-channel affine parameters for scale + BatchNorm (inference-style)
        # fused_mul = scale * (gamma / sqrt(running_var + eps))
        # fused_add = beta - gamma * running_mean / sqrt(running_var + eps)
        # Note: This uses running statistics as in inference mode.
        denom = torch.sqrt(self.bn.running_var + self.bn.eps)
        bn_mul = self.bn.weight / denom
        bn_add = self.bn.bias - self.bn.weight * self.bn.running_mean / denom
        fused_mul = self.scale * bn_mul
        fused_add = bn_add
        # Custom kernel applies per-feature affine: y = y * fused_mul + fused_add
        out = custom_ops_lib.gemm_scale_batch_norm_custom(y, fused_mul, fused_add)
        return out
'''
