host_operator_src="""
#include "group_norm_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/tensor.h"
#include "graph/utils/type_utils.h"

namespace {
inline void GetShapeNC1HWTail(const gert::Shape& shape, uint32_t &n, uint32_t &c, uint32_t &h, uint32_t &w) {
    auto dimNum = shape.GetDimNum();
    // 支持 (N,C), (N,C,H), (N,C,H,W) 三种，缺省维度按1处理
    n = dimNum > 0 ? static_cast<uint32_t>(shape.GetDim(0)) : 1;
    c = dimNum > 1 ? static_cast<uint32_t>(shape.GetDim(1)) : 1;
    h = dimNum > 2 ? static_cast<uint32_t>(shape.GetDim(2)) : 1;
    w = dimNum > 3 ? static_cast<uint32_t>(shape.GetDim(3)) : 1;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GroupNormCustomTilingData tiling;

    const gert::Shape* inShape = context->GetInputShape(0);
    auto origin = inShape->GetOriginShape();
    uint32_t n=1, c=1, h=1, w=1;
    GetShapeNC1HWTail(origin, n, c, h, w);

    // 读取属性 num_groups
    uint32_t groupNum = 1;
    // 若接口不可用，可按需替换为实际获取属性的API
    (void)context->GetAttr("num_groups", groupNum);

    // 设置并保存基本形状字段
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_group(groupNum);

    // 依据文档计算 GroupNorm 的 tiling 信息
    std::vector<int64_t> vec = {static_cast<int64_t>(n), static_cast<int64_t>(c),
                                static_cast<int64_t>(h), static_cast<int64_t>(w)};
    ge::Shape srcShape(vec);

    uint32_t maxSize = 0;
    uint32_t minSize = 0;
    // 输入类型为 float，typeSize = 4
    AscendC::GetGroupNormMaxMinTmpSize(srcShape, 4, false, groupNum, maxSize, minSize);
    // 为了更好性能，使用可用的最大值（实际可依据UB余量选择）
    AscendC::GetGroupNormNDTilingInfo(srcShape, maxSize, 4, false, groupNum, tiling.groupNormTilingData);

    context->SetBlockDim(BLOCK_DIM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    context->SetTilingKey(1);

    // 无需额外workspace
    size_t* workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // y 与 x 同形状；mean/variance 为简化，给定与 x 相同形状（足够的空间）
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    gert::Shape* mean_shape = context->GetOutputShape(1);
    gert::Shape* var_shape = context->GetOutputShape(2);

    *y_shape = *x_shape;
    *mean_shape = *x_shape;
    *var_shape = *x_shape;

    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    context->SetOutputDataType(1, inputDataType);
    context->SetOutputDataType(2, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class GroupNormCustom : public OpDef {
public:
    explicit GroupNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("gamma")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("beta")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("mean")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("variance")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        // 属性：分组数
        this->Attr("num_groups").Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(GroupNormCustom);
}

#include "kernel_operator.h"

extern "C" __global__ __aicore__ void group_norm_custom(GM_ADDR inputX_gm,
                                                        GM_ADDR gamma_gm,
                                                        GM_ADDR beta_gm,
                                                        GM_ADDR output_gm,
                                                        GM_ADDR outputMean_gm,
                                                        GM_ADDR outputVariance_gm,
                                                        GM_ADDR workspace,
                                                        GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    // 使用高阶API的GroupNorm内核封装，数据类型为float
    KernelGroupNorm<float, false> op;
    op.Init(inputX_gm, gamma_gm, beta_gm, output_gm, outputMean_gm, outputVariance_gm,
            tilingData.groupNormTilingData);

    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor group_norm_custom_impl_npu(const at::Tensor& x,
                                      const at::Tensor& gamma,
                                      const at::Tensor& beta,
                                      int64_t num_groups) {
    at::Tensor y = at::empty_like(x);
    // 为简化处理，mean与variance分配与x相同shape的缓冲区
    at::Tensor mean = at::empty_like(x);
    at::Tensor variance = at::empty_like(x);
    EXEC_NPU_CMD(aclnnGroupNormCustom, x, gamma, beta, y, mean, variance, num_groups);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("group_norm_custom", &group_norm_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("group_norm_custom", &group_norm_custom_impl_npu, "GroupNorm custom kernel");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC算子进行GroupNorm
    """
    def __init__(self, num_features: int, num_groups: int):
        super().__init__()
        self.weight = torch.nn.Parameter(torch.ones(num_features))
        self.bias = torch.nn.Parameter(torch.zeros(num_features))
        self.num_groups = int(num_groups)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.group_norm_custom(x, self.weight, self.bias, self.num_groups)
