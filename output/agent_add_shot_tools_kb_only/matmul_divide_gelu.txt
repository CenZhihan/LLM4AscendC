kernel_src="""
#include "kernel_operator.h"

// 使用 GEMM 计算 X @ Wt，随后按行加上 bias_div，并计算 GELU。
// GELU(x) = 0.5 * x * (1 + erf(x / sqrt(2)))

class KernelMatmulDivideGelu {
public:
    __aicore__ inline KernelMatmulDivideGelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w_t, GM_ADDR bias_div, GM_ADDR z,
                                uint32_t m, uint32_t k, uint32_t n)
    {
        this->m = m;
        this->k = k;
        this->n = n;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), m * k);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w_t), k * n);
        bGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bias_div), n);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(z), m * n);

        // 为一次性 GEMM 分配缓冲区（示例简化实现）
        pipe.InitBuffer(inQueueA1, 1, m * k * sizeof(float));
        pipe.InitBuffer(inQueueB1, 1, k * n * sizeof(float));
        pipe.InitBuffer(outQueueCO1, 1, m * n * sizeof(float));

        // 行处理缓冲区
        pipe.InitBuffer(rowQueue, 1, n * sizeof(float));
        pipe.InitBuffer(biasQueue, 1, n * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        CopyInGemm();
        ComputeGemm();
        CopyOutGemm();
        ElementwiseBiasGelu();
    }

private:
    __aicore__ inline void CopyInGemm()
    {
        AscendC::LocalTensor<float> aLocal = inQueueA1.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB1.AllocTensor<float>();
        AscendC::DataCopy(aLocal, xGm, m * k);
        AscendC::DataCopy(bLocal, wGm, k * n);
        inQueueA1.EnQue(aLocal);
        inQueueB1.EnQue(bLocal);
    }

    __aicore__ inline void ComputeGemm()
    {
        AscendC::LocalTensor<float> aLocal = inQueueA1.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB1.DeQue<float>();
        AscendC::LocalTensor<float> cLocal = outQueueCO1.AllocTensor<float>();

        AscendC::GemmTiling tilling = AscendC::GetGemmTiling<float>(m, k, n);
        AscendC::Gemm(cLocal, aLocal, bLocal, m, k, n, tilling, false, 1);

        outQueueCO1.EnQue<float>(cLocal);
        inQueueA1.FreeTensor(aLocal);
        inQueueB1.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOutGemm()
    {
        AscendC::LocalTensor<float> cLocal = outQueueCO1.DeQue<float>();
        // 拷贝 GEMM 结果到 GM 输出
        AscendC::DataCopy(zGm, cLocal, m * n);
        outQueueCO1.FreeTensor(cLocal);
    }

    __aicore__ inline void ElementwiseBiasGelu()
    {
        // 拷贝 bias_div 到本地
        AscendC::LocalTensor<float> bLocal = biasQueue.AllocTensor<float>();
        AscendC::DataCopy(bLocal, bGm, n);
        biasQueue.EnQue(bLocal);
        bLocal = biasQueue.DeQue<float>();

        // 常量
        const float inv_sqrt2 = 0.7071067811865476f; // 1/sqrt(2)

        for (uint32_t i = 0; i < m; ++i) {
            // 每行处理
            AscendC::LocalTensor<float> rowLocal = rowQueue.AllocTensor<float>();
            // 从 GM 读取第 i 行
            AscendC::DataCopy(rowLocal, zGm[i * n], n);

            // 加上 bias_div
            AscendC::Add(rowLocal, rowLocal, bLocal, n);

            // GELU: 0.5 * x * (1 + erf(x / sqrt(2)))
            // t = x / sqrt(2)
            AscendC::Muls(rowLocal, rowLocal, inv_sqrt2, n); // rowLocal = t
            AscendC::Erf(rowLocal, rowLocal, n);             // rowLocal = erf(t)
            AscendC::Adds(rowLocal, rowLocal, 1.0f, n);      // rowLocal = erf(t) + 1
            AscendC::Muls(rowLocal, rowLocal, 0.5f, n);      // rowLocal = 0.5 * (1 + erf(t))

            // 需要与原 x 相乘：先把原行再次读入到临时缓冲，或分步计算
            AscendC::LocalTensor<float> origRow = rowQueue.AllocTensor<float>();
            AscendC::DataCopy(origRow, zGm[i * n], n);
            AscendC::Add(origRow, origRow, bLocal, n);       // origRow = x + bias_div

            AscendC::Mul(origRow, origRow, rowLocal, n);     // origRow = (x + b) * 0.5 * (1 + erf(.))

            // 写回第 i 行
            AscendC::DataCopy(zGm[i * n], origRow, n);

            rowQueue.FreeTensor(rowLocal);
            rowQueue.FreeTensor(origRow);
        }
        biasQueue.FreeTensor(bLocal);
    }

private:
    AscendC::TPipe pipe;
    // GEMM 队列
    AscendC::TQue<AscendC::QuePosition::A1, 1> inQueueA1;
    AscendC::TQue<AscendC::QuePosition::B1, 1> inQueueB1;
    AscendC::TQue<AscendC::QuePosition::CO1, 1> outQueueCO1;
    // 行与偏置处理队列
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> rowQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> biasQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t m;
    uint32_t k;
    uint32_t n;
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w_t, GM_ADDR bias_div, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulDivideGelu op;
    op.Init(x, w_t, bias_div, z, tiling_data.m, tiling_data.k, tiling_data.n);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    融合算子版本：使用自定义 AscendC 内核实现 X @ (W/div).T + (b/div) 后接 GELU。
    其中除法已预先吸收到权重与偏置，核内执行 Matmul + BiasAdd + GELU。
    """
    def __init__(self, input_size, output_size, divisor):
        super().__init__()
        self.linear = torch.nn.Linear(input_size, output_size, bias=True)
        self.divisor = float(divisor)

    def forward(self, x):
        # 将除法吸收到权重与偏置，以减少核内标量处理复杂度
        w_scaled = self.linear.weight / self.divisor    # [out, in]
        b_scaled = self.linear.bias / self.divisor      # [out]

        # 算子需要 W 的转置（GEMM B 矩阵为 [k, n]）
        z = custom_ops_lib.matmul_divide_gelu_custom(x, w_scaled.t().contiguous(), b_scaled.contiguous())
        return z
