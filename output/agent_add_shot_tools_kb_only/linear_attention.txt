    int64_t Dk = k_shape->GetOriginShape().GetDim(2);

kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

constexpr int32_t TILE_ELEMS = 4096; // 处理ReLU流的分块元素数
constexpr DTYPE EPS = static_cast<DTYPE>(1e-6f);

class KernelLinearAttention {
public:
    __aicore__ inline KernelLinearAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y,
                                GM_ADDR workspace,
                                uint32_t batch, uint32_t q_len, uint32_t k_len, uint32_t d_model)
    {
        this->B = batch;
        this->QL = q_len;
        this->KL = k_len;
        this->D = d_model;

        uint32_t bidx = AscendC::GetBlockIdx();
        if (bidx >= B) {
            return;
        }

        // 每个batch的偏移（元素数）
        perBatchQElems  = static_cast<uint64_t>(QL) * static_cast<uint64_t>(D);
        perBatchKElems  = static_cast<uint64_t>(KL) * static_cast<uint64_t>(D);
        perBatchKVElems = static_cast<uint64_t>(D)  * static_cast<uint64_t>(D);
        perBatchElems   = perBatchQElems + perBatchKElems + perBatchKVElems;

        // 设置输入/输出GM（按batch切片）
        qGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(q) + bidx * perBatchQElems, perBatchQElems);
        kGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(k) + bidx * perBatchKElems, perBatchKElems);
        vGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(v) + bidx * perBatchKElems, perBatchKElems);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(y) + bidx * perBatchQElems, perBatchQElems);

        // 工作空间切片： [Q_ | K_ | KV]
        __gm__ DTYPE* wsBase = reinterpret_cast<__gm__ DTYPE*>(workspace);
        __gm__ DTYPE* wsThis = wsBase + bidx * perBatchElems;

        qWsGm.SetGlobalBuffer(wsThis, perBatchQElems);
        kWsGm.SetGlobalBuffer(wsThis + perBatchQElems, perBatchKElems);
        kvWsGm.SetGlobalBuffer(wsThis + perBatchQElems + perBatchKElems, perBatchKVElems);

        // 初始化本地缓冲
        // ReLU流的输入/输出临时缓冲
        pipe.InitBuffer(reluInBuf, TILE_ELEMS * sizeof(DTYPE));
        pipe.InitBuffer(reluOutBuf, TILE_ELEMS * sizeof(DTYPE));

        // 计算阶段行向量缓存：长度为D
        uint32_t rowBytes = D * sizeof(DTYPE);
        pipe.InitBuffer(kvRowBuf, rowBytes);
        pipe.InitBuffer(vRowBuf, rowBytes);
        pipe.InitBuffer(qRowBuf, rowBytes);
        pipe.InitBuffer(outRowBuf, rowBytes);
        pipe.InitBuffer(scalarBuf, sizeof(DTYPE)); // 存放单标量
    }

    __aicore__ inline void Process()
    {
        if (AscendC::GetBlockIdx() >= B) {
            return;
        }
        // 1) 预处理：Q_ = relu(Q) + eps，K_ = relu(K) + eps
        PreprocessReluPlusEps(qGm, qWsGm, perBatchQElems);
        PreprocessReluPlusEps(kGm, kWsGm, perBatchKElems);

        // 2) 计算 KV = K_^T @ V， 其中 K_:[K,D], V:[K,D], KV:[D,D]
        ComputeKV();

        // 3) 计算 out = Q_ @ KV， 其中 Q_:[Q,D], KV:[D,D], out:[Q,D]
        ComputeOut();
    }

private:
    __aicore__ inline void PreprocessReluPlusEps(AscendC::GlobalTensor<DTYPE>& src,
                                                 AscendC::GlobalTensor<DTYPE>& dst,
                                                 uint64_t total_elems)
    {
        uint64_t processed = 0;
        while (processed < total_elems) {
            uint32_t cur = static_cast<uint32_t>((total_elems - processed) > TILE_ELEMS ? TILE_ELEMS : (total_elems - processed));

            AscendC::LocalTensor<DTYPE> inLocal  = reluInBuf.Get<DTYPE>(cur);
            AscendC::LocalTensor<DTYPE> outLocal = reluOutBuf.Get<DTYPE>(cur);

            AscendC::DataCopy(inLocal, src[processed], cur);
            AscendC::Relu(outLocal, inLocal, cur);
            AscendC::Adds(outLocal, outLocal, EPS, cur);
            AscendC::DataCopy(dst[processed], outLocal, cur);

            processed += cur;
        }
    }

    __aicore__ inline void ComputeKV()
    {
        // 对于每个 d in [0, D):
        //   kv_row = Kahan-like累加：kv_row = k_0[d]*V_0 + k_1[d]*V_1 + ... + k_{K-1}[d]*V_{K-1}
        AscendC::LocalTensor<DTYPE> kvRow = kvRowBuf.Get<DTYPE>(D);
        AscendC::LocalTensor<DTYPE> vRow  = vRowBuf.Get<DTYPE>(D);
        AscendC::LocalTensor<DTYPE> kScalar = scalarBuf.Get<DTYPE>(1);

        for (uint32_t d = 0; d < D; ++d) {
            // init using first k = 0
            // load V row 0
            AscendC::DataCopy(vRow, vGm[0ULL * D], D);
            // load K_[0, d] scalar
            AscendC::DataCopy(kScalar, kWsGm[0ULL * D + d], 1);
            // kvRow = vRow * kScalar
            AscendC::DataCopy(kvRow, vRow, D);
            AscendC::Muls(kvRow, kvRow, kScalar.GetValue(0), D);

            // accumulate remaining k
            for (uint32_t k = 1; k < KL; ++k) {
                AscendC::DataCopy(vRow, vGm[static_cast<uint64_t>(k) * D], D);
                AscendC::DataCopy(kScalar, kWsGm[static_cast<uint64_t>(k) * D + d], 1);
                AscendC::Axpy(kvRow, vRow, kScalar.GetValue(0), D);
            }

            // write kv_row to GM: row-major [d, :]
            AscendC::DataCopy(kvWsGm[static_cast<uint64_t>(d) * D], kvRow, D);
        }
    }

    __aicore__ inline void ComputeOut()
    {
        // 对每个 q in [0, Q):
        //   out_row = q_row[0]*KV[0,:] + q_row[1]*KV[1,:] + ... + q_row[D-1]*KV[D-1,:]
        AscendC::LocalTensor<DTYPE> qRow   = qRowBuf.Get<DTYPE>(D);
        AscendC::LocalTensor<DTYPE> kvRow  = kvRowBuf.Get<DTYPE>(D);
        AscendC::LocalTensor<DTYPE> outRow = outRowBuf.Get<DTYPE>(D);

        for (uint32_t q = 0; q < QL; ++q) {
            // load q_row
            AscendC::DataCopy(qRow, qWsGm[static_cast<uint64_t>(q) * D], D);

            // init with d=0
            AscendC::DataCopy(kvRow, kvWsGm[0ULL * D], D);
            AscendC::DataCopy(outRow, kvRow, D);
            AscendC::Muls(outRow, outRow, qRow.GetValue(0), D);

            // accumulate remaining d
            for (uint32_t d = 1; d < D; ++d) {
                AscendC::DataCopy(kvRow, kvWsGm[static_cast<uint64_t>(d) * D], D);
                AscendC::Axpy(outRow, kvRow, qRow.GetValue(d), D);
            }

            // write out_row
            AscendC::DataCopy(yGm[static_cast<uint64_t>(q) * D], outRow, D);
        }
    }

private:
    AscendC::TPipe pipe;

    // 全局张量
    AscendC::GlobalTensor<DTYPE> qGm;
    AscendC::GlobalTensor<DTYPE> kGm;
    AscendC::GlobalTensor<DTYPE> vGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    // 工作空间：Q_、K_、KV
    AscendC::GlobalTensor<DTYPE> qWsGm;
    AscendC::GlobalTensor<DTYPE> kWsGm;
    AscendC::GlobalTensor<DTYPE> kvWsGm;

    // 本地缓冲
    AscendC::TBuf<AscendC::TPosition::VECCALC> reluInBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> reluOutBuf;

    AscendC::TBuf<AscendC::TPosition::VECCALC> kvRowBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> vRowBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> qRowBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> outRowBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> scalarBuf;

    // 维度信息
    uint32_t B;
    uint32_t QL;
    uint32_t KL;
    uint32_t D;

    // 元素计数（每batch）
    uint64_t perBatchQElems;
    uint64_t perBatchKElems;
    uint64_t perBatchKVElems;
    uint64_t perBatchElems;
};

extern "C" __global__ __aicore__ void linear_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLinearAttention op;
    op.Init(q, k, v, y, workspace,
            tiling_data.batch, tiling_data.q_len, tiling_data.k_len, tiling_data.d_model);
    op.Process();
}
