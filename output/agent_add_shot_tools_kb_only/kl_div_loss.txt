kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelKlDivLoss {
public:
    __aicore__ inline KernelKlDivLoss() {}
    __aicore__ inline void Init(GM_ADDR pred, GM_ADDR target, GM_ADDR loss, uint32_t totalLength, uint32_t tileNum, uint32_t batchSize)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->batchSize = batchSize;
        this->invBatch = 1.0f / static_cast<float>(batchSize);

        predGm.SetGlobalBuffer((__gm__ DTYPE_X *)pred + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        tgtGm.SetGlobalBuffer((__gm__ DTYPE_Y *)target + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        lossGm.SetGlobalBuffer((__gm__ DTYPE_Z *)loss, 1);

        pipe.InitBuffer(inQueuePred, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueTgt, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        // VECCALC buffer used for temporary tensors: log(pred), log(target), diff, lossVec
        pipe.InitBuffer(tmpQueue, BUFFER_NUM * 4, this->tileLength * sizeof(DTYPE_Z));
        // VECOUT buffer for scalar partial sum per tile
        pipe.InitBuffer(outQueueLoss, BUFFER_NUM, sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> predLocal = inQueuePred.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> tgtLocal = inQueueTgt.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(predLocal, predGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(tgtLocal, tgtGm[progress * this->tileLength], this->tileLength);
        inQueuePred.EnQue(predLocal);
        inQueueTgt.EnQue(tgtLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> predLocal = inQueuePred.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> tgtLocal = inQueueTgt.DeQue<DTYPE_Y>();

        AscendC::LocalTensor<DTYPE_Z> logPredLocal = tmpQueue.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> logTgtLocal = tmpQueue.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> diffLocal = tmpQueue.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> lossVecLocal = tmpQueue.AllocTensor<DTYPE_Z>();

        // log(pred) and log(target)
        AscendC::Log(logPredLocal, predLocal, this->tileLength);
        AscendC::Log(logTgtLocal, tgtLocal, this->tileLength);

        // diff = log(target) - log(pred)
        AscendC::Sub(diffLocal, logTgtLocal, logPredLocal, this->tileLength);

        // lossVec = target * diff
        AscendC::Mul(lossVecLocal, diffLocal, tgtLocal, this->tileLength);

        // partial scalar sum for this tile
        AscendC::LocalTensor<DTYPE_Z> partialLocal = outQueueLoss.AllocTensor<DTYPE_Z>();
        AscendC::ReduceSum(partialLocal, lossVecLocal, this->tileLength);

        // scale by 1 / batchSize to implement 'batchmean'
        AscendC::Muls(partialLocal, partialLocal, this->invBatch, 1);

        // atomic add to global loss scalar
        AscendC::SetAtomicAdd<DTYPE_Z>();
        AscendC::DataCopy(lossGm, partialLocal, 1);
        AscendC::SetAtomicNone();

        // free locals and queues
        outQueueLoss.EnQue<DTYPE_Z>(partialLocal);
        outQueueLoss.FreeTensor(partialLocal);

        tmpQueue.FreeTensor(lossVecLocal);
        tmpQueue.FreeTensor(diffLocal);
        tmpQueue.FreeTensor(logTgtLocal);
        tmpQueue.FreeTensor(logPredLocal);

        inQueuePred.FreeTensor(predLocal);
        inQueueTgt.FreeTensor(tgtLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePred, inQueueTgt;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM * 4> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueLoss;

    AscendC::GlobalTensor<DTYPE_X> predGm;
    AscendC::GlobalTensor<DTYPE_Y> tgtGm;
    AscendC::GlobalTensor<DTYPE_Z> lossGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t batchSize;
    float invBatch;
};

extern "C" __global__ __aicore__ void kl_div_loss_custom(GM_ADDR pred, GM_ADDR target, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelKlDivLoss op;
    op.Init(pred, target, loss, tiling_data.totalLength, tiling_data.tileNum, tiling_data.batchSize);
    op.Process();
}
"""
