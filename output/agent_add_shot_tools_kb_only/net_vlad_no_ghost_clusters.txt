kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

// 计算 v[b,k,:] = sum_n assignment[b,n,k] * x[b,n,:] - (sum_n assignment[b,n,k]) * clusters2_kd[k,:]
// 输入: x [B,N,D], assignment [B,N,K], clusters2_kd [K,D]
// 输出: vlad_bkd [B,K,D] (D 为连续内存，便于一次性写回)
class KernelNetVladNoGhostClusters {
public:
    __aicore__ inline KernelNetVladNoGhostClusters() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR assignment,
                                GM_ADDR clusters2_kd,
                                GM_ADDR vlad_bkd,
                                const uint32_t B,
                                const uint32_t N,
                                const uint32_t D,
                                const uint32_t K,
                                const uint32_t pairsPerBlock)
    {
        this->B = B;
        this->N = N;
        this->D = D;
        this->K = K;
        this->pairsPerBlock = pairsPerBlock;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, (uint64_t)B * N * D);
        asgGm.SetGlobalBuffer((__gm__ DTYPE*)assignment, (uint64_t)B * N * K);
        c2Gm.SetGlobalBuffer((__gm__ DTYPE*)clusters2_kd, (uint64_t)K * D);
        outGm.SetGlobalBuffer((__gm__ DTYPE*)vlad_bkd, (uint64_t)B * K * D);

        // 本 kernel 简单按向量长度 D 进行局部缓冲
        pipe.InitBuffer(xVecQue, 1, D * sizeof(DTYPE));
        pipe.InitBuffer(tmpVecQue, 1, D * sizeof(DTYPE));
        pipe.InitBuffer(accVecQue, 1, D * sizeof(DTYPE));
        pipe.InitBuffer(c2VecQue, 1, D * sizeof(DTYPE));
        pipe.InitBuffer(outVecQue, 1, D * sizeof(DTYPE));
        pipe.InitBuffer(asgScalarQue, 1, sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        const uint32_t totalPairs = B * K;
        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t startPair = blockIdx * pairsPerBlock;
        uint32_t endPair = startPair + pairsPerBlock;
        if (endPair > totalPairs) endPair = totalPairs;

        for (uint32_t p = startPair; p < endPair; ++p) {
            uint32_t b = p / K;
            uint32_t k = p % K;

            // 累加向量 acc = sum_n (assignment[b,n,k] * x[b,n,:])
            AccumulateBK(b, k);

            // 加载 clusters2_kd[k,:]
            AscendC::LocalTensor<DTYPE> c2Local = c2VecQue.AllocTensor<DTYPE>();
            AscendC::DataCopy(c2Local, c2Gm[k * D], D);
            c2VecQue.EnQue(c2Local);

            // 从队列取出 acc
            AscendC::LocalTensor<DTYPE> accLocal = accVecQue.DeQue<DTYPE>();
            // 取出 c2 向量
            c2Local = c2VecQue.DeQue<DTYPE>();

            // 标量 s = sum_n assignment[b,n,k]
            DTYPE s = sSumScalar;

            // tmp = c2Local * s
            AscendC::LocalTensor<DTYPE> tmpLocal = tmpVecQue.AllocTensor<DTYPE>();
            AscendC::Muls(tmpLocal, c2Local, s, D);

            // out = acc - tmp
            AscendC::LocalTensor<DTYPE> outLocal = outVecQue.AllocTensor<DTYPE>();
            AscendC::Sub(outLocal, accLocal, tmpLocal, D);

            // 写回到输出 [B,K,D]，D 连续
            AscendC::DataCopy(outGm[(uint64_t)b * K * D + (uint64_t)k * D], outLocal, D);

            // 释放局部张量
            c2VecQue.FreeTensor(c2Local);
            accVecQue.FreeTensor(accLocal);
            tmpVecQue.FreeTensor(tmpLocal);
            outVecQue.FreeTensor(outLocal);
        }
    }

private:
    __aicore__ inline void AccumulateBK(uint32_t b, uint32_t k)
    {
        // acc = sum_n (asg[b,n,k] * x[b,n,:])
        bool inited = false;
        sSumScalar = (DTYPE)0;

        AscendC::LocalTensor<DTYPE> accLocal = accVecQue.AllocTensor<DTYPE>();

        for (uint32_t n = 0; n < N; ++n) {
            // load x[b,n,:]
            AscendC::LocalTensor<DTYPE> xLocal = xVecQue.AllocTensor<DTYPE>();
            const uint64_t xOffset = ((uint64_t)b * N + n) * D;
            AscendC::DataCopy(xLocal, xGm[xOffset], D);
            xVecQue.EnQue(xLocal);

            // load assignment scalar a = assignment[b,n,k]
            AscendC::LocalTensor<DTYPE> asgScLocal = asgScalarQue.AllocTensor<DTYPE>();
            const uint64_t asgOffset = ((uint64_t)b * N + n) * K + k;
            AscendC::DataCopy(asgScLocal, asgGm[asgOffset], 1);
            asgScalarQue.EnQue(asgScLocal);

            xLocal = xVecQue.DeQue<DTYPE>();
            asgScLocal = asgScalarQue.DeQue<DTYPE>();
            DTYPE aScalar = asgScLocal.GetValue(0); // 获取标量

            // tmp = xLocal * aScalar
            AscendC::LocalTensor<DTYPE> tmpLocal = tmpVecQue.AllocTensor<DTYPE>();
            AscendC::Muls(tmpLocal, xLocal, aScalar, D);

            if (!inited) {
                // acc = tmp
                AscendC::DataCopy(accLocal, tmpLocal, D);
                inited = true;
            } else {
                // acc += tmp
                AscendC::Add(accLocal, accLocal, tmpLocal, D);
            }

            sSumScalar += aScalar;

            // 释放本轮局部
            xVecQue.FreeTensor(xLocal);
            asgScalarQue.FreeTensor(asgScLocal);
            tmpVecQue.FreeTensor(tmpLocal);
        }

        accVecQue.EnQue(accLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> xVecQue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpVecQue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> accVecQue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> c2VecQue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outVecQue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> asgScalarQue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> asgGm;
    AscendC::GlobalTensor<DTYPE> c2Gm;
    AscendC::GlobalTensor<DTYPE> outGm;

    uint32_t B;
    uint32_t N;
    uint32_t D;
    uint32_t K;
    uint32_t pairsPerBlock;

    DTYPE sSumScalar;
};

extern "C" __global__ __aicore__ void net_vlad_no_ghost_clusters_custom(
    GM_ADDR x, GM_ADDR assignment, GM_ADDR clusters2_kd, GM_ADDR vlad_bkd, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelNetVladNoGhostClusters op;
    op.Init(x,
            assignment,
            clusters2_kd,
            vlad_bkd,
            tiling_data.bLength(),
            tiling_data.nLength(),
            tiling_data.dLength(),
            tiling_data.kLength(),
            tiling_data.pairsPerBlock());
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(ModelNew, self).__init__()
        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        import math
        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        self.clusters = nn.Parameter(init_sc * torch.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        self.clusters2 = nn.Parameter(init_sc * torch.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size

    def forward(self, x, mask=None):
        # x: [B, N, D]
        B, N, D = x.shape
        x_flat = x.reshape(-1, self.feature_size)  # [B*N, D]

        if x_flat.device != self.clusters.device:
            raise ValueError(f"x.device {x_flat.device} != cluster.device {self.clusters.device}")

        assignment = torch.matmul(x_flat, self.clusters)  # [B*N, K+G]
        assignment = self.batch_norm(assignment)
        assignment = F.softmax(assignment, dim=1)  # [B*N, K+G]

        # 去除 ghost cluster
        assignment = assignment[:, :self.cluster_size]  # [B*N, K]
        assignment = assignment.view(B, N, self.cluster_size)  # [B, N, K]

        # clusters2: [1, D, K] -> 准备为 [K, D] 连续存储以便 kernel 一次性加载
        clusters2_kd = self.clusters2.squeeze(0).permute(1, 0).contiguous()  # [K, D]

        # 自定义核: 计算 vlad_core = sum_n a[b,n,k]*x[b,n,:] - (sum_n a[b,n,k])*c2[k,:]
        vlad_bkd = custom_ops_lib.net_vlad_no_ghost_clusters_custom(x, assignment, clusters2_kd)  # [B, K, D]
        vlad = vlad_bkd.permute(0, 2, 1).contiguous()  # [B, D, K]

        # L2 intra norm
        vlad = F.normalize(vlad, dim=1)

        # 展平 + L2 norm
        vlad = vlad.reshape(B, self.cluster_size * self.feature_size)
        vlad = F.normalize(vlad, dim=1)
        return vlad
'''
