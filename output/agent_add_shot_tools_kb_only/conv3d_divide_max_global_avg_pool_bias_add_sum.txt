kernel_src="""
#include "kernel_operator.h"

// 该自定义核实现：对输入 y([N, C, 1, 1, 1]) 与 bias([C, 1, 1, 1]) 逐元素相加，
// 然后沿通道维进行求和，输出 z([N, 1, 1, 1, 1])。
// Conv3d/除法/MaxPool/全局平均池化在 PyTorch 侧完成。
// 我们在核中使用 AscendC::Add 和 ReduceSum 做加法与通道求和。

constexpr int32_t BUFFER_NUM = 1;

class KernelConv3dDivideMaxGlobalAvgPoolBiasAddSum {
public:
    __aicore__ inline KernelConv3dDivideMaxGlobalAvgPoolBiasAddSum() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z, uint32_t batch, uint32_t channels, uint32_t tileNum)
    {
        this->batch = batch;
        this->channels = channels;
        this->tileNum = tileNum;
        this->tileLength = channels / tileNum;  // 简单等分通道

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, batch * channels);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_BIAS *)bias, channels);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, batch); // 每个 batch 输出一个标量

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueBias, BUFFER_NUM, this->tileLength * sizeof(DTYPE_BIAS));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, 32); // workLocal 最小空间，参考 Reduce 示例
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, sizeof(DTYPE_Z)); // 每次写出一个标量
    }

    __aicore__ inline void Process()
    {
        // 先将 bias 整体读入到 GM（分块读取到 UB）
        for (uint32_t b = 0; b < batch; ++b) {
            // 每个 batch 的通道向量：x[b, :, 1,1,1]
            DTYPE_Z acc = 0; // 通道求和累计（可作为局部标量）
            for (uint32_t t = 0; t < tileNum; ++t) {
                CopyIn(b, t);
                DTYPE_Z part = ComputeTile(); // 返回该 tile 的部分和
                acc += part;
            }
            // 输出写回
            CopyOut(b, acc);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t b, uint32_t t)
    {
        AscendC::LocalTensor<DTYPE_X>   xLocal   = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_BIAS> biasLocal = inQueueBias.AllocTensor<DTYPE_BIAS>();

        uint32_t offsetX   = b * channels + t * tileLength;
        uint32_t offsetBias = t * tileLength;
        AscendC::DataCopy(xLocal,   xGm[offsetX],   tileLength);
        AscendC::DataCopy(biasLocal, biasGm[offsetBias], tileLength);

        inQueueX.EnQue(xLocal);
        inQueueBias.EnQue(biasLocal);
    }

    __aicore__ inline DTYPE_Z ComputeTile()
    {
        AscendC::LocalTensor<DTYPE_X>   xLocal   = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_BIAS> biasLocal = inQueueBias.DeQue<DTYPE_BIAS>();

        // 元素加法
        AscendC::LocalTensor<DTYPE_Z> addLocal = outQueueZ.AllocTensor<DTYPE_Z>(); // 复用 out 队列做中间存储
        AscendC::Add(addLocal, xLocal, biasLocal, tileLength);

        // ReduceSum 通道分片求和（需要 workLocal）
        AscendC::LocalTensor<uint8_t> workLocal = tmpQueue.AllocTensor<uint8_t>();
        AscendC::LocalTensor<DTYPE_Z> sumLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // 假设 ReduceSum 接口原型与 ReduceMax 类似：ReduceSum(dstLocal, srcLocal, workLocal, count)
        AscendC::ReduceSum<DTYPE_Z>(sumLocal, addLocal, workLocal, tileLength);

        // 释放不再使用的本地张量
        outQueueZ.FreeTensor(addLocal);
        tmpQueue.FreeTensor(workLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueBias.FreeTensor(biasLocal);

        // 取出该 tile 的部分和
        DTYPE_Z part = sumLocal.GetValue(0); // 取标量
        outQueueZ.FreeTensor(sumLocal);
        return part;
    }

    __aicore__ inline void CopyOut(uint32_t b, DTYPE_Z acc)
    {
        // 将标量 acc 写回到 zGm[b]
        AscendC::LocalTensor<DTYPE_Z> outScalar = outQueueZ.AllocTensor<DTYPE_Z>();
        outScalar.SetValue(0, acc);
        AscendC::DataCopy(zGm[b], outScalar, 1);
        outQueueZ.FreeTensor(outScalar);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueBias;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X>   xGm;
    AscendC::GlobalTensor<DTYPE_BIAS> biasGm;
    AscendC::GlobalTensor<DTYPE_Z>   zGm;

    uint32_t batch;
    uint32_t channels;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv3d_divide_max_global_avg_pool_bias_add_sum_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dDivideMaxGlobalAvgPoolBiasAddSum op;
    op.Init(x, bias, z, tiling_data.batch, tiling_data.channels, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版模型：保留 Conv3d/除法/MaxPool/全局平均池化在 PyTorch，融合 bias 加法与通道求和到 AscendC 自定义核。
    """
    def __init__(self, in_channels, out_channels, kernel_size, divisor, pool_size, bias_shape, sum_dim):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.divisor = divisor
        self.max_pool = torch.nn.MaxPool3d(pool_size)
        self.global_avg_pool = torch.nn.AdaptiveAvgPool3d((1, 1, 1))
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.sum_dim = sum_dim  # 未在核中使用（核固定对通道维求和）

    def forward(self, x):
        x = self.conv(x)
        x = x / self.divisor
        x = self.max_pool(x)
        x = self.global_avg_pool(x)          # 形状变为 [N, C, 1, 1, 1]
        z = custom_ops_lib.conv3d_divide_max_global_avg_pool_bias_add_sum_custom(x, self.bias)
        # 输出 z: [N, 1, 1, 1, 1]，与原模型在 sum_dim=1 的语义一致
        return z
