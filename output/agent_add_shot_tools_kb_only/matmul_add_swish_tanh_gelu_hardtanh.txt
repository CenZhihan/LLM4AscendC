kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulAddSwishTanhGeluHardtanhCustom {
public:
    __aicore__ inline KernelMatmulAddSwishTanhGeluHardtanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add_value, GM_ADDR y,
                                uint32_t totalLength, uint32_t outDim, uint32_t tileNum)
    {
        this->totalLength = totalLength;
        this->outDim = outDim;
        this->tileNum = tileNum;

        uint32_t totalRows = totalLength / outDim;
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t rowsPerBlock = (totalRows + blocks - 1) / blocks;
        this->rowsPerBlock = rowsPerBlock;

        uint32_t startRow = rowsPerBlock * AscendC::GetBlockIdx();
        uint32_t validRows = totalRows > startRow ? (AscendC::Min(rowsPerBlock, totalRows - startRow)) : 0;
        this->validRows = validRows;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x) + startRow * outDim, validRows * outDim);
        addGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(add_value), outDim);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y) + startRow * outDim, validRows * outDim);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, outDim * sizeof(float));
        pipe.InitBuffer(inQueueAdd, BUFFER_NUM, outDim * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, outDim * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (validRows == 0) {
            return;
        }
        // Preload add_value once into queue
        AscendC::LocalTensor<float> addLocalOnce = inQueueAdd.AllocTensor<float>();
        AscendC::DataCopy(addLocalOnce, addGm, outDim);
        inQueueAdd.EnQue(addLocalOnce);

        for (uint32_t r = 0; r < validRows; ++r) {
            CopyIn(r);
            Compute();
            CopyOut(r);
        }

        AscendC::LocalTensor<float> addLocalFree = inQueueAdd.DeQue<float>();
        inQueueAdd.FreeTensor(addLocalFree);
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * outDim], outDim);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        // Fetch x and add_value
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> addLocal = inQueueAdd.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        // y = x + add_value (broadcast on last dim)
        AscendC::Add(yLocal, xLocal, addLocal, outDim);

        // Swish(y) with beta=1.0
        AscendC::Swish(yLocal, yLocal, outDim, 1.0f);

        // Tanh(y)
        AscendC::Tanh(yLocal, yLocal, outDim);

        // GELU(y)
        AscendC::Gelu(yLocal, yLocal, outDim);

        // Hardtanh clamp to [-1, 1]
        AscendC::HardTanh(yLocal, yLocal, outDim, -1.0f, 1.0f);

        outQueueY.EnQue<float>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueAdd.EnQue<float>(addLocal); // push back add_value for reuse
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[rowIdx * outDim], yLocal, outDim);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAdd;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> addGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t totalLength;
    uint32_t outDim;
    uint32_t tileNum;
    uint32_t rowsPerBlock;
    uint32_t validRows;
};

extern "C" __global__ __aicore__ void matmul_add_swish_tanh_gelu_hardtanh_custom(GM_ADDR x, GM_ADDR add_value, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulAddSwishTanhGeluHardtanhCustom op;
    op.Init(x, add_value, y, tiling_data.totalLength, tiling_data.outDim, tiling_data.tileNum);
    op.Process();
}
