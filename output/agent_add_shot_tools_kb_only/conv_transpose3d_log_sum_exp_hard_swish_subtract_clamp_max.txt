kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose3dLogSumExpHardSwishSubtractClampMax {
public:
    __aicore__ inline KernelConvTranspose3dLogSumExpHardSwishSubtractClampMax() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y, uint32_t totalSpatial, uint32_t channels, uint32_t tileNum)
    {
        this->totalSpatial = totalSpatial;
        this->channels = channels;
        this->tileNum = tileNum;

        // Split only along spatial dimension
        this->blockSpatial = totalSpatial / AscendC::GetBlockNum();
        this->tileLength = this->blockSpatial / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_Y*)bias);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y);

        // queues for data movement and temporaries
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));   // for per-channel loads
        pipe.InitBuffer(tmpQueue0, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));  // max buffer
        pipe.InitBuffer(tmpQueue1, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));  // sum buffer
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));  // output
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress)
    {
        uint32_t blockStartSpatial = this->blockSpatial * AscendC::GetBlockIdx();
        uint32_t startSpatial = blockStartSpatial + progress * this->tileLength;

        // Allocate temporaries: maxVec, sumVec
        AscendC::LocalTensor<DTYPE_X> maxVec = tmpQueue0.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> sumVec = tmpQueue1.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Initialize maxVec to a very small number and sumVec to 0
        DTYPE_X neg_large = static_cast<DTYPE_X>(-3.402823e+38f);
        DTYPE_X zero_val = static_cast<DTYPE_X>(0.0f);
        AscendC::Duplicate(maxVec, neg_large, this->tileLength);
        AscendC::Duplicate(sumVec, zero_val, this->tileLength);

        // First pass: compute element-wise max over channels (for numerical stability)
        for (uint32_t c = 0; c < channels; ++c) {
            AscendC::LocalTensor<DTYPE_X> chanVec = inQueueX.AllocTensor<DTYPE_X>();
            uint64_t gmOffset = static_cast<uint64_t>(c) * static_cast<uint64_t>(totalSpatial) + static_cast<uint64_t>(startSpatial);
            AscendC::DataCopy(chanVec, xGm[gmOffset], this->tileLength);
            AscendC::Max(maxVec, maxVec, chanVec, this->tileLength);
            inQueueX.FreeTensor(chanVec);
        }

        // Second pass: compute sum(exp(x - max))
        for (uint32_t c = 0; c < channels; ++c) {
            AscendC::LocalTensor<DTYPE_X> chanVec = inQueueX.AllocTensor<DTYPE_X>();
            uint64_t gmOffset = static_cast<uint64_t>(c) * static_cast<uint64_t>(totalSpatial) + static_cast<uint64_t>(startSpatial);
            AscendC::DataCopy(chanVec, xGm[gmOffset], this->tileLength);

            // temp = chanVec - maxVec
            AscendC::Sub(chanVec, chanVec, maxVec, this->tileLength);
            // temp = exp(temp)
            AscendC::Exp(chanVec, chanVec, this->tileLength);
            // sumVec += temp
            AscendC::Add(sumVec, sumVec, chanVec, this->tileLength);

            inQueueX.FreeTensor(chanVec);
        }

        // lse = ln(sumVec) + maxVec
        AscendC::Ln(sumVec, sumVec, this->tileLength);
        AscendC::Add(yLocal, sumVec, maxVec, this->tileLength);

        // HardSwish-like: y = y * sigmoid(y + 3) / 6
        // tmp1 = y + 3
        AscendC::Adds(yLocal, yLocal, static_cast<DTYPE_Z>(3.0f), this->tileLength);
        // tmp1 = -tmp1
        AscendC::Muls(yLocal, yLocal, static_cast<DTYPE_Z>(-1.0f), this->tileLength);
        // tmp1 = exp(tmp1)
        AscendC::Exp(yLocal, yLocal, this->tileLength);
        // denom = 1 + tmp1
        AscendC::Adds(yLocal, yLocal, static_cast<DTYPE_Z>(1.0f), this->tileLength);
        // sig = 1/denom
        AscendC::Reciprocal(yLocal, yLocal, this->tileLength);
        // hs = original_lse * sig
        // We need original lse again: recompute lse into sumVec to reuse buffers
        AscendC::Duplicate(sumVec, zero_val, this->tileLength);
        // Recompute lse into sumVec = ln(sum) + max stored in previous step: we still have ln(sum) in tmp? Not kept.
        // Instead, reconstruct using yLocal currently holds sigmoid(y + 3) where y is lse + 3 before neg/exp/add/recip
        // We stored lse in 'maxVec'? No. Recompute quickly: Not feasible here, so approximate by using yLocal as sigmoid
        // Workaround: keep lse in 'maxVec' before overwriting:
        // Note: Adjust order to preserve lse:
        // For clarity in code, the actual flow below stores lse into sumVec first.

        // Correction: Restart from after computing lse into yLocal:
        // Move lse into sumVec, then compute sigmoid into maxVec, then hs = sumVec * maxVec / 6

        // recompute lse to sumVec
        // (Recompute quickly using maxVec + ln(sumVec_original)). Since sumVec currently holds ln(sum) and maxVec holds max
        // sumVec = ln(sum) + max
        AscendC::Add(sumVec, sumVec, maxVec, this->tileLength);
        // maxVec will hold sigmoid(sumVec + 3)
        AscendC::Adds(maxVec, sumVec, static_cast<DTYPE_X>(3.0f), this->tileLength);
        AscendC::Muls(maxVec, maxVec, static_cast<DTYPE_X>(-1.0f), this->tileLength);
        AscendC::Exp(maxVec, maxVec, this->tileLength);
        AscendC::Adds(maxVec, maxVec, static_cast<DTYPE_X>(1.0f), this->tileLength);
        AscendC::Reciprocal(maxVec, maxVec, this->tileLength);
        // yLocal = sumVec * maxVec
        AscendC::Mul(yLocal, sumVec, maxVec, this->tileLength);
        AscendC::Muls(yLocal, yLocal, static_cast<DTYPE_Z>(1.0f/6.0f), this->tileLength);

        // Subtract bias (broadcast scalar)
        AscendC::LocalTensor<DTYPE_Y> biasLocal = inQueueX.AllocTensor<DTYPE_Y>();
        // Load scalar bias (assume bias tensor shape is 1 element)
        AscendC::DataCopy(biasLocal, biasGm, 1);
        DTYPE_Y biasVal = biasLocal.GetValue(0);
        inQueueX.FreeTensor(biasLocal);
        // yLocal = yLocal - biasVal
        AscendC::Adds(yLocal, yLocal, static_cast<DTYPE_Z>(-biasVal), this->tileLength);

        // Clamp to [-1, 1]
        AscendC::Maxs(yLocal, yLocal, static_cast<DTYPE_Z>(-1.0f), this->tileLength);
        AscendC::Mins(yLocal, yLocal, static_cast<DTYPE_Z>(1.0f), this->tileLength);

        // Enqueue output and free temps
        outQueueY.EnQue<DTYPE_Z>(yLocal);
        tmpQueue0.FreeTensor(maxVec);
        tmpQueue1.FreeTensor(sumVec);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        uint32_t blockStartSpatial = this->blockSpatial * AscendC::GetBlockIdx();
        uint32_t startSpatial = blockStartSpatial + progress * this->tileLength;
        AscendC::DataCopy(yGm[startSpatial], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue0, tmpQueue1;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> biasGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t totalSpatial;
    uint32_t channels;
    uint32_t tileNum;
    uint32_t blockSpatial;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_log_sum_exp_hard_swish_subtract_clamp_max_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dLogSumExpHardSwishSubtractClampMax op;
    op.Init(x, bias, y, tiling_data.totalSpatial, tiling_data.channels, tiling_data.tileNum);
    op.Process();
}
