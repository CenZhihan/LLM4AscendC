kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelGemmGroupNormHardtanh {
public:
    __aicore__ inline KernelGemmGroupNormHardtanh() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wT, GM_ADDR b, GM_ADDR gnW, GM_ADDR gnB,
                                GM_ADDR y,
                                uint32_t B, uint32_t K, uint32_t N,
                                uint32_t G, uint32_t Gsize, uint32_t tileK,
                                float htMin, float htMax, float eps)
    {
        batch = B;
        inFeatures = K;
        outFeatures = N;
        numGroups = G;
        groupSize = Gsize;
        kTile = tileK;
        hardtanhMin = htMin;
        hardtanhMax = htMax;
        epsilon = eps;

        // Split rows across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockId = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (batch + blocks - 1) / blocks;
        rowStart = rowsPerBlock * blockId;
        rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > batch) rowEnd = batch;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, (uint32_t)batch * inFeatures);
        wTGm.SetGlobalBuffer((__gm__ DTYPE*)wT, (uint32_t)inFeatures * outFeatures); // [K, N]
        bGm.SetGlobalBuffer((__gm__ DTYPE*)b, outFeatures);
        gnWGm.SetGlobalBuffer((__gm__ DTYPE*)gnW, outFeatures);
        gnBGm.SetGlobalBuffer((__gm__ DTYPE*)gnB, outFeatures);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, (uint32_t)batch * outFeatures);

        // Simple queues for local buffers
        pipe.InitBuffer(inQueueX, 1, sizeof(DTYPE) * inFeatures);
        pipe.InitBuffer(outQueueZ, 1, sizeof(DTYPE) * groupSize);
        pipe.InitBuffer(auxQueue, 1, sizeof(DTYPE) * groupSize);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            // For each group (tile on N aligned to groupSize so that we can do per-group normalization)
            for (uint32_t g = 0; g < numGroups; ++g) {
                uint32_t n0 = g * groupSize;

                // Local output vector for this row-group
                AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.AllocTensor<DTYPE>();
                // zero init
                AscendC::Duplicate(zLocal, (DTYPE)0.0f, groupSize);

                // GEMM: y[r, n0:n0+groupSize] = dot(x[r,:], wT[:, n0:n0+groupSize]) + b[n0:n0+groupSize]
                // Accumulate over K in scalar fashion for simplicity
                for (uint32_t k = 0; k < inFeatures; ++k) {
                    DTYPE xval = *((__gm__ DTYPE*)xGm.GetPhyAddr() + r * inFeatures + k);
                    // load wT row k slice [n0 : n0+groupSize]
                    for (uint32_t j = 0; j < groupSize; ++j) {
                        DTYPE wval = *((__gm__ DTYPE*)wTGm.GetPhyAddr() + k * outFeatures + (n0 + j));
                        // zLocal[j] += xval * wval
                        AscendC::LocalTensor<DTYPE> tmp = auxQueue.AllocTensor<DTYPE>();
                        AscendC::Duplicate(tmp, (DTYPE)(xval * wval), 1);
                        // element-wise add 1 element at position j
                        // Write-accumulate: using scalar add to vector position j
                        // Fallback: copy current zLocal[j] to scalar, add, and write back
                        // Note: This is a conceptual implementation to illustrate fusion
                        // Real kernel should vectorize for performance
                        // Read zLocal[j] -> tmp2
                        // Here we simulate by DataCopy 1 element
                        AscendC::DataCopy(tmp, zLocal[j], 1);
                        DTYPE acc = (*((DTYPE*)tmp.GetPhyAddr()));
                        acc += xval * wval;
                        // store back
                        (*((DTYPE*)tmp.GetPhyAddr())) = acc;
                        AscendC::DataCopy(zLocal[j], tmp, 1);
                        auxQueue.FreeTensor(tmp);
                    }
                }
                // Add bias
                for (uint32_t j = 0; j < groupSize; ++j) {
                    DTYPE bval = *((__gm__ DTYPE*)bGm.GetPhyAddr() + (n0 + j));
                    AscendC::LocalTensor<DTYPE> tmp = auxQueue.AllocTensor<DTYPE>();
                    AscendC::DataCopy(tmp, zLocal[j], 1);
                    DTYPE v = (*((DTYPE*)tmp.GetPhyAddr()));
                    v += bval;
                    (*((DTYPE*)tmp.GetPhyAddr())) = v;
                    AscendC::DataCopy(zLocal[j], tmp, 1);
                    auxQueue.FreeTensor(tmp);
                }

                // GroupNorm (per row per group)
                // Compute mean
                DTYPE mean = 0.f;
                for (uint32_t j = 0; j < groupSize; ++j) {
                    AscendC::LocalTensor<DTYPE> tmp = auxQueue.AllocTensor<DTYPE>();
                    AscendC::DataCopy(tmp, zLocal[j], 1);
                    mean += (*((DTYPE*)tmp.GetPhyAddr()));
                    auxQueue.FreeTensor(tmp);
                }
                mean /= (DTYPE)groupSize;
                // Compute variance
                DTYPE var = 0.f;
                for (uint32_t j = 0; j < groupSize; ++j) {
                    AscendC::LocalTensor<DTYPE> tmp = auxQueue.AllocTensor<DTYPE>();
                    AscendC::DataCopy(tmp, zLocal[j], 1);
                    DTYPE d = (*((DTYPE*)tmp.GetPhyAddr())) - mean;
                    var += d * d;
                    auxQueue.FreeTensor(tmp);
                }
                var /= (DTYPE)groupSize;
                DTYPE invStd = (DTYPE)1.0f / (DTYPE)AscendC::sqrt(var + epsilon);

                // Apply affine and HardTanh, then write to GM
                for (uint32_t j = 0; j < groupSize; ++j) {
                    DTYPE gamma = *((__gm__ DTYPE*)gnWGm.GetPhyAddr() + (n0 + j));
                    DTYPE beta = *((__gm__ DTYPE*)gnBGm.GetPhyAddr() + (n0 + j));
                    AscendC::LocalTensor<DTYPE> tmp = auxQueue.AllocTensor<DTYPE>();
                    AscendC::DataCopy(tmp, zLocal[j], 1);
                    DTYPE v = (*((DTYPE*)tmp.GetPhyAddr()));
                    // normalize
                    v = (v - mean) * invStd;
                    // affine
                    v = v * gamma + beta;
                    // hardtanh
                    if (v < hardtanhMin) v = hardtanhMin;
                    if (v > hardtanhMax) v = hardtanhMax;
                    // write to output y
                    *((__gm__ DTYPE*)yGm.GetPhyAddr() + r * outFeatures + (n0 + j)) = v;
                    auxQueue.FreeTensor(tmp);
                }

                outQueueZ.FreeTensor(zLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> auxQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wTGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> gnWGm;
    AscendC::GlobalTensor<DTYPE> gnBGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t batch;
    uint32_t inFeatures;
    uint32_t outFeatures;
    uint32_t numGroups;
    uint32_t groupSize;
    uint32_t kTile;
    uint32_t rowStart;
    uint32_t rowEnd;
    float hardtanhMin;
    float hardtanhMax;
    float epsilon;
};

extern "C" __global__ __aicore__ void gemm_group_norm_hardtanh_custom(
    GM_ADDR x, GM_ADDR weight_t, GM_ADDR bias, GM_ADDR gn_weight, GM_ADDR gn_bias,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelGemmGroupNormHardtanh op;
    op.Init(x, weight_t, bias, gn_weight, gn_bias, y,
            tilingData.batchSize, tilingData.inFeatures, tilingData.outFeatures,
            tilingData.numGroups, tilingData.groupSize, tilingData.tileK,
            tilingData.hardtanhMin, tilingData.hardtanhMax, tilingData.epsilon);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused GEMM + GroupNorm + HardTanh using custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, num_groups, hardtanh_min, hardtanh_max):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features, bias=True)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features, affine=True)
        self.num_groups = num_groups
        self.hardtanh_min = float(hardtanh_min)
        self.hardtanh_max = float(hardtanh_max)
        self.epsilon = 1.0e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Pass weight as transposed [K, N] to match kernel assumption
        wt = self.gemm.weight.t().contiguous()
        b = self.gemm.bias
        gn_w = self.group_norm.weight
        gn_b = self.group_norm.bias
        # Kernel expects float or float16 inputs, output is float
        return custom_ops_lib.gemm_group_norm_hardtanh_custom(
            x, wt, b, gn_w, gn_b, self.num_groups, self.hardtanh_min, self.hardtanh_max, self.epsilon
        )
