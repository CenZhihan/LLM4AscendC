kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t TILE_ELEMS = 4096;

class KernelInceptionConcatC {
public:
    __aicore__ inline KernelInceptionConcatC() {}

    __aicore__ inline void Init(GM_ADDR b1, GM_ADDR b2, GM_ADDR b3, GM_ADDR b4, GM_ADDR z,
                                uint32_t n, uint32_t h, uint32_t w,
                                uint32_t c1, uint32_t c2, uint32_t c3, uint32_t c4, uint32_t /*tileNum*/) {
        this->n = n;
        this->h = h;
        this->w = w;
        this->hw = h * w;
        this->c1 = c1;
        this->c2 = c2;
        this->c3 = c3;
        this->c4 = c4;
        this->ct = c1 + c2 + c3 + c4;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t base = n / blockNum;
        uint32_t rem = n % blockNum;
        uint32_t localN = base + (blockIdx < rem ? 1 : 0);
        uint32_t startN = blockIdx * base + (blockIdx < rem ? blockIdx : rem);

        this->localN = localN;

        uint64_t offsetB1 = static_cast<uint64_t>(startN) * c1 * hw;
        uint64_t offsetB2 = static_cast<uint64_t>(startN) * c2 * hw;
        uint64_t offsetB3 = static_cast<uint64_t>(startN) * c3 * hw;
        uint64_t offsetB4 = static_cast<uint64_t>(startN) * c4 * hw;
        uint64_t offsetZ  = static_cast<uint64_t>(startN) * ct * hw;

        b1Gm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(b1) + offsetB1, static_cast<uint32_t>(localN * c1 * hw));
        b2Gm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(b2) + offsetB2, static_cast<uint32_t>(localN * c2 * hw));
        b3Gm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(b3) + offsetB3, static_cast<uint32_t>(localN * c3 * hw));
        b4Gm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(b4) + offsetB4, static_cast<uint32_t>(localN * c4 * hw));
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(z) + offsetZ, static_cast<uint32_t>(localN * ct * hw));

        pipe.InitBuffer(outQueue, BUFFER_NUM, TILE_ELEMS * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint64_t cumC1 = 0;
        uint64_t cumC2 = c1;
        uint64_t cumC3 = c1 + c2;
        uint64_t cumC4 = c1 + c2 + c3;

        for (uint32_t ni = 0; ni < localN; ++ni) {
            uint64_t zBase = static_cast<uint64_t>(ni) * ct * hw;

            CopyBranch(b1Gm, c1, zBase + cumC1 * hw);
            CopyBranch(b2Gm, c2, zBase + cumC2 * hw);
            CopyBranch(b3Gm, c3, zBase + cumC3 * hw);
            CopyBranch(b4Gm, c4, zBase + cumC4 * hw);
        }
    }

private:
    __aicore__ inline void CopyBranch(AscendC::GlobalTensor<float>& src, uint32_t c, uint64_t zOffset) {
        uint64_t branchSize = static_cast<uint64_t>(c) * hw;
        uint64_t copied = 0;
        while (copied < branchSize) {
            uint32_t curr = static_cast<uint32_t>(AscendC::min<uint64_t>(TILE_ELEMS, branchSize - copied));
            AscendC::LocalTensor<float> buf = outQueue.AllocTensor<float>();
            AscendC::DataCopy(buf, src[copied], curr);
            AscendC::DataCopy(zGm[zOffset + copied], buf, curr);
            outQueue.FreeTensor(buf);
            copied += curr;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<float> b1Gm;
    AscendC::GlobalTensor<float> b2Gm;
    AscendC::GlobalTensor<float> b3Gm;
    AscendC::GlobalTensor<float> b4Gm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t n, h, w, hw;
    uint32_t c1, c2, c3, c4, ct;
    uint32_t localN;
};

extern "C" __global__ __aicore__ void googlenet_inception_module_custom(GM_ADDR b1, GM_ADDR b2, GM_ADDR b3, GM_ADDR b4, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelInceptionConcatC op;
    op.Init(b1, b2, b3, b4, z,
            tilingData.n(), tilingData.h(), tilingData.w(),
            tilingData.c1(), tilingData.c2(), tilingData.c3(), tilingData.c4(),
            tilingData.tileNum());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_1x1, reduce_3x3, out_3x3, reduce_5x5, out_5x5, pool_proj):
        super(ModelNew, self).__init__()
        self.branch1x1 = torch.nn.Conv2d(in_channels, out_1x1, kernel_size=1)
        self.branch3x3 = torch.nn.Sequential(
            torch.nn.Conv2d(in_channels, reduce_3x3, kernel_size=1),
            torch.nn.Conv2d(reduce_3x3, out_3x3, kernel_size=3, padding=1)
        )
        self.branch5x5 = torch.nn.Sequential(
            torch.nn.Conv2d(in_channels, reduce_5x5, kernel_size=1),
            torch.nn.Conv2d(reduce_5x5, out_5x5, kernel_size=5, padding=2)
        )
        self.branch_pool = torch.nn.Sequential(
            torch.nn.MaxPool2d(kernel_size=3, stride=1, padding=1),
            torch.nn.Conv2d(in_channels, pool_proj, kernel_size=1)
        )

    def forward(self, x):
        b1 = self.branch1x1(x)
        b2 = self.branch3x3(x)
        b3 = self.branch5x5(x)
        b4 = self.branch_pool(x)
        return custom_ops_lib.googlenet_inception_module_custom(b1, b2, b3, b4)
'''
