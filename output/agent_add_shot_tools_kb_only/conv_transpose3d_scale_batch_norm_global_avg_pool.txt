kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGAP3D {
public:
    __aicore__ inline KernelGAP3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t n, uint32_t c, uint32_t dhw, uint32_t tileLen)
    {
        this->n = n;
        this->c = c;
        this->dhw = dhw;
        this->tileLen = tileLen;

        totalPairs = n * c;

        // 简单按 (n*c) 维度切分给各个block
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        pairsPerBlock = (totalPairs + blockNum - 1) / blockNum;
        startPair = pairsPerBlock * blockIdx;
        endPair = startPair + pairsPerBlock;
        if (endPair > totalPairs) endPair = totalPairs;

        // 为当前block设置GM窗口
        uint32_t startElem = startPair * dhw;
        uint32_t elemCount = (endPair > startPair) ? (endPair - startPair) * dhw : 0;

        xGm.SetGlobalBuffer((__gm__ float*)x + startElem, elemCount);
        yGm.SetGlobalBuffer((__gm__ float*)y + startPair, (endPair > startPair) ? (endPair - startPair) : 0);

        // 缓冲区初始化
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(float));
        pipe.InitBuffer(tmpQueueA, 1, this->tileLen * sizeof(float)); // ReduceSum dst/work临时缓冲
        pipe.InitBuffer(tmpQueueB, 1, this->tileLen * sizeof(float));
        pipe.InitBuffer(outQueueY, 1, 1 * sizeof(float)); // 单元素输出
    }

    __aicore__ inline void Process()
    {
        if (startPair >= endPair) {
            return;
        }
        uint32_t localPairs = endPair - startPair;
        for (uint32_t p = 0; p < localPairs; ++p) {
            // 对于pair p（对应某个(n,c)），累加其DHW维度
            uint32_t remain = dhw;
            uint64_t baseOffset = static_cast<uint64_t>(p) * dhw; // 在当前block窗口内的偏移
            float sumVal = 0.0f;

            uint32_t prog = 0;
            while (remain > 0) {
                uint32_t len = remain > tileLen ? tileLen : remain;

                // 拷贝tile
                AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm[baseOffset + prog * tileLen], len);

                // ReduceSum tile
                AscendC::LocalTensor<float> dstLocal = tmpQueueA.AllocTensor<float>();
                AscendC::LocalTensor<float> workLocal = tmpQueueB.AllocTensor<float>();
                AscendC::ReduceSum(dstLocal, xLocal, workLocal, len);
                float part = AscendC::GetAccVal<float>();
                sumVal += part;

                tmpQueueA.FreeTensor(dstLocal);
                tmpQueueB.FreeTensor(workLocal);
                inQueueX.EnQue(xLocal); // 立即释放
                inQueueX.FreeTensor(xLocal);

                remain -= len;
                ++prog;
            }

            float meanVal = sumVal / static_cast<float>(dhw);

            // 将mean写回GM
            AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
            AscendC::Duplicate(yLocal, meanVal, 1);
            AscendC::DataCopy(yGm[p], yLocal, 1);
            outQueueY.EnQue<float>(yLocal);
            outQueueY.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueA;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueB;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t dhw;
    uint32_t tileLen;

    uint32_t totalPairs;
    uint32_t pairsPerBlock;
    uint32_t startPair;
    uint32_t endPair;
};

extern "C" __global__ __aicore__ void conv_transpose3d_scale_batch_norm_global_avg_pool_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    KernelGAP3D op;
    op.Init(x, y,
            tilingData.n(),
            tilingData.c(),
            tilingData.dhw(),
            tilingData.tileLen());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版本：
    - 仍使用PyTorch的ConvTranspose3d完成反卷积；
    - 在PyTorch侧完成scale放大；
    - 使用自定义AscendC算子计算全局平均池化（对 N,C 维的每个通道在 D,H,W 上取均值）；
    - 最后在PyTorch侧完成BatchNorm的仿射变换：gamma * ((avg - mean)/sqrt(var+eps)) + beta。
    """
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor, eps=1e-5, momentum=0.1):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size)
        self.scale_factor = scale_factor
        self.batch_norm = torch.nn.BatchNorm3d(out_channels, eps=eps, momentum=momentum)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = x * self.scale_factor
        # 自定义AscendC核：全局平均池化
        avg = custom_ops_lib.conv_transpose3d_scale_batch_norm_global_avg_pool_custom(x)  # [N,C,1,1,1]
        # BN 仿射：gamma * ((avg - mean)/sqrt(var+eps)) + beta
        gamma = self.batch_norm.weight.view(1, -1, 1, 1, 1)
        beta = self.batch_norm.bias.view(1, -1, 1, 1, 1)
        mean = self.batch_norm.running_mean.view(1, -1, 1, 1, 1)
        var = self.batch_norm.running_var.view(1, -1, 1, 1, 1)
        y = gamma * (avg - mean) / torch.sqrt(var + self.batch_norm.eps) + beta
        return y
'''
