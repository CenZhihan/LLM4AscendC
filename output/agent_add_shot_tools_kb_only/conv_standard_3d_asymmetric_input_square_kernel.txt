kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_Y = float;

constexpr int32_t ONE_ELEM = 1;

class KernelConvStandard3dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, const uint32_t n, const uint32_t cIn,
                                const uint32_t cOut, const uint32_t h, const uint32_t w_in,
                                const uint32_t d, const uint32_t k, const uint32_t hOut, const uint32_t wOut) {
        this->n = n;
        this->cIn = cIn;
        this->cOut = cOut;
        this->h = h;
        this->w = w_in;
        this->d = d;
        this->k = k;
        this->hOut = hOut;
        this->wOut = wOut;

        uint64_t xLen = (uint64_t)n * cIn * h * w * d;
        uint64_t wLen = (uint64_t)cOut * cIn * k * k;
        uint64_t yLen = (uint64_t)n * cOut * hOut * wOut * d;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_X *>(x), xLen);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_W *>(w), wLen);
        yGm.SetGlobalBuffer(reinterpret_cast__gm__ DTYPE_Y *>(y), yLen);

        pipe.InitBuffer(inQueueX, 1, sizeof(DTYPE_X) * ONE_ELEM);
        pipe.InitBuffer(inQueueW, 1, sizeof(DTYPE_W) * ONE_ELEM);
        pipe.InitBuffer(tmpQueue, 1, sizeof(DTYPE_Y) * ONE_ELEM);
        pipe.InitBuffer(outQueue, 1, sizeof(DTYPE_Y) * ONE_ELEM);

        // split output channels across blocks
        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();
        this->ocPerBlock = (cOut + blockNum - 1) / blockNum;
        this->ocStart = this->blockIdx * this->ocPerBlock;
        this->ocEnd = (this->ocStart + this->ocPerBlock) > cOut ? cOut : (this->ocStart + this->ocPerBlock);
    }

    __aicore__ inline void Process() {
        // Loop over N, D, OC range for this block, spatial Hout x Wout
        for (uint32_t nIdx = 0; nIdx < n; ++nIdx) {
            for (uint32_t dIdx = 0; dIdx < d; ++dIdx) {
                for (uint32_t oc = ocStart; oc < ocEnd; ++oc) {
                    for (uint32_t oh = 0; oh < hOut; ++oh) {
                        for (uint32_t ow = 0; ow < wOut; ++ow) {
                            ComputeOne(nIdx, dIdx, oc, oh, ow);
                        }
                    }
                }
            }
        }
    }

private:
    __aicore__ inline uint64_t XOffset(uint32_t nIdx, uint32_t cIdx, uint32_t hIdx, uint32_t wIdx, uint32_t dIdx) const {
        // Layout: [N, C, H, W, D]
        return (((((uint64_t)nIdx * cIn + cIdx) * h + hIdx) * w + wIdx) * d + dIdx);
    }
    __aicore__ inline uint64_t WOffset(uint32_t oc, uint32_t ic, uint32_t kh, uint32_t kw) const {
        // Layout: [Cout, Cin, K, K]
        return (((uint64_t)oc * cIn + ic) * k + kh) * k + kw;
    }
    __aicore__ inline uint64_t YOffset(uint32_t nIdx, uint32_t oc, uint32_t oh, uint32_t ow, uint32_t dIdx) const {
        // Layout: [N, Cout, Hout, Wout, D]
        return (((((uint64_t)nIdx * cOut + oc) * hOut + oh) * wOut + ow) * d + dIdx);
    }

    __aicore__ inline void ComputeOne(uint32_t nIdx, uint32_t dIdx, uint32_t oc, uint32_t oh, uint32_t ow) {
        // accLocal = 0
        AscendC::LocalTensor<DTYPE_Y> accLocal = outQueue.AllocTensor<DTYPE_Y>();
        AscendC::Duplicate(accLocal, static_cast<DTYPE_Y>(0), ONE_ELEM);

        for (uint32_t ic = 0; ic < cIn; ++ic) {
            for (uint32_t kh = 0; kh < k; ++kh) {
                for (uint32_t kw = 0; kw < k; ++kw) {
                    // Load one input scalar
                    AscendC::LocalTensor<DTYPE_X> inLocalBuf = inQueueX.AllocTensor<DTYPE_X>();
                    uint64_t xoff = XOffset(nIdx, ic, oh + kh, ow + kw, dIdx);
                    AscendC::DataCopy(inLocalBuf, xGm[xoff], ONE_ELEM);
                    inQueueX.EnQue(inLocalBuf);

                    // Load one weight scalar
                    AscendC::LocalTensor<DTYPE_W> wLocalBuf = inQueueW.AllocTensor<DTYPE_W>();
                    uint64_t woff = WOffset(oc, ic, kh, kw);
                    AscendC::DataCopy(wLocalBuf, wGm[woff], ONE_ELEM);
                    inQueueW.EnQue(wLocalBuf);

                    // DeQue for compute
                    AscendC::LocalTensor<DTYPE_X> inLocal = inQueueX.DeQue<DTYPE_X>();
                    AscendC::LocalTensor<DTYPE_W> wtLocal = inQueueW.DeQue<DTYPE_W>();

                    // tmp = in * w
                    AscendC::LocalTensor<DTYPE_Y> tmpLocal = tmpQueue.AllocTensor<DTYPE_Y>();
                    AscendC::Mul(tmpLocal, inLocal, wtLocal, ONE_ELEM);

                    // acc += tmp
                    AscendC::Add(accLocal, accLocal, tmpLocal, ONE_ELEM);

                    // free
                    tmpQueue.FreeTensor(tmpLocal);
                    inQueueX.FreeTensor(inLocal);
                    inQueueW.FreeTensor(wtLocal);
                }
            }
        }

        // store acc to output
        uint64_t yoff = YOffset(nIdx, oc, oh, ow, dIdx);
        AscendC::DataCopy(yGm[yoff], accLocal, ONE_ELEM);
        outQueue.FreeTensor(accLocal);
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue;

    uint32_t n, cIn, cOut, h, w, d, k, hOut, wOut;
    uint32_t blockNum, blockIdx;
    uint32_t ocPerBlock, ocStart, ocEnd;
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dAsymmetricInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c_in, tiling_data.c_out,
            tiling_data.h, tiling_data.w, tiling_data.d,
            tiling_data.k, tiling_data.hout, tiling_data.wout);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super().__init__()
        # Keep a Conv3d module only to own the parameters (weights)
        # Kernel is (K, K, 1) as required
        self.conv3d = torch.nn.Conv3d(in_channels, out_channels, (kernel_size, kernel_size, 1),
                                      stride=stride, padding=padding, dilation=dilation, groups=groups, bias=bias)

        # Note: The custom kernel here implements stride=1, padding=0, dilation=1, groups=1, bias=False.

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x is expected as [N, C, H, W, D]
        # Weight from Conv3d is [Cout, Cin, Kd(=1), Kh, Kw], take the 2D slice at d=0
        weight_2d = self.conv3d.weight[:,:,0,:,:].contiguous()
        return custom_ops_lib.conv_standard_3d_asymmetric_input_square_kernel_custom(x, weight_2d)
