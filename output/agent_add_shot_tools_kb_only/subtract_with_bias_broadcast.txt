kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelSubtractWithBiasBroadcast {
public:
    __aicore__ inline KernelSubtractWithBiasBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z, uint32_t outter, uint32_t inner)
    {
        this->rowsTotal = outter;
        this->inner = inner;
        this->rowsPerBlock = outter / AscendC::GetBlockNum();

        xGm.SetGlobalBuffer((__gm__ float *)x + AscendC::GetBlockIdx() * rowsPerBlock * inner, rowsPerBlock * inner);
        zGm.SetGlobalBuffer((__gm__ float *)z + AscendC::GetBlockIdx() * rowsPerBlock * inner, rowsPerBlock * inner);
        biasGm.SetGlobalBuffer((__gm__ float *)bias, inner);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, inner * sizeof(float));
        pipe.InitBuffer(inQueueBias, BUFFER_NUM, inner * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, inner * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsPerBlock; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<float> biasLocal = inQueueBias.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * inner], inner);
        AscendC::DataCopy(biasLocal, biasGm, inner);
        inQueueX.EnQue(xLocal);
        inQueueBias.EnQue(biasLocal);
    }

    __aicore__ inline void Compute(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> biasLocal = inQueueBias.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

        AscendC::Sub(zLocal, xLocal, biasLocal, inner);

        outQueueZ.EnQue<float>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueBias.FreeTensor(biasLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm[rowIdx * inner], zLocal, inner);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueBias;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> biasGm;
    AscendC::GlobalTensor<float> zGm;
    uint32_t rowsTotal;
    uint32_t rowsPerBlock;
    uint32_t inner;
};

extern "C" __global__ __aicore__ void subtract_with_bias_broadcast_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSubtractWithBiasBroadcast op;
    op.Init(x, bias, z, tiling_data.outter, tiling_data.inner);
    op.Process();
}
