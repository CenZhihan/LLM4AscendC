kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelMatmulScaleResidualAddClampLogSumExpMish {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t batch, uint32_t hidden,
                                uint32_t tileBatch, float scale, float clampMin, float clampMax)
    {
        this->batch = batch;
        this->hidden = hidden;
        this->tileBatch = tileBatch;
        this->scale = scale;
        this->clampMin = clampMin;
        this->clampMax = clampMax;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // shard by batch rows
        uint32_t batchPerBlock = (batch + blockNum - 1) / blockNum;
        uint32_t startRow = batchPerBlock * blockIdx;
        if (startRow > batch) startRow = batch;
        uint32_t endRow = startRow + batchPerBlock;
        if (endRow > batch) endRow = batch;
        this->rowsThisBlock = (endRow > startRow) ? (endRow - startRow) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + startRow * hidden, rowsThisBlock * hidden);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + startRow * 1, rowsThisBlock * 1);

        uint32_t tileRows = this->tileBatch;
        if (tileRows == 0) tileRows = 1;
        if (tileRows > rowsThisBlock) tileRows = rowsThisBlock;

        this->tileRows = tileRows;
        // Buffers:
        // - main working buffer: [tileRows, hidden]
        // - tmp for LogSoftMax: use tmpQueue
        // - output buffer: [tileRows, 1]
        pipe.InitBuffer(inQueueX, BUFFER_NUM, tileRows * hidden * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, (tileRows * hidden + tileRows * 64) * sizeof(uint8_t)); // temp
        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileRows * 1 * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (rowsThisBlock == 0) { return; }
        int32_t loops = (rowsThisBlock + tileRows - 1) / tileRows;
        for (int32_t i = 0; i < loops; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint32_t curRows = GetCurRows(progress);
        if (curRows == 0) { return; }

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // move [curRows * hidden]
        AscendC::DataCopy(xLocal, xGm[progress * tileRows * hidden], curRows * hidden);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        uint32_t curRows = GetCurRows(progress);
        if (curRows == 0) { return; }

        // tensors
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<uint8_t> tmpLocal = tmpQueue.AllocTensor<uint8_t>();

        // Working buffer view
        AscendC::LocalTensor<DTYPE_X> workLocal = xLocal; // shape [curRows, hidden]

        // 1) scale: x = x * scale
        AscendC::Muls(workLocal, workLocal, static_cast<DTYPE_X>(scale), curRows * hidden);

        // 2) residual add: x = x + x
        AscendC::Add(workLocal, workLocal, workLocal, curRows * hidden);

        // 3) clamp: x = min(max(x, clampMin), clampMax)
        AscendC::Maxs(workLocal, workLocal, static_cast<DTYPE_X>(clampMin), curRows * hidden);
        AscendC::Mins(workLocal, workLocal, static_cast<DTYPE_X>(clampMax), curRows * hidden);

        // 4) logsumexp along last axis (hidden) -> [curRows, 1]
        //    Use LogSoftMax to get maxTensor and sumTensor, then LSE = log(sum) + max.
        //    Allocate aux buffers from tmpLocal
        // layout tmp: first part for logsoftmax dst(dummy), second for max,sum
        // Prepare shape info for softmax last-axis reduction
        AscendC::SoftMaxShapeInfo softmaxInfo = {curRows, hidden, curRows, hidden};
        // carve tmp buffers
        // dst dummy same size as workLocal
        AscendC::LocalTensor<DTYPE_X> lsmDst = reinterpret_cast<AscendC::LocalTensor<DTYPE_X>&>(tmpLocal);
        lsmDst.SetSize(curRows * hidden);
        // move ptr forward by curRows*hidden*sizeof(DTYPE_X)
        AscendC::LocalTensor<uint8_t> tmpRemain = tmpLocal[curRows * hidden * sizeof(DTYPE_X)];
        // max and sum buffers: broadcasted per-row blocks
        AscendC::LocalTensor<DTYPE_X> maxLocal = reinterpret_cast<AscendC::LocalTensor<DTYPE_X>&>(tmpRemain);
        maxLocal.SetSize(curRows * 8); // one data block (32B) => 8 floats per row
        AscendC::LocalTensor<uint8_t> tmpRemain2 = tmpRemain[curRows * 8 * sizeof(DTYPE_X)];
        AscendC::LocalTensor<DTYPE_X> sumLocal = reinterpret_cast<AscendC::LocalTensor<DTYPE_X>&>(tmpRemain2);
        sumLocal.SetSize(curRows * 8);

        // call LogSoftMax to populate maxLocal and sumLocal (dst ignored)
        // Note: need tiling for LogSoftMax; use a default tiling holder
        AscendC::LogSoftMax<DTYPE_X, false>(lsmDst, sumLocal, maxLocal, workLocal, tmpRemain2, softmaxtiling_, softmaxInfo);

        // compute lseLocal = log(sumLocal) + maxLocal; result broadcasted over 8 elements per row
        AscendC::Ln(sumLocal, sumLocal, curRows * 8);
        AscendC::Add(sumLocal, sumLocal, maxLocal, curRows * 8); // sumLocal now holds lse broadcasted

        // 5) mish(lse) = lse * tanh(softplus(lse)), softplus = ln(1 + exp(lse))
        //    Then y = lse * mish(lse) = lse^2 * tanh(softplus(lse))
        //    We only need one value per row; take the first element of each row from sumLocal
        //    Prepare temporaries for softplus/tanh on the first column
        // Extract first-column LSE into yLocal [curRows, 1]
        for (uint32_t r = 0; r < curRows; ++r) {
            // sumLocal index r*8 is the first in its data block
            AscendC::DataCopy(yLocal[r], sumLocal[r * 8], 1);
        }

        // Compute softplus/tanh in-place on yLocal
        AscendC::LocalTensor<DTYPE_Y> tmpCol = yLocal; // alias
        // softplus: ln(1 + exp(y))
        AscendC::Exp(tmpCol, yLocal, curRows);      // tmpCol = exp(y)
        AscendC::Adds(tmpCol, tmpCol, static_cast<DTYPE_Y>(1.0f), curRows); // tmpCol = 1 + exp(y)
        AscendC::Ln(tmpCol, tmpCol, curRows);       // tmpCol = ln(1 + exp(y))
        // tanh(softplus)
        AscendC::Tanh(tmpCol, tmpCol, curRows);     // tmpCol = tanh(softplus(y))
        // y = y * y
        AscendC::Mul(yLocal, yLocal, yLocal, curRows);
        // y = y * tanh(softplus(y))
        AscendC::Mul(yLocal, yLocal, tmpCol, curRows);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        uint32_t curRows = GetCurRows(progress);
        if (curRows == 0) { return; }

        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * tileRows * 1], yLocal, curRows * 1);
        outQueueY.FreeTensor(yLocal);
    }

    __aicore__ inline uint32_t GetCurRows(int32_t progress) const
    {
        uint32_t offset = progress * tileRows;
        if (offset >= rowsThisBlock) return 0;
        uint32_t remain = rowsThisBlock - offset;
        return remain > tileRows ? tileRows : remain;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t batch;
    uint32_t hidden;
    uint32_t tileBatch;
    uint32_t tileRows;
    uint32_t rowsThisBlock;

    float scale;
    float clampMin;
    float clampMax;

    // placeholder tiling object for LogSoftMax
    AscendC::LogSoftMaxTiling softmaxtiling_;
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScaleResidualAddClampLogSumExpMish op;
    op.Init(x, y,
            tiling_data.batch(),
            tiling_data.hidden(),
            tiling_data.tileBatch(),
            tiling_data.scale(),
            tiling_data.clampMin(),
            tiling_data.clampMax());
    op.Process();
}
