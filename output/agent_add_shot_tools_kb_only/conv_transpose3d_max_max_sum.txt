kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 双缓冲
using DTYPE = float;

class KernelConvTranspose3dMaxMaxSum {
public:
    __aicore__ inline KernelConvTranspose3dMaxMaxSum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d,
                                uint32_t h, uint32_t w, uint32_t tileW)
    {
        this->N = n;
        this->C = c;
        this->D = d;
        this->H = h;
        this->W = w;
        this->tileW = tileW;

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x, (uint64_t)N * C * D * H * W);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y, (uint64_t)N * 1 * D * H * W);

        // 每次处理一个行片段 [wStart, wStart+tileW)
        uint32_t lineBytes = this->tileW * sizeof(DTYPE);
        pipe.InitBuffer(inQueueX, BUFFER_NUM, lineBytes);
        pipe.InitBuffer(outQueueY, BUFFER_NUM, lineBytes);
    }

    __aicore__ inline void Process()
    {
        // 按batch划分block
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t nPerBlock = (N + blocks - 1) / blocks;
        uint32_t nStart = blockIdx * nPerBlock;
        uint32_t nEnd = AscendC::Min<uint32_t>(N, nStart + nPerBlock);

        for (uint32_t n = nStart; n < nEnd; ++n) {
            for (uint32_t di = 0; di < D; ++di) {
                for (uint32_t hi = 0; hi < H; ++hi) {
                    for (uint32_t wStart = 0; wStart < W; wStart += tileW) {
                        uint32_t curLen = AscendC::Min<uint32_t>(tileW, W - wStart);

                        // 分配累加buffer
                        AscendC::LocalTensor<DTYPE> accLocal = outQueueY.AllocTensor<DTYPE>();

                        bool first = true;
                        for (uint32_t ci = 0; ci < C; ++ci) {
                            AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
                            uint64_t xOffset = IndexOffset(n, ci, di, hi, wStart);
                            AscendC::DataCopy(xLocal, xGm[xOffset], curLen);
                            if (first) {
                                AscendC::DataCopy(accLocal, xLocal, curLen);
                                first = false;
                            } else {
                                AscendC::Add(accLocal, accLocal, xLocal, curLen);
                            }
                            inQueueX.FreeTensor(xLocal);
                        }

                        uint64_t yOffset = IndexOffsetY(n, di, hi, wStart);
                        AscendC::DataCopy(yGm[yOffset], accLocal, curLen);
                        outQueueY.FreeTensor(accLocal);
                    }
                }
            }
        }
    }

private:
    __aicore__ inline uint64_t IndexOffset(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        // NCDHW 连续内存
        return (((uint64_t)n * C + c) * D + d) * H * W + (uint64_t)h * W + w;
    }

    __aicore__ inline uint64_t IndexOffsetY(uint32_t n, uint32_t d, uint32_t h, uint32_t w) const
    {
        // 输出形状 [N,1,D,H,W]，通道固定为0
        return (((uint64_t)n * 1 + 0) * D + d) * H * W + (uint64_t)h * W + w;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t N, C, D, H, W;
    uint32_t tileW;
};

extern "C" __global__ __aicore__ void conv_transpose3d_max_max_sum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dMaxMaxSum op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.d,
            tiling_data.h, tiling_data.w, tiling_data.tileW);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC算子将 MaxPool3d(2) + MaxPool3d(3) + 通道求和 融合为一个自定义核。
    ConvTranspose3d 仍使用原生实现。
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.max_pool1 = torch.nn.MaxPool3d(kernel_size=2)
        self.max_pool2 = torch.nn.MaxPool3d(kernel_size=3)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.max_pool1(x)
        x = self.max_pool2(x)
        # 使用自定义算子在通道维度执行 sum，保持维度
        x = custom_ops_lib.conv_transpose3d_max_max_sum_custom(x)
        return x
