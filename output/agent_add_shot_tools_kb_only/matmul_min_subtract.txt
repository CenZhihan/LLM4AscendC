kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_B = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulMinSubtract {
public:
    __aicore__ inline KernelMatmulMinSubtract() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR cst, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileNum)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileNum = tileNum;

        // 按block划分输出工作量
        this->blockLength = (M * N) / AscendC::GetBlockNum();
        this->tileLength = (this->blockLength / (tileNum * BUFFER_NUM));

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_X *>(x), M * K);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_W *>(w), N * K);
        bGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_B *>(b), N);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_Y *>(y), M * N);

        constPtr = reinterpret_cast<__gm__ DTYPE_Y *>(cst);
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // 采用简单块循环: 每块处理连续tileLength个输出元素
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            ComputeTile(i);
            PostProcessTile(i);
            StoreTile(i);
        }
    }

private:
    __aicore__ inline void ComputeTile(int32_t progress)
    {
        // 计算输出线性区间
        uint32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();
        uint32_t startIdx = blockOffset + progress * this->tileLength;

        // 分配本地缓冲
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();

        // 逐元素计算: y[idx] = sum_k x[i,k] * w[j,k] + b[j]
        for (uint32_t t = 0; t < this->tileLength; ++t) {
            uint32_t idx = startIdx + t;
            if (idx >= M * N) {
                yLocal.SetValue(t, static_cast<DTYPE_Y>(0));
                continue;
            }
            uint32_t i = idx / N;
            uint32_t j = idx % N;

            DTYPE_Y acc = static_cast<DTYPE_Y>(0);
            // 加bias
            acc += bGm.GetValue(j);

            // x[i, k] * w[j, k] 的内积
            uint32_t xBase = i * K;
            uint32_t wBase = j * K;
            for (uint32_t k = 0; k < K; ++k) {
                DTYPE_X xv = xGm.GetValue(xBase + k);
                DTYPE_W wv = wGm.GetValue(wBase + k);
                acc += xv * wv;
            }
            yLocal.SetValue(t, acc);
        }
        inQueueY.EnQue(yLocal);
    }

    __aicore__ inline void PostProcessTile(int32_t)
    {
        // 从队列取出，执行逐元素 min(y, c) - c
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> yOutLocal = outQueueY.AllocTensor<DTYPE_Y>();

        DTYPE_Y c = *constPtr; // 常量标量

        for (uint32_t t = 0; t < this->tileLength; ++t) {
            DTYPE_Y v = yLocal.GetValue(t);
            if (v > c) v = c;
            v = v - c;
            yOutLocal.SetValue(t, v);
        }

        outQueueY.EnQue<DTYPE_Y>(yOutLocal);
        inQueueY.FreeTensor(yLocal);
    }

    __aicore__ inline void StoreTile(int32_t progress)
    {
        // 将结果写回GM
        AscendC::LocalTensor<DTYPE_Y> yOutLocal = outQueueY.DeQue<DTYPE_Y>();
        uint32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();
        uint32_t startIdx = blockOffset + progress * this->tileLength;

        // 顺序写回
        for (uint32_t t = 0; t < this->tileLength; ++t) {
            uint32_t idx = startIdx + t;
            if (idx < M * N) {
                yGm.SetValue(idx, yOutLocal.GetValue(t));
            }
        }
        outQueueY.FreeTensor(yOutLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    __gm__ DTYPE_Y* constPtr;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileNum;
    uint32_t blockLength;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void matmul_min_subtract_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                 GM_ADDR constant, GM_ADDR y,
                                                                 GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMinSubtract op;
    op.Init(x, weight, bias, constant, y,
            tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused: y = (x @ W^T + b).clamp_max(constant) - constant
    """
    def __init__(self, in_features, out_features, constant):
        super().__init__()
        self.linear = torch.nn.Linear(in_features, out_features)
        # 将常量保持为参数以便与原模型一致
        self.constant = torch.nn.Parameter(torch.tensor(constant, dtype=torch.float32))

    def forward(self, x):
        # 使用自定义融合算子: 输入 x, weight, bias, constant
        # constant 作为 1 元素张量传入
        const_tensor = self.constant
        return custom_ops_lib.matmul_min_subtract_custom(x, self.linear.weight, self.linear.bias, const_tensor)
