host_operator_src="""
#include "multi_query_attention_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"
#include <vector>

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 获取输入形状，按最后一维做SoftMax
    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin_shape = in_shape->GetOriginShape();
    int64_t dim_num = origin_shape.GetDimNum();
    std::vector<int64_t> dims;
    dims.reserve(dim_num);
    for (int64_t i = 0; i < dim_num; ++i) {
        dims.push_back(origin_shape.GetDim(i));
    }

    uint64_t total_elems = 1;
    for (auto d : dims) total_elems *= (uint64_t)d;
    uint32_t K = static_cast<uint32_t>(dims.back());
    uint64_t M64 = (K == 0) ? 0 : (total_elems / (uint64_t)K);
    uint32_t M = static_cast<uint32_t>(M64);

    ge::Shape srcShape(dims);

    // 计算SoftMax所需本地最小临时空间，并获取tiling参数
    const uint32_t localWorkSpaceSize = AscendC::GetSoftMaxMinTmpSize(srcShape, sizeof(float), false);

    MultiQueryAttentionCustomTilingData tiling;
    tiling.set_srcM(M);
    tiling.set_srcK(K);
    AscendC::SoftMaxTilingFunc(srcShape, sizeof(float), localWorkSpaceSize, tiling.softmaxTilingData);

    // 设置block维度（SoftMax高阶API内部根据tiling执行，此处简单设置）
    context->SetBlockDim(1);

    // 保存tiling数据
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    context->SetTilingKey(1);

    // 无额外workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

template <typename T>
class KernelSoftmaxOp {
public:
    __aicore__ inline KernelSoftmaxOp() {}
    __aicore__ inline void Init(__gm__ uint8_t *xGm, __gm__ uint8_t *yGm, const SoftMaxTiling &tilingIn, uint32_t srcMIn, uint32_t srcKIn)
    {
        tiling = tilingIn;
        srcM = srcMIn;
        srcK = srcKIn;
        elementNumPerBlk = 32 / sizeof(T);

        xGlobal.SetGlobalBuffer((__gm__ T*)xGm);
        yGlobal.SetGlobalBuffer((__gm__ T*)yGm);

        // 分配本地队列缓冲（示例按整块分配，实际运行中以硬件UB大小为准由tiling控制）
        pipe.InitBuffer(inQueue, 1, srcM * srcK * sizeof(T));
        pipe.InitBuffer(maxQueue, 1, srcM * elementNumPerBlk * sizeof(T));
        pipe.InitBuffer(sumQueue, 1, srcM * elementNumPerBlk * sizeof(T));
        pipe.InitBuffer(outQueue, 1, srcM * srcK * sizeof(T));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<T> xLocal = inQueue.AllocTensor<T>();
        AscendC::DataCopy(xLocal, xGlobal, srcM * srcK);
        inQueue.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<T> xLocal = inQueue.DeQue<T>();
        AscendC::LocalTensor<T> sumTempLocal = sumQueue.AllocTensor<T>();
        AscendC::LocalTensor<T> maxTempLocal = maxQueue.AllocTensor<T>();
        AscendC::LocalTensor<T> yLocal = outQueue.AllocTensor<T>();

        AscendC::SoftMaxShapeInfo shapeInfo = {srcM, srcK, srcM, srcK};
        AscendC::SoftMax<T>(yLocal, sumTempLocal, maxTempLocal, xLocal, tiling, shapeInfo);

        outQueue.EnQue<T>(yLocal);
        maxQueue.FreeTensor(maxTempLocal);
        sumQueue.FreeTensor(sumTempLocal);
        inQueue.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<T> yLocal = outQueue.DeQue<T>();
        AscendC::DataCopy(yGlobal, yLocal, srcM * srcK);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> maxQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> sumQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue;

    AscendC::GlobalTensor<T> xGlobal, yGlobal;
    SoftMaxTiling tiling;
    uint32_t srcM = 0;
    uint32_t srcK = 0;
    uint32_t elementNumPerBlk = 0;
};

extern "C" __global__ __aicore__ void multi_query_attention_custom(__gm__ uint8_t *x, __gm__ uint8_t *y, __gm__ uint8_t *workspace, __gm__ uint8_t *tilingGm)
{
    GET_TILING_DATA(tilingData, tilingGm);
    KernelSoftmaxOp<float> op;
    op.Init((__gm__ uint8_t*)x, (__gm__ uint8_t*)y, tilingData.softmaxTilingData, tilingData.srcM, tilingData.srcK);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}

model_src='''
import math
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Multi-Query Attention (MQA) with custom AscendC softmax kernel.
    The custom op computes softmax over the last dimension of attention logits.
    """
    def __init__(self, d_model=4096, num_heads=32):
        super().__init__()
        self.q_proj = torch.nn.Linear(d_model, d_model)
        self.k_proj = torch.nn.Linear(d_model, d_model // num_heads)  # shared across heads
        self.v_proj = torch.nn.Linear(d_model, d_model // num_heads)
        self.out_proj = torch.nn.Linear(d_model, d_model)
        self.num_heads = num_heads

    def forward(self, x):
        B, N, D = x.shape
        H = self.num_heads
        d_h = D // H

        Q = self.q_proj(x).view(B, N, H, d_h)        # [B, N, H, d_h]
        K = self.k_proj(x).unsqueeze(2)              # [B, K, 1, d_h]
        V = self.v_proj(x).unsqueeze(2)              # [B, K, 1, d_h]

        # 计算注意力logits: [B, H, N, K]
        logits = torch.einsum("bnhd,bkhd->bhnk", Q, K)
        logits = logits / math.sqrt(d_h)

        # 使用自定义AscendC Softmax算子，最后一维做softmax
        attn = custom_ops_lib.multi_query_attention_custom(logits)

        # 加权求和得到输出: [B, N, H, d_h] -> [B, N, D]
        out = torch.einsum("bhnk,bkhd->bnhd", attn, V).reshape(B, N, D)
        return self.out_proj(out)
'''
