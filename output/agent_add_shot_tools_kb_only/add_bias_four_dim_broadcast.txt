kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // 每次处理一个 [H*W] 通道平面

class KernelAddBias4DBroadcast {
public:
    __aicore__ inline KernelAddBias4DBroadcast() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z, uint32_t n, uint32_t c, uint32_t h, uint32_t w)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->hw = h * w;

        // 为避免跨通道，按 (N*C) 的片段划分到不同 block
        uint32_t totalNc = n * c;
        uint32_t startNc = (totalNc * AscendC::GetBlockIdx()) / AscendC::GetBlockNum();
        uint32_t endNc   = (totalNc * (AscendC::GetBlockIdx() + 1)) / AscendC::GetBlockNum();
        this->startNc = startNc;
        this->ncCount = endNc - startNc;

        // 设置GM指针，长度以元素计
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + startNc * this->hw, this->ncCount * this->hw);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias, c);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + startNc * this->hw, this->ncCount * this->hw);

        // 初始化本地队列缓冲
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->hw * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->hw * sizeof(DTYPE_Z));
        pipe.InitBuffer(biasQueue, BUFFER_NUM, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t idx = 0; idx < this->ncCount; ++idx) {
            CopyIn(idx);
            Compute(idx);
            CopyOut(idx);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t idx)
    {
        // 复制一个通道平面 [H*W] 的数据
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[idx * this->hw], this->hw);

        // 复制该通道的bias标量
        uint32_t cIdx = (this->startNc + idx) % this->c;
        AscendC::LocalTensor<DTYPE_Y> biasLocal = biasQueue.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(biasLocal, biasGm[cIdx], 1);

        inQueueX.EnQue(xLocal);
        biasQueue.EnQue(biasLocal);
    }

    __aicore__ inline void Compute(uint32_t idx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> biasLocal = biasQueue.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // 从本地Tensor取出标量值
        DTYPE_Y biasVal = biasLocal.GetValue(0);

        // 对该通道平面执行向量 + 标量
        AscendC::Adds(zLocal, xLocal, biasVal, static_cast<int32_t>(this->hw));

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        biasQueue.FreeTensor(biasLocal);
    }

    __aicore__ inline void CopyOut(uint32_t idx)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[idx * this->hw], zLocal, this->hw);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> biasQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> biasGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t n, c, h, w, hw;
    uint32_t startNc;
    uint32_t ncCount;
};

extern "C" __global__ __aicore__ void add_bias_four_dim_broadcast_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAddBias4DBroadcast op;
    op.Init(x, bias, z, tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w);
    op.Process();
}
