kernel_src="""
#include "kernel_operator.h"

class KernelFused {
public:
    __aicore__ inline KernelFused() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR subtract, GM_ADDR z,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w)
    {
        this->N = n;
        this->C = c;
        this->D = d;
        this->H = h;
        this->W = w;
        this->spatial = d * h * w;

        xGm.SetGlobalBuffer((__gm__ float*)x, N * C * D * H * W);
        subGm.SetGlobalBuffer((__gm__ float*)subtract, C);
        zGm.SetGlobalBuffer((__gm__ float*)z, N * D * H * W);
    }

    __aicore__ inline void Process()
    {
        // Simple scalar-loop implementation over positions to compute:
        // softmax over channel, subtract per-channel bias, swish, then max over channel.
        for (uint32_t n = 0; n < N; ++n) {
            uint32_t base_nc = n * C * spatial;
            for (uint32_t s = 0; s < spatial; ++s) {
                // 1) load channels
                float maxVal = -3.4e38f;
                for (uint32_t c = 0; c < C; ++c) {
                    float val = xGm[base_nc + c * spatial + s];
                    if (val > maxVal) maxVal = val;
                }
                // 2) stable softmax: exp(x - max) and sum
                float sumExp = 0.f;
                for (uint32_t c = 0; c < C; ++c) {
                    float val = xGm[base_nc + c * spatial + s] - maxVal;
                    float e = AscendC::Exp(val);
                    // Temporarily store back to xGm scratch is not appropriate; compute on the fly
                    tmpExp[c] = e; // use small local buffer
                    sumExp += e;
                }
                float invSum = 1.f / sumExp;

                // 3) subtract bias, swish, and track max over channel
                float outMax = -3.4e38f;
                for (uint32_t c = 0; c < C; ++c) {
                    float soft = tmpExp[c] * invSum;
                    float y = soft - subGm[c];
                    // swish: y * sigmoid(y) = y / (1 + exp(-y))
                    float sig = 1.f / (1.f + AscendC::Exp(-y));
                    float sw = y * sig;
                    if (sw > outMax) outMax = sw;
                }
                // 4) write output [N, D, H, W] flattened as N*spatial
                zGm[n * spatial + s] = outMax;
            }
        }
    }

private:
    // Small local temporary buffer for per-position exp values, assuming C is reasonably small.
    static const uint32_t MAX_C = 1024;
    float tmpExp[MAX_C];

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> subGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t N;
    uint32_t C;
    uint32_t D;
    uint32_t H;
    uint32_t W;
    uint32_t spatial;
};

extern "C" __global__ __aicore__ void conv_transpose3d_max_pool_softmax_subtract_swish_max_custom(
    GM_ADDR x, GM_ADDR subtract, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelFused op;
    op.Init(x, subtract, z, tilingData.n, tilingData.c, tilingData.d, tilingData.h, tilingData.w);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model:
      - Keep ConvTranspose3d and MaxPool3d as standard modules
      - Fuse Softmax (over channels), Subtract (per-channel), Swish, and Max-over-channels into a custom AscendC kernel
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, pool_kernel_size, pool_stride, pool_padding):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.max_pool = torch.nn.MaxPool3d(kernel_size=pool_kernel_size, stride=pool_stride, padding=pool_padding)
        self.subtract = torch.nn.Parameter(torch.randn(out_channels))

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.max_pool(x)
        # Fused kernel expects [N, C, D, H, W] tensor and per-channel vector
        x = custom_ops_lib.conv_transpose3d_max_pool_softmax_subtract_swish_max_custom(x, self.subtract)
        return x
