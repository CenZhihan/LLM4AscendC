    ScatterCustomTilingData tiling;

    tiling.set_nLength(nLength);
    tiling.set_kLength(kLength);

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelScatter {
public:
    __aicore__ inline KernelScatter() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR updates, GM_ADDR z,
                                uint32_t bLength, uint32_t nLength, uint32_t kLength, uint32_t tileK)
    {
        this->bLength = bLength;
        this->nLength = nLength;
        this->kLength = kLength;
        this->tileK = tileK;

        xPtr = reinterpret_cast<__gm__ float*>(x);
        idxPtr = reinterpret_cast<__gm__ int64_t*>(idx);
        updPtr = reinterpret_cast<__gm__ float*>(updates);
        zPtr = reinterpret_cast<__gm__ float*>(z);

        // one full row buffer for copy x[row] -> z[row]
        pipe.InitBuffer(rowBufQueueIn, BUFFER_NUM, this->nLength * sizeof(float));
        pipe.InitBuffer(rowBufQueueOut, BUFFER_NUM, this->nLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();
        for (uint32_t row = blockIdx; row < bLength; row += blockNum) {
            CopyRow(row);
            ApplyUpdates(row);
        }
    }

private:
    __aicore__ inline void CopyRow(uint32_t row)
    {
        // Copy one row from x -> local -> z
        AscendC::GlobalTensor<float> xRowGm;
        AscendC::GlobalTensor<float> zRowGm;
        xRowGm.SetGlobalBuffer(xPtr + static_cast<uint64_t>(row) * nLength, nLength);
        zRowGm.SetGlobalBuffer(zPtr + static_cast<uint64_t>(row) * nLength, nLength);

        AscendC::LocalTensor<float> tmpLocal = rowBufQueueIn.AllocTensor<float>();
        AscendC::DataCopy(tmpLocal, xRowGm, nLength);
        rowBufQueueIn.EnQue(tmpLocal);

        AscendC::LocalTensor<float> tmpLocalDeq = rowBufQueueIn.DeQue<float>();
        AscendC::DataCopy(zRowGm, tmpLocalDeq, nLength);
        rowBufQueueIn.FreeTensor(tmpLocalDeq);
    }

    __aicore__ inline void ApplyUpdates(uint32_t row)
    {
        // For each (idx, update) in this row, write to z
        const uint64_t idxBase = static_cast<uint64_t>(row) * kLength;
        const uint64_t zBase = static_cast<uint64_t>(row) * nLength;

        // Scalar global reads/writes for flexibility with arbitrary indices
        for (uint32_t j = 0; j < kLength; ++j) {
            int64_t col = idxPtr[idxBase + j];
            float val = updPtr[idxBase + j];
            // Assume 0 <= col < nLength
            zPtr[zBase + static_cast<uint64_t>(col)] = val;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> rowBufQueueIn;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> rowBufQueueOut;

    __gm__ float* xPtr;
    __gm__ int64_t* idxPtr;
    __gm__ float* updPtr;
    __gm__ float* zPtr;

    uint32_t bLength;
    uint32_t nLength;
    uint32_t kLength;
    uint32_t tileK;
};

extern "C" __global__ __aicore__ void scatter_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR updates, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelScatter op;
    op.Init(x, idx, updates, z, tiling_data.bLength, tiling_data.nLength, tiling_data.kLength, tiling_data.tileK);
    op.Process();
}
