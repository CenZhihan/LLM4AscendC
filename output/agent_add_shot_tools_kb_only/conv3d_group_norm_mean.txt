kernel_src="""
#include "kernel_operator.h"

// 本自定义核实现对输入张量 x（期望为 GN 之后的 (B, C, D, H, W)）
// 沿 [C, D, H, W] 求均值，输出 y 形状为 (B,)。
// 为简化演示与通用性，采用逐 batch 的分块 ReduceSum，然后乘以 1/innerLength 得到均值。

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReduce {
public:
    __aicore__ inline KernelMeanReduce() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t innerLength, uint32_t tileNum)
    {
        this->batch = batch;
        this->innerLength = innerLength;
        this->tileNum = tileNum;

        // 每个 block 负责的样本数
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        this->batchPerBlock = (batch + blocks - 1) / blocks;
        this->batchStart = idx * this->batchPerBlock;
        this->batchEnd = this->batchStart + this->batchPerBlock;
        if (this->batchEnd > batch) this->batchEnd = batch;
        this->procBatch = (this->batchEnd > this->batchStart) ? (this->batchEnd - this->batchStart) : 0;

        // 设定 GM 基址和范围
        xGm.SetGlobalBuffer((__gm__ float*)x + (uint64_t)this->batchStart * innerLength,
                            (uint64_t)this->procBatch * innerLength);
        yGm.SetGlobalBuffer((__gm__ float*)y + this->batchStart, this->procBatch);

        // 分块长度（假定可整除）
        this->loopCount = this->tileNum * BUFFER_NUM;
        this->tileLength = (this->loopCount > 0) ? (this->innerLength / this->loopCount) : this->innerLength;
        if (this->tileLength == 0) this->tileLength = this->innerLength;

        // 初始化缓冲
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        // workLocal 可按 Method 2 任意大小传入，这里给定 tileLength 大小
        pipe.InitBuffer(tmpWorkQue, BUFFER_NUM, this->tileLength * sizeof(float));
        // 输出与临时标量缓冲（大小 1）
        pipe.InitBuffer(outQueueY, BUFFER_NUM, 1 * sizeof(float));
        pipe.InitBuffer(tmpScalarQue, BUFFER_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (this->procBatch == 0) { return; }

        for (uint32_t b = 0; b < this->procBatch; ++b) {
            // 累加器 accLocal
            AscendC::LocalTensor<float> accLocal = outQueueY.AllocTensor<float>();
            float zero = 0.0f;
            AscendC::Duplicate(accLocal, zero, 1);

            for (uint32_t i = 0; i < this->loopCount; ++i) {
                // 读入一块
                AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                uint64_t gmOffset = (uint64_t)b * this->innerLength + (uint64_t)i * this->tileLength;
                AscendC::DataCopy(xLocal, xGm[gmOffset], this->tileLength);
                inQueueX.EnQue(xLocal);

                // ReduceSum -> 得到该块和
                AscendC::LocalTensor<float> xTile = inQueueX.DeQue<float>();
                AscendC::LocalTensor<float> workLocal = tmpWorkQue.AllocTensor<float>();
                AscendC::LocalTensor<float> sumLocal = tmpScalarQue.AllocTensor<float>();
                AscendC::ReduceSum(sumLocal, xTile, workLocal, this->tileLength);

                // acc += sumLocal
                AscendC::Add(accLocal, accLocal, sumLocal, 1);

                // 释放
                tmpScalarQue.FreeTensor(sumLocal);
                tmpWorkQue.FreeTensor(workLocal);
                inQueueX.FreeTensor(xTile);
            }

            // 均值：accLocal *= (1.0 / innerLength)
            AscendC::LocalTensor<float> scaleLocal = tmpScalarQue.AllocTensor<float>();
            float scale = 1.0f / static_cast<float>(this->innerLength);
            AscendC::Duplicate(scaleLocal, scale, 1);
            AscendC::Mul(accLocal, accLocal, scaleLocal, 1);
            tmpScalarQue.FreeTensor(scaleLocal);

            // 写回
            outQueueY.EnQue<float>(accLocal);
            AscendC::LocalTensor<float> outLocal = outQueueY.DeQue<float>();
            AscendC::DataCopy(yGm[b], outLocal, 1);
            outQueueY.FreeTensor(outLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpWorkQue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpScalarQue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t batch = 0;
    uint32_t innerLength = 0;
    uint32_t tileNum = 1;
    uint32_t loopCount = 1;
    uint32_t tileLength = 1;

    uint32_t batchPerBlock = 0;
    uint32_t batchStart = 0;
    uint32_t batchEnd = 0;
    uint32_t procBatch = 0;
};

extern "C" __global__ __aicore__ void conv3d_group_norm_mean_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReduce op;
    op.Init(x, y, tiling_data.batch, tiling_data.innerLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Conv3d -> GroupNorm -> Mean(C,D,H,W) fused为自定义AscendC均值核（仅替换最终Mean阶段）
    """
    def __init__(self, in_channels, out_channels, kernel_size, num_groups):
        super().__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_channels)

    def forward(self, x):
        x = self.conv(x)
        x = self.group_norm(x)
        # 使用自定义核计算 (C,D,H,W) 维度上的均值，输出形状为 (B,)
        y = custom_ops_lib.conv3d_group_norm_mean_custom(x)
        return y
