#include "elu_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    EluCustomTilingData tiling;
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class EluCustom : public OpDef {

    explicit EluCustom(const char* name) : OpDef(name)

        this->Input("x")

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数量

class KernelElu {
public:
    __aicore__ inline KernelElu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR alpha, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        aGm.SetGlobalBuffer((__gm__ DTYPE_Y *)alpha + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(calcBufs, this->tileLength * sizeof(DTYPE_X) * 2); // 两个中间buffer
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> aLocal = inQueueA.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(aLocal, aGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueA.EnQue(aLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> aLocal = inQueueA.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        AscendC::LocalTensor<DTYPE_X> tmp1 = calcBufs.Get<DTYPE_X>(); // 保存min(x,0)及其后续变换
        AscendC::LocalTensor<DTYPE_X> tmp2 = calcBufs.Get<DTYPE_X>(); // 保存alpha*(exp(x)-1)

        // y = max(0, x) + alpha * (exp(min(0, x)) - 1)
        // 1) yLocal <- max(x, 0)
        AscendC::Maxs(yLocal, xLocal, static_cast<DTYPE_X>(0), this->tileLength);

        // 2) tmp1 <- min(x, 0)
        AscendC::Mins(tmp1, xLocal, static_cast<DTYPE_X>(0), this->tileLength);

        // 3) tmp1 <- exp(tmp1)
        AscendC::Exp(tmp1, tmp1, this->tileLength);

        // 4) tmp1 <- tmp1 - 1
        AscendC::Adds(tmp1, tmp1, static_cast<DTYPE_X>(-1.0f), this->tileLength);

        // 5) tmp2 <- tmp1 * alpha (alpha为向量输入)
        AscendC::Mul(tmp2, tmp1, aLocal, this->tileLength);

        // 6) yLocal <- yLocal + tmp2
        AscendC::Add(yLocal, yLocal, tmp2, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueA.FreeTensor(aLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueA;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBufs;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> aGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void elu_custom(GM_ADDR x, GM_ADDR alpha, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelElu op;
    op.Init(x, alpha, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
"""
