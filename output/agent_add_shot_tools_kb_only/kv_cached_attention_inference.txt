kernel_src="""
#include "kernel_operator.h"

// 注意：该核面向B=1, Q=1的自回归推理KV缓存常见场景，数据类型float。
// 实现思路：
// 1) 将q拷入UB；按行逐行扫描k，基于分块(tileD)做分段点积累加，得到scores[kvLen]；
// 2) 对scores执行scaled + softmax（两次遍历做max与sum）；
// 3) 按d_model分块(tileD)，遍历kv行，累加prob[i] * v[i, d:d+tileD]，得到输出out[1,1,d_model]；
// 该实现强调可读性，便于验证功能正确性。实际优化可考虑更细致的向量化与并行化。

constexpr int32_t Q_BUF_NUM = 1;
constexpr int32_t SCRATCH_BUF_NUM = 1;

class KernelKvCachedAttention {
public:
    __aicore__ inline KernelKvCachedAttention() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR out,
                                uint32_t batch, uint32_t qLen, uint32_t kvLen,
                                uint32_t dModel, uint32_t tileD, float scale)
    {
        // 简化：仅实现batch=1, qLen=1
        this->batch = batch;
        this->qLen = qLen;
        this->kvLen = kvLen;
        this->dModel = dModel;
        this->tileD = tileD;
        this->scale = scale;

        qGm.SetGlobalBuffer((__gm__ float*)q, batch * qLen * dModel);
        kGm.SetGlobalBuffer((__gm__ float*)k, batch * kvLen * dModel);
        vGm.SetGlobalBuffer((__gm__ float*)v, batch * kvLen * dModel);
        outGm.SetGlobalBuffer((__gm__ float*)out, batch * qLen * dModel);

        // UB分配
        // qLocal: dModel
        pipe.InitBuffer(qQueue, Q_BUF_NUM, dModel * sizeof(float));
        // scores / probs: kvLen
        pipe.InitBuffer(scoresQueue, 1, kvLen * sizeof(float));
        pipe.InitBuffer(probsQueue, 1, kvLen * sizeof(float));
        // scratch块：tileD
        pipe.InitBuffer(kTileQueue, SCRATCH_BUF_NUM, tileD * sizeof(float));
        pipe.InitBuffer(vTileQueue, SCRATCH_BUF_NUM, tileD * sizeof(float));
        pipe.InitBuffer(outTileQueue, SCRATCH_BUF_NUM, tileD * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // 假设batch=1, qLen=1
        CopyInQ();
        ComputeScores();
        SoftmaxScores();
        WeightedSumV();
        CopyOut();
    }

private:
    __aicore__ inline void CopyInQ()
    {
        // 将q拷入UB
        qLocal = qQueue.AllocTensor<float>();
        AscendC::DataCopy(qLocal, qGm, dModel);
        qQueue.EnQue(qLocal);
    }

    __aicore__ inline void ComputeScores()
    {
        AscendC::LocalTensor<float> qVec = qQueue.DeQue<float>();

        // scores缓冲
        scoresLocal = scoresQueue.AllocTensor<float>();
        // 逐行计算点积
        for (uint32_t i = 0; i < kvLen; ++i) {
            float acc = 0.0f;
            // 按tileD分段
            for (uint32_t d0 = 0; d0 < dModel; d0 += tileD) {
                uint32_t cur = AscendC::Min(tileD, dModel - d0);
                AscendC::LocalTensor<float> kTile = kTileQueue.AllocTensor<float>();
                // 从K中取出一行的分段
                AscendC::DataCopy(kTile, kGm[i * dModel + d0], cur);
                // 对tile做标量MAC（为简洁，这里用标量循环）
                for (uint32_t t = 0; t < cur; ++t) {
                    acc += qVec[d0 + t] * kTile[t];
                }
                kTileQueue.FreeTensor(kTile);
            }
            // 记录scaled分数
            scoresLocal[i] = acc * scale;
        }
        scoresQueue.EnQue(scoresLocal);
        qQueue.FreeTensor(qVec);
    }

    __aicore__ inline void SoftmaxScores()
    {
        AscendC::LocalTensor<float> scores = scoresQueue.DeQue<float>();
        probsLocal = probsQueue.AllocTensor<float>();

        // 先求max
        float maxVal = -3.4e38f;
        for (uint32_t i = 0; i < kvLen; ++i) {
            float v = scores[i];
            maxVal = v > maxVal ? v : maxVal;
        }
        // 计算exp并累积sum
        float sumVal = 0.0f;
        for (uint32_t i = 0; i < kvLen; ++i) {
            float e = AscendC::Exp(scores[i] - maxVal);
            probsLocal[i] = e;
            sumVal += e;
        }
        // 归一化
        float invSum = 1.0f / sumVal;
        for (uint32_t i = 0; i < kvLen; ++i) {
            probsLocal[i] = probsLocal[i] * invSum;
        }

        probsQueue.EnQue(probsLocal);
        scoresQueue.FreeTensor(scores);
    }

    __aicore__ inline void WeightedSumV()
    {
        AscendC::LocalTensor<float> probs = probsQueue.DeQue<float>();
        // 输出按d_model分块累加
        for (uint32_t d0 = 0; d0 < dModel; d0 += tileD) {
            uint32_t cur = AscendC::Min(tileD, dModel - d0);
            AscendC::LocalTensor<float> outTile = outTileQueue.AllocTensor<float>();
            // outTile清零
            for (uint32_t t = 0; t < cur; ++t) { outTile[t] = 0.0f; }

            for (uint32_t i = 0; i < kvLen; ++i) {
                AscendC::LocalTensor<float> vTile = vTileQueue.AllocTensor<float>();
                AscendC::DataCopy(vTile, vGm[i * dModel + d0], cur);
                float p = probs[i];
                for (uint32_t t = 0; t < cur; ++t) {
                    outTile[t] += p * vTile[t];
                }
                vTileQueue.FreeTensor(vTile);
            }
            // 回写到outGm
            AscendC::DataCopy(outGm[d0], outTile, cur);
            outTileQueue.FreeTensor(outTile);
        }

        probsQueue.FreeTensor(probs);
    }

    __aicore__ inline void CopyOut()
    {
        // 已在WeightedSumV内按块写回
        return;
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::TPosition::VECIN, Q_BUF_NUM> qQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> scoresQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> probsQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, SCRATCH_BUF_NUM> kTileQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, SCRATCH_BUF_NUM> vTileQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, SCRATCH_BUF_NUM> outTileQueue;

    AscendC::GlobalTensor<float> qGm, kGm, vGm, outGm;

    AscendC::LocalTensor<float> qLocal;
    AscendC::LocalTensor<float> scoresLocal;
    AscendC::LocalTensor<float> probsLocal;

    uint32_t batch;
    uint32_t qLen;
    uint32_t kvLen;
    uint32_t dModel;
    uint32_t tileD;
    float scale;
};

extern "C" __global__ __aicore__ void kv_cached_attention_inference_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelKvCachedAttention op;
    op.Init(q, k, v, out,
            tiling_data.batch(),
            tiling_data.qLen(),
            tiling_data.kvLen(),
            tiling_data.dModel(),
            tiling_data.tileD(),
            tiling_data.scale());
    op.Process();
}
