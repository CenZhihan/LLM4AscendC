#include "gru_custom_tiling.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GruCustomTilingData tiling;

    // x: [seq_len, batch, input_size] or flattened as [M, K]
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    // Infer M, K, N
    // Assume x is [M, K] or [seq_len, batch, K]
    uint64_t dims = x_shape->GetDimNum();
    uint32_t K = 0;
    uint32_t M = 0;
    if (dims >= 2) {
        K = static_cast<uint32_t>(x_shape->GetDim(x_shape->GetDimNum() - 1));
        uint64_t mprod = 1;
        for (uint64_t i = 0; i + 1 < dims; ++i) {
            mprod *= x_shape->GetDim(i);
        }
        M = static_cast<uint32_t>(mprod);
    } else {
        // Fallback
        K = static_cast<uint32_t>(x_shape->GetOriginShape().GetShapeSize());
        M = 1;
    }
    // w_ih: [3*hidden, input_size] => N = 3*hidden, K_w = input_size
    uint32_t N = 0;
    if (w_shape->GetDimNum() == 2) {
        N = static_cast<uint32_t>(w_shape->GetDim(0));
    } else {
        N = static_cast<uint32_t>(w_shape->GetOriginShape().GetShapeSize() / K);
    }

    // Simple tiling: split rows (M) across blocks
    const uint32_t BLOCK_DIM = 8;
    context->SetBlockDim(BLOCK_DIM);

    uint32_t blockM = (M + BLOCK_DIM - 1) / BLOCK_DIM;
    uint32_t tileRows = 8; // rows per tile inside one block
    if (tileRows > blockM) tileRows = blockM == 0 ? 1 : blockM;
    uint32_t tilesPerBlock = (blockM + tileRows - 1) / tileRows;

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_blockM(blockM);
    tiling.set_tileRows(tileRows);
    tiling.set_tilesPerBlock(tilesPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // gate_x = x @ w_ih^T + b_ih
    // x: [M, K] or [seq_len, batch, K], w_ih: [N, K], output: [M, N] or [seq_len, batch, N]
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    gert::Shape* y_shape = context->GetOutputShape(0);

    if (x_shape->GetDimNum() >= 2 && w_shape->GetDimNum() == 2) {
        // keep leading dims of x, replace last dim with N
        *y_shape = *x_shape;
        y_shape->SetDim(y_shape->GetDimNum() - 1, w_shape->GetDim(0));
    } else {
        // fallback to 2D
        y_shape->SetDimNum(2);
        y_shape->SetDim(0, x_shape->GetOriginShape().GetShapeSize());
        y_shape->SetDim(1, 1);
    }
    return GRAPH_SUCCESS;
}

    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);

kernel_src="""
#include "kernel_operator.h"

// Kernel purpose: compute gate_x = x @ w_ih^T + b_ih
// x shape: [M, K] (flattened from [seq_len, batch, K])
// w_ih shape: [N, K], b_ih: [N]
// gate_x shape: [M, N]

constexpr int32_t BUFFER_NUM = 2;

class KernelGruLinear {
public:
    __aicore__ inline KernelGruLinear() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t blockM, uint32_t tileRows, uint32_t tilesPerBlock)
    {
        this->M = M;
        this->K = K;
        this->N = N;
        this->blockM = blockM;
        this->tileRows = tileRows;
        this->tilesPerBlock = tilesPerBlock;

        // Block ownership on M dimension
        uint32_t blockStart = AscendC::GetBlockIdx() * blockM;
        if (blockStart > M) blockStart = M;
        uint32_t blockLen = blockM;
        if (blockStart + blockLen > M) blockLen = (M > blockStart) ? (M - blockStart) : 0;
        this->blockStartRow = blockStart;
        this->blockRows = blockLen;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + blockStartRow * K, blockRows * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)w, N * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)b, N);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + blockStartRow * N, blockRows * N);

        // Each tile processes 'tileRows' rows
        pipe.InitBuffer(inQueueX, BUFFER_NUM, tileRows * K * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, 1, N * sizeof(DTYPE_Y)); // bias cached once
        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileRows * N * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // Load bias once
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(bLocal, bGm, N);
        inQueueB.EnQue(bLocal);

        uint32_t tiles = (blockRows + tileRows - 1) / tileRows;
        if (tiles == 0) { inQueueB.FreeTensor(bLocal); return; }

        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t curRows = tileRows;
            if ((t + 1) * tileRows > blockRows) {
                curRows = blockRows - t * tileRows;
            }
            CopyIn(t, curRows);
            Compute(t, curRows);
            CopyOut(t, curRows);
        }
        bLocal = inQueueB.DeQue<DTYPE_Y>();
        inQueueB.FreeTensor(bLocal);
    }

private:
    __aicore__ inline void CopyIn(uint32_t tileIdx, uint32_t curRows)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // start row offset in GM for this tile
        uint32_t rowOffset = tileIdx * tileRows;
        AscendC::DataCopy(xLocal, xGm[rowOffset * K], curRows * K);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t tileIdx, uint32_t curRows)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Naive GEMM: for each row r in the tile, for each n in [0, N)
        // y[r, n] = b[n] + sum_{k=0..K-1} x[r, k] * w[n, k]
        // Note: For demonstration; for performance, replace with Matmul API and tiling.
        for (uint32_t r = 0; r < curRows; ++r) {
            for (uint32_t n = 0; n < N; ++n) {
                float acc = ((const DTYPE_Y*)bLocal.GetPhyAddr())[n];
                const DTYPE_X* xRow = ((const DTYPE_X*)xLocal.GetPhyAddr()) + r * K;
                const DTYPE_Y* wRow = ((const DTYPE_Y*)wGm.GetPhyAddr()) + n * K;
                for (uint32_t k = 0; k < K; ++k) {
                    acc += static_cast<float>(xRow[k]) * static_cast<float>(wRow[k]);
                }
                ((DTYPE_Z*)yLocal.GetPhyAddr())[r * N + n] = static_cast<DTYPE_Z>(acc);
            }
        }

        outQueueY.EnQue(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.EnQue(bLocal); // keep bias cached
    }

    __aicore__ inline void CopyOut(uint32_t tileIdx, uint32_t curRows)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        uint32_t rowOffset = tileIdx * tileRows;
        AscendC::DataCopy(yGm[rowOffset * N], yLocal, curRows * N);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t M;
    uint32_t K;
    uint32_t N;
    uint32_t blockM;
    uint32_t tileRows;
    uint32_t tilesPerBlock;
    uint32_t blockStartRow;
    uint32_t blockRows;
};

extern "C" __global__ __aicore__ void gru_custom(GM_ADDR x, GM_ADDR w_ih, GM_ADDR b_ih, GM_ADDR gate_x, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGruLinear op;
    op.Init(x, w_ih, b_ih, gate_x,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.blockM, tiling_data.tileRows, tiling_data.tilesPerBlock);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import torch.nn.functional as F

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_size, num_layers=3, bias=True, batch_first=False):
        super().__init__()
        assert bias, "This optimized path currently assumes bias=True"
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.batch_first = batch_first

        # Keep an nn.GRU to manage parameters and initialization; we will consume its weights
        self.gru = torch.nn.GRU(input_size, hidden_size, num_layers, bias=bias, batch_first=batch_first, dropout=0, bidirectional=False)

    def forward(self, x, h0):
        # x: (seq_len, batch, input_size) if batch_first=False
        if self.batch_first:
            x = x.transpose(0, 1)  # (seq, batch, input)
        seq_len, batch, _ = x.shape

        h_prev = h0  # shape: (num_layers, batch, hidden)

        layer_input = x
        outputs_per_layer = None

        for l in range(self.num_layers):
            # Extract GRU layer l parameters from PyTorch
            w_ih = getattr(self.gru, f"weight_ih_l{l}")      # [3*hidden, input_size]
            w_hh = getattr(self.gru, f"weight_hh_l{l}")      # [3*hidden, hidden_size]
            b_ih = getattr(self.gru, f"bias_ih_l{l}")        # [3*hidden]
            b_hh = getattr(self.gru, f"bias_hh_l{l}")        # [3*hidden]

            # Compute gate_x = x_t @ w_ih^T + b_ih for all t via custom kernel
            gate_x = custom_ops_lib.gru_custom(layer_input, w_ih, b_ih)  # (seq, batch, 3*hidden)

            # Recurrent loop over time using PyTorch math for h_t update
            h_t = h_prev[l]  # (batch, hidden)
            layer_outputs = []
            for t in range(seq_len):
                gx = gate_x[t]  # (batch, 3*hidden)
                gh = F.linear(h_t, w_hh, b_hh)  # (batch, 3*hidden)

                gates = gx + gh
                # Split gates: r, z, n
                r, z, n = gates.chunk(3, dim=-1)
                r = torch.sigmoid(r)
                z = torch.sigmoid(z)
                n = torch.tanh(n)
                h_t = n + z * (h_t - n)
                layer_outputs.append(h_t.unsqueeze(0))

            outputs_per_layer = torch.cat(layer_outputs, dim=0)  # (seq, batch, hidden)
            # Prepare next layer input
            layer_input = outputs_per_layer
            # Update h_prev for next layer
            h_prev = h_prev.clone()
            h_prev[l] = h_t

        out = outputs_per_layer
        if self.batch_first:
            out = out.transpose(0, 1)
        return out
'''
