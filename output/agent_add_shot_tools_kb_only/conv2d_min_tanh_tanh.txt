kernel_src="""
#include "kernel_operator.h"

// Fused kernel: channel-wise ReduceMin (keepdim) + Tanh + Tanh
// Input: x [N, C, H, W] (float)
// Output: y [N, 1, H, W] (float)

constexpr int32_t IN_BUF_NUM = 1;    // per-channel tile buffer
constexpr int32_t TMP_BUF_NUM = 2;   // runMin + tanh tmp
constexpr int32_t OUT_BUF_NUM = 2;   // double-buffer output

class KernelConv2dMinTanhTanh {
public:
    __aicore__ inline KernelConv2dMinTanhTanh() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t tileNum)
    {
        this->N = n;
        this->C = c;
        this->H = h;
        this->W = w;
        this->HW = h * w;
        this->tileNum = tileNum;
        // Assign batches evenly to blocks
        this->batchPerBlock = this->N / AscendC::GetBlockNum();
        this->batchStart = this->batchPerBlock * AscendC::GetBlockIdx();

        // Tile along spatial HW inside each batch
        this->tileLength = (this->HW) / (tileNum * OUT_BUF_NUM);  // expect divisible in sample
        if (this->tileLength == 0) {
            this->tileLength = this->HW; // fallback
        }

        xGm.SetGlobalBuffer((__gm__ float*)x, (int64_t)this->N * this->C * this->HW);
        yGm.SetGlobalBuffer((__gm__ float*)y, (int64_t)this->N * this->HW);

        pipe.InitBuffer(inQueue, IN_BUF_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue, TMP_BUF_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueue, OUT_BUF_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t bn = 0; bn < this->batchPerBlock; ++bn) {
            int32_t loopCount = this->tileNum * OUT_BUF_NUM;
            for (int32_t p = 0; p < loopCount; ++p) {
                ComputeOneTile(bn, p);
                CopyOut(bn, p);
            }
        }
    }

private:
    __aicore__ inline void ComputeOneTile(uint32_t bn, int32_t progress)
    {
        uint32_t nIdx = this->batchStart + bn;
        uint32_t tileOffset = progress * this->tileLength; // within HW

        // Allocate runMin buffer
        AscendC::LocalTensor<float> runMin = tmpQueue.AllocTensor<float>();
        // Load first channel tile
        int64_t baseIn0 = ((int64_t)nIdx * this->C + 0) * this->HW + tileOffset;
        AscendC::DataCopy(runMin, xGm[baseIn0], this->tileLength);

        // Loop remaining channels and compute element-wise min
        for (uint32_t cIdx = 1; cIdx < this->C; ++cIdx) {
            AscendC::LocalTensor<float> srcLocal = inQueue.AllocTensor<float>();
            int64_t baseIn = ((int64_t)nIdx * this->C + cIdx) * this->HW + tileOffset;
            AscendC::DataCopy(srcLocal, xGm[baseIn], this->tileLength);
            AscendC::Min(runMin, runMin, srcLocal, (int32_t)this->tileLength);
            inQueue.FreeTensor(srcLocal);
        }

        // Apply Tanh twice: runMin -> tanhTmp -> dstLocal
        AscendC::LocalTensor<float> tanhTmp = tmpQueue.AllocTensor<float>();
        AscendC::Tanh(tanhTmp, runMin, (uint32_t)this->tileLength);

        AscendC::LocalTensor<float> dstLocal = outQueue.AllocTensor<float>();
        AscendC::Tanh(dstLocal, tanhTmp, (uint32_t)this->tileLength);

        outQueue.EnQue<float>(dstLocal);
        tmpQueue.FreeTensor(tanhTmp);
        tmpQueue.FreeTensor(runMin);
    }

    __aicore__ inline void CopyOut(uint32_t bn, int32_t progress)
    {
        uint32_t nIdx = this->batchStart + bn;
        uint32_t tileOffset = progress * this->tileLength; // within HW

        AscendC::LocalTensor<float> dstLocal = outQueue.DeQue<float>();
        int64_t baseOut = (int64_t)nIdx * this->HW + tileOffset; // [N,1,H,W] contiguous plane
        AscendC::DataCopy(yGm[baseOut], dstLocal, this->tileLength);
        outQueue.FreeTensor(dstLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t N, C, H, W;
    uint32_t HW;
    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t batchPerBlock;
    uint32_t batchStart;
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanh op;
    op.Init(x, y,
            tiling_data.n,
            tiling_data.c,
            tiling_data.h,
            tiling_data.w,
            tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Model that performs conv2d in PyTorch, then fused channel-wise min + tanh + tanh with custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        x = self.conv(x)  # [N, C, H_out, W_out]
        # fused: channel-wise min (keepdim=True) + tanh + tanh
        x = custom_ops_lib.conv2d_min_tanh_tanh_custom(x)
        return x
