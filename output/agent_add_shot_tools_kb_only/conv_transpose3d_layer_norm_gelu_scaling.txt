kernel_src="""
#include "kernel_operator.h"

template <typename dataType, bool isReuseSource = false>
class KernelFusedLayerNormGeluScale {
public:
    __aicore__ inline KernelFusedLayerNormGeluScale() {}

    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR gammaGm, GM_ADDR betaGm, GM_ADDR yGm,
                                uint32_t bLen, uint32_t sLen, uint32_t hLen, float epsilon, float scale,
                                const LayerNormTiling &lnTiling)
    {
        bLength = bLen;
        sLength = sLen;
        hLength = hLen;
        bshLength = bLength * sLength * hLength;
        eps = epsilon;
        scl = scale;
        tiling = lnTiling;

        xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(xGm), bshLength);
        gammaGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(gammaGm), hLength);
        betaGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(betaGm), hLength);
        yGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ dataType *>(yGm), bshLength);

        pipe.InitBuffer(inQueueX, 1, sizeof(dataType) * bshLength);
        pipe.InitBuffer(inQueueGamma, 1, sizeof(dataType) * hLength);
        pipe.InitBuffer(inQueueBeta, 1, sizeof(dataType) * hLength);
        pipe.InitBuffer(outQueueY, 1, sizeof(dataType) * bshLength);
        pipe.InitBuffer(outQueueMean, 1, sizeof(dataType) * (bLength * sLength));
        pipe.InitBuffer(outQueueVar, 1, sizeof(dataType) * (bLength * sLength));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<dataType> xLocal = inQueueX.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> gammaLocal = inQueueGamma.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> betaLocal = inQueueBeta.AllocTensor<dataType>();
        AscendC::DataCopy(xLocal, xGlobal, bshLength);
        AscendC::DataCopy(gammaLocal, gammaGlobal, hLength);
        AscendC::DataCopy(betaLocal, betaGlobal, hLength);
        inQueueX.EnQue(xLocal);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<dataType> xLocal = inQueueX.DeQue<dataType>();
        AscendC::LocalTensor<dataType> gammaLocal = inQueueGamma.DeQue<dataType>();
        AscendC::LocalTensor<dataType> betaLocal = inQueueBeta.DeQue<dataType>();
        AscendC::LocalTensor<dataType> yLocal = outQueueY.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> meanLocal = outQueueMean.AllocTensor<dataType>();
        AscendC::LocalTensor<dataType> varLocal = outQueueVar.AllocTensor<dataType>();

        // LayerNorm: 输出 yLocal、均值与方差
        AscendC::LayerNorm<dataType, isReuseSource>(
            yLocal, meanLocal, varLocal, xLocal, gammaLocal, betaLocal, (dataType)eps, tiling);

        // GELU
        AscendC::Gelu(yLocal, yLocal, bshLength);

        // Scaling（可通过 tiling 传入，默认 1.0）
        if (scl != (dataType)1.0f) {
            AscendC::Muls(yLocal, yLocal, (dataType)scl, bshLength);
        }

        outQueueY.EnQue(yLocal);
        outQueueMean.EnQue(meanLocal);
        outQueueVar.EnQue(varLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueGamma.FreeTensor(gammaLocal);
        inQueueBeta.FreeTensor(betaLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<dataType> yLocal = outQueueY.DeQue<dataType>();
        AscendC::LocalTensor<dataType> meanLocal = outQueueMean.DeQue<dataType>();
        AscendC::LocalTensor<dataType> varLocal = outQueueVar.DeQue<dataType>();

        AscendC::DataCopy(yGlobal, yLocal, bshLength);

        outQueueY.FreeTensor(yLocal);
        outQueueMean.FreeTensor(meanLocal);
        outQueueVar.FreeTensor(varLocal);
    }

private:
    AscendC::GlobalTensor<dataType> xGlobal;
    AscendC::GlobalTensor<dataType> gammaGlobal;
    AscendC::GlobalTensor<dataType> betaGlobal;
    AscendC::GlobalTensor<dataType> yGlobal;

    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueGamma;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueBeta;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueMean;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueVar;

    uint32_t bLength;
    uint32_t sLength;
    uint32_t hLength;
    uint32_t bshLength;
    dataType eps;
    dataType scl;
    LayerNormTiling tiling;
};

extern "C" __global__ __aicore__ void conv_transpose3d_layer_norm_gelu_scaling_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGM)
{
    GET_TILING_DATA(tilingData, tilingGM);
    KernelFusedLayerNormGeluScale<float, false> op;
    op.Init(x, gamma, beta, y,
            tilingData.bLength, tilingData.sLength, tilingData.hLength,
            tilingData.epsilon, tilingData.scale, tilingData.layernormTilingData);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    结构：ConvTranspose3d -> (重排到[B,S,H]) -> 自定义AscendC算子( LayerNorm + GELU + 可选缩放 ) -> (重排回[N,C,D,H,W])。
    其中 LayerNorm 的归一化维度为通道 C（作为最后一维 H）。
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, bias=True, eps=1e-5, scaling_factor=1.0):
        super().__init__()
        self.conv_transpose = nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        # 仅复用参数，不直接调用 PyTorch 的 LayerNorm 计算
        self.layer_norm = nn.LayerNorm(out_channels, eps=eps)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        # x: [N, C_in, D, H, W]
        x = self.conv_transpose(x)  # -> [N, C_out, D', H', W']

        # 准备归一化权重与偏置
        gamma = self.layer_norm.weight    # [C_out]
        beta = self.layer_norm.bias       # [C_out]

        # 将通道移到最后一维，并展平成 [B, S, H] 以便 LayerNorm 在 H 维归一化
        # [N, C, D, H, W] -> [N, D, H, W, C]
        x_cl = x.permute(0, 2, 3, 4, 1).contiguous()
        N, Dp, Hp, Wp, C = x_cl.shape
        B = N * Dp * Hp * Wp
        S = 1
        H = C
        x_bsh = x_cl.view(B, S, H).contiguous()

        # 调用自定义 AscendC 内核：输出与输入形状一致 [B, S, H]
        y_bsh = custom_ops_lib.conv_transpose3d_layer_norm_gelu_scaling_custom(x_bsh, gamma, beta)

        # 还原回 [N, C, D, H, W]
        y_cl = y_bsh.view(N, Dp, Hp, Wp, C)
        y = y_cl.permute(0, 4, 1, 2, 3).contiguous()

        # 缩放（可选，默认 1.0）
        if self.scaling_factor != 1.0:
            y = y * self.scaling_factor
        return y

def get_inputs():
    # 输入: [batch_size, in_channels, D, H, W]
    batch_size = 32
    in_channels = 32
    D, H, W = 16, 32, 32
    return [torch.rand(batch_size, in_channels, D, H, W)]

def get_init_inputs():
    # 与原模型一致
    in_channels = 32
    out_channels = 64
    kernel_size = 4
    stride = 2
    padding = 1
    bias = True
    eps = 1e-5
    scaling_factor = 1.0
    return [in_channels, out_channels, kernel_size, stride, padding, bias, eps, scaling_factor]
