kernel_src="""
#include "kernel_operator.h"

class KernelScatterAdd {
public:
    __aicore__ inline KernelScatterAdd() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR updates, GM_ADDR z,
                                uint32_t rows, uint32_t cols, uint32_t k,
                                uint32_t rowTile, uint32_t kTile)
    {
        this->rows = rows;
        this->cols = cols;
        this->k = k;
        this->rowTile = rowTile;
        this->kTile = kTile;

        uint32_t row_id = AscendC::GetBlockIdx();
        // 每个block绑定一行
        xRow.SetGlobalBuffer((__gm__ float*)x + row_id * cols, cols);
        zRow.SetGlobalBuffer((__gm__ float*)z + row_id * cols, cols);
        idxRow.SetGlobalBuffer((__gm__ int32_t*)idx + row_id * k, k);
        updatesRow.SetGlobalBuffer((__gm__ float*)updates + row_id * k, k);

        // 缓冲区：行拷贝使用inQueueX作为输入本地缓冲，outQueueZ作为输出本地缓冲
        pipe.InitBuffer(inQueueX, 1, rowTile * sizeof(float));
        pipe.InitBuffer(outQueueZ, 1, rowTile * sizeof(float));

        // 更新阶段：索引缓冲采用inQueueIdx，更新值缓冲采用outQueueUpd（保证VECOUT->GM路径以支持原子加）
        pipe.InitBuffer(inQueueIdx, 1, kTile * sizeof(int32_t));
        pipe.InitBuffer(outQueueUpd, 1, kTile * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // 1) 初始化z = x（按列分块拷贝）
        for (uint32_t offset = 0; offset < cols; offset += rowTile) {
            uint32_t curLen = AscendC::MIN(rowTile, cols - offset);
            CopyRowChunk(offset, curLen);
        }

        // 2) scatter_add: 对每个更新块，执行按索引的原子加
        for (uint32_t off = 0; off < k; off += kTile) {
            uint32_t curLen = AscendC::MIN(kTile, k - off);
            ScatterAddChunk(off, curLen);
        }
    }

private:
    __aicore__ inline void CopyRowChunk(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xRow[offset], len);
        AscendC::DataCopy(zRow[offset], xLocal, len);
        inQueueX.FreeTensor(xLocal);
        outQueueZ.FreeTensor(zLocal);
    }

    __aicore__ inline void ScatterAddChunk(uint32_t off, uint32_t len)
    {
        // 加载索引与更新值
        AscendC::LocalTensor<int32_t> idxLocal = inQueueIdx.AllocTensor<int32_t>();
        AscendC::LocalTensor<float> updLocal = outQueueUpd.AllocTensor<float>();
        AscendC::DataCopy(idxLocal, idxRow[off], len);
        AscendC::DataCopy(updLocal, updatesRow[off], len);

        // 针对每个元素执行原子加：z[row, idx[i]] += updates[row, i]
        // 为确保VECOUT->GM原子路径，源缓冲为VECOUT（updLocal）
        for (uint32_t i = 0; i < len; ++i) {
            // 读取索引并做范围保护
            int32_t col = idxLocal.GetValue(i);
            if (col >= 0 && (uint32_t)col < cols) {
                AscendC::SetAtomicAdd<float>();
                // 单元素原子加
                AscendC::DataCopy(zRow[col], updLocal[i], 1);
                AscendC::SetAtomicNone();
            }
        }

        inQueueIdx.FreeTensor(idxLocal);
        outQueueUpd.FreeTensor(updLocal);
    }

private:
    AscendC::TPipe pipe;

    // 队列
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;

    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueUpd;

    // 行视图的GM张量
    AscendC::GlobalTensor<float> xRow;
    AscendC::GlobalTensor<float> zRow;
    AscendC::GlobalTensor<int32_t> idxRow;
    AscendC::GlobalTensor<float> updatesRow;

    // 形状与分块
    uint32_t rows;
    uint32_t cols;
    uint32_t k;
    uint32_t rowTile;
    uint32_t kTile;
};

extern "C" __global__ __aicore__ void scatter_add_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR updates, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelScatterAdd op;
    op.Init(x, idx, updates, z,
            tiling_data.rows, tiling_data.cols, tiling_data.k,
            tiling_data.rowTile, tiling_data.kTile);
    op.Process();
}
