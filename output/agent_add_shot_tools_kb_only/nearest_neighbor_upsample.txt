kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // 每个队列的tensor数
// 最近邻上采样，固定scale=4（由tiling传入），输入输出均为float类型（DTYPE_X/DTYPE_Y由框架映射）
class KernelNearestNeighborUpsample {
public:
    __aicore__ inline KernelNearestNeighborUpsample() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t out_h, uint32_t out_w, uint32_t scale)
    {
        this->n = n; this->c = c; this->h = h; this->w = w;
        this->out_h = out_h; this->out_w = out_w; this->scale = scale;

        // 设置全局buffer
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, static_cast<uint64_t>(n)*c*h*w);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, static_cast<uint64_t>(n)*c*out_h*out_w);

        // UB行缓冲: 输入一行(w)与输出一行(out_w)
        pipe.InitBuffer(inRowQueue, BUFFER_NUM, w * sizeof(DTYPE_X));
        pipe.InitBuffer(outRowQueue, BUFFER_NUM, out_w * sizeof(DTYPE_Y));

        // 行总数(N*C*H)，按core划分
        totalRows = n * c * h;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (totalRows + blockNum - 1) / blockNum;
        startRow = rowsPerBlock * blockIdx;
        endRow = startRow + rowsPerBlock;
        if (endRow > totalRows) endRow = totalRows;
    }

    __aicore__ inline void Process()
    {
        for (uint32_t row = startRow; row < endRow; ++row) {
            CopyIn(row);
            Compute(row);
            CopyOut(row);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_X> inRowLocal = inRowQueue.AllocTensor<DTYPE_X>();
        // 输入第row行起始偏移（每行长度为w）
        uint64_t inOffset = static_cast<uint64_t>(row) * w;
        AscendC::DataCopy(inRowLocal, xGm[inOffset], w);
        inRowQueue.EnQue(inRowLocal);
    }

    __aicore__ inline void Compute(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_X> inRowLocal = inRowQueue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> outRowLocal = outRowQueue.AllocTensor<DTYPE_Y>();

        // 水平方向最近邻放大: 每个元素复制scale次
        // 通过对LocalTensor的切片与DataCopy实现逐元素复制
        for (uint32_t col = 0; col < w; ++col) {
            for (uint32_t r = 0; r < scale; ++r) {
                // 目标位置为 col*scale + r，长度1
                AscendC::DataCopy(outRowLocal[col * scale + r], inRowLocal[col], 1);
            }
        }

        outRowQueue.EnQue<DTYPE_Y>(outRowLocal);
        inRowQueue.FreeTensor(inRowLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_Y> outRowLocal = outRowQueue.DeQue<DTYPE_Y>();

        // 计算(n, c, h)索引
        uint32_t ch = c * h;
        uint32_t nIdx = row / ch;
        uint32_t rem = row % ch;
        uint32_t cIdx = rem / h;
        uint32_t hIdx = rem % h;

        // 垂直方向最近邻放大: 将扩大后的单行复制scale次到输出
        for (uint32_t vr = 0; vr < scale; ++vr) {
            uint64_t outRowIndex = ((static_cast<uint64_t>(nIdx) * c + cIdx) * (static_cast<uint64_t>(out_h)) + (static_cast<uint64_t>(hIdx) * scale + vr));
            uint64_t outOffset = outRowIndex * out_w;
            AscendC::DataCopy(yGm[outOffset], outRowLocal, out_w);
        }

        outRowQueue.FreeTensor(outRowLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inRowQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outRowQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, h, w;
    uint32_t out_h, out_w;
    uint32_t scale;

    uint32_t totalRows;
    uint32_t startRow;
    uint32_t endRow;
};

extern "C" __global__ __aicore__ void nearest_neighbor_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelNearestNeighborUpsample op;
    op.Init(x, y,
            tiling_data.n(), tiling_data.c(), tiling_data.h(), tiling_data.w(),
            tiling_data.out_h(), tiling_data.out_w(), tiling_data.scale());
    op.Process();
}

at::Tensor nearest_neighbor_upsample_custom_impl_npu(const at::Tensor& self) {
    TORCH_CHECK(self.dim() == 4, \"input must be 4D [N,C,H,W]\");
    int64_t scale = 4;
    auto sizes = self.sizes();
    int64_t N = sizes[0];
    int64_t C = sizes[1];
    int64_t H = sizes[2];
    int64_t W = sizes[3];
    at::IntArrayRef out_sizes{N, C, H * scale, W * scale};
    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnNearestNeighborUpsampleCustom, self, result);
    return result;
}

    m.impl(\"nearest_neighbor_upsample_custom\", &nearest_neighbor_upsample_custom_impl_npu);
