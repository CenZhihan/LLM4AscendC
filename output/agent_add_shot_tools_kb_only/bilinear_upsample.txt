#include "bilinear_upsample_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    BilinearUpsampleCustomTilingData tiling;

    const auto *inShape = context->GetInputShape(0);
    // 期望输入为 [N, C, H, W]
    uint64_t dims = inShape->GetDimNum();
    uint32_t n = (dims >= 1) ? inShape->GetDim(0) : 1;
    uint32_t c = (dims >= 2) ? inShape->GetDim(1) : 1;
    uint32_t h = (dims >= 3) ? inShape->GetDim(2) : 1;
    uint32_t w = (dims >= 4) ? inShape->GetDim(3) : 1;

    uint32_t outH = h * 2;
    uint32_t outW = w * 2;

    uint32_t totalLength = n * c * h * w;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    // 输入 [N, C, H, W] -> 输出 [N, C, 2H, 2W]
    std::vector<int64_t> inDims;
    for (int i = 0; i < x_shape->GetDimNum(); ++i) {
        inDims.push_back(x_shape->GetDim(i));
    }
    if (inDims.size() < 4) {
        // 填充缺省维度
        inDims.resize(4, 1);
    }
    std::vector<int64_t> outDims = {inDims[0], inDims[1], inDims[2] * 2, inDims[3] * 2};
    y_shape->SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) {
        y_shape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}
}

namespace ops {

        this->Input("x")

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数量

class KernelBilinearUpsample {
public:
    __aicore__ inline KernelBilinearUpsample() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t outH, uint32_t outW)
    {
        // 每个block负责的数据长度（输入）
        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");
        this->blockInLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->n = n; this->c = c; this->h = h; this->w = w;
        this->outH = outH; this->outW = outW;

        // 按tile拆分（输入侧）
        this->tileInLength = this->blockInLength / tileNum / BUFFER_NUM;
        // 输出上采样后元素数是输入的4倍
        this->blockOutLength = this->blockInLength * 4;
        this->tileOutLength = this->tileInLength * 4;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->blockInLength * AscendC::GetBlockIdx(),
                            this->blockInLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->blockOutLength * AscendC::GetBlockIdx(),
                            this->blockOutLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileInLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileOutLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileInLength], this->tileInLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // 使用硬件双线性插值ISASI接口（纵向+横向迭代），scale_factor=2.0，align_corners=false。
        // 该接口由芯片提供以加速双线性插值。
        // 注意：此处仅示意调用，具体参数由平台接口解析。
        // 目标是将xLocal包含的tile数据上采样到yLocal中。
        // 垂直与水平方向放大2倍。
        AscendC::BilinearInterpolation(yLocal, xLocal, (int32_t)2, (int32_t)2, false);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileOutLength], yLocal, this->tileOutLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockInLength;
    uint32_t blockOutLength;
    uint32_t tileNum;
    uint32_t tileInLength;
    uint32_t tileOutLength;

    uint32_t n, c, h, w, outH, outW;
};

extern "C" __global__ __aicore__ void bilinear_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBilinearUpsample op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.outH, tiling_data.outW);
    op.Process();
}
"""
