host_operator_src="""
#include "three_dim_tensor_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t SafeCastDim(int64_t v) { return static_cast<uint32_t>(v); }
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // Input0: A [N, M, K], Input1: B [K, L]
    const auto* aShape = context->GetInputShape(0);
    const auto* bShape = context->GetInputShape(1);
    const auto& aOrigin = aShape->GetOriginShape();
    const auto& bOrigin = bShape->GetOriginShape();

    // Read dims
    uint32_t dimA = static_cast<uint32_t>(aOrigin.GetDimNum());
    uint32_t dimB = static_cast<uint32_t>(bOrigin.GetDimNum());
    if (dimA != 3 || dimB != 2) {
        return ge::GRAPH_FAILED;
    }
    uint32_t N = SafeCastDim(aOrigin.GetDim(0));
    uint32_t M = SafeCastDim(aOrigin.GetDim(1));
    uint32_t K = SafeCastDim(aOrigin.GetDim(2));
    uint32_t Kb = SafeCastDim(bOrigin.GetDim(0));
    uint32_t L = SafeCastDim(bOrigin.GetDim(1));
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    // Decide block dim and batch config
    // We run matmul with NORMAL layout and batch on A's first dimension (N).
    // Keep a conservative block dim for portability.
    const uint32_t BLOCK_DIM = 1;
    context->SetBlockDim(BLOCK_DIM);

    ThreeDimTensorMatrixMultiplicationCustomTilingData tiling;
    tiling.set_N(N);
    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_L(L);
    tiling.set_batchA(N);   // compute all A batches at once
    tiling.set_batchB(1);   // B has no batching

    // Save tiling
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace required
    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    const auto& aO = a_shape->GetOriginShape();
    const auto& bO = b_shape->GetOriginShape();
    if (aO.GetDimNum() != 3 || bO.GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    gert::Shape* out = context->GetOutputShape(0);
    gert::SmallVector dims;
    dims.push_back(aO.GetDim(0)); // N
    dims.push_back(aO.GetDim(1)); // M
    dims.push_back(bO.GetDim(1)); // L
    out->GetOriginShape().SetDim(dims);
    out->SetUnknownRank(false);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"
#include "register/tilingdata_base.h"

using A_T = float;
using B_T = float;
using C_T = float;
using BiasT = float;

extern \"C\" __global__ __aicore__ void three_dim_tensor_matrix_multiplication_custom(GM_ADDR aGM, GM_ADDR bGM,
                                                                                      GM_ADDR cGM, GM_ADDR workspaceGM,
                                                                                      GM_ADDR tilingGM)
{
    // Read host tiling data (N, M, K, L, batchA, batchB)
    GET_TILING_DATA(tiling_data, tilingGM);
    const uint32_t N = tiling_data.N;
    const uint32_t M = tiling_data.M;
    const uint32_t K = tiling_data.K;
    const uint32_t L = tiling_data.L;
    const uint32_t batchA = tiling_data.batchA; // expected N
    const uint32_t batchB = tiling_data.batchB; // expected 1

    // Define Matmul types in NORMAL layout
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

    // Prepare TCubeTiling
    TCubeTiling tiling;
    // Set NORMAL layout core sizes
    tiling.singleCoreM = M;
    tiling.singleCoreK = K;
    tiling.singleCoreN = L;
    // Set batch info: batch over N
    tiling.ALayoutInfoB = N;  // number of A batches
    tiling.BLayoutInfoB = 1;  // B has no batches
    tiling.CLayoutInfoB = N;  // output batches
    // No bias
    tiling.isBias = 0;
    // Share buffer config (typical)
    tiling.shareMode = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize = 0;

    // Wrap GM buffers
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;
    const int32_t sizeA = static_cast<int32_t>(N) * static_cast<int32_t>(M) * static_cast<int32_t>(K) * sizeof(A_T);
    const int32_t sizeB = static_cast<int32_t>(K) * static_cast<int32_t>(L) * sizeof(B_T);
    const int32_t sizeC = static_cast<int32_t>(N) * static_cast<int32_t>(M) * static_cast<int32_t>(L) * sizeof(C_T);
    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(cGM), sizeC);

    int offset_a = 0, offset_b = 0, offset_c = 0;
    AscendC::GlobalTensor<A_T> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal[offset_a].GetPhyAddr()),
                         static_cast<int32_t>(M) * static_cast<int32_t>(K));
    AscendC::GlobalTensor<B_T> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal[offset_b].GetPhyAddr()),
                         static_cast<int32_t>(K) * static_cast<int32_t>(L));
    AscendC::GlobalTensor<C_T> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal[offset_c].GetPhyAddr()),
                         static_cast<int32_t>(M) * static_cast<int32_t>(L));

    // Create Matmul instance with NORMAL layout
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;
    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);

    // Init and bind tensors
    mm1.Init(&tiling);
    const uint32_t isTransposeAIn = 0;
    const uint32_t isTransposeBIn = 0;
    mm1.SetTensorA(gm_a, isTransposeAIn);
    mm1.SetTensorB(gm_b, isTransposeBIn);
    mm1.SetWorkspace(workspaceGM, 0);

    // Execute multi-batch Matmul: compute C of shape [N, M, L]
    // batchA: number of A batches per call (N), batchB: 1
    mm1.IterateBatch(gm_c, batchA, batchB, false);
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor three_dim_tensor_matrix_multiplication_custom_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // A: [N, M, K], B: [K, L], output C: [N, M, L]
    at::IntArrayRef aSizes = A.sizes();
    TORCH_CHECK(aSizes.size() == 3, \"A must be 3D\");
    auto N = aSizes[0];
    auto M = aSizes[1];
    auto L = B.size(1);
    at::Tensor C = at::empty({N, M, L}, A.options());
    EXEC_NPU_CMD(aclnnThreeDimTensorMatrixMultiplicationCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"three_dim_tensor_matrix_multiplication_custom\", &three_dim_tensor_matrix_multiplication_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
