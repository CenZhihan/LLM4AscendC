kernel_src="""
#include "kernel_operator.h"

// 本自定义算子实现：对输入张量逐元素执行 LeakyReLU -> LeakyReLU -> GELU -> GELU 的链式计算。
// 注：GEMM 与 LogSumExp 在 PyTorch 侧完成，本核对其输出进行进一步非线性加速。
constexpr int32_t BUFFER_NUM = 2; // 每个队列中的张量数量，双缓冲

class KernelFusedAct {
public:
    __aicore__ inline KernelFusedAct() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        // 每个AI Core的分块再细分为 tileNum * BUFFER_NUM 片
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ float *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ float *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
        // 分配中间计算缓冲区（两段临时缓冲）
        pipe.InitBuffer(calcBufs, this->tileLength * sizeof(float) * 2);
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();

        // 两段中间缓冲
        AscendC::LocalTensor<float> tmp1 = calcBufs.Get<float>(0);
        AscendC::LocalTensor<float> tmp2 = calcBufs.Get<float>(this->tileLength * sizeof(float));

        // LeakyReLU 1: tmp1 = LeakyRelu(xLocal, 0.01)
        const float slope = 0.01f;
        AscendC::LeakyRelu(tmp1, xLocal, slope, this->tileLength);

        // LeakyReLU 2: tmp2 = LeakyRelu(tmp1, 0.01)
        AscendC::LeakyRelu(tmp2, tmp1, slope, this->tileLength);

        // GELU 1: tmp1 = Gelu(tmp2)
        AscendC::Gelu(tmp1, tmp2, this->tileLength);

        // GELU 2: yLocal = Gelu(tmp1)
        AscendC::Gelu(yLocal, tmp1, this->tileLength);

        outQueueY.EnQue<float>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBufs;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedAct op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    计算流程：
    1) 线性层（GEMM）；
    2) 按 dim=1 执行 LogSumExp，保持 keepdim=True；
    3) 使用自定义AscendC算子执行 LeakyReLU -> LeakyReLU -> GELU -> GELU。
    """
    def __init__(self, in_features, out_features, bias=True):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features, bias=bias)

    def forward(self, x):
        x = self.linear(x)  # GEMM
        x = torch.logsumexp(x, dim=1, keepdim=True)  # LogSumExp
        x = custom_ops_lib.gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(x)  # LeakyReLU*2 + GELU*2
        return x
