#include "where_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    WhereBroadcastCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(1);
    const auto& dims = x_shape->GetOriginShape().GetDims();

    uint32_t d0 = 1, d1 = 1, d2 = 1;
    if (dims.size() >= 3) {
        d0 = static_cast<uint32_t>(dims[dims.size() - 3]);
        d1 = static_cast<uint32_t>(dims[dims.size() - 2]);
        d2 = static_cast<uint32_t>(dims[dims.size() - 1]);
    } else if (dims.size() == 2) {
        d0 = 1;
        d1 = static_cast<uint32_t>(dims[0]);
        d2 = static_cast<uint32_t>(dims[1]);
    } else if (dims.size() == 1) {
        d0 = 1;
        d1 = 1;
        d2 = static_cast<uint32_t>(dims[0]);
    }

    tiling.set_d0(d0);
    tiling.set_d1(d1);
    tiling.set_d2(d2);

    // 为 SelectWithBytesMask 预留的临时空间，按最后一维长度配置
    tiling.set_tmpSize(d2);

    // 设置并行块数
    uint32_t blockDim = d0 < 8 ? d0 : 8;
    if (blockDim == 0) {
        blockDim = 1;
    }
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(1);
    gert::Shape* z_shape = context->GetOutputShape(0);
    *z_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto xType = context->GetInputDataType(1);
    context->SetOutputDataType(0, xType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

class KernelWhereBroadcast {
public:
    __aicore__ inline KernelWhereBroadcast() {}

    __aicore__ inline void Init(GM_ADDR cond, GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t d0, uint32_t d1, uint32_t d2, uint32_t tmpSize)
    {
        this->d0 = d0;
        this->d1 = d1;
        this->d2 = d2;
        this->planeStride = d1 * d2;

        // 按第0维（batch维）划分 block
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t per = (d0 + blockNum - 1) / blockNum;
        this->iStart = per * blockIdx;
        if (this->iStart > d0) this->iStart = d0;
        uint32_t iEnd = this->iStart + per;
        if (iEnd > d0) iEnd = d0;
        this->iCount = (iEnd >= this->iStart) ? (iEnd - this->iStart) : 0;

        // 设置全局缓冲区（从该 block 的起始位置偏移）
        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x) + (uint64_t)this->iStart * this->planeStride,
                            (uint64_t)this->iCount * this->planeStride);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y) + (uint64_t)this->iStart * this->planeStride,
                            (uint64_t)this->iCount * this->planeStride);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(z) + (uint64_t)this->iStart * this->planeStride,
                            (uint64_t)this->iCount * this->planeStride);

        // cond 仅依赖中间维（d1），如示例 (1, 512, 1)，只需长度为 d1 的布尔向量
        condGm.SetGlobalBuffer(reinterpret_cast<__gm__ bool*>(cond), d1);

        // 初始化队列与本地缓冲
        pipe.InitBuffer(inQueueX, 1, d2 * sizeof(float));
        pipe.InitBuffer(inQueueY, 1, d2 * sizeof(float));
        pipe.InitBuffer(outQueueZ, 1, d2 * sizeof(float));
        pipe.InitBuffer(maskQueue, 1, sizeof(bool));                  // 每行一个布尔
        pipe.InitBuffer(tmpQueue, 1, tmpSize);                        // SelectWithBytesMask 的临时缓冲
    }

    __aicore__ inline void Process()
    {
        if (iCount == 0) { return; }
        AscendC::SelectWithBytesMaskShapeInfo info;
        info.firstAxis = 1;           // 每次处理一行
        info.srcLastAxis = d2;        // 行长度
        info.maskLastAxis = 1;        // 每行单个布尔，按最后一维广播

        for (uint32_t ii = 0; ii < iCount; ++ii) {
            for (uint32_t j = 0; j < d1; ++j) {
                uint64_t baseOffset = (uint64_t)ii * planeStride + (uint64_t)j * d2;

                // 拷贝一行 x, y
                AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                AscendC::LocalTensor<float> yLocal = inQueueY.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm[baseOffset], d2);
                AscendC::DataCopy(yLocal, yGm[baseOffset], d2);
                inQueueX.EnQue(xLocal);
                inQueueY.EnQue(yLocal);

                // 拷贝该行对应的布尔条件（单个数据）
                AscendC::LocalTensor<bool> maskLocal = maskQueue.AllocTensor<bool>();
                AscendC::DataCopy(maskLocal, condGm[j], 1);
                maskQueue.EnQue(maskLocal);

                // 选择并写回
                AscendC::LocalTensor<float> xL = inQueueX.DeQue<float>();
                AscendC::LocalTensor<float> yL = inQueueY.DeQue<float>();
                AscendC::LocalTensor<bool> mL = maskQueue.DeQue<bool>();
                AscendC::LocalTensor<uint8_t> tmpLocal = tmpQueue.AllocTensor<uint8_t>();
                AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

                // 根据布尔掩码选择：z = cond ? x : y
                AscendC::SelectWithBytesMask<float, bool>(zLocal, xL, yL, mL, tmpLocal, info);

                outQueueZ.EnQue<float>(zLocal);

                // 释放局部
                inQueueX.FreeTensor(xL);
                inQueueY.FreeTensor(yL);
                maskQueue.FreeTensor(mL);
                tmpQueue.FreeTensor(tmpLocal);

                // 回写到 GM
                AscendC::LocalTensor<float> zL = outQueueZ.DeQue<float>();
                AscendC::DataCopy(zGm[baseOffset], zL, d2);
                outQueueZ.FreeTensor(zL);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueY;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> maskQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueZ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<float> zGm;
    AscendC::GlobalTensor<bool>  condGm;

    uint32_t d0 = 0, d1 = 0, d2 = 0;
    uint32_t planeStride = 0;
    uint32_t iStart = 0, iCount = 0;
};

extern "C" __global__ __aicore__ void where_broadcast_custom(GM_ADDR cond, GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelWhereBroadcast op;
    op.Init(cond, x, y, z, tilingData.d0, tilingData.d1, tilingData.d2, tilingData.tmpSize);
    op.Process();
}
"""
