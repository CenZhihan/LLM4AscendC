kernel_src="""
#include "kernel_operator.h"

using DTYPE_A = float;
using DTYPE_B = float;
using DTYPE_C = float;

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulWithDiagonalMatrices {
public:
    __aicore__ inline KernelMatmulWithDiagonalMatrices() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t nRows, uint32_t nCols, uint32_t tileCols)
    {
        this->nRows = nRows;
        this->nCols = nCols;
        this->tileCols = tileCols;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // 行切分到block
        this->rowsPerBlock = (nRows + blockNum - 1) / blockNum;
        this->startRow = blockIdx * this->rowsPerBlock;
        uint32_t tentativeEnd = this->startRow + this->rowsPerBlock;
        this->endRow = (tentativeEnd > nRows) ? nRows : tentativeEnd;
        if (this->startRow >= this->endRow) {
            this->rowsPerBlock = 0;
        }

        // 设置全局内存窗口
        aGm.SetGlobalBuffer((__gm__ DTYPE_A*)a + this->startRow, (this->endRow > this->startRow) ? (this->endRow - this->startRow) : 0);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b + (uint64_t)this->startRow * this->nCols, (this->endRow - this->startRow) * this->nCols);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C*)c + (uint64_t)this->startRow * this->nCols, (this->endRow - this->startRow) * this->nCols);

        // 初始化队列缓冲
        pipe.InitBuffer(inQueueScalar, BUFFER_NUM, sizeof(DTYPE_A));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileCols * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, this->tileCols * sizeof(DTYPE_C));
    }

    __aicore__ inline void Process()
    {
        if (this->rowsPerBlock == 0) {
            return;
        }

        for (uint32_t r = 0; r < (this->endRow - this->startRow); ++r) {
            // 拷贝当前行对应的标量 A[startRow + r]
            AscendC::LocalTensor<DTYPE_A> aLocal = inQueueScalar.AllocTensor<DTYPE_A>();
            AscendC::DataCopy(aLocal, aGm[r], 1);
            DTYPE_A alpha = aLocal.GetValue(0);
            inQueueScalar.FreeTensor(aLocal);

            // 按列分块处理 B 的这一行，输出到 C
            for (uint32_t c = 0; c < this->nCols; c += this->tileCols) {
                uint32_t len = (c + this->tileCols <= this->nCols) ? this->tileCols : (this->nCols - c);

                AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
                AscendC::DataCopy(bLocal, bGm[(uint64_t)r * this->nCols + c], len);

                AscendC::LocalTensor<DTYPE_C> cLocal = outQueueC.AllocTensor<DTYPE_C>();
                // 每个元素乘以标量 alpha
                AscendC::Muls(cLocal, bLocal, alpha, static_cast<int32_t>(len));

                AscendC::DataCopy(cGm[(uint64_t)r * this->nCols + c], cLocal, len);

                inQueueB.FreeTensor(bLocal);
                outQueueC.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueScalar;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t nRows;
    uint32_t nCols;
    uint32_t tileCols;

    uint32_t rowsPerBlock;
    uint32_t startRow;
    uint32_t endRow;
};

extern "C" __global__ __aicore__ void matmul_with_diagonal_matrices_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithDiagonalMatrices op;
    op.Init(a, b, c, tiling_data.nRows, tiling_data.nCols, tiling_data.tileCols);
    op.Process();
}
