kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelGemmDivideSumScaling {
public:
    __aicore__ inline KernelGemmDivideSumScaling() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR scale, GM_ADDR y,
                                uint32_t batchSize, uint32_t inputSize, uint32_t hiddenSize)
    {
        this->B = batchSize;
        this->I = inputSize;
        this->H = hiddenSize;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(x), static_cast<int64_t>(B) * I);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(weight), static_cast<int64_t>(H) * I);
        sGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(scale), 1);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(y), B);

        // 本算子局部缓冲区初始化
        pipe.InitBuffer(inQueueW, 1, I * sizeof(DTYPE));     // weight 行缓存
        pipe.InitBuffer(inQueueX, 1, I * sizeof(DTYPE));     // x 行缓存
        pipe.InitBuffer(tmpQueue, 3, I * sizeof(DTYPE));     // vAcc/vTmp/中间mul
        pipe.InitBuffer(alphaVecQ, 1, I * sizeof(DTYPE));    // alpha 向量
        pipe.InitBuffer(alphaQ, 2, sizeof(DTYPE));           // 标量 alpha/0.5
        pipe.InitBuffer(workQ, 1, I * sizeof(DTYPE));        // ReduceSum 工作区
        pipe.InitBuffer(outScalarQ, 1, sizeof(DTYPE));       // 输出标量
    }

    __aicore__ inline void Process()
    {
        // 1) 计算 v = sum_j weight[j, :]  (按行累加到长度为 I 的向量)
        AscendC::LocalTensor<DTYPE> vAcc = tmpQueue.AllocTensor<DTYPE>();
        AscendC::Duplicate(vAcc, (DTYPE)0.0f, I);

        for (uint32_t r = 0; r < H; ++r) {
            AscendC::LocalTensor<DTYPE> rowLocal = inQueueW.AllocTensor<DTYPE>();
            AscendC::DataCopy(rowLocal, wGm[static_cast<int64_t>(r) * I], I);

            AscendC::LocalTensor<DTYPE> vTmp = tmpQueue.AllocTensor<DTYPE>();
            AscendC::Add(vTmp, vAcc, rowLocal, I);
            // 将 vTmp 拷贝回 vAcc 以便继续累加
            AscendC::DataCopy(vAcc, vTmp, I);

            tmpQueue.FreeTensor(vTmp);
            inQueueW.FreeTensor(rowLocal);
        }

        // 2) 计算 alpha = scale * 0.5，并广播到长度为 I 的向量
        AscendC::LocalTensor<DTYPE> halfLocal = alphaQ.AllocTensor<DTYPE>();
        AscendC::Duplicate(halfLocal, (DTYPE)0.5f, 1);

        AscendC::LocalTensor<DTYPE> scaleLocal = alphaQ.AllocTensor<DTYPE>();
        AscendC::DataCopy(scaleLocal, sGm, 1);

        AscendC::LocalTensor<DTYPE> alphaLocal = alphaQ.AllocTensor<DTYPE>();
        AscendC::Mul(alphaLocal, scaleLocal, halfLocal, 1);

        AscendC::LocalTensor<DTYPE> alphaVec = alphaVecQ.AllocTensor<DTYPE>();
        uint32_t dstShape[1] = { I };
        uint32_t srcShape[1] = { 1 };
        AscendC::BroadCast<DTYPE, 1, 0>(alphaVec, alphaLocal, dstShape, srcShape);

        // 3) v = v * alpha
        AscendC::LocalTensor<DTYPE> vScaled = tmpQueue.AllocTensor<DTYPE>();
        AscendC::Mul(vScaled, vAcc, alphaVec, I);
        AscendC::DataCopy(vAcc, vScaled, I);

        tmpQueue.FreeTensor(vScaled);
        alphaVecQ.FreeTensor(alphaVec);
        alphaQ.FreeTensor(alphaLocal);
        alphaQ.FreeTensor(scaleLocal);
        alphaQ.FreeTensor(halfLocal);

        // 4) 对每个 batch 行：y[b] = sum_k ( x[b,k] * v[k] )
        for (uint32_t b = 0; b < B; ++b) {
            AscendC::LocalTensor<DTYPE> xRow = inQueueX.AllocTensor<DTYPE>();
            AscendC::DataCopy(xRow, xGm[static_cast<int64_t>(b) * I], I);

            AscendC::LocalTensor<DTYPE> mulTmp = tmpQueue.AllocTensor<DTYPE>();
            AscendC::Mul(mulTmp, xRow, vAcc, I);

            AscendC::LocalTensor<DTYPE> workLocal = workQ.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> outScalar = outScalarQ.AllocTensor<DTYPE>();
            AscendC::ReduceSum(outScalar, mulTmp, workLocal, I);

            AscendC::DataCopy(yGm[b], outScalar, 1);

            outScalarQ.FreeTensor(outScalar);
            workQ.FreeTensor(workLocal);
            tmpQueue.FreeTensor(mulTmp);
            inQueueX.FreeTensor(xRow);
        }

        tmpQueue.FreeTensor(vAcc);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueW, inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 3> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> alphaVecQ;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 2> alphaQ;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> workQ;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outScalarQ;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> sGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t B{0}, I{0}, H{0};
};

extern "C" __global__ __aicore__ void gemm_divide_sum_scaling_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmDivideSumScaling op;
    op.Init(x, weight, scale, y, tiling_data.batchSize(), tiling_data.inputSize(), tiling_data.hiddenSize());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    将 x @ weight.T / 2 再按 dim=1 求和并乘以 scaling_factor 转换为：
    输出[b] = sum_k x[b,k] * (scaling_factor/2 * sum_j weight[j,k])
    在自定义 AscendC 算子中实现为：先按行累计 weight 得到向量，再进行逐 batch 的点积。
    """
    def __init__(self, input_size, hidden_size, scaling_factor):
        super(ModelNew, self).__init__()
        self.weight = torch.nn.Parameter(torch.randn(hidden_size, input_size))
        self.scaling_factor = scaling_factor

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        scale = torch.tensor([self.scaling_factor], dtype=x.dtype, device=x.device)
        return custom_ops_lib.gemm_divide_sum_scaling_custom(x, self.weight, scale)
'''
