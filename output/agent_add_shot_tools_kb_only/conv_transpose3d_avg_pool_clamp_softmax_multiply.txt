kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

class KernelClampSoftmaxMultiply {
public:
    __aicore__ inline KernelClampSoftmaxMultiply() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR clamp_min, GM_ADDR clamp_max, GM_ADDR z,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t b, uint32_t c, uint32_t d, uint32_t h, uint32_t w)
    {
        this->totalLength = totalLength;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->B = b; this->C = c; this->D = d; this->H = h; this->W = w;
        this->DHW = d * h * w;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE*)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE*)scale);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        // 缓冲区：scale向量（长度C），以及clamp上下限（各1个元素）
        pipe.InitBuffer(scaleQueue, 1, this->C * sizeof(DTYPE));
        pipe.InitBuffer(minQueue, 1, sizeof(DTYPE));
        pipe.InitBuffer(maxQueue, 1, sizeof(DTYPE));

        // 将clamp标量与scale拷贝到本地一次
        AscendC::LocalTensor<DTYPE> scaleLocal = scaleQueue.AllocTensor<DTYPE>();
        AscendC::DataCopy(scaleLocal, scaleGm, this->C);
        scaleQueue.EnQue(scaleLocal);

        // clamp_min
        clampMinGlobal.SetGlobalBuffer((__gm__ DTYPE*)clamp_min);
        clampMaxGlobal.SetGlobalBuffer((__gm__ DTYPE*)clamp_max);
        AscendC::LocalTensor<DTYPE> minLocal = minQueue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> maxLocal = maxQueue.AllocTensor<DTYPE>();
        AscendC::DataCopy(minLocal, clampMinGlobal, 1);
        AscendC::DataCopy(maxLocal, clampMaxGlobal, 1);
        minQueue.EnQue(minLocal);
        maxQueue.EnQue(maxLocal);
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.AllocTensor<DTYPE>();

        // 取出scale与clamp标量
        AscendC::LocalTensor<DTYPE> scaleLocal = scaleQueue.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> minLocal = minQueue.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> maxLocal = maxQueue.DeQue<DTYPE>();
        DTYPE vmin = minLocal.GetValue(0);
        DTYPE vmax = maxLocal.GetValue(0);

        // 逐元素：clamp
        for (uint32_t i = 0; i < this->tileLength; ++i) {
            DTYPE v = xLocal.GetValue(i);
            if (v < vmin) v = vmin;
            if (v > vmax) v = vmax;
            zLocal.SetValue(i, v);
        }

        // 软最大化（按通道内空间维度归一化）：
        // 将tile映射到全局索引，按行(M=B*C)与列(N=DHW)做分块处理。
        // 简化处理：对每个通道的片段在tile内执行按DHW的softmax。若跨越边界，仅在当前tile范围内归一化。
        // 计算softmax: 先按通道段计算最大值与和，再归一化。
        uint64_t tileStart = (uint64_t)this->blockLength * AscendC::GetBlockIdx() + (uint64_t)progress * this->tileLength;
        uint32_t offset = 0;
        while (offset < this->tileLength) {
            uint64_t gidx = tileStart + offset;
            uint64_t pos_in_dhw = gidx % this->DHW;
            uint32_t remain_in_row = this->DHW - static_cast<uint32_t>(pos_in_dhw);
            uint32_t run = AscendC::Min((uint32_t)(this->tileLength - offset), remain_in_row);

            // 计算该段的最大值
            DTYPE seg_max = -3.4e38f;
            for (uint32_t j = 0; j < run; ++j) {
                DTYPE v = zLocal.GetValue(offset + j);
                if (v > seg_max) seg_max = v;
            }
            // 计算exp和
            DTYPE seg_sum = 0.f;
            for (uint32_t j = 0; j < run; ++j) {
                DTYPE v = zLocal.GetValue(offset + j);
                DTYPE e = AscendC::Exp(v - seg_max);
                zLocal.SetValue(offset + j, e);
                seg_sum += e;
            }
            // 归一化
            DTYPE inv_sum = 1.f / seg_sum;
            for (uint32_t j = 0; j < run; ++j) {
                DTYPE e = zLocal.GetValue(offset + j);
                zLocal.SetValue(offset + j, e * inv_sum);
            }
            offset += run;
        }

        // 乘以每通道scale
        for (uint32_t i = 0; i < this->tileLength; ++i) {
            uint64_t gidx = tileStart + i;
            uint32_t cidx = static_cast<uint32_t>((gidx / this->DHW) % this->C);
            DTYPE s = scaleLocal.GetValue(cidx);
            DTYPE v = zLocal.GetValue(i);
            zLocal.SetValue(i, v * s);
        }

        outQueueZ.EnQue<DTYPE>(zLocal);
        inQueueX.FreeTensor(xLocal);
        // 将scale与clamp标量放回队列供下次迭代使用
        scaleQueue.EnQue(scaleLocal);
        minQueue.EnQue(minLocal);
        maxQueue.EnQue(maxLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.DeQue<DTYPE>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scaleQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> minQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> maxQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> zGm;
    AscendC::GlobalTensor<DTYPE> scaleGm;
    AscendC::GlobalTensor<DTYPE> clampMinGlobal, clampMaxGlobal;

    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t B, C, D, H, W;
    uint32_t DHW;
};

extern "C" __global__ __aicore__ void conv_transpose3d_avg_pool_clamp_softmax_multiply_custom(
    GM_ADDR x, GM_ADDR scale, GM_ADDR clamp_min, GM_ADDR clamp_max, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelClampSoftmaxMultiply op;
    op.Init(x, scale, clamp_min, clamp_max, z,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.batch, tiling_data.channels, tiling_data.depth, tiling_data.height, tiling_data.width);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    AvgPool3d + ConvTranspose3d 保持不变，后续的 clamp + softmax(空间维度) + 按通道scale 相乘
    使用自定义AscendC融合算子。
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, pool_kernel_size, clamp_min, clamp_max):
        super().__init__()
        self.avg_pool = torch.nn.AvgPool3d(pool_kernel_size)
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size,
                                                       stride=stride, padding=padding, output_padding=output_padding)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        self.scale = torch.nn.Parameter(torch.ones(1, out_channels, 1, 1, 1))

    def forward(self, x):
        x = self.avg_pool(x)
        x = self.conv_transpose(x)
        # 自定义算子执行：clamp -> softmax(在最后一维，即空间flatten后) -> multiply(scale)
        clamp_min_t = torch.tensor(self.clamp_min, dtype=x.dtype, device=x.device)
        clamp_max_t = torch.tensor(self.clamp_max, dtype=x.dtype, device=x.device)
        x = custom_ops_lib.conv_transpose3d_avg_pool_clamp_softmax_multiply_custom(x, self.scale, clamp_min_t, clamp_max_t)
        return x
