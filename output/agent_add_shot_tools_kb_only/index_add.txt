kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelIndexAdd {
public:
    __aicore__ inline KernelIndexAdd() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, GM_ADDR values, GM_ADDR z,
                                uint32_t rows, uint32_t cols, uint32_t n,
                                uint32_t tileRows, uint32_t copyTileRows)
    {
        this->rows = rows;
        this->cols = cols;
        this->n = n;
        this->tileRows = tileRows;
        this->copyTileRows = copyTileRows;

        xGm.SetGlobalBuffer((__gm__ float*)x, rows * cols);
        idxGm.SetGlobalBuffer((__gm__ uint32_t*)indices, n);
        valGm.SetGlobalBuffer((__gm__ float*)values, n * cols);
        zGm.SetGlobalBuffer((__gm__ float*)z, rows * cols);

        pipe.InitBuffer(copyInQueue, BUFFER_NUM, copyTileRows * cols * sizeof(float));
        pipe.InitBuffer(idxInQueue, BUFFER_NUM, tileRows * sizeof(uint32_t));
        pipe.InitBuffer(valOutQueue, BUFFER_NUM, tileRows * cols * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        CopyXtoZPerCore();
        IndexAddPerCore();
    }

private:
    __aicore__ inline void CopyXtoZPerCore()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t rowStart = (rows * blockIdx) / blockNum;
        uint32_t rowEnd = (rows * (blockIdx + 1)) / blockNum;

        for (uint32_t r = rowStart; r < rowEnd; r += copyTileRows) {
            uint32_t curRows = AscendC::min(copyTileRows, rowEnd - r);
            AscendC::LocalTensor<float> tmpLocal = copyInQueue.AllocTensor<float>();
            tmpLocal.SetSize(curRows * cols);
            AscendC::DataCopy(tmpLocal, xGm[r * cols], curRows * cols);
            AscendC::DataCopy(zGm[r * cols], tmpLocal, curRows * cols);
            copyInQueue.FreeTensor(tmpLocal);
        }
    }

    __aicore__ inline void IndexAddPerCore()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t nStart = (n * blockIdx) / blockNum;
        uint32_t nEnd = (n * (blockIdx + 1)) / blockNum;

        for (uint32_t p = nStart; p < nEnd; p += tileRows) {
            uint32_t curRows = AscendC::min(tileRows, nEnd - p);

            AscendC::LocalTensor<uint32_t> idxLocal = idxInQueue.AllocTensor<uint32_t>();
            idxLocal.SetSize(curRows);
            AscendC::DataCopy(idxLocal, idxGm[p], curRows);

            AscendC::LocalTensor<float> valLocal = valOutQueue.AllocTensor<float>();
            valLocal.SetSize(curRows * cols);
            AscendC::DataCopy(valLocal, valGm[p * cols], curRows * cols);

            AscendC::SetAtomicAdd<float>();
            for (uint32_t j = 0; j < curRows; ++j) {
                uint32_t rowIdx = idxLocal.GetValue(j);
                AscendC::LocalTensor<float> rowLocal = valLocal[j * cols];
                AscendC::DataCopy(zGm[rowIdx * cols], rowLocal, cols);
            }
            AscendC::SetAtomicNone();

            valOutQueue.FreeTensor(valLocal);
            idxInQueue.FreeTensor(idxLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> copyInQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> idxInQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> valOutQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<uint32_t> idxGm;
    AscendC::GlobalTensor<float> valGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t n;
    uint32_t tileRows;
    uint32_t copyTileRows;
};

extern "C" __global__ __aicore__ void index_add_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR values, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelIndexAdd op;
    op.Init(x, indices, values, z,
            tiling_data.rows, tiling_data.cols, tiling_data.n,
            tiling_data.tileRows, tiling_data.copyTileRows);
    op.Process();
}
