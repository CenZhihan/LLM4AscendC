static ge::graphStatus TilingFunc(gert::TilingContext* context)

    uint32_t m = static_cast<uint32_t>(A_origin.GetDim(1));
    uint32_t k = static_cast<uint32_t>(A_origin.GetDim(2));
    uint32_t n = static_cast<uint32_t>(B_origin.GetDim(2));

    tiling.set_batch(batch);

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;

    const gert::Shape* A_shape = context->GetInputShape(0);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelBMM {
public:
    __aicore__ inline KernelBMM() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C, uint32_t m, uint32_t n, uint32_t k, uint32_t batch) {
        this->m_ = m;
        this->n_ = n;
        this->k_ = k;
        this->batch_ = batch;
        A_ptr_ = reinterpret_cast<__gm__ float*>(A);
        B_ptr_ = reinterpret_cast<__gm__ float*>(B);
        C_ptr_ = reinterpret_cast<__gm__ float*>(C);
    }

    __aicore__ inline void Process() {
        // 朴素实现：C[b, i, j] = sum_k A[b, i, k] * B[b, k, j]
        for (uint32_t b = 0; b < batch_; ++b) {
            uint64_t baseA = static_cast<uint64_t>(b) * m_ * k_;
            uint64_t baseB = static_cast<uint64_t>(b) * k_ * n_;
            uint64_t baseC = static_cast<uint64_t>(b) * m_ * n_;
            for (uint32_t i = 0; i < m_; ++i) {
                for (uint32_t j = 0; j < n_; ++j) {
                    float acc = 0.0f;
                    for (uint32_t kk = 0; kk < k_; ++kk) {
                        float a_val = A_ptr_[baseA + static_cast<uint64_t>(i) * k_ + kk];
                        float b_val = B_ptr_[baseB + static_cast<uint64_t>(kk) * n_ + j];
                        acc += a_val * b_val;
                    }
                    C_ptr_[baseC + static_cast<uint64_t>(i) * n_ + j] = acc;
                }
            }
        }
    }

private:
    uint32_t m_;
    uint32_t n_;
    uint32_t k_;
    uint32_t batch_;
    __gm__ float* A_ptr_;
    __gm__ float* B_ptr_;
    __gm__ float* C_ptr_;
};

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBMM op;
    op.Init(A, B, C, tiling_data.m, tiling_data.n, tiling_data.k, tiling_data.batch);
    op.Process();
}
