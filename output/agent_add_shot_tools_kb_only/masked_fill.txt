#include "masked_fill_custom_tiling.h"
#include "register/op_def_registry.h"
#include "kernel_operator.h"
#include <limits>
#include <vector>

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaskedFillCustomTilingData tiling;

    auto originShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t dimNum = static_cast<uint32_t>(originShape.GetDimNum());
    uint32_t srcLastAxis = static_cast<uint32_t>(originShape.GetDim(dimNum - 1));
    uint64_t totalElems = static_cast<uint64_t>(originShape.GetShapeSize());
    uint32_t firstAxis = static_cast<uint32_t>(totalElems / srcLastAxis);

    // 设置 block dim
    context->SetBlockDim(BLOCK_DIM);

    // 简化实现：每个 tile 处理 1 行，保证 UB 缓冲可用且满足对齐要求
    uint32_t tileRows = 1;

    // 计算 SelectWithBytesMask 的临时缓冲大小（按 tile 形状）
    std::vector<int64_t> src0Vec = {static_cast<int64_t>(tileRows), static_cast<int64_t>(srcLastAxis)};
    std::vector<int64_t> src1Vec = {1}; // scalar
    std::vector<int64_t> maskVec = {static_cast<int64_t>(tileRows), static_cast<int64_t>(srcLastAxis)};
    ge::Shape src0Shape(src0Vec);
    ge::Shape src1Shape(src1Vec);
    ge::Shape maskShape(maskVec);
    uint32_t maxTmp = 0, minTmp = 0;
    AscendC::GetSelectWithBytesMaskMaxMinTmpSize(src0Shape, src1Shape, 4, maskShape, 1, true, maxTmp, minTmp);

    tiling.set_firstAxis(firstAxis);
    tiling.set_srcLastAxis(srcLastAxis);
    tiling.set_maskLastAxis(srcLastAxis);
    tiling.set_tileRows(tileRows);
    tiling.set_tmpSize(minTmp);
    tiling.set_scalarValue(-std::numeric_limits<float>::infinity());

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // 无需 workspace
    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* z_shape = context->GetOutputShape(0);
    *z_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // 简化：单缓冲 tile 流水

class KernelMaskedFill {
public:
    __aicore__ inline KernelMaskedFill() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR z,
                                uint32_t firstAxis, uint32_t srcLastAxis,
                                uint32_t maskLastAxis, uint32_t tileRows,
                                uint32_t tmpSize, float scalarValue)
    {
        this->firstAxis = firstAxis;
        this->srcLastAxis = srcLastAxis;
        this->maskLastAxis = maskLastAxis;
        this->tileRows = tileRows;
        this->tmpSize = tmpSize;
        this->scalar = static_cast<float>(scalarValue);

        // 每个 block 负责 firstAxis / blockNum 行
        this->rowsPerBlock = firstAxis / AscendC::GetBlockNum();
        this->elemPerRow = srcLastAxis;
        this->elemsPerTile = this->tileRows * this->elemPerRow;
        this->loopCount = this->rowsPerBlock / this->tileRows;

        uint32_t blockRowStart = this->rowsPerBlock * AscendC::GetBlockIdx();
        uint32_t blockElemStart = blockRowStart * this->elemPerRow;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x) + blockElemStart,
                            this->rowsPerBlock * this->elemPerRow);
        maskGm.SetGlobalBuffer(reinterpret_cast<__gm__ bool *>(mask) + blockElemStart,
                               this->rowsPerBlock * this->elemPerRow);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(z) + blockElemStart,
                            this->rowsPerBlock * this->elemPerRow);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->elemsPerTile * sizeof(float));
        pipe.InitBuffer(inQueueMask, BUFFER_NUM, this->elemsPerTile * sizeof(bool));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->elemsPerTile * sizeof(float));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tmpSize);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<bool> maskLocal = inQueueMask.AllocTensor<bool>();
        uint32_t offset = progress * elemsPerTile;
        AscendC::DataCopy(xLocal, xGm[offset], elemsPerTile);
        AscendC::DataCopy(maskLocal, maskGm[offset], elemsPerTile);
        inQueueX.EnQue(xLocal);
        inQueueMask.EnQue(maskLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progress*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<bool> maskLocal = inQueueMask.DeQue<bool>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tmpLocal = tmpQueue.AllocTensor<uint8_t>();

        AscendC::SelectWithBytesMaskShapeInfo info;
        info.firstAxis = tileRows;
        info.srcLastAxis = srcLastAxis;
        info.maskLastAxis = maskLastAxis;

        AscendC::SelectWithBytesMask(zLocal, xLocal, scalar, maskLocal, tmpLocal, info);

        outQueueZ.EnQue<float>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueMask.FreeTensor(maskLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        uint32_t offset = progress * elemsPerTile;
        AscendC::DataCopy(zGm[offset], zLocal, elemsPerTile);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueMask;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<bool> maskGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t firstAxis;
    uint32_t srcLastAxis;
    uint32_t maskLastAxis;
    uint32_t tileRows;
    uint32_t rowsPerBlock;
    uint32_t elemPerRow;
    uint32_t elemsPerTile;
    uint32_t loopCount;
    uint32_t tmpSize;
    float scalar;
};

extern "C" __global__ __aicore__ void masked_fill_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedFill op;
    op.Init(x, mask, z,
            tiling_data.firstAxis(),
            tiling_data.srcLastAxis(),
            tiling_data.maskLastAxis(),
            tiling_data.tileRows(),
            tiling_data.tmpSize(),
            tiling_data.scalarValue());
    op.Process();
}
