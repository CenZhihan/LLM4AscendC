    SqueezeNetCustomTilingData tiling;

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    return ge::GRAPH_SUCCESS;

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* y_shape = context->GetInputShape(1);
    gert::Shape* z_shape = context->GetOutputShape(0);

    // 默认拷贝x形状
    *z_shape = *x_shape;

    // 将C维(维度1)设为相加
    if (x_shape->GetDimNum() == y_shape->GetDimNum() && x_shape->GetDimNum() >= 2) {

        int64_t c0 = x_shape->GetDim(1);
        int64_t c1 = y_shape->GetDim(1);
        z_shape->SetDim(1, c0 + c1);

    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelCatC {
public:
    __aicore__ inline KernelCatC() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t xTotalLength, uint32_t yTotalLength, uint32_t tileNum)
    {
        this->xTotal = xTotalLength;
        this->yTotal = yTotalLength;
        this->tileNum = tileNum;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // 按block平均划分
        this->xBlockLen = this->xTotal / blockNum;
        this->yBlockLen = this->yTotal / blockNum;

        this->xBlockStart = this->xBlockLen * blockIdx;
        this->yBlockStart = this->yBlockLen * blockIdx;

        // 分块tile长度
        this->xTileLen = this->xBlockLen / this->tileNum / BUFFER_NUM;
        this->yTileLen = this->yBlockLen / this->tileNum / BUFFER_NUM;

        // GM视图
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->xBlockStart, this->xBlockLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->yBlockStart, this->yBlockLen);
        // z使用完整长度，写出时带偏移
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, this->xTotal + this->yTotal);

        uint32_t outTileMax = this->xTileLen > this->yTileLen ? this->xTileLen : this->yTileLen;

        // 队列buffer
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->xTileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->yTileLen * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, outTileMax * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // 复制X -> Z的前半段
        int32_t xLoopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < xLoopCount; i++) {
            CopyInX(i);
            ComputeX(i);
            CopyOutX(i);
        }
        // 复制Y -> Z的后半段（偏移xTotal）
        int32_t yLoopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < yLoopCount; i++) {
            CopyInY(i);
            ComputeY(i);
            CopyOutY(i);
        }
    }

private:
    __aicore__ inline void CopyInX(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->xTileLen], this->xTileLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeX(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        // 直接搬运
        AscendC::DataCopy(zLocal, xLocal, this->xTileLen);
        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutX(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        uint32_t zOffset = this->xBlockStart + progress * this->xTileLen;
        AscendC::DataCopy(zGm[zOffset], zLocal, this->xTileLen);
        outQueueZ.FreeTensor(zLocal);
    }

    __aicore__ inline void CopyInY(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(yLocal, yGm[progress * this->yTileLen], this->yTileLen);
        inQueueY.EnQue(yLocal);
    }

    __aicore__ inline void ComputeY(int32_t)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        // 直接搬运
        AscendC::DataCopy(zLocal, yLocal, this->yTileLen);
        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueY.FreeTensor(yLocal);
    }

    __aicore__ inline void CopyOutY(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        uint32_t zOffset = this->xTotal + this->yBlockStart + progress * this->yTileLen;
        AscendC::DataCopy(zGm[zOffset], zLocal, this->yTileLen);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t xTotal;
    uint32_t yTotal;
    uint32_t xBlockLen;
    uint32_t yBlockLen;
    uint32_t xBlockStart;
    uint32_t yBlockStart;
    uint32_t tileNum;
    uint32_t xTileLen;
    uint32_t yTileLen;
};

extern "C" __global__ __aicore__ void squeeze_net_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCatC op;
    op.Init(x, y, z, tiling_data.xTotalLength, tiling_data.yTotalLength, tiling_data.tileNum);
    op.Process();
}
"""

#include <torch/library.h>

at::Tensor squeeze_net_custom_impl_npu(const at::Tensor& x, const at::Tensor& y) {
    TORCH_CHECK(x.dim() == 4 && y.dim() == 4, "Inputs must be 4D (NCHW).");
    TORCH_CHECK(x.size(0) == y.size(0) && x.size(2) == y.size(2) && x.size(3) == y.size(3),

    auto sizes = x.sizes().vec();
    sizes[1] = x.size(1) + y.size(1); // C拼接
    at::Tensor result = at::empty(sizes, x.options());
    EXEC_NPU_CMD(aclnnSqueezeNetCustom, x.contiguous(), y.contiguous(), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("squeeze_net_custom", &squeeze_net_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("squeeze_net_custom", &squeeze_net_custom_impl_npu, "channel concat (x, y) -> z");

model_src='''
import torch
import torch_npu
import custom_ops_lib

class FireModule(torch.nn.Module):
    def __init__(self, in_channels, squeeze_channels, expand1x1_channels, expand3x3_channels):
        super(FireModule, self).__init__()
        self.squeeze = torch.nn.Conv2d(in_channels, squeeze_channels, kernel_size=1)
        self.squeeze_activation = torch.nn.ReLU(inplace=True)
        self.expand1x1 = torch.nn.Conv2d(squeeze_channels, expand1x1_channels, kernel_size=1)
        self.expand1x1_activation = torch.nn.ReLU(inplace=True)
        self.expand3x3 = torch.nn.Conv2d(squeeze_channels, expand3x3_channels, kernel_size=3, padding=1)
        self.expand3x3_activation = torch.nn.ReLU(inplace=True)
    
    def forward(self, x):
        x = self.squeeze_activation(self.squeeze(x))
        out1 = self.expand1x1_activation(self.expand1x1(x))
        out3 = self.expand3x3_activation(self.expand3x3(x))
        # 使用自定义AscendC核替换torch.cat（沿C维拼接）
        return custom_ops_lib.squeeze_net_custom(out1, out3)

class ModelNew(torch.nn.Module):
    def __init__(self, num_classes=1000):
        super(ModelNew, self).__init__()
        self.features = torch.nn.Sequential(
            torch.nn.Conv2d(3, 96, kernel_size=7, stride=2),
            torch.nn.ReLU(inplace=True),
            torch.nn.MaxPool2d(kernel_size=3, stride=2, ceil_mode=True),
            FireModule(96, 16, 64, 64),
            FireModule(128, 16, 64, 64),
            FireModule(128, 32, 128, 128),
            torch.nn.MaxPool2d(kernel_size=3, stride=2, ceil_mode=True),
            FireModule(256, 32, 128, 128),
            FireModule(256, 48, 192, 192),
            FireModule(384, 48, 192, 192),
            FireModule(384, 64, 256, 256),
            torch.nn.MaxPool2d(kernel_size=3, stride=2, ceil_mode=True),
            FireModule(512, 64, 256, 256),
        )
        self.classifier = torch.nn.Sequential(
            torch.nn.Dropout(p=0.0),
            torch.nn.Conv2d(512, num_classes, kernel_size=1),
            torch.nn.ReLU(inplace=True),
            torch.nn.AdaptiveAvgPool2d((1, 1))
        )
    
    def forward(self, x):
        x = self.features(x)
        x = self.classifier(x)
        return torch.flatten(x, 1)

def get_inputs():
    batch_size = 64
    input_channels = 3
    height = 512
    width = 512
    return [torch.rand(batch_size, input_channels, height, width)]

def get_init_inputs():
    num_classes = 1000
    return [num_classes]
'''
