kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConv2dNaive {
public:
    __aicore__ inline KernelConv2dNaive() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR z,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t inH, uint32_t inW,
                                uint32_t kH, uint32_t kW,
                                uint32_t outH, uint32_t outW,
                                uint64_t totalOut)
    {
        this->n = n;
        this->cin = cin;
        this->cout = cout;
        this->inH = inH;
        this->inW = inW;
        this->kH = kH;
        this->kW = kW;
        this->outH = outH;
        this->outW = outW;
        this->totalOut = totalOut;

        uint64_t xSize = (uint64_t)n * cin * inH * inW;
        uint64_t wSize = (uint64_t)cout * cin * kH * kW;
        uint64_t zSize = (uint64_t)n * cout * outH * outW;

        // Partition output by blocks
        this->blockOut = totalOut / AscendC::GetBlockNum();
        this->startOut = this->blockOut * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm.SetGlobalBuffer((__gm__ float*)w, wSize);
        zGm.SetGlobalBuffer((__gm__ float*)z, zSize);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, sizeof(float));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, sizeof(float));
        pipe.InitBuffer(tmpQueue , BUFFER_NUM, sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Each block computes its own contiguous output range
        uint64_t endOut = this->startOut + this->blockOut;
        for (uint64_t outIdx = this->startOut; outIdx < endOut; ++outIdx) {
            ComputeOne(outIdx);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint64_t outIndex)
    {
        if (outH == 0 || outW == 0) {
            return;
        }
        // Map linear output index to (n, co, oh, ow)
        uint64_t tmp = outIndex;
        uint32_t ow = tmp % outW; tmp /= outW;
        uint32_t oh = tmp % outH; tmp /= outH;
        uint32_t co = tmp % cout; tmp /= cout;
        uint32_t nIdx = tmp % n;

        // Allocate accumulator (1 element)
        AscendC::LocalTensor<float> accLocal = outQueueZ.AllocTensor<float>();
        bool accInitialized = false;

        // Accumulate over ci, kh, kw
        for (uint32_t ciIdx = 0; ciIdx < cin; ++ciIdx) {
            for (uint32_t khIdx = 0; khIdx < kH; ++khIdx) {
                for (uint32_t kwIdx = 0; kwIdx < kW; ++kwIdx) {
                    // Compute flattened indices
                    uint64_t xOffset = (((uint64_t)nIdx * cin + ciIdx) * inH + (oh + khIdx)) * inW + (ow + kwIdx);
                    uint64_t wOffset = ((((uint64_t)co * cin + ciIdx) * kH + khIdx) * kW + kwIdx);

                    // Load one input and one weight element
                    AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                    AscendC::LocalTensor<float> wLocal = inQueueW.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm[xOffset], 1);
                    AscendC::DataCopy(wLocal, wGm[wOffset], 1);

                    AscendC::LocalTensor<float> prodLocal = tmpQueue.AllocTensor<float>();
                    // prod = x * w (vector length 1)
                    AscendC::Mul(prodLocal, xLocal, wLocal, 1);

                    if (!accInitialized) {
                        // acc = prod
                        AscendC::DataCopy(accLocal, prodLocal, 1);
                        accInitialized = true;
                    } else {
                        // acc = acc + prod
                        AscendC::Add(accLocal, accLocal, prodLocal, 1);
                    }

                    // Free temporaries
                    inQueueX.FreeTensor(xLocal);
                    inQueueW.FreeTensor(wLocal);
                    tmpQueue.FreeTensor(prodLocal);
                }
            }
        }

        // Store result
        uint64_t zOffset = (((uint64_t)nIdx * cout + co) * outH + oh) * outW + ow;
        AscendC::DataCopy(zGm[zOffset], accLocal, 1);
        outQueueZ.FreeTensor(accLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN , BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN , BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t n, cin, cout, inH, inW, kH, kW, outH, outW;
    uint64_t totalOut;
    uint64_t blockOut;
    uint64_t startOut;
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv2dNaive op;
    op.Init(x, w, z,
            tilingData.n, tilingData.cin, tilingData.cout,
            tilingData.inH, tilingData.inW,
            tilingData.kH, tilingData.kW,
            tilingData.outH, tilingData.outW,
            tilingData.totalOut);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}
