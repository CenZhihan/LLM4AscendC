#include "cumsum_reverse_custom_tiling.h"

namespace {
static uint64_t Product(const std::vector<int64_t>& vec) {
    uint64_t p = 1;
    for (auto v : vec) { p *= static_cast<uint64_t>(v); }
    return p;
}
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CumsumReverseCustomTilingData tiling;

    // 仅支持沿最后一维做反向cumsum
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = 0;
    if (attrs != nullptr && attrs->GetAttrNum() >= 1) {
        dim = *(attrs->GetAttrPointer<int32_t>(0));
    }

    const gert::StorageShape *shape_ptr = context->GetInputShape(0);
    const auto &origin = shape_ptr->GetOriginShape();
    int64_t rank = origin.GetDimNum();
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }

    // 校正负轴
    int32_t axis = dim < 0 ? (int32_t)(rank + dim) : dim;
    // 仅支持 axis == 最后一维
    if (axis != (int32_t)rank - 1) {
        return ge::GRAPH_FAILED;
    }

    // 计算 outter 和 n
    std::vector<int64_t> dims;
    dims.reserve(rank);
    for (int i = 0; i < rank; ++i) dims.push_back(origin.GetDim(i));
    uint64_t n = static_cast<uint64_t>(dims.back());
    uint64_t total = Product(dims);
    if (n == 0) return ge::GRAPH_FAILED;
    uint64_t outter = total / n;

    // 类型字节数
    auto dt = context->GetInputTensor(0)->GetDataType();
    uint32_t typeSize = (dt == ge::DT_FLOAT16) ? 2 : 4;

    // inner 按 32B 对齐
    uint64_t innerAligned = ((n * typeSize + 32 - 1) / 32) * 32 / typeSize;

    tiling.set_outter((uint32_t)outter);
    tiling.set_n((uint32_t)n);
    tiling.set_inner((uint32_t)innerAligned);

    context->SetBlockDim(8);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using DType = DTYPE_X;

class KernelCumsumReverse {
public:
    __aicore__ inline KernelCumsumReverse() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outter, uint32_t inner, uint32_t n)
    {
        this->outter = outter;
        this->inner = inner;   // 32B 对齐后的宽
        this->n = n;           // 实际宽
        xGm.SetGlobalBuffer((__gm__ DType *)x, (uint64_t)outter * n);
        yGm.SetGlobalBuffer((__gm__ DType *)y, (uint64_t)outter * n);

        // 分配本地缓冲
        pipe.InitBuffer(inQueueX, 1, inner * sizeof(DType));     // 输入行
        pipe.InitBuffer(prefixQueue, 1, inner * sizeof(DType));  // 前缀和行
        pipe.InitBuffer(tmpQueue, 1, inner * sizeof(DType));     // 临时（broadcast Total）
        pipe.InitBuffer(outQueueY, 1, inner * sizeof(DType));    // 输出行
        pipe.InitBuffer(sumQueue, 1, AscendC::ONE_BLK_SIZE);     // 行和（1个元素，32B对齐）
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerCore = (outter + blockNum - 1) / blockNum;
        uint32_t rowStart = rowsPerCore * blockIdx;
        uint32_t rowEnd = rowStart + rowsPerCore;
        if (rowEnd > outter) rowEnd = outter;

        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            ComputeOneRow(r);
        }
    }

private:
    __aicore__ inline void ComputeOneRow(uint32_t rowIdx)
    {
        // 申请本地tensor
        AscendC::LocalTensor<DType> xLocal = inQueueX.AllocTensor<DType>();
        AscendC::LocalTensor<DType> prefixLocal = prefixQueue.AllocTensor<DType>();
        AscendC::LocalTensor<DType> tmpLocal = tmpQueue.AllocTensor<DType>();
        AscendC::LocalTensor<DType> yLocal = outQueueY.AllocTensor<DType>();
        AscendC::LocalTensor<DType> sumLocal = sumQueue.AllocTensor<DType>();

        // 读入一行（前 inner 已对齐，首 n 为真实数据）
        AscendC::DataCopy(xLocal, xGm[(uint64_t)rowIdx * n], n);
        // 若 inner > n，尾部内容无需关心（不回写）

        // 计算该行的和（Sum）
        AscendC::SumParams sumParams{1, inner, n};
        // sumLocal 大小按 1 个元素对齐至 32B
        AscendC::Sum<DType>(sumLocal, xLocal, sumParams);

        // 计算该行的前缀和（沿最后一维 CumSum）
        static constexpr AscendC::CumSumConfig cfg{true, false, false};
        const AscendC::CumSumInfo info{1, inner};
        AscendC::CumSum<DType, cfg>(prefixLocal, tmpLocal /*占位未使用*/, xLocal, info);

        // 将行和广播到整行：这里使用 Duplicate 将标量填充到 tmpLocal
        // 假设可从 sumLocal 读取第 0 个标量作为 totalSum
        DType totalSum = sumLocal.GetValue(0);
        AscendC::Duplicate<DType>(tmpLocal, totalSum, inner);

        // y = totalSum - prefix + x
        AscendC::Sub(yLocal, tmpLocal, prefixLocal, inner);
        AscendC::Add(yLocal, yLocal, xLocal, inner);

        // 写回前 n 个元素
        AscendC::DataCopy(yGm[(uint64_t)rowIdx * n], yLocal, n);

        // 释放本地tensor
        inQueueX.FreeTensor(xLocal);
        prefixQueue.FreeTensor(prefixLocal);
        tmpQueue.FreeTensor(tmpLocal);
        outQueueY.FreeTensor(yLocal);
        sumQueue.FreeTensor(sumLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> prefixQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> sumQueue;

    AscendC::GlobalTensor<DType> xGm;
    AscendC::GlobalTensor<DType> yGm;

    uint32_t outter;
    uint32_t inner;
    uint32_t n;
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverse op;
    op.Init(x, y, tiling_data.outter, tiling_data.inner, tiling_data.n);
    op.Process();
}
