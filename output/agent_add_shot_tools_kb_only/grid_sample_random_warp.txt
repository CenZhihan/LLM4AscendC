kernel_src="""
#include "kernel_operator.h"

class KernelGridSampleRandomWarp {
public:
    __aicore__ inline KernelGridSampleRandomWarp() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR grid, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W, uint32_t totalPixels)
    {
        this->N = N;
        this->C = C;
        this->H = H;
        this->W = W;
        this->totalPixels = totalPixels;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint32_t)(N * C * H * W));
        gridGm.SetGlobalBuffer((__gm__ float*)grid, (uint32_t)(N * H * W * 2));
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint32_t)(N * C * H * W));
    }

    __aicore__ inline void Process()
    {
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockId = AscendC::GetBlockIdx();

        uint32_t base = totalPixels / blocks;
        uint32_t rem = totalPixels % blocks;
        uint32_t start = blockId * base + (blockId < rem ? blockId : rem);
        uint32_t length = base + (blockId < rem ? 1 : 0);

        __gm__ float* x_ptr = xGm.GetPhyAddr();
        __gm__ float* grid_ptr = gridGm.GetPhyAddr();
        __gm__ float* y_ptr = yGm.GetPhyAddr();

        uint64_t HW = (uint64_t)H * (uint64_t)W;
        uint64_t CHW = (uint64_t)C * HW;

        for (uint32_t idx = 0; idx < length; ++idx) {
            uint32_t g = start + idx;
            uint32_t n = g / (H * W);
            uint32_t hw = g % (H * W);
            uint32_t h = hw / W;
            uint32_t w = hw % W;

            uint64_t grid_index = ((uint64_t)n * H * W + (uint64_t)h * W + (uint64_t)w) * 2ULL;
            float gx = grid_ptr[grid_index + 0];
            float gy = grid_ptr[grid_index + 1];

            // align_corners = false mapping
            float ix = (gx + 1.f) * (float)W * 0.5f - 0.5f;
            float iy = (gy + 1.f) * (float)H * 0.5f - 0.5f;

            int x0 = (int)AscendC::Floor(ix);
            int x1 = x0 + 1;
            int y0 = (int)AscendC::Floor(iy);
            int y1 = y0 + 1;

            float dx = ix - (float)x0;
            float dy = iy - (float)y0;

            // bounds mask for zero padding
            float bx0 = (x0 >= 0 && x0 < (int)W) ? 1.f : 0.f;
            float bx1 = (x1 >= 0 && x1 < (int)W) ? 1.f : 0.f;
            float by0 = (y0 >= 0 && y0 < (int)H) ? 1.f : 0.f;
            float by1 = (y1 >= 0 && y1 < (int)H) ? 1.f : 0.f;

            int rx0 = x0 < 0 ? 0 : (x0 >= (int)W ? (int)W - 1 : x0);
            int rx1 = x1 < 0 ? 0 : (x1 >= (int)W ? (int)W - 1 : x1);
            int ry0 = y0 < 0 ? 0 : (y0 >= (int)H ? (int)H - 1 : y0);
            int ry1 = y1 < 0 ? 0 : (y1 >= (int)H ? (int)H - 1 : y1);

            float w00 = (1.f - dx) * (1.f - dy) * bx0 * by0;
            float w01 = (dx)         * (1.f - dy) * bx1 * by0;
            float w10 = (1.f - dx) * (dy)         * bx0 * by1;
            float w11 = (dx)         * (dy)         * bx1 * by1;

            uint64_t row0_off = (uint64_t)ry0 * W;
            uint64_t row1_off = (uint64_t)ry1 * W;

            for (uint32_t c = 0; c < C; ++c) {
                uint64_t base_off = (uint64_t)n * CHW + (uint64_t)c * HW;
                float v00 = x_ptr[base_off + row0_off + (uint64_t)rx0];
                float v01 = x_ptr[base_off + row0_off + (uint64_t)rx1];
                float v10 = x_ptr[base_off + row1_off + (uint64_t)rx0];
                float v11 = x_ptr[base_off + row1_off + (uint64_t)rx1];

                float outv = v00 * w00 + v01 * w01 + v10 * w10 + v11 * w11;

                y_ptr[base_off + (uint64_t)h * W + (uint64_t)w] = outv;
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> gridGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t N;
    uint32_t C;
    uint32_t H;
    uint32_t W;
    uint32_t totalPixels;
};

extern "C" __global__ __aicore__ void grid_sample_random_warp_custom(GM_ADDR x, GM_ADDR grid, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGridSampleRandomWarp op;
    op.Init(x, grid, y, tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W, tiling_data.totalPixels);
    op.Process();
}
