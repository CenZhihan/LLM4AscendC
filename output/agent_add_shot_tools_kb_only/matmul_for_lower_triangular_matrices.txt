kernel_src="""
#include "kernel_operator.h"

struct TriTiling {
    uint32_t N;
};

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    // 读取tiling数据
    TriTiling tri;
    {
        auto src = reinterpret_cast<__gm__ uint8_t*>(tiling);
        auto dst = reinterpret_cast<uint8_t*>(&tri);
        for (uint32_t i = 0; i < sizeof(TriTiling); ++i) {
            dst[i] = src[i];
        }
    }

    const uint32_t N = tri.N;
    __gm__ float* Agm = reinterpret_cast<__gm__ float*>(A);
    __gm__ float* Bgm = reinterpret_cast__gm__ (reinterpret_cast<float*>(B)); // avoid warning

    // 修正上一行的写法（编译时等价）
    Bgm = reinterpret_cast<__gm__ float*>(B);
    __gm__ float* Cgm = reinterpret_cast<__gm__ float*>(C);

    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockId  = AscendC::GetBlockIdx();

    // 将行按block划分，简化负载均衡
    uint32_t rowsPerBlock = (N + blockNum - 1) / blockNum;
    uint32_t rowStart = blockId * rowsPerBlock;
    uint32_t rowEnd = rowStart + rowsPerBlock;
    if (rowEnd > N) rowEnd = N;

    // 对每个分配到的行进行计算：仅计算下三角部分（j <= i），上三角直接置0
    for (uint32_t i = rowStart; i < rowEnd; ++i) {
        // 上三角清零
        for (uint32_t j = i + 1; j < N; ++j) {
            Cgm[i * N + j] = 0.0f;
        }
        // 下三角与对角线：C[i,j] = sum_{k=j..i} A[i,k] * B[k,j] （A、B均为下三角）
        for (uint32_t j = 0; j <= i; ++j) {
            float acc = 0.0f;
            // 利用下三角性质缩小k范围
            for (uint32_t k = j; k <= i; ++k) {
                acc += Agm[i * N + k] * Bgm[k * N + j];
            }
            Cgm[i * N + j] = acc;
        }
    }
}
