#include "matmul_for_symmetric_matrices_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    // 仅处理方阵场景 (N, N)
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    uint64_t a_m = a_shape->GetDim(0);
    uint64_t a_k = a_shape->GetDim(1);
    uint64_t b_k = b_shape->GetDim(0);
    uint64_t b_n = b_shape->GetDim(1);
    if (a_k != b_k || a_m != b_n || a_m != a_k) {
        // 简化：N×N 与 N×N
        return ge::GRAPH_FAILED;
    }

    // 设置 blockDim，简化为根据平台核心数选择
    context->SetBlockDim(8);

    MatmulForSymmetricMatricesCustomTilingData tiling;
    tiling.set_m(static_cast<uint32_t>(a_m));
    tiling.set_n(static_cast<uint32_t>(b_n));
    tiling.set_k(static_cast<uint32_t>(a_k));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    uint64_t m = a_shape->GetDim(0);
    uint64_t k1 = a_shape->GetDim(1);
    uint64_t k2 = b_shape->GetDim(0);
    uint64_t n = b_shape->GetDim(1);
    if (k1 != k2) {
        return GRAPH_FAILED;
    }
    gert::Shape* c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, m);
    c_shape->SetDim(1, n);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    auto dt_a = context->GetInputDataType(0);
    auto dt_b = context->GetInputDataType(1);
    if (dt_a != dt_b) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, dt_a);
    return GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using aType = matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
using bType = matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
using cType = matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
using biasType = cType;
using MatmulOp = matmul::Matmul<aType, bType, cType, biasType>;

class KernelMatmulSym {
public:
    __aicore__ inline KernelMatmulSym() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t m, uint32_t n, uint32_t k)
    {
        this->m = m;
        this->n = n;
        this->k = k;
        aGm = reinterpret_cast<__gm__ float*>(a);
        bGm = reinterpret_cast<__gm__ float*>(b);
        cGm = reinterpret_cast<__gm__ float*>(c);
    }

    __aicore__ inline void Process()
    {
        // 使用高阶 Matmul API 实现 C = A x B
        MatmulOp mm;
        // 步骤：初始化 -> 设置 A/B/C -> 运行 -> 结束
        mm.Init();
        mm.SetTensorA(aGm, m, k, false);
        mm.SetTensorB(bGm, k, n, false);
        mm.SetTensorC(cGm, m, n);
        mm.Run();
        mm.End();
    }

private:
    __gm__ float* aGm;
    __gm__ float* bGm;
    __gm__ float* cGm;
    uint32_t m, n, k;
};

extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSym op;
    op.Init(a, b, c, tiling_data.m, tiling_data.n, tiling_data.k);
    op.Process();
}
