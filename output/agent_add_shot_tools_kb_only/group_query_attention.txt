kernel_src="""
#include "kernel_operator.h"
#include <cmath>

struct GQATiling {
    uint32_t batch;
    uint32_t seqlen;
    uint32_t num_heads;
    uint32_t head_dim;
};

class KernelGroupQueryAttention {
public:
    __aicore__ inline KernelGroupQueryAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR scale, GM_ADDR y, const GQATiling &t)
    {
        qGm = reinterpret_cast<__gm__ float*>(q);
        kGm = reinterpret_cast<__gm__ float*>(k);
        vGm = reinterpret_cast<__gm__ float*>(v);
        yGm = reinterpret_cast<__gm__ float*>(y);
        scaleGm = reinterpret_cast<__gm__ float*>(scale);
        tiling = t;
        headsPerBatch = tiling.num_heads;
        totalHeads = tiling.batch * tiling.num_heads;
        strideRow = tiling.num_heads * tiling.head_dim; // elements between adjacent rows for a fixed head
        strideHeadBase = tiling.head_dim;              // contiguous over head_dim
        scaleVal = *scaleGm;
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        for (uint32_t headIdx = blockIdx; headIdx < totalHeads; headIdx += blockNum) {
            uint32_t b = headIdx / headsPerBatch;
            uint32_t h = headIdx % headsPerBatch;
            ComputeOneHead(b, h);
        }
    }

private:
    __aicore__ inline void LoadRow(__gm__ float* basePtr, uint32_t rowIndex, float* dst)
    {
        // basePtr points to tensor start [B, L, H, Dh]; rowIndex is sequence index l; head fixed
        __gm__ float* rowPtr = basePtr + rowIndex * strideRow + currentHeadOffset;
        for (uint32_t d = 0; d < tiling.head_dim; ++d) {
            dst[d] = rowPtr[d];
        }
    }

    __aicore__ inline void StoreRow(__gm__ float* basePtr, uint32_t rowIndex, const float* src)
    {
        __gm__ float* rowPtr = basePtr + rowIndex * strideRow + currentHeadOffset;
        for (uint32_t d = 0; d < tiling.head_dim; ++d) {
            rowPtr[d] = src[d];
        }
    }

    __aicore__ inline float DotScaled(const float* qa, uint32_t keyRowIdx)
    {
        float acc = 0.f;
        __gm__ float* kRowPtr = kGmBase + keyRowIdx * strideRow;
        for (uint32_t d = 0; d < tiling.head_dim; ++d) {
            acc += qa[d] * kRowPtr[d];
        }
        return acc * scaleVal;
    }

    __aicore__ inline void AccumulateOut(float weight, uint32_t keyRowIdx, float* outAcc)
    {
        __gm__ float* vRowPtr = vGmBase + keyRowIdx * strideRow;
        for (uint32_t d = 0; d < tiling.head_dim; ++d) {
            outAcc[d] += weight * vRowPtr[d];
        }
    }

    __aicore__ inline void ComputeOneHead(uint32_t b, uint32_t h)
    {
        // compute base offsets for the given (b,h)
        // offset (b, l=0, h, d=0)
        uint64_t headBaseIndex = (static_cast<uint64_t>(b) * tiling.seqlen * tiling.num_heads + h) * tiling.head_dim;
        currentHeadOffset = h * tiling.head_dim; // within a row
        qGmBase = qGm + headBaseIndex - h * tiling.head_dim; // set row 0 base at (b,0,0,0), we'll add l*strideRow + h*Dh
        kGmBase = kGm + headBaseIndex - h * tiling.head_dim;
        vGmBase = vGm + headBaseIndex - h * tiling.head_dim;
        yGmBase = yGm + headBaseIndex - h * tiling.head_dim;

        // local buffers with conservative upper bounds
        const uint32_t MAX_DH = 256;
        __ubuf__ float qVec[MAX_DH];
        __ubuf__ float outVec[MAX_DH];

        uint32_t Dh = tiling.head_dim;
        // process each query row
        for (uint32_t l = 0; l < tiling.seqlen; ++l) {
            // load q[l]
            LoadRow(qGm, l, qVec);

            // first pass: find max score over all keys
            float maxScore = -1e30f;
            for (uint32_t m = 0; m < tiling.seqlen; ++m) {
                float s = 0.f;
                __gm__ float* kRowPtr = kGmBase + m * strideRow;
                for (uint32_t d = 0; d < Dh; ++d) {
                    s += qVec[d] * kRowPtr[d];
                }
                s = s * scaleVal;
                if (s > maxScore) maxScore = s;
            }

            // second pass: accumulate numerator and denominator
            for (uint32_t d = 0; d < Dh; ++d) {
                outVec[d] = 0.f;
            }
            float denom = 0.f;

            for (uint32_t m = 0; m < tiling.seqlen; ++m) {
                float s = 0.f;
                __gm__ float* kRowPtr2 = kGmBase + m * strideRow;
                for (uint32_t d = 0; d < Dh; ++d) {
                    s += qVec[d] * kRowPtr2[d];
                }
                s = s * scaleVal;
                float w = expf(s - maxScore);
                denom += w;

                __gm__ float* vRowPtr = vGmBase + m * strideRow;
                for (uint32_t d = 0; d < Dh; ++d) {
                    outVec[d] += w * vRowPtr[d];
                }
            }

            float invDen = 1.0f / denom;
            for (uint32_t d = 0; d < Dh; ++d) {
                outVec[d] *= invDen;
            }

            // store y[l]
            StoreRow(yGm, l, outVec);
        }
    }

private:
    __gm__ float* qGm {nullptr};
    __gm__ float* kGm {nullptr};
    __gm__ float* vGm {nullptr};
    __gm__ float* yGm {nullptr};
    __gm__ float* scaleGm {nullptr};

    __gm__ float* qGmBase {nullptr};
    __gm__ float* kGmBase {nullptr};
    __gm__ float* vGmBase {nullptr};
    __gm__ float* yGmBase {nullptr};

    GQATiling tiling;
    uint32_t headsPerBatch {0};
    uint32_t totalHeads {0};
    uint32_t strideRow {0};       // H*Dh
    uint32_t strideHeadBase {0};  // Dh
    uint32_t currentHeadOffset {0};
    float scaleVal {1.f};
};

extern "C" __global__ __aicore__ void group_query_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGM) {
    GET_TILING_DATA(tiling_data, tilingGM);
    GQATiling t;
    t.batch = tiling_data.batch();
    t.seqlen = tiling_data.seqlen();
    t.num_heads = tiling_data.num_heads();
    t.head_dim = tiling_data.head_dim();

    KernelGroupQueryAttention op;
    op.Init(q, k, v, scale, y, t);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import math
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, d_model=1024, num_heads=16, num_kv_heads=4):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be divisible by num_kv_heads"
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.group_size = num_heads // num_kv_heads

        self.q_proj = nn.Linear(d_model, d_model)
        self.k_proj = nn.Linear(d_model, d_model * num_kv_heads // num_heads)
        self.v_proj = nn.Linear(d_model, d_model * num_kv_heads // num_heads)
        self.out_proj = nn.Linear(d_model, d_model)

    def forward(self, x):
        B, L, D = x.shape
        H, H_kv = self.num_heads, self.num_kv_heads
        head_dim = D // H

        Q = self.q_proj(x).view(B, L, H, head_dim).contiguous()
        K = self.k_proj(x).view(B, L, H_kv, head_dim).contiguous()
        V = self.v_proj(x).view(B, L, H_kv, head_dim).contiguous()

        if self.group_size > 1:
            K = K.repeat_interleave(self.group_size, dim=2).contiguous()
            V = V.repeat_interleave(self.group_size, dim=2).contiguous()

        scale = torch.tensor([1.0 / math.sqrt(head_dim)], dtype=Q.dtype, device=Q.device)
        # fused attention: [B, L, H, Dh] = f(Q, K, V, scale)
        Y = custom_ops_lib.group_query_attention_custom(Q, K, V, scale)
        Y = Y.reshape(B, L, D)
        return self.out_proj(Y)
'''
