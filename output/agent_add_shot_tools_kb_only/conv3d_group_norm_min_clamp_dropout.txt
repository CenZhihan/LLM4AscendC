kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConv3dGroupNormMinClampDropout {
public:
    __aicore__ inline KernelConv3dGroupNormMinClampDropout() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, float minValue, float maxValue)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->minValue = minValue;
        this->maxValue = maxValue;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // 构造min/max常量向量
        AscendC::LocalTensor<DTYPE_X> minLocal = tmpQueue.AllocTensor<DTYPE_X>();
        AscendC::Duplicate(minLocal, (DTYPE_X)minValue, this->tileLength);

        // y = min(x, min_value)
        AscendC::Min(yLocal, xLocal, minLocal, this->tileLength);

        // clamp到[max_value]上界
        AscendC::LocalTensor<DTYPE_X> maxLocal = tmpQueue.AllocTensor<DTYPE_X>();
        AscendC::Duplicate(maxLocal, (DTYPE_X)maxValue, this->tileLength);
        AscendC::Min(yLocal, yLocal, maxLocal, this->tileLength);

        // clamp到[min_value]下界
        AscendC::Max(yLocal, yLocal, minLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        tmpQueue.FreeTensor(minLocal);
        tmpQueue.FreeTensor(maxLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    float minValue;
    float maxValue;
};

extern "C" __global__ __aicore__ void conv3d_group_norm_min_clamp_dropout_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dGroupNormMinClampDropout op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.minValue, tiling_data.maxValue);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    以PyTorch完成Conv3d与GroupNorm，随后用自定义AscendC算子融合执行min与clamp，再进行Dropout。
    """
    def __init__(self, in_channels, out_channels, kernel_size, groups, min_value, max_value, dropout_p):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.norm = torch.nn.GroupNorm(groups, out_channels)
        self.dropout = torch.nn.Dropout(dropout_p)

    def forward(self, x):
        x = self.conv(x)
        x = self.norm(x)
        # 自定义AscendC核：conv3d_group_norm_min_clamp_dropout_custom，仅对min+clamp阶段加速
        x = custom_ops_lib.conv3d_group_norm_min_clamp_dropout_custom(x)
        x = self.dropout(x)
        return x
