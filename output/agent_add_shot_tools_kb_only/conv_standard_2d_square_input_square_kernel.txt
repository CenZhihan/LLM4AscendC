#include "conv_standard_2d_square_input_square_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
constexpr int32_t KERNEL_H = 11;
constexpr int32_t KERNEL_W = 11;
constexpr int32_t STRIDE   = 4;
constexpr int32_t PADDING  = 2;
}

namespace optiling {
const uint32_t BLOCK_DIM = 32;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // x: [N,C,H,W], w: [OC,C,KH,KW], b: [OC] (optional)
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    uint32_t N  = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t C  = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t H  = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t W  = static_cast<uint32_t>(x_shape->GetDim(3));

    uint32_t OC = static_cast<uint32_t>(w_shape->GetDim(0));
    uint32_t KH = static_cast<uint32_t>(w_shape->GetDim(2));
    uint32_t KW = static_cast<uint32_t>(w_shape->GetDim(3));

    if (KH != KERNEL_H || KW != KERNEL_W) {
        return ge::GRAPH_FAILED;
    }

    uint32_t OH = static_cast<uint32_t>((static_cast<int32_t>(H) + 2 * PADDING - KERNEL_H) / STRIDE + 1);
    uint32_t OW = static_cast<uint32_t>((static_cast<int32_t>(W) + 2 * PADDING - KERNEL_W) / STRIDE + 1);

    optiling::ConvStandard2dSquareInputSquareKernelCustomTilingData tiling;
    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_oc(OC);
    tiling.set_oh(OH);
    tiling.set_ow(OW);

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return GRAPH_FAILED;
    }

    constexpr int32_t STRIDE = 4;
    constexpr int32_t PAD = 2;
    constexpr int32_t KH = 11;
    constexpr int32_t KW = 11;

    int64_t N = x_shape->GetDim(0);
    int64_t C = x_shape->GetDim(1);
    int64_t H = x_shape->GetDim(2);
    int64_t W = x_shape->GetDim(3);

    (void)C; // unused directly here
    int64_t OC = w_shape->GetDim(0);

    int64_t OH = (H + 2 * PAD - KH) / STRIDE + 1;
    int64_t OW = (W + 2 * PAD - KW) / STRIDE + 1;

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, OC);
    y_shape->SetDim(2, OH);
    y_shape->SetDim(3, OW);
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

namespace {
constexpr int32_t KERNEL_H = 11;
constexpr int32_t KERNEL_W = 11;
constexpr int32_t STRIDE   = 4;
constexpr int32_t PADDING  = 2;
}

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N  = tiling_data.n;
    const uint32_t C  = tiling_data.c;
    const uint32_t H  = tiling_data.h;
    const uint32_t W  = tiling_data.w;
    const uint32_t OC = tiling_data.oc;
    const uint32_t OH = tiling_data.oh;
    const uint32_t OW = tiling_data.ow;

    __gm__ float* x_gm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_gm = reinterpret_cast<__gm__ float*>(w);
    __gm__ float* b_gm = reinterpret_cast<__gm__ float*>(b);
    __gm__ float* y_gm = reinterpret_cast<__gm__ float*>(y);

    const uint64_t totalOut = static_cast<uint64_t>(N) * OC * OH * OW;
    const uint64_t blockNum = AscendC::GetBlockNum();
    const uint64_t blockIdx = AscendC::GetBlockIdx();

    const uint64_t base = (totalOut / blockNum) * blockIdx;
    const uint64_t tail = (blockIdx == (blockNum - 1)) ? totalOut : (base + totalOut / blockNum);

    for (uint64_t linearIdx = base; linearIdx < tail; ++linearIdx) {
        uint64_t t = linearIdx;
        const uint32_t ow = static_cast<uint32_t>(t % OW); t /= OW;
        const uint32_t oh = static_cast<uint32_t>(t % OH); t /= OH;
        const uint32_t oc = static_cast<uint32_t>(t % OC); t /= OC;
        const uint32_t n  = static_cast<uint32_t>(t);

        float acc = 0.0f;
        if (b_gm != nullptr) {
            acc = b_gm[oc];
        }

        const int32_t in_h_base = static_cast<int32_t>(oh) * STRIDE - PADDING;
        const int32_t in_w_base = static_cast<int32_t>(ow) * STRIDE - PADDING;

        for (uint32_t c0 = 0; c0 < C; ++c0) {
            for (int32_t kh = 0; kh < KERNEL_H; ++kh) {
                const int32_t ih = in_h_base + kh;
                if (ih < 0 || ih >= static_cast<int32_t>(H)) {
                    continue;
                }
                for (int32_t kw = 0; kw < KERNEL_W; ++kw) {
                    const int32_t iw = in_w_base + kw;
                    if (iw < 0 || iw >= static_cast<int32_t>(W)) {
                        continue;
                    }
                    const uint64_t x_idx = (((static_cast<uint64_t>(n) * C + c0) * H + static_cast<uint32_t>(ih)) * W + static_cast<uint32_t>(iw));
                    const uint64_t w_idx = (((static_cast<uint64_t>(oc) * C + c0) * KERNEL_H + kh) * KERNEL_W + kw);
                    acc += x_gm[x_idx] * w_gm[w_idx];
                }
            }
        }

        const uint64_t y_idx = (((static_cast<uint64_t>(n) * OC + oc) * OH + oh) * OW + ow);
        y_gm[y_idx] = acc;
    }
}

#include <torch/extension.h>

static at::Tensor to_contiguous_if_needed(const at::Tensor& t) {
    return t.is_contiguous() ? t : t.contiguous();
}

at::Tensor conv_standard_2d_square_input_square_kernel_custom_impl_npu(const at::Tensor& x,
                                                                       const at::Tensor& w,
                                                                       c10::optional<at::Tensor> b_opt) {
    at::Tensor x_c = to_contiguous_if_needed(x);
    at::Tensor w_c = to_contiguous_if_needed(w);
    at::Tensor y = at::empty({x.size(0), w.size(0),
                              (x.size(2) + 2 * 2 - 11) / 4 + 1,
                              (x.size(3) + 2 * 2 - 11) / 4 + 1},
                              x.options());
    if (b_opt.has_value() && b_opt.value().defined()) {
        at::Tensor b_c = to_contiguous_if_needed(b_opt.value());
        EXEC_NPU_CMD(aclnnConvStandard2dSquareInputSquareKernelCustom, x_c, w_c, b_c, y);
    } else {
        // Pass an undefined tensor for optional bias
        EXEC_NPU_CMD(aclnnConvStandard2dSquareInputSquareKernelCustom, x_c, w_c, at::Tensor(), y);
    }
    return y;
}
