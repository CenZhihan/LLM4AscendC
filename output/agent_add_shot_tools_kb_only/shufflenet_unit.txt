kernel_src="""
#include "kernel_operator.h"

// 将通道打散并重排（Channel Shuffle）
// 公式：Cg = C / G
// out[:, c_out, :, :] = in[:, c_in, :, :]
// 其中 c_in = (c_out % G) * Cg + (c_out / G)

constexpr int32_t BUFFER_NUM = 2;

typedef float DTYPE_X;
typedef float DTYPE_Y;

class KernelShuffleUnit {
public:
    __aicore__ inline KernelShuffleUnit() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t groups, uint32_t tileSize)
    {
        this->nLength = n;
        this->cLength = c;
        this->hLength = h;
        this->wLength = w;
        this->groups = groups > 0 ? groups : 1;
        this->sLength = h * w;
        this->cgLength = c / this->groups;

        // 每次从同一通道块中搬运的空间范围
        this->tileLength = tileSize;
        if (this->tileLength == 0) this->tileLength = 1024;

        // 分块到不同的 block 上（按通道维拆分）
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t basePerBlock = cLength / blockNum;
        this->startC = basePerBlock * blockIdx;
        this->endC = (blockIdx == blockNum - 1) ? cLength : (this->startC + basePerBlock);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, nLength * cLength * sLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, nLength * cLength * sLength);

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // 遍历 batch 与分配到当前 block 的输出通道
        for (uint32_t n = 0; n < nLength; ++n) {
            for (uint32_t co = startC; co < endC; ++co) {
                uint32_t ci = (co % groups) * cgLength + (co / groups);
                uint64_t baseOut = static_cast<uint64_t>(n) * cLength * sLength + static_cast<uint64_t>(co) * sLength;
                uint64_t baseIn  = static_cast<uint64_t>(n) * cLength * sLength + static_cast<uint64_t>(ci) * sLength;

                uint32_t pos = 0;
                while (pos < sLength) {
                    uint32_t curr = AscendC::min(this->tileLength, sLength - pos);
                    CopyIn(baseIn + pos, curr);
                    Compute(curr); // 本算子为纯通道重排，计算阶段为空操作
                    CopyOut(baseOut + pos, curr);
                    pos += curr;
                }
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t inOffset, uint32_t currLen)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[inOffset], currLen);
        inQueue.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t currLen)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.AllocTensor<DTYPE_Y>();
        // 直接搬运（数据不变），仅通道维的块之间做重排，单块内保持顺序
        AscendC::Copy(yLocal, xLocal, currLen);
        outQueue.EnQue<DTYPE_Y>(yLocal);
        inQueue.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint64_t outOffset, uint32_t currLen)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[outOffset], yLocal, currLen);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t nLength;
    uint32_t cLength;
    uint32_t hLength;
    uint32_t wLength;
    uint32_t sLength;   // H * W
    uint32_t groups;
    uint32_t cgLength;  // C / groups
    uint32_t tileLength;

    uint32_t startC;
    uint32_t endC;
};

extern "C" __global__ __aicore__ void shufflenet_unit_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelShuffleUnit op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.groups, tiling_data.tileSize);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, groups=3):
        super(ModelNew, self).__init__()
        assert out_channels % 4 == 0
        mid_channels = out_channels // 4

        self.groups = groups

        # First 1x1 group convolution
        self.conv1 = nn.Conv2d(in_channels, mid_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn1 = nn.BatchNorm2d(mid_channels)
        
        # Depthwise 3x3 convolution
        self.conv2 = nn.Conv2d(mid_channels, mid_channels, kernel_size=3, stride=1, padding=1, groups=mid_channels, bias=False)
        self.bn2 = nn.BatchNorm2d(mid_channels)
        
        # Second 1x1 group convolution
        self.conv3 = nn.Conv2d(mid_channels, out_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn3 = nn.BatchNorm2d(out_channels)
        
        # Shortcut connection if input and output channels are the same
        if in_channels == out_channels:
            self.shortcut = nn.Sequential()
        else:
            self.shortcut = nn.Sequential(
                nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=False),
                nn.BatchNorm2d(out_channels)
            )

    def forward(self, x):
        out = F.relu(self.bn1(self.conv1(x)))
        out = self.bn2(self.conv2(out))
        # 使用自定义 AscendC 算子做 Channel Shuffle
        out = custom_ops_lib.shufflenet_unit_custom(out, self.groups)
        out = F.relu(self.bn3(self.conv3(out)))
        out = out + self.shortcut(x)
        return out
'''
