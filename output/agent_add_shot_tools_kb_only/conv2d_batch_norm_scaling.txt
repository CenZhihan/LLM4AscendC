kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1;

class KernelBnScale {
public:
    __aicore__ inline KernelBnScale() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR a, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->hw = h * w;
        this->nc = n * c;
        xGm.SetGlobalBuffer((__gm__ DTYPE *)x, static_cast<uint32_t>(n) * c * h * w);
        aGm.SetGlobalBuffer((__gm__ DTYPE *)a, c);
        bGm.SetGlobalBuffer((__gm__ DTYPE *)b, c);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y, static_cast<uint32_t>(n) * c * h * w);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->hw * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->hw * sizeof(DTYPE));
        pipe.InitBuffer(inQueueScalar, BUFFER_NUM, sizeof(DTYPE)); // for a[c], b[c]
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t segPerBlock = (nc + blockNum - 1) / blockNum;
        uint32_t segStart = blockIdx * segPerBlock;
        uint32_t segEnd = segStart + segPerBlock;
        if (segEnd > nc) segEnd = nc;

        for (uint32_t seg = segStart; seg < segEnd; ++seg) {
            uint32_t nIdx = seg / c;
            uint32_t cIdx = seg % c;
            uint32_t base = (nIdx * c + cIdx) * hw;

            // load x[nIdx, cIdx, :, :]
            AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
            AscendC::DataCopy(xLocal, xGm[base], hw);

            // load a[cIdx]
            AscendC::LocalTensor<DTYPE> aLocal = inQueueScalar.AllocTensor<DTYPE>();
            AscendC::DataCopy(aLocal, aGm[cIdx], 1);

            // load b[cIdx]
            AscendC::LocalTensor<DTYPE> bLocal = inQueueScalar.AllocTensor<DTYPE>();
            AscendC::DataCopy(bLocal, bGm[cIdx], 1);

            // compute y = a[c] * x + b[c]
            AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
            DTYPE aVal = aLocal.GetValue(0);
            DTYPE bVal = bLocal.GetValue(0);
            AscendC::Muls(yLocal, xLocal, aVal, hw);
            AscendC::Adds(yLocal, yLocal, bVal, hw);

            inQueueX.FreeTensor(xLocal);
            inQueueScalar.FreeTensor(aLocal);
            inQueueScalar.FreeTensor(bLocal);
            outQueueY.EnQue(yLocal);

            AscendC::LocalTensor<DTYPE> yOutLocal = outQueueY.DeQue<DTYPE>();
            AscendC::DataCopy(yGm[base], yOutLocal, hw);
            outQueueY.FreeTensor(yOutLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueScalar;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> aGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> yGm;
    uint32_t n;
    uint32_t c;
    uint32_t h;
    uint32_t w;
    uint32_t hw;
    uint32_t nc;
};

extern "C" __global__ __aicore__ void conv2d_batch_norm_scaling_custom(GM_ADDR x, GM_ADDR a, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBnScale op;
    op.Init(x, a, b, y, tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    将 BatchNorm 和缩放融合为自定义 AscendC 内核：先做 Conv2d，再用 y = a*x + b 完成 BN 与缩放。
    其中 a = scaling_factor * gamma * rsqrt(var + eps)
         b = scaling_factor * beta - a * mean
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = torch.nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        # 先使用原生 Conv2d
        x = self.conv(x)
        # 从 BN 模块中读取参数与统计量（按推理公式融合）
        gamma = self.bn.weight
        beta = self.bn.bias
        running_mean = self.bn.running_mean
        running_var = self.bn.running_var
        eps = self.bn.eps

        # 预计算 a, b，避免在设备侧做复杂开方与广播
        rstd = torch.rsqrt(running_var + eps)
        a = self.scaling_factor * gamma * rstd
        b = self.scaling_factor * beta - a * running_mean

        # 调用自定义 NPU 内核：对每个输出通道执行 y = a[c] * x + b[c]
        y = custom_ops_lib.conv2d_batch_norm_scaling_custom(x, a, b)
        return y
'''
