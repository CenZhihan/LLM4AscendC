kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelGemmMulLeakyRelu {
public:
    __aicore__ inline KernelGemmMulLeakyRelu() : multiplier(1.0f), negSlope(0.1f), scalarLoaded(false) {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR multiplierGm, GM_ADDR negSlopeGm, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ float *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ float *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mulGm.SetGlobalBuffer((__gm__ float *)multiplierGm, 1);
        slopeGm.SetGlobalBuffer((__gm__ float *)negSlopeGm, 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(scalarQueue, 1, 2 * sizeof(float));
        LoadScalars();
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void LoadScalars()
    {
        if (!scalarLoaded) {
            AscendC::LocalTensor<float> tmp = scalarQueue.AllocTensor<float>();
            // load multiplier
            AscendC::DataCopy(tmp, mulGm, 1);
            multiplier = tmp.GetValue(0);
            // load negative slope
            AscendC::DataCopy(tmp, slopeGm, 1);
            negSlope = tmp.GetValue(0);
            scalarQueue.FreeTensor(tmp);
            scalarLoaded = true;
        }
    }

    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        // scale by multiplier then apply LeakyReLU
        AscendC::LocalTensor<float> tmpLocal = outQueueY.AllocTensor<float>();
        AscendC::Muls(tmpLocal, xLocal, multiplier, this->tileLength);
        AscendC::LeakyRelu(yLocal, tmpLocal, negSlope, this->tileLength);

        outQueueY.EnQue<float>(yLocal);
        outQueueY.FreeTensor(tmpLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scalarQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<float> mulGm;
    AscendC::GlobalTensor<float> slopeGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    float multiplier;
    float negSlope;
    bool scalarLoaded;
};

extern "C" __global__ __aicore__ void gemm_multiply_leakyrelu_custom(GM_ADDR x, GM_ADDR multiplier,
                                                                      GM_ADDR negative_slope, GM_ADDR y,
                                                                      GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMulLeakyRelu op;
    op.Init(x, multiplier, negative_slope, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep GEMM with nn.Linear, fuse subsequent multiply and LeakyReLU into one AscendC kernel.
    """
    def __init__(self, in_features, out_features, multiplier, negative_slope):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.register_buffer("_multiplier_buf", torch.tensor([multiplier], dtype=torch.float32))
        self.register_buffer("_neg_slope_buf", torch.tensor([negative_slope], dtype=torch.float32))

    def forward(self, x):
        x = self.gemm(x)
        # fused: y = LeakyReLU(x * multiplier, negative_slope)
        y = custom_ops_lib.gemm_multiply_leakyrelu_custom(x, self._multiplier_buf, self._neg_slope_buf)
        return y
