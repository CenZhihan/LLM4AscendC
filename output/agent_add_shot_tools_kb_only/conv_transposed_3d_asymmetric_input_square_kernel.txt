kernel_src="""
#include "kernel_operator.h"

// Only supports: stride=1, padding=0, dilation=1, output_padding=0, groups=1, no bias.
// Input x: [N, Cin, D, H, W]
// Weight w: [Cin, Cout, K, K, K] (square kernel)
// Output y: [N, Cout, D + K - 1, H + K - 1, W + K - 1]
using DTYPE = float;

class KernelConvTransposed3D {
public:
    __aicore__ inline KernelConvTransposed3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t in_d, uint32_t in_h, uint32_t in_w, uint32_t k,
                                uint32_t out_d, uint32_t out_h, uint32_t out_w)
    {
        this->N = n; this->Cin = cin; this->Cout = cout;
        this->Din = in_d; this->Hin = in_h; this->Win = in_w;
        this->K = k;
        this->Dout = out_d; this->Hout = out_h; this->Wout = out_w;

        // Partition by batch
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t batchPerBlock = (N + blockNum - 1) / blockNum;
        this->batchStart = blockIdx * batchPerBlock;
        this->batchEnd = this->batchStart + batchPerBlock;
        if (this->batchEnd > N) this->batchEnd = N;

        // Set GM buffers to the start of this block's batch range
        uint64_t xStrideB = (uint64_t)Cin * Din * Hin * Win;
        uint64_t yStrideB = (uint64_t)Cout * Dout * Hout * Wout;
        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + xStrideB * this->batchStart, xStrideB * (this->batchEnd - this->batchStart));
        wGm.SetGlobalBuffer((__gm__ DTYPE*)w, (uint64_t)Cin * Cout * K * K * K);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + yStrideB * this->batchStart, yStrideB * (this->batchEnd - this->batchStart));

        // Init queues for row buffers
        // x row length = Win, y row length = max(Win, Wout) but we process W each time
        uint32_t xRowBytes = Win * sizeof(DTYPE);
        uint32_t yRowBytes = Win * sizeof(DTYPE);
        pipe.InitBuffer(xRowQue, 1, xRowBytes);
        pipe.InitBuffer(yRowQue, 1, yRowBytes);
        pipe.InitBuffer(zeroRowQue, 1, (uint32_t)(Wout * sizeof(DTYPE)));
    }

    __aicore__ inline void Process()
    {
        if (batchStart >= batchEnd) { return; }
        // 1) zero initialize output for the assigned batch range
        ZeroOutput();

        // 2) compute transposed conv (naive, tiled by rows on width dimension)
        for (uint32_t b = batchStart; b < batchEnd; ++b) {
            uint64_t xBaseB = (uint64_t)(b - batchStart) * Cin * Din * Hin * Win;
            uint64_t yBaseB = (uint64_t)(b - batchStart) * Cout * Dout * Hout * Wout;

            for (uint32_t ic = 0; ic < Cin; ++ic) {
                uint64_t xBaseC = xBaseB + (uint64_t)ic * Din * Hin * Win;

                for (uint32_t kd = 0; kd < K; ++kd) {
                    for (uint32_t kh = 0; kh < K; ++kh) {
                        for (uint32_t d = 0; d < Din; ++d) {
                            uint32_t zOut = d + kd;
                            for (uint32_t h = 0; h < Hin; ++h) {
                                uint32_t yOut = h + kh;

                                // load x row [W]
                                AscendC::LocalTensor<DTYPE> xLocal = xRowQue.AllocTensor<DTYPE>();
                                uint64_t xRowOff = xBaseC + (uint64_t)d * Hin * Win + (uint64_t)h * Win;
                                AscendC::DataCopy(xLocal, xGm[xRowOff], Win);

                                // for each kernel kw and output channel
                                for (uint32_t kw = 0; kw < K; ++kw) {
                                    uint32_t xOffsetInOutW = kw; // left shift on output width
                                    for (uint32_t oc = 0; oc < Cout; ++oc) {
                                        // weight index: [ic, oc, kd, kh, kw]
                                        uint64_t wOff = (((uint64_t)ic * Cout + oc) * K + kd) * K * K
                                                        + (uint64_t)kh * K + kw;
                                        DTYPE wScalar = wGm[wOff];

                                        // y row base for this (b, oc, zOut, yOut), starting at xOffsetInOutW
                                        AscendC::LocalTensor<DTYPE> yLocal = yRowQue.AllocTensor<DTYPE>();
                                        uint64_t yRowBase = yBaseB
                                            + (uint64_t)oc * Dout * Hout * Wout
                                            + (uint64_t)zOut * Hout * Wout
                                            + (uint64_t)yOut * Wout
                                            + (uint64_t)xOffsetInOutW;
                                        AscendC::DataCopy(yLocal, yGm[yRowBase], Win);

                                        // yLocal = wScalar * xLocal + yLocal
                                        AscendC::Axpy(yLocal, xLocal, wScalar, Win);

                                        AscendC::DataCopy(yGm[yRowBase], yLocal, Win);
                                        yRowQue.FreeTensor(yLocal);
                                    } // oc
                                } // kw

                                xRowQue.FreeTensor(xLocal);
                            } // h
                        } // d
                    } // kh
                } // kd
            } // ic
        } // b
    }

private:
    __aicore__ inline void ZeroOutput()
    {
        // Zero y for each assigned batch
        AscendC::LocalTensor<DTYPE> zeroLocal = zeroRowQue.AllocTensor<DTYPE>();
        // Fill zeroLocal with zeros
        // Use Duplicate to quickly set to 0
        AscendC::Duplicate(zeroLocal, (DTYPE)0, Wout);

        for (uint32_t b = batchStart; b < batchEnd; ++b) {
            uint64_t yBaseB = (uint64_t)(b - batchStart) * Cout * Dout * Hout * Wout;
            for (uint32_t oc = 0; oc < Cout; ++oc) {
                for (uint32_t z = 0; z < Dout; ++z) {
                    for (uint32_t y = 0; y < Hout; ++y) {
                        uint64_t yRowBase = yBaseB
                            + (uint64_t)oc * Dout * Hout * Wout
                            + (uint64_t)z * Hout * Wout
                            + (uint64_t)y * Wout;
                        AscendC::DataCopy(yGm[yRowBase], zeroLocal, Wout);
                    }
                }
            }
        }
        zeroRowQue.FreeTensor(zeroLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  xRowQue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yRowQue;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> zeroRowQue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t N, Cin, Cout;
    uint32_t Din, Hin, Win;
    uint32_t K;
    uint32_t Dout, Hout, Wout;

    uint32_t batchStart, batchEnd;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvTransposed3D op;
    op.Init(x, w, y,
            tilingData.n, tilingData.cin, tilingData.cout,
            tilingData.in_d, tilingData.in_h, tilingData.in_w,
            tilingData.k, tilingData.out_d, tilingData.out_h, tilingData.out_w);
    op.Process();
}
