kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // 每次处理1行，简化对齐与UB使用

class KernelL1Norm {
public:
    __aicore__ inline KernelL1Norm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t bLength, uint32_t hLength, uint32_t rowsPerTile)
    {
        this->bLength = bLength;
        this->hLength = hLength;
        this->rowsPerTile = rowsPerTile;

        // 多核按行均分
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t baseRows = bLength / blockNum;
        uint32_t rem = bLength % blockNum;

        if (blockIdx < rem) {
            this->startRow = blockIdx * (baseRows + 1);
            this->rowsThisBlock = baseRows + 1;
        } else {
            this->startRow = rem * (baseRows + 1) + (blockIdx - rem) * baseRows;
            this->rowsThisBlock = baseRows;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)startRow * hLength, (uint64_t)rowsThisBlock * hLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + (uint64_t)startRow * hLength, (uint64_t)rowsThisBlock * hLength);

        // 队列与缓冲：一行输入、一行输出、ReduceSum中间缓冲与结果
        pipe.InitBuffer(inQueueX,  BUFFER_NUM, hLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, hLength * sizeof(DTYPE_Z));
        // work缓冲给ReduceSum使用（大小保守取一行）
        pipe.InitBuffer(workQueue, BUFFER_NUM, hLength * sizeof(DTYPE_X));
        // ReduceSum输出标量（对齐32B）
        pipe.InitBuffer(sumQueue,  BUFFER_NUM, AscendC::ONE_BLK_SIZE);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsThisBlock; ++r) {
            // 1) 取该行数据到UB
            AscendC::LocalTensor<DTYPE_X> xRow = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xRow, xGm[r * hLength], hLength);

            // 2) 计算 |x| 的和，使用ReduceSum + GetAccVal
            AscendC::LocalTensor<DTYPE_X> workLocal = workQueue.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> sumLocal  = sumQueue.AllocTensor<DTYPE_X>();
            AscendC::Abs(xRow, xRow, hLength);
            AscendC::ReduceSum(sumLocal, xRow, workLocal, hLength);
            DTYPE_X sumAbs = AscendC::GetAccVal<DTYPE_X>();

            // 3) 计算 1/mean(|x|) = H / sumAbs
            DTYPE_X invMean = (DTYPE_X)hLength / sumAbs;

            // 4) 再次从GM取原始x行，乘以 invMean，写回
            AscendC::LocalTensor<DTYPE_Z> yRow = outQueueY.AllocTensor<DTYPE_Z>();
            AscendC::DataCopy(yRow, xGm[r * hLength], hLength);
            AscendC::Muls(yRow, yRow, invMean, hLength);

            // 写回GM
            AscendC::DataCopy(yGm[r * hLength], yRow, hLength);

            // 释放本次使用的本地缓冲
            outQueueY.FreeTensor(yRow);
            sumQueue.FreeTensor(sumLocal);
            workQueue.FreeTensor(workLocal);
            inQueueX.FreeTensor(xRow);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> workQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> sumQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t bLength;
    uint32_t hLength;
    uint32_t rowsPerTile;

    uint32_t startRow;
    uint32_t rowsThisBlock;
};

extern "C" __global__ __aicore__ void l1_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelL1Norm op;
    op.Init(x, y, tiling_data.bLength, tiling_data.hLength, tiling_data.rowsPerTile);
    op.Process();
}
"""
