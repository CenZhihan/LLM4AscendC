kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2; // 每个队列的张量数量

class KernelVgg16Relu {
public:
    __aicore__ inline KernelVgg16Relu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        this->blockLength = (totalLength + blockNum - 1) / blockNum; // 简单均分，最后一块按需处理
        this->tileNum = tileNum;
        // 每个tile的长度（包含尾块处理，向上取整）
        uint32_t denom = tileNum * BUFFER_NUM;
        this->tileLength = (this->blockLength + denom - 1) / denom;

        uint32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();
        xGm.SetGlobalBuffer((__gm__ DTYPE *)x + blockOffset, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y + blockOffset, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        if (this->blockLength == 0) {
            return;
        }
        int32_t loopCount = (this->blockLength + this->tileLength - 1) / this->tileLength;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline uint32_t GetCurrLen(int32_t progress) const
    {
        uint32_t offset = progress * this->tileLength;
        uint32_t remain = (offset < this->blockLength) ? (this->blockLength - offset) : 0;
        return remain > this->tileLength ? this->tileLength : remain;
    }

    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint32_t currLen = GetCurrLen(progress);
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], currLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        uint32_t currLen = GetCurrLen(progress);
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
        // 基于向量单元的ReLU计算
        AscendC::Relu(yLocal, xLocal, static_cast<int32_t>(currLen));
        outQueueY.EnQue<DTYPE>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        uint32_t currLen = GetCurrLen(progress);
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, currLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;
    uint32_t blockLength = 0;
    uint32_t tileNum = 0;
    uint32_t tileLength = 0;
};

extern "C" __global__ __aicore__ void vgg16_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelVgg16Relu op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class NPUReLU(torch.nn.Module):
    def __init__(self):
        super().__init__()
    def forward(self, x):
        return custom_ops_lib.vgg16_custom(x)

class ModelNew(torch.nn.Module):
    def __init__(self, num_classes=1000):
        super(ModelNew, self).__init__()
        self.features = torch.nn.Sequential(
            # Block 1
            torch.nn.Conv2d(3, 64, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.Conv2d(64, 64, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.MaxPool2d(kernel_size=2, stride=2),

            # Block 2
            torch.nn.Conv2d(64, 128, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.Conv2d(128, 128, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.MaxPool2d(kernel_size=2, stride=2),

            # Block 3
            torch.nn.Conv2d(128, 256, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.Conv2d(256, 256, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.Conv2d(256, 256, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.MaxPool2d(kernel_size=2, stride=2),

            # Block 4
            torch.nn.Conv2d(256, 512, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.Conv2d(512, 512, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.Conv2d(512, 512, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.MaxPool2d(kernel_size=2, stride=2),

            # Block 5
            torch.nn.Conv2d(512, 512, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.Conv2d(512, 512, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.Conv2d(512, 512, kernel_size=3, padding=1),
            NPUReLU(),
            torch.nn.MaxPool2d(kernel_size=2, stride=2)
        )

        self.classifier = torch.nn.Sequential(
            torch.nn.Linear(512 * 7 * 7, 4096),
            NPUReLU(),
            torch.nn.Dropout(p=0.0),
            torch.nn.Linear(4096, 4096),
            NPUReLU(),
            torch.nn.Dropout(p=0.0),
            torch.nn.Linear(4096, num_classes)
        )

    def forward(self, x):
        x = self.features(x)
        x = torch.flatten(x, 1)
        x = self.classifier(x)
        return x
'''
