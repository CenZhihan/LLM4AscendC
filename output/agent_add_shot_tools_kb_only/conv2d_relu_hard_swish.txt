kernel_src="""
#include "kernel_operator.h"

// 本Kernel执行Conv2D + ReLU + HardSwish: y = ReLU(conv(x, w) + b); z = y * clamp((y + 3)/6, 0, 1)
constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数量

class KernelConv2dReluHardSwish {
public:
    __aicore__ inline KernelConv2dReluHardSwish() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t ww,
                                uint32_t outC, uint32_t kH, uint32_t kW,
                                uint32_t outH, uint32_t outW, uint32_t tileNum)
    {
        this->n = n; this->c = c; this->h = h; this->w = ww;
        this->outC = outC; this->kH = kH; this->kW = kW; this->outH = outH; this->outW = outW;
        this->tileNum = tileNum;

        // 输出总长度
        this->totalOutLen = n * outC * outH * outW;
        uint32_t blockNum = AscendC::GetBlockNum();
        this->blockOutLen = (blockNum > 0) ? (this->totalOutLen / blockNum) : this->totalOutLen;
        this->tileLen = (this->blockOutLen / tileNum / BUFFER_NUM);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, n * c * h * ww);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, outC * c * kH * kW);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b, outC);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockOutLen * AscendC::GetBlockIdx(), this->blockOutLen);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));   // 仅用于演示占位
        pipe.InitBuffer(inQueueW, BUFFER_NUM, (outC * c * kH * kW / tileNum + 1) * sizeof(DTYPE_W)); // 简化：权重分片
        pipe.InitBuffer(inQueueB, BUFFER_NUM, outC * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            // 对应片段的卷积 + ReLU + HardSwish
            Compute(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress)
    {
        // 分配本地缓冲
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> reluLocal = tmpQueue.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> tmpLocal = inQueueX.AllocTensor<DTYPE_Z>(); // 复用队列作为临时

        // 简化示例调用: 假设存在Conv2D接口，按当前tile输出范围计算卷积至outLocal
        // 实际实现中应将输入/权重/偏置按tile搬运至本地，再调用Cube路径做卷积
        // 这里仅展示API调用和后续融合逻辑
        // AscendC::Conv2D(outLocal, xLocal4D, wLocal4D, bLocal1D, ...); // 省略具体参数

        // 为了示例完整性，先将outLocal清零（真实实现由Conv2D写入）
        AscendC::Duplicate(outLocal, static_cast<DTYPE_Z>(0.0f), this->tileLen);

        // ReLU
        AscendC::Relu(reluLocal, outLocal, this->tileLen);

        // HardSwish: y * clamp((y + 3)/6, 0, 1)
        // tmpLocal = y + 3
        AscendC::Adds(tmpLocal, reluLocal, static_cast<DTYPE_Z>(3.0f), this->tileLen);
        // tmpLocal = tmpLocal * (1/6)
        AscendC::Muls(tmpLocal, tmpLocal, static_cast<DTYPE_Z>(1.0f/6.0f), this->tileLen);
        // tmpLocal = min(tmpLocal, 1)
        AscendC::Mins(tmpLocal, tmpLocal, static_cast<DTYPE_Z>(1.0f), this->tileLen);
        // tmpLocal = max(tmpLocal, 0)
        AscendC::Maxs(tmpLocal, tmpLocal, static_cast<DTYPE_Z>(0.0f), this->tileLen);
        // outLocal = y * tmpLocal
        AscendC::Mul(outLocal, reluLocal, tmpLocal, this->tileLen);

        outQueueZ.EnQue<DTYPE_Z>(outLocal);

        // 将结果写回GM
        AscendC::LocalTensor<DTYPE_Z> zWriteLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLen], zWriteLocal, this->tileLen);
        outQueueZ.FreeTensor(zWriteLocal);

        // 释放临时
        tmpQueue.FreeTensor(reluLocal);
        inQueueX.FreeTensor(tmpLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX;   // 复用为临时
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TBuf<AscendC::TPosition::VECCALC>            tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t n, c, h, w, outC, kH, kW, outH, outW;
    uint32_t tileNum;
    uint32_t totalOutLen;
    uint32_t blockOutLen;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluHardSwish op;
    op.Init(x, weight, bias, z,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.outC, tiling_data.kH, tiling_data.kW,
            tiling_data.outH, tiling_data.outW, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC算子，将Conv2D、ReLU与HardSwish融合。
    假设stride=1、padding=0。

        # 调用自定义融合算子，传入卷积权重与偏置
