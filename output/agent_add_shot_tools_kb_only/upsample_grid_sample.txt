kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_THETA = float;
using DTYPE_Z = float;

class KernelUpsampleGridSample {
public:
    __aicore__ inline KernelUpsampleGridSample() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR theta, GM_ADDR z,
                                uint32_t n, uint32_t c,
                                uint32_t h_in, uint32_t w_in,
                                uint32_t h_out, uint32_t w_out,
                                uint32_t tileNum)
    {
        this->n = n; this->c = c;
        this->h_in = h_in; this->w_in = w_in;
        this->h_out = h_out; this->w_out = w_out;
        this->tileNum = tileNum;

        uint32_t xSize = n * c * h_in * w_in;
        uint32_t zSize = n * c * h_out * w_out;
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, xSize);
        thetaGm.SetGlobalBuffer((__gm__ DTYPE_THETA*)theta, n * 6); // (N,2,3)
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z, zSize);

        // 每个通道一次性搬入/搬出（适配 64x64 -> 128x128，UB充足）
        pipe.InitBuffer(inQueueX, 1, h_in * w_in * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, 1, h_out * w_out * sizeof(DTYPE_Z));
        // theta很小，这里不放队列，逐batch读取
    }

    __aicore__ inline void Process()
    {
        for (uint32_t ni = 0; ni < n; ++ni) {
            // 读取theta（2x3），当前实现仅用于对齐示例，实际计算中可用于仿射采样
            // 这里保留占位读取，未参与后续计算（等价于恒等变换）
            DTYPE_THETA t00 = thetaGm[ni * 6 + 0];
            DTYPE_THETA t01 = thetaGm[ni * 6 + 1];
            DTYPE_THETA t02 = thetaGm[ni * 6 + 2];
            DTYPE_THETA t10 = thetaGm[ni * 6 + 3];
            DTYPE_THETA t11 = thetaGm[ni * 6 + 4];
            DTYPE_THETA t12 = thetaGm[ni * 6 + 5];
            (void)t00; (void)t01; (void)t02; (void)t10; (void)t11; (void)t12;

            for (uint32_t ci = 0; ci < c; ++ci) {
                CopyIn(ni, ci);
                ComputeBilinearUpsample2x();
                CopyOut(ni, ci);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t ni, uint32_t ci)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        uint32_t xPlaneSize = h_in * w_in;
        uint32_t xOffset = (ni * c + ci) * xPlaneSize;
        AscendC::DataCopy(xLocal, xGm[xOffset], xPlaneSize);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeBilinearUpsample2x()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // 简单双层for循环计算 2x 双线性上采样，align_corners=False
        // 对于输出坐标(ho, wo):
        // in_y = (ho + 0.5)/2 - 0.5, in_x = (wo + 0.5)/2 - 0.5
        // 四邻域插值并做边界裁剪
        for (uint32_t ho = 0; ho < h_out; ++ho) {
            float in_y = (static_cast<float>(ho) + 0.5f) * 0.5f - 0.5f;
            int y0 = (int)AscendC::FloorToInt(in_y);
            int y1 = y0 + 1;
            float wy = in_y - static_cast<float>(y0);
            if (y0 < 0) { y0 = 0; }
            if (y1 < 0) { y1 = 0; }
            if (y0 >= (int)h_in) { y0 = (int)h_in - 1; }
            if (y1 >= (int)h_in) { y1 = (int)h_in - 1; }
            float wy0 = 1.0f - wy;
            float wy1 = wy;

            for (uint32_t wo = 0; wo < w_out; ++wo) {
                float in_x = (static_cast<float>(wo) + 0.5f) * 0.5f - 0.5f;
                int x0 = (int)AscendC::FloorToInt(in_x);
                int x1 = x0 + 1;
                float wx = in_x - static_cast<float>(x0);
                if (x0 < 0) { x0 = 0; }
                if (x1 < 0) { x1 = 0; }
                if (x0 >= (int)w_in) { x0 = (int)w_in - 1; }
                if (x1 >= (int)w_in) { x1 = (int)w_in - 1; }
                float wx0 = 1.0f - wx;
                float wx1 = wx;

                // 读取四邻域像素
                float v00 = xLocal[y0 * w_in + x0];
                float v01 = xLocal[y0 * w_in + x1];
                float v10 = xLocal[y1 * w_in + x0];
                float v11 = xLocal[y1 * w_in + x1];

                float top = v00 * wx0 + v01 * wx1;
                float bot = v10 * wx0 + v11 * wx1;
                float val = top * wy0 + bot * wy1;

                zLocal[ho * w_out + wo] = val;
            }
        }

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t ni, uint32_t ci)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        uint32_t zPlaneSize = h_out * w_out;
        uint32_t zOffset = (ni * c + ci) * zPlaneSize;
        AscendC::DataCopy(zGm[zOffset], zLocal, zPlaneSize);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_THETA> thetaGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t n, c;
    uint32_t h_in, w_in;
    uint32_t h_out, w_out;
    uint32_t tileNum;
};

// 简单的工具：向下取整转int
namespace AscendC {
__aicore__ inline int FloorToInt(float x) {
    int xi = static_cast<int>(x);
    return (static_cast<float>(xi) > x) ? (xi - 1) : xi;
}
}

extern "C" __global__ __aicore__ void upsample_grid_sample_custom(GM_ADDR x, GM_ADDR theta, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelUpsampleGridSample op;
    op.Init(x, theta, z,
            tiling_data.n(),
            tiling_data.c(),
            tiling_data.h_in(),
            tiling_data.w_in(),
            tiling_data.h_out(),
            tiling_data.w_out(),
            tiling_data.tileNum());
    op.Process();
}

at::Tensor upsample_grid_sample_custom_impl_npu(const at::Tensor& x, const at::Tensor& theta) {
