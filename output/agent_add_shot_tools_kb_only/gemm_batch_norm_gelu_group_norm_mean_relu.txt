kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

using DTYPE = float;

class KernelFusedBnGeluRelu {
public:
    __aicore__ inline KernelFusedBnGeluRelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR shift, GM_ADDR y,
                                uint32_t totalRows, uint32_t features, uint32_t rowsPerBlock)
    {
        this->totalRows = totalRows;
        this->features = features;
        this->rowsPerBlock = rowsPerBlock;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > totalRows) { endRow = totalRows; }
        this->myStartRow = startRow;
        this->myRows = (endRow > startRow) ? (endRow - startRow) : 0;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(x) + static_cast<uint64_t>(myStartRow) * features,
                            static_cast<uint64_t>(myRows) * features);
        scaleGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(scale), features);
        shiftGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(shift), features);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(y) + static_cast<uint64_t>(myStartRow) * features,
                            static_cast<uint64_t>(myRows) * features);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, features * sizeof(DTYPE));
        pipe.InitBuffer(inQueueScale, 1, features * sizeof(DTYPE));
        pipe.InitBuffer(inQueueShift, 1, features * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, features * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < myRows; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> scaleLocal = inQueueScale.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> shiftLocal = inQueueShift.AllocTensor<DTYPE>();

        AscendC::DataCopy(xLocal, xGm[static_cast<uint64_t>(rowIdx) * features], features);
        AscendC::DataCopy(scaleLocal, scaleGm, features);
        AscendC::DataCopy(shiftLocal, shiftGm, features);

        inQueueX.EnQue(xLocal);
        inQueueScale.EnQue(scaleLocal);
        inQueueShift.EnQue(shiftLocal);
    }

    __aicore__ inline void Compute(uint32_t /*rowIdx*/)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> scaleLocal = inQueueScale.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> shiftLocal = inQueueShift.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();

        // y = x * scale + shift
        AscendC::Mul(yLocal, xLocal, scaleLocal, features);
        AscendC::Add(yLocal, yLocal, shiftLocal, features);

        // GELU
        AscendC::Gelu(yLocal, yLocal, features);

        // ReLU
        AscendC::Relu(yLocal, yLocal, features);

        outQueueY.EnQue<DTYPE>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueScale.FreeTensor(scaleLocal);
        inQueueShift.FreeTensor(shiftLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[static_cast<uint64_t>(rowIdx) * features], yLocal, features);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueScale;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueShift;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> scaleGm;
    AscendC::GlobalTensor<DTYPE> shiftGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t totalRows;
    uint32_t features;
    uint32_t rowsPerBlock;
    uint32_t myStartRow;
    uint32_t myRows;
};

extern "C" __global__ __aicore__ void gemm_batch_norm_gelu_group_norm_mean_relu_custom(
    GM_ADDR x, GM_ADDR scale, GM_ADDR shift, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedBnGeluRelu op;
    op.Init(x, scale, shift, y,
            tiling_data.totalRows,
            tiling_data.features,
            tiling_data.rowsPerBlock);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.batch_norm = torch.nn.BatchNorm1d(out_features)

    def forward(self, x):
        # GEMM (Linear)
        y = self.gemm(x)

        # Precompute BatchNorm inference affine params on-the-fly:
        # scale = gamma / sqrt(running_var + eps)
        # shift = beta - running_mean * scale
        bn = self.batch_norm
        # Ensure tensors are on the same device/dtype as y
        running_var = bn.running_var.to(device=y.device, dtype=y.dtype)
        running_mean = bn.running_mean.to(device=y.device, dtype=y.dtype)
        weight = (bn.weight if bn.weight is not None else torch.ones_like(running_var)).to(device=y.device, dtype=y.dtype)
        bias = (bn.bias if bn.bias is not None else torch.zeros_like(running_var)).to(device=y.device, dtype=y.dtype)
        eps = torch.tensor(bn.eps, device=y.device, dtype=y.dtype)

        scale = weight / torch.sqrt(running_var + eps)
        shift = bias - running_mean * scale

        # Fused BatchNorm + GELU + ReLU custom op
        y = custom_ops_lib.gemm_batch_norm_gelu_group_norm_mean_relu_custom(y, scale, shift)
        return y
'''
