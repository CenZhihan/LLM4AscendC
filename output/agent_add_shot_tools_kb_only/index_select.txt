#include "index_select_custom_tiling.h"

namespace {
inline uint32_t GetUint32Dim(const gert::Shape* shape, size_t idx) {
    if (shape == nullptr) return 0;
    auto os = shape->GetOriginShape();
    if (idx >= os.GetDimNum()) return 0;
    int64_t d = os.GetDim(idx);
    return d > 0 ? static_cast<uint32_t>(d) : 0;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    IndexSelectCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* idx_shape = context->GetInputShape(1);

    uint32_t rows = GetUint32Dim(x_shape, 0);
    uint32_t cols = GetUint32Dim(x_shape, 1);
    uint32_t indexLength = static_cast<uint32_t>(idx_shape->GetOriginShape().GetShapeSize());

    // 简单按索引长度分块，避免UB溢出
    uint32_t tileK = indexLength;
    const uint32_t MAX_TILE_K = 1024; // 可调
    if (tileK > MAX_TILE_K) tileK = MAX_TILE_K;

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_indexLength(indexLength);
    tiling.set_tileK(tileK);

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* idx_shape = context->GetInputShape(1);

    uint64_t n = x_shape->GetOriginShape().GetDim(0);
    uint64_t m = idx_shape->GetOriginShape().GetShapeSize();
    gert::Shape* y_shape = context->GetOutputShape(0);
    gert::Shape out_shape({static_cast<int64_t>(n), static_cast<int64_t>(m)});
    *y_shape = out_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Z = float;
using DTYPE_INDICES = int32_t;

class KernelIndexSelect {
public:
    __aicore__ inline KernelIndexSelect() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t indexLength, uint32_t tileK)
    {
        this->rows = rows;
        this->cols = cols;
        this->indexLength = indexLength;
        this->tileK = tileK;

        // 计算每个block负责的行范围（最后一个block兜底）
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (rows + blockNum - 1) / blockNum;
        this->rowStart = rowsPerBlock * blockIdx;
        uint32_t rowEnd = this->rowStart + rowsPerBlock;
        if (rowEnd > rows) rowEnd = rows;
        this->procRows = (rowEnd > this->rowStart) ? (rowEnd - this->rowStart) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + static_cast<uint64_t>(this->rowStart) * cols, static_cast<uint64_t>(this->procRows) * cols);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_INDICES *)indices, indexLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + static_cast<uint64_t>(this->rowStart) * indexLength, static_cast<uint64_t>(this->procRows) * indexLength);

        // 本地缓存：一行输入、一个索引tile、一个输出tile（对应一行）
        pipe.InitBuffer(xRowQueue, 1, cols * sizeof(DTYPE_X));
        pipe.InitBuffer(idxQueue, 1, tileK * sizeof(DTYPE_INDICES));
        pipe.InitBuffer(yRowQueue, 1, tileK * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (procRows == 0 || indexLength == 0) {
            return;
        }

        for (uint32_t t = 0; t < indexLength; t += tileK) {
            uint32_t curTile = indexLength - t;
            if (curTile > tileK) curTile = tileK;

            // 复制索引到UB
            AscendC::LocalTensor<DTYPE_INDICES> idxLocal = idxQueue.AllocTensor<DTYPE_INDICES>();
            AscendC::DataCopy(idxLocal, idxGm[t], curTile);
            idxQueue.EnQue(idxLocal);
            idxLocal = idxQueue.DeQue<DTYPE_INDICES>();

            for (uint32_t r = 0; r < procRows; ++r) {
                // 复制当前行到UB
                AscendC::LocalTensor<DTYPE_X> xRowLocal = xRowQueue.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xRowLocal, xGm[static_cast<uint64_t>(r) * cols], cols);

                // 目标行tile
                AscendC::LocalTensor<DTYPE_Z> yRowLocal = yRowQueue.AllocTensor<DTYPE_Z>();

                // 使用Gather根据索引从行缓冲中取数
                // Gather: 将xRowLocal中由idxLocal指定位置的元素收集到yRowLocal，长度为curTile
                AscendC::Gather(yRowLocal, xRowLocal, idxLocal, curTile);

                // 写回到输出（目标为当前行在[y]中的对应tile位置）
                AscendC::DataCopy(yGm[static_cast<uint64_t>(r) * indexLength + t], yRowLocal, curTile);

                // 释放本地tensor
                xRowQueue.FreeTensor(xRowLocal);
                yRowQueue.FreeTensor(yRowLocal);
            }
            idxQueue.FreeTensor(idxLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> xRowQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> idxQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> yRowQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_INDICES> idxGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t indexLength;
    uint32_t tileK;

    uint32_t rowStart;
    uint32_t procRows;
};

extern "C" __global__ __aicore__ void index_select_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelIndexSelect op;
    op.Init(x, indices, y,
            tilingData.rows(),
            tilingData.cols(),
            tilingData.indexLength(),
            tilingData.tileK());
    op.Process();
}
"""

at::Tensor index_select_custom_impl_npu(const at::Tensor& self, const at::Tensor& indices) {

    auto out_sizes = std::vector<int64_t>{self.size(0), indices.size(0)};
    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnIndexSelectCustom, self, indices, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"index_select_custom\", &index_select_custom_impl_npu);
}
