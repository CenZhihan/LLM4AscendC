kernel_src="""
#include "kernel_operator.h"

// Naive reference implementation (per-element) for ConvTranspose1d with dilation on AscendC.
// Assumes weight shape [Cin, Cout, K], input [N, Cin, Lin], output [N, Cout, Lout].
// Lout = (Lin - 1) * stride - 2 * padding + dilation * (K - 1) + 1.
constexpr int32_t BUFFER_NUM = 1;

class KernelConvTransposed1dDilated {
public:
    __aicore__ inline KernelConvTransposed1dDilated() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Lin, uint32_t Lout, uint32_t K,
                                int32_t stride, int32_t padding, int32_t dilation,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->N = N;
        this->Cin = Cin;
        this->Cout = Cout;
        this->Lin = Lin;
        this->Lout = Lout;
        this->K = K;
        this->stride = stride;
        this->padding = padding;
        this->dilation = dilation;
        this->tileNum = tileNum;

        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileLength = (this->blockLength + tileNum * BUFFER_NUM - 1) / (tileNum * BUFFER_NUM);

        xGm = reinterpret_cast<__gm__ DTYPE_X *>(x);
        wGm = reinterpret_cast<__gm__ DTYPE_W *>(w);
        yGm = reinterpret_cast<__gm__ DTYPE_Y *>(y);

        // Initialize output block to zero (naively)
        // Each block processes a contiguous range; we'll set to zero before accumulation
        uint64_t blockOffset = static_cast<uint64_t>(this->blockLength) * AscendC::GetBlockIdx();
        for (uint64_t i = 0; i < this->blockLength; ++i) {
            yGm[blockOffset + i] = static_cast<DTYPE_Y>(0);
        }
    }

    __aicore__ inline void Process()
    {
        // Process by tiles; each output element computed independently by summation
        uint32_t loopCount = this->tileNum * BUFFER_NUM;
        for (uint32_t t = 0; t < loopCount; ++t) {
            ComputeTile(t);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t progress)
    {
        uint64_t blockBase = static_cast<uint64_t>(this->blockLength) * AscendC::GetBlockIdx();
        uint64_t start = blockBase + static_cast<uint64_t>(progress) * this->tileLength;
        uint64_t end = AscendC::Min<uint64_t>(blockBase + this->blockLength, start + this->tileLength);

        for (uint64_t gidx = start; gidx < end; ++gidx) {
            // Map global index to (n, oc, pos)
            uint64_t idx = gidx;
            uint32_t n   = static_cast<uint32_t>(idx / (static_cast<uint64_t>(Cout) * Lout));
            idx          = idx % (static_cast<uint64_t>(Cout) * Lout);
            uint32_t oc  = static_cast<uint32_t>(idx / Lout);
            uint32_t pos = static_cast<uint32_t>(idx % Lout);

            float acc = 0.f;
            for (uint32_t ic = 0; ic < Cin; ++ic) {
                for (uint32_t k = 0; k < K; ++k) {
                    // in_pos must satisfy pos + padding - k*dilation == in_pos * stride
                    int64_t numer = static_cast<int64_t>(pos) + static_cast<int64_t>(padding) - static_cast<int64_t>(k) * static_cast<int64_t>(dilation);
                    if (numer < 0) continue;
                    if (numer % stride != 0) continue;
                    uint32_t in_pos = static_cast<uint32_t>(numer / stride);
                    if (in_pos >= Lin) continue;

                    uint64_t x_off = ((static_cast<uint64_t>(n) * Cin + ic) * Lin + in_pos);
                    uint64_t w_off = ((static_cast<uint64_t>(ic) * Cout + oc) * K + k);

                    float xv = static_cast<float>(xGm[x_off]);
                    float wv = static_cast<float>(wGm[w_off]);
                    acc += xv * wv;
                }
            }
            yGm[gidx] = static_cast<DTYPE_Y>(acc);
        }
    }

private:
    __gm__ DTYPE_X *xGm;
    __gm__ DTYPE_W *wGm;
    __gm__ DTYPE_Y *yGm;

    uint32_t N, Cin, Cout, Lin, Lout, K;
    int32_t stride, padding, dilation;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1dDilated op;
    op.Init(x, weight, y,
            tiling_data.N(), tiling_data.Cin(), tiling_data.Cout(),
            tiling_data.Lin(), tiling_data.Lout(), tiling_data.K(),
            tiling_data.stride(), tiling_data.padding(), tiling_data.dilation(),
            tiling_data.totalLength(), tiling_data.tileNum());
    op.Process();
}
