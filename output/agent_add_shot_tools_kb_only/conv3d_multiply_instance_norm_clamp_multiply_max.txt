kernel_src="""
#include "kernel_operator.h"

#define DTYPE_X float
#define DTYPE_Z float

// 本kernel实现: 对输入x(N,C,D,H,W)进行通道维max归约, 输出y(N,D,H,W).
// 上游在PyTorch中完成: Conv3d -> 逐通道乘法 -> InstanceNorm3d -> Clamp -> 再乘法
// 本kernel聚焦于通道方向的最大值归约, 提升性能的关键在于按(N, D*H*W)分块并在UB内循环C。

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxAlongC {
public:
    __aicore__ inline KernelMaxAlongC() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t d, uint32_t h, uint32_t w,
                                uint32_t s, uint32_t tileInner, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->d = d;
        this->h = h;
        this->w = w;
        this->s = s;
        this->tileInner = tileInner;
        this->tileNum = tileNum;

        // 将N维在多核之间均分
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t nPerBlock = n / blocks + ((blockIdx < (n % blocks)) ? 1 : 0);
        uint32_t nStart = (n / blocks) * blockIdx + (blockIdx < (n % blocks) ? blockIdx : (n % blocks));
        this->nStart = nStart;
        this->nCount = nPerBlock;

        // 全局tensor绑定到该block处理的起始位置
        uint64_t xOffset = static_cast<uint64_t>(nStart) * c * s;
        uint64_t yOffset = static_cast<uint64_t>(nStart) * s;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + xOffset, nPerBlock * c * s);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + yOffset, nPerBlock * s);

        // 初始化队列缓存: 每个tile大小为tileInner
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileInner * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueueMax, BUFFER_NUM, this->tileInner * sizeof(DTYPE_Z));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileInner * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t ni = 0; ni < nCount; ++ni) {
            for (uint32_t t = 0; t < tileNum; ++t) {
                uint32_t innerStart = t * tileInner;
                uint32_t innerLen = (t == tileNum - 1) ? (s - innerStart) : tileInner;

                // 初始化max缓冲为极小值
                AscendC::LocalTensor<DTYPE_Z> maxLocal = tmpQueueMax.AllocTensor<DTYPE_Z>();
                DTYPE_Z negInf = static_cast<DTYPE_Z>(-3.4e38f);
                AscendC::Duplicate(maxLocal, negInf, innerLen);

                for (uint32_t ci = 0; ci < c; ++ci) {
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                    // 计算当前通道的GM偏移
                    uint64_t off = static_cast<uint64_t>(ni) * c * s + static_cast<uint64_t>(ci) * s + innerStart;
                    AscendC::DataCopy(xLocal, xGm[off], innerLen);

                    // 逐通道更新max
                    AscendC::Max(maxLocal, maxLocal, xLocal, innerLen);

                    inQueueX.EnQue(xLocal);
                    inQueueX.FreeTensor(xLocal);
                }

                // 写回输出: y shape [n, d*h*w] 按同样内分块布局
                AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
                AscendC::DataCopy(yLocal, maxLocal, innerLen);
                uint64_t yOff = static_cast<uint64_t>(ni) * s + innerStart;
                AscendC::DataCopy(yGm[yOff], yLocal, innerLen);

                outQueueY.EnQue(yLocal);
                outQueueY.FreeTensor(yLocal);
                tmpQueueMax.EnQue(maxLocal);
                tmpQueueMax.FreeTensor(maxLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueMax;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t n, c, d, h, w, s;
    uint32_t tileInner, tileNum;
    uint32_t nStart, nCount;
};

extern "C" __global__ __aicore__ void conv3d_multiply_instance_norm_clamp_multiply_max_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxAlongC op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.s, tiling_data.tileInner, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版本：
    - 保留Conv3d/InstanceNorm3d在PyTorch中（由底层库高效实现）
    - 自定义AscendC算子完成通道方向最大值归约（dim=1），减少数据往返与调度开销
    - 其余逐通道乘法和clamp保留在PyTorch以确保数值一致性
    """
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape, clamp_min, clamp_max):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.multiplier = torch.nn.Parameter(torch.randn(multiplier_shape))
        self.instance_norm = torch.nn.InstanceNorm3d(out_channels)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

    def forward(self, x):
        # Conv3d
        x = self.conv(x)
        # 逐通道乘法 (broadcast on N,D,H,W)
        x = x * self.multiplier
        # InstanceNorm3d
        x = self.instance_norm(x)
        # clamp 与再次乘法（保持原模型序）
        x = torch.clamp(x, self.clamp_min, self.clamp_max)
        x = x * self.multiplier
        # 自定义AscendC核: 对通道维取max (dim=1)
        y = custom_ops_lib.conv3d_multiply_instance_norm_clamp_multiply_max_custom(x)
        return y
