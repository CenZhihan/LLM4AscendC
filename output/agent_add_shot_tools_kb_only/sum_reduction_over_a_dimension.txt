#include "sum_reduction_over_a_dimension_custom_tiling.h"

namespace {
inline void ComputeBeforeAfter(const std::vector<int64_t>& dims, int dim, uint32_t& before, uint32_t& reduce, uint32_t& after) {
    before = 1;
    for (int i = 0; i < dim; ++i) {
        before *= static_cast<uint32_t>(dims[i]);
    }
    reduce = static_cast<uint32_t>(dims[dim]);
    after = 1;
    for (size_t i = dim + 1; i < dims.size(); ++i) {
        after *= static_cast<uint32_t>(dims[i]);
    }
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t DEFAULT_TILE_INNER = 1024;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    SumReductionOverADimensionCustomTilingData tiling;

    // 读取 dim 属性
    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    int32_t dim = 0;
    if (attrs != nullptr) {
        dim = *(attrs->GetAttrPointer<int32_t>(0));
    }

    // 获取输入形状
    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();
    std::vector<int64_t> dims = origin.GetDims();

    // 规范 dim（支持负数）
    if (dim < 0) {
        dim += static_cast<int32_t>(dims.size());
    }

    uint32_t beforeOuter = 0, reduceLen = 0, afterInner = 0;
    ComputeBeforeAfter(dims, dim, beforeOuter, reduceLen, afterInner);

    // 简单内轴分块策略
    uint32_t tileInner = afterInner > 0 ? (afterInner < DEFAULT_TILE_INNER ? afterInner : DEFAULT_TILE_INNER) : 0;

    // 设置 BlockDim
    context->SetBlockDim(BLOCK_DIM);

    // 写入 tiling 数据
    tiling.set_beforeOuter(beforeOuter);
    tiling.set_reduceLen(reduceLen);
    tiling.set_afterInner(afterInner);
    tiling.set_tileInner(tileInner);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // 不申请 workspace
    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // 读取 dim 属性
    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    int32_t dim = 0;
    if (attrs != nullptr) {
        dim = *(attrs->GetAttrPointer<int32_t>(0));
    }

    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();
    std::vector<int64_t> dims = origin.GetDims();
    if (dim < 0) {
        dim += static_cast<int32_t>(dims.size());

    out_shape->SetOriginShape(dims);
    out_shape->SetStorageShape(dims);
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // 简化：每次处理一个切片

class KernelSumReductionOverADimension {
public:
    __aicore__ inline KernelSumReductionOverADimension() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t beforeOuter, uint32_t reduceLen, uint32_t afterInner, uint32_t tileInner)
    {
        this->beforeOuter = beforeOuter;
        this->reduceLen = reduceLen;
        this->afterInner = afterInner;
        this->tileInner = (tileInner == 0) ? afterInner : tileInner;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y);

        // 分配UB缓冲：一个累加缓冲 + 一个临时读入缓冲
        pipe.InitBuffer(accQueue, BUFFER_NUM, this->tileInner * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileInner * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        // 以outer为并行单元，均匀划分到block
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t outerStart = (beforeOuter * blockIdx) / blockNum;
        uint32_t outerEnd   = (beforeOuter * (blockIdx + 1)) / blockNum;

        for (uint32_t outer = outerStart; outer < outerEnd; ++outer) {
            // 内轴按 tileInner 分块
            for (uint32_t innerOffset = 0; innerOffset < afterInner; innerOffset += tileInner) {
                uint32_t curTile = (innerOffset + tileInner <= afterInner) ? tileInner : (afterInner - innerOffset);

                // 累加缓冲置零
                AscendC::LocalTensor<DTYPE_Y> accLocal = accQueue.AllocTensor<DTYPE_Y>();
                DTYPE_Y zeroVal(0.0f);
                AscendC::Duplicate<DTYPE_Y>(accLocal, zeroVal, curTile);

                // 沿 reduceLen 维度累加
                for (uint32_t r = 0; r < reduceLen; ++r) {
                    uint64_t xOffset = ((uint64_t)outer * reduceLen + r) * afterInner + innerOffset;
                    AscendC::LocalTensor<DTYPE_X> tmpLocal = tmpQueue.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(tmpLocal, xGm[xOffset], curTile);
                    AscendC::Add<DTYPE_Y, DTYPE_Y, DTYPE_X>(accLocal, accLocal, tmpLocal, curTile);
                    tmpQueue.FreeTensor(tmpLocal);
                }

                // 写回到输出（keepdim=True => 输出shape: [beforeOuter, 1, afterInner]）
                uint64_t yOffset = ((uint64_t)outer * /*1*/ 1 + 0) * afterInner + innerOffset;
                AscendC::DataCopy(yGm[yOffset], accLocal, curTile);
                accQueue.FreeTensor(accLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> accQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t beforeOuter;
    uint32_t reduceLen;
    uint32_t afterInner;
    uint32_t tileInner;
};

extern "C" __global__ __aicore__ void sum_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSumReductionOverADimension op;
    op.Init(x, y, tiling_data.beforeOuter, tiling_data.reduceLen, tiling_data.afterInner, tiling_data.tileInner);
    op.Process();
}
