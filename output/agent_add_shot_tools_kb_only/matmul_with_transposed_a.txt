host_operator_src="""
#include "register/op_def_registry.h"
#include "register/tilingdata_base.h"

// Matmul tiling API headers are provided by AscendC toolchain
// Use MatmulApiTiling to generate TCubeTiling for kernel
namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // A is (K, M), but kernel computes A^T * B, so logical M = A.shape[1], K = A.shape[0]
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    uint64_t a0 = a_shape->GetDim(0); // K
    uint64_t a1 = a_shape->GetDim(1); // M
    uint64_t b0 = b_shape->GetDim(0); // K
    uint64_t b1 = b_shape->GetDim(1); // N
    if (a0 != b0) {
        return ge::GRAPH_FAILED;
    }

    // Build tiling for NORMAL layout, A transposed, B not transposed, C float
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling tiling(ascendcPlatform);
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT, /*isTrans*/ true);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT, /*isTrans*/ false);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);
    // Logical matmul computes C[M, N] = (A^T)[M, K] x B[K, N]
    tiling.SetShape(static_cast<int64_t>(a1), static_cast<int64_t>(b1), static_cast<int64_t>(a0));
    tiling.SetOrgShape(static_cast<int64_t>(a1), static_cast<int64_t>(b1), static_cast<int64_t>(a0));
    tiling.SetBias(false);
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Set block dim (can also be decided by tiling internally; using a safe default)
    context->SetBlockDim(8);
    // Save TCubeTiling into raw tiling buffer
    auto raw = context->GetRawTilingData();
    tilingData.SaveToBuffer(raw->GetData(), raw->GetCapacity());
    raw->SetDataSize(tilingData.GetDataSize());

    // Workspace (not used)
    size_t *workspace_sizes = context->GetWorkspaceSizes(1);
    workspace_sizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Output shape is (M, N) = (A.shape[1], B.shape[1])
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    gert::Shape* c_shape = context->GetOutputShape(0);

    gert::Shape out;
    out.SetDimNum(2);
    out.SetDim(0, a_shape->GetDim(1));
    out.SetDim(1, b_shape->GetDim(1));
    *c_shape = out;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class MatmulWithTransposedACustom : public OpDef {
public:
    explicit MatmulWithTransposedACustom(const char* name) : OpDef(name)
    {
        this->Input("a")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("b")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("c")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(MatmulWithTransposedACustom);
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = float;
using B_T = float;
using C_T = float;

extern "C" __global__ __aicore__ void matmul_with_transposed_a_custom(GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM,
                                                                      GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // Load TCubeTiling from GM
    TCubeTiling tiling;
    auto tgm = (__gm__ uint32_t*)tilingGM;
    auto tloc = (uint32_t*)&tiling;
    for (int i = 0; i < static_cast<int>(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tgm, ++tloc) {
        *tloc = *tgm;
    }

    // Define MatmulType for NORMAL layout, A transposed at runtime
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, true,  LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, false, LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T> biasType;

    // Prepare GM tensors
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;

    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(C_T);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(cGM), sizeC);

    int offset_a = 0, offset_b = 0, offset_c = 0;

    AscendC::GlobalTensor<A_T> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal[offset_a].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreK);

    AscendC::GlobalTensor<B_T> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal[offset_b].GetPhyAddr()),
                         tiling.singleCoreK * tiling.singleCoreN);

    AscendC::GlobalTensor<C_T> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal[offset_c].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreN);

    // Create and initialize Matmul instance
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm;
    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
    mm.Init(&tiling);

    // Set A transposed, B not transposed
    mm.SetTensorA(gm_a, true);
    mm.SetTensorB(gm_b, false);
    mm.SetWorkspace(workspaceGM, 0);

    // Execute single matmul
    mm.IterateAll(gm_c);
}
