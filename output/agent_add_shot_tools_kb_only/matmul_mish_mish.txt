    uint32_t k2 = static_cast<uint32_t>(wt_shape->GetDim(0));
    uint32_t n = static_cast<uint32_t>(wt_shape->GetDim(1));
    if (k != k2) {
        return ge::GRAPH_FAILED;

    tiling.set_m(m);
    tiling.set_k(k);
    tiling.set_n(n);

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""
#include "kernel_operator.h"

class KernelMatmulMishMish {
public:
    __aicore__ inline KernelMatmulMishMish() {}
    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR wtGm, GM_ADDR biasBroadGm, GM_ADDR zGm, uint32_t m, uint32_t k, uint32_t n)
    {
        this->m = m;
        this->k = k;
        this->n = n;
        sizeA = static_cast<uint32_t>(m) * k;     // half
        sizeB = static_cast<uint32_t>(k) * n;     // half
        sizeC = static_cast<uint32_t>(m) * n;     // float
        sizeBias = static_cast<uint32_t>(m) * n;  // float (broadcasted bias)

        aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half *>(xGm), sizeA);
        bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half *>(wtGm), sizeB);
        biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(biasBroadGm), sizeBias);
        cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(zGm), sizeC);

        pipe.InitBuffer(inQueueA, 1, sizeA * sizeof(half));
        pipe.InitBuffer(inQueueB, 1, sizeB * sizeof(half));
        pipe.InitBuffer(inQueueBias, 1, sizeBias * sizeof(float));
        pipe.InitBuffer(outQueueC, 1, sizeC * sizeof(float));
        pipe.InitBuffer(tmpQueue, 1, sizeC * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<half> aLocal = inQueueA.AllocTensor<half>();
        AscendC::LocalTensor<half> bLocal = inQueueB.AllocTensor<half>();
        AscendC::LocalTensor<float> biasLocal = inQueueBias.AllocTensor<float>();

        AscendC::DataCopy(aLocal, aGlobal, sizeA);
        AscendC::DataCopy(bLocal, bGlobal, sizeB);
        AscendC::DataCopy(biasLocal, biasGlobal, sizeBias);

        inQueueA.EnQue(aLocal);
        inQueueB.EnQue(bLocal);
        inQueueBias.EnQue(biasLocal);
    }

    __aicore__ inline void MishOnce(AscendC::LocalTensor<float>& inout, AscendC::LocalTensor<float>& tmp, uint32_t count)
    {
        // tmp = exp(inout)
        AscendC::Exp(tmp, inout, count);
        // tmp = tmp + 1
        const float one = 1.0f;
        AscendC::Adds(tmp, tmp, one, static_cast<int32_t>(count));
        // tmp = log(tmp)
        AscendC::Log(tmp, tmp, count);
        // tmp = tanh(tmp)
        AscendC::Tanh(tmp, tmp, static_cast<int32_t>(count));
        // inout = inout * tmp
        AscendC::Mul(inout, inout, tmp, count);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<half> aLocal = inQueueA.DeQue<half>();
        AscendC::LocalTensor<half> bLocal = inQueueB.DeQue<half>();
        AscendC::LocalTensor<float> biasLocal = inQueueBias.DeQue<float>();
        AscendC::LocalTensor<float> cLocal = outQueueC.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();

        AscendC::GemmTiling tilling = AscendC::GetGemmTiling<half>(m, k, n);
        // matmul: [m,k] x [k,n] -> [m,n]
        AscendC::Gemm(cLocal, aLocal, bLocal, m, k, n, tilling, false, 1);

        // cLocal = cLocal + biasLocal (bias already broadcast to [m,n])
        AscendC::Add(cLocal, cLocal, biasLocal, sizeC);

        // Mish twice: c = mish(mish(c))
        MishOnce(cLocal, tmpLocal, sizeC);
        MishOnce(cLocal, tmpLocal, sizeC);

        outQueueC.EnQue(cLocal);
        inQueueA.FreeTensor(aLocal);
        inQueueB.FreeTensor(bLocal);
        inQueueBias.FreeTensor(biasLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<float> cLocal = outQueueC.DeQue<float>();
        AscendC::DataCopy(cGlobal, cLocal, sizeC);
        outQueueC.FreeTensor(cLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueA;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueB;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueBias;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueC;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue;

    AscendC::GlobalTensor<half> aGlobal;
    AscendC::GlobalTensor<half> bGlobal;
    AscendC::GlobalTensor<float> biasGlobal;
    AscendC::GlobalTensor<float> cGlobal;

    uint32_t m;
    uint32_t k;
    uint32_t n;
    uint32_t sizeA;
    uint32_t sizeB;
    uint32_t sizeC;
    uint32_t sizeBias;
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR w_t, GM_ADDR bias_broadcast, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMishMish op;
    op.Init(x, w_t, bias_broadcast, z, tiling_data.m, tiling_data.k, tiling_data.n);
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor matmul_mish_mish_custom_impl_npu(const at::Tensor& x, const at::Tensor& w, const at::Tensor& bias) {
    // x: [m, k] (float32), w: [n, k] from weight.t(), bias: [n] (float32)
    TORCH_CHECK(x.dim() == 2, "x dim must be 2");
    TORCH_CHECK(w.dim() == 2, "w dim must be 2");
    TORCH_CHECK(bias.dim() == 1, "bias dim must be 1");

    auto m = x.size(0);
    auto n = w.size(0);
    TORCH_CHECK(x.size(1) == w.size(1), "k mismatch between x and w");
    TORCH_CHECK(bias.size(0) == n, "bias size mismatch");

    // Prepare inputs: half for x and w^T, float for bias broadcast and output
    at::Tensor x_half = x.to(at::kHalf);
    at::Tensor w_half = w.to(at::kHalf).contiguous();
    at::Tensor bias_broadcast = bias.to(at::kFloat).unsqueeze(0).expand({m, n}).contiguous();
    at::Tensor result = at::empty({m, n}, x.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnMatmulMishMishCustom, x_half, w_half, bias_broadcast, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_mish_mish_custom", &matmul_mish_mish_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_mish_mish_custom", &matmul_mish_mish_custom_impl_npu, "fused matmul + mish + mish");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused MatMul + Mish + Mish using custom AscendC kernel.
    """
    def __init__(self, in_features, out_features):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features)

    def forward(self, x):
        # weight.t(): [out_features, in_features] -> [in_features, out_features]? actually linear.weight: [out, in]
        # we need [n, k] where n=out_features, k=in_features
        w_t = self.linear.weight  # shape [out, in] == [n, k]
        b = self.linear.bias      # shape [n]
        # Custom op expects: x_half [m,k], w_half [n,k], bias_broadcast [m,n] (created inside binding)
        return custom_ops_lib.matmul_mish_mish_custom(x, w_t, b)

def get_inputs():
    # randomly generate input tensors based on the model architecture
    batch_size = 1024
    in_features = 8192
    return [torch.rand(batch_size, in_features)]

def get_init_inputs():
    # randomly generate tensors required for initialization based on the model architecture
    in_features = 8192
    out_features = 8192
    return [in_features, out_features]
