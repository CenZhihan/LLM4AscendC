kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using DTYPE_X = float;
using DTYPE_Y = float;

constexpr int32_t IN_BUFFER_NUM = 2;   // 输入队列缓冲数
constexpr int32_t OUT_BUFFER_NUM = 2;  // 输出队列缓冲数
constexpr int32_t TMP_BUFFER_NUM = 3;  // 临时缓冲（平方与ReduceSum工作区）

class KernelFrobeniusNorm {
public:
    __aicore__ inline KernelFrobeniusNorm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        uint32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();

        this->tileNum = tileNum;
        this->loopCount = this->tileNum * IN_BUFFER_NUM;

        // 可变tile切分：尽量均分，其余余数前若干tile+1
        this->tileBase = this->blockLength / this->loopCount;
        this->tileTail = this->blockLength % this->loopCount;
        this->tileMax = this->tileBase + ((this->tileTail > 0) ? 1 : 0);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + blockOffset, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + blockOffset, this->blockLength);

        pipe.InitBuffer(inQueueX, IN_BUFFER_NUM, this->tileMax * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, OUT_BUFFER_NUM, this->tileMax * sizeof(DTYPE_Y));
        // 临时缓冲：平方/ReduceSum中间缓冲
        pipe.InitBuffer(tmpQueue, TMP_BUFFER_NUM, this->tileMax * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // 第一遍：计算平方和 -> Frobenius范数
        float sumSquares = 0.0f;
        uint32_t gmOffset = 0;
        for (int32_t i = 0; i < static_cast<int32_t>(this->loopCount); ++i) {
            uint32_t copyLen = this->tileBase + ((static_cast<uint32_t>(i) < this->tileTail) ? 1u : 0u);
            if (copyLen == 0) { continue; }

            // CopyIn
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[gmOffset], copyLen);
            inQueueX.EnQue(xLocal);

            // Compute partial sum of squares
            AscendC::LocalTensor<DTYPE_X> xLocalCalc = inQueueX.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Y> tmpSquare = tmpQueue.AllocTensor<DTYPE_Y>();
            AscendC::Mul(tmpSquare, xLocalCalc, xLocalCalc, copyLen); // x^2

            AscendC::LocalTensor<DTYPE_Y> dstBuf = tmpQueue.AllocTensor<DTYPE_Y>();
            AscendC::LocalTensor<DTYPE_Y> workBuf = tmpQueue.AllocTensor<DTYPE_Y>();
            AscendC::ReduceSum(dstBuf, tmpSquare, workBuf, copyLen);
            float partial = AscendC::GetAccVal<float>();
            sumSquares += partial;

            tmpQueue.FreeTensor(workBuf);
            tmpQueue.FreeTensor(dstBuf);
            tmpQueue.FreeTensor(tmpSquare);

            inQueueX.FreeTensor(xLocalCalc);

            gmOffset += copyLen;
        }

        // 计算归一化系数 (避免除0)
        const float eps = 1e-12f;
        float scale = 0.0f;
        if (sumSquares > 0.0f) {
            scale = 1.0f / sqrtf(sumSquares + eps);
        }

        // 第二遍：按相同切分输出 y = x * scale
        gmOffset = 0;
        for (int32_t i = 0; i < static_cast<int32_t>(this->loopCount); ++i) {
            uint32_t copyLen = this->tileBase + ((static_cast<uint32_t>(i) < this->tileTail) ? 1u : 0u);
            if (copyLen == 0) { continue; }

            // CopyIn
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[gmOffset], copyLen);
            inQueueX.EnQue(xLocal);

            // Compute: Muls with scalar scale
            AscendC::LocalTensor<DTYPE_X> xLocalCalc = inQueueX.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
            AscendC::Muls(yLocal, xLocalCalc, static_cast<DTYPE_Y>(scale), copyLen);

            outQueueY.EnQue<DTYPE_Y>(yLocal);
            inQueueX.FreeTensor(xLocalCalc);

            // CopyOut
            AscendC::LocalTensor<DTYPE_Y> yLocalOut = outQueueY.DeQue<DTYPE_Y>();
            AscendC::DataCopy(yGm[gmOffset], yLocalOut, copyLen);
            outQueueY.FreeTensor(yLocalOut);

            gmOffset += copyLen;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, TMP_BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t loopCount;
    uint32_t tileBase;
    uint32_t tileTail;
    uint32_t tileMax;
};

extern "C" __global__ __aicore__ void frobenius_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFrobeniusNorm op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
