#include "lstm_custom_tiling.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    if (context == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const gert::Shape* in_shape = context->GetInputShape(0);
    if (in_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }
    const auto& dims = in_shape->GetOriginShape().GetDims();
    if (dims.size() < 3) {
        return ge::GRAPH_FAILED;
    }

    uint32_t batchSize  = static_cast<uint32_t>(dims[0]);
    uint32_t seqLen     = static_cast<uint32_t>(dims[1]);
    uint32_t hiddenSize = static_cast<uint32_t>(dims[2]);

    // 配置BlockDim，按batch并行
    uint32_t blockDim = std::min<uint32_t>(batchSize, 32);
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    LstmCustomTilingData tiling;
    tiling.set_batchSize(batchSize);
    tiling.set_seqLen(seqLen);
    tiling.set_hiddenSize(hiddenSize);

    // 简单切分hidden维做流水拷贝
    uint32_t tileNum = 8;
    if (hiddenSize < 512) tileNum = 2;
    tiling.set_tileNum(tileNum);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // 无workspace需求
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    if (context == nullptr) {
        return GRAPH_FAILED;
    }
    const gert::Shape* x_shape = context->GetInputShape(0);
    if (x_shape == nullptr) {
        return GRAPH_FAILED;
    }
    const auto& dims = x_shape->GetOriginShape().GetDims();
    if (dims.size() < 3) {
        return GRAPH_FAILED;
    }
    // 输出形状为 (batch, hidden)
    std::vector<int64_t> out_dims = {dims[0], dims[2]};
    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLstmLastStep {
public:
    __aicore__ inline KernelLstmLastStep() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t seqLen,
                                uint32_t hiddenSize, uint32_t tileNum)
    {
        this->batchSize = batchSize;
        this->seqLen = seqLen;
        this->hiddenSize = hiddenSize;
        this->tileNum = tileNum;

        uint32_t totalX = batchSize * seqLen * hiddenSize;
        uint32_t totalY = batchSize * hiddenSize;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), totalX);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), totalY);

        // 每个block处理的batch范围
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->batchPerBlock = (batchSize + blockNum - 1) / blockNum;
        this->startBatch = blockIdx * this->batchPerBlock;
        uint32_t endBatchCalc = this->startBatch + this->batchPerBlock;
        this->endBatch = endBatchCalc > batchSize ? batchSize : endBatchCalc;

        // 计算tile长度，避免为0
        uint32_t chunks = tileNum * BUFFER_NUM;
        this->tileLength = (hiddenSize + chunks - 1) / chunks;
        if (this->tileLength == 0) this->tileLength = hiddenSize;

        // 初始化一个本地缓冲区用于拷贝
        pipe.InitBuffer(tmpBuf, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (startBatch >= endBatch) {
            return;
        }
        for (uint32_t b = startBatch; b < endBatch; ++b) {
            uint32_t srcBase = b * seqLen * hiddenSize + (seqLen - 1) * hiddenSize;
            uint32_t dstBase = b * hiddenSize;

            for (uint32_t off = 0; off < hiddenSize; off += tileLength) {
                uint32_t len = tileLength;
                if (off + len > hiddenSize) {
                    len = hiddenSize - off;
                }
                AscendC::LocalTensor<float> tmpLocal = tmpBuf.Get<float>();
                AscendC::DataCopy(tmpLocal, xGm[srcBase + off], len);
                AscendC::DataCopy(yGm[dstBase + off], tmpLocal, len);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t batchSize;
    uint32_t seqLen;
    uint32_t hiddenSize;
    uint32_t tileNum;

    uint32_t batchPerBlock;
    uint32_t startBatch;
    uint32_t endBatch;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void lstm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLstmLastStep op;
    op.Init(x, y,
            tiling_data.batchSize,
            tiling_data.seqLen,
            tiling_data.hiddenSize,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, output_size, dropout=0.0):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.lstm = torch.nn.LSTM(input_size, hidden_size, num_layers,
                                  batch_first=True, dropout=dropout, bidirectional=False)
        self.fc = torch.nn.Linear(hidden_size, output_size)

    def forward(self, x, h0=None, c0=None):
        batch_size = x.size(0)
        if h0 is None:
            h0 = torch.randn(self.num_layers, batch_size, self.hidden_size, device=x.device, dtype=x.dtype)
        if c0 is None:
            c0 = torch.randn(self.num_layers, batch_size, self.hidden_size, device=x.device, dtype=x.dtype)

        out, _ = self.lstm(x, (h0, c0))  # (batch, seq, hidden)
        last = custom_ops_lib.lstm_custom(out)  # (batch, hidden)
        out = self.fc(last)  # (batch, output)
        return out
'''
