#include "matmul_sigmoid_sum_custom_tiling.h"

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());

    // 解析输入形状
    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }
    // x: [B, K], weight: [N, K] (线性层参数为[out, in]，在kernel中对B进行转置处理)
    uint64_t B = static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(0));
    uint64_t K = static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(1));
    uint64_t N = static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(0));

    // 设置BlockDim（根据实际硬件可微调）
    const uint32_t BLOCK_DIM = 8;
    context->SetBlockDim(BLOCK_DIM);

    // 使用Matmul单核tiling API生成TCubeTiling
    matmul_tiling::MatmulApiTiling tiling(ascendcPlatform);
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetShape(static_cast<uint32_t>(B), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    tiling.SetOrgShape(static_cast<uint32_t>(B), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    tiling.SetBias(true);
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret == -1) {
        return ge::GRAPH_FAILED;
    }

    // 将TCubeTiling序列化到RawTilingData
    auto *raw = context->GetRawTilingData();
    if (raw == nullptr || raw->GetCapacity() < sizeof(optiling::TCubeTiling)) {
        return ge::GRAPH_FAILED;
    }
    std::memcpy(raw->GetData(), &tilingData, sizeof(optiling::TCubeTiling));
    raw->SetDataSize(sizeof(optiling::TCubeTiling));

    // 设置workspace：存放C = Matmul结果 (B x N)
    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = static_cast<size_t>(B * N * sizeof(float));

    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = float;   // x
using B_T = float;   // weight
using C_T = float;   // matmul result
using BiasT = float; // bias

// 简单Sigmoid实现：dst = 1 / (1 + exp(-src))
__aicore__ inline void SigmoidLocal(const AscendC::LocalTensor<float>& dst, const AscendC::LocalTensor<float>& src, uint32_t len) {
    // 使用高阶API，如果Sigmoid不可用可替换为近似实现
    AscendC::Sigmoid(dst, src, len);
}

extern "C" __global__ __aicore__ void matmul_sigmoid_sum_custom(GM_ADDR xGM, GM_ADDR wGM, GM_ADDR yGM, GM_ADDR biasGM, GM_ADDR tilingGM, GM_ADDR workspaceGM)
{
    set_atomic_none();

    // 读取TCubeTiling
    TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < sizeof(TCubeTiling) / sizeof(int32_t); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // 构造GM张量
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;
    AscendC::GlobalTensor<BiasT> biasGlobal;
    AscendC::GlobalTensor<C_T> yGlobal;

    // NORMAL布局：x: [B, K], w: [N, K], c: [B, N], bias: [N], y: [B, 1]
    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(C_T);
    int32_t sizeBias = tiling.CLayoutInfoB * tiling.singleCoreN * sizeof(BiasT);
    int32_t sizeY = tiling.CLayoutInfoB * tiling.singleCoreM * sizeof(C_T);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(xGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(wGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(workspaceGM), sizeC); // 中间结果存workspace
    biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ BiasT*>(biasGM), sizeBias);
    yGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(yGM), sizeY);

    // 创建Matmul实例（NORMAL）
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
    // PyTorch Linear的权重为[out, in]，将B设置为转置使其参与计算为KxN
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true, LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

    AscendC::TPipe pipe;
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
    mm.Init(&tiling);
    mm.SetTensorA(aGlobal, false);
    mm.SetTensorB(bGlobal, true);
    mm.SetWorkspace(workspaceGM, 0);
    if (tiling.isBias) {
        mm.SetBias(biasGlobal);
    }
    // 执行单次批处理Matmul
    mm.IterateBatch(cGlobal, 1, 1, false);

    // 后处理：对每个样本行做Sigmoid并按列求和 => 输出[B, 1]
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue;
    pipe.InitBuffer(inQueue, 1, tiling.singleCoreN * sizeof(C_T));
    pipe.InitBuffer(tmpQueue, 1, tiling.singleCoreN * sizeof(C_T));
    pipe.InitBuffer(outQueue, 1, sizeof(C_T)); // 每次输出一个标量

    for (uint32_t row = 0; row < tiling.singleCoreM; ++row) {
        // 从C的第row行读取到Local
        AscendC::LocalTensor<C_T> rowLocal = inQueue.AllocTensor<C_T>();
        // 通过workspaceGM指针计算行起始位置
        __gm__ C_T* cBase = reinterpret_cast<__gm__ C_T*>(workspaceGM);
        __gm__ C_T* rowPtr = cBase + row * tiling.singleCoreN;
        AscendC::GlobalTensor<C_T> rowGlobal;
        rowGlobal.SetGlobalBuffer(rowPtr, tiling.singleCoreN);
        AscendC::DataCopy(rowLocal, rowGlobal, tiling.singleCoreN);
        inQueue.EnQue(rowLocal);

        // Sigmoid
        AscendC::LocalTensor<C_T> rowLocalIn = inQueue.DeQue<C_T>();
        AscendC::LocalTensor<C_T> rowLocalOut = tmpQueue.AllocTensor<C_T>();
        SigmoidLocal(rowLocalOut, rowLocalIn, tiling.singleCoreN);

        // ReduceSum得到标量
        AscendC::LocalTensor<C_T> reduceWork = tmpQueue.AllocTensor<C_T>();
        AscendC::LocalTensor<C_T> reduceDst = outQueue.AllocTensor<C_T>(); // 作为占位符
        AscendC::ReduceSum(reduceDst, rowLocalOut, reduceWork, tiling.singleCoreN);
        C_T acc = AscendC::GetAccVal<C_T>();

        // 写回结果到y的第row个元素
        AscendC::LocalTensor<C_T> yScalarLocal = outQueue.AllocTensor<C_T>();
        // 将标量写入Local，再拷贝到GM
        // 这里假设LocalTensor支持直接索引赋值
        yScalarLocal.SetValue(0, acc);
        AscendC::GlobalTensor<C_T> yRowGlobal;
        __gm__ C_T* yBase = reinterpret_cast<__gm__ C_T*>(yGM);
        yRowGlobal.SetGlobalBuffer(yBase + row, 1);
        AscendC::DataCopy(yRowGlobal, yScalarLocal, 1);

        // 释放队列资源
        inQueue.FreeTensor(rowLocalIn);
        tmpQueue.FreeTensor(rowLocalOut);
        tmpQueue.FreeTensor(reduceWork);
        outQueue.FreeTensor(reduceDst);
        outQueue.FreeTensor(yScalarLocal);
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_size):
        super(ModelNew, self).__init__()
        # 与原始Linear等价的参数定义
        self.weight = torch.nn.Parameter(torch.randn(hidden_size, input_size))
        self.bias = torch.nn.Parameter(torch.randn(hidden_size))
