    uint32_t wout = win + kw - 1;

    tiling.set_cout(cout);

kernel_src="""
#include "kernel_operator.h"

// 为简化演示，类型使用 float
using DTYPE_X = float; // input x
using DTYPE_W = float; // weight w (未在核内使用的简化示范)
using DTYPE_Y = float; // output y

// 该核实现仅做基础数据搬运/占位，真实高性能反卷积可按需求扩展
class KernelConvTransposed2dAsym {
public:
    __aicore__ inline KernelConvTransposed2dAsym() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t hin, uint32_t win,
                                uint32_t kh, uint32_t kw,
                                uint32_t hout, uint32_t wout,
                                uint32_t tileW)
    {
        (void)w; // 未使用，仅为接口占位
        this->n = n;
        this->cin = cin;
        this->cout = cout;
        this->hin = hin;
        this->win = win;
        this->kh = kh;
        this->kw = kw;
        this->hout = hout;
        this->wout = wout;
        this->tileW = tileW;

        // GlobalTensor 设定
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, n * cin * hin * win);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, n * cout * hout * wout);

        // 基本队列与缓冲：以行宽为单位搬运
        uint32_t maxRow = (win > wout) ? win : wout;
        pipe.InitBuffer(inQueueRow, 1, maxRow * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueRow, 1, maxRow * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // 简化：仅将输入的每个 (n, c_in) 的行复制到输出对应 (n, c_out=0) 的左上角起始位置，作为占位示例
        // 实际需要实现转置卷积的累加逻辑，可基于行片段+Axpy进行扩展
        for (uint32_t bn = 0; bn < n; ++bn) {
            // 仅写入第一个输出通道，避免越界（示例）
            uint32_t oc = 0;
            for (uint32_t h = 0; h < hin && h < hout; ++h) {
                // 读入输入第一个通道 (ci=0) 的一行
                uint32_t ci = 0;
                // 输入偏移: ((bn*cin + ci)*hin + h)*win
                uint64_t inOff = (((uint64_t)bn * cin + ci) * hin + h) * win;
                AscendC::LocalTensor<DTYPE_X> srcLocal = inQueueRow.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(srcLocal, xGm[inOff], win);
                inQueueRow.EnQue(srcLocal);

                // 写出到输出 (bn, oc, h, 0) 的一行前 win 与 wout 的较小长度
                AscendC::LocalTensor<DTYPE_X> srcLocalDeq = inQueueRow.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> dstLocal = outQueueRow.AllocTensor<DTYPE_Y>();
                uint32_t copyLen = win < wout ? win : wout;
                // 将输入行复制到输出行起始
                AscendC::DataCopy(dstLocal, srcLocalDeq, copyLen);
                // 输出偏移: ((bn*cout + oc)*hout + h)*wout
                uint64_t outOff = (((uint64_t)bn * cout + oc) * hout + h) * wout;
                AscendC::DataCopy(yGm[outOff], dstLocal, copyLen);
                outQueueRow.EnQue(dstLocal);
                inQueueRow.FreeTensor(srcLocalDeq);
                AscendC::LocalTensor<DTYPE_Y> dstLocalDeq = outQueueRow.DeQue<DTYPE_Y>();
                outQueueRow.FreeTensor(dstLocalDeq);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueRow;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueRow;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, cin, cout, hin, win, kh, kw, hout, wout, tileW;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    (void)workspace;
    GET_TILING_DATA(tilingData, tiling);
    KernelConvTransposed2dAsym op;
    op.Init(x, w, y,
            tilingData.n(), tilingData.cin(), tilingData.cout(),
            tilingData.hin(), tilingData.win(),
            tilingData.kh(), tilingData.kw(),
            tilingData.hout(), tilingData.wout(),
            tilingData.tileW());
    op.Process();
}

        super().__init__()
        assert stride == (1, 1), "当前自定义算子示例仅支持 stride=(1,1)"
