kernel_src="""
#include "kernel_operator.h"

class KernelShuffle {
public:
    __aicore__ inline KernelShuffle() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR shape, GM_ADDR g, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLen)
    {
        totalLen = totalLength;
        tileSize = tileLen;

        xGm.SetGlobalBuffer((__gm__ float*)x, totalLen);
        yGm.SetGlobalBuffer((__gm__ float*)y, totalLen);

        // 读取shape与groups
        shapeGm.SetGlobalBuffer((__gm__ int32_t*)shape, 4);
        gGm.SetGlobalBuffer((__gm__ int32_t*)g, 1);

        pipe.InitBuffer(bufQueue, 1, tileSize * sizeof(float));
        pipe.InitBuffer(auxQueue, 1, 4 * sizeof(int32_t));

        // 将shape (N,C,H,W) 与 g 读入UB
        AscendC::LocalTensor<int32_t> shapeLocal = auxQueue.AllocTensor<int32_t>();
        AscendC::DataCopy(shapeLocal, shapeGm, 4);
        N = static_cast<uint32_t>(shapeLocal.GetValue(0));
        C = static_cast<uint32_t>(shapeLocal.GetValue(1));
        H = static_cast<uint32_t>(shapeLocal.GetValue(2));
        W = static_cast<uint32_t>(shapeLocal.GetValue(3));
        auxQueue.FreeTensor(shapeLocal);

        AscendC::LocalTensor<int32_t> gLocal = auxQueue.AllocTensor<int32_t>();
        AscendC::DataCopy(gLocal, gGm, 1);
        groups = static_cast<uint32_t>(gLocal.GetValue(0));
        auxQueue.FreeTensor(gLocal);

        Cg = C / groups;
        HW = H * W;
    }

    __aicore__ inline void Process()
    {
        // NCHW 连续内存；对每个n、每个输出通道c_out，找到输入通道src_c并按HW分块搬运
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t c_out = 0; c_out < C; ++c_out) {
                uint32_t src_c = (c_out % Cg) * groups + (c_out / Cg);

                uint64_t src_plane_off = (static_cast<uint64_t>(n) * C + src_c) * HW;
                uint64_t dst_plane_off = (static_cast<uint64_t>(n) * C + c_out) * HW;

                uint64_t remaining = HW;
                uint64_t prog = 0;
                while (remaining > 0) {
                    uint32_t step = remaining > tileSize ? tileSize : static_cast<uint32_t>(remaining);

                    AscendC::LocalTensor<float> tmp = bufQueue.AllocTensor<float>();
                    AscendC::DataCopy(tmp, xGm[src_plane_off + prog], step);
                    AscendC::DataCopy(yGm[dst_plane_off + prog], tmp, step);
                    bufQueue.FreeTensor(tmp);

                    prog += step;
                    remaining -= step;
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> auxQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> bufQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<int32_t> shapeGm;
    AscendC::GlobalTensor<int32_t> gGm;

    uint32_t totalLen = 0;
    uint32_t tileSize = 0;

    uint32_t N = 0, C = 0, H = 0, W = 0, HW = 0;
    uint32_t groups = 1, Cg = 1;
};

extern "C" __global__ __aicore__ void shufflenet_custom(GM_ADDR x, GM_ADDR shape, GM_ADDR g, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelShuffle op;
    op.Init(x, shape, g, y, tiling_data.totalLength, tiling_data.tileLen);
    op.Process();
}
"""

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor shufflenet_custom_impl_npu(const at::Tensor& x, int64_t groups) {
    TORCH_CHECK(x.dim() == 4, "shufflenet_custom expects 4D NCHW tensor");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "shufflenet_custom expects float32 input");
    auto sizes = x.sizes();
    int64_t N = sizes[0], C = sizes[1], H = sizes[2], W = sizes[3];
    TORCH_CHECK(C % groups == 0, "channels must be divisible by groups");

    at::Tensor y = at::empty_like(x);
    at::Tensor shape = at::empty({4}, at::dtype(at::kInt).device(x.device()));
    at::Tensor g = at::empty({1}, at::dtype(at::kInt).device(x.device()));

    // 填充shape与g
    auto shape_acc = shape.cpu();
    auto g_acc = g.cpu();
    auto* sp = shape_acc.data_ptr<int32_t>();
    sp[0] = static_cast<int32_t>(N);
    sp[1] = static_cast<int32_t>(C);
    sp[2] = static_cast<int32_t>(H);
    sp[3] = static_cast<int32_t>(W);
    auto* gp = g_acc.data_ptr<int32_t>();
    gp[0] = static_cast<int32_t>(groups);
    shape.copy_(shape_acc);
    g.copy_(g_acc);

    EXEC_NPU_CMD(aclnnShufflenetCustom, x, shape, g, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("shufflenet_custom", &shufflenet_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("shufflenet_custom", &shufflenet_custom_impl_npu, "Channel Shuffle");

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ChannelShuffleCustom(nn.Module):
    def __init__(self, groups: int):
        super().__init__()
        self.groups = groups

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.shufflenet_custom(x, self.groups)

class ShuffleNetUnit(nn.Module):
    def __init__(self, in_channels, out_channels, groups=3):
        super(ShuffleNetUnit, self).__init__()
        assert out_channels % 4 == 0
        mid_channels = out_channels // 4

        self.conv1 = nn.Conv2d(in_channels, mid_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn1 = nn.BatchNorm2d(mid_channels)

        self.conv2 = nn.Conv2d(mid_channels, mid_channels, kernel_size=3, stride=1, padding=1, groups=mid_channels, bias=False)
        self.bn2 = nn.BatchNorm2d(mid_channels)

        self.conv3 = nn.Conv2d(mid_channels, out_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn3 = nn.BatchNorm2d(out_channels)

        self.shuffle = ChannelShuffleCustom(groups)

        if in_channels == out_channels:
            self.shortcut = nn.Sequential()
        else:
            self.shortcut = nn.Sequential(
                nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=False),
                nn.BatchNorm2d(out_channels)
            )

    def forward(self, x):
        out = F.relu(self.bn1(self.conv1(x)))
        out = self.bn2(self.conv2(out))
        out = self.shuffle(out)
        out = F.relu(self.bn3(self.conv3(out)))
        out = out + self.shortcut(x)
        return out

class ModelNew(nn.Module):
    def __init__(self, num_classes=1000, groups=3, stages_repeats=[3, 7, 3], stages_out_channels=[24, 240, 480, 960]):
        super(ModelNew, self).__init__()
        self.conv1 = nn.Conv2d(3, stages_out_channels[0], kernel_size=3, stride=2, padding=1, bias=False)
        self.bn1 = nn.BatchNorm2d(stages_out_channels[0])
        self.maxpool = nn.MaxPool2d(kernel_size=3, stride=2, padding=1)

        self.stage2 = self._make_stage(stages_out_channels[0], stages_out_channels[1], stages_repeats[0], groups)
        self.stage3 = self._make_stage(stages_out_channels[1], stages_out_channels[2], stages_repeats[1], groups)
        self.stage4 = self._make_stage(stages_out_channels[2], stages_out_channels[3], stages_repeats[2], groups)

        self.conv5 = nn.Conv2d(stages_out_channels[3], 1024, kernel_size=1, stride=1, padding=0, bias=False)
        self.bn5 = nn.BatchNorm2d(1024)
        self.fc = nn.Linear(1024, num_classes)

    def _make_stage(self, in_channels, out_channels, repeats, groups):
        layers = []
        layers.append(ShuffleNetUnit(in_channels, out_channels, groups))
        for _ in range(1, repeats):
            layers.append(ShuffleNetUnit(out_channels, out_channels, groups))
        return nn.Sequential(*layers)

    def forward(self, x):
        x = F.relu(self.bn1(self.conv1(x)))
        x = self.maxpool(x)

        x = self.stage2(x)
        x = self.stage3(x)
        x = self.stage4(x)

        x = F.relu(self.bn5(self.conv5(x)))
        x = F.adaptive_avg_pool2d(x, (1, 1))
        x = x.view(x.size(0), -1)
        x = self.fc(x)
        return x
'''
