#include "max_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline void Get3dDims(const gert::Shape& inShape, uint32_t &d0, uint32_t &d1, uint32_t &d2) {
    // 假设输入为最多3维，若不足则左侧补1
    int32_t dimNum = inShape.GetDimNum();
    if (dimNum == 1) {
        d0 = 1; d1 = 1; d2 = static_cast<uint32_t>(inShape.GetDim(0));
    } else if (dimNum == 2) {
        d0 = 1; d1 = static_cast<uint32_t>(inShape.GetDim(0)); d2 = static_cast<uint32_t>(inShape.GetDim(1));
    } else {
        d0 = static_cast<uint32_t>(inShape.GetDim(0));
        d1 = static_cast<uint32_t>(inShape.GetDim(1));
        d2 = static_cast<uint32_t>(inShape.GetDim(2));
    }
}
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 读取输入形状，转换为3D
    const gert::Shape& xShape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape& yShape = context->GetInputShape(1)->GetOriginShape();
    uint32_t ax0, ax1, ax2; Get3dDims(xShape, ax0, ax1, ax2);
    uint32_t by0, by1, by2; Get3dDims(yShape, by0, by1, by2);

    // 广播后的输出形状
    uint32_t dz0 = std::max(ax0, by0);
    uint32_t dz1 = std::max(ax1, by1);
    uint32_t dz2 = std::max(ax2, by2);

    // 这里选择最保守的tile方案，沿最后一维按1元素计算，便于处理广播
    uint32_t tileK = 1;

    MaxBroadcastCustomTilingData tiling;
    tiling.set_d0(dz0);
    tiling.set_d1(dz1);
    tiling.set_d2(dz2);
    tiling.set_a0(ax0);
    tiling.set_a1(ax1);
    tiling.set_a2(ax2);
    tiling.set_b0(by0);
    tiling.set_b1(by1);
    tiling.set_b2(by2);
    tiling.set_tileK(tileK);

    context->SetBlockDim(1);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* y_shape = context->GetInputShape(1);
    const gert::Shape& xs = x_shape->GetOriginShape();
    const gert::Shape& ys = y_shape->GetOriginShape();

    auto dim_max = [](int64_t a, int64_t b) -> int64_t { return a > b ? a : b; };

    // 统一到3D推导
    int32_t xd = xs.GetDimNum(), yd = ys.GetDimNum();
    int32_t od = std::max(xd, yd);
    int64_t xd0=1, xd1=1, xd2=1;
    int64_t yd0=1, yd1=1, yd2=1;
    if (xd == 1) { xd2 = xs.GetDim(0); }
    else if (xd == 2) { xd1 = xs.GetDim(0); xd2 = xs.GetDim(1); }
    else if (xd >= 3) { xd0 = xs.GetDim(0); xd1 = xs.GetDim(1); xd2 = xs.GetDim(2); }
    if (yd == 1) { yd2 = ys.GetDim(0); }
    else if (yd == 2) { yd1 = ys.GetDim(0); yd2 = ys.GetDim(1); }
    else if (yd >= 3) { yd0 = ys.GetDim(0); yd1 = ys.GetDim(1); yd2 = ys.GetDim(2); }

    int64_t oz0 = dim_max(xd0, yd0);
    int64_t oz1 = dim_max(xd1, yd1);
    int64_t oz2 = dim_max(xd2, yd2);

    // 输出保持为3D（与kernel假设一致）
    gert::Shape* out_shape = context->GetOutputShape(0);
    out_shape->SetDimNum(3);
    out_shape->SetDim(0, oz0);
    out_shape->SetDim(1, oz1);
    out_shape->SetDim(2, oz2);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMaxBroadcast {
public:
    __aicore__ inline KernelMaxBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t d0, uint32_t d1, uint32_t d2,
                                uint32_t a0, uint32_t a1, uint32_t a2,
                                uint32_t b0, uint32_t b1, uint32_t b2,
                                uint32_t tileK)
    {
        this->d0 = d0; this->d1 = d1; this->d2 = d2;
        this->a0 = a0; this->a1 = a1; this->a2 = a2;
        this->b0 = b0; this->b1 = b1; this->b2 = b2;
        this->tileK = tileK;

        uint32_t aTotal = a0 * a1 * a2;
        uint32_t bTotal = b0 * b1 * b2;
        uint32_t zTotal = d0 * d1 * d2;

        aGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x), aTotal);
        bGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(y), bTotal);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(z), zTotal);

        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileK * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < d0; ++i) {
            for (uint32_t j = 0; j < d1; ++j) {
                for (uint32_t k = 0; k < d2; k += tileK) {
                    uint32_t curK = (k + tileK <= d2) ? tileK : (d2 - k);
                    CopyIn(i, j, k, curK);
                    Compute(curK);
                    CopyOut(i, j, k, curK);
                }
            }
        }
    }

private:
    __aicore__ inline uint32_t OffsetA(uint32_t i, uint32_t j, uint32_t k) const
    {
        uint32_t ai = (a0 == 1) ? 0 : i;
        uint32_t aj = (a1 == 1) ? 0 : j;
        uint32_t ak = (a2 == 1) ? 0 : k;
        return (ai * a1 * a2) + (aj * a2) + ak;
    }

    __aicore__ inline uint32_t OffsetB(uint32_t i, uint32_t j, uint32_t k) const
    {
        uint32_t bi = (b0 == 1) ? 0 : i;
        uint32_t bj = (b1 == 1) ? 0 : j;
        uint32_t bk = (b2 == 1) ? 0 : k;
        return (bi * b1 * b2) + (bj * b2) + bk;
    }

    __aicore__ inline uint32_t OffsetZ(uint32_t i, uint32_t j, uint32_t k) const
    {
        return (i * d1 * d2) + (j * d2) + k;
    }

    __aicore__ inline void CopyIn(uint32_t i, uint32_t j, uint32_t k, uint32_t curK)
    {
        AscendC::LocalTensor<float> aLocal = inQueueA.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB.AllocTensor<float>();

        // 由于tileK=1，这里每次拷贝1个元素，方便处理任意维度的广播
        AscendC::DataCopy(aLocal, aGm[OffsetA(i, j, k)], curK);
        AscendC::DataCopy(bLocal, bGm[OffsetB(i, j, k)], curK);

        inQueueA.EnQue(aLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t curK)
    {
        AscendC::LocalTensor<float> aLocal = inQueueA.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

        AscendC::Max(zLocal, aLocal, bLocal, static_cast<int32_t>(curK));

        outQueueZ.EnQue<float>(zLocal);
        inQueueA.FreeTensor(aLocal);
        inQueueB.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(uint32_t i, uint32_t j, uint32_t k, uint32_t curK)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm[OffsetZ(i, j, k)], zLocal, curK);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<float> aGm, bGm, zGm;

    uint32_t d0, d1, d2;
    uint32_t a0, a1, a2;
    uint32_t b0, b1, b2;
    uint32_t tileK;
};

extern "C" __global__ __aicore__ void max_broadcast_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxBroadcast op;
    op.Init(x, y, z,
            tiling_data.d0, tiling_data.d1, tiling_data.d2,
            tiling_data.a0, tiling_data.a1, tiling_data.a2,
            tiling_data.b0, tiling_data.b1, tiling_data.b2,
            tiling_data.tileK);
    op.Process();
}
