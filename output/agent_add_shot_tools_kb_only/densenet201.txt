kernel_src="""
#include "kernel_operator.h"

// 基于NCHW按通道拼接：z = concat(x, y) on dim=1
// 假设输入输出均为float32 ND，连续存储

constexpr int32_t BUFFER_NUM = 2;

class KernelConcatC {
public:
    __aicore__ inline KernelConcatC() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t n, uint32_t c0, uint32_t c1, uint32_t h, uint32_t w,
                                uint32_t tileNum)
    {
        this->n = n;
        this->c0 = c0;
        this->c1 = c1;
        this->h = h;
        this->w = w;
        this->tileNum = tileNum;

        this->xBatchSize = static_cast<uint64_t>(c0) * h * w;
        this->yBatchSize = static_cast<uint64_t>(c1) * h * w;
        this->zBatchSize = static_cast<uint64_t>(c0 + c1) * h * w;

        // 每批按元素tile，粗略划分
        uint64_t perBatch = zBatchSize;
        uint64_t denom = static_cast<uint64_t>(tileNum) * BUFFER_NUM;
        this->tileLen = static_cast<uint32_t>((perBatch + denom - 1) / denom);
        if (this->tileLen == 0) this->tileLen = 1;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->nStart = (static_cast<uint64_t>(n) * blockIdx) / blockNum;
        this->nEnd   = (static_cast<uint64_t>(n) * (blockIdx + 1)) / blockNum;
        this->localN = (nEnd > nStart) ? (nEnd - nStart) : 0;

        // 绑定GM起始地址到本核片段
        xGm.SetGlobalBuffer((__gm__ float *)x + nStart * xBatchSize, localN * xBatchSize);
        yGm.SetGlobalBuffer((__gm__ float *)y + nStart * yBatchSize, localN * yBatchSize);
        zGm.SetGlobalBuffer((__gm__ float *)z + nStart * zBatchSize, localN * zBatchSize);

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLen * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint64_t b = 0; b < localN; ++b) {
            // 1) 拷贝x到z（每批的前c0*h*w段）
            CopyContiguous(xGm, b * xBatchSize, zGm, b * zBatchSize, xBatchSize);
            // 2) 拷贝y到z（每批的后c1*h*w段）
            CopyContiguous(yGm, b * yBatchSize, zGm, b * zBatchSize + xBatchSize, yBatchSize);
        }
    }

private:
    __aicore__ inline void CopyContiguous(AscendC::GlobalTensor<float>& srcGm, uint64_t srcOff,
                                          AscendC::GlobalTensor<float>& dstGm, uint64_t dstOff,
                                          uint64_t totalElems)
    {
        uint64_t copied = 0;
        while (copied < totalElems) {
            uint32_t curr = static_cast<uint32_t>(AscendC::min<uint64_t>(tileLen, totalElems - copied));
            AscendC::LocalTensor<float> buf = inQueue.AllocTensor<float>();
            AscendC::DataCopy(buf, srcGm[srcOff + copied], curr);
            inQueue.EnQue(buf);

            AscendC::LocalTensor<float> buf2 = inQueue.DeQue<float>();
            AscendC::DataCopy(dstGm[dstOff + copied], buf2, curr);
            inQueue.FreeTensor(buf2);

            copied += curr;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t n, c0, c1, h, w, tileNum;
    uint64_t xBatchSize, yBatchSize, zBatchSize;
    uint32_t tileLen;
    uint64_t nStart, nEnd, localN;
};

extern "C" __global__ __aicore__ void densenet201_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConcatC op;
    op.Init(x, y, z,
            tiling_data.n, tiling_data.c0, tiling_data.c1, tiling_data.h, tiling_data.w,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class DenseBlockNew(nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        super(DenseBlockNew, self).__init__()
        layers = []
        for i in range(num_layers):
            layers.append(self._make_layer(num_input_features + i * growth_rate, growth_rate))
        self.layers = nn.ModuleList(layers)

    def _make_layer(self, in_features: int, growth_rate: int):
        return nn.Sequential(
            nn.BatchNorm2d(in_features),
            nn.ReLU(inplace=True),
            nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
            nn.Dropout(0.0)
        )

    def forward(self, x):
        # 逐层拼接：x_{k+1} = concat(x_k, new_feature_k) 沿通道维度
        for layer in self.layers:
            new_feature = layer(x)
            x = custom_ops_lib.densenet201_custom(x, new_feature)
        return x

class TransitionLayer(nn.Module):
    def __init__(self, num_input_features: int, num_output_features: int):
        super(TransitionLayer, self).__init__()
        self.transition = nn.Sequential(
            nn.BatchNorm2d(num_input_features),
            nn.ReLU(inplace=True),
            nn.Conv2d(num_input_features, num_output_features, kernel_size=1, bias=False),
            nn.AvgPool2d(kernel_size=2, stride=2)
        )

    def forward(self, x):
        return self.transition(x)

class ModelNew(nn.Module):
    def __init__(self, growth_rate: int = 32, num_classes: int = 1000):
        super(ModelNew, self).__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 64, kernel_size=7, stride=2, padding=3, bias=False),
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(kernel_size=3, stride=2, padding=1)
        )

        num_features = 64
        block_layers = [6, 12, 48, 32]

        self.dense_blocks = nn.ModuleList()
        self.transition_layers = nn.ModuleList()

        for i, num_layers in enumerate(block_layers):
            block = DenseBlockNew(num_layers=num_layers, num_input_features=num_features, growth_rate=growth_rate)
            self.dense_blocks.append(block)
            num_features = num_features + num_layers * growth_rate

            if i != len(block_layers) - 1:
                transition = TransitionLayer(num_input_features=num_features, num_output_features=num_features // 2)
                self.transition_layers.append(transition)
                num_features = num_features // 2

        self.final_bn = nn.BatchNorm2d(num_features)
        self.classifier = nn.Linear(num_features, num_classes)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = self.features(x)
        for i, block in enumerate(self.dense_blocks):
            x = block(x)
            if i != len(self.dense_blocks) - 1:
                x = self.transition_layers[i](x)
        x = self.final_bn(x)
        x = F.relu(x, inplace=True)
        x = F.adaptive_avg_pool2d(x, (1, 1)).view(x.size(0), -1)
        x = self.classifier(x)
        return x
'''
