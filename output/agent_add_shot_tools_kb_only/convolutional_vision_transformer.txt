kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCViTCat {
public:
    __aicore__ inline KernelCViTCat() {}
    __aicore__ inline void Init(GM_ADDR cls, GM_ADDR x, GM_ADDR y, uint32_t batch, uint32_t embedDim, uint32_t tileE)
    {
        this->batch = batch;
        this->embedDim = embedDim;
        this->tileE = tileE;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();

        uint32_t base = batch / blocks;
        uint32_t rem = batch % blocks;

        if (idx < rem) {
            startB = idx * (base + 1);
            blockB = base + 1;
        } else {
            startB = rem * (base + 1) + (idx - rem) * base;
            blockB = base;
        }

        // Set GM buffers with offsets
        clsGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(cls) + startB * embedDim, blockB * embedDim);
        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x) + startB * embedDim, blockB * embedDim);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(y) + startB * 2 * embedDim, blockB * 2 * embedDim);

        pipe.InitBuffer(inQueueCls, BUFFER_NUM, this->tileE * sizeof(float));
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileE * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileE * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (blockB == 0 || embedDim == 0) {
            return;
        }
        for (uint32_t b = 0; b < blockB; ++b) {
            // write channel 0 from cls
            for (uint32_t pos = 0; pos < embedDim; pos += tileE) {
                uint32_t cur = (pos + tileE <= embedDim) ? tileE : (embedDim - pos);
                CopyInCls(b, pos, cur);
                ComputeCopyToY(b, 0, pos, cur); // channel 0
                CopyOutY(b, 0, pos, cur);
            }
            // write channel 1 from x
            for (uint32_t pos = 0; pos < embedDim; pos += tileE) {
                uint32_t cur = (pos + tileE <= embedDim) ? tileE : (embedDim - pos);
                CopyInX(b, pos, cur);
                ComputeCopyToY(b, 1, pos, cur); // channel 1
                CopyOutY(b, 1, pos, cur);
            }
        }
    }

private:
    __aicore__ inline void CopyInCls(uint32_t b, uint32_t pos, uint32_t len)
    {
        AscendC::LocalTensor<float> clsLocal = inQueueCls.AllocTensor<float>();
        AscendC::DataCopy(clsLocal, clsGm[b * embedDim + pos], len);
        inQueueCls.EnQue(clsLocal);
    }

    __aicore__ inline void CopyInX(uint32_t b, uint32_t pos, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[b * embedDim + pos], len);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeCopyToY(uint32_t b, uint32_t ch, uint32_t pos, uint32_t len)
    {
        AscendC::LocalTensor<float> srcLocal;
        if (ch == 0) {
            srcLocal = inQueueCls.DeQue<float>();
        } else {
            srcLocal = inQueueX.DeQue<float>();
        }
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        AscendC::DataCopy(yLocal, srcLocal, len);
        outQueueY.EnQue(yLocal);
        if (ch == 0) {
            inQueueCls.FreeTensor(srcLocal);
        } else {
            inQueueX.FreeTensor(srcLocal);
        }
    }

    __aicore__ inline void CopyOutY(uint32_t b, uint32_t ch, uint32_t pos, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        // y layout: [B, 2, E] flattened. Offset for channel ch is ch * E
        AscendC::DataCopy(yGm[b * 2 * embedDim + ch * embedDim + pos], yLocal, len);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueCls;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<float> clsGm;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t batch;
    uint32_t embedDim;
    uint32_t tileE;

    uint32_t startB;
    uint32_t blockB;
};

extern "C" __global__ __aicore__ void convolutional_vision_transformer_custom(GM_ADDR cls, GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCViTCat op;
    op.Init(cls, x, y, tiling_data.batch, tiling_data.embedDim, tiling_data.tileE);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_classes, embed_dim=512, num_heads=8, num_layers=6, 
                 mlp_ratio=4.0, patch_size=4, in_channels=3, image_size=32):
        super().__init__()
        self.patch_size = patch_size
        self.image_size = image_size
        self.embed_dim = embed_dim

        self.conv1 = torch.nn.Conv2d(in_channels, embed_dim, kernel_size=patch_size, stride=patch_size)
        num_patches = (image_size // patch_size) ** 2
        self.linear_proj = torch.nn.Linear(embed_dim * num_patches, embed_dim)

        self.transformer_layers = torch.nn.ModuleList([
            torch.nn.TransformerEncoderLayer(
                d_model=embed_dim,
                nhead=num_heads,
                dim_feedforward=int(embed_dim * mlp_ratio),
                dropout=0.0,
                batch_first=True
            ) for _ in range(num_layers)
        ])

        self.cls_token = torch.nn.Parameter(torch.zeros(1, 1, embed_dim))
        self.fc_out = torch.nn.Linear(embed_dim, num_classes)

    def forward(self, x):
        B = x.size(0)
        x = self.conv1(x)                 # (B, E, H/ps, W/ps)
        x = x.flatten(start_dim=1)        # (B, E * num_patches)
        x = self.linear_proj(x)           # (B, E)

        cls_tokens = self.cls_token.expand(B, -1, -1).contiguous()     # (B, 1, E)
        # Replace cat with custom kernel: (B, 2, E)
        x = custom_ops_lib.convolutional_vision_transformer_custom(cls_tokens, x)

        for layer in self.transformer_layers:
            x = layer(x)

        return self.fc_out(x[:, 0])       # (B, num_classes)
'''
