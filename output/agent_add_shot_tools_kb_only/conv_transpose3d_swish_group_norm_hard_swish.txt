#include "kernel_operator.h"

namespace CustomOp {

struct MyTiling {
  GroupNormTiling groupNormTilingData;
  float epsilon;
  uint32_t totalLength;
  uint32_t tileNum;
  uint32_t n, c, d, h, w;
  uint32_t groupNum;
};

class KernelConvTranspose3dSwishGroupNormHardSwish {
public:
  __aicore__ inline KernelConvTranspose3dSwishGroupNormHardSwish() {}
  __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, MyTiling tilingData) {
    this->tiling_ = tilingData;
    n_ = tiling_.n; c_ = tiling_.c; d_ = tiling_.d; h_ = tiling_.h; w_ = tiling_.w;
    g_ = tiling_.groupNum;
    total_ = tiling_.totalLength;

    xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), total_);
    gammaGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(gamma), c_);
    betaGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(beta), c_);
    yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), total_);

    // Allocate UB queues
    pipe_.InitBuffer(inQueueX_, 1, sizeof(float) * total_);
    pipe_.InitBuffer(inQueueGamma_, 1, ((sizeof(float) * c_ + 31) / 32) * 32);
    pipe_.InitBuffer(inQueueBeta_, 1, ((sizeof(float) * c_ + 31) / 32) * 32);
    pipe_.InitBuffer(outQueueY_, 1, sizeof(float) * total_);
    pipe_.InitBuffer(meanBuf_, ((sizeof(float) * n_ * g_ + 31) / 32) * 32);
    pipe_.InitBuffer(varBuf_, ((sizeof(float) * n_ * g_ + 31) / 32) * 32);
    pipe_.InitBuffer(tmpBuf_, 1, sizeof(float) * total_);
  }

  __aicore__ inline void Process() {
    CopyIn();
    Compute();
    CopyOut();
  }

private:
  __aicore__ inline void CopyIn() {
    auto xLocal = inQueueX_.AllocTensor<float>();
    AscendC::DataCopy(xLocal, xGm_, total_);
    inQueueX_.EnQue(xLocal);

    auto gammaLocal = inQueueGamma_.AllocTensor<float>();
    AscendC::DataCopy(gammaLocal, gammaGm_, c_);
    inQueueGamma_.EnQue(gammaLocal);

    auto betaLocal = inQueueBeta_.AllocTensor<float>();
    AscendC::DataCopy(betaLocal, betaGm_, c_);
    inQueueBeta_.EnQue(betaLocal);
  }

  __aicore__ inline void Compute() {
    auto xLocal = inQueueX_.DeQue<float>();
    auto gammaLocal = inQueueGamma_.DeQue<float>();
    auto betaLocal = inQueueBeta_.DeQue<float>();
    auto yLocal = outQueueY_.AllocTensor<float>();
    auto meanLocal = meanBuf_.Get<float>();
    auto varLocal = varBuf_.Get<float>();
    auto tmpLocal = tmpBuf_.Get<float>(); // workspace for activations

    // Step 1: Swish (SiLU) -> tmpLocal = SiLU(xLocal)
    AscendC::Silu(tmpLocal, xLocal, total_);

    // Step 2: GroupNorm -> yLocal = GN(tmpLocal)
    AscendC::GroupNorm<float, false>(yLocal, meanLocal, varLocal, tmpLocal, gammaLocal, betaLocal,
                                     (float)tiling_.epsilon, tiling_.groupNormTilingData);

    // Step 3: HardSwish: y = y * clamp(y + 3, 0, 6) / 6
    // tmpLocal = y + 3
    AscendC::Adds(tmpLocal, yLocal, (float)3.0f, total_);
    // tmpLocal = clamp_min(tmpLocal, 0)
    AscendC::ClampMin(tmpLocal, tmpLocal, (float)0.0f, total_);
    // tmpLocal = clamp_max(tmpLocal, 6)
    AscendC::ClampMax(tmpLocal, tmpLocal, (float)6.0f, total_);
    // tmpLocal = tmpLocal * (1/6)
    AscendC::Muls(tmpLocal, tmpLocal, (float)(1.0f/6.0f), total_);
    // yLocal = yLocal * tmpLocal
    AscendC::Mul(yLocal, yLocal, tmpLocal, total_);

    outQueueY_.EnQue<float>(yLocal);
    inQueueX_.FreeTensor(xLocal);
    inQueueGamma_.FreeTensor(gammaLocal);
    inQueueBeta_.FreeTensor(betaLocal);
  }

  __aicore__ inline void CopyOut() {
    auto yLocal = outQueueY_.DeQue<float>();
    AscendC::DataCopy(yGm_, yLocal, total_);
    outQueueY_.FreeTensor(yLocal);
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX_;
  AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueGamma_;
  AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueBeta_;
  AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY_;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> meanBuf_;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> varBuf_;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> tmpBuf_;
  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> gammaGm_;
  AscendC::GlobalTensor<float> betaGm_;
  AscendC::GlobalTensor<float> yGm_;
  MyTiling tiling_;
  uint32_t n_, c_, d_, h_, w_, g_;
  uint32_t total_;
};

} // namespace CustomOp

extern "C" __global__ __aicore__ void conv_transpose3d_swish_group_norm_hard_swish_custom(GM_ADDR x,
                                                                                           GM_ADDR gamma,
                                                                                           GM_ADDR beta,
                                                                                           GM_ADDR y,
                                                                                           GM_ADDR workspace,
                                                                                           GM_ADDR tiling) {
  if ASCEND_IS_AIC {
    return;
  }
  CustomOp::MyTiling t;
  CopyTiling(&t, tiling);
  CustomOp::KernelConvTranspose3dSwishGroupNormHardSwish op;
  op.Init(x, gamma, beta, y, t);
  if (TILING_KEY_IS(1)) {
    op.Process();
  }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep ConvTranspose3d; fuse Swish + GroupNorm + HardSwish into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, groups, eps, bias=True):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        self.groups = groups
        self.eps = eps
        # Create a GroupNorm module only to host learnable affine parameters (gamma/beta)
        self.group_norm = torch.nn.GroupNorm(num_groups=groups, num_channels=out_channels, eps=eps)

    def forward(self, x):
        x = self.conv_transpose(x)  # Standard ConvTranspose3d
        # Fused Swish + GroupNorm + HardSwish
        return custom_ops_lib.conv_transpose3d_swish_group_norm_hard_swish_custom(
            x, self.group_norm.weight, self.group_norm.bias, self.groups, float(self.eps)
        )
