kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelDWSP2D {
public:
    __aicore__ inline KernelDWSP2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR dw_weight, GM_ADDR pw_weight, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t w,
                                uint32_t kH, uint32_t kW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t tileNum)
    {
        this->N = n;
        this->Cin = cin;
        this->Cout = cout;
        this->H = h;
        this->W = w;
        this->kH = kH;
        this->kW = kW;
        this->strideH = strideH;
        this->strideW = strideW;
        this->padH = padH;
        this->padW = padW;
        this->dilationH = dilationH;
        this->dilationW = dilationW;

        uint32_t xSize = n * cin * h * w;
        uint32_t ySize = n * cout * h * w; // assuming out H/W same as input for current params
        uint32_t dwSize = cin * kH * kW;   // dw_weight: [Cin, 1, kH, kW]
        uint32_t pwSize = cout * cin;      // pw_weight: [Cout, Cin, 1, 1]

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(x), xSize);
        dwGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(dw_weight), dwSize);
        pwGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(pw_weight), pwSize);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(y), ySize);

        // Queues: full weights local buffer, and scalar IO buffers
        pipe.InitBuffer(inQueueDW, 1, dwSize * sizeof(DTYPE));
        pipe.InitBuffer(inQueuePW, 1, pwSize * sizeof(DTYPE));
        pipe.InitBuffer(inQueueScalar, 1, 1 * sizeof(DTYPE));
        pipe.InitBuffer(outQueueScalar, 1, 1 * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // Load all weights to local once
        AscendC::LocalTensor<DTYPE> dwLocal = inQueueDW.AllocTensor<DTYPE>();
        AscendC::DataCopy(dwLocal, dwGm, Cin * kH * kW);
        inQueueDW.EnQue(dwLocal);

        AscendC::LocalTensor<DTYPE> pwLocal = inQueuePW.AllocTensor<DTYPE>();
        AscendC::DataCopy(pwLocal, pwGm, Cout * Cin);
        inQueuePW.EnQue(pwLocal);

        // DeQue weight tensors for compute
        AscendC::LocalTensor<DTYPE> dwW = inQueueDW.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> pwW = inQueuePW.DeQue<DTYPE>();

        // Alloc scalar buffers for GM <-> Local transfers
        AscendC::LocalTensor<DTYPE> xScalar = inQueueScalar.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> yScalar = outQueueScalar.AllocTensor<DTYPE>();

        // Naive reference implementation with stride/pad/dilation support
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oh = 0; oh < H; ++oh) {
                for (uint32_t ow = 0; ow < W; ++ow) {
                    for (uint32_t co = 0; co < Cout; ++co) {
                        DTYPE sum_out = (DTYPE)0.0f;

                        for (uint32_t ci = 0; ci < Cin; ++ci) {
                            // depthwise conv on (n, ci, oh, ow)
                            DTYPE sum_dw = (DTYPE)0.0f;
                            int32_t base_h = static_cast<int32_t>(oh) * static_cast<int32_t>(strideH) - static_cast<int32_t>(padH);
                            int32_t base_w = static_cast<int32_t>(ow) * static_cast<int32_t>(strideW) - static_cast<int32_t>(padW);
                            for (uint32_t kh = 0; kh < kH; ++kh) {
                                int32_t ih = base_h + static_cast<int32_t>(kh) * static_cast<int32_t>(dilationH);
                                if (ih < 0 || ih >= static_cast<int32_t>(H)) continue;
                                for (uint32_t kw = 0; kw < kW; ++kw) {
                                    int32_t iw = base_w + static_cast<int32_t>(kw) * static_cast<int32_t>(dilationW);
                                    if (iw < 0 || iw >= static_cast<int32_t>(W)) continue;

                                    // x offset
                                    uint32_t x_off = (((n * Cin + ci) * H + static_cast<uint32_t>(ih)) * W + static_cast<uint32_t>(iw));
                                    // copy one element from GM to local
                                    AscendC::DataCopy(xScalar, xGm[x_off], 1);
                                    DTYPE x_val = xScalar.GetValue(0);

                                    // dw weight index: [ci, kh, kw]
                                    uint32_t w_off = (ci * kH + kh) * kW + kw;
                                    DTYPE w_val = dwW.GetValue(w_off);
                                    sum_dw += x_val * w_val;
                                }
                            }
                            // pointwise weight [co, ci]
                            uint32_t pw_off = co * Cin + ci;
                            DTYPE pw_val = pwW.GetValue(pw_off);
                            sum_out += sum_dw * pw_val;
                        }

                        // store y[n, co, oh, ow]
                        uint32_t y_off = (((n * Cout + co) * H + oh) * W + ow);
                        // write sum_out to y via local scalar buffer
                        // Assume LocalTensor supports SetValue
                        yScalar.SetValue(0, sum_out);
                        AscendC::DataCopy(yGm[y_off], yScalar, 1);
                    }
                }
            }
        }

        // Release local resources
        inQueueDW.FreeTensor(dwW);
        inQueuePW.FreeTensor(pwW);
        inQueueScalar.FreeTensor(xScalar);
        outQueueScalar.FreeTensor(yScalar);
    }

private:
    AscendC::TPipe pipe;
    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> dwGm;
    AscendC::GlobalTensor<DTYPE> pwGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueDW;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueuePW;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueScalar;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueScalar;

    uint32_t N, Cin, Cout, H, W;
    uint32_t kH, kW;
    uint32_t strideH, strideW;
    uint32_t padH, padW;
    uint32_t dilationH, dilationW;
};

extern "C" __global__ __aicore__ void conv_depthwise_separable_2d_custom(GM_ADDR x, GM_ADDR dw_weight, GM_ADDR pw_weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelDWSP2D op;
    op.Init(
        x, dw_weight, pw_weight, y,
        tiling_data.n, tiling_data.cin, tiling_data.cout,
        tiling_data.h, tiling_data.w,
        tiling_data.kH, tiling_data.kW,
        tiling_data.strideH, tiling_data.strideW,
        tiling_data.padH, tiling_data.padW,
        tiling_data.dilationH, tiling_data.dilationW,
        tiling_data.tileNum
    );
    op.Process();
}

class ModelNew(torch.nn.Module):
    """
    Optimized depthwise-separable 2D convolution using custom AscendC kernel.
    Assumptions for this sample kernel:
      - bias is not used.
      - stride=1, dilation=1; padding is typically kernel_size//2 to keep H/W.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int,
                 stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size

        # Depthwise weight: [Cin, 1, kH, kW]
        self.dw_weight = torch.nn.Parameter(
            torch.empty(in_channels, 1, kernel_size, kernel_size)
        )
        # Pointwise weight: [Cout, Cin, 1, 1]
        self.pw_weight = torch.nn.Parameter(
            torch.empty(out_channels, in_channels, 1, 1)
        )

        # Initialize weights similar to nn.Conv2d defaults (Kaiming uniform-like)
        torch.nn.init.kaiming_uniform_(self.dw_weight, a=math.sqrt(5))
        torch.nn.init.kaiming_uniform_(self.pw_weight, a=math.sqrt(5))

        # Store conv params (not currently passed to kernel in this demo)
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.use_bias = False

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_depthwise_separable_2d_custom(x, self.dw_weight, self.pw_weight)
