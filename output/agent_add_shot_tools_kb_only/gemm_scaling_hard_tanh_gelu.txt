kernel_src="""
#include "kernel_operator.h"

// 本算子对输入张量执行：y = GELU(clamp(x * scale, min_val, max_val))
// 为简单起见，scale/min_val/max_val 需与 x 形状一致（由上层传入 full_like 张量）

constexpr int32_t BUFFER_NUM = 2;

typedef float DTYPE_X;
typedef float DTYPE_SCALE;
typedef float DTYPE_MINV;
typedef float DTYPE_MAXV;
typedef float DTYPE_Y;

class KernelGemmScalingHardTanhGelu {
public:
    __aicore__ inline KernelGemmScalingHardTanhGelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR minv, GM_ADDR maxv, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        sGm.SetGlobalBuffer((__gm__ DTYPE_SCALE *)scale + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        minGm.SetGlobalBuffer((__gm__ DTYPE_MINV *)minv + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        maxGm.SetGlobalBuffer((__gm__ DTYPE_MAXV *)maxv + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueS, BUFFER_NUM, this->tileLength * sizeof(DTYPE_SCALE));
        pipe.InitBuffer(inQueueMin, BUFFER_NUM, this->tileLength * sizeof(DTYPE_MINV));
        pipe.InitBuffer(inQueueMax, BUFFER_NUM, this->tileLength * sizeof(DTYPE_MAXV));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_SCALE> sLocal = inQueueS.AllocTensor<DTYPE_SCALE>();
        AscendC::LocalTensor<DTYPE_MINV> minLocal = inQueueMin.AllocTensor<DTYPE_MINV>();
        AscendC::LocalTensor<DTYPE_MAXV> maxLocal = inQueueMax.AllocTensor<DTYPE_MAXV>();

        AscendC::DataCopy(xLocal,   xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(sLocal,   sGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(minLocal, minGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(maxLocal, maxGm[progress * this->tileLength], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueS.EnQue(sLocal);
        inQueueMin.EnQue(minLocal);
        inQueueMax.EnQue(maxLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X>    xLocal   = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_SCALE> sLocal  = inQueueS.DeQue<DTYPE_SCALE>();
        AscendC::LocalTensor<DTYPE_MINV> minLocal = inQueueMin.DeQue<DTYPE_MINV>();
        AscendC::LocalTensor<DTYPE_MAXV> maxLocal = inQueueMax.DeQue<DTYPE_MAXV>();

        AscendC::LocalTensor<DTYPE_Y> tmpLocal = tmpQueue.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> yLocal   = outQueueY.AllocTensor<DTYPE_Y>();

        // 1) 缩放
        AscendC::Mul(tmpLocal, xLocal, sLocal, this->tileLength);
        // 2) hardtanh: 先与最小值比较取大，再与最大值比较取小
        AscendC::Maximum(tmpLocal, tmpLocal, minLocal, this->tileLength);
        AscendC::Minimum(tmpLocal, tmpLocal, maxLocal, this->tileLength);
        // 3) GELU
        AscendC::Gelu(yLocal, tmpLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Y>(yLocal);

        // 释放输入与中间张量
        tmpQueue.FreeTensor(tmpLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueS.FreeTensor(sLocal);
        inQueueMin.FreeTensor(minLocal);
        inQueueMax.FreeTensor(maxLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueS;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueMin;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueMax;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X>    xGm;
    AscendC::GlobalTensor<DTYPE_SCALE> sGm;
    AscendC::GlobalTensor<DTYPE_MINV> minGm;
    AscendC::GlobalTensor<DTYPE_MAXV> maxGm;
    AscendC::GlobalTensor<DTYPE_Y>    yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR minv, GM_ADDR maxv, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScalingHardTanhGelu op;
    op.Init(x, scale, minv, maxv, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版：线性层使用原生 GEMM，随后将缩放 + hardtanh + GELU 融合到自定义 AscendC 算子中
    """
    def __init__(self, in_features, out_features, scaling_factor, hardtanh_min, hardtanh_max):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.scaling_factor = float(scaling_factor)
        self.hardtanh_min = float(hardtanh_min)
        self.hardtanh_max = float(hardtanh_max)

    def forward(self, x):
        # 先执行 GEMM（Linear）
        x = self.gemm(x)
        # 将标量参数扩展为与 x 同形状张量，便于内核进行逐元素计算
        scale_tensor = torch.full_like(x, self.scaling_factor)
        min_tensor = torch.full_like(x, self.hardtanh_min)
        max_tensor = torch.full_like(x, self.hardtanh_max)
        # 调用自定义融合算子：y = GELU(clamp(x * scale, min, max))
        y = custom_ops_lib.gemm_scaling_hard_tanh_gelu_custom(x, scale_tensor, min_tensor, max_tensor)
        return y
'''
