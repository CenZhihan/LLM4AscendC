#include "matmul_with_irregular_shapes_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MatmulWithIrregularShapesCustomTilingData tiling;

    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    // A: [M, K], B: [K, N]
    uint32_t m = static_cast<uint32_t>(a_shape->GetDim(0));
    uint32_t k = static_cast<uint32_t>(a_shape->GetDim(1));
    uint32_t n = static_cast<uint32_t>(b_shape->GetDim(1));

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_m(m);
    tiling.set_k(k);
    tiling.set_n(n);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    gert::Shape* c_shape = context->GetOutputShape(0);

    // A: [M, K], B: [K, N] -> C: [M, N]
    std::vector<int64_t> y_dims(2);
    y_dims[0] = a_shape->GetDim(0);
    y_dims[1] = b_shape->GetDim(1);
    c_shape->SetDim(y_dims);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // A/B: float16, C: float32
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

class KernelMatmulIrregular {
public:
    __aicore__ inline KernelMatmulIrregular() {}
    __aicore__ inline void Init(__gm__ uint8_t* aGm, __gm__ uint8_t* bGm, __gm__ uint8_t* cGm,
                                uint32_t mInput, uint32_t kInput, uint32_t nInput)
    {
        m = mInput;
        k = kInput;
        n = nInput;

        // sizes
        aSize = static_cast<uint32_t>(m) * k;
        bSize = static_cast<uint32_t>(k) * n;
        cSize = static_cast<uint32_t>(m) * n;

        // aligned sizes (for output copy)
        roundm = AscendC::DivCeil(m, 16) * 16;
        roundn = AscendC::DivCeil(n, 16) * 16;
        roundk = AscendC::DivCeil(k, c0Size) * c0Size;

        aGlobal.SetGlobalBuffer((__gm__ half*)aGm);
        bGlobal.SetGlobalBuffer((__gm__ half*)bGm);
        cGlobal.SetGlobalBuffer((__gm__ float*)cGm);

        // Init queues: copy entire matrices (note: for large shapes, real kernels should tile)
        pipe.InitBuffer(inQueueA1, 1, aSize * sizeof(half));
        pipe.InitBuffer(inQueueB1, 1, bSize * sizeof(half));
        pipe.InitBuffer(outQueueCO1, 1, cSize * sizeof(float));
        pipe.InitBuffer(outQueueUB, 1, cSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyUB();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<half> aLocal = inQueueA1.AllocTensor<half>();
        AscendC::LocalTensor<half> bLocal = inQueueB1.AllocTensor<half>();
        AscendC::DataCopy(aLocal, aGlobal, aSize);
        AscendC::DataCopy(bLocal, bGlobal, bSize);
        inQueueA1.EnQue(aLocal);
        inQueueB1.EnQue(bLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<half> aLocal = inQueueA1.DeQue<half>();
        AscendC::LocalTensor<half> bLocal = inQueueB1.DeQue<half>();
        AscendC::LocalTensor<float> cLocal = outQueueCO1.AllocTensor<float>();

        AscendC::GemmTiling tiling = AscendC::GetGemmTiling<half>(m, k, n);
        tiling.loopMode = AscendC::LoopMode::MODE_NM;

        // Single GEMM call: C = A(m,k) * B(k,n), result moved to GM; dst does not need to be initialized
        AscendC::Gemm(cLocal, aLocal, bLocal, m, k, n, tiling, false, /*initValue*/ 1);

        outQueueCO1.EnQue<float>(cLocal);
        inQueueA1.FreeTensor(aLocal);
        inQueueB1.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyUB()
    {
        AscendC::LocalTensor<float> cLocal = outQueueCO1.DeQue<float>();
        AscendC::LocalTensor<float> cUB = outQueueUB.AllocTensor<float>();

        AscendC::DataCopyParams params;
        params.blockCount = 1;
        params.blockLen = roundm * roundn * sizeof(float) / 1024;

        AscendC::DataCopyEnhancedParams enhanced;
        enhanced.blockMode = BlockMode::BLOCK_MODE_MATRIX;

        AscendC::DataCopy(cUB, cLocal, params, enhanced);
        outQueueUB.EnQue<float>(cUB);
        outQueueCO1.FreeTensor(cLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<float> cUB = outQueueUB.DeQue<float>();
        AscendC::DataCopy(cGlobal, cUB, cSize);
        outQueueUB.FreeTensor(cUB);
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::QuePosition::A1, 1> inQueueA1;
    AscendC::TQue<AscendC::QuePosition::B1, 1> inQueueB1;
    AscendC::TQue<AscendC::QuePosition::CO1, 1> outQueueCO1;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueUB;

    AscendC::GlobalTensor<half> aGlobal, bGlobal;
    AscendC::GlobalTensor<float> cGlobal;

    uint32_t m, k, n;
    uint32_t aSize, bSize, cSize;

    uint32_t roundm, roundk, roundn;
    uint32_t c0Size = 16;
};

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(__gm__ uint8_t* aGm,
                                                                          __gm__ uint8_t* bGm,
                                                                          __gm__ uint8_t* cGm,
                                                                          GM_ADDR workspace,
                                                                          GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelMatmulIrregular op;
    op.Init(aGm, bGm, cGm, tilingData.m, tilingData.k, tilingData.n);
    op.Process();
}

at::Tensor matmul_with_irregular_shapes_custom_impl_npu(const at::Tensor& A, const at::Tensor& B) {

    at::Tensor A_half = A.to(at::kHalf);

    EXEC_NPU_CMD(aclnnMatmulWithIrregularShapesCustom, A_half, B_half, C_float);
    return C_float;
}

    m.impl("matmul_with_irregular_shapes_custom", &matmul_with_irregular_shapes_custom_impl_npu);
