static ge::graphStatus TilingFunc(gert::TilingContext* context)

    TakeAlongDimCustomTilingData tiling;

    const gert::Shape* xShape = context->GetInputShape(0);
    const gert::Shape* idxShape = context->GetInputShape(1);

    uint32_t n = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t k = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    uint32_t m = static_cast<uint32_t>(idxShape->GetOriginShape().GetDim(1));

    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_m(m);

    // Set a reasonable block dim
    uint32_t blockDim = 8;
    if (n < blockDim) blockDim = n == 0 ? 1 : n;
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelTakeAlongDim {
public:
    __aicore__ inline KernelTakeAlongDim() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR y, uint32_t n, uint32_t k, uint32_t m)
    {
        this->n = n;
        this->k = k;
        this->m = m;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t baseRows = n / blockNum;
        uint32_t tail = n % blockNum;

        this->rows = baseRows + (blockIdx < tail ? 1U : 0U);
        this->rowStart = blockIdx * baseRows + (blockIdx < tail ? blockIdx : tail);

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), static_cast<uint32_t>(n) * k);
        idxGm.SetGlobalBuffer(reinterpret_cast<__gm__ uint32_t*>(idx), static_cast<uint32_t>(n) * m);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), static_cast<uint32_t>(n) * m);

        pipe.InitBuffer(inQueueX, 1, k * sizeof(float));
        pipe.InitBuffer(inQueueIdx, 1, m * sizeof(uint32_t));
        pipe.InitBuffer(outQueueY, 1, m * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rows; ++r) {
            uint32_t rowIdx = rowStart + r;
            CopyIn(rowIdx);
            Compute();
            CopyOut(rowIdx);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<uint32_t> idxLocal = inQueueIdx.AllocTensor<uint32_t>();

        AscendC::DataCopy(xLocal, xGm[rowIdx * k], k);
        AscendC::DataCopy(idxLocal, idxGm[rowIdx * m], m);

        // Convert index values (elements) to byte offsets by *4 using two adds (idx*2 then *4)
        AscendC::Add(idxLocal, idxLocal, idxLocal, m);   // idxLocal = 2 * idxLocal
        AscendC::Add(idxLocal, idxLocal, idxLocal, m);   // idxLocal = 4 * idxLocal (bytes)

        inQueueX.EnQue(xLocal);
        inQueueIdx.EnQue(idxLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<uint32_t> offLocal = inQueueIdx.DeQue<uint32_t>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        // Gather m elements from xLocal using byte offsets in offLocal
        AscendC::Gather(yLocal, xLocal, offLocal, (uint32_t)0, m);

        outQueueY.EnQue<float>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueIdx.FreeTensor(offLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[rowIdx * m], yLocal, m);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX, inQueueIdx;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<uint32_t> idxGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n;
    uint32_t k;
    uint32_t m;

    uint32_t rows;
    uint32_t rowStart;
};

extern "C" __global__ __aicore__ void take_along_dim_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTakeAlongDim op;
    op.Init(x, idx, y, tiling_data.n, tiling_data.k, tiling_data.m);
    op.Process();
}
