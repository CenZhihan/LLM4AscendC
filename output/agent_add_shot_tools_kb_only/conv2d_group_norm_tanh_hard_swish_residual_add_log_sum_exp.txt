kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR tiling_gm)
    {
        GET_TILING_DATA(tilingData, tiling_gm);
        n = tilingData.n();
        c = tilingData.c();
        h = tilingData.h();
        w = tilingData.w();
        groups = tilingData.groups();
        epsilon = tilingData.epsilon();

        total_elems = n * c * h * w;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(x), total_elems);
        gammaGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(gamma), c);
        betaGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(beta), c);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(y), total_elems);

        uint32_t hwAlignSize = ((sizeof(DTYPE) * h * w + ONE_BLK_SIZE - 1) / ONE_BLK_SIZE * ONE_BLK_SIZE) / sizeof(DTYPE);
        aligned_hw = hwAlignSize;

        pipe.InitBuffer(inQueueX, BUFFER_NUM, sizeof(DTYPE) * n * c * aligned_hw);
        pipe.InitBuffer(inQueueGamma, BUFFER_NUM, (sizeof(DTYPE) * c + 31) / 32 * 32);
        pipe.InitBuffer(inQueueBeta, BUFFER_NUM, (sizeof(DTYPE) * c + 31) / 32 * 32);
        pipe.InitBuffer(outQueueY, BUFFER_NUM, sizeof(DTYPE) * n * c * aligned_hw);

        pipe.InitBuffer(meanBuffer, (sizeof(DTYPE) * groups * n + 31) / 32 * 32);
        pipe.InitBuffer(varBuffer, (sizeof(DTYPE) * groups * n + 31) / 32 * 32);

        // Scratch buffers for element-wise activations
        pipe.InitBuffer(scratchA, 1, sizeof(DTYPE) * n * c * aligned_hw);
        pipe.InitBuffer(scratchB, 1, sizeof(DTYPE) * n * c * aligned_hw);
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        // Copy input X with width padding alignment
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopyParams copyParamsX{static_cast<uint16_t>(n*c), static_cast<uint16_t>(h*w*sizeof(DTYPE)), 0, 0};
        AscendC::DataCopyPadParams padParamsX{true, 0, static_cast<uint8_t>(aligned_hw - h * w), 0};
        AscendC::DataCopyPad(xLocal, xGm, copyParamsX, padParamsX);
        inQueueX.EnQue(xLocal);

        // Copy gamma and beta
        AscendC::LocalTensor<DTYPE> gammaLocal = inQueueGamma.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> betaLocal  = inQueueBeta.AllocTensor<DTYPE>();
        AscendC::DataCopyParams copyParamsGB{1, static_cast<uint16_t>(c*sizeof(DTYPE)), 0, 0};
        AscendC::DataCopyPadParams padParamsGB{false, 0, 0, 0};
        AscendC::DataCopyPad(gammaLocal, gammaGm, copyParamsGB, padParamsGB);
        AscendC::DataCopyPad(betaLocal,  betaGm,  copyParamsGB, padParamsGB);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> gammaLocal = inQueueGamma.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> betaLocal  = inQueueBeta.DeQue<DTYPE>();

        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> meanLocal = meanBuffer.Get<DTYPE>();
        AscendC::LocalTensor<DTYPE> varLocal  = varBuffer.Get<DTYPE>();

        // Prepare GN tiling info
        uint32_t inputShape[4] = {n, c, h, w};
        AscendC::ShapeInfo shapeInfo{(uint8_t)4, inputShape, (uint8_t)4, inputShape, AscendC::DataFormat::ND};

        uint32_t stackBufferSize = 0;
        {
            AscendC::LocalTensor<float> stackBuffer;
            bool ok = AscendC::PopStackBuffer<float, AscendC::TPosition::LCM>(stackBuffer);
            (void)ok;
            stackBufferSize = stackBuffer.GetSize();
        }

        AscendC::GroupNormTiling groupNormTiling;
        AscendC::GetGroupNormNDTillingInfo(shapeInfo, stackBufferSize, sizeof(DTYPE), false, groups, groupNormTiling);

        // Step 1: GroupNorm on xLocal -> yLocal
        AscendC::GroupNorm<DTYPE, false>(yLocal, meanLocal, varLocal, xLocal, gammaLocal, betaLocal, (DTYPE)epsilon, groupNormTiling);
        AscendC::PipeBarrier<AscendC::PIPE_ALL>();

        // Step 2: tanh(yLocal) using exp-based formulation: tanh(z) = (e^{2z}-1)/(e^{2z}+1)
        AscendC::LocalTensor<DTYPE> tmpA = scratchA.Get<DTYPE>(); // store e^{2z}
        AscendC::LocalTensor<DTYPE> tmpB = scratchB.Get<DTYPE>(); // store denominator

        uint32_t lengthAll = n * c * aligned_hw;

        AscendC::Muls(tmpA, yLocal, (DTYPE)2.0f, lengthAll);     // tmpA = 2*z
        AscendC::Exp(tmpA, tmpA, lengthAll);                     // tmpA = exp(2*z)
        AscendC::Adds(tmpB, tmpA, (DTYPE)1.0f, lengthAll);       // tmpB = exp(2*z) + 1
        AscendC::Adds(yLocal, tmpA, (DTYPE)(-1.0f), lengthAll);  // yLocal = exp(2*z) - 1
        AscendC::Div(yLocal, yLocal, tmpB, lengthAll);           // yLocal = tanh(z)

        // Step 3: HardSwish on tanh output: hswish(u) = u * clamp(u+3, 0, 6) / 6
        AscendC::Adds(tmpA, yLocal, (DTYPE)3.0f, lengthAll);     // tmpA = yLocal + 3
        AscendC::Maxs(tmpA, tmpA, (DTYPE)0.0f, lengthAll);       // tmpA = max(tmpA, 0)
        AscendC::Mins(tmpA, tmpA, (DTYPE)6.0f, lengthAll);       // tmpA = min(tmpA, 6)
        AscendC::Mul(tmpA, tmpA, yLocal, lengthAll);             // tmpA = yLocal * clamp(...)
        AscendC::Muls(tmpA, tmpA, (DTYPE)(1.0f/6.0f), lengthAll);// tmpA = hardswish(yLocal)

        // Step 4: Residual add with original xLocal (broadcast-aligned buffer)
        AscendC::Add(yLocal, xLocal, tmpA, lengthAll);           // yLocal = xLocal + hswish(tanh(GN(x)))

        outQueueY.EnQue<DTYPE>(yLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueGamma.FreeTensor(gammaLocal);
        inQueueBeta.FreeTensor(betaLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        // Copy back removing width padding
        AscendC::DataCopyParams copyParamsY{static_cast<uint16_t>(n*c), static_cast<uint16_t>(h*w*sizeof(DTYPE)), 0, 0};
        AscendC::DataCopyPadParams padParamsY{true, 0, static_cast<uint8_t>(aligned_hw - h * w), 0};
        AscendC::DataCopyPad(yGm, yLocal, copyParamsY, padParamsY);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM>  inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM>  inQueueGamma;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM>  inQueueBeta;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::TBuf<AscendC::QuePosition::VECCALC> meanBuffer;
    AscendC::TBuf<AscendC::QuePosition::VECCALC> varBuffer;
    AscendC::TBuf<AscendC::QuePosition::VECCALC> scratchA;
    AscendC::TBuf<AscendC::QuePosition::VECCALC> scratchB;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> gammaGm;
    AscendC::GlobalTensor<DTYPE> betaGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t n{1}, c{1}, h{1}, w{1};
    uint32_t groups{1};
    DTYPE epsilon{1e-5f};
    uint32_t total_elems{0};
    uint32_t aligned_hw{0};
};

extern "C" __global__ __aicore__ void conv2d_group_norm_tanh_hard_swish_residual_add_log_sum_exp_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelFused op;
    op.Init(x, gamma, beta, y, tiling);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep Conv2d on PyTorch, fuse GroupNorm + Tanh + HardSwish + ResidualAdd into a single AscendC kernel,
    then compute LogSumExp on channel dimension.
    """
    def __init__(self, in_channels, out_channels, kernel_size, groups, eps=1e-5):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = torch.nn.GroupNorm(groups, out_channels, eps=eps)
        self.groups = groups
        self.eps = eps

    def forward(self, x):
        x_conv = self.conv(x)
        # Fused: GroupNorm -> Tanh -> HardSwish -> ResidualAdd (with x_conv)
        x_residual = custom_ops_lib.conv2d_group_norm_tanh_hard_swish_residual_add_log_sum_exp_custom(
            x_conv, self.group_norm.weight, self.group_norm.bias, self.groups, self.eps
        )
        # Final LogSumExp across channel dimension with keepdim=True
        x_logsumexp = torch.logsumexp(x_residual, dim=1, keepdim=True)
        return x_logsumexp
