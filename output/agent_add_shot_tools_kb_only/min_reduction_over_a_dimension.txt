kernel_src="""
#include "kernel_operator.h"

// 该内核对输入张量按最后一维进行最小值归约，输出去掉最后一维
// 为确保与 AscendC ReduceMin API 的支持范围兼容，建议输入 dtype 为半精度或按工程配置的 DTYPE_X/DTYPE_Y

class KernelMinReduceLastDim {
public:
    __aicore__ inline KernelMinReduceLastDim() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t outerCount, uint32_t reduceLen)
    {
        this->totalLength = totalLength;
        this->outerCount  = outerCount;   // 行数（除最后一维外展开）
        this->reduceLen   = reduceLen;    // 最后一维长度
        // 基于 block 划分行范围
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockId  = AscendC::GetBlockIdx();
        uint32_t baseRows = (outerCount == 0) ? 0 : (outerCount / blockNum);
        uint32_t remRows  = (outerCount == 0) ? 0 : (outerCount % blockNum);
        this->myRows      = baseRows + (blockId < remRows ? 1u : 0u);
        this->myStartRow  = baseRows * blockId + (blockId < remRows ? blockId : remRows);

        // 设置 GM 指针（按行展开）
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + static_cast<uint64_t>(myStartRow) * reduceLen,
                            static_cast<uint64_t>(myRows) * reduceLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + myStartRow, myRows);

        // 本地缓冲：一行数据、工作区和一个标量输出
        pipe.InitBuffer(inQueueSrc, 1, reduceLen * sizeof(DTYPE_X));
        // 参考文档示例，工作区最小空间按 32 个元素分配（半/浮不同字节数由 DTYPE_X 决定）
        pipe.InitBuffer(workQueue, 1, 32 * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueDst, 1, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < myRows; ++r) {
            CopyIn(r);
            Compute();
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_X> srcLocal = inQueueSrc.AllocTensor<DTYPE_X>();
        // 将一整行（最后一维长度）拷入 UB
        AscendC::DataCopy(srcLocal, xGm[row * reduceLen], reduceLen);
        inQueueSrc.EnQue(srcLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE_X> srcLocal = inQueueSrc.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> dstLocal = outQueueDst.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_X> workLocal = workQueue.AllocTensor<DTYPE_X>();
        // 对该行做最小值归约，仅返回最小值
        AscendC::ReduceMin<DTYPE_X>(dstLocal, srcLocal, workLocal, static_cast<int32_t>(reduceLen), false);
        outQueueDst.EnQue<DTYPE_Y>(dstLocal);
        inQueueSrc.FreeTensor(srcLocal);
        workQueue.FreeTensor(workLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_Y> dstLocal = outQueueDst.DeQue<DTYPE_Y>();
        // 将该行的 min 标量写回
        AscendC::DataCopy(yGm[row], dstLocal, 1);
        outQueueDst.FreeTensor(dstLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueSrc;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> workQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueDst;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalLength;
    uint32_t outerCount;
    uint32_t reduceLen;
    uint32_t myRows;
    uint32_t myStartRow;
};

extern "C" __global__ __aicore__ void min_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMinReduceLastDim op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.outerCount, tiling_data.reduceLen);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用 AscendC 自定义核对指定维度做最小值归约。
    内核固定对最后一维归约，因此在调用前将目标维度置于最后一维。
    """
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # 将目标归约维度移动到最后一维，然后调用自定义算子
        d = self.dim if self.dim >= 0 else (x.dim() + self.dim)
        assert 0 <= d < x.dim(), "dim 超出范围"
        if d != x.dim() - 1:
            perm = [i for i in range(x.dim()) if i != d] + [d]
            x = x.permute(perm).contiguous()
        return custom_ops_lib.min_reduction_over_a_dimension_custom(x)

def get_inputs():
    batch_size = 128
    dim1 = 4096
    dim2 = 4096
    x = torch.rand(batch_size, dim1, dim2, device="npu")
    return [x]

def get_init_inputs():
    return [1]
