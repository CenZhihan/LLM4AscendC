BEGIN_TILING_DATA_DEF(MatmulWithLargeKDimensionCustomTilingData)
  TILING_DATA_FIELD_DEF(uint32_t, dummy);
END_TILING_DATA_DEF;

REGISTER_TILING_DATA_CLASS(MatmulWithLargeKDimensionCustom, MatmulWithLargeKDimensionCustomTilingData)

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto platformInfo = context->GetPlatformInfo();
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(platformInfo);

    int64_t M = context->GetInputShape(0)->GetOriginShape().GetDim(0);
    int64_t K = context->GetInputShape(0)->GetOriginShape().GetDim(1);
    int64_t N = context->GetInputShape(1)->GetOriginShape().GetDim(1);

    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);
    tiling.SetDim(8); // 使用8个核心参与计算
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBias(false);
    tiling.SetShape(M, N, K);
    tiling.SetSingleShape(M, N, K);
    tiling.SetOrgShape(M, N, K);
    tiling.SetBufferSpace(-1, -1, -1); // 默认使用可用的AI处理器全部空间

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret == -1) {
        return ge::GRAPH_FAILED;
    }

    uint32_t blockDim = tiling.GetCoreNum();
    context->SetBlockDim(blockDim);

    auto raw = context->GetRawTilingData();
    std::memcpy(raw->GetData(), &tilingData, sizeof(optiling::TCubeTiling));
    raw->SetDataSize(sizeof(optiling::TCubeTiling));
    context->SetTilingKey(1);

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

host_operator_src="""
#include "matmul_with_large_k_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
extern ge::graphStatus TilingFunc(gert::TilingContext* context);
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // 输入0: A(M, K)，输入1: B(K, N)，输出: C(M, N)
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    gert::Shape* c_shape = context->GetOutputShape(0);

    std::vector<int64_t> out_dims(2);
    out_dims[0] = a_shape->GetOriginShape().GetDim(0);
    out_dims[1] = b_shape->GetOriginShape().GetDim(1);

    c_shape->SetDimNum(2);
    c_shape->SetOrigin(out_dims.data(), out_dims.size());
    c_shape->SetStorageShape(out_dims.data(), out_dims.size());
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // 继承输入A的数据类型到输出C
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

extern "C" __global__ __aicore__ void matmul_with_large_k_dimension_custom(GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    typedef float A_T;
    typedef float B_T;
    typedef float C_T;
    typedef float BiasT;

    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, false, LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

    // 读取TCubeTiling
    TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(int32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // 初始化GM张量
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;

    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(C_T);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(cGM), sizeC);

    tiling.shareMode = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize = 0;

    AscendC::GlobalTensor<A_T> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal[0].GetPhyAddr()), tiling.singleCoreM * tiling.singleCoreK);

    AscendC::GlobalTensor<B_T> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal[0].GetPhyAddr()), tiling.singleCoreK * tiling.singleCoreN);

    AscendC::GlobalTensor<C_T> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal[0].GetPhyAddr()), tiling.singleCoreM * tiling.singleCoreN);

    // 创建并初始化Matmul实例
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;
    TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);
    mm1.Init(&tiling);
    mm1.SetTensorA(gm_a, false);
    mm1.SetTensorB(gm_b, false);
    mm1.SetWorkspace(workspaceGM, 0);

    // 单次批量计算(设置batchA=1, batchB=1)
    mm1.IterateBatch(gm_c, 1, 1, false);
}
"""
