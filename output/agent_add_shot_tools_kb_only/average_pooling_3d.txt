inline int64_t floor_div(int64_t a, int64_t b) {
    return a / b;

    return floor_div(in + 2 * p - k, s) + 1;
}

kernel_src="""
#include "kernel_operator.h"

#define DTYPE_X float
#define DTYPE_Z float

class KernelAvgPool3D {
public:
    __aicore__ inline KernelAvgPool3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t k, uint32_t s, uint32_t p,
                                uint32_t od, uint32_t oh, uint32_t ow)
    {
        this->n = n; this->c = c; this->d = d; this->h = h; this->w = w;
        this->k = k; this->s = s; this->p = p;
        this->od = od; this->oh = oh; this->ow = ow;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint64_t>(n) * c * d * h * w);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, static_cast<uint64_t>(n) * c * od * oh * ow);
    }

    __aicore__ inline void Process()
    {
        uint32_t totalNC = n * c;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t base = (totalNC / blockNum) * blockIdx;
        uint32_t tail = (blockIdx == blockNum - 1) ? totalNC : (base + (totalNC / blockNum));
        if (blockIdx == blockNum - 1) {
            tail = totalNC;
        }

        uint64_t inStrideW = 1;
        uint64_t inStrideH = w;
        uint64_t inStrideD = static_cast<uint64_t>(h) * w;
        uint64_t inStrideC = static_cast<uint64_t>(d) * h * w;
        uint64_t inStrideN = static_cast<uint64_t>(c) * d * h * w;

        uint64_t outStrideW = 1;
        uint64_t outStrideH = ow;
        uint64_t outStrideD = static_cast<uint64_t>(oh) * ow;
        uint64_t outStrideC = static_cast<uint64_t>(od) * oh * ow;
        uint64_t outStrideN = static_cast<uint64_t>(c) * od * oh * ow;

        for (uint32_t ncIdx = base; ncIdx < tail; ++ncIdx) {
            uint32_t nn = ncIdx / c;
            uint32_t cc = ncIdx % c;
            uint64_t inBase = nn * inStrideN + cc * inStrideC;
            uint64_t outBase = nn * outStrideN + cc * outStrideC;

            for (uint32_t z = 0; z < od; ++z) {
                int64_t startD = static_cast<int64_t>(z) * s - static_cast<int64_t>(p);
                for (uint32_t y = 0; y < oh; ++y) {
                    int64_t startH = static_cast<int64_t>(y) * s - static_cast<int64_t>(p);
                    for (uint32_t x = 0; x < ow; ++x) {
                        int64_t startW = static_cast<int64_t>(x) * s - static_cast<int64_t>(p);
                        float acc = 0.0f;
                        uint32_t cnt = 0U;

                        for (uint32_t kd = 0; kd < k; ++kd) {
                            int64_t id = startD + static_cast<int64_t>(kd);
                            if (id < 0 || id >= static_cast<int64_t>(d)) continue;
                            for (uint32_t kh = 0; kh < k; ++kh) {
                                int64_t ih = startH + static_cast<int64_t>(kh);
                                if (ih < 0 || ih >= static_cast<int64_t>(h)) continue;
                                for (uint32_t kw = 0; kw < k; ++kw) {
                                    int64_t iw = startW + static_cast<int64_t>(kw);
                                    if (iw < 0 || iw >= static_cast<int64_t>(w)) continue;
                                    uint64_t inIndex = inBase + static_cast<uint64_t>(id) * inStrideD
                                                      + static_cast<uint64_t>(ih) * inStrideH
                                                      + static_cast<uint64_t>(iw) * inStrideW;
                                    acc += *((__gm__ DTYPE_X*)xGm.GetGlobalBuffer() + inIndex);
                                    ++cnt;
                                }
                            }
                        }
                        float outVal = cnt > 0U ? (acc / static_cast<float>(cnt)) : 0.0f;
                        uint64_t outIndex = outBase + static_cast<uint64_t>(z) * outStrideD
                                          + static_cast<uint64_t>(y) * outStrideH
                                          + static_cast<uint64_t>(x) * outStrideW;
                        *((__gm__ DTYPE_Z*)zGm.GetGlobalBuffer() + outIndex) = outVal;
                    }
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t n, c, d, h, w;
    uint32_t k, s, p;
    uint32_t od, oh, ow;
};

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelAvgPool3D op;
    op.Init(x, z,
            tilingData.n(), tilingData.c(), tilingData.d(), tilingData.h(), tilingData.w(),
            tilingData.kd(), tilingData.sd(), tilingData.pd(),
            tilingData.od(), tilingData.oh(), tilingData.ow());
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor average_pooling_3d_custom_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding) {
    TORCH_CHECK(self.dim() == 5, "Input must be 5D NCDHW");
    auto sizes = self.sizes();
    int64_t N = sizes[0], C = sizes[1], D = sizes[2], H = sizes[3], W = sizes[4];

    auto calc = [](int64_t in, int64_t k, int64_t s, int64_t p) {
        return (in + 2 * p - k) / s + 1;
    };
    int64_t Do = calc(D, kernel_size, stride, padding);
    int64_t Ho = calc(H, kernel_size, stride, padding);
    int64_t Wo = calc(W, kernel_size, stride, padding);

    at::Tensor result = at::empty({N, C, Do, Ho, Wo}, self.options());
    EXEC_NPU_CMD(aclnnAveragePooling3dCustom, self, kernel_size, stride, padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("average_pooling_3d_custom", &average_pooling_3d_custom_impl_npu);
}
