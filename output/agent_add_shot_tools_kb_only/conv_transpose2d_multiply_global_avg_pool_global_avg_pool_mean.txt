kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR multiplier, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t totalNc, uint32_t ncPerBlock) {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->planeSize = h * w;
        this->totalNc = totalNc;
        this->ncPerBlock = ncPerBlock;
        this->tileLen = 1024; // tile for spatial reduce

        uint32_t startNc = AscendC::GetBlockIdx() * this->ncPerBlock;
        if (startNc > this->totalNc) startNc = this->totalNc;
        uint32_t endNc = startNc + this->ncPerBlock;
        if (endNc > this->totalNc) endNc = this->totalNc;
        this->procNc = (endNc > startNc) ? (endNc - startNc) : 0;

        xGm.SetGlobalBuffer((__gm__ float*)x + static_cast<uint64_t>(startNc) * this->planeSize,
                            static_cast<uint64_t>(this->procNc) * this->planeSize);
        mulGm.SetGlobalBuffer((__gm__ float*)multiplier, 1);
        yGm.SetGlobalBuffer((__gm__ float*)y + startNc, this->procNc);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(float));
        pipe.InitBuffer(inQueueMul, 1, sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (this->procNc == 0) {
            return;
        }

        // Load multiplier once
        AscendC::LocalTensor<float> mulLocal = inQueueMul.AllocTensor<float>();
        AscendC::DataCopy(mulLocal, mulGm, 1);
        inQueueMul.EnQue(mulLocal);
        mulLocal = inQueueMul.DeQue<float>();
        float m = mulLocal[0];
        inQueueMul.FreeTensor(mulLocal);

        // Iterate each (n,c) plane assigned to this block
        for (uint32_t idx = 0; idx < this->procNc; ++idx) {
            float acc = 0.0f;
            uint32_t remain = this->planeSize;
            uint64_t base = static_cast<uint64_t>(idx) * this->planeSize;
            uint32_t offset = 0;

            while (remain > 0) {
                uint32_t len = remain > this->tileLen ? this->tileLen : remain;
                AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm[base + offset], len);
                // multiply by scalar
                AscendC::Muls(xLocal, xLocal, m, len);
                // accumulate to scalar
                for (uint32_t i = 0; i < len; ++i) {
                    acc += xLocal[i];
                }
                inQueueX.FreeTensor(xLocal);
                offset += len;
                remain -= len;
            }

            float meanVal = acc / static_cast<float>(this->planeSize); // second mean is identical

            AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
            AscendC::Duplicate(yLocal, meanVal, 1);
            // write to GM: y shape [N,C,1,1] flattened as [Nc]
            AscendC::DataCopy(yGm[idx], yLocal, 1);
            outQueueY.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueMul;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> mulGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n, c, h, w;
    uint32_t planeSize;
    uint32_t totalNc;
    uint32_t ncPerBlock;
    uint32_t procNc;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void conv_transpose2d_multiply_global_avg_pool_global_avg_pool_mean_custom(GM_ADDR x, GM_ADDR multiplier, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, multiplier, y,
            tiling_data.n(), tiling_data.c(), tiling_data.h(), tiling_data.w(),
            tiling_data.totalNc(), tiling_data.ncPerBlock());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom AscendC op replaces: multiply + global avg pool + second global avg pool
    The transposed convolution remains as PyTorch op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, multiplier):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.multiplier = multiplier

    def forward(self, x):
        x = self.conv_transpose(x)
        mult = torch.tensor([self.multiplier], dtype=x.dtype, device=x.device)
        # fused: (x * multiplier) -> mean over H,W (keepdim=True), then mean again
        return custom_ops_lib.conv_transpose2d_multiply_global_avg_pool_global_avg_pool_mean_custom(x, mult)
