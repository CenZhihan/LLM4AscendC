namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    LogicAndBroadcastCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    uint32_t totalLength = x_shape->GetOriginShape().GetShapeSize();

    // Assume ND with at least 3 dims as per example: (N, 512, 1024)
    // dim1 = second dim size, dim2 = last dim size
    uint32_t rank = x_shape->GetOriginShape().GetDimNum();
    uint32_t dim1 = 1;
    uint32_t dim2 = 1;
    if (rank >= 2) {
        dim1 = x_shape->GetOriginShape().GetDim(rank - 2);
    }
    if (rank >= 1) {
        dim2 = x_shape->GetOriginShape().GetDim(rank - 1);
    }

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_dim1(dim1);
    tiling.set_dim2(dim2);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue
 
class KernelLogicAndBroadcast {
public:
    __aicore__ inline KernelLogicAndBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t dim1, uint32_t dim2)
    {
        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();

        this->totalLength = totalLength;
        this->dim1 = dim1;   // middle dimension size (e.g., 512)
        this->dim2 = dim2;   // last dimension size (e.g., 1024)

        this->blockLength = totalLength / this->blockNum;
        this->blockStart = this->blockLength * this->blockIdx;
        // Handle tail block length
        uint32_t blockEnd = (this->blockIdx == this->blockNum - 1) ? totalLength : (this->blockStart + this->blockLength);
        this->blockLength = blockEnd - this->blockStart;

        this->tileNum = tileNum;
        // ensure tileLength is a multiple of dim2 to simplify broadcasting along last dim
        uint32_t baseTile = (this->blockLength / tileNum / BUFFER_NUM);
        uint32_t mult = (baseTile / this->dim2);
        this->tileLength = mult > 0 ? (mult * this->dim2) : this->dim2;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockStart, this->blockLength);
        // y has shape (1, dim1, 1). Keep a single global buffer for y with length dim1.
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, this->dim1);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockStart, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y)); // for expanded y
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));

        // small scalar buffer for fetching a single y value
        pipe.InitBuffer(tmpScalarQue, 1, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        uint32_t loopCount = DivCeil(this->blockLength, this->tileLength);
        for (uint32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline uint32_t DivCeil(uint32_t a, uint32_t b) {
        return (a + b - 1) / b;
    }

    __aicore__ inline void CopyIn(uint32_t progress)
    {
        uint32_t remain = this->blockLength - progress * this->tileLength;
        this->curCalLen = remain > this->tileLength ? this->tileLength : remain;

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->curCalLen);
        inQueueX.EnQue(xLocal);

        // Prepare expanded yLocal according to broadcasting of y (1, dim1, 1)
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();
        // Fill yLocal in blocks of dim2 with the same scalar
        uint32_t globalBase = this->blockStart + progress * this->tileLength;
        uint32_t filled = 0;
        while (filled < this->curCalLen) {
            uint32_t idx = globalBase + filled;
            // j = ((idx / dim2) % dim1)
            uint32_t j = (idx / this->dim2) % this->dim1;

            // load scalar y[j]
            AscendC::LocalTensor<DTYPE_Y> yScalarLocal = tmpScalarQue.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(yScalarLocal, yGm[j], 1);
            DTYPE_Y val = yScalarLocal.GetValue(0);
            tmpScalarQue.FreeTensor(yScalarLocal);

            // number of elements to fill in this block
            uint32_t blkRemainInTile = this->curCalLen - filled;
            uint32_t fillCount = blkRemainInTile < this->dim2 ? blkRemainInTile : this->dim2;

            // Fill yLocal[filled ... filled+fillCount) with val
            for (uint32_t t = 0; t < fillCount; ++t) {
                yLocal.SetValue(filled + t, val);
            }

            filled += fillCount;
        }
        inQueueY.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // Elementwise logical AND on bool tensors
        for (uint32_t i = 0; i < this->curCalLen; ++i) {
            DTYPE_X a_val = xLocal.GetValue(i);
            DTYPE_Y b_val = yLocal.GetValue(i);
            DTYPE_Z out_val = static_cast<DTYPE_Z>(static_cast<bool>(a_val) & static_cast<bool>(b_val));
            zLocal.SetValue(i, out_val);
        }

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueY.FreeTensor(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->curCalLen);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpScalarQue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t blockNum;
    uint32_t blockIdx;
    uint32_t totalLength;
    uint32_t blockStart;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t curCalLen;

    uint32_t dim1; // second dimension
    uint32_t dim2; // last dimension
};

extern "C" __global__ __aicore__ void logic_and_broadcast_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogicAndBroadcast op;
    op.Init(x, y, z, tiling_data.totalLength, tiling_data.tileNum, tiling_data.dim1, tiling_data.dim2);
    op.Process();
}
