static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    // Input shape info
    const gert::Shape* inShape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(inShape->GetOriginShape().GetShapeSize());
    int64_t rank = inShape->GetOriginShape().GetDimNum();

    // Read attr dim if available, default to 1
    int64_t dimAttr = 1;
    // Depending on runtime, attribute retrieval may differ; default used if not present.
    // try get attr "dim"
    if (context->HasAttr("dim")) {
        dimAttr = context->GetAttr<int64_t>("dim");
    }
    // Normalize negative dims
    if (dimAttr < 0) {
        dimAttr += rank;
    }
    if (dimAttr < 0) dimAttr = 0;
    if (dimAttr >= rank) dimAttr = rank > 0 ? (rank - 1) : 0;

    // Compute inner/outter
    int64_t inner = inShape->GetOriginShape().GetDim(dimAttr);
    int64_t outter = (inner == 0) ? 0 : (inShape->GetOriginShape().GetShapeSize() / inner);

    // Fill tiling base fields
    tiling.set_totalLength(totalLength);
    tiling.set_inner(static_cast<int32_t>(inner));
    tiling.set_outter(static_cast<int32_t>(outter));
    tiling.set_n(static_cast<int32_t>(inner));
    tiling.set_k(1);
    tiling.set_islargest(true);
    // For simplicity, pass isinitindex=true to align with KernelTopK template below
    tiling.set_isinitindex(true);
    tiling.set_dim(static_cast<int32_t>(dimAttr));

    // Platform info and TopK tiling
    auto platformInfo = context->GetPlatformInfo();
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(platformInfo);

    uint32_t maxsize = 0;
    uint32_t minsize = 0;
    // Input dtype float => dataTypeSize=4
    const uint32_t dtypesize = 4;

    // Get tiling for TopK (k=1 argmax)
    // Note: mode uses TOPK_NORMAL; isLargest=true
    AscendC::TopKTilingFunc(ascendcPlatform,
                            tiling.inner(),
                            tiling.outter(),
                            tiling.k(),
                            dtypesize,
                            tiling.isinitindex(),
                            AscendC::TopKMode::TOPK_NORMAL,
                            tiling.islargest(),
                            tiling.topkTilingData());

    AscendC::GetTopKMaxMinTmpSize(ascendcPlatform,
                                  tiling.inner(),
                                  tiling.outter(),
                                  /*isReuseSource*/false,
                                  tiling.isinitindex(),
                                  AscendC::TopKMode::TOPK_NORMAL,
                                  tiling.islargest(),
                                  dtypesize,
                                  maxsize,
                                  minsize);

    // Choose minimal tmp size that guarantees correctness
    tiling.set_tmpsize(minsize);

    // Set block dim
    const uint32_t BLOCK_DIM = 8;
    context->SetBlockDim(BLOCK_DIM);

    // Save tiling
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());

    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

host_operator_src="""
#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
extern ge::graphStatus TilingFunc(gert::TilingContext* context);
}

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Get input shape and attr dim
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);

    int64_t rank = in_shape->GetOriginShape().GetDimNum();
    int64_t dimAttr = 1;
    if (context->HasAttr("dim")) {
        dimAttr = context->GetAttr<int64_t>("dim");
    }
    if (dimAttr < 0) dimAttr += rank;
    if (dimAttr < 0) dimAttr = 0;
    if (dimAttr >= rank) dimAttr = rank > 0 ? (rank - 1) : 0;

    // Build output dims by removing the specified dimension
    std::vector<int64_t> outDims;
    outDims.reserve(rank > 0 ? (rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dimAttr) continue;
        outDims.push_back(in_shape->GetOriginShape().GetDim(i));
    }
    out_shape->SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) {
        out_shape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    // Argmax outputs indices; set to INT32
    context->SetOutputDataType(0, ge::DT_INT32);
    return ge::GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

class ArgmaxOverADimensionCustom : public OpDef {
public:
    explicit ArgmaxOverADimensionCustom(const char* name) : OpDef(name)

};

kernel_src="""
#include "kernel_operator.h"

// Use TopK high-level API with k=1 to implement Argmax.

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x,
                                                                      GM_ADDR indices,
                                                                      GM_ADDR workspace,
                                                                      GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    // Template params follow TopK example: <T, isReuseSrc, isInitIndex, keepTopKIdx, sorted, Mode>
    KernelTopK<float, false, true, false, false, AscendC::TopKMode::TOPK_NORMAL> op;

    // Partition workspace for required GM buffers (srcIdx, finishLocal, dstVal)
    // Use tmpsize as a safe segment size.
    __gm__ uint8_t* ws_base = reinterpret_cast<__gm__ uint8_t*>(workspace);
    GM_ADDR srcIdx      = reinterpret_cast<GM_ADDR>(ws_base);
    GM_ADDR finishLocal = reinterpret_cast<GM_ADDR>(ws_base + tilingData.tmpsize);
    GM_ADDR dstVal      = reinterpret_cast<GM_ADDR>(ws_base + tilingData.tmpsize * 2);

    // Initialize and run TopK with k=1 (argmax)
    op.Init(x,
            srcIdx,
            finishLocal,
            dstVal,
            indices,
            tilingData.k(),            // 1
            tilingData.islargest(),    // true
            tilingData.tmpsize(),      // tmp size
            tilingData.outter(),       // outer axis length
            tilingData.inner(),        // inner axis length
            tilingData.n(),            // same as inner
            tilingData.topkTilingData());

    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include \"pytorch_npu_helper.hpp\"
#include <torch/extension.h>

at::Tensor argmax_over_a_dimension_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    // Normalize dim
    int64_t rank = self.dim();
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank > 0 ? (rank - 1) : 0;

    // Build output shape (remove 'dim')
    std::vector<int64_t> out_sizes;
    out_sizes.reserve(rank > 0 ? (rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out_sizes.push_back(self.size(i));
    }

    at::Tensor result = at::empty(out_sizes, self.options().dtype(at::kInt));

    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"argmax_over_a_dimension_custom\", &argmax_over_a_dimension_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"argmax_over_a_dimension_custom\", &argmax_over_a_dimension_custom_impl_npu, \"Argmax over a dimension\");
