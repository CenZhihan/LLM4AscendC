kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // 每个队列的tensor数量

class KernelClampBroadcast {
public:
    __aicore__ inline KernelClampBroadcast() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR minv, GM_ADDR maxv, GM_ADDR z,
                                uint32_t totalLength, uint32_t cols, uint32_t tileNum)
    {
        this->cols = cols;
        this->tileNum = tileNum;
        this->tileCols = cols / tileNum;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->rows = this->blockLength / this->cols;

        xGm.SetGlobalBuffer((__gm__ float*)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ float*)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // min/max为广播向量，不随block分段
        minGm.SetGlobalBuffer((__gm__ float*)minv, this->cols);
        maxGm.SetGlobalBuffer((__gm__ float*)maxv, this->cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileCols * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileCols * sizeof(float));
        pipe.InitBuffer(tmpQue, 2, this->tileCols * sizeof(float)); // 用于存放min/max切片
    }

    __aicore__ inline void Process()
    {
        // 以列方向分块，每块tileCols列，行全覆盖
        for (uint32_t t = 0; t < this->tileNum; ++t) {
            uint32_t colOffset = t * this->tileCols;

            // 加载本tile对应的min/max切片到本地
            AscendC::LocalTensor<float> minLocal = tmpQue.AllocTensor<float>();
            AscendC::LocalTensor<float> maxLocal = tmpQue.AllocTensor<float>();
            AscendC::DataCopy(minLocal, minGm[colOffset], this->tileCols);
            AscendC::DataCopy(maxLocal, maxGm[colOffset], this->tileCols);

            for (uint32_t r = 0; r < this->rows; ++r) {
                uint32_t base = r * this->cols + colOffset;

                // CopyIn
                AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm[base], this->tileCols);
                inQueueX.EnQue(xLocal);

                // Compute
                AscendC::LocalTensor<float> xLocalDeq = inQueueX.DeQue<float>();
                AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
                // 先做max(x, min), 再做min(., max)
                AscendC::Max(zLocal, xLocalDeq, minLocal, this->tileCols);
                AscendC::Min(zLocal, zLocal, maxLocal, this->tileCols);
                outQueueZ.EnQue(zLocal);
                inQueueX.FreeTensor(xLocalDeq);

                // CopyOut
                AscendC::LocalTensor<float> zLocalDeq = outQueueZ.DeQue<float>();
                AscendC::DataCopy(zGm[base], zLocalDeq, this->tileCols);
                outQueueZ.FreeTensor(zLocalDeq);
            }

            tmpQue.FreeTensor(minLocal);
            tmpQue.FreeTensor(maxLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> minGm;
    AscendC::GlobalTensor<float> maxGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t cols;
    uint32_t rows;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileCols;
};

extern "C" __global__ __aicore__ void clamp_broadcast_custom(GM_ADDR x, GM_ADDR min_val, GM_ADDR max_val, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelClampBroadcast op;
    op.Init(x, min_val, max_val, z, tiling_data.totalLength, tiling_data.cols, tiling_data.tileNum);
    op.Process();
}
