host_operator_src="""
#include "embedding_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    EmbeddingCustomTilingData tiling;

    // indices shape: [B, S]
    const gert::Shape* idxShape = context->GetInputShape(0);
    uint32_t totalTokens = idxShape->GetOriginShape().GetShapeSize(); // B*S

    // weight shape: [V, E]
    const gert::Shape* wShape = context->GetInputShape(1);
    std::vector<int64_t> wDims = wShape->GetOriginShape().GetDims();
    uint32_t embDim = (wDims.size() >= 2) ? static_cast<uint32_t>(wDims.back()) : 0;
    uint32_t weightLength = wShape->GetOriginShape().GetShapeSize(); // V*E

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalTokens(totalTokens);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_embDim(embDim);
    tiling.set_weightLength(weightLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Output shape = indices shape + [emb_dim]
    const gert::Shape* idxShape = context->GetInputShape(0);
    const gert::Shape* wShape = context->GetInputShape(1);

    std::vector<int64_t> idxDims = idxShape->GetOriginShape().GetDims();
    std::vector<int64_t> wDims = wShape->GetOriginShape().GetDims();

    int64_t embDim = (wDims.size() >= 2) ? wDims.back() : 0;
    std::vector<int64_t> outDims = idxDims;
    outDims.push_back(embDim);

    gert::Shape* outShape = context->GetOutputShape(0);
    outShape->SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) {
        outShape->SetDim(i, outDims[i]);
    }
    outShape->SetFormat(ge::FORMAT_ND);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    // Output dtype follows weight dtype
    const auto weightDt = context->GetInputDataType(1);
    context->SetOutputDataType(0, weightDt);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class EmbeddingCustom : public OpDef {
public:
    explicit EmbeddingCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT32})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("z")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(EmbeddingCustom);
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelEmbedding {
public:
    __aicore__ inline KernelEmbedding() {}
    __aicore__ inline void Init(GM_ADDR indices, GM_ADDR weight, GM_ADDR output,
                                uint32_t totalTokens, uint32_t tileNum, uint32_t embDim, uint32_t weightLength)
    {
        this->tileNum = tileNum;
        this->embDim = embDim;
        this->blockTokens = totalTokens / AscendC::GetBlockNum();
        this->tileTokens = this->blockTokens / tileNum / BUFFER_NUM;

        idxGm.SetGlobalBuffer((__gm__ DTYPE_X *)indices + this->blockTokens * AscendC::GetBlockIdx(), this->blockTokens);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)weight, weightLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z *)output + (uint64_t)this->blockTokens * (uint64_t)this->embDim * AscendC::GetBlockIdx(),
                              (uint64_t)this->blockTokens * (uint64_t)this->embDim);

        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, this->tileTokens * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueOut, BUFFER_NUM, (uint64_t)this->tileTokens * (uint64_t)this->embDim * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueueRow, 1, this->embDim * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> idxLocal = inQueueIdx.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(idxLocal, idxGm[progress * this->tileTokens], this->tileTokens);
        inQueueIdx.EnQue(idxLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> idxLocal = inQueueIdx.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Y> rowLocal = tmpQueueRow.AllocTensor<DTYPE_Y>();

        // For each token in the tile, copy one embedding row (embDim elements)
        for (uint32_t t = 0; t < this->tileTokens; ++t) {
            int32_t indexVal = static_cast<int32_t>(idxLocal.GetValue(t));
            // Copy one row from weight to rowLocal
            AscendC::DataCopy(rowLocal, wGm[(uint64_t)indexVal * (uint64_t)this->embDim], this->embDim);
            // Copy rowLocal to outLocal at proper offset
            AscendC::DataCopy(outLocal[(uint64_t)t * (uint64_t)this->embDim], rowLocal, this->embDim);
        }

        outQueueOut.EnQue<DTYPE_Z>(outLocal);
        tmpQueueRow.FreeTensor(rowLocal);
        inQueueIdx.FreeTensor(idxLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.DeQue<DTYPE_Z>();
        AscendC::DataCopy(outGm[(uint64_t)progress * (uint64_t)this->tileTokens * (uint64_t)this->embDim],
                          outLocal, (uint64_t)this->tileTokens * (uint64_t)this->embDim);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueRow;

    AscendC::GlobalTensor<DTYPE_X> idxGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;

    uint32_t tileNum;
    uint32_t embDim;
    uint32_t blockTokens;
    uint32_t tileTokens;
};

extern "C" __global__ __aicore__ void embedding_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelEmbedding op;
    op.Init(x, y, z, tiling_data.totalTokens, tiling_data.tileNum, tiling_data.embDim, tiling_data.weightLength);
    op.Process();
}
"""

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor embedding_custom_impl_npu(const at::Tensor& indices, const at::Tensor& weight) {
    // Ensure indices are int32
    at::Tensor idx_i32 = indices.to(torch::kInt32);
    // Output shape: [B, S, E]
    auto out_sizes = std::vector<int64_t>{idx_i32.size(0), idx_i32.size(1), weight.size(1)};
    at::Tensor result = at::empty(out_sizes, weight.options());
    EXEC_NPU_CMD(aclnnEmbeddingCustom, idx_i32, weight, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("embedding_custom", &embedding_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("embedding_custom", &embedding_custom_impl_npu, "embedding lookup");
}
