namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 读取输入形状 NCHW
    const gert::Shape* inShape = context->GetInputShape(0);
    auto oriShape = inShape->GetOriginShape();
    uint32_t dimNum = oriShape.GetDimNum();
    // 若非4D，按NCHW展开
    uint32_t n = 1, c = 1, h = 1, w = 1;
    if (dimNum == 4) {
        n = static_cast<uint32_t>(oriShape.GetDim(0));
        c = static_cast<uint32_t>(oriShape.GetDim(1));
        h = static_cast<uint32_t>(oriShape.GetDim(2));
        w = static_cast<uint32_t>(oriShape.GetDim(3));
    } else {
        // 兜底：将所有维度乘到H上，N=C=1，W=1
        uint64_t sz = oriShape.GetShapeSize();
        h = static_cast<uint32_t>(sz);
        w = 1;
    }

    // 读取属性（如无法获取，则使用默认值）
    uint32_t kernel_size = 2, stride = 2, padding = 0;
    // 兼容不同平台可能的接口，失败时使用默认值
    do {
        auto attrKernel = context->GetOpAttr("kernel_size");
        if (attrKernel != nullptr) kernel_size = static_cast<uint32_t>(attrKernel->GetInt());
        auto attrStride = context->GetOpAttr("stride");
        if (attrStride != nullptr) stride = static_cast<uint32_t>(attrStride->GetInt());
        auto attrPadding = context->GetOpAttr("padding");
        if (attrPadding != nullptr) padding = static_cast<uint32_t>(attrPadding->GetInt());
    } while (0);

    uint32_t kh = kernel_size, kw = kernel_size;
    uint32_t sh = stride, sw = stride;
    uint32_t ph = padding, pw = padding;

    // 计算输出尺寸（与PyTorch MaxPool2d一致：floor）
    auto out_h = (int64_t)((static_cast<int64_t>(h) + 2LL * ph - static_cast<int64_t>(kh)) / static_cast<int64_t>(sh)) + 1LL;
    auto out_w = (int64_t)((static_cast<int64_t>(w) + 2LL * pw - static_cast<int64_t>(kw)) / static_cast<int64_t>(sw)) + 1LL;
    if (out_h < 0) out_h = 0;
    if (out_w < 0) out_w = 0;

    MaxPooling2dCustomTilingData tiling;
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_kh(kh);
    tiling.set_kw(kw);
    tiling.set_sh(sh);
    tiling.set_sw(sw);
    tiling.set_ph(ph);
    tiling.set_pw(pw);
    tiling.set_oh(static_cast<uint32_t>(out_h));
    tiling.set_ow(static_cast<uint32_t>(out_w));
    tiling.set_planeSizeIn(h * w);
    tiling.set_planeSizeOut(static_cast<uint32_t>(out_h) * static_cast<uint32_t>(out_w));

    // block并行按 N*C 切分
    uint32_t totalPlanes = n * c;
    uint32_t blockNum = totalPlanes;
    if (blockNum == 0) blockNum = 1;
    if (blockNum > 8) blockNum = 8; // 简单设置一个上限
    context->SetBlockDim(blockNum);
    tiling.set_blockNum(blockNum);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    context->SetTilingKey(1);

    // 本算子不需要workspace
    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* inShape = context->GetInputShape(0);
    auto ori = inShape->GetOriginShape();
    // 获取属性（默认同上）
    int64_t kernel_size = 2, stride = 2, padding = 0;
    do {
        auto attrKernel = context->GetOpAttr("kernel_size");
        if (attrKernel != nullptr) kernel_size = attrKernel->GetInt();
        auto attrStride = context->GetOpAttr("stride");
        if (attrStride != nullptr) stride = attrStride->GetInt();
        auto attrPadding = context->GetOpAttr("padding");
        if (attrPadding != nullptr) padding = attrPadding->GetInt();
    } while (0);

    int64_t n = 1, c = 1, h = 1, w = 1;
    if (ori.GetDimNum() == 4) {
        n = ori.GetDim(0);
        c = ori.GetDim(1);
        h = ori.GetDim(2);
        w = ori.GetDim(3);
    } else {
        h = ori.GetShapeSize();
        w = 1;
    }
    int64_t oh = (h + 2 * padding - kernel_size) / stride + 1;
    int64_t ow = (w + 2 * padding - kernel_size) / stride + 1;
    if (oh < 0) oh = 0;
    if (ow < 0) ow = 0;

    gert::Shape* outShape = context->GetOutputShape(0);
    outShape->SetDimNum(4);
    outShape->SetDim(0, n);
    outShape->SetDim(1, c);
    outShape->SetDim(2, oh);
    outShape->SetDim(3, ow);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMaxPool2d {
public:
    __aicore__ inline KernelMaxPool2d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t kh, uint32_t kw, uint32_t sh, uint32_t sw,
                                uint32_t ph, uint32_t pw, uint32_t oh, uint32_t ow,
                                uint32_t blockNum)
    {
        this->n = n; this->c = c; this->h = h; this->w = w;
        this->kh = kh; this->kw = kw; this->sh = sh; this->sw = sw;
        this->ph = ph; this->pw = pw; this->oh = oh; this->ow = ow;

        planeIn = h * w;
        planeOut = oh * ow;

        totalPlanes = n * c;
        blocks = (blockNum == 0) ? 1 : blockNum;

        // 每个block处理的plane数
        planesPerBlock = (totalPlanes + blocks - 1) / blocks;
        startPlane = AscendC::GetBlockIdx() * planesPerBlock;
        endPlane = startPlane + planesPerBlock;
        if (endPlane > totalPlanes) endPlane = totalPlanes;

        // GM绑定到本block负责的起始plane地址
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + static_cast<uint64_t>(startPlane) * planeIn, (endPlane - startPlane) * planeIn);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + static_cast<uint64_t>(startPlane) * planeOut, (endPlane - startPlane) * planeOut);

        // UB缓冲：输入一个plane，输出一个plane
        pipe.InitBuffer(inQueueX, BUFFER_NUM, planeIn * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, planeOut * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (startPlane >= endPlane) return;
        // 逐plane处理
        for (uint32_t p = startPlane; p < endPlane; ++p) {
            uint32_t localIdx = p - startPlane;
            CopyIn(localIdx);
            Compute();
            CopyOut(localIdx);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t localIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[static_cast<uint64_t>(localIdx) * planeIn], planeIn);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // 2D MaxPool，padding越界视为-Inf
        // 简单实现：在UB上进行循环
        for (uint32_t ohIdx = 0; ohIdx < oh; ++ohIdx) {
            int64_t ih0 = static_cast<int64_t>(ohIdx) * static_cast<int64_t>(sh) - static_cast<int64_t>(ph);
            for (uint32_t owIdx = 0; owIdx < ow; ++owIdx) {
                int64_t iw0 = static_cast<int64_t>(owIdx) * static_cast<int64_t>(sw) - static_cast<int64_t>(pw);
                DTYPE_Y maxVal = static_cast<DTYPE_Y>(-3.402823466e+38F); // -INF for float
                for (uint32_t khIdx = 0; khIdx < kh; ++khIdx) {
                    int64_t ihCur = ih0 + static_cast<int64_t>(khIdx);
                    if (ihCur < 0 || ihCur >= static_cast<int64_t>(h)) continue;
                    for (uint32_t kwIdx = 0; kwIdx < kw; ++kwIdx) {
                        int64_t iwCur = iw0 + static_cast<int64_t>(kwIdx);
                        if (iwCur < 0 || iwCur >= static_cast<int64_t>(w)) continue;
                        uint64_t inOffset = static_cast<uint64_t>(ihCur) * static_cast<uint64_t>(w) + static_cast<uint64_t>(iwCur);
                        DTYPE_Y v = static_cast<DTYPE_Y>(xLocal[inOffset]);
                        maxVal = (v > maxVal) ? v : maxVal;
                    }
                }
                uint64_t outOffset = static_cast<uint64_t>(ohIdx) * static_cast<uint64_t>(ow) + static_cast<uint64_t>(owIdx);
                yLocal[outOffset] = maxVal;
            }
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t localIdx)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[static_cast<uint64_t>(localIdx) * planeOut], yLocal, planeOut);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, h, w;
    uint32_t kh, kw, sh, sw, ph, pw;
    uint32_t oh, ow;

    uint32_t planeIn, planeOut;
    uint32_t totalPlanes, blocks, planesPerBlock, startPlane, endPlane;
};

extern "C" __global__ __aicore__ void max_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPool2d op;
    op.Init(x, y,
            tiling_data.n(), tiling_data.c(), tiling_data.h(), tiling_data.w(),
            tiling_data.kh(), tiling_data.kw(), tiling_data.sh(), tiling_data.sw(),
            tiling_data.ph(), tiling_data.pw(), tiling_data.oh(), tiling_data.ow(),
            tiling_data.blockNum());
    op.Process();
}
