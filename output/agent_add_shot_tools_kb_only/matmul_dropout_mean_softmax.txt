kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

using DTYPE_X = half;
using DTYPE_W = half;
using DTYPE_B = float;
using DTYPE_Z = float;

class KernelMatmulDropoutSoftmax {
public:
    __aicore__ inline KernelMatmulDropoutSoftmax() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR p, GM_ADDR z,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileNumM, uint32_t tileNumN)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileNumM = tileNumM;
        this->tileNumN = tileNumN;

        // 每个核处理的行数
        rowsPerBlock = M / AscendC::GetBlockNum();
        startRow = rowsPerBlock * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + startRow * K, rowsPerBlock * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, N * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, N);
        pGm.SetGlobalBuffer((__gm__ DTYPE_B*)p, 1);  // p标量，float
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z + startRow * N, rowsPerBlock * N);

        // 行分块=1，列分块tileNumN
        colTile = N / (tileNumN * BUFFER_NUM);
        if (colTile == 0) { colTile = N; } // 兜底

        // 为输入、权重分块以及输出开辟本地队列
        pipe.InitBuffer(inQueueX, BUFFER_NUM, K * sizeof(DTYPE_X));              // 一行x
        pipe.InitBuffer(inQueueW, BUFFER_NUM, colTile * K * sizeof(DTYPE_W));    // 权重块[n_tile, K]
        pipe.InitBuffer(inQueueB, BUFFER_NUM, colTile * sizeof(DTYPE_B));        // 偏置块[n_tile]
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, colTile * sizeof(DTYPE_Z));       // 输出块[n_tile]
        pipe.InitBuffer(tmpQueueSoftmax, BUFFER_NUM, colTile * sizeof(DTYPE_Z)); // softmax中间缓存
    }

    __aicore__ inline void Process()
    {
        // 读取dropout p
        DTYPE_B pVal = *((__gm__ DTYPE_B*)pGm.GetPhyAddr());
        DTYPE_B keepProb = 1.0f - pVal;
        // 主循环：逐行处理
        for (uint32_t r = 0; r < rowsPerBlock; ++r) {
            // 拷贝当前行的x到本地
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[r * K], K);
            inQueueX.EnQue(xLocal);

            // 计算当前行的 matmul + bias，按列分块
            DTYPE_B rowMax = -3.402823e38f; // -FLT_MAX
            // 第一遍：计算输出并找最大值（用于softmax稳定性）
            for (int32_t t = 0; t < tileNumN * BUFFER_NUM; ++t) {
                // 读入权重和bias分块
                AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.AllocTensor<DTYPE_W>();
                AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
                uint32_t colStart = t * colTile;
                AscendC::DataCopy(wLocal, wGm[colStart * K], colTile * K);
                AscendC::DataCopy(bLocal, bGm[colStart], colTile);
                inQueueW.EnQue(wLocal);
                inQueueB.EnQue(bLocal);

                // 取出x、w、b，计算该分块的y_tile = x * w^T + b
                AscendC::LocalTensor<DTYPE_X> xRow = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_W> wBlk = inQueueW.DeQue<DTYPE_W>();
                AscendC::LocalTensor<DTYPE_B> bBlk = inQueueB.DeQue<DTYPE_B>();
                AscendC::LocalTensor<DTYPE_Z> zTile = outQueueZ.AllocTensor<DTYPE_Z>();

                // 逐列点积
                for (uint32_t c = 0; c < colTile; ++c) {
                    DTYPE_B acc = 0.0f;
                    // wBlk布局: [colTile, K]
                    for (uint32_t kk = 0; kk < K; ++kk) {
                        acc += (DTYPE_B)xRow.GetValue(kk) * (DTYPE_B)wBlk.GetValue(c * K + kk);
                    }
                    acc += (DTYPE_B)bBlk.GetValue(c);
                    // dropout: 采用期望缩放（不采样） acc *= keepProb
                    acc *= keepProb;
                    zTile.SetValue(c, acc);
                    if (acc > rowMax) { rowMax = acc; }
                }

                outQueueZ.EnQue(zTile);
                inQueueW.FreeTensor(wBlk);
                inQueueB.FreeTensor(bBlk);
                // xRow留在队列中，不释放，后续还需要
            }

            // 第二遍：softmax归一化
            // 计算 sum(exp(z - rowMax))
            DTYPE_B sumExp = 0.0f;
            for (int32_t t = 0; t < tileNumN * BUFFER_NUM; ++t) {
                AscendC::LocalTensor<DTYPE_Z> zTile = outQueueZ.DeQue<DTYPE_Z>();
                AscendC::LocalTensor<DTYPE_Z> tmpTile = tmpQueueSoftmax.AllocTensor<DTYPE_Z>();
                for (uint32_t c = 0; c < colTile; ++c) {
                    DTYPE_B val = zTile.GetValue(c);
                    DTYPE_B e = AscendC::Exp(val - rowMax);
                    tmpTile.SetValue(c, e);
                    sumExp += e;
                }
                tmpQueueSoftmax.EnQue(tmpTile);
                outQueueZ.FreeTensor(zTile);
            }

            // 写回softmax结果
            for (int32_t t = 0; t < tileNumN * BUFFER_NUM; ++t) {
                AscendC::LocalTensor<DTYPE_Z> tmpTile = tmpQueueSoftmax.DeQue<DTYPE_Z>();
                AscendC::LocalTensor<DTYPE_Z> zTileOut = outQueueZ.AllocTensor<DTYPE_Z>();
                for (uint32_t c = 0; c < colTile; ++c) {
                    DTYPE_B e = tmpTile.GetValue(c);
                    zTileOut.SetValue(c, e / sumExp);
                }
                // 输出到GM
                uint32_t colStart = t * colTile;
                AscendC::DataCopy(zGm[r * N + colStart], zTileOut, colTile);
                outQueueZ.FreeTensor(zTileOut);
            }

            // 清理x行本地缓存
            AscendC::LocalTensor<DTYPE_X> xRowClear = inQueueX.DeQue<DTYPE_X>();
            inQueueX.FreeTensor(xRowClear);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueW, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueueSoftmax;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_B> pGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t M, N, K;
    uint32_t tileNumM, tileNumN;
    uint32_t rowsPerBlock, startRow;
    uint32_t colTile;
};

extern "C" __global__ __aicore__ void matmul_dropout_mean_softmax_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR p, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    // 读取自定义tiling数据
    GET_TILING_DATA(tiling_data, tiling);
    uint32_t M = tiling_data.M;
    uint32_t N = tiling_data.N;
    uint32_t K = tiling_data.K;
    uint32_t tileNumM = tiling_data.tileNumM;
    uint32_t tileNumN = tiling_data.tileNumN;

    KernelMatmulDropoutSoftmax op;
    op.Init(x, weight, bias, p, z, M, N, K, tileNumM, tileNumN);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused Matmul + Dropout(scale) + Softmax using custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, dropout_p):
        super(ModelNew, self).__init__()
        # 权重与偏置，匹配Linear
        self.weight = torch.nn.Parameter(torch.randn(out_features, in_features, dtype=torch.float16))
        self.bias = torch.nn.Parameter(torch.zeros(out_features, dtype=torch.float))
        self.dropout_p = float(dropout_p)

    def forward(self, x):
        # x半精度以匹配Matmul高阶API约束
        x_fp16 = x.to(torch.float16)
        p = torch.tensor([self.dropout_p], dtype=torch.float, device=x.device)
        return custom_ops_lib.matmul_dropout_mean_softmax_custom(x_fp16, self.weight, self.bias, p)
