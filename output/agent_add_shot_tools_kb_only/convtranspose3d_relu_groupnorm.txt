kernel_src="""
#include "kernel_operator.h"
#include <cmath>

extern "C" __global__ __aicore__ void convtranspose3d_relu_groupnorm_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    const uint32_t N = tilingData.n;
    const uint32_t C = tilingData.c;
    const uint32_t D = tilingData.d;
    const uint32_t H = tilingData.h;
    const uint32_t W = tilingData.w;
    const uint32_t G = tilingData.groups;
    const float eps = tilingData.eps;

    if (C == 0 || G == 0) {
        return;
    }

    const uint32_t cPerGroup = C / G;
    const uint64_t inner = static_cast<uint64_t>(D) * H * W;
    const uint64_t groupElems = static_cast<uint64_t>(cPerGroup) * inner;

    __gm__ float* xGm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* yGm = reinterpret_cast<__gm__ float*>(y);
    __gm__ float* gammaGm = reinterpret_cast<__gm__ float*>(gamma);
    __gm__ float* betaGm = reinterpret_cast<__gm__ float*>(beta);

    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t blocksPerN = G;
    const uint32_t nIdx = (blocksPerN == 0) ? 0 : (blockIdx / blocksPerN);
    const uint32_t gIdx = (blocksPerN == 0) ? 0 : (blockIdx % blocksPerN);
    if (nIdx >= N) {
        return;
    }

    const uint32_t cStart = gIdx * cPerGroup;

    // First pass: compute mean and variance over ReLU(x) within this (n, group)
    float sum = 0.0f;
    float sumsq = 0.0f;

    for (uint32_t cOff = 0; cOff < cPerGroup; ++cOff) {
        const uint32_t cIdx = cStart + cOff;
        const uint64_t base = (static_cast<uint64_t>(nIdx) * C + cIdx) * inner;
        for (uint64_t i = 0; i < inner; ++i) {
            float v = xGm[base + i];
            if (v < 0.f) v = 0.f; // ReLU
            sum += v;
            sumsq += v * v;
        }
    }

    const float mean = sum / static_cast<float>(groupElems);
    const float var = sumsq / static_cast<float>(groupElems) - mean * mean;
    const float rstd = 1.0f / sqrtf(var + eps);

    // Second pass: normalize and affine using gamma/beta per channel
    for (uint32_t cOff = 0; cOff < cPerGroup; ++cOff) {
        const uint32_t cIdx = cStart + cOff;
        const float g = gammaGm[cIdx];
        const float b = betaGm[cIdx];
        const uint64_t base = (static_cast<uint64_t>(nIdx) * C + cIdx) * inner;
        for (uint64_t i = 0; i < inner; ++i) {
            float v = xGm[base + i];
            if (v < 0.f) v = 0.f; // ReLU
            float nv = (v - mean) * rstd;
            yGm[base + i] = nv * g + b;
        }
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized: keep ConvTranspose3d in PyTorch, fuse ReLU + GroupNorm into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, groups, bias=False):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, bias=bias)
        self.groups = groups
        # GroupNorm params (gamma/beta) are kept to be passed into custom op
        self.group_norm = torch.nn.GroupNorm(num_groups=groups, num_channels=out_channels)

    def forward(self, x):
        y = self.conv_transpose(x)
        # Use custom fused kernel: ReLU + GroupNorm
        y = custom_ops_lib.convtranspose3d_relu_groupnorm_custom(
            y, self.group_norm.weight, self.group_norm.bias, self.group_norm.num_groups, float(self.group_norm.eps))
        return y
