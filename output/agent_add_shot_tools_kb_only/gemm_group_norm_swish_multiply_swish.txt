kernel_src="""
#include "kernel_operator.h"

using DTYPE_T = float;
constexpr int32_t BUFFER_NUM = 1;

// Fused: GEMM -> GroupNorm -> Swish -> Multiply -> Swish
class KernelGemmGroupNormSwishMultiplySwish {
public:
    __aicore__ inline KernelGemmGroupNormSwishMultiplySwish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR gn_w, GM_ADDR gn_b, GM_ADDR mul_w,
                                GM_ADDR y,
                                uint32_t batch, uint32_t inF, uint32_t outF,
                                uint32_t numGroups, uint32_t tileNum, uint32_t blockDim, float epsilon)
    {
        this->B = batch;
        this->K = inF;
        this->N = outF;
        this->G = numGroups;
        this->eps = epsilon;
        this->tileNum = tileNum;

        // Split batch across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t thisBlock = AscendC::GetBlockIdx();
        this->BperBlock = (B + blocks - 1) / blocks;
        uint32_t bStart = thisBlock * this->BperBlock;
        if (bStart + this->BperBlock > B) { this->BperBlock = B - bStart; }
        this->validB = this->BperBlock;

        xGm.SetGlobalBuffer((__gm__ DTYPE_T*)x + bStart * K, this->validB * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE_T*)w, N * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_T*)b, N);
        gnWGm.SetGlobalBuffer((__gm__ DTYPE_T*)gn_w, N);
        gnBGm.SetGlobalBuffer((__gm__ DTYPE_T*)gn_b, N);
        mulWGm.SetGlobalBuffer((__gm__ DTYPE_T*)mul_w, N);
        yGm.SetGlobalBuffer((__gm__ DTYPE_T*)y + bStart * N, this->validB * N);

        // One-shot local buffers
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->validB * K * sizeof(DTYPE_T));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, N * K * sizeof(DTYPE_T));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, N * sizeof(DTYPE_T));
        pipe.InitBuffer(inQueueGNW, BUFFER_NUM, N * sizeof(DTYPE_T));
        pipe.InitBuffer(inQueueGNB, BUFFER_NUM, N * sizeof(DTYPE_T));
        pipe.InitBuffer(inQueueMulW, BUFFER_NUM, N * sizeof(DTYPE_T));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->validB * N * sizeof(DTYPE_T));
    }

    __aicore__ inline void Process()
    {
        // CopyIn all operands
        AscendC::LocalTensor<DTYPE_T> xLocal = inQueueX.AllocTensor<DTYPE_T>();
        AscendC::LocalTensor<DTYPE_T> wLocal = inQueueW.AllocTensor<DTYPE_T>();
        AscendC::LocalTensor<DTYPE_T> bLocal = inQueueB.AllocTensor<DTYPE_T>();
        AscendC::LocalTensor<DTYPE_T> gnWLocal = inQueueGNW.AllocTensor<DTYPE_T>();
        AscendC::LocalTensor<DTYPE_T> gnBLocal = inQueueGNB.AllocTensor<DTYPE_T>();
        AscendC::LocalTensor<DTYPE_T> mulWLocal = inQueueMulW.AllocTensor<DTYPE_T>();
        AscendC::LocalTensor<DTYPE_T> yLocal = outQueueY.AllocTensor<DTYPE_T>();

        AscendC::DataCopy(xLocal, xGm, this->validB * K);
        AscendC::DataCopy(wLocal, wGm, N * K);
        AscendC::DataCopy(bLocal, bGm, N);
        AscendC::DataCopy(gnWLocal, gnWGm, N);
        AscendC::DataCopy(gnBLocal, gnBGm, N);
        AscendC::DataCopy(mulWLocal, mulWGm, N);

        // GEMM: [B,K] x [N,K]^T -> [B,N], then +bias
        // Note: Assume high-level MatMul API is available. Transpose weight on the fly.
        // yLocal = xLocal @ wLocal^T + bLocal
        AscendC::MatMul(yLocal, xLocal, wLocal, this->validB, N, K, /*transA=*/false, /*transB=*/true);
        // Add bias
        for (uint32_t i = 0; i < this->validB; ++i) {
            AscendC::Add(yLocal[i * N], yLocal[i * N], bLocal, N);
        }

        // GroupNorm over channel axis (N) for each batch sample, with G groups
        // Per sample: split N into G groups of size S = N/G
        uint32_t S = N / (G == 0 ? 1 : G);
        for (uint32_t i = 0; i < this->validB; ++i) {
            for (uint32_t g = 0; g < G; ++g) {
                uint32_t offset = i * N + g * S;
                // Compute mean and variance for this group
                DTYPE_T meanVal = 0.0f;
                for (uint32_t s = 0; s < S; ++s) { meanVal += yLocal[offset + s]; }
                meanVal /= static_cast<DTYPE_T>(S);
                DTYPE_T varVal = 0.0f;
                for (uint32_t s = 0; s < S; ++s) {
                    DTYPE_T diff = yLocal[offset + s] - meanVal;
                    varVal += diff * diff;
                }
                varVal /= static_cast<DTYPE_T>(S);
                DTYPE_T rstd = static_cast<DTYPE_T>(1.0f) / AscendC::Sqrt(varVal + static_cast<DTYPE_T>(eps));

                // Normalize and apply gn_weight/bias
                for (uint32_t s = 0; s < S; ++s) {
                    uint32_t c = g * S + s;
                    DTYPE_T xn = (yLocal[offset + s] - meanVal) * rstd;
                    yLocal[offset + s] = xn * gnWLocal[c] + gnBLocal[c];
                }
            }
        }

        // Swish: x * sigmoid(x)
        for (uint32_t i = 0; i < this->validB * N; ++i) {
            DTYPE_T val = yLocal[i];
            DTYPE_T sig = AscendC::Sigmoid(val);
            yLocal[i] = val * sig;
        }

        // Multiply with vector multiply_weight (broadcast over batch)
        for (uint32_t i = 0; i < this->validB; ++i) {
            AscendC::Mul(yLocal[i * N], yLocal[i * N], mulWLocal, N);
        }

        // Swish again
        for (uint32_t i = 0; i < this->validB * N; ++i) {
            DTYPE_T val = yLocal[i];
            DTYPE_T sig = AscendC::Sigmoid(val);
            yLocal[i] = val * sig;
        }

        // Write back
        AscendC::DataCopy(yGm, yLocal, this->validB * N);

        // Free
        inQueueX.FreeTensor(xLocal);
        inQueueW.FreeTensor(wLocal);
        inQueueB.FreeTensor(bLocal);
        inQueueGNW.FreeTensor(gnWLocal);
        inQueueGNB.FreeTensor(gnBLocal);
        inQueueMulW.FreeTensor(mulWLocal);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueGNW;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueGNB;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueMulW;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_T> xGm;
    AscendC::GlobalTensor<DTYPE_T> wGm;
    AscendC::GlobalTensor<DTYPE_T> bGm;
    AscendC::GlobalTensor<DTYPE_T> gnWGm;
    AscendC::GlobalTensor<DTYPE_T> gnBGm;
    AscendC::GlobalTensor<DTYPE_T> mulWGm;
    AscendC::GlobalTensor<DTYPE_T> yGm;

    uint32_t B, K, N, G;
    uint32_t BperBlock;
    uint32_t validB;
    uint32_t tileNum;
    float eps;
};

extern "C" __global__ __aicore__ void gemm_group_norm_swish_multiply_swish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR gn_weight, GM_ADDR gn_bias, GM_ADDR multiply_weight,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmGroupNormSwishMultiplySwish op;
    op.Init(x, weight, bias, gn_weight, gn_bias, multiply_weight, y,
            tiling_data.batch, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.numGroups, tiling_data.tileNum, tiling_data.blockDim, tiling_data.epsilon);
    op.Process();
}
