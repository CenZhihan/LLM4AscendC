const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    AdamCustomTilingData tiling;
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    context->SetBlockDim(BLOCK_DIM);

    // 使用与原始模型一致的默认超参
    const float beta1 = 0.9f;
    const float beta2 = 0.999f;
    const float lr = 1e-3f;
    const float eps = 1e-8f;
    const int step = 1;

    const float oneMinusBeta1 = 1.0f - beta1;
    const float oneMinusBeta2 = 1.0f - beta2;
    const float bc1 = 1.0f / (1.0f - std::pow(beta1, static_cast<float>(step)));
    const float bc2 = 1.0f / (1.0f - std::pow(beta2, static_cast<float>(step)));

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_beta1(beta1);
    tiling.set_oneMinusBeta1(oneMinusBeta1);
    tiling.set_beta2(beta2);
    tiling.set_oneMinusBeta2(oneMinusBeta2);
    tiling.set_lr(lr);
    tiling.set_eps(eps);
    tiling.set_bc1(bc1);
    tiling.set_bc2(bc2);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelAdam {
public:
    __aicore__ inline KernelAdam() {}

    __aicore__ inline void Init(GM_ADDR param, GM_ADDR grad, GM_ADDR m, GM_ADDR v, GM_ADDR out,
                                uint32_t totalLength, uint32_t tileNum,
                                float beta1, float oneMinusBeta1,
                                float beta2, float oneMinusBeta2,
                                float lr, float eps, float bc1, float bc2)
    {
        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        ASSERT(tileNum != 0 && "tile num can not be zero!");
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        paramGm.SetGlobalBuffer((__gm__ float*)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gradGm.SetGlobalBuffer((__gm__ float*)grad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mGm.SetGlobalBuffer((__gm__ float*)m + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ float*)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ float*)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueParam, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueGrad,  BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueM,     BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueV,     BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueOut,  BUFFER_NUM, this->tileLength * sizeof(float));

        this->beta1 = beta1;
        this->oneMinusBeta1 = oneMinusBeta1;
        this->beta2 = beta2;
        this->oneMinusBeta2 = oneMinusBeta2;
        this->lr = lr;
        this->eps = eps;
        this->bc1 = bc1; // 1/(1 - beta1^step)
        this->bc2 = bc2; // 1/(1 - beta2^step)
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> paramLocal = inQueueParam.AllocTensor<float>();
        AscendC::LocalTensor<float> gradLocal  = inQueueGrad.AllocTensor<float>();
        AscendC::LocalTensor<float> mLocal     = inQueueM.AllocTensor<float>();
        AscendC::LocalTensor<float> vLocal     = inQueueV.AllocTensor<float>();

        AscendC::DataCopy(paramLocal, paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gradLocal,  gradGm[progress * this->tileLength],  this->tileLength);
        AscendC::DataCopy(mLocal,     mGm[progress * this->tileLength],     this->tileLength);
        AscendC::DataCopy(vLocal,     vGm[progress * this->tileLength],     this->tileLength);

        inQueueParam.EnQue(paramLocal);
        inQueueGrad.EnQue(gradLocal);
        inQueueM.EnQue(mLocal);
        inQueueV.EnQue(vLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<float> paramLocal = inQueueParam.DeQue<float>();
        AscendC::LocalTensor<float> gradLocal  = inQueueGrad.DeQue<float>();
        AscendC::LocalTensor<float> mLocal     = inQueueM.DeQue<float>();
        AscendC::LocalTensor<float> vLocal     = inQueueV.DeQue<float>();
        AscendC::LocalTensor<float> outLocal   = outQueueOut.AllocTensor<float>();

        // m_new = beta1 * m + (1 - beta1) * grad
        AscendC::Muls(mLocal, mLocal, this->beta1, this->tileLength);                 // mLocal = beta1 * m
        AscendC::Muls(outLocal, gradLocal, this->oneMinusBeta1, this->tileLength);    // outLocal = (1-beta1)*grad
        AscendC::Add(mLocal, mLocal, outLocal, this->tileLength);                     // mLocal = m_new

        // v_new = beta2 * v + (1 - beta2) * grad^2
        AscendC::Muls(vLocal, vLocal, this->beta2, this->tileLength);                 // vLocal = beta2 * v
        AscendC::Mul(outLocal, gradLocal, gradLocal, this->tileLength);               // outLocal = grad^2
        AscendC::Muls(outLocal, outLocal, this->oneMinusBeta2, this->tileLength);     // outLocal = (1-beta2)*grad^2
        AscendC::Add(vLocal, vLocal, outLocal, this->tileLength);                     // vLocal = v_new

        // m_hat = m_new * bc1 ; v_hat = v_new * bc2
        AscendC::Muls(mLocal, mLocal, this->bc1, this->tileLength);                   // mLocal = m_hat
        AscendC::Muls(vLocal, vLocal, this->bc2, this->tileLength);                   // vLocal = v_hat

        // denom = sqrt(v_hat) + eps
        AscendC::Sqrt(vLocal, vLocal, this->tileLength);                               // vLocal = sqrt(v_hat)
        AscendC::Adds(vLocal, vLocal, this->eps, this->tileLength);                    // vLocal = sqrt(v_hat) + eps

        // update = lr * m_hat / denom = lr * m_hat * reciprocal(denom)
        AscendC::Reciprocal(vLocal, vLocal, this->tileLength);                         // vLocal = 1 / denom
        AscendC::Mul(outLocal, mLocal, vLocal, this->tileLength);                      // outLocal = m_hat / denom
        AscendC::Muls(outLocal, outLocal, this->lr, this->tileLength);                 // outLocal = lr * m_hat / denom

        // param_out = param - update
        AscendC::Sub(outLocal, paramLocal, outLocal, this->tileLength);

        outQueueOut.EnQue<float>(outLocal);
        inQueueParam.FreeTensor(paramLocal);
        inQueueGrad.FreeTensor(gradLocal);
        inQueueM.FreeTensor(mLocal);
        inQueueV.FreeTensor(vLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> outLocal = outQueueOut.DeQue<float>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inQueueParam;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inQueueGrad;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inQueueM;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inQueueV;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueOut;

    AscendC::GlobalTensor<float> paramGm;
    AscendC::GlobalTensor<float> gradGm;
    AscendC::GlobalTensor<float> mGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> outGm;

    uint32_t blockLength = 0;
    uint32_t tileNum = 0;
    uint32_t tileLength = 0;

    float beta1 = 0.9f;
    float oneMinusBeta1 = 0.1f;
    float beta2 = 0.999f;
    float oneMinusBeta2 = 0.001f;
    float lr = 1e-3f;
    float eps = 1e-8f;
    float bc1 = 1.0f;
    float bc2 = 1.0f;
};

extern "C" __global__ __aicore__ void adam_custom(GM_ADDR param, GM_ADDR grad, GM_ADDR m, GM_ADDR v,
                                                  GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelAdam op;
    op.Init(param, grad, m, v, out,
            tilingData.totalLength, tilingData.tileNum,
            tilingData.beta1, tilingData.oneMinusBeta1,
            tilingData.beta2, tilingData.oneMinusBeta2,
            tilingData.lr, tilingData.eps, tilingData.bc1, tilingData.bc2);
    op.Process();
}
