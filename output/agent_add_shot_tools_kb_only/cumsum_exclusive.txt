kernel_src="""
#include "kernel_operator.h"

class KernelCumsumExclusiveCustom {
public:
    __aicore__ inline KernelCumsumExclusiveCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outter, uint32_t inner)
    {
        this->outter = outter;
        this->inner = inner;

        // 多核切分：按行(outter)维度均分
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (outter + blockNum - 1) / blockNum;
        this->startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = AscendC::Min(this->startRow + rowsPerBlock, outter);
        this->rowsThisBlock = (endRow > this->startRow) ? (endRow - this->startRow) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->startRow * this->inner, this->rowsThisBlock * this->inner);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->startRow * this->inner, this->rowsThisBlock * this->inner);

        pipe.InitBuffer(inQueueX, 1, this->inner * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, 1, this->inner * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, 1, this->inner * sizeof(DTYPE_X));      // inclusive结果临时缓冲
        pipe.InitBuffer(lastRowQueue, 1, this->inner * sizeof(DTYPE_X));  // CumSum接口所需lastRow缓冲
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsThisBlock; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->inner], this->inner);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*rowIdx*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> tmpLocal = tmpQueue.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> lastRowLocal = lastRowQueue.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // 配置：沿最后一维做累加，不复用源，不输出lastRow
        static constexpr AscendC::CumSumConfig cumSumConfig{true, false, false};
        const AscendC::CumSumInfo info{1, this->inner}; // 单行处理：outter=1, inner=内层长度

        // 先算inclusive cumsum到tmpLocal
        AscendC::CumSum<DTYPE_X, cumSumConfig>(tmpLocal, lastRowLocal, xLocal, info);

        // exclusive = [0, inclusive[:-1]]
        DTYPE_Y zeroVal = static_cast<DTYPE_Y>(0);
        AscendC::Duplicate<DTYPE_Y>(yLocal, zeroVal, this->inner);
        if (this->inner > 1) {
            AscendC::DataCopy(yLocal[1], tmpLocal, this->inner - 1);
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        tmpQueue.FreeTensor(tmpLocal);
        lastRowQueue.FreeTensor(lastRowLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[rowIdx * this->inner], yLocal, this->inner);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> lastRowQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t outter;
    uint32_t inner;
    uint32_t startRow;
    uint32_t rowsThisBlock;
};

extern "C" __global__ __aicore__ void cumsum_exclusive_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    if (TILING_KEY_IS(1)) {
        KernelCumsumExclusiveCustom op;
        op.Init(x, y, tiling_data.outter, tiling_data.inner);
        op.Process();
    }
}
"""
