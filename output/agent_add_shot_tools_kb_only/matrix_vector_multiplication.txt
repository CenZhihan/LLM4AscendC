#include "matrix_vector_multiplication_custom_tiling.h"

namespace optiling {
static const uint32_t BLOCK_DIM = 8;
static const uint32_t DEFAULT_TILE_K = 4096; // 每次按K维处理的块大小

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MatrixVectorMultiplicationCustomTilingData tiling;

    // A: [M, K], B: [K, 1]
    // 使用A总元素数除以B总元素数得到M，并直接使用B总元素数作为K
    uint64_t aSize = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    uint64_t bSize = context->GetInputShape(1)->GetOriginShape().GetShapeSize(); // K*1=K
    if (bSize == 0) {
        return ge::GRAPH_FAILED;
    }
    uint64_t M = aSize / bSize;
    uint64_t K = bSize;

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalM(static_cast<uint32_t>(M));
    tiling.set_totalK(static_cast<uint32_t>(K));
    tiling.set_tileK(DEFAULT_TILE_K);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // 输出形状 [M, 1]，其中M来自A[0]，1来自B[1]
    const gert::Shape* a_shape = context->GetInputShape(0);
    gert::Shape* c_shape = context->GetOutputShape(0);
    // 简化处理：设为与A相同的未知形状格式下的占位，实际执行时由tiling与kernel保证正确拷贝范围
    *c_shape = *a_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t QUEUE_DEPTH = 1;

class KernelMatVec {
public:
    __aicore__ inline KernelMatVec() {}
    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t totalM, uint32_t totalK, uint32_t tileK)
    {
        this->totalM = totalM;
        this->totalK = totalK;
        this->tileK = tileK;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // 采用按比例切分行的方式，避免除不尽问题
        this->startRow = (totalM * blockIdx) / blockNum;
        this->endRow = (totalM * (blockIdx + 1)) / blockNum;
        this->blockRows = (this->endRow > this->startRow) ? (this->endRow - this->startRow) : 0;

        aGm.SetGlobalBuffer((__gm__ float*)A + static_cast<uint64_t>(this->startRow) * this->totalK,
                            static_cast<uint64_t>(this->blockRows) * this->totalK);
        bGm.SetGlobalBuffer((__gm__ float*)B, this->totalK);
        cGm.SetGlobalBuffer((__gm__ float*)C + this->startRow, this->blockRows);

        pipe.InitBuffer(inQueueA, QUEUE_DEPTH, this->tileK * sizeof(float));
        pipe.InitBuffer(inQueueB, QUEUE_DEPTH, this->tileK * sizeof(float));
        pipe.InitBuffer(calcQueueTmp, QUEUE_DEPTH, this->tileK * sizeof(float));
        pipe.InitBuffer(calcQueueWork, QUEUE_DEPTH, this->tileK * sizeof(float));
        pipe.InitBuffer(outQueueC, QUEUE_DEPTH, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (this->blockRows == 0) {
            return;
        }
        for (uint32_t r = 0; r < this->blockRows; ++r) {
            float acc = 0.0f;
            for (uint32_t k0 = 0; k0 < this->totalK; k0 += this->tileK) {
                uint32_t currTile = this->tileK;
                uint32_t remain = this->totalK - k0;
                if (remain < currTile) currTile = remain;

                AscendC::LocalTensor<float> bLocal = inQueueB.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm[k0], currTile);

                AscendC::LocalTensor<float> aLocal = inQueueA.AllocTensor<float>();
                // A按行连续存储，取r行的[k0, k0+currTile)
                AscendC::DataCopy(aLocal, aGm[(static_cast<uint64_t>(r) * this->totalK) + k0], currTile);

                AscendC::LocalTensor<float> tmpLocal = calcQueueTmp.AllocTensor<float>();
                AscendC::Mul(tmpLocal, aLocal, bLocal, currTile);

                AscendC::LocalTensor<float> workLocal = calcQueueWork.AllocTensor<float>();
                // 使用ReduceSum对tmpLocal进行求和
                AscendC::ReduceSum(tmpLocal, tmpLocal, workLocal, currTile);
                float part = AscendC::GetAccVal<float>();
                acc += part;

                calcQueueWork.FreeTensor(workLocal);
                calcQueueTmp.FreeTensor(tmpLocal);
                inQueueA.FreeTensor(aLocal);
                inQueueB.FreeTensor(bLocal);
            }

            AscendC::LocalTensor<float> cScalar = outQueueC.AllocTensor<float>();
            AscendC::Duplicate(cScalar, acc, 1);
            AscendC::DataCopy(cGm[r], cScalar, 1);
            outQueueC.FreeTensor(cScalar);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, QUEUE_DEPTH> inQueueA, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECCALC, QUEUE_DEPTH> calcQueueTmp, calcQueueWork;
    AscendC::TQue<AscendC::TPosition::VECOUT, QUEUE_DEPTH> outQueueC;

    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> cGm;

    uint32_t totalM;
    uint32_t totalK;
    uint32_t tileK;

    uint32_t startRow;
    uint32_t endRow;
    uint32_t blockRows;
};

extern "C" __global__ __aicore__ void matrix_vector_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatVec op;
    op.Init(A, B, C, tiling_data.totalM, tiling_data.totalK, tiling_data.tileK);
    op.Process();
}
"""
