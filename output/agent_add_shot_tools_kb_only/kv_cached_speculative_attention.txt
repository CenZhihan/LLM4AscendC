namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto qShape = context->GetInputShape(0)->GetOriginShape();
    auto kShape = context->GetInputShape(1)->GetOriginShape();
    auto vShape = context->GetInputShape(2)->GetOriginShape();
    // Expect ND: [B, q_len, d], [B, kv_len, d], [B, kv_len, d]
    uint32_t b = static_cast<uint32_t>(qShape.GetDim(0));
    uint32_t q_len = static_cast<uint32_t>(qShape.GetDim(1));
    uint32_t d_model = static_cast<uint32_t>(qShape.GetDim(2));
    uint32_t kv_len = static_cast<uint32_t>(kShape.GetDim(1));

    AttnHeader header{};
    header.b = b;
    header.q_len = q_len;
    header.kv_len = kv_len;
    header.d_model = d_model;
    header.scale = 1.0f / std::sqrt(static_cast<float>(d_model));
    header.reserved = 0;

    // Build two Matmul tilings
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());

    // Tiling 1: [q_len, d] x [d, kv_len] -> [q_len, kv_len]
    matmul_tiling::MultiCoreMatmulTiling tiling1(ascendcPlatform);
    tiling1.SetDim(1);
    tiling1.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling1.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling1.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling1.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling1.SetShape(q_len, kv_len, d_model);
    tiling1.SetOrgShape(q_len, kv_len, d_model);
    tiling1.SetBias(false);
    tiling1.SetBufferSpace(-1, -1, -1);
    optiling::TCubeTiling cube1{};
    tiling1.GetTiling(cube1);

    // Tiling 2: [q_len, kv_len] x [kv_len, d] -> [q_len, d]
    matmul_tiling::MultiCoreMatmulTiling tiling2(ascendcPlatform);
    tiling2.SetDim(1);
    tiling2.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling2.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling2.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling2.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling2.SetShape(q_len, d_model, kv_len);
    tiling2.SetOrgShape(q_len, d_model, kv_len);
    tiling2.SetBias(false);
    tiling2.SetBufferSpace(-1, -1, -1);
    optiling::TCubeTiling cube2{};
    tiling2.GetTiling(cube2);

    // Write raw tiling buffer: header + cube1 + cube2
    uint8_t* rawBuf = reinterpret_cast<uint8_t*>(context->GetRawTilingData()->GetData());
    size_t cap = context->GetRawTilingData()->GetCapacity();

    size_t headerSize = sizeof(AttnHeader);
    size_t cube1Size = sizeof(optiling::TCubeTiling);
    size_t cube2Size = sizeof(optiling::TCubeTiling);
    size_t totalSize = headerSize + cube1Size + cube2Size;
    if (cap < totalSize) {
        return ge::GRAPH_FAILED;
    }
    std::memcpy(rawBuf, &header, headerSize);
    std::memcpy(rawBuf + headerSize, &cube1, cube1Size);
    std::memcpy(rawBuf + headerSize + cube1Size, &cube2, cube2Size);
    context->GetRawTilingData()->SetDataSize(static_cast<uint32_t>(totalSize));

    // Set block dim and workspace size: allocate two buffers [B * q_len * kv_len] for scores and probs
    const uint32_t BLOCK_DIM = 1;
    context->SetBlockDim(BLOCK_DIM);
    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    size_t perBatchScores = static_cast<size_t>(q_len) * static_cast<size_t>(kv_len) * sizeof(float);
    workspaceSizes[0] = static_cast<size_t>(b) * perBatchScores * 2; // scores + probs
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using namespace AscendC;

#pragma pack(push, 1)
struct AttnHeader {
  uint32_t b;
  uint32_t q_len;
  uint32_t kv_len;
  uint32_t d_model;
  float    scale;
  uint32_t reserved;
};
#pragma pack(pop)

constexpr int32_t BUFFER_NUM = 1;

class KernelKvCachedSpecAttn {
public:
    __aicore__ inline KernelKvCachedSpecAttn() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling)
    {
        // Parse header + two cube tilings from tiling GM
        auto tilingPtr = reinterpret_cast<__gm__ uint8_t*>(tiling);
        // Read header
        AttnHeader headerTmp{};
        __gm__ uint8_t* hdrGm = tilingPtr;
        // Copy header fields
        __gm__ uint32_t* p32 = reinterpret_cast<__gm__ uint32_t*>(hdrGm);
        header.b = p32[0];
        header.q_len = p32[1];
        header.kv_len = p32[2];
        header.d_model = p32[3];
        // float scale at p32[4] interpreted separately
        __gm__ float* pf = reinterpret_cast<__gm__ float*>(hdrGm + 16);
        header.scale = *pf;

        // offsets
        uint32_t headerSize = sizeof(AttnHeader);
        __gm__ uint32_t* cube1Ptr = reinterpret_cast<__gm__ uint32_t*>(tilingPtr + headerSize);
        __gm__ uint32_t* cube2Ptr = reinterpret_cast<__gm__ uint32_t*>(tilingPtr + headerSize + sizeof(TCubeTiling));

        // Load tilings
        auto t1 = (__gm__ uint32_t*)cube1Ptr;
        auto t2 = (__gm__ uint32_t*)cube2Ptr;
        for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i) {
            ((uint32_t*)&cubeTiling1)[i] = t1[i];
            ((uint32_t*)&cubeTiling2)[i] = t2[i];
        }

        // Set GM tensors
        qGm.SetGlobalBuffer((__gm__ float*)q);
        kGm.SetGlobalBuffer((__gm__ float*)k);
        vGm.SetGlobalBuffer((__gm__ float*)v);
        oGm.SetGlobalBuffer((__gm__ float*)o);

        // Workspace: scores and probs, per-batch contiguous
        workspaceBase = (__gm__ float*)workspace;

        // Local buffers for softmax per row
        int64_t rowSize = header.kv_len; // number of floats per score row
        pipe.InitBuffer(rowInQueue, BUFFER_NUM, rowSize * sizeof(float));
        pipe.InitBuffer(rowOutQueue, BUFFER_NUM, rowSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Types for Matmul 1 and 2
        typedef matmul::MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> aType;
        typedef matmul::MatmulType<TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL> bTypeT;
        typedef matmul::MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> bTypeN;
        typedef matmul::MatmulType<TPosition::GM, CubeFormat::ND, float> cType;
        typedef matmul::MatmulType<TPosition::GM, CubeFormat::ND, float> biasType;

        // Matmul objects
        matmul::Matmul<aType, bTypeT, cType, biasType> mm1;
        matmul::Matmul<aType, bTypeN, cType, biasType> mm2;

        // Register to TPipe
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm2);

        // Init with tilings
        mm1.Init(&cubeTiling1);
        mm2.Init(&cubeTiling2);

        // Iterate batches
        for (uint32_t bidx = 0; bidx < header.b; ++bidx) {
            // Offsets in elements
            int64_t qBatchStride = (int64_t)header.q_len * header.d_model;
            int64_t kvBatchStride = (int64_t)header.kv_len * header.d_model;

            // Pointers per batch
            GlobalTensor<float> qBatch, kBatch, vBatch, sBatch, pBatch, oBatch;
            qBatch.SetGlobalBuffer(const_cast<__gm__ float*>(qGm[bidx * qBatchStride].GetPhyAddr()),
                                   header.q_len * header.d_model);
            kBatch.SetGlobalBuffer(const_cast<__gm__ float*>(kGm[bidx * kvBatchStride].GetPhyAddr()),
                                   header.kv_len * header.d_model);
            vBatch.SetGlobalBuffer(const_cast<__gm__ float*>(vGm[bidx * kvBatchStride].GetPhyAddr()),
                                   header.kv_len * header.d_model);

            // Workspace slices for this batch
            int64_t scoresBatchOffset = bidx * (int64_t)header.q_len * header.kv_len;
            int64_t probsBatchOffset  = scoresBatchOffset + (int64_t)header.q_len * header.kv_len;
            GlobalTensor<float> sGm, pGm;
            sGm.SetGlobalBuffer(const_cast<__gm__ float*>(workspaceBase + scoresBatchOffset),
                                header.q_len * header.kv_len);
            pGm.SetGlobalBuffer(const_cast<__gm__ float*>(workspaceBase + probsBatchOffset),
                                header.q_len * header.kv_len);

            // Matmul 1: S = Q @ K^T
            mm1.SetTensorA(qBatch, false);
            mm1.SetTensorB(kBatch, true);
            mm1.SetWorkspace((GM_ADDR)workspaceBase, 0);
            mm1.IterateBatch(sGm, 1, 1, false);

            // Row-wise: scale and softmax S -> P
            for (uint32_t qi = 0; qi < header.q_len; ++qi) {
                // Copy row qi from S to local
                LocalTensor<float> sLocal = rowInQueue.AllocTensor<float>();
                DataCopy(sLocal, sGm[qi * header.kv_len], header.kv_len);
                // scale
                Muls(sLocal, sLocal, header.scale, header.kv_len);
                // softmax
                LocalTensor<float> pLocal = rowOutQueue.AllocTensor<float>();
                SoftMax(pLocal, sLocal, smTiling); // rely on internal tiling check
                rowOutQueue.EnQue<float>(pLocal);
                rowInQueue.EnQue<float>(sLocal);

                // Copy back to GM P
                LocalTensor<float> pLocal2 = rowOutQueue.DeQue<float>();
                DataCopy(pGm[qi * header.kv_len], pLocal2, header.kv_len);
                rowOutQueue.FreeTensor(pLocal2);

                // free input local
                LocalTensor<float> sLocal2 = rowInQueue.DeQue<float>();
                rowInQueue.FreeTensor(sLocal2);
            }

            // Matmul 2: O = P @ V
            oBatch.SetGlobalBuffer(const_cast<__gm__ float*>(oGm[bidx * qBatchStride].GetPhyAddr()),
                                   header.q_len * header.d_model);
            mm2.SetTensorA(pGm, false);
            mm2.SetTensorB(vBatch, false);
            mm2.SetWorkspace((GM_ADDR)workspaceBase, 0);
            mm2.IterateBatch(oBatch, 1, 1, false);
        }
    }

private:
    AttnHeader header{};
    TCubeTiling cubeTiling1{};
    TCubeTiling cubeTiling2{};

    TPipe pipe;
    TQue<TPosition::VECIN, BUFFER_NUM> rowInQueue;
    TQue<TPosition::VECOUT, BUFFER_NUM> rowOutQueue;

    GlobalTensor<float> qGm;
    GlobalTensor<float> kGm;
    GlobalTensor<float> vGm;
    GlobalTensor<float> oGm;

    __gm__ float* workspaceBase = nullptr;

    // softmax tiling placeholder; API will recompute when isCheckTiling=true
    SoftMaxTiling smTiling{};
};

extern "C" __global__ __aicore__ void kv_cached_speculative_attention_custom(GM_ADDR Q, GM_ADDR KCache, GM_ADDR VCache, GM_ADDR O, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelKvCachedSpecAttn op;
    op.Init(Q, KCache, VCache, O, workspace, tiling);
    op.Process();
}
