namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    RmspropCustomTilingData tiling;
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);

    // 获取算子属性（lr、alpha、eps），并写入tiling数据
    // 注意：以下属性获取接口为常见用法示例，实际项目中请根据SDK提供的API获取属性。
    float lr = 1e-3f;
    float alpha = 0.99f;
    float eps = 1e-8f;
    auto lrAttr = context->GetOpAttr(\"lr\");
    if (lrAttr != nullptr) { lr = lrAttr->GetFloat(); }
    auto alphaAttr = context->GetOpAttr(\"alpha\");
    if (alphaAttr != nullptr) { alpha = alphaAttr->GetFloat(); }
    auto epsAttr = context->GetOpAttr(\"eps\");
    if (epsAttr != nullptr) { eps = epsAttr->GetFloat(); }

    tiling.set_lr(lr);
    tiling.set_alpha(alpha);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    context->SetTilingKey(1);
    return ge::GRAPH_SUCCESS;
}
}

    *out_shape = *param_shape;
    return GRAPH_SUCCESS;

    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);

}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelRmsprop {
public:
    __aicore__ inline KernelRmsprop() {}
    __aicore__ inline void Init(GM_ADDR param, GM_ADDR grad, GM_ADDR v, GM_ADDR z,
                                float lr, float alpha, float eps,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        this->lr = lr;
        this->alpha = alpha;
        this->eps = eps;

        paramGm.SetGlobalBuffer((__gm__ float*)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gradGm.SetGlobalBuffer((__gm__ float*)grad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ float*)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ float*)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueParam, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueGrad, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueV, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(float));

        pipe.InitBuffer(tmpQueue1, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue2, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue3, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> paramLocal = inQueueParam.AllocTensor<float>();
        AscendC::LocalTensor<float> gradLocal = inQueueGrad.AllocTensor<float>();
        AscendC::LocalTensor<float> vLocal = inQueueV.AllocTensor<float>();
        AscendC::DataCopy(paramLocal, paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gradLocal, gradGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(vLocal, vGm[progress * this->tileLength], this->tileLength);
        inQueueParam.EnQue(paramLocal);
        inQueueGrad.EnQue(gradLocal);
        inQueueV.EnQue(vLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<float> paramLocal = inQueueParam.DeQue<float>();
        AscendC::LocalTensor<float> gradLocal = inQueueGrad.DeQue<float>();
        AscendC::LocalTensor<float> vLocal = inQueueV.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

        AscendC::LocalTensor<float> t1 = tmpQueue1.AllocTensor<float>(); // g^2 or v_new
        AscendC::LocalTensor<float> t2 = tmpQueue2.AllocTensor<float>(); // alpha*v or sqrt(v_new)+eps
        AscendC::LocalTensor<float> t3 = tmpQueue3.AllocTensor<float>(); // (1-alpha)*g^2 or lr * (g/denom)

        // g^2
        AscendC::Mul(t1, gradLocal, gradLocal, this->tileLength);
        // alpha * v
        AscendC::Muls(t2, vLocal, this->alpha, this->tileLength);
        // (1 - alpha) * g^2
        float one_minus_alpha = 1.0f - this->alpha;
        AscendC::Muls(t3, t1, one_minus_alpha, this->tileLength);
        // v_new = alpha*v + (1-alpha)*g^2 -> reuse t1
        AscendC::Add(t1, t2, t3, this->tileLength);
        // sqrt(v_new)
        AscendC::Sqrt(t2, t1, this->tileLength);
        // denom = sqrt(v_new) + eps
        AscendC::Adds(t2, t2, this->eps, this->tileLength);
        // g / denom
        AscendC::Div(t3, gradLocal, t2, this->tileLength);
        // lr * (g / denom)
        AscendC::Muls(t3, t3, this->lr, this->tileLength);
        // param - lr * g / denom
        AscendC::Sub(zLocal, paramLocal, t3, this->tileLength);

        outQueueZ.EnQue<float>(zLocal);
        inQueueParam.FreeTensor(paramLocal);
        inQueueGrad.FreeTensor(gradLocal);
        inQueueV.FreeTensor(vLocal);
        tmpQueue1.FreeTensor(t1);
        tmpQueue2.FreeTensor(t2);
        tmpQueue3.FreeTensor(t3);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueParam;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueGrad;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueV;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue1;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue2;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue3;

    AscendC::GlobalTensor<float> paramGm;
    AscendC::GlobalTensor<float> gradGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> zGm;

    float lr;
    float alpha;
    float eps;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void rmsprop_custom(GM_ADDR param, GM_ADDR grad, GM_ADDR v, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelRmsprop op;
    op.Init(param, grad, v, z, tilingData.lr, tilingData.alpha, tilingData.eps, tilingData.totalLength, tilingData.tileNum);
    if (TILING_KEY_IS(1)) {
        op.Process();
    } else {
        op.Process();
    }
}
"""
