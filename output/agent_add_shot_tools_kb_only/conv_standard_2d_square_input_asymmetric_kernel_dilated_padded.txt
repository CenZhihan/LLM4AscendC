    if ((v = ctx->GetAttr("dil_w")) != nullptr) v->GetValue(&dil_w, sizeof(int64_t));

    int64_t effKW = (kW - 1) * dil_w + 1;

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    const uint32_t N = tilingData.N;
    const uint32_t Cin = tilingData.Cin;
    const uint32_t Hin = tilingData.Hin;
    const uint32_t Win = tilingData.Win;
    const uint32_t Cout = tilingData.Cout;
    const uint32_t kH = tilingData.kH;
    const uint32_t kW = tilingData.kW;
    const uint32_t padH = tilingData.padH;
    const uint32_t padW = tilingData.padW;
    const uint32_t strideH = tilingData.strideH;
    const uint32_t strideW = tilingData.strideW;
    const uint32_t dilH = tilingData.dilH;
    const uint32_t dilW = tilingData.dilW;
    const uint32_t Hout = tilingData.Hout;
    const uint32_t Wout = tilingData.Wout;

    __gm__ float* xGm = (__gm__ float*)x;
    __gm__ float* wGm = (__gm__ float*)w;
    __gm__ float* zGm = (__gm__ float*)z;

    // Simple block split by output channel
    uint32_t blkNum = AscendC::GetBlockNum();
    uint32_t blkIdx = AscendC::GetBlockIdx();
    uint32_t ocPerBlk = (Cout + blkNum - 1) / blkNum;
    uint32_t ocBeg = blkIdx * ocPerBlk;
    uint32_t ocEnd = ocBeg + ocPerBlk;
    if (ocEnd > Cout) ocEnd = Cout;

    // Layout:
    // x: [N, Cin, Hin, Win] - contiguous NCHW
    // w: [Cout, Cin, kH, kW]
    // z: [N, Cout, Hout, Wout]
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t oc = ocBeg; oc < ocEnd; ++oc) {
            for (uint32_t oh = 0; oh < Hout; ++oh) {
                for (uint32_t ow = 0; ow < Wout; ++ow) {
                    float acc = 0.0f;
                    // Convolution sum
                    for (uint32_t ic = 0; ic < Cin; ++ic) {
                        for (uint32_t kh = 0; kh < kH; ++kh) {
                            int32_t in_h = static_cast<int32_t>(oh * strideH) + static_cast<int32_t>(kh * dilH) - static_cast<int32_t>(padH);
                            if (in_h < 0 || in_h >= static_cast<int32_t>(Hin)) continue;
                            for (uint32_t kw = 0; kw < kW; ++kw) {
                                int32_t in_w = static_cast<int32_t>(ow * strideW) + static_cast<int32_t>(kw * dilW) - static_cast<int32_t>(padW);
                                if (in_w < 0 || in_w >= static_cast<int32_t>(Win)) continue;

                                uint64_t xIdx = ((static_cast<uint64_t>(n) * Cin + ic) * Hin + static_cast<uint32_t>(in_h)) * Win + static_cast<uint32_t>(in_w);
                                uint64_t wIdx = (((static_cast<uint64_t>(oc) * Cin + ic) * kH + kh) * kW + kw);
                                acc += xGm[xIdx] * wGm[wIdx];
                            }
                        }
                    }
                    uint64_t zIdx = ((static_cast<uint64_t>(n) * Cout + oc) * Hout + oh) * Wout + ow;
                    zGm[zIdx] = acc;
                }
            }
        }
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline int64_t calc_out_dim(int64_t in, int64_t pad, int64_t k, int64_t dil, int64_t stride) {
    const int64_t effK = (k - 1) * dil + 1;
    return (in + 2 * pad - effK) / stride + 1;
}

at::Tensor conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom_impl_npu(
    const at::Tensor& x,
    const at::Tensor& w,
    int64_t stride_h, int64_t stride_w,
    int64_t pad_h, int64_t pad_w,
    int64_t dil_h, int64_t dil_w) {

    TORCH_CHECK(x.dim() == 4, "Input x must be 4D NCHW");
    TORCH_CHECK(w.dim() == 4, "Weight w must be 4D [Cout, Cin, kH, kW]");
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto Hin = x.size(2);
    auto Win = x.size(3);
    auto Cout = w.size(0);
    auto kH = w.size(2);
    auto kW = w.size(3);
    TORCH_CHECK(w.size(1) == Cin, "Weight Cin mismatch");

    int64_t Hout = calc_out_dim(Hin, pad_h, kH, dil_h, stride_h);
    int64_t Wout = calc_out_dim(Win, pad_w, kW, dil_w, stride_w);
    TORCH_CHECK(Hout >= 0 && Wout >= 0, "Invalid output size");

    at::Tensor y = at::empty({N, Cout, Hout, Wout}, x.options());

    // Launch custom op
    EXEC_NPU_CMD(aclnnConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom,
                 x, w,
                 stride_h, stride_w, pad_h, pad_w, dil_h, dil_w,
                 y);
    return y;
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用 AscendC 自定义算子替换 PyTorch Conv2d：支持方形输入、非对称卷积核、膨胀与填充。
    权重来自于内置的 nn.Conv2d 模块，bias 默认为 False。
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: int = 1, padding: tuple = (0, 0), dilation: tuple = (1, 1), bias: bool = False):
        super().__init__()
        assert not bias, "当前自定义算子不支持 bias=True"
        self.conv2d = torch.nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, bias=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        stride_h, stride_w = (self.conv2d.stride if isinstance(self.conv2d.stride, tuple) else (self.conv2d.stride, self.conv2d.stride))
        pad_h, pad_w = (self.conv2d.padding if isinstance(self.conv2d.padding, tuple) else (self.conv2d.padding, self.conv2d.padding))
        dil_h, dil_w = (self.conv2d.dilation if isinstance(self.conv2d.dilation, tuple) else (self.conv2d.dilation, self.conv2d.dilation))
        return custom_ops_lib.conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
            x, self.conv2d.weight, stride_h, stride_w, pad_h, pad_w, dil_h, dil_w
        )
