kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGruHidden {
public:
    __aicore__ inline KernelGruHidden() {}
    __aicore__ inline void Init(GM_ADDR h, GM_ADDR out, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        hGm.SetGlobalBuffer((__gm__ DTYPE_X *)h + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueH, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueOut, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> hLocal = inQueueH.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(hLocal, hGm[progress * this->tileLength], this->tileLength);
        inQueueH.EnQue(hLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> hLocal = inQueueH.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(outLocal, hLocal, this->tileLength);
        outQueueOut.EnQue<DTYPE_Z>(outLocal);
        inQueueH.FreeTensor(hLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.DeQue<DTYPE_Z>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueH;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut;
    AscendC::GlobalTensor<DTYPE_X> hGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gru_hidden_custom(GM_ADDR h, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGruHidden op;
    op.Init(h, out, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
