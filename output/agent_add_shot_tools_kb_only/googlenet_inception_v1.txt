kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

using DTYPE = float;

class KernelGooglenetInceptionV1Custom {
public:
    __aicore__ inline KernelGooglenetInceptionV1Custom() {}
    __aicore__ inline void Init(GM_ADDR x0, GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR y,
                                uint32_t len0, uint32_t len1, uint32_t len2, uint32_t len3,
                                uint32_t tileNum)
    {
        this->len0 = len0;
        this->len1 = len1;
        this->len2 = len2;
        this->len3 = len3;
        this->tileNum = tileNum;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // 每个block处理的分段长度
        blkLen0 = len0 / blockNum;
        blkLen1 = len1 / blockNum;
        blkLen2 = len2 / blockNum;
        blkLen3 = len3 / blockNum;

        // 输出本block的起始偏移
        blkLenY = blkLen0 + blkLen1 + blkLen2 + blkLen3;

        x0Gm.SetGlobalBuffer((__gm__ DTYPE*)x0 + blkLen0 * blockIdx, blkLen0);
        x1Gm.SetGlobalBuffer((__gm__ DTYPE*)x1 + blkLen1 * blockIdx, blkLen1);
        x2Gm.SetGlobalBuffer((__gm__ DTYPE*)x2 + blkLen2 * blockIdx, blkLen2);
        x3Gm.SetGlobalBuffer((__gm__ DTYPE*)x3 + blkLen3 * blockIdx, blkLen3);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y  + blkLenY  * blockIdx,  blkLenY);

        // 为不同输入分配各自的tile长度
        tileLen0 = (blkLen0 > 0) ? AscendC::GetCeil(blkLen0 / (tileNum * BUFFER_NUM), 1) : 0;
        tileLen1 = (blkLen1 > 0) ? AscendC::GetCeil(blkLen1 / (tileNum * BUFFER_NUM), 1) : 0;
        tileLen2 = (blkLen2 > 0) ? AscendC::GetCeil(blkLen2 / (tileNum * BUFFER_NUM), 1) : 0;
        tileLen3 = (blkLen3 > 0) ? AscendC::GetCeil(blkLen3 / (tileNum * BUFFER_NUM), 1) : 0;

        // 初始化队列
        if (blkLen0 > 0) {
            pipe.InitBuffer(inQueue0, BUFFER_NUM, tileLen0 * sizeof(DTYPE));
            pipe.InitBuffer(outQueue0, BUFFER_NUM, tileLen0 * sizeof(DTYPE));
        }
        if (blkLen1 > 0) {
            pipe.InitBuffer(inQueue1, BUFFER_NUM, tileLen1 * sizeof(DTYPE));
            pipe.InitBuffer(outQueue1, BUFFER_NUM, tileLen1 * sizeof(DTYPE));
        }
        if (blkLen2 > 0) {
            pipe.InitBuffer(inQueue2, BUFFER_NUM, tileLen2 * sizeof(DTYPE));
            pipe.InitBuffer(outQueue2, BUFFER_NUM, tileLen2 * sizeof(DTYPE));
        }
        if (blkLen3 > 0) {
            pipe.InitBuffer(inQueue3, BUFFER_NUM, tileLen3 * sizeof(DTYPE));
            pipe.InitBuffer(outQueue3, BUFFER_NUM, tileLen3 * sizeof(DTYPE));
        }
    }

    __aicore__ inline void Process()
    {
        uint32_t yOffset = 0;
        // 顺序拼接四个分支到y，按通道维（内存上为整块追加）
        if (blkLen0 > 0) {
            CopyStream(x0Gm, yGm, inQueue0, outQueue0, blkLen0, tileLen0, yOffset);
            yOffset += blkLen0;
        }
        if (blkLen1 > 0) {
            CopyStream(x1Gm, yGm, inQueue1, outQueue1, blkLen1, tileLen1, yOffset);
            yOffset += blkLen1;
        }
        if (blkLen2 > 0) {
            CopyStream(x2Gm, yGm, inQueue2, outQueue2, blkLen2, tileLen2, yOffset);
            yOffset += blkLen2;
        }
        if (blkLen3 > 0) {
            CopyStream(x3Gm, yGm, inQueue3, outQueue3, blkLen3, tileLen3, yOffset);
            yOffset += blkLen3;
        }
    }

private:
    template<typename InQ, typename OutQ>
    __aicore__ inline void CopyStream(AscendC::GlobalTensor<DTYPE>& x,
                                      AscendC::GlobalTensor<DTYPE>& y,
                                      InQ& inQ, OutQ& outQ,
                                      uint32_t blkLen, uint32_t tLen, uint32_t yBegin)
    {
        uint32_t copied = 0;
        while (copied < blkLen) {
            uint32_t curr = AscendC::Min(tLen, blkLen - copied);
            AscendC::LocalTensor<DTYPE> xLocal = inQ.AllocTensor<DTYPE>();
            AscendC::DataCopy(xLocal, x[copied], curr);

            AscendC::LocalTensor<DTYPE> yLocal = outQ.AllocTensor<DTYPE>();
            // 将局部数据写回到y的对应位置
            AscendC::DataCopy(y[yBegin + copied], yLocal, curr);

            outQ.EnQue<DTYPE>(yLocal);
            inQ.EnQue<DTYPE>(xLocal);

            outQ.FreeTensor(yLocal);
            inQ.FreeTensor(xLocal);
            copied += curr;
        }
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueue0, inQueue1, inQueue2, inQueue3;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue0, outQueue1, outQueue2, outQueue3;

    AscendC::GlobalTensor<DTYPE> x0Gm, x1Gm, x2Gm, x3Gm, yGm;

    uint32_t len0{0}, len1{0}, len2{0}, len3{0};
    uint32_t blkLen0{0}, blkLen1{0}, blkLen2{0}, blkLen3{0}, blkLenY{0};
    uint32_t tileNum{1};
    uint32_t tileLen0{0}, tileLen1{0}, tileLen2{0}, tileLen3{0};
};

extern "C" __global__ __aicore__ void googlenet_inception_v1_custom(
    GM_ADDR x0, GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelGooglenetInceptionV1Custom op;
    op.Init(x0, x1, x2, x3, y,
            tilingData.totalLength0(),
            tilingData.totalLength1(),
            tilingData.totalLength2(),
            tilingData.totalLength3(),
            tilingData.tileNum());
    op.Process();
}
"""

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class InceptionModule(nn.Module):
    def __init__(self, in_channels, out_1x1, reduce_3x3, out_3x3, reduce_5x5, out_5x5, pool_proj):
        super(InceptionModule, self).__init__()
        self.branch1x1 = nn.Conv2d(in_channels, out_1x1, kernel_size=1)
        self.branch3x3 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_3x3, kernel_size=1),
            nn.Conv2d(reduce_3x3, out_3x3, kernel_size=3, padding=1)
        )
        self.branch5x5 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_5x5, kernel_size=1),
            nn.Conv2d(reduce_5x5, out_5x5, kernel_size=5, padding=2)
        )
        self.branch_pool = nn.Sequential(
            nn.MaxPool2d(kernel_size=3, stride=1, padding=1),
            nn.Conv2d(in_channels, pool_proj, kernel_size=1)
        )

    def forward(self, x):
        b1 = self.branch1x1(x)
        b3 = self.branch3x3(x)
        b5 = self.branch5x5(x)
        bp = self.branch_pool(x)
        # 使用自定义AscendC核替代torch.cat沿通道拼接
        return custom_ops_lib.googlenet_inception_v1_custom(b1, b3, b5, bp)

class ModelNew(nn.Module):
    def __init__(self, num_classes=1000):
        super(ModelNew, self).__init__()
        self.conv1 = nn.Conv2d(3, 64, kernel_size=7, stride=2, padding=3)
        self.maxpool1 = nn.MaxPool2d(3, stride=2, padding=1)
        self.conv2 = nn.Conv2d(64, 64, kernel_size=1)
        self.conv3 = nn.Conv2d(64, 192, kernel_size=3, padding=1)
        self.maxpool2 = nn.MaxPool2d(3, stride=2, padding=1)

        self.inception3a = InceptionModule(192, 64, 96, 128, 16, 32, 32)
        self.inception3b = InceptionModule(256, 128, 128, 192, 32, 96, 64)
        self.maxpool3 = nn.MaxPool2d(3, stride=2, padding=1)

        self.inception4a = InceptionModule(480, 192, 96, 208, 16, 48, 64)
        self.inception4b = InceptionModule(512, 160, 112, 224, 24, 64, 64)
        self.inception4c = InceptionModule(512, 128, 128, 256, 24, 64, 64)
        self.inception4d = InceptionModule(512, 112, 144, 288, 32, 64, 64)
        self.inception4e = InceptionModule(528, 256, 160, 320, 32, 128, 128)
        self.maxpool4 = nn.MaxPool2d(3, stride=2, padding=1)

        self.inception5a = InceptionModule(832, 256, 160, 320, 32, 128, 128)
        self.inception5b = InceptionModule(832, 384, 192, 384, 48, 128, 128)

        self.avgpool = nn.AdaptiveAvgPool2d((1, 1))
        self.dropout = nn.Dropout(0.0)
        self.fc = nn.Linear(1024, num_classes)

    def forward(self, x):
        x = self.maxpool1(F.relu(self.conv1(x)))
        x = F.relu(self.conv2(x))
        x = self.maxpool2(F.relu(self.conv3(x)))

        x = self.inception3a(x)
        x = self.inception3b(x)
        x = self.maxpool3(x)

        x = self.inception4a(x)
        x = self.inception4b(x)
        x = self.inception4c(x)
        x = self.inception4d(x)
        x = self.inception4e(x)
        x = self.maxpool4(x)

        x = self.inception5a(x)
        x = self.inception5b(x)

        x = self.avgpool(x)
        x = torch.flatten(x, 1)
        x = self.dropout(x)
        x = self.fc(x)
        return x
'''
