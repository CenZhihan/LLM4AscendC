#include "register/tilingdata_base.h"
#include "tiling/platform/platform_ascendc.h"
#include "tiling/tiling_api.h"

namespace optiling {

BEGIN_TILING_DATA_DEF(CrossModalAttentionCustomTilingData)
  // 直接使用TCubeTiling的原始字节存入RawTilingData
  TILING_DATA_FIELD_DEF(uint32_t, m);
  TILING_DATA_FIELD_DEF(uint32_t, n);
  TILING_DATA_FIELD_DEF(uint32_t, k);
  TILING_DATA_FIELD_DEF(uint32_t, batch);
  // 仅作占位，实际TCubeTiling通过RawTilingData传递
END_TILING_DATA_DEF;

REGISTER_TILING_DATA_CLASS(CrossModalAttentionCustom, CrossModalAttentionCustomTilingData)

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 输入x: [B, M, K], 输入y: [B, K, N]，输出z: [B, M, N]
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* y_shape = context->GetInputShape(1);

    // 兼容2D/3D，若为2D则batch=1
    uint32_t dims_x = x_shape->GetDimNum();
    uint32_t dims_y = y_shape->GetDimNum();

    uint32_t batch = 1;
    uint32_t M = 0, Kx = 0, Ky = 0, N = 0;
    if (dims_x == 3) {
        batch = static_cast<uint32_t>(x_shape->GetDim(0));
        M = static_cast<uint32_t>(x_shape->GetDim(1));
        Kx = static_cast<uint32_t>(x_shape->GetDim(2));
    } else if (dims_x == 2) {
        batch = 1;
        M = static_cast<uint32_t>(x_shape->GetDim(0));
        Kx = static_cast<uint32_t>(x_shape->GetDim(1));
    } else {
        return ge::GRAPH_FAILED;
    }

    if (dims_y == 3) {
        uint32_t by = static_cast<uint32_t>(y_shape->GetDim(0));
        Ky = static_cast<uint32_t>(y_shape->GetDim(1));
        N = static_cast<uint32_t>(y_shape->GetDim(2));
        if (by != batch) return ge::GRAPH_FAILED;
    } else if (dims_y == 2) {
        Ky = static_cast<uint32_t>(y_shape->GetDim(0));
        N = static_cast<uint32_t>(y_shape->GetDim(1));
    } else {
        return ge::GRAPH_FAILED;
    }

    if (Kx != Ky) return ge::GRAPH_FAILED;

    // 设置blockDim
    const uint32_t maxBlock = 32;
    uint32_t blockDim = batch < maxBlock ? batch : maxBlock;
    context->SetBlockDim(blockDim);

    // 使用BatchMatmulTiling获取TCubeTiling
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::BatchMatmulTiling bmmTiling(ascendcPlatform);
    bmmTiling.SetDim(blockDim);
    // A/B: fp16，C: fp32
    bmmTiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    bmmTiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    bmmTiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    bmmTiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    bmmTiling.SetBias(false);

    // 设置单次Matmul形状与原始形状
    bmmTiling.SetShape(M, N, Kx);
    bmmTiling.SetSingleShape(M, N, Kx);
    bmmTiling.SetOrgShape(M, N, Kx);
    bmmTiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = bmmTiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // 将TCubeTiling原始字节写入RawTilingData
    auto *raw = context->GetRawTilingData();
    size_t needBytes = sizeof(optiling::TCubeTiling);
    if (raw->GetCapacity() < needBytes) {
        return ge::GRAPH_FAILED;
    }
    // 拷贝TCubeTiling
    void *dst = raw->GetData();
    const void *src = reinterpret_cast<const void *>(&tilingData);
    (void)memcpy(dst, src, needBytes);
    raw->SetDataSize(needBytes);

    // 设置workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    const gert::Shape* y_shape = context->GetInputShape(1);

    uint32_t dims_x = x_shape->GetDimNum();
    uint32_t dims_y = y_shape->GetDimNum();

    int64_t M = 0, Kx = 0, Ky = 0, N = 0;

        batch = x_shape->GetDim(0);
        M = x_shape->GetDim(1);
        Kx = x_shape->GetDim(2);
    } else if (dims_x == 2) {
        batch = 1;
        M = x_shape->GetDim(0);
        Kx = x_shape->GetDim(1);
    } else {
        return GRAPH_FAILED;
    }

    if (dims_y == 3) {
        if (y_shape->GetDim(0) != batch) return GRAPH_FAILED;

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = half;
using B_T = half;
using C_T = float;

extern "C" __global__ __aicore__ void cross_modal_attention_custom(GM_ADDR xGM, GM_ADDR yGM, GM_ADDR zGM, GM_ADDR workspaceGM, GM_ADDR tilingGM) {
    // 从GM中读取TCubeTiling
    optiling::TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(optiling::TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // 定义Matmul类型：A/B = fp16, C = fp32, ND NORMAL
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, A_T> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, B_T> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T> biasType;

    // 设置全局Tensor
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;

    // 简单设置：直接绑定整块buffer
    // 实际大小由框架保证
    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(xGM), 0);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(yGM), 0);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(zGM), 0);

    // 创建Matmul对象并初始化
    AscendC::TPipe pipe;
    matmul::Matmul<aType, bType, cType, biasType> mm;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
    mm.Init(&tiling);

    // 绑定输入输出
    mm.SetTensorA(aGlobal[0], false);
    mm.SetTensorB(bGlobal[0], false);
    mm.SetWorkspace(workspaceGM, 0);

    // 执行BMM/Matmul。对于NORMAL布局，host侧已完成tiling设置
    // 这里执行一次IterateNBatch，按tiling批次循环
    const uint32_t batchLoop = 1;
    const uint32_t batchA = 1;
    const uint32_t batchB = 1;
    const bool enSequentialWrite = false;
    mm.IterateNBatch(batchLoop, batchA, batchB, enSequentialWrite);
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import math
from torch import nn

class ModelNew(nn.Module):
    """
    使用自定义AscendC核（CrossModalAttentionCustom）替换注意力中的两次矩阵乘法：
      1) scores = (Q @ K^T) / sqrt(dh)
      2) out_heads = softmax(scores) @ V
    其中CrossModalAttentionCustom实现通用的batched matmul: [B, M, K] x [B, K, N] -> [B, M, N]
    """
    def __init__(self, d_model_text=256, d_model_img=512, num_heads=8):
        super().__init__()
        self.d_model_text = d_model_text
        self.d_model_img = d_model_img
        self.num_heads = num_heads
        self.q_proj = nn.Linear(d_model_text, d_model_text)
        self.kv_proj = nn.Linear(d_model_img, d_model_text)
        # 输出投影可复用原MHA的效果
        self.out_proj = nn.Linear(d_model_text, d_model_text)

    def forward(self, text, img):
        # 线性投影
        q = self.q_proj(text)              # [B, Lt, D]
        kv = self.kv_proj(img)             # [B, Li, D]

        B, Lt, D = q.shape
        Li = kv.shape[1]
        H = self.num_heads
        dh = D // H

        # 切分多头并reshape为[B*H, L, dh]
        qh = q.view(B, Lt, H, dh).permute(0, 2, 1, 3).reshape(B*H, Lt, dh)
        kh = kv.view(B, Li, H, dh).permute(0, 2, 1, 3).reshape(B*H, Li, dh)
        vh = kv.view(B, Li, H, dh).permute(0, 2, 1, 3).reshape(B*H, Li, dh)

        # 1) scores = Q @ K^T
        q_fp16 = qh.to(torch.float16)
        kT_fp16 = kh.transpose(1, 2).contiguous().to(torch.float16)   # [B*H, dh, Li]
        scores_fp32 = custom_ops_lib.cross_modal_attention_custom(q_fp16, kT_fp16)  # [B*H, Lt, Li] fp32

        # 缩放+softmax
        scores_fp32 = scores_fp32 / math.sqrt(dh)
        attn = torch.softmax(scores_fp32, dim=-1)                     # [B*H, Lt, Li] fp32

        # 2) out_heads = attn @ V
        attn_fp16 = attn.to(torch.float16)
        v_fp16 = vh.to(torch.float16)                                 # [B*H, Li, dh]
        out_heads_fp32 = custom_ops_lib.cross_modal_attention_custom(attn_fp16, v_fp16)  # [B*H, Lt, dh] fp32

        # 合并多头
        out_heads = out_heads_fp32.reshape(B, H, Lt, dh).permute(0, 2, 1, 3).reshape(B, Lt, D)

        # 输出投影
        out = self.out_proj(out_heads)
        return out
'''
