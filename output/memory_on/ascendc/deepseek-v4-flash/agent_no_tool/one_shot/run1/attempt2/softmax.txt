host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for softmax per row (online two-pass):
//   inQueue: 2 * tile (copy-in)
//   outQueue: 2 * tile (copy-out)
// No extra temp (use out buffer for exp)
// Total floats = 4 * tile => bytes = 16 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MAX_TILE_ELEMS = 8192;        // cap to avoid large latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Assume 2D shape: [B, D] (dim=1 softmax). rowLength = D.
    const uint32_t rowLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));

    // Tile length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 16;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        // Row fits in one tile; process row as a whole.
        tileLen = AlignDown(rowLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowLength;
    }

    // Number of rows
    const uint32_t numRows = totalLength / rowLength;

    // Block dim: based on rows, but cap
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t tileLength)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        numRows_ = totalLen_ / rowLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // Temp buffer for exponent intermediate and scalar reductions
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        // Dedicated 1-element buffers for reduction results
        pipe_.InitBuffer(scalarQueue_, 2, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        const uint32_t rowsPerCore = CeilDiv(numRows_, blockNum);
        const uint32_t startRow = blockIdx * rowsPerCore;
        if (startRow >= numRows_) return;
        const uint32_t endRow = (startRow + rowsPerCore > numRows_) ? numRows_ : startRow + rowsPerCore;

        const int32_t tileBlocksPerRow = CeilDiv(rowLen_, tileLen_);

        for (uint32_t r = startRow; r < endRow; ++r) {
            ProcessRow(r, tileBlocksPerRow);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx, int32_t numChunks)
    {
        const uint32_t rowBase = rowIdx * rowLen_;

        // ===== Phase 1: compute row max and sum =====
        float rowMax = -1e38f;
        float rowSum = 0.0f;

        for (int32_t c = 0; c < numChunks; ++c) {
            const uint32_t offset = rowBase + c * tileLen_;
            const uint32_t curLen = (c == numChunks - 1) ? (rowLen_ - c * tileLen_) : tileLen_;

            // Read chunk
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> xVec = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();  // for exp
            AscendC::LocalTensor<float> scalar = scalarQueue_.AllocTensor<float>();  // 1-element

            // Compute max of this chunk
            AscendC::ReduceMax(scalar, xVec, curLen);
            float chunkMax = scalar(0);
            scalarQueue_.FreeTensor(scalar);

            float newMax = (rowMax > chunkMax) ? rowMax : chunkMax;

            // Compute exp(x - newMax)
            AscendC::Sub(tmp, xVec, newMax, curLen);
            AscendC::Exp(tmp, tmp, curLen);

            // Sum the exponentials
            scalar = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceSum(scalar, tmp, curLen);
            float chunkSum = scalar(0);
            scalarQueue_.FreeTensor(scalar);

            // Update rowSum: rowSum = rowSum * exp(rowMax - newMax) + chunkSum
            float expDiff = AscendC::Exp(rowMax - newMax);  // scalar exp
            rowSum = rowSum * expDiff + chunkSum;
            rowMax = newMax;

            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xVec);
        }

        // ===== Phase 2: normalize output =====
        for (int32_t c = 0; c < numChunks; ++c) {
            const uint32_t offset = rowBase + c * tileLen_;
            const uint32_t curLen = (c == numChunks - 1) ? (rowLen_ - c * tileLen_) : tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> xVec = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // y = exp(x - rowMax) / rowSum
            AscendC::Sub(yLocal, xVec, rowMax, curLen);
            AscendC::Exp(yLocal, yLocal, curLen);
            AscendC::Div(yLocal, yLocal, rowSum, curLen);

            outQueueY_.EnQue<float>(yLocal);
            inQueueX_.FreeTensor(xVec);

            // Write out
            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yOut, curLen);
            outQueueY_.FreeTensor(yOut);
        }
    }

    __aicore__ inline static int32_t CeilDiv(int32_t a, int32_t b)
    {
        return (a + b - 1) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLength, tiling_data.tileLength);
    op.Process();
}
