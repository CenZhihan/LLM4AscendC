#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;
    const auto* input_shape = context->GetInputShape(0);
    const auto& origin_shape = input_shape->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(origin_shape.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim_ptr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dim_ptr;
    int64_t ndim = origin_shape.GetDimNum();
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= ndim) dim = 1; // fallback

    uint32_t innerSize = static_cast<uint32_t>(origin_shape.GetDim(dim));
    uint32_t outerSize = totalLength / innerSize;

    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (innerSize < maxTile) ? innerSize : maxTile;
    if (tileLen == 0) tileLen = 1;

    uint32_t blockDim = outerSize;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outerSize,
                                uint32_t innerSize, uint32_t tileLength,
                                uint32_t blockDim) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        outerSize_ = outerSize;
        innerSize_ = innerSize;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        totalLen_ = totalLength;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || innerSize_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = blockDim_;

        for (uint32_t outerIdx = blockIdx; outerIdx < outerSize_; outerIdx += blockNum) {
            uint32_t baseOffset = outerIdx * innerSize_;
            float mx = ComputeMax(baseOffset);
            float sumExp = ComputeSumExp(baseOffset, mx);
            ComputeOutput(baseOffset, mx, sumExp);
        }
    }

private:
    __aicore__ inline float ComputeMax(uint32_t baseOffset) {
        float globalMax = -1e38f;
        uint32_t remaining = innerSize_;
        uint32_t offset = baseOffset;

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> maxScalar = scalarQueue_.AllocTensor<float>();
            // ReduceMax: dst, src, dstIdx, srcLen, reduceLogic
            AscendC::ReduceMax(maxScalar, xLocal, 0, curLen, false);
            float localMax = maxScalar.GetValue(0);
            if (localMax > globalMax) globalMax = localMax;
            scalarQueue_.FreeTensor(maxScalar);
            inQueueX_.FreeTensor(xLocal);

            offset += curLen;
            remaining -= curLen;
        }
        return globalMax;
    }

    __aicore__ inline float ComputeSumExp(uint32_t baseOffset, float mx) {
        float sum = 0.0f;
        uint32_t remaining = innerSize_;
        uint32_t offset = baseOffset;

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            // x - mx (scalar subtraction)
            AscendC::Sub(tLocal, xLocal, mx, curLen);
            // exp
            AscendC::Exp(tLocal, tLocal, curLen);
            // ReduceSum: dst, src, dstIdx, srcLen, reduceLogic
            AscendC::LocalTensor<float> sumScalar = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumScalar, tLocal, 0, curLen, false);
            sum += sumScalar.GetValue(0);
            scalarQueue_.FreeTensor(sumScalar);

            tmpQueue_.FreeTensor(tLocal);
            inQueueX_.FreeTensor(xLocal);

            offset += curLen;
            remaining -= curLen;
        }
        return sum;
    }

    __aicore__ inline void ComputeOutput(uint32_t baseOffset, float mx, float sumExp) {
        // Compute logSum = log(sumExp) using a scalar tensor
        AscendC::LocalTensor<float> logScalar = scalarQueue_.AllocTensor<float>();
        logScalar.SetValue(0, sumExp);
        AscendC::Log(logScalar, logScalar, 1);
        float logSum = logScalar.GetValue(0);
        scalarQueue_.FreeTensor(logScalar);

        uint32_t remaining = innerSize_;
        uint32_t offset = baseOffset;

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            // y = x - mx - logSum
            AscendC::Sub(tLocal, xLocal, mx, curLen);
            AscendC::Sub(yLocal, tLocal, logSum, curLen);

            outQueueY_.EnQue<float>(yLocal);
            tmpQueue_.FreeTensor(tLocal);
            inQueueX_.FreeTensor(xLocal);

            // Copy out
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);

            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t outerSize_ {0};
    uint32_t innerSize_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
    uint32_t totalLen_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.outerSize,
            tiling_data.innerSize, tiling_data.tileLength, tiling_data.blockDim);
    op.Process();
}
"""
