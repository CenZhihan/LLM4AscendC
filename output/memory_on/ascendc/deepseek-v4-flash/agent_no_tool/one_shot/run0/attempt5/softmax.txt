host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: we need 2 tiles (x and exp) plus temporary buffers for broadcast
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  
static constexpr uint32_t ALIGN_ELEMS = 2048;            
static constexpr uint32_t MAX_TILE_ELEMS = 8192;         

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const auto &originShape = inputShape->GetOriginShape();
    uint32_t batchSize = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t dimSize = static_cast<uint32_t>(originShape.GetDim(1));

    // tileLen: number of columns per tile (must not exceed dimSize)
    uint32_t maxTile = UB_BUDGET_BYTES / 8;  // 2 tiles * 4 bytes
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = (dimSize < maxTile) ? dimSize : maxTile;

    // blockDim: each block processes one or more rows
    uint32_t blockDim = batchSize;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim > batchSize) blockDim = batchSize;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_dimSize(dimSize);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t dimSize,
                                uint32_t tileLen)
    {
        batchSize_ = batchSize;
        dimSize_ = dimSize;
        tileLen_ = tileLen;
        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * dimSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * dimSize);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, dimSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, dimSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, dimSize_ * sizeof(float));       // for broadcasts and exp
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));              // for reduction scalars
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || dimSize_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t rowsPerBlock = (batchSize_ + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > batchSize_) endRow = batchSize_;

        for (uint32_t row = startRow; row < endRow; ++row) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t offset = row * dimSize_;

        // Load x into input queue
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offset], dimSize_);

        // Step 1: reduce max along dim
        AscendC::LocalTensor<float> maxReduced = scalarQueue_.AllocTensor<float>();
        AscendC::ReduceMax(maxReduced, xLocal, 0, 0);
        float rowMax = maxReduced.GetValue(0);
        scalarQueue_.FreeTensor(maxReduced);

        // Broadcast max to full row
        AscendC::LocalTensor<float> maxBroadcast = tmpQueue_.AllocTensor<float>();
        AscendC::BroadcastVar(maxBroadcast, rowMax, dimSize_);

        // exp(x - max)
        AscendC::LocalTensor<float> expTmp = tmpQueue_.AllocTensor<float>();
        AscendC::Sub(expTmp, xLocal, maxBroadcast, dimSize_);
        AscendC::Exp(expTmp, expTmp, dimSize_);
        inQueueX_.FreeTensor(xLocal);
        tmpQueue_.FreeTensor(maxBroadcast);

        // Step 2: reduce sum
        AscendC::LocalTensor<float> sumReduced = scalarQueue_.AllocTensor<float>();
        AscendC::ReduceSum(sumReduced, expTmp, 0, 0);
        float rowSum = sumReduced.GetValue(0);
        scalarQueue_.FreeTensor(sumReduced);

        // Broadcast sum to full row
        AscendC::LocalTensor<float> sumBroadcast = tmpQueue_.AllocTensor<float>();
        AscendC::BroadcastVar(sumBroadcast, rowSum, dimSize_);

        // Normalize
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Div(yLocal, expTmp, sumBroadcast, dimSize_);
        tmpQueue_.FreeTensor(expTmp);
        tmpQueue_.FreeTensor(sumBroadcast);
        outQueueY_.EnQue(yLocal);

        // Copy out
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], yOut, dimSize_);
        outQueueY_.FreeTensor(yOut);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t dimSize_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.batchSize, tiling_data.dimSize, tiling_data.tileLen);
    op.Process();
}

    return result;

}
