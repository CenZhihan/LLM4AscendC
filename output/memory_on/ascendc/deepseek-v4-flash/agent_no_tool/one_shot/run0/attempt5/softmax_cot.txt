constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t dimSize,
                                uint32_t tileLen)
    {
        batchSize_ = batchSize;
        dimSize_ = dimSize;
        tileLen_ = tileLen;
        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * dimSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * dimSize);

        // Allocate pipe buffers: 2 queues for input, 2 for output, and 3 temp tiles (max, exp, sum broadcast, etc.)
        // We'll allocate temporary tensor for intermediate results.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, dimSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, dimSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, dimSize_ * sizeof(float)); // for expTmp and broadcast
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float)); // for reduction result
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || dimSize_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t rowsPerBlock = (batchSize_ + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > batchSize_) endRow = batchSize_;

        for (uint32_t row = startRow; row < endRow; ++row) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t offset = row * dimSize_;

        // Load x into input queue
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offset], dimSize_);

        // Compute max along row
        AscendC::LocalTensor<float> maxReduced = scalarQueue_.AllocTensor<float>(); // size 1
        AscendC::ReduceMax(maxReduced, xLocal, 0, 0);
        float rowMax = maxReduced.GetValue(0);
        scalarQueue_.FreeTensor(maxReduced);

        // Broadcast max to full row
        AscendC::LocalTensor<float> maxBroadcast = tmpQueue_.AllocTensor<float>();
        AscendC::Dup(maxBroadcast, rowMax, dimSize_);

        // Compute exp(x - max) into expTmp (reuse xLocal? need separate)
        AscendC::LocalTensor<float> expTmp = tmpQueue_.AllocTensor<float>(); // second temp, but we need two, we already allocated maxBroadcast; we can reuse the other temp queue
        AscendC::Sub(expTmp, xLocal, maxBroadcast, dimSize_);

        // Free xLocal and maxBroadcast

        // Compute sum of exp
