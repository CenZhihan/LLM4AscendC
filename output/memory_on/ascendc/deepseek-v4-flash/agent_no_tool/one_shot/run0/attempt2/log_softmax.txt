host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: 
//   inQueueX: 2 * tile (float) => 8 * tile bytes
//   tmpQueue: 2 * tile (float) => 8 * tile bytes + small overhead
// Total ~ 16 * tile bytes (buffers) + 2 scalars. We use 192KB UB per core.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t BYTES_PER_ELEM = 4;
static constexpr uint32_t NUM_BUFFERS = 4; // x, tmp (tile), scalars (ignored)
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape &shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalLength = shape.GetShapeSize();
    const uint32_t numDims = shape.GetDimNum();

    // Assume 2D: batch size and feature dim. For simplicity, we use dim=1 (last axis).
    // We'll extract batchSize and dimSize.
    uint32_t batchSize = 1, dimSize = 1;
    if (numDims >= 1) batchSize = static_cast<uint32_t>(shape.GetDim(0));
    if (numDims >= 2) dimSize = static_cast<uint32_t>(shape.GetDim(1));
    // If dim==0, swap? Not needed for given model.

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimPtr ? *dimPtr : 1);

    // Compute tile length from UB budget.
    // We need: inBuffer: 1 * tile (x), tmpBuffer: 2 * tile (exp, diff) => 3*tile floats = 12*tile bytes
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * BYTES_PER_ELEM); // 16384
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (dimSize < tileLen) {
        uint32_t aligned = AlignDown(dimSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : dimSize;
    }

    // Block dimension: assign rows to blocks.
    uint32_t blockDim = batchSize;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batchSize(batchSize);
    tiling.set_dimSize(dimSize);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t dimSize,
                                int32_t dim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        dimSize_ = dimSize;
        // dim attribute is not used; we assume reduction on last dimension.

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (dimSize_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();

        // Each block processes rows in a grid-stride loop.
        for (uint32_t row = blockIdx; row < batchSize_; row += blockNum) {
            uint32_t base = row * dimSize_;

            // Phase 1: compute max over the entire row.
            // Initialize rowMax to a very small value (like -1e38f) - AscendC does not have std::numeric_limits.
            float rowMax = -1e38f;
            for (uint32_t t = 0; t < dimSize_; t += tileLen_) {
                uint32_t curLen = (t + tileLen_ <= dimSize_) ? tileLen_ : dimSize_ - t;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base + t], curLen);

                // Compute max of the tile using AscendC::ReduceMax (not deprecated VecReduceMax).
                AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceMax(maxLocal, xLocal, curLen);
                float tileMax = maxLocal(0);

                if (tileMax > rowMax) rowMax = tileMax;

                tmpQueue_.FreeTensor(maxLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            // Phase 2: compute sum of exp(x - rowMax) over the row.
            float rowSum = 0.0f;
            for (uint32_t t = 0; t < dimSize_; t += tileLen_) {
                uint32_t curLen = (t + tileLen_ <= dimSize_) ? tileLen_ : dimSize_ - t;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base + t], curLen);

                // Compute x - rowMax (use Subs for scalar subtraction)
                AscendC::LocalTensor<float> diffLocal = tmpQueue_.AllocTensor<float>();
                AscendC::Subs(diffLocal, xLocal, rowMax, curLen);

                // Compute exp(diff)
                AscendC::Exp(diffLocal, diffLocal, curLen);

                // Compute sum of exp values in the tile using AscendC::ReduceSum (not deprecated VecReduceSum).
                AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceSum(sumLocal, diffLocal, curLen);
                float tileSum = sumLocal(0);
                rowSum += tileSum;

                tmpQueue_.FreeTensor(sumLocal);
                tmpQueue_.FreeTensor(diffLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            // Compute log(rowSum) using a scalar tensor.
            AscendC::LocalTensor<float> logSumTensor = tmpQueue_.AllocTensor<float>();
            logSumTensor(0) = rowSum;
            AscendC::Log(logSumTensor, logSumTensor, 1);
            float logSum = logSumTensor(0);
            tmpQueue_.FreeTensor(logSumTensor);

            // Phase 3: compute output y = x - rowMax - logSum
            for (uint32_t t = 0; t < dimSize_; t += tileLen_) {
                uint32_t curLen = (t + tileLen_ <= dimSize_) ? tileLen_ : dimSize_ - t;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base + t], curLen);

                // y = (x - rowMax) - logSum  (use Subs for scalar subtraction)
                AscendC::LocalTensor<float> yLocal = tmpQueue_.AllocTensor<float>();
                AscendC::Subs(yLocal, xLocal, rowMax, curLen);
                AscendC::Subs(yLocal, yLocal, logSum, curLen);

                // Write back to global memory.
                AscendC::DataCopy(yGm_[base + t], yLocal, curLen);

                tmpQueue_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t dimSize_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.batchSize, tiling_data.dimSize, tiling_data.dim);
    op.Process();
}
