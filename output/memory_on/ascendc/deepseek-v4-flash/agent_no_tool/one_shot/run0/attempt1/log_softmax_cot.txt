class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t dimSize,
                                int32_t dim)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        dimSize_ = dimSize;
        // ignore dim

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileLen_ * sizeof(float)); // for exp or max reduction
    }

    __aicore__ inline void Process()
    {
        if (dimSize_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();

        // outer loop over rows with grid-stride
        for (uint32_t row = blockIdx; row < batchSize_; row += blockNum) {
            uint32_t base = row * dimSize_;
            // Phase 1: compute max over row
            float rowMax = -std::numeric_limits<float>::infinity();
            for (uint32_t t = 0; t < dimSize_; t += tileLen_) {
                uint32_t curLen = (t + tileLen_ <= dimSize_) ? tileLen_ : dimSize_ - t;
                // Load x tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();

                // Free

                inQueueX_.FreeTensor(xLocal);
