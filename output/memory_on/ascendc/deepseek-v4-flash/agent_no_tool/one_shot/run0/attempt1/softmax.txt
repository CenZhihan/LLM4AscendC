// UB budget: we need 2 tiles (x and exp) plus scalar temporaries.
// Each tile: float => 4 bytes. Two tiles => 8 * tileLen bytes.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  
static constexpr uint32_t ALIGN_ELEMS = 2048;            
static constexpr uint32_t MAX_TILE_ELEMS = 8192;         // avoid too-large tiles

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const int64_t *dims = inputShape->GetOriginShape().GetDims();
    uint32_t batchSize = static_cast<uint32_t>(dims[0]);
    uint32_t dimSize = static_cast<uint32_t>(dims[1]);

    // tileLen = number of columns per tile (must not exceed dimSize)
    uint32_t maxTile = UB_BUDGET_BYTES / 8;  // 2 tiles * 4 bytes
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = (dimSize < maxTile) ? dimSize : maxTile;

    // blockDim = number of blocks, each block processes rows
    uint32_t blockDim = batchSize;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    // Ensure at most one row per block for simplicity
    if (blockDim > batchSize) blockDim = batchSize;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_dimSize(dimSize);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

// ---- Simplified kernel for full-row UB ----

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t dimSize,
                                uint32_t tileLen)
    {
        batchSize_ = batchSize;
        dimSize_ = dimSize;
        // tileLen assumed equal to dimSize for simplicity
        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * dimSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * dimSize);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, dimSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, dimSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, dimSize_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float)); // for scalars
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || dimSize_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t rowsPerBlock = (batchSize_ + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > batchSize_) endRow = batchSize_;

        for (uint32_t row = startRow; row < endRow; ++row) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t offset = row * dimSize_;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offset], dimSize_);
        inQueueX_.EnQue(xLocal);

        // First pass: compute max
        AscendC::LocalTensor<float> x1 = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maxTmp = tmpQueue_.AllocTensor<float>();
        AscendC::ReduceMax(maxTmp, x1, dimSize_);
        float rowMax = maxTmp.GetValue(0);
        tmpQueue_.FreeTensor(maxTmp);
        inQueueX_.FreeTensor(x1); // We need x again, so we'll re-read

        // Second pass: compute exp(x - rowMax) and sum
        AscendC::LocalTensor<float> x2 = inQueueX_.AllocTensor<float>(); // new local tensor
        AscendC::DataCopy(x2, xGm_[offset], dimSize_);
        AscendC::LocalTensor<float> expTmp = tmpQueue_.AllocTensor<float>();
        AscendC::Sub(expTmp, x2, rowMax, dimSize_);
        AscendC::Exp(expTmp, expTmp, dimSize_);
        // Compute sum
        AscendC::LocalTensor<float> sumTmp = scalarQueue_.AllocTensor<float>();
        AscendC::ReduceSum(sumTmp, expTmp, dimSize_);
        float rowSum = sumTmp.GetValue(0);
        scalarQueue_.FreeTensor(sumTmp);
        // Now normalize: copy expTmp to yLocal and divide by rowSum
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Div(yLocal, expTmp, rowSum, dimSize_);
        outQueueY_.EnQue(yLocal);
        tmpQueue_.FreeTensor(expTmp);
        inQueueX_.FreeTensor(x2);

        // Copy output to global
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], yOut, dimSize_);
        outQueueY_.FreeTensor(yOut);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t dimSize_ {0};
    uint32_t tileLen_ {0}; // unused in this simplified version
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.batchSize, tiling_data.dimSize, tiling_data.tileLen);
    op.Process();
}
