kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -1e38f;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t numFeatures) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        numFeatures_ = numFeatures;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileLen_ * sizeof(float));  // for max and exp
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid‑stride loop over rows
        uint32_t rowStride = blockNum;
        uint32_t rowBase = blockIdx;
        while (rowBase < batchSize_) {
            ProcessRow(rowBase);
            rowBase += rowStride;
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx) {
        uint32_t rowOffset = rowIdx * numFeatures_;

        // ---------- Pass1: compute row maximum ----------
        float rowMax = NEG_INF;
        AscendC::LocalTensor<float> maxTemp = tmpQueue_.AllocTensor<float>();
        uint32_t colBase = 0;
        while (colBase < numFeatures_) {
            uint32_t curLen = numFeatures_ - colBase;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            // Load tile of x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + colBase], curLen);
            inQueueX_.EnQue(xLocal);

            // Compute max of this tile (reduction to 1-element tensor)
            AscendC::ReduceMax<float, AscendC::REDUCE_MAX>(maxTemp, xLocal, curLen, 0);
            float tileMax = maxTemp.GetValue(0);
            if (tileMax > rowMax) rowMax = tileMax;

            AscendC::LocalTensor<float> xLocalDeq = inQueueX_.DeQue<float>();
            inQueueX_.FreeTensor(xLocalDeq);

            colBase += tileLen_;
        }
        tmpQueue_.FreeTensor(maxTemp);

        // ---------- Pass2: compute sum of exp(x - rowMax) and write exp to y ----------
        colBase = 0;
        float rowSum = 0.0f;
        AscendC::LocalTensor<float> sumTemp = tmpQueue_.AllocTensor<float>();
        while (colBase < numFeatures_) {
            uint32_t curLen = numFeatures_ - colBase;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            // Load x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + colBase], curLen);
            inQueueX_.EnQue(xLocal);

            // Allocate tensor for exp
            AscendC::LocalTensor<float> expLocal = outQueueY_.AllocTensor<float>();

            // Compute exp(x - rowMax)
            AscendC::Sub(expLocal, xLocal, rowMax, curLen);  // subtract scalar rowMax
            AscendC::Exp(expLocal, expLocal, curLen);

            // Accumulate row sum (reduction to 1-element tensor)
            AscendC::ReduceSum<float, AscendC::REDUCE_SUM>(sumTemp, expLocal, curLen, 0);
            float tileSum = sumTemp.GetValue(0);
            rowSum += tileSum;

            // Write exp to y (temporary)
            AscendC::DataCopy(yGm_[rowOffset + colBase], expLocal, curLen);

            outQueueY_.FreeTensor(expLocal);
            AscendC::LocalTensor<float> xLocalDeq = inQueueX_.DeQue<float>();
            inQueueX_.FreeTensor(xLocalDeq);

            colBase += tileLen_;
        }
        tmpQueue_.FreeTensor(sumTemp);

        // ---------- Pass3: divide each element by rowSum ----------
        float invRowSum = 1.0f / rowSum;
        colBase = 0;
        while (colBase < numFeatures_) {
            uint32_t curLen = numFeatures_ - colBase;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            // Load temporary exp from y
            AscendC::LocalTensor<float> expLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(expLocal, yGm_[rowOffset + colBase], curLen);
            inQueueX_.EnQue(expLocal);

            // Divide by rowSum
            AscendC::LocalTensor<float> resultLocal = outQueueY_.AllocTensor<float>();
            AscendC::Muls(resultLocal, expLocal, invRowSum, curLen);

            // Write final result to y
            AscendC::DataCopy(yGm_[rowOffset + colBase], resultLocal, curLen);

            outQueueY_.FreeTensor(resultLocal);
            AscendC::LocalTensor<float> expLocalDeq = inQueueX_.DeQue<float>();
            inQueueX_.FreeTensor(expLocalDeq);

            colBase += tileLen_;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t numFeatures_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.batchSize, tiling_data.numFeatures);
    op.Process();
}
"""
