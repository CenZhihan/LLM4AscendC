host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget (conservative per-core share)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// Align for efficient DMA
static constexpr uint32_t ALIGN_ELEMS = 2048;
// Hard cap to avoid too large per-tile latency
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SoftmaxCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::StorageShape &origin = x_shape->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(origin.GetShapeSize());
    uint32_t batchSize = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t numFeatures = static_cast<uint32_t>(origin.GetDim(1));

    // softmax requires three passes over each row, so per iteration we need:
    // inQueueX: 1 tile
    // outQueueY: 1 tile (for intermediate exp)
    // tmpQueue: 1 tile (for reductions)
    // Total floats per iteration ≈ 3 * tileLen
    // Budget in floats: UB_BUDGET_BYTES / sizeof(float) / 3
    uint32_t maxTile = (UB_BUDGET_BYTES / sizeof(float)) / 3;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    // If numFeatures smaller than tileLen, tileLen = numFeatures (aligned)
    if (numFeatures < tileLen) {
        uint32_t aligned = AlignDown(numFeatures, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : numFeatures;
    }

    // Block dimension: at most number of rows, but capped
    uint32_t blockDim = batchSize;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batchSize(batchSize);
    tiling.set_numFeatures(numFeatures);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t numFeatures) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        numFeatures_ = numFeatures;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        // outQueueY will be used in pass2 and pass3
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid‑stride loop over rows
        uint32_t rowStride = blockNum;
        uint32_t rowBase = blockIdx;
        while (rowBase < batchSize_) {
            // Process row rowBase
            ProcessRow(rowBase);
            rowBase += rowStride;
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx) {
        uint32_t rowOffset = rowIdx * numFeatures_;

        // ---------- Pass1: compute row max ----------
        float rowMax = -1e38f;  // sufficiently small
        uint32_t colBase = 0;
        while (colBase < numFeatures_) {
            uint32_t curLen = numFeatures_ - colBase;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            // Load tile of x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + colBase], curLen);
            inQueueX_.EnQue(xLocal);

            // Compute max of this tile
            AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceMax(maxLocal, xLocal, curLen);
            // Retrieve scalar max
            float tileMax = maxLocal.GetValue(0);
            tmpQueue_.FreeTensor(maxLocal);
            if (tileMax > rowMax) rowMax = tileMax;

            // Recycle xLocal
            AscendC::LocalTensor<float> xLocalDeq = inQueueX_.DeQue<float>();
            inQueueX_.FreeTensor(xLocalDeq);

            colBase += tileLen_;
        }

        // ---------- Pass2: compute sum of exp(x - rowMax) and write exp to y ----------
        colBase = 0;
        float rowSum = 0.0f;
        while (colBase < numFeatures_) {
            uint32_t curLen = numFeatures_ - colBase;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            // Load x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + colBase], curLen);
            inQueueX_.EnQue(xLocal);

            // Allocate tensor for exp and sum reduction
            AscendC::LocalTensor<float> expLocal = outQueueY_.AllocTensor<float>();

            // Compute exp(x - rowMax)
            AscendC::Sub(expLocal, xLocal, rowMax, curLen);
            AscendC::Exp(expLocal, expLocal, curLen);

            // Accumulate row sum (we'll add by vector reduction)
            AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumLocal, expLocal, curLen);
            float tileSum = sumLocal.GetValue(0);
            tmpQueue_.FreeTensor(sumLocal);

            rowSum += tileSum;

            // Write exp to y (temporary)
            AscendC::DataCopy(yGm_[rowOffset + colBase], expLocal, curLen);

            // Free tensors
            outQueueY_.FreeTensor(expLocal);
            AscendC::LocalTensor<float> xLocalDeq = inQueueX_.DeQue<float>();
            inQueueX_.FreeTensor(xLocalDeq);

            colBase += tileLen_;
        }

        // ---------- Pass3: divide by rowSum ----------
        colBase = 0;
        while (colBase < numFeatures_) {
            uint32_t curLen = numFeatures_ - colBase;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            // Load temporary exp from y
            AscendC::LocalTensor<float> expLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(expLocal, yGm_[rowOffset + colBase], curLen);
            inQueueX_.EnQue(expLocal);

            // Divide by rowSum
            AscendC::LocalTensor<float> resultLocal = outQueueY_.AllocTensor<float>();
            AscendC::Muls(resultLocal, expLocal, 1.0f / rowSum, curLen);

            // Write final result to y
            AscendC::DataCopy(yGm_[rowOffset + colBase], resultLocal, curLen);

            // Free tensors
            outQueueY_.FreeTensor(resultLocal);
            AscendC::LocalTensor<float> expLocalDeq = inQueueX_.DeQue<float>();
            inQueueX_.FreeTensor(expLocalDeq);

            colBase += tileLen_;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t numFeatures_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.batchSize, tiling_data.numFeatures);
    op.Process();
}
"""
