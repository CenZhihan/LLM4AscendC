host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: need buffers for input, output, tmp for exp, max, sum, etc.
// For one row tile we need roughly 5 * tileLen floats => 20 * tileLen bytes.
// Use conservative budget.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    uint32_t dim = static_cast<uint32_t>(*dimPtr);

    // Use GetOriginShape() to get dimensions; it returns gert::Shape with GetDim().
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t numDims = shape.GetDimNum();
    // Assume the reduction is along the last dimension.
    uint32_t rowLen = static_cast<uint32_t>(shape.GetDim(numDims - 1));
    uint32_t numRows = (rowLen > 0) ? totalLength / rowLen : 0;

    // Tile size: fraction of row that fits in UB
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // 20 bytes per element (5 floats)
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLen < tileLen) {
        tileLen = AlignDown(rowLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowLen;
    }

    // Each block processes one row at a time, so blockDim = number of rows
    uint32_t blockDim = numRows;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);
    tiling.set_rowLen(rowLen);
    tiling.set_numRows(numRows);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)

kernel_src="""

#include "kernel_operator.h"
#include <cmath>   // for logf, fmaxf

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t dim, uint32_t rowLen, uint32_t numRows)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        rowLen_ = rowLen;
        numRows_ = numRows;
        (void)dim; // unused in kernel

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(maxBuf_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sumBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes rows in a grid-stride manner
        for (uint32_t rowIdx = blockIdx; rowIdx < numRows_; rowIdx += blockNum) {
            uint32_t rowOffset = rowIdx * rowLen_;
            float rowMax = computeRowMax(rowOffset);
            float rowSum = computeRowSum(rowOffset, rowMax);
            computeOutput(rowOffset, rowMax, rowSum);
        }
    }

private:
    __aicore__ inline float computeRowMax(uint32_t rowOffset)
    {
        float maxVal = -FLT_MAX;
        uint32_t remaining = rowLen_;
        uint32_t cur = 0;
        while (remaining > 0) {
            uint32_t len = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + cur], len);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> maxLocal = maxBuf_.AllocTensor<float>();
            AscendC::LocalTensor<float> buf = tmpQueue_.AllocTensor<float>();
            // Broadcast current maxVal and element-wise max
            AscendC::BroadCast(buf, maxVal, len);
            AscendC::Max(maxLocal, xLocal, buf, len);
            // Reduce to scalar via manual loop
            for (uint32_t i = 0; i < len; i++) {
                float v = maxLocal(i);
                if (v > maxVal) maxVal = v;
            }

            tmpQueue_.FreeTensor(buf);
            maxBuf_.FreeTensor(maxLocal);
            inQueueX_.FreeTensor(xLocal);
            cur += len;
            remaining -= len;
        }
        return maxVal;
    }

    __aicore__ inline float computeRowSum(uint32_t rowOffset, float rowMax)
    {
        float sumVal = 0.0f;
        uint32_t remaining = rowLen_;
        uint32_t cur = 0;
        while (remaining > 0) {
            uint32_t len = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + cur], len);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            // t = x - rowMax
            AscendC::Sub(tLocal, xLocal, rowMax, len);
            // t = exp(t)
            AscendC::Exp(tLocal, tLocal, len);
            // accumulate sum
            for (uint32_t i = 0; i < len; i++) {
                sumVal += tLocal(i);
            }

            tmpQueue_.FreeTensor(tLocal);
            inQueueX_.FreeTensor(xLocal);
            cur += len;
            remaining -= len;
        }
        return sumVal;
    }

    __aicore__ inline void computeOutput(uint32_t rowOffset, float rowMax, float rowSum)
    {
        float logSum = logf(rowSum);
        uint32_t remaining = rowLen_;
        uint32_t cur = 0;
        while (remaining > 0) {
            uint32_t len = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + cur], len);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            // y = x - rowMax - logSum
            AscendC::Sub(tLocal, xLocal, rowMax, len);
            AscendC::Sub(yLocal, tLocal, logSum, len);

            AscendC::DataCopy(yGm_[rowOffset + cur], yLocal, len);

            tmpQueue_.FreeTensor(tLocal);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
            cur += len;
            remaining -= len;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maxBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.dim,
            tiling_data.rowLen,
            tiling_data.numRows);
    op.Process();
}
