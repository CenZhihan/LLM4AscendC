namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k05321CustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin_shape = x_shape->GetOriginShape();

    int32_t ndim = origin_shape.GetDimNum();
    uint32_t N = (ndim >= 1) ? origin_shape.GetDim(0) : 1;
    uint32_t C = (ndim >= 2) ? origin_shape.GetDim(1) : 1;
    uint32_t H = (ndim >= 3) ? origin_shape.GetDim(2) : 1;
    uint32_t W = (ndim >= 4) ? origin_shape.GetDim(3) : 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scale_ptr = attrs->GetAttrPointer<float>(0);
    float scale = (scale_ptr != nullptr) ? *scale_ptr : 2.0f;

    uint32_t H_out = static_cast<uint32_t>(static_cast<float>(H) * scale + 0.5f);
    uint32_t W_out = static_cast<uint32_t>(static_cast<float>(W) * scale + 0.5f);
    uint32_t totalLength = N * C * H_out * W_out;

    // Tile size based on UB budget: each tile processes output elements.
    // Kernel reads input tile based on output tile bounding box.
    // For simplicity, tile size = number of output elements processed per batch.
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float) / 8;  // conservative
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;
    if (tileLen == 0) tileLen = 1;

    uint32_t blockDim = 8;  // fixed, can be tuned
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_scale(scale);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k05321Custom {
public:
    __aicore__ inline KernelCa6k05321Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t H_out, uint32_t W_out, float scale)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N; C_ = C; H_ = H; W_ = W;
        H_out_ = H_out; W_out_ = W_out;
        scale_ = scale;
        invScale_ = 1.0f / scale;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Allocate local buffer: input tile size (upsample area for output tile)
        // For each output tile, we need a rectangle of input that covers that output tile.
        // That rectangle is roughly (ceil(H_out / scale) * ceil(W_out / scale)) per channel.
        // We allocate a fixed max size based on worst-case (stretch with small stride).
        // For simplicity, allocate large enough for 64x64 output tile → ~128x128 input.
        const uint32_t maxInputTile = 128 * 128;  // per channel
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, maxInputTile * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            ProcessTile(base, curLen);

            base += strideElems;
            if (base >= totalLen_) break;
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t outOffset, uint32_t len)
    {
        // Determine output indices for this tile.
        // Output layout: N * C * H_out * W_out (row-major)
        // For each output element, we compute source coordinates and interpolate.
        // Since we have at most tileLen_ elements, we can loop elementwise.
        // For better performance, we would vectorize, but for correctness we use scalar.
        // AscendC supports vector operations but bilinear by nature is scalar.
        // We'll use a simple loop; should be okay for moderate tile sizes.

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // For each element in the tile, compute bilinear interpolation.
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t globalIdx = outOffset + i;
            // Decompose to (n,c,oh,ow)
            uint32_t oh = globalIdx % W_out_;
            uint32_t rest1 = globalIdx / W_out_;
            uint32_t ow = rest1 % H_out_;
            uint32_t rest2 = rest1 / H_out_;
            uint32_t c = rest2 % C_;
            uint32_t n = rest2 / C_;

            // source coordinates in input (center-aligned? PyTorch align_corners=False default)
            float h = (static_cast<float>(oh) + 0.5f) * invScale_ - 0.5f;
            float w = (static_cast<float>(ow) + 0.5f) * invScale_ - 0.5f;

            // clamp to [0, H_-1], [0, W_-1]
            h = AscendC::Max(0.0f, AscendC::Min(static_cast<float>(H_ - 1), h));
            w = AscendC::Max(0.0f, AscendC::Min(static_cast<float>(W_ - 1), w));

            int32_t ih0 = static_cast<int32_t>(h);
            int32_t iw0 = static_cast<int32_t>(w);
            int32_t ih1 = AscendC::Min(ih0 + 1, static_cast<int32_t>(H_ - 1));
            int32_t iw1 = AscendC::Min(iw0 + 1, static_cast<int32_t>(W_ - 1));

            float dy = h - static_cast<float>(ih0);
            float dx = w - static_cast<float>(iw0);

            // Compute input base index for this (n,c)
            uint32_t baseIdx = n * C_ * H_ * W_ + c * H_ * W_;

            float v00 = xGm_.GetValue(baseIdx + ih0 * W_ + iw0);
            float v10 = xGm_.GetValue(baseIdx + ih1 * W_ + iw0);
            float v01 = xGm_.GetValue(baseIdx + ih0 * W_ + iw1);
            float v11 = xGm_.GetValue(baseIdx + ih1 * W_ + iw1);

            float v0 = v00 * (1.0f - dx) + v01 * dx;
            float v1 = v10 * (1.0f - dx) + v11 * dx;
            float result = v0 * (1.0f - dy) + v1 * dy;

            yLocal.SetValue(i, result);
        }

        // Copy out
        AscendC::DataCopy(yGm_[outOffset], yLocal, len);

        outQueueY_.FreeTensor(yLocal);
        tmpQueue_.FreeTensor(tLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_ {0}, C_ {0}, H_ {0}, W_ {0};
    uint32_t H_out_ {0}, W_out_ {0};
    float scale_ {2.0f};
    float invScale_ {0.5f};
};

extern "C" __global__ __aicore__ void ca6k_05321_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k05321Custom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.H_out, tiling_data.W_out, tiling_data.scale);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor ca6k_05321_impl_npu(const at::Tensor& self, double scale_factor) {
    at::Tensor result = at::empty({}, self.options());
    // Infer output shape from input and scale_factor
    int64_t N = self.size(0);
    int64_t C = self.size(1);
    int64_t H = self.size(2);
    int64_t W = self.size(3);
    int64_t H_out = static_cast<int64_t>(static_cast<double>(H) * scale_factor + 0.5);
    int64_t W_out = static_cast<int64_t>(static_cast<double>(W) * scale_factor + 0.5);
    result.resize_({N, C, H_out, W_out});
    EXEC_NPU_CMD(aclnnCa6k05321Custom, self, scale_factor, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_05321_custom", &ca6k_05321_impl_npu);
}

model_src = '''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

_FN = "ca6k_05321_custom"

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, scale_factor):
        super(ModelNew, self).__init__()
        self.scale_factor = scale_factor
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1)

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        x = fn(x, self.scale_factor)
        x = self.conv(x)
        return x
