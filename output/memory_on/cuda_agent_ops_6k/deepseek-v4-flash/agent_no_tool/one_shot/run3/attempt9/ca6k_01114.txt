host_operator_src = """
#include "ca6k01114_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/types.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k01114CustomTilingData tiling;
    const gert::StorageShape *storageShape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(storageShape->GetOriginShape().GetShapeSize());
    uint32_t rows = static_cast<uint32_t>(storageShape->GetOriginShape().GetDim(0));
    uint32_t cols = static_cast<uint32_t>(storageShape->GetOriginShape().GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t diagonal = 0;
    if (attrs && attrs->GetAttrPointer<int32_t>(0)) {
        diagonal = *attrs->GetAttrPointer<int32_t>(0);
    }

    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    // Use setter methods generated by TILING_DATA_FIELD_DEF
    tiling.set_totalLength(totalLength);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_diagonal(diagonal);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    gert::StorageShape *y_shape = context->GetOutputShape(0);

}

kernel_src = """
#include "kernel_operator.h"
#include "ca6k01114_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01114Custom {
public:
    __aicore__ inline KernelCa6k01114Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rows, uint32_t cols,
                                uint32_t tileLength, int32_t diagonal) {
        totalLen_ = totalLength;
        rows_ = rows;
        cols_ = cols;
        tileLen_ = tileLength;
        diagonal_ = diagonal;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = (base + tileLen_ > totalLen_) ? totalLen_ - base : tileLen_;
            CopyIn(base, curLen);
            Compute(curLen, base);
            CopyOut(base, curLen);
            base += strideElems;
            if (base >= totalLen_) break;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len, uint32_t baseOffset) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        for (uint32_t i = 0; i < len; i++) {
            uint32_t globalIdx = baseOffset + i;
            int32_t row = static_cast<int32_t>(globalIdx / cols_);
            int32_t col = static_cast<int32_t>(globalIdx % cols_);
            if (col >= row + diagonal_) {
                yLocal[i] = xLocal[i];
            } else {
                yLocal[i] = 0.0f;
            }
        }
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t offset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileLen_ {0};
    int32_t diagonal_ {0};
};

extern "C" __global__ __aicore__ void ca6k_01114_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01114Custom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.rows,
            tiling_data.cols,
            tiling_data.tileLength,
            tiling_data.diagonal);
    op.Process();
}
"""
