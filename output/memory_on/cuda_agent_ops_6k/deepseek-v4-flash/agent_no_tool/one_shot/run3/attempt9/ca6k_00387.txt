kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;
constexpr float NEG_INF = -1e10f;

class KernelCa6k00387Custom {
public:
    __aicore__ inline KernelCa6k00387Custom() {}

    __aicore__ inline void Init(GM_ADDR logProbs, GM_ADDR targets,
                                GM_ADDR inputLengths, GM_ADDR targetLengths,
                                GM_ADDR loss,
                                uint32_t batchSize, float batchSizeFloat,
                                uint32_t maxTime,
                                uint32_t maxTargetLen, uint32_t numClasses,
                                int32_t blank, uint8_t reductionMean)
    {
        batchSize_ = batchSize;
        batchSizeFloat_ = batchSizeFloat;
        maxTime_ = maxTime;
        maxTargetLen_ = maxTargetLen;
        numClasses_ = numClasses;
        blank_ = blank;
        reductionMean_ = reductionMean;

        logProbsGm_.SetGlobalBuffer((__gm__ float*)logProbs, maxTime * batchSize * numClasses);
        targetsGm_.SetGlobalBuffer((__gm__ int32_t*)targets, batchSize * maxTargetLen);
        inputLengthsGm_.SetGlobalBuffer((__gm__ int32_t*)inputLengths, batchSize);
        targetLengthsGm_.SetGlobalBuffer((__gm__ int32_t*)targetLengths, batchSize);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        // Use TQue for target copy (single buffered)
        pipe_.InitBuffer(inQueueTargets_, BUFFER_NUM, maxTargetLen_ * sizeof(int32_t));

        // Use TBuf for logProbs row and temporary computation
        pipe_.InitBuffer(logProbsRowBuf_, numClasses_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 2 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        float totalLoss = 0.0f;

        for (uint32_t batchIdx = 0; batchIdx < batchSize_; ++batchIdx) {
            int32_t T = inputLengthsGm_.GetValue(batchIdx);
            int32_t L = targetLengthsGm_.GetValue(batchIdx);
            if (T <= 0 || L <= 0) continue;

            // Copy target sequence into local via TQue
            AscendC::LocalTensor<int32_t> targetsLocal = inQueueTargets_.AllocTensor<int32_t>();
            AscendC::DataCopy(targetsLocal, targetsGm_[batchIdx * maxTargetLen_], L);
            inQueueTargets_.EnQue(targetsLocal);
            AscendC::LocalTensor<int32_t> tgt = inQueueTargets_.DeQue<int32_t>();

            // Build extended target sequence (U = 2*L+1 <= 61)
            int32_t extended[61];
            extended[0] = blank_;
            for (int32_t i = 0; i < L; ++i) {
                extended[2 * i + 1] = tgt.GetValue(i);
                extended[2 * i + 2] = blank_;
            }
            int32_t U = 2 * L + 1;

            // Allocate local tensor for logProbs row from TBuf
            AscendC::LocalTensor<float> logProbsRow = logProbsRowBuf_.AllocTensor<float>();

            // Copy row for t=0
            AscendC::DataCopy(logProbsRow,
                              logProbsGm_[batchIdx * numClasses_],
                              numClasses_);

            // Initialize alpha at t=0
            float alphaPrev[61];
            float alphaCurr[61];
            alphaCurr[0] = logProbsRow.GetValue(extended[0]);
            for (int32_t u = 1; u < U; ++u) {
                if (u == 1) {
                    alphaCurr[1] = logProbsRow.GetValue(extended[1]);
                } else {
                    alphaCurr[u] = NEG_INF;
                }
            }

            // Loop over time steps
            for (int32_t t = 1; t < T; ++t) {
                // Copy alphaCurr to alphaPrev
                for (int32_t u = 0; u < U; ++u) {
                    alphaPrev[u] = alphaCurr[u];
                }

                // Copy current time step row into local
                uint32_t gmOffset = t * batchSize_ * numClasses_ + batchIdx * numClasses_;
                AscendC::DataCopy(logProbsRow,
                                  logProbsGm_[gmOffset],
                                  numClasses_);

                // Update alphaCurr for this time step
                for (int32_t u = 0; u < U; ++u) {
                    float logProb = logProbsRow.GetValue(extended[u]);
                    float best = alphaPrev[u] + logProb;
                    if (u > 0) {
                        float val = alphaPrev[u - 1] + logProb;
                        if (val > best) best = val;
                    }
                    if (u > 1 && extended[u] != blank_ && extended[u] != extended[u - 2]) {
                        float val = alphaPrev[u - 2] + logProb;
                        if (val > best) best = val;
                    }
                    alphaCurr[u] = best;
                }
            }

            // Free logProbsRow (no longer needed for this batch)
            logProbsRowBuf_.FreeTensor(logProbsRow);

            // Compute loss for this sample using log-sum-exp
            float val1 = alphaCurr[U - 1];
            float val2 = alphaCurr[U - 2];
            float maxVal = (val1 > val2) ? val1 : val2;

            AscendC::LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>();
            tmp.SetValue(0, val1 - maxVal);
            tmp.SetValue(1, val2 - maxVal);
            AscendC::Exp(tmp, tmp, 2);
            float sumExp = tmp.GetValue(0) + tmp.GetValue(1);
            tmp.SetValue(0, sumExp);
            AscendC::Log(tmp, tmp, 1);
            float logSumExp = tmp.GetValue(0);
            tmpBuf_.FreeTensor(tmp);

            float lossSample = -(maxVal + logSumExp);
            totalLoss += lossSample;

            inQueueTargets_.FreeTensor(tgt);
        }

        if (reductionMean_) {
            totalLoss /= batchSizeFloat_;
        }
        lossGm_.SetValue(0, totalLoss);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTargets_;
    AscendC::TBuf<AscendC::TPosition::VECIN> logProbsRowBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;

    AscendC::GlobalTensor<float> logProbsGm_;
    AscendC::GlobalTensor<int32_t> targetsGm_;
    AscendC::GlobalTensor<int32_t> inputLengthsGm_;
    AscendC::GlobalTensor<int32_t> targetLengthsGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t batchSize_;
    float batchSizeFloat_;
    uint32_t maxTime_;
    uint32_t maxTargetLen_;
    uint32_t numClasses_;
    int32_t blank_;
    uint8_t reductionMean_;
};

extern "C" __global__ __aicore__ void ca6k00387_custom(GM_ADDR logProbs, GM_ADDR targets,
                                                         GM_ADDR inputLengths, GM_ADDR targetLengths,
                                                         GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00387Custom op;
    op.Init(logProbs, targets, inputLengths, targetLengths, loss,
            tiling_data.batchSize, tiling_data.batchSizeFloat,
            tiling_data.maxTime,
            tiling_data.maxTargetLen, tiling_data.numClasses,
            tiling_data.blank, tiling_data.reductionMean);
    op.Process();
}
