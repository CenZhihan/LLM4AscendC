namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k02116CustomTilingData tiling;

    auto originShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batch = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t channels = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t inputLength = static_cast<uint32_t>(originShape.GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *output_size_ptr = attrs->GetAttrPointer<int64_t>(0);
    uint32_t outputSize = static_cast<uint32_t>(*output_size_ptr);

    uint32_t totalOutputElems = batch * channels * outputSize;
    uint32_t blockDim = 24;
    uint32_t tileLength = (totalOutputElems + blockDim - 1) / blockDim;

    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_inputLength(inputLength);
    tiling.set_outputSize(outputSize);
    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src = '''
#include "kernel_operator.h"
#include "ca6k02116_custom_tiling.h"

class KernelCa6k02116Custom {
public:
    __aicore__ inline KernelCa6k02116Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t channels,
                                uint32_t inputLength, uint32_t outputSize,
                                uint32_t totalOutputElems, uint32_t tileLength)
    {
        batch_ = batch;
        channels_ = channels;
        inputLength_ = inputLength;
        outputSize_ = outputSize;
        totalOutputElems_ = totalOutputElems;
        tileLength_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * channels * inputLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElems);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t baseIdx = blockIdx * tileLength_;
        if (baseIdx >= totalOutputElems_) return;

        uint32_t count = tileLength_;
        if (baseIdx + count > totalOutputElems_) {
            count = totalOutputElems_ - baseIdx;
        }

        for (uint32_t i = 0; i < count; i++) {
            uint32_t globalIdx = baseIdx + i;

            // linear index -> (n, c, out_idx)
            uint32_t tmp = globalIdx / outputSize_;
            uint32_t out_idx = globalIdx - tmp * outputSize_;
            uint32_t n = tmp / channels_;
            uint32_t c = tmp - n * channels_;

            // Adaptive max pooling: map output index to input range
            uint32_t start = out_idx * inputLength_ / outputSize_;
            uint32_t end = ((out_idx + 1) * inputLength_ + outputSize_ - 1) / outputSize_ - 1;
            if (end >= inputLength_) end = inputLength_ - 1;

            uint32_t inputBase = (n * channels_ + c) * inputLength_ + start;

            // Initialize max with first element
            float maxVal = xGm_[inputBase];
            for (uint32_t j = 1; j <= end - start; j++) {
                float val = xGm_[inputBase + j];
                if (val > maxVal) maxVal = val;
            }
            yGm_[globalIdx] = maxVal;
        }
    }

private:
    uint32_t batch_;
    uint32_t channels_;
    uint32_t inputLength_;
    uint32_t outputSize_;
    uint32_t totalOutputElems_;
    uint32_t tileLength_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
};

extern "C" __global__ __aicore__ void ca6k_02116_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02116Custom op;
    op.Init(x, y,
            tiling_data.batch, tiling_data.channels,
            tiling_data.inputLength, tiling_data.outputSize,
            tiling_data.totalOutputElems, tiling_data.tileLength);
    op.Process();
}
'''
