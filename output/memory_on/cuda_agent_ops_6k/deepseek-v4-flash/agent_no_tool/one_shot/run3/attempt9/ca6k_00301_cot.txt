class MatMul {
public:
    __aicore__ inline void SetTensorA(const GlobalTensor<AType>& A, uint32_t M, uint32_t K, bool transA);
    __aicore__ inline void SetTensorB(const GlobalTensor<BType>& B, uint32_t K, uint32_t N, bool transB);

};
