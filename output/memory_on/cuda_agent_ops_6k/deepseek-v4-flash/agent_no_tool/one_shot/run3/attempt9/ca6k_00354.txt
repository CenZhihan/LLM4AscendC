host_operator_src = '''
#include "ca6k_00354_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00354CustomTilingData tiling;

    const gert::StorageShape *inShape = context->GetInputShape(0);
    uint32_t N = inShape->GetOriginShape().GetDim(0);
    uint32_t C = inShape->GetOriginShape().GetDim(1);
    uint32_t H = inShape->GetOriginShape().GetDim(2);
    uint32_t W = inShape->GetOriginShape().GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *padSizeH = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *padSizeW = attrs->GetAttrPointer<int32_t>(1);

    int32_t padH = *padSizeH;
    int32_t padW = *padSizeW;
    uint32_t outH = H + 2 * padH;
    uint32_t outW = W + 2 * padW;
    uint32_t totalLength = N * C * outH * outW;

    const uint32_t UB_BUDGET_BYTES = 192 * 1024;
    const uint32_t ALIGN_ELEMS = 2048;
    const uint32_t HARD_CAP_ELEMS = 32768;

    uint32_t maxTile = UB_BUDGET_BYTES / 4;
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }
    if (tileLen > HARD_CAP_ELEMS) tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_padSizeH(padH);
    tiling.set_padSizeW(padW);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    int32_t padH = 0, padW = 0;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs != nullptr) {
        const int32_t *padSizeH = attrs->GetAttrPointer<int32_t>(0);
        const int32_t *padSizeW = attrs->GetAttrPointer<int32_t>(1);
        if (padSizeH) padH = *padSizeH;
        if (padSizeW) padW = *padSizeW;
    }
    y_shape->SetDim(2, x_shape->GetDim(2) + 2 * padH);
    y_shape->SetDim(3, x_shape->GetDim(3) + 2 * padW);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class Ca6k00354Custom : public OpDef {
public:
    explicit Ca6k00354Custom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("pad_size_h").AttrType(OPTIONAL).Int(0);
        this->Attr("pad_size_w").AttrType(OPTIONAL).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(Ca6k00354Custom);
} // namespace ops

kernel_src = '''
#include "kernel_operator.h"

class KernelCa6k00354Custom {
public:
    __aicore__ inline KernelCa6k00354Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t padSizeH, int32_t padSizeW,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t outH, uint32_t outW)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        padH_ = padSizeH;
        padW_ = padSizeW;
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        outH_ = outH;
        outW_ = outW;

        uint32_t inLen = N_ * C_ * H_ * W_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, inLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || N_ == 0 || C_ == 0 || H_ == 0 || W_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t start = blockIdx * tileLen_;
        uint32_t end = start + tileLen_;
        if (start >= totalLen_) return;
        if (end > totalLen_) end = totalLen_;

        uint32_t strideC = outH_ * outW_;
        uint32_t strideN = C_ * strideC;
        uint32_t strideInC = H_ * W_;
        uint32_t strideInN = C_ * strideInC;

        for (uint32_t i = start; i < end; ++i) {
            uint32_t tmp = i;
            uint32_t n = tmp / strideN;
            tmp = tmp % strideN;
            uint32_t c = tmp / strideC;
            tmp = tmp % strideC;
            uint32_t h = tmp / outW_;
            uint32_t w = tmp % outW_;

            int32_t inH = static_cast<int32_t>(h) - padH_;
            inH = (inH < 0) ? 0 : (inH >= static_cast<int32_t>(H_))
                                    ? static_cast<int32_t>(H_) - 1 : inH;
            int32_t inW = static_cast<int32_t>(w) - padW_;
            inW = (inW < 0) ? 0 : (inW >= static_cast<int32_t>(W_))
                                    ? static_cast<int32_t>(W_) - 1 : inW;

            uint32_t inIdx = n * strideInN + c * strideInC +
                             static_cast<uint32_t>(inH) * W_ +
                             static_cast<uint32_t>(inW);
            float val = xGm_[inIdx];
            yGm_[i] = val;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t padH_ {0};
    int32_t padW_ {0};
    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00354_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00354Custom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.padSizeH,
            tiling_data.padSizeW,
            tiling_data.N,
            tiling_data.C,
            tiling_data.H,
            tiling_data.W,
            tiling_data.outH,
            tiling_data.outW);
    op.Process();
}
'''
