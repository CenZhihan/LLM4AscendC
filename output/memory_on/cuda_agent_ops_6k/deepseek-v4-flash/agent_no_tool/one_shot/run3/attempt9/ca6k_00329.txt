#include "ca6k_00329_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00329CustomTilingData tiling;
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const auto &origin_shape = x_shape->GetOriginShape();
    uint32_t dimNum = origin_shape.GetDimNum();
    uint32_t batch = 1, channels = 1, height = 1, width = 1;
    if (dimNum == 4) {
        batch = origin_shape.GetDim(0);
        channels = origin_shape.GetDim(1);
        height = origin_shape.GetDim(2);
        width = origin_shape.GetDim(3);
    } else if (dimNum == 3) {
        batch = origin_shape.GetDim(0);
        channels = origin_shape.GetDim(1);
        height = origin_shape.GetDim(2);
        width = 1;
    } else if (dimNum == 2) {
        batch = origin_shape.GetDim(0);
        channels = origin_shape.GetDim(1);
        height = 1;
        width = 1;
    } else {
        return ge::GRAPH_FAILED;
    }

    uint32_t totalLength = batch * channels * height * width;
    uint32_t spatialSize = height * width;
    uint32_t elementsPerChannel = batch * spatialSize;

    // UB budget: we need per-channel sum (float), sum squares (float), and tile of input/output
    // Rough: 2 * tileLen for input/output + 2 * channels for sum/sumSq -> channels is small.
    uint32_t maxTilePerChannel = (UB_BUDGET_BYTES / sizeof(float)) / 2;  // half for input/output
    maxTilePerChannel = AlignDown(maxTilePerChannel, ALIGN_ELEMS);
    if (maxTilePerChannel < ALIGN_ELEMS) maxTilePerChannel = ALIGN_ELEMS;
    if (maxTilePerChannel > HARD_CAP_ELEMS) maxTilePerChannel = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // tileLength: number of elements per channel to process per iteration
    // We'll process one channel at a time, tiling the spatial dimension
    uint32_t tileLen = maxTilePerChannel;
    if (spatialSize < tileLen) {
        tileLen = AlignDown(spatialSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = spatialSize;
    }

    // blockDim: one block per channel for simplicity (channel parallelism)
    uint32_t blockDim = channels;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_spatialSize(spatialSize);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00329Custom {
public:
    __aicore__ inline KernelCa6k00329Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batch, uint32_t channels,
                                uint32_t spatialSize, float eps)
    {
        batch_ = batch;
        channels_ = channels;
        spatialSize_ = spatialSize;
        tileLen_ = tileLength;
        eps_ = eps;
        totalLen_ = totalLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        // For per-channel sum and sum squares
        pipe_.InitBuffer(sumBuf_, 1, sizeof(float));
        pipe_.InitBuffer(sumSqBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= channels_) return;

        // Each block handles one channel
        uint32_t c = blockIdx;
        uint32_t batchStride = channels_ * spatialSize_;
        uint32_t channelStride = spatialSize_;

        // Compute sum and sum of squares for this channel across all batch and spatial
        float sum = 0.0f;
        float sumSq = 0.0f;
        uint32_t elementCount = batch_ * spatialSize_;
        uint32_t elementsProcessed = 0;

        // Tiled reduction
        while (elementsProcessed < elementCount) {
            uint32_t remaining = elementCount - elementsProcessed;
            uint32_t curLen = (remaining < tileLen_) ? remaining : tileLen_;

            // Load data for this tile
            uint32_t base = c * spatialSize_ + elementsProcessed;  // within one batch? Need batch offset
            // Actually, data layout: [batch, channels, H, W] -> offset = b * batchStride + c * channelStride + hw
            // We need to iterate over batch as well. For simplicity, we'll load contiguous per batch-tile? 
            // Better: We'll load elements in order: for b in 0..batch-1, for hw in 0..spatialSize-1
            // But that's not contiguous. To keep simple, we'll process single element at a time (slow but correct).
            // Since tileLen is large, we can still process large chunks if we map.
            // We'll just do a scalar loop for reduction, but that's inefficient.
            // For correctness, we'll use scalar accumulation inside the loop.
            // We'll compute sum locally and write to global later.
            // But we need to use vector operations. Given complexity, we'll implement a simple but correct version:
            // Use a single scalar reduction on the whole channel. We'll load and accumulate.
            // This will pass correctness even if not optimal.

            // We'll load one element at a time (for correctness)
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t globalIdx = (elementsProcessed + i); // need to compute correct index
                uint32_t b = globalIdx / spatialSize_;
                uint32_t hw = globalIdx % spatialSize_;
                globalIdx = b * batchStride + c * channelStride + hw;
                float val = xGm_.GetValue(globalIdx);
                sum += val;
                sumSq += val * val;
            }

            elementsProcessed += curLen;
        }

        // Now compute mean and variance
        float mean = sum / (float)elementCount;
        float var = sumSq / (float)elementCount - mean * mean;
        float invStd = 1.0f / sqrt(var + eps_);

        // Apply normalization: y = (x - mean) * invStd
        elementsProcessed = 0;
        while (elementsProcessed < elementCount) {
            uint32_t remaining = elementCount - elementsProcessed;
            uint32_t curLen = (remaining < tileLen_) ? remaining : tileLen_;

            // Process tile with vector operations
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            // Load tile
            // We need to copy from global in correct order; for simplicity we copy element by element
            // Better: use DataCopy with stride? Not easy. We'll load via loop for correctness.
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t globalIdx = (elementsProcessed + i);
                uint32_t b = globalIdx / spatialSize_;
                uint32_t hw = globalIdx % spatialSize_;
                globalIdx = b * batchStride + c * channelStride + hw;
                xLocal.SetValue(i, xGm_.GetValue(globalIdx));
            }

            // Compute (x - mean) * invStd
            AscendC::Adds(tLocal, xLocal, -mean, curLen);
            AscendC::Muls(yLocal, tLocal, invStd, curLen);

            // Write back
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t globalIdx = (elementsProcessed + i);
                uint32_t b = globalIdx / spatialSize_;
                uint32_t hw = globalIdx % spatialSize_;
                globalIdx = b * batchStride + c * channelStride + hw;
                yGm_.SetValue(globalIdx, yLocal.GetValue(i));
            }

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            tmpQueue_.FreeTensor(tLocal);

            elementsProcessed += curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_ {0};
    uint32_t channels_ {0};
    uint32_t spatialSize_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
    uint32_t totalLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00329_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00329Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.batch, tiling_data.channels,
            tiling_data.spatialSize, tiling_data.eps);
    op.Process();
}
