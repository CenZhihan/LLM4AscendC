static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00294CustomTilingData tiling;
    const gert::StorageShape *indicesShape = context->GetInputShape(0);
    const gert::StorageShape *weightShape = context->GetInputShape(1);

    uint32_t totalIndices = indicesShape->GetOriginShape().GetShapeSize();
    uint32_t embeddingDim = weightShape->GetOriginShape().GetDim(1);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *paddingIdx = attrs->GetAttrPointer<int64_t>(0);

    uint32_t rowCost = 2 * embeddingDim * sizeof(float) + sizeof(int64_t);
    uint32_t maxTile = 192 * 1024 / rowCost;
    if (maxTile > 256) maxTile = 256;
    maxTile = (maxTile / 32) * 32;
    if (maxTile < 32) maxTile = 32;

    uint32_t tileIndices = (totalIndices < maxTile) ? totalIndices : maxTile;
    if (tileIndices == 0) tileIndices = 1;

    uint32_t blockDim = (totalIndices + tileIndices - 1) / tileIndices;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalIndices(totalIndices);
    tiling.set_tileIndices(tileIndices);
    tiling.set_embeddingDim(embeddingDim);
    tiling.set_paddingIdx(*paddingIdx);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00294Custom {
public:
    __aicore__ inline KernelCa6k00294Custom() {}

    __aicore__ inline void Init(GM_ADDR indices, GM_ADDR weight, GM_ADDR output,
                                uint32_t totalIndices, uint32_t tileIndices,
                                uint32_t embeddingDim, int64_t paddingIdx)
    {
        totalIndices_ = totalIndices;
        tileIndices_ = tileIndices;
        embeddingDim_ = embeddingDim;
        paddingIdx_ = paddingIdx;

        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices, totalIndices_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalIndices_ * embeddingDim_);

        pipe_.InitBuffer(inQueueIndices_, BUFFER_NUM, tileIndices_ * sizeof(int64_t));
        pipe_.InitBuffer(outQueueY_, 1, embeddingDim_ * sizeof(float));
        pipe_.InitBuffer(weightRowTmp_, 1, embeddingDim_ * sizeof(float));
        pipe_.InitBuffer(zeroRowTmp_, 1, embeddingDim_ * sizeof(float));
        AscendC::SetZero(zeroRowTmp_, embeddingDim_);
    }

    __aicore__ inline void Process()
    {
        if (totalIndices_ == 0 || tileIndices_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t strideRows = blockNum * tileIndices_;
        uint32_t startRow = blockIdx * tileIndices_;

        // grid-stride loop over tiles
        for (uint32_t base = startRow; base < totalIndices_; base += strideRows) {
            uint32_t curTile = (base + tileIndices_ > totalIndices_) ? (totalIndices_ - base) : tileIndices_;
            // load indices
            AscendC::LocalTensor<int64_t> idxLocal = inQueueIndices_.AllocTensor<int64_t>();
            AscendC::DataCopy(idxLocal, indicesGm_[base], curTile);
            inQueueIndices_.EnQue(idxLocal);

            // process each index in the tile
            AscendC::LocalTensor<int64_t> idxDeque = inQueueIndices_.DeQue<int64_t>();
            for (uint32_t i = 0; i < curTile; ++i) {
                int64_t idx = idxDeque.GetValue(i);
                uint32_t outOffset = (base + i) * embeddingDim_;
                if (idx == paddingIdx_) {
                    // zero row
                    AscendC::DataCopy(outputGm_[outOffset], zeroRowTmp_, embeddingDim_);
                } else {
                    // copy weight row
                    AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                    uint32_t weightOffset = static_cast<uint32_t>(idx) * embeddingDim_;
                    AscendC::DataCopy(yLocal, weightGm_[weightOffset], embeddingDim_);
                    AscendC::DataCopy(outputGm_[outOffset], yLocal, embeddingDim_);
                    outQueueY_.FreeTensor(yLocal);
                }
            }
            inQueueIndices_.FreeTensor(idxDeque);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIndices_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightRowTmp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> zeroRowTmp_;

    AscendC::GlobalTensor<int64_t> indicesGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t totalIndices_ {0};
    uint32_t tileIndices_ {0};
    uint32_t embeddingDim_ {0};
    int64_t paddingIdx_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00294_custom(GM_ADDR indices, GM_ADDR weight, GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00294Custom op;
    op.Init(indices, weight, output,
            tiling_data.totalIndices, tiling_data.tileIndices,
            tiling_data.embeddingDim, tiling_data.paddingIdx);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00294_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_embeddings, embedding_dim, padding_idx):
        super(ModelNew, self).__init__()
        self.weight = torch.nn.Parameter(torch.randn(num_embeddings, embedding_dim, dtype=torch.float))
        self.padding_idx = padding_idx

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        # x shape: (batch, seq_length)
        batch, seq = x.shape
        result = fn(x.contiguous().view(-1), self.weight, self.padding_idx)
        result = result.view(batch, seq, -1)
        return result
'''
