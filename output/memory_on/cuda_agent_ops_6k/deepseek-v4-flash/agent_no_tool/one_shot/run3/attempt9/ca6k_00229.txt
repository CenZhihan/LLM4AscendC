static constexpr uint32_t ALIGN_ROWS = 1;
static constexpr uint32_t MAX_TILE_ROWS = 128;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00229CustomTilingData tiling;

    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::Shape &x_shape = x_storage->GetOriginShape();
    uint32_t batch = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t channels = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t inputH = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t inputW = static_cast<uint32_t>(x_shape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t padTop = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(0)[0]);
    uint32_t padBottom = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(1)[0]);
    uint32_t padLeft = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(2)[0]);
    uint32_t padRight = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(3)[0]);

    uint32_t outputH = inputH + padTop + padBottom;
    uint32_t outputW = inputW + padLeft + padRight;
    uint32_t totalOutputRows = batch * channels * outputH;

    // UB budget: each row needs one local tensor of outputW floats (for double buffering we need 2)
    // plus a small constant for pipeline overhead.
    uint32_t maxTileRows = (UB_BUDGET_BYTES / sizeof(float)) / (2 * outputW);
    if (maxTileRows == 0) maxTileRows = 1;
    if (maxTileRows > MAX_TILE_ROWS) maxTileRows = MAX_TILE_ROWS;
    uint32_t tileRows = maxTileRows;
    if (tileRows > totalOutputRows) tileRows = totalOutputRows;
    if (tileRows == 0) tileRows = 1;

    uint32_t numTiles = CeilDiv(totalOutputRows, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_padTop(padTop);
    tiling.set_padBottom(padBottom);
    tiling.set_padLeft(padLeft);
    tiling.set_padRight(padRight);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_totalOutputRows(totalOutputRows);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

kernel_src = '''
#include "kernel_operator.h"
#include "ca6k_00229_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00229Custom {
public:
    __aicore__ inline KernelCa6k00229Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                GM_ADDR workspace, GM_ADDR tiling) {
        xGm_ = (__gm__ float*)x;
        yGm_ = (__gm__ float*)y;
        tilingBuf_ = tiling;
        // Initialize pipe with double-buffered output row tensors
        // We will allocate them per-row in Process, so no fixed-size init needed.
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 0);  // allocated dynamically per row
    }

    __aicore__ inline void Process() {
        Ca6k00229CustomTilingData tiling;
        GET_TILING_DATA(tiling, tilingBuf_);

        uint32_t batch = tiling.batch;
        uint32_t channels = tiling.channels;
        uint32_t inputH = tiling.inputH;
        uint32_t inputW = tiling.inputW;
        uint32_t padTop = tiling.padTop;
        uint32_t padBottom = tiling.padBottom;
        uint32_t padLeft = tiling.padLeft;
        uint32_t padRight = tiling.padRight;
        uint32_t outputH = tiling.outputH;
        uint32_t outputW = tiling.outputW;
        uint32_t totalOutputRows = tiling.totalOutputRows;
        uint32_t tileRows = tiling.tileRows;
        uint32_t blockDim = tiling.blockDim;

        if (totalOutputRows == 0 || tileRows == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t baseRow = blockIdx * tileRows;
        if (baseRow >= totalOutputRows) return;

        // Pre-allocate a zero buffer for padding (constant zeros)
        AscendC::LocalTensor<float> zeroBuf;
        pipe_.InitBuffer(zeroBuf, 1, outputW * sizeof(float));
        AscendC::SetZero(zeroBuf, outputW);

        // Grid-stride loop over tiles
        while (baseRow < totalOutputRows) {
            uint32_t rowsThisTile = (baseRow + tileRows <= totalOutputRows) ? tileRows : (totalOutputRows - baseRow);

            for (uint32_t r = 0; r < rowsThisTile; ++r) {
                uint32_t rowIdx = baseRow + r;
                // Compute the batch, channel, output row index
                uint32_t planesPerChannel = outputH;
                uint32_t planeIdx = rowIdx / planesPerChannel;  // which (batch, channel) plane
                uint32_t outRow = rowIdx % planesPerChannel;

                // Allocate output local tensor for this row
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                // Zero-initialize from zeroBuf
                AscendC::DataCopy(yLocal, zeroBuf, outputW);

                if (outRow >= padTop && outRow < padTop + inputH) {
                    uint32_t inRow = outRow - padTop;
                    // Input row offset within the plane: (planeIdx * inputH + inRow) * inputW
                    uint32_t inputPlaneStride = inputH * inputW;
                    uint32_t inputRowOffset = planeIdx * inputPlaneStride + inRow * inputW;
                    // Copy input row into yLocal at padLeft offset
                    AscendC::DataCopy(yLocal[padLeft], xGm_[inputRowOffset], inputW);
                }
                // Else row remains zero (already initialized)

                // Output row offset: (planeIdx * outputH + outRow) * outputW
                uint32_t outputPlaneStride = outputH * outputW;
                uint32_t outputRowOffset = planeIdx * outputPlaneStride + outRow * outputW;
                AscendC::DataCopy(yGm_[outputRowOffset], yLocal, outputW);

                outQueueY_.FreeTensor(yLocal);
            }

            baseRow += blockDim * tileRows;  // grid-stride
        }
        pipe_.FreeTensor(zeroBuf);
    }

private:
    __gm__ float* xGm_;
    __gm__ float* yGm_;
    __gm__ void* tilingBuf_;
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
};

extern "C" __global__ __aicore__ void ca6k_00229_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    KernelCa6k00229Custom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}
'''
