host_operator_src = '''
#include "ca6k_04582_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04582CustomTiling tiling;

    const gert::StorageShape* xShape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t D = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    uint32_t totalLen = N * (N - 1) / 2;

    // Choose blockDim. For N=512 we can use up to 24 blocks.
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    // Assign contiguous i ranges to each block.
    uint32_t iStart, iEnd, outOffset = 0;
    uint32_t pairsPerBlock = (totalLen + blockDim - 1) / blockDim;
    uint32_t i = 0;
    uint32_t cum = 0;
    for (uint32_t b = 0; b < blockDim; b++) {
        uint32_t target = (b + 1) * pairsPerBlock;
        if (target > totalLen) target = totalLen;
        uint32_t pairsThisBlock = target - cum;
        if (pairsThisBlock == 0) {
            // Not enough work for all blocks; set empty range for this block
            tiling.set_N(N);
            tiling.set_D(D);
            tiling.set_iStart(0);
            tiling.set_iEnd(0);
            tiling.set_outOffset(outOffset);
            // tiling saved per block is not done here; we need to set per block? Actually we will set blockDim and save tiling data once.
            // AscendC tiling expects same tiling data for all blocks? Actually each block gets the same tiling data, but we can assign work via blockIdx.
            // So we must compute iStart and iEnd for this block inside the kernel based on blockIdx.
            // Better: Pass blockDim and totalN, and let kernel compute its own range.
            // To keep consistency, we will pass N, D, blockDim, and let kernel compute work using grid stride.
            // This is simpler and avoids complex host tiling.
            // For simplicity, we revert to grid-stride approach.
            // But since we already started, let's fix: use grid stride across pairs.
        }
        cum = target;
    }
    // Simpler: Use grid stride over pairs. Pass N, D only.
    // Update tiling fields accordingly.
    tiling.set_N(N);
    tiling.set_D(D);
    tiling.set_iStart(0);
    tiling.set_iEnd(0);
    tiling.set_outOffset(0);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape* xShape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t M = N * (N - 1) / 2;
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(1);
    y_shape->SetDim(0, static_cast<int64_t>(M));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''
#include "kernel_operator.h"
#include "ca6k_04582_custom_tiling.h"
#include <math.h>

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k04582Custom {
public:
    __aicore__ inline KernelCa6k04582Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t D)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, N * D);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * (N - 1) / 2);
        N_ = N;
        D_ = D;
        totalPairs_ = N * (N - 1) / 2;

        pipe_.InitBuffer(bufRowI_, 1, D_ * sizeof(float));
        pipe_.InitBuffer(bufRowJ_, 1, D_ * sizeof(float));
        pipe_.InitBuffer(bufDiffSq_, 1, D_ * sizeof(float));
        pipe_.InitBuffer(bufSum_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Grid-stride over all pairs
        for (uint32_t idx = blockIdx; idx < totalPairs_; idx += blockNum) {
            // Compute i and j from linear index idx
            uint32_t i = 0;
            // use a while loop to find i: number of pairs before row i is i*(2*N_-i-1)/2
            uint32_t start = 0;
            uint32_t end = 0;
            while (true) {
                uint32_t nextStart = (i + 1) * (2 * N_ - i - 2) / 2;
                if (nextStart > idx) {
                    break;
                }
                i++;
                start = nextStart;
            }
            uint32_t j = idx - (i * (2 * N_ - i - 1) / 2) + i + 1;

            // Allocate local tensors
            AscendC::LocalTensor<float> rowI = bufRowI_.AllocTensor<float>();
            AscendC::LocalTensor<float> rowJ = bufRowJ_.AllocTensor<float>();
            AscendC::LocalTensor<float> diffSq = bufDiffSq_.AllocTensor<float>();
            AscendC::LocalTensor<float> sum = bufSum_.AllocTensor<float>();

            // Load rows
            AscendC::DataCopy(rowI, xGm_[i * D_], D_);
            AscendC::DataCopy(rowJ, xGm_[j * D_], D_);

            // Compute diff = rowI - rowJ, then square and reduce sum
            AscendC::Sub(rowI, rowJ, diffSq, D_);
            AscendC::Mul(diffSq, diffSq, diffSq, D_);
            AscendC::ReduceAdd(sum, diffSq, D_);

            // Get scalar sum and sqrt
            float s = sum(0);
            float dist = sqrtf(s);

            // Write result
            yGm_.SetValue(idx, dist);

            // Free tensors
            bufRowI_.FreeTensor(rowI);
            bufRowJ_.FreeTensor(rowJ);
            bufDiffSq_.FreeTensor(diffSq);
            bufSum_.FreeTensor(sum);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufRowI_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufRowJ_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufDiffSq_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufSum_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t D_ {0};
    uint32_t totalPairs_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04582_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04582Custom op;
    op.Init(x, y, tiling_data.N, tiling_data.D);
    op.Process();
}
'''
