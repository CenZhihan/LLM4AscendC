kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04593Custom {
public:
    __aicore__ inline KernelCa6k04593Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batch, uint32_t channels,
                                uint32_t depth, uint32_t height, uint32_t width,
                                float eps)
    {
        spatial_ = depth * height * width;
        totalRows_ = batch * channels;
        eps_ = eps;
        tileLen_ = tileLength > spatial_ ? spatial_ : tileLength;
        if (tileLen_ == 0) tileLen_ = spatial_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (spatial_ == 0 || totalRows_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t row = blockIdx;
        while (row < totalRows_) {
            uint32_t offset = row * spatial_;
            uint32_t len = spatial_;

            // ----- load input -----
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();

            // ----- compute mean -----
            AscendC::LocalTensor<float> sumScalar = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumScalar, xLocal, len);
            float mean = sumScalar(0) / (float)len;
            scalarQueue_.FreeTensor(sumScalar);

            // ----- center -----
            AscendC::LocalTensor<float> meanVec = tmpQueue_.AllocTensor<float>();
            AscendC::Duplicate(meanVec, mean, len);
            AscendC::Sub(xLocal, xLocal, meanVec, len);
            tmpQueue_.FreeTensor(meanVec);

            // ----- compute variance (sum of squares) -----
            AscendC::LocalTensor<float> sqBuf = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(sqBuf, xLocal, xLocal, len);
            AscendC::LocalTensor<float> varScalar = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceSum(varScalar, sqBuf, len);
            float variance = varScalar(0) / (float)len;
            scalarQueue_.FreeTensor(varScalar);
            tmpQueue_.FreeTensor(sqBuf);

            // ----- normalize -----
            float denom = variance + eps_;
            float invStd = 1.0f / sqrtf(denom);
            AscendC::Muls(xLocal, xLocal, invStd, len);

            // ----- store output -----
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, xLocal, len);
            AscendC::DataCopy(yGm_[offset], yLocal, len);
            outQueueY_.FreeTensor(yLocal);

            inQueueX_.FreeTensor(xLocal);

            row += blockNum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t spatial_;
    uint32_t totalRows_;
    uint32_t tileLen_;
    float eps_;
};

extern "C" __global__ __aicore__ void ca6k_04593_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04593Custom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.batch, tiling_data.channels,
            tiling_data.depth, tiling_data.height, tiling_data.width,
            tiling_data.eps);
    op.Process();
}
