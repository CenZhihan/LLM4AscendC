kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k03241Custom {
public:
    __aicore__ inline KernelCa6k03241Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t totalCols)
    {
        totalRows_ = totalRows;
        totalCols_ = totalCols;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows * totalCols);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalCols);

        pipe_.InitBuffer(rowBuf_, 1, totalCols_ * sizeof(float));
        pipe_.InitBuffer(maxBuf_, 1, totalCols_ * sizeof(float));
        pipe_.InitBuffer(sumBuf_, 1, totalCols_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, totalCols_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || totalCols_ == 0) return;

        AscendC::LocalTensor<float> maxLocal = maxBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = sumBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> rowLocal = rowBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpBuf_.AllocTensor<float>();

        const float negInf = -3.402823466e+38f;
        AscendC::SetValue(maxLocal, totalCols_, negInf);
        AscendC::SetValue(sumLocal, totalCols_, 0.0f);

        for (uint32_t i = 0; i < totalRows_; i++) {
            AscendC::DataCopy(rowLocal, xGm_[i * totalCols_], totalCols_);
            AscendC::Max(maxLocal, maxLocal, rowLocal, totalCols_);
        }

        for (uint32_t i = 0; i < totalRows_; i++) {
            AscendC::DataCopy(rowLocal, xGm_[i * totalCols_], totalCols_);
            AscendC::Sub(tmpLocal, rowLocal, maxLocal, totalCols_);
            AscendC::Exp(tmpLocal, tmpLocal, totalCols_);
            AscendC::Add(sumLocal, sumLocal, tmpLocal, totalCols_);
        }

        AscendC::Log(tmpLocal, sumLocal, totalCols_);
        AscendC::Add(yGm_, tmpLocal, maxLocal, totalCols_);

        maxBuf_.FreeTensor(maxLocal);
        sumBuf_.FreeTensor(sumLocal);
        rowBuf_.FreeTensor(rowLocal);
        tmpBuf_.FreeTensor(tmpLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> rowBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maxBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t totalCols_ {0};
};

extern "C" __global__ __aicore__ void ca6k_03241_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k03241Custom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.totalCols);
    op.Process();
}
"""
