#include "kernel_operator.h"

class KernelPdist {
public:
    __aicore__ inline KernelPdist() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t d,
                                uint32_t totalPairs, float p)
    {
        n_ = n;
        d_ = d;
        totalPairs_ = totalPairs;
        p_ = p;
        inv_p_ = 1.0f / p;
        xGm_.SetGlobalBuffer((__gm__ float*)x, n * d);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalPairs);

        // Initialize local buffers: row_i, row_j, diff, abs, pow, p_tensor, inv_p_tensor, sum_tensor
        pipe_.InitBuffer(bufRowI_, d_ * sizeof(float));
        pipe_.InitBuffer(bufRowJ_, d_ * sizeof(float));
        pipe_.InitBuffer(bufDiff_, d_ * sizeof(float));
        pipe_.InitBuffer(bufAbs_, d_ * sizeof(float));
        pipe_.InitBuffer(bufPow_, d_ * sizeof(float));
        pipe_.InitBuffer(bufPTensor_, d_ * sizeof(float)); // constant p
        pipe_.InitBuffer(bufInvPTensor_, 1 * sizeof(float)); // constant inv_p
        pipe_.InitBuffer(bufSum_, 1 * sizeof(float));
        pipe_.InitBuffer(bufDist_, 1 * sizeof(float));

        // Fill constant tensor p
        AscendC::LocalTensor<float> pLocal = bufPTensor_.Get<float>();
        AscendC::LocalTensor<float> invPLocal = bufInvPTensor_.Get<float>();
        AscendC::Duplicate(pLocal, p_, d_);
        AscendC::Duplicate(invPLocal, inv_p_, 1);
    }

    __aicore__ inline void Process()
    {
        if (totalPairs_ == 0) return;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t rowsPerBlock = n_ / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = (blockIdx == blockNum - 1) ? n_ : startRow + rowsPerBlock;
        if (startRow >= n_) return;

        for (uint32_t i = startRow; i < endRow; i++) {
            // Load row i
            AscendC::LocalTensor<float> rowI = bufRowI_.Get<float>();
            AscendC::DataCopy(rowI, xGm_[i * d_], d_);

            uint32_t offset_i = i * (2 * n_ - i - 1) / 2;
            for (uint32_t j = i + 1; j < n_; j++) {
                // Load row j
                AscendC::LocalTensor<float> rowJ = bufRowJ_.Get<float>();
                AscendC::DataCopy(rowJ, xGm_[j * d_], d_);

                // Compute diff = rowI - rowJ
                AscendC::LocalTensor<float> diff = bufDiff_.Get<float>();
                AscendC::Sub(diff, rowI, rowJ, d_);

                // Compute abs_diff
                AscendC::LocalTensor<float> absDiff = bufAbs_.Get<float>();
                AscendC::Abs(absDiff, diff, d_);

                // Compute pow(abs_diff, p)
                AscendC::LocalTensor<float> powDiff = bufPow_.Get<float>();
                AscendC::LocalTensor<float> pTensor = bufPTensor_.Get<float>();
                AscendC::Pow(powDiff, absDiff, pTensor, d_);

                // Reduce sum to get sum of powers
                AscendC::LocalTensor<float> sum = bufSum_.Get<float>();
                AscendC::ReduceSum(sum, powDiff, d_);

                // Compute dist = sum^(1/p)
                AscendC::LocalTensor<float> dist = bufDist_.Get<float>();
                AscendC::LocalTensor<float> invPTensor = bufInvPTensor_.Get<float>();
                AscendC::Pow(dist, sum, invPTensor, 1);

                // Store to global
                AscendC::DataCopy(yGm_[offset_i + (j - i - 1)], dist, 1);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufRowI_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufRowJ_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufDiff_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufAbs_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufPow_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufPTensor_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufInvPTensor_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufSum_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufDist_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t n_;
    uint32_t d_;
    uint32_t totalPairs_;
    float p_;
    float inv_p_;
};
