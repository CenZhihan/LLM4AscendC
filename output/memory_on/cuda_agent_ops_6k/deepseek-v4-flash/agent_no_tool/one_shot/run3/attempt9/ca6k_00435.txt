kernel_src = """
#include "kernel_operator.h"

using namespace AscendC;

class KernelCa6k00435Custom {
public:
    __aicore__ inline KernelCa6k00435Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inputN, uint32_t inputC,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t upscaleFactor)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        totalLen_ = totalLength;
        inputN_ = inputN;
        inputC_ = inputC;
        inputH_ = inputH;
        inputW_ = inputW;
        upscaleFactor_ = upscaleFactor;
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        uint32_t r = upscaleFactor_;
        uint32_t outC = inputC_ * r * r;
        uint32_t outH = inputH_ / r;
        uint32_t outW = inputW_ / r;

        // grid-stride loop over output elements
        for (uint32_t idx = blockIdx; idx < totalLen_; idx += blockNum) {
            // compute output coordinates (n, oc, oh, ow)
            uint32_t ow = idx % outW;
            uint32_t rem = idx / outW;
            uint32_t oh = rem % outH;
            rem /= outH;
            uint32_t oc = rem % outC;
            uint32_t n = rem / outC;

            // compute input coordinates
            uint32_t cin = oc / (r * r);
            uint32_t localIdx = oc % (r * r);
            uint32_t i = localIdx / r;  // row offset within block
            uint32_t j = localIdx % r;  // col offset within block

            uint32_t ih = oh * r + i;
            uint32_t iw = ow * r + j;

            // linear index in input
            uint32_t srcIdx = ((n * inputC_ + cin) * inputH_ + ih) * inputW_ + iw;

            // copy scalar value
            yGm_[idx] = xGm_[srcIdx];
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    uint32_t totalLen_;
    uint32_t inputN_;
    uint32_t inputC_;
    uint32_t inputH_;
    uint32_t inputW_;
    uint32_t upscaleFactor_;
};

extern "C" __global__ __aicore__ void ca6k_00435_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00435Custom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.inputN, tiling_data.inputC,
            tiling_data.inputH, tiling_data.inputW,
            tiling_data.upscaleFactor);
    op.Process();
}
"""
