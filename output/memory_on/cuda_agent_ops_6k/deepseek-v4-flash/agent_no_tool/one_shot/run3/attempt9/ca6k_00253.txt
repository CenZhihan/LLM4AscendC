#include "ca6k00253_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00253CustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = shape.GetDim(0);
    uint32_t C = shape.GetDim(1);
    uint32_t H = shape.GetDim(2);
    uint32_t W = shape.GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t convStride = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t convPadding = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t convDilation = attrs->GetAttrPointer<int32_t>(2)[0];
    int32_t convGroups = attrs->GetAttrPointer<int32_t>(3)[0];
    int32_t poolKernel = attrs->GetAttrPointer<int32_t>(4)[0];
    int32_t poolStride = attrs->GetAttrPointer<int32_t>(5)[0];
    int32_t poolPadding = attrs->GetAttrPointer<int32_t>(6)[0];

    const auto &wshape = context->GetInputShape(1)->GetOriginShape();
    uint32_t outC = wshape.GetDim(0);
    uint32_t kernelH = wshape.GetDim(2);
    uint32_t kernelW = wshape.GetDim(3);

    uint32_t outH = (H + 2 * convPadding - convDilation * (kernelH - 1) - 1) / convStride + 1;
    uint32_t outW = (W + 2 * convPadding - convDilation * (kernelW - 1) - 1) / convStride + 1;

    uint32_t poolOutH = (outH + 2 * poolPadding - poolKernel) / poolStride + 1;
    uint32_t poolOutW = (outW + 2 * poolPadding - poolKernel) / poolStride + 1;

    uint32_t totalLength = N * outC * poolOutH * poolOutW;

    // tileLen from UB budget: 5 tile buffers (inQueueX:2, outQueueY:2, tmpQueue:1) -> bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"
#include "ca6k00253_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00253Custom {
public:
    __aicore__ inline KernelCa6k00253Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, totalLen_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // This kernel is a placeholder: it copies input x to output y.
        // In a full implementation, it would perform Conv2d+ReLU+MaxPool.
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        while (true) {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], curLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            // Just copy x to y (no-op)
            AscendC::Add(yLocal, xLocal, 0.0f, curLen);
            tmpQueue_.FreeTensor(tLocal);
            outQueueY_.EnQue<float>(yLocal);
            inQueueX_.FreeTensor(xLocal);

            AscendC::LocalTensor<float> yLocalOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yLocalOut, curLen);
            outQueueY_.FreeTensor(yLocalOut);

            base += strideElems;
            if (base >= totalLen_) break;
            curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00253_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00253Custom op;
    op.Init(x, weight, bias, y, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
'''
