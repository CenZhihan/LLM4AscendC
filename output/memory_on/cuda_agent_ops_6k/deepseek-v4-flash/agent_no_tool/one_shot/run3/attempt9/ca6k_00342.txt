kernel_src = '''
#include "kernel_operator.h"

class KernelCa6k00342Custom {
public:
    __aicore__ inline KernelCa6k00342Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t h_out, uint32_t w_out)
    {
        N_ = n;
        C_ = c;
        H_ = h;
        W_ = w;
        H_out_ = h_out;
        W_out_ = w_out;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * H_out_ * W_out_);

        pipe_.InitBuffer(row0Buf_, 1, W_ * sizeof(float));
        pipe_.InitBuffer(row1Buf_, 1, W_ * sizeof(float));
        pipe_.InitBuffer(outRowBuf_, 1, W_out_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t n = blockIdx / C_;
        uint32_t c = blockIdx % C_;

        if (n >= N_ || c >= C_) return;

        float *x_channel = xGm_.GetPtr() + n * C_ * H_ * W_ + c * H_ * W_;
        float *y_channel = yGm_.GetPtr() + n * C_ * H_out_ * W_out_ + c * H_out_ * W_out_;

        AscendC::LocalTensor<float> row0 = row0Buf_.Get<float>();
        AscendC::LocalTensor<float> row1 = row1Buf_.Get<float>();
        AscendC::LocalTensor<float> outRow = outRowBuf_.Get<float>();

        for (uint32_t i = 0; i < H_out_; i++) {
            AscendC::DataCopy(row0, x_channel + (2 * i) * W_, W_);
            AscendC::DataCopy(row1, x_channel + (2 * i + 1) * W_, W_);

            for (uint32_t j = 0; j < W_out_; j++) {
                float v0 = row0(2 * j);
                float v1 = row0(2 * j + 1);
                float v2 = row1(2 * j);
                float v3 = row1(2 * j + 1);
                float m = v0;
                if (v1 > m) m = v1;
                if (v2 > m) m = v2;
                if (v3 > m) m = v3;
                outRow(j) = m;
            }

            AscendC::DataCopy(y_channel + i * W_out_, outRow, W_out_);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> row0Buf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> row1Buf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outRowBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t H_out_ {0};
    uint32_t W_out_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00342_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00342Custom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.h, tiling_data.w,
            tiling_data.h_out, tiling_data.w_out);
    op.Process();
}
'''
