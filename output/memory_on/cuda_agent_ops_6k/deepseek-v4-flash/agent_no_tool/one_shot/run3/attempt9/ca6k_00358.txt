kernel_src="""

#include "kernel_operator.h"

class KernelCa6k00358Custom {
public:
    __aicore__ inline KernelCa6k00358Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElems,
                                uint32_t inN, uint32_t inC, uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW)
    {
        totalElems_ = totalOutputElems;
        inN_ = inN;
        inC_ = inC;
        inH_ = inH;
        inW_ = inW;
        outH_ = outH;
        outW_ = outW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, inN * inC * inH * inW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElems_);
    }

    __aicore__ inline void Process()
    {
        if (totalElems_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // grid-stride loop over output elements
        for (uint32_t idx = blockIdx; idx < totalElems_; idx += blockNum) {
            // decode indices
            uint32_t tmp = idx;
            uint32_t out_w = tmp % outW_;
            tmp /= outW_;
            uint32_t out_h = tmp % outH_;
            tmp /= outH_;
            uint32_t c = tmp % inC_;
            tmp /= inC_;
            uint32_t n = tmp;

            // compute input window bounds
            uint32_t start_h = (out_h * inH_) / outH_;
            uint32_t end_h = ((out_h + 1) * inH_ + outH_ - 1) / outH_;
            uint32_t start_w = (out_w * inW_) / outW_;
            uint32_t end_w = ((out_w + 1) * inW_ + outW_ - 1) / outW_;

            uint32_t baseOffset = (n * inC_ + c) * inH_ * inW_;
            float maxVal = -INFINITY;

            for (uint32_t ih = start_h; ih < end_h; ++ih) {
                uint32_t rowOffset = baseOffset + ih * inW_;
                for (uint32_t iw = start_w; iw < end_w; ++iw) {
                    float val = xGm_[rowOffset + iw];
                    if (val > maxVal) maxVal = val;
                }
            }

            yGm_[idx] = maxVal;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalElems_ {0};
    uint32_t inN_ {0};
    uint32_t inC_ {0};
    uint32_t inH_ {0};
    uint32_t inW_ {0};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00358_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00358Custom op;
    op.Init(x, y,
            tiling_data.totalOutputElems,
            tiling_data.inN, tiling_data.inC, tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW);
    op.Process();
}
