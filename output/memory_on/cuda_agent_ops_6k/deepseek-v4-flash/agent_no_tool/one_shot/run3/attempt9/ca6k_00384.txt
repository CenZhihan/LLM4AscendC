#include "ca6k00384_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00384CustomTilingData tiling;

    const gert::StorageShape* storageShape = context->GetInputShape(0);
    uint32_t totalLength = storageShape->GetOriginShape().GetShapeSize();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);

    uint32_t maxTile = UB_BUDGET_BYTES / (7 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}  // namespace optiling

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00384Custom {
public:
    __aicore__ inline KernelCa6k00384Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mean, GM_ADDR var,
                                GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength, float eps) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        meanGm_.SetGlobalBuffer((__gm__ float*)mean, totalLen_);
        varGm_.SetGlobalBuffer((__gm__ float*)var, totalLen_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, totalLen_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMean_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueVar_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueWeight_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueBias_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;

        for (uint32_t base = blockIdx * tileLen_; base < totalLen_; base += strideElems) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            CopyIn(base, curLen);
            Compute(curLen);
            CopyOut(base, curLen);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> meanLocal = inQueueMean_.AllocTensor<float>();
        AscendC::LocalTensor<float> varLocal = inQueueVar_.AllocTensor<float>();
        AscendC::LocalTensor<float> weightLocal = inQueueWeight_.AllocTensor<float>();
        AscendC::LocalTensor<float> biasLocal = inQueueBias_.AllocTensor<float>();

        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        AscendC::DataCopy(meanLocal, meanGm_[gmOffset], len);
        AscendC::DataCopy(varLocal, varGm_[gmOffset], len);
        AscendC::DataCopy(weightLocal, weightGm_[gmOffset], len);
        AscendC::DataCopy(biasLocal, biasGm_[gmOffset], len);

        inQueueX_.EnQue(xLocal);
        inQueueMean_.EnQue(meanLocal);
        inQueueVar_.EnQue(varLocal);
        inQueueWeight_.EnQue(weightLocal);
        inQueueBias_.EnQue(biasLocal);
    }

    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> meanLocal = inQueueMean_.DeQue<float>();
        AscendC::LocalTensor<float> varLocal = inQueueVar_.DeQue<float>();
        AscendC::LocalTensor<float> weightLocal = inQueueWeight_.DeQue<float>();
        AscendC::LocalTensor<float> biasLocal = inQueueBias_.DeQue<float>();

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Compute (x - mean)
        AscendC::Sub(yLocal, xLocal, meanLocal, len);
        // Compute rsqrt(var + eps)
        AscendC::Adds(tLocal, varLocal, eps_, len);
        AscendC::Rsqrt(tLocal, tLocal, len);
        // y = (x - mean) * rsqrt(var + eps)
        AscendC::Mul(yLocal, yLocal, tLocal, len);
        // y = y * weight
        AscendC::Mul(yLocal, yLocal, weightLocal, len);
        // y = y + bias
        AscendC::Add(yLocal, yLocal, biasLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);

        inQueueX_.FreeTensor(xLocal);
        inQueueMean_.FreeTensor(meanLocal);
        inQueueVar_.FreeTensor(varLocal);
        inQueueWeight_.FreeTensor(weightLocal);
        inQueueBias_.FreeTensor(biasLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMean_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueVar_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWeight_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBias_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> meanGm_;
    AscendC::GlobalTensor<float> varGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    float eps_{1e-5};
};

extern "C" __global__ __aicore__ void ca6k00384_custom(GM_ADDR x, GM_ADDR mean, GM_ADDR var,
                                                        GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00384Custom op;
    op.Init(x, mean, var, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.eps);
    op.Process();
}

model_src = '''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

_FN = 'ca6k00384_custom'

class ModelNew(nn.Module):
    def __init__(self, norm_type: str, num_channels: int):
        super().__init__()
        self.norm_type = norm_type
        self.num_channels = num_channels
        if norm_type == "batch":
            self.norm = nn.BatchNorm2d(num_channels)
        elif norm_type == "instance":
            self.norm = nn.InstanceNorm2d(num_channels)
        elif norm_type == "layer":
            self.norm = nn.LayerNorm(num_channels)
        else:
            raise ValueError("Invalid normalization type")
        # store reference to affine parameters and eps
        self.eps = self.norm.eps if hasattr(self.norm, 'eps') else 1e-5

    def forward(self, x):
        norm = self.norm
        # compute mean and var exactly as the reference module would
        if self.norm_type == "batch":
            if norm.training:
                # use batch statistics
                mean = torch.mean(x, dim=[0,2,3], keepdim=True)
                var = torch.var(x, dim=[0,2,3], keepdim=True, unbiased=False)
            else:
                # use running statistics
                mean = norm.running_mean.view(1, -1, 1, 1)
                var = norm.running_var.view(1, -1, 1, 1)
        elif self.norm_type == "instance":
            mean = torch.mean(x, dim=[2,3], keepdim=True)
            var = torch.var(x, dim=[2,3], keepdim=True, unbiased=False)
        elif self.norm_type == "layer":
            mean = torch.mean(x, dim=[1,2,3], keepdim=True)
            var = torch.var(x, dim=[1,2,3], keepdim=True, unbiased=False)
        else:
            raise ValueError("Invalid norm_type")

        # expand mean, var, weight, bias to full shape
        mean = mean.expand_as(x)
        var = var.expand_as(x)

        if hasattr(norm, 'weight') and norm.weight is not None:
            weight = norm.weight
            if self.norm_type == "layer":
                weight = weight.view(1, -1, 1, 1)
            else:
                weight = weight.view(1, -1, 1, 1)
            weight = weight.expand_as(x)
        else:
            weight = torch.ones_like(x)

        if hasattr(norm, 'bias') and norm.bias is not None:
            bias = norm.bias
            if self.norm_type == "layer":
                bias = bias.view(1, -1, 1, 1)
            else:
                bias = bias.view(1, -1, 1, 1)
            bias = bias.expand_as(x)
        else:
            bias = torch.zeros_like(x)

        fn = getattr(custom_ops_lib, _FN)
        return fn(x, mean, var, weight, bias, self.eps)
'''
