constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00477Custom {
public:
    __aicore__ inline KernelCa6k00477Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t numSlices, uint32_t spatialSize,
                                float spatialSizeFloat, float eps)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        numSlices_ = numSlices;
        spatialSize_ = spatialSize;
        spatialSizeFloat_ = spatialSizeFloat;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(mulBuf_, 1, tileLen_ * sizeof(float));   // for elementwise multiply
        pipe_.InitBuffer(redBuf_, 1, sizeof(float));              // scalar reduction result
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t slice = blockIdx; slice < numSlices_; slice += blockNum) {
            uint32_t sliceOffset = slice * spatialSize_;

            // ----- first pass: compute sum and sum of squares -----
            float sum_x = 0.0f;
            float sum_x2 = 0.0f;
            uint32_t remaining = spatialSize_;
            uint32_t base = sliceOffset;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base], curLen);

                // sum of x
                AscendC::LocalTensor<float> red = redBuf_.AllocTensor<float>();
                AscendC::VecReduceSum(red, xLocal, curLen);
                sum_x += red(0);
                redBuf_.FreeTensor(red);

                // sum of x^2
                AscendC::LocalTensor<float> mul = mulBuf_.AllocTensor<float>();
                AscendC::Mul(mul, xLocal, xLocal, curLen);
                red = redBuf_.AllocTensor<float>();
                AscendC::VecReduceSum(red, mul, curLen);
                sum_x2 += red(0);
                redBuf_.FreeTensor(red);
                mulBuf_.FreeTensor(mul);

                inQueueX_.FreeTensor(xLocal);
                base += curLen;
                remaining -= curLen;
            }

            // compute mean and invStd
