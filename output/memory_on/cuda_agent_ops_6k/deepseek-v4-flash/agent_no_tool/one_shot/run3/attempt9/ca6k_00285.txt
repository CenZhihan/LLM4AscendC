kernel_src="""

#include "kernel_operator.h"
using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00285Custom {
public:
    __aicore__ inline KernelCa6k00285Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C_in, uint32_t H_in, uint32_t W_in,
                                uint32_t factor, uint32_t totalLength, uint32_t tileLength)
    {
        N_ = N;
        C_in_ = C_in;
        H_in_ = H_in;
        W_in_ = W_in;
        factor_ = factor;
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        C_out_ = C_in_ * factor_ * factor_;
        H_out_ = H_in_ / factor_;
        W_out_ = W_in_ / factor_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Single-core processing
        uint32_t cnt = 0;
        // Process the entire tensor tile by tile using double buffer
        uint32_t remaining = totalLen_;
        uint32_t offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            CopyIn(offset, curLen);
            Compute(offset, curLen);
            CopyOut(offset, curLen);
            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // For each element in the tile, compute input -> output mapping
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t globalIdx = gmOffset + i;
            // Decode globalIdx into input coordinates (n, c, h, w)
            uint32_t n = globalIdx / (C_in_ * H_in_ * W_in_);
            uint32_t rest = globalIdx % (C_in_ * H_in_ * W_in_);
            uint32_t c = rest / (H_in_ * W_in_);
            rest = rest % (H_in_ * W_in_);
            uint32_t h = rest / W_in_;
            uint32_t w = rest % W_in_;

            // Compute output coordinates
            uint32_t out_c = c * factor_ * factor_ + (h % factor_) * factor_ + (w % factor_);
            uint32_t out_h = h / factor_;
            uint32_t out_w = w / factor_;

            uint32_t outIdx = ((n * C_out_ + out_c) * H_out_ + out_h) * W_out_ + out_w;
            // Write the value to the correct output location
            yLocal.SetValue(i, xLocal.GetValue(i));
            // But we need to scatter; we cannot directly set yLocal at outIdx because local tensor is sequential.
            // Instead, we'll use global output tensor for scatter.
            // This approach uses local tensor only to accumulate, then write to global.
            yGm_.SetValue(outIdx, xLocal.GetValue(i));
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        // We already wrote to global in Compute, so we just release the local tensor
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_;
    uint32_t C_in_;
    uint32_t H_in_;
    uint32_t W_in_;
    uint32_t C_out_;
    uint32_t H_out_;
    uint32_t W_out_;
    uint32_t factor_;
    uint32_t totalLen_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void ca6k_00285_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00285Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C_in, tiling_data.H_in, tiling_data.W_in,
            tiling_data.factor, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
