#include "ca6k00397_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00397CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *lambd = attrs->GetAttrPointer<float>(0);

    // UB budget: 5 floats per element (2 in, 2 out, 1 tmp) => 20 bytes per element
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    // Direct assignment to public fields (no setter methods)
    tiling.totalLength = totalLength;
    tiling.tileLength = tileLen;
    tiling.blockDim = blockDim;
    tiling.lambd = *lambd;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00397Custom {
public:
    __aicore__ inline KernelCa6k00397Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                float lambd)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        lambd_ = lambd;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Tile-granularity grid-stride
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Prefetch first tile
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> signLocal = tmpQueue_.AllocTensor<float>(); // reuse same tmp queue? We'll use separate allocation; but we only have one tmp queue. We'll allocate two temporary tensors from the same queue? Actually tmpQueue_ has one buffer, cannot allocate two simultaneously. So we need a different approach: use only one temporary tensor by computing in steps.
        // Better: reuse the same tLocal for intermediate results.
        // We'll compute: abs_x = abs(x), sub = abs_x - lambd, max_sub = max(sub, 0), sign = (x > 0) ? 1 : (x < 0) ? -1 : 0, then y = max_sub * sign.

        // Step 1: abs_x = |x|
        AscendC::Abs(tLocal, xLocal, len);  // tLocal = abs(x)
        // Step 2: sub = abs_x - lambd
        AscendC::Subs(tLocal, tLocal, lambd_, len); // tLocal = abs(x) - lambd
        // Step 3: max_sub = max(sub, 0)
        AscendC::Maxs(tLocal, tLocal, 0.0f, len); // tLocal = max(abs(x)-lambd, 0)

        // Step 4: sign = (x > 0) - (x < 0)
        // We need two temp tensors: we only have one previous tLocal. But we can reuse the input xLocal now?
        // Actually after step 3, we can reuse xLocal since we no longer need it. But careful: xLocal is still needed for sign? We have xLocal tensor from dequeue. We can compute sign using xLocal and store in a temporary from tmpQueue_. But tmpQueue_ only has one buffer. However, we can allocate signLocal from the same pipe by using a different queue? Let's simplify: Use two buffers from inQueueX and outQueueY? Not recommended. Alternative: compute sign using Mins and Muls without extra tensor? 
        // We'll use a second temporary tensor from a second tmp queue? We can add another tmp queue in the class, but that increases UB usage. Simpler: compute y in two steps using only one temporary.
        // Approach: compute pos = x > 0? AscendC has Compare with mask? Unclear.
        // We'll implement a simpler formula: y = max(x - lambd, 0) + min(x + lambd, 0). This avoids sign.
        // Then we only need two temporary tensors: one for max part, one for min part? Actually we can compute y in place using xLocal and one tmp.
        // Let's use: 
        //   tmp = x + lambd, tmp = min(tmp, 0)  // negative part
        //   y = x - lambd, y = max(y, 0)  // positive part
        //   y = y + tmp
        // But we need two tmp tensors? We can compute positive part into yLocal, negative part into tLocal, then add.
        // That requires one tmp. Good.

        // Compute positive part: yLocal = x - lambd, then max with 0
        AscendC::Subs(yLocal, xLocal, lambd_, len); // y = x - lambd
        AscendC::Maxs(yLocal, yLocal, 0.0f, len); // y = max(x - lambd, 0)

        // Compute negative part: tLocal = x + lambd, then min with 0
        AscendC::Adds(tLocal, xLocal, lambd_, len); // t = x + lambd
        AscendC::Mins(tLocal, tLocal, 0.0f, len); // t = min(x + lambd, 0)

        // Add
        AscendC::Add(yLocal, yLocal, tLocal, len); // y = max(x-lambd,0) + min(x+lambd,0)

        // Free temporary and enqueue output
        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    float lambd_ {0.5f};
};

extern "C" __global__ __aicore__ void ca6k00397_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00397Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.lambd);
    op.Process();
}
