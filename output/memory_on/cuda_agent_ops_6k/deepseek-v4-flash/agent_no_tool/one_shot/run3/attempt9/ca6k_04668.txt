#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04668Custom {
public:
    __aicore__ inline KernelCa6k04668Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElems, uint32_t tileLen,
                                uint32_t blockDim,
                                uint32_t inputN, uint32_t inputC,
                                uint32_t inputD, uint32_t inputH, uint32_t inputW,
                                uint32_t outputD, uint32_t outputH, uint32_t outputW)
    {
        totalOutputElems_ = totalOutputElems;
        tileLen_ = tileLen;
        blockDim_ = blockDim;
        inputN_ = inputN;
        inputC_ = inputC;
        inputD_ = inputD;
        inputH_ = inputH;
        inputW_ = inputW;
        outputD_ = outputD;
        outputH_ = outputH;
        outputW_ = outputW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, inputN * inputC * inputD * inputH * inputW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElems);

        uint32_t maxSliceSize = inputW * inputH;
        uint32_t ubSize = maxSliceSize * sizeof(float) + sizeof(float);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, ubSize);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0) return;

        for (uint32_t idx = 0; idx < totalOutputElems_; ++idx) {
            uint32_t od = (idx / (outputH_ * outputW_)) % outputD_;
            uint32_t oh = (idx / outputW_) % outputH_;
            uint32_t ow = idx % outputW_;
            uint32_t c = (idx / (outputD_ * outputH_ * outputW_)) % inputC_;
            uint32_t n = idx / (inputC_ * outputD_ * outputH_ * outputW_);

            uint32_t startD = (od * inputD_) / outputD_;
            uint32_t endD = ((od + 1) * inputD_ + outputD_ - 1) / outputD_;
            if (endD > inputD_) endD = inputD_;
            uint32_t startH = (oh * inputH_) / outputH_;
            uint32_t endH = ((oh + 1) * inputH_ + outputH_ - 1) / outputH_;
            if (endH > inputH_) endH = inputH_;
            uint32_t startW = (ow * inputW_) / outputW_;
            uint32_t endW = ((ow + 1) * inputW_ + outputW_ - 1) / outputW_;
            if (endW > inputW_) endW = inputW_;

            float maxVal = -3.402823e+38f;
            for (uint32_t d = startD; d < endD; ++d) {
                uint32_t sliceOffset = n * inputC_ * inputD_ * inputH_ * inputW_
                                      + c * inputD_ * inputH_ * inputW_
                                      + d * inputH_ * inputW_;
                uint32_t sliceSize = inputH_ * inputW_;
                AscendC::LocalTensor<float> sliceLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(sliceLocal, xGm_[sliceOffset], sliceSize);
                for (uint32_t h = startH; h < endH; ++h) {
                    for (uint32_t w = startW; w < endW; ++w) {
                        float val = sliceLocal(h * inputW_ + w);
                        if (val > maxVal) maxVal = val;
                    }
                }
                inQueueX_.FreeTensor(sliceLocal);
            }

            AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();
            outLocal(0) = maxVal;
            AscendC::DataCopy(yGm_[idx], outLocal, 1);
            outQueueY_.FreeTensor(outLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutputElems_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
    uint32_t inputN_ {0};
    uint32_t inputC_ {0};
    uint32_t inputD_ {0};
    uint32_t inputH_ {0};
    uint32_t inputW_ {0};
    uint32_t outputD_ {0};
    uint32_t outputH_ {0};
    uint32_t outputW_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04668_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04668Custom op;
    op.Init(x, y,
            tiling_data.totalOutputElems, tiling_data.tileLen, tiling_data.blockDim,
            tiling_data.inputN, tiling_data.inputC,
            tiling_data.inputD, tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputD, tiling_data.outputH, tiling_data.outputW);
    op.Process();
}
