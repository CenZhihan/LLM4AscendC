kernel_src = """
#include "kernel_operator.h"
constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04603Custom {
public:
    __aicore__ inline KernelCa6k04603Custom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t outC, uint32_t kernelH, uint32_t kernelW,
                                uint32_t stride, uint32_t padding, uint32_t tileSize)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        outC_ = outC; kernelH_ = kernelH; kernelW_ = kernelW;
        stride_ = stride; padding_ = padding; tileSize_ = tileSize;
        xGm_.SetGlobalBuffer((__gm__ float*)x, N*C*H*W);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outC*C*kernelH*kernelW);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, outC);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N*outC*H*W);
        uint32_t inTile = tileSize * C * H * W;
        uint32_t wTile = outC * C * kernelH * kernelW;
        uint32_t outTile = tileSize * outC * H * W;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inTile * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, wTile * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, outC * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outTile * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, outTile * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideBlocks = blockNum * tileSize_;
        uint32_t base = blockIdx * tileSize_;
        if (base >= N_) return;
        while (true) {
            uint32_t curN = N_ - base;
            if (curN > tileSize_) curN = tileSize_;
            // For simplicity, we load entire weight and bias once (they are small)
            // and iterate over batches.
            // Prefetch weight and bias for first block (shared across all blocks)
            // Actually, weight and bias are constant; we can load once.
            // But each block needs its own local copy.
            // We'll load them inside the loop for simplicity.
            CopyIn(base, curN);
            Compute(curN);
            CopyOut(base, curN);
            base += strideBlocks;
            if (base >= N_) break;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t batchOffset, uint32_t curN)
    {
        // Load input patch (batch portion)
        uint32_t inSize = curN * C_ * H_ * W_;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[batchOffset * C_ * H_ * W_], inSize);
        inQueueX_.EnQue(xLocal);

        // Load weight (full kernel)
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, outC_ * C_ * kernelH_ * kernelW_);
        inQueueW_.EnQue(wLocal);

        // Load bias
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        AscendC::DataCopy(bLocal, bGm_, outC_);
        inQueueB_.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t curN)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Perform convolution: y = conv(x, w) + bias, then relu
        // AscendC provides Conv2D: output shape [curN, outC, H, W]
        AscendC::Conv2D(tLocal, xLocal, wLocal, curN, C_, H_, W_, outC_, kernelH_, kernelW_, stride_, padding_);
        // Add bias: broadcast along batch and spatial dims
        AscendC::Add(tLocal, tLocal, bLocal, curN * outC_ * H_ * W_);
        // ReLU
        AscendC::Relu(yLocal, tLocal, curN * outC_ * H_ * W_);

        inQueueX_.FreeTensor(xLocal);
        inQueueW_.FreeTensor(wLocal);
        inQueueB_.FreeTensor(bLocal);
        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t batchOffset, uint32_t curN)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint32_t outSize = curN * outC_ * H_ * W_;
        AscendC::DataCopy(yGm_[batchOffset * outC_ * H_ * W_], yLocal, outSize);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t N_, C_, H_, W_, outC_, kernelH_, kernelW_, stride_, padding_, tileSize_;
};

extern "C" __global__ __aicore__ void ca6k_04603_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04603Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.outC, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.stride, tiling_data.padding, tiling_data.tileSize);
    op.Process();
}
"""

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = "ca6k_04603_custom"

class ModelNew(torch.nn.Module):
    def __init__(self, module_dict):
        super(ModelNew, self).__init__()
        self.conv1_weight = module_dict["conv1"].weight.npu().contiguous()
        self.conv1_bias = module_dict["conv1"].bias.npu().contiguous()
        self.conv2_weight = module_dict["conv2"].weight.npu().contiguous()
        self.conv2_bias = module_dict["conv2"].bias.npu().contiguous()
    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        h = fn(x.npu(), self.conv1_weight, self.conv1_bias, 1, 1, 3)
        out = fn(h, self.conv2_weight, self.conv2_bias, 1, 1, 3)
        return out
'''
