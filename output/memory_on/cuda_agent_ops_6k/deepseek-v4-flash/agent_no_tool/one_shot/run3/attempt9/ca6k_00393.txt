#include "ca6k00393_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00393CustomTilingData tiling;
    // Input shapes: log_prob [batch, numClasses], target [batch]
    const gert::StorageShape *logProbShape = context->GetInputShape(0);
    const auto dims = logProbShape->GetOriginShape().GetDimNum();
    uint32_t batch_size = 1;
    uint32_t num_classes = 1;
    if (dims >= 2) {
        batch_size = logProbShape->GetOriginShape().GetDim(0);
        num_classes = logProbShape->GetOriginShape().GetDim(1);
    } else if (dims == 1) {
        batch_size = logProbShape->GetOriginShape().GetDim(0);
    }

    uint32_t block_dim = 1;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);  // for target reads only, but safe
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tile_len = maxTile;
    if (batch_size < tile_len) {
        tile_len = batch_size;
        uint32_t aligned = AlignDown(tile_len, ALIGN_ELEMS);
        if (aligned > 0) tile_len = aligned;
    }

    context->SetBlockDim(block_dim);

    tiling.set_batch_size(batch_size);
    tiling.set_num_classes(num_classes);
    tiling.set_tile_len(tile_len);
    tiling.set_block_dim(block_dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
