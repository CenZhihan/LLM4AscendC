host_operator_src = """

#include "ca6k_00265_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00265CustomTilingData tiling;

    const gert::StorageShape *x_shape_storage = context->GetInputShape(0);
    const auto &x_shape = x_shape_storage->GetOriginShape();
    int64_t batch = x_shape.GetDim(0);
    int64_t channel = x_shape.GetDim(1);
    int64_t H = x_shape.GetDim(2);
    int64_t W = x_shape.GetDim(3);
    uint32_t spatialSize = static_cast<uint32_t>(H * W);
    uint32_t channelSize = static_cast<uint32_t>(channel);
    uint32_t totalLength = static_cast<uint32_t>(batch * channel * spatialSize);

    // x, t1, t2, y in UB => 4 * tileLen floats => 16 * tileLen bytes
    uint32_t maxTile = UB_BUDGET_BYTES / 16;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_channelSize(channelSize);
    tiling.set_spatialSize(spatialSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

}  // namespace ge

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00265Custom {
public:
    __aicore__ inline KernelCa6k00265Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR tensor1, GM_ADDR tensor2,
                                GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t channelSize, uint32_t spatialSize) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        channelSize_ = channelSize;
        spatialSize_ = spatialSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        t1Gm_.SetGlobalBuffer((__gm__ float*)tensor1, totalLen_);
        t2Gm_.SetGlobalBuffer((__gm__ float*)tensor2, totalLen_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, channelSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueT1_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueT2_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(biasBuf_, 1, channelSize_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Load bias into local
        AscendC::LocalTensor<float> biasLocal = biasBuf_.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, biasGm_, channelSize_);

        // Grid-stride loop over tiles
        for (uint32_t base = blockIdx * tileLen_; base < totalLen_; base += blockNum * tileLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Allocate and copy inputs
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], curLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> t1Local = inQueueT1_.AllocTensor<float>();
            AscendC::DataCopy(t1Local, t1Gm_[base], curLen);
            inQueueT1_.EnQue(t1Local);

            AscendC::LocalTensor<float> t2Local = inQueueT2_.AllocTensor<float>();
            AscendC::DataCopy(t2Local, t2Gm_[base], curLen);
            inQueueT2_.EnQue(t2Local);

            // Dequeue input tensors
            xLocal = inQueueX_.DeQue<float>();
            t1Local = inQueueT1_.DeQue<float>();
            t2Local = inQueueT2_.DeQue<float>();

            // Compute: y = x + t1/t2 + bias (per-channel broadcast)
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            for (uint32_t i = 0; i < curLen; i++) {
                float xVal = xLocal(i);
                float t1Val = t1Local(i);
                float t2Val = t2Local(i);
                uint32_t linearIdx = base + i;
                uint32_t c = (linearIdx / spatialSize_) % channelSize_;
                float biasVal = biasLocal(c);
                yLocal(i) = xVal + (t1Val / t2Val) + biasVal;
            }

            // Free input tensors
            inQueueX_.FreeTensor(xLocal);
            inQueueT1_.FreeTensor(t1Local);
            inQueueT2_.FreeTensor(t2Local);

            // Enqueue output and copy to global
            outQueueY_.EnQue(yLocal);
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);
        }

        biasBuf_.FreeTensor(biasLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueT1_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueT2_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> biasBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> t1Gm_;
    AscendC::GlobalTensor<float> t2Gm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ = 0;
    uint32_t tileLen_ = 0;
    uint32_t channelSize_ = 0;
    uint32_t spatialSize_ = 0;
};

extern "C" __global__ __aicore__ void ca6k_00265_custom(
    GM_ADDR x, GM_ADDR tensor1, GM_ADDR tensor2, GM_ADDR bias, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00265Custom op;
    op.Init(x, tensor1, tensor2, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.channelSize, tiling_data.spatialSize);
    op.Process();
}
