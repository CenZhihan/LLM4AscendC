static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00935CustomTilingData tiling;

    const gert::StorageShape* xShape = context->GetInputShape(0);
    const gert::StorageShape* wShape = context->GetInputShape(1);

    uint32_t batch = xShape->GetOriginShape().GetDim(0);
    uint32_t inChannels = xShape->GetOriginShape().GetDim(1);
    uint32_t inLength = xShape->GetOriginShape().GetDim(2);
    uint32_t outChannels = wShape->GetOriginShape().GetDim(1);
    uint32_t kernelSize = wShape->GetOriginShape().GetDim(2);

    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    uint32_t stride = *attrs->GetAttrPointer<uint32_t>(0);
    int32_t padding = *attrs->GetAttrPointer<int32_t>(1);
    uint32_t outputPadding = *attrs->GetAttrPointer<uint32_t>(2);
    uint32_t outLength = (inLength - 1) * stride - 2 * padding + kernelSize + outputPadding;

    bool hasBias = false;
    if (context->GetInputShape(2) != nullptr) {
        const gert::StorageShape* biasShape = context->GetInputShape(2);
        hasBias = (biasShape->GetOriginShape().GetDimNum() > 0);
    }

    uint32_t totalCols = outLength;
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * (inChannels * kernelSize + outChannels));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    uint32_t tileLen = maxTile;
    if (totalCols < tileLen) {
        tileLen = AlignDown(totalCols, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalCols;
    }

    uint32_t numTiles = CeilDiv(totalCols, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    // Use setter methods (macro generates set_*)
    tiling.set_batch(batch);
    tiling.set_inChannels(inChannels);
    tiling.set_inLength(inLength);
    tiling.set_outChannels(outChannels);
    tiling.set_outLength(outLength);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_outputPadding(outputPadding);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1 : 0);

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00935Custom {
public:
    __aicore__ inline KernelCa6k00935Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t batch, uint32_t inChannels, uint32_t inLength,
                                uint32_t outChannels, uint32_t outLength,
                                uint32_t kernelSize, uint32_t stride,
                                int32_t padding, uint32_t outputPadding,
                                uint32_t tileLength, bool hasBias)
    {
        batch_ = batch;
        inChannels_ = inChannels;
        inLength_ = inLength;
        outChannels_ = outChannels;
        outLength_ = outLength;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        outputPadding_ = outputPadding;
        tileLength_ = tileLength;
        hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * inChannels * inLength);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, inChannels * outChannels * kernelSize);
        if (hasBias) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, outChannels);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * outChannels * outLength);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * inChannels_ * kernelSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * outChannels_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLength_ * inChannels_ * kernelSize_ * sizeof(float));
        pipe_.InitBuffer(wLocalBuf_, 1, inChannels_ * outChannels_ * kernelSize_ * sizeof(float));
        if (hasBias_)
            pipe_.InitBuffer(biasLocalBuf_, 1, outChannels_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outLength_ == 0 || tileLength_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t colStride = blockNum * tileLength_;
        uint32_t colStart = blockIdx * tileLength_;
        if (colStart >= outLength_) return;

        AscendC::LocalTensor<float> wLocal = wLocalBuf_.Get<float>();
        AscendC::DataCopy(wLocal, wGm_, inChannels_ * outChannels_ * kernelSize_);

        AscendC::LocalTensor<float> biasLocal;
        if (hasBias_) {
            biasLocal = biasLocalBuf_.Get<float>();
            AscendC::DataCopy(biasLocal, biasGm_, outChannels_);
        }

        for (uint32_t b = 0; b < batch_; ++b) {
            uint32_t batchOffset = b * inChannels_ * inLength_;
            uint32_t outBatchOffset = b * outChannels_ * outLength_;

            uint32_t colCur = colStart;
            while (colCur < outLength_) {
                uint32_t tileLen = tileLength_;
                if (colCur + tileLen > outLength_) tileLen = outLength_ - colCur;

                AscendC::LocalTensor<float> colLocal = inQueueX_.AllocTensor<float>();
                BuildIm2Col(colLocal, xGm_[batchOffset], inChannels_, inLength_, kernelSize_,
                            stride_, padding_, colCur, tileLen);

                AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();
                PerformMatMul(outLocal, wLocal, colLocal, outChannels_, tileLen, inChannels_ * kernelSize_);

                if (hasBias_) {
                    for (uint32_t oc = 0; oc < outChannels_; ++oc) {
                        float biasVal = biasLocal[oc];
                        for (uint32_t n = 0; n < tileLen; ++n) {
                            outLocal[oc * tileLen + n] += biasVal;
                        }
                    }
                }

                AscendC::DataCopy(yGm_[outBatchOffset + colCur * outChannels_], outLocal, tileLen * outChannels_);
                outQueueY_.FreeTensor(outLocal);
                inQueueX_.FreeTensor(colLocal);

                colCur += colStride;
            }
        }
    }

private:
    __aicore__ inline void BuildIm2Col(AscendC::LocalTensor<float>& col,
                                       AscendC::GlobalTensor<float>& input,
                                       uint32_t C, uint32_t L,
                                       uint32_t K, uint32_t S, int32_t P,
                                       uint32_t colStart, uint32_t colLen)
    {
        for (uint32_t i = 0; i < colLen; ++i) {
            uint32_t outPos = colStart + i;
            int32_t inStart = static_cast<int32_t>(outPos) * static_cast<int32_t>(S) - P;
            for (uint32_t c = 0; c < C; ++c) {
                for (uint32_t k = 0; k < K; ++k) {
                    int32_t inIdx = inStart + static_cast<int32_t>(k);
                    float val = 0.0f;
                    if (inIdx >= 0 && inIdx < static_cast<int32_t>(L)) {
                        val = input[c * L + inIdx];
                    }
                    col[i * (C * K) + c * K + k] = val;
                }
            }
        }
    }

    __aicore__ inline void PerformMatMul(AscendC::LocalTensor<float>& out,
                                         AscendC::LocalTensor<float>& weight,
                                         AscendC::LocalTensor<float>& col,
                                         uint32_t M, uint32_t N, uint32_t K)
    {
        for (uint32_t m = 0; m < M; ++m) {
            for (uint32_t n = 0; n < N; ++n) {
                float sum = 0.0f;
                for (uint32_t k = 0; k < K; ++k) {
                    sum += weight[k * M + m] * col[n * K + k];
                }
                out[m * N + n] = sum;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> wLocalBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> biasLocalBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t inChannels_;
    uint32_t inLength_;
    uint32_t outChannels_;
    uint32_t outLength_;
    uint32_t kernelSize_;
    uint32_t stride_;
    int32_t padding_;
    uint32_t outputPadding_;
    uint32_t tileLength_;
    bool hasBias_;
};

extern "C" __global__ __aicore__ void ca6k_00935_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                                       GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00935Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.get_batch(), tiling_data.get_inChannels(), tiling_data.get_inLength(),
            tiling_data.get_outChannels(), tiling_data.get_outLength(),
            tiling_data.get_kernelSize(), tiling_data.get_stride(),
            tiling_data.get_padding(), tiling_data.get_outputPadding(),
            tiling_data.get_tileLength(), tiling_data.get_hasBias() != 0);
    op.Process();
}

import torch.nn as nn

_FN = 'ca6k_00935_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0, output_padding=0, bias=False):
        super(ModelNew, self).__init__()
        # Initialize weight and bias exactly as ConvTranspose1d default
        self.weight = nn.Parameter(torch.empty(in_channels, out_channels, kernel_size))
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        if bias:
            self.bias = nn.Parameter(torch.zeros(out_channels))
        else:
            self.bias = None
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.dilation = 1
        self.groups = 1

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        if self.bias is not None:
            return fn(x, self.weight, self.bias, self.stride, self.padding, self.output_padding, self.dilation)
        else:
            return fn(x, self.weight, None, self.stride, self.padding, self.output_padding, self.dilation)
