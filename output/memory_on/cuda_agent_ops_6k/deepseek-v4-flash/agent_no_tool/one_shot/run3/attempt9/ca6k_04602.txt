#include "ca6k_04602_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"

namespace optiling {

static constexpr uint32_t UB_SIZE_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 32;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    auto shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = shape.GetDim(0);
    uint32_t C = shape.GetDim(1);
    uint32_t H = shape.GetDim(2);
    uint32_t W = shape.GetDim(3);

    int32_t kernelSize = 3;
    int32_t stride = 1;
    int32_t padding = 1;

    uint32_t outH = (H + 2 * padding - kernelSize) / stride + 1;
    uint32_t outW = (W + 2 * padding - kernelSize) / stride + 1;

    uint32_t tileC = C;
    uint32_t tileH = H;
    uint32_t tileW = W;
    if (tileH * tileW * sizeof(float) > UB_SIZE_BYTES / 2) {
        tileH = 128;
        tileW = 128;
    }
    tileH = (tileH / ALIGN) * ALIGN;
    tileW = (tileW / ALIGN) * ALIGN;
    if (tileH == 0) tileH = ALIGN;
    if (tileW == 0) tileW = ALIGN;

    uint32_t totalOutputElems = N * C * outH * outW;
    uint32_t blockDim = (totalOutputElems + 1023) / 1024;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;
    context->SetBlockDim(blockDim);

    Ca6k04602CustomTilingData tiling;
    tiling.N = N;
    tiling.C = C;
    tiling.H = H;
    tiling.W = W;
    tiling.tileC = tileC;
    tiling.tileH = tileH;
    tiling.tileW = tileW;
    tiling.blockDim = blockDim;
    tiling.kernelSize = kernelSize;
    tiling.stride = stride;
    tiling.padding = padding;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include <float.h>

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04602Custom {
public:
    __aicore__ inline KernelCa6k04602Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t tileC, uint32_t tileH, uint32_t tileW,
                                int32_t kernelSize, int32_t stride, int32_t padding)
    {
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        tileC_ = tileC;
        tileH_ = tileH;
        tileW_ = tileW;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * H * W);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C * H * W);
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || H_ == 0 || W_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t tilesN = N_;
        uint32_t tilesC = (C_ + tileC_ - 1) / tileC_;
        uint32_t tilesH = (H_ + tileH_ - 1) / tileH_;
        uint32_t tilesW = (W_ + tileW_ - 1) / tileW_;
        uint32_t totalTiles = tilesN * tilesC * tilesH * tilesW;
        uint32_t strideTiles = blockNum;
        uint32_t tileIdx = blockIdx;

        while (tileIdx < totalTiles) {
            uint32_t idx = tileIdx;
            uint32_t tn = idx / (tilesC * tilesH * tilesW);
            idx %= (tilesC * tilesH * tilesW);
            uint32_t tc = idx / (tilesH * tilesW);
            idx %= (tilesH * tilesW);
            uint32_t th = idx / tilesW;
            uint32_t tw = idx % tilesW;

            uint32_t n = tn;
            uint32_t cStart = tc * tileC_;
            uint32_t hStart = th * tileH_;
            uint32_t wStart = tw * tileW_;
            uint32_t cEnd = (cStart + tileC_ > C_) ? C_ : cStart + tileC_;
            uint32_t hEnd = (hStart + tileH_ > H_) ? H_ : hStart + tileH_;
            uint32_t wEnd = (wStart + tileW_ > W_) ? W_ : wStart + tileW_;

            for (uint32_t c = cStart; c < cEnd; ++c) {
                for (uint32_t h = hStart; h < hEnd; ++h) {
                    for (uint32_t w = wStart; w < wEnd; ++w) {
                        float maxVal = -FLT_MAX;
                        for (int32_t kh = 0; kh < kernelSize_; ++kh) {
                            for (int32_t kw = 0; kw < kernelSize_; ++kw) {
                                int32_t ih = static_cast<int32_t>(h) + kh - padding_;
                                int32_t iw = static_cast<int32_t>(w) + kw - padding_;
                                float val;
                                if (ih >= 0 && ih < static_cast<int32_t>(H_) &&
                                    iw >= 0 && iw < static_cast<int32_t>(W_)) {
                                    val = xGm_[n * (C_ * H_ * W_) + c * (H_ * W_) + ih * W_ + iw];
                                } else {
                                    val = -FLT_MAX;
                                }
                                if (val > maxVal) maxVal = val;
                            }
                        }
                        yGm_[n * (C_ * H_ * W_) + c * (H_ * W_) + h * W_ + w] = maxVal;
                    }
                }
            }
            tileIdx += strideTiles;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, H_, W_;
    uint32_t tileC_, tileH_, tileW_;
    int32_t kernelSize_, stride_, padding_;
};

extern "C" __global__ __aicore__ void ca6k_04602_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04602Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.tileC, tiling_data.tileH, tiling_data.tileW,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_04602_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        x = fn(x)
        x = self.conv(x)
