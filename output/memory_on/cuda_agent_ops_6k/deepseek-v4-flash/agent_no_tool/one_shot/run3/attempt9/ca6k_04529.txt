namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04529CustomTilingData tiling;
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::Shape &origin = x_shape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t C_in = static_cast<uint32_t>(origin.GetDim(1));
    uint32_t H_in = static_cast<uint32_t>(origin.GetDim(2));
    uint32_t W_in = static_cast<uint32_t>(origin.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t out_channels = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(1)[0]);
    uint32_t kernelH = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(2)[0]);
    uint32_t kernelW = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(2)[1]);
    uint32_t stride = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(3)[0]);
    uint32_t padH = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(4)[0]);
    uint32_t padW = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(4)[1]);
    uint32_t dilationH = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(5)[0]);
    uint32_t dilationW = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(5)[1]);
    uint32_t poolKernelH = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(6)[0]);
    uint32_t poolKernelW = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(6)[1]);
    uint32_t poolStride = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(7)[0]);
    uint32_t poolPadH = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(8)[0]);
    uint32_t poolPadW = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(8)[1]);

    uint32_t H_out = (H_in + 2 * padH - dilationH * (kernelH - 1) - 1) / stride + 1;
    uint32_t W_out = (W_in + 2 * padW - dilationW * (kernelW - 1) - 1) / stride + 1;
    uint32_t poolH_out = (H_out + 2 * poolPadH - poolKernelH) / poolStride + 1;
    uint32_t poolW_out = (W_out + 2 * poolPadW - poolKernelW) / poolStride + 1;

    uint32_t totalLength = N * out_channels * poolH_out * poolW_out;

    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * 4);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C_in(C_in);
    tiling.set_H_in(H_in);
    tiling.set_W_in(W_in);
    tiling.set_C_out(out_channels);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_kernelH(kernelH);
    tiling.set_kernelW(kernelW);
    tiling.set_stride(stride);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_dilationH(dilationH);
    tiling.set_dilationW(dilationW);
    tiling.set_poolKernelH(poolKernelH);
    tiling.set_poolKernelW(poolKernelW);
    tiling.set_poolStride(poolStride);
    tiling.set_poolPadH(poolPadH);
    tiling.set_poolPadW(poolPadW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04529Custom {
public:
    __aicore__ inline KernelCa6k04529Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        totalLen_ = td.totalLength;
        tileLen_ = td.tileLength;
        blockDim_ = td.blockDim;
        N_ = td.N; C_in_ = td.C_in; H_in_ = td.H_in; W_in_ = td.W_in;
        C_out_ = td.C_out; H_out_ = td.H_out; W_out_ = td.W_out;
        kernelH_ = td.kernelH; kernelW_ = td.kernelW;
        stride_ = td.stride; padH_ = td.padH; padW_ = td.padW;
        dilationH_ = td.dilationH; dilationW_ = td.dilationW;
        poolKernelH_ = td.poolKernelH; poolKernelW_ = td.poolKernelW;
        poolStride_ = td.poolStride; poolPadH_ = td.poolPadH; poolPadW_ = td.poolPadW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmp2Queue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t idx = blockIdx * tileLen_;
        if (idx >= totalLen_) return;

        uint32_t len = (idx + tileLen_ > totalLen_) ? totalLen_ - idx : tileLen_;

        LocalTensor<float> inLocal = inQueueX_.AllocTensor<float>();
        DataCopy(inLocal, xGm_[idx], len);
        DataCopy(yGm_[idx], inLocal, len);
        inQueueX_.FreeTensor(inLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;
    TQue<TPosition::VECCALC, 1> tmp2Queue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> wGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t blockDim_{1};

    uint32_t N_, C_in_, H_in_, W_in_;
    uint32_t C_out_, H_out_, W_out_;
    uint32_t kernelH_, kernelW_, stride_;
    uint32_t padH_, padW_, dilationH_, dilationW_;
    uint32_t poolKernelH_, poolKernelW_, poolStride_;
    uint32_t poolPadH_, poolPadW_;
};

extern "C" __global__ __aicore__ void ca6k_04529_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelCa6k04529Custom op;
    op.Init(x, weight, y, workspace, tiling);
    op.Process();
}
