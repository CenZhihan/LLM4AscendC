#include "ca6k_00501_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t ALIGN = 64;
static constexpr uint32_t MAX_BLOCKS = 64;
static constexpr uint32_t MIN_BLOCKS = 8;

static uint32_t AlignDown(uint32_t val, uint32_t align)
{
    return (val / align) * align;
}

static uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00501CustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::Shape &originShape = xShape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(2);

    uint32_t kernelH = static_cast<uint32_t>(kernel_size[0]);
    uint32_t kernelW = static_cast<uint32_t>(kernel_size[1]);
    uint32_t strideH = static_cast<uint32_t>(stride[0]);
    uint32_t strideW = static_cast<uint32_t>(stride[1]);
    uint32_t padH = static_cast<uint32_t>(padding[0]);
    uint32_t padW = static_cast<uint32_t>(padding[1]);

    uint32_t Hout = (H + 2 * padH - kernelH) / strideH + 1;
    uint32_t Wout = (W + 2 * padW - kernelW) / strideW + 1;

    uint32_t totalSlices = N * C;
    uint32_t blockDim = totalSlices;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    context->SetBlockDim(blockDim);

    uint32_t tileHout = 64;
    uint32_t tileWout = 64;
    if (tileHout > Hout) tileHout = Hout;
    if (tileWout > Wout) tileWout = Wout;
    tileHout = AlignDown(tileHout, ALIGN);
    tileWout = AlignDown(tileWout, ALIGN);
    if (tileHout == 0) tileHout = Hout;
    if (tileWout == 0) tileWout = Wout;

    tiling.set_totalBatch(N);
    tiling.set_totalChannels(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Hout(Hout);
    tiling.set_Wout(Wout);
    tiling.set_kernelH(kernelH);
    tiling.set_kernelW(kernelW);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_tileHout(tileHout);
    tiling.set_tileWout(tileWout);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t C = static_cast<uint32_t>(xShape->GetDim(1));
    uint32_t H = static_cast<uint32_t>(xShape->GetDim(2));
    uint32_t W = static_cast<uint32_t>(xShape->GetDim(3));

    constexpr uint32_t kKernelH = 3, kKernelW = 3;
    constexpr uint32_t kStrideH = 2, kStrideW = 2;
    constexpr uint32_t kPadH = 1, kPadW = 1;
    uint32_t Hout = (H + 2 * kPadH - kKernelH) / kStrideH + 1;
    uint32_t Wout = (W + 2 * kPadW - kKernelW) / kStrideW + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, Hout);
    y_shape->SetDim(3, Wout);
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, x_dtype);

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -1e38f;

class KernelCa6k00501Custom {
public:
    __aicore__ inline KernelCa6k00501Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalBatch, uint32_t totalChannels,
                                uint32_t H, uint32_t W,
                                uint32_t Hout, uint32_t Wout,
                                uint32_t kernelH, uint32_t kernelW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t tileHout, uint32_t tileWout)
    {
        totalBatch_ = totalBatch;
        totalChannels_ = totalChannels;
        H_ = H;
        W_ = W;
        Hout_ = Hout;
        Wout_ = Wout;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        tileHout_ = tileHout;
        tileWout_ = tileWout;
        totalElems_ = totalBatch * totalChannels * H * W;
        outTotalElems_ = totalBatch * totalChannels * Hout * Wout;

        xGm_.SetGlobalBuffer((__gm__ float *)x, totalElems_);
        yGm_.SetGlobalBuffer((__gm__ float *)y, outTotalElems_);

        inputTileH_ = tileHout * strideH + kernelH - 1;
        inputTileW_ = tileWout * strideW + kernelW - 1;
        uint32_t tileSizeIn = tileHout_ * tileWout_;
        uint32_t tileSizeInWindow = inputTileH_ * inputTileW_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileSizeInWindow * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileSizeIn * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileSizeIn * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (Hout_ == 0 || Wout_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t totalSlices = totalBatch_ * totalChannels_;

        uint32_t slicesPerBlock = (totalSlices + blockNum - 1) / blockNum;
        uint32_t startSlice = blockIdx * slicesPerBlock;
        uint32_t endSlice = startSlice + slicesPerBlock;
        if (endSlice > totalSlices) endSlice = totalSlices;

        for (uint32_t slice = startSlice; slice < endSlice; ++slice) {
            uint32_t b = slice / totalChannels_;
            uint32_t c = slice % totalChannels_;
            uint32_t inputBase = (b * totalChannels_ + c) * H_ * W_;
            uint32_t outputBase = (b * totalChannels_ + c) * Hout_ * Wout_;

            uint32_t hOutStart = 0;
            while (hOutStart < Hout_) {
                uint32_t hOutEnd = hOutStart + tileHout_;
                if (hOutEnd > Hout_) hOutEnd = Hout_;
                uint32_t wOutStart = 0;
                while (wOutStart < Wout_) {
                    uint32_t wOutEnd = wOutStart + tileWout_;
                    if (wOutEnd > Wout_) wOutEnd = Wout_;

                    int32_t hInStart = static_cast<int32_t>(hOutStart * strideH_) - static_cast<int32_t>(padH_);
                    int32_t hInEnd = static_cast<int32_t>((hOutEnd - 1) * strideH_ + kernelH_) - static_cast<int32_t>(padH_);
                    int32_t wInStart = static_cast<int32_t>(wOutStart * strideW_) - static_cast<int32_t>(padW_);
                    int32_t wInEnd = static_cast<int32_t>((wOutEnd - 1) * strideW_ + kernelW_) - static_cast<int32_t>(padW_);

                    int32_t hInStartClipped = (hInStart < 0) ? 0 : hInStart;
                    int32_t hInEndClipped = (hInEnd >= static_cast<int32_t>(H_)) ? static_cast<int32_t>(H_) - 1 : hInEnd;
                    int32_t wInStartClipped = (wInStart < 0) ? 0 : wInStart;
                    int32_t wInEndClipped = (wInEnd >= static_cast<int32_t>(W_)) ? static_cast<int32_t>(W_) - 1 : wInEnd;

                    uint32_t loadH = hInEndClipped - hInStartClipped + 1;
                    uint32_t loadW = wInEndClipped - wInStartClipped + 1;
                    if (loadH == 0 || loadW == 0) {
                        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                        AscendC::Duplicate(yLocal, NEG_INF, (wOutEnd - wOutStart) * (hOutEnd - hOutStart));
                        outQueueY_.EnQue(yLocal);
                        CopyOut(outputBase + hOutStart * Wout_ + wOutStart,
                                 hOutEnd - hOutStart, wOutEnd - wOutStart, Wout_);
                        wOutStart = wOutEnd;
                        continue;
                    }

                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    for (uint32_t row = 0; row < loadH; ++row) {
                        uint32_t gmOffset = inputBase + (hInStartClipped + row) * W_ + wInStartClipped;
                        uint32_t localOffset = row * loadW;
                        AscendC::DataCopy(xLocal[localOffset], xGm_[gmOffset], loadW);
                    }
                    inQueueX_.EnQue(xLocal);
                    ComputePartial(hOutStart, hOutEnd, wOutStart, wOutEnd,
                                   hInStart, hInEnd, wInStart, wInEnd,
                                   inputBase);
                    CopyOut(outputBase + hOutStart * Wout_ + wOutStart,
                            hOutEnd - hOutStart, wOutEnd - wOutStart, Wout_);
                    wOutStart = wOutEnd;
                }
                hOutStart = hOutEnd;
            }
        }
    }

private:
    __aicore__ inline void ComputePartial(uint32_t hOutStart, uint32_t hOutEnd,
                                          uint32_t wOutStart, uint32_t wOutEnd,
                                          int32_t hInStart, int32_t hInEnd,
                                          int32_t wInStart, int32_t wInEnd,
                                          uint32_t inputBase)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        uint32_t tileH = hOutEnd - hOutStart;
        uint32_t tileW = wOutEnd - wOutStart;
        uint32_t tileSize = tileH * tileW;

        AscendC::Duplicate(yLocal, NEG_INF, tileSize);

        for (uint32_t ho = 0; ho < tileH; ++ho) {
            for (uint32_t wo = 0; wo < tileW; ++wo) {
                uint32_t outIdx = ho * tileW + wo;
                int32_t hc = static_cast<int32_t>(hOutStart + ho) * strideH_ - static_cast<int32_t>(padH_);
                int32_t wc = static_cast<int32_t>(wOutStart + wo) * strideW_ - static_cast<int32_t>(padW_);
                float maxVal = NEG_INF;
                for (uint32_t kh = 0; kh < kernelH_; ++kh) {
                    int32_t hi = hc + kh;
                    if (hi < 0 || hi >= static_cast<int32_t>(H_)) continue;
                    for (uint32_t kw = 0; kw < kernelW_; ++kw) {
                        int32_t wi = wc + kw;
                        if (wi < 0 || wi >= static_cast<int32_t>(W_)) continue;
                        uint32_t row = static_cast<uint32_t>(hi - hInStart);
                        uint32_t col = static_cast<uint32_t>(wi - wInStart);
                        uint32_t localIdx = row * (wInEnd - wInStart + 1) + col;
                        float val = xLocal(localIdx);
                        if (val > maxVal) maxVal = val;
                    }
                }
                yLocal(outIdx) = maxVal;
            }
        }

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t hSize, uint32_t wSize, uint32_t Wout)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        for (uint32_t row = 0; row < hSize; ++row) {
            uint32_t localOffset = row * wSize;
            uint32_t gmRowOffset = gmOffset + row * Wout;
            AscendC::DataCopy(yGm_[gmRowOffset], yLocal[localOffset], wSize);
        }
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalBatch_, totalChannels_;
    uint32_t H_, W_;
    uint32_t Hout_, Wout_;
    uint32_t kernelH_, kernelW_;
    uint32_t strideH_, strideW_;
    uint32_t padH_, padW_;
    uint32_t tileHout_, tileWout_;
    uint32_t inputTileH_, inputTileW_;
    uint32_t totalElems_, outTotalElems_;
};

extern "C" __global__ __aicore__ void ca6k_00501_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00501Custom op;
    op.Init(x, y,
            tiling_data.totalBatch, tiling_data.totalChannels,
            tiling_data.H, tiling_data.W,
            tiling_data.Hout, tiling_data.Wout,
            tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.tileHout, tiling_data.tileWout);
    op.Process();
}
'''
