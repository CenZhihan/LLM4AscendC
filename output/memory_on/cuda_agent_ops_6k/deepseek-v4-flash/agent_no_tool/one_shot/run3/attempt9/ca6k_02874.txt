kernel_src="""

#include "kernel_operator.h"

class KernelCa6k02874Custom {
public:
    __aicore__ inline KernelCa6k02874Custom() {}

    __aicore__ inline void Init(GM_ADDR bias, GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalRows, uint32_t totalCols,
                                uint32_t tileRow, uint32_t tileCol)
    {
        totalRows_ = totalRows;
        totalCols_ = totalCols;
        tileRow_ = tileRow;
        tileCol_ = tileCol;

        biasGm_.SetGlobalBuffer((__gm__ float*)bias, totalCols_);
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * totalCols_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, totalCols_ * totalCols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * totalCols_);

        // Allocate local buffers: bias, xRow, weightRow, outRow (each totalCols floats)
        pipe_.InitBuffer(biasBuf_, 1, totalCols_ * sizeof(float));
        pipe_.InitBuffer(xRowBuf_, 1, totalCols_ * sizeof(float));
        pipe_.InitBuffer(weightRowBuf_, 1, totalCols_ * sizeof(float));
        pipe_.InitBuffer(outRowBuf_, 1, totalCols_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || totalCols_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t startRow = blockIdx * tileRow_;
        uint32_t endRow = startRow + tileRow_;
        if (startRow >= totalRows_) return;
        if (endRow > totalRows_) endRow = totalRows_;

        // Load bias once (same for all rows)
        AscendC::LocalTensor<float> biasLocal = biasBuf_.Get<float>();
        AscendC::DataCopy(biasLocal, biasGm_, totalCols_);

        for (uint32_t r = startRow; r < endRow; ++r) {
            // Load x row
            AscendC::LocalTensor<float> xRowLocal = xRowBuf_.Get<float>();
            AscendC::DataCopy(xRowLocal, xGm_[r * totalCols_], totalCols_);

            // Initialize output with bias
            AscendC::LocalTensor<float> outRowLocal = outRowBuf_.Get<float>();
            AscendC::DataCopy(outRowLocal, biasLocal, totalCols_);

            // For each k, accumulate weight[k,:] scaled by x[r,k]
            for (uint32_t k = 0; k < totalCols_; ++k) {
                AscendC::LocalTensor<float> weightRowLocal = weightRowBuf_.Get<float>();
                AscendC::DataCopy(weightRowLocal, weightGm_[k * totalCols_], totalCols_);

                float xVal = xRowLocal(k);
                AscendC::Muls(weightRowLocal, weightRowLocal, xVal, totalCols_);
                AscendC::Add(outRowLocal, outRowLocal, weightRowLocal, totalCols_);
            }

            // Write output row to global memory
            AscendC::DataCopy(yGm_[r * totalCols_], outRowLocal, totalCols_);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> biasBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> xRowBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> weightRowBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> outRowBuf_;

    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t totalCols_ {0};
    uint32_t tileRow_ {0};
    uint32_t tileCol_ {0};
};

extern "C" __global__ __aicore__ void ca6k02874_custom(GM_ADDR bias, GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02874Custom op;
    op.Init(bias, x, weight, y,
            tiling_data.totalRows, tiling_data.totalCols,
            tiling_data.tileRow, tiling_data.tileCol);
    op.Process();
}
