kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k03803Custom {
public:
    __aicore__ inline KernelCa6k03803Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(tilingData, tiling);
        totalLen_ = tilingData.totalLength;
        tileLen_ = tilingData.tileLength;
        blockDim_ = tilingData.blockDim;
        N_ = tilingData.N;
        C_in_ = tilingData.C_in;
        C_out_ = tilingData.C_out;
        H_in_ = tilingData.H_in;
        W_in_ = tilingData.W_in;
        H_out_ = tilingData.H_out;
        W_out_ = tilingData.W_out;
        K_h_ = tilingData.K_h;
        K_w_ = tilingData.K_w;
        stride_ = tilingData.stride;
        padding_ = tilingData.padding;
        output_padding_ = tilingData.output_padding;
        dilation_h_ = tilingData.dilation_h;
        dilation_w_ = tilingData.dilation_w;
        weightSize_ = tilingData.weightSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight); // shape: C_in x C_out x K_h x K_w
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Buffers: two for input tiles, two for output, one for weight, one for workspace
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(wBuf_, 1, weightSize_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        // Load full weight into local memory (shared across tiles)
        LoadWeight();

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t totalTiles = (totalLen_ + tileLen_ - 1) / tileLen_;
        // grid stride over tiles
        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            uint32_t start = tileIdx * tileLen_;
            uint32_t len = (tileLen_ <= totalLen_ - start) ? tileLen_ : (totalLen_ - start);
            if (len == 0) continue;

            ComputeTile(start, len);
        }
    }

private:
    __aicore__ inline void LoadWeight()
    {
        AscendC::LocalTensor<float> wLocal = wBuf_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, weightSize_);
        wBuf_.EnQue(wLocal);
    }

    __aicore__ inline void ComputeTile(uint32_t start, uint32_t len)
    {
        // start is linear offset in output y (N, C_out, H_out, W_out)
        AscendC::LocalTensor<float> wLocal = wBuf_.DeQue<float>();
        AscendC::LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // For each output element, compute index and accumulate
        for (uint32_t i = 0; i < len; i++) {
            uint32_t idx = start + i;
            // decode idx to (n, c_out, h_out, w_out)
            uint32_t n = idx / (C_out_ * H_out_ * W_out_);
            uint32_t rem = idx % (C_out_ * H_out_ * W_out_);
            uint32_t c_out = rem / (H_out_ * W_out_);
            uint32_t hw = rem % (H_out_ * W_out_);
            uint32_t h_out = hw / W_out_;
            uint32_t w_out = hw % W_out_;

            float sum = 0.0f;
            // iterate over input channels and kernel elements
            for (uint32_t ic = 0; ic < C_in_; ic++) {
                for (uint32_t kh = 0; kh < K_h_; kh++) {
                    for (uint32_t kw = 0; kw < K_w_; kw++) {
                        // input spatial position with dilation
                        int32_t h_in = static_cast<int32_t>(h_out) - (static_cast<int32_t>(padding_)
                                    - static_cast<int32_t>(dilation_h_ * kh));
                        int32_t w_in = static_cast<int32_t>(w_out) - (static_cast<int32_t>(padding_)
                                    - static_cast<int32_t>(dilation_w_ * kw));
                        // apply stride? For transposed convolution, the input is at positions
                        // h_in = (h_out - padding - dilation*(K_h-1 - kh)) / stride?
                        // Since stride=1, it's simpler.
                        // For standard transposed convolution without stride, the output position
                        // h_out corresponds to input positions h_in such that
                        // h_out = h_in + padding + dilation_h * kh ? The formula is opposite.
                        // Let's use the im2col interpretation:
                        // output at (h_out, w_out) receives contributions from input at
                        // (h_out - padding - dilation_h * kh, w_out - padding - dilation_w * kw)
                        // with stride=1.
                        // Check boundaries
                        if (h_in >= 0 && h_in < static_cast<int32_t>(H_in_) &&
                            w_in >= 0 && w_in < static_cast<int32_t>(W_in_)) {
                            // compute linear index in input
                            uint32_t xIdx = ((n * C_in_ + ic) * H_in_ + h_in) * W_in_ + w_in;
                            // weight index: (ic, c_out, kh, kw)
                            uint32_t wIdx = ((ic * C_out_ + c_out) * K_h_ + kh) * K_w_ + kw;
                            float xVal = xGm_.GetValue(xIdx);
                            float wVal = wLocal.GetValue(wIdx);
                            sum += xVal * wVal;
                        }
                    }
                }
            }
            // Store result (bias not used)
            yGm_.SetValue(idx, sum);
        }

        tmpBuf_.FreeTensor(tmp);
        outQueueY_.FreeTensor(yLocal);
        wBuf_.FreeTensor(wLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t blockDim_;
    uint32_t N_, C_in_, C_out_, H_in_, W_in_, H_out_, W_out_;
    uint32_t K_h_, K_w_, stride_, padding_, output_padding_;
    uint32_t dilation_h_, dilation_w_;
    uint32_t weightSize_;
};

extern "C" __global__ __aicore__ void ca6k_03803_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelCa6k03803Custom op;
    op.Init(x, weight, y, workspace, tiling);
    op.Process();
}

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0,
                 output_padding=0, dilation=(1,1)):
        super().__init__()
        # Store parameters for later use
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.dilation = dilation
        # Use a real ConvTranspose2d module to maintain parameter names (state dict compatibility)

            in_channels, out_channels, kernel_size,
            stride=stride, padding=padding,
            output_padding=output_padding, dilation=dilation,
            bias=False
        )

        # Extract weight from the internal module and call custom op
        w = self.conv_transpose.weight
        return custom_ops_lib.ca6k_03803_custom(
            x, w,
            self.in_channels, self.out_channels,
            self.kernel_size[0], self.kernel_size[1],
            self.stride, self.padding, self.output_padding,
