#include "ca6k_00303_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t TILE_ALIGN = 1024;
static constexpr uint32_t MAX_TILE_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
  return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k00303CustomTilingData tiling;

  const gert::Shape *in_shape = context->GetInputShape(0);
  uint32_t N = static_cast<uint32_t>(in_shape->GetDim(0));
  uint32_t C_in = static_cast<uint32_t>(in_shape->GetDim(1));
  uint32_t H = static_cast<uint32_t>(in_shape->GetDim(2));
  uint32_t W = static_cast<uint32_t>(in_shape->GetDim(3));

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int64_t scaleFactor = attrs->GetInt(0);

  uint32_t r = static_cast<uint32_t>(scaleFactor);
  uint32_t C_out = C_in / (r * r);
  uint32_t H_out = H * r;
  uint32_t W_out = W * r;
  uint32_t totalLength = N * C_out * H_out * W_out;

  uint32_t tileLength = 1024;
  if (tileLength > MAX_TILE_ELEMS) tileLength = MAX_TILE_ELEMS;
  tileLength = AlignDown(tileLength, TILE_ALIGN);
  if (tileLength < TILE_ALIGN) tileLength = TILE_ALIGN;

  uint32_t numTiles = CeilDiv(totalLength, tileLength);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_batchSize(N);
  tiling.set_inChannels(C_in);
  tiling.set_outChannels(C_out);
  tiling.set_inHeight(H);
  tiling.set_inWidth(W);
  tiling.set_scaleFactor(r);
  tiling.set_outHeight(H_out);
  tiling.set_outWidth(W_out);
  tiling.set_totalLength(totalLength);
  tiling.set_tileLength(tileLength);
  tiling.set_blockDim(blockDim);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape *in_shape = context->GetInputShape(0);
  uint32_t N = static_cast<uint32_t>(in_shape->GetDim(0));
  uint32_t C_in = static_cast<uint32_t>(in_shape->GetDim(1));
  uint32_t H = static_cast<uint32_t>(in_shape->GetDim(2));
  uint32_t W = static_cast<uint32_t>(in_shape->GetDim(3));

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int64_t scaleFactor = attrs->GetInt(0);
  uint32_t r = static_cast<uint32_t>(scaleFactor);
  uint32_t C_out = C_in / (r * r);
  uint32_t H_out = H * r;
  uint32_t W_out = W * r;

}

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelPixelShuffleCustom {
public:
  __aicore__ inline KernelPixelShuffleCustom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                              uint32_t batchSize, uint32_t inChannels,
                              uint32_t outChannels, uint32_t inHeight,
                              uint32_t inWidth, uint32_t scaleFactor,
                              uint32_t outHeight, uint32_t outWidth,
                              uint32_t totalLength, uint32_t tileLength) {
    batchSize_ = batchSize;
    inChannels_ = inChannels;
    outChannels_ = outChannels;
    inHeight_ = inHeight;
    inWidth_ = inWidth;
    scaleFactor_ = scaleFactor;
    outHeight_ = outHeight;
    outWidth_ = outWidth;
    totalLength_ = totalLength;
    tileLength_ = tileLength;

    xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inChannels * inHeight * inWidth);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
  }

  __aicore__ inline void Process() {
    if (totalLength_ == 0) return;

    uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    for (uint32_t idx = blockIdx; idx < totalLength_; idx += blockNum) {
      uint32_t tail = idx;
      uint32_t n = tail / (outChannels_ * outHeight_ * outWidth_);
      tail %= (outChannels_ * outHeight_ * outWidth_);
      uint32_t c = tail / (outHeight_ * outWidth_);
      tail %= (outHeight_ * outWidth_);
      uint32_t h = tail / outWidth_;
      uint32_t w = tail % outWidth_;

      uint32_t in_h = h / scaleFactor_;
      uint32_t in_w = w / scaleFactor_;
      uint32_t r = scaleFactor_;
      uint32_t in_c = c * r * r + (h % r) * r + (w % r);

      uint32_t in_idx = n * (inChannels_ * inHeight_ * inWidth_) +
                        in_c * (inHeight_ * inWidth_) +
                        in_h * inWidth_ + in_w;

      float val = xGm_[in_idx];
      yGm_[idx] = val;
    }
  }

private:
  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> yGm_;

  uint32_t batchSize_;
  uint32_t inChannels_;
  uint32_t outChannels_;
  uint32_t inHeight_;
  uint32_t inWidth_;
  uint32_t scaleFactor_;
  uint32_t outHeight_;
  uint32_t outWidth_;
  uint32_t totalLength_;
  uint32_t tileLength_;
};

extern "C" __global__ __aicore__ void ca6k_00303_custom(GM_ADDR x, GM_ADDR y,
                                                        GM_ADDR workspace,
                                                        GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelPixelShuffleCustom op;
  op.Init(x, y,
          tiling_data.batchSize, tiling_data.inChannels,
          tiling_data.outChannels, tiling_data.inHeight,
          tiling_data.inWidth, tiling_data.scaleFactor,
          tiling_data.outHeight, tiling_data.outWidth,
          tiling_data.totalLength, tiling_data.tileLength);
  op.Process();
}
'''
