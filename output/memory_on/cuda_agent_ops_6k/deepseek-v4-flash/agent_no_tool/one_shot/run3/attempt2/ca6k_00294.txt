#include "ca6k_00294_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00294CustomTilingData tiling;
    const gert::StorageShape *indicesShape = context->GetInputShape(0);
    const gert::StorageShape *weightShape = context->GetInputShape(1);

    uint32_t totalIndices = indicesShape->GetOriginShape().GetShapeSize();
    uint32_t embeddingDim = weightShape->GetOriginShape().GetDim(1);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *paddingIdx = attrs->GetAttrPointer<int64_t>(0);

    // UB budget estimation: each row needs 2*embeddingDim*sizeof(float) + sizeof(int64_t)
    // double buffering -> factor 2
    uint32_t rowCost = 2 * embeddingDim * 4 + 8;
    uint32_t maxTile = 192 * 1024 / rowCost;
    if (maxTile > 64) maxTile = 64;
    maxTile = (maxTile / 32) * 32;
    if (maxTile < 32) maxTile = 32;

    uint32_t tileIndices = (totalIndices < maxTile) ? totalIndices : maxTile;
    if (tileIndices == 0) tileIndices = 1;

    uint32_t blockDim = (totalIndices + tileIndices - 1) / tileIndices;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.totalIndices_ = totalIndices;
    tiling.tileIndices_ = tileIndices;
    tiling.embeddingDim_ = embeddingDim;
    tiling.paddingIdx_ = static_cast<uint32_t>(*paddingIdx);
    tiling.blockDim_ = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *indicesShape = context->GetInputShape(0);
    const gert::Shape *weightShape = context->GetInputShape(1);
    gert::Shape *yShape = context->GetOutputShape(0);

    uint32_t ndim = indicesShape->GetDimNum();
    uint32_t embeddingDim = weightShape->GetDim(1);   // assume weight is 2D

    yShape->SetDimNum(ndim + 1);
    for (uint32_t i = 0; i < ndim; ++i) {
        yShape->SetDim(i, indicesShape->GetDim(i));
    }
    yShape->SetDim(ndim, static_cast<int64_t>(embeddingDim));
    return ge::GRAPH_SUCCESS;
}

{
    ge::DataType weightDtype = context->GetInputDataType(1);
    context->SetOutputDataType(0, weightDtype);
    return ge::GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00294Custom {
public:
    __aicore__ inline KernelCa6k00294Custom() {}

    __aicore__ inline void Init(GM_ADDR indices, GM_ADDR weight, GM_ADDR output,
                                uint32_t totalIndices, uint32_t tileIndices,
                                uint32_t embeddingDim, uint32_t paddingIdx)
    {
        totalIndices_ = totalIndices;
        tileIndices_ = tileIndices;
        embeddingDim_ = embeddingDim;
        paddingIdx_ = paddingIdx;

        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices, totalIndices_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalIndices_ * embeddingDim_);

        pipe_.InitBuffer(inQueueIndices_, BUFFER_NUM, tileIndices_ * sizeof(int64_t));
        pipe_.InitBuffer(outQueueY_, 1, embeddingDim_ * sizeof(float));
        pipe_.InitBuffer(weightRowTmp_, 1, embeddingDim_ * sizeof(float));
        pipe_.InitBuffer(zeroRowTmp_, 1, embeddingDim_ * sizeof(float));
        AscendC::SetZero(zeroRowTmp_, embeddingDim_);
    }

    __aicore__ inline void Process()
    {
        if (totalIndices_ == 0 || tileIndices_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t strideRows = blockNum * tileIndices_;
        uint32_t startRow = blockIdx * tileIndices_;

        while (startRow < totalIndices_) {
            uint32_t curRows = tileIndices_;
            if (startRow + curRows > totalIndices_) {
                curRows = totalIndices_ - startRow;
            }

            LoadIndices(startRow, curRows);
            ProcessRows(startRow, curRows);

            startRow += strideRows;
        }
    }

private:
    __aicore__ inline void LoadIndices(uint32_t rowOffset, uint32_t rows)
    {
        AscendC::LocalTensor<int64_t> idxLocal = inQueueIndices_.AllocTensor<int64_t>();
        AscendC::DataCopy(idxLocal, indicesGm_[rowOffset], rows);
        inQueueIndices_.EnQue(idxLocal);
    }

    __aicore__ inline void ProcessRows(uint32_t rowOffset, uint32_t rows)
    {
        AscendC::LocalTensor<int64_t> idxLocal = inQueueIndices_.DeQue<int64_t>();
        AscendC::LocalTensor<float> weightRow = weightRowTmp_.AllocTensor<float>();
        AscendC::LocalTensor<float> zeroRow = zeroRowTmp_;
        AscendC::LocalTensor<float> outRow = outQueueY_.AllocTensor<float>();

        for (uint32_t r = 0; r < rows; ++r) {
            int64_t idx = idxLocal(r);
            uint32_t outBase = (rowOffset + r) * embeddingDim_;

            if (static_cast<int64_t>(paddingIdx_) == idx) {
                // Write zeros
                AscendC::DataCopy(outputGm_[outBase], zeroRow, embeddingDim_);
            } else {
                uint32_t weightBase = static_cast<uint32_t>(idx) * embeddingDim_;
                // Load weight row
                AscendC::DataCopy(weightRow, weightGm_[weightBase], embeddingDim_);
                // Copy to output
                outRow = weightRow;
                AscendC::DataCopy(outputGm_[outBase], outRow, embeddingDim_);
            }
        }

        outQueueY_.FreeTensor(outRow);
        weightRowTmp_.FreeTensor(weightRow);
        inQueueIndices_.FreeTensor(idxLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIndices_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightRowTmp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> zeroRowTmp_;

    AscendC::GlobalTensor<int64_t> indicesGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t totalIndices_ {0};
    uint32_t tileIndices_ {0};
    uint32_t embeddingDim_ {0};
    uint32_t paddingIdx_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00294_custom(GM_ADDR indices, GM_ADDR weight, GM_ADDR output,
                                                         GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00294Custom op;
    op.Init(indices, weight, output,
            tiling_data.totalIndices_, tiling_data.tileIndices_,
            tiling_data.embeddingDim_, tiling_data.paddingIdx_);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00294_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_embeddings, embedding_dim, padding_idx):
        super(ModelNew, self).__init__()
        self.embedding = torch.nn.Embedding(num_embeddings, embedding_dim, padding_idx=padding_idx)
        self.padding_idx = padding_idx

    def forward(self, x):
        flat = x.reshape(-1)
        result = getattr(custom_ops_lib, _FN)(flat, self.embedding.weight, self.padding_idx)
        return result.reshape(*x.shape, self.embedding.weight.size(-1))
