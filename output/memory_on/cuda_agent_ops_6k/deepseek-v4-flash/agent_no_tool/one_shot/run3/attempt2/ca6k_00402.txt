#include "ca6k_00402_custom_tiling.h"

namespace optiling {

static inline uint32_t productExceptDim(const gert::Shape *shape, int32_t dim) {
    uint32_t prod = 1;
    uint32_t ndim = static_cast<uint32_t>(shape->GetDimNum());
    for (uint32_t i = 0; i < ndim; i++) {
        if (static_cast<int32_t>(i) != dim)
            prod *= static_cast<uint32_t>(shape->GetDim(i));
    }
    return prod;
}

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00402CustomTilingData tiling;
    const gert::Shape *shape = context->GetInputShape(0);
    const uint32_t totalLength = static_cast<uint32_t>(shape->GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    uint32_t ndim = static_cast<uint32_t>(shape->GetDimNum());
    if (dim < 0) dim += ndim;
    if (dim < 0 || static_cast<uint32_t>(dim) >= ndim) {
        return ge::GRAPH_FAILED;
    }

    uint32_t reductionSize = static_cast<uint32_t>(shape->GetDim(static_cast<uint32_t>(dim)));
    uint32_t outerSize = productExceptDim(shape, static_cast<int32_t>(dim));

    uint32_t blockDim = outerSize;
    if (blockDim > 24) blockDim = 24;
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    uint32_t chunkSize = (outerSize + blockDim - 1) / blockDim;
    if (chunkSize == 0) chunkSize = 1;

    tiling.set_totalLength(totalLength);
    tiling.set_outerSize(outerSize);
    tiling.set_reductionSize(reductionSize);
    tiling.set_chunkSize(chunkSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    uint32_t ndim = static_cast<uint32_t>(x_shape->GetDimNum());
    if (dim < 0) dim += ndim;
    if (dim < 0 || static_cast<uint32_t>(dim) >= ndim) {
        return GRAPH_FAILED;

            y_shape->SetDim(idx, x_shape->GetDim(i));

        }
    }
    return GRAPH_SUCCESS;

kernel_src = '''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00402Custom {
public:
    __aicore__ inline KernelCa6k00402Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerSize, uint32_t reductionSize,
                                uint32_t chunkSize) {
        outerSize_ = outerSize;
        reductionSize_ = reductionSize;
        chunkSize_ = chunkSize;
        xGm_.SetGlobalBuffer((__gm__ float*)x, outerSize_ * reductionSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outerSize_);
        pipe_.InitBuffer(inQueue_, BUFFER_NUM, reductionSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, reductionSize_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();
        if (blockNum == 0) return;
        uint32_t startRow = blockIdx * chunkSize_;
        uint32_t endRow = startRow + chunkSize_;
        if (endRow > outerSize_) endRow = outerSize_;

        for (uint32_t row = startRow; row < endRow; row++) {
            LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
            DataCopy(xLocal, xGm_[row * reductionSize_], reductionSize_);
            inQueue_.EnQue(xLocal);

            xLocal = inQueue_.DeQue<float>();
            LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
            ReduceAdd(tmpLocal, xLocal, reductionSize_);
            float sum = tmpLocal.GetValue(0);
            tmpQueue_.FreeTensor(tmpLocal);
            inQueue_.FreeTensor(xLocal);

            float mean = sum / static_cast<float>(reductionSize_);

            LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
            yLocal.SetValue(0, mean);
            outQueue_.EnQue(yLocal);

            yLocal = outQueue_.DeQue<float>();
            DataCopy(yGm_[row], yLocal, 1);
            outQueue_.FreeTensor(yLocal);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueue_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueue_;
    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    uint32_t outerSize_;
    uint32_t reductionSize_;
    uint32_t chunkSize_;
};

extern "C" __global__ __aicore__ void ca6k_00402_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00402Custom op;
    op.Init(x, y, tiling_data.outerSize, tiling_data.reductionSize, tiling_data.chunkSize);
    op.Process();
}
'''
