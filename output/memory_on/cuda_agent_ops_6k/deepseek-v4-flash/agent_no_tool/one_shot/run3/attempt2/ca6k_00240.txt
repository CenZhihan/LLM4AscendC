host_operator_src = '''
#include "ca6k_00240_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00240CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Each tile needs: x (input) + y_true (input) + y (output) + tmp for intermediate = 4 vectors
    // plus possibly extra for log computation. Assume 4 * tileLen floats => 16 * tileLen bytes.
    uint32_t maxTile = UB_BUDGET_BYTES / 16;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00240Custom {
public:
    __aicore__ inline KernelCa6k00240Custom() {}

    __aicore__ inline void Init(GM_ADDR y_pred, GM_ADDR y_true, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        predGm_.SetGlobalBuffer((__gm__ float*)y_pred, totalLen_);
        trueGm_.SetGlobalBuffer((__gm__ float*)y_true, totalLen_);
        outGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueuePred_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueTrue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmp2Queue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> predLocal = inQueuePred_.AllocTensor<float>();
        AscendC::LocalTensor<float> trueLocal = inQueueTrue_.AllocTensor<float>();
        AscendC::DataCopy(predLocal, predGm_[gmOffset], len);
        AscendC::DataCopy(trueLocal, trueGm_[gmOffset], len);
        inQueuePred_.EnQue(predLocal);
        inQueueTrue_.EnQue(trueLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> predLocal = inQueuePred_.DeQue<float>();
        AscendC::LocalTensor<float> trueLocal = inQueueTrue_.DeQue<float>();
        AscendC::LocalTensor<float> outLocal = outQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2Local = tmp2Queue_.AllocTensor<float>();

        // Compute binary_cross_entropy_with_logits element-wise:
        // loss = max(pred, 0) - pred * true + log(1 + exp(-abs(pred)))
        // Use: max(0, pred) - pred * true + log(1 + exp(-abs(pred)))
        // Compute part by part:
        // t1 = max(0, pred)
        AscendC::Maxs(tLocal, predLocal, 0.0f, len);
        // t2 = pred * true
        AscendC::Mul(t2Local, predLocal, trueLocal, len);
        // out = t1 - t2
        AscendC::Sub(outLocal, tLocal, t2Local, len);
        // t1 = abs(pred)
        AscendC::Abs(tLocal, predLocal, len);
        // t2 = exp(-t1)
        AscendC::Neg(tLocal, tLocal, len); // -abs(pred)
        AscendC::Exp(t2Local, tLocal, len);
        // t1 = 1 + t2
        AscendC::Adds(tLocal, t2Local, 1.0f, len);
        // t2 = log(t1)
        AscendC::Log(t2Local, tLocal, len);
        // out = out + t2
        AscendC::Add(outLocal, outLocal, t2Local, len);

        tmp2Queue_.FreeTensor(t2Local);
        tmpQueue_.FreeTensor(tLocal);
        outQueue_.EnQue<float>(outLocal);
        inQueueTrue_.FreeTensor(trueLocal);
        inQueuePred_.FreeTensor(predLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> outLocal = outQueue_.DeQue<float>();
        AscendC::DataCopy(outGm_[gmOffset], outLocal, len);
        outQueue_.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePred_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTrue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp2Queue_;

    AscendC::GlobalTensor<float> predGm_;
    AscendC::GlobalTensor<float> trueGm_;
    AscendC::GlobalTensor<float> outGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00240_custom(GM_ADDR y_pred, GM_ADDR y_true, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00240Custom op;
    op.Init(y_pred, y_true, y, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
'''
