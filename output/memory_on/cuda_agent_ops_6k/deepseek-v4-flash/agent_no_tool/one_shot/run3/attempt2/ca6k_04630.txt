static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04630CustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    uint32_t totalElements = 1;
    for (int i = 0; i < xShape->GetDimNum(); ++i) {
        totalElements *= xShape->GetDim(i);
    }
    uint32_t N = xShape->GetDim(0);
    uint32_t C = xShape->GetDim(1);
    uint32_t H = xShape->GetDim(2);
    uint32_t W = xShape->GetDim(3);
    uint32_t halfC = C / 2;
    uint32_t spatialSize = H * W;
    uint32_t totalPoints = N * spatialSize;

    tiling.totalLength = totalPoints;
    tiling.inFeatures = C;
    tiling.halfFeatures = halfC;
    tiling.spatialStride = spatialSize;
    tiling.width = W;

    uint32_t maxTile = 1;  // process one point per tile
    uint32_t tileLen = maxTile;
    if (totalPoints < tileLen) tileLen = totalPoints;

    uint32_t numTiles = CeilDiv(totalPoints, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);
    tiling.blockDim = blockDim;
    tiling.tileLength = tileLen;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFusedGLULinear {
public:
    __aicore__ inline KernelFusedGLULinear() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inFeatures, uint32_t halfFeatures,
                                uint32_t spatialStride, uint32_t width)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inFeat_ = inFeatures;
        halfFeat_ = halfFeatures;
        spatialStride_ = spatialStride;
        width_ = width;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_ * inFeat_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, halfFeat_ * halfFeat_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, halfFeat_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_ * halfFeat_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inFeat_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, halfFeat_ * sizeof(float));
        pipe_.InitBuffer(weightLocal_, 1, halfFeat_ * halfFeat_ * sizeof(float));
        pipe_.InitBuffer(biasLocal_, 1, halfFeat_ * sizeof(float));
        pipe_.InitBuffer(tmpNeg_, 1, halfFeat_ * sizeof(float));
        pipe_.InitBuffer(tmpExp_, 1, halfFeat_ * sizeof(float));
        pipe_.InitBuffer(tmpDenom_, 1, halfFeat_ * sizeof(float));
        pipe_.InitBuffer(tmpSigmoid_, 1, halfFeat_ * sizeof(float));
        pipe_.InitBuffer(tmpMulOut_, 1, halfFeat_ * sizeof(float));
        pipe_.InitBuffer(matMulBuf_, 1, halfFeat_ * sizeof(float));

        // Pre-load weight and bias to local memory
        AscendC::LocalTensor<float> wLocal = weightLocal_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, weightGm_, halfFeat_ * halfFeat_);
        weightLocal_.EnQue(wLocal);

        AscendC::LocalTensor<float> bLocal = biasLocal_.AllocTensor<float>();
        AscendC::DataCopy(bLocal, biasGm_, halfFeat_);
        biasLocal_.EnQue(bLocal);
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t stridePoints = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            for (uint32_t i = 0; i < curLen; ++i) {
                uint32_t ptIdx = base + i;
                ProcessOnePoint(ptIdx);
            }

            base += stridePoints;
        }
    }

private:
    __aicore__ inline void ProcessOnePoint(uint32_t ptIdx) {
        // Spatial point index into (n, h, w)
        uint32_t n = ptIdx / spatialStride_;
        uint32_t rem = ptIdx % spatialStride_;
        uint32_t h = rem / width_;
        uint32_t w = rem % width_;

        // Base offset in global memory for this point's channel 0
        uint32_t baseOffset = n * inFeat_ * spatialStride_ + h * width_ + w;

        // Allocate a local tensor to hold the contiguous input vector of length inFeat_
        AscendC::LocalTensor<float> xCont = inQueueX_.AllocTensor<float>();
        // Load inFeat_ elements with stride spatialStride_ using DataCopyPad
        AscendC::DataCopyPad(xCont, xGm_[baseOffset], 1, inFeat_, spatialStride_ - 1, 0);
        inQueueX_.EnQue(xCont);

        // Dequeue for compute
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();

        // ----- GLU -----
        // Split into first half (a) and second half (b)
        AscendC::LocalTensor<float> a = xLocal;                                // points to first half
        AscendC::LocalTensor<float> b = xLocal + halfFeat_;                    // second half (assumes halfFeat_ aligned to vector size)

        // Compute sigmoid(a)
        AscendC::LocalTensor<float> negA = tmpNeg_;        // reuse
        AscendC::LocalTensor<float> expNegA = tmpExp_;
        AscendC::LocalTensor<float> denom = tmpDenom_;
        AscendC::LocalTensor<float> sigA = tmpSigmoid_;

        AscendC::Muls(negA, a, -1.0f, halfFeat_);
        AscendC::Exp(expNegA, negA, halfFeat_);
        AscendC::Adds(denom, expNegA, 1.0f, halfFeat_);
        AscendC::Reciprocal(sigA, denom, halfFeat_);

        // Multiply element-wise: glu_out = sigA * b
        AscendC::LocalTensor<float> gluOut = tmpMulOut_;
        AscendC::Mul(gluOut, sigA, b, halfFeat_);

        // ----- Linear -----
        // Retrieve weight and bias from local queues
        AscendC::LocalTensor<float> wLocal = weightLocal_.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = biasLocal_.DeQue<float>();

        // Compute y = gluOut * weight^T + bias  (weight shape [halfFeat, halfFeat], row-major)
        // For each output element i: sum_j gluOut[j] * weight[i * halfFeat + j] + bias[i]
        AscendC::LocalTensor<float> linearOut = matMulBuf_;
        for (uint32_t i = 0; i < halfFeat_; ++i) {
            // Load row i of weight
            AscendC::LocalTensor<float> wRow = wLocal + i * halfFeat_;
            // Dot product
            AscendC::LocalTensor<float> tmp = tmpExp_;  // reuse temporary
            AscendC::Mul(tmp, gluOut, wRow, halfFeat_);
            // Reduce sum
            AscendC::LocalTensor<float> sum = linearOut + i;
            AscendC::ReduceAdd(sum, tmp, halfFeat_);
            // Add bias
            AscendC::Adds(sum, sum, bLocal(i), 1);
        }

        // Write output y for this point: stride spatialStride_ between channels
        // Use DataCopyPad to scatter halfFeat_ elements
        uint32_t yOffset = n * halfFeat_ * spatialStride_ + h * width_ + w;
        AscendC::DataCopyPad(yGm_[yOffset], linearOut, 1, halfFeat_, 0, spatialStride_ - 1);

        // Free tensors
        weightLocal_.EnQue(wLocal);
        biasLocal_.EnQue(bLocal);
        tmpNeg_.FreeTensor(negA);
        tmpExp_.FreeTensor(expNegA);
        tmpDenom_.FreeTensor(denom);
        tmpSigmoid_.FreeTensor(sigA);
        tmpMulOut_.FreeTensor(gluOut);
        matMulBuf_.FreeTensor(linearOut);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightLocal_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> biasLocal_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpNeg_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpExp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDenom_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSigmoid_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMulOut_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> matMulBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t inFeat_ {0};
    uint32_t halfFeat_ {0};
    uint32_t spatialStride_ {0};
    uint32_t width_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04630_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedGLULinear op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.inFeatures, tiling_data.halfFeatures,
            tiling_data.spatialStride, tiling_data.width);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_04630_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_features):
        super(ModelNew, self).__init__()
        assert in_features % 2 == 0, "in_features must be even"
        self.in_features = in_features
        self.half_features = in_features // 2
        # The fused operator does not own weight/bias; they are passed in forward.
        # For compatibility, we create them here (as parameters) but they are not used in the custom op.
        # Instead, the operator expects weight and bias as inputs.
        # In actual usage, the op is invoked with weight and bias provided.
        # To match reference model, we store weight/bias for the linear layer.
        self.weight = torch.nn.Parameter(torch.empty(self.half_features, self.half_features))
        self.bias = torch.nn.Parameter(torch.empty(self.half_features))
        torch.nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        torch.nn.init.zeros_(self.bias)

    def forward(self, x):
        # x shape: (N, in_features, H, W)
        # Use custom op: requires weight and bias as tensors (not parameters)
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias)
'''
