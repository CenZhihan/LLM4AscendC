host_operator_src = """
#include "ca6k_00435_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00435CustomTilingData tiling;

    // Get shape dimensions via GetOriginShape() (fixes GetDim error)
    const gert::StorageShape *inShape = context->GetInputShape(0);
    uint32_t inputN = inShape->GetOriginShape().GetDim(0);
    uint32_t inputC = inShape->GetOriginShape().GetDim(1);
    uint32_t inputH = inShape->GetOriginShape().GetDim(2);
    uint32_t inputW = inShape->GetOriginShape().GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t upscaleFactor = static_cast<uint32_t>(attrs->GetAttrPointer<int>(0)[0]);
    uint32_t totalLength = inputN * inputC * inputH * inputW;

    // Compute tile length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = totalLength;

    uint32_t blockDim = CeilDiv(totalLength, tileLen);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    // Use setter methods (fields are private, TILING_DATA_FIELD_DEF)
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_inputN(inputN);
    tiling.set_inputC(inputC);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_upscaleFactor(upscaleFactor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    gert::Shape *yShape = context->GetOutputShape(0);

    uint32_t N = xShape->GetDim(0);
    uint32_t C = xShape->GetDim(1);
    uint32_t H = xShape->GetDim(2);
    uint32_t W = xShape->GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t r = static_cast<uint32_t>(attrs->GetAttrPointer<int>(0)[0]);

    uint32_t outC = C * r * r;
    uint32_t outH = H / r;
    uint32_t outW = W / r;

    yShape->SetDim(0, N);
    yShape->SetDim(1, outC);
    yShape->SetDim(2, outH);
    yShape->SetDim(3, outW);

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class Ca6k00435Custom : public OpDef {
public:
    explicit Ca6k00435Custom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

            .Format({ge::FORMAT_ND})

        this->Output("y")

            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("upscale_factor").AttrType(OPTIONAL).Int(2);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(Ca6k00435Custom);
} // namespace ops

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00435Custom {
public:
    __aicore__ inline KernelCa6k00435Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inputN, uint32_t inputC,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t upscaleFactor)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inN_ = inputN;
        inC_ = inputC;
        inH_ = inputH;
        inW_ = inputW;
        r_ = upscaleFactor;
        outC_ = inC_ * r_ * r_;
        outH_ = inH_ / r_;
        outW_ = inW_ / r_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(tmpBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stride = blockNum * tileLen_;
        uint32_t start = blockIdx * tileLen_;
        if (start >= totalLen_) return;

        AscendC::LocalTensor<float> scalar = tmpBuf_.Get<float>();

        for (uint32_t idx = start; idx < totalLen_; idx += stride) {
            uint32_t end = idx + tileLen_;
            if (end > totalLen_) end = totalLen_;
            for (uint32_t i = idx; i < end; ++i) {
                // Compute output coordinates
                uint32_t n = i / (outC_ * outH_ * outW_);
                uint32_t rem1 = i % (outC_ * outH_ * outW_);
                uint32_t oc = rem1 / (outH_ * outW_);
                uint32_t rem2 = rem1 % (outH_ * outW_);
                uint32_t oh = rem2 / outW_;
                uint32_t ow = rem2 % outW_;

                // Map to input coordinates
                uint32_t ic = oc / (r_ * r_);
                uint32_t localIdx = oc % (r_ * r_);
                uint32_t ih = oh * r_ + localIdx / r_;
                uint32_t iw = ow * r_ + localIdx % r_;

                // Compute input global index
                uint32_t inIdx = ((n * inC_ + ic) * inH_ + ih) * inW_ + iw;

                // Read one element from global
                AscendC::DataCopy(scalar, xGm_[inIdx], 1);
                // Write one element to global
                AscendC::DataCopy(yGm_[i], scalar, 1);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t inN_{0};
    uint32_t inC_{0};
    uint32_t inH_{0};
    uint32_t inW_{0};
    uint32_t r_{2};
    uint32_t outC_{0};
    uint32_t outH_{0};
    uint32_t outW_{0};
};

extern "C" __global__ __aicore__ void ca6k_00435_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelCa6k00435Custom op;
    op.Init(x, y,
            tilingData.totalLength, tilingData.tileLength,
            tilingData.inputN, tilingData.inputC,
            tilingData.inputH, tilingData.inputW,
            tilingData.upscaleFactor);
    op.Process();
}
"""
