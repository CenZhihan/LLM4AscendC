#include "ca6k_04629_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t MAX_BLOCK_DIM = 24;
static constexpr uint32_t MIN_BLOCK_DIM = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04629CustomTilingData tiling;
    // input shape: N, C, H, W
    const gert::Shape *inShape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(inShape->GetDim(0));
    uint32_t C = static_cast<uint32_t>(inShape->GetDim(1));
    uint32_t H = static_cast<uint32_t>(inShape->GetDim(2));
    uint32_t W = static_cast<uint32_t>(inShape->GetDim(3));
    uint32_t totalLength = N * C * H * W;
    uint32_t totalRows = N * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t groups = attrs->GetAttrPointer<int32_t>(0)[0];
    if (groups <= 0) groups = 2;
    if (C % groups != 0) {
        return ge::GRAPH_FAILED;
    }

    // UB requirement (3 * tileRows * C floats): in, out, temp
    uint32_t maxTileRows = UB_BUDGET_BYTES / (3 * C * sizeof(float));
    if (maxTileRows < 1) maxTileRows = 1;
    uint32_t tileRows = maxTileRows;
    if (tileRows > totalRows) tileRows = totalRows;
    tileRows = AlignDown(tileRows, ALIGN_ELEMS);
    if (tileRows == 0) tileRows = 1;

    uint32_t tileLength = tileRows * C;

    uint32_t numTiles = CeilDiv(totalRows, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < MIN_BLOCK_DIM) blockDim = MIN_BLOCK_DIM;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_groups(static_cast<uint32_t>(groups));
    tiling.set_C(C);
    tiling.set_totalRows(totalRows);
    tiling.set_tileRows(tileRows);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04629Custom {
public:
    __aicore__ inline KernelCa6k04629Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t groups, uint32_t C,
                                uint32_t totalRows, uint32_t tileRows) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        groups_ = groups;
        C_ = C;
        totalRows_ = totalRows;
        tileRows_ = tileRows;
        rowLen_ = C;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalRows_ == 0 || tileRows_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes interleaved tiles
        uint32_t baseRow = blockIdx * tileRows_;
        while (baseRow < totalRows_) {
            // Compute tile size for this iteration
            uint32_t curRows = tileRows_;
            if (baseRow + curRows > totalRows_) {
                curRows = totalRows_ - baseRow;
            }
            uint32_t curLen = curRows * rowLen_;

            // Copy in
            uint32_t gmOffset = baseRow * rowLen_;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], curLen);
            inQueueX_.EnQue(xLocal);

            // Compute and copy out
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            xLocal = inQueueX_.DeQue<float>();

            // For each row perform channel shuffle
            for (uint32_t r = 0; r < curRows; r++) {
                uint32_t rowBase = r * rowLen_;
                uint32_t chunk = rowLen_ / groups_;  // channels per group
                for (uint32_t j = 0; j < chunk; j++) {
                    for (uint32_t g = 0; g < groups_; g++) {
                        float val = xLocal(rowBase + g * chunk + j);
                        yLocal(rowBase + j * groups_ + g) = val;
                    }
                }
            }

            outQueueY_.EnQue<float>(yLocal);
            inQueueX_.FreeTensor(xLocal);

            // Copy out
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[gmOffset], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);

            baseRow += blockNum * tileRows_;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t groups_;
    uint32_t C_;
    uint32_t totalRows_;
    uint32_t tileRows_;
    uint32_t rowLen_;
};

extern "C" __global__ __aicore__ void ca6k_04629_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04629Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.groups, tiling_data.C,
            tiling_data.totalRows, tiling_data.tileRows);
    op.Process();
}
'''
