kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 1;
constexpr float NEG_INF = -1e10f;
constexpr int32_t MAX_T = 2048;
constexpr int32_t MAX_U = 60;

class KernelCa6k00387Custom {
public:
    __aicore__ inline KernelCa6k00387Custom() {}

    __aicore__ inline void Init(GM_ADDR logProbs, GM_ADDR targets,
                                GM_ADDR inputLengths, GM_ADDR targetLengths,
                                GM_ADDR loss,
                                uint32_t batchSize, uint32_t maxTime,
                                uint32_t maxTargetLen, uint32_t numClasses,
                                int32_t blank, uint8_t reductionMean)
    {
        batchSize_ = batchSize;
        maxTime_ = maxTime;
        maxTargetLen_ = maxTargetLen;
        numClasses_ = numClasses;
        blank_ = blank;
        reductionMean_ = reductionMean;

        logProbsGm_.SetGlobalBuffer((__gm__ float*)logProbs, batchSize * maxTime * numClasses);
        targetsGm_.SetGlobalBuffer((__gm__ int32_t*)targets, batchSize * maxTargetLen);
        inputLengthsGm_.SetGlobalBuffer((__gm__ int32_t*)inputLengths, batchSize);
        targetLengthsGm_.SetGlobalBuffer((__gm__ int32_t*)targetLengths, batchSize);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inQueueLogProbs_, BUFFER_NUM, maxTime * numClasses * sizeof(float));
        pipe_.InitBuffer(inQueueTargets_, BUFFER_NUM, maxTargetLen * sizeof(int32_t));
        pipe_.InitBuffer(inQueueInputLen_, BUFFER_NUM, 1 * sizeof(int32_t));
        pipe_.InitBuffer(inQueueTargetLen_, BUFFER_NUM, 1 * sizeof(int32_t));
        pipe_.InitBuffer(tmpQueue_, 1, MAX_U * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, MAX_U * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= batchSize_) return;

        uint32_t baseLogProbs = blockIdx * maxTime_ * numClasses_;
        uint32_t baseTargets = blockIdx * maxTargetLen_;

        AscendC::LocalTensor<float> logProbsLocal = inQueueLogProbs_.AllocTensor<float>();
        AscendC::DataCopy(logProbsLocal, logProbsGm_[baseLogProbs], maxTime_ * numClasses_);
        inQueueLogProbs_.EnQue(logProbsLocal);

        AscendC::LocalTensor<int32_t> targetsLocal = inQueueTargets_.AllocTensor<int32_t>();
        AscendC::DataCopy(targetsLocal, targetsGm_[baseTargets], maxTargetLen_);
        inQueueTargets_.EnQue(targetsLocal);

        AscendC::LocalTensor<int32_t> inputLenLocal = inQueueInputLen_.AllocTensor<int32_t>();
        AscendC::DataCopy(inputLenLocal, inputLengthsGm_[blockIdx], 1);
        inQueueInputLen_.EnQue(inputLenLocal);

        AscendC::LocalTensor<int32_t> targetLenLocal = inQueueTargetLen_.AllocTensor<int32_t>();
        AscendC::DataCopy(targetLenLocal, targetLengthsGm_[blockIdx], 1);
        inQueueTargetLen_.EnQue(targetLenLocal);

        AscendC::LocalTensor<float> lp = inQueueLogProbs_.DeQue<float>();
        AscendC::LocalTensor<int32_t> tgt = inQueueTargets_.DeQue<int32_t>();
        AscendC::LocalTensor<int32_t> inLen = inQueueInputLen_.DeQue<int32_t>();
        AscendC::LocalTensor<int32_t> tgtLen = inQueueTargetLen_.DeQue<int32_t>();

        int32_t T = inLen.GetValue(0);
        int32_t L = tgtLen.GetValue(0);
        if (T <= 0 || L <= 0) {
            inQueueLogProbs_.FreeTensor(lp);
            inQueueTargets_.FreeTensor(tgt);
            inQueueInputLen_.FreeTensor(inLen);
            inQueueTargetLen_.FreeTensor(tgtLen);
            return;
        }

        int32_t extended[MAX_U];
        extended[0] = blank_;
        for (int32_t i = 0; i < L; ++i) {
            extended[2 * i + 1] = tgt.GetValue(i);
            extended[2 * i + 2] = blank_;
        }
        int32_t U = 2 * L + 1;

        AscendC::LocalTensor<float> alphaPrev = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> alphaCurr = tmpQueue2_.AllocTensor<float>();
        alphaCurr.SetValue(0, lp.GetValue(blank_));
        if (U > 1) {
            alphaCurr.SetValue(1, lp.GetValue(extended[1]));
        }
        for (int32_t u = 2; u < U; ++u) {
            alphaCurr.SetValue(u, NEG_INF);
        }

        for (int32_t t = 1; t < T; ++t) {
            AscendC::DataCopy(alphaPrev, alphaCurr, U);
            for (int32_t u = 0; u < U; ++u) {
                float logProb = lp.GetValue(t * numClasses_ + extended[u]);
                float best = NEG_INF;
                best = alphaPrev.GetValue(u) + logProb;
                if (u > 0) {
                    float val = alphaPrev.GetValue(u - 1) + logProb;
                    if (val > best) best = val;
                }
                if (u > 1 && extended[u] != blank_ && extended[u] != extended[u - 2]) {
                    float val = alphaPrev.GetValue(u - 2) + logProb;
                    if (val > best) best = val;
                }
                alphaCurr.SetValue(u, best);
            }
        }

        float maxVal = (alphaCurr.GetValue(U - 1) > alphaCurr.GetValue(U - 2)) ? alphaCurr.GetValue(U - 1) : alphaCurr.GetValue(U - 2);
        float sum = expf(alphaCurr.GetValue(U - 1) - maxVal) + expf(alphaCurr.GetValue(U - 2) - maxVal);
        float lossSample = -(maxVal + logf(sum));

        // Use atomic add for reduction (mean)
        AscendC::AtomicAdd(lossGm_[0], lossSample);

        tmpQueue_.FreeTensor(alphaPrev);
        tmpQueue2_.FreeTensor(alphaCurr);
        inQueueLogProbs_.FreeTensor(lp);
        inQueueTargets_.FreeTensor(tgt);
        inQueueInputLen_.FreeTensor(inLen);
        inQueueTargetLen_.FreeTensor(tgtLen);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueLogProbs_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTargets_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueInputLen_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTargetLen_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> logProbsGm_;
    AscendC::GlobalTensor<int32_t> targetsGm_;
    AscendC::GlobalTensor<int32_t> inputLengthsGm_;
    AscendC::GlobalTensor<int32_t> targetLengthsGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t batchSize_;
    uint32_t maxTime_;
    uint32_t maxTargetLen_;
    uint32_t numClasses_;
    int32_t blank_;
    uint8_t reductionMean_;
};

extern "C" __global__ __aicore__ void ca6k_00387_custom(GM_ADDR logProbs, GM_ADDR targets,
                                                         GM_ADDR inputLengths, GM_ADDR targetLengths,
                                                         GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00387Custom op;
    op.Init(logProbs, targets, inputLengths, targetLengths, loss,
            tiling_data.batchSize, tiling_data.maxTime,
            tiling_data.maxTargetLen, tiling_data.numClasses,
            tiling_data.blank, tiling_data.reductionMean);
    op.Process();
}
