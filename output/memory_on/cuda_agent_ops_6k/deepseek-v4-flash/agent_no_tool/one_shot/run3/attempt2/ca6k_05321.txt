#include "ca6k_05321_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k05321CustomTilingData tiling;

    const gert::StorageShape *xStorageShape = context->GetInputShape(0);
    const auto &originShape = xStorageShape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scalePtr = attrs->GetAttrPointer<float>(0);
    float scale = *scalePtr;

    uint32_t H_out = static_cast<uint32_t>(static_cast<double>(H) * scale);
    uint32_t W_out = static_cast<uint32_t>(static_cast<double>(W) * scale);

    uint32_t totalLength = N * C * H_out * W_out;

    // Determine tile size: 64x64 for input (so output 128x128 at scale=2)
    uint32_t tileH = 64;
    uint32_t tileW = 64;
    // Adjust if image smaller
    if (tileH > H) tileH = H;
    if (tileW > W) tileW = W;
    // UB budget: input tile with 1-pixel border: (tileH+1)*(tileW+1) floats
    // output tile: (tileH*scale)*(tileW*scale) floats
    // temporary: same size as output tile
    uint32_t inputTileSize = (tileH + 1) * (tileW + 1) * sizeof(float);
    uint32_t outputTileSize = static_cast<uint32_t>(tileH * scale) *
                              static_cast<uint32_t>(tileW * scale) * sizeof(float);
    uint32_t totalTileBytes = inputTileSize + 2 * outputTileSize;
    if (totalTileBytes > UB_BUDGET_BYTES) {
        // Reduce tile size
        tileH = 32;
        tileW = 32;
        inputTileSize = (tileH + 1) * (tileW + 1) * sizeof(float);
        outputTileSize = static_cast<uint32_t>(tileH * scale) *
                         static_cast<uint32_t>(tileW * scale) * sizeof(float);
        totalTileBytes = inputTileSize + 2 * outputTileSize;
    }

    uint32_t blockDim = 8; // fixed for this kernel
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_scale(scale);
    tiling.set_totalLength(totalLength);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    int64_t N = xShape->GetDim(0);
    int64_t C = xShape->GetDim(1);
    int64_t H = xShape->GetDim(2);
    int64_t W = xShape->GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scalePtr = attrs->GetAttrPointer<float>(0);
    float scale = *scalePtr;

    int64_t H_out = static_cast<int64_t>(static_cast<double>(H) * scale);
    int64_t W_out = static_cast<int64_t>(static_cast<double>(W) * scale);

    gert::Shape outShape;
    outShape.SetDimNum(4);
    outShape.SetDim(0, N);
    outShape.SetDim(1, C);
    outShape.SetDim(2, H_out);
    outShape.SetDim(3, W_out);
    gert::Shape *yShape = context->GetOutputShape(0);
    *yShape = outShape;
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
