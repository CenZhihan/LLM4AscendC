kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k02886Custom {
public:
    __aicore__ inline KernelCa6k02886Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength, float epsilon) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        eps_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, tileLen_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, tileLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        if (tileLen_ == 0 || totalLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        LoadWeightBias();

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            CopyIn(base, curLen);

            uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void LoadWeightBias() {
        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, tileLen_);
        AscendC::DataCopy(bLocal, bGm_, tileLen_);
        wQueue_.EnQue(wLocal);
        bQueue_.EnQue(bLocal);
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> mean = sQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> var = sQueue_.AllocTensor<float>();

        // mean = sum(x) / N
        AscendC::ReduceSum(xLocal, mean, false);
        float invN = 1.0f / static_cast<float>(len);
        AscendC::Muls(mean, mean, invN, 1);

        // var = sum(x^2) / N - mean^2
        AscendC::Mul(tmp, xLocal, xLocal, len);
        AscendC::ReduceSum(tmp, var, false);
        AscendC::Muls(var, var, invN, 1);
        AscendC::Mul(tmp, mean, mean, 1);
        AscendC::Sub(var, var, tmp, 1);

        // inv_std = 1 / sqrt(var + eps)
        AscendC::Adds(tmp, var, eps_, 1);
        AscendC::Rsqrt(tmp, tmp, 1);

        // normalize
        AscendC::Sub(yLocal, xLocal, mean, len);
        AscendC::Mul(yLocal, yLocal, tmp, len);

        AscendC::LocalTensor<float> wLocal = wQueue_.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = bQueue_.DeQue<float>();
        AscendC::Mul(yLocal, yLocal, wLocal, len);
        AscendC::Add(yLocal, yLocal, bLocal, len);
        wQueue_.EnQue(wLocal);
        bQueue_.EnQue(bLocal);

        sQueue_.FreeTensor(mean);
        sQueue_.FreeTensor(var);
        tmpQueue_.FreeTensor(tmp);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5};
};

extern "C" __global__ __aicore__ void ca6k02886_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02886Custom op;
    op.Init(x, weight, bias, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.epsilon);
    op.Process();
}
