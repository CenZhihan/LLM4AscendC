host_operator_src = r'''
#include "ca6k_04616_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cstdint>

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04616CustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(0);

    // Input shape: N, C, D, H, W
    auto input_shape = x_shape->GetOriginShape();
    int64_t N = input_shape.GetDim(0);
    int64_t C = input_shape.GetDim(1);
    int64_t D = input_shape.GetDim(2);
    int64_t H = input_shape.GetDim(3);
    int64_t W = input_shape.GetDim(4);

    int32_t padLeft   = static_cast<int32_t>(padding[0]);
    int32_t padRight  = static_cast<int32_t>(padding[1]);
    int32_t padTop    = static_cast<int32_t>(padding[2]);
    int32_t padBottom = static_cast<int32_t>(padding[3]);
    int32_t padFront  = static_cast<int32_t>(padding[4]);
    int32_t padBack   = static_cast<int32_t>(padding[5]);

    int64_t Wout = W + padLeft + padRight;
    int64_t Hout = H + padTop + padBottom;
    int64_t Dout = D + padFront + padBack;
    uint32_t totalOutputLength = static_cast<uint32_t>(N * C * Dout * Hout * Wout);

    // Simple block dim = 8 (number of available cores)
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_totalOutputLength(totalOutputLength);
    tiling.set_C(static_cast<uint32_t>(C));
    tiling.set_D(static_cast<uint32_t>(D));
    tiling.set_H(static_cast<uint32_t>(H));
    tiling.set_W(static_cast<uint32_t>(W));
    tiling.set_Dout(static_cast<uint32_t>(Dout));
));
));
