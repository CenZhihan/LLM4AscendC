static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k04670CustomTilingData tiling;

  const gert::StorageShape *xShape = context->GetInputShape(0);
  const uint32_t N = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
  const uint32_t C = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
  const uint32_t H = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
  const uint32_t W = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));
  const uint32_t totalLength = N * C * H * W;

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  const float *eps = attrs->GetAttrPointer<float>(0);
  uint32_t numFeatures = static_cast<uint32_t>(attrs->GetAttrPointer<int>(1)[0]);

  // UB budget: two copies of a tile (one for input, one for output) + temporaries
  constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
  constexpr uint32_t ALIGN_ELEMS = 2048;
  constexpr uint32_t HARD_CAP_ELEMS = 65536;

  uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t spatialPerChannel = H * W;

  tiling.set_tileLength(tileLen);

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04670Custom {
public:
  __aicore__ inline KernelCa6k04670Custom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                              uint32_t totalLength, uint32_t tileLength,
                              uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                              float eps) {
    totalLen_ = totalLength;
    tileLen_ = tileLength;
    N_ = N;
    C_ = C;
    H_ = H;
    W_ = W;
    eps_ = eps;
    spatialLen_ = H * W;
    xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    weightGm_.SetGlobalBuffer((__gm__ float*)weight, C_);
    biasGm_.SetGlobalBuffer((__gm__ float*)bias, C_);

    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    // Each block processes multiple channels in a grid-stride loop
    for (uint32_t c = blockIdx; c < C_; c += blockNum) {
      float mean, var;
      ComputeSumAndSumSq(c, mean, var);
      NormalizeChannel(c, mean, var);
    }
  }

private:
  __aicore__ inline void ComputeSumAndSumSq(uint32_t c, float &mean, float &var) {
    float sum = 0.0f;
    float sumSq = 0.0f;
    uint32_t channelOffset = c * spatialLen_;

    for (uint32_t n = 0; n < N_; ++n) {
      uint32_t batchOffset = n * C_ * spatialLen_;
      uint32_t base = batchOffset + channelOffset;
      uint32_t remaining = spatialLen_;
      uint32_t tileStart = 0;

      while (remaining > 0) {
        uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[base + tileStart], curLen);
        inQueueX_.EnQue(xLocal);

        xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
        AscendC::Mul(tLocal, xLocal, xLocal, curLen);

        AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumSqLocal = tmpQueue_.AllocTensor<float>();
        AscendC::ReduceSum(sumLocal, xLocal, curLen);
        AscendC::ReduceSum(sumSqLocal, tLocal, curLen);

        sum += sumLocal.GetValue(0);
        sumSq += sumSqLocal.GetValue(0);

        tmpQueue_.FreeTensor(tLocal);
        tmpQueue_.FreeTensor(sumLocal);
        tmpQueue_.FreeTensor(sumSqLocal);
        inQueueX_.FreeTensor(xLocal);

        tileStart += curLen;
        remaining -= curLen;
      }
    }

    float totalElements = static_cast<float>(N_ * spatialLen_);
    mean = sum / totalElements;
    float sqMean = sumSq / totalElements;
    var = sqMean - mean * mean;
  }

  __aicore__ inline void NormalizeChannel(uint32_t c, float mean, float var) {
    float invStd = 1.0f / AscendC::Sqrt(var + eps_);
    float weight = weightGm_.GetValue(c);
    float bias = biasGm_.GetValue(c);
    uint32_t channelOffset = c * spatialLen_;

    for (uint32_t n = 0; n < N_; ++n) {
      uint32_t batchOffset = n * C_ * spatialLen_;
      uint32_t base = batchOffset + channelOffset;
      uint32_t remaining = spatialLen_;
      uint32_t tileStart = 0;

      while (remaining > 0) {
        uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[base + tileStart], curLen);
        inQueueX_.EnQue(xLocal);

        xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        AscendC::Sub(tLocal, xLocal, mean, curLen);
        AscendC::Muls(tLocal, tLocal, invStd, curLen);
        AscendC::Muls(yLocal, tLocal, weight, curLen);
        AscendC::Adds(yLocal, yLocal, bias, curLen);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);

        yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[base + tileStart], yLocal, curLen);
        outQueueY_.FreeTensor(yLocal);

        tileStart += curLen;
        remaining -= curLen;
      }
    }
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
  AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> yGm_;
  AscendC::GlobalTensor<float> weightGm_;
  AscendC::GlobalTensor<float> biasGm_;
  uint32_t totalLen_{0};
  uint32_t tileLen_{0};
  uint32_t N_{0}, C_{0}, H_{0}, W_{0};
  uint32_t spatialLen_{0};
  float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_04670_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelCa6k04670Custom op;
  op.Init(x, y, weight, bias, tiling_data.totalLength, tiling_data.tileLength,
          tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W, tiling_data.eps);
  op.Process();
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_04670_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_features, affine=True, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.eps = eps
        self.affine = affine
        if affine:
            self.weight = torch.nn.Parameter(torch.ones(num_features))
            self.bias = torch.nn.Parameter(torch.zeros(num_features))
        else:
            self.register_parameter('weight', None)
            self.register_parameter('bias', None)

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        if self.affine:
            return fn(x, self.weight, self.bias, self.eps)
        else:
            # pass ones/zeros as weight/bias when not affine
            ones = torch.ones(1, device=x.device).expand(self.num_features)
            zeros = torch.zeros(1, device=x.device).expand(self.num_features)
            return fn(x, ones, zeros, self.eps)
