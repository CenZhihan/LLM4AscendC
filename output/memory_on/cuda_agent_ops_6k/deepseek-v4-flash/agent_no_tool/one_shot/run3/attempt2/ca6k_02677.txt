#include "ca6k_02677_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k02677CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // 5 floats per element: x1, x2, y, tmpA, tmpB
    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    // Direct assignment to public fields (with underscore suffix per macro)
    tiling.totalLength_ = totalLength;
    tiling.tileLength_ = tileLen;
    tiling.blockDim_ = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k02677Custom {
public:
    __aicore__ inline KernelCa6k02677Custom() {}

    __aicore__ inline void Init(GM_ADDR x1, GM_ADDR x2, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        x1Gm_.SetGlobalBuffer((__gm__ float*)x1, totalLen_);
        x2Gm_.SetGlobalBuffer((__gm__ float*)x2, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX1_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueX2_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueA_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueB_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> x1Local = inQueueX1_.AllocTensor<float>();
        AscendC::DataCopy(x1Local, x1Gm_[gmOffset], len);
        inQueueX1_.EnQue(x1Local);

        AscendC::LocalTensor<float> x2Local = inQueueX2_.AllocTensor<float>();
        AscendC::DataCopy(x2Local, x2Gm_[gmOffset], len);
        inQueueX2_.EnQue(x2Local);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> x1Local = inQueueX1_.DeQue<float>();
        AscendC::LocalTensor<float> x2Local = inQueueX2_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpA = tmpQueueA_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpB = tmpQueueB_.AllocTensor<float>();

        // stable logaddexp: max + log(exp(x1-max) + exp(x2-max))
        AscendC::Max(tmpA, x1Local, x2Local, len);                // tmpA = max(x1,x2)
        AscendC::Sub(tmpB, x1Local, tmpA, len);                   // tmpB = x1 - max
        AscendC::Exp(tmpB, tmpB, len);                             // tmpB = exp(x1-max)
        AscendC::Sub(x1Local, x2Local, tmpA, len);                // reuse x1Local: x1Local = x2 - max (safe, x1 no longer needed)
        AscendC::Exp(x1Local, x1Local, len);                       // x1Local = exp(x2-max)
        AscendC::Add(tmpB, tmpB, x1Local, len);                    // tmpB = exp(x1-max) + exp(x2-max)
        AscendC::Log(tmpB, tmpB, len);                             // tmpB = log(sum)
        AscendC::Add(yLocal, tmpA, tmpB, len);                     // y = max + log(sum)

        tmpQueueA_.FreeTensor(tmpA);
        tmpQueueB_.FreeTensor(tmpB);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX1_.FreeTensor(x1Local);
        inQueueX2_.FreeTensor(x2Local);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX1_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX2_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueA_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueB_;

    AscendC::GlobalTensor<float> x1Gm_;
    AscendC::GlobalTensor<float> x2Gm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_02677_custom(GM_ADDR x1, GM_ADDR x2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02677Custom op;
    op.Init(x1, x2, y, tiling_data.totalLength_, tiling_data.tileLength_);
    op.Process();
}
'''
