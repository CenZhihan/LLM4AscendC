host_operator_src = '''
#include "ca6k_00517_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00517CustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    uint32_t dimCount = inShape->GetDimCount();
    // Expect 3D shape: (N, C, L)
    if (dimCount != 3) return ge::GRAPH_FAILED;
    uint32_t N = static_cast<uint32_t>(inShape->GetDim(0));
    uint32_t C = static_cast<uint32_t>(inShape->GetDim(1));
    uint32_t L = static_cast<uint32_t>(inShape->GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *paddingAttr = attrs->GetAttrPointer<int64_t>(0);
    uint32_t pad = static_cast<uint32_t>(*paddingAttr);

    uint32_t totalSlices = N * C;
    uint32_t sliceLen = L;
    uint32_t sliceOutLen = L + 2 * pad;

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = maxTile;
    if (sliceLen < tileLen) {
        tileLen = sliceLen;
        tileLen = AlignDown(tileLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = sliceLen;
    }

    tiling.set_totalSlices(totalSlices);
    tiling.set_sliceLen(sliceLen);
    tiling.set_sliceOutLen(sliceOutLen);
    tiling.set_padding(pad);
    tiling.set_tileLen(tileLen);

    uint32_t numTiles = CeilDiv(totalSlices, 1u);
    uint32_t blockDim = (numTiles < 8u) ? 8u : numTiles;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    // Adjust last dimension: L_out = L + 2*padding
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *paddingAttr = attrs->GetAttrPointer<int64_t>(0);
    int64_t newL = x_shape->GetDim(2) + 2 * (*paddingAttr);
    y_shape->SetDim(2, newL);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
}  // namespace ge

namespace ops {
class Ca6k00517Custom : public OpDef {
public:
    explicit Ca6k00517Custom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Attr("padding").AttrType(OPTIONAL).Int(16);
        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(Ca6k00517Custom);
}  // namespace ops

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00517Custom {
public:
    __aicore__ inline KernelCa6k00517Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalSlices, uint32_t sliceLen,
                                uint32_t sliceOutLen, uint32_t padding,
                                uint32_t tileLen) {
        totalSlices_ = totalSlices;
        sliceLen_ = sliceLen;
        sliceOutLen_ = sliceOutLen;
        padding_ = padding;
        tileLen_ = tileLen;
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        uint32_t maxBufSize = (padding_ > tileLen_) ? padding_ : tileLen_;
        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, maxBufSize * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalSlices_ == 0 || sliceLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t s = blockIdx; s < totalSlices_; s += blockNum) {
            ProcessSlice(s);
        }
    }

private:
    __aicore__ inline void ProcessSlice(uint32_t sliceIdx) {
        uint32_t inBase = sliceIdx * sliceLen_;
        uint32_t outBase = sliceIdx * sliceOutLen_;

        // Copy tail (last padding elements of slice) to output start
        if (padding_ > 0) {
            AscendC::LocalTensor<float> tailBuf = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(tailBuf, xGm_[inBase + sliceLen_ - padding_], padding_);
            AscendC::DataCopy(yGm_[outBase], tailBuf, padding_);
            tmpQueue_.FreeTensor(tailBuf);
        }

        // Copy middle (entire slice) to output offset padding
        uint32_t midInBase = inBase;
        uint32_t midOutBase = outBase + padding_;
        uint32_t midRemaining = sliceLen_;
        uint32_t midOffset = 0;
        while (midRemaining > 0) {
            uint32_t curTile = (midRemaining > tileLen_) ? tileLen_ : midRemaining;
            AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[midInBase + midOffset], curTile);
            AscendC::DataCopy(yGm_[midOutBase + midOffset], xLocal, curTile);
            outQueue_.FreeTensor(yLocal);
            inQueue_.FreeTensor(xLocal);
            midOffset += curTile;
            midRemaining -= curTile;
        }

        // Copy head (first padding elements of slice) to output end
        if (padding_ > 0) {
            uint32_t headOutStart = outBase + sliceLen_ + padding_;
            AscendC::LocalTensor<float> headBuf = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(headBuf, xGm_[inBase], padding_);
            AscendC::DataCopy(yGm_[headOutStart], headBuf, padding_);
            tmpQueue_.FreeTensor(headBuf);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalSlices_ {0};
    uint32_t sliceLen_ {0};
    uint32_t sliceOutLen_ {0};
    uint32_t padding_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00517_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00517Custom op;
    op.Init(x, y,
            tiling_data.totalSlices, tiling_data.sliceLen,
            tiling_data.sliceOutLen, tiling_data.padding,
            tiling_data.tileLen);
    op.Process();
}
'''
