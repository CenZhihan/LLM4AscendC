namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04596CustomTilingData tiling;
    uint32_t totalLength = static_cast<uint32_t>(
        context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(1);

    context->SetBlockDim(1);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04596Custom {
public:
    __aicore__ inline KernelCa6k04596Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR count,
                                uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        countGm_.SetGlobalBuffer((__gm__ int32_t*)count, 1);

        pipe_.InitBuffer(inQueueX_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx != 0) return; // single block processing

        uint32_t globalOffset = 0;

        for (uint32_t base = 0; base < totalLen_; base += tileLen_) {
            uint32_t curLen = (base + tileLen_ <= totalLen_) ? tileLen_ : (totalLen_ - base);

            // Load tile from x and mask
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
            AscendC::DataCopy(maskLocal, maskGm_[base], curLen);
            inQueueMask_.EnQue(maskLocal);
            maskLocal = inQueueMask_.DeQue<float>();

            AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();

            uint32_t localCount = 0;
            for (uint32_t i = 0; i < curLen; ++i) {
                AscendC::Scalar<float> mv = maskLocal.GetValue(i);
                float maskVal = mv.GetValue();
                if (maskVal > 0.0f) {
                    AscendC::Scalar<float> xv = xLocal.GetValue(i);
                    outLocal.SetValue(localCount, xv.GetValue());
                    ++localCount;
                }
            }

            if (localCount > 0) {
                AscendC::DataCopy(yGm_[globalOffset], outLocal, localCount);
                globalOffset += localCount;
            }

            inQueueX_.FreeTensor(xLocal);
            inQueueMask_.FreeTensor(maskLocal);
            outQueueY_.FreeTensor(outLocal);
        }

        countGm_.SetValue(0, static_cast<int32_t>(globalOffset));
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> maskGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<int32_t> countGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04596_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR count, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04596Custom op;
    op.Init(x, mask, y, count, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
