kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelAvgPool3d {
public:
    __aicore__ inline KernelAvgPool3d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t N, int32_t C, int32_t D, int32_t H, int32_t W,
                                int32_t kernelD, int32_t kernelH, int32_t kernelW,
                                int32_t strideD, int32_t strideH, int32_t strideW,
                                int32_t padD, int32_t padH, int32_t padW,
                                int32_t outD, int32_t outH, int32_t outW)
    {
        N_ = N; C_ = C; D_ = D; H_ = H; W_ = W;
        kernelD_ = kernelD; kernelH_ = kernelH; kernelW_ = kernelW;
        strideD_ = strideD; strideH_ = strideH; strideW_ = strideW;
        padD_ = padD; padH_ = padH; padW_ = padW;
        outD_ = outD; outH_ = outH; outW_ = outW;
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint32_t)(N_ * C_ * outD_ * outH_ * outW_));
        // Double buffer for reading rows from global memory
        pipe_.InitBuffer(rowQueue_, BUFFER_NUM, (uint32_t)kernelW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t totalOutElems = static_cast<uint32_t>(N_ * C_ * outD_ * outH_ * outW_);
        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutElems) return;

        while (true) {
            uint32_t curTileLen = tileLen_;
            if (base + curTileLen > totalOutElems) curTileLen = totalOutElems - base;
            for (uint32_t i = 0; i < curTileLen; i++) {
                uint32_t outIdx = base + i;
                ComputeOne(outIdx);
            }
            base += strideElems;
            if (base >= totalOutElems) break;
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t outIdx)
    {
        int32_t ow = outIdx % outW_;
        int32_t remaining = outIdx / outW_;
        int32_t oh = remaining % outH_;
        remaining = remaining / outH_;
        int32_t od = remaining % outD_;
        remaining = remaining / outD_;
        int32_t c = remaining % C_;
        int32_t n = remaining / C_;

        int32_t startD = od * strideD_ - padD_;
        int32_t startH = oh * strideH_ - padH_;
        int32_t startW = ow * strideW_ - padW_;

        float sum = 0.0f;
        uint32_t count = 0;

        for (int32_t kd = 0; kd < kernelD_; kd++) {
            int32_t d = startD + kd;
            if (d < 0 || d >= D_) continue;
            for (int32_t kh = 0; kh < kernelH_; kh++) {
                int32_t h = startH + kh;
                if (h < 0 || h >= H_) continue;
                int32_t wStart = startW;
                int32_t wEnd = startW + kernelW_;
                int32_t actualStart = (wStart < 0) ? 0 : wStart;
                int32_t actualEnd = (wEnd > W_) ? W_ : wEnd;
                if (actualStart >= actualEnd) continue;
                uint32_t rowLen = static_cast<uint32_t>(actualEnd - actualStart);
                uint32_t rowOffset = (uint32_t)(((n * C_ + c) * D_ + d) * H_ + h) * W_ + actualStart;
                
                // allocate and copy row
                AscendC::LocalTensor<float> rowLocal = rowQueue_.AllocTensor<float>();
                AscendC::DataCopy(rowLocal, xGm_[rowOffset], rowLen);
                
                // accumulate
                for (uint32_t w = 0; w < rowLen; w++) {
                    sum += rowLocal(w);
                }
                count += rowLen;
                rowQueue_.FreeTensor(rowLocal);
            }
        }
        yGm_[outIdx] = sum / ((float)count);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> rowQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_;
    uint32_t tileLen_;
    int32_t N_, C_, D_, H_, W_;
    int32_t kernelD_, kernelH_, kernelW_;
    int32_t strideD_, strideH_, strideW_;
    int32_t padD_, padH_, padW_;
    int32_t outD_, outH_, outW_;
};

extern "C" __global__ __aicore__ void ca6k_04607_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAvgPool3d op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C, tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW);
    op.Process();
}
"""

#include <torch/library.h>

at::Tensor avgpool3d_impl_npu(const at::Tensor& self,
                               std::vector<int64_t> kernel_size,
                               std::vector<int64_t> stride,
                               std::vector<int64_t> padding) {
    auto N = self.size(0);
    auto C = self.size(1);
    auto D = self.size(2);
    auto H = self.size(3);
    auto W = self.size(4);
    int64_t outD = (D + 2 * padding[0] - kernel_size[0]) / stride[0] + 1;
    int64_t outH = (H + 2 * padding[1] - kernel_size[1]) / stride[1] + 1;
    int64_t outW = (W + 2 * padding[2] - kernel_size[2]) / stride[2] + 1;
    at::Tensor result = at::empty({N, C, outD, outH, outW}, self.options());
    EXEC_NPU_CMD(aclnnCa6k04607Custom, self, kernel_size, stride, padding, result);
    return result;
}

    m.impl("ca6k_04607_custom", &avgpool3d_impl_npu);
