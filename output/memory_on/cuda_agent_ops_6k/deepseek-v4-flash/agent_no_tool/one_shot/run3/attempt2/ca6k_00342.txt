kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;   // double buffering for input rows
constexpr int32_t KERNEL_SIZE = 2;

class KernelCa6k00342Custom {
public:
    __aicore__ inline KernelCa6k00342Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                int32_t N, int32_t C, int32_t H, int32_t W,
                                int32_t H_out, int32_t W_out)
    {
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        H_out_ = H_out;
        W_out_ = W_out;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * H_out_ * W_out_);

        // Buffer for two consecutive input rows, double buffered
        pipe_.InitBuffer(inRowsBuf_, BUFFER_NUM, 2 * W_ * sizeof(float));
        // Buffer for one output row
        pipe_.InitBuffer(outRowBuf_, 1, W_out_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const int32_t blockIdx = AscendC::GetBlockIdx();   // 0 .. N*C-1
        int32_t n = blockIdx / C_;
        int32_t c = blockIdx % C_;

        if (n >= N_) return;

        int32_t inRowStride = W_;
        int32_t outRowStride = W_out_;

        // Loop over output rows
        for (int32_t h_out = 0; h_out < H_out_; ++h_out) {
            int32_t rowA = 2 * h_out;
            int32_t rowB = 2 * h_out + 1;

            // Allocate buffer for two input rows
            AscendC::LocalTensor<float> inRows = inRowsBuf_.AllocTensor<float>();
            // Copy rowA
            int32_t offsetA = (n * C_ + c) * H_ * W_ + rowA * W_;
            AscendC::DataCopy(inRows, xGm_[offsetA], W_);
            // Copy rowB into second half of buffer
            int32_t offsetB = (n * C_ + c) * H_ * W_ + rowB * W_;
            AscendC::DataCopy(inRows[W_], xGm_[offsetB], W_);

            // Compute output row
            AscendC::LocalTensor<float> outRow = outRowBuf_.AllocTensor<float>();
            for (int32_t w_out = 0; w_out < W_out_; ++w_out) {
                int32_t colA = 2 * w_out;
                int32_t colB = 2 * w_out + 1;
                float v00 = inRows(colA);
                float v01 = inRows(colB);
                float v10 = inRows[W_ + colA];
                float v11 = inRows[W_ + colB];

                float max = v00;
                if (v01 > max) max = v01;
                if (v10 > max) max = v10;
                if (v11 > max) max = v11;
                outRow(w_out) = max;
            }

            // Write output row to global
            int32_t outOffset = (n * C_ + c) * H_out_ * W_out_ + h_out * W_out_;
            AscendC::DataCopy(yGm_[outOffset], outRow, W_out_);

            outRowBuf_.FreeTensor(outRow);
            inRowsBuf_.FreeTensor(inRows);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inRowsBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> outRowBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    int32_t N_, C_, H_, W_, H_out_, W_out_;
};

extern "C" __global__ __aicore__ void ca6k_00342_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00342Custom op;
    op.Init(x, y, tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.H_out, tiling_data.W_out);
    op.Process();
}
'''
