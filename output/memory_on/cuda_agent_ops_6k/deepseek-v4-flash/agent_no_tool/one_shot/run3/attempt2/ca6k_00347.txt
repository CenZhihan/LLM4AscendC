        blockDim = 16;

    tiling.channels_ = channels;

    tiling.width_ = width;
    tiling.kernelH_ = kernelH;
    tiling.kernelW_ = kernelW;
    tiling.dilationH_ = dilationH;
    tiling.dilationW_ = dilationW;
    tiling.stride_ = stride;
    tiling.padding_ = padding;
    tiling.outputH_ = outputH;
    tiling.outputW_ = outputW;

#include "kernel_operator.h"

class KernelCa6k00347Custom {
public:
    __aicore__ inline KernelCa6k00347Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                Ca6k00347CustomTilingData td)
    {
        xGm_ = (__gm__ float*)x;
        yGm_ = (__gm__ float*)y;

        batchSize_ = td.batchSize_;
        channels_ = td.channels_;
        height_ = td.height_;
        width_ = td.width_;
        kernelH_ = td.kernelH_;
        kernelW_ = td.kernelW_;
        dilationH_ = td.dilationH_;
        dilationW_ = td.dilationW_;
        stride_ = td.stride_;
        padding_ = td.padding_;
        outputH_ = td.outputH_;
        outputW_ = td.outputW_;
        outputRows_ = td.outputRows_;
        totalOutputElems_ = td.totalOutputElems_;
        blockDim_ = td.blockDim_;
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();

        for (uint64_t elemIdx = blockIdx; elemIdx < totalOutputElems_; elemIdx += blockNum) {
            // compute batch, row, spatial position from flattened index
            uint32_t batch = elemIdx / (outputRows_ * outputH_ * outputW_);
            uint32_t rem = elemIdx % (outputRows_ * outputH_ * outputW_);
            uint32_t r = rem / (outputH_ * outputW_);         // which output row (channel group)
            uint32_t p = rem % (outputH_ * outputW_);         // spatial position
            uint32_t h_out = p / outputW_;
            uint32_t w_out = p % outputW_;

            // The output pixel at (batch, r, h_out, w_out) corresponds to a single value from input
            // Actually unfold produces a column: each output element corresponds to one entry in the sliding window.
            // We need to compute the exact input index.
            // r = c * kernelH_ * kernelW_ + kh * kernelW_ + kw
            uint32_t c = r / (kernelH_ * kernelW_);
            uint32_t kr = (r % (kernelH_ * kernelW_)) / kernelW_;
            uint32_t kc = (r % (kernelH_ * kernelW_)) % kernelW_;

            uint32_t h_in = h_out * stride_ + kr * dilationH_ - padding_;
            uint32_t w_in = w_out * stride_ + kc * dilationW_ - padding_;

            float val = 0.0f;
            if (h_in < height_ && w_in < width_) {
                uint64_t_t_t
