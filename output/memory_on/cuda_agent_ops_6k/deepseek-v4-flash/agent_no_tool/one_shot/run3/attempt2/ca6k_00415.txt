namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00415CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *numGroups = attrs->GetAttrPointer<int64_t>(0);
    const float *eps = attrs->GetAttrPointer<float>(1);

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(shape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(shape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(shape.GetDim(3));

    // tileLen from UB budget: we need 3 tiles (in, out, sum/sumsq temporary) + small workspace
    // roughly 4 * tileLen floats = 16 * tileLen bytes
    uint32_t maxTile = UB_BUDGET_BYTES / 16;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t spatial = H * W;
    uint32_t tileLen = maxTile;
    if (spatial < tileLen) {
        tileLen = AlignDown(spatial, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = spatial;
    }

    // Each block handles one (batch, group) pair
    uint32_t totalBlocks = N * static_cast<uint32_t>(*numGroups);
    uint32_t blockDim = totalBlocks;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 512) blockDim = 512;

    context->SetBlockDim(blockDim);

    // Use setter methods (generated by TILING_DATA_FIELD_DEF)
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_numGroups(static_cast<uint32_t>(*numGroups));
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00415Custom {
public:
    __aicore__ inline KernelCa6k00415Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t numGroups, uint32_t N, uint32_t C,
                                uint32_t H, uint32_t W, float eps) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        numGroups_ = numGroups;
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        eps_ = eps;
        groupSize_ = C / numGroups;
        spatialSize_ = H * W;
        numTilePerChannel_ = (spatialSize_ + tileLen_ - 1) / tileLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, C_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, C_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        // Buffer for per-channel gamma and beta (groupSize floats each)
        pipe_.InitBuffer(gammaBuffer_, 1, groupSize_ * sizeof(float));
        pipe_.InitBuffer(betaBuffer_, 1, groupSize_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t totalBlocks = N_ * numGroups_;
        if (blockIdx >= totalBlocks) return;

        uint32_t batchIdx = blockIdx / numGroups_;
        uint32_t groupIdx = blockIdx % numGroups_;
        uint32_t groupStartChan = groupIdx * groupSize_;

        // Load gamma and beta for this group into local buffers
        AscendC::LocalTensor<float> gammaLocal = gammaBuffer_.AllocTensor<float>();
        AscendC::LocalTensor<float> betaLocal = betaBuffer_.AllocTensor<float>();
        AscendC::DataCopy(gammaLocal, gammaGm_[groupStartChan], groupSize_);
        AscendC::DataCopy(betaLocal, betaGm_[groupStartChan], groupSize_);

        // --- Pass 1: compute group-level mean and variance ---
        // We'll accumulate sum and sumsq in scalar registers (using local tensors for reduce)
        AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumsqLocal = tmpQueue_.AllocTensor<float>();
        sumLocal.SetValue(0, 0.0f);
        sumsqLocal.SetValue(0, 0.0f);

        for (uint32_t c = 0; c < groupSize_; ++c) {
            uint32_t globalChan = groupStartChan + c;
            uint32_t chanOffset = batchIdx * C_ * spatialSize_ + globalChan * spatialSize_;
            for (uint32_t t = 0; t < numTilePerChannel_; ++t) {
                uint32_t tileBase = chanOffset + t * tileLen_;
                uint32_t curLen = tileLen_;
                if (t == numTilePerChannel_ - 1) {
                    curLen = spatialSize_ - t * tileLen_;
                }

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[tileBase], curLen);

                // sum += reduce_sum(x)
                AscendC::ReduceSum(sumLocal, xLocal, false, curLen);
                // sumsq += reduce_sum(x * x)
                AscendC::Mul(tmpQueue_.Get<float>(), xLocal, xLocal, curLen);
                AscendC::ReduceSum(sumsqLocal, tmpQueue_.Get<float>(), false, curLen);

                inQueueX_.FreeTensor(xLocal);
            }
        }

        float totalElems = static_cast<float>(groupSize_ * spatialSize_);
        float mean = sumLocal.GetValue(0) / totalElems;
        float var = sumsqLocal.GetValue(0) / totalElems - mean * mean;
        float invStd = 1.0f / AscendC::Sqrt(var + eps_);

        tmpQueue_.FreeTensor(sumLocal);
        tmpQueue_.FreeTensor(sumsqLocal);

        // --- Pass 2: normalize and apply affine ---
        for (uint32_t c = 0; c < groupSize_; ++c) {
            uint32_t globalChan = groupStartChan + c;
            uint32_t chanOffset = batchIdx * C_ * spatialSize_ + globalChan * spatialSize_;
            float gammaVal = gammaLocal.GetValue(c);
            float betaVal = betaLocal.GetValue(c);

            for (uint32_t t = 0; t < numTilePerChannel_; ++t) {
                uint32_t tileBase = chanOffset + t * tileLen_;
                uint32_t curLen = tileLen_;
                if (t == numTilePerChannel_ - 1) {
                    curLen = spatialSize_ - t * tileLen_;
                }

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[tileBase], curLen);

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // y = gamma * ( (x - mean) * invStd ) + beta
                AscendC::Sub(yLocal, xLocal, mean, curLen);
                AscendC::Mul(yLocal, yLocal, invStd, curLen);
                AscendC::Muls(yLocal, yLocal, gammaVal, curLen);
                AscendC::Adds(yLocal, yLocal, betaVal, curLen);

                AscendC::DataCopy(yGm_[tileBase], yLocal, curLen);

                inQueueX_.FreeTensor(xLocal);
                outQueueY_.FreeTensor(yLocal);
            }
        }

        gammaBuffer_.FreeTensor(gammaLocal);
        betaBuffer_.FreeTensor(betaLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> gammaBuffer_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> betaBuffer_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t numGroups_{0};
    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t groupSize_{0};
    uint32_t spatialSize_{0};
    uint32_t numTilePerChannel_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_00415_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                                         GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00415Custom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.numGroups, tiling_data.N, tiling_data.C,
            tiling_data.H, tiling_data.W, tiling_data.eps);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00415_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_channels: int, num_groups: int, *init_inputs):
        super(ModelNew, self).__init__()
        self.num_channels = num_channels
        self.num_groups = num_groups
        # Learnable gamma (weight) and beta (bias) as in nn.GroupNorm
        self.gamma = torch.nn.Parameter(torch.ones(num_channels))
        self.beta = torch.nn.Parameter(torch.zeros(num_channels))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Move parameters to same device as input
        gamma = self.gamma.to(x.device)
        beta = self.beta.to(x.device)
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, gamma, beta, self.num_groups, 1e-5)
