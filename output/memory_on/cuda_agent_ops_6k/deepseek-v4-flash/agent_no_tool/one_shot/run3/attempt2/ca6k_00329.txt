kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00329Custom {
public:
    __aicore__ inline KernelCa6k00329Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength, uint32_t channels,
                                uint32_t spatialCount, float eps)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        channels_ = channels;
        spatialCount_ = spatialCount;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, channels_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, channels_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        for (uint32_t c = 0; c < channels_; ++c) {
            float mean = 0.0f;
            float var = 0.0f;
            ComputeChannelStats(c, mean, var);
            NormalizeChannel(c, mean, var);
        }
    }

private:
    __aicore__ inline void ComputeChannelStats(uint32_t c, float& mean, float& var)
    {
        float sum = 0.0f;
        float sumSq = 0.0f;
        uint32_t base = c * spatialCount_;

        for (uint32_t offset = 0; offset < spatialCount_; offset += tileLen_) {
            uint32_t len = (offset + tileLen_ <= spatialCount_) ? tileLen_ : (spatialCount_ - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], len);
            for (uint32_t i = 0; i < len; ++i) {
                float val = xLocal(i);
                sum += val;
                sumSq += val * val;
            }
            inQueueX_.FreeTensor(xLocal);
        }

        float count = static_cast<float>(spatialCount_);
        mean = sum / count;
        var = sumSq / count - mean * mean;
        if (var < 0.0f) var = 0.0f;
    }

    __aicore__ inline void NormalizeChannel(uint32_t c, float mean, float var)
    {
        float scale = 1.0f / sqrtf(var + eps_);
        float gamma = weightGm_.GetValue(c);
        float beta = biasGm_.GetValue(c);
        uint32_t base = c * spatialCount_;

        for (uint32_t offset = 0; offset < spatialCount_; offset += tileLen_) {
            uint32_t len = (offset + tileLen_ <= spatialCount_) ? tileLen_ : (spatialCount_ - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], len);
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            for (uint32_t i = 0; i < len; ++i) {
                float normalized = (xLocal(i) - mean) * scale;
                yLocal(i) = normalized * gamma + beta;
            }
            AscendC::DataCopy(yGm_[base + offset], yLocal, len);
            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t channels_{0};
    uint32_t spatialCount_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_00329_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00329Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.channels, tiling_data.spatialCount,
            tiling_data.eps);
    op.Process();
}
