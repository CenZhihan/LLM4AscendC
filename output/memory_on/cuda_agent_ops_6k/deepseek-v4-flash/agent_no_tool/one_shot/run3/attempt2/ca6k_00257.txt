static constexpr uint32_t ALIGN = 32;

static ge::graphStatus TilingFunc(gert::TilingContext *context)

    Ca6k00257CustomTilingData tiling;

    const int32_t *batchPtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *heightPtr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *widthPtr = attrs->GetAttrPointer<int32_t>(2);
    const float *scalePtr = attrs->GetAttrPointer<float>(3);
    const bool *alignCornersPtr = attrs->GetAttrPointer<bool>(4);

    uint32_t batch = static_cast<uint32_t>(*batchPtr);
    uint32_t height = static_cast<uint32_t>(*heightPtr);
    uint32_t width = static_cast<uint32_t>(*widthPtr);
    float scale = *scalePtr;
    int32_t align_corners = *alignCornersPtr ? 1 : 0;

    uint32_t totalRows = batch * height;
    uint32_t blockDim = (totalRows + ALIGN - 1) / ALIGN;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.totalRows_ = totalRows;
    tiling.height_ = height;
    tiling.width_ = width;
    tiling.batch_ = batch;
    tiling.scale_ = scale;
    tiling.align_corners_ = align_corners;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelCa6k00257Custom {
public:
    __aicore__ inline KernelCa6k00257Custom() {}

    __aicore__ inline void Init(GM_ADDR theta, GM_ADDR grid,
                                uint32_t totalRows, uint32_t height, uint32_t width,
                                uint32_t batch, float scale, int32_t align_corners)
    {
        totalRows_ = totalRows;
        height_ = height;
        width_ = width;
        batch_ = batch;
        scale_ = scale;
        alignCorners_ = align_corners;

        thetaGm_.SetGlobalBuffer((__gm__ float*)theta, batch * 6);
        gridGm_.SetGlobalBuffer((__gm__ float*)grid, totalRows * width * 2);
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t rowIdx = blockIdx; rowIdx < totalRows_; rowIdx += blockNum) {
            uint32_t b = rowIdx / height_;
            uint32_t h = rowIdx % height_;

            // Load theta for this batch (6 floats)
            float theta00, theta01, theta02, theta10, theta11, theta12;
            theta00 = thetaGm_[b * 6 + 0];
            theta01 = thetaGm_[b * 6 + 1];
            theta02 = thetaGm_[b * 6 + 2];
            theta10 = thetaGm_[b * 6 + 3];
            theta11 = thetaGm_[b * 6 + 4];
            theta12 = thetaGm_[b * 6 + 5];

            uint32_t baseIdx = (b * height_ + h) * width_ * 2;

            for (uint32_t w = 0; w < width_; ++w) {
                float x_out = static_cast<float>(w);
                float y_out = static_cast<float>(h);

                float x_norm, y_norm;
                if (alignCorners_) {
                    if (width_ > 1) {
                        x_norm = x_out / static_cast<float>(width_ - 1) * 2.0f - 1.0f;
                    } else {
                        x_norm = 0.0f;
                    }
                    if (height_ > 1) {
                        y_norm = y_out / static_cast<float>(height_ - 1) * 2.0f - 1.0f;
                    } else {
                        y_norm = 0.0f;
                    }
                } else {
                    x_norm = (x_out + 0.5f) / static_cast<float>(width_) * 2.0f - 1.0f;
                    y_norm = (y_out + 0.5f) / static_cast<float>(height_) * 2.0f - 1.0f;
                }

                float x_src = theta00 * x_norm + theta01 * y_norm + theta02;
                float y_src = theta10 * x_norm + theta11 * y_norm + theta12;

                x_src *= scale_;
                y_src *= scale_;

                gridGm_[baseIdx + w * 2 + 0] = x_src;
                gridGm_[baseIdx + w * 2 + 1] = y_src;
            }
        }
    }

private:
    AscendC::GlobalTensor<float> thetaGm_;
    AscendC::GlobalTensor<float> gridGm_;

    uint32_t totalRows_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    uint32_t batch_ {0};
    float scale_ {0.5f};
    int32_t alignCorners_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00257_custom(GM_ADDR theta, GM_ADDR grid, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00257Custom op;
    op.Init(theta, grid, tiling_data.totalRows, tiling_data.height, tiling_data.width,
            tiling_data.batch, tiling_data.scale, tiling_data.align_corners);
    op.Process();
}
