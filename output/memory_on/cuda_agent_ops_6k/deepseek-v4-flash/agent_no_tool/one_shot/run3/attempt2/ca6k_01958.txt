kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;
constexpr float ZERO = 0.0f;
constexpr float ONE = 1.0f;

class KernelCa6k01958Custom {
public:
    __aicore__ inline KernelCa6k01958Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalLen, uint32_t batchSize, uint32_t numChannels,
                                uint32_t height, uint32_t width, uint32_t tileLen,
                                float eps, bool affine) {
        totalLen_ = totalLen;
        batchSize_ = batchSize;
        numChannels_ = numChannels;
        height_ = height;
        width_ = width;
        spatialLen_ = height * width;
        tileLen_ = tileLen;
        eps_ = eps;
        affine_ = affine;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        if (affine_ && weight != nullptr) {
            weightGm_.SetGlobalBuffer((__gm__ float*)weight, numChannels_);
        }
        if (affine_ && bias != nullptr) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, numChannels_);
        }

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(reduceQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0) return;
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        // Each block processes one (n, c) pair
        if (blockIdx >= batchSize_ * numChannels_) return;

        uint32_t n = blockIdx / numChannels_;
        uint32_t c = blockIdx % numChannels_;
        uint32_t channelOffset = (n * numChannels_ + c) * spatialLen_;

        // ---- First pass: compute sum of x ----
        uint32_t offset = channelOffset;
        uint32_t remaining = spatialLen_;
        float sum = 0.0f;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            AscendC::LocalTensor<float> tmp = reduceQueue_.AllocTensor<float>();
            AscendC::Sum(tmp, xLocal, curLen);
            AscendC::LocalScalar<float> scalar;
            AscendC::ScalarMul(tmp, ONE, scalar, curLen);
            sum += scalar.GetValue(0);
            reduceQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }
        float mean = sum / static_cast<float>(spatialLen_);

        // ---- Second pass: compute sum of (x - mean)^2 ----
        offset = channelOffset;
        remaining = spatialLen_;
        float varSum = 0.0f;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            AscendC::LocalTensor<float> tmp = reduceQueue_.AllocTensor<float>();
            AscendC::Sub(tmp, xLocal, mean, curLen);
            AscendC::Mul(tmp, tmp, tmp, curLen);
            AscendC::LocalTensor<float> partSum = reduceQueue_.AllocTensor<float>();
            AscendC::Sum(partSum, tmp, curLen);
            AscendC::LocalScalar<float> scalar;
            AscendC::ScalarMul(partSum, ONE, scalar, curLen);
            varSum += scalar.GetValue(0);
            reduceQueue_.FreeTensor(partSum);
            reduceQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }
        float variance = varSum / static_cast<float>(spatialLen_);
        float invStd = 1.0f / sqrtf(variance + eps_);

        // ---- Normalize and apply affine ----
        float gamma = affine_ ? weightGm_.GetValue(c) : 1.0f;
        float beta = affine_ ? biasGm_.GetValue(c) : 0.0f;

        offset = channelOffset;
        remaining = spatialLen_;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Sub(yLocal, xLocal, mean, curLen);
            AscendC::Muls(yLocal, yLocal, invStd, curLen);
            AscendC::Muls(yLocal, yLocal, gamma, curLen);
            AscendC::Adds(yLocal, yLocal, beta, curLen);
            AscendC::DataCopy(yGm_[offset], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> reduceQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t numChannels_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    uint32_t spatialLen_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
    bool affine_ {true};
};

extern "C" __global__ __aicore__ void ca6k_01958_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01958Custom op;
    op.Init(x, y, weight, bias, tiling_data.totalLength_, tiling_data.batchSize_,
            tiling_data.numChannels_, tiling_data.height_, tiling_data.width_,
            tiling_data.tileLength_, tiling_data.eps_, tiling_data.affine_);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

_FN = "ca6k_01958_custom"

class ModelNew(nn.Module):
    def __init__(self, num_features, eps=1e-5, affine=True):
        super().__init__()
        self.eps = eps
        self.affine = affine
        if affine:
            self.weight = nn.Parameter(torch.ones(num_features))
            self.bias = nn.Parameter(torch.zeros(num_features))
        else:
            self.register_parameter("weight", None)
            self.register_parameter("bias", None)

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        if self.affine:
            return fn(x, self.eps, self.affine, self.weight, self.bias)
        else:
            return fn(x, self.eps, self.affine, None, None)
