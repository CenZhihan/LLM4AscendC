kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k05313Custom {
public:
    __aicore__ inline KernelCa6k05313Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t H, uint32_t W,
                                uint32_t kH, uint32_t kW,
                                uint32_t sH, uint32_t sW,
                                uint32_t pH, uint32_t pW,
                                uint32_t outH, uint32_t outW,
                                uint32_t tileLen)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, N*C*H*W);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N*C*outH*outW);
        N_ = N; C_ = C; H_ = H; W_ = W;
        kH_ = kH; kW_ = kW; sH_ = sH; sW_ = sW;
        pH_ = pH; pW_ = pW;
        outH_ = outH; outW_ = outW;
        tileLen_ = tileLen;
        blockDim_ = AscendC::GetBlockNum();
        blockIdx_ = AscendC::GetBlockIdx();
        totalJobs_ = N_ * C_ * outH_ * outW_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, kW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, kH_ * kW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalJobs_ == 0) return;
        uint32_t jobPerBlock = (totalJobs_ + blockDim_ - 1) / blockDim_;
        uint32_t startJob = blockIdx_ * jobPerBlock;
        uint32_t endJob = (startJob + jobPerBlock) > totalJobs_ ? totalJobs_ : (startJob + jobPerBlock);
        for (uint32_t jobId = startJob; jobId < endJob; ++jobId) {
            uint32_t ow = jobId % outW_;
            uint32_t oh = (jobId / outW_) % outH_;
            uint32_t c = (jobId / (outW_ * outH_)) % C_;
            uint32_t n = jobId / (outW_ * outH_ * C_);

            int32_t hStart = static_cast<int32_t>(oh * sH_) - static_cast<int32_t>(pH_);
            int32_t wStart = static_cast<int32_t>(ow * sW_) - static_cast<int32_t>(pW_);

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            float sum = 0.0f;
            uint32_t validCount = 0;
            for (int32_t kh = 0; kh < static_cast<int32_t>(kH_); ++kh) {
                for (int32_t kw = 0; kw < static_cast<int32_t>(kW_); ++kw) {
                    int32_t h = hStart + kh;
                    int32_t w = wStart + kw;
                    float val = 0.0f;
                    if (h >= 0 && h < static_cast<int32_t>(H_) && w >= 0 && w < static_cast<int32_t>(W_)) {
                        uint32_t gmIdx = ((n * C_ + c) * H_ + h) * W_ + w;
                        AscendC::DataCopy(xLocal, xGm_[gmIdx], 1);
                        val = xLocal(0);
                        if (val < 0.0f) val = 0.0f;
                        ++validCount;
                    }
                    sum += val;
                }
            }
            float avg = (validCount > 0) ? (sum / static_cast<float>(validCount)) : 0.0f;
            yLocal(0) = avg;
            uint32_t outIdx = (n * C_ + c) * outH_ * outW_ + oh * outW_ + ow;
            AscendC::DataCopy(yGm_[outIdx], yLocal, 1);

            inQueueX_.FreeTensor(xLocal);
            tmpQueue_.FreeTensor(tLocal);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, H_, W_;
    uint32_t kH_, kW_, sH_, sW_;
    uint32_t pH_, pW_;
    uint32_t outH_, outW_;
    uint32_t tileLen_;
    uint32_t blockDim_;
    uint32_t blockIdx_;
    uint32_t totalJobs_;
};

extern "C" __global__ __aicore__ void ca6k_05313_custom(GM_ADDR x, GM_ADDR y, GM_ADDR kernel_size, GM_ADDR stride, GM_ADDR padding, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k05313Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.kH, tiling_data.kW,
            tiling_data.sH, tiling_data.sW,
            tiling_data.pH, tiling_data.pW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.tileLength);
    op.Process();
}
