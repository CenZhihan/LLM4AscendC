#include "ca6k_00414_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00414CustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = shape.GetDim(0);
    uint32_t C = shape.GetDim(1);
    uint32_t H = shape.GetDim(2);
    uint32_t W = shape.GetDim(3);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float scale = *attrs->GetAttrPointer<float>(0);

    uint32_t H_out = static_cast<uint32_t>(H * scale);
    uint32_t W_out = static_cast<uint32_t>(W * scale);

    uint32_t tileH = 16;
    uint32_t tileW = 16;
    uint32_t tileElems = tileH * tileW;
    uint32_t ubNeeded = tileElems * 20 * sizeof(float);
    if (ubNeeded > UB_BUDGET_BYTES) {
        tileH = AlignDown(UB_BUDGET_BYTES / (20 * sizeof(float)), 8);
        tileW = tileH;
    }

    uint32_t totalTiles = CeilDiv(H_out, tileH) * CeilDiv(W_out, tileW) * N * C;
    uint32_t blockDim = totalTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
static constexpr float BICUBIC_A = -0.75f;

class KernelCa6k00414Custom {
public:
    __aicore__ inline KernelCa6k00414Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t H, uint32_t W,
                                uint32_t H_out, uint32_t W_out,
                                uint32_t tileH, uint32_t tileW,
                                uint32_t blockDim) {
        N_ = N; C_ = C; H_ = H; W_ = W; H_out_ = H_out; W_out_ = W_out;
        tileH_ = tileH; tileW_ = tileW; blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * H * W);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C * H_out * W_out);

        patchH_ = tileH_ / 2 + 3;
        patchW_ = tileW_ / 2 + 3;
        uint32_t patchElems = patchH_ * patchW_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, patchElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileH_ * tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileH_ * tileW_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blkIdx = AscendC::GetBlockIdx();
        uint32_t blkNum = AscendC::GetBlockNum();
        uint32_t tileCols = (W_out_ + tileW_ - 1) / tileW_;
        uint32_t tileRows = (H_out_ + tileH_ - 1) / tileH_;
        uint32_t totalWork = N_ * C_ * tileRows * tileCols;
        for (uint32_t bid = blkIdx; bid < totalWork; bid += blkNum) {
            uint32_t temp = bid;
            uint32_t tc = temp % tileCols;
            temp /= tileCols;
            uint32_t tr = temp % tileRows;
            temp /= tileRows;
            uint32_t c = temp % C_;
            uint32_t b = temp / C_;

            uint32_t outRowStart = tr * tileH_;
            uint32_t outColStart = tc * tileW_;
            uint32_t outRowEnd = outRowStart + tileH_;
            if (outRowEnd > H_out_) outRowEnd = H_out_;
            uint32_t outColEnd = outColStart + tileW_;
            if (outColEnd > W_out_) outColEnd = W_out_;

            float x0 = outColStart / 2.0f;
            float y0 = outRowStart / 2.0f;
            float x1 = (outColEnd - 1) / 2.0f;
            float y1 = (outRowEnd - 1) / 2.0f;
            int inColStart = (int)(x0) - 1;
            if (inColStart < 0) inColStart = 0;
            int inRowStart = (int)(y0) - 1;
            if (inRowStart < 0) inRowStart = 0;
            int inColEnd = (int)(x1) + 2;
            if (inColEnd >= (int)W_) inColEnd = (int)W_ - 1;
            int inRowEnd = (int)(y1) + 2;
            if (inRowEnd >= (int)H_) inRowEnd = (int)H_ - 1;
            int patchRows = inRowEnd - inRowStart + 1;
            int patchCols = inColEnd - inColStart + 1;

            AscendC::LocalTensor<float> patch = inQueueX_.AllocTensor<float>();
            for (int r = 0; r < patchRows; ++r) {
                uint32_t gmOffset = (b * C_ + c) * H_ * W_ + (inRowStart + r) * W_ + inColStart;
                AscendC::DataCopy(patch[r * patchW_], xGm_[gmOffset], patchCols);
            }
            inQueueX_.EnQue(patch);

            AscendC::LocalTensor<float> output = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> work = tmpQueue_.AllocTensor<float>();
            int idx = 0;
            for (uint32_t r = outRowStart; r < outRowEnd; ++r) {
                for (uint32_t col = outColStart; col < outColEnd; ++col) {
                    float yF = r / 2.0f;
                    float xF = col / 2.0f;
                    int iy = (int)yF;
                    int ix = (int)xF;
                    float du = xF - ix;
                    float dv = yF - iy;

                    float wx[4], wy[4];
                    for (int k = 0; k < 4; ++k) {
                        float u = (float)(k - 1) - du;
                        float v = (float)(k - 1) - dv;
                        wx[k] = bicubicWeight(u, BICUBIC_A);
                        wy[k] = bicubicWeight(v, BICUBIC_A);
                    }

                    float sum = 0.0f;
                    for (int j = 0; j < 4; ++j) {
                        int row = iy + j - 1;
                        if (row < 0) row = 0;
                        if (row >= (int)H_) row = (int)H_ - 1;
                        float rowSum = 0.0f;
                        for (int k = 0; k < 4; ++k) {
                            int colIn = ix + k - 1;
                            if (colIn < 0) colIn = 0;
                            if (colIn >= (int)W_) colIn = (int)W_ - 1;
                            uint32_t gIdx = (b * C_ + c) * H_ * W_ + row * W_ + colIn;
                            rowSum += xGm_.GetValue(gIdx) * wx[k];
                        }
                        sum += rowSum * wy[j];
                    }

                    output.SetValue(idx, sum);
                    idx++;
                }
            }
            tmpQueue_.FreeTensor(work);
            outQueueY_.EnQue<float>(output);

            AscendC::LocalTensor<float> outLocal = outQueueY_.DeQue<float>();
            int outIdx = 0;
            for (uint32_t r = outRowStart; r < outRowEnd; ++r) {
                uint32_t gmOffset = (b * C_ + c) * H_out_ * W_out_ + r * W_out_ + outColStart;
                uint32_t len = outColEnd - outColStart;
                AscendC::DataCopy(yGm_[gmOffset], outLocal[outIdx], len);
                outIdx += len;
            }
            outQueueY_.FreeTensor(outLocal);

            AscendC::LocalTensor<float> patchFree = inQueueX_.DeQue<float>();
            inQueueX_.FreeTensor(patchFree);
        }
    }

private:
    __aicore__ inline float bicubicWeight(float x, float a) {
        float absX = x;
        if (absX < 0.0f) absX = -absX;
        if (absX <= 1.0f) {
            return (a + 2.0f) * absX * absX * absX - (a + 3.0f) * absX * absX + 1.0f;
        } else if (absX < 2.0f) {
            return a * absX * absX * absX - 5.0f * a * absX * absX + 8.0f * a * absX - 4.0f * a;
        } else {
            return 0.0f;
        }
    }

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, H_, W_, H_out_, W_out_;
    uint32_t tileH_, tileW_, blockDim_;
    uint32_t patchH_, patchW_;
};

extern "C" __global__ __aicore__ void ca6k_00414_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00414Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C,
            tiling_data.H, tiling_data.W,
            tiling_data.H_out, tiling_data.W_out,
            tiling_data.tileH, tiling_data.tileW,
            tiling_data.blockDim);
    op.Process();
}
"""
