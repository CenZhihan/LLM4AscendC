kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00317Custom {
public:
    __aicore__ inline KernelCa6k00317Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalInputElems, uint32_t totalOutputElems,
                                uint32_t tileLen, uint32_t blockDim,
                                uint32_t n, uint32_t c_in, uint32_t c_out,
                                uint32_t d_in, uint32_t h_in, uint32_t w_in,
                                uint32_t d_out, uint32_t h_out, uint32_t w_out,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                int32_t stride_d, int32_t stride_h, int32_t stride_w,
                                int32_t pad_d, int32_t pad_h, int32_t pad_w,
                                int32_t output_pad_d, int32_t output_pad_h, int32_t output_pad_w)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputElems);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, c_out * c_in * kd * kh * kw);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElems);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen * sizeof(float));

        n_ = n; c_in_ = c_in; c_out_ = c_out;
        d_in_ = d_in; h_in_ = h_in; w_in_ = w_in;
        d_out_ = d_out; h_out_ = h_out; w_out_ = w_out;
        kd_ = kd; kh_ = kh; kw_ = kw;
        stride_d_ = stride_d; stride_h_ = stride_h; stride_w_ = stride_w;
        pad_d_ = pad_d; pad_h_ = pad_h; pad_w_ = pad_w;
        output_pad_d_ = output_pad_d; output_pad_h_ = output_pad_h; output_pad_w_ = output_pad_w;
    }

    __aicore__ inline void Process() {
        for (uint32_t batch = 0; batch < n_; batch++) {
            for (uint32_t oc = 0; oc < c_out_; oc++) {
                for (uint32_t od = 0; od < d_out_; od++) {
                    for (uint32_t oh = 0; oh < h_out_; oh++) {
                        for (uint32_t ow = 0; ow < w_out_; ow++) {
                            float sum = 0.0f;
                            for (uint32_t ic = 0; ic < c_in_; ic++) {
                                for (uint32_t kd = 0; kd < kd_; kd++) {
                                    for (uint32_t kh = 0; kh < kh_; kh++) {
                                        for (uint32_t kw = 0; kw < kw_; kw++) {
                                            int32_t id = static_cast<int32_t>(od) - static_cast<int32_t>(pad_d_) + static_cast<int32_t>(kd);
                                            int32_t ih = static_cast<int32_t>(oh) - static_cast<int32_t>(pad_h_) + static_cast<int32_t>(kh);
                                            int32_t iw = static_cast<int32_t>(ow) - static_cast<int32_t>(pad_w_) + static_cast<int32_t>(kw);
                                            if (id >= 0 && id < static_cast<int32_t>(d_in_) &&
                                                ih >= 0 && ih < static_cast<int32_t>(h_in_) &&
                                                iw >= 0 && iw < static_cast<int32_t>(w_in_) &&
                                                (id - static_cast<int32_t>(kd_)) % static_cast<int32_t>(stride_d_) == 0 &&
                                                (ih - static_cast<int32_t>(kh_)) % static_cast<int32_t>(stride_h_) == 0 &&
                                                (iw - static_cast<int32_t>(kw_)) % static_cast<int32_t>(stride_w_) == 0) {
                                                uint32_t id_src = (id - static_cast<int32_t>(pad_d_) + static_cast<int32_t>(kd_)) / static_cast<int32_t>(stride_d_);
                                                uint32_t ih_src = (ih - static_cast<int32_t>(pad_h_) + static_cast<int32_t>(kh_)) / static_cast<int32_t>(stride_h_);
                                                uint32_t iw_src = (iw - static_cast<int32_t>(pad_w_) + static_cast<int32_t>(kw_)) / static_cast<int32_t>(stride_w_);
                                                uint32_t x_idx = batch * c_in_ * d_in_ * h_in_ * w_in_ + ic * d_in_ * h_in_ * w_in_ + id_src * h_in_ * w_in_ + ih_src * w_in_ + iw_src;
                                                uint32_t w_idx = oc * c_in_ * kd_ * kh_ * kw_ + ic * kd_ * kh_ * kw_ + kd * kh_ * kw_ + kh * kw_ + kw;
                                                sum += xGm_.GetValue(x_idx) * wGm_.GetValue(w_idx);
                                            }
                                        }
                                    }
                                }
                            }
                            uint32_t y_idx = batch * c_out_ * d_out_ * h_out_ * w_out_ + oc * d_out_ * h_out_ * w_out_ + od * h_out_ * w_out_ + oh * w_out_ + ow;
                            yGm_.SetValue(y_idx, sum);
                        }
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_, c_in_, c_out_, d_in_, h_in_, w_in_, d_out_, h_out_, w_out_;
    uint32_t kd_, kh_, kw_;
    int32_t stride_d_, stride_h_, stride_w_;
    int32_t pad_d_, pad_h_, pad_w_;
    int32_t output_pad_d_, output_pad_h_, output_pad_w_;
};

extern "C" __global__ __aicore__ void ca6k_00317_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00317Custom op;
    op.Init(x, weight, y,
            tiling_data.totalInputElems, tiling_data.totalOutputElems,
            tiling_data.tileLen, tiling_data.blockDim,
            tiling_data.n, tiling_data.c_in, tiling_data.c_out,
            tiling_data.d_in, tiling_data.h_in, tiling_data.w_in,
            tiling_data.d_out, tiling_data.h_out, tiling_data.w_out,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.stride_d, tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_d, tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.output_pad_d, tiling_data.output_pad_h, tiling_data.output_pad_w);
    op.Process();
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00317_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias):
        super().__init__()
        self.weight = torch.nn.Parameter(torch.randn(out_channels, in_channels, *kernel_size))
        self.bias = torch.nn.Parameter(torch.zeros(out_channels)) if bias else None
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.stride, self.padding, self.output_padding)
