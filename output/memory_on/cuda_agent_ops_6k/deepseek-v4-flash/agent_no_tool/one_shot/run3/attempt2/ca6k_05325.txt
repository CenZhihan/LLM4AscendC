host_operator_src = """
#include "ca6k_05325_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
  return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k05325CustomTilingData tiling;

  const gert::StorageShape *xShape = context->GetInputShape(0);
  const auto &originShape = xShape->GetOriginShape();
  uint32_t totalOrig = static_cast<uint32_t>(originShape.GetShapeSize());
  // Input shape: (B, 2*D, H, W)
  uint32_t batch = static_cast<uint32_t>(originShape.GetDim(0));
  uint32_t channels = static_cast<uint32_t>(originShape.GetDim(1));
  uint32_t height = static_cast<uint32_t>(originShape.GetDim(2));
  uint32_t width = static_cast<uint32_t>(originShape.GetDim(3));
  uint32_t halfChannels = channels / 2;  // D
  uint32_t spatialSize = batch * height * width; // N
  uint32_t totalLength = spatialSize * halfChannels; // number of output elements

  // UB buffer usage per position:
  //   inBuf:  2*halfChannels floats (left+right)
  //   outBuf: halfChannels floats
  //   temp:   3*halfChannels floats (glu, weightRow, tmp)
  // Total floats per position = 2*D + D + 3*D = 6*D
  uint32_t perPosBytes = 6 * halfChannels * sizeof(float);
  uint32_t maxTile = (UB_BUDGET_BYTES / perPosBytes);
  if (maxTile < 1) maxTile = 1;
  maxTile = AlignDown(maxTile, 1); // no alignment constraint as long as buffer multiple of 32B
  uint32_t tileLen = maxTile;
  if (totalLength < tileLen) {
    tileLen = totalLength;
  }

  uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_totalLength(totalLength);
  tiling.set_tileLength(tileLen);
  tiling.set_blockDim(blockDim);
  tiling.set_halfChannels(halfChannels);
  tiling.set_spatialSize(spatialSize);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape *x_shape = context->GetInputShape(0);
  gert::Shape *y_shape = context->GetOutputShape(0);
  // output shape: (B, D, H, W)
  auto x_dims = x_shape->GetDimNum();
  gert::Shape out_shape(*x_shape);
  out_shape.SetDim(1, x_shape->GetDim(1) / 2);
  *y_shape = out_shape;
  return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
  context->SetOutputDataType(0, context->GetInputDataType(0));
  return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k05325Custom {
public:
  __aicore__ inline KernelCa6k05325Custom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                              uint32_t totalLength, uint32_t tileLength,
                              uint32_t halfChannels, uint32_t spatialSize) {
    totalLen_ = totalLength;
    tileLen_ = tileLength;
    halfChannels_ = halfChannels;
    spatialSize_ = spatialSize;
    inputStride_ = 2 * halfChannels;  // floats per spatial position in input
    outputStride_ = halfChannels;     // floats per spatial position in output

    xGm_.SetGlobalBuffer((__gm__ float*)x, spatialSize_ * inputStride_);
    weightGm_.SetGlobalBuffer((__gm__ float*)weight, halfChannels_ * halfChannels_);
    biasGm_.SetGlobalBuffer((__gm__ float*)bias, halfChannels_);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

    uint32_t inBufBytes = tileLen_ * inputStride_ * sizeof(float);
    uint32_t outBufBytes = tileLen_ * outputStride_ * sizeof(float);
    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inBufBytes);
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outBufBytes);
    pipe_.InitBuffer(tmpBuf_, 3 * halfChannels_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;

    uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    uint32_t stridePos = blockNum * tileLen_;
    uint32_t posBase = blockIdx * tileLen_;

    while (posBase < spatialSize_) {
      uint32_t curTileLen = spatialSize_ - posBase;
      if (curTileLen > tileLen_) curTileLen = tileLen_;

      // Copy input tile
      AscendC::LocalTensor<float> inLocal = inQueueX_.AllocTensor<float>();
      AscendC::DataCopy(inLocal, xGm_[posBase * inputStride_], curTileLen * inputStride_);
      inQueueX_.EnQue(inLocal);

      // Copy output tile (zeroed later)
      AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();

      // Compute GLU + Linear for each position in tile
      AscendC::LocalTensor<float> tmpBuf = tmpBuf_.Get<float>(0);
      AscendC::LocalTensor<float> gluBuf = tmpBuf;                  // D floats
      AscendC::LocalTensor<float> weightRowBuf = tmpBuf + halfChannels_;  // D floats
      AscendC::LocalTensor<float> tmp2Buf = tmpBuf + 2 * halfChannels_;   // D floats

      for (uint32_t i = 0; i < curTileLen; i++) {
        // Input offsets
        uint32_t inOffset = i * inputStride_;
        AscendC::LocalTensor<float> leftBuf = inLocal[inOffset];
        AscendC::LocalTensor<float> rightBuf = inLocal[inOffset + halfChannels_];

        // GLU: glu = sigmoid(left) * right
        AscendC::Sigmoid(gluBuf, leftBuf, halfChannels_);
        AscendC::Mul(gluBuf, gluBuf, rightBuf, halfChannels_);

        // Linear: for each output channel
        for (uint32_t j = 0; j < halfChannels_; j++) {
          // Load weight row j (global -> local)
          AscendC::DataCopy(weightRowBuf, weightGm_[j * halfChannels_], halfChannels_);
          // tmp = glu * weightRow
          AscendC::Mul(tmp2Buf, gluBuf, weightRowBuf, halfChannels_);
          // reduce sum
          float dot = AscendC::ReduceSum(tmp2Buf, halfChannels_);
          // add bias
          float res = dot + biasGm_.GetValue(j);
          outLocal[i * outputStride_ + j].SetValue(res);
        }
      }

      outQueueY_.EnQue(outLocal);
      inQueueX_.FreeTensor(inLocal);

      // Write output tile to global
      AscendC::LocalTensor<float> outLocalDeq = outQueueY_.DeQue<float>();
      AscendC::DataCopy(yGm_[posBase * outputStride_], outLocalDeq, curTileLen * outputStride_);
      outQueueY_.FreeTensor(outLocalDeq);

      posBase += stridePos;
    }
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
  AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> weightGm_;
  AscendC::GlobalTensor<float> biasGm_;
  AscendC::GlobalTensor<float> yGm_;

  uint32_t totalLen_ {0};
  uint32_t tileLen_ {0};
  uint32_t halfChannels_ {0};
  uint32_t spatialSize_ {0};
  uint32_t inputStride_ {0};
  uint32_t outputStride_ {0};
};

extern "C" __global__ __aicore__ void ca6k_05325_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelCa6k05325Custom op;
  op.Init(x, weight, bias, y,
          tiling_data.totalLength, tiling_data.tileLength,
          tiling_data.halfChannels, tiling_data.spatialSize);
  op.Process();
}
