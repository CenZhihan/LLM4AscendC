kernel_src="""
#include "kernel_operator.h"

class KernelCa6k00246Custom {
public:
    __aicore__ inline KernelCa6k00246Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t r, uint32_t totalOut, uint32_t tileOut)
    {
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        r_ = r;
        totalOut_ = totalOut;
        tileOut_ = tileOut;

        Hr_ = H_ * r_;
        Wr_ = W_ * r_;
        inC_ = C_ * r_ * r_;
        inStrideHW_ = H_ * W_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * inC_ * inStrideHW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut_);
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileOut_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride over output elements
        for (uint32_t outBase = blockIdx * tileOut_; outBase < totalOut_; outBase += blockNum * tileOut_) {
            uint32_t curTile = tileOut_;
            if (outBase + curTile > totalOut_) curTile = totalOut_ - outBase;

            for (uint32_t i = 0; i < curTile; ++i) {
                uint32_t outIdx = outBase + i;

                // decode output index: [n, c, h_out, w_out]
                uint32_t n = outIdx / (C_ * Hr_ * Wr_);
                uint32_t rem = outIdx % (C_ * Hr_ * Wr_);
                uint32_t c = rem / (Hr_ * Wr_);
                rem = rem % (Hr_ * Wr_);
                uint32_t hOut = rem / Wr_;
                uint32_t wOut = rem % Wr_;

                // map to input: h = hOut / r_, w = wOut / r_, i_ = hOut % r_, j_ = wOut % r_
                uint32_t hIn = hOut / r_;
                uint32_t wIn = wOut / r_;
                uint32_t i_ = hOut % r_;
                uint32_t j_ = wOut % r_;
                uint32_t inC = c * r_ * r_ + i_ * r_ + j_;

                // input index: [n, inC, hIn, wIn] -> linear
                uint32_t inIdx = (n * inC_ + inC) * inStrideHW_ + hIn * W_ + wIn;

                yGm_[outIdx] = xGm_[inIdx];
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t N_;
    uint32_t C_;
    uint32_t H_;
    uint32_t W_;
    uint32_t r_;
    uint32_t Hr_;
    uint32_t Wr_;
    uint32_t inC_;
    uint32_t inStrideHW_;
    uint32_t totalOut_;
    uint32_t tileOut_;
};

extern "C" __global__ __aicore__ void ca6k_00246_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00246Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.r, tiling_data.totalOut, tiling_data.tileOut);
    op.Process();
}
