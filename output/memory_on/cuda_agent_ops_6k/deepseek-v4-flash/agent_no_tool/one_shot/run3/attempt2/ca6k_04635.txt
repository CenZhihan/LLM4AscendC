namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t MAX_TILE_ELEMS = 4096;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04635CustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::Shape &origin = x_shape->GetOriginShape();
    uint32_t batch = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t channels = static_cast<uint32_t>(origin.GetDim(1));
    uint32_t inputD = static_cast<uint32_t>(origin.GetDim(2));
    uint32_t inputH = static_cast<uint32_t>(origin.GetDim(3));
    uint32_t inputW = static_cast<uint32_t>(origin.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *output_size = attrs->GetAttrPointer<int64_t>(0);
    uint32_t outputD = static_cast<uint32_t>(output_size[0]);
    uint32_t outputH = static_cast<uint32_t>(output_size[1]);
    uint32_t outputW = static_cast<uint32_t>(output_size[2]);

    uint32_t totalOutputElems = batch * channels * outputD * outputH * outputW;

    uint32_t strideD = inputD / outputD;
    uint32_t kernelD = inputD - (outputD - 1u) * strideD;
    uint32_t strideH = inputH / outputH;
    uint32_t kernelH = inputH - (outputH - 1u) * strideH;
    uint32_t strideW = inputW / outputW;
    uint32_t kernelW = inputW - (outputW - 1u) * strideW;

    uint32_t tileLen = totalOutputElems;
    if (tileLen > MAX_TILE_ELEMS) tileLen = MAX_TILE_ELEMS;

    uint32_t blockDim = CeilDiv(totalOutputElems, tileLen);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_inputD(inputD);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_outputD(outputD);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_strideD(strideD);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_kernelD(kernelD);
    tiling.set_kernelH(kernelH);
    tiling.set_kernelW(kernelW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::Shape &origin = x_shape->GetOriginShape();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *output_size = attrs->GetAttrPointer<int64_t>(0);
    uint32_t outputD = static_cast<uint32_t>(output_size[0]);
    uint32_t outputH = static_cast<uint32_t>(output_size[1]);
    uint32_t outputW = static_cast<uint32_t>(output_size[2]);

    gert::Shape *y_shape = context->GetOutputShape(0);
    // Copy first two dims (batch, channels)
    y_shape->SetDim(0, origin.GetDim(0));
    y_shape->SetDim(1, origin.GetDim(1));
    y_shape->SetDim(2, static_cast<int64_t>(outputD));
    y_shape->SetDim(3, static_cast<int64_t>(outputH));
    y_shape->SetDim(4, static_cast<int64_t>(outputW));
    y_shape->SetDimCount(5);
    return GRAPH_SUCCESS;
}
