host_operator_src = """
#include "register/op_def_registry.h"
#include "Ca6k04589Custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04589CustomTilingData tiling;

    const gert::StorageShape *inShape = context->GetInputShape(0);
    uint32_t N = (uint32_t)inShape->GetDim(0);
    uint32_t C = (uint32_t)inShape->GetDim(1);
    uint32_t H = (uint32_t)inShape->GetDim(2);
    uint32_t W = (uint32_t)inShape->GetDim(3);

    uint32_t outH = H / 2;
    uint32_t outW = W / 2;
    uint32_t channelSizeOut = outH * outW;
    uint32_t channelSizeIn = H * W;
    uint32_t totalLength = N * C * channelSizeOut;

    // UB budget: output tile elements (plus input tile rows? We'll tile on output elements)
    // Kernel reads up to 2 rows of input per output tile row. For each output tile of length tileLength,
    // need input of 2 * (tileLength * 2) if tileLength spans multiple output columns? Actually better to tile by rows.
    // Simpler: tile by output elements. per output element we need 4 input values. We'll read 2 rows of input at a time.
    // For a tile of tileLen output elements (contiguous in row-major order), we need input of at most 2 * (tileLen * 2) if tile spans multiple output rows? But tileLen is small, we can just read input for each output element individually.
    // To simplify and avoid UB overflow, we keep tileLen small.
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float) * 2); // each output requires reading 4 floats, plus temp buffer for input row?
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    if (maxTile > outW) maxTile = outW;
    if (maxTile < 1) maxTile = 1;
    uint32_t tileLength = maxTile;

    uint32_t numTiles = (tileLength == 0) ? 1u : CeilDiv(totalLength, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.totalLength = totalLength;
    tiling.tileLength = tileLength;
    tiling.blockDim = blockDim;
    tiling.inputW = W;
    tiling.inputH = H;
    tiling.outputW = outW;
    tiling.outputH = outH;
    tiling.channelSizeOut = channelSizeOut;
    tiling.channelSizeIn = channelSizeIn;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    gert::StorageShape *outShape = context->GetOutputShape(0);

    uint32_t C = inShape->GetDim(1);
    uint32_t H = inShape->GetDim(2);
    uint32_t W = inShape->GetDim(3);
    outShape->SetDimNum(4);
    outShape->SetDim(0, N);
    outShape->SetDim(1, C);
    outShape->SetDim(2, H / 2);
    outShape->SetDim(3, W / 2);
    return GRAPH_SUCCESS;
}

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float INV_4 = 0.25f;

class KernelCa6k04589Custom {
public:
    __aicore__ inline KernelCa6k04589Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inputW, uint32_t inputH,
                                uint32_t outputW, uint32_t outputH,
                                uint32_t channelSizeOut, uint32_t channelSizeIn) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inputW_ = inputW;
        inputH_ = inputH;
        outputW_ = outputW;
        outputH_ = outputH;
        channelSizeOut_ = channelSizeOut;
        channelSizeIn_ = channelSizeIn;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Buffer for 2 rows of input (each row = inputW floats)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, 2 * inputW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, 2 * inputW_ * sizeof(float)); // for intermediate sum
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stride = blockNum * tileLen_;
        uint32_t outIdx = blockIdx * tileLen_;
        if (outIdx >= totalLen_) return;

        while (outIdx < totalLen_) {
            uint32_t curTile = tileLen_;
            if (outIdx + curTile > totalLen_) curTile = totalLen_ - outIdx;

            // Process output elements contiguously in row-major order
            // output element linear index = outIdx
            uint32_t batchChannel = outIdx / channelSizeOut_; // batch*C + channel
            uint32_t outOffset = outIdx - batchChannel * channelSizeOut_; // within channel
            uint32_t oh = outOffset / outputW_;
            uint32_t ow = outOffset % outputW_;

            // The output tile spans multiple rows if curTile > outputW_ - ow.
            // To simplify, we process each output element individually within tile.
            // Since tileLen is small (max outputW), we can loop.
            uint32_t cnt = 0;
            while (cnt < curTile) {
                uint32_t row = oh + (ow + cnt) / outputW_;
                uint32_t col = (ow + cnt) % outputW_;
                if (row >= outputH_) break; // safety

                // Input row and column for 2x2 block
                uint32_t inRow = row * 2;
                uint32_t inCol = col * 2;

                // Load 4 input values: (inRow, inCol), (inRow, inCol+1), (inRow+1, inCol), (inRow+1, inCol+1)
                uint32_t baseIn = batchChannel * channelSizeIn_ + inRow * inputW_ + inCol;
                float v00 = xGm_.GetValue(baseIn);
                float v01 = xGm_.GetValue(baseIn + 1);
                float v10 = xGm_.GetValue(baseIn + inputW_);
                float v11 = xGm_.GetValue(baseIn + inputW_ + 1);

                float avg = (v00 + v01 + v10 + v11) * INV_4;
                yGm_.SetValue(outIdx + cnt, avg);
                cnt++;
            }

            outIdx += stride;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t inputW_ {0};
    uint32_t inputH_ {0};
    uint32_t outputW_ {0};
    uint32_t outputH_ {0};
    uint32_t channelSizeOut_ {0};
    uint32_t channelSizeIn_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04589_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04589Custom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.inputW,
            tiling_data.inputH,
            tiling_data.outputW,
            tiling_data.outputH,
            tiling_data.channelSizeOut,
            tiling_data.channelSizeIn);
    op.Process();
}
"""
