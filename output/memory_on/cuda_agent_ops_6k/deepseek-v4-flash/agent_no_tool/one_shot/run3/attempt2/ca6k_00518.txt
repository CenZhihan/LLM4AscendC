// UB model: at most 2 tile-sized float tensors alive simultaneously => 8 bytes per element
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00518CustomTilingData tiling;

    const gert::StorageShape *inShape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(inShape->GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    int64_t rank = inShape->GetDimNum();
    if (dim < 0) dim += rank;
    uint32_t dimSize = static_cast<uint32_t>(inShape->GetDim(dim));

    uint32_t numVectors = totalLength / dimSize;

    // tile length from UB budget (2 tensors, each of float)
    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLength = maxTile;
    if (dimSize < tileLength) {
        tileLength = AlignDown(dimSize, ALIGN_ELEMS);
        if (tileLength == 0) tileLength = dimSize;
    }

    // block dim: at most 24, but also limited by number of vectors
    uint32_t blockDim = numVectors;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_dimSize(dimSize);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

#include "kernel_operator.h"
#include <limits>

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00518Custom {
public:
    __aicore__ inline KernelCa6k00518Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t dimSize,
                                uint32_t tileLength, uint32_t blockDim)
    {
        totalLen_ = totalLength;
        dimSize_ = dimSize;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || dimSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t numVectors = totalLen_ / dimSize_;

        for (uint32_t vecIdx = blockIdx; vecIdx < numVectors; vecIdx += blockNum) {
            uint32_t baseOff = vecIdx * dimSize_;

            // ---------- Pass 1: find global max ----------
            float globalMax = -std::numeric_limits<float>::max();
            for (uint32_t tileOff = 0; tileOff < dimSize_; tileOff += tileLen_) {
                uint32_t curLen = (tileOff + tileLen_ <= dimSize_) ? tileLen_ : (dimSize_ - tileOff);
                AscendC::LocalTensor<float> xLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
                AscendC::DataCopy(xLocal, xGm_[baseOff + tileOff], curLen);

                AscendC::LocalTensor<float> localMax = pipe_.AllocTensor<float>(sizeof(float));
                AscendC::ReduceMax<float, float>(localMax, xLocal, true);
                float lm = localMax.GetValue(0);
                if (lm > globalMax) globalMax = lm;

                pipe_.FreeTensor(localMax);
                pipe_.FreeTensor(xLocal);
            }

            // ---------- Pass 2: compute sum of exp(x-globalMax) ----------
            float globalSum = 0.0f;
            for (uint32_t tileOff = 0; tileOff < dimSize_; tileOff += tileLen_) {
                uint32_t curLen = (tileOff + tileLen_ <= dimSize_) ? tileLen_ : (dimSize_ - tileOff);

                AscendC::LocalTensor<float> xLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
                AscendC::DataCopy(xLocal, xGm_[baseOff + tileOff], curLen);

                AscendC::LocalTensor<float> diff = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
                AscendC::Sub(diff, xLocal, globalMax, curLen);
                pipe_.FreeTensor(xLocal);

                AscendC::LocalTensor<float> expLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
                AscendC::Exp(expLocal, diff, curLen);
                pipe_.FreeTensor(diff);

                AscendC::LocalTensor<float> localSum = pipe_.AllocTensor<float>(sizeof(float));
                AscendC::ReduceSum<float, float>(localSum, expLocal, true);
                globalSum += localSum.GetValue(0);

                pipe_.FreeTensor(localSum);
                pipe_.FreeTensor(expLocal);
            }

            // ---------- Pass 3: normalize and write out ----------
            float invSum = 1.0f / globalSum;
            for (uint32_t tileOff = 0; tileOff < dimSize_; tileOff += tileLen_) {
                uint32_t curLen = (tileOff + tileLen_ <= dimSize_) ? tileLen_ : (dimSize_ - tileOff);

                AscendC::LocalTensor<float> xLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
                AscendC::DataCopy(xLocal, xGm_[baseOff + tileOff], curLen);

                AscendC::LocalTensor<float> diff = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
                AscendC::Sub(diff, xLocal, globalMax, curLen);
                pipe_.FreeTensor(xLocal);

                AscendC::LocalTensor<float> expLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
                AscendC::Exp(expLocal, diff, curLen);
                pipe_.FreeTensor(diff);

                AscendC::LocalTensor<float> yLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
                AscendC::Muls(yLocal, expLocal, invSum, curLen);
                pipe_.FreeTensor(expLocal);

                AscendC::DataCopy(yGm_[baseOff + tileOff], yLocal, curLen);
                pipe_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t dimSize_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00518_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00518Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.dimSize, tiling_data.tileLength, tiling_data.blockDim);
    op.Process();
}
