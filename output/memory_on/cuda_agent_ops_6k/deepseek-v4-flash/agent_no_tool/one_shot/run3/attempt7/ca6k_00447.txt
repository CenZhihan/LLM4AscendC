#include "ca6k_00447_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
  return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k00447CustomTilingData tiling;
  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int32_t pad_left = *attrs->GetAttrPointer<int32_t>(0);
  int32_t pad_right = *attrs->GetAttrPointer<int32_t>(1);
  int32_t pad_top = *attrs->GetAttrPointer<int32_t>(2);
  int32_t pad_bottom = *attrs->GetAttrPointer<int32_t>(3);
  int32_t pad_front = *attrs->GetAttrPointer<int32_t>(4);
  int32_t pad_back = *attrs->GetAttrPointer<int32_t>(5);

  const gert::StorageShape *storage_shape = context->GetInputShape(0);
  const gert::Shape &x_shape = storage_shape->GetOriginShape();
  uint32_t origN = x_shape.GetDim(0);
  uint32_t origC = x_shape.GetDim(1);
  uint32_t origD = x_shape.GetDim(2);
  uint32_t origH = x_shape.GetDim(3);
  uint32_t origW = x_shape.GetDim(4);

  uint32_t outD = origD + pad_front + pad_back;
  uint32_t outH = origH + pad_top + pad_bottom;
  uint32_t outW = origW + pad_left + pad_right;

  uint32_t totalOutLen = origN * origC * outD * outH * outW;

  uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = maxTile;
  if (totalOutLen < tileLen) {
    uint32_t aligned = AlignDown(totalOutLen, ALIGN_ELEMS);
    tileLen = (aligned > 0) ? aligned : totalOutLen;
  }

  uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutLen, tileLen);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_totalOutLen(totalOutLen);
  tiling.set_tileLen(tileLen);
  tiling.set_blockDim(blockDim);
  tiling.set_pad_left(pad_left);
  tiling.set_pad_right(pad_right);
  tiling.set_pad_top(pad_top);
  tiling.set_pad_bottom(pad_bottom);
  tiling.set_pad_front(pad_front);
  tiling.set_pad_back(pad_back);
  tiling.set_origN(origN);
  tiling.set_origC(origC);
  tiling.set_origD(origD);
  tiling.set_origH(origH);
  tiling.set_origW(origW);
  tiling.set_outD(outD);
  tiling.set_outH(outH);
  tiling.set_outW(outW);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00447Custom {
public:
    __aicore__ inline KernelCa6k00447Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalOutLen, uint32_t tileLen,
                                int32_t padLeft, int32_t padRight,
                                int32_t padTop, int32_t padBottom,
                                int32_t padFront, int32_t padBack,
                                uint32_t origN, uint32_t origC,
                                uint32_t origD, uint32_t origH, uint32_t origW,
                                uint32_t outD, uint32_t outH, uint32_t outW)
    {
        totalOutLen_ = totalOutLen;
        tileLen_ = tileLen;
        padLeft_ = padLeft;
        padRight_ = padRight;
        padTop_ = padTop;
        padBottom_ = padBottom;
        padFront_ = padFront;
        padBack_ = padBack;
        origN_ = origN;
        origC_ = origC;
        origD_ = origD;
        origH_ = origH;
        origW_ = origW;
        outD_ = outD;
        outH_ = outH;
        outW_ = outW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, origN * origC * origD * origH * origW);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, origC);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutLen_) return;

        while (base < totalOutLen_) {
            uint32_t curLen = totalOutLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            ProcessTile(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t outOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; i++) {
            uint32_t globalIdx = outOffset + i;
            uint32_t n5 = globalIdx / (origC_ * outD_ * outH_ * outW_);
            uint32_t rem1 = globalIdx % (origC_ * outD_ * outH_ * outW_);
            uint32_t c5 = rem1 / (outD_ * outH_ * outW_);
            uint32_t rem2 = rem1 % (outD_ * outH_ * outW_);
            uint32_t d5 = rem2 / (outH_ * outW_);
            uint32_t rem3 = rem2 % (outH_ * outW_);
            uint32_t h5 = rem3 / outW_;
            uint32_t w5 = rem3 % outW_;

            int32_t origD = static_cast<int32_t>(d5) - padFront_;
            int32_t origH = static_cast<int32_t>(h5) - padTop_;
            int32_t origW = static_cast<int32_t>(w5) - padLeft_;

            float xVal = 0.0f;
            if (origD >= 0 && origD < static_cast<int32_t>(origD_) &&
                origH >= 0 && origH < static_cast<int32_t>(origH_) &&
                origW >= 0 && origW < static_cast<int32_t>(origW_)) {
                uint32_t xIdx = n5 * origC_ * origD_ * origH_ * origW_ +
                                c5 * origD_ * origH_ * origW_ +
                                static_cast<uint32_t>(origD) * origH_ * origW_ +
                                static_cast<uint32_t>(origH) * origW_ +
                                static_cast<uint32_t>(origW);
                xVal = xGm_.GetValue(xIdx);
            }
            float biasVal = biasGm_.GetValue(c5);
            float yVal = xVal + biasVal;
            yLocal.SetValue(i, yVal);
        }

        AscendC::DataCopy(yGm_[outOffset], yLocal, len);

        outQueueY_.FreeTensor(yLocal);
        tmpQueue_.FreeTensor(tLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLen_ {0};
    uint32_t tileLen_ {0};
    int32_t padLeft_ {0};
    int32_t padRight_ {0};
    int32_t padTop_ {0};
    int32_t padBottom_ {0};
    int32_t padFront_ {0};
    int32_t padBack_ {0};
    uint32_t origN_ {0};
    uint32_t origC_ {0};
    uint32_t origD_ {0};
    uint32_t origH_ {0};
    uint32_t origW_ {0};
    uint32_t outD_ {0};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00447_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00447Custom op;
    op.Init(x, bias, y,
            tiling_data.totalOutLen, tiling_data.tileLen,
            tiling_data.pad_left, tiling_data.pad_right,
            tiling_data.pad_top, tiling_data.pad_bottom,
            tiling_data.pad_front, tiling_data.pad_back,
            tiling_data.origN, tiling_data.origC,
            tiling_data.origD, tiling_data.origH, tiling_data.origW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW);
    op.Process();
}
'''

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00447_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, pad_size, bias_shape):
        super(ModelNew, self).__init__()
        self.pad_left, self.pad_right, self.pad_top, self.pad_bottom, self.pad_front, self.pad_back = pad_size
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.bias,
                  self.pad_left, self.pad_right,
                  self.pad_top, self.pad_bottom,
                  self.pad_front, self.pad_back)
