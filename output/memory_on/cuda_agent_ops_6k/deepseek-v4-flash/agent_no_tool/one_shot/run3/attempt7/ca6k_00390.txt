#include "ca6k_00390_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00390CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *groups = attrs->GetAttrPointer<int32_t>(0);

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batchSize = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t channels = static_cast<uint32_t>(shape.GetDim(1));
    uint32_t height = static_cast<uint32_t>(shape.GetDim(2));
    uint32_t width = static_cast<uint32_t>(shape.GetDim(3));
    uint32_t spatialSize = height * width;
    uint32_t sampleLen = channels * spatialSize;

    // UB budget: in+out+tmp = 5 * tileLen * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, sampleLen);  // align to whole samples
    if (maxTile < sampleLen) maxTile = sampleLen;
    // cap to avoid huge tile
    constexpr uint32_t HARD_CAP_ELEMS = 65536;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, sampleLen);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, sampleLen);
        if (tileLen == 0) tileLen = totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_groups(*groups);
    tiling.set_channels(channels);
    tiling.set_spatialSize(spatialSize);
    tiling.set_batchSize(batchSize);
    tiling.set_sampleLen(sampleLen);
    tiling.set_channelsPerGroup(channels / *groups);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00390Custom {
public:
    __aicore__ inline KernelCa6k00390Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t groups,
                                uint32_t channels, uint32_t spatialSize, uint32_t batchSize,
                                uint32_t sampleLen, uint32_t channelsPerGroup)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        groups_ = groups;
        channels_ = channels;
        spatialSize_ = spatialSize;
        batchSize_ = batchSize;
        sampleLen_ = sampleLen;
        channelsPerGroup_ = channelsPerGroup;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride loop over tiles (each tile aligned to sample boundaries)
        const uint32_t strideElems = blockNum * tileLen_;
        for (uint32_t base = blockIdx * tileLen_; base < totalLen_; base += strideElems)
        {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            CopyIn(base, curLen);
            Compute(base, curLen);
            CopyOut(base, curLen);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t gmBase, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // The tile is aligned to sample boundaries, so len is multiple of sampleLen.
        // Within each sample in the tile, we perform channel shuffle.
        uint32_t numSamples = len / sampleLen_;
        for (uint32_t s = 0; s < numSamples; ++s) {
            uint32_t baseSample = s * sampleLen_;
            for (uint32_t c = 0; c < channels_; ++c) {
                uint32_t cNew = (c % groups_) * channelsPerGroup_ + (c / groups_);
                uint32_t srcOffset = baseSample + c * spatialSize_;
                uint32_t dstOffset = baseSample + cNew * spatialSize_;
                // Copy spatial block of size spatialSize_ from xLocal to tLocal at correct position
                // Use DataCopy for contiguous block (spatial elements are contiguous)
                AscendC::DataCopy(tLocal[dstOffset], xLocal[srcOffset], spatialSize_);
            }
        }
        // Copy tLocal (permuted) to yLocal
        AscendC::DataCopy(yLocal, tLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t groups_ {2};
    uint32_t channels_ {0};
    uint32_t spatialSize_ {0};
    uint32_t batchSize_ {0};
    uint32_t sampleLen_ {0};
    uint32_t channelsPerGroup_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00390_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00390Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.groups, tiling_data.channels, tiling_data.spatialSize, tiling_data.batchSize,
            tiling_data.sampleLen, tiling_data.channelsPerGroup);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00390_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        # init_inputs: [num_channels, groups, block_size]
        if len(init_inputs) >= 2:
            self.groups = init_inputs[1]
        else:
            self.groups = 2

    def forward(self, *inputs):
        # inputs: [x]
        x = inputs[0]
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.groups)
