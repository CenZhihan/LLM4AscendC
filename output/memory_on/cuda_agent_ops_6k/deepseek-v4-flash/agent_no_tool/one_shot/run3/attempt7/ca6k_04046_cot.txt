namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k04046CustomTilingData tiling;
  const gert::StorageShape *storageShape = context->GetInputShape(0);
  const auto& inShape = storageShape->GetOriginShape();
  uint32_t batch = inShape.GetDim(0);
  uint32_t channels = inShape.GetDim(1);
  uint32_t depth = inShape.GetDim(2);
  uint32_t height = inShape.GetDim(3);
  uint32_t width = inShape.GetDim(4);

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int32_t pad_front = attrs->GetAttrPointer<int32_t>(0)[0];
  int32_t pad_back = attrs->GetAttrPointer<int32_t>(1)[0];
  int32_t pad_top = attrs->GetAttrPointer<int32_t>(2)[0];
  int32_t pad_bottom = attrs->GetAttrPointer<int32_t>(3)[0];
  int32_t pad_left = attrs->GetAttrPointer<int32_t>(4)[0];
  int32_t pad_right = attrs->GetAttrPointer<int32_t>(5)[0];

  uint32_t dPad = depth + (uint32_t)pad_front + (uint32_t)pad_back;
  uint32_t hPad = height + (uint32_t)pad_top + (uint32_t)pad_bottom;
  uint32_t wPad = width + (uint32_t)pad_left + (uint32_t)pad_right;

  uint64_t totalOutput = (uint64_t)batch * channels * dPad * hPad * wPad;
  if (totalOutput >= 0xFFFFFFFFull) totalOutput = 0xFFFFFFFFul;
  uint32_t totalOutputLength = (uint32_t)totalOutput;

  uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = maxTile;
  if (totalOutputLength < tileLen) {
    tileLen = AlignDown(totalOutputLength, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = totalOutputLength;
  }

  uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutputLength, tileLen);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  // Direct assignment (not set_* methods)

  tiling.channels = channels;
  tiling.depth = depth;
  tiling.height = height;
  tiling.width = width;
  tiling.pad_front = pad_front;
  tiling.pad_back = pad_back;
  tiling.pad_top = pad_top;
  tiling.pad_bottom = pad_bottom;
  tiling.pad_left = pad_left;
  tiling.pad_right = pad_right;
  tiling.totalOutputLength = totalOutputLength;
  tiling.tileLength = tileLen;
  tiling.blockDim = blockDim;

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *workspace = context->GetWorkspaceSizes(1);
  workspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}
