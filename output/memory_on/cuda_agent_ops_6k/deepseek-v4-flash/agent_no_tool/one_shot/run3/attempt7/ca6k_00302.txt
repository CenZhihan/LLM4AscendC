#include "ca6k00302_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00302CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // UB budget for two input tiles + one diff tile + one scalar = 4 tiles
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;

    const uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00302Custom {
public:
    __aicore__ inline KernelCa6k00302Custom() {}

    __aicore__ inline void Init(GM_ADDR input, GM_ADDR target, GM_ADDR output,
                                uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inputGm_.SetGlobalBuffer((__gm__ float*)input, totalLen_);
        targetGm_.SetGlobalBuffer((__gm__ float*)target, totalLen_);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, 1);
        pipe_.InitBuffer(inQueueIn_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueTgt_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        float totalSum = 0.0f;
        uint32_t remain = totalLen_;
        uint32_t offset = 0;

        while (remain > 0) {
            uint32_t curLen = (remain > tileLen_) ? tileLen_ : remain;
            CopyIn(offset, curLen);
            Compute(curLen, totalSum);
            offset += curLen;
            remain -= curLen;
        }

        // Write mean = totalSum / totalLen_ as scalar
        AscendC::LocalTensor<float> scalarLocal = outQueue_.AllocTensor<float>();
        scalarLocal.SetValue(0, totalSum / static_cast<float>(totalLen_));
        AscendC::DataCopy(outputGm_, scalarLocal, 1);
        outQueue_.FreeTensor(scalarLocal);
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t len) {
        AscendC::LocalTensor<float> inLocal = inQueueIn_.AllocTensor<float>();
        AscendC::DataCopy(inLocal, inputGm_[offset], len);
        inQueueIn_.EnQue(inLocal);
        AscendC::LocalTensor<float> tgtLocal = inQueueTgt_.AllocTensor<float>();
        AscendC::DataCopy(tgtLocal, targetGm_[offset], len);
        inQueueTgt_.EnQue(tgtLocal);
    }

    __aicore__ inline void Compute(uint32_t len, float &totalSum) {
        AscendC::LocalTensor<float> inLocal = inQueueIn_.DeQue<float>();
        AscendC::LocalTensor<float> tgtLocal = inQueueTgt_.DeQue<float>();
        AscendC::LocalTensor<float> diffLocal = tmpQueue_.AllocTensor<float>();

        // diff = in - tgt
        AscendC::Sub(diffLocal, inLocal, tgtLocal, len);
        // abs(diff)
        AscendC::Abs(diffLocal, diffLocal, len);
        // reduce sum
        float tileSum = AscendC::ReduceAdd(diffLocal, len);
        totalSum += tileSum;

        tmpQueue_.FreeTensor(diffLocal);
        inQueueIn_.FreeTensor(inLocal);
        inQueueTgt_.FreeTensor(tgtLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIn_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTgt_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::GlobalTensor<float> inputGm_;
    AscendC::GlobalTensor<float> targetGm_;
    AscendC::GlobalTensor<float> outputGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00302_custom(GM_ADDR input, GM_ADDR target, GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00302Custom op;
    op.Init(input, target, output, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
'''
