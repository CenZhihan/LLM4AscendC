static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    const gert::StorageShape *xShape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t C = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    uint32_t L = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *outputSizePtr = attrs->GetAttrPointer<int64_t>(0);
    uint32_t outputSize = static_cast<uint32_t>(*outputSizePtr);

    uint32_t tileLen = AlignDown((UB_BUDGET_BYTES - outputSize * sizeof(float)) / sizeof(float), ALIGN_ELEMS);
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;
    if (tileLen > 65536u) tileLen = 65536u;

    uint32_t totalSlices = N * C;
    uint32_t blockDim = totalSlices;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    Ca6k00343CustomTilingData tiling;
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_L(L);
    tiling.set_outputSize(outputSize);
    tiling.set_tilingLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -1e38f;

class KernelCa6k00343Custom {
public:
    __aicore__ inline KernelCa6k00343Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t L,
                                uint32_t outputSize, uint32_t tilingLength) {
        N_ = N;
        C_ = C;
        L_ = L;
        outputSize_ = outputSize;
        tileLen_ = tilingLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * L_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * outputSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outputSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, outputSize_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (N_ == 0 || C_ == 0 || L_ == 0 || outputSize_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t totalSlices = N_ * C_;
        uint32_t slicesPerBlock = CeilDiv(totalSlices, blockNum);
        uint32_t startSlice = blockIdx * slicesPerBlock;
        uint32_t endSlice = startSlice + slicesPerBlock;
        if (endSlice > totalSlices) endSlice = totalSlices;

        for (uint32_t sliceIdx = startSlice; sliceIdx < endSlice; ++sliceIdx) {
            uint32_t n = sliceIdx / C_;
            uint32_t c = sliceIdx % C_;
            ProcessSlice(n, c);
        }
    }

private:
    __aicore__ inline void ProcessSlice(uint32_t n, uint32_t c) {
        uint32_t sliceOffset = (n * C_ + c) * L_;
        uint32_t outOffset = (n * C_ + c) * outputSize_;

        AscendC::LocalTensor<float> maxLocal = outQueueY_.AllocTensor<float>();
        AscendC::Duplicate<float>(maxLocal, NEG_INF, outputSize_);

        uint32_t remaining = L_;
        uint32_t readPos = 0;

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[sliceOffset + readPos], curLen);

            for (uint32_t i = 0; i < curLen; ++i) {
                uint32_t globalIdx = readPos + i;
                uint32_t bin = (static_cast<uint64_t>(globalIdx) * outputSize_) / L_;
                float val = xLocal(i);
                if (val > maxLocal(bin)) {
                    maxLocal(bin) = val;
                }
            }

            inQueueX_.FreeTensor(xLocal);
            readPos += curLen;
            remaining -= curLen;
        }

        outQueueY_.EnQue<float>(maxLocal);
        AscendC::LocalTensor<float> outLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[outOffset], outLocal, outputSize_);
        outQueueY_.FreeTensor(outLocal);
    }

    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t L_ {0};
    uint32_t outputSize_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00343_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00343Custom op;
    op.Init(x, y, tiling_data.N, tiling_data.C, tiling_data.L,
            tiling_data.outputSize, tiling_data.tilingLength);
    op.Process();
}
'''
