    for (uint32_t i = 0; i < x_shape.GetDimNum(); i++) {
        totalLength *= x_shape.GetDim(i);
    }
    uint32_t numChannels = x_shape.GetDim(1);
    uint32_t spatialSize = totalLength / numChannels; // N*H*W

    const gert::RuntimeAttrs *attrs = context->GetAttrs();

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04586Custom {
public:
    __aicore__ inline KernelCa6k04586Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR running_mean, GM_ADDR running_var,
                                GM_ADDR running_mean_out, GM_ADDR running_var_out,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t numChannels,
                                uint32_t spatialSize, uint32_t training,
                                float momentum, float eps)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numChannels);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, numChannels);
        runningMeanGm_.SetGlobalBuffer((__gm__ float*)running_mean, numChannels);
        runningVarGm_.SetGlobalBuffer((__gm__ float*)running_var, numChannels);
        runningMeanOutGm_.SetGlobalBuffer((__gm__ float*)running_mean_out, numChannels);
        runningVarOutGm_.SetGlobalBuffer((__gm__ float*)running_var_out, numChannels);

        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        numChannels_ = numChannels;
        spatialSize_ = spatialSize;
        training_ = training != 0;
        momentum_ = momentum;
        eps_ = eps;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = blockDim_;

        // Each block processes a contiguous set of channels
        uint32_t channelsPerBlock = (numChannels_ + blockNum - 1) / blockNum;
        uint32_t startChannel = blockIdx * channelsPerBlock;
        uint32_t endChannel = startChannel + channelsPerBlock;
        if (endChannel > numChannels_) endChannel = numChannels_;
        if (startChannel >= numChannels_) return;

        for (uint32_t c = startChannel; c < endChannel; c++) {
            float w = weightGm_.GetValue(c);
            float b = biasGm_.GetValue(c);
            float rm = runningMeanGm_.GetValue(c);
            float rv = runningVarGm_.GetValue(c);

            float mean, variance;
            if (training_) {
                // First pass: compute sum and sum_sq for this channel
                float sum = 0.0f;
                float sum_sq = 0.0f;
                uint32_t remaining = spatialSize_;
                uint32_t offset = c * spatialSize_;
                while (remaining > 0) {
                    uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                    // Load tile
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[offset], curLen);
                    inQueueX_.EnQue(xLocal);

                    xLocal = inQueueX_.DeQue<float>();
                    AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

                    // sum
                    AscendC::ReduceSum(tLocal, xLocal, curLen);
                    sum += tLocal.GetValue(0);
                    // sum_sq
                    AscendC::Mul(tLocal, xLocal, xLocal, curLen);
                    AscendC::ReduceSum(tLocal, tLocal, curLen);
                    sum_sq += tLocal.GetValue(0);

                    tmpQueue_.FreeTensor(tLocal);
                    inQueueX_.FreeTensor(xLocal);
                    offset += curLen;
                    remaining -= curLen;
                }
                float inv_size = 1.0f / static_cast<float>(spatialSize_);
                mean = sum * inv_size;
                variance = sum_sq * inv_size - mean * mean;
                if (variance < 0.0f) variance = 0.0f;

                // Update running statistics (in-place writing)
                float newMean = (1.0f - momentum_) * rm + momentum_ * mean;
                float newVar = (1.0f - momentum_) * rv + momentum_ * variance;
                runningMeanOutGm_.SetValue(c, newMean);
                runningVarOutGm_.SetValue(c, newVar);
            } else {
                mean = rm;
                variance = rv;
                runningMeanOutGm_.SetValue(c, rm);
                runningVarOutGm_.SetValue(c, rv);
            }

            // Second pass: normalize and apply scale/shift
            float inv_std = 1.0f / sqrt(variance + eps_);
            uint32_t remaining = spatialSize_;
            uint32_t offset = c * spatialSize_;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

                // Load x
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curLen);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> tLocal2 = tmpQueue2_.AllocTensor<float>();

                // (x - mean)
                AscendC::Sub(tLocal, xLocal, mean, curLen);
                // * inv_std
                AscendC::Muls(tLocal, tLocal, inv_std, curLen);
                // * weight
                AscendC::Muls(yLocal, tLocal, w, curLen);
                // + bias
                AscendC::Add(yLocal, yLocal, b, curLen);

                tmpQueue_.FreeTensor(tLocal);
                tmpQueue2_.FreeTensor(tLocal2);
                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xLocal);

                // Store output
                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[offset], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);

                offset += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> runningMeanGm_;
    AscendC::GlobalTensor<float> runningVarGm_;
    AscendC::GlobalTensor<float> runningMeanOutGm_;
    AscendC::GlobalTensor<float> runningVarOutGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t blockDim_;
    uint32_t numChannels_;
    uint32_t spatialSize_;
    bool training_;
    float momentum_;
    float eps_;
};

extern "C" __global__ __aicore__ void ca6k_04586_custom(GM_ADDR x, GM_ADDR y,
                                                        GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR running_mean, GM_ADDR running_var,
                                                        GM_ADDR running_mean_out, GM_ADDR running_var_out,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04586Custom op;
    op.Init(x, y, weight, bias, running_mean, running_var,
            running_mean_out, running_var_out,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.blockDim, tiling_data.numChannels,
            tiling_data.spatialSize, tiling_data.training,
            tiling_data.momentum, tiling_data.eps);
    op.Process();
}

class ModelNew(torch.nn.Module):
    def __init__(self, num_features, momentum=0.1, eps=1e-05, affine=True):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.momentum = momentum
        self.eps = eps
        self.weight = torch.nn.Parameter(torch.ones(num_features))
        self.bias = torch.nn.Parameter(torch.zeros(num_features))
        self.register_buffer('running_mean', torch.zeros(num_features))
        self.register_buffer('running_var', torch.ones(num_features))
        self.training = True

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        y, rm_out, rv_out = fn(x, self.weight, self.bias, self.running_mean, self.running_var,
                                self.training, self.momentum, self.eps)
        # Update running statistics buffers
