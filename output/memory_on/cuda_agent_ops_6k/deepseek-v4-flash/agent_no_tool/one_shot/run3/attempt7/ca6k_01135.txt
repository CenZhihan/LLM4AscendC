#include "ca6k_01135_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t TILE_H = 32;
static constexpr uint32_t TILE_W = 32;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k01135CustomTilingData tiling;

    auto originShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(3));

    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    float scale = *(attrs->GetAttrPointer<float>(0));
    uint32_t H_out = static_cast<uint32_t>(H * scale);
    uint32_t W_out = static_cast<uint32_t>(W * scale);

    // total output size for blockDim heuristic
    uint32_t totalOutputElems = N * C * H_out * W_out;
    uint32_t numTiles = CeilDiv(H_out, TILE_H) * CeilDiv(W_out, TILE_W);
    numTiles *= (N * C);  // per channel per batch
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_inputH(H);
    tiling.set_inputW(W);
    tiling.set_outputH(H_out);
    tiling.set_outputW(W_out);
    tiling.set_scaleFactor(scale);
    tiling.set_tileOutputH(TILE_H);
    tiling.set_tileOutputW(TILE_W);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src = '''
#include "kernel_operator.h"
#include "ca6k_01135_custom_tiling.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01135Custom {
public:
    __aicore__ inline KernelCa6k01135Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t outputH, uint32_t outputW,
                                float scaleFactor,
                                uint32_t tileOutputH, uint32_t tileOutputW)
    {
        N_ = N; C_ = C;
        inputH_ = inputH; inputW_ = inputW;
        outputH_ = outputH; outputW_ = outputW;
        scaleFactor_ = scaleFactor;
        tileOutputH_ = tileOutputH; tileOutputW_ = tileOutputW;
        tileInputH_ = (tileOutputH_ + scaleFactor_ - 1) / scaleFactor_;
        tileInputW_ = (tileOutputW_ + scaleFactor_ - 1) / scaleFactor_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * inputH_ * inputW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * outputH_ * outputW_);
        // local tensors for input tile and output tile
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileInputH_ * tileInputW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOutputH_ * tileOutputW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outputH_ == 0 || outputW_ == 0) return;
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();
        // distribute tiles among blocks (tiles over output pixels)
        uint32_t tilesPerBatchChannel = ((outputH_ + tileOutputH_ - 1) / tileOutputH_) *
                                         ((outputW_ + tileOutputW_ - 1) / tileOutputW_);
        uint32_t totalTiles = N_ * C_ * tilesPerBatchChannel;
        uint32_t tilesPerBlock = (totalTiles + blockNum - 1) / blockNum;
        uint32_t tileStart = blockIdx * tilesPerBlock;
        uint32_t tileEnd = tileStart + tilesPerBlock;
        if (tileEnd > totalTiles) tileEnd = totalTiles;

        for (uint32_t tileId = tileStart; tileId < tileEnd; tileId++) {
            // decode tileId: n, c, tileRow, tileCol
            uint32_t tmp = tileId;
            uint32_t localTileId = tmp % tilesPerBatchChannel; tmp /= tilesPerBatchChannel;
            uint32_t c = tmp % C_; tmp /= C_;
            uint32_t n = tmp;

            uint32_t tilesPerRow = (outputW_ + tileOutputW_ - 1) / tileOutputW_;
            uint32_t tileRow = localTileId / tilesPerRow;
            uint32_t tileCol = localTileId % tilesPerRow;

            uint32_t outStartH = tileRow * tileOutputH_;
            uint32_t outStartW = tileCol * tileOutputW_;
            uint32_t outEndH = outStartH + tileOutputH_;
            if (outEndH > outputH_) outEndH = outputH_;
            uint32_t outEndW = outStartW + tileOutputW_;
            if (outEndW > outputW_) outEndW = outputW_;
            uint32_t actualTileOutH = outEndH - outStartH;
            uint32_t actualTileOutW = outEndW - outStartW;

            // corresponding input tile (nearest neighbor mapping)
            uint32_t inStartH = outStartH / scaleFactor_;
            uint32_t inStartW = outStartW / scaleFactor_;
            uint32_t inEndH = (outEndH - 1) / scaleFactor_ + 1;
            if (inEndH > inputH_) inEndH = inputH_;
            uint32_t inEndW = (outEndW - 1) / scaleFactor_ + 1;
            if (inEndW > inputW_) inEndW = inputW_;
            uint32_t actualTileInH = inEndH - inStartH;
            uint32_t actualTileInW = inEndW - inStartW;

            // copy input tile
            LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            // copy row by row (contiguous rows)
            for (uint32_t r = 0; r < actualTileInH; r++) {
                uint32_t srcOffset = ((n * C_ + c) * inputH_ + (inStartH + r)) * inputW_ + inStartW;
                uint32_t dstOffset = r * tileInputW_;  // tileInputW_ is max width, actualTileInW <= tileInputW_
                DataCopy(xLocal[dstOffset], xGm_[srcOffset], actualTileInW);
            }
            inQueueX_.EnQue(xLocal);

            // compute output tile
            LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // zero out first
            for (uint32_t i = 0; i < tileOutputH_ * tileOutputW_; i++) {
                yLocal.SetValue(i, 0.0f);
            }
            // fill
            for (uint32_t oh = 0; oh < actualTileOutH; oh++) {
                uint32_t ih = (outStartH + oh) / scaleFactor_ - inStartH;
                for (uint32_t ow = 0; ow < actualTileOutW; ow++) {
                    uint32_t iw = (outStartW + ow) / scaleFactor_ - inStartW;
                    float val = xLocal[ih * tileInputW_ + iw];
                    yLocal[oh * tileOutputW_ + ow] = val;
                }
            }
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);

            // copy output tile
            LocalTensor<float> yLocalOut = outQueueY_.DeQue<float>();
            for (uint32_t r = 0; r < actualTileOutH; r++) {
                uint32_t dstOffset = ((n * C_ + c) * outputH_ + (outStartH + r)) * outputW_ + outStartW;
                uint32_t srcOffset = r * tileOutputW_;
                DataCopy(yGm_[dstOffset], yLocalOut[srcOffset], actualTileOutW);
            }
            outQueueY_.FreeTensor(yLocalOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_;
    uint32_t inputH_, inputW_;
    uint32_t outputH_, outputW_;
    float scaleFactor_;
    uint32_t tileOutputH_, tileOutputW_;
    uint32_t tileInputH_, tileInputW_;
};

extern "C" __global__ __aicore__ void ca6k_01135_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01135Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C,
            tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputH, tiling_data.outputW,
            tiling_data.scaleFactor,
            tiling_data.tileOutputH, tiling_data.tileOutputW);
    op.Process();
}
'''

at::Tensor ca6k_01135_impl_npu(const at::Tensor& self, double scale_factor) {
    int64_t N = self.size(0);
    int64_t C = self.size(1);
    int64_t H = self.size(2);
    int64_t W = self.size(3);
