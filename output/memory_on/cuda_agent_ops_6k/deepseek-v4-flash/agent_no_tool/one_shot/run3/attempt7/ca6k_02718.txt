kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -1e30f;

class KernelCa6k02718Custom {
public:
    __aicore__ inline KernelCa6k02718Custom() {}

    __aicore__ inline void Init(GM_ADDR log_probs, GM_ADDR targets,
                                GM_ADDR input_lengths, GM_ADDR target_lengths,
                                GM_ADDR loss,
                                uint32_t totalBatch,
                                uint32_t seqLength,
                                uint32_t numClasses, uint32_t maxTargetLength,
                                int32_t blank, int32_t reduction)
    {
        totalBatch_ = totalBatch;
        seqLength_ = seqLength;
        numClasses_ = numClasses;
        maxTargetLength_ = maxTargetLength;
        blank_ = blank;
        reduction_ = reduction;

        logProbsGm_.SetGlobalBuffer((__gm__ float*)log_probs, totalBatch_ * seqLength_ * numClasses_);
        targetsGm_.SetGlobalBuffer((__gm__ int32_t*)targets, totalBatch_ * maxTargetLength_);
        inputLengthsGm_.SetGlobalBuffer((__gm__ int32_t*)input_lengths, totalBatch_);
        targetLengthsGm_.SetGlobalBuffer((__gm__ int32_t*)target_lengths, totalBatch_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        // small per-time-step log_probs buffer
        pipe_.InitBuffer(inQueueLogProbs_, BUFFER_NUM, numClasses_ * sizeof(float));
        // targets for one sample
        pipe_.InitBuffer(inQueueTargets_, BUFFER_NUM, maxTargetLength_ * sizeof(int32_t));
        // alpha arrays size = 2 * maxTargetLength + 1 + 1 (for safety)
        uint32_t alphaSize = (maxTargetLength_ * 2 + 4) * sizeof(float);
        pipe_.InitBuffer(alphaPrevQue_, 1, alphaSize);
        pipe_.InitBuffer(alphaCurrQue_, 1, alphaSize);
        // single element for log-sum-exp
        pipe_.InitBuffer(scalarQue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalBatch_ == 0) return;
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx != 0) return; // single core

        float totalLoss = 0.0f;
        uint32_t offsetLogProbs = 0;
        uint32_t offsetTargets = 0;

        for (uint32_t b = 0; b < totalBatch_; ++b) {
            int32_t T = inputLengthsGm_.GetValue(b);
            int32_t U = targetLengthsGm_.GetValue(b);
            if (U <= 0 || T <= 0) {
                offsetLogProbs += seqLength_ * numClasses_;
                offsetTargets += maxTargetLength_;
                continue;
            }

            // Copy targets for this sample
            AscendC::LocalTensor<int32_t> targetsLocal = inQueueTargets_.AllocTensor<int32_t>();
            AscendC::DataCopy(targetsLocal, targetsGm_[offsetTargets], maxTargetLength_);
            inQueueTargets_.EnQue(targetsLocal);

            // Allocate alpha buffers
            AscendC::LocalTensor<float> alphaPrev = alphaPrevQue_.AllocTensor<float>();
            AscendC::LocalTensor<float> alphaCurr = alphaCurrQue_.AllocTensor<float>();
            uint32_t S = 2u * static_cast<uint32_t>(U) + 1u;

            // Initialize alpha for t=0
            for (uint32_t s = 0; s < S; ++s) {
                alphaPrev.SetValue(s, NEG_INF);
            }
            // alpha0(0) = log_probs[0, blank]
            float firstProb = logProbsGm_.GetValue(offsetLogProbs + blank_);
            alphaPrev.SetValue(0, firstProb);
            if (U > 0) {
                alphaPrev.SetValue(1, logProbsGm_.GetValue(offsetLogProbs + targetsLocal.GetValue(0)));
            }

            // Loop over time
            for (int32_t t = 1; t < T; ++t) {
                // Load log_probs for this time step
                AscendC::LocalTensor<float> logProbsLocal = inQueueLogProbs_.AllocTensor<float>();
                AscendC::DataCopy(logProbsLocal, logProbsGm_[offsetLogProbs + t * numClasses_], numClasses_);
                inQueueLogProbs_.EnQue(logProbsLocal);
                AscendC::LocalTensor<float> logP = inQueueLogProbs_.DeQue<float>();

                for (uint32_t s = 0; s < S; ++s) {
                    alphaCurr.SetValue(s, NEG_INF);
                }

                for (uint32_t s = 0; s < S; ++s) {
                    int32_t lbl = (s % 2 == 0) ? blank_ : targetsLocal.GetValue(static_cast<int32_t>(s / 2));
                    float prob = logP.GetValue(lbl);

                    float sum = alphaPrev.GetValue(s) + prob;
                    alphaCurr.SetValue(s, LogSumExp(alphaCurr.GetValue(s), sum));

                    if (s > 0) {
                        sum = alphaPrev.GetValue(s - 1) + prob;
                        alphaCurr.SetValue(s, LogSumExp(alphaCurr.GetValue(s), sum));
                    }

                    if (s > 1) {
                        int32_t lbl2 = (s % 2 == 0) ? blank_ : targetsLocal.GetValue(static_cast<int32_t>(s / 2));
                        int32_t lblPrev2 = ((s - 2) % 2 == 0) ? blank_ : targetsLocal.GetValue(static_cast<int32_t>((s - 2) / 2));
                        if (lbl2 != lblPrev2) {
                            sum = alphaPrev.GetValue(s - 2) + prob;
                            alphaCurr.SetValue(s, LogSumExp(alphaCurr.GetValue(s), sum));
                        }
                    }
                }

                // Free logP and swap
                inQueueLogProbs_.FreeTensor(logP);
                AscendC::LocalTensor<float> tmp = alphaPrev;
                alphaPrev = alphaCurr;
                alphaCurr = tmp;
            }

            float sampleLoss = -LogSumExp(alphaPrev.GetValue(S - 1), alphaPrev.GetValue(S - 2));
            totalLoss += sampleLoss;

            // Free alpha buffers and target buffer
            alphaPrevQue_.FreeTensor(alphaPrev);
            alphaCurrQue_.FreeTensor(alphaCurr);
            AscendC::LocalTensor<int32_t> tgt = inQueueTargets_.DeQue<int32_t>();
            inQueueTargets_.FreeTensor(tgt);

            offsetLogProbs += seqLength_ * numClasses_;
            offsetTargets += maxTargetLength_;
        }

        if (reduction_ == 1) { // mean
            totalLoss = totalLoss / static_cast<float>(totalBatch_);
        }
        lossGm_.SetValue(0, totalLoss);
    }

private:
    __aicore__ inline float LogSumExp(float a, float b)
    {
        if (a < -1e20f && b < -1e20f) return NEG_INF;
        float maxVal = (a > b) ? a : b;
        float diff = (a > b) ? (b - a) : (a - b);
        AscendC::LocalTensor<float> scalar = scalarQue_.AllocTensor<float>();
        scalar.SetValue(0, diff);
        AscendC::Exp(scalar, scalar, 1);
        float expDiff = scalar.GetValue(0);
        // 1.0f + expDiff
        scalar.SetValue(0, 1.0f + expDiff);
        AscendC::Log(scalar, scalar, 1);
        float logVal = scalar.GetValue(0);
        scalarQue_.FreeTensor(scalar);
        return maxVal + logVal;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueLogProbs_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTargets_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> alphaPrevQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> alphaCurrQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQue_;

    AscendC::GlobalTensor<float> logProbsGm_;
    AscendC::GlobalTensor<int32_t> targetsGm_;
    AscendC::GlobalTensor<int32_t> inputLengthsGm_;
    AscendC::GlobalTensor<int32_t> targetLengthsGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalBatch_;
    uint32_t seqLength_;
    uint32_t numClasses_;
    uint32_t maxTargetLength_;
    int32_t blank_;
    int32_t reduction_;
};

extern "C" __global__ __aicore__ void ca6k02718_custom(GM_ADDR log_probs, GM_ADDR targets,
                                                         GM_ADDR input_lengths, GM_ADDR target_lengths,
                                                         GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02718Custom op;
    op.Init(log_probs, targets, input_lengths, target_lengths, loss,
            tiling_data.totalBatch,
            tiling_data.seqLength, tiling_data.numClasses,
            tiling_data.maxTargetLength, tiling_data.blank, tiling_data.reduction);
    op.Process();
}
