static constexpr uint32_t MAX_BLOCK_DIM = 24;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00346CustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::Shape &origin = x_shape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t C = static_cast<uint32_t>(origin.GetDim(1));
    uint32_t inputH = static_cast<uint32_t>(origin.GetDim(2));
    uint32_t inputW = static_cast<uint32_t>(origin.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scaleFactor = attrs->GetAttrPointer<float>(0);
    float sf = *scaleFactor;
    uint32_t scaleInt = static_cast<uint32_t>(sf + 0.5f); // assume integer scale

    uint32_t outputH = static_cast<uint32_t>(inputH * sf);
    uint32_t outputW = static_cast<uint32_t>(inputW * sf);

    uint32_t tileH = 16;
    uint32_t tileW = 16;

    uint32_t numTilesH = CeilDiv(outputH, tileH);
    uint32_t numTilesW = CeilDiv(outputW, tileW);
    uint32_t totalTiles = N * C * numTilesH * numTilesW;

    uint32_t blockDim = (totalTiles < MAX_BLOCK_DIM) ? totalTiles : MAX_BLOCK_DIM;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_totalTiles(totalTiles);
    tiling.set_scaleInt(scaleInt);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src='''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_H_OUT = 16;
constexpr int32_t TILE_W_OUT = 16;

class KernelCa6k00346Custom {
public:
    __aicore__ inline KernelCa6k00346Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t outputH, uint32_t outputW,
                                uint32_t tileH, uint32_t tileW,
                                uint32_t totalTiles,
                                uint32_t scaleInt) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * inputH * inputW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C * outputH * outputW);

        N_ = N;
        C_ = C;
        inputH_ = inputH;
        inputW_ = inputW;
        outputH_ = outputH;
        outputW_ = outputW;
        tileHOut_ = tileH;
        tileWOut_ = tileW;
        totalTiles_ = totalTiles;
        scaleInt_ = scaleInt;

        numTilesH_ = (outputH_ + tileHOut_ - 1) / tileHOut_;
        numTilesW_ = (outputW_ + tileWOut_ - 1) / tileWOut_;
        tilesPerChannel_ = numTilesH_ * numTilesW_;

        // Each output tile maps to an input tile of size (tileHOut/scaleInt) x (tileWOut/scaleInt) plus one extra for bilinear
        inputTileH_ = (tileHOut_ + scaleInt_ - 1) / scaleInt_ + 1;
        inputTileW_ = (tileWOut_ + scaleInt_ - 1) / scaleInt_ + 1;
        maxInputElems_ = inputTileH_ * inputTileW_;

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, maxInputElems_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileHOut_ * tileWOut_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t tilesPerBlock = (totalTiles_ + blockNum - 1) / blockNum;
        uint32_t tileStart = blockIdx * tilesPerBlock;
        uint32_t tileEnd = tileStart + tilesPerBlock;
        if (tileEnd > totalTiles_) tileEnd = totalTiles_;

        for (uint32_t linearTileIdx = tileStart; linearTileIdx < tileEnd; ++linearTileIdx) {
            ProcessOneTile(linearTileIdx);
        }
    }

private:
    __aicore__ inline void ProcessOneTile(uint32_t linearTileIdx) {
        uint32_t n = linearTileIdx / (C_ * tilesPerChannel_);
        uint32_t rem = linearTileIdx % (C_ * tilesPerChannel_);
        uint32_t c = rem / tilesPerChannel_;
        uint32_t tileIdxInChannel = rem % tilesPerChannel_;
        uint32_t tileRow = tileIdxInChannel / numTilesW_;
        uint32_t tileCol = tileIdxInChannel % numTilesW_;

        uint32_t oyStart = tileRow * tileHOut_;
        uint32_t oxStart = tileCol * tileWOut_;
        uint32_t oyEnd = oyStart + tileHOut_;
        if (oyEnd > outputH_) oyEnd = outputH_;
        uint32_t oxEnd = oxStart + tileWOut_;
        if (oxEnd > outputW_) oxEnd = outputW_;
        uint32_t actualTileH = oyEnd - oyStart;
        uint32_t actualTileW = oxEnd - oxStart;

        // Compute input tile boundaries (integer arithmetic)
        uint32_t iyStart = oyStart / scaleInt_;
        uint32_t iyEndInput = (oyEnd - 1) / scaleInt_ + 1; // inclusive last input row needed (for interpolation)
        // Add one extra row for bilinear (needs iy+1 for odd rows)
        uint32_t iyEnd = iyEndInput + 1;
        if (iyEnd > inputH_) iyEnd = inputH_;
        uint32_t ixStart = oxStart / scaleInt_;
        uint32_t ixEndInput = (oxEnd - 1) / scaleInt_ + 1;
        uint32_t ixEnd = ixEndInput + 1;
        if (ixEnd > inputW_) ixEnd = inputW_;

        uint32_t inputRows = iyEnd - iyStart;
        uint32_t inputCols = ixEnd - ixStart;

        // Allocate local tensor for input patch
        AscendC::LocalTensor<float> inputPatch = inQueue_.AllocTensor<float>();

        // Load input patch row by row
        for (uint32_t r = 0; r < inputRows; ++r) {
            uint32_t globalBase = (n * C_ * inputH_ * inputW_) +
                                  (c * inputH_ * inputW_) +
                                  ((iyStart + r) * inputW_) + ixStart;
            AscendC::DataCopy(inputPatch[r * inputCols], xGm_[globalBase], inputCols);
        }

        // Compute output tile using vectorized bilinear interpolation
        AscendC::LocalTensor<float> outTile = outQueue_.AllocTensor<float>();

        // For each output row, compute from input rows using linear interpolation
        for (uint32_t oy = 0; oy < actualTileH; ++oy) {
            uint32_t globalIy = oyStart + oy;
            uint32_t iy0 = globalIy / scaleInt_;
            uint32_t iy1 = iy0 + 1;
            if (iy1 >= inputH_) iy1 = iy0; // clamp
            float dy = static_cast<float>(globalIy % scaleInt_) / static_cast<float>(scaleInt_);

            // Get slices of input rows
            uint32_t localIy0 = iy0 - iyStart;
            uint32_t localIy1 = iy1 - iyStart;

            // For each output column, compute interpolation along X then Y
            for (uint32_t ox = 0; ox < actualTileW; ++ox) {
                uint32_t globalIx = oxStart + ox;
                uint32_t ix0 = globalIx / scaleInt_;
                uint32_t ix1 = ix0 + 1;
                if (ix1 >= inputW_) ix1 = ix0;
                float dx = static_cast<float>(globalIx % scaleInt_) / static_cast<float>(scaleInt_);

                // Gather four pixel values from inputPatch
                // Since we cannot index LocalTensor directly, we load 2x2 block into scalar tensors using DataCopy
                // We'll use a 1-element helper tensor
                AscendC::LocalTensor<float> tmp = pipe_.AllocTensor<float>(1);
                uint32_t localIx0 = ix0 - ixStart;
                uint32_t localIx1 = ix1 - ixStart;

                // v00
                AscendC::DataCopy(tmp, inputPatch[localIy0 * inputCols + localIx0], 1);
                float v00 = tmp(0);
                // v01
                AscendC::DataCopy(tmp, inputPatch[localIy0 * inputCols + localIx1], 1);
                float v01 = tmp(0);
                // v10
                AscendC::DataCopy(tmp, inputPatch[localIy1 * inputCols + localIx0], 1);
                float v10 = tmp(0);
                // v11
                AscendC::DataCopy(tmp, inputPatch[localIy1 * inputCols + localIx1], 1);
                float v11 = tmp(0);
                pipe_.FreeTensor(tmp);

                float top = v00 + dx * (v01 - v00);
                float bottom = v10 + dx * (v11 - v10);
                float val = top + dy * (bottom - top);

                // Write val to outTile using DataCopy single element
                AscendC::LocalTensor<float> outElem = pipe_.AllocTensor<float>(1);
                outElem.SetValue(0, val);
                AscendC::DataCopy(outTile[oy * actualTileW + ox], outElem, 1);
                pipe_.FreeTensor(outElem);
            }
        }

        // Write output tile to global memory row by row
        for (uint32_t r = 0; r < actualTileH; ++r) {
            uint32_t globalOutBase = (n * C_ * outputH_ * outputW_) +
                                     (c * outputH_ * outputW_) +
                                     ((oyStart + r) * outputW_) + oxStart;
            AscendC::DataCopy(yGm_[globalOutBase], outTile[r * actualTileW], actualTileW);
        }

        // Free local tensors
        inQueue_.FreeTensor(inputPatch);
        outQueue_.FreeTensor(outTile);
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, inputH_, inputW_, outputH_, outputW_;
    uint32_t tileHOut_, tileWOut_, numTilesH_, numTilesW_, tilesPerChannel_, totalTiles_;
    uint32_t inputTileH_, inputTileW_, maxInputElems_;
    uint32_t scaleInt_;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
};

extern "C" __global__ __aicore__ void ca6k00346_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00346Custom op;
    op.Init(x, y,
            tiling_data.N,
            tiling_data.C,
            tiling_data.inputH,
            tiling_data.inputW,
            tiling_data.outputH,
            tiling_data.outputW,
            tiling_data.tileH,
            tiling_data.tileW,
            tiling_data.totalTiles,
            tiling_data.scaleInt);
    op.Process();
}

#include <torch/library.h>

at::Tensor ca6k00346_impl_npu(const at::Tensor& self, double scale_factor) {
    int64_t H = self.size(2);
    int64_t W = self.size(3);
    int64_t outH = static_cast<int64_t>(H * scale_factor);
    int64_t outW = static_cast<int64_t>(W * scale_factor);
    at::Tensor result = at::empty({self.size(0), self.size(1), outH, outW}, self.options());
