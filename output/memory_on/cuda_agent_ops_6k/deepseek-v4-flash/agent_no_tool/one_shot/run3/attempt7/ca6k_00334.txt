host_operator_src="""
#include "ca6k_00334_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00334CustomTilingData tiling;
    const gert::Shape &shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalLength = static_cast<uint32_t>(shape.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim_ptr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dim_ptr != nullptr) ? *dim_ptr : 0;

    const uint32_t ndim = static_cast<uint32_t>(shape.GetDimNum());
    // Clamp dim to valid range
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= static_cast<int64_t>(ndim)) dim = 0;
    uint32_t udim = static_cast<uint32_t>(dim);

    uint32_t dimSize = static_cast<uint32_t>(shape.GetDim(udim));

    // compute outerSize (product of dims before dim) and innerSize (product after dim)
    uint32_t outerSize = 1;
    for (uint32_t i = 0; i < udim; ++i)
        outerSize *= static_cast<uint32_t>(shape.GetDim(i));
    uint32_t innerSize = 1;
    for (uint32_t i = udim + 1; i < ndim; ++i)
        innerSize *= static_cast<uint32_t>(shape.GetDim(i));

    uint32_t stride = innerSize; // step between consecutive elements along dim
    uint32_t numSlices = outerSize * innerSize; // number of independent cumprod sequences

    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = (totalLength < maxTile) ? AlignDown(totalLength, ALIGN_ELEMS) : maxTile;
    if (tileLen == 0) tileLen = totalLength;

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dimSize(dimSize);
    tiling.set_stride(stride);
    tiling.set_numSlices(numSlices);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    gert::Shape *y_shape = context->GetOutputShape(0);

    context->SetOutputDataType(0, x_dtype);

}

kernel_src="""
#include "kernel_operator.h"
#include "ca6k_00334_custom_tiling.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00334Custom {
public:
    __aicore__ inline KernelCa6k00334Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t dimSize,
                                uint32_t stride, uint32_t numSlices) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        dimSize_ = dimSize;
        stride_ = stride;
        numSlices_ = numSlices;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        // Each block processes a subset of slices
        for (uint32_t s = blockIdx; s < numSlices_; s += blockNum) {
            // Compute the starting index of this slice
            // The slice is at base + i*stride for i in [0, dimSize_)
            // base = (s / innerSize) * (dimSize_ * stride_) + (s % innerSize)
            uint32_t outerIdx = s / stride_; // number of full strides within outer*inner?? Actually correct formula:
            // Let stride = innerSize. Then for a given slice index s (0..outerSize*innerSize-1),
            // the starting offset is: (s / innerSize) * (dimSize_ * stride_) + (s % innerSize)
            // Because s = outerIdx * innerSize + innerIdx, and the start offset = outerIdx * (dimSize_*stride) + innerIdx
            uint32_t innerIdx = s % stride_;
            uint32_t outerIdxSlice = s / stride_;
            uint32_t base = outerIdxSlice * (dimSize_ * stride_) + innerIdx;

            float cumulative = 1.0f;

            // Process each element of the slice
            for (uint32_t i = 0; i < dimSize_; ++i) {
                uint32_t gmIndex = base + i * stride_;
                float xVal = xGm_.GetValue(gmIndex);  // direct GM read for simplicity
                cumulative *= xVal;
                yGm_.SetValue(gmIndex, cumulative);
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
    uint32_t dimSize_ {0};
    uint32_t stride_ {1};
    uint32_t numSlices_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00334_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00334Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.blockDim, tiling_data.dimSize,
            tiling_data.stride, tiling_data.numSlices);
    op.Process();
}

import torch

_FN = 'ca6k_00334_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        # init_inputs[0] is shape (not used directly), init_inputs[1] is dim
