#include "ca6k00071_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00071CustomTilingData tiling;
    const gert::StorageShape *xStorage = context->GetInputShape(0);
    const gert::Shape &xShape = xStorage->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(xShape.GetShapeSize());
    uint32_t ndim = static_cast<uint32_t>(xShape.GetDimNum());
    uint32_t numRows = (ndim >= 1) ? static_cast<uint32_t>(xShape.GetDim(0)) : 0;
    uint32_t numCols = (ndim >= 2) ? static_cast<uint32_t>(xShape.GetDim(1)) : 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t dim = *(attrs->GetAttrPointer<int32_t>(0));

    uint32_t maxColTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxColTile = AlignDown(maxColTile, ALIGN_ELEMS);
    if (maxColTile < ALIGN_ELEMS) maxColTile = ALIGN_ELEMS;
    if (maxColTile > HARD_CAP_ELEMS) maxColTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (numCols < maxColTile) ? numCols : maxColTile;
    if (tileLen == 0) tileLen = 1;

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_numRows(numRows);
    tiling.set_numCols(numCols);
    tiling.set_tileLength(tileLen);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    gert::Shape *y_shape = context->GetOutputShape(0);

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00071Custom {
public:
    __aicore__ inline KernelCa6k00071Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t numRows, uint32_t numCols,
                                uint32_t tileLength, int32_t dim)
    {
        totalLen_ = totalLength;
        N_ = numRows;
        C_ = numCols;
        tileLen_ = tileLength;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0u || tileLen_ == 0u) return;
        if (dim_ != 0) return; // only dim 0 implemented

        uint32_t colTile = (tileLen_ < C_) ? tileLen_ : C_;
        uint32_t cStart = 0u;
        while (cStart < C_) {
            uint32_t curColTile = (cStart + colTile <= C_) ? colTile : (C_ - cStart);

            AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();
            // Initialize accumulator to zero
            AscendC::Duplicate(acc, 0.0f, curColTile);
            for (uint32_t i = 0u; i < N_; ++i) {
                uint32_t offset = i * C_ + cStart;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curColTile);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
                AscendC::Add(acc, acc, xDeq, curColTile);

                AscendC::DataCopy(yGm_[offset], acc, curColTile);

                inQueueX_.FreeTensor(xDeq);
            }
            accQueue_.FreeTensor(acc);
            cStart += colTile;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0u};
    uint32_t N_ {0u};
    uint32_t C_ {0u};
    uint32_t tileLen_ {0u};
    int32_t dim_ {0};
};

extern "C" __global__ __aicore__ void ca6k00071_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
)
