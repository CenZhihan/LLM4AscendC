#include "ca6k00415_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00415CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *numGroups = attrs->GetAttrPointer<int64_t>(0);
    const float *eps = attrs->GetAttrPointer<float>(1);

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(shape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(shape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(shape.GetDim(3));

    uint32_t channelsPerGroup = C / static_cast<uint32_t>(*numGroups);
    uint32_t groupSize = channelsPerGroup * H * W;

    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (groupSize < tileLen) {
        tileLen = AlignDown(groupSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = groupSize;
    }

    uint32_t blockDim = N * static_cast<uint32_t>(*numGroups);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 512) blockDim = 512;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_numGroups(static_cast<uint32_t>(*numGroups));
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_channelsPerGroup(channelsPerGroup);
    tiling.set_groupSize(groupSize);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00415Custom {
public:
    __aicore__ inline KernelCa6k00415Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength, uint32_t blockDim,
                                uint32_t numGroups, uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t channelsPerGroup, uint32_t groupSize, float eps)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        numGroups_ = numGroups;
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        channelsPerGroup_ = channelsPerGroup;
        groupSize_ = groupSize;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, C_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, C_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, sizeof(float));
        pipe_.InitBuffer(tmpBuf2_, 1, sizeof(float));
        pipe_.InitBuffer(gammaBuf_, channelsPerGroup_ * sizeof(float));
        pipe_.InitBuffer(betaBuf_, channelsPerGroup_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (groupSize_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= N_ * numGroups_) return;

        uint32_t batch = blockIdx / numGroups_;
        uint32_t groupInBatch = blockIdx % numGroups_;
        uint32_t groupStartChannel = groupInBatch * channelsPerGroup_;
        uint32_t groupOffset = batch * C_ * H_ * W_ + groupStartChannel * H_ * W_;

        AscendC::LocalTensor<float> gammaLocal = gammaBuf_.Get<float>();
        AscendC::LocalTensor<float> betaLocal = betaBuf_.Get<float>();
        AscendC::DataCopy(gammaLocal, gammaGm_[groupStartChannel], channelsPerGroup_);
        AscendC::DataCopy(betaLocal, betaGm_[groupStartChannel], channelsPerGroup_);

        // Pass 1: compute mean and variance (Welford)
        AscendC::LocalTensor<float> mean = tmpBuf_.Get<float>();
        mean.SetValue(0, 0.0f);
        AscendC::LocalTensor<float> m2 = tmpBuf2_.Get<float>();
        m2.SetValue(0, 0.0f);
        uint32_t count = 0;
        uint32_t remaining = groupSize_;
        uint32_t offset = groupOffset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            for (uint32_t i = 0; i < curLen; i++) {
                float x_val = xLocal.GetValue(i);
                count++;
                float delta = x_val - mean.GetValue(0);
                mean.SetValue(0, mean.GetValue(0) + delta / count);
                float delta2 = x_val - mean.GetValue(0);
                m2.SetValue(0, m2.GetValue(0) + delta * delta2);
            }
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }

        float var_val = m2.GetValue(0) / groupSize_;
        float rstd_val = 1.0f / sqrtf(var_val + eps_);
        float mean_val = mean.GetValue(0);

        // Pass 2: normalize, scale, shift, write
        remaining = groupSize_;
        offset = groupOffset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t spatialSize = H_ * W_;
                uint32_t channelLocal = (offset + i - groupOffset) / spatialSize;
                float x_val = xLocal.GetValue(i);
                float norm_val = (x_val - mean_val) * rstd_val;
                float gamma_val = gammaLocal.GetValue(channelLocal);
                float beta_val = betaLocal.GetValue(channelLocal);
                yLocal.SetValue(i, norm_val * gamma_val + beta_val);
            }
            AscendC::DataCopy(yGm_[offset], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf2_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> gammaBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> betaBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
    uint32_t numGroups_ {0};
    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t channelsPerGroup_ {0};
    uint32_t groupSize_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_00415_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00415Custom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim,
            tiling_data.numGroups, tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.channelsPerGroup, tiling_data.groupSize, tiling_data.eps);
    op.Process();
}
