kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00393Custom {
public:
    __aicore__ inline KernelCa6k00393Custom() {}

    __aicore__ inline void Init(GM_ADDR log_prob, GM_ADDR target, GM_ADDR loss,
                                uint32_t batchSize, uint32_t numClasses,
                                uint32_t tileLen)
    {
        batchSize_ = batchSize;
        numClasses_ = numClasses;
        tileLen_ = tileLen;
        logProbGm_.SetGlobalBuffer((__gm__ float*)log_prob, batchSize * numClasses);
        targetGm_.SetGlobalBuffer((__gm__ int64_t*)target, batchSize);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0) return;

        if (AscendC::GetBlockIdx() != 0) return;

        float sum_loss = 0.0f;
        uint32_t processed = 0;
        while (processed < batchSize_) {
            uint32_t cur = batchSize_ - processed;
            if (cur > tileLen_) cur = tileLen_;

            for (uint32_t i = 0; i < cur; i++) {
                uint32_t idx = processed + i;
                int64_t classIdx = targetGm_[idx];
                float logp = logProbGm_[idx * numClasses_ + classIdx];
                sum_loss -= logp;
            }
            processed += cur;
        }

        sum_loss /= static_cast<float>(batchSize_);

        lossGm_[0] = sum_loss;
    }

private:
    AscendC::GlobalTensor<float> logProbGm_;
    AscendC::GlobalTensor<int64_t> targetGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t batchSize_ {0};
    uint32_t numClasses_ {1};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00393_custom(GM_ADDR log_prob, GM_ADDR target, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00393Custom op;
    op.Init(log_prob, target, loss,
            tiling_data.batchSize, tiling_data.numClasses,
            tiling_data.tileLen);
    op.Process();
}
