  const uint32_t H = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
  const uint32_t W = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04670Custom {
public:
  __aicore__ inline KernelCa6k04670Custom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                              GM_ADDR mean, GM_ADDR var,
                              uint32_t totalLength, uint32_t tileLength,
                              uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                              float eps) {
    totalLen_ = totalLength;
    tileLen_ = tileLength;
    N_ = N;
    C_ = C;
    H_ = H;
    W_ = W;
    spatialLen_ = H * W;
    eps_ = eps;

    xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    weightGm_.SetGlobalBuffer((__gm__ float*)weight, C_);
    biasGm_.SetGlobalBuffer((__gm__ float*)bias, C_);
    meanGm_.SetGlobalBuffer((__gm__ float*)mean, C_);
    varGm_.SetGlobalBuffer((__gm__ float*)var, C_);

    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    pipe_.InitBuffer(scaleQueue_, 1, tileLen_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    // Grid-stride loop over tiles (each tile is spatialLen_ elements flat)
    const uint32_t strideElems = blockNum * tileLen_;
    uint32_t base = blockIdx * tileLen_;
    if (base >= totalLen_) return;

    // Prefetch first tile
    uint32_t curLen = totalLen_ - base;
    if (curLen > tileLen_) curLen = tileLen_;
    CopyIn(base, curLen);

    while (true) {
      uint32_t nextBase = base + strideElems;
      uint32_t nextLen = 0;
      if (nextBase < totalLen_) {
        nextLen = totalLen_ - nextBase;
        if (nextLen > tileLen_) nextLen = tileLen_;
        CopyIn(nextBase, nextLen);
      }

      Compute(base, curLen);
      CopyOut(base, curLen);

      if (nextLen == 0) break;
      base = nextBase;
      curLen = nextLen;
    }
  }

private:
  __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
    AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
    inQueueX_.EnQue(xLocal);
  }

  __aicore__ inline void Compute(uint32_t gmOffset, uint32_t len) {
    AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
    AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
    AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
    AscendC::LocalTensor<float> scale = scaleQueue_.AllocTensor<float>();

    uint32_t startGlobal = gmOffset;
    uint32_t remaining = len;
    uint32_t offset = 0;
    while (remaining > 0) {
      uint32_t globalIdx = startGlobal + offset;
      uint32_t posInChannel = globalIdx % spatialLen_;
      uint32_t c = (globalIdx / spatialLen_) % C_;
      uint32_t avail = spatialLen_ - posInChannel;
      if (avail > remaining) avail = remaining;

      float meanVal = meanGm_.GetValue(c);
      float varVal = varGm_.GetValue(c);
      float weightVal = weightGm_.GetValue(c);
      float biasVal = biasGm_.GetValue(c);
      // Fix: use sqrt instead of sqrtf
      float invStd = 1.0f / sqrt(varVal + eps_);

      // Fix: use AscendC::Sub instead of Subs
      AscendC::Sub(tmp, xLocal[offset], meanVal, avail);
      AscendC::Muls(tmp, tmp, invStd, avail);
      AscendC::Muls(tmp, tmp, weightVal, avail);
      AscendC::Adds(yLocal[offset], tmp, biasVal, avail);

      offset += avail;
      remaining -= avail;
    }

    tmpQueue_.FreeTensor(tmp);
    scaleQueue_.FreeTensor(scale);
    outQueueY_.EnQue<float>(yLocal);
    inQueueX_.FreeTensor(xLocal);
  }

  __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
    AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
    AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
    outQueueY_.FreeTensor(yLocal);
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
  AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
  AscendC::TQue<AscendC::TPosition::VECCALC, 1> scaleQueue_;

  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> yGm_;
  AscendC::GlobalTensor<float> weightGm_;
  AscendC::GlobalTensor<float> biasGm_;
  AscendC::GlobalTensor<float> meanGm_;
  AscendC::GlobalTensor<float> varGm_;

  uint32_t totalLen_ {0};
  uint32_t tileLen_ {0};
  uint32_t N_ {0};
  uint32_t C_ {0};
  uint32_t H_ {0};
  uint32_t W_ {0};
  uint32_t spatialLen_ {0};
  float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void ca6k04670_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR mean, GM_ADDR var,
                                                        GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelCa6k04670Custom op;
  op.Init(x, y, weight, bias, mean, var,
          tiling_data.totalLength, tiling_data.tileLength,
          tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
          tiling_data.eps);
  op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k04670_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_features: int, affine: bool = True, eps: float = 1e-5, momentum: float = 0.1):
        super().__init__()
        self.num_features = num_features
        self.affine = affine
        self.eps = eps
        self.momentum = momentum
        if affine:
            self.weight = torch.nn.Parameter(torch.ones(num_features))
            self.bias = torch.nn.Parameter(torch.zeros(num_features))
        else:
            self.register_parameter('weight', None)
            self.register_parameter('bias', None)
        self.register_buffer('running_mean', torch.zeros(num_features))
        self.register_buffer('running_var', torch.ones(num_features))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if not self.affine:
            w = torch.ones(self.num_features, dtype=x.dtype, device=x.device)
            b = torch.zeros(self.num_features, dtype=x.dtype, device=x.device)
        else:
            w = self.weight
            b = self.bias
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, w, b, self.running_mean, self.running_var, self.eps)
