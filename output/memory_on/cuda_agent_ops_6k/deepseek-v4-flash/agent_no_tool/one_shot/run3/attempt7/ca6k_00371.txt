kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TMP_BUFFER_NUM = 2;

class KernelCa6k00371Custom {
public:
    __aicore__ inline KernelCa6k00371Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t dimC, uint32_t totalSpatial,
                                int32_t size, float alpha, float beta, float k) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        dimC_ = dimC;
        totalSpatial_ = totalSpatial;
        halfSize_ = size / 2;
        alpha_ = alpha;
        beta_ = beta;
        k_ = k;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // two temporary tensors: squares and denominator
        pipe_.InitBuffer(tmpQueue_, TMP_BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalSpatial_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t spIdx = blockIdx;

        while (spIdx < totalSpatial_) {
            uint32_t gmOffset = spIdx * dimC_;
            CopyIn(gmOffset);
            Compute(tileLen_);
            CopyOut(gmOffset);
            spIdx += blockNum;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], tileLen_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> sqLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> denomLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // compute squares
        AscendC::Mul(sqLocal, xLocal, xLocal, len);

        // compute sliding sum over channels, store in denomLocal
        for (uint32_t c = 0; c < dimC_; ++c) {
            int32_t start = static_cast<int32_t>(c) - halfSize_;
            int32_t end = static_cast<int32_t>(c) + halfSize_;
            if (start < 0) start = 0;
            if (end >= static_cast<int32_t>(dimC_)) end = static_cast<int32_t>(dimC_) - 1;
            float sum = 0.0f;
            for (int32_t j = start; j <= end; ++j) {
                sum += sqLocal[j];
            }
            denomLocal.SetValue(c, k_ + alpha_ * sum);
        }

        // denominator = (k + alpha * sum) ^ (-beta)
        AscendC::Pow(denomLocal, denomLocal, -beta_, len);
        // y = x * denominator
        AscendC::Mul(yLocal, xLocal, denomLocal, len);

        tmpQueue_.FreeTensor(sqLocal);
        tmpQueue_.FreeTensor(denomLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, tileLen_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t dimC_ {0};
    uint32_t totalSpatial_ {0};
    int32_t halfSize_ {0};
    float alpha_ {0.0001f};
    float beta_ {0.75f};
    float k_ {1.0f};
};

extern "C" __global__ __aicore__ void ca6k_00371_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00371Custom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.dimC, tiling_data.totalSpatial,
            tiling_data.size, tiling_data.alpha, tiling_data.beta, tiling_data.k);
    op.Process();
}
"""
