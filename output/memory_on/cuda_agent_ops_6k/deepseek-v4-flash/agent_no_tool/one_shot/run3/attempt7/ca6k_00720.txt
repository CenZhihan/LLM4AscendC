host_operator_src = '''
#include "ca6k00720_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00720CustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::StorageShape *w_shape = context->GetInputShape(1);

    uint32_t batch = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t seq = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t inFeatures = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    uint32_t outFeatures = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));

    uint32_t totalRows = batch * seq;

    // Tiling: process all output features at once (they fit UB) and tile rows moderately.
    uint32_t tileOut = outFeatures;
    uint32_t tileRows = 8;  // process 8 rows per iteration; adjust to fit UB
    bool fit = false;
    while (!fit) {
        uint32_t weightBytes = tileOut * inFeatures * sizeof(float);
        uint32_t xBytes = tileRows * inFeatures * sizeof(float);
        uint32_t yBytes = tileRows * tileOut * sizeof(float);
        uint32_t biasBytes = tileOut * sizeof(float);
        uint32_t totalBytes = weightBytes + xBytes + yBytes + biasBytes;
        if (totalBytes <= UB_BUDGET_BYTES) {
            fit = true;
        } else {
            if (tileRows > 1) {
                tileRows = tileRows / 2;
            } else if (tileOut > 1) {
                tileOut = AlignDown(tileOut / 2, ALIGN_ELEMS);
                if (tileOut == 0) tileOut = ALIGN_ELEMS;
            } else {
                break;
            }
        }
    }
    if (tileOut == 0) tileOut = ALIGN_ELEMS;
    if (tileRows == 0) tileRows = 1;

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    // Direct assignment (not setter methods)
    tiling.totalRows = totalRows;
    tiling.inFeatures = inFeatures;
    tiling.outFeatures = outFeatures;
    tiling.tileRows = tileRows;
    tiling.tileOut = tileOut;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, x_shape->GetDim(0));
    y_shape->SetDim(1, x_shape->GetDim(1));
    y_shape->SetDim(2, w_shape->GetDim(0));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge
'''

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00720Custom {
public:
    __aicore__ inline KernelCa6k00720Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalRows, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t tileRows, uint32_t tileOut) {
        totalRows_ = totalRows;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        tileRows_ = tileRows;
        tileOut_ = tileOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * inFeatures_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures_ * inFeatures_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * outFeatures_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileRows_ * inFeatures_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpWeight_, 1, tileOut_ * inFeatures_ * sizeof(float));
        pipe_.InitBuffer(tmpBias_, 1, tileOut_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalRows_ == 0 || inFeatures_ == 0 || outFeatures_ == 0) return;
        if (AscendC::GetBlockIdx() != 0) return;

        // Outer loop over output feature tiles
        for (uint32_t oStart = 0; oStart < outFeatures_; oStart += tileOut_) {
            uint32_t curTileOut = tileOut_;
            if (oStart + curTileOut > outFeatures_) {
                curTileOut = outFeatures_ - oStart;
            }

            // Load weight and bias tiles once per output tile
            AscendC::LocalTensor<float> wLocal = tmpWeight_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[oStart * inFeatures_], curTileOut * inFeatures_);
            AscendC::LocalTensor<float> bLocal = tmpBias_.AllocTensor<float>();
            AscendC::DataCopy(bLocal, bGm_[oStart], curTileOut);

            // Inner loop over rows
            for (uint32_t rStart = 0; rStart < totalRows_; rStart += tileRows_) {
                uint32_t curTileRows = tileRows_;
                if (rStart + curTileRows > totalRows_) {
                    curTileRows = totalRows_ - rStart;
                }

                // Load x tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rStart * inFeatures_], curTileRows * inFeatures_);
                // Allocate y tile
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // Compute y = x * W^T + bias  (W is [outFeatures, inFeatures])
                for (uint32_t r = 0; r < curTileRows; ++r) {
                    for (uint32_t oc = 0; oc < curTileOut; ++oc) {
                        float sum = bLocal[oc];
                        for (uint32_t ic = 0; ic < inFeatures_; ++ic) {
                            sum += xLocal[r * inFeatures_ + ic] * wLocal[oc * inFeatures_ + ic];
                        }
                        yLocal[r * curTileOut + oc] = sum;
                    }
                }

                // Write y tile
                AscendC::DataCopy(yGm_[rStart * outFeatures_ + oStart], yLocal, curTileRows * curTileOut);

                // Free local tensors
                inQueueX_.FreeTensor(xLocal);
                outQueueY_.FreeTensor(yLocal);
            }

            // Free weight and bias tensors
            tmpWeight_.FreeTensor(wLocal);
            tmpBias_.FreeTensor(bLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpWeight_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBias_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_;
    uint32_t inFeatures_;
    uint32_t outFeatures_;
    uint32_t tileRows_;
    uint32_t tileOut_;
};

extern "C" __global__ __aicore__ void ca6k_00720_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00720Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalRows, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.tileRows, tiling_data.tileOut);
    op.Process();
}
'''
