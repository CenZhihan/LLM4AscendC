#include "ca6k_00286_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00286CustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> dims = x_shape->GetOriginShape().GetDims();
    uint32_t N = static_cast<uint32_t>(dims[0]);
    uint32_t C = static_cast<uint32_t>(dims[1]);
    uint32_t H = static_cast<uint32_t>(dims[2]);
    uint32_t W = static_cast<uint32_t>(dims[3]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *scaleFactor = attrs->GetAttrPointer<int32_t>(0);
    uint32_t sf = static_cast<uint32_t>(*scaleFactor);

    uint32_t totalInputRows = N * C * H;
    uint32_t inputWidth = W;
    uint32_t outputWidth = W * sf;

    // Compute max rows per block from UB budget
    // Per row: input row (W floats) + output rows (2*W floats) = 3*W floats
    // Double buffering not used, so we use 1 buffer each: total = 3*W*4 bytes
    uint32_t bytesPerRow = 3 * inputWidth * sizeof(float);
    uint32_t maxRows = (bytesPerRow > 0) ? (UB_BUDGET_BYTES / bytesPerRow) : 1;
    if (maxRows < 1) maxRows = 1;
    if (maxRows > 32) maxRows = 32;

    // Determine tile rows and block dim
    uint32_t tileRows = maxRows;
    if (totalInputRows < tileRows) {
        tileRows = totalInputRows;
    }
    uint32_t blockDim = CeilDiv(totalInputRows, tileRows);
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;
    // Recompute tile rows to fit block dim
    tileRows = CeilDiv(totalInputRows, blockDim);

    context->SetBlockDim(blockDim);

    tiling.set_totalInputRows(totalInputRows);
    tiling.set_inputWidth(inputWidth);
    tiling.set_outputWidth(outputWidth);
    tiling.set_scaleFactor(sf);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> dims = x_shape->GetOriginShape().GetDims();
    // Assume input is 4D
    int64_t N = dims[0];
    int64_t C = dims[1];
    int64_t H = dims[2];
    int64_t W = dims[3];
    // Need scale_factor from attrs (not available in InferShape? Usually InferShape does not have attrs.
    // We'll assume scale_factor=2 as default. The runtime will have correct shape based on input.
    // For infer shape, we need to get attr. But in this context we don't have attrs. We'll use default 2.
    int64_t sf = 2;
    int64_t outH = H * sf;
    int64_t outW = W * sf;
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, outH);
    y_shape->SetDim(3, outW);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
}

namespace ops {
class Ca6k00286Custom : public OpDef {
public:
    explicit Ca6k00286Custom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("scale_factor").AttrType(OPTIONAL).Int(2);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(Ca6k00286Custom);
}

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00286Custom {
public:
    __aicore__ inline KernelCa6k00286Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputRows,
                                uint32_t inputWidth,
                                uint32_t outputWidth,
                                uint32_t scaleFactor,
                                uint32_t tileRows) {
        totalInputRows_ = totalInputRows;
        inputWidth_ = inputWidth;
        outputWidth_ = outputWidth;
        scaleFactor_ = scaleFactor;
        tileRows_ = tileRows;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputRows_ * inputWidth_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalInputRows_ * outputWidth_);

        // Buffers for one input row and one output row (two rows)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputWidth_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outputWidth_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalInputRows_ == 0 || inputWidth_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockDim = static_cast<uint32_t>(AscendC::GetBlockNum());
        if (blockDim == 0) blockDim = 1;

        uint32_t startRow = blockIdx * tileRows_;
        if (startRow >= totalInputRows_) return;

        uint32_t endRow = startRow + tileRows_;
        if (endRow > totalInputRows_) endRow = totalInputRows_;

        for (uint32_t r = startRow; r < endRow; ++r) {
            // Load input row
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[r * inputWidth_], inputWidth_);

            // Compute output row: duplicate each element and apply ReLU
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            for (uint32_t i = 0; i < inputWidth_; ++i) {
                float val = xLocal(i);
                float relu_val = (val > 0.0f) ? val : 0.0f;
                yLocal(2 * i) = relu_val;
                yLocal(2 * i + 1) = relu_val;
            }

            // Store output rows
            AscendC::DataCopy(yGm_[r * outputWidth_], yLocal, outputWidth_);

            outQueueY_.FreeTensor(yLocal);
);
);
