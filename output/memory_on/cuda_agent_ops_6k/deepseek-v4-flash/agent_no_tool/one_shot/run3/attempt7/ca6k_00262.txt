kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t ALIGN_ELEMS = 2048;
constexpr int32_t UB_BUDGET_BYTES = 192 * 1024;

class KernelCa6k00262Custom {
public:
    __aicore__ inline KernelCa6k00262Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR src, GM_ADDR y,
                                uint32_t totalLength, uint32_t batch,
                                uint32_t H, uint32_t W,
                                uint32_t diagLen, int32_t dim1, int32_t dim2, int32_t offset)
    {
        totalLen_ = totalLength;
        batch_ = batch;
        H_ = H;
        W_ = W;
        diagLen_ = diagLen;
        dim1_ = dim1;
        dim2_ = dim2;
        offset_ = offset;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        srcGm_.SetGlobalBuffer((__gm__ float*)src, batch_ * diagLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Tile size: each tile holds tileRows_ rows of the matrix (full width).
        // UB: 2 tiles * tileRows_ * W * sizeof(float) + src tile * diagLen_ * sizeof(float)
        uint32_t maxRows = (UB_BUDGET_BYTES - diagLen_ * sizeof(float)) / (2 * W_ * sizeof(float));
        maxRows = (maxRows / 8) * 8;
        if (maxRows < 1) maxRows = 1;
        tileRows_ = maxRows;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileRows_ * W_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * W_ * sizeof(float));
        pipe_.InitBuffer(srcLocal_, 1, diagLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || batch_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t b = blockIdx; b < batch_; b += blockNum) {
            uint32_t batchOffset = b * H_ * W_;
            uint32_t diagOffset = b * diagLen_;

            // Load src for this batch into local
            AscendC::LocalTensor<float> srcLocal = srcLocal_.AllocTensor<float>();
            AscendC::DataCopy(srcLocal, srcGm_[diagOffset], diagLen_);

            uint32_t rowsProcessed = 0;
            while (rowsProcessed < H_) {
                uint32_t curRows = (tileRows_ < (H_ - rowsProcessed)) ? tileRows_ : (H_ - rowsProcessed);
                uint32_t curLen = curRows * W_;
                // Align curLen to ALIGN_ELEMS for DataCopy
                uint32_t alignedLen = (curLen + ALIGN_ELEMS - 1) & ~(ALIGN_ELEMS - 1);
                if (alignedLen > curLen) {
                    // Need to handle leftover? For simplicity, use alignedLen but ensure we only write curLen later.
                    // DataCopy requires len to be multiple of some alignment? Use curLen as is.
                    curLen = curLen; // keep original
                }

                // Load x tile
                AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xTile, xGm_[batchOffset + rowsProcessed * W_], curLen);

                // Write to y first (copy)
                AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();
                AscendC::DataCopy(yTile, xTile, curLen);

                // Overwrite diagonal elements within the tile
                // dim1=1, dim2=2, offset=0 => diagonal at (i,i) for i in [0,diagLen)
                for (uint32_t r = 0; r < curRows; r++) {
                    uint32_t globalRow = rowsProcessed + r;
                    if (globalRow < diagLen_) {
                        int32_t col = static_cast<int32_t>(globalRow) + offset_;
                        if (col >= 0 && static_cast<uint32_t>(col) < W_) {
                            uint32_t idx = r * W_ + static_cast<uint32_t>(col);
                            yTile(idx) = srcLocal(globalRow);
                        }
                    }
                }

                // Copy from local y to global
                AscendC::DataCopy(yGm_[batchOffset + rowsProcessed * W_], yTile, curLen);

                outQueueY_.FreeTensor(yTile);
                inQueueX_.FreeTensor(xTile);

                rowsProcessed += curRows;
            }

            srcLocal_.FreeTensor(srcLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> srcLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> srcGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t batch_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t diagLen_ {0};
    int32_t dim1_ {0};
    int32_t dim2_ {0};
    int32_t offset_ {0};
    uint32_t tileRows_ {0};
};

extern "C" __global__ __aicore__ void ca6k00262_custom(GM_ADDR x, GM_ADDR src, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00262Custom op;
    op.Init(x, src, y,
            tiling_data.totalLength, tiling_data.batch,
            tiling_data.H, tiling_data.W,
            tiling_data.diagLen, tiling_data.dim1, tiling_data.dim2, tiling_data.offset);
    op.Process();
}

model_src = '''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

_FN = 'ca6k00262_custom'

class ModelNew(nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        # init_inputs: [input_shape, src_shape, dim1, dim2, offset]
        self.src = nn.Parameter(torch.randn(init_inputs[1]))
        self.dim1 = init_inputs[2]
        self.dim2 = init_inputs[3]
        self.offset = init_inputs[4]

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.src, self.dim1, self.dim2, self.offset)
