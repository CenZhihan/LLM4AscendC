kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

__aicore__ inline void conv2d_with_relu(
    AscendC::LocalTensor<float> &out,
    AscendC::GlobalTensor<float> &xGm,
    AscendC::GlobalTensor<float> &wGm,
    uint32_t batch, uint32_t inC, uint32_t inH, uint32_t inW,
    uint32_t outC, uint32_t outH, uint32_t outW,
    uint32_t kernelH, uint32_t kernelW,
    uint32_t strideH, uint32_t strideW,
    uint32_t padH, uint32_t padW,
    uint32_t dilationH, uint32_t dilationW)
{
    // Loop over output: for each (n, oc, oh, ow) compute sum
    for (uint32_t n = 0; n < batch; n++) {
        for (uint32_t oc = 0; oc < outC; oc++) {
            for (uint32_t oh = 0; oh < outH; oh++) {
                for (uint32_t ow = 0; ow < outW; ow++) {
                    float acc = 0.0f;
                    int32_t base_h = static_cast<int32_t>(oh) * static_cast<int32_t>(strideH) -
                                     static_cast<int32_t>(padH);
                    int32_t base_w = static_cast<int32_t>(ow) * static_cast<int32_t>(strideW) -
                                     static_cast<int32_t>(padW);
                    for (uint32_t ic = 0; ic < inC; ic++) {
                        for (uint32_t kh = 0; kh < kernelH; kh++) {
                            for (uint32_t kw = 0; kw < kernelW; kw++) {
                                int32_t ih = base_h + static_cast<int32_t>(kh) * static_cast<int32_t>(dilationH);
                                int32_t iw = base_w + static_cast<int32_t>(kw) * static_cast<int32_t>(dilationW);
                                if (ih >= 0 && ih < static_cast<int32_t>(inH) &&
                                    iw >= 0 && iw < static_cast<int32_t>(inW)) {
                                    float x_val = xGm[n * inC * inH * inW + ic * inH * inW + static_cast<uint32_t>(ih) * inW + static_cast<uint32_t>(iw)];
                                    float w_val = wGm[oc * inC * kernelH * kernelW + ic * kernelH * kernelW + kh * kernelW + kw];
                                    acc += x_val * w_val;
                                }
                            }
                        }
                    }
                    // ReLU
                    if (acc < 0.0f) acc = 0.0f;
                    out[n * outC * outH * outW + oc * outH * outW + oh * outW + ow] = acc;
                }
            }
        }
    }
}

__aicore__ inline void avgpool2d(
    AscendC::LocalTensor<float> &out,
    AscendC::LocalTensor<float> &in,
    uint32_t batch, uint32_t C, uint32_t inH, uint32_t inW,
    uint32_t outH, uint32_t outW,
    uint32_t kernelH, uint32_t kernelW,
    uint32_t strideH, uint32_t strideW,
    uint32_t padH, uint32_t padW)
{
    for (uint32_t n = 0; n < batch; n++) {
        for (uint32_t c = 0; c < C; c++) {
            for (uint32_t oh = 0; oh < outH; oh++) {
                for (uint32_t ow = 0; ow < outW; ow++) {
                    float sum = 0.0f;
                    float count = 0.0f;
                    int32_t base_h = static_cast<int32_t>(oh) * static_cast<int32_t>(strideH) -
                                     static_cast<int32_t>(padH);
                    int32_t base_w = static_cast<int32_t>(ow) * static_cast<int32_t>(strideW) -
                                     static_cast<int32_t>(padW);
                    for (uint32_t kh = 0; kh < kernelH; kh++) {
                        for (uint32_t kw = 0; kw < kernelW; kw++) {
                            int32_t ih = base_h + static_cast<int32_t>(kh);
                            int32_t iw = base_w + static_cast<int32_t>(kw);
                            if (ih >= 0 && ih < static_cast<int32_t>(inH) &&
                                iw >= 0 && iw < static_cast<int32_t>(inW)) {
                                sum += in[n * C * inH * inW + c * inH * inW + static_cast<uint32_t>(ih) * inW + static_cast<uint32_t>(iw)];
                                count += 1.0f;
                            }
                        }
                    }
                    out[n * C * outH * outW + c * outH * outW + oh * outW + ow] = (count > 0.0f) ? sum / count : 0.0f;
                }
            }
        }
    }
}

extern "C" __global__ __aicore__ void ca6k_05313_conv_relu(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    // Use a local buffer for output, then copy to global.
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    pipe.InitBuffer(inQueue, 1, td.totalLength * sizeof(float));
    pipe.InitBuffer(outQueue, 1, td.totalLength * sizeof(float));

    AscendC::LocalTensor<float> outLocal = outQueue.AllocTensor<float>();
    AscendC::GlobalTensor<float> xGm, wGm, yGm;
    xGm.SetGlobalBuffer((__gm__ float*)x, td.N * td.inC * td.inH * td.inW);
    wGm.SetGlobalBuffer((__gm__ float*)weight, td.outC * td.inC * td.kernelH * td.kernelW);
    yGm.SetGlobalBuffer((__gm__ float*)y, td.totalLength);

    conv2d_with_relu(outLocal, xGm, wGm,
                     td.N, td.inC, td.inH, td.inW,
                     td.outC, td.outH, td.outW,
                     td.kernelH, td.kernelW,
                     td.strideH, td.strideW,
                     td.padH, td.padW,
                     td.dilationH, td.dilationW);

    // Copy output from local to global
    AscendC::DataCopy(yGm, outLocal, td.totalLength);
    outQueue.FreeTensor(outLocal);
}

extern "C" __global__ __aicore__ void ca6k_05313_avgpool(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    pipe.InitBuffer(inQueue, 1, td.totalLength * sizeof(float));
    pipe.InitBuffer(outQueue, 1, td.totalLength * sizeof(float));

    AscendC::LocalTensor<float> inLocal = inQueue.AllocTensor<float>();
    AscendC::LocalTensor<float> outLocal = outQueue.AllocTensor<float>();
    AscendC::GlobalTensor<float> xGm, yGm;
    uint32_t inputSize = td.N * td.C * td.inH * td.inW;
    xGm.SetGlobalBuffer((__gm__ float*)x, inputSize);
