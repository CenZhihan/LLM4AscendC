kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04605Custom {
public:
    __aicore__ inline KernelCa6k04605Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t OC, uint32_t KH, uint32_t KW,
                                uint32_t OH, uint32_t OW,
                                int32_t strideH, int32_t strideW,
                                int32_t padTop, int32_t padBottom,
                                int32_t padLeft, int32_t padRight,
                                int32_t dilationH, int32_t dilationW,
                                int32_t groups)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        OC_ = OC; KH_ = KH; KW_ = KW;
        OH_ = OH; OW_ = OW;
        strideH_ = strideH; strideW_ = strideW;
        padTop_ = padTop; padBottom_ = padBottom;
        padLeft_ = padLeft; padRight_ = padRight;
        dilationH_ = dilationH; dilationW_ = dilationW;
        groups_ = groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * H * W);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, OC * (C / groups) * KH * KW);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, OC);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * OC * OH * OW);
    }

    __aicore__ inline void Process()
    {
        ConvParam convParam;
        convParam.SetInputShape(N_, C_, H_, W_);
        convParam.SetWeightShape(OC_, C_ / groups_, KH_, KW_);
        convParam.SetStride(strideH_, strideW_);
        convParam.SetPadding(padTop_, padBottom_, padLeft_, padRight_);
        convParam.SetDilation(dilationH_, dilationW_);
        convParam.SetGroups(groups_);
        convParam.SetBias(true);

        Convolution<float, float, float> conv;
        conv.SetConvParam(convParam);
        conv.Process(xGm_, weightGm_, biasGm_, yGm_);
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> weightGm_;
    GlobalTensor<float> biasGm_;
    GlobalTensor<float> yGm_;
    uint32_t N_, C_, H_, W_, OC_, KH_, KW_, OH_, OW_;
    int32_t strideH_, strideW_;
    int32_t padTop_, padBottom_, padLeft_, padRight_;
    int32_t dilationH_, dilationW_;
    int32_t groups_;
};

extern "C" __global__ __aicore__ void ca6k_04605_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04605Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.OC, tiling_data.KH, tiling_data.KW,
            tiling_data.OH, tiling_data.OW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padTop, tiling_data.padBottom,
            tiling_data.padLeft, tiling_data.padRight,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups);
    op.Process();
}
