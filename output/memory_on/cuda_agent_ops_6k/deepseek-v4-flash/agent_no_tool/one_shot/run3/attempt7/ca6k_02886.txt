static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k02886CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t dim0 = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(0));
    const uint32_t dim1 = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));
    const uint32_t dim2 = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(2));
    const uint32_t normSize = dim1 * dim2;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilon = attrs->GetAttrPointer<float>(0);

    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (normSize < tileLen) {
        uint32_t aligned = AlignDown(normSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : normSize;
    }

    uint32_t blockDim = dim0;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_normSize(normSize);
    tiling.set_normSizeFloat(static_cast<float>(normSize));
    tiling.set_epsilon(*epsilon);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k02886Custom {
public:
    __aicore__ inline KernelCa6k02886Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t normSize, float normSizeFloat, float epsilon) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        normSize_ = normSize;
        normSizeFloat_ = normSizeFloat;
        eps_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, normSize_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, normSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileLen_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sQueue_, 10, sizeof(float));  // scalars
    }

    __aicore__ inline void Process() {
        if (normSize_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t sampleBase = blockIdx * normSize_;

        // First pass: accumulate sum and sumSq over the sample
        AscendC::LocalTensor<float> totalSum = sQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> totalSumSq = sQueue_.AllocTensor<float>();
        AscendC::Dups(totalSum, 0.0f, 1);
        AscendC::Dups(totalSumSq, 0.0f, 1);

        uint32_t remaining = normSize_;
        uint32_t base = 0;
        while (remaining > 0) {
            uint32_t len = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[sampleBase + base], len);

            // sum
            AscendC::LocalTensor<float> sumScalar = sQueue_.AllocTensor<float>();
            AscendC::ReduceSum<AscendC::ReducePattern::SUM>(sumScalar, xLocal, len, false);
            AscendC::Add(totalSum, totalSum, sumScalar, 1);
            sQueue_.FreeTensor(sumScalar);

            // sum of squares
            AscendC::LocalTensor<float> sqLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(sqLocal, xLocal, xLocal, len);
            AscendC::LocalTensor<float> sqScalar = sQueue_.AllocTensor<float>();
            AscendC::ReduceSum<AscendC::ReducePattern::SUM>(sqScalar, sqLocal, len, false);
            AscendC::Add(totalSumSq, totalSumSq, sqScalar, 1);
            tmpQueue_.FreeTensor(sqLocal);
            sQueue_.FreeTensor(sqScalar);

            inQueueX_.FreeTensor(xLocal);
            base += len;
            remaining -= len;
        }

        // Compute mean, variance, and invStd
        AscendC::LocalTensor<float> mean = sQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> var = sQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> lenFloat = sQueue_.AllocTensor<float>();
        lenFloat.SetValue(0, normSizeFloat_);
        AscendC::Div(mean, totalSum, lenFloat, 1);

        AscendC::Div(var, totalSumSq, lenFloat, 1);
        AscendC::LocalTensor<float> meanSq = sQueue_.AllocTensor<float>();
        AscendC::Mul(meanSq, mean, mean, 1);
        AscendC::Sub(var, var, meanSq, 1);
        AscendC::Adds(var, var, eps_, 1);
        AscendC::Rsqrt(var, var, 1);  // var now holds invStd

        sQueue_.FreeTensor(meanSq);
        sQueue_.FreeTensor(lenFloat);
        sQueue_.FreeTensor(totalSum);
        sQueue_.FreeTensor(totalSumSq);

        // Second pass: normalize and apply weight and bias
        remaining = normSize_;
        base = 0;
        while (remaining > 0) {
            uint32_t len = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[sampleBase + base], len);

            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[base], len);
            AscendC::DataCopy(bLocal, bGm_[base], len);

            AscendC::LocalTensor<float> tmp1 = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmp2 = tmpQueue_.AllocTensor<float>();

            // Normalize: (x - mean) * invStd
            AscendC::Sub(tmp2, xLocal, mean, len);
            AscendC::Mul(tmp2, tmp2, var, len);  // var is invStd

            // Apply weight and bias
            AscendC::Mul(tmp1, tmp2, wLocal, len);
            AscendC::Add(tmp1, tmp1, bLocal, len);

            AscendC::DataCopy(yGm_[sampleBase + base], tmp1, len);

            tmpQueue_.FreeTensor(tmp1);
            tmpQueue_.FreeTensor(tmp2);
            inQueueX_.FreeTensor(xLocal);
            wQueue_.FreeTensor(wLocal);
            bQueue_.FreeTensor(bLocal);

            base += len;
            remaining -= len;
        }

        sQueue_.FreeTensor(mean);
        sQueue_.FreeTensor(var);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 10> sQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t normSize_ {0};
    float normSizeFloat_ {0.0f};
    float eps_ {1e-5};
};

extern "C" __global__ __aicore__ void ca6k02886_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02886Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.normSize, tiling_data.normSizeFloat, tiling_data.epsilon);
    op.Process();
}
"""

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k02886_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape):
        super().__init__()
        self.weight = torch.nn.Parameter(torch.ones(normalized_shape))
        self.bias = torch.nn.Parameter(torch.zeros(normalized_shape))
        self.epsilon = 1e-5

    def forward(self, x):
        w = self.weight.view(-1)
        b = self.bias.view(-1)
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, w, b, self.epsilon)
