    GET_TILING_DATA(td, tiling);
    uint32_t n = td.n;
    uint32_t d = td.d;
    float p = td.p;
    uint32_t totalPairs = td.totalPairs;
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t pairsPerBlock = totalPairs / blockNum;
    uint32_t rem = totalPairs % blockNum;
    uint32_t startPair = blockIdx * pairsPerBlock + (blockIdx < rem ? blockIdx : rem);
    uint32_t endPair = startPair + pairsPerBlock + (blockIdx < rem ? 1 : 0);

    __gm__ float *xGm = (__gm__ float*)x;
    __gm__ float *yGm = (__gm__ float*)y;

    for (uint32_t k = startPair; k < endPair; k++) {
        // compute i,j
        float i_f = (2.0f * n - 1.0f - sqrtf((2.0f * n - 1.0f) * (2.0f * n - 1.0f) - 8.0f * k)) / 2.0f;
