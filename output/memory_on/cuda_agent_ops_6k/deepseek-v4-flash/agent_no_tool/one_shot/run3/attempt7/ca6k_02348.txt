#include "kernel_operator.h"
#include "ca6k_02348_custom_tiling.h"

using namespace AscendC;

class KernelCa6k02348Custom {
public:
    __aicore__ inline KernelCa6k02348Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR y, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        n_ = td.n;
        c_in_ = td.c_in;
        h_in_ = td.h_in;
        w_in_ = td.w_in;
        c_out_ = td.c_out;
        k_h_ = td.k_h;
        k_w_ = td.k_w;
        stride_h_ = td.stride_h;
        stride_w_ = td.stride_w;
        pad_h_ = td.pad_h;
        pad_w_ = td.pad_w;
        out_pad_h_ = td.out_pad_h;
        out_pad_w_ = td.out_pad_w;
        h_out_ = td.h_out;
        w_out_ = td.w_out;
        totalLength_ = td.totalLength;
        tileLength_ = td.tileLength;
        hasBias_ = td.hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        if (hasBias_ != 0) {
            bGm_.SetGlobalBuffer((__gm__ float*)bias);
        }
    }

    __aicore__ inline void Process()
    {
        if (totalLength_ == 0) return;
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        uint32_t stride = blockNum * tileLength_;
        uint32_t start = blockIdx * tileLength_;
        if (start >= totalLength_) return;

        uint32_t remaining = totalLength_ - start;
        uint32_t end = start + ((remaining > tileLength_) ? tileLength_ : remaining);

        for (uint32_t outIdx = start; outIdx < end; outIdx++) {
            uint32_t tmp = outIdx;
            uint32_t ow = tmp % w_out_;
            tmp /= w_out_;
            uint32_t oh = tmp % h_out_;
            tmp /= h_out_;
            uint32_t co = tmp % c_out_;
            uint32_t n = tmp / c_out_;

            float acc = 0.0f;
            for (uint32_t ci = 0; ci < c_in_; ci++) {
                for (uint32_t kh = 0; kh < k_h_; kh++) {
                    for (uint32_t kw = 0; kw < k_w_; kw++) {
                        int32_t ih = static_cast<int32_t>(oh) + static_cast<int32_t>(pad_h_) - static_cast<int32_t>(kh);
                        int32_t iw = static_cast<int32_t>(ow) + static_cast<int32_t>(pad_w_) - static_cast<int32_t>(kw);
                        if (ih >= 0 && iw >= 0 && (ih % static_cast<int32_t>(stride_h_) == 0) && (iw % static_cast<int32_t>(stride_w_) == 0)) {
                            uint32_t ihi = static_cast<uint32_t>(ih) / stride_h_;
                            uint32_t iwi = static_cast<uint32_t>(iw) / stride_w_;
                            if (ihi < h_in_ && iwi < w_in_) {
                                uint32_t xIdx = n * c_in_ * h_in_ * w_in_ +
                                                ci * h_in_ * w_in_ +
                                                ihi * w_in_ + iwi;
                                uint32_t wIdx = co * c_in_ * k_h_ * k_w_ +
                                                ci * k_h_ * k_w_ +
                                                kh * k_w_ + kw;
                                acc += xGm_.GetValue(xIdx) * wGm_.GetValue(wIdx);
                            }
                        }
                    }
                }
            }
            if (hasBias_ != 0) {
                acc += bGm_.GetValue(co);
            }
            yGm_.SetValue(outIdx, acc);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_, c_in_, h_in_, w_in_, c_out_, k_h_, k_w_;
    uint32_t stride_h_, stride_w_, pad_h_, pad_w_, out_pad_h_, out_pad_w_;
    uint32_t h_out_, w_out_, totalLength_, tileLength_;
    uint32_t hasBias_;
};

extern "C" __global__ __aicore__ void ca6k_02348_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                         GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelCa6k02348Custom op;
    op.Init(x, weight, bias, y, tiling);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import math

_FN = "ca6k_02348_custom"

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias):
        super(ModelNew, self).__init__()
        self.weight = torch.nn.Parameter(torch.empty(out_channels, in_channels, kernel_size, kernel_size))
        fan_in = out_channels * kernel_size * kernel_size
        bound = 1.0 / math.sqrt(fan_in)
        torch.nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        if bias:
            self.bias = torch.nn.Parameter(torch.empty(out_channels))
            torch.nn.init.uniform_(self.bias, -bound, bound)
        else:
            self.bias = None
        self.stride = stride if isinstance(stride, tuple) else (stride, stride)
        self.padding = padding if isinstance(padding, tuple) else (padding, padding)
        self.output_padding = output_padding if isinstance(output_padding, tuple) else (output_padding, output_padding)

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        if self.bias is not None:
            return fn(x, self.weight, self.bias, self.stride, self.padding, self.output_padding)
        else:
            return fn(x, self.weight, torch.zeros(0, device=x.device, dtype=x.dtype), self.stride, self.padding, self.output_padding)
'''
