kernel_src = '''
#include "kernel_operator.h"
#include <math.h>

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k04582Custom {
public:
    __aicore__ inline KernelCa6k04582Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t N, uint32_t D)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, N * D);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        N_ = N;
        D_ = D;
        totalLength_ = totalLength;
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t stride = blockNum;
        for (uint32_t idx = blockIdx; idx < totalLength_; idx += stride) {
            // Compute pair (i, j) from linear index idx
            uint32_t i = N_ - 2 - (uint32_t)((sqrt(4.0f * N_ * (N_ - 1) - 8.0f * idx - 7.0f) - 1.0f) / 2.0f);
            // Adjust if needed (due to floating point rounding)
            uint32_t rowStart = i * N_ - i * (i + 1) / 2;
            while (i > 0 && rowStart > idx) {
                i--;
                rowStart = i * N_ - i * (i + 1) / 2;
            }
            while (rowStart + (N_ - i - 1) <= idx) {
                i++;
                rowStart = i * N_ - i * (i + 1) / 2;
            }
            uint32_t j = idx - rowStart + i + 1;

            // Load two rows into local memory
            AscendC::LocalTensor<float> xRow0 = pipe_.AllocTensor<float>(D_);
            AscendC::LocalTensor<float> xRow1 = pipe_.AllocTensor<float>(D_);

            AscendC::DataCopy(xRow0, xGm_[i * D_], D_);
            AscendC::DataCopy(xRow1, xGm_[j * D_], D_);

            float sumSq = 0.0f;
            for (uint32_t k = 0; k < D_; k++) {
                float diff = xRow0(k) - xRow1(k);
                sumSq += diff * diff;
            }

            pipe_.FreeTensor(xRow0);
            pipe_.FreeTensor(xRow1);

            float result = sqrtf(sumSq);
            yGm_.SetValue(idx, result);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLength_ {0};
    uint32_t N_ {0};
    uint32_t D_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04582_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04582Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.N, tiling_data.D);
    op.Process();
}
'''
