#include "ca6k_00229_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/types.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00229CustomTilingData tiling;

    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::Shape &x_shape = x_storage->GetOriginShape();
    uint32_t batch = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t channels = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t inputH = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t inputW = static_cast<uint32_t>(x_shape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t padTop = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(0)[0]);
    uint32_t padBottom = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(1)[0]);
    uint32_t padLeft = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(2)[0]);
    uint32_t padRight = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(3)[0]);

    uint32_t outputH = inputH + padTop + padBottom;
    uint32_t outputW = inputW + padLeft + padRight;
    uint32_t totalOutputLen = batch * channels * outputH * outputW;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = (totalOutputLen < maxTile) ? totalOutputLen : maxTile;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = 1;

    uint32_t numTiles = CeilDiv(totalOutputLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_padTop(padTop);
    tiling.set_padBottom(padBottom);
    tiling.set_padLeft(padLeft);
    tiling.set_padRight(padRight);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_totalOutputLen(totalOutputLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t padTop = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t padBottom = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padLeft = attrs->GetAttrPointer<int32_t>(2)[0];
    int32_t padRight = attrs->GetAttrPointer<int32_t>(3)[0];

    int64_t N = x_shape->GetDim(0);
    int64_t C = x_shape->GetDim(1);
    int64_t H = x_shape->GetDim(2);
    int64_t W = x_shape->GetDim(3);

    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, H + padTop + padBottom);
    y_shape->SetDim(3, W + padLeft + padRight);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"
#include "ca6k_00229_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00229Custom {
public:
    __aicore__ inline KernelCa6k00229Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                GM_ADDR workspace, GM_ADDR tiling) {
        xGm_ = (__gm__ float*)x;
        yGm_ = (__gm__ float*)y;
        tilingBuf_ = tiling;
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process() {
        Ca6k00229CustomTilingData tiling;
        GET_TILING_DATA(tiling, tilingBuf_);

        uint32_t totalLen = tiling.totalOutputLen;
        uint32_t tileLen = tiling.tileLen;
        if (totalLen == 0 || tileLen == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t strideElems = blockNum * tileLen;
        uint32_t base = blockIdx * tileLen;
        if (base >= totalLen) return;

        uint32_t batch = tiling.batch;
        uint32_t channels = tiling.channels;
        uint32_t inputH = tiling.inputH;
        uint32_t inputW = tiling.inputW;
        uint32_t padTop = tiling.padTop;
        uint32_t padBottom = tiling.padBottom;
        uint32_t padLeft = tiling.padLeft;
        uint32_t padRight = tiling.padRight;
        uint32_t outputH = tiling.outputH;
        uint32_t outputW = tiling.outputW;

        while (base < totalLen) {
            uint32_t curLen = (totalLen - base < tileLen) ? totalLen - base : tileLen;
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t linearIdx = base + i;
                uint32_t planeSize = outputH * outputW;
                uint32_t channelStride = channels * planeSize;
                uint32_t b = linearIdx / channelStride;
                uint32_t rem = linearIdx % channelStride;
                uint32_t c = rem / planeSize;
                uint32_t hw = rem % planeSize;
                uint32_t outRow = hw / outputW;
                uint32_t outCol = hw % outputW;

                float val = 0.0f;
                if (outRow >= padTop && outRow < padTop + inputH &&
                    outCol >= padLeft && outCol < padLeft + inputW) {
                    uint32_t inRow = outRow - padTop;
                    uint32_t inCol = outCol - padLeft;
                    uint32_t inPlaneSize = inputH * inputW;
                    uint32_t inChannelStride = channels * inPlaneSize;
                    uint32_t inIdx = b * inChannelStride + c * inPlaneSize + inRow * inputW + inCol;
                    val = xGm_[inIdx];
                }
                yLocal.SetValue(i, val);
            }

            AscendC::DataCopy(yGm_[base], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);

            base += strideElems;
        }
    }

private:
    __gm__ float* xGm_;
    __gm__ float* yGm_;
    __gm__ void* tilingBuf_;
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
};

extern "C" __global__ __aicore__ void ca6k_00229_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    KernelCa6k00229Custom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}
'''
