namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00378CustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::Shape &originShape = xShape->GetOriginShape();
    uint32_t ndim = static_cast<uint32_t>(originShape.GetDimNum());
    uint32_t totalLength = 1;
    for (uint32_t i = 0; i < ndim; ++i) totalLength *= static_cast<uint32_t>(originShape.GetDim(i));
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(3));
    uint32_t spatialPerBatch = H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    const float *momentumPtr = attrs->GetAttrPointer<float>(1);
    const bool *trainingPtr = attrs->GetAttrPointer<bool>(2);

    uint32_t maxTile = UB_BUDGET_BYTES / static_cast<uint32_t>(sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    uint32_t tileLen = maxTile;
    if (spatialPerBatch < tileLen) {
        uint32_t aligned = AlignDown(spatialPerBatch, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : spatialPerBatch;
    }
    if (tileLen == 0) tileLen = 1;

    uint32_t blockDim = C;
    if (blockDim > 32) blockDim = 32;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_numChannels(C);
    tiling.set_spatialSize(spatialPerBatch);
    tiling.set_batchSize(N);
    tiling.set_eps(*epsPtr);
    tiling.set_momentum(*momentumPtr);
    tiling.set_training(*trainingPtr);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00378Custom {
public:
    __aicore__ inline KernelCa6k00378Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR running_mean, GM_ADDR running_var,
                                GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t numChannels, uint32_t spatialPerBatch,
                                uint32_t batchSize, float eps, float momentumVal, bool training)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        numChannels_ = numChannels;
        spatialPerBatch_ = spatialPerBatch;
        batchSize_ = batchSize;
        eps_ = eps;
        momentum_ = momentumVal;
        training_ = training;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        meanGm_.SetGlobalBuffer((__gm__ float*)running_mean, numChannels_);
        varGm_.SetGlobalBuffer((__gm__ float*)running_var, numChannels_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numChannels_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, numChannels_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // Local memory for scalar stats (single float per buffer)
        pipe_.InitBuffer(meanBuf_, sizeof(float));
        pipe_.InitBuffer(varBuf_, sizeof(float));
        pipe_.InitBuffer(weightBuf_, sizeof(float));
        pipe_.InitBuffer(biasBuf_, sizeof(float));
        // Temporary buffer for accumulation in training mode
        pipe_.InitBuffer(accumBuf_, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t channelsPerBlock = (numChannels_ + blockNum - 1) / blockNum;
        uint32_t startChannel = blockIdx * channelsPerBlock;
        uint32_t endChannel = startChannel + channelsPerBlock;
        if (endChannel > numChannels_) endChannel = numChannels_;

        for (uint32_t c = startChannel; c < endChannel; ++c) {
            // Load per-channel scalars
            float meanVal, varVal, weightVal, biasVal;
            {
                auto scalarLoad = [&](AscendC::GlobalTensor<float> &src, float &dst) {
                    auto t = accumBuf_.AllocTensor<float>();
                    AscendC::DataCopy(t, src[c], 1);
                    dst = t(0);
                    accumBuf_.FreeTensor(t);
                };
                scalarLoad(meanGm_, meanVal);
                scalarLoad(varGm_, varVal);
                scalarLoad(weightGm_, weightVal);
                scalarLoad(biasGm_, biasVal);
            }

            if (training_) {
                // Compute batch mean and variance over (N, H, W) for channel c
                float sum = 0.0f;
                float sumSq = 0.0f;
                uint32_t totalPixels = batchSize_ * spatialPerBatch_;
                for (uint32_t b = 0; b < batchSize_; ++b) {
                    uint32_t base = (b * numChannels_ + c) * spatialPerBatch_;
                    for (uint32_t off = 0; off < spatialPerBatch_; off += tileLen_) {
                        uint32_t curLen = spatialPerBatch_ - off;
                        if (curLen > tileLen_) curLen = tileLen_;
                        auto xTile = inQueueX_.AllocTensor<float>();
                        AscendC::DataCopy(xTile, xGm_[base + off], curLen);
                        // vectorized sum
                        auto temp = accumBuf_.AllocTensor<float>();
                        AscendC::ReduceMax(temp, xxTile, curLen); // not needed
                        // Use local reduction
                        // Simpler: scalar loop (inefficient but correct)
                        for (uint32_t i = 0; i < curLen; ++i) {
                            sum += xTile(i);
                        }
                        ascendsq = accumBuf_.AllocTensor<float>();
                        AscendC::Mul(ascendsq, xTile, xTile, curLen);
                        for (uint32_t i = 0; i < curLen; ++i) {
                            sumSq += ascendsq(i);
                        }
                        accumBuf_.FreeTensor(ascendsq);
                        inQueueX_.FreeTensor(xTile);
                    }
                }
                float recip = 1.0f / static_cast<float>(totalPixels);
                float batchMean = sum * recip;
                float batchVar = sumSq * recip - batchMean * batchMean;
                // Update running stats
                float newMean = (1.0f - momentum_) * meanVal + momentum_ * batchMean;
                float newVar  = (1.0f - momentum_) * varVal  + momentum_ * batchVar;
                // Write back updated stats (in-place)
                {
                    auto t = accumBuf_.AllocTensor<float>();
                    t(0) = newMean;
                    AscendC::DataCopy(meanGm_[c], t, 1);
                    accumBuf_.FreeTensor(t);
                }
                {
                    auto t = accumBuf_.AllocTensor<float>();
                    t(0) = newVar;
                    AscendC::DataCopy(varGm_[c], t, 1);
                    accumBuf_.FreeTensor(t);
                }
                meanVal = batchMean;
                varVal = batchVar;
            }

            float invStd = 1.0f / sqrt(varVal + eps_);

            // Normalize each batch for this channel
            for (uint32_t b = 0; b < batchSize_; ++b) {
                uint32_t base = (b * numChannels_ + c) * spatialPerBatch_;
                uint32_t remaining = spatialPerBatch_;
                uint32_t offset = 0;
                while (remaining > 0) {
                    uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                    auto xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
                    auto yLocal = outQueueY_.AllocTensor<float>();
                    // y = (x - mean) * invStd * weight + bias
                    AscendC::Adds(yLocal, xLocal, -meanVal, curLen);
                    AscendC::Muls(yLocal, yLocal, invStd, curLen);
                    AscendC::Muls(yLocal, yLocal, weightVal, curLen);
                    AscendC::Adds(yLocal, yLocal, biasVal, curLen);
                    AscendC::DataCopy(yGm_[base + offset], yLocal, curLen);
                    inQueueX_.FreeTensor(xLocal);
                    outQueueY_.FreeTensor(yLocal);
                    offset += curLen;
                    remaining -= curLen;
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> meanBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> varBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> weightBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> biasBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> accumBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> meanGm_;
    AscendC::GlobalTensor<float> varGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numChannels_ {0};
    uint32_t spatialPerBatch_ {0};
    uint32_t batchSize_ {0};
    float eps_ {1e-05f};
    float momentum_ {0.1f};
    bool training_ {false};
};

extern "C" __global__ __aicore__ void ca6k00378_custom(GM_ADDR x, GM_ADDR running_mean, GM_ADDR running_var,
                                                        GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00378Custom op;
    op.Init(x, running_mean, running_var, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.numChannels, tiling_data.spatialSize,
            tiling_data.batchSize, tiling_data.eps, tiling_data.momentum, tiling_data.training);
    op.Process();
}
'''

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00378_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_features, eps=1e-05, momentum=0.1, affine=True, track_running_stats=True):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.eps = eps
        self.momentum = momentum
        self.affine = affine
        self.register_buffer('running_mean', torch.zeros(num_features))
        self.register_buffer('running_var', torch.ones(num_features))
        if affine:
            self.weight = torch.nn.Parameter(torch.ones(num_features))
            self.bias = torch.nn.Parameter(torch.zeros(num_features))
        else:
            self.register_buffer('weight', torch.ones(num_features))
            self.register_buffer('bias', torch.zeros(num_features))

    def forward(self, x):
        return getattr(custom_ops_lib, _FN)(
            x,
            self.running_mean,
            self.running_var,
            self.weight,
            self.bias,
            eps=self.eps,
            momentum=self.momentum,
            training=self.training
