#include "ca6k00449_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t FLOAT_SIZE = 4;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00449CustomTilingData tiling;

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const gert::Shape &originShape = inputShape->GetOriginShape();
    const uint32_t dimNum = originShape.GetDimNum();
    uint32_t N = (dimNum >= 1) ? static_cast<uint32_t>(originShape.GetDim(0)) : 1;
    uint32_t C = (dimNum >= 2) ? static_cast<uint32_t>(originShape.GetDim(1)) : 1;
    uint32_t H = (dimNum >= 3) ? static_cast<uint32_t>(originShape.GetDim(2)) : 1;
    uint32_t W = (dimNum >= 4) ? static_cast<uint32_t>(originShape.GetDim(3)) : 1;

    uint32_t totalLength = N * C * H * W;
    uint32_t spatialPerBatch = H * W;
    uint32_t totalSpatial = N * spatialPerBatch;

    uint32_t maxTile = UB_BUDGET_BYTES / (3 * C * FLOAT_SIZE);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > spatialPerBatch) maxTile = AlignDown(spatialPerBatch, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = spatialPerBatch;

    uint32_t tileLen = maxTile;
    if (totalSpatial < tileLen) {
        tileLen = AlignDown(totalSpatial, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalSpatial;
    }

    uint32_t numTiles = CeilDiv(totalSpatial, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    // direct assignment instead of setters
    tiling.totalLength = totalLength;
    tiling.tileLength = tileLen;
    tiling.blockDim = blockDim;
    tiling.C = C;
    tiling.H = H;
    tiling.W = W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    tiling.normSize = static_cast<uint32_t>(attrs->GetAttrPointer<int>(0)[0]);
    tiling.alpha = attrs->GetAttrPointer<float>(1)[0];
    tiling.beta = attrs->GetAttrPointer<float>(2)[0];
    tiling.k = attrs->GetAttrPointer<float>(3)[0];

    context->SetBlockDim(blockDim);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "ca6k00449_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00449Custom {
public:
    __aicore__ inline KernelCa6k00449Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t C, uint32_t H, uint32_t W,
                                uint32_t normSize, float alpha, float beta, float k) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        C_ = C;
        H_ = H;
        W_ = W;
        normSize_ = normSize;
        halfWin_ = normSize_ / 2;
        alpha_ = alpha;
        beta_ = beta;
        k_ = k;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(bufX_, BUFFER_NUM, C_ * sizeof(float));
        pipe_.InitBuffer(bufY_, BUFFER_NUM, C_ * sizeof(float));
        pipe_.InitBuffer(bufSq_, BUFFER_NUM, C_ * sizeof(float));
        pipe_.InitBuffer(bufPowBase_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(bufPowExp_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(bufPowRes_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t spatialPerBatch = H_ * W_;
        uint32_t totalSpatial = totalLen_ / C_;
        uint32_t tileCount = (totalSpatial + tileLen_ - 1) / tileLen_;

        for (uint32_t tileIdx = blockIdx; tileIdx < tileCount; tileIdx += blockNum) {
            uint32_t spatialStart = tileIdx * tileLen_;
            uint32_t curSpatialLen = totalSpatial - spatialStart;
            if (curSpatialLen > tileLen_) curSpatialLen = tileLen_;

            for (uint32_t s = 0; s < curSpatialLen; ++s) {
                uint32_t spatialIdx = spatialStart + s;
                uint32_t batch = spatialIdx / spatialPerBatch;
                uint32_t posInBatch = spatialIdx % spatialPerBatch;

                // Copy input channel slice
                AscendC::LocalTensor<float> xLocal = bufX_.AllocTensor<float>();
                for (uint32_t c = 0; c < C_; ++c) {
                    uint32_t gmIdx = batch * C_ * spatialPerBatch + c * spatialPerBatch + posInBatch;
                    xLocal(c) = xGm_(gmIdx);
                }

                // Compute squares
                AscendC::LocalTensor<float> sqLocal = bufSq_.AllocTensor<float>();
                for (uint32_t c = 0; c < C_; ++c) {
                    float val = xLocal(c);
                    sqLocal(c) = val * val;
                }

                AscendC::LocalTensor<float> yLocal = bufY_.AllocTensor<float>();
                for (uint32_t c = 0; c < C_; ++c) {
                    float sum = 0.0f;
                    int32_t start = static_cast<int32_t>(c) - static_cast<int32_t>(halfWin_);
                    if (start < 0) start = 0;
                    int32_t end = static_cast<int32_t>(c) + static_cast<int32_t>(halfWin_);
                    if (end >= static_cast<int32_t>(C_)) end = static_cast<int32_t>(C_) - 1;
                    for (int32_t j = start; j <= end; ++j) {
                        sum += sqLocal(j);
                    }
                    float base = k_ + alpha_ * sum;

                    // pow(base, beta) via AscendC vector
                    AscendC::LocalTensor<float> powBase = bufPowBase_.Get<float>();
                    AscendC::LocalTensor<float> powExp = bufPowExp_.Get<float>();
                    AscendC::LocalTensor<float> powRes = bufPowRes_.Get<float>();
                    powBase(0) = base;
                    powExp(0) = beta_;
                    AscendC::Pow(powRes, powBase, powExp, 1);
                    float denom = powRes(0);

                    yLocal(c) = xLocal(c) / denom;
                }

                // Write output
                for (uint32_t c = 0; c < C_; ++c) {
                    uint32_t gmIdx = batch * C_ * spatialPerBatch + c * spatialPerBatch + posInBatch;
                    yGm_(gmIdx) = yLocal(c);
                }

                // Free tensors
                bufX_.FreeTensor(xLocal);
                bufY_.FreeTensor(yLocal);
                bufSq_.FreeTensor(sqLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bufX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bufY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bufSq_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufPowBase_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufPowExp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufPowRes_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t C_{0};
    uint32_t H_{0};
    uint32_t W_{0};
    uint32_t normSize_{0};
    uint32_t halfWin_{0};
    float alpha_{0.0f};
    float beta_{0.0f};
    float k_{2.0f};
};

extern "C" __global__ __aicore__ void ca6k_00449_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00449Custom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.normSize, tiling_data.alpha, tiling_data.beta, tiling_data.k);
    op.Process();
}
