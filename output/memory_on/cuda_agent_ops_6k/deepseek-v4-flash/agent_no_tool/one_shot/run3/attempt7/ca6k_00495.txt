kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00495Custom {
public:
    __aicore__ inline KernelCa6k00495Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputLength, uint32_t n,
                                uint32_t c, uint32_t h, uint32_t w)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        totalOutputLen_ = totalOutputLength;
        n_ = n;
        c_ = c;
        h_ = h;
        w_ = w;
        hOut_ = h / 2;
        wOut_ = w / 2;
        strideHW_ = w * h;
        strideC_ = strideHW_ * c;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, 4 * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));

        // Allocate a local buffer for the 2x2 block (free from queue management)
        AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf;
        pipe_.InitBuffer(tmpBuf, 4 * sizeof(float));
        localIn_ = tmpBuf.Get<float>();
    }

    __aicore__ inline void Process()
    {
        if (totalOutputLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t start = blockIdx;
        uint32_t step = blockNum;

        for (uint32_t idx = start; idx < totalOutputLen_; idx += step) {
            uint32_t temp = idx;
            uint32_t outCol = temp % wOut_;
            temp /= wOut_;
            uint32_t outRow = temp % hOut_;
            temp /= hOut_;
            uint32_t ch = temp % c_;
            uint32_t batch = temp / c_;

            uint32_t baseOffset = batch * strideC_ + ch * strideHW_ + 2 * outRow * w_;
            uint32_t colBase = baseOffset + 2 * outCol;

            // Copy 2x2 block to local buffer
            AscendC::LocalTensor<float> localIn = localIn_;
            AscendC::DataCopy(localIn, xGm_[colBase], 4);

            // Compute max of the four values
            float v0 = localIn(0);
            float v1 = localIn(1);
            float v2 = localIn(2);
            float v3 = localIn(3);
            float maxVal = v0;
            if (v1 > maxVal) maxVal = v1;
            if (v2 > maxVal) maxVal = v2;
            if (v3 > maxVal) maxVal = v3;

            // Write output
            AscendC::LocalTensor<float> localOut = outQueueY_.AllocTensor<float>();
            localOut(0) = maxVal;
            outQueueY_.EnQue<float>(localOut);

            AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[idx], yLocal, 1);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::LocalTensor<float> localIn_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalOutputLen_ {0};
    uint32_t n_ {1};
    uint32_t c_ {1};
    uint32_t h_ {1};
    uint32_t w_ {1};
    uint32_t hOut_ {0};
    uint32_t wOut_ {0};
    uint32_t strideHW_ {0};
    uint32_t strideC_ {0};
};

extern "C" __global__ __aicore__ void ca6k00495_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00495Custom op;
    op.Init(x, y, tiling_data.totalOutputLength, tiling_data.n,
            tiling_data.c, tiling_data.h, tiling_data.w);
    op.Process();
}
