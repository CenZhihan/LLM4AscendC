namespace optiling {

static constexpr uint32_t MAX_BLOCK = 24u;
static constexpr uint32_t TILE_ELEMS = 4096u;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00246CustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t r = attrs->GetAttrPointer<int64_t>(0)[0];
    if (r <= 0) r = 1;

    const auto& x_shape = context->GetInputShape(0)->GetOriginShape();
    int64_t N = x_shape.GetDim(0);
    int64_t inC = x_shape.GetDim(1);
    int64_t H = x_shape.GetDim(2);
    int64_t W = x_shape.GetDim(3);
    int64_t C = inC / (r * r);
    int64_t Hr = H * r;
    int64_t Wr = W * r;
    int64_t totalOut = N * C * Hr * Wr;

    uint32_t tileOut = TILE_ELEMS;
    if (tileOut > totalOut) tileOut = totalOut;
    uint32_t blockDim = (totalOut + tileOut - 1) / tileOut;
    if (blockDim > MAX_BLOCK) blockDim = MAX_BLOCK;
    if (blockDim < 8) blockDim = 8;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_r(r);
    tiling.set_inC(inC);
    tiling.set_Hr(Hr);
    tiling.set_Wr(Wr);
    tiling.set_totalOut(totalOut);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelCa6k00246Custom {
public:
    __aicore__ inline KernelCa6k00246Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t r, uint32_t inC, uint32_t Hr, uint32_t Wr,
                                uint32_t totalOut, uint32_t tileOut)
    {
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        r_ = r;
        inC_ = inC;
        Hr_ = Hr;
        Wr_ = Wr;
        totalOut_ = totalOut;
        tileOut_ = tileOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalOut_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut_);
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileOut_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t start = blockIdx * tileOut_;
        uint32_t end = start + tileOut_;
        if (end > totalOut_) end = totalOut_;

        uint32_t strideHrWr = Hr_ * Wr_;
        uint32_t strideCHrWr = C_ * strideHrWr;
        uint32_t strideHW = H_ * W_;
        uint32_t strideinCHW = inC_ * strideHW;
        uint32_t strideCW = C_ * W_;
        // Not used: strideNCHrWr = N_ * strideCHrWr? Not needed.

        for (uint32_t idx = start; idx < end; ++idx) {
            // n = idx // (C_ * Hr_ * Wr_)
            uint32_t tmp1 = idx;
            uint32_t n = tmp1 / strideCHrWr;
            tmp1 -= n * strideCHrWr;
            uint32_t c = tmp1 / strideHrWr;
            tmp1 -= c * strideHrWr;
            uint32_t h_out = tmp1 / Wr_;
            uint32_t w_out = tmp1 % Wr_;

            uint32_t h = h_out / r_;
            uint32_t ri = h_out % r_;
            uint32_t w = w_out / r_;
            uint32_t rj = w_out % r_;

            // srcIdx = n * inC_ * H_ * W_ + (c * r_ * r_ + ri * r_ + rj) * H_ * W_ + h * W_ + w
            uint32_t srcIdx = n * strideinCHW
                            + (c * r_ * r_ + ri * r_ + rj) * strideHW
                            + h * W_
                            + w;

            float val = xGm_[srcIdx];
            yGm_[idx] = val;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t r_ {1};
    uint32_t inC_ {0};
    uint32_t Hr_ {0};
    uint32_t Wr_ {0};
    uint32_t totalOut_ {0};
    uint32_t tileOut_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00246_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00246Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.r, tiling_data.inC, tiling_data.Hr, tiling_data.Wr,
            tiling_data.totalOut, tiling_data.tileOut);
    op.Process();
}
