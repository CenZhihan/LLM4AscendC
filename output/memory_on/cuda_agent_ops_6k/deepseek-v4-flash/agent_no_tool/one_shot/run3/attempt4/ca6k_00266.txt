host_operator_src = '''
#include "ca6k_00266_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t TILE_HARD_CAP = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00266CustomTilingData tiling;

    auto indices_shape = context->GetInputShape(0)->GetOriginShape();
    auto offsets_shape = context->GetInputShape(1)->GetOriginShape();
    auto weight_shape = context->GetInputShape(2)->GetOriginShape();

    uint32_t totalIndices = static_cast<uint32_t>(indices_shape.GetShapeSize());
    uint32_t numOffsets = static_cast<uint32_t>(offsets_shape.GetShapeSize());
    uint32_t numEmbeddings = static_cast<uint32_t>(weight_shape.GetDim(0));
    uint32_t embeddingDim = static_cast<uint32_t>(weight_shape.GetDim(1));
    uint32_t totalBags = numOffsets - 1;

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > TILE_HARD_CAP) maxTile = AlignDown(TILE_HARD_CAP, ALIGN_ELEMS);
    uint32_t tileDim = (embeddingDim < maxTile) ? embeddingDim : maxTile;
    tileDim = AlignDown(tileDim, ALIGN_ELEMS);
    if (tileDim == 0) tileDim = ALIGN_ELEMS;

    uint32_t blockDim = totalBags;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalBags(totalBags);
    tiling.set_embeddingDim(embeddingDim);
    tiling.set_numEmbeddings(numEmbeddings);
    tiling.set_tileDim(tileDim);
    tiling.set_totalIndices(totalIndices);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling
