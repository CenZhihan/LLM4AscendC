    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = *attrs->GetAttrPointer<int32_t>(0);
    int32_t strideVal = *attrs->GetAttrPointer<int32_t>(1);
    int32_t paddingVal = *attrs->GetAttrPointer<int32_t>(2);

    // Estimate tileH from UB budget
    // For pooling: we need input rows = tileH + 2*padding, output rows = tileH
    // we also need local buffer for input tile and output tile (2*BUFFER_NUM each)
    // Rough estimate: UB bytes for one plane: inTileSize + outTileSize <= budget/2
    uint32_t outH = (H + 2 * static_cast<uint32_t>(paddingVal) - static_cast<uint32_t>(kernelSize)) / static_cast<uint32_t>(strideVal) + 1;
    uint32_t outW = (W + 2 * static_cast<uint32_t>(paddingVal) - static_cast<uint32_t>(kernelSize)) / static_cast<uint32_t>(strideVal) + 1;
    uint32_t maxBytesPerTile = static_cast<uint32_t>(UB_BUDGET_BYTES * 0.9f);
    uint32_t estimatedTileH = maxBytesPerTile / (4 * ( (kernelSize + 2*paddingVal) * W + outW )); // conservative

    if (estimatedTileH > 64) estimatedTileH = 64;
