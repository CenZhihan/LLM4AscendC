#include "ca6k04524_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04524CustomTilingData tiling;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = *attrs->GetAttrPointer<float>(0);

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t n = shape.GetDim(0);
    uint32_t c = shape.GetDim(1);
    uint32_t h = shape.GetDim(2);
    uint32_t w = shape.GetDim(3);
    uint32_t groupSize = h * w;

    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));  // 2 queues + reduction temps
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (groupSize < tileLen) {
        tileLen = AlignDown(groupSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = groupSize;
    }

    uint32_t totalGroups = n * c;
    uint32_t blockDim = totalGroups;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04524Custom {
public:
    __aicore__ inline KernelCa6k04524Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t h, uint32_t w,
                                uint32_t tileLength, float eps) {
        n_ = n;
        c_ = c;
        h_ = h;
        w_ = w;
        tileLen_ = tileLength;
        eps_ = eps;
        groupSize_ = h_ * w_;
        totalGroups_ = n_ * c_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * c_ * groupSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * c_ * groupSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalGroups_ == 0 || groupSize_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        if (blockIdx >= totalGroups_) return;

        // Each block handles one (n,c) group
        uint32_t startIdx = blockIdx * groupSize_;  // global offset for this group

        // Phase 1: compute mean and variance
        float sum = 0.0f;
        float sumSq = 0.0f;
        uint32_t remaining = groupSize_;
        uint32_t offset = 0;

        while (remaining > 0) {
            uint32_t curTile = (remaining > tileLen_) ? tileLen_ : remaining;
            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[startIdx + offset], curTile);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            // Compute sum
            float tileSum = AscendC::ReduceAdd(xLocal, curTile);
            sum += tileSum;
            // Compute sum of squares
            AscendC::LocalTensor<float> sqLocal = tmpBuf_.AllocTensor<float>();
            AscendC::Mul(sqLocal, xLocal, xLocal, curTile);
            float tileSumSq = AscendC::ReduceAdd(sqLocal, curTile);
            sumSq += tileSumSq;
            tmpBuf_.FreeTensor(sqLocal);
            inQueueX_.FreeTensor(xLocal);

            offset += curTile;
            remaining -= curTile;
        }

        float mean = sum / (float)groupSize_;
        float var = sumSq / (float)groupSize_ - mean * mean;
        float invStd = 1.0f / sqrtf(var + eps_);

        // Phase 2: normalize and output
        remaining = groupSize_;
        offset = 0;
        while (remaining > 0) {
            uint32_t curTile = (remaining > tileLen_) ? tileLen_ : remaining;
            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[startIdx + offset], curTile);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            // y = (x - mean) * invStd
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Sub(yLocal, xLocal, mean, curTile);
            AscendC::Mul(yLocal, yLocal, invStd, curTile);
            outQueueY_.EnQue(yLocal);

            // Copy out
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[startIdx + offset], yLocal, curTile);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);

            offset += curTile;
            remaining -= curTile;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_ {0};
    uint32_t c_ {0};
    uint32_t h_ {0};
    uint32_t w_ {0};
    uint32_t tileLen_ {0};
    uint32_t groupSize_ {0};
    uint32_t totalGroups_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_04524_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04524Custom op;
    op.Init(x, y, tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.tileLength, tiling_data.eps);
    op.Process();
}
"""
