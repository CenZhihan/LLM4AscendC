host_operator_src = """
#include "ca6k_02872_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k02872CustomTilingData tiling;
    uint32_t totalX = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    uint32_t divisorLen = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());

    // UB model: 2 tiles in (x and divisor chunk) + 2 tiles out (y) + 1 temp (reciprocal) + 1 divisor local
    // We keep divisor local out of tile loop. So per tile: x tile, y tile, temp (all float* tileLen)
    // plus divisor (divisorLen floats). Total floats = 3*tileLen + divisorLen.
    uint32_t maxTileFloat = (49152 - divisorLen) / 3;
    if (maxTileFloat < 256) maxTileFloat = 256;
    uint32_t tileLen = AlignDown(maxTileFloat, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = ALIGN_ELEMS;
    if (tileLen > HARD_CAP_ELEMS) tileLen = HARD_CAP_ELEMS;

    // Ensure tileLen is a multiple of divisorLen for simple broadcast
    if (divisorLen > 0) {
        uint32_t mult = CeilDiv(tileLen, divisorLen) * divisorLen;
        if (mult > HARD_CAP_ELEMS) mult = AlignDown(HARD_CAP_ELEMS, divisorLen);
        if (mult < divisorLen) mult = divisorLen;
        tileLen = mult;
    }

    if (totalX < tileLen) {
        uint32_t aligned = AlignDown(totalX, divisorLen);
        tileLen = (aligned > 0) ? aligned : totalX;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalX, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalX);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_divisorLength(divisorLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k02872Custom {
public:
    __aicore__ inline KernelCa6k02872Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR divisor, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t divisorLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        divisorLen_ = divisorLength;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        divisorGm_.SetGlobalBuffer((__gm__ float*)divisor, divisorLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));

        // Load entire divisor into local memory once (small, max 512)
        pipe_.InitBuffer(divisorLocal_, 1, divisorLen_ * sizeof(float));
        AscendC::LocalTensor<float> dLocal = divisorLocal_.AllocTensor<float>();
        AscendC::DataCopy(dLocal, divisorGm_, divisorLen_);
        divisorLocal_.EnQue(dLocal);
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Dequeue divisor once per compute (same for all tiles)
        AscendC::LocalTensor<float> dLocal = divisorLocal_.DeQue<float>();

        // Process in groups of divisorLen for simple broadcast
        uint32_t processed = 0;
        while (processed < len) {
            uint32_t chunk = len - processed;
            if (chunk > divisorLen_) chunk = divisorLen_;

            AscendC::LocalTensor<float> dChunk = divisorLocal_.AllocTensor<float>(chunk);
            AscendC::DataCopy(dChunk, dLocal[0], chunk);
            AscendC::Reciprocal(tLocal, dChunk, chunk);
            AscendC::Mul(yLocal[processed], xLocal[processed], tLocal, chunk);
            divisorLocal_.FreeTensor(dChunk);

            processed += chunk;
        }

        divisorLocal_.EnQue(dLocal);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> divisorLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> divisorGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t divisorLen_{0};
};

extern "C" __global__ __aicore__ void ca6k_02872_custom(GM_ADDR x, GM_ADDR divisor, GM_ADDR y,
                                                         GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02872Custom op;
    op.Init(x, divisor, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.divisorLength);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_02872_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, bias_shape):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, *inputs):
        x = inputs[0]
        x = self.linear(x)
        fn = getattr(custom_ops_lib, _FN)
