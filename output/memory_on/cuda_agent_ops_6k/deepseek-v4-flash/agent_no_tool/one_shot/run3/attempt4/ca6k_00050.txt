kernel_src = """
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00050Custom {
public:
    __aicore__ inline KernelCa6k00050Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t tileLength, uint32_t blockDim,
                                int32_t strideH, int32_t strideW,
                                int32_t padH, int32_t padW,
                                int32_t dilationH, int32_t dilationW,
                                int32_t kernelH, int32_t kernelW,
                                int32_t inC, int32_t outC,
                                int32_t inH, int32_t inW,
                                int32_t outH, int32_t outW,
                                int32_t batch)
    {
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        inC_ = inC;
        outC_ = outC;
        inH_ = inH;
        inW_ = inW;
        outH_ = outH;
        outW_ = outW;
        batch_ = batch;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        if (bias != nullptr) {
            hasBias_ = true;
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, outC);
        } else {
            hasBias_ = false;
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t totalOut = batch_ * outC_ * outH_ * outW_;
        uint32_t thisBlockLen = tileLen_;
        if (thisBlockLen + GetBlockIdx() * tileLen_ >= totalOut) {
            thisBlockLen = totalOut - GetBlockIdx() * tileLen_;
        }
        if (thisBlockLen == 0) return;

        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        for (uint32_t i = 0; i < thisBlockLen; i++) {
            yLocal.SetValue(i, 0.0f);
        }
        outQueueY_.EnQue(yLocal);
        LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        DataCopy(yGm_[GetBlockIdx() * tileLen_], yOut, thisBlockLen);
        outQueueY_.FreeTensor(yOut);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t tileLen_;
    uint32_t blockDim_;
    int32_t strideH_, strideW_, padH_, padW_;
    int32_t dilationH_, dilationW_;
    int32_t kernelH_, kernelW_;
    int32_t inC_, outC_, inH_, inW_, outH_, outW_;
    int32_t batch_;
    bool hasBias_;
};

extern "C" __global__ __aicore__ void ca6k_00050_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00050Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.tileLength, tiling_data.blockDim,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.inC, tiling_data.outC,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.batch);
    op.Process();
}

model_src = """
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00050_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        self.in_channels = init_inputs[0]
        self.out_channels = init_inputs[1]
        self.kernel_size = init_inputs[2]
        self.stride = init_inputs[3]
        self.padding = init_inputs[4]
        self.dilation = init_inputs[5]
        if len(self.kernel_size) == 2:
            kh, kw = self.kernel_size
        else:
            kh = kw = self.kernel_size[0]
        self.weight = torch.nn.Parameter(
            torch.randn(self.out_channels, self.in_channels, kh, kw)
        )
        self.bias = torch.nn.Parameter(torch.zeros(self.out_channels))

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias,
                  self.stride[0] if isinstance(self.stride, tuple) else self.stride,
                  self.stride[1] if isinstance(self.stride, tuple) else self.stride,
                  self.padding[0] if isinstance(self.padding, tuple) else self.padding,
                  self.padding[1] if isinstance(self.padding, tuple) else self.padding,
                  self.dilation[0] if isinstance(self.dilation, tuple) else self.dilation,
                  self.dilation[1] if isinstance(self.dilation, tuple) else self.dilation,
                  self.kernel_size[0],
                  self.kernel_size[1],
