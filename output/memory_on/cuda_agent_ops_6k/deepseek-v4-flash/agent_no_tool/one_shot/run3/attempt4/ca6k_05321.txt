static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k05321CustomTilingData tiling;

    const gert::StorageShape *xStorageShape = context->GetInputShape(0);
    const auto &originShape = xStorageShape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scalePtr = attrs->GetAttrPointer<float>(0);
    float scale = *scalePtr;

    uint32_t H_out = static_cast<uint32_t>(static_cast<double>(H) * scale);
    uint32_t W_out = static_cast<uint32_t>(static_cast<double>(W) * scale);

    uint32_t totalLength = N * C * H_out * W_out;

    // Determine tile size: output tile of size 64x64
    uint32_t tileH = 64;
    uint32_t tileW = 64;
    // Adjust for output dimensions
    if (tileH > H_out) tileH = H_out;
    if (tileW > W_out) tileW = W_out;
    // Corresponding input tile (with border)
    float invScale = 1.0f / scale;
    uint32_t inputTileH = static_cast<uint32_t>(std::ceil(tileH * invScale)) + 1;
    uint32_t inputTileW = static_cast<uint32_t>(std::ceil(tileW * invScale)) + 1;
    uint32_t inputTileBytes = inputTileH * inputTileW * sizeof(float);
    uint32_t outputTileBytes = tileH * tileW * sizeof(float);
    uint32_t totalTileBytes = inputTileBytes + 2 * outputTileBytes; // one input, one output, one temporary
    if (totalTileBytes > UB_BUDGET_BYTES) {
        tileH = 32;
        tileW = 32;
        inputTileH = static_cast<uint32_t>(std::ceil(tileH * invScale)) + 1;
        inputTileW = static_cast<uint32_t>(std::ceil(tileW * invScale)) + 1;
        inputTileBytes = inputTileH * inputTileW * sizeof(float);
        outputTileBytes = tileH * tileW * sizeof(float);
        totalTileBytes = inputTileBytes + 2 * outputTileBytes;
    }

    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_scale(scale);
    tiling.set_totalLength(totalLength);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k05321Custom {
public:
    __aicore__ inline KernelCa6k05321Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t H_out, uint32_t W_out, float scale,
                                uint32_t tileH, uint32_t tileW)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        H_out_ = H_out; W_out_ = W_out; scale_ = scale;
        tileH_ = tileH; tileW_ = tileW;
        invScale_ = 1.0f / scale;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * H * W);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C * H_out * W_out);

        // UB buffer: input tile with 1-pixel border for bilinear
        uint32_t inputTileH = static_cast<uint32_t>(AscendC::Ceil(tileH * invScale_)) + 1;
        uint32_t inputTileW = static_cast<uint32_t>(AscendC::Ceil(tileW * invScale_)) + 1;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputTileH * inputTileW * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileH_ * tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, tileH_ * tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (H_out_ == 0 || W_out_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Block-level grid stride over output slices (each slice is a channel of a batch)
        const uint32_t outSliceSize = H_out_ * W_out_;
        const uint32_t totalSlices = N_ * C_;
        const uint32_t slicesPerBlock = (totalSlices + blockNum - 1) / blockNum;
        const uint32_t startSlice = blockIdx * slicesPerBlock;
        const uint32_t endSlice = AscendC::Min(startSlice + slicesPerBlock, totalSlices);

        for (uint32_t sc = startSlice; sc < endSlice; ++sc) {
            uint32_t n = sc / C_;
            uint32_t c = sc % C_;
            // Process output tile rows
            for (uint32_t hOut0 = 0; hOut0 < H_out_; hOut0 += tileH_) {
                uint32_t tileHOut = AscendC::Min(tileH_, H_out_ - hOut0);
                for (uint32_t wOut0 = 0; wOut0 < W_out_; wOut0 += tileW_) {
                    uint32_t tileWOut = AscendC::Min(tileW_, W_out_ - wOut0);
                    ProcessTile(n, c, hOut0, wOut0, tileHOut, tileWOut);
                }
            }
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t n, uint32_t c,
                                       uint32_t hOut0, uint32_t wOut0,
                                       uint32_t tileHOut, uint32_t tileWOut)
    {
        // Compute input bounding box with border
        float hInStartF = hOut0 * invScale_;
        float wInStartF = wOut0 * invScale_;
        float hInEndF = (hOut0 + tileHOut - 1) * invScale_;
        float wInEndF = (wOut0 + tileWOut - 1) * invScale_;

        uint32_t hInStart = static_cast<uint32_t>(AscendC::Floor(hInStartF));
        uint32_t wInStart = static_cast<uint32_t>(AscendC::Floor(wInStartF));
        uint32_t hInEnd = static_cast<uint32_t>(AscendC::Ceil(hInEndF));
        uint32_t wInEnd = static_cast<uint32_t>(AscendC::Ceil(wInEndF));
        // Clamp to input bounds
        if (hInEnd >= H_) hInEnd = H_ - 1;
        if (wInEnd >= W_) wInEnd = W_ - 1;
        if (hInStart > hInEnd) hInStart = hInEnd;
        if (wInStart > wInEnd) wInStart = wInEnd;

        uint32_t inputH = hInEnd - hInStart + 1;
        uint32_t inputW = wInEnd - wInStart + 1;

        // Copy input tile to local
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        for (uint32_t ih = 0; ih < inputH; ++ih) {
            uint32_t globalRow = n * C_ * H_ * W_ + c * H_ * W_ + (hInStart + ih) * W_;
            uint32_t localOffset = ih * inputW;
            AscendC::DataCopy(xLocal[localOffset], xGm_[globalRow + wInStart], inputW);
        }
        inQueueX_.EnQue(xLocal);

        xLocal = inQueueX_.DeQue<float>();

        // Compute bilinear output tile
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpBuf_.Get<float>();

        for (uint32_t ho = 0; ho < tileHOut; ++ho) {
            float yf = hOut0 + ho;
            float xf = wOut0; // will loop inside
            float srcYf = yf * invScale_;
            float srcXfBase = xf * invScale_;
            uint32_t srcY0 = static_cast<uint32_t>(AscendC::Floor(srcYf));
            uint32_t srcY1 = AscendC::Min(srcY0 + 1, inputH - 1);
            float dy = srcYf - static_cast<float>(srcY0);
            for (uint32_t wo = 0; wo < tileWOut; ++wo) {
                float srcXf = srcXfBase + wo * invScale_;
                uint32_t srcX0 = static_cast<uint32_t>(AscendC::Floor(srcXf));
                uint32_t srcX1 = AscendC::Min(srcX0 + 1, inputW - 1);
                float dx = srcXf - static_cast<float>(srcX0);

                // bilinear interpolation
                float v00 = xLocal[srcY0 * inputW + srcX0];
                float v01 = xLocal[srcY0 * inputW + srcX1];
                float v10 = xLocal[srcY1 * inputW + srcX0];
                float v11 = xLocal[srcY1 * inputW + srcX1];
                float v0 = v00 + (v01 - v00) * dx;
                float v1 = v10 + (v11 - v10) * dx;
                float v = v0 + (v1 - v0) * dy;

                uint32_t outIdx = ho * tileW_ + wo;
                tmpLocal.SetValue(outIdx, v);
            }
        }

        // Copy from tmp to yLocal
        AscendC::DataCopy(yLocal, tmpLocal, tileHOut * tileWOut);
        outQueueY_.EnQue(yLocal);

        // Copy out to global
        yLocal = outQueueY_.DeQue<float>();
        for (uint32_t ho = 0; ho < tileHOut; ++ho) {
            uint32_t globalRow = n * C_ * H_out_ * W_out_ + c * H_out_ * W_out_ + (hOut0 + ho) * W_out_;
            uint32_t localOffset = ho * tileW_;
            AscendC::DataCopy(yGm_[globalRow + wOut0], yLocal[localOffset], tileWOut);
        }
        outQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, H_, W_, H_out_, W_out_;
    float scale_;
    float invScale_;
    uint32_t tileH_, tileW_;
};

extern "C" __global__ __aicore__ void ca6k_05321_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k05321Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.H_out, tiling_data.W_out, tiling_data.scale,
            tiling_data.tileH, tiling_data.tileW);
    op.Process();
}
