static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
  return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static ge::graphStatus TilingFuncGlu(gert::TilingContext *context) {
  Ca6k05325GluTilingData tiling;
  const gert::StorageShape *xShape = context->GetInputShape(0);
  const auto &originShape = xShape->GetOriginShape();
  uint32_t batch = static_cast<uint32_t>(originShape.GetDim(0));
  uint32_t channels = static_cast<uint32_t>(originShape.GetDim(1));
  uint32_t height = static_cast<uint32_t>(originShape.GetDim(2));
  uint32_t width = static_cast<uint32_t>(originShape.GetDim(3));
  uint32_t halfChannels = channels / 2;
  uint32_t spatialSize = batch * height * width;
  uint32_t totalLength = spatialSize * halfChannels;

  uint32_t perPosBytes = 2 * halfChannels * sizeof(float) + halfChannels * sizeof(float) + halfChannels * sizeof(float); // in+out+tmp

  if (maxTile == 0) maxTile = 1;

kernel_src = """
#include "kernel_operator.h"
#include "ca6k05325_glu_tiling.h"
#include "ca6k05325_linear_tiling.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k05325Glu {
public:
  __aicore__ inline KernelCa6k05325Glu() {}
  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                              uint32_t totalLength, uint32_t tileLength,
                              uint32_t halfChannels, uint32_t spatialSize) {
    totalLen_ = totalLength;
    tileLen_ = tileLength;
    halfChannels_ = halfChannels;
    spatialSize_ = spatialSize;
    inputStride_ = halfChannels_ * 2;
    outputStride_ = halfChannels_;

    xGm_.SetGlobalBuffer((__gm__ float*)x, spatialSize_ * inputStride_);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * inputStride_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * outputStride_ * sizeof(float));
    pipe_.InitBuffer(tmpBuf_, halfChannels_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;

    uint32_t blockNum = GetBlockNum();
    uint32_t blockIdx = GetBlockIdx();

    uint32_t stridePos = blockNum * tileLen_;
    uint32_t posBase = blockIdx * tileLen_;

    while (posBase < spatialSize_) {
      uint32_t curTileLen = spatialSize_ - posBase;
      if (curTileLen > tileLen_) curTileLen = tileLen_;

      // Copy input tile
      LocalTensor<float> inLocal = inQueueX_.AllocTensor<float>();
      DataCopy(inLocal, xGm_[posBase * inputStride_], curTileLen * inputStride_);
      inQueueX_.EnQue(inLocal);

      LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();

      for (uint32_t i = 0; i < curTileLen; i++) {
        uint32_t inOffset = i * inputStride_;
        LocalTensor<float> left = inLocal[inOffset];
        LocalTensor<float> right = inLocal[inOffset + halfChannels_];

        // temporaries
        LocalTensor<float> sig = tmpBuf_.Get<float>(0);

        // sigmoid(left)
        Sigmoid(sig, left, halfChannels_);
        // glu = sig * right
        Mul(sig, sig, right, halfChannels_);
        // copy to output
        DataCopy(outLocal[i * outputStride_], sig, halfChannels_);
      }

      outQueueY_.EnQue(outLocal);
      inQueueX_.FreeTensor(inLocal);

      // write back
      LocalTensor<float> outDeq = outQueueY_.DeQue<float>();
      DataCopy(yGm_[posBase * outputStride_], outDeq, curTileLen * outputStride_);
      outQueueY_.FreeTensor(outDeq);

      posBase += stridePos;
    }
  }

private:
  TPipe pipe_;
  TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
  TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
  TBuf<TPosition::VECCALC> tmpBuf_;

  GlobalTensor<float> xGm_;
  GlobalTensor<float> yGm_;

  uint32_t totalLen_ {0};
  uint32_t tileLen_ {0};
  uint32_t halfChannels_ {0};
  uint32_t spatialSize_ {0};
  uint32_t inputStride_ {0};
  uint32_t outputStride_ {0};
};

class KernelCa6k05325Linear {
public:
  __aicore__ inline KernelCa6k05325Linear() {}
  __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                              uint32_t totalLength, uint32_t tileLength,
                              uint32_t halfChannels, uint32_t spatialSize) {
    totalLen_ = totalLength;
    tileLen_ = tileLength;
    halfChannels_ = halfChannels;
    spatialSize_ = spatialSize;
    stride_ = halfChannels_;

    xGm_.SetGlobalBuffer((__gm__ float*)x, spatialSize_ * stride_);
    weightGm_.SetGlobalBuffer((__gm__ float*)weight, halfChannels_ * halfChannels_);
    biasGm_.SetGlobalBuffer((__gm__ float*)bias, halfChannels_);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * stride_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * stride_ * sizeof(float));
    pipe_.InitBuffer(tmpBuf1_, halfChannels_ * sizeof(float));
    pipe_.InitBuffer(tmpBuf2_, halfChannels_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;

    uint32_t blockNum = GetBlockNum();
    uint32_t blockIdx = GetBlockIdx();

    uint32_t stridePos = blockNum * tileLen_;
    uint32_t posBase = blockIdx * tileLen_;

    while (posBase < spatialSize_) {
      uint32_t curTileLen = spatialSize_ - posBase;
      if (curTileLen > tileLen_) curTileLen = tileLen_;

      // Copy input tile (all positions)
      LocalTensor<float> inLocal = inQueueX_.AllocTensor<float>();
      DataCopy(inLocal, xGm_[posBase * stride_], curTileLen * stride_);
      inQueueX_.EnQue(inLocal);

      LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();

      for (uint32_t i = 0; i < curTileLen; i++) {
        LocalTensor<float> vec = inLocal[i * stride_];
        // For each output channel j
        for (uint32_t j = 0; j < halfChannels_; j++) {
          // Load weight row j
          LocalTensor<float> wRow = tmpBuf1_.Get<float>(0);
          DataCopy(wRow, weightGm_[j * halfChannels_], halfChannels_);
          // Multiply vec with wRow element-wise
          LocalTensor<float> prod = tmpBuf2_.Get<float>(0);
          Mul(prod, vec, wRow, halfChannels_);
          // Reduce sum
          float dot = ReduceSum(prod, halfChannels_);
          // Add bias
          float res = dot + biasGm_.GetValue(j);
          // Set output element
          outLocal[i * stride_ + j] = res;
        }
      }

      outQueueY_.EnQue(outLocal);
      inQueueX_.FreeTensor(inLocal);

      LocalTensor<float> outDeq = outQueueY_.DeQue<float>();
      DataCopy(yGm_[posBase * stride_], outDeq, curTileLen * stride_);
      outQueueY_.FreeTensor(outDeq);

      posBase += stridePos;
    }
  }

private:
  TPipe pipe_;
  TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
  TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
  TBuf<TPosition::VECCALC> tmpBuf1_;
  TBuf<TPosition::VECCALC> tmpBuf2_;

  GlobalTensor<float> xGm_;
  GlobalTensor<float> weightGm_;
  GlobalTensor<float> biasGm_;
  GlobalTensor<float> yGm_;

  uint32_t totalLen_ {0};
  uint32_t tileLen_ {0};
  uint32_t halfChannels_ {0};
  uint32_t spatialSize_ {0};
  uint32_t stride_ {0};
};

extern "C" __global__ __aicore__ void ca6k_05325_glu(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelCa6k05325Glu op;
  op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
          tiling_data.halfChannels, tiling_data.spatialSize);
  op.Process();
}

extern "C" __global__ __aicore__ void ca6k_05325_linear(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                         GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelCa6k05325Linear op;
  op.Init(x, weight, bias, y, tiling_data.totalLength, tiling_data.tileLength,
          tiling_data.halfChannels, tiling_data.spatialSize);
  op.Process();
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN_GLU = 'ca6k_05325_glu'
_FN_LINEAR = 'ca6k_05325_linear'

class ModelNew(torch.nn.Module):
    def __init__(self, in_features: int):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features // 2, in_features // 2)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        glu_fn = getattr(custom_ops_lib, _FN_GLU)
        linear_fn = getattr(custom_ops_lib, _FN_LINEAR)
        glu_out = glu_fn(x)
        out = linear_fn(glu_out, self.linear.weight, self.linear.bias)
        return out
