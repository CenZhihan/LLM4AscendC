static constexpr uint32_t ALIGN_ELEMS = 32;

static uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k02874CustomTilingData tiling;
    const gert::StorageShape *x_shape = context->GetInputShape(1);  // x is second input
    uint32_t batch_size = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t in_features = static_cast<uint32_t>(x_shape->GetDim(1));

    uint32_t totalRows = batch_size;
    uint32_t totalCols = in_features;

    // UB budget for one row block: each tile computation needs:
    //   x_tile: tileRows * totalCols floats
    //   w_tile: totalCols * tileCols floats
    //   y_tile: tileRows * tileCols floats
    //   bias: totalCols floats (loaded once, reused)
    //   temp: totalCols floats (for each column)
    // Let tileRows = 2, compute max tileCols.
    uint32_t tileRows = 2;
    uint32_t maxTileCols = (UB_BUDGET_BYTES / sizeof(float) - tileRows * totalCols - totalCols) / (totalCols + tileRows);
    if (maxTileCols < 1) {
        tileRows = 1;
        maxTileCols = (UB_BUDGET_BYTES / sizeof(float) - tileRows * totalCols - totalCols) / (totalCols + tileRows);
    }
    if (maxTileCols < 1) maxTileCols = 1;
    uint32_t tileCols = maxTileCols;
    if (tileCols > totalCols) tileCols = totalCols;

    uint32_t blockDim = totalRows < MAX_BLOCKS ? totalRows : MAX_BLOCKS;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_totalCols(totalCols);
    tiling.set_tileRows(tileRows);
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k02874Custom {
public:
    __aicore__ inline KernelCa6k02874Custom() {}

    __aicore__ inline void Init(GM_ADDR bias, GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalRows, uint32_t totalCols,
                                uint32_t tileRows, uint32_t tileCols)
    {
        totalRows_ = totalRows;
        totalCols_ = totalCols;
        tileRows_ = tileRows;
        tileCols_ = tileCols;

        biasGm_.SetGlobalBuffer((__gm__ float*)bias, totalCols_);
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * totalCols_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, totalCols_ * totalCols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * totalCols_);

        // Allocate local buffers
        pipe_.InitBuffer(biasBuf_, 1, totalCols_ * sizeof(float));
        pipe_.InitBuffer(xBuf_, 1, tileRows_ * totalCols_ * sizeof(float));
        pipe_.InitBuffer(wBuf_, 1, totalCols_ * tileCols_ * sizeof(float));
        pipe_.InitBuffer(yBuf_, 1, tileRows_ * tileCols_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, totalCols_ * sizeof(float));  // for dot product
        pipe_.InitBuffer(scalarBuf_, 1, sizeof(float));            // scalar for reduce

        // Load bias once
        AscendC::LocalTensor<float> biasLocal = biasBuf_.Get<float>();
        AscendC::DataCopy(biasLocal, biasGm_, totalCols_);
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || totalCols_ == 0 || tileRows_ == 0 || tileCols_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t rowStride = blockNum * tileRows_;
        uint32_t rowStart = blockIdx * tileRows_;
        uint32_t rowEnd = totalRows_ < (rowStart + tileRows_) ? totalRows_ : rowStart + tileRows_;

        while (rowStart < totalRows_) {
            uint32_t curRows = (rowEnd - rowStart);  // actual rows (may be less than tileRows)

            // Load x tile
            AscendC::LocalTensor<float> xLocal = xBuf_.Get<float>();
            AscendC::DataCopy(xLocal, xGm_[rowStart * totalCols_], curRows * totalCols_);

            // Process column tiles
            uint32_t colStart = 0;
            while (colStart < totalCols_) {
                uint32_t curCols = totalCols_ - colStart;
                if (curCols > tileCols_) curCols = tileCols_;

                // Load weight tile (totalCols_ x curCols) – row-major, rows = totalCols_, cols = curCols
                AscendC::LocalTensor<float> wLocal = wBuf_.Get<float>();
                for (uint32_t k = 0; k < totalCols_; k++) {
                    AscendC::DataCopy(wLocal[k * curCols], weightGm_[k * totalCols_ + colStart], curCols);
                }

                // For each row in the current row block
                AscendC::LocalTensor<float> yLocal = yBuf_.Get<float>();
                for (uint32_t r = 0; r < curRows; r++) {
                    // For each output column in the current col block
                    for (uint32_t c = 0; c < curCols; c++) {
                        // Compute dot product of x row (totalCols_) and weight column (totalCols_)
                        AscendC::LocalTensor<float> xRow = xLocal[r * totalCols_];
                        AscendC::LocalTensor<float> wCol = wLocal[c];  // stride = curCols? careful

                        // Actually wLocal is packed as totalCols_ rows of curCols cols.
                        // We need the column vector: we can extract by strided access:
                        // Build temp vector from wLocal: for each k, wLocal[k * curCols + c]
                        // Since we don't have strided load, we can loop manually:
                        AscendC::LocalTensor<float> tmp = tmpBuf_.Get<float>();
                        for (uint32_t k = 0; k < totalCols_; k++) {
                            tmp[k] = wLocal[k * curCols + c];
                        }
                        AscendC::Mul(tmp, xRow, tmp, totalCols_);
                        AscendC::ReduceAdd(scalarBuf_.Get<float>(), tmp, totalCols_, false);
                        float dot = scalarBuf_.Get<float>()[0];
                        // Add bias and store
                        float biasVal = biasBuf_.Get<float>()[colStart + c];
                        yLocal[r * curCols + c] = biasVal + dot;
                        tmpBuf_.FreeTensor(tmp);
                    }
                }

                // Copy y tile to global
                for (uint32_t r = 0; r < curRows; r++) {
                    AscendC::DataCopy(yGm_[(rowStart + r) * totalCols_ + colStart],
                                      yLocal[r * curCols], curCols);
                }

                colStart += curCols;
            }

            rowStart += rowStride;
            if (rowStart + tileRows_ > totalRows_)
                rowEnd = totalRows_;
            else
                rowEnd = rowStart + tileRows_;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECIN> biasBuf_;
    AscendC::TBuf<AscendC::TPosition::VECIN> xBuf_;
    AscendC::TBuf<AscendC::TPosition::VECIN> wBuf_;
    AscendC::TBuf<AscendC::TPosition::VECOUT> yBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> scalarBuf_;

    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t totalCols_ {0};
    uint32_t tileRows_ {0};
    uint32_t tileCols_ {0};
};

extern "C" __global__ __aicore__ void ca6k_02874_custom(GM_ADDR bias, GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02874Custom op;
    op.Init(bias, x, weight, y,
            tiling_data.totalRows, tiling_data.totalCols,
            tiling_data.tileRows, tiling_data.tileCols);
    op.Process();
}
