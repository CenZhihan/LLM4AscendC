kernel_src = '''
#include "kernel_operator.h"

class KernelCa6k04616Custom {
public:
    __aicore__ inline KernelCa6k04616Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputLength,
                                uint32_t N, uint32_t C, uint32_t D, uint32_t H, uint32_t W,
                                uint32_t Dout, uint32_t Hout, uint32_t Wout,
                                int32_t padLeft, int32_t padRight,
                                int32_t padTop, int32_t padBottom,
                                int32_t padFront, int32_t padBack,
                                uint32_t tileLength)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        totalOutputLength_ = totalOutputLength;
        N_ = N; C_ = C; D_ = D; H_ = H; W_ = W;
        Dout_ = Dout; Hout_ = Hout; Wout_ = Wout;
        padLeft_ = padLeft; padRight_ = padRight;
        padTop_ = padTop; padBottom_ = padBottom;
        padFront_ = padFront; padBack_ = padBack;
        tileLength_ = tileLength;

        inputStrideC_ = D * H * W;
        inputStrideD_ = H * W;
        inputStrideH_ = W;
        inputStrideW_ = 1;

        outputStrideC_ = Dout * Hout * Wout;
        outputStrideD_ = Hout * Wout;
        outputStrideH_ = Wout;
        outputStrideW_ = 1;
    }

    __aicore__ inline void Process()
    {
        if (totalOutputLength_ == 0 || tileLength_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t start = blockIdx * tileLength_;
        uint32_t end = start + tileLength_;
        if (start >= totalOutputLength_) return;
        if (end > totalOutputLength_) end = totalOutputLength_;

        for (uint32_t offset = start; offset < end; offset++) {
            uint32_t w_out = offset % Wout_;
            uint32_t h_out = (offset / Wout_) % Hout_;
            uint32_t d_out = (offset / (Hout_ * Wout_)) % Dout_;
            uint32_t c = (offset / (Dout_ * Hout_ * Wout_)) % C_;
            uint32_t n = offset / (C_ * Dout_ * Hout_ * Wout_);

            int32_t d_src = (static_cast<int32_t>(d_out) - padFront_) % static_cast<int32_t>(D_);
            if (d_src < 0) d_src += D_;
            int32_t h_src = (static_cast<int32_t>(h_out) - padTop_) % static_cast<int32_t>(H_);
            if (h_src < 0) h_src += H_;
            int32_t w_src = (static_cast<int32_t>(w_out) - padLeft_) % static_cast<int32_t>(W_);
            if (w_src < 0) w_src += W_;

            uint32_t srcLinear = n * (C_ * D_ * H_ * W_) + c * inputStrideC_ +
                                 static_cast<uint32_t>(d_src) * inputStrideD_ +
                                 static_cast<uint32_t>(h_src) * inputStrideH_ +
                                 static_cast<uint32_t>(w_src);
            yGm_.SetValue(offset, xGm_.GetValue(srcLinear));
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutputLength_;
    uint32_t N_, C_, D_, H_, W_;
    uint32_t Dout_, Hout_, Wout_;
    int32_t padLeft_, padRight_, padTop_, padBottom_, padFront_, padBack_;
    uint32_t tileLength_;

    uint32_t inputStrideC_, inputStrideD_, inputStrideH_, inputStrideW_;
    uint32_t outputStrideC_, outputStrideD_, outputStrideH_, outputStrideW_;
};

extern "C" __global__ __aicore__ void ca6k_04616_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04616Custom op;
    op.Init(x, y,
            tiling_data.totalOutputLength,
            tiling_data.N, tiling_data.C, tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.Dout, tiling_data.Hout, tiling_data.Wout,
            tiling_data.padLeft, tiling_data.padRight,
            tiling_data.padTop, tiling_data.padBottom,
            tiling_data.padFront, tiling_data.padBack,
            tiling_data.tileLength);
    op.Process();
}
'''
