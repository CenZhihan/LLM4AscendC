#include "ca6k_00236_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00236CustomTilingData tiling;

    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &originShape = storageShape->GetOriginShape();
    uint32_t batch = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t height = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t width = static_cast<uint32_t>(originShape.GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dimAttr;

    // totalRows = batch * width (each row of length height)
    uint32_t totalRows = batch * width;

    constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    // Per row: we need input, output, temp for exponentials (three buffers of height floats each)
    // This means 3 * height * sizeof(float) per row, plus small scalar buffers.
    // For height=512, that is 3*512*4 = 6144 bytes per row. Max rows: 192*1024/6144 = 32 rows.
    // We also need aligned DMA transfers. Use 2048-aligned tileRows.
    constexpr uint32_t ALIGN = 8;
    uint32_t maxRows = UB_BUDGET_BYTES / (3 * height * sizeof(float));
    if (maxRows < 1) maxRows = 1;
    uint32_t tileRows = (maxRows / ALIGN) * ALIGN;
    if (tileRows == 0) tileRows = ALIGN;
    if (tileRows > totalRows) tileRows = totalRows;

    uint32_t numTiles = (totalRows + tileRows - 1) / tileRows;
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 32) blockDim = 32;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_tileRows(tileRows);
    tiling.set_height(height);
    tiling.set_width(width);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00236Custom {
public:
    __aicore__ inline KernelCa6k00236Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t tileRows,
                                uint32_t height, uint32_t width, int32_t dim)
    {
        totalRows_ = totalRows;
        tileRows_ = tileRows;
        height_ = height;
        width_ = width;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(xQueue_, BUFFER_NUM, height_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, height_ * sizeof(float));
        pipe_.InitBuffer(tmpExp_, 1, height_ * sizeof(float));
        pipe_.InitBuffer(tmpMax_, 1, height_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || tileRows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride across rows
        const uint32_t stride = blockNum * tileRows_;
        uint32_t rowStart = blockIdx * tileRows_;
        while (rowStart < totalRows_) {
            uint32_t curTileRows = totalRows_ - rowStart;
            if (curTileRows > tileRows_) curTileRows = tileRows_;

            // For each row in the tile
            for (uint32_t r = 0; r < curTileRows; ++r) {
                uint32_t rowIdx = rowStart + r;
                ProcessOneRow(rowIdx);
            }

            rowStart += stride;
        }
    }

private:
    __aicore__ inline void ProcessOneRow(uint32_t rowIdx)
    {
        // Load one row of length height_ from global memory
        AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowIdx * height_], height_);
        xQueue_.EnQue(xLocal);

        AscendC::LocalTensor<float> x = xQueue_.DeQue<float>();

        // Compute max(x) along the row (we store x negated: softmin = softmax(-x))
        // Actually softmin = softmax(-x). We'll compute neg_x first.
        AscendC::LocalTensor<float> negX = tmpExp_.Get();
        AscendC::Muls(negX, x, -1.0f, height_);

        // Compute max of negX along the row
        AscendC::LocalTensor<float> maxBuf = tmpMax_.Get();
        AscendC::LocalTensor<float> maxVal = tmpMax_.Get(); // reuse
        // ReduceMax returns scalar, but we need broadcast subtract.
        // Use ReduceMax to get max, then broadcast subtract.
        AscendC::ReduceMax(maxVal, negX, height_); // produces single element
        // Broadcast subtract: negX - maxVal
        AscendC::Sub(negX, negX, maxVal, height_);

        // Exponentiate
        AscendC::Exp(negX, negX, height_);

        // Compute sum of exp values along row
        AscendC::LocalTensor<float> sumExp = tmpExp_.Get();
        AscendC::ReduceSum(sumExp, negX, height_); // scalar

        // Divide: negX / sumExp
        AscendC::Div(negX, negX, sumExp, height_);

        // Write result to global memory (rowIdx in output)
        AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, negX, height_); // copy result into yLocal
        yQueue_.EnQue(yLocal);

        // Actually we need to copy from yLocal to global
        AscendC::LocalTensor<float> y = yQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[rowIdx * height_], y, height_);
        yQueue_.FreeTensor(y);
        xQueue_.FreeTensor(x);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpExp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMax_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t tileRows_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00236_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00236Custom op;
    op.Init(x, y,
            tiling_data.totalRows,
            tiling_data.tileRows,
            tiling_data.height,
            tiling_data.width,
            tiling_data.dim);
    op.Process();
}
