kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04595Custom {
public:
    __aicore__ inline KernelCa6k04595Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                int32_t N, int32_t C, int32_t dIn, int32_t hIn, int32_t wIn,
                                int32_t dOut, int32_t hOut, int32_t wOut,
                                int32_t totalOutElems)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(N*C*dIn*hIn*wIn));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(N*C*dOut*hOut*wOut));
        N_ = N;
        C_ = C;
        dIn_ = dIn;
        hIn_ = hIn;
        wIn_ = wIn;
        dOut_ = dOut;
        hOut_ = hOut;
        wOut_ = wOut;
        totalOutElems_ = totalOutElems;

        // Set strides (contiguous NCDHW)
        strideC_ = dIn * hIn * wIn;
        strideD_ = hIn * wIn;
        strideH_ = wIn;
        strideW_ = 1;
    }

    __aicore__ inline void Process()
    {
        if (totalOutElems_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride loop over output elements
        for (int32_t idx = static_cast<int32_t>(blockIdx); idx < totalOutElems_; idx += static_cast<int32_t>(blockNum))
        {
            // Decode linear index to (b, c, od, oh, ow)
            int32_t tmp = idx;
            int32_t ow = tmp % wOut_; tmp /= wOut_;
            int32_t oh = tmp % hOut_; tmp /= hOut_;
            int32_t od = tmp % dOut_; tmp /= dOut_;
            int32_t c = tmp % C_; tmp /= C_;
            int32_t b = tmp % N_;

            // Compute window start and end indices (floor mapping)
            int32_t startD = (od * dIn_) / dOut_;
            int32_t endD   = ((od + 1) * dIn_) / dOut_;
            int32_t startH = (oh * hIn_) / hOut_;
            int32_t endH   = ((oh + 1) * hIn_) / hOut_;
            int32_t startW = (ow * wIn_) / wOut_;
            int32_t endW   = ((ow + 1) * wIn_) / wOut_;

            // Base offset for input batch and channel
            int32_t baseN = b * C_ * strideC_;
            int32_t baseC = c * strideC_;

            float maxVal = -1.0e38f;
            for (int32_t d = startD; d < endD; ++d) {
                int32_t offsetD = d * strideD_;
                for (int32_t h = startH; h < endH; ++h) {
                    int32_t offsetH = h * strideH_;
                    for (int32_t w = startW; w < endW; ++w) {
                        int32_t linearIdx = baseN + baseC + offsetD + offsetH + w;
                        float val = xGm_.GetValue(linearIdx);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }

            // Write output at corresponding location
            int32_t outOff = (b * C_ * dOut_ * hOut_ * wOut_) + (c * dOut_ * hOut_ * wOut_) +
                             (od * hOut_ * wOut_) + (oh * wOut_) + ow;
            yGm_.SetValue(outOff, maxVal);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    int32_t N_, C_, dIn_, hIn_, wIn_;
    int32_t dOut_, hOut_, wOut_;
    int32_t totalOutElems_;
    int32_t strideC_, strideD_, strideH_, strideW_;
};

extern "C" __global__ __aicore__ void ca6k_04595_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04595Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C,
            tiling_data.dIn, tiling_data.hIn, tiling_data.wIn,
            tiling_data.dOut, tiling_data.hOut, tiling_data.wOut,
            tiling_data.totalOutElems);
    op.Process();
}
'''
