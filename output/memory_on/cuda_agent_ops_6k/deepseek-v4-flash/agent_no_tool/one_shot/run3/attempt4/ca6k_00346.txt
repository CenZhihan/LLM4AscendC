kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_H = 16;
constexpr int32_t TILE_W = 16;

class KernelCa6k00346Custom {
public:
    __aicore__ inline KernelCa6k00346Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t outputH, uint32_t outputW,
                                uint32_t tileH, uint32_t tileW,
                                float scaleFactor) {
        xGm_ = (__gm__ float*)x;
        yGm_ = (__gm__ float*)y;
        N_ = N;
        C_ = C;
        inputH_ = inputH;
        inputW_ = inputW;
        outputH_ = outputH;
        outputW_ = outputW;
        tileH_ = tileH;
        tileW_ = tileW;
        scaleFactor_ = scaleFactor;

        // Precompute reciprocal for fast division
        invScale_ = 1.0f / scaleFactor;

        // Calculate number of tiles per dim
        numTilesH_ = (outputH_ + tileH_ - 1) / tileH_;
        numTilesW_ = (outputW_ + tileW_ - 1) / tileW_;
        tilesPerChannel_ = numTilesH_ * numTilesW_;

        // Allocate local buffer for one output tile (float)
        pipe_.InitBuffer(outTileBuf_, 1, tileH_ * tileW_ * sizeof(float));
        // For reading input rows: we will load two input rows at a time
        // Each row has inputW_ floats, but we only need a segment covering the tile.
        // Actually for each output tile, we need input rows from iy_start to iy_end inclusive.
        // We'll load the needed input rows into local buffers on the fly.
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Determine which (n,c,tileRow,tileCol) this block handles
        uint32_t linearTileIdx = blockIdx;

        uint32_t n = linearTileIdx / (C_ * tilesPerChannel_);
        uint32_t rem = linearTileIdx % (C_ * tilesPerChannel_);
        uint32_t c = rem / tilesPerChannel_;
        uint32_t tileIdxInChannel = rem % tilesPerChannel_;
        uint32_t tileRow = tileIdxInChannel / numTilesW_;
        uint32_t tileCol = tileIdxInChannel % numTilesW_;

        // Output tile bounds
        uint32_t oyStart = tileRow * tileH_;
        uint32_t oxStart = tileCol * tileW_;
        uint32_t oyEnd = oyStart + tileH_;
        if (oyEnd > outputH_) oyEnd = outputH_;
        uint32_t oxEnd = oxStart + tileW_;
        if (oxEnd > outputW_) oxEnd = outputW_;
        uint32_t actualTileH = oyEnd - oyStart;
        uint32_t actualTileW = oxEnd - oxStart;

        // Corresponding input region (with margin for interpolation)
        // For each out pixel (oy,ox), we need input pixels:
        //   (floor(oy * invScale), floor(ox * invScale))
        //   and floor+1 (if within bounds)
        // So input row range: floor(oyStart * invScale) to ceil(oyEnd * invScale)
        // For safety, we load from floor(oyStart * invScale) to ceil(oyEnd * invScale)
        float inv = invScale_;
        uint32_t iyStart = (uint32_t)(oyStart * inv);
        uint32_t iyEnd = (uint32_t)((oyEnd - 1) * inv) + 2; // +1 for ceil, +1 for next neighbor
        if (iyEnd > inputH_) iyEnd = inputH_;
        uint32_t ixStart = (uint32_t)(oxStart * inv);
        uint32_t ixEnd = (uint32_t)((oxEnd - 1) * inv) + 2;
        if (ixEnd > inputW_) ixEnd = inputW_;
        uint32_t inputRows = iyEnd - iyStart;
        uint32_t inputCols = ixEnd - ixStart;

        // We'll process each output row, loading required input rows.
        // For each output row oy, we need two input rows: iy and iy+1.
        // We'll maintain two row buffers.
        AscendC::LocalTensor<float> rowLocal0, rowLocal1;
        // Actually we'll load a full input tile of size inputRows x inputCols into a local buffer.
        // But to reduce memory, we can load row by row.
        // Since we have limited UB, we will load rows into local tensors one at a time.
        // We'll load one row at a time and compute output rows that depend on it.
        // For bilinear, each output row uses two consecutive input rows.
        // We'll use a sliding window: load first two rows, compute output rows that need them, then shift.
        // We'll allocate two row buffers of length inputCols (max tileW+1 due to margin).
        uint32_t maxRowLen = (inputCols > 0) ? inputCols : 1;
        pipe_.InitBuffer(rowBuf0_, 1, maxRowLen * sizeof(float));
        pipe_.InitBuffer(rowBuf1_, 1, maxRowLen * sizeof(float));

        // Compute number of output rows that use a given pair of input rows
        // For each output row oy, the required iy = floor(oy * inv) and iy+1.
        // When iy increases, we slide.
        // We'll iterate oy from oyStart to oyEnd, but we need to load input rows as needed.

        // Precompute starting iy for first output row
        uint32_t currentIy = (uint32_t)(oyStart * inv);
        // Load the first two input rows (currentIy and currentIy+1)
        LoadInputRow(rowBuf0_, n, c, currentIy, ixStart, inputCols);
        if (currentIy + 1 < inputH_) {
            LoadInputRow(rowBuf1_, n, c, currentIy + 1, ixStart, inputCols);
        } else {
            // Edge: duplicate last row
            LoadInputRow(rowBuf1_, n, c, inputH_ - 1, ixStart, inputCols);
        }

        // Allocate output tile buffer
        AscendC::LocalTensor<float> outTile = outTileBuf_.Get<float>();

        // Iterate over output rows
        uint32_t outRowIdx = 0;
        for (uint32_t oy = oyStart; oy < oyEnd; oy++) {
            float inY = oy * inv;
            uint32_t iy = (uint32_t)inY;
            float dy = inY - iy;

            // Check if we need to slide rows
            if (iy > currentIy) {
                // Shift: rowBuf0_ becomes previous rowBuf1_, load new row
                AscendC::LocalTensor<float> tmp = rowBuf0_;
                rowBuf0_ = rowBuf1_;
                rowBuf1_ = tmp;
                // Load new row iy+1 (or duplicate if at edge)
                uint32_t nextIy = iy + 1;
                if (nextIy >= inputH_) nextIy = inputH_ - 1;
                LoadInputRow(rowBuf1_, n, c, nextIy, ixStart, inputCols);
                currentIy = iy;
            }

            AscendC::LocalTensor<float> row0 = rowBuf0_.Get<float>();
            AscendC::LocalTensor<float> row1 = rowBuf1_.Get<float>();

            for (uint32_t ox = oxStart; ox < oxEnd; ox++) {
                float inX = ox * inv;
                uint32_t ix = (uint32_t)inX;
                float dx = inX - ix;

                // Clamp indices to valid range (should already be within loaded region)
                uint32_t localIx0 = ix - ixStart;
                uint32_t localIx1 = localIx0 + 1;
                if (localIx1 >= inputCols) localIx1 = inputCols - 1;

                // Read four input values
                float v00 = row0(localIx0);
                float v01 = row0(localIx1);
                float v10 = row1(localIx0);
                float v11 = row1(localIx1);

                // Bilinear interpolation
                float top = v00 + dx * (v01 - v00);
                float bottom = v10 + dx * (v11 - v10);
                float val = top + dy * (bottom - top);

                // Store to output tile
                uint32_t localOffset = outRowIdx * actualTileW + (ox - oxStart);
                outTile(localOffset) = val;
            }
            outRowIdx++;
        }

        // Copy output tile to global memory
        uint32_t outBase = n * C_ * outputH_ * outputW_ + c * outputH_ * outputW_ + oyStart * outputW_ + oxStart;
        for (uint32_t i = 0; i < actualTileH; i++) {
            AscendC::DataCopy(yGm_[outBase + i * outputW_], outTile[i * actualTileW], actualTileW);
        }
    }

private:
    __aicore__ inline void LoadInputRow(AscendC::LocalTensor<float> &localBuf,
                                        uint32_t n, uint32_t c,
                                        uint32_t row, uint32_t colStart, uint32_t len) {
        uint32_t inBase = n * C_ * inputH_ * inputW_ + c * inputH_ * inputW_ + row * inputW_ + colStart;
        AscendC::DataCopy(localBuf, xGm_[inBase], len);
    }

private:
    __gm__ float *xGm_;
    __gm__ float *yGm_;

    uint32_t N_, C_, inputH_, inputW_, outputH_, outputW_;
    uint32_t tileH_, tileW_, numTilesH_, numTilesW_, tilesPerChannel_;
    float scaleFactor_, invScale_;

    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> outTileBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> rowBuf0_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> rowBuf1_;
};

extern "C" __global__ __aicore__ void ca6k_00346_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00346Custom op;
    op.Init(x, y,
            tiling_data.N,
            tiling_data.C,
            tiling_data.inputH,
            tiling_data.inputW,
            tiling_data.outputH,
            tiling_data.outputW,
            tiling_data.tileH,
            tiling_data.tileW,
            tiling_data.scaleFactor);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor ca6k_00346_impl_npu(const at::Tensor& self, double scale_factor) {
    at::Tensor result = at::empty_like(self);
    // Determine output shape from scale factor
    int64_t H = self.size(2);
    int64_t W = self.size(3);
    int64_t outH = static_cast<int64_t>(H * scale_factor);
    int64_t outW = static_cast<int64_t>(W * scale_factor);
    result = at::empty({self.size(0), self.size(1), outH, outW}, self.options());
    EXEC_NPU_CMD(aclnnCa6k00346Custom, self, scale_factor, result);
    return result;
}

    m.impl("ca6k_00346_custom", &ca6k_00346_impl_npu);
}

import torch
import torch_npu

_FN = 'ca6k_00346_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        if len(init_inputs) > 0:
            self.scale_factor = init_inputs[0]
        else:
            self.scale_factor = 2.0

    def forward(self, *inputs):
        fn = getattr(custom_ops_lib, _FN)
        x = inputs[0]
        return fn(x, self.scale_factor)
