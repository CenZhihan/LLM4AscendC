kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t MAX_COL_FLT = 16384;  // max elements per column that fit in UB (conservative)

class KernelMedian {
public:
    __aicore__ inline KernelMedian() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace,
                                uint32_t rows, uint32_t cols)
    {
        rows_ = rows;
        cols_ = cols;
        xGm_.SetGlobalBuffer((__gm__ float*)x, rows * cols);
        yGm_.SetGlobalBuffer((__gm__ float*)y, cols);
        wsGm_.SetGlobalBuffer((__gm__ float*)workspace, rows * cols);
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = GetBlockIdx();
        if (blockIdx >= cols_) return;

        // compute median for column blockIdx
        uint32_t colOffset = blockIdx; // columns are contiguous in memory (row-major)
        uint32_t step = cols_;

        // Copy column into workspace (contiguous) to simplify random access
        for (uint32_t r = 0; r < rows_; r += MAX_COL_FLT) {
            uint32_t copyLen = (r + MAX_COL_FLT <= rows_) ? MAX_COL_FLT : (rows_ - r);
            LocalTensor<float> localBuf = inQueue_.AllocTensor<float>();
            DataCopy(localBuf, xGm_[r * cols_ + blockIdx * 1], copyLen);
            inQueue_.EnQue(localBuf);
            LocalTensor<float> localTmp = inQueue_.DeQue<float>();
            DataCopy(wsGm_[r * cols_ + blockIdx * 1], localTmp, copyLen);
            inQueue_.FreeTensor(localTmp);
        }

        // Now workspace column is contiguous in wsGm at offsets: blockIdx for each row
        // Use quickselect on workspace column
        int32_t left = 0;
        int32_t right = static_cast<int32_t>(rows_) - 1;
        int32_t k = static_cast<int32_t>((rows_ - 1) / 2);

        while (left < right) {
            // pick pivot (first element)
            float pivot = wsGm_[left * cols_ + blockIdx * 1];
            int32_t i = left - 1;
            int32_t j = right + 1;
            while (true) {
                do { i++; } while (wsGm_[i * cols_ + blockIdx * 1] < pivot);
                do { j--; } while (wsGm_[j * cols_ + blockIdx * 1] > pivot);
                if (i >= j) break;
                float tmp = wsGm_[i * cols_ + blockIdx * 1];
                wsGm_[i * cols_ + blockIdx * 1] = wsGm_[j * cols_ + blockIdx * 1];
                wsGm_[j * cols_ + blockIdx * 1] = tmp;
            }
            if (k <= j) right = j;
            else left = j + 1;
        }

        float median = wsGm_[k * cols_ + blockIdx * 1];
        yGm_[blockIdx] = median;
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> wsGm_;
    uint32_t rows_;
    uint32_t cols_;
    TPipe pipe_;
    TQue<TPosition::VECIN, 1> inQueue_;
};

extern "C" __global__ __aicore__ void ca6k_03088_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(td, tiling);
    KernelMedian op;
    op.Init(x, y, workspace, td.rows, td.cols);
    op.Process();
}
"""
