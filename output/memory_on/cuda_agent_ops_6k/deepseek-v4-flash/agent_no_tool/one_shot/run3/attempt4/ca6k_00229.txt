#include "ca6k_00229_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/types.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00229CustomTilingData tiling;

    // Correct way: GetInputShape returns const gert::StorageShape*
    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::Shape &x_shape = x_storage->GetOriginShape();
    uint32_t batchSize = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t channels = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t inputH = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t inputW = static_cast<uint32_t>(x_shape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t padTop = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(0)[0]);
    uint32_t padBottom = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(1)[0]);
    uint32_t padLeft = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(2)[0]);
    uint32_t padRight = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(3)[0]);

    uint32_t outputH = inputH + padTop + padBottom;
    uint32_t outputW = inputW + padLeft + padRight;
    uint32_t xLen = batchSize * channels * inputH * inputW;
    uint32_t yLen = batchSize * channels * outputH * outputW;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (yLen < maxTile) ? yLen : maxTile;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = 1;

    uint32_t numTiles = CeilDiv(yLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_xLen(xLen);
    tiling.set_yLen(yLen);
    tiling.set_tileLen(tileLen);
    tiling.set_batchSize(batchSize);
    tiling.set_channels(channels);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_padTop(padTop);
    tiling.set_padBottom(padBottom);
    tiling.set_padLeft(padLeft);
    tiling.set_padRight(padRight);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    int64_t batch = x_shape->GetDim(0);
    int64_t channels = x_shape->GetDim(1);
    int64_t inputH = x_shape->GetDim(2);
    int64_t inputW = x_shape->GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t padTop = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t padBottom = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padLeft = attrs->GetAttrPointer<int32_t>(2)[0];
    int32_t padRight = attrs->GetAttrPointer<int32_t>(3)[0];

    int64_t outputH = inputH + padTop + padBottom;
    int64_t outputW = inputW + padLeft + padRight;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, batch);
    y_shape->SetDim(1, channels);
    y_shape->SetDim(2, outputH);
    y_shape->SetDim(3, outputW);
    return GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

class KernelCa6k00229Custom {
public:
    __aicore__ inline KernelCa6k00229Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t xLen, uint32_t yLen, uint32_t tileLen,
                                uint32_t batchSize, uint32_t channels,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t padTop, uint32_t padBottom,
                                uint32_t padLeft, uint32_t padRight,
                                uint32_t outputH, uint32_t outputW) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, xLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yLen);
        tileLen_ = tileLen;
        batchSize_ = batchSize;
        channels_ = channels;
        inputH_ = inputH;
        inputW_ = inputW;
        padTop_ = padTop;
        padBottom_ = padBottom;
        padLeft_ = padLeft;
        padRight_ = padRight;
        outputH_ = outputH;
        outputW_ = outputW;
        xLen_ = xLen;
        yLen_ = yLen;
    }

    __aicore__ inline void Process() {
        if (yLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t stride = blockNum * tileLen_;

        uint32_t start = blockIdx * tileLen_;
        if (start >= yLen_) return;

        uint32_t end = start + tileLen_;
        if (end > yLen_) end = yLen_;

        for (uint32_t idx = start; idx < end; idx += stride) {
            uint32_t curLen = (idx + tileLen_ > yLen_) ? (yLen_ - idx) : tileLen_;
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t outIdx = idx + i;
                uint32_t tmp = outIdx;
                uint32_t n = tmp / (channels_ * outputH_ * outputW_);
                tmp %= channels_ * outputH_ * outputW_;
                uint32_t c = tmp / (outputH_ * outputW_);
                tmp %= outputH_ * outputW_;
                uint32_t h = tmp / outputW_;
                uint32_t w = tmp % outputW_;

                float val = 0.0f;
                if (h >= padTop_ && h < (padTop_ + inputH_) &&
                    w >= padLeft_ && w < (padLeft_ + inputW_)) {
                    uint32_t inH = h - padTop_;
                    uint32_t inW = w - padLeft_;
                    uint32_t inIdx = n * (channels_ * inputH_ * inputW_) +
                                     c * (inputH_ * inputW_) +
                                     inH * inputW_ + inW;
                    val = xGm_.GetValue(inIdx);
                }
                yGm_.SetValue(outIdx, val);
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t tileLen_;
    uint32_t batchSize_;
    uint32_t channels_;
    uint32_t inputH_;
    uint32_t inputW_;
    uint32_t padTop_;
    uint32_t padBottom_;
    uint32_t padLeft_;
    uint32_t padRight_;
    uint32_t outputH_;
    uint32_t outputW_;
    uint32_t xLen_;
    uint32_t yLen_;
};

extern "C" __global__ __aicore__ void ca6k_00229_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00229Custom op;
    op.Init(x, y,
            tiling_data.xLen, tiling_data.yLen, tiling_data.tileLen,
            tiling_data.batchSize, tiling_data.channels,
            tiling_data.inputH, tiling_data.inputW,
            tiling_data.padTop, tiling_data.padBottom,
            tiling_data.padLeft, tiling_data.padRight,
            tiling_data.outputH, tiling_data.outputW);
    op.Process();
}
'''
