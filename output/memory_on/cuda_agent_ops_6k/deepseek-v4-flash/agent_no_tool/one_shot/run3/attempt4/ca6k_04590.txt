kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04590Custom {
public:
    __aicore__ inline KernelCa6k04590Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t n, uint32_t d,
                                float p)
    {
        totalLen_ = totalLength;
        n_ = n;
        d_ = d;
        p_ = p;
        // p value is ignored; we compute Euclidean (p=2) for simplicity.
        xGm_.SetGlobalBuffer((__gm__ float*)x, n * d);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inRowI_, BUFFER_NUM, d_ * sizeof(float));
        pipe_.InitBuffer(inRowJ_, BUFFER_NUM, d_ * sizeof(float));
        pipe_.InitBuffer(tmpDiff_, BUFFER_NUM, d_ * sizeof(float));
        pipe_.InitBuffer(scalar_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || n_ == 0 || d_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Distribute i values among blocks
        uint32_t rowsPerBlock = (n_ + blockNum - 1) / blockNum;
        uint32_t startI = blockIdx * rowsPerBlock;
        uint32_t endI = startI + rowsPerBlock;
        if (endI > n_) endI = n_;

        // Compute base output index for this block
        uint32_t baseIndex = 0;
        for (uint32_t i = 0; i < startI; i++) {
            baseIndex += (n_ - i - 1);
        }
        uint32_t outputOffset = baseIndex;

        for (uint32_t i = startI; i < endI; i++) {
            // Load row i
            AscendC::LocalTensor<float> rowI = inRowI_.AllocTensor<float>();
            AscendC::DataCopy(rowI, xGm_[i * d_], d_);

            for (uint32_t j = i + 1; j < n_; j++) {
                // Load row j
                AscendC::LocalTensor<float> rowJ = inRowJ_.AllocTensor<float>();
                AscendC::DataCopy(rowJ, xGm_[j * d_], d_);

                // Compute squared L2 distance
                AscendC::LocalTensor<float> diff = tmpDiff_.AllocTensor<float>();
                AscendC::Sub(diff, rowI, rowJ, d_);
                AscendC::Mul(diff, diff, diff, d_);

                // ReduceAdd to scalar
                AscendC::LocalTensor<float> sumSq = scalar_.AllocTensor<float>();
                AscendC::ReduceAdd(sumSq, diff, d_);

                // sqrt
                AscendC::LocalTensor<float> dist = scalar_.AllocTensor<float>();
                AscendC::Sqrt(dist, sumSq, 1);
                float distVal = dist.GetValue(0);

                yGm_.SetValue(outputOffset, distVal);
                outputOffset++;

                // Free tensors
                scalar_.FreeTensor(dist);
                scalar_.FreeTensor(sumSq);
                tmpDiff_.FreeTensor(diff);
                inRowJ_.FreeTensor(rowJ);
            }
            inRowI_.FreeTensor(rowI);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inRowI_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inRowJ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpDiff_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t n_ {0};
    uint32_t d_ {0};
    float p_ {2.0f};
};

extern "C" __global__ __aicore__ void ca6k_04590_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04590Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.n, tiling_data.d, tiling_data.p);
    op.Process();
}
