#include "ca6k02880_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k02880CustomTilingData tiling;

    const gert::StorageShape *xShape = context->GetInputShape(0);
    uint32_t batch = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t channels = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    uint32_t inputLength = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];

    int32_t outLen = (inputLength + 2 * padding - kernelSize) / stride + 1;
    if (outLen <= 0) outLen = 1;
    uint32_t outputLength = static_cast<uint32_t>(outLen);
    uint32_t totalLength = batch * channels * outputLength;

    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float) + sizeof(int64_t));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    // Use setter methods (AscendC macro generates setters)
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_inputLength(inputLength);
    tiling.set_outputLength(outputLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -3.40282347e+38f;

class KernelCa6k02880Custom {
public:
    __aicore__ inline KernelCa6k02880Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR indices,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t kernelSize, int32_t stride, int32_t padding,
                                uint32_t batch, uint32_t channels,
                                uint32_t inputLength, uint32_t outputLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        batch_ = batch;
        channels_ = channels;
        inputLen_ = inputLength;
        outputLen_ = outputLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        idxGm_.SetGlobalBuffer((__gm__ int64_t*)indices);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            Compute(base, curLen);

            base += strideElems;
        }
    }

private:
    __aicore__ inline void Compute(uint32_t base, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t elemIdx = base + i;
            uint32_t n = elemIdx / (channels_ * outputLen_);
            uint32_t rem = elemIdx % (channels_ * outputLen_);
            uint32_t c = rem / outputLen_;
            uint32_t o = rem % outputLen_;

            int32_t start = static_cast<int32_t>(o) * stride_ - padding_;
            float best = NEG_INF;
            int64_t bestIdx = 0;
            for (int32_t k = 0; k < kernelSize_; ++k) {
                int32_t inIdx = start + k;
                if (inIdx >= 0 && inIdx < static_cast<int32_t>(inputLen_)) {
                    uint32_t flatIdx = n * channels_ * inputLen_ + c * inputLen_ + static_cast<uint32_t>(inIdx);
                    float val = xGm_[flatIdx];
                    if (val > best) {
                        best = val;
                        bestIdx = static_cast<int64_t>(inIdx);
                    }
                }
            }
            yGm_[elemIdx] = best;
            idxGm_[elemIdx] = bestIdx;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<int64_t> idxGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t kernelSize_ {3};
    int32_t stride_ {1};
    int32_t padding_ {1};
    uint32_t batch_ {1};
    uint32_t channels_ {1};
    uint32_t inputLen_ {1};
    uint32_t outputLen_ {1};
};

extern "C" __global__ __aicore__ void ca6k_02880_custom(GM_ADDR x, GM_ADDR y, GM_ADDR indices, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02880Custom op;
    op.Init(x, y, indices,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.batch, tiling_data.channels,
            tiling_data.inputLength, tiling_data.outputLength);
    op.Process();
}

model_src = '''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

_FN = 'ca6k_02880_custom'

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv1d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)

    def forward(self, x):
        conv_out = self.conv(x)
        fn = getattr(custom_ops_lib, _FN)
        out, indices = fn(conv_out, kernel_size=3, stride=1, padding=1)
        return out, indices
