kernel_src='''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;
constexpr float NEG_INF = -1e30f;
constexpr float LOG_EPS = 1e-20f;

class KernelCa6k02718Custom {
public:
    __aicore__ inline KernelCa6k02718Custom() {}

    __aicore__ inline void Init(GM_ADDR log_probs, GM_ADDR targets,
                                GM_ADDR input_lengths, GM_ADDR target_lengths,
                                GM_ADDR loss,
                                uint32_t totalBatch,
                                uint32_t seqLength,
                                uint32_t numClasses, uint32_t maxTargetLength,
                                float totalBatchFloat,
                                int32_t blank, int32_t reduction)
    {
        totalBatch_ = totalBatch;
        totalBatchFloat_ = totalBatchFloat;
        seqLength_ = seqLength;
        numClasses_ = numClasses;
        maxTargetLength_ = maxTargetLength;
        blank_ = blank;
        reduction_ = reduction;
        batchStride_ = totalBatch_ * numClasses_; // stride between time steps

        logProbsGm_.SetGlobalBuffer((__gm__ float*)log_probs, totalBatch_ * seqLength_ * numClasses_);
        targetsGm_.SetGlobalBuffer((__gm__ int32_t*)targets, totalBatch_ * maxTargetLength_);
        inputLengthsGm_.SetGlobalBuffer((__gm__ int32_t*)input_lengths, totalBatch_);
        targetLengthsGm_.SetGlobalBuffer((__gm__ int32_t*)target_lengths, totalBatch_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        // buffers: one time step of log_probs (numClasses)
        pipe_.InitBuffer(inQueueLogProbs_, BUFFER_NUM, numClasses_ * sizeof(float));
        // targets for one sample (max size)
        pipe_.InitBuffer(inQueueTargets_, BUFFER_NUM, maxTargetLength_ * sizeof(int32_t));
        // alpha arrays: max possible S = (maxTargetLength_*2 + 1)
        uint32_t maxS = maxTargetLength_ * 2 + 1;
        pipe_.InitBuffer(alphaPrevQue_, 1, maxS * sizeof(float));
        pipe_.InitBuffer(alphaCurrQue_, 1, maxS * sizeof(float));
        // temporary for log-sum-exp (2 floats, but we use only first)
        pipe_.InitBuffer(tmpQue_, 1, 2 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalBatch_ == 0) return;
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());
        if (blockIdx != 0) return; // single core

        float totalLoss = 0.0f;

        for (uint32_t b = 0; b < totalBatch_; ++b) {
            int32_t T = inputLengthsGm_.GetValue(b);
            int32_t U = targetLengthsGm_.GetValue(b);

            uint32_t S = (U == 0) ? 1 : (2u * static_cast<uint32_t>(U) + 1u);

            // Copy targets for this sample if U>0
            LocalTensor<int32_t> targetsLocal;
            if (U > 0) {
                targetsLocal = inQueueTargets_.AllocTensor<int32_t>();
                DataCopy(targetsLocal, targetsGm_[b * maxTargetLength_], maxTargetLength_);
                inQueueTargets_.EnQue(targetsLocal);
                targetsLocal = inQueueTargets_.DeQue<int32_t>();
            }

            // Allocate alpha buffers
            LocalTensor<float> alphaPrev = alphaPrevQue_.AllocTensor<float>();
            LocalTensor<float> alphaCurr = alphaCurrQue_.AllocTensor<float>();

            // Initialize alpha for t=0
            // base offset for sample b in global log_probs (first time step)
            uint32_t sampleBase = b * numClasses_;
            for (uint32_t s = 0; s < S; ++s) {
                alphaPrev.SetValue(s, NEG_INF);
            }
            // alpha[0] = log_probs[0, b, blank]
            alphaPrev.SetValue(0, logProbsGm_.GetValue(0 * batchStride_ + sampleBase + blank_));
            if (U > 0) {
                // alpha[1] = log_probs[0, b, target[0]]
                alphaPrev.SetValue(1, logProbsGm_.GetValue(0 * batchStride_ + sampleBase + targetsLocal.GetValue(0)));
            }

            // Loop over time steps t = 1 ... T-1
            for (int32_t t = 1; t < T; ++t) {
                // Load log_probs for this time step
                LocalTensor<float> logProbsLocal = inQueueLogProbs_.AllocTensor<float>();
                DataCopy(logProbsLocal, logProbsGm_[t * batchStride_ + sampleBase], numClasses_);
                inQueueLogProbs_.EnQue(logProbsLocal);
                LocalTensor<float> logP = inQueueLogProbs_.DeQue<float>();

                // Initialize alphaCurr to -inf
                for (uint32_t s = 0; s < S; ++s) {
                    alphaCurr.SetValue(s, NEG_INF);
                }

                // Recurrence
                for (uint32_t s = 0; s < S; ++s) {
                    int32_t lbl;
                    if (s % 2 == 0) {
                        lbl = blank_;
                    } else {
                        lbl = targetsLocal.GetValue(static_cast<int32_t>(s / 2));
                    }
                    float prob = logP.GetValue(lbl);

                    // from alphaPrev[s]
                    float sum = alphaPrev.GetValue(s) + prob;
                    float newVal = logSumExp(alphaCurr.GetValue(s), sum);
                    alphaCurr.SetValue(s, newVal);

                    // from alphaPrev[s-1] if s>0
                    if (s > 0) {
                        sum = alphaPrev.GetValue(s - 1) + prob;
                        newVal = logSumExp(alphaCurr.GetValue(s), sum);
                        alphaCurr.SetValue(s, newVal);
                    }

                    // from alphaPrev[s-2] if s>1 and labels differ
                    if (s > 1) {
                        int32_t lbl2 = (s % 2 == 0) ? blank_ : targetsLocal.GetValue(static_cast<int32_t>(s / 2));
                        int32_t lblPrev2 = ((s - 2) % 2 == 0) ? blank_ : targetsLocal.GetValue(static_cast<int32_t>((s - 2) / 2));
                        if (lbl2 != lblPrev2) {
                            sum = alphaPrev.GetValue(s - 2) + prob;
                            newVal = logSumExp(alphaCurr.GetValue(s), sum);
                            alphaCurr.SetValue(s, newVal);
                        }
                    }
                }

                // Swap alpha buffers
                LocalTensor<float> tmp = alphaPrev;
                alphaPrev = alphaCurr;
                alphaCurr = tmp;

                // Free logP
                inQueueLogProbs_.FreeTensor(logP);
            }

            // Final loss: -(log(alpha_{T-1}[S-1] + alpha_{T-1}[S-2]) for S>=2 else -alpha[0])
            float sampleLoss;
            if (S >= 2) {
                sampleLoss = -logSumExp(alphaPrev.GetValue(S - 1), alphaPrev.GetValue(S - 2));
            } else {
                sampleLoss = -alphaPrev.GetValue(0);
            }
            totalLoss += sampleLoss;

            // Free alpha buffers
            alphaPrevQue_.FreeTensor(alphaPrev);
            alphaCurrQue_.FreeTensor(alphaCurr);
            if (U > 0) {
                inQueueTargets_.FreeTensor(targetsLocal);
            }
        }

        if (reduction_ == 1) { // mean
            totalLoss = totalLoss / totalBatchFloat_;
        }
        lossGm_.SetValue(0, totalLoss);
    }

private:
    __aicore__ inline float logSumExp(float a, float b)
    {
        if (a <= NEG_INF + 1.0f && b <= NEG_INF + 1.0f) return NEG_INF;
        float maxVal = (a > b) ? a : b;
        float diff = (a > b) ? (b - a) : (a - b);
        LocalTensor<float> tmp = tmpQue_.AllocTensor<float>();
        tmp.SetValue(0, diff);
        Exp(tmp, tmp, 1);
        tmp.SetValue(0, 1.0f + tmp.GetValue(0));
        Log(tmp, tmp, 1);
        float result = maxVal + tmp.GetValue(0);
        tmpQue_.FreeTensor(tmp);
        return result;
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueLogProbs_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueTargets_;
    TQue<TPosition::VECCALC, 1> alphaPrevQue_;
    TQue<TPosition::VECCALC, 1> alphaCurrQue_;
    TQue<TPosition::VECCALC, 1> tmpQue_;

    GlobalTensor<float> logProbsGm_;
    GlobalTensor<int32_t> targetsGm_;
    GlobalTensor<int32_t> inputLengthsGm_;
    GlobalTensor<int32_t> targetLengthsGm_;
    GlobalTensor<float> lossGm_;

    uint32_t totalBatch_;
    float totalBatchFloat_;
    uint32_t seqLength_;
    uint32_t numClasses_;
    uint32_t maxTargetLength_;
    uint32_t batchStride_; // = totalBatch * numClasses
    int32_t blank_;
    int32_t reduction_;
};

extern "C" __global__ __aicore__ void ca6k02718_custom(GM_ADDR log_probs, GM_ADDR targets,
                                                         GM_ADDR input_lengths, GM_ADDR target_lengths,
                                                         GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02718Custom op;
    op.Init(log_probs, targets, input_lengths, target_lengths, loss,
            tiling_data.totalBatch,
            tiling_data.seqLength, tiling_data.numClasses,
            tiling_data.maxTargetLength,
            tiling_data.totalBatchFloat,
            tiling_data.blank, tiling_data.reduction);
    op.Process();
}
