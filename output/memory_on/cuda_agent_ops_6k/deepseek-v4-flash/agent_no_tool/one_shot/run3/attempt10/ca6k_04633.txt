kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04633Custom {
public:
    __aicore__ inline KernelCa6k04633Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR values, GM_ADDR indices,
                                uint32_t totalLength, uint32_t innerSize,
                                uint32_t outerSize, uint32_t topK,
                                uint32_t tileLength, uint32_t blockDim)
    {
        totalLen_ = totalLength;
        innerSize_ = innerSize;
        outerSize_ = outerSize;
        topK_ = topK;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        negInf_ = -1.0e38f;  // approximate -inf

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        vGm_.SetGlobalBuffer((__gm__ float*)values, outerSize_ * topK_);
        iGm_.SetGlobalBuffer((__gm__ int64_t*)indices, outerSize_ * topK_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueV_, BUFFER_NUM, topK_ * sizeof(float));
        pipe_.InitBuffer(outQueueI_, BUFFER_NUM, topK_ * sizeof(int64_t));
        pipe_.InitBuffer(tmpQueue_, 1, (tileLen_ + 1) * sizeof(float));
        pipe_.InitBuffer(idxQueue_, 1, 1 * sizeof(int32_t));
    }

    __aicore__ inline void Process()
    {
        if (outerSize_ == 0 || innerSize_ == 0 || topK_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t outerPerBlock = (outerSize_ + blockNum - 1) / blockNum;
        uint32_t startOuter = blockIdx * outerPerBlock;
        uint32_t endOuter = startOuter + outerPerBlock;
        if (endOuter > outerSize_) endOuter = outerSize_;

        for (uint32_t o = startOuter; o < endOuter; ++o) {
            // Load one inner slice
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[o * innerSize_], innerSize_);

            AscendC::LocalTensor<float> valLocal = outQueueV_.AllocTensor<float>();
            AscendC::LocalTensor<int64_t> idxLocal = outQueueI_.AllocTensor<int64_t>();
            AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<int32_t> idxBuf = idxQueue_.AllocTensor<int32_t>();

            for (uint32_t k = 0; k < topK_; ++k) {
                // Find argmax in xLocal
                AscendC::ArgMax(idxBuf, xLocal, innerSize_);
                int32_t maxIdx = AscendC::GetValue(idxBuf, 0);
                float maxVal = AscendC::GetValue(xLocal, maxIdx);

                AscendC::SetValue(valLocal, k, maxVal);
                AscendC::SetValue(idxLocal, k, static_cast<int64_t>(maxIdx));

                // Mask found max to -inf
                AscendC::SetValue(xLocal, maxIdx, negInf_);
            }

            // Write out
            AscendC::DataCopy(vGm_[o * topK_], valLocal, topK_);
            AscendC::DataCopy(iGm_[o * topK_], idxLocal, topK_);

            idxQueue_.FreeTensor(idxBuf);
            tmpQueue_.FreeTensor(tmpLocal);
            outQueueI_.FreeTensor(idxLocal);
            outQueueV_.FreeTensor(valLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueV_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueI_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> idxQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> vGm_;
    AscendC::GlobalTensor<int64_t> iGm_;

    uint32_t totalLen_ {0};
    uint32_t innerSize_ {0};
    uint32_t outerSize_ {0};
    uint32_t topK_ {16};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
    float negInf_ {-1.0e38f};
};

extern "C" __global__ __aicore__ void ca6k_04633_custom(GM_ADDR x, GM_ADDR values, GM_ADDR indices, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04633Custom op;
    op.Init(x, values, indices,
            tiling_data.totalLength, tiling_data.innerSize,
            tiling_data.outerSize, tiling_data.topK,
            tiling_data.tileLength, tiling_data.blockDim);
    op.Process();
}
