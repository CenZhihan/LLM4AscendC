#include "ca6k00333_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00333CustomTilingData tiling;
    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &originShape = storageShape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(3));
    uint32_t totalSpatial = N * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *sizePtr = attrs->GetAttrPointer<int32_t>(0);
    const float *alphaPtr = attrs->GetAttrPointer<float>(1);
    const float *betaPtr = attrs->GetAttrPointer<float>(2);
    const float *kPtr = attrs->GetAttrPointer<float>(3);

    uint32_t maxTileSpatial = UB_BUDGET_BYTES / ((C + 1) * sizeof(float));
    if (maxTileSpatial < 1) maxTileSpatial = 1;
    uint32_t tileSpatial = AlignDown(maxTileSpatial, ALIGN_ELEMS);
    if (tileSpatial < ALIGN_ELEMS) tileSpatial = ALIGN_ELEMS;
    if (tileSpatial > totalSpatial) tileSpatial = AlignDown(totalSpatial, ALIGN_ELEMS);
    if (tileSpatial == 0) tileSpatial = totalSpatial;

    uint32_t numTiles = CeilDiv(totalSpatial, tileSpatial);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalSpatial(totalSpatial);
    tiling.set_tileSpatial(tileSpatial);
    tiling.set_blockDim(blockDim);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_size(*sizePtr);
    tiling.set_alpha(*alphaPtr);
    tiling.set_beta(*betaPtr);
    tiling.set_k(*kPtr);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

}

kernel_src = '''
#include "kernel_operator.h"
#include "ca6k00333_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00333Custom {
public:
    __aicore__ inline KernelCa6k00333Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalSpatial, uint32_t tileSpatial,
                                uint32_t blockDim, uint32_t C,
                                uint32_t H, uint32_t W,
                                int32_t size, float alpha, float beta, float k) {
        totalSpatial_ = totalSpatial;
        tileSpatial_ = tileSpatial;
        C_ = C;
        H_ = H;
        W_ = W;
        halfSize_ = size / 2;
        alpha_ = alpha;
        beta_ = beta;
        k_ = k;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalSpatial_ * C_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalSpatial_ * C_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileSpatial_ * C_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileSpatial_ * C_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileSpatial_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalSpatial_ == 0 || tileSpatial_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t strideSpatial = blockNum * tileSpatial_;
        uint32_t baseSpatial = blockIdx * tileSpatial_;
        if (baseSpatial >= totalSpatial_) return;

        while (true) {
            uint32_t curSpatial = totalSpatial_ - baseSpatial;
            if (curSpatial > tileSpatial_) curSpatial = tileSpatial_;

            PrefetchTile(baseSpatial, curSpatial);
            ComputeTile(baseSpatial, curSpatial);
            WriteTile(baseSpatial, curSpatial);

            baseSpatial += strideSpatial;
            if (baseSpatial >= totalSpatial_) break;
        }
    }

private:
    __aicore__ inline void PrefetchTile(uint32_t baseSpatial, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        for (uint32_t i = 0; i < len; i++) {
            uint32_t spatialIdx = baseSpatial + i;
            uint32_t n = spatialIdx / (H_ * W_);
            uint32_t rem = spatialIdx % (H_ * W_);
            uint32_t h = rem / W_;
            uint32_t w = rem % W_;
            for (uint32_t c = 0; c < C_; c++) {
                uint32_t globalOffset = n * C_ * H_ * W_ + c * H_ * W_ + h * W_ + w;
                xLocal(i * C_ + c) = xGm_(globalOffset);
            }
        }
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeTile(uint32_t baseSpatial, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumSq = tmpQueue_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; i++) {
            for (uint32_t c = 0; c < C_; c++) {
                float sum = 0.0f;
                int32_t start = (c >= halfSize_) ? (c - halfSize_) : 0;
                int32_t end = (c + halfSize_ < C_) ? (c + halfSize_) : (C_ - 1);
                for (int32_t j = start; j <= end; j++) {
                    float val = xLocal(i * C_ + j);
                    sum += val * val;
                }
                float denom = k_ + alpha_ * sum;
                // compute denom^{-beta} using exp(-beta * log(denom))
                float scale = expf(-beta_ * logf(denom));
                yLocal(i * C_ + c) = xLocal(i * C_ + c) * scale;
            }
        }

        tmpQueue_.FreeTensor(sumSq);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void WriteTile(uint32_t baseSpatial, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        for (uint32_t i = 0; i < len; i++) {
            uint32_t spatialIdx = baseSpatial + i;
            uint32_t n = spatialIdx / (H_ * W_);
            uint32_t rem = spatialIdx % (H_ * W_);
            uint32_t h = rem / W_;
            uint32_t w = rem % W_;
            for (uint32_t c = 0; c < C_; c++) {
                uint32_t globalOffset = n * C_ * H_ * W_ + c * H_ * W_ + h * W_ + w;
                yGm_(globalOffset) = yLocal(i * C_ + c);
            }
        }
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalSpatial_ {0};
    uint32_t tileSpatial_ {0};
    uint32_t C_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t halfSize_ {0};
    float alpha_ {1e-4f};
    float beta_ {0.75f};
    float k_ {1.0f};
};

extern "C" __global__ __aicore__ void ca6k_00333_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00333Custom op;
    op.Init(x, y,
            tiling_data.totalSpatial, tiling_data.tileSpatial,
            tiling_data.blockDim, tiling_data.C,
            tiling_data.H, tiling_data.W,
            tiling_data.size, tiling_data.alpha, tiling_data.beta, tiling_data.k);
    op.Process();
}
'''
