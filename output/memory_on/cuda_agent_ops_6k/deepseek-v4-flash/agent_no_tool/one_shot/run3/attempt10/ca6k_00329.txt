#include "ca6k00329_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00329CustomTilingData tiling;
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const auto &origin_shape = x_shape->GetOriginShape();
    uint32_t dimNum = origin_shape.GetDimNum();
    uint32_t batch = 1, channels = 1, height = 1, width = 1;
    // Parse NHWC? Actually input is NCHW (batch, channels, H, W)
    if (dimNum == 4) {
        batch = origin_shape.GetDim(0);
        channels = origin_shape.GetDim(1);
        height = origin_shape.GetDim(2);
        width = origin_shape.GetDim(3);
    } else {
        return ge::GRAPH_FAILED;
    }

    uint32_t totalLength = batch * channels * height * width;
    uint32_t spatialSize = height * width;

    // Tile length: we will process one (batch, channel) slice at a time
    // The spatial dimension fits in one tile for each batch/channel.
    uint32_t maxTile = spatialSize;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = spatialSize;
    // Ensure we don't exceed UB budget: we need 2 * tileLen for input/output + a small buffer
    uint32_t ubElements = UB_BUDGET_BYTES / sizeof(float);
    if (maxTile * 2 > ubElements) {
        maxTile = AlignDown(ubElements / 2, ALIGN_ELEMS);
        if (maxTile == 0) maxTile = ALIGN_ELEMS;
    }
    uint32_t tileLen = maxTile;

    // Block dim: each block handles one channel (parallel over channels)
    uint32_t blockDim = channels;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_spatialSize(spatialSize);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00329Custom {
public:
    __aicore__ inline KernelCa6k00329Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batch, uint32_t channels,
                                uint32_t spatialSize, float eps)
    {
        batch_ = batch;
        channels_ = channels;
        spatialSize_ = spatialSize;
        tileLen_ = tileLength;
        eps_ = eps;
        totalLen_ = totalLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Each tile processes one (batch, channel) slice of spatialSize elements
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, spatialSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, spatialSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, spatialSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || spatialSize_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= channels_) return;

        // Each block handles one channel c
        uint32_t c = blockIdx;
        uint32_t stride = channels_ * spatialSize_;  // per batch stride

        // ---- first pass: compute sum and sumSq over all batches for this channel ----
        float sum = 0.0f;
        float sumSq = 0.0f;

        for (uint32_t b = 0; b < batch_; b++) {
            uint32_t base = b * stride + c * spatialSize_;
            uint32_t len = spatialSize_;
            // Load a tile (entire spatial slice for this batch and channel)
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], len);
            // Scalar reduction over the local tensor (simple but correct)
            for (uint32_t i = 0; i < len; i++) {
                float val = xLocal.GetValue(i);
                sum += val;
                sumSq += val * val;
            }
            inQueueX_.FreeTensor(xLocal);
        }

        uint32_t elementCount = batch_ * spatialSize_;
        float mean = sum / (float)elementCount;
        float var = sumSq / (float)elementCount - mean * mean;
        float invStd = 1.0f / sqrt(var + eps_);

        // ---- second pass: normalize ----
        for (uint32_t b = 0; b < batch_; b++) {
            uint32_t base = b * stride + c * spatialSize_;
            uint32_t len = spatialSize_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            AscendC::DataCopy(xLocal, xGm_[base], len);
            // y = (x - mean) * invStd
            AscendC::Adds(tLocal, xLocal, -mean, len);
            AscendC::Muls(yLocal, tLocal, invStd, len);

            AscendC::DataCopy(yGm_[base], yLocal, len);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            tmpQueue_.FreeTensor(tLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_ {0};
    uint32_t channels_ {0};
    uint32_t spatialSize_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
    uint32_t totalLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k00329_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00329Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.batch, tiling_data.channels,
            tiling_data.spatialSize, tiling_data.eps);
    op.Process();
}
