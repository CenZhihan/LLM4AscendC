kernel_src = """
#include "kernel_operator.h"
#include "ca6k_00414_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float BICUBIC_A = -0.75f;

class KernelCa6k00414Custom {
public:
    __aicore__ inline KernelCa6k00414Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                const Ca6k00414CustomTilingData &tilingData) {
        n_ = tilingData.get_n();
        c_ = tilingData.get_c();
        h_ = tilingData.get_h();
        w_ = tilingData.get_w();
        hOut_ = tilingData.get_hOut();
        wOut_ = tilingData.get_wOut();
        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * c_ * h_ * w_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * c_ * hOut_ * wOut_);
        totalOut_ = n_ * c_ * hOut_ * wOut_;
    }

    __aicore__ inline void Process() {
        if (totalOut_ == 0) return;
        uint32_t blkIdx = AscendC::GetBlockIdx();
        uint32_t blkNum = AscendC::GetBlockNum();
        for (uint32_t idx = blkIdx; idx < totalOut_; idx += blkNum) {
            uint32_t tmp = idx;
            uint32_t wIdx = tmp % wOut_; tmp /= wOut_;
            uint32_t hIdx = tmp % hOut_; tmp /= hOut_;
            uint32_t cIdx = tmp % c_; 
            uint32_t bIdx = tmp / c_;

            float xF = static_cast<float>(wIdx) / 2.0f;
            float yF = static_cast<float>(hIdx) / 2.0f;
            int ix = static_cast<int>(xF);
            int iy = static_cast<int>(yF);
            float du = xF - static_cast<float>(ix);
            float dv = yF - static_cast<float>(iy);

            float wx[4], wy[4];
            for (int k = 0; k < 4; ++k) {
                float u = static_cast<float>(k - 1) - du;
                float v = static_cast<float>(k - 1) - dv;
                wx[k] = bicubicWeight(u);
                wy[k] = bicubicWeight(v);
            }

            float sum = 0.0f;
            for (int j = 0; j < 4; ++j) {
                int row = iy + j - 1;
                if (row < 0) row = 0;
                if (row >= static_cast<int>(h_)) row = static_cast<int>(h_) - 1;
                float rowSum = 0.0f;
                for (int k = 0; k < 4; ++k) {
                    int col = ix + k - 1;
                    if (col < 0) col = 0;
                    if (col >= static_cast<int>(w_)) col = static_cast<int>(w_) - 1;
                    uint32_t gIdx = (bIdx * c_ + cIdx) * h_ * w_ + static_cast<uint32_t>(row) * w_ + static_cast<uint32_t>(col);
                    rowSum += xGm_.GetValue(gIdx) * wx[k];
                }
                sum += rowSum * wy[j];
            }

            yGm_.SetValue(idx, sum);
        }
    }

private:
    __aicore__ inline float bicubicWeight(float x) {
        float absX = x;
        if (absX < 0.0f) absX = -absX;
        if (absX <= 1.0f) {
            return (BICUBIC_A + 2.0f) * absX * absX * absX
                - (BICUBIC_A + 3.0f) * absX * absX + 1.0f;
        } else if (absX < 2.0f) {
            return BICUBIC_A * absX * absX * absX
                - 5.0f * BICUBIC_A * absX * absX
                + 8.0f * BICUBIC_A * absX - 4.0f * BICUBIC_A;
        } else {
            return 0.0f;
        }
    }

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t n_, c_, h_, w_, hOut_, wOut_;
    uint32_t totalOut_;
};

extern "C" __global__ __aicore__ void ca6k_00414_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00414Custom op;
    op.Init(x, y, tiling_data);
    op.Process();
}
"""
