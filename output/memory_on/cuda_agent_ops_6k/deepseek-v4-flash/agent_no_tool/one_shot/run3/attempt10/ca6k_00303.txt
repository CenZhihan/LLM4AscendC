static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00303CustomTiling tiling;
    const gert::StorageShape *storage = context->GetInputShape(0);
    const gert::Shape &in_shape = storage->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(in_shape.GetDim(0));
    uint32_t C_in = static_cast<uint32_t>(in_shape.GetDim(1));
    uint32_t H_in = static_cast<uint32_t>(in_shape.GetDim(2));
    uint32_t W_in = static_cast<uint32_t>(in_shape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t scale_factor = attrs->GetAttrPointer<int32_t>(0)[0];
    uint32_t scale = static_cast<uint32_t>(scale_factor);

    uint32_t C_out = C_in / (scale * scale);
    uint32_t H_out = H_in * scale;
    uint32_t W_out = W_in * scale;
    uint32_t totalElemsPerBatch = C_out * H_out * W_out;

    context->SetBlockDim(N);

    tiling.set_N(N);
    tiling.set_C_in(C_in);
    tiling.set_H_in(H_in);
    tiling.set_W_in(W_in);
    tiling.set_scaleFactor(scale);
    tiling.set_C_out(C_out);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_totalElemsPerBatch(totalElemsPerBatch);
    tiling.set_blockDim(N);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelPixelShuffle {
public:
    __aicore__ inline KernelPixelShuffle() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C_in, uint32_t H_in, uint32_t W_in,
                                uint32_t scaleFactor,
                                uint32_t C_out, uint32_t H_out, uint32_t W_out,
                                uint32_t totalElemsPerBatch)
    {
        N_ = N;
        C_in_ = C_in;
        H_in_ = H_in;
        W_in_ = W_in;
        scale_ = scaleFactor;
        C_out_ = C_out;
        H_out_ = H_out;
        W_out_ = W_out;
        totalElemsPerBatch_ = totalElemsPerBatch;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Allocate a single-element local buffer for scalar transfers.
        pipe_.InitBuffer(tmpElemBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= N_) return;

        uint32_t n = blockIdx;
        uint32_t batchOffIn = n * C_in_ * H_in_ * W_in_;
        uint32_t batchOffOut = n * C_out_ * H_out_ * W_out_;

        // Iterate over all output elements for this batch.
        for (uint32_t linearOut = 0; linearOut < totalElemsPerBatch_; ++linearOut) {
            // Decode (c_out, h_out, w_out)
            uint32_t c_out = linearOut / (H_out_ * W_out_);
            uint32_t rest = linearOut % (H_out_ * W_out_);
            uint32_t h_out = rest / W_out_;
            uint32_t w_out = rest % W_out_;

            // Compute input indices
            uint32_t c_in_base = c_out * scale_ * scale_;
            uint32_t h_in = h_out / scale_;
            uint32_t w_in = w_out / scale_;
            uint32_t r = h_out % scale_;
            uint32_t c = w_out % scale_;
            uint32_t c_in = c_in_base + r * scale_ + c;

            uint32_t inputIdx = batchOffIn + (c_in * H_in_ + h_in) * W_in_ + w_in;
            uint32_t outputIdx = batchOffOut + linearOut;

            // Copy single element via local buffer
            AscendC::LocalTensor<float> tmpElem = tmpElemBuf_.Get<float>();
            AscendC::DataCopy(tmpElem, xGm_[inputIdx], 1);
            AscendC::DataCopy(yGm_[outputIdx], tmpElem, 1);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpElemBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_in_ {0};
    uint32_t H_in_ {0};
    uint32_t W_in_ {0};
    uint32_t scale_ {2};
    uint32_t C_out_ {0};
    uint32_t H_out_ {0};
    uint32_t W_out_ {0};
    uint32_t totalElemsPerBatch_ {0};
};

extern "C" __global__ __aicore__ void ca6k00303_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelPixelShuffle op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C_in, tiling_data.H_in, tiling_data.W_in,
            tiling_data.scaleFactor,
            tiling_data.C_out, tiling_data.H_out, tiling_data.W_out,
            tiling_data.totalElemsPerBatch);
    op.Process();
}
'''
