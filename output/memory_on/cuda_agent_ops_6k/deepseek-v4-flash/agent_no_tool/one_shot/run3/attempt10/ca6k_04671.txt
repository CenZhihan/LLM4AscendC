#include "ca6k04671_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04671CustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const auto &originShape = xShape->GetOriginShape();
    const uint32_t dims[4] = {
        static_cast<uint32_t>(originShape.GetDim(0)),
        static_cast<uint32_t>(originShape.GetDim(1)),
        static_cast<uint32_t>(originShape.GetDim(2)),
        static_cast<uint32_t>(originShape.GetDim(3))
    };
    uint32_t totalLength = dims[0] * dims[1] * dims[2] * dims[3];
    uint32_t batchSize = dims[0];
    uint32_t channels = dims[1];
    uint32_t spatialSize = dims[2] * dims[3];

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = attrs->GetAttrPointer<float>(0) ? *attrs->GetAttrPointer<float>(0) : 1e-5f;
    float momentum = attrs->GetAttrPointer<float>(1) ? *attrs->GetAttrPointer<float>(1) : 0.1f;
    bool training = attrs->GetAttrPointer<bool>(2) ? *attrs->GetAttrPointer<bool>(2) : false;

    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * 3);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;
    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batchSize(batchSize);
    tiling.set_channels(channels);
    tiling.set_spatialSize(spatialSize);
    tiling.set_eps(eps);
    tiling.set_momentum(momentum);
    tiling.set_training(training);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04671Custom {
public:
    __aicore__ inline KernelCa6k04671Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                GM_ADDR running_mean, GM_ADDR running_var,
                                GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t channels,
                                uint32_t spatialSize, float eps, float momentum,
                                bool training)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        channels_ = channels;
        spatialSize_ = spatialSize;
        eps_ = eps;
        momentum_ = momentum;
        training_ = training;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        meanGm_.SetGlobalBuffer((__gm__ float*)running_mean, channels_);
        varGm_.SetGlobalBuffer((__gm__ float*)running_var, channels_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, channels_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, channels_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = (totalLen_ - base) > tileLen_ ? tileLen_ : (totalLen_ - base);
            CopyIn(base, curLen);
            Compute(base, curLen);
            CopyOut(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t base, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; i++) {
            uint32_t globalIdx = base + i;
            uint32_t batch = globalIdx / (channels_ * spatialSize_);
            uint32_t channel = (globalIdx % (channels_ * spatialSize_)) / spatialSize_;
            float mean = meanGm_(channel);
            float invStd = 1.0f / sqrt(varGm_(channel) + eps_);
            float w = weightGm_(channel);
            float b = biasGm_(channel);
            float xVal = xLocal(i);
            yLocal(i) = w * (xVal - mean) * invStd + b;
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> meanGm_;
    AscendC::GlobalTensor<float> varGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t channels_ {0};
    uint32_t spatialSize_ {0};
    float eps_ {1e-5f};
    float momentum_ {0.1f};
    bool training_ {false};
};

extern "C" __global__ __aicore__ void ca6k04671_custom(GM_ADDR x, GM_ADDR y,
    GM_ADDR running_mean, GM_ADDR running_var, GM_ADDR weight, GM_ADDR bias,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04671Custom op;
    op.Init(x, y, running_mean, running_var, weight, bias,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.batchSize, tiling_data.channels,
            tiling_data.spatialSize, tiling_data.eps, tiling_data.momentum,
            tiling_data.training);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k04671_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_features, eps=1e-05, momentum=0.1, affine=True, track_running_stats=True):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.eps = eps
        self.momentum = momentum
        self.affine = affine
        self.track_running_stats = track_running_stats
        self.register_buffer('running_mean', torch.zeros(num_features))
        self.register_buffer('running_var', torch.ones(num_features))
        if affine:
            self.weight = torch.nn.Parameter(torch.ones(num_features))
            self.bias = torch.nn.Parameter(torch.zeros(num_features))
        else:
            self.register_buffer('weight', torch.ones(num_features))
            self.register_buffer('bias', torch.zeros(num_features))

    def forward(self, x, training=False, momentum=None):
        fn = getattr(custom_ops_lib, _FN)
        mom = momentum if momentum is not None else self.momentum
        return fn(x, self.running_mean, self.running_var, self.weight, self.bias,
                  self.eps, mom, training)
