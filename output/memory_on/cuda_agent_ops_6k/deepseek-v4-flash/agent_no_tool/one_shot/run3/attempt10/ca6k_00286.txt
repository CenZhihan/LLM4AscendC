#include "ca6k_00286_custom_tiling.h"

namespace optiling {

static constexpr uint32_t TILE_SIZE = 4096; // output elements per tile

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00286CustomTilingData tiling;
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const auto &origin_shape = x_shape->GetOriginShape();
    const std::vector<int64_t> dims = origin_shape.GetDims();
    uint32_t N = static_cast<uint32_t>(dims[0]);
    uint32_t C = static_cast<uint32_t>(dims[1]);
    uint32_t H = static_cast<uint32_t>(dims[2]);
    uint32_t W = static_cast<uint32_t>(dims[3]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *scale_factor_ptr = attrs->GetAttrPointer<int32_t>(0);
    uint32_t sf = static_cast<uint32_t>(*scale_factor_ptr);

    uint32_t outputH = H * sf;
    uint32_t outputW = W * sf;
    uint32_t totalOutputElems = N * C * outputH * outputW;

    uint32_t tileSize = TILE_SIZE;
    if (tileSize > totalOutputElems) tileSize = totalOutputElems;
    uint32_t blockDim = (totalOutputElems + tileSize - 1) / tileSize;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_scaleFactor(sf);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_tileSize(tileSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    int64_t n = x_shape->GetDim(0);
    int64_t c = x_shape->GetDim(1);
    int64_t h = x_shape->GetDim(2);

kernel_src = """
#include "kernel_operator.h"

class KernelCa6k00286Custom {
public:
    __aicore__ inline KernelCa6k00286Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t scaleFactor,
                                uint32_t outputH, uint32_t outputW,
                                uint32_t totalOutputElems, uint32_t tileSize)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        sf_ = scaleFactor;
        outputH_ = outputH; outputW_ = outputW;
        totalOutputElems_ = totalOutputElems;
        tileSize_ = tileSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * H * W);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElems);
    }

    __aicore__ inline void Process() {
        if (totalOutputElems_ == 0 || tileSize_ == 0) return;

        uint32_t blockDim = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t start = blockIdx * tileSize_;
        if (start >= totalOutputElems_) return;
        uint32_t end = start + tileSize_;
        if (end > totalOutputElems_) end = totalOutputElems_;

        uint32_t inputHW = H_ * W_;
        uint32_t outputHW = outputH_ * outputW_;
        uint32_t CinputHW = C_ * inputHW;
        uint32_t CoutputHW = C_ * outputHW;

        for (uint32_t out_idx = start; out_idx < end; ++out_idx) {
            uint32_t b = out_idx / CoutputHW;
            uint32_t rem1 = out_idx % CoutputHW;
            uint32_t c = rem1 / outputHW;
            uint32_t rem2 = rem1 % outputHW;
            uint32_t h_out = rem2 / outputW_;
            uint32_t w_out = rem2 % outputW_;
            uint32_t h_in = h_out / sf_;
            uint32_t w_in = w_out / sf_;
            uint32_t in_idx = b * CinputHW + c * inputHW + h_in * W_ + w_in;

            float val = xGm_[in_idx];
            yGm_[out_idx] = (val > 0.0f) ? val : 0.0f;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t N_, C_, H_, W_, sf_;
    uint32_t outputH_, outputW_;
    uint32_t totalOutputElems_;
    uint32_t tileSize_;
};

extern "C" __global__ __aicore__ void ca6k_00286_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00286Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.scaleFactor,
            tiling_data.outputH, tiling_data.outputW,
            tiling_data.totalOutputElems, tiling_data.tileSize);
    op.Process();
}
"""
