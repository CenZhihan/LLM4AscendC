#include "ca6k_04587_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04587CustomTilingData tiling;
    auto inputShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t n = static_cast<uint32_t>(inputShape.GetDim(0));
    uint32_t c = static_cast<uint32_t>(inputShape.GetDim(1));
    uint32_t inD = static_cast<uint32_t>(inputShape.GetDim(2));
    uint32_t inH = static_cast<uint32_t>(inputShape.GetDim(3));
    uint32_t inW = static_cast<uint32_t>(inputShape.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(2);
    uint32_t kd = static_cast<uint32_t>(kernel_size[0]);
    uint32_t kh = static_cast<uint32_t>(kernel_size[1]);
    uint32_t kw = static_cast<uint32_t>(kernel_size[2]);
    uint32_t sd = static_cast<uint32_t>(stride[0]);
    uint32_t sh = static_cast<uint32_t>(stride[1]);
    uint32_t sw = static_cast<uint32_t>(stride[2]);
    uint32_t pd = static_cast<uint32_t>(padding[0]);
    uint32_t ph = static_cast<uint32_t>(padding[1]);
    uint32_t pw = static_cast<uint32_t>(padding[2]);

    uint32_t od = (inD + 2 * pd - kd) / sd + 1;
    uint32_t oh = (inH + 2 * ph - kh) / sh + 1;
    uint32_t ow = (inW + 2 * pw - kw) / sw + 1;

    uint32_t totalLength = n * c * od * oh * ow;

    // UB budget per thread: we need to load one output tile and its corresponding input window
    // For simplicity, process 1 output element per iteration with double buffering
    // We'll use 2 buffers for input window (max window size = kd*kh*kw floats)
    // and 2 buffers for output max values.
    // Also need workspace for min/max reduction. Use simple loop.
    // Minimal per-tile elements: at least 1 output element.
    // We'll allocate tileLength output elements; compute required input window area per element.
    // To simplify, process 1 output element per iteration -> tileLength = 1.
    // But that is inefficient; we can process multiple spatial locations in parallel.
    // Considering UB size, we can process up to ~4096 floats (16KB) per iteration.
    // Let tileLength be the number of output elements per iteration.
    uint32_t window_size = kd * kh * kw;
    uint32_t bytes_per_iter = (window_size + 1) * sizeof(float) * 2; // double buffered
    uint32_t max_tile_output = UB_BUDGET_BYTES / bytes_per_iter;
    if (max_tile_output > 1024) max_tile_output = 1024;
    if (max_tile_output < 1) max_tile_output = 1;
    uint32_t tileLength = max_tile_output;
    if (tileLength > totalLength) tileLength = totalLength;

    // blockDim: reasonable parallelism
    uint32_t numTiles = CeilDiv(totalLength, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inD(inD);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_kd(kd);
    tiling.set_kh(kh);
    tiling.set_kw(kw);
    tiling.set_sd(sd);
    tiling.set_sh(sh);
    tiling.set_sw(sw);
    tiling.set_od(od);
    tiling.set_oh(oh);
    tiling.set_ow(ow);
    tiling.set_pad_d(pd);
    tiling.set_pad_h(ph);
    tiling.set_pad_w(pw);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04587Custom {
public:
    __aicore__ inline KernelCa6k04587Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        totalLen_ = td.totalLength;
        tileLen_ = td.tileLength;
        n_ = td.n;
        c_ = td.c;
        inD_ = td.inD;
        inH_ = td.inH;
        inW_ = td.inW;
        kd_ = td.kd;
        kh_ = td.kh;
        kw_ = td.kw;
        sd_ = td.sd;
        sh_ = td.sh;
        sw_ = td.sw;
        od_ = td.od;
        oh_ = td.oh;
        ow_ = td.ow;
        pd_ = td.pad_d;
        ph_ = td.pad_h;
        pw_ = td.pad_w;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // We will use double buffering for output elements (local max values)
        // and a local buffer for the input window.
        // Maximum window size = kd_ * kh_ * kw_
        windowSize_ = kd_ * kh_ * kw_;

        // Allocate local tensors: output buffer for tile (tileLen_ floats)
        // and input buffer for the window (windowSize_ floats)
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        // For input window, we load per output element, so we need only one window buffer.
        pipe_.InitBuffer(winBuffer_, 1, windowSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output elements
        uint32_t start = blockIdx * tileLen_;
        uint32_t end = start + tileLen_;
        if (start >= totalLen_) return;
        if (end > totalLen_) end = totalLen_;

        // Process all elements assigned to this block (by looping over tiles)
        for (uint32_t g = start; g < end; g++) {
            // Compute output indices
            uint32_t linear = g;
            uint32_t ow_ = linear % ow_;
            linear /= ow_;
            uint32_t oh_ = linear % oh_;
            linear /= oh_;
            uint32_t od_ = linear % od_;
            linear /= od_;
            uint32_t c_ = linear % c_;
            uint32_t n_ = linear / c_;

            // Compute input region start
            int64_t d_start = static_cast<int64_t>(od_) * sd_ - static_cast<int64_t>(pd_);
            int64_t h_start = static_cast<int64_t>(oh_) * sh_ - static_cast<int64_t>(ph_);
            int64_t w_start = static_cast<int64_t>(ow_) * sw_ - static_cast<int64_t>(pw_);

            float max_val = -FLT_MAX;

            // Iterate over the window
            for (uint32_t kd = 0; kd < kd_; ++kd) {
                int64_t d_idx = d_start + static_cast<int64_t>(kd);
                if (d_idx < 0 || d_idx >= static_cast<int64_t>(inD_)) continue;
                for (uint32_t kh = 0; kh < kh_; ++kh) {
                    int64_t h_idx = h_start + static_cast<int64_t>(kh);
                    if (h_idx < 0 || h_idx >= static_cast<int64_t>(inH_)) continue;
                    for (uint32_t kw = 0; kw < kw_; ++kw) {
                        int64_t w_idx = w_start + static_cast<int64_t>(kw);
                        if (w_idx < 0 || w_idx >= static_cast<int64_t>(inW_)) continue;
                        // Compute input linear index
                        uint32_t inLinear = ((n_ * c_ + c_) * inD_ + static_cast<uint32_t>(d_idx)) * inH_ * inW_
                                            + static_cast<uint32_t>(h_idx) * inW_
                                            + static_cast<uint32_t>(w_idx);
                        float val = xGm_.GetValue(inLinear);
                        if (val > max_val) max_val = val;
                    }
                }
            }

            // Write max to output
            yGm_.SetValue(g, max_val);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> winBuffer_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t n_ {0}, c_ {0}, inD_ {0}, inH_ {0}, inW_ {0};
    uint32_t kd_ {0}, kh_ {0}, kw_ {0};
    uint32_t sd_ {0}, sh_ {0}, sw_ {0};
    uint32_t od_ {0}, oh_ {0}, ow_ {0};
    uint32_t pd_ {0}, ph_ {0}, pw_ {0};
    uint32_t windowSize_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04587_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelCa6k04587Custom op;
    op.Init(x, y, tiling);
    op.Process();
}
'''

python_bind_src = '''
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor ca6k_04587_impl_npu(const at::Tensor& self,
                                std::vector<int64_t> kernel_size,
                                std::vector<int64_t> stride,
                                std::vector<int64_t> padding) {
    at::Tensor result = at::empty_like(self);
    // The output shape must be computed; for simplicity let the kernel handle it
    // Actually we need to compute output size and allocate accordingly.
    // For max_pool3d, output size is determined by formula.
    int64_t N = self.size(0);
    int64_t C = self.size(1);
    int64_t D = self.size(2);
    int64_t H = self.size(3);
    int64_t W = self.size(4);
    int64_t kd = kernel_size[0];
    int64_t kh = kernel_size[1];
    int64_t kw = kernel_size[2];
    int64_t sd = stride[0];
    int64_t sh = stride[1];
    int64_t sw = stride[2];
    int64_t pd = padding[0];
    int64_t ph = padding[1];
    int64_t pw = padding[2];
    int64_t od = (D + 2 * pd - kd) / sd + 1;
    int64_t oh = (H + 2 * ph - kh) / sh + 1;
    int64_t ow = (W + 2 * pw - kw) / sw + 1;
    result = at::empty({N, C, od, oh, ow}, self.options());
    EXEC_NPU_CMD(aclnnCa6k04587Custom, self, kernel_size, stride, padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_04587_custom", &ca6k_04587_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("ca6k_04587_custom", &ca6k_04587_impl_npu, "MaxPool3d custom (Ca6k04587Custom on NPU)");
}
