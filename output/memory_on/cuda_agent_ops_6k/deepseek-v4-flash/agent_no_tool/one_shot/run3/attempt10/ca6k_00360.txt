kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00360Custom {
public:
    __aicore__ inline KernelCa6k00360Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR output, GM_ADDR h_n,
                                GM_ADDR weight_ih_0, GM_ADDR weight_hh_0,
                                GM_ADDR bias_ih_0, GM_ADDR bias_hh_0,
                                GM_ADDR weight_ih_1, GM_ADDR weight_hh_1,
                                GM_ADDR bias_ih_1, GM_ADDR bias_hh_1,
                                uint32_t totalTimesteps, uint32_t batchSize,
                                uint32_t inputSize, uint32_t hiddenSize,
                                uint32_t numLayers, uint8_t useBias)
    {
        totalTimesteps_ = totalTimesteps;
        batchSize_ = batchSize;
        inputSize_ = inputSize;
        hiddenSize_ = hiddenSize;
        numLayers_ = numLayers;
        useBias_ = useBias != 0;

        uint32_t totalX = totalTimesteps * batchSize * inputSize;
        uint32_t totalOut = totalTimesteps * batchSize * hiddenSize;
        uint32_t totalHN = numLayers * batchSize * hiddenSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalX);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalOut);
        h_nGm_.SetGlobalBuffer((__gm__ float*)h_n, totalHN);

        // Global pointers for weights and biases
        weightIH_[0].SetGlobalBuffer((__gm__ float*)weight_ih_0, hiddenSize * inputSize);
        weightHH_[0].SetGlobalBuffer((__gm__ float*)weight_hh_0, hiddenSize * hiddenSize);
        if (useBias_) {
            biasIH_[0].SetGlobalBuffer((__gm__ float*)bias_ih_0, hiddenSize);
            biasHH_[0].SetGlobalBuffer((__gm__ float*)bias_hh_0, hiddenSize);
        }
        weightIH_[1].SetGlobalBuffer((__gm__ float*)weight_ih_1, hiddenSize * inputSize);
        weightHH_[1].SetGlobalBuffer((__gm__ float*)weight_hh_1, hiddenSize * hiddenSize);
        if (useBias_) {
            biasIH_[1].SetGlobalBuffer((__gm__ float*)bias_ih_1, hiddenSize);
            biasHH_[1].SetGlobalBuffer((__gm__ float*)bias_hh_1, hiddenSize);
        }

        // Allocate UB buffers
        // Weights for each layer (loaded once)
        pipe_.InitBuffer(wIH_local_, 2, hiddenSize_ * inputSize_ * sizeof(float));
        pipe_.InitBuffer(wHH_local_, 2, hiddenSize_ * hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(biasIH_local_, 2, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(biasHH_local_, 2, hiddenSize_ * sizeof(float));

        // Load weights into UB
        for (uint32_t layer = 0; layer < 2; layer++) {
            float* wIH_ptr = (layer == 0) ? wIH_local_.Get<float>() : wIH_local_.Get<float>() + hiddenSize_ * inputSize_;
            float* wHH_ptr = (layer == 0) ? wHH_local_.Get<float>() : wHH_local_.Get<float>() + hiddenSize_ * hiddenSize_;
            float* bIH_ptr = (layer == 0) ? biasIH_local_.Get<float>() : biasIH_local_.Get<float>() + hiddenSize_;
            float* bHH_ptr = (layer == 0) ? biasHH_local_.Get<float>() : biasHH_local_.Get<float>() + hiddenSize_;
            AscendC::LocalTensor<float> wIH_tmp = (layer == 0) ? wIH_local_.Get<float>() : wIH_local_.Get<float>() + hiddenSize_ * inputSize_;
            AscendC::LocalTensor<float> wHH_tmp = (layer == 0) ? wHH_local_.Get<float>() : wHH_local_.Get<float>() + hiddenSize_ * hiddenSize_;
            AscendC::LocalTensor<float> bIH_tmp = (layer == 0) ? biasIH_local_.Get<float>() : biasIH_local_.Get<float>() + hiddenSize_;
            AscendC::LocalTensor<float> bHH_tmp = (layer == 0) ? biasHH_local_.Get<float>() : biasHH_local_.Get<float>() + hiddenSize_;
            AscendC::DataCopy(wIH_tmp, weightIH_[layer], hiddenSize_ * inputSize_);
            AscendC::DataCopy(wHH_tmp, weightHH_[layer], hiddenSize_ * hiddenSize_);
            if (useBias_) {
                AscendC::DataCopy(bIH_tmp, biasIH_[layer], hiddenSize_);
                AscendC::DataCopy(bHH_tmp, biasHH_[layer], hiddenSize_);
            } else {
                AscendC::SetZero(bIH_tmp, hiddenSize_);
                AscendC::SetZero(bHH_tmp, hiddenSize_);
            }
        }

        // Per-sample buffers
        pipe_.InitBuffer(xVec_, 1, inputSize_ * sizeof(float));
        pipe_.InitBuffer(hPrev0_all_, 1, batchSize_ * hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(hPrev1_all_, 1, batchSize_ * hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(hCurr0_, 1, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(hCurr1_, 1, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(linVec_, 1, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(tempVec_, 1, (inputSize_ > hiddenSize_ ? inputSize_ : hiddenSize_) * sizeof(float));
        pipe_.InitBuffer(weightCol_, 1, (inputSize_ > hiddenSize_ ? inputSize_ : hiddenSize_) * sizeof(float));
        pipe_.InitBuffer(dotScalar_, 1, sizeof(float));

        // Initialize hidden states to zero
        AscendC::SetZero(hPrev0_all_.Get<float>(), batchSize_ * hiddenSize_);
        AscendC::SetZero(hPrev1_all_.Get<float>(), batchSize_ * hiddenSize_);
    }

    __aicore__ inline void Process()
    {
        if (totalTimesteps_ == 0 || batchSize_ == 0 || hiddenSize_ == 0) return;

        float* wIH0 = wIH_local_.Get<float>();
        float* wHH0 = wHH_local_.Get<float>() + 0 * hiddenSize_ * hiddenSize_;
        float* bIH0 = biasIH_local_.Get<float>();
        float* bHH0 = biasHH_local_.Get<float>() + 0 * hiddenSize_;
        float* wIH1 = wIH_local_.Get<float>() + 1 * hiddenSize_ * inputSize_;
        float* wHH1 = wHH_local_.Get<float>() + 1 * hiddenSize_ * hiddenSize_;
        float* bIH1 = biasIH_local_.Get<float>() + 1 * hiddenSize_;
        float* bHH1 = biasHH_local_.Get<float>() + 1 * hiddenSize_;

        float* hPrev0_all = hPrev0_all_.Get<float>();
        float* hPrev1_all = hPrev1_all_.Get<float>();
        float* hCurr0 = hCurr0_.Get<float>();
        float* hCurr1 = hCurr1_.Get<float>();
        float* linVec = linVec_.Get<float>();
        float* tempBuf = tempVec_.Get<float>();
        float* weightCol = weightCol_.Get<float>();
        float* dotScalar = dotScalar_.Get<float>();

        for (uint32_t t = 0; t < totalTimesteps_; t++) {
            uint32_t outBase = t * batchSize_ * hiddenSize_;
            for (uint32_t b = 0; b < batchSize_; b++) {
                // Load input x[t,b]
                uint32_t xBase = (t * batchSize_ + b) * inputSize_;
                AscendC::DataCopy(xVec_.Get<float>(), xGm_[xBase], inputSize_);

                // Layer 0: lin = x * W_ih0 + h_prev0 * W_hh0 + bias_ih0 + bias_hh0
                // Matrix-vector multiplication: compute each output element
                for (uint32_t j = 0; j < hiddenSize_; j++) {
                    // Load column j of weight_ih0 (size inputSize)
                    for (uint32_t k = 0; k < inputSize_; k++) {
                        weightCol[k] = wIH0[j * inputSize_ + k];
                    }
                    AscendC::Mul(tempBuf, xVec_.Get<float>(), weightCol, inputSize_);
                    float sum = 0.0f;
                    for (uint32_t k = 0; k < inputSize_; k++) {
                        sum += tempBuf[k];
                    }
                    linVec[j] = sum;
                }
                // Add contribution from previous hidden state (size hiddenSize)
                float* hPrev_b = hPrev0_all + b * hiddenSize_;
                for (uint32_t j = 0; j < hiddenSize_; j++) {
                    // Load column j of weight_hh0 (size hiddenSize)
                    for (uint32_t k = 0; k < hiddenSize_; k++) {
                        weightCol[k] = wHH0[j * hiddenSize_ + k];
                    }
                    AscendC::Mul(tempBuf, hPrev_b, weightCol, hiddenSize_);
                    float sum = 0.0f;
                    for (uint32_t k = 0; k < hiddenSize_; k++) {
                        sum += tempBuf[k];
                    }
                    linVec[j] += sum;
                }
                // Add biases
                for (uint32_t j = 0; j < hiddenSize_; j++) {
                    linVec[j] += bIH0[j] + bHH0[j];
                }
                // Tanh activation
                AscendC::Tanh(hCurr0, linVec, hiddenSize_);

                // Layer 1: input = hCurr0
                for (uint32_t j = 0; j < hiddenSize_; j++) {
                    // Load column j of weight_ih1 (size hiddenSize)
                    for (uint32_t k = 0; k < hiddenSize_; k++) {
                        weightCol[k] = wIH1[j * hiddenSize_ + k];
                    }
                    AscendC::Mul(tempBuf, hCurr0, weightCol, hiddenSize_);
                    float sum = 0.0f;
                    for (uint32_t k = 0; k < hiddenSize_; k++) {
                        sum += tempBuf[k];
                    }
                    linVec[j] = sum;
                }
                // Add hPrev1 * weight_hh1
                float* hPrev1_b = hPrev1_all + b * hiddenSize_;
                for (uint32_t j = 0; j < hiddenSize_; j++) {
                    for (uint32_t k = 0; k < hiddenSize_; k++) {
                        weightCol[k] = wHH1[j * hiddenSize_ + k];
                    }
                    AscendC::Mul(tempBuf, hPrev1_b, weightCol, hiddenSize_);
                    float sum = 0.0f;
                    for (uint32_t k = 0; k < hiddenSize_; k++) {
                        sum += tempBuf[k];
                    }
                    linVec[j] += sum;
                }
                // Add biases
                for (uint32_t j = 0; j < hiddenSize_; j++) {
                    linVec[j] += bIH1[j] + bHH1[j];
                }
                // Tanh
                AscendC::Tanh(hCurr1, linVec, hiddenSize_);

                // Write output[t,b] = hCurr1
                for (uint32_t j = 0; j < hiddenSize_; j++) {
                    outputGm_[outBase + b * hiddenSize_ + j] = hCurr1[j];
                }

                // Update hidden states for next timestep
                for (uint32_t j = 0; j < hiddenSize_; j++) {
                    hPrev0_all[b * hiddenSize_ + j] = hCurr0[j];
                    hPrev1_all[b * hiddenSize_ + j] = hCurr1[j];
                }
            }
        }

        // Write h_n: final hidden state for each layer
        // h_n shape: [numLayers, batchSize, hiddenSize]
        for (uint32_t layer = 0; layer < 2; layer++) {
            float* src = (layer == 0) ? hPrev0_all_.Get<float>() : hPrev1_all_.Get<float>();
            for (uint32_t b = 0; b < batchSize_; b++) {
                for (uint32_t j = 0; j < hiddenSize_; j++) {
                    h_nGm_[layer * batchSize_ * hiddenSize_ + b * hiddenSize_ + j] = src[b * hiddenSize_ + j];
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> wIH_local_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> wHH_local_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> biasIH_local_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> biasHH_local_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> xVec_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> hPrev0_all_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> hPrev1_all_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> hCurr0_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> hCurr1_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> linVec_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tempVec_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> weightCol_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> dotScalar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> outputGm_;
    AscendC::GlobalTensor<float> h_nGm_;
    AscendC::GlobalTensor<float> weightIH_[2];
    AscendC::GlobalTensor<float> weightHH_[2];
    AscendC::GlobalTensor<float> biasIH_[2];
    AscendC::GlobalTensor<float> biasHH_[2];

    uint32_t totalTimesteps_ = 0;
    uint32_t batchSize_ = 0;
    uint32_t inputSize_ = 0;
    uint32_t hiddenSize_ = 0;
    uint32_t numLayers_ = 0;
    bool useBias_ = false;
};

extern "C" __global__ __aicore__ void ca6k_00360_custom(
    GM_ADDR x, GM_ADDR output, GM_ADDR h_n,
    GM_ADDR weight_ih_0, GM_ADDR weight_hh_0,
    GM_ADDR bias_ih_0, GM_ADDR bias_hh_0,

model_src = '''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, bias=True):
        super().__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.bias = bias
        ref_rnn = torch.nn.RNN(input_size, hidden_size, num_layers, bias=bias)
        self.weight_ih_0 = torch.nn.Parameter(ref_rnn.weight_ih_l0.clone())
        self.weight_hh_0 = torch.nn.Parameter(ref_rnn.weight_hh_l0.clone())
        self.weight_ih_1 = torch.nn.Parameter(ref_rnn.weight_ih_l1.clone())
        self.weight_hh_1 = torch.nn.Parameter(ref_rnn.weight_hh_l1.clone())
        if bias:
            self.bias_ih_0 = torch.nn.Parameter(ref_rnn.bias_ih_l0.clone())
            self.bias_hh_0 = torch.nn.Parameter(ref_rnn.bias_hh_l0.clone())
            self.bias_ih_1 = torch.nn.Parameter(ref_rnn.bias_ih_l1.clone())
            self.bias_hh_1 = torch.nn.Parameter(ref_rnn.bias_hh_l1.clone())
        else:
            self.bias_ih_0 = torch.nn.Parameter(torch.zeros(hidden_size, dtype=torch.float32))
            self.bias_hh_0 = torch.nn.Parameter(torch.zeros(hidden_size, dtype=torch.float32))
            self.bias_ih_1 = torch.nn.Parameter(torch.zeros(hidden_size, dtype=torch.float32))
