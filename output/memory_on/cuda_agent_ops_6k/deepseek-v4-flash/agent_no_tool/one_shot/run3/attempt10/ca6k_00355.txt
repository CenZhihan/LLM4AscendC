#include "ca6k_00355_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00355CustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const auto &origin_shape = x_shape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(origin_shape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(origin_shape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(origin_shape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(origin_shape.GetDim(3));

    uint32_t totalElements = N * C * H * W;
    uint32_t spatialDim = H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = (attrs != nullptr) ? attrs->GetAttrPointer<float>(1)[0] : 1e-5f;

    // tile length: we process one channel at a time across spatial dims
    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float)); // x and y local buffers
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    // tile length should be at least spatialDim to process full channels efficiently
    uint32_t tileLen = maxTile;
    if (spatialDim < tileLen) {
        tileLen = spatialDim;
        tileLen = AlignDown(tileLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = spatialDim;
    }

    // block dim: one block per channel
    uint32_t blockDim = C;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalElements(totalElements);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_C(C);
    tiling.set_spatialDim(spatialDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00355Custom {
public:
    __aicore__ inline KernelCa6k00355Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR running_mean, GM_ADDR running_var,
                                GM_ADDR y,
                                uint32_t totalElements, uint32_t tileLength,
                                uint32_t C, uint32_t spatialDim,
                                float eps)
    {
        totalLen_ = totalElements;
        tileLen_ = tileLength;
        C_ = C;
        spatialDim_ = spatialDim;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, C_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, C_);
        meanGm_.SetGlobalBuffer((__gm__ float*)running_mean, C_);
        varGm_.SetGlobalBuffer((__gm__ float*)running_var, C_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));

        // Load per-channel data into local memory
        pipe_.InitBuffer(localWeight_, C_ * sizeof(float));
        pipe_.InitBuffer(localBias_, C_ * sizeof(float));
        pipe_.InitBuffer(localMean_, C_ * sizeof(float));
        pipe_.InitBuffer(localVar_, C_ * sizeof(float));

        AscendC::LocalTensor<float> lW = localWeight_.Get<float>();
        AscendC::DataCopy(lW, weightGm_, C_);
        AscendC::LocalTensor<float> lB = localBias_.Get<float>();
        AscendC::DataCopy(lB, biasGm_, C_);
        AscendC::LocalTensor<float> lM = localMean_.Get<float>();
        AscendC::DataCopy(lM, meanGm_, C_);
        AscendC::LocalTensor<float> lV = localVar_.Get<float>();
        AscendC::DataCopy(lV, varGm_, C_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a subset of channels
        uint32_t channelsPerBlock = (C_ + blockNum - 1) / blockNum;
        uint32_t chStart = blockIdx * channelsPerBlock;
        if (chStart >= C_) return;
        uint32_t chEnd = chStart + channelsPerBlock;
        if (chEnd > C_) chEnd = C_;

        // For each channel, process spatial data in tiles
        for (uint32_t c = chStart; c < chEnd; ++c) {
            // pointers to per-channel arrays
            AscendC::LocalTensor<float> wLocal = localWeight_.Get<float>();
            AscendC::LocalTensor<float> bLocal = localBias_.Get<float>();
            AscendC::LocalTensor<float> mLocal = localMean_.Get<float>();
            AscendC::LocalTensor<float> vLocal = localVar_.Get<float>();

            float weightVal = wLocal(c);
            float biasVal = bLocal(c);
            float meanVal = mLocal(c);
            float varVal = vLocal(c);
            float invStd = 1.0f / sqrtf(varVal + eps_);

            // Process spatial elements for this channel in tiles
            uint32_t spatialRemaining = spatialDim_;
            uint32_t globalOffset = c * spatialDim_;
            while (spatialRemaining > 0) {
                uint32_t curTile = (spatialRemaining < tileLen_) ? spatialRemaining : tileLen_;
                // Copy in
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[globalOffset], curTile);
                inQueueX_.EnQue(xLocal);

                // Compute
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                // y = (x - mean) * invStd * weight + bias
                AscendC::LocalTensor<float> tLocal = pipe_.TmpBuf1.Get<float>();
                AscendC::Subs(tLocal, xLocal, meanVal, curTile);
                AscendC::Muls(tLocal, tLocal, invStd * weightVal, curTile);
                AscendC::Adds(yLocal, tLocal, biasVal, curTile);
                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xLocal);

                // Copy out
                AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[globalOffset], yOut, curTile);
                outQueueY_.FreeTensor(yOut);

                globalOffset += curTile;
                spatialRemaining -= curTile;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::TBuf<AscendC::TPosition::VECCALC> localWeight_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> localBias_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> localMean_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> localVar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> meanGm_;
    AscendC::GlobalTensor<float> varGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t C_ {0};
    uint32_t spatialDim_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_00355_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                         GM_ADDR running_mean, GM_ADDR running_var,
                                                         GM_ADDR y,
                                                         GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00355Custom op;
    op.Init(x, weight, bias, running_mean, running_var, y,
            tiling_data.totalElements, tiling_data.tileLength,
            tiling_data.C, tiling_data.spatialDim,
            tiling_data.eps);
    op.Process();
}
