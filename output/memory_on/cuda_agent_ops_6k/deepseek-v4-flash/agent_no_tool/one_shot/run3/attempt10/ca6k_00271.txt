static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MAX_BLOCK_DIM = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00271CustomTilingData tiling;
    const gert::Shape &inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t dims = inShape.GetDimNum();
    uint32_t N = inShape.GetDim(0);
    uint32_t C = inShape.GetDim(1);
    uint32_t D = (dims > 2) ? inShape.GetDim(2) : 1;
    uint32_t H = (dims > 3) ? inShape.GetDim(3) : 1;
    uint32_t W = (dims > 4) ? inShape.GetDim(4) : 1;
    uint32_t spatialSize = D * H * W;

    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = maxTile;
    if (spatialSize < tileLen) {
        tileLen = AlignDown(spatialSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = spatialSize;
    }

    uint32_t blockDim = N * C;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_batch(N);
    tiling.set_channels(C);
    tiling.set_spatialSize(spatialSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00271Custom {
public:
    __aicore__ inline KernelCa6k00271Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t channels,
                                uint32_t spatialSize, uint32_t tileLength)
    {
        batch_ = batch;
        channels_ = channels;
        spatialSize_ = spatialSize;
        tileLen_ = tileLength;
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        if (blockIdx >= batch_ * channels_) return;

        uint32_t n = blockIdx / channels_;
        uint32_t c = blockIdx % channels_;

        uint32_t sliceOffset = (n * channels_ + c) * spatialSize_;

        float sum = 0.0f;

        uint32_t remaining = spatialSize_;
        uint32_t offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining < tileLen_) ? remaining : tileLen_;

            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[sliceOffset + offset], curLen);
            inQueueX_.EnQue(xLocal);

            // Wait for data and accumulate
            xLocal = inQueueX_.DeQue<float>();
            for (uint32_t i = 0; i < curLen; ++i) {
                sum += xLocal.GetValue(i);
            }
            inQueueX_.FreeTensor(xLocal);

            offset += curLen;
            remaining -= curLen;
        }

        float avg = sum / static_cast<float>(spatialSize_);
        yGm_.SetValue(blockIdx, avg);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_ {0};
    uint32_t channels_ {0};
    uint32_t spatialSize_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00271_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00271Custom op;
    op.Init(x, y, tiling_data.batch, tiling_data.channels,
            tiling_data.spatialSize, tiling_data.tileLength);
    op.Process();
}
"""
