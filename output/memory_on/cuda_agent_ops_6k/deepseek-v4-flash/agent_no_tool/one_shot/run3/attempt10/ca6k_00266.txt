    Ca6k00266CustomTilingData tiling;

    const gert::Shape *offsets_shape = context->GetInputShape(1);
    const gert::Shape *weight_shape = context->GetInputShape(2);

    uint32_t batchSize = static_cast<uint32_t>(offsets_shape->GetOriginShape().GetShapeSize()) - 1;

    // Tile embedding dimension
    uint32_t maxTile = UB_BUDGET / (2 * sizeof(float));  // accumulator + load buffer
    maxTile = (maxTile / ALIGN) * ALIGN;
    uint32_t tileLen = (embeddingDim < maxTile) ? embeddingDim : maxTile;
    if (tileLen < ALIGN) tileLen = ALIGN;

    uint32_t blockDim = 1;

    context->SetBlockDim(blockDim);
    tiling.set_numEmbeddings(numEmbeddings);
    tiling.set_embeddingDim(embeddingDim);
    tiling.set_batchSize(batchSize);
    tiling.set_totalIndices(totalIndices);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00266Custom {
public:
    __aicore__ inline KernelCa6k00266Custom() {}

    __aicore__ inline void Init(GM_ADDR indices, GM_ADDR offsets, GM_ADDR weight,
                                GM_ADDR output,
                                uint32_t numEmbeddings, uint32_t embeddingDim,
                                uint32_t batchSize, uint32_t totalIndices,
                                uint32_t tileLen)
    {
        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices, totalIndices);
        offsetsGm_.SetGlobalBuffer((__gm__ int64_t*)offsets, batchSize + 1);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numEmbeddings * embeddingDim);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, batchSize * embeddingDim);

        numEmbeddings_ = numEmbeddings;
        embeddingDim_ = embeddingDim;
        batchSize_ = batchSize;
        totalIndices_ = totalIndices;
        tileLen_ = tileLen;

        pipe_.InitBuffer(accBuf_, tileLen_ * sizeof(float));
        pipe_.InitBuffer(embBuf_, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || embeddingDim_ == 0) return;

        for (uint32_t b = 0; b < batchSize_; ++b) {
            int64_t start = offsetsGm_.GetValue(b);
            int64_t end = offsetsGm_.GetValue(b + 1);

            for (uint32_t d = 0; d < embeddingDim_; d += tileLen_) {
                uint32_t curLen = tileLen_;
                if (d + curLen > embeddingDim_) curLen = embeddingDim_ - d;

                // Init accumulator to zero
                AscendC::LocalTensor<float> accLocal = accBuf_.Get<float>();
                AscendC::Duplicated(accLocal, 0.0f, curLen);

                for (int64_t i = start; i < end; ++i) {
                    int64_t idx = indicesGm_.GetValue(i);
                    // Load embedding row slice
                    AscendC::LocalTensor<float> embLocal = embBuf_.Get<float>();
                    AscendC::DataCopy(embLocal, weightGm_[idx * embeddingDim_ + d], curLen);
                    AscendC::Add(accLocal, accLocal, embLocal, curLen);
                }
                // Write output
                AscendC::DataCopy(outputGm_[b * embeddingDim_ + d], accLocal, curLen);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> accBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> embBuf_;

    AscendC::GlobalTensor<int64_t> indicesGm_;
    AscendC::GlobalTensor<int64_t> offsetsGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t numEmbeddings_ {0};
    uint32_t embeddingDim_ {0};
    uint32_t batchSize_ {0};
    uint32_t totalIndices_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00266_custom(GM_ADDR indices, GM_ADDR offsets, GM_ADDR weight,
                                                        GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00266Custom op;
    op.Init(indices, offsets, weight, output,
            tiling_data.numEmbeddings, tiling_data.embeddingDim,
            tiling_data.batchSize, tiling_data.totalIndices,
            tiling_data.tileLen);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00266_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_embeddings, embedding_dim, mode="sum"):
        super(ModelNew, self).__init__()
        self.num_embeddings = num_embeddings
        self.embedding_dim = embedding_dim
        self.mode = mode
        # weight is not an input; we embed it directly in the op?
        # But reference has EmbeddingBag with its own weight. We'll
        # re-create a weight parameter here and pass it to the op.
        self.weight = torch.nn.Parameter(torch.randn(num_embeddings, embedding_dim))

    def forward(self, input, offsets):
        fn = getattr(custom_ops_lib, _FN)
        return fn(input, offsets, self.weight, self.mode)
