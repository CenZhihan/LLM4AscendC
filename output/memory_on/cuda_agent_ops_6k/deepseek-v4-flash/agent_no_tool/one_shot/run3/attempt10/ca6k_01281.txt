kernel_src='''

#include "kernel_operator.h"
#include "ca6k_01281_custom_tiling.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01281Custom {
public:
    __aicore__ inline KernelCa6k01281Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR hx, GM_ADDR y, GM_ADDR hy,
                                uint32_t seqLen, uint32_t batch, uint32_t inputSize,
                                uint32_t hiddenSize, uint32_t numLayers, bool bias,
                                bool batchFirst, float dropout,
                                uint32_t tileLen, uint32_t blockDim)
    {
        seqLen_ = seqLen;
        batch_ = batch;
        inputSize_ = inputSize;
        hiddenSize_ = hiddenSize;
        numLayers_ = numLayers;
        bias_ = bias;
        batchFirst_ = batchFirst;
        dropout_ = dropout;
        tileLen_ = tileLen;

        uint32_t xSize = seqLen * batch * inputSize;
        uint32_t hxSize = numLayers * batch * hiddenSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        hxGm_.SetGlobalBuffer((__gm__ float*)hx, hxSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, xSize);
        hyGm_.SetGlobalBuffer((__gm__ float*)hy, hxSize);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t totalX = seqLen_ * batch_ * inputSize_;
        uint32_t offset = 0;
        while (offset < totalX) {
            uint32_t curLen = tileLen_;
            if (offset + curLen > totalX) curLen = totalX - offset;
            CopyIn(offset, curLen);
            CopyOut(offset, curLen);
            offset += curLen;
        }

        uint32_t hxSize = numLayers_ * batch_ * hiddenSize_;
        AscendC::LocalTensor<float> hxLocal;
        pipe_.InitBuffer(tmpQue_, 1, hxSize * sizeof(float));
        hxLocal = tmpQue_.AllocTensor<float>();
        AscendC::DataCopy(hxLocal, hxGm_, hxSize);
        AscendC::DataCopy(hyGm_, hxLocal, hxSize);
        tmpQue_.FreeTensor(hxLocal);
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], xLocal, len);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> hxGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> hyGm_;

    uint32_t seqLen_;
    uint32_t batch_;
    uint32_t inputSize_;
    uint32_t hiddenSize_;
    uint32_t numLayers_;
    bool bias_;
    bool batchFirst_;
    float dropout_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void ca6k_01281_custom(GM_ADDR x, GM_ADDR hx, GM_ADDR y, GM_ADDR hy,
                                                          GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelCa6k01281Custom op;
    op.Init(x, hx, y, hy,
            tilingData.seqLen, tilingData.batchSize,
            tilingData.inputSize, tilingData.hiddenSize,
            tilingData.numLayers, tilingData.bias,
            tilingData.batchFirst, tilingData.dropout,
            tilingData.tileLen, tilingData.blockDim);
    op.Process();
}
