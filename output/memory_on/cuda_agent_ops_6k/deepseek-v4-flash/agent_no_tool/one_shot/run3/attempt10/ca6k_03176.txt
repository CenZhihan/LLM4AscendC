#include "ca6k03176_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MAX_TILE_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k03176CustomTilingData tiling;

    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::Shape &originShape = xShape->GetOriginShape();
    uint32_t totalLength = 1;
    for (uint32_t d = 0; d < originShape.GetDimNum(); ++d)
        totalLength *= static_cast<uint32_t>(originShape.GetDim(d));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += originShape.GetDimNum();
    uint32_t rowSize = static_cast<uint32_t>(originShape.GetDim(dim));
    uint32_t columns = totalLength / rowSize;

    uint32_t tileLen = UB_BUDGET_BYTES / (2 * sizeof(float));
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen > MAX_TILE_ELEMS) tileLen = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;
    if (tileLen > rowSize) tileLen = AlignDown(rowSize, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = rowSize;

    uint32_t blockDim = columns;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_columns(columns);
    tiling.set_rowSize(rowSize);
    tiling.set_tileLength(tileLen);
    tiling.set_dim(dim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += x_shape->GetDimNum();

    gert::Shape *values_shape = context->GetOutputShape(0);
    uint32_t outDims = x_shape->GetDimNum() - 1;
    if (outDims == 0) outDims = 1;
    values_shape->SetDimNum(outDims);
    uint32_t idx = 0;
    for (uint32_t d = 0; d < x_shape->GetDimNum(); ++d) {
        if (static_cast<int32_t>(d) != dim) {
            values_shape->SetDim(idx, x_shape->GetDim(d));
            ++idx;
        }
    }
    if (outDims == 1 && x_shape->GetDimNum() == 1) {
        values_shape->SetDim(0, 1);
    }

    gert::Shape *indices_shape = context->GetOutputShape(1);
    indices_shape->SetDimNum(values_shape->GetDimNum());
    for (uint32_t d = 0; d < values_shape->GetDimNum(); ++d)
        indices_shape->SetDim(d, values_shape->GetDim(d));

    return GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k03176Custom {
public:
    __aicore__ inline KernelCa6k03176Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR values, GM_ADDR indices,
                                uint32_t totalLength, uint32_t columns, uint32_t rowSize,
                                uint32_t tileLength, int32_t dim)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        valuesGm_.SetGlobalBuffer((__gm__ float*)values, columns);
        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices, columns);
        columns_ = columns;
        rowSize_ = rowSize;
        tileLen_ = tileLength;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueXCopy_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(oneBuf_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(pivotBuf_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(maskBuf_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (columns_ == 0 || rowSize_ == 0) return;
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= columns_) return;

        uint32_t colStart = blockIdx * rowSize_;

        // First pass: count non-NaN elements using vectorized operations
        float nonNanCountF = 0.0f;
        for (uint32_t offset = 0; offset < rowSize_; offset += tileLen_) {
            uint32_t curLen = (offset + tileLen_ > rowSize_) ? rowSize_ - offset : tileLen_;
            if (curLen == 0) break;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[colStart + offset], curLen);
            inQueueX_.EnQue(xLocal);
            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();

            // Create copy for NaN detection (compare with self)
            AscendC::LocalTensor<float> xCopy = inQueueXCopy_.AllocTensor<float>();
            AscendC::DataCopy(xCopy, xGm_[colStart + offset], curLen);
            // xCopy is same as xDeque, but we keep it for comparison
            AscendC::LocalTensor<float> nanMask = maskBuf_.AllocTensor<float>();
            AscendC::Compare(nanMask, xDeque, xCopy, AscendC::CMPMODE::NE, curLen);
            // nanMask = 1.0 for NaN, 0.0 for non-NaN
            AscendC::LocalTensor<float> sumTmp = tmpBuf_.AllocTensor<float>();
            AscendC::ReduceSum(sumTmp, nanMask, curLen);
            float nanCount = sumTmp.GetValue(0);
            nonNanCountF += (static_cast<float>(curLen) - nanCount);

            inQueueX_.FreeTensor(xDeque);
            inQueueXCopy_.FreeTensor(xCopy);
            maskBuf_.FreeTensor(nanMask);
            tmpBuf_.FreeTensor(sumTmp);
        }

        if (nonNanCountF < 0.5f) {
            valuesGm_.SetValue(blockIdx, 0.0f);
            indicesGm_.SetValue(blockIdx, 0);
            return;
        }

        float kFloat = (nonNanCountF - 1.0f) * 0.5f; // median index (0-based)

        // Quickselect using vectorized comparisons and ReduceSum
        int64_t lowIndex = 0;
        int64_t highIndex = static_cast<int64_t>(rowSize_) - 1;
        float medianVal = 0.0f;
        int64_t medianIdx = 0;
        uint32_t maxIter = 200;

        while (lowIndex <= highIndex && maxIter > 0) {
            --maxIter;

            // Pick pivot from low, mid, high (avoid NaN)
            int64_t mid = lowIndex + (highIndex - lowIndex) / 2;
            float vLow = ReadOneFloat(colStart, lowIndex);
            float vMid = ReadOneFloat(colStart, mid);
            float vHigh = ReadOneFloat(colStart, highIndex);
            float pivotVal;
            int64_t pivotPos;
            if (vLow == vLow) { pivotVal = vLow; pivotPos = lowIndex; }
            else if (vMid == vMid) { pivotVal = vMid; pivotPos = mid; }
            else if (vHigh == vHigh) { pivotVal = vHigh; pivotPos = highIndex; }
            else {
                bool found = false;
                for (int64_t pos = lowIndex; pos <= highIndex; ++pos) {
                    float v = ReadOneFloat(colStart, pos);
                    if (v == v) {
                        pivotVal = v;
                        pivotPos = pos;
                        found = true;
                        break;
                    }
                }
                if (!found) break;
            }

            // Broadcast pivot to a tensor
            AscendC::LocalTensor<float> pivotLocal = pivotBuf_.AllocTensor<float>();
            AscendC::Muls(pivotLocal, 0.0f, tileLen_);
            AscendC::Adds(pivotLocal, pivotVal, tileLen_);

            float lessCount = 0.0f;
            float eqCount = 0.0f;
            int64_t lastEqPos = -1;

            for (uint32_t offset = 0; offset < rowSize_; offset += tileLen_) {
                uint32_t curLen = (offset + tileLen_ > rowSize_) ? rowSize_ - offset : tileLen_;
                if (curLen == 0) break;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[colStart + offset], curLen);
                inQueueX_.EnQue(xLocal);
                AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();

                // Create copy for NaN detection
                AscendC::LocalTensor<float> xCopy = inQueueXCopy_.AllocTensor<float>();
                AscendC::DataCopy(xCopy, xGm_[colStart + offset], curLen);

                // valid mask: 1 for non-NaN, 0 for NaN
                AscendC::LocalTensor<float> nanMask = maskBuf_.AllocTensor<float>();
                AscendC::Compare(nanMask, xDeque, xCopy, AscendC::CMPMODE::NE, curLen);
                // nanMask = 1 for NaN; invert
                AscendC::LocalTensor<float> validMask = tmpBuf_.AllocTensor<float>();
                AscendC::Muls(validMask, 1.0f, curLen);
                AscendC::Sub(validMask, validMask, nanMask, curLen);

                // less than pivot mask
                AscendC::LocalTensor<float> lessMask = maskBuf_.AllocTensor<float>();
                AscendC::Compare(lessMask, xDeque, pivotLocal, AscendC::CMPMODE::LT, curLen);
                AscendC::Mul(lessMask, lessMask, validMask, curLen);
                AscendC::LocalTensor<float> sumTmp = tmpBuf_.AllocTensor<float>();
                AscendC::ReduceSum(sumTmp, lessMask, curLen);
                lessCount += sumTmp.GetValue(0);

                // equal to pivot mask
                AscendC::Compare(lessMask, xDeque, pivotLocal, AscendC::CMPMODE::EQ, curLen);
                AscendC::Mul(lessMask, lessMask, validMask, curLen);
                AscendC::ReduceSum(sumTmp, lessMask, curLen);
                float eqAdd = sumTmp.GetValue(0);
                eqCount += eqAdd;

                // record last equal position in this tile
                if (eqAdd > 0.5f) {
                    for (uint32_t i = 0; i < curLen; ++i) {
                        int64_t absPos = static_cast<int64_t>(offset + i);
                        if (absPos >= lowIndex && absPos <= highIndex) {
                            float v = xDeque.GetValue(i);
                            if (v == v && v == pivotVal) {
                                lastEqPos = absPos;
                            }
                        }
                    }
                }

                inQueueX_.FreeTensor(xDeque);
                inQueueXCopy_.FreeTensor(xCopy);
                maskBuf_.FreeTensor(nanMask);
                tmpBuf_.FreeTensor(validMask);
                // sumTmp is reused via tmpBuf_, no need to free here
            }

            pivotBuf_.FreeTensor(pivotLocal);

            if (kFloat < lessCount) {
                highIndex = pivotPos - 1;
            } else if (kFloat < lessCount + eqCount) {
                medianVal = pivotVal;
                medianIdx = lastEqPos;
                break;
            } else {
                kFloat -= (lessCount + eqCount);
                lowIndex = pivotPos + 1;
            }
        }

        valuesGm_.SetValue(blockIdx, medianVal);
        indicesGm_.SetValue(blockIdx, medianIdx);
    }

private:
    __aicore__ inline float ReadOneFloat(uint32_t colStart, int64_t pos)
    {
        uint32_t gmOffset = colStart + static_cast<uint32_t>(pos);
        AscendC::LocalTensor<float> tmp = scalarBuf_.AllocTensor<float>();
        AscendC::DataCopy(tmp, xGm_[gmOffset], 1);
        float val = tmp.GetValue(0);
        scalarBuf_.FreeTensor(tmp);
        return val;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueXCopy_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> oneBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> pivotBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maskBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> valuesGm_;
    AscendC::GlobalTensor<int64_t> indicesGm_;

    uint32_t columns_;
    uint32_t rowSize_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void ca6k03176_custom(GM_ADDR x, GM_ADDR values, GM_ADDR indices,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k03176Custom op;
    op.Init(x, values, indices,
            tiling_data.totalLength, tiling_data.columns, tiling_data.rowSize,
            tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
'''
