kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01096Custom {
public:
    __aicore__ inline KernelCa6k01096Custom() {}

    __aicore__ inline void Init(
        uint32_t batch, uint32_t inChannels, uint32_t inHeight, uint32_t inWidth,
        uint32_t outChannels, uint32_t outHeight, uint32_t outWidth,
        uint32_t kH, uint32_t kW,
        int32_t strideH, int32_t strideW, int32_t padH, int32_t padW,
        int32_t outputPadH, int32_t outputPadW,
        uint32_t totalOutputElems,
        __gm__ float *x, __gm__ float *w, __gm__ float *y
    ) {
        batch_ = batch;
        inChannels_ = inChannels;
        inHeight_ = inHeight;
        inWidth_ = inWidth;
        outChannels_ = outChannels;
        outHeight_ = outHeight;
        outWidth_ = outWidth;
        kH_ = kH;
        kW_ = kW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        outputPadH_ = outputPadH;
        outputPadW_ = outputPadW;
        totalOutputElems_ = totalOutputElems;

        xGm_.SetGlobalBuffer(x, batch_ * inChannels_ * inHeight_ * inWidth_);
        // weight shape: (inChannels, outChannels, kH, kW)
        wGm_.SetGlobalBuffer(w, inChannels_ * outChannels_ * kH_ * kW_);
        yGm_.SetGlobalBuffer(y, totalOutputElems_);
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        for (uint32_t idx = blockIdx; idx < totalOutputElems_; idx += blockNum) {
            uint32_t n = idx / (outChannels_ * outHeight_ * outWidth_);
            uint32_t rem = idx - n * outChannels_ * outHeight_ * outWidth_;
            uint32_t co = rem / (outHeight_ * outWidth_);
            rem -= co * outHeight_ * outWidth_;
            uint32_t oh = rem / outWidth_;
            uint32_t ow = rem - oh * outWidth_;

            float sum = 0.0f;
            for (uint32_t ci = 0; ci < inChannels_; ci++) {
                for (uint32_t kh = 0; kh < kH_; kh++) {
                    for (uint32_t kw = 0; kw < kW_; kw++) {
                        int32_t ih = static_cast<int32_t>(oh + padH_) - static_cast<int32_t>(kh);
                        int32_t iw = static_cast<int32_t>(ow + padW_) - static_cast<int32_t>(kw);
                        if (ih % strideH_ != 0 || iw % strideW_ != 0) continue;
                        ih /= strideH_;
                        iw /= strideW_;
                        if (ih < 0 || static_cast<uint32_t>(ih) >= inHeight_ ||
                            iw < 0 || static_cast<uint32_t>(iw) >= inWidth_) continue;
                        float xVal = xGm_[n * inChannels_ * inHeight_ * inWidth_ +
                                          ci * inHeight_ * inWidth_ +
                                          static_cast<uint32_t>(ih) * inWidth_ + static_cast<uint32_t>(iw)];
                        // weight index: [ci][co][kh][kw]
                        float wVal = wGm_[ci * outChannels_ * kH_ * kW_ +
                                          co * kH_ * kW_ +
                                          kh * kW_ + kw];
                        sum += xVal * wVal;
                    }
                }
            }
            yGm_[idx] = sum;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t inChannels_;
    uint32_t inHeight_;
    uint32_t inWidth_;
    uint32_t outChannels_;
    uint32_t outHeight_;
    uint32_t outWidth_;
    uint32_t kH_;
    uint32_t kW_;
    int32_t strideH_;
    int32_t strideW_;
    int32_t padH_;
    int32_t padW_;
    int32_t outputPadH_;
    int32_t outputPadW_;
    uint32_t totalOutputElems_;
};

extern "C" __global__ __aicore__ void ca6k_01096_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01096Custom op;
    op.Init(
        tiling_data.batch, tiling_data.inChannels, tiling_data.inHeight, tiling_data.inWidth,
        tiling_data.outChannels, tiling_data.outHeight, tiling_data.outWidth,
        tiling_data.kernelHeight, tiling_data.kernelWidth,
        tiling_data.strideH, tiling_data.strideW, tiling_data.padH, tiling_data.padW,
        tiling_data.outputPadH, tiling_data.outputPadW,
        tiling_data.totalOutputElems,
        (__gm__ float *)x, (__gm__ float *)weight, (__gm__ float *)y
    );
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_01096_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias):
        super(ModelNew, self).__init__()
        # weight shape must match PyTorch's ConvTranspose2d: (in_channels, out_channels, kH, kW)
        self.weight = torch.nn.Parameter(torch.empty(in_channels, out_channels, kernel_size[0], kernel_size[1]))
        torch.nn.init.kaiming_uniform_(self.weight, a=torch.sqrt(torch.tensor(5.0)))
        self.bias = None
        if bias:
            self.bias = torch.nn.Parameter(torch.empty(out_channels))
            torch.nn.init.zeros_(self.bias)
        if isinstance(stride, tuple):
            self.stride_h = stride[0]
            self.stride_w = stride[1]
        else:
            self.stride_h = stride
            self.stride_w = stride
        if isinstance(padding, tuple):
            self.pad_h = padding[0]
            self.pad_w = padding[1]
        else:
            self.pad_h = padding
            self.pad_w = padding
        if isinstance(output_padding, tuple):
            self.output_pad_h = output_padding[0]
            self.output_pad_w = output_padding[1]
        else:
            self.output_pad_h = output_padding
            self.output_pad_w = output_padding

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        out = fn(x, self.weight, self.stride_h, self.stride_w,
                 self.pad_h, self.pad_w, self.output_pad_h, self.output_pad_w)
        if self.bias is not None:
            out = out + self.bias.view(1, -1, 1, 1)
        return out
'''
