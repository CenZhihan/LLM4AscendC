host_operator_src = """

#include "ca6k00289_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00289CustomTilingData tiling;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    int32_t kernelD = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t kernelH = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t kernelW = attrs->GetAttrPointer<int32_t>(2)[0];
    int32_t strideD = attrs->GetAttrPointer<int32_t>(3)[0];
    int32_t strideH = attrs->GetAttrPointer<int32_t>(4)[0];
    int32_t strideW = attrs->GetAttrPointer<int32_t>(5)[0];
    int32_t padD = attrs->GetAttrPointer<int32_t>(6)[0];
    int32_t padH = attrs->GetAttrPointer<int32_t>(7)[0];
    int32_t padW = attrs->GetAttrPointer<int32_t>(8)[0];

    const auto& shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batchSize = shape.GetDim(0);
    uint32_t channels = shape.GetDim(1);
    uint32_t inDepth = shape.GetDim(2);
    uint32_t inHeight = shape.GetDim(3);
    uint32_t inWidth = shape.GetDim(4);

    uint32_t outDepth = (inDepth + 2 * padD - kernelD) / strideD + 1;
    uint32_t outHeight = (inHeight + 2 * padH - kernelH) / strideH + 1;
    uint32_t outWidth = (inWidth + 2 * padW - kernelW) / strideW + 1;

    uint32_t totalOutputElements = batchSize * channels * outDepth * outHeight * outWidth;

    uint32_t blockDim = 16;
    uint32_t tileLen = 4096;
    if (totalOutputElements < tileLen * blockDim) {
        tileLen = (totalOutputElements + blockDim - 1) / blockDim;
    }

    context->SetBlockDim(blockDim);

    // Use set_ prefix (lowercase) as generated by TILING_DATA_FIELD_DEF macro
    tiling.set_batchSize(batchSize);
    tiling.set_channels(channels);
    tiling.set_inDepth(inDepth);
    tiling.set_inHeight(inHeight);
    tiling.set_inWidth(inWidth);
    tiling.set_outDepth(outDepth);
    tiling.set_outHeight(outHeight);
    tiling.set_outWidth(outWidth);
    tiling.set_blockDim(blockDim);
    tiling.set_tileLen(tileLen);
    tiling.set_kernelD(kernelD);
    tiling.set_kernelH(kernelH);
    tiling.set_kernelW(kernelW);
    tiling.set_strideD(strideD);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padD(padD);
    tiling.set_padH(padH);
    tiling.set_padW(padW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
