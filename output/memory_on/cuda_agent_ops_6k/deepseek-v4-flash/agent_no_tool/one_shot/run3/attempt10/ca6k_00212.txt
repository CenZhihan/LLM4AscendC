namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00212CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    // 3 buffers: input, target, diff. Each tile floats.
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;
    if (tileLen == 0) tileLen = 1;

    // single block for scalar reduction
    context->SetBlockDim(1);
    tiling.totalLength = totalLength;
    tiling.tileLength = tileLen;
    tiling.blockDim = 1;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00212Custom {
public:
    __aicore__ inline KernelCa6k00212Custom() {}

    __aicore__ inline void Init(GM_ADDR input, GM_ADDR target, GM_ADDR loss,
                                uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inputGm_.SetGlobalBuffer((__gm__ float*)input, totalLen_);
        targetGm_.SetGlobalBuffer((__gm__ float*)target, totalLen_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inQueueInput_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueTarget_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(diffQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(partialSumQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        // Initialize local scalar sum to 0
        AscendC::LocalTensor<float> sumLocal = partialSumQueue_.AllocTensor<float>();
        sumLocal.SetValue(0, 0.0f);

        uint32_t offset = 0;
        while (offset < totalLen_) {
            uint32_t curLen = (offset + tileLen_ <= totalLen_) ? tileLen_ : (totalLen_ - offset);

            // Load input tile
            AscendC::LocalTensor<float> inLocal = inQueueInput_.AllocTensor<float>();
            AscendC::DataCopy(inLocal, inputGm_[offset], curLen);
            inQueueInput_.EnQue(inLocal);

            // Load target tile
            AscendC::LocalTensor<float> tarLocal = inQueueTarget_.AllocTensor<float>();
            AscendC::DataCopy(tarLocal, targetGm_[offset], curLen);
            inQueueTarget_.EnQue(tarLocal);

            // Dequeue both
            AscendC::LocalTensor<float> xLocal = inQueueInput_.DeQue<float>();
            AscendC::LocalTensor<float> tLocal = inQueueTarget_.DeQue<float>();

            // Compute absolute difference
            AscendC::LocalTensor<float> diffLocal = diffQueue_.AllocTensor<float>();
            AscendC::Sub(diffLocal, xLocal, tLocal, curLen);
            AscendC::Abs(diffLocal, diffLocal, curLen);

            // Reduce tile sum
            AscendC::LocalTensor<float> tileSum = diffQueue_.AllocTensor<float>();
            AscendC::ReduceSum(tileSum, diffLocal, curLen);
            float tSumVal = tileSum.GetValue(0);
            diffQueue_.FreeTensor(tileSum);

            // Accumulate into global sum
            float curSum = sumLocal.GetValue(0);
            curSum += tSumVal;
            sumLocal.SetValue(0, curSum);

            // Free intermediate tensors
            diffQueue_.FreeTensor(diffLocal);
            inQueueTarget_.FreeTensor(tLocal);
            inQueueInput_.FreeTensor(xLocal);

            offset += curLen;
        }

        // Compute mean and write to global memory
        float lossVal = sumLocal.GetValue(0) / static_cast<float>(totalLen_);
        lossGm_.SetValue(0, lossVal);
        partialSumQueue_.FreeTensor(sumLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueInput_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTarget_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> diffQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> partialSumQueue_;

    AscendC::GlobalTensor<float> inputGm_;
    AscendC::GlobalTensor<float> targetGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00212_custom(GM_ADDR input, GM_ADDR target, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00212Custom op;
    op.Init(input, target, loss, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
"""
