#include "ca6k04592_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ROWS = 16;
static constexpr uint32_t HARD_CAP_ROWS = 1024;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04592CustomTilingData tiling;
    const gert::Shape &originShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalElements = static_cast<uint32_t>(originShape.GetShapeSize());
    const uint32_t ndim = originShape.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= static_cast<int32_t>(ndim)) dim = 1;

    uint32_t C = static_cast<uint32_t>(originShape.GetDim(dim));
    uint32_t rows = totalElements / C;

    // Per-tile memory: xLocal (tileRows*C), yLocal (tileRows*C), maxBuf (C), expBuf (C), sumBuf (1), oneBuf (C)
    // Total floats = 2*tileRows*C + 3*C + 1 ≈ 2*tileRows*C (dominant)
    // target ~80% of UB budget
    uint32_t maxTileRows = (UB_BUDGET_BYTES * 4 / 5) / (2 * C * sizeof(float));
    maxTileRows = AlignDown(maxTileRows, ALIGN_ROWS);
    if (maxTileRows < ALIGN_ROWS) maxTileRows = ALIGN_ROWS;
    if (maxTileRows > HARD_CAP_ROWS) maxTileRows = HARD_CAP_ROWS;

    uint32_t tileRows = maxTileRows;
    if (rows < tileRows) {
        tileRows = AlignDown(rows, ALIGN_ROWS);
        if (tileRows == 0) tileRows = rows;
    }

    const uint32_t numTiles = CeilDiv(rows, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    // Direct assignment (no setter methods)
    tiling.totalElements = totalElements;
    tiling.C = C;
    tiling.rows = rows;
    tiling.tileRows = tileRows;
    tiling.blockDim = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "ca6k04592_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04592Custom {
public:
    __aicore__ inline KernelCa6k04592Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t C,
                                uint32_t tileRows)
    {
        rows_ = rows;
        C_ = C;
        tileRows_ = tileRows;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileRows_ * C_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * C_ * sizeof(float));
        pipe_.InitBuffer(maxBuf_, 1, C_ * sizeof(float));
        pipe_.InitBuffer(expBuf_, 1, C_ * sizeof(float));
        pipe_.InitBuffer(sumBuf_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(oneBuf_, 1, C_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || C_ == 0 || tileRows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideRows = blockNum * tileRows_;
        uint32_t baseRow = blockIdx * tileRows_;
        if (baseRow >= rows_) return;

        uint32_t curRows = rows_ - baseRow;
        if (curRows > tileRows_) curRows = tileRows_;
        CopyIn(baseRow, curRows);

        while (true) {
            uint32_t nextBaseRow = baseRow + strideRows;
            uint32_t nextRows = 0;
            if (nextBaseRow < rows_) {
                nextRows = rows_ - nextBaseRow;
                if (nextRows > tileRows_) nextRows = tileRows_;
                CopyIn(nextBaseRow, nextRows);
            }

            Compute(curRows);
            CopyOut(baseRow, curRows);

            if (nextRows == 0) break;
            baseRow = nextBaseRow;
            curRows = nextRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowOffset, uint32_t numRows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        uint32_t len = numRows * C_;
        AscendC::DataCopy(xLocal, xGm_[rowOffset * C_], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t numRows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = maxBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> expLocal = expBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = sumBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> oneLocal = oneBuf_.AllocTensor<float>();

        for (uint32_t r = 0; r < numRows; ++r) {
            uint32_t offset = r * C_;
            AscendC::LocalTensor<float> rowX = xLocal[offset];
            AscendC::LocalTensor<float> rowY = yLocal[offset];

            // copy row to maxLocal to compute max
            AscendC::DataCopy(maxLocal, rowX, C_);
            // compute max
            AscendC::ReduceMax(maxLocal, maxLocal, C_);
            // broadcast max to full row (store in expLocal as temp)
            AscendC::Broadcast(expLocal, maxLocal, 1, C_);

            // x - max
            AscendC::Sub(rowY, rowX, expLocal, C_);

            // exp(x-max)
            AscendC::Exp(expLocal, rowY, C_);

            // sum(exp)
            AscendC::ReduceAdd(sumLocal, expLocal, C_);

            // log(sum)
            AscendC::Log(sumLocal, sumLocal, 1);

            // broadcast logSum to full row
            AscendC::Broadcast(oneLocal, sumLocal, 1, C_);

            // y = (x - max) - logSum
            AscendC::Sub(rowY, rowY, oneLocal, C_);
        }

        maxBuf_.FreeTensor(maxLocal);
        expBuf_.FreeTensor(expLocal);
        sumBuf_.FreeTensor(sumLocal);
        oneBuf_.FreeTensor(oneLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowOffset, uint32_t numRows)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint32_t len = numRows * C_;
        AscendC::DataCopy(yGm_[rowOffset * C_], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maxBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> expBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> oneBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t C_ {0};
    uint32_t tileRows_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04592_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04592Custom op;
    op.Init(x, y, tiling_data.rows, tiling_data.C, tiling_data.tileRows);
    op.Process();
}
