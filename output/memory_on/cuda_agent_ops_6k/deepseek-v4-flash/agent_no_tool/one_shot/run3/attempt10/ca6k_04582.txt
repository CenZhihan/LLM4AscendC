kernel_src = '''
#include "kernel_operator.h"
#include "ca6k_04582_custom_tiling.h"
#include <math.h>

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k04582Custom {
public:
    __aicore__ inline KernelCa6k04582Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t D)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, N * D);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * (N - 1) / 2);
        N_ = N;
        D_ = D;
        totalPairs_ = N * (N - 1) / 2;

        pipe_.InitBuffer(inQueueXi_, BUFFER_NUM, D_ * sizeof(float));
        pipe_.InitBuffer(inQueueXj_, BUFFER_NUM, D_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, D_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        for (uint32_t idx = blockIdx; idx < totalPairs_; idx += blockNum) {
            // Compute i, j from linear index idx
            uint32_t i = 0;
            while (true) {
                uint32_t nextStart = (i + 1) * (2 * N_ - i - 2) / 2;
                if (nextStart > idx) break;
                i++;
            }
            uint32_t j = idx - (i * (2 * N_ - i - 1) / 2) + i + 1;

            // Load row i
            AscendC::LocalTensor<float> rowI = inQueueXi_.AllocTensor<float>();
            AscendC::DataCopy(rowI, xGm_[i * D_], D_);
            inQueueXi_.EnQue(rowI);

            // Load row j
            AscendC::LocalTensor<float> rowJ = inQueueXj_.AllocTensor<float>();
            AscendC::DataCopy(rowJ, xGm_[j * D_], D_);
            inQueueXj_.EnQue(rowJ);

            // Dequeue both rows
            AscendC::LocalTensor<float> rowI_deq = inQueueXi_.DeQue<float>();
            AscendC::LocalTensor<float> rowJ_deq = inQueueXj_.DeQue<float>();

            // Compute diff squared and reduce sum
            AscendC::LocalTensor<float> diffSq = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(diffSq, rowI_deq, rowJ_deq, D_);
            AscendC::Mul(diffSq, diffSq, diffSq, D_);

            AscendC::LocalTensor<float> sum = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceAdd(sum, diffSq, D_);

            float s = sum(0);
            float dist = sqrtf(s);
            yGm_.SetValue(idx, dist);

            // Free tensors
            inQueueXi_.FreeTensor(rowI_deq);
            inQueueXj_.FreeTensor(rowJ_deq);
            tmpQueue_.FreeTensor(diffSq);
            scalarQueue_.FreeTensor(sum);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueXi_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueXj_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t D_ {0};
    uint32_t totalPairs_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04582_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04582Custom op;
    op.Init(x, y, tiling_data.N, tiling_data.D);
    op.Process();
}
'''
