#include "ca6k04608_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04608CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t numChannels =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float eps = attrs->GetAttrPointer<float>(0)[0];

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    const uint32_t perChannelLen = totalLength / numChannels;
    uint32_t tileLen = maxTile;
    if (perChannelLen < tileLen) {
        tileLen = AlignDown(perChannelLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = perChannelLen;
    }

    const uint32_t blockDim = numChannels;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    size_t dimNum = x_shape->GetDimNum();
    y_shape->SetDimNum(dimNum);
    for (size_t i = 0; i < dimNum; i++) {
        y_shape->SetDim(i, x_shape->GetDim(i));
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04608Custom {
public:
    __aicore__ inline KernelCa6k04608Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalLength, uint32_t tileLength,
                                float eps, uint32_t blockDim) {
        totalLength_ = totalLength;
        tileLength_ = tileLength;
        eps_ = eps;
        blockDim_ = blockDim;
        perChannelLen_ = totalLength_ / blockDim_;

        channelIdx_ = AscendC::GetBlockIdx();
        if (channelIdx_ >= blockDim_) return;

        xGm_.SetGlobalBuffer((__gm__ float *)x);
        yGm_.SetGlobalBuffer((__gm__ float *)y);
        weightGm_.SetGlobalBuffer((__gm__ float *)weight, blockDim_);
        biasGm_.SetGlobalBuffer((__gm__ float *)bias, blockDim_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLength_ * sizeof(float));

        weightVal_ = weightGm_[channelIdx_];
        biasVal_ = biasGm_[channelIdx_];
    }

    __aicore__ inline void Process() {
        if (perChannelLen_ == 0 || tileLength_ == 0) return;

        uint32_t offset = channelIdx_ * perChannelLen_;
        uint32_t remaining = perChannelLen_;
        uint32_t curLen;
        AscendC::LocalTensor<float> xLocal;

        float sum = 0.0f;
        float sumSq = 0.0f;

        while (remaining > 0) {
            curLen = (remaining > tileLength_) ? tileLength_ : remaining;
            xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            for (uint32_t i = 0; i < curLen; i++) {
                float v = xLocal(i);
                sum += v;
                sumSq += v * v;
            }
            inQueueX_.FreeTensor(xLocal);
            remaining -= curLen;
            offset += curLen;
        }

        float mean = sum / perChannelLen_;
        float var = sumSq / perChannelLen_ - mean * mean;
        float invStd = 1.0f / sqrt(var + eps_);

        offset = channelIdx_ * perChannelLen_;
        remaining = perChannelLen_;
        while (remaining > 0) {
            curLen = (remaining > tileLength_) ? tileLength_ : remaining;
            xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            AscendC::Sub(yLocal, xLocal, mean, curLen);
            AscendC::Muls(yLocal, yLocal, invStd, curLen);
            AscendC::Muls(yLocal, yLocal, weightVal_, curLen);
            AscendC::Add(yLocal, yLocal, biasVal_, curLen);

            outQueueY_.EnQue<float>(yLocal);
            inQueueX_.FreeTensor(xLocal);

            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);

            remaining -= curLen;
            offset += curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalLength_ {0};
    uint32_t tileLength_ {0};
    uint32_t blockDim_ {0};
    uint32_t perChannelLen_ {0};
    uint32_t channelIdx_ {0};
    float weightVal_ {1.0f};
    float biasVal_ {0.0f};
    float eps_ {1e-5};
};

extern "C" __global__ __aicore__ void ca6k_04608_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                                         GM_ADDR running_mean, GM_ADDR running_var,
                                                         GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04608Custom op;
    op.Init(x, y, weight, bias, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.eps, tiling_data.blockDim);
    op.Process();
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_04608_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_features, eps=1e-5, momentum=0.1, affine=True, track_running_stats=True):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.eps = eps
        self.momentum = momentum
        self.affine = affine
        self.track_running_stats = track_running_stats
        if affine:
            self.weight = torch.nn.Parameter(torch.ones(num_features))
            self.bias = torch.nn.Parameter(torch.zeros(num_features))
        else:
            self.register_buffer('weight', torch.ones(num_features))
            self.register_buffer('bias', torch.zeros(num_features))
        if track_running_stats:
            self.register_buffer('running_mean', torch.zeros(num_features))
            self.register_buffer('running_var', torch.ones(num_features))
        else:
            self.register_buffer('running_mean', None)
            self.register_buffer('running_var', None)

    def forward(self, x):
        weight = self.weight
        bias = self.bias
        running_mean = self.running_mean if self.running_mean is not None else torch.zeros(self.num_features, device=x.device)
        running_var = self.running_var if self.running_var is not None else torch.ones(self.num_features, device=x.device)
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, weight, bias, running_mean, running_var, self.eps)
'''
