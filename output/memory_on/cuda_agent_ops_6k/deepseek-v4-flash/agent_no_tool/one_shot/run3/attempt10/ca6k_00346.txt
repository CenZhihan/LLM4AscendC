static constexpr uint32_t MAX_BLOCK_DIM = 24;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00346CustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::Shape &origin = x_shape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t C = static_cast<uint32_t>(origin.GetDim(1));
    uint32_t inputH = static_cast<uint32_t>(origin.GetDim(2));
    uint32_t inputW = static_cast<uint32_t>(origin.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scaleFactor = attrs->GetAttrPointer<float>(0);
    float sf = *scaleFactor;

    uint32_t outputH = static_cast<uint32_t>(static_cast<float>(inputH) * sf);
    uint32_t outputW = static_cast<uint32_t>(static_cast<float>(inputW) * sf);

    // Tile size: 16x16 output pixels
    uint32_t tileH = 16;
    uint32_t tileW = 16;

    uint32_t numTilesH = CeilDiv(outputH, tileH);
    uint32_t numTilesW = CeilDiv(outputW, tileW);
    uint32_t totalTiles = N * C * numTilesH * numTilesW;

    uint32_t blockDim = (totalTiles < MAX_BLOCK_DIM) ? totalTiles : MAX_BLOCK_DIM;
    if (blockDim < 1) blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_scaleFactor(sf);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src='''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00346Custom {
public:
    __aicore__ inline KernelCa6k00346Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t outputH, uint32_t outputW,
                                uint32_t tileH, uint32_t tileW,
                                float scaleFactor)
    {
        N_ = N;
        C_ = C;
        inputH_ = inputH;
        inputW_ = inputW;
        outputH_ = outputH;
        outputW_ = outputW;
        tileH_ = tileH;
        tileW_ = tileW;
        scaleFactor_ = scaleFactor;
        // Compute reciprocal for division
        scaleRcp_ = 1.0f / scaleFactor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * inputH * inputW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C * outputH * outputW);

        // Allocate local buffer for one output tile (tileH*tileW floats) + one line of local tensors for neighbors
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileH_ * tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileH_ * tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileH_ * tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outputH_ == 0 || outputW_ == 0) return;

        const uint32_t blockDim = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();

        // Total number of output tiles (spatial only, per (N,C) slice)
        uint32_t tilesH = (outputH_ + tileH_ - 1) / tileH_;
        uint32_t tilesW = (outputW_ + tileW_ - 1) / tileW_;
        uint32_t totalTiles = tilesH * tilesW;

        // Grid-stride loop over tiles
        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockDim) {
            uint32_t tileRow = tileIdx / tilesW;
            uint32_t tileCol = tileIdx % tilesW;
            uint32_t startH = tileRow * tileH_;
            uint32_t startW = tileCol * tileW_;
            uint32_t endH = startH + tileH_;
            uint32_t endW = startW + tileW_;
            if (endH > outputH_) endH = outputH_;
            if (endW > outputW_) endW = outputW_;
            uint32_t tileHeight = endH - startH;
            uint32_t tileWidth = endW - startW;
            if (tileHeight == 0 || tileWidth == 0) continue;

            // Process each (N,C) slice
            for (uint32_t n = 0; n < N_; ++n) {
                for (uint32_t c = 0; c < C_; ++c) {
                    // Compute output tile pixels
                    ComputeTile(n, c, startH, endH, startW, endW, tileHeight, tileWidth);
                }
            }
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t n, uint32_t c,
                                       uint32_t startH, uint32_t endH,
                                       uint32_t startW, uint32_t endW,
                                       uint32_t tileH, uint32_t tileW)
    {
        // Local tensor for output tile
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // We'll process row by row within the tile to limit local memory usage
        for (uint32_t i = startH; i < endH; ++i) {
            for (uint32_t j = startW; j < endW; ++j) {
                // bilinear interpolation
                // output pixel (j,i) in spatial (W,H)
                // input coordinate: x_in = (j + 0.5)/scale - 0.5, y_in = (i + 0.5)/scale - 0.5
                float x_in = (static_cast<float>(j) + 0.5f) * scaleRcp_ - 0.5f;
                float y_in = (static_cast<float>(i) + 0.5f) * scaleRcp_ - 0.5f;
                // Clamp to valid range
                int32_t x0 = static_cast<int32_t>(floorf(x_in));
                int32_t y0 = static_cast<int32_t>(floorf(y_in));
                // fractional part
                float dx = x_in - static_cast<float>(x0);
                float dy = y_in - static_cast<float>(y0);
                // Clamp to [0, inputW-1] and [0, inputH-1]
                int32_t x1 = x0 + 1;
                int32_t y1 = y0 + 1;
                if (x0 < 0) x0 = 0;
                if (x0 >= static_cast<int32_t>(inputW_)) x0 = inputW_ - 1;
                if (x1 < 0) x1 = 0;
                if (x1 >= static_cast<int32_t>(inputW_)) x1 = inputW_ - 1;
                if (y0 < 0) y0 = 0;
                if (y0 >= static_cast<int32_t>(inputH_)) y0 = inputH_ - 1;
                if (y1 < 0) y1 = 0;
                if (y1 >= static_cast<int32_t>(inputH_)) y1 = inputH_ - 1;

                // Read four neighbors from global memory
                uint32_t baseIdx = n * C_ * inputH_ * inputW_ + c * inputH_ * inputW_;
                float v00 = xGm_[baseIdx + static_cast<uint32_t>(y0) * inputW_ + static_cast<uint32_t>(x0)];
                float v01 = xGm_[baseIdx + static_cast<uint32_t>(y0) * inputW_ + static_cast<uint32_t>(x1)];
                float v10 = xGm_[baseIdx + static_cast<uint32_t>(y1) * inputW_ + static_cast<uint32_t>(x0)];
                float v11 = xGm_[baseIdx + static_cast<uint32_t>(y1) * inputW_ + static_cast<uint32_t>(x1)];
                // Bilinear interpolation
                float w00 = (1.0f - dx) * (1.0f - dy);
                float w01 = dx * (1.0f - dy);
                float w10 = (1.0f - dx) * dy;
                float w11 = dx * dy;
                float out = v00 * w00 + v01 * w01 + v10 * w10 + v11 * w11;
                // Store to local tensor
                uint32_t localIdx = (i - startH) * tileW_ + (j - startW);
                yLocal.SetValue(localIdx, out);
            }
        }

        // Copy output tile to global
        uint32_t yBase = n * C_ * outputH_ * outputW_ + c * outputH_ * outputW_ + startH * outputW_ + startW;
        // We need to copy tile rows individually because global memory is row-major
        for (uint32_t row = 0; row < tileH; ++row) {
            uint32_t gmOffset = yBase + row * outputW_;
            uint32_t localOffset = row * tileW_;
            AscendC::DataCopy(yGm_[gmOffset], yLocal[localOffset], tileW);
        }

        outQueueY_.FreeTensor(yLocal);
        tmpQueue_.FreeTensor(tLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t inputH_ {0};
    uint32_t inputW_ {0};
    uint32_t outputH_ {0};
    uint32_t outputW_ {0};
    uint32_t tileH_ {16};
    uint32_t tileW_ {16};
    float scaleFactor_ {1.0f};
    float scaleRcp_ {1.0f};
};

extern "C" __global__ __aicore__ void ca6k00346_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00346Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C,
            tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputH, tiling_data.outputW,
            tiling_data.tileH, tiling_data.tileW,
            tiling_data.scaleFactor);
    op.Process();
}
'''

import torch

import custom_ops_lib

_FN = 'ca6k00346_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super().__init__()
        # init_inputs[0] is scale_factor, but we don't need to store it because op gets it from input
        self.scale_factor = init_inputs[0] if len(init_inputs) > 0 else 2.0

    def forward(self, *inputs):
        x = inputs[0]
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.scale_factor)
