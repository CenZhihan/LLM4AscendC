static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04599CustomTilingData tiling;

    const gert::Shape &xShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t inN = xShape.GetDim(0);
    uint32_t inC = xShape.GetDim(1);
    uint32_t inD = xShape.GetDim(2);
    uint32_t inH = xShape.GetDim(3);
    uint32_t inW = xShape.GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(0);
    uint32_t padLeft   = static_cast<uint32_t>(padding[0]);
    uint32_t padRight  = static_cast<uint32_t>(padding[1]);
    uint32_t padTop    = static_cast<uint32_t>(padding[2]);
    uint32_t padBottom = static_cast<uint32_t>(padding[3]);
    uint32_t padFront  = static_cast<uint32_t>(padding[4]);
    uint32_t padBack   = static_cast<uint32_t>(padding[5]);

    uint32_t outW = inW + padLeft + padRight;
    uint32_t outH = inH + padTop + padBottom;
    uint32_t outD = inD + padFront + padBack;
    uint32_t totalLength = inN * inC * outD * outH * outW;

    uint32_t elemPerRow = inW;
    uint32_t maxRows = UB_BUDGET_BYTES / (sizeof(float) * elemPerRow);
    if (maxRows > 256) maxRows = 256;
    uint32_t tileLen = maxRows * elemPerRow;
    if (tileLen > totalLength) {
        tileLen = AlignDown(totalLength, elemPerRow);
        if (tileLen == 0) tileLen = totalLength;

    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_inN(inN);
    tiling.set_inC(inC);
    tiling.set_inD(inD);
    tiling.set_inH(inH);
    tiling.set_inW(inW);

    tiling.set_outH(outH);

    tiling.set_padTop(padTop);
    tiling.set_padBottom(padBottom);
    tiling.set_padFront(padFront);
    tiling.set_padBack(padBack);

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04599Custom {
public:
    __aicore__ inline KernelCa6k04599Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inN, uint32_t inC, uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint32_t padLeft, uint32_t padRight, uint32_t padTop, uint32_t padBottom,
                                uint32_t padFront, uint32_t padBack)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inN_ = inN; inC_ = inC; inD_ = inD; inH_ = inH; inW_ = inW;
        outD_ = outD; outH_ = outH; outW_ = outW;
        padLeft_ = padLeft; padRight_ = padRight;
        padTop_ = padTop; padBottom_ = padBottom;
        padFront_ = padFront; padBack_ = padBack;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t totalRows = inN_ * inC_ * inD_ * inH_;
        uint32_t rowsPerTile = tileLen_ / inW_;
        if (rowsPerTile == 0) rowsPerTile = 1;

        uint32_t rowStart = blockIdx * rowsPerTile;
        uint32_t rowEnd = rowStart + rowsPerTile;
        if (rowEnd > totalRows) rowEnd = totalRows;

        uint32_t inStrideC = inD_ * inH_ * inW_;
        uint32_t inStrideD = inH_ * inW_;
        uint32_t inStrideH = inW_;
        uint32_t outStrideC = outD_ * outH_ * outW_;
        uint32_t outStrideD = outH_ * outW_;
        uint32_t outStrideH = outW_;

        if (rowStart < rowEnd) {
            uint32_t row = rowStart;
            PrefetchRow(row, inStrideC, inStrideD, inStrideH);
        }

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            if (row + 1 < rowEnd) {
                PrefetchRow(row + 1, inStrideC, inStrideD, inStrideH);
            }

            uint32_t n, c, d, h;
            ComputeIndices(row, n, c, d, h);
            uint32_t outOff = n * outStrideC * inC_ + c * outStrideC + (d + padFront_) * outStrideD + (h + padTop_) * outStrideH + padLeft_;
            uint32_t inOff = n * inStrideC * inC_ + c * inStrideC + d * inStrideD + h * inStrideH;

            AscendC::LocalTensor<float> rowBuf = inQueueX_.DeQue<float>();
            AscendC::DataCopy(yGm_[outOff], rowBuf, inW_);
            inQueueX_.FreeTensor(rowBuf);
        }
    }

private:
    __aicore__ inline void PrefetchRow(uint32_t row,
                                        uint32_t inStrideC, uint32_t inStrideD, uint32_t inStrideH)
    {
        uint32_t n, c, d, h;
        ComputeIndices(row, n, c, d, h);
        uint32_t inOff = n * inStrideC * inC_ + c * inStrideC + d * inStrideD + h * inStrideH;
        AscendC::LocalTensor<float> rowBuf = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(rowBuf, xGm_[inOff], inW_);
        inQueueX_.EnQue(rowBuf);
    }

    __aicore__ inline void ComputeIndices(uint32_t row, uint32_t &n, uint32_t &c, uint32_t &d, uint32_t &h)
    {
        uint32_t totalPerN = inC_ * inD_ * inH_;
        n = row / totalPerN;
        uint32_t rem = row - n * totalPerN;
        uint32_t totalPerC = inD_ * inH_;
        c = rem / totalPerC;
        rem = rem - c * totalPerC;
        d = rem / inH_;
        h = rem - d * inH_;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_, tileLen_;
    uint32_t inN_, inC_, inD_, inH_, inW_;
    uint32_t outD_, outH_, outW_;
    uint32_t padLeft_, padRight_, padTop_, padBottom_, padFront_, padBack_;
};

extern "C" __global__ __aicore__ void ca6k_04599_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04599Custom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.inN, tiling_data.inC, tiling_data.inD, tiling_data.inH, tiling_data.inW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.padLeft, tiling_data.padRight, tiling_data.padTop, tiling_data.padBottom,
            tiling_data.padFront, tiling_data.padBack);
    op.Process();
}
