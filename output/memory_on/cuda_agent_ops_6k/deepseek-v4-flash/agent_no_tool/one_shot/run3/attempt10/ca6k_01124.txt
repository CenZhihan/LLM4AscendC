#include "ca6k_01124_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k01124CustomTilingData tiling;

    const gert::StorageShape *storage = context->GetInputShape(0);
    const gert::Shape &inShape = storage->GetOriginShape();
    int32_t B = static_cast<int32_t>(inShape.GetDim(0));
    int32_t C = static_cast<int32_t>(inShape.GetDim(1));
    int32_t D = static_cast<int32_t>(inShape.GetDim(2));
    int32_t H = static_cast<int32_t>(inShape.GetDim(3));
    int32_t W = static_cast<int32_t>(inShape.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float normType = attrs->GetAttrPointer<float>(0)[0];
    const int32_t *kernelSize = attrs->GetAttrPointer<int32_t>(1);
    int32_t kD = kernelSize[0];
    int32_t kH = kernelSize[1];
    int32_t kW = kernelSize[2];

    int32_t outD = (D - kD) / kD + 1;
    int32_t outH = (H - kH) / kH + 1;
    int32_t outW = (W - kW) / kW + 1;

    uint32_t totalOutputElems =
        static_cast<uint32_t>(B) * static_cast<uint32_t>(C) *
        static_cast<uint32_t>(outD) * static_cast<uint32_t>(outH) * static_cast<uint32_t>(outW);

    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * 8);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (totalOutputElems < maxTile) ? totalOutputElems : maxTile;

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutputElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    int32_t strideC = D * H * W;
    int32_t strideD = H * W;
    int32_t strideH = W;

    int32_t outStrideC = outD * outH * outW;
    int32_t outStrideD = outH * outW;
    int32_t outStrideH = outW;

    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_normType(normType);
    tiling.set_kernelD(kD);
    tiling.set_kernelH(kH);
    tiling.set_kernelW(kW);
    tiling.set_inD(D);
    tiling.set_inH(H);
    tiling.set_inW(W);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_strideC(strideC);
    tiling.set_strideD(strideD);
    tiling.set_strideH(strideH);
    tiling.set_outStrideC(outStrideC);
    tiling.set_outStrideD(outStrideD);
    tiling.set_outStrideH(outStrideH);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *inShape = context->GetInputShape(0);
    int32_t B = inShape->GetDim(0);
    int32_t C = inShape->GetDim(1);
    int32_t D = inShape->GetDim(2);
    int32_t H = inShape->GetDim(3);
    int32_t W = inShape->GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernelSize = attrs->GetAttrPointer<int32_t>(1);
    int32_t kD = kernelSize[0];
    int32_t kH = kernelSize[1];
    int32_t kW = kernelSize[2];

    int32_t outD = (D - kD) / kD + 1;
    int32_t outH = (H - kH) / kH + 1;

    y_shape->SetDim(2, outD);
    y_shape->SetDim(3, outH);
    y_shape->SetDim(4, outW);
    return GRAPH_SUCCESS;
}

kernel_src = '''

#include "kernel_operator.h"
#include <cmath>

class KernelCa6k01124Custom {
public:
    __aicore__ inline KernelCa6k01124Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElems, uint32_t tileLength,
                                float normType,
                                int32_t kD, int32_t kH, int32_t kW,
                                int32_t inD, int32_t inH, int32_t inW,
                                int32_t outD, int32_t outH, int32_t outW,
                                int32_t strideC, int32_t strideD, int32_t strideH,
                                int32_t outStrideC, int32_t outStrideD, int32_t outStrideH) {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        totalOutputElems_ = totalOutputElems;
        tileLen_ = tileLength;
        normType_ = normType;
        kD_ = kD; kH_ = kH; kW_ = kW;
        inD_ = inD; inH_ = inH; inW_ = inW;
        outD_ = outD; outH_ = outH; outW_ = outW;
        strideC_ = strideC; strideD_ = strideD; strideH_ = strideH;
        outStrideC_ = outStrideC; outStrideD_ = outStrideD; outStrideH_ = outStrideH;
    }

    __aicore__ inline void Process() {
        if (totalOutputElems_ == 0) return;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        (void)blockNum;
        uint32_t begin = blockIdx * tileLen_;
        uint32_t end = begin + tileLen_;
        if (end > totalOutputElems_) end = totalOutputElems_;

        for (uint32_t idx = begin; idx < end; ++idx) {
            uint32_t outIdx = idx;
            int32_t ow = static_cast<int32_t>(outIdx % outW_); outIdx /= outW_;
            int32_t oh = static_cast<int32_t>(outIdx % outH_); outIdx /= outH_;
            int32_t od = static_cast<int32_t>(outIdx % outD_); outIdx /= outD_;
            int32_t c = static_cast<int32_t>(outIdx % static_cast<uint32_t>(outStrideC_));
            outIdx /= outStrideC_;
            int32_t b = static_cast<int32_t>(outIdx);

            int32_t inBase = b * strideC_ + c * inD_ * inH_ * inW_ + od * kD_ * strideD_ + oh * kH_ * strideH_ + ow * kW_;

            float sum = 0.0f;
            for (int32_t kd = 0; kd < kD_; ++kd) {
                for (int32_t kh = 0; kh < kH_; ++kh) {
                    for (int32_t kw = 0; kw < kW_; ++kw) {
                        int32_t inIdx = inBase + kd * strideD_ + kh * strideH_ + kw;
                        float val = xGm_[inIdx];
                        float absVal = fabsf(val);
                        if (absVal == 0.0f) {
                            sum += 0.0f;
                        } else {
                            sum += expf(logf(absVal) * normType_);
                        }
                    }
                }
            }
            float outVal = expf(logf(sum) * (1.0f / normType_));
            yGm_[idx] = outVal;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutputElems_ {0};
    uint32_t tileLen_ {0};
    float normType_ {2.0f};
    int32_t kD_ {2}, kH_ {2}, kW_ {2};
    int32_t inD_ {0}, inH_ {0}, inW_ {0};
    int32_t outD_ {0}, outH_ {0}, outW_ {0};
    int32_t strideC_ {0}, strideD_ {0}, strideH_ {0};
    int32_t outStrideC_ {0}, outStrideD_ {0}, outStrideH_ {0};
};

extern "C" __global__ __aicore__ void ca6k_01124_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01124Custom op;
    op.Init(x, y,
            tiling_data.totalOutputElems, tiling_data.tileLength,
            tiling_data.normType,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.inD, tiling_data.inH, tiling_data.inW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.strideC, tiling_data.strideD, tiling_data.strideH,
            tiling_data.outStrideC, tiling_data.outStrideD, tiling_data.outStrideH);
    op.Process();
}
'''
