kernel_src = """

#include "kernel_operator.h"

class KernelCa6k00495Custom {
public:
    __aicore__ inline KernelCa6k00495Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputLength, uint32_t n,
                                uint32_t c, uint32_t h, uint32_t w)
    {
        uint32_t totalInputElems = n * c * h * w;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputElems);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputLength);
        totalOutputLen_ = totalOutputLength;
        n_ = n;
        c_ = c;
        h_ = h;
        w_ = w;
    }

    __aicore__ inline void Process()
    {
        if (totalOutputLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t h2 = h_ / 2;
        uint32_t w2 = w_ / 2;

        for (uint32_t idx = blockIdx; idx < totalOutputLen_; idx += blockNum) {
            uint32_t temp = idx;
            uint32_t outCol = temp % w2;
            temp /= w2;
            uint32_t outRow = temp % h2;
            temp /= h2;
            uint32_t ch = temp % c_;
            uint32_t batch = temp / c_;

            uint32_t inBase = batch * c_ * h_ * w_ + ch * h_ * w_ + 2 * outRow * w_ + 2 * outCol;

            float v0 = xGm_[inBase];
            float v1 = xGm_[inBase + 1];
            float v2 = xGm_[inBase + w_];
            float v3 = xGm_[inBase + w_ + 1];

            float maxVal = v0;
            if (v1 > maxVal) maxVal = v1;
            if (v2 > maxVal) maxVal = v2;
            if (v3 > maxVal) maxVal = v3;

            yGm_[idx] = maxVal;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalOutputLen_ {0};
    uint32_t n_ {1};
    uint32_t c_ {1};
    uint32_t h_ {1};
    uint32_t w_ {1};
};

extern "C" __global__ __aicore__ void ca6k00495_custom(GM_ADDR x, GM_ADDR y, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00495Custom op;
    op.Init(x, y, tiling_data.totalOutputLength, tiling_data.n,
            tiling_data.c, tiling_data.h, tiling_data.w);
    op.Process();
}
"""
