kernel_src = """
#include "kernel_operator.h"

class KernelCa6k00262Custom {
public:
    __aicore__ inline KernelCa6k00262Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR src, GM_ADDR y,
                                uint32_t totalLength, uint32_t batch,
                                uint32_t H, uint32_t W,
                                uint32_t diagLen, int32_t offset)
    {
        totalLen_ = totalLength;
        batch_ = batch;
        H_ = H;
        W_ = W;
        diagLen_ = diagLen;
        offset_ = offset;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        srcGm_.SetGlobalBuffer((__gm__ float*)src, batch_ * diagLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t idx = blockIdx; idx < totalLen_; idx += blockNum) {
            // compute batch, row, column
            uint32_t b = idx / (H_ * W_);
            uint32_t rem = idx - b * H_ * W_;
            uint32_t i = rem / W_;
            uint32_t j = rem % W_;

            // check if element is on the diagonal (dim1=1, dim2=2)
            bool is_diag = (static_cast<int32_t>(j) == static_cast<int32_t>(i) + offset_)
                           && static_cast<int32_t>(i) >= 0
                           && static_cast<int32_t>(i) < static_cast<int32_t>(diagLen_)
                           && static_cast<int32_t>(j) >= 0
                           && static_cast<int32_t>(j) < static_cast<int32_t>(W_);

            if (is_diag) {
                yGm_.SetValue(idx, srcGm_.GetValue(b * diagLen_ + i));
            } else {
                yGm_.SetValue(idx, xGm_.GetValue(idx));
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> srcGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t batch_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t diagLen_ {0};
    int32_t offset_ {0};
};

extern "C" __global__ __aicore__ void ca6k00262_custom(GM_ADDR x, GM_ADDR src, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00262Custom op;
    op.Init(x, src, y,
            tiling_data.totalLength, tiling_data.batch,
            tiling_data.H, tiling_data.W,
            tiling_data.diagLen, tiling_data.offset);
    op.Process();
}
"""
