#include "ca6k00518_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00518CustomTilingData tiling;
    const gert::Shape &originShape = context->GetInputShape(0)->GetOriginShape();
    const int64_t rank = originShape.GetDimNum();
    uint32_t totalLength = static_cast<uint32_t>(originShape.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += rank;
    uint32_t dimSize = static_cast<uint32_t>(originShape.GetDim(dim));
    uint32_t numVectors = totalLength / dimSize;

    // UB budget for 3 floats per element (read, write, temp)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLength = dimSize < maxTile ? dimSize : maxTile;

    uint32_t blockDim = numVectors;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_dimSize(dimSize);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {

    y_shape->SetDimNum(dimNum);
    for (int64_t i = 0; i < dimNum; ++i) {
        y_shape->SetDim(i, x_shape->GetDim(i));
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "ca6k00518_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00518Custom {
public:
    __aicore__ inline KernelCa6k00518Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t dimSize,
                                uint32_t tileLength)
    {
        totalLen_ = totalLength;
        dimSize_ = dimSize;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // temp buffer for negative and exp
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || dimSize_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t numVectors = totalLen_ / dimSize_;

        for (uint32_t v = blockIdx; v < numVectors; v += blockNum) {
            uint32_t base = v * dimSize_;
            ProcessVector(base);
        }
    }

private:
    __aicore__ inline void ProcessVector(uint32_t base)
    {
        // First pass: find global max of -x over all tiles
        float globalMax = -1e38f; // sufficiently small
        for (uint32_t offset = 0; offset < dimSize_; offset += tileLen_) {
            uint32_t curLen = dimSize_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
            // compute local max of -x
            AscendC::Neg(xLocal, xLocal, curLen);
            AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceMax(maxLocal, xLocal, curLen);
            float localMax = maxLocal.GetValue(0);
            if (localMax > globalMax) globalMax = localMax;
            tmpQueue_.FreeTensor(maxLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        // Second pass: compute exp(-x - globalMax), sum, and write normalized
        float globalSum = 0.0f;
        // First accumulate sum
        for (uint32_t offset = 0; offset < dimSize_; offset += tileLen_) {
            uint32_t curLen = dimSize_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
            AscendC::Neg(xLocal, xLocal, curLen);
            AscendC::Adds(xLocal, xLocal, globalMax, curLen);
            AscendC::Exp(xLocal, xLocal, curLen);
            // Save exp values to temp for later use
            AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(expLocal, xLocal, curLen);
            // partial sum
            AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumLocal, xLocal, curLen);
            float partSum = sumLocal.GetValue(0);
            globalSum += partSum;
            tmpQueue_.FreeTensor(sumLocal);
            // Free xLocal, but we need exp values; store them in a separate buffer? 
            // Actually we stored in expLocal, but we will reuse tmpQueue for sumLocal; we need to keep exp values for writing.
            // Better: write directly to global after computing sum? We need sum first. So we compute sum in first sub-pass,
            // then second sub-pass to write. So we need to recompute exp. That's fine, it's cheap.
            inQueueX_.FreeTensor(xLocal);
            tmpQueue_.FreeTensor(expLocal);
        }

        // Use a scalar for 1/sum
        float invSum = 1.0f / globalSum;

        // Third pass: recompute exp and write normalized
        for (uint32_t offset = 0; offset < dimSize_; offset += tileLen_) {
            uint32_t curLen = dimSize_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
            AscendC::Neg(xLocal, xLocal, curLen);
            AscendC::Adds(xLocal, xLocal, globalMax, curLen);
            AscendC::Exp(xLocal, xLocal, curLen);
            AscendC::Muls(xLocal, xLocal, invSum, curLen);
            AscendC::DataCopy(yGm_[base + offset], xLocal, curLen);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t dimSize_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00518_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00518Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.dimSize, tiling_data.tileLength);
    op.Process();
}
