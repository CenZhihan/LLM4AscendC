kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float INF_VAL = 3.402823e+38f;

class KernelCa6k00188Custom {
public:
    __aicore__ inline KernelCa6k00188Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t totalCols)
    {
        totalRows_ = totalRows;
        totalCols_ = totalCols;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows * totalCols);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows * totalCols);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, totalCols_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, totalCols_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, totalCols_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || totalCols_ == 0) return;

        // persistent tensor for running min
        AscendC::LocalTensor<float> prevMins = tmpQueue_.AllocTensor<float>();
        AscendC::SetVal(prevMins, totalCols_, INF_VAL);

        for (uint32_t row = 0; row < totalRows_; row++) {
            uint32_t offset = row * totalCols_;

            // CopyIn
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], totalCols_);
            inQueueX_.EnQue(xLocal);

            // DeQue input (waits for copy completion)
            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();

            // Compute: y = min(x, prevMins)
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Min(yLocal, xDeque, prevMins, totalCols_);
            AscendC::Muls(prevMins, yLocal, 1.0f, totalCols_);   // prevMins = y
            outQueueY_.EnQue(yLocal);

            // CopyOut
            AscendC::LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yDeque, totalCols_);
            outQueueY_.FreeTensor(yDeque);
            inQueueX_.FreeTensor(xDeque);
        }

        tmpQueue_.FreeTensor(prevMins);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t totalCols_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00188_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00188Custom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.totalCols);
    op.Process();
}
'''
