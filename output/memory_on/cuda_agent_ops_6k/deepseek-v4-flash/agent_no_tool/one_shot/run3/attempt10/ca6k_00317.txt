static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *stride_attrs = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *padding_attrs = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *output_padding_attrs = attrs->GetAttrPointer<int64_t>(2);

    int32_t n = x_shape->GetDim(0);
    int32_t c_out = w_shape->GetDim(0);
    int32_t d_in = x_shape->GetDim(2);
    int32_t h_in = x_shape->GetDim(3);
    int32_t w_in = x_shape->GetDim(4);
    int32_t kd = w_shape->GetDim(2);
    int32_t kh = w_shape->GetDim(3);
    int32_t kw = w_shape->GetDim(4);

    int32_t d_out = (d_in - 1) * stride_attrs[0] - 2 * padding_attrs[0] + (kd - 1) + output_padding_attrs[0] + 1;
    int32_t h_out = (h_in - 1) * stride_attrs[1] - 2 * padding_attrs[1] + (kh - 1) + output_padding_attrs[1] + 1;
    int32_t w_out = (w_in - 1) * stride_attrs[2] - 2 * padding_attrs[2] + (kw - 1) + output_padding_attrs[2] + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, c_out);
    y_shape->SetDim(2, d_out);
    y_shape->SetDim(3, h_out);
    y_shape->SetDim(4, w_out);
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;

kernel_src = '''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00317Custom {
public:
    __aicore__ inline KernelCa6k00317Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                const Ca6k00317CustomTilingData &tiling) {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)w);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        n_ = tiling.n;
        c_in_ = tiling.c_in;
        c_out_ = tiling.c_out;
        d_in_ = tiling.d_in;
        h_in_ = tiling.h_in;
        w_in_ = tiling.w_in;
        d_out_ = tiling.d_out;
        h_out_ = tiling.h_out;
        w_out_ = tiling.w_out;
        kd_ = tiling.kd;
        kh_ = tiling.kh;
        kw_ = tiling.kw;
        stride_d_ = tiling.stride_d;
        stride_h_ = tiling.stride_h;
        stride_w_ = tiling.stride_w;
        pad_d_ = tiling.pad_d;
        pad_h_ = tiling.pad_h;
        pad_w_ = tiling.pad_w;
        output_pad_d_ = tiling.output_pad_d;
        output_pad_h_ = tiling.output_pad_h;
        output_pad_w_ = tiling.output_pad_w;
    }

    __aicore__ inline void Process() {
        if (n_ == 0 || c_out_ == 0 || d_out_ == 0 || h_out_ == 0 || w_out_ == 0) return;

        const uint32_t blockNum = GetBlockNum();
        const uint32_t blockIdx = GetBlockIdx();

        // Each block handles one (n, co) pair
        if (blockIdx >= n_ * c_out_) return;
        uint32_t bn = blockIdx / c_out_;
        uint32_t co = blockIdx % c_out_;

        // Strides
        uint32_t x_stride_n = c_in_ * d_in_ * h_in_ * w_in_;
        uint32_t x_stride_c = d_in_ * h_in_ * w_in_;
        uint32_t x_stride_d = h_in_ * w_in_;
        uint32_t x_stride_h = w_in_;

        uint32_t w_stride_co = c_in_ * kd_ * kh_ * kw_;
        uint32_t w_stride_ci = kd_ * kh_ * kw_;
        uint32_t w_stride_kd = kh_ * kw_;
        uint32_t w_stride_kh = kw_;

        uint32_t y_stride_n = c_out_ * d_out_ * h_out_ * w_out_;
        uint32_t y_stride_co = d_out_ * h_out_ * w_out_;
        uint32_t y_stride_d = h_out_ * w_out_;
        uint32_t y_stride_h = w_out_;

        // Loop over output positions
        for (uint32_t od = 0; od < d_out_; od++) {
            for (uint32_t oh = 0; oh < h_out_; oh++) {
                for (uint32_t ow = 0; ow < w_out_; ow++) {
                    float sum = 0.0f;

                    // Map output position to input positions
                    int32_t id_start = static_cast<int32_t>(od) + pad_d_ - (static_cast<int32_t>(kd_) - 1);
                    int32_t ih_start = static_cast<int32_t>(oh) + pad_h_ - (static_cast<int32_t>(kh_) - 1);
                    int32_t iw_start = static_cast<int32_t>(ow) + pad_w_ - (static_cast<int32_t>(kw_) - 1);

                    for (uint32_t kd = 0; kd < kd_; kd++) {
                        int32_t id = id_start + static_cast<int32_t>(kd) * stride_d_;
                        if (id < 0 || static_cast<uint32_t>(id) >= d_in_) continue;
                        uint32_t id_u = static_cast<uint32_t>(id);

                        for (uint32_t kh = 0; kh < kh_; kh++) {
                            int32_t ih = ih_start + static_cast<int32_t>(kh) * stride_h_;
                            if (ih < 0 || static_cast<uint32_t>(ih) >= h_in_) continue;
                            uint32_t ih_u = static_cast<uint32_t>(ih);

                            for (uint32_t kw = 0; kw < kw_; kw++) {
                                int32_t iw = iw_start + static_cast<int32_t>(kw) * stride_w_;
                                if (iw < 0 || static_cast<uint32_t>(iw) >= w_in_) continue;
                                uint32_t iw_u = static_cast<uint32_t>(iw);

                                for (uint32_t ci = 0; ci < c_in_; ci++) {
                                    float x_val = xGm_[
                                        bn * x_stride_n +
                                        ci * x_stride_c +
                                        id_u * x_stride_d +
                                        ih_u * x_stride_h +
                                        iw_u
                                    ];
                                    float w_val = wGm_[
                                        co * w_stride_co +
                                        ci * w_stride_ci +
                                        kd * w_stride_kd +
                                        kh * w_stride_kh +
                                        kw
                                    ];
                                    sum += x_val * w_val;
                                }
                            }
                        }
                    }

                    // Write output
                    yGm_[
                        bn * y_stride_n +
                        co * y_stride_co +
                        od * y_stride_d +
                        oh * y_stride_h +
                        ow
                    ] = sum;
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_ {0};
    uint32_t c_in_ {0};
    uint32_t c_out_ {0};
    uint32_t d_in_ {0};
    uint32_t h_in_ {0};
    uint32_t w_in_ {0};
    uint32_t d_out_ {0};
    uint32_t h_out_ {0};
    uint32_t w_out_ {0};
    uint32_t kd_ {0};
    uint32_t kh_ {0};
    uint32_t kw_ {0};
    int32_t stride_d_ {1};
    int32_t stride_h_ {2};
    int32_t stride_w_ {2};
    int32_t pad_d_ {0};
    int32_t pad_h_ {0};
    int32_t pad_w_ {0};
    int32_t output_pad_d_ {0};
    int32_t output_pad_h_ {0};
    int32_t output_pad_w_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00317_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00317Custom op;
    op.Init(x, w, y, tiling_data);
    op.Process();
}
'''

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias):
        super().__init__()
        # Create weight parameter with shape (out_channels, in_channels, *kernel_size) for ConvTranspose3d

        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        # bias is not used since bias=False in reference

    def forward(self, x):
        fn = getattr(custom_ops_lib, 'ca6k_00317_custom')
        return fn(x, self.weight, self.stride, self.padding, self.output_padding)
