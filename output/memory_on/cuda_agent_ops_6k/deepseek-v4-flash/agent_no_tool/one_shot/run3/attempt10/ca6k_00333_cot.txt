public:
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalSpatial, uint32_t tileSpatial, uint32_t C, uint32_t size, float alpha, float beta, float k) {
        totalSpatial_ = totalSpatial;
        tileSpatial_ = tileSpatial;
        C_ = C;
        halfSize_ = size / 2;
        alpha_ = alpha;
        beta_ = beta;
        k_ = k;
        // buffers for input and output tiles
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileSpatial_ * C_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileSpatial_ * C_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileSpatial_ * sizeof(float)); // for squares sum
        // Initialize global tensors
        xGm_ = x;
        yGm_ = y;
    }

    __aicore__ void Process() {
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        uint32_t strideSpatial = blockNum * tileSpatial_;
        uint32_t baseSpatial = blockIdx * tileSpatial_;
        while (baseSpatial < totalSpatial_) {
            uint32_t curSpatial = totalSpatial_ - baseSpatial;
            if (curSpatial > tileSpatial_) curSpatial = tileSpatial_;
            // Prefetch input tile: for each spatial index, load all channels
            PrefetchTile(baseSpatial, curSpatial);
            // Compute normalization
            ComputeTile(baseSpatial, curSpatial);
            // Write output
            WriteTile(baseSpatial, curSpatial);
            baseSpatial += strideSpatial;
        }
    }

private:
    void PrefetchTile(uint32_t baseSpatial, uint32_t len) {
        // For each spatial point, load all channels
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        for (uint32_t i = 0; i < len; i++) {
            uint32_t spatialIdx = baseSpatial + i;
            // Compute n, h, w from spatialIdx: spatialIdx = n * H * W + h * W + w
            uint32_t n = spatialIdx / (H_ * W_);
            uint32_t rem = spatialIdx % (H_ * W_);
            uint32_t h = rem / W_;
            uint32_t w = rem % W_;
            for (uint32_t c = 0; c < C_; c++) {
                uint32_t globalOffset = n * C_ * H_ * W_ + c * H_ * W_ + h * W_ + w;
                xLocal(i * C_ + c) = xGm_(globalOffset);
            }
        }
        inQueueX_.EnQue(xLocal);
    }

    void ComputeTile(uint32_t baseSpatial, uint32_t len) {
        LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        LocalTensor<float> sumSq = tmpQueue_.AllocTensor<float>();
        // For each spatial point, compute sum of squares over channels within window
        for (uint32_t i = 0; i < len; i++) {
            for (uint32_t c = 0; c < C_; c++) {
