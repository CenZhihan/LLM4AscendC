kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k05320Custom {
public:
    __aicore__ inline KernelCa6k05320Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t batch, uint32_t inChannels, uint32_t outChannels,
                                uint32_t inDepth, uint32_t inHeight, uint32_t inWidth,
                                uint32_t outDepth, uint32_t outHeight, uint32_t outWidth,
                                uint32_t kernelD, uint32_t kernelH, uint32_t kernelW,
                                int32_t strideD, int32_t strideH, int32_t strideW,
                                int32_t padD, int32_t padH, int32_t padW,
                                int32_t dilationD, int32_t dilationH, int32_t dilationW,
                                bool hasBias)
    {
        batch_ = batch;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        inDepth_ = inDepth;
        inHeight_ = inHeight;
        inWidth_ = inWidth;
        outDepth_ = outDepth;
        outHeight_ = outHeight;
        outWidth_ = outWidth;
        kernelD_ = kernelD;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        padD_ = padD;
        padH_ = padH;
        padW_ = padW;
        dilationD_ = dilationD;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        bGm_.SetGlobalBuffer((__gm__ float*)bias);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t outChannelsPerBlock = (outChannels_ + blockNum - 1) / blockNum;
        uint32_t startK = blockIdx * outChannelsPerBlock;
        uint32_t endK = startK + outChannelsPerBlock;
        if (endK > outChannels_) endK = outChannels_;
        if (startK >= outChannels_) return;

        for (uint32_t n = 0; n < batch_; ++n) {
            for (uint32_t k = startK; k < endK; ++k) {
                for (uint32_t d = 0; d < outDepth_; ++d) {
                    for (uint32_t h = 0; h < outHeight_; ++h) {
                        for (uint32_t w = 0; w < outWidth_; ++w) {
                            float sum = 0.0f;
                            for (uint32_t kd = 0; kd < kernelD_; ++kd) {
                                for (uint32_t kh = 0; kh < kernelH_; ++kh) {
                                    for (uint32_t kw = 0; kw < kernelW_; ++kw) {
                                        int32_t id = static_cast<int32_t>(d) * strideD_ + kd * dilationD_ - padD_;
                                        int32_t ih = static_cast<int32_t>(h) * strideH_ + kh * dilationH_ - padH_;
                                        int32_t iw = static_cast<int32_t>(w) * strideW_ + kw * dilationW_ - padW_;
                                        if (id >= 0 && id < static_cast<int32_t>(inDepth_) &&
                                            ih >= 0 && ih < static_cast<int32_t>(inHeight_) &&
                                            iw >= 0 && iw < static_cast<int32_t>(inWidth_)) {
                                            for (uint32_t c = 0; c < inChannels_; ++c) {
                                                float xval = xGm_[n * inChannels_ * inDepth_ * inHeight_ * inWidth_ +
                                                                  c * inDepth_ * inHeight_ * inWidth_ +
                                                                  id * inHeight_ * inWidth_ +
                                                                  ih * inWidth_ + iw];
                                                float wval = wGm_[k * inChannels_ * kernelD_ * kernelH_ * kernelW_ +
                                                                  c * kernelD_ * kernelH_ * kernelW_ +
                                                                  kd * kernelH_ * kernelW_ +
                                                                  kh * kernelW_ + kw];
                                                sum += xval * wval;
                                            }
                                        }
                                    }
                                }
                            }
                            if (hasBias_) {
                                sum += bGm_[k];
                            }
                            yGm_[n * outChannels_ * outDepth_ * outHeight_ * outWidth_ +
                                 k * outDepth_ * outHeight_ * outWidth_ +
                                 d * outHeight_ * outWidth_ +
                                 h * outWidth_ + w] = sum;
                        }
                    }
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t inChannels_;
    uint32_t outChannels_;
    uint32_t inDepth_;
    uint32_t inHeight_;
    uint32_t inWidth_;
    uint32_t outDepth_;
    uint32_t outHeight_;
    uint32_t outWidth_;
    uint32_t kernelD_;
    uint32_t kernelH_;
    uint32_t kernelW_;
    int32_t strideD_;
    int32_t strideH_;
    int32_t strideW_;
    int32_t padD_;
    int32_t padH_;
    int32_t padW_;
    int32_t dilationD_;
    int32_t dilationH_;
    int32_t dilationW_;
    bool hasBias_;
};

extern "C" __global__ __aicore__ void ca6k_05320_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k05320Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.batch, tiling_data.inChannels, tiling_data.outChannels,
            tiling_data.inDepth, tiling_data.inHeight, tiling_data.inWidth,
            tiling_data.outDepth, tiling_data.outHeight, tiling_data.outWidth,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW,
            tiling_data.dilationD, tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.hasBias);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import math

_FN = 'ca6k_05320_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        # init_inputs: in_channels, out_channels, kernel_size, stride, padding, dilation
        self.in_channels = init_inputs[0]
        self.out_channels = init_inputs[1]
        self.kernel_size = init_inputs[2]  # tuple (D, H, W)
        self.stride = init_inputs[3]
        self.padding = init_inputs[4]
        self.dilation = init_inputs[5]
        self.weight = None
        self.bias = None

    def forward(self, *inputs):
        x = inputs[0]
        if self.weight is None:
            in_channels = x.shape[1]
            # Kaiming uniform initialization for conv3d
            weight = torch.empty(self.out_channels, in_channels, *self.kernel_size, device=x.device)
            torch.nn.init.kaiming_uniform_(weight, a=math.sqrt(5))
            self.weight = torch.nn.Parameter(weight)
            bias = torch.empty(self.out_channels, device=x.device)
            fan_in = in_channels * self.kernel_size[0] * self.kernel_size[1] * self.kernel_size[2]
            bound = 1.0 / math.sqrt(fan_in) if fan_in > 0 else 0
            torch.nn.init.uniform_(bias, -bound, bound)
            self.bias = torch.nn.Parameter(bias)
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias, self.stride, self.padding, self.dilation)
'''
