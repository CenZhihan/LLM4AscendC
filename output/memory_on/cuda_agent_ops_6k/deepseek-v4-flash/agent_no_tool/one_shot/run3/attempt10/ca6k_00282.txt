#include "ca6k00282_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00282CustomTilingData tiling;

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const gert::Shape &origin = inputShape->GetOriginShape();
    uint32_t totalLength = 1;
    for (uint32_t i = 0; i < origin.GetDimNum(); i++) {
        totalLength *= origin.GetDim(i);
    }
    uint32_t spatialSize = 1;
    for (uint32_t i = 2; i < origin.GetDimNum(); i++) {
        spatialSize *= origin.GetDim(i);
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps_ptr = attrs->GetAttrPointer<float>(0);
    float eps = (eps_ptr != nullptr) ? *eps_ptr : 1e-5f;

    uint32_t maxTile = UB_BUDGET_BYTES / (4u * 4u);  // 4 tensors (x, y, sum, sumsq) each float
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = maxTile;
    if (spatialSize < tileLen) {
        uint32_t aligned = AlignDown(spatialSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : spatialSize;
    }

    uint32_t batchChannels = totalLength / spatialSize;
    uint32_t blockDim = batchChannels;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_spatialSize(spatialSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);
    tiling.set_invSpatialSize(1.0f / static_cast<float>(spatialSize));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00282Custom {
public:
    __aicore__ inline KernelCa6k00282Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t spatialSize,
                                uint32_t tileLength, uint32_t blockDim,
                                float eps, float invSpatialSize)
    {
        totalLen_ = totalLength;
        spatialSize_ = spatialSize;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        eps_ = eps;
        invSpatialSize_ = invSpatialSize;
        mean_ = 0.0f;
        var_ = 0.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tempSum_, BUFFER_NUM, 1u * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t featPerBlock = spatialSize_;
        uint32_t featOffset = blockIdx * featPerBlock;
        if (featOffset >= totalLen_) return;

        // First pass: compute sum and sum_sq over spatial dimension
        float sum = 0.0f;
        float sumSq = 0.0f;

        uint32_t tiles = (featPerBlock + tileLen_ - 1) / tileLen_;
        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t offset = featOffset + t * tileLen_;
            uint32_t curLen = (offset + tileLen_ > totalLen_) ? (totalLen_ - offset) : tileLen_;
            if (curLen == 0) break;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> acc1 = tempSum_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);

            // sum part
            AscendC::ReduceAdd(acc1, xLocal, curLen);
            sum += acc1(0);

            // sum of squares part
            AscendC::Mul(xLocal, xLocal, xLocal, curLen);
            AscendC::ReduceAdd(acc1, xLocal, curLen);
            sumSq += acc1(0);

            tempSum_.FreeTensor(acc1);
            inQueueX_.FreeTensor(xLocal);
        }

        float mean = sum * invSpatialSize_;
        float var = sumSq * invSpatialSize_ - mean * mean;
        if (var < 0.0f) var = 0.0f;
        float rstd = 1.0f / sqrt(var + eps_);

        // Second pass: normalize using computed mean and rstd
        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t offset = featOffset + t * tileLen_;
            uint32_t curLen = (offset + tileLen_ > totalLen_) ? (totalLen_ - offset) : tileLen_;
            if (curLen == 0) break;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);

            AscendC::Subs(yLocal, xLocal, mean, curLen);
            AscendC::Mul(yLocal, yLocal, rstd, curLen);

            AscendC::DataCopy(yGm_[offset], yLocal, curLen);

            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tempSum_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t spatialSize_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
    float eps_ {1e-5f};
    float invSpatialSize_ {0.0f};
    float mean_ {0.0f};
    float var_ {0.0f};
};

extern "C" __global__ __aicore__ void ca6k_00282_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00282Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.spatialSize,
            tiling_data.tileLength, tiling_data.blockDim,
            tiling_data.eps, tiling_data.invSpatialSize);
    op.Process();
}
