kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;
constexpr float NEG_INF = -1e10f;

class KernelCa6k00387Custom {
public:
    __aicore__ inline KernelCa6k00387Custom() {}

    __aicore__ inline void Init(GM_ADDR logProbs, GM_ADDR targets,
                                GM_ADDR inputLengths, GM_ADDR targetLengths,
                                GM_ADDR loss,
                                uint32_t batchSize, float batchSizeFloat,
                                uint32_t maxTime,
                                uint32_t maxTargetLen, uint32_t numClasses,
                                int32_t blank, uint8_t reductionMean)
    {
        batchSize_ = batchSize;
        batchSizeFloat_ = batchSizeFloat;
        maxTime_ = maxTime;
        maxTargetLen_ = maxTargetLen;
        numClasses_ = numClasses;
        blank_ = blank;
        reductionMean_ = reductionMean;

        logProbsGm_.SetGlobalBuffer((__gm__ float*)logProbs, maxTime * batchSize * numClasses);
        targetsGm_.SetGlobalBuffer((__gm__ int32_t*)targets, batchSize * maxTargetLen);
        inputLengthsGm_.SetGlobalBuffer((__gm__ int32_t*)inputLengths, batchSize);
        targetLengthsGm_.SetGlobalBuffer((__gm__ int32_t*)targetLengths, batchSize);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        // Use a single pipe buffer for target copy and temporary row data
        // Buffer size = (maxTargetLen + numClasses + 2) floats => enough for int32 and float
        uint32_t bufSize = (maxTargetLen_ + numClasses_ + 2) * sizeof(float);
        pipe_.InitBuffer(tmpBuf_, 1, bufSize);
    }

    __aicore__ inline void Process()
    {
        // Cast buffer address to float* for reuse
        volatile float* bufPtr = reinterpret_cast<volatile float*>(tmpBuf_.Get());

        float totalLoss = 0.0f;

        for (uint32_t batchIdx = 0; batchIdx < batchSize_; ++batchIdx) {
            int32_t T = inputLengthsGm_.GetValue(batchIdx);
            int32_t L = targetLengthsGm_.GetValue(batchIdx);
            if (T <= 0 || L <= 0) continue;

            // Copy target sequence into buffer (first L ints as float to reuse buffer)
            __gm__ int32_t* targetBase = targetsGm_.GetPtr() + batchIdx * maxTargetLen_;
            for (int32_t i = 0; i < L; ++i) {
                // use float* buffer, store int as float bits (will cast back when reading)
                ((volatile int32_t*)bufPtr)[i] = targetBase[i];
            }
            volatile int32_t* tgt = (volatile int32_t*)bufPtr;

            // Build extended target sequence (U = 2*L+1 <= 61)
            int32_t extended[61];
            extended[0] = blank_;
            for (int32_t i = 0; i < L; ++i) {
                extended[2 * i + 1] = tgt[i];
                extended[2 * i + 2] = blank_;
            }
            int32_t U = 2 * L + 1;

            // Copy first row of logProbs into buffer after targets
            volatile float* logRow = bufPtr + maxTargetLen_; // use region after targets
            __gm__ float* logProbsBase = logProbsGm_.GetPtr();
            for (uint32_t i = 0; i < numClasses_; ++i) {
                logRow[i] = logProbsBase[batchIdx * numClasses_ + i];
            }

            // Initialize alpha at t=0
            float alphaPrev[61];
            float alphaCurr[61];
            alphaCurr[0] = logRow[extended[0]];
            for (int32_t u = 1; u < U; ++u) {
                if (u == 1) {
                    alphaCurr[1] = logRow[extended[1]];
                } else {
                    alphaCurr[u] = NEG_INF;
                }
            }

            // Loop over time steps
            for (int32_t t = 1; t < T; ++t) {
                // Copy alphaCurr to alphaPrev
                for (int32_t u = 0; u < U; ++u) {
                    alphaPrev[u] = alphaCurr[u];
                }

                // Copy current time step row into local buffer
                uint32_t gmOffset = t * batchSize_ * numClasses_ + batchIdx * numClasses_;
                for (uint32_t i = 0; i < numClasses_; ++i) {
                    logRow[i] = logProbsBase[gmOffset + i];
                }

                // Update alphaCurr for this time step
                for (int32_t u = 0; u < U; ++u) {
                    float logProb = logRow[extended[u]];
                    float best = alphaPrev[u] + logProb;
                    if (u > 0) {
                        float val = alphaPrev[u - 1] + logProb;
                        if (val > best) best = val;
                    }
                    if (u > 1 && extended[u] != blank_ && extended[u] != extended[u - 2]) {
                        float val = alphaPrev[u - 2] + logProb;
                        if (val > best) best = val;
                    }
                    alphaCurr[u] = best;
                }
            }

            // Compute loss for this sample using log-sum-exp
            float val1 = alphaCurr[U - 1];
            float val2 = alphaCurr[U - 2];
            float maxVal = (val1 > val2) ? val1 : val2;

            // Compute exp(val1 - maxVal) + exp(val2 - maxVal) using scalar operations
            // Use built-in exp via scalar function
            float expSum = 0.0f;
            // Use a simple polynomial approximation? Use scalar exp from math? Not available.
            // Use AscendC intrinsic for single element exp if possible, but we cannot here.
            // We'll compute using a loop with 2 elements via local tensor from buffer
            // Allocate small local tensor from buffer tail
            volatile float* expBuf = bufPtr + maxTargetLen_ + numClasses_;
            expBuf[0] = val1 - maxVal;
            expBuf[1] = val2 - maxVal;

            // Use AscendC::Exp only if we have a proper LocalTensor; workaround:
            // Create a LocalTensor from buffer address
            AscendC::LocalTensor<float> tmpExp;
            tmpExp.SetValuePtr(expBuf, 2); // careful: not standard, but allowed via SetValuePtr
            AscendC::Exp(tmpExp, tmpExp, 2);
            float sumExp = tmpExp.GetValue(0) + tmpExp.GetValue(1);
            tmpExp.SetValue(0, sumExp);
            AscendC::Log(tmpExp, tmpExp, 1);
            float logSumExp = tmpExp.GetValue(0);

            float lossSample = -(maxVal + logSumExp);
            totalLoss += lossSample;
        }

        if (reductionMean_) {
            totalLoss /= batchSizeFloat_;
        }
        lossGm_.SetValue(0, totalLoss);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECIN> tmpBuf_;

    AscendC::GlobalTensor<float> logProbsGm_;
    AscendC::GlobalTensor<int32_t> targetsGm_;
    AscendC::GlobalTensor<int32_t> inputLengthsGm_;
    AscendC::GlobalTensor<int32_t> targetLengthsGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t batchSize_;
    float batchSizeFloat_;
    uint32_t maxTime_;
    uint32_t maxTargetLen_;
    uint32_t numClasses_;
    int32_t blank_;
    uint8_t reductionMean_;
};

extern "C" __global__ __aicore__ void ca6k00387_custom(GM_ADDR logProbs, GM_ADDR targets,
                                                         GM_ADDR inputLengths, GM_ADDR targetLengths,
                                                         GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00387Custom op;
    op.Init(logProbs, targets, inputLengths, targetLengths, loss,
            tiling_data.batchSize, tiling_data.batchSizeFloat,
            tiling_data.maxTime,
            tiling_data.maxTargetLen, tiling_data.numClasses,
            tiling_data.blank, tiling_data.reductionMean);
    op.Process();
}
