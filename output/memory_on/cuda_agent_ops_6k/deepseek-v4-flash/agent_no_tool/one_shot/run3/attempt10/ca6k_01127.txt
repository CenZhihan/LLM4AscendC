kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01127Custom {
public:
    __aicore__ inline KernelCa6k01127Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR target, GM_ADDR y,
                                uint32_t batchSize, uint32_t featureSize)
    {
        batchSize_ = batchSize;
        featureSize_ = featureSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * featureSize_);
        targetGm_.SetGlobalBuffer((__gm__ int64_t*)target, batchSize_ * featureSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, featureSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueT_, BUFFER_NUM, featureSize_ * sizeof(int64_t));
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || featureSize_ == 0) return;

        float totalLoss = 0.0f;
        AscendC::LocalTensor<float> lossScalar = tmpQueue_.AllocTensor<float>();

        for (uint32_t s = 0; s < batchSize_; ++s) {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<int64_t> tLocal = inQueueT_.AllocTensor<int64_t>();

            uint32_t offset = s * featureSize_;
            AscendC::DataCopy(xLocal, xGm_[offset], featureSize_);
            AscendC::DataCopy(tLocal, targetGm_[offset], featureSize_);

            // Count positive labels
            uint32_t posCount = 0;
            for (uint32_t j = 0; j < featureSize_; ++j) {
                if (tLocal.GetValue(j) != -1) {
                    posCount++;
                }
            }

            if (posCount > 1) {
                float sum = 0.0f;
                for (uint32_t p = 0; p < featureSize_; ++p) {
                    int64_t tp = tLocal.GetValue(p);
                    if (tp == -1) continue;
                    float xp = xLocal.GetValue(p);
                    for (uint32_t q = 0; q < featureSize_; ++q) {
                        if (p == q) continue;
                        int64_t tq = tLocal.GetValue(q);
                        if (tq == -1) continue;
                        float margin = 1.0f - (xp - xLocal.GetValue(q));
                        if (margin > 0.0f) sum += margin;
                    }
                }
                float pairs = static_cast<float>(posCount * (posCount - 1));
                totalLoss += sum / pairs;
            }

            inQueueX_.FreeTensor(xLocal);
            inQueueT_.FreeTensor(tLocal);
        }

        float finalLoss = totalLoss / static_cast<float>(batchSize_);
        lossScalar.SetValue(0, finalLoss);
        AscendC::DataCopy(yGm_, lossScalar, 1);
        tmpQueue_.FreeTensor(lossScalar);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueT_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> targetGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t featureSize_ {0};
};

extern "C" __global__ __aicore__ void ca6k_01127_custom(GM_ADDR x, GM_ADDR target, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01127Custom op;
    op.Init(x, target, y, tiling_data.batchSize, tiling_data.featureSize);
    op.Process();
}
