kernel_src="""

#include "kernel_operator.h"
#include "ca6k04617_custom_tiling.h"
#include <cfloat>

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04617Custom {
public:
    __aicore__ inline KernelCa6k04617Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength, uint32_t blockDim,
                                uint32_t n, uint32_t cIn, uint32_t hIn, uint32_t wIn,
                                uint32_t cOut, uint32_t hOut, uint32_t wOut,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding,
                                uint32_t poolKernelSize, uint32_t poolStride)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        n_ = n; cIn_ = cIn; hIn_ = hIn; wIn_ = wIn;
        cOut_ = cOut; hOut_ = hOut; wOut_ = wOut;
        kernelSize_ = kernelSize; stride_ = stride; padding_ = padding;
        poolKernelSize_ = poolKernelSize; poolStride_ = poolStride;
        hOutPool_ = (hOut_ - poolKernelSize_) / poolStride_ + 1;
        wOutPool_ = (wOut_ - poolKernelSize_) / poolStride_ + 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n * cIn * hIn * wIn);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, cOut * cIn * kernelSize * kernelSize);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, cOut);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t offset = blockIdx * tileLen_;
        if (offset >= totalLen_) return;

        while (offset < totalLen_) {
            uint32_t workLen = totalLen_ - offset;
            if (workLen > tileLen_) workLen = tileLen_;

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            for (uint32_t i = 0; i < workLen; i++) {
                uint32_t linearIdx = offset + i;
                uint32_t batch = linearIdx / (cOut_ * hOutPool_ * wOutPool_);
                uint32_t rem = linearIdx % (cOut_ * hOutPool_ * wOutPool_);
                uint32_t co = rem / (hOutPool_ * wOutPool_);
                uint32_t spatialRem = rem % (hOutPool_ * wOutPool_);
                uint32_t hp = spatialRem / wOutPool_;
                uint32_t wp = spatialRem % wOutPool_;

                float result = -FLT_MAX;
                for (uint32_t dy = 0; dy < poolKernelSize_; dy++) {
                    for (uint32_t dx = 0; dx < poolKernelSize_; dx++) {
                        uint32_t convH = hp * poolStride_ + dy;
                        uint32_t convW = wp * poolStride_ + dx;
                        if (convH >= hOut_ || convW >= wOut_) continue;

                        float convVal = 0.0f;
                        for (uint32_t ci = 0; ci < cIn_; ci++) {
                            for (uint32_t kh = 0; kh < kernelSize_; kh++) {
                                int32_t inh = static_cast<int32_t>(convH * stride_ + kh) - static_cast<int32_t>(padding_);
                                if (inh < 0 || inh >= static_cast<int32_t>(hIn_)) continue;
                                for (uint32_t kw = 0; kw < kernelSize_; kw++) {
                                    int32_t inw = static_cast<int32_t>(convW * stride_ + kw) - static_cast<int32_t>(padding_);
                                    if (inw < 0 || inw >= static_cast<int32_t>(wIn_)) continue;
                                    uint32_t xIdx = batch * cIn_ * hIn_ * wIn_ +
                                                    ci * hIn_ * wIn_ +
                                                    static_cast<uint32_t>(inh) * wIn_ + static_cast<uint32_t>(inw);
                                    uint32_t wIdx = co * cIn_ * kernelSize_ * kernelSize_ +
                                                    ci * kernelSize_ * kernelSize_ +
                                                    kh * kernelSize_ + kw;
                                    convVal += xGm_.GetValue(xIdx) * wGm_.GetValue(wIdx);
                                }
                            }
                        }
                        convVal += bGm_.GetValue(co);
                        if (convVal < 0.0f) convVal = 0.0f;
                        if (convVal > result) result = convVal;
                    }
                }
                yLocal.SetValue(i, result);
            }

            AscendC::DataCopy(yGm_[offset], yLocal, workLen);
            outQueueY_.FreeTensor(yLocal);

            offset += strideElems;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t n_, cIn_, hIn_, wIn_;
    uint32_t cOut_, hOut_, wOut_;
    uint32_t kernelSize_, stride_, padding_;
    uint32_t poolKernelSize_, poolStride_;
    uint32_t hOutPool_, wOutPool_;
};

extern "C" __global__ __aicore__ void ca6k_04617_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04617Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim,
            tiling_data.n, tiling_data.cIn, tiling_data.hIn, tiling_data.wIn,
            tiling_data.cOut, tiling_data.hOut, tiling_data.wOut,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.poolKernelSize, tiling_data.poolStride);
    op.Process();
}
