#include "ca6k00363_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/types.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
  return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k00363CustomTilingData tiling;
  const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
  uint32_t batchSize = static_cast<uint32_t>(x_shape.GetDim(0));
  uint32_t inputChannels = static_cast<uint32_t>(x_shape.GetDim(1));
  uint32_t inputHeight = static_cast<uint32_t>(x_shape.GetDim(2));
  uint32_t inputWidth = static_cast<uint32_t>(x_shape.GetDim(3));

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  const int32_t *upscale_factor = attrs->GetAttrPointer<int32_t>(0);
  uint32_t r = static_cast<uint32_t>(*upscale_factor);

  uint32_t outC = inputChannels * r * r;
  uint32_t outH = inputHeight / r;
  uint32_t outW = inputWidth / r;
  uint32_t totalLength = batchSize * outC * outH * outW;

  uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = maxTile;
  if (totalLength < tileLen) {
    uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
    tileLen = (aligned > 0) ? aligned : totalLength;
  }

  uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  tiling.SetBatchSize(batchSize);
  tiling.SetInputChannels(inputChannels);
  tiling.SetInputHeight(inputHeight);
  tiling.SetInputWidth(inputWidth);
  tiling.SetUpscaleFactor(r);
  tiling.SetTotalLength(totalLength);
  tiling.SetTileLength(tileLen);
  tiling.SetBlockDim(blockDim);

  context->SetBlockDim(blockDim);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"
#include "ca6k00363_custom_tiling.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;

class KernelPixelUnshuffle {
public:
  __aicore__ inline KernelPixelUnshuffle() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                              uint32_t batchSize, uint32_t inputChannels,
                              uint32_t inputHeight, uint32_t inputWidth,
                              uint32_t upscaleFactor, uint32_t totalLength,
                              uint32_t tileLength) {
    n_ = batchSize;
    c_ = inputChannels;
    h_ = inputHeight;
    w_ = inputWidth;
    r_ = upscaleFactor;
    totalLen_ = totalLength;
    tileLen_ = tileLength;

    xGm_.SetGlobalBuffer((__gm__ float *)x);
    yGm_.SetGlobalBuffer((__gm__ float *)y);

    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;

    uint32_t outC = c_ * r_ * r_;
    uint32_t outH = h_ / r_;
    uint32_t outW = w_ / r_;

    uint32_t blockNum = GetBlockNum();
    uint32_t blockIdx = GetBlockIdx();

    uint32_t stride = blockNum * tileLen_;
    uint32_t start = blockIdx * tileLen_;
    if (start >= totalLen_) return;

    for (uint32_t outOffset = start; outOffset < totalLen_; outOffset += stride) {
      uint32_t len = tileLen_;
      if (outOffset + len > totalLen_) len = totalLen_ - outOffset;

      LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
      for (uint32_t i = 0; i < len; i++) {
        uint32_t outIdx = outOffset + i;
        uint32_t n_idx = outIdx / (outC * outH * outW);
        uint32_t tmp = outIdx % (outC * outH * outW);
        uint32_t oc = tmp / (outH * outW);
        uint32_t oh = (tmp % (outH * outW)) / outW;
        uint32_t ow = tmp % outW;
        uint32_t ic = oc / (r_ * r_);
        uint32_t pixel = oc % (r_ * r_);
        uint32_t ph = pixel / r_;
        uint32_t pw = pixel % r_;
        uint32_t inIdx = n_idx * (c_ * h_ * w_) + ic * (h_ * w_) + (oh * r_ + ph) * w_ + (ow * r_ + pw);
        xLocal.SetValue(i, xGm_.GetValue(inIdx));
      }
      inQueueX_.EnQue(xLocal);

      LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
      xLocal = inQueueX_.DeQue<float>();
      for (uint32_t i = 0; i < len; i++) {
        yLocal.SetValue(i, xLocal.GetValue(i));
      }
      outQueueY_.EnQue(yLocal);
      inQueueX_.FreeTensor(xLocal);

      yLocal = outQueueY_.DeQue<float>();
      for (uint32_t i = 0; i < len; i++) {
        yGm_.SetValue(outOffset + i, yLocal.GetValue(i));
      }
      outQueueY_.FreeTensor(yLocal);
    }
  }

private:
  TPipe pipe_;
  TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
  TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
  GlobalTensor<float> xGm_;
  GlobalTensor<float> yGm_;
  uint32_t n_ {0};
  uint32_t c_ {0};
  uint32_t h_ {0};
  uint32_t w_ {0};
  uint32_t r_ {2};
  uint32_t totalLen_ {0};
  uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00363_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelPixelUnshuffle op;
  op.Init(x, y,
          tiling_data.batchSize, tiling_data.inputChannels,
          tiling_data.inputHeight, tiling_data.inputWidth,
          tiling_data.upscaleFactor, tiling_data.totalLength,
          tiling_data.tileLength);
  op.Process();
}
'''
