static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k05320CustomTilingData tiling;

    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::StorageShape *wShape = context->GetInputShape(1);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    uint32_t N = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t C = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    uint32_t D = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
    uint32_t H = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));
    uint32_t W = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(4));

    uint32_t K = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(0));
    uint32_t kD = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(2));
    uint32_t kH = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(3));
    uint32_t kW = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(4));

    const int64_t *stride = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *dilation = attrs->GetAttrPointer<int64_t>(2);

    int32_t sD = static_cast<int32_t>(stride[0]);
    int32_t sH = static_cast<int32_t>(stride[1]);
    int32_t sW = static_cast<int32_t>(stride[2]);
    int32_t pD = static_cast<int32_t>(padding[0]);
    int32_t pH = static_cast<int32_t>(padding[1]);
    int32_t pW = static_cast<int32_t>(padding[2]);
    int32_t dD = static_cast<int32_t>(dilation[0]);
    int32_t dH = static_cast<int32_t>(dilation[1]);
    int32_t dW = static_cast<int32_t>(dilation[2]);

    int32_t outD = static_cast<int32_t>((D + 2 * pD - dD * (kD - 1) - 1) / sD + 1);
    int32_t outH = static_cast<int32_t>((H + 2 * pH - dH * (kH - 1) - 1) / sH + 1);
    int32_t outW = static_cast<int32_t>((W + 2 * pW - dW * (kW - 1) - 1) / sW + 1);

    bool hasBias = (context->GetInputShape(2) != nullptr);

    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_batch(N);
    tiling.set_inChannels(C);
    tiling.set_outChannels(K);
    tiling.set_inDepth(D);
    tiling.set_inHeight(H);
    tiling.set_inWidth(W);
    tiling.set_outDepth(outD);
    tiling.set_outHeight(outH);
    tiling.set_outWidth(outW);
    tiling.set_kernelD(kD);
    tiling.set_kernelH(kH);
    tiling.set_kernelW(kW);
    tiling.set_strideD(sD);
    tiling.set_strideH(sH);
    tiling.set_strideW(sW);
    tiling.set_padD(pD);
    tiling.set_padH(pH);
    tiling.set_padW(pW);
    tiling.set_dilationD(dD);
    tiling.set_dilationH(dH);
    tiling.set_dilationW(dW);
    tiling.set_hasBias(hasBias);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
