class KernelCa6k00535Custom {
public:
    __aicore__ inline KernelCa6k00535Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t blockDim)
    {
        rows_ = rows;
        cols_ = cols;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, rows_ * cols_);

        // Allocate local buffers for one column
        pipe_.InitBuffer(inBufX_, 1, rows_ * sizeof(float));
        pipe_.InitBuffer(outBufY_, 1, rows_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t colsPerBlock = (cols_ + blockDim_ - 1) / blockDim_;
        uint32_t startCol = blockIdx * colsPerBlock;
        uint32_t endCol = startCol + colsPerBlock;
        if (endCol > cols_) endCol = cols_;

        for (uint32_t col = startCol; col < endCol; col++) {
            // Copy column x to local
            LocalTensor<float> xLocal = inBufX_.Get<float>();
            DataCopy(xLocal, xGm_[col * rows_], rows_);
            
            // Compute cumulative max
            LocalTensor<float> yLocal = outBufY_.Get<float>();
            float curMax = -3.4028235e+38f;
            for (uint32_t i = 0; i < rows_; i++) {
                float val = xLocal(i);
                if (val > curMax) curMax = val;
                yLocal(i) = curMax;
            }
            
            // Copy back to global
            DataCopy(yGm_[col * rows_], yLocal, rows_);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> inBufX_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> outBufY_;
