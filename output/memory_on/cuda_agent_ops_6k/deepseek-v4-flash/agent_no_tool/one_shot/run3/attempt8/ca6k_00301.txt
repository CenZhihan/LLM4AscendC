#include "ca6k00301_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00301CustomTilingData tiling;

    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::StorageShape *y_storage = context->GetInputShape(1);
    const gert::Shape &x_shape = x_storage->GetOriginShape();
    const gert::Shape &y_shape = y_storage->GetOriginShape();
    if (x_shape.GetDimNum() != 3 || y_shape.GetDimNum() != 3)
        return ge::GRAPH_FAILED;

    uint32_t batch = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t M = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t K = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t N = static_cast<uint32_t>(y_shape.GetDim(2));

    // Compute conservative tile sizes that fit in UB for float
    uint32_t available = (UB_BUDGET_BYTES * 3 / 4) / sizeof(float); // leave margin
    uint32_t tileM = 64;
    uint32_t tileN = 64;
    uint32_t tileK = 128;
    while (tileM * tileK + tileK * tileN + tileM * tileN > available && tileM > 16) {
        tileM /= 2; tileN /= 2; tileK /= 2;
    }
    if (tileM < 16) { tileM = 16; tileN = 16; tileK = 16; }

    tiling.set_batch(batch);
    tiling.set_m(M);
    tiling.set_n(N);
    tiling.set_k(K);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);

    uint32_t blockDim = batch; // one block per batch element
    context->SetBlockDim(blockDim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *y_shape = context->GetInputShape(1);
    gert::Shape *z_shape = context->GetOutputShape(0);
    int64_t dims[3] = {x_shape->GetDim(0), x_shape->GetDim(1), y_shape->GetDim(2)};
    z_shape->SetDimNum(3);
    for (int64_t i = 0; i < 3; ++i)
        z_shape->SetDim(i, dims[i]);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

}  // namespace ge

kernel_src="""

#include "kernel_operator.h"
#include "matmul_intf.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00301Custom {
public:
    __aicore__ inline KernelCa6k00301Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t batch, uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        batch_ = batch;
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileK_ = tileK;
        tileN_ = tileN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * M * K);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * K * N);
        zGm_.SetGlobalBuffer((__gm__ float*)z, batch * M * N);
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        if (blockIdx >= batch_) return;

        uint32_t b = blockIdx;

        // Set pointers for this batch
        float *xBatch = xGm_.GetPtr() + b * M_ * K_;
        float *yBatch = yGm_.GetPtr() + b * K_ * N_;
        float *zBatch = zGm_.GetPtr() + b * M_ * N_;

        // Create MatMul object for floating point
        MatMul<float, float, float> mm;
        mm.SetMatrixA(M_, K_, false, xBatch, M_ * K_);
        mm.SetMatrixB(K_, N_, false, yBatch, K_ * N_);
        mm.SetMatrixC(M_, N_, false, zBatch, M_ * N_);
        mm.SetConfig(1); // normal matmul
        mm.SetTiling(TPosition::VECCALC, tileM_, tileK_, tileN_);
        mm.IterateAll();
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> zGm_;

    uint32_t batch_ {0};
    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {0};
    uint32_t tileK_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void ca6k00301_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00301Custom op;
    op.Init(x, y, z,
            tiling_data.batch, tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
