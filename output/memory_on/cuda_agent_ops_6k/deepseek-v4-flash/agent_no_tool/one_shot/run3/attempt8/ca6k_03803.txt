#include "kernel_operator.h"
#include "ca6k_03803_custom_tiling.h"
#include <cstdio>

class KernelCa6k03803Custom {
public:
    __aicore__ inline KernelCa6k03803Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t N, uint32_t C_in,
                                uint32_t C_out, uint32_t H_in, uint32_t W_in,
                                uint32_t H_out, uint32_t W_out, uint32_t K_h,
                                uint32_t K_w, uint32_t stride, uint32_t padding,
                                uint32_t output_padding, uint32_t dilation_h,
                                uint32_t dilation_w, uint32_t weightSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        N_ = N; C_in_ = C_in; C_out_ = C_out;
        H_in_ = H_in; W_in_ = W_in;
        H_out_ = H_out; W_out_ = W_out;
        K_h_ = K_h; K_w_ = K_w;
        stride_ = stride; padding_ = padding;
        output_padding_ = output_padding;
        dilation_h_ = dilation_h; dilation_w_ = dilation_w;
        weightSize_ = weightSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, weightSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t totalTiles = (totalLen_ + tileLen_ - 1) / tileLen_;
        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            uint32_t start = tileIdx * tileLen_;
            uint32_t len = (tileLen_ <= totalLen_ - start) ? tileLen_ : (totalLen_ - start);
            if (len == 0) continue;
            ComputeTile(start, len);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t start, uint32_t len)
    {
        for (uint32_t i = 0; i < len; i++) {
            uint32_t idx = start + i;
            uint32_t n = idx / (C_out_ * H_out_ * W_out_);
            uint32_t rem = idx % (C_out_ * H_out_ * W_out_);
            uint32_t c_out = rem / (H_out_ * W_out_);
            uint32_t hw = rem % (H_out_ * W_out_);
            uint32_t h_out = hw / W_out_;
            uint32_t w_out = hw % W_out_;

            float sum = 0.0f;
            for (uint32_t ic = 0; ic < C_in_; ic++) {
                for (uint32_t kh = 0; kh < K_h_; kh++) {
                    int32_t diff_h = static_cast<int32_t>(h_out) + static_cast<int32_t>(padding_)
                                     - static_cast<int32_t>(dilation_h_ * kh);
                    if (diff_h < 0) continue;
                    if (diff_h % static_cast<int32_t>(stride_) != 0) continue;
                    int32_t h_in = diff_h / static_cast<int32_t>(stride_);
                    if (static_cast<uint32_t>(h_in) >= H_in_) continue;

                    for (uint32_t kw = 0; kw < K_w_; kw++) {
                        int32_t diff_w = static_cast<int32_t>(w_out) + static_cast<int32_t>(padding_)
                                         - static_cast<int32_t>(dilation_w_ * kw);
                        if (diff_w < 0) continue;
                        if (diff_w % static_cast<int32_t>(stride_) != 0) continue;
                        int32_t w_in = diff_w / static_cast<int32_t>(stride_);
                        if (static_cast<uint32_t>(w_in) >= W_in_) continue;

                        uint32_t xIdx = ((n * C_in_ + ic) * H_in_ + static_cast<uint32_t>(h_in)) * W_in_ + static_cast<uint32_t>(w_in);
                        uint32_t wIdx = ((ic * C_out_ + c_out) * K_h_ + kh) * K_w_ + kw;
                        float xVal = xGm_.GetValue(xIdx);
                        float wVal = wGm_.GetValue(wIdx);
                        sum += xVal * wVal;
                    }
                }
            }
            yGm_.SetValue(idx, sum);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_, tileLen_, blockDim_;
    uint32_t N_, C_in_, C_out_, H_in_, W_in_, H_out_, W_out_;
    uint32_t K_h_, K_w_, stride_, padding_, output_padding_;
    uint32_t dilation_h_, dilation_w_, weightSize_;
};

extern "C" __global__ __aicore__ void ca6k_03803_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    KernelCa6k03803Custom op;
    op.Init(x, weight, y,
            td.totalLength, td.tileLength, td.blockDim,
            td.N, td.C_in, td.C_out, td.H_in, td.W_in,
            td.H_out, td.W_out, td.K_h, td.K_w,
            td.stride, td.padding, td.output_padding,
            td.dilation_h, td.dilation_w, td.weightSize);
    op.Process();
}

import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0,
                 output_padding=0, dilation=(1,1)):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.dilation = dilation
        self.conv_transpose = nn.ConvTranspose2d(
            in_channels, out_channels, kernel_size,
            stride=stride, padding=padding,
            output_padding=output_padding, dilation=dilation,
            bias=False
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        w = self.conv_transpose.weight
        return custom_ops_lib.ca6k_03803_custom(
            x, w,
            self.in_channels, self.out_channels,
            self.kernel_size[0], self.kernel_size[1],
            self.stride, self.padding, self.output_padding,
            self.dilation[0], self.dilation[1]
