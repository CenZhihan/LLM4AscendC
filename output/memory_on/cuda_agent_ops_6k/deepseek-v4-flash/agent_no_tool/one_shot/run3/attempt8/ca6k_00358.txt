    uint32_t outW = static_cast<uint32_t>(outputWidth);
    uint32_t totalOutput = n * c * outH * outW;

    uint32_t blockDim = totalOutput / 4096;
    if (blockDim < MIN_BLOCK_DIM) blockDim = MIN_BLOCK_DIM;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;

    tiling.set_inW(w);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_blockDim(blockDim);
