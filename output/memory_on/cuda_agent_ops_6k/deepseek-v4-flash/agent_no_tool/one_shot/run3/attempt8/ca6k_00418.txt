kernel_src="""

#include "kernel_operator.h"
#include "ca6k_00418_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLpPool2d {
public:
    __aicore__ inline KernelLpPool2d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElems,
                                uint32_t inputN, uint32_t inputC,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t outputH, uint32_t outputW,
                                uint32_t kernelSize, uint32_t stride,
                                float normType, bool ceilMode) {
        totalElems_ = totalOutputElems;
        n_ = inputN; c_ = inputC;
        h_ = inputH; w_ = inputW;
        oh_ = outputH; ow_ = outputW;
        ks_ = kernelSize; st_ = stride;
        ceil_ = ceilMode;
        normType_ = normType;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(tmpBuf_, 1 * sizeof(float));
        pipe_.InitBuffer(sqrtBuf_, 1 * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalElems_ == 0) return;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        for (uint32_t idx = blockIdx; idx < totalElems_; idx += blockNum) {
            uint32_t tmp = idx;
            uint32_t ow_idx = tmp % ow_;
            tmp /= ow_;
            uint32_t oh_idx = tmp % oh_;
            tmp /= oh_;
            uint32_t cc = tmp % c_;
            uint32_t nn = tmp / c_;
            compute_one(nn, cc, oh_idx, ow_idx, idx);
        }
    }

private:
    __aicore__ inline void compute_one(uint32_t nn, uint32_t cc,
                                       uint32_t oh_idx, uint32_t ow_idx,
                                       uint32_t outIdx) {
        float sum = 0.0f;
        uint32_t baseH = oh_idx * st_;
        uint32_t baseW = ow_idx * st_;
        for (uint32_t kh = 0; kh < ks_; ++kh) {
            uint32_t inH = baseH + kh;
            if (inH >= h_) continue;
            for (uint32_t kw = 0; kw < ks_; ++kw) {
                uint32_t inW = baseW + kw;
                if (inW >= w_) continue;
                uint32_t inIdx = ((nn * c_ + cc) * h_ + inH) * w_ + inW;
                float val = xGm_.GetValue(inIdx);
                sum += val * val;
            }
        }

        AscendC::LocalTensor<float> src = tmpBuf_.Get<float>();
        src.SetValue(0, sum);
        AscendC::LocalTensor<float> dst = sqrtBuf_.Get<float>();
        AscendC::Sqrt(dst, src, 1);
        float yVal = dst.GetValue(0);
        yGm_.SetValue(outIdx, yVal);
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalElems_;
    uint32_t n_, c_, h_, w_;
    uint32_t oh_, ow_;
    uint32_t ks_, st_;
    bool ceil_;
    float normType_;

    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> sqrtBuf_;
};

extern "C" __global__ __aicore__ void ca6k_00418_custom(GM_ADDR x, GM_ADDR y,
                                                       GM_ADDR workspace,
                                                       GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLpPool2d op;
    op.Init(x, y,
            tiling_data.totalOutputElems,
            tiling_data.inputN, tiling_data.inputC,
            tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputH, tiling_data.outputW,
            tiling_data.kernelSize, tiling_data.stride,
            tiling_data.normType, tiling_data.blockDim);
    op.Process();
}

at::Tensor lp_pool2d_impl_npu(const at::Tensor& self,
                              double norm_type,
                              int64_t kernel_size,
                              int64_t stride,
                              bool ceil_mode) {
    int64_t N = self.size(0);
    int64_t C = self.size(1);
    int64_t H = self.size(2);
    int64_t W = self.size(3);
    int64_t H_out, W_out;
    if (ceil_mode) {
        H_out = static_cast<int64_t>(std::ceil((H - kernel_size) / static_cast<double>(stride))) + 1;
        W_out = static_cast<int64_t>(std::ceil((W - kernel_size) / static_cast<double>(stride))) + 1;
    } else {
        H_out = static_cast<int64_t>(std::floor((H - kernel_size) / static_cast<double>(stride))) + 1;
        W_out = static_cast<int64_t>(std::floor((W - kernel_size) / static_cast<double>(stride))) + 1;
    }
    at::Tensor result = at::empty({N, C, H_out, W_out}, self.options());
    EXEC_NPU_CMD(aclnnCa6k00418Custom, self, norm_type, kernel_size, stride, ceil_mode, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_00418_custom", &lp_pool2d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
