    int32_t scale_factor = attrs->GetAttrPointer<int32_t>(0)[0];

    uint32_t scale = static_cast<uint32_t>(scale_factor);
    uint32_t C_out = C_in / (scale * scale);
    uint32_t H_out = H_in * scale;
    uint32_t W_out = W_in * scale;

    context->SetBlockDim(N);

    tiling.set_N(N);
    tiling.set_C_in(C_in);
    tiling.set_H_in(H_in);
    tiling.set_W_in(W_in);
    tiling.set_scaleFactor(scale);
    tiling.set_C_out(C_out);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_blockDim(N);
