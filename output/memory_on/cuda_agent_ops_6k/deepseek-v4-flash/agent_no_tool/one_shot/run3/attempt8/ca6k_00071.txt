kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00071Custom {
public:
    __aicore__ inline KernelCa6k00071Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t numRows, uint32_t numCols,
                                uint32_t tileLength, int32_t dim)
    {
        totalLen_ = totalLength;
        N_ = numRows;
        C_ = numCols;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, C_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, BUFFER_NUM, C_ * sizeof(float));

        // Initialize accumulator to zero
        AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();
        AscendC::Duplicate(acc, 0.0f, C_);
        accQueue_.EnQue(acc);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0u || C_ == 0u) return;
        if (dim_ != 0) return; // only dim 0 implemented

        for (uint32_t i = 0u; i < N_; ++i) {
            uint32_t offset = i * C_;

            // Load one row of input
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], C_);
            inQueueX_.EnQue(xLocal);

            // Deque input and accumulator
            AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> acc = accQueue_.DeQue<float>();

            // acc = acc + xDeq
            AscendC::Add(acc, acc, xDeq, C_);

            // Write result to global memory
            AscendC::DataCopy(yGm_[offset], acc, C_);

            // Free temporary tensors and re-enque accumulator
            inQueueX_.FreeTensor(xDeq);
            accQueue_.EnQue(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0u};
    uint32_t N_ {0u};
    uint32_t C_ {0u};
    int32_t dim_ {0};
};

extern "C" __global__ __aicore__ void ca6k00071_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00071Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.numRows, tiling_data.numCols, tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
