kernel_src="""

#include "kernel_operator.h"
#include <math.h>

class KernelPdist {
public:
    __aicore__ inline KernelPdist() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t d,
                                uint32_t totalPairs, float p)
    {
        n_ = n;
        d_ = d;
        totalPairs_ = totalPairs;
        p_ = p;
        xGm_.SetGlobalBuffer((__gm__ float*)x, n * d);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalPairs);
    }

    __aicore__ inline void Process()
    {
        if (totalPairs_ == 0) return;
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t pairsPerBlock = totalPairs_ / blockNum;
        uint32_t rem = totalPairs_ % blockNum;
        uint32_t startPair = blockIdx * pairsPerBlock + (blockIdx < rem ? blockIdx : rem);
        uint32_t endPair = startPair + pairsPerBlock + (blockIdx < rem ? 1 : 0);

        for (uint32_t k = startPair; k < endPair; k++) {
            // compute row indices i and j (i < j)
            float i_f = (2.0f * n_ - 1.0f - sqrtf((2.0f * n_ - 1.0f) * (2.0f * n_ - 1.0f) - 8.0f * k)) / 2.0f;
            uint32_t i = static_cast<uint32_t>(i_f);
            uint32_t offset_i = i * (2 * n_ - i - 1) / 2;
            uint32_t j = k - offset_i + i + 1;

            // compute Lp distance
            float sum = 0.0f;
            for (uint32_t dim = 0; dim < d_; dim++) {
                float diff = xGm_[i * d_ + dim] - xGm_[j * d_ + dim];
                float abs_diff = fabs(diff);
                sum += powf(abs_diff, p_);
            }
            float dist = powf(sum, 1.0f / p_);
            yGm_[k] = dist;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t n_;
    uint32_t d_;
    uint32_t totalPairs_;
    float p_;
};

extern "C" __global__ __aicore__ void ca6k04590_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelPdist op;
    op.Init(x, y, tiling_data.n, tiling_data.d, tiling_data.totalPairs, tiling_data.p);
    op.Process();
}

#include <torch/library.h>

at::Tensor pdist_impl_npu(const at::Tensor& self, double p) {
    int64_t n = self.size(0);
    int64_t totalPairs = n * (n - 1) / 2;
    at::Tensor result = at::empty({totalPairs}, self.options());
    EXEC_NPU_CMD(aclnnCa6k04590Custom, self, p, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k04590_custom", &pdist_impl_npu);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
