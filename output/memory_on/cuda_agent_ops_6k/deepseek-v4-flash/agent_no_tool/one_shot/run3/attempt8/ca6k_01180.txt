kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01180Custom {
public:
    __aicore__ inline KernelCa6k01180Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLength)
    {
        totalRows_ = totalRows;
        rowLength_ = rowLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows * rowLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, rowLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(tmpSortBuf_, 1, rowLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowLength_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Distribute rows across blocks (grid stride)
        uint32_t rowsPerBlock = (totalRows_ + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > totalRows_) endRow = totalRows_;

        for (uint32_t r = startRow; r < endRow; r++) {
            // Load row data into input queue
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[r * rowLength_], rowLength_);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            float modeVal = ComputeMode(xLocal, rowLength_);
            inQueueX_.FreeTensor(xLocal);

            // Write output
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            yLocal.SetValue(0, modeVal);
            AscendC::DataCopy(yGm_[r], yLocal, 1);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    __aicore__ inline float ComputeMode(const AscendC::LocalTensor<float>& xRow, uint32_t len)
    {
        // Sort the row and find the value with longest consecutive run
        AscendC::LocalTensor<float> sorted = tmpSortBuf_.AllocTensor<float>();
        AscendC::DataCopy(sorted, xRow, len);

        // Sort ascending
        AscendC::Sort(sorted, sorted, false, len);

        float bestVal = sorted.GetValue(0);
        uint32_t bestCount = 1;
        uint32_t curCount = 1;

        for (uint32_t i = 1; i < len; i++) {
            float cur = sorted.GetValue(i);
            float prev = sorted.GetValue(i - 1);
            if (cur == prev) {
                curCount++;
            } else {
                if (curCount > bestCount) {
                    bestCount = curCount;
                    bestVal = prev;
                }
                curCount = 1;
            }
        }
        // Check last run
        if (curCount > bestCount) {
            bestVal = sorted.GetValue(len - 1);
        }

        tmpSortBuf_.FreeTensor(sorted);
        return bestVal;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSortBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t rowLength_ {0};
};

extern "C" __global__ __aicore__ void ca6k01180_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01180Custom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.rowLength);
    op.Process();
}
