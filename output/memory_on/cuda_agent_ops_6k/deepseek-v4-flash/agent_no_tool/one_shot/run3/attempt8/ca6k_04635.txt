    uint32_t inputW = static_cast<uint32_t>(origin.GetDim(4));

#include "kernel_operator.h"

class KernelCa6k04635Custom {
public:
    __aicore__ inline KernelCa6k04635Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElems,
                                uint32_t tileLen,
                                uint32_t batch,
                                uint32_t channels,
                                uint32_t inputD, uint32_t inputH, uint32_t inputW,
                                uint32_t outputD, uint32_t outputH, uint32_t outputW,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t kernelD, uint32_t kernelH, uint32_t kernelW) {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        totalOutputElems_ = totalOutputElems;
        tileLen_ = tileLen;
        batch_ = batch;
        channels_ = channels;
        inputD_ = inputD; inputH_ = inputH; inputW_ = inputW;
        outputD_ = outputD; outputH_ = outputH; outputW_ = outputW;
        strideD_ = strideD; strideH_ = strideH; strideW_ = strideW;
        kernelD_ = kernelD; kernelH_ = kernelH; kernelW_ = kernelW;

        inputStrideN_ = channels_ * inputD_ * inputH_ * inputW_;
        inputStrideC_ = inputD_ * inputH_ * inputW_;
        inputStrideD_ = inputH_ * inputW_;
        inputStrideH_ = inputW_;
    }

    __aicore__ inline void Process() {
        if (totalOutputElems_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Use grid-stride loop over output elements
        for (uint32_t idx = blockIdx * tileLen_; idx < totalOutputElems_; idx += blockNum * tileLen_) {
            uint32_t end = idx + tileLen_;
            if (end > totalOutputElems_) end = totalOutputElems_;

            for (uint32_t o = idx; o < end; o++) {
                // decompose o into (n, c, dz, hy, wx)
                uint32_t tmp = o;
                uint32_t wx = tmp % outputW_; tmp /= outputW_;
                uint32_t hy = tmp % outputH_; tmp /= outputH_;
                uint32_t dz = tmp % outputD_; tmp /= outputD_;
                uint32_t c = tmp % channels_; tmp /= channels_;
                uint32_t n = tmp;

                uint32_t inStartD = dz * strideD_;
                uint32_t inStartH = hy * strideH_;
                uint32_t inStartW = wx * strideW_;
                uint32_t inEndD = inStartD + kernelD_;
                uint32_t inEndH = inStartH + kernelH_;
                uint32_t inEndW = inStartW + kernelW_;

                uint32_t inputBase = n * inputStrideN_ + c * inputStrideC_;

                float sum = 0.0f;
                for (uint32_t d = inStartD; d < inEndD; d++) {
                    for (uint32_t h = inStartH; h < inEndH; h++) {
                        for (uint32_t w = inStartW; w < inEndW; w++) {
                            uint32_t inIdx = inputBase + d * inputStrideD_ + h * inputStrideH_ + w;
                            sum += xGm_.GetValue(inIdx);
                        }
                    }
                }
                float windowSize = static_cast<float>(kernelD_ * kernelH_ * kernelW_);
                yGm_.SetValue(o, sum / windowSize);
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalOutputElems_ {0};
    uint32_t tileLen_ {0};
    uint32_t batch_ {0};
    uint32_t channels_ {0};
    uint32_t inputD_ {0}, inputH_ {0}, inputW_ {0};
    uint32_t outputD_ {0}, outputH_ {0}, outputW_ {0};
    uint32_t strideD_ {0}, strideH_ {0}, strideW_ {0};
    uint32_t kernelD_ {0}, kernelH_ {0}, kernelW_ {0};
    uint32_t inputStrideN_ {0}, inputStrideC_ {0}, inputStrideD_ {0}, inputStrideH_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04635_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04635Custom op;
    op.Init(x, y,
            tiling_data.totalOutputElems,
            tiling_data.tileLen,
            tiling_data.batch,
            tiling_data.channels,
            tiling_data.inputD, tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputD, tiling_data.outputH, tiling_data.outputW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW);
    op.Process();
}
