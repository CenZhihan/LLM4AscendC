kernel_src = """
#include "kernel_operator.h"
#include "ca6k_04528_custom_tiling.h"

using namespace AscendC;

class KernelMaxPool3D {
public:
    __aicore__ inline KernelMaxPool3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                const Ca6k04528CustomTilingData &tiling) {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        totalLength_ = tiling.totalLength;
        N_ = tiling.N;
        C_ = tiling.C;
        D_ = tiling.D;
        H_ = tiling.H;
        W_ = tiling.W;
        D_out_ = tiling.D_out;
        H_out_ = tiling.H_out;
        W_out_ = tiling.W_out;
        kernel_size_ = tiling.kernel_size;
        stride_ = tiling.stride;
        padding_ = tiling.padding;
    }

    __aicore__ inline void Process() {
        uint32_t total = totalLength_;
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        uint32_t perBlock = (total + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * perBlock;
        uint32_t end = (start + perBlock > total) ? total : start + perBlock;

        for (uint32_t idx = start; idx < end; idx++) {
            uint32_t temp = idx;
            int32_t wo = temp % W_out_; temp /= W_out_;
            int32_t ho = temp % H_out_; temp /= H_out_;
            int32_t do_ = temp % D_out_; temp /= D_out_;
            int32_t c = temp % C_; temp /= C_;
            int32_t b = temp;

            int32_t di_start = do_ * stride_ - padding_;
            int32_t hi_start = ho * stride_ - padding_;
            int32_t wi_start = wo * stride_ - padding_;

            float max_val = -1e38f;
            for (int32_t kd = 0; kd < kernel_size_; kd++) {
                int32_t di = di_start + kd;
                if (di < 0 || di >= D_) continue;
                for (int32_t kh = 0; kh < kernel_size_; kh++) {
                    int32_t hi = hi_start + kh;
                    if (hi < 0 || hi >= H_) continue;
                    for (int32_t kw = 0; kw < kernel_size_; kw++) {
                        int32_t wi = wi_start + kw;
                        if (wi < 0 || wi >= W_) continue;
                        float val = xGm_.GetValue(((b * C_ + c) * D_ + di) * H_ * W_ + hi * W_ + wi);
                        if (val > max_val) max_val = val;
                    }
                }
            }
            yGm_.SetValue(idx, max_val);
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    uint32_t totalLength_;
    int32_t N_, C_, D_, H_, W_;
    int32_t D_out_, H_out_, W_out_;
    int32_t kernel_size_, stride_, padding_;
};

extern "C" __global__ __aicore__ void ca6k_04528_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPool3D op;
    op.Init(x, y, tiling_data);
    op.Process();
}
"""

model_src = """
import torch
import torch.nn as nn
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape):
        super(ModelNew, self).__init__()
        self.custom_pool = custom_ops_lib.ca6k_04528_custom
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.bias = nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        x = self.custom_pool(x, self.kernel_size, self.stride, self.padding)
        x = self.conv(x)
        x = x + self.bias
        return x
