#include "ca6k00253_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00253CustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = shape.GetDim(0);
    uint32_t C = shape.GetDim(1);
    uint32_t H = shape.GetDim(2);
    uint32_t W = shape.GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t convStride = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t convPadding = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t convDilation = attrs->GetAttrPointer<int32_t>(2)[0];
    int32_t convGroups = attrs->GetAttrPointer<int32_t>(3)[0];
    int32_t poolKernel = attrs->GetAttrPointer<int32_t>(4)[0];
    int32_t poolStride = attrs->GetAttrPointer<int32_t>(5)[0];
    int32_t poolPadding = attrs->GetAttrPointer<int32_t>(6)[0];

    const auto &wshape = context->GetInputShape(1)->GetOriginShape();
    uint32_t outC = wshape.GetDim(0);
    uint32_t kernelH = wshape.GetDim(2);
    uint32_t kernelW = wshape.GetDim(3);

    uint32_t outH = (H + 2 * convPadding - convDilation * (kernelH - 1) - 1) / convStride + 1;
    uint32_t outW = (W + 2 * convPadding - convDilation * (kernelW - 1) - 1) / convStride + 1;

    uint32_t poolOutH = (outH + 2 * poolPadding - poolKernel) / poolStride + 1;
    uint32_t poolOutW = (outW + 2 * poolPadding - poolKernel) / poolStride + 1;

    uint32_t totalLength = N * outC * poolOutH * poolOutW;

    // tileLen from UB budget: 5 tile buffers (inQueueX:2, outQueueY:2, tmpQueue:1) -> bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

    int32_t convDilation = attrs->GetAttrPointer<int32_t>(2)[0];
    int32_t poolKernel = attrs->GetAttrPointer<int32_t>(4)[0];
    int32_t poolStride = attrs->GetAttrPointer<int32_t>(5)[0];
    int32_t poolPadding = attrs->GetAttrPointer<int32_t>(6)[0];

    uint32_t outH = (H + 2 * convPadding - convDilation * (kH - 1) - 1) / convStride + 1;
    uint32_t outW = (W + 2 * convPadding - convDilation * (kW - 1) - 1) / convStride + 1;
    uint32_t poolOutH = (outH + 2 * poolPadding - poolKernel) / poolStride + 1;
    uint32_t poolOutW = (outW + 2 * poolPadding - poolKernel) / poolStride + 1;
