#include "ca6k_05321_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t TILE_SIZE = 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k05321CustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin_shape = x_shape->GetOriginShape();

    int32_t ndim = origin_shape.GetDimNum();
    uint32_t N = (ndim >= 1) ? origin_shape.GetDim(0) : 1;
    uint32_t C = (ndim >= 2) ? origin_shape.GetDim(1) : 1;
    uint32_t H = (ndim >= 3) ? origin_shape.GetDim(2) : 1;
    uint32_t W = (ndim >= 4) ? origin_shape.GetDim(3) : 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scale_ptr = attrs->GetAttrPointer<float>(0);
    float scale = (scale_ptr != nullptr) ? *scale_ptr : 2.0f;

    uint32_t H_out = static_cast<uint32_t>(static_cast<float>(H) * scale + 0.5f);
    uint32_t W_out = static_cast<uint32_t>(static_cast<float>(W) * scale + 0.5f);
    uint32_t totalLength = N * C * H_out * W_out;

    // Simple tiling: each block processes a contiguous chunk of output elements
    uint32_t blockDim = 8;  // fixed for simplicity
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_scale(scale);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
