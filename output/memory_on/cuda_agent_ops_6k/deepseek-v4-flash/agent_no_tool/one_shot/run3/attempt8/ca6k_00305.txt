static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00305CustomTilingData tiling;
    const gert::StorageShape *inputStorageShape = context->GetInputShape(0);
    const gert::Shape &originShape = inputStorageShape->GetOriginShape();
    const uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    const uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    const uint32_t L = static_cast<uint32_t>(originShape.GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int *kernelSize = attrs->GetAttrPointer<int>(0);
    const int *stride = attrs->GetAttrPointer<int>(1);
    uint32_t K = static_cast<uint32_t>(*kernelSize);
    uint32_t S = static_cast<uint32_t>(*stride);

    uint32_t L_out = (L - K) / S + 1;
    uint32_t totalLength = N * C * L_out;
    uint32_t blockDim = N * C;

    constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    constexpr uint32_t ALIGN_ELEMS = 64;
    uint32_t tileLen = L_out;
    if (L_out > 10000) {
        uint32_t maxTile = (UB_BUDGET_BYTES / 4 - K) / (S + 2);
        maxTile = (maxTile / ALIGN_ELEMS) * ALIGN_ELEMS;
        if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
        if (maxTile < 64) maxTile = 64;
        tileLen = maxTile > L_out ? L_out : maxTile;
    }

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_inputWidth(L);
    tiling.set_outputWidth(L_out);
    tiling.set_kernelSize(K);
    tiling.set_stride(S);
    tiling.set_invKernelSize(1.0f / static_cast<float>(K));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00305Custom {
public:
    __aicore__ inline KernelCa6k00305Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t inputWidth,
                                uint32_t outputWidth, uint32_t kernelSize,
                                uint32_t stride, float invKernelSize,
                                uint32_t tileLength)
    {
        totalLen_ = totalLength;
        inputWidth_ = inputWidth;
        outputWidth_ = outputWidth;
        kernelSize_ = kernelSize;
        stride_ = stride;
        invKernelSize_ = invKernelSize;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Buffer sizes: max input slice for one tile output
        uint32_t maxInputSlice = (tileLen_ > 0) ? ((tileLen_ - 1) * stride_ + kernelSize_) : 0u;
        if (maxInputSlice < kernelSize_) maxInputSlice = kernelSize_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, maxInputSlice * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outputWidth_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        // Each block processes one row (channel)
        uint32_t baseIn = blockIdx * inputWidth_;
        uint32_t baseOut = blockIdx * outputWidth_;

        uint32_t curStart = 0;
        while (curStart < outputWidth_) {
            // Determine tile output length
            uint32_t curTileLen = outputWidth_ - curStart;
            if (curTileLen > tileLen_) curTileLen = tileLen_;

            // Corresponding input slice length
            uint32_t inputSliceLen = (curTileLen - 1) * stride_ + kernelSize_;
            uint32_t inputOff = curStart * stride_;

            // Copy input slice
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[baseIn + inputOff], inputSliceLen);
            inQueueX_.EnQue(xLocal);

            // Compute output tile
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            xLocal = inQueueX_.DeQue<float>();

            for (uint32_t i = 0; i < curTileLen; ++i) {
                float sum = 0.0f;
                for (uint32_t k = 0; k < kernelSize_; ++k) {
                    float val = xLocal[i * stride_ + k];
                    sum += val * val;
                }
                yLocal[i] = AscendC::Sqrt(sum * invKernelSize_);
            }

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.EnQue(yLocal);

            // Copy output tile to global
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[baseOut + curStart], yLocal, curTileLen);
            outQueueY_.FreeTensor(yLocal);

            curStart += curTileLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t inputWidth_ {0};
    uint32_t outputWidth_ {0};
    uint32_t kernelSize_ {0};
    uint32_t stride_ {0};
    float invKernelSize_ {0.0f};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00305_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00305Custom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.inputWidth,
            tiling_data.outputWidth,
            tiling_data.kernelSize,
            tiling_data.stride,
            tiling_data.invKernelSize,
            tiling_data.tileLength);
    op.Process();
}
