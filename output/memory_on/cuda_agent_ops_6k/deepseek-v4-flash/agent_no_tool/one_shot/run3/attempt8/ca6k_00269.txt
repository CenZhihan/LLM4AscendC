    int32_t strideH = attrs->GetAttrPointer<int32_t>(3)[0];
    int32_t strideW = attrs->GetAttrPointer<int32_t>(4)[0];

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    int32_t N = static_cast<int32_t>(x_shape->GetOriginShape().GetDim(0));
    int32_t C = static_cast<int32_t>(x_shape->GetOriginShape().GetDim(1));

    int32_t W = static_cast<int32_t>(x_shape->GetOriginShape().GetDim(3));

    int32_t outW = (W - kernelSizeW) / strideW + 1;

    if (tileLength < ALIGN_ELEMS) tileLength = ALIGN_ELEMS;

        if (tileLength == 0) tileLength = totalLength;

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00269Custom {
public:
    __aicore__ inline KernelCa6k00269Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t N, int32_t C, int32_t H, int32_t W,
                                int32_t outH, int32_t outW,
                                int32_t kernelSizeH, int32_t kernelSizeW,
                                int32_t strideH, int32_t strideW,
                                int32_t p)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N; C_ = C; H_ = H; W_ = W;
        outH_ = outH; outW_ = outW;
        kernelSizeH_ = kernelSizeH; kernelSizeW_ = kernelSizeW;
        strideH_ = strideH; strideW_ = strideW;
        p_ = p;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t flatIdx = base + i;
                int32_t ow = flatIdx % outW_;
                int32_t oh = (flatIdx / outW_) % outH_;
                int32_t c = (flatIdx / (outW_ * outH_)) % C_;
                int32_t n = flatIdx / (outW_ * outH_ * C_);

                int32_t startH = oh * strideH_;
                int32_t startW = ow * strideW_;

                float sum = 0.0f;
                for (int32_t kh = 0; kh < kernelSizeH_; kh++) {
                    for (int32_t kw = 0; kw < kernelSizeW_; kw++) {
                        int32_t h = startH + kh;
                        int32_t w = startW + kw;
                        uint32_t inIdx = ((n * C_ + c) * H_ + h) * W_ + w;
                        float val = xGm_.GetValue(inIdx);
                        sum += val * val;
                    }
                }
                // use AscendC Sqrt on a scalar via 1-element LocalTensor
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                tLocal.SetValue(0, sum);
                AscendC::Sqrt(tLocal, tLocal, 1);
                float outVal = tLocal.GetValue(0);
                tmpQueue_.FreeTensor(tLocal);

                yLocal.SetValue(i, outVal);
            }

            outQueueY_.EnQue(yLocal);
            AscendC::LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yDeque, curLen);
            outQueueY_.FreeTensor(yDeque);

            base += strideElems;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t N_, C_, H_, W_;
    int32_t outH_, outW_;
    int32_t kernelSizeH_, kernelSizeW_;
    int32_t strideH_, strideW_;
    int32_t p_;
};

extern "C" __global__ __aicore__ void ca6k00269_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00269Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.outH, tiling_data.outW,
            tiling_data.kernelSizeH, tiling_data.kernelSizeW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.p);
    op.Process();
}
