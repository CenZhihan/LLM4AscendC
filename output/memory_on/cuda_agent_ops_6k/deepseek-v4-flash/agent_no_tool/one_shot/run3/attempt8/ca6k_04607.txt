#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelAvgPool3d {
public:
    __aicore__ inline KernelAvgPool3d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputLength, uint32_t tileLength,
                                int32_t N, int32_t C, int32_t D, int32_t H, int32_t W,
                                int32_t kernelD, int32_t kernelH, int32_t kernelW,
                                int32_t strideD, int32_t strideH, int32_t strideW,
                                int32_t padD, int32_t padH, int32_t padW,
                                int32_t outD, int32_t outH, int32_t outW,
                                int32_t totalOutputElems)
    {
        N_ = N; C_ = C; D_ = D; H_ = H; W_ = W;
        kernelD_ = kernelD; kernelH_ = kernelH; kernelW_ = kernelW;
        strideD_ = strideD; strideH_ = strideH; strideW_ = strideW;
        padD_ = padD; padH_ = padH; padW_ = padW;
        outD_ = outD; outH_ = outH; outW_ = outW;
        totalOutputElems_ = totalOutputElems;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElems);
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0 || tileLen_ == 0) return;
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= (uint32_t)totalOutputElems_) return;

        while (true) {
            uint32_t curLen = tileLen_;
            if (base + curLen > (uint32_t)totalOutputElems_) {
                curLen = totalOutputElems_ - base;
            }
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t outIdx = base + i;
                ComputeOne(outIdx);
            }
            base += strideElems;
            if (base >= (uint32_t)totalOutputElems_) break;
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t outIdx)
    {
        int32_t ow = outIdx % outW_;
        int32_t rem = outIdx / outW_;
        int32_t oh = rem % outH_;
        rem = rem / outH_;
        int32_t od = rem % outD_;
        rem = rem / outD_;
        int32_t c = rem % C_;
        int32_t n = rem / C_;

        int32_t startD = od * strideD_ - padD_;
        int32_t startH = oh * strideH_ - padH_;
        int32_t startW = ow * strideW_ - padW_;

        float sum = 0.0f;
        uint32_t count = 0;

        for (int32_t kd = 0; kd < kernelD_; kd++) {
            int32_t d = startD + kd;
            if (d < 0 || d >= D_) continue;
            for (int32_t kh = 0; kh < kernelH_; kh++) {
                int32_t h = startH + kh;
                if (h < 0 || h >= H_) continue;
                for (int32_t kw = 0; kw < kernelW_; kw++) {
                    int32_t w = startW + kw;
                    if (w < 0 || w >= W_) continue;
                    uint32_t inIdx = static_cast<uint32_t>(
                        ((n * C_ + c) * D_ + d) * H_ + h) * W_ + w;
                    sum += xGm_[inIdx];
                    count++;
                }
            }
        }

        float avg = (count > 0) ? (sum / (float)count) : 0.0f;
        yGm_[outIdx] = avg;
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    int32_t N_, C_, D_, H_, W_;
    int32_t kernelD_, kernelH_, kernelW_;
    int32_t strideD_, strideH_, strideW_;
    int32_t padD_, padH_, padW_;
    int32_t outD_, outH_, outW_;
    int32_t totalOutputElems_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void ca6k_04607_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAvgPool3d op;
    op.Init(x, y, tiling_data.totalInputLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C, tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.totalOutputElems);
    op.Process();
}
