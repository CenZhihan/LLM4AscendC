namespace optiling {

static uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00523CustomTilingData tiling;

    const auto &origShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t ndim = origShape.GetDimNum();
    uint32_t shapeArr[4] = {0};
    for (uint32_t i = 0; i < ndim && i < 4; ++i)
        shapeArr[i] = static_cast<uint32_t>(origShape.GetDim(i));

    uint32_t totalLength = static_cast<uint32_t>(origShape.GetShapeSize());

    // For this model, flip dims is always [2] (height in 4D B C H W)
    uint32_t flipMask = 1u << 2;

    uint32_t tileLength = (ndim > 0) ? shapeArr[ndim - 1] : 0;
    if (tileLength == 0) tileLength = 1;

    uint32_t totalRows = (tileLength > 0) ? (totalLength / tileLength) : 0;
    uint32_t blockDim = 8;
    if (totalRows > 32) blockDim = 24;
    else if (totalRows > 16) blockDim = 16;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_ndim(ndim);
    tiling.set_shape0((ndim >= 1) ? shapeArr[0] : 0);
    tiling.set_shape1((ndim >= 2) ? shapeArr[1] : 0);
    tiling.set_shape2((ndim >= 3) ? shapeArr[2] : 0);
    tiling.set_shape3((ndim >= 4) ? shapeArr[3] : 0);
    tiling.set_flipMask(flipMask);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFlipCustom {
public:
    __aicore__ inline KernelFlipCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t ndim,
                                uint32_t shape0, uint32_t shape1, uint32_t shape2, uint32_t shape3,
                                uint32_t flipMask)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        ndim_ = ndim;
        shape0_ = shape0;
        shape1_ = shape1;
        shape2_ = shape2;
        shape3_ = shape3;
        flipMask_ = flipMask;

        height_ = shape2_;
        width_ = shape3_;
        totalRows_ = totalLen_ / tileLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride over rows
        uint32_t rowsPerBlock = (totalRows_ + blockNum - 1u) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = (blockIdx + 1u) * rowsPerBlock;
        if (endRow > totalRows_) endRow = totalRows_;
        if (startRow >= totalRows_) return;

        for (uint32_t r = startRow; r < endRow; ++r) {
            uint32_t imageIndex = r / height_;
            uint32_t h_in = height_ - 1 - (r % height_);  // reversed height
            uint32_t srcRow = imageIndex * height_ + h_in;
            uint32_t srcOffset = srcRow * tileLen_;
            uint32_t dstOffset = r * tileLen_;

            // copy one row using double buffer
            AscendC::LocalTensor<float> local = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(local, xGm_[srcOffset], tileLen_);
            AscendC::DataCopy(yGm_[dstOffset], local, tileLen_);
            inQueueX_.FreeTensor(local);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
    uint32_t ndim_ {0};
    uint32_t shape0_ {0};
    uint32_t shape1_ {0};
    uint32_t shape2_ {0};
    uint32_t shape3_ {0};
    uint32_t flipMask_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    uint32_t totalRows_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00523_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFlipCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.blockDim, tiling_data.ndim,
            tiling_data.shape0, tiling_data.shape1, tiling_data.shape2, tiling_data.shape3,
            tiling_data.flipMask);
    op.Process();
}
'''
