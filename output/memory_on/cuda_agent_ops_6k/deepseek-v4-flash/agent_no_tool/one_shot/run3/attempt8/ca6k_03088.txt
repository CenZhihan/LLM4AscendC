kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t MAX_CHUNK = 4096;

class KernelMedian {
public:
    __aicore__ inline KernelMedian() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR values, GM_ADDR indices,
                                GM_ADDR workspace, uint32_t totalLength,
                                uint32_t tileLength, uint32_t blockDim, int32_t dim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        dim_ = dim;
        rows_ = totalLen_ / tileLen_;
        cols_ = tileLen_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        valuesGm_.SetGlobalBuffer((__gm__ float*)values, cols_);
        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices, cols_);
        wsGm_.SetGlobalBuffer((__gm__ float*)workspace, totalLen_);
        pipe_.InitBuffer(inQueue_, BUFFER_NUM, MAX_CHUNK * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = GetBlockIdx();
        if (blockIdx >= cols_) return;

        uint32_t colOffset = blockIdx;

        // Copy entire column to workspace (stride = cols_)
        uint32_t copied = 0;
        while (copied < rows_) {
            uint32_t chunk = rows_ - copied;
            if (chunk > MAX_CHUNK) chunk = MAX_CHUNK;
            LocalTensor<float> localBuf = inQueue_.AllocTensor<float>();
            DataCopy(localBuf, xGm_[copied * cols_ + colOffset], chunk);
            inQueue_.EnQue(localBuf);
            LocalTensor<float> localTmp = inQueue_.DeQue<float>();
            DataCopy(wsGm_[copied * cols_ + colOffset], localTmp, chunk);
            inQueue_.FreeTensor(localTmp);
            copied += chunk;
        }

        // Quickselect to find median value
        int32_t left = 0;
        int32_t right = static_cast<int32_t>(rows_) - 1;
        int32_t k = static_cast<int32_t>((rows_ - 1) / 2);
        while (left < right) {
            float pivot = wsGm_[left * cols_ + colOffset];
            int32_t i = left - 1;
            int32_t j = right + 1;
            while (true) {
                do { i++; } while (wsGm_[i * cols_ + colOffset] < pivot);
                do { j--; } while (wsGm_[j * cols_ + colOffset] > pivot);
                if (i >= j) break;
                float tmp = wsGm_[i * cols_ + colOffset];
                wsGm_[i * cols_ + colOffset] = wsGm_[j * cols_ + colOffset];
                wsGm_[j * cols_ + colOffset] = tmp;
            }
            if (k <= j) right = j;
            else left = j + 1;
        }
        float medianVal = wsGm_[k * cols_ + colOffset];

        // Find first occurrence of medianVal in the original column
        int64_t medianIdx = -1;
        for (uint32_t r = 0; r < rows_; ++r) {
            if (xGm_[r * cols_ + colOffset] == medianVal) {
                medianIdx = static_cast<int64_t>(r);
                break;
            }
        }

        valuesGm_[blockIdx] = medianVal;
        indicesGm_[blockIdx] = medianIdx;
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> valuesGm_;
    GlobalTensor<int64_t> indicesGm_;
    GlobalTensor<float> wsGm_;
    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t blockDim_;
    int32_t dim_;
    uint32_t rows_;
    uint32_t cols_;
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueue_;
};

extern "C" __global__ __aicore__ void ca6k_03088_custom(
    GM_ADDR x, GM_ADDR values, GM_ADDR indices, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(td, tiling);
    KernelMedian op;
    op.Init(x, values, indices, workspace, td.totalLength, td.tileLength, td.blockDim, td.dim);
    op.Process();
}
"""
