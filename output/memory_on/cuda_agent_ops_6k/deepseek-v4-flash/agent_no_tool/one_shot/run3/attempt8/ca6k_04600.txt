static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04600CustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0);
    const uint32_t N = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    const uint32_t C = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    const uint32_t H = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
    const uint32_t W = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));
    uint32_t totalLength = N * C * H * W;
    uint32_t tileLen = H * W; // one channel spatial size

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = attrs->GetAttrPointer<float>(0)[0];
    bool training = attrs->GetAttrPointer<bool>(1)[0];

    // blockDim: number of blocks to parallelize over channels
    uint32_t numChannels = N * C;
    uint32_t blockDim = (numChannels < 8) ? 8u : (numChannels > 24u ? 24u : numChannels);
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_eps(eps);
    tiling.set_training(training ? 1 : 0);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // not used for double buffer, kept for structure

class KernelCa6k04600Custom {
public:
    __aicore__ inline KernelCa6k04600Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t H, uint32_t W, float eps, uint8_t training)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        H_ = H;
        W_ = W;
        eps_ = eps;
        training_ = training;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, totalLen_ / (H_ * W_)); // size = N*C
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, totalLen_ / (H_ * W_));

        // allocate local buffers for one spatial tile
        pipe_.InitBuffer(inQueueX_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileLen_ * sizeof(float));    // store diff or intermediate
        pipe_.InitBuffer(meanBuf_, 1, sizeof(float));
        pipe_.InitBuffer(varBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t numChannels = totalLen_ / tileLen_; // N * C
        uint32_t channelsPerBlock = (numChannels + blockNum - 1) / blockNum;
        uint32_t startCh = blockIdx * channelsPerBlock;
        uint32_t endCh = (startCh + channelsPerBlock > numChannels) ? numChannels : startCh + channelsPerBlock;

        for (uint32_t ch = startCh; ch < endCh; ++ch) {
            uint32_t base = ch * tileLen_;
            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], tileLen_);

            // Compute mean
            AscendC::LocalTensor<float> meanLocal = meanBuf_.Get<float>();
            AscendC::LocalTensor<float> varLocal = varBuf_.Get<float>();
            AscendC::ReduceSum(meanLocal, xLocal, tileLen_);
            meanLocal(0) = meanLocal(0) / static_cast<float>(tileLen_);

            // Compute variance: sum((x - mean)^2)
            AscendC::LocalTensor<float> tmpLocal = tmpBuf_.Get<float>();
            AscendC::Sub(tmpLocal, xLocal, meanLocal(0), tileLen_);
            AscendC::Mul(tmpLocal, tmpLocal, tmpLocal, tileLen_);
            AscendC::ReduceSum(varLocal, tmpLocal, tileLen_);
            float invStd = 1.0f / sqrt(varLocal(0) / static_cast<float>(tileLen_) + eps_);

            // Normalize: y = (x - mean) * invStd * weight[ch] + bias[ch]
            float w = weightGm_.GetValue(ch);
            float b = biasGm_.GetValue(ch);
            AscendC::Sub(yLocal_, xLocal, meanLocal(0), tileLen_);
            AscendC::Muls(yLocal_, yLocal_, invStd, tileLen_);
            AscendC::Muls(yLocal_, yLocal_, w, tileLen_);
            AscendC::Adds(yLocal_, yLocal_, b, tileLen_);

            // Write back
            AscendC::DataCopy(yGm_[base], yLocal_, tileLen_);

            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> meanBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> varBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    // local tensor for output during compute (reuse across channels)
    AscendC::LocalTensor<float> yLocal_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t H_{0};
    uint32_t W_{0};
    float eps_{1e-5f};
    uint8_t training_{1};
};

extern "C" __global__ __aicore__ void ca6k_04600_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04600Custom op;
    op.Init(x, y, weight, bias,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.H, tiling_data.W, tiling_data.eps, tiling_data.training);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor instance_norm_custom_impl_npu(
    const at::Tensor& x,
    const at::Tensor& running_mean,
    const at::Tensor& running_var,
    const at::Tensor& weight,
    const at::Tensor& bias,
    bool training,
    double momentum,
    double eps)
{
    // momentum is ignored (we always compute stats when training)
    at::Tensor result = at::empty_like(x);
    EXEC_NPU_CMD(aclnnCa6k04600Custom, x, running_mean, running_var, weight, bias, eps, training, result);
    return result;
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_04600_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_features, eps, momentum):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.eps = eps
        self.momentum = momentum
        # these parameters are not used by the custom op (training=True by default)
        self.running_mean = torch.nn.Parameter(torch.zeros(num_features), requires_grad=False)
        self.running_var = torch.nn.Parameter(torch.ones(num_features), requires_grad=False)
        self.weight = torch.nn.Parameter(torch.ones(num_features))
        self.bias = torch.nn.Parameter(torch.zeros(num_features))

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        # order matches operator input_desc: x, running_mean, running_var, weight, bias
        return fn(x, self.running_mean, self.running_var, self.weight, self.bias)
'''
