    Ca6k02874CustomTilingData tiling;
    const gert::StorageShape *x_shape = context->GetInputShape(1);

    uint32_t totalCols = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));

    // We need UB for: one row of x (tileCol floats), one tile of weight (tileCol),
    // partial output row (tileCol), plus temps for multiply and reduce.
    // Approx: 3 * tileCol floats. So max tileCol = UB_BUDGET_BYTES / (3*4) = 16384.
    uint32_t maxTileCol = UB_BUDGET_BYTES / (3 * sizeof(float));

    if (maxTileCol > totalCols) maxTileCol = AlignDown(totalCols, ALIGN);
    if (maxTileCol == 0) maxTileCol = totalCols;

    // Number of blocks: we assign each block a chunk of rows.
    // We'll compute tileRow such that each block gets at least one row.
    uint32_t blockDim = 16; // default
    if (totalRows < 32) blockDim = totalRows;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    uint32_t tileRow = CeilDiv(totalRows, blockDim);
    if (tileRow == 0) tileRow = 1;

    tiling.set_totalRows(totalRows);
    tiling.set_totalCols(totalCols);
    tiling.set_tileRow(tileRow);
    tiling.set_tileCol(maxTileCol);
    tiling.set_blockDim(blockDim);
