host_operator_src = '''
#include "ca6k_00363_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/types.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
  return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k00363CustomTilingData tiling;
  const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
  int32_t n = x_shape.GetDim(0);
  int32_t c = x_shape.GetDim(1);
  int32_t h = x_shape.GetDim(2);
  int32_t w = x_shape.GetDim(3);

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  const int32_t *upscale_factor = attrs->GetAttrPointer<int32_t>(0);
  int32_t r = *upscale_factor;

  int32_t out_c = c * r * r;
  int32_t out_h = h / r;
  int32_t out_w = w / r;
  int32_t totalLength = n * out_c * out_h * out_w;

  uint32_t maxTile = UB_BUDGET_BYTES / 4;
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = maxTile;
  uint32_t totalLen = static_cast<uint32_t>(totalLength);
  if (totalLen < tileLen) {
    uint32_t aligned = AlignDown(totalLen, ALIGN_ELEMS);
    tileLen = (aligned > 0) ? aligned : totalLen;
  }

  uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLen, tileLen);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  tiling.set_n(n);
  tiling.set_c(c);
  tiling.set_h(h);
  tiling.set_w(w);
  tiling.set_upscaleFactor(r);
  tiling.set_totalLength(totalLength);
  tiling.set_tileLength(static_cast<int32_t>(tileLen));
  tiling.set_blockDim(static_cast<int32_t>(blockDim));

  context->SetBlockDim(tiling.get_blockDim());

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

  int32_t n = x_shape->GetDim(0);
  int32_t c = x_shape->GetDim(1);
  int32_t h = x_shape->GetDim(2);
  int32_t w = x_shape->GetDim(3);

  const gert::RuntimeAttrs *attrs = context->GetAttrs();

kernel_src = '''
#include "kernel_operator.h"
#include "ca6k_00363_custom_tiling.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;

class KernelPixelUnshuffle {
public:
  __aicore__ inline KernelPixelUnshuffle() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                              int32_t n, int32_t c, int32_t h, int32_t w,
                              int32_t upscaleFactor, int32_t totalLength,
                              int32_t tileLength) {
    n_ = n;
    c_ = c;
    h_ = h;
    w_ = w;
    r_ = upscaleFactor;
    totalLen_ = totalLength;
    tileLen_ = tileLength;

    xGm_.SetGlobalBuffer((__gm__ float *)x);
    yGm_.SetGlobalBuffer((__gm__ float *)y);

    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;

    const uint32_t blockNum = GetBlockNum();
    const uint32_t blockIdx = GetBlockIdx();

    int32_t outC = c_ * r_ * r_;
    int32_t outH = h_ / r_;
    int32_t outW = w_ / r_;
    int32_t stride = blockNum * tileLen_;

    int32_t start = blockIdx * tileLen_;
    if (start >= totalLen_) return;

    int32_t offset = start;
    while (offset < totalLen_) {
      int32_t curLen = (offset + tileLen_ <= totalLen_) ? tileLen_ : (totalLen_ - offset);
      CopyIn(offset, curLen, outC, outH, outW);
      Compute(offset, curLen);
      CopyOut(offset, curLen);
      offset += stride;
    }
  }

private:
  __aicore__ inline void CopyIn(int32_t outOffset, int32_t len,
                                int32_t outC, int32_t outH, int32_t outW) {
    LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
    for (int32_t i = 0; i < len; i++) {
      int32_t outIdx = outOffset + i;
      int32_t n_idx = outIdx / (outC * outH * outW);
      int32_t tmp = outIdx % (outC * outH * outW);
      int32_t oc = tmp / (outH * outW);
      int32_t oh = (tmp % (outH * outW)) / outW;
      int32_t ow = tmp % outW;
      int32_t ic = oc / (r_ * r_);
      int32_t pixel = oc % (r_ * r_);
      int32_t ph = pixel / r_;
      int32_t pw = pixel % r_;
      int32_t inIdx = n_idx * (c_ * h_ * w_) + ic * (h_ * w_) + (oh * r_ + ph) * w_ + (ow * r_ + pw);
      xLocal.SetValue(i, xGm_.GetValue(inIdx));
    }
    inQueueX_.EnQue(xLocal);
  }

  __aicore__ inline void Compute(int32_t outOffset, int32_t len) {
    LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
    LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
    for (int32_t i = 0; i < len; i++) {
      yLocal.SetValue(i, xLocal.GetValue(i));
    }
    outQueueY_.EnQue(yLocal);
    inQueueX_.FreeTensor(xLocal);
  }

  __aicore__ inline void CopyOut(int32_t outOffset, int32_t len) {
    LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
    for (int32_t i = 0; i < len; i++) {
      yGm_.SetValue(outOffset + i, yLocal.GetValue(i));
    }
    outQueueY_.FreeTensor(yLocal);
  }

private:
  TPipe pipe_;
  TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
  TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
  GlobalTensor<float> xGm_;
  GlobalTensor<float> yGm_;
  int32_t n_ {0};
  int32_t c_ {0};
  int32_t h_ {0};
  int32_t w_ {0};
  int32_t r_ {2};
  int32_t totalLen_ {0};
  int32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00363_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelPixelUnshuffle op;
  op.Init(x, y,
          tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
          tiling_data.upscaleFactor, tiling_data.totalLength, tiling_data.tileLength);
  op.Process();
}
'''

at::Tensor pixel_unshuffle_impl_npu(const at::Tensor& self, int64_t upscale_factor) {
    int64_t n = self.size(0);
    int64_t c = self.size(1);
    int64_t h = self.size(2);
    int64_t w = self.size(3);
    int64_t out_c = c * upscale_factor * upscale_factor;
    int64_t out_h = h / upscale_factor;
    int64_t out_w = w / upscale_factor;
    at::Tensor result = at::empty({n, out_c, out_h, out_w}, self.options());
