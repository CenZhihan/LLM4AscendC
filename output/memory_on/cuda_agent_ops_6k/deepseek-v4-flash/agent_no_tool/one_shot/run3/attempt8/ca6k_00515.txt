kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00515Custom {
public:
    __aicore__ inline KernelCa6k00515Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t outputH, uint32_t outputW)
    {
        totalLen_ = totalLength;
        inputH_ = inputH;
        inputW_ = inputW;
        outputH_ = outputH;
        outputW_ = outputW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inRowQueue_, BUFFER_NUM, inputW_ * sizeof(float));
        pipe_.InitBuffer(tmpScalar_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t idx = blockIdx;
        while (idx < totalLen_) {
            ComputeOneOutput(idx);
            idx += blockNum;
        }
    }

private:
    __aicore__ inline void ComputeOneOutput(uint32_t flatIdx)
    {
        uint32_t perChannel = outputH_ * outputW_;
        uint32_t batchChannel = flatIdx / perChannel;
        uint32_t spatialIdx = flatIdx % perChannel;
        uint32_t oy = spatialIdx / outputW_;
        uint32_t ox = spatialIdx % outputW_;

        float scaleH = static_cast<float>(inputH_) / static_cast<float>(outputH_);
        float scaleW = static_cast<float>(inputW_) / static_cast<float>(outputW_);
        uint32_t startH = static_cast<uint32_t>(oy * scaleH);
        uint32_t endH   = static_cast<uint32_t>((oy + 1) * scaleH);
        uint32_t startW = static_cast<uint32_t>(ox * scaleW);
        uint32_t endW   = static_cast<uint32_t>((ox + 1) * scaleW);
        if (endH > inputH_) endH = inputH_;
        if (endW > inputW_) endW = inputW_;
        if (endH <= startH) startH = endH - 1;
        if (endW <= startW) startW = endW - 1;
        uint32_t hRegion = endH - startH;
        uint32_t wRegion = endW - startW;

        uint32_t inputBase = batchChannel * inputH_ * inputW_;

        AscendC::LocalTensor<float> sumLocal = tmpScalar_.AllocTensor<float>();
        sumLocal.SetValue(0, 0.0f);

        for (uint32_t ry = startH; ry < endH; ry++) {
            uint32_t rowOffset = inputBase + ry * inputW_ + startW;
            AscendC::LocalTensor<float> rowLocal = inRowQueue_.AllocTensor<float>();
            AscendC::DataCopy(rowLocal, xGm_[rowOffset], wRegion);
            inRowQueue_.EnQue(rowLocal);
            rowLocal = inRowQueue_.DeQue<float>();
            for (uint32_t i = 0; i < wRegion; i++) {
                float val = rowLocal(i);
                sumLocal.SetValue(0, sumLocal(0) + val);
            }
            inRowQueue_.FreeTensor(rowLocal);
        }

        float count = static_cast<float>(hRegion * wRegion);
        float avg = sumLocal(0) / count;

        uint32_t outputBase = batchChannel * perChannel + oy * outputW_ + ox;
        yGm_.SetValue(outputBase, avg);

        tmpScalar_.FreeTensor(sumLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpScalar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t inputH_ {0};
    uint32_t inputW_ {0};
    uint32_t outputH_ {0};
    uint32_t outputW_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00515_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00515Custom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputH, tiling_data.outputW);
    op.Process();
}
