host_operator_src = """
#include "ca6k_04603_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04603CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    uint32_t maxTile = UB_BUDGET_BYTES / 4;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
}  // namespace ge

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04603Custom {
public:
    __aicore__ inline KernelCa6k04603Custom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w1, GM_ADDR b1, GM_ADDR w2, GM_ADDR b2, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        w1Gm_.SetGlobalBuffer((__gm__ float*)w1, totalLen_);
        b1Gm_.SetGlobalBuffer((__gm__ float*)b1, totalLen_);
        w2Gm_.SetGlobalBuffer((__gm__ float*)w2, totalLen_);
        b2Gm_.SetGlobalBuffer((__gm__ float*)b2, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW1_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueB1_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW2_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueB2_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }
    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        // Copy all inputs for the first tile
        CopyIn(base, curLen, xGm_, inQueueX_);
        CopyIn(base, curLen, w1Gm_, inQueueW1_);
        CopyIn(base, curLen, b1Gm_, inQueueB1_);
        CopyIn(base, curLen, w2Gm_, inQueueW2_);
        CopyIn(base, curLen, b2Gm_, inQueueB2_);
        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen, xGm_, inQueueX_);
                CopyIn(nextBase, nextLen, w1Gm_, inQueueW1_);
                CopyIn(nextBase, nextLen, b1Gm_, inQueueB1_);
                CopyIn(nextBase, nextLen, w2Gm_, inQueueW2_);
                CopyIn(nextBase, nextLen, b2Gm_, inQueueB2_);
            }
            Compute(curLen);
            CopyOut(base, curLen);
            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }
private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len,
                                  AscendC::GlobalTensor<float>& gmSrc,
                                  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>& q) {
        AscendC::LocalTensor<float> local = q.AllocTensor<float>();
        AscendC::DataCopy(local, gmSrc[gmOffset], len);
        q.EnQue(local);
    }
    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> w1Local = inQueueW1_.DeQue<float>();
        AscendC::LocalTensor<float> b1Local = inQueueB1_.DeQue<float>();
        AscendC::LocalTensor<float> w2Local = inQueueW2_.DeQue<float>();
        AscendC::LocalTensor<float> b2Local = inQueueB2_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // conv1: y = conv(x, w1) + b1, then relu
        AscendC::Conv2D(tLocal, xLocal, w1Local, len);
        AscendC::Add(tLocal, tLocal, b1Local, len);
        AscendC::Relu(tLocal, tLocal, len);

        // conv2: y = conv(tLocal, w2) + b2, then relu
        AscendC::Conv2D(yLocal, tLocal, w2Local, len);
        AscendC::Add(yLocal, yLocal, b2Local, len);
        AscendC::Relu(yLocal, yLocal, len);

        inQueueX_.FreeTensor(xLocal);
        inQueueW1_.FreeTensor(w1Local);
        inQueueB1_.FreeTensor(b1Local);
        inQueueW2_.FreeTensor(w2Local);
        inQueueB2_.FreeTensor(b2Local);
        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
    }
    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }
private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW1_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB1_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW2_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB2_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> w1Gm_;
    AscendC::GlobalTensor<float> b1Gm_;
    AscendC::GlobalTensor<float> w2Gm_;
    AscendC::GlobalTensor<float> b2Gm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04603_custom(GM_ADDR x, GM_ADDR w1, GM_ADDR b1, GM_ADDR w2, GM_ADDR b2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04603Custom op;
    op.Init(x, w1, b1, w2, b2, y, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
"""
