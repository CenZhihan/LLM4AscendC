static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_BYTES = 32;
static constexpr uint32_t MAX_INNER_SIZE = 512;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04633CustomTilingData tiling;

    const gert::StorageShape *shape = context->GetInputShape(0);
    const uint32_t totalLength = static_cast<uint32_t>(shape->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int *topk_attr = attrs->GetAttrPointer<int>(0);
    const int *dim_attr = attrs->GetAttrPointer<int>(1);

    int32_t dim = *dim_attr;
    int64_t ndim = static_cast<int64_t>(shape->GetOriginShape().GetDimNum());
    if (dim < 0) dim += static_cast<int32_t>(ndim);
    uint32_t innerSize = static_cast<uint32_t>(shape->GetOriginShape().GetDim(dim));
    uint32_t topK = static_cast<uint32_t>(*topk_attr);
    if (topK > innerSize) topK = innerSize;
    uint32_t outerSize = totalLength / innerSize;

    uint32_t bytes_per_vector = innerSize * (sizeof(float) + sizeof(int64_t));
    uint32_t max_vectors_tile = UB_BUDGET_BYTES / ( bytes_per_vector + 128 );
    if (max_vectors_tile > 256) max_vectors_tile = 256;
    if (max_vectors_tile == 0) max_vectors_tile = 1;
    uint32_t tileVectors = max_vectors_tile;
    if (tileVectors > outerSize) tileVectors = outerSize;

    uint32_t tileLength = tileVectors * innerSize;
    uint32_t blockDim = CeilDiv(outerSize, tileVectors);
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 8) blockDim = 8;

    context->SetBlockDim(blockDim);

    tiling.SetTotalLength(totalLength);
    tiling.SetInnerSize(innerSize);
    tiling.SetTopK(topK);
    tiling.SetOuterSize(outerSize);
    tiling.SetTileLength(tileLength);
    tiling.SetTileVectors(tileVectors);
    tiling.SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04633Custom {
public:
    __aicore__ inline KernelCa6k04633Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR values, GM_ADDR indices,
                                uint32_t totalLength, uint32_t innerSize,
                                uint32_t topK, uint32_t outerSize,
                                uint32_t tileLength, uint32_t tileVectors)
    {
        totalLen_ = totalLength;
        innerSize_ = innerSize;
        topK_ = topK;
        outerSize_ = outerSize;
        tileLen_ = tileLength;
        tileVecs_ = tileVectors;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        valuesGm_.SetGlobalBuffer((__gm__ float*)values, outerSize_ * topK_);
        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices, outerSize_ * topK_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueValues_, BUFFER_NUM, tileVecs_ * topK_ * sizeof(float));
        pipe_.InitBuffer(outQueueIndices_, BUFFER_NUM, tileVecs_ * topK_ * sizeof(int64_t));
        pipe_.InitBuffer(tmpVals_, 1, innerSize_ * sizeof(float));
        pipe_.InitBuffer(tmpIdx_, 1, innerSize_ * sizeof(int64_t));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || innerSize_ == 0 || topK_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t vecsPerBlock = tileVecs_;
        uint32_t startVec = blockIdx * vecsPerBlock;
        if (startVec >= outerSize_) return;

        while (startVec < outerSize_) {
            uint32_t curVecs = (startVec + vecsPerBlock <= outerSize_) ? vecsPerBlock : (outerSize_ - startVec);
            uint32_t curTileLen = curVecs * innerSize_;
            uint32_t gmOffset = startVec * innerSize_;
            CopyIn(gmOffset, curTileLen);
            Compute(curVecs);
            uint32_t outOffset = startVec * topK_;
            CopyOut(outOffset, curVecs);
            startVec += blockNum * vecsPerBlock;
            if (curVecs < vecsPerBlock) break;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t numVecs) {
        AscendC::LocalTensor<float> inVals = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> outVals = outQueueValues_.AllocTensor<float>();
        AscendC::LocalTensor<int64_t> outIdx = outQueueIndices_.AllocTensor<int64_t>();
        AscendC::LocalTensor<float> tmpVals = tmpVals_.Get<float>();
        AscendC::LocalTensor<int64_t> tmpIdx = tmpIdx_.Get<int64_t>();

        for (uint32_t v = 0; v < numVecs; v++) {
            uint32_t base = v * innerSize_;
            for (uint32_t i = 0; i < innerSize_; i++) {
                tmpVals(i) = inVals(base + i);
                tmpIdx(i) = static_cast<int64_t>(i);
            }
            // bubble sort descending
            for (uint32_t i = 0; i < innerSize_ - 1; i++) {
                for (uint32_t j = 0; j < innerSize_ - 1 - i; j++) {
                    if (tmpVals(j) < tmpVals(j + 1)) {
                        float t = tmpVals(j);
                        tmpVals(j) = tmpVals(j + 1);
                        tmpVals(j + 1) = t;
                        int64_t t_idx = tmpIdx(j);
                        tmpIdx(j) = tmpIdx(j + 1);
                        tmpIdx(j + 1) = t_idx;
                    }
                }
            }
            uint32_t outBase = v * topK_;
            for (uint32_t i = 0; i < topK_; i++) {
                outVals(outBase + i) = tmpVals(i);
                outIdx(outBase + i) = tmpIdx(i);
            }
        }

        outQueueValues_.EnQue<float>(outVals);
        outQueueIndices_.EnQue<int64_t>(outIdx);
        inQueueX_.FreeTensor(inVals);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t numVecs) {
        AscendC::LocalTensor<float> outVals = outQueueValues_.DeQue<float>();
        AscendC::LocalTensor<int64_t> outIdx = outQueueIndices_.DeQue<int64_t>();
        AscendC::DataCopy(valuesGm_[gmOffset], outVals, numVecs * topK_);
        AscendC::DataCopy(indicesGm_[gmOffset], outIdx, numVecs * topK_);
        outQueueValues_.FreeTensor(outVals);
        outQueueIndices_.FreeTensor(outIdx);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueValues_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueIndices_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVals_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpIdx_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> valuesGm_;
    AscendC::GlobalTensor<int64_t> indicesGm_;

    uint32_t totalLen_ {0};
    uint32_t innerSize_ {0};
    uint32_t topK_ {0};
    uint32_t outerSize_ {0};
    uint32_t tileLen_ {0};
    uint32_t tileVecs_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04633_custom(GM_ADDR x, GM_ADDR values, GM_ADDR indices,
                                                         GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04633Custom op;
    op.Init(x, values, indices,
            tiling_data.totalLength, tiling_data.innerSize, tiling_data.topK,
            tiling_data.outerSize, tiling_data.tileLength, tiling_data.tileVectors);
    op.Process();
}
