static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    // Compute output shape from attributes
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *stride_attrs = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *padding_attrs = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *output_padding_attrs = attrs->GetAttrPointer<int64_t>(2);

    int32_t n = x_shape->GetDim(0);
    int32_t c_out = w_shape->GetDim(0);
    int32_t d_in = x_shape->GetDim(2);
    int32_t h_in = x_shape->GetDim(3);
    int32_t w_in = x_shape->GetDim(4);
    int32_t kd = w_shape->GetDim(2);
    int32_t kh = w_shape->GetDim(3);
    int32_t kw = w_shape->GetDim(4);

    int32_t d_out = (d_in - 1) * stride_attrs[0] - 2 * padding_attrs[0] + (kd - 1) + output_padding_attrs[0] + 1;
    int32_t h_out = (h_in - 1) * stride_attrs[1] - 2 * padding_attrs[1] + (kh - 1) + output_padding_attrs[1] + 1;
    int32_t w_out = (w_in - 1) * stride_attrs[2] - 2 * padding_attrs[2] + (kw - 1) + output_padding_attrs[2] + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, c_out);
    y_shape->SetDim(2, d_out);
    y_shape->SetDim(3, h_out);
    y_shape->SetDim(4, w_out);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00317Custom {
public:
    __aicore__ inline KernelCa6k00317Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                const Ca6k00317CustomTilingData &tiling) {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)w);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        n_ = tiling.n;
        c_in_ = tiling.c_in;
        c_out_ = tiling.c_out;
        d_in_ = tiling.d_in;
        h_in_ = tiling.h_in;
        w_in_ = tiling.w_in;
        d_out_ = tiling.d_out;
        h_out_ = tiling.h_out;
        w_out_ = tiling.w_out;
        kd_ = tiling.kd;
        kh_ = tiling.kh;
        kw_ = tiling.kw;
        stride_d_ = tiling.stride_d;
        stride_h_ = tiling.stride_h;
        stride_w_ = tiling.stride_w;
        pad_d_ = tiling.pad_d;
        pad_h_ = tiling.pad_h;
        pad_w_ = tiling.pad_w;
        output_pad_d_ = tiling.output_pad_d;
        output_pad_h_ = tiling.output_pad_h;
        output_pad_w_ = tiling.output_pad_w;
    }

    __aicore__ inline void Process() {
        if (n_ == 0 || c_out_ == 0 || d_out_ == 0 || h_out_ == 0 || w_out_ == 0) return;

        const uint32_t blockNum = GetBlockNum();
        const uint32_t blockIdx = GetBlockIdx();

        // Each block handles one (n, co) pair
        if (blockIdx >= n_ * c_out_) return;
        uint32_t bn = blockIdx / c_out_;
        uint32_t co = blockIdx % c_out_;

        // Strides for input and weight
        uint32_t x_stride_n = c_in_ * d_in_ * h_in_ * w_in_;
        uint32_t x_stride_c = d_in_ * h_in_ * w_in_;
        uint32_t x_stride_d = h_in_ * w_in_;
        uint32_t x_stride_h = w_in_;

        uint32_t w_stride_co = c_in_ * kd_ * kh_ * kw_;
        uint32_t w_stride_ci = kd_ * kh_ * kw_;
        uint32_t w_stride_kd = kh_ * kw_;
        uint32_t w_stride_kh = kw_;

        uint32_t y_stride_n = c_out_ * d_out_ * h_out_ * w_out_;
        uint32_t y_stride_co = d_out_ * h_out_ * w_out_;
        uint32_t y_stride_d = h_out_ * w_out_;
        uint32_t y_stride_h = w_out_;

        // Compute output for this (bn, co)
        for (uint32_t od = 0; od < d_out_; od++) {
            for (uint32_t oh = 0; oh < h_out_; oh++) {
                for (uint32_t ow = 0; ow < w_out_; ow++) {
                    float sum = 0.0f;
                    // Map output position back to input range
                    for (uint32_t kd = 0; kd < kd_; kd++) {
                        // id = (od - kd - output_pad_d_ + pad_d_) / stride_d_ , but we need integer check
                        int32_t id = (static_cast<int32_t>(od) - static_cast<int32_t>(kd) - static_cast<int32_t>(output_pad_d_) + pad_d_);
                        if (id < 0) continue;
                        if (id % stride_d_ != 0) continue;
                        id /= stride_d_;
                        if (static_cast<uint32_t>(id) >= d_in_) continue;
                        for (uint32_t kh = 0; kh < kh_; kh++) {
                            int32_t ih = (static_cast<int32_t>(oh) - static_cast<int32_t>(kh) - static_cast<int32_t>(output_pad_h_) + pad_h_);
                            if (ih < 0) continue;
                            if (ih % stride_h_ != 0) continue;
                            ih /= stride_h_;
                            if (static_cast<uint32_t>(ih) >= h_in_) continue;
                            for (uint32_t kw = 0; kw < kw_; kw++) {
                                int32_t iw = (static_cast<int32_t>(ow) - static_cast<int32_t>(kw) - static_cast<int32_t>(output_pad_w_) + pad_w_);
                                if (iw < 0) continue;
                                if (iw % stride_w_ != 0) continue;
                                iw /= stride_w_;
                                if (static_cast<uint32_t>(iw) >= w_in_) continue;
                                // Sum over input channel
                                for (uint32_t ci = 0; ci < c_in_; ci++) {
                                    uint32_t x_idx = bn * x_stride_n + ci * x_stride_c + static_cast<uint32_t>(id) * x_stride_d + static_cast<uint32_t>(ih) * x_stride_h + static_cast<uint32_t>(iw);
                                    uint32_t w_idx = co * w_stride_co + ci * w_stride_ci + kd * w_stride_kd + kh * w_stride_kh + kw;
                                    sum += xGm_.GetValue(x_idx) * wGm_.GetValue(w_idx);
                                }
                            }
                        }
                    }
                    uint32_t y_idx = bn * y_stride_n + co * y_stride_co + od * y_stride_d + oh * y_stride_h + ow;
                    yGm_.SetValue(y_idx, sum);
                }
            }
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t n_, c_in_, c_out_, d_in_, h_in_, w_in_, d_out_, h_out_, w_out_;
    uint32_t kd_, kh_, kw_;
    int32_t stride_d_, stride_h_, stride_w_;
    int32_t pad_d_, pad_h_, pad_w_;
    int32_t output_pad_d_, output_pad_h_, output_pad_w_;
};

extern "C" __global__ __aicore__ void ca6k00317_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00317Custom op;
    op.Init(x, w, y, tiling_data);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias):
        super(ModelNew, self).__init__()
        # LazyConvTranspose3d will create weight with correct shape
        # We store parameters to create weight lazily
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.bias = bias
        # Weight will be created on first forward call
        self.weight = None

    def forward(self, x):
        if self.weight is None:
            # Create weight (out_channels, in_channels, *kernel_size)
            self.weight = torch.nn.Parameter(
                torch.empty(self.out_channels, x.shape[1], *self.kernel_size, device=x.device)
            )
            torch.nn.init.kaiming_uniform_(self.weight, a=0.0)
        return custom_ops_lib.ca6k00317_custom(x, self.weight, self.stride, self.padding, self.output_padding)
