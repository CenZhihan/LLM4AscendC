    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_scale(scale);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k05321Custom {
public:
    __aicore__ inline KernelCa6k05321Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t blockDim,
                                uint32_t N, uint32_t C,
                                uint32_t H, uint32_t W,
                                uint32_t H_out, uint32_t W_out,
                                float scale)
    {
        totalLen_ = totalLength;
        blockDim_ = blockDim;
        N_ = N; C_ = C; H_ = H; W_ = W;
        H_out_ = H_out; W_out_ = W_out;
        scale_ = scale;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t chunk = (totalLen_ + blockDim_ - 1) / blockDim_;
        uint32_t start = blockIdx * chunk;
        uint32_t end = (start + chunk > totalLen_) ? totalLen_ : start + chunk;

        for (uint32_t idx = start; idx < end; ++idx) {
            // decode flat index to (n, c, oy, ox)
            uint32_t nchw = idx;
            uint32_t ox = nchw % W_out_;
            nchw /= W_out_;
            uint32_t oy = nchw % H_out_;
            nchw /= H_out_;
            uint32_t c = nchw % C_;
            uint32_t n = nchw / C_;

            // source coordinates (align_corners=False, scale_factor)
            float srcY = static_cast<float>(oy) / scale_;
            float srcX = static_cast<float>(ox) / scale_;

            int32_t y0 = static_cast<int32_t>(srcY);
            int32_t x0 = static_cast<int32_t>(srcX);
            int32_t y1 = (y0 + 1 < static_cast<int32_t>(H_)) ? y0 + 1 : y0;
            int32_t x1 = (x0 + 1 < static_cast<int32_t>(W_)) ? x0 + 1 : x0;

            float dy = srcY - static_cast<float>(y0);
            float dx = srcX - static_cast<float>(x0);

            // read four input values
            uint32_t base = n * C_ * H_ * W_ + c * H_ * W_;
            float v00 = xGm_[base + y0 * W_ + x0];
            float v01 = xGm_[base + y0 * W_ + x1];
            float v10 = xGm_[base + y1 * W_ + x0];
            float v11 = xGm_[base + y1 * W_ + x1];

            // bilinear interpolation
            float top = v00 + (v01 - v00) * dx;
            float bottom = v10 + (v11 - v10) * dx;
            float result = top + (bottom - top) * dy;

            yGm_[idx] = result;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t blockDim_ {1};
    uint32_t N_ {1};
    uint32_t C_ {1};
    uint32_t H_ {1};
    uint32_t W_ {1};
    uint32_t H_out_ {1};
    uint32_t W_out_ {1};
    float scale_ {2.0f};
};

extern "C" __global__ __aicore__ void ca6k_05321_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k05321Custom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.blockDim,
            tiling_data.N, tiling_data.C,
            tiling_data.H, tiling_data.W,
            tiling_data.H_out, tiling_data.W_out,
            tiling_data.scale);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = "ca6k_05321_custom"

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, scale_factor):
        super(ModelNew, self).__init__()
        self.scale_factor = scale_factor
        # note: conv is not fused; keep as native op on NPU
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1)

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        x_up = fn(x, self.scale_factor)
        x = self.conv(x_up)
        return x
'''
