kernel_src = """
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04600Custom {
public:
    __aicore__ inline KernelCa6k04600Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR running_mean, GM_ADDR running_var,
                                GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalLength, uint32_t tileLength, uint32_t blockDim,
                                float eps, uint32_t numFeatures, uint32_t spatialSize,
                                uint8_t training) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        eps_ = eps;
        numFeatures_ = numFeatures;
        spatialSize_ = spatialSize;
        training_ = training;
        numRows_ = totalLength / spatialSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        meanGm_.SetGlobalBuffer((__gm__ float*)running_mean, numFeatures_);
        varGm_.SetGlobalBuffer((__gm__ float*)running_var, numFeatures_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numFeatures_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, numFeatures_);

        // Buffers for per-channel parameters (read from global once)
        pipe_.InitBuffer(meanBuf_, numFeatures_ * sizeof(float));
        meanLocal_ = meanBuf_.Get<float>();
        pipe_.InitBuffer(varBuf_, numFeatures_ * sizeof(float));
        varLocal_ = varBuf_.Get<float>();
        pipe_.InitBuffer(weightBuf_, numFeatures_ * sizeof(float));
        weightLocal_ = weightBuf_.Get<float>();
        pipe_.InitBuffer(biasBuf_, numFeatures_ * sizeof(float));
        biasLocal_ = biasBuf_.Get<float>();

        DataCopy(meanLocal_, meanGm_, numFeatures_);
        DataCopy(varLocal_, varGm_, numFeatures_);
        DataCopy(weightLocal_, weightGm_, numFeatures_);
        DataCopy(biasLocal_, biasGm_, numFeatures_);
    }

    __aicore__ inline void Process() {
        if (numRows_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        for (uint32_t row = blockIdx; row < numRows_; row += blockNum) {
            uint32_t channel = row % numFeatures_;

            float mean_val, var_val;
            if (training_ != 0) {
                // Compute mean and variance from the data for this channel
                float sum = 0.0f, sum_sq = 0.0f;
                uint32_t base_row = row * spatialSize_;
                for (uint32_t j = 0; j < spatialSize_; ++j) {
                    float val = xGm_.GetValue(base_row + j);
                    sum += val;
                    sum_sq += val * val;
                }
                float invN = 1.0f / (float)spatialSize_;
                mean_val = sum * invN;
                float muSq = mean_val * mean_val;
                var_val = sum_sq * invN - muSq;
                // Use unbiased variance? Instance norm typically uses biased variance (N)
                // F.instance_norm uses unbiased when training? Actually it uses the same as batch norm: biased when training? Check documentation: "The mean and standard-deviation are calculated per-dimension separately for each object in a mini-batch. Also, the argument `affine` is not used here. The formula is: y = (x - mean) / sqrt(var + eps) * gamma + beta."
                // Standard instance norm uses biased variance (divide by N, not N-1). So we keep as is.
            } else {
                mean_val = meanLocal_.GetValue(channel);
                var_val = varLocal_.GetValue(channel);
            }

            float weight_val = weightLocal_.GetValue(channel);
            float bias_val = biasLocal_.GetValue(channel);

            float inv_std = 1.0f / Sqrt(var_val + eps_);
            float scale = weight_val * inv_std;
            float shift = bias_val - scale * mean_val;

            uint32_t base = row * spatialSize_;
            uint32_t remaining = spatialSize_;

            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

                LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                DataCopy(xLocal, xGm_[base], curLen);
                inQueueX_.EnQue(xLocal);

                LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
                LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                Muls(yLocal, xDeque, scale, curLen);
                Adds(yLocal, yLocal, shift, curLen);
                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xDeque);

                LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
                DataCopy(yGm_[base], yDeque, curLen);
                outQueueY_.FreeTensor(yDeque);

                base += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    TBuf<TPosition::VECCALC> meanBuf_;
    TBuf<TPosition::VECCALC> varBuf_;
    TBuf<TPosition::VECCALC> weightBuf_;
    TBuf<TPosition::VECCALC> biasBuf_;
    LocalTensor<float> meanLocal_;
    LocalTensor<float> varLocal_;
    LocalTensor<float> weightLocal_;
    LocalTensor<float> biasLocal_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> meanGm_;
    GlobalTensor<float> varGm_;
    GlobalTensor<float> weightGm_;
    GlobalTensor<float> biasGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
    float eps_ {1e-5f};
    uint32_t numFeatures_ {0};
    uint32_t spatialSize_ {0};
    uint32_t numRows_ {0};
    uint8_t training_ {1};
};

extern "C" __global__ __aicore__ void ca6k_04600_custom(GM_ADDR x, GM_ADDR y, GM_ADDR running_mean, GM_ADDR running_var,
                                                         GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04600Custom op;
    op.Init(x, y, running_mean, running_var, weight, bias,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim,
            tiling_data.eps, tiling_data.numFeatures, tiling_data.spatialSize,
            tiling_data.training);
    op.Process();
}

model_src = """
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_04600_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_features, eps, momentum):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.eps = eps
        self.momentum = momentum
        self.running_mean = torch.nn.Parameter(torch.zeros(num_features), requires_grad=False)
        self.running_var = torch.nn.Parameter(torch.ones(num_features), requires_grad=False)
        self.weight = torch.nn.Parameter(torch.ones(num_features))
        self.bias = torch.nn.Parameter(torch.zeros(num_features))

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.running_mean, self.running_var, self.weight, self.bias, self.eps, self.training)
