kernel_src = """

#include "kernel_operator.h"
#include <float.h>

using namespace AscendC;

class KernelMaxPool3d {
public:
    __aicore__ inline KernelMaxPool3d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t channels,
                                uint32_t inDepth, uint32_t inHeight, uint32_t inWidth,
                                uint32_t outDepth, uint32_t outHeight, uint32_t outWidth,
                                int32_t kernelD, int32_t kernelH, int32_t kernelW,
                                int32_t strideD, int32_t strideH, int32_t strideW,
                                int32_t padD, int32_t padH, int32_t padW)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        batchSize_ = batchSize;
        channels_ = channels;
        inDepth_ = inDepth;
        inHeight_ = inHeight;
        inWidth_ = inWidth;
        outDepth_ = outDepth;
        outHeight_ = outHeight;
        outWidth_ = outWidth;
        kernelD_ = kernelD;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        padD_ = padD;
        padH_ = padH;
        padW_ = padW;
        totalOut_ = batchSize_ * channels_ * outDepth_ * outHeight_ * outWidth_;
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        for (uint32_t idx = blockIdx; idx < totalOut_; idx += blockNum) {
            uint32_t tmp = idx;
            uint32_t w = tmp % outWidth_; tmp /= outWidth_;
            uint32_t h = tmp % outHeight_; tmp /= outHeight_;
            uint32_t d = tmp % outDepth_;
            tmp /= outDepth_;
            uint32_t c = tmp % channels_;
            uint32_t n = tmp / channels_;

            int32_t h_start = static_cast<int32_t>(h * strideH_) - padH_;
            int32_t w_start = static_cast<int32_t>(w * strideW_) - padW_;
            int32_t d_start = static_cast<int32_t>(d * strideD_) - padD_;

            float maxVal = -FLT_MAX;
            for (int32_t kz = 0; kz < kernelD_; ++kz) {
                int32_t in_d = d_start + kz;
                if (in_d < 0 || in_d >= static_cast<int32_t>(inDepth_)) continue;
                for (int32_t ky = 0; ky < kernelH_; ++ky) {
                    int32_t in_h = h_start + ky;
                    if (in_h < 0 || in_h >= static_cast<int32_t>(inHeight_)) continue;
                    for (int32_t kx = 0; kx < kernelW_; ++kx) {
                        int32_t in_w = w_start + kx;
                        if (in_w < 0 || in_w >= static_cast<int32_t>(inWidth_)) continue;
                        uint32_t inIdx = ((n * channels_ + c) * inDepth_ + in_d) * inHeight_ * inWidth_ + in_h * inWidth_ + in_w;
                        float val = xGm_.GetValue(inIdx);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }
            yGm_.SetValue(idx, maxVal);
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    uint32_t batchSize_, channels_, inDepth_, inHeight_, inWidth_;
    uint32_t outDepth_, outHeight_, outWidth_;
    uint32_t totalOut_;
    int32_t kernelD_, kernelH_, kernelW_;
    int32_t strideD_, strideH_, strideW_;
    int32_t padD_, padH_, padW_;
};

extern "C" __global__ __aicore__ void ca6k_00289_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPool3d op;
    op.Init(x, y,
            tiling_data.batchSize, tiling_data.channels,
            tiling_data.inDepth, tiling_data.inHeight, tiling_data.inWidth,
            tiling_data.outDepth, tiling_data.outHeight, tiling_data.outWidth,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW);
    op.Process();
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00289_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.kd, self.kh, self.kw = kernel_size
        self.sd, self.sh, self.sw = stride
        self.pd, self.ph, self.pw = padding

    def forward(self, *inputs):
        x = inputs[0]
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kd, self.kh, self.kw,
                  self.sd, self.sh, self.sw,
                  self.pd, self.ph, self.pw)
'''
