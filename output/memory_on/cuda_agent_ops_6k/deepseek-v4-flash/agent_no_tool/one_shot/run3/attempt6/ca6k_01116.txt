    uint32_t C = xShape->GetOriginShape().GetDim(1);
    uint32_t L = xShape->GetOriginShape().GetDim(2);
    uint32_t O = wShape->GetOriginShape().GetDim(0);
    uint32_t K = wShape->GetOriginShape().GetDim(2);

kernel_src = '''
#include "ca6k01116_custom_tiling.h"
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01116Custom {
public:
    __aicore__ inline KernelCa6k01116Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t L, uint32_t O, uint32_t K,
                                int32_t stride, int32_t padding, int32_t dilation, uint32_t groups,
                                uint32_t L_out, uint32_t totalOutput, uint32_t hasBias,
                                uint32_t blockDim)
    {
        N_ = N;
        C_ = C;
        L_ = L;
        O_ = O;
        K_ = K;
        stride_ = stride;
        padding_ = padding;
        dilation_ = dilation;
        groups_ = groups;
        L_out_ = L_out;
        totalOutput_ = totalOutput;
        hasBias_ = hasBias;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * L_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, O_ * (C_ / groups_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutput_);

        uint32_t sliceSize = (C_ / groups_) * K_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, sliceSize * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, sliceSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, sliceSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutput_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // grid stride over output elements
        uint32_t start = blockIdx;
        uint32_t step = blockNum;
        for (uint32_t globalOutIdx = start; globalOutIdx < totalOutput_; globalOutIdx += step) {
            uint32_t n = globalOutIdx / (O_ * L_out_);
            uint32_t rem = globalOutIdx % (O_ * L_out_);
            uint32_t o = rem / L_out_;
            uint32_t l_out = rem % L_out_;
            int32_t l_in_start = static_cast<int32_t>(l_out) * stride_ - padding_;

            uint32_t sliceSize = (C_ / groups_) * K_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

            // Load input slice for this output
            for (uint32_t c = 0; c < (C_ / groups_); ++c) {
                for (uint32_t k = 0; k < K_; ++k) {
                    int32_t idx = l_in_start + static_cast<int32_t>(k) * dilation_;
                    float val = 0.0f;
                    if (idx >= 0 && idx < static_cast<int32_t>(L_)) {
                        val = xGm_.GetValue(n * C_ * L_ + c * L_ + static_cast<uint32_t>(idx));
                    }
                    xLocal.SetValue(c * K_ + k, val);
                }
            }

            // Load weight slice for output channel o
            uint32_t wOffset = o * (C_ / groups_) * K_;
            for (uint32_t i = 0; i < sliceSize; ++i) {
                wLocal.SetValue(i, wGm_.GetValue(wOffset + i));
            }

            // Compute dot product
            float sum = 0.0f;
            for (uint32_t i = 0; i < sliceSize; ++i) {
                sum += xLocal.GetValue(i) * wLocal.GetValue(i);
            }

            // Add bias if present
            if (hasBias_) {
                sum += bGm_.GetValue(o);
            }

            // Store output
            yGm_.SetValue(globalOutIdx, sum);

            // Free tensors
            inQueueX_.FreeTensor(xLocal);
            inQueueW_.FreeTensor(wLocal);
            tmpQueue_.FreeTensor(tmp);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t L_ {0};
    uint32_t O_ {0};
    uint32_t K_ {0};
    int32_t stride_ {1};
    int32_t padding_ {0};
    int32_t dilation_ {1};
    uint32_t groups_ {1};
    uint32_t L_out_ {0};
    uint32_t totalOutput_ {0};
    uint32_t hasBias_ {0};
    uint32_t blockDim_ {8};
};

extern "C" __global__ __aicore__ void ca6k01116_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01116Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.C, tiling_data.L, tiling_data.O, tiling_data.K,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation, tiling_data.groups,
            tiling_data.L_out, tiling_data.totalOutput, tiling_data.hasBias,
            tiling_data.blockDim);
    op.Process();
}
'''

                          c10::optional<at::Tensor> bias,
                          int64_t stride, int64_t padding, int64_t dilation, int64_t groups) {
    at::Tensor result = at::empty({}, self.options());
    if (bias.has_value()) {
        EXEC_NPU_CMD(aclnnCa6k01116Custom, self, weight, bias.value(), result, stride, padding, dilation, groups);

        EXEC_NPU_CMD(aclnnCa6k01116Custom, self, weight, emptyBias, result, stride, padding, dilation, groups);
    }
    return result;

model_src = '''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0, dilation=1, groups=1):
        super(ModelNew, self).__init__()
        # Use nn.Conv1d to get identical weight/bias initialization as reference
        self.conv = nn.Conv1d(in_channels, out_channels, kernel_size,
                              stride=stride, padding=padding, dilation=dilation,
                              groups=groups, bias=True)
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups

    def forward(self, x):
        fn = getattr(custom_ops_lib, 'ca6k01116_custom')
        return fn(x, self.conv.weight, self.conv.bias,
                  self.stride, self.padding, self.dilation, self.groups)
