#include "ca6k_00417_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00417CustomTilingData tiling;

    auto xShape = context->GetInputShape(0)->GetOriginShape();
    int32_t n = xShape.GetDim(0);
    int32_t c = xShape.GetDim(1);
    int32_t h = xShape.GetDim(2);
    int32_t w = xShape.GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(2);

    int32_t kernel_h = static_cast<int32_t>(kernel_size[0]);
    int32_t kernel_w = static_cast<int32_t>(kernel_size[1]);
    int32_t stride_h = static_cast<int32_t>(stride[0]);
    int32_t stride_w = static_cast<int32_t>(stride[1]);
    int32_t pad_h = static_cast<int32_t>(padding[0]);
    int32_t pad_w = static_cast<int32_t>(padding[1]);

    int32_t out_h = (h + 2 * pad_h - kernel_h) / stride_h + 1;
    int32_t out_w = (w + 2 * pad_w - kernel_w) / stride_w + 1;

    uint32_t totalLength = static_cast<uint32_t>(n) * static_cast<uint32_t>(c) *
                           static_cast<uint32_t>(out_h) * static_cast<uint32_t>(out_w);

    uint32_t maxTile = UB_BUDGET_BYTES / 4;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_kernel_h(kernel_h);
    tiling.set_kernel_w(kernel_w);
    tiling.set_stride_h(stride_h);
    tiling.set_stride_w(stride_w);
    tiling.set_pad_h(pad_h);
    tiling.set_pad_w(pad_w);
    tiling.set_out_h(out_h);
    tiling.set_out_w(out_w);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

#include "kernel_operator.h"
#include <limits>

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00417Custom {
public:
    __aicore__ inline KernelCa6k00417Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t n, int32_t c, int32_t h, int32_t w,
                                int32_t kernel_h, int32_t kernel_w,
                                int32_t stride_h, int32_t stride_w,
                                int32_t pad_h, int32_t pad_w,
                                int32_t out_h, int32_t out_w)
    {
        totalLength_ = totalLength;
        tileLength_ = tileLength;
        n_ = n;
        c_ = c;
        h_ = h;
        w_ = w;
        kernel_h_ = kernel_h;
        kernel_w_ = kernel_w;
        stride_h_ = stride_h;
        stride_w_ = stride_w;
        pad_h_ = pad_h;
        pad_w_ = pad_w;
        out_h_ = out_h;
        out_w_ = out_w;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * c_ * h_ * w_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength_);
    }

    __aicore__ inline void Process()
    {
        if (totalLength_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLength_;
        uint32_t base = blockIdx * tileLength_;

        while (base < totalLength_) {
            uint32_t curLen = totalLength_ - base;
            if (curLen > tileLength_) curLen = tileLength_;
            ComputeRange(base, base + curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeRange(uint32_t start, uint32_t end)
    {
        for (uint32_t idx = start; idx < end; ++idx) {
            uint32_t w_out = idx % static_cast<uint32_t>(out_w_);
            uint32_t tmp = idx / static_cast<uint32_t>(out_w_);
            uint32_t h_out = tmp % static_cast<uint32_t>(out_h_);
            tmp /= static_cast<uint32_t>(out_h_);
            uint32_t c_idx = tmp % static_cast<uint32_t>(c_);
            uint32_t n_idx = tmp / static_cast<uint32_t>(c_);

            int h_in_start = static_cast<int>(h_out) * stride_h_ - pad_h_;
            int w_in_start = static_cast<int>(w_out) * stride_w_ - pad_w_;

            float max_val = -std::numeric_limits<float>::infinity();
            for (int kh = 0; kh < kernel_h_; ++kh) {
                int h_in = h_in_start + kh;
                if (h_in < 0 || h_in >= h_) continue;
                for (int kw = 0; kw < kernel_w_; ++kw) {
                    int w_in = w_in_start + kw;
                    if (w_in < 0 || w_in >= w_) continue;
                    uint32_t in_idx = ((n_idx * static_cast<uint32_t>(c_) + c_idx) * static_cast<uint32_t>(h_) + static_cast<uint32_t>(h_in)) * static_cast<uint32_t>(w_) + static_cast<uint32_t>(w_in);
                    float val = xGm_[in_idx];
                    if (val > max_val) max_val = val;
                }
            }
            yGm_[idx] = max_val;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLength_ {0};
    uint32_t tileLength_ {0};

    int32_t n_ {0};
    int32_t c_ {0};
    int32_t h_ {0};
    int32_t w_ {0};
    int32_t kernel_h_ {0};
    int32_t kernel_w_ {0};
    int32_t stride_h_ {1};
    int32_t stride_w_ {1};
    int32_t pad_h_ {0};
    int32_t pad_w_ {0};
    int32_t out_h_ {0};
    int32_t out_w_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00417_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00417Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kernel_h, tiling_data.kernel_w,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.out_h, tiling_data.out_w);
    op.Process();
}
