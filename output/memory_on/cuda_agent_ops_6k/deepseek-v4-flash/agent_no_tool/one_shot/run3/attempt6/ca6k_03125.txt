#include "ca6k_03125_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k03125CustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t* splitSizesPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t splitSize0 = splitSizesPtr[0];
    int64_t splitSize1 = splitSizesPtr[1];

    const gert::StorageShape *xShape = context->GetInputShape(0);
    const uint32_t dimNum = xShape->GetOriginShape().GetDimNum();
    const uint32_t innerDim = (dimNum >= 2) ? xShape->GetOriginShape().GetDim(1) : 1u;
    const uint32_t totalLength0 = static_cast<uint32_t>(splitSize0) * innerDim;
    const uint32_t totalLength1 = static_cast<uint32_t>(splitSize1) * innerDim;

    uint32_t blockDim = 2u;
    context->SetBlockDim(blockDim);

    uint32_t tileLen = AlignDown(UB_BUDGET_BYTES / sizeof(float), ALIGN_ELEMS);
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;
    uint32_t maxTile0 = AlignDown(totalLength0, ALIGN_ELEMS);
    uint32_t maxTile1 = AlignDown(totalLength1, ALIGN_ELEMS);
    if (tileLen > maxTile0) tileLen = maxTile0;
    if (tileLen > maxTile1) tileLen = maxTile1;
    if (tileLen == 0) tileLen = ALIGN_ELEMS;

    tiling.set_totalLength0(totalLength0);
    tiling.set_totalLength1(totalLength1);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_splitSize0(static_cast<int32_t>(splitSize0));
    tiling.set_splitSize1(static_cast<int32_t>(splitSize1));
    tiling.set_innerDim(innerDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k03125Custom {
public:
    __aicore__ inline KernelCa6k03125Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR out0, GM_ADDR out1,
                                uint32_t totalLen0, uint32_t totalLen1,
                                uint32_t tileLen, int32_t splitSize0, int32_t splitSize1,
                                uint32_t innerDim)
    {
        totalLen0_ = totalLen0;
        totalLen1_ = totalLen1;
        tileLen_ = tileLen;
        splitSize0_ = splitSize0;
        splitSize1_ = splitSize1;
        innerDim_ = innerDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen0_ + totalLen1_);
        out0Gm_.SetGlobalBuffer((__gm__ float*)out0, totalLen0_);
        out1Gm_.SetGlobalBuffer((__gm__ float*)out1, totalLen1_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx == 0) {
            ProcessOne(totalLen0_, xGm_, out0Gm_, 0u);
        } else if (blockIdx == 1) {
            uint32_t xOffset = static_cast<uint32_t>(splitSize0_) * innerDim_;
            ProcessOne(totalLen1_, xGm_, out1Gm_, xOffset);
        }
    }

private:
    __aicore__ inline void ProcessOne(uint32_t totalLen,
                                      AscendC::GlobalTensor<float>& srcGm,
                                      AscendC::GlobalTensor<float>& dstGm,
                                      uint32_t srcOffset)
    {
        uint32_t offset = 0;
        while (offset < totalLen) {
            uint32_t curLen = (totalLen - offset > tileLen_) ? tileLen_ : (totalLen - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, srcGm[srcOffset + offset], curLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, xLocal, curLen);
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);

            AscendC::LocalTensor<float> yLocalDeq = outQueueY_.DeQue<float>();
            AscendC::DataCopy(dstGm[offset], yLocalDeq, curLen);
            outQueueY_.FreeTensor(yLocalDeq);

            offset += curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> out0Gm_;
    AscendC::GlobalTensor<float> out1Gm_;

    uint32_t totalLen0_ {0};
    uint32_t totalLen1_ {0};
    uint32_t tileLen_ {0};
    int32_t splitSize0_ {0};
    int32_t splitSize1_ {0};
    uint32_t innerDim_ {1};
};

extern "C" __global__ __aicore__ void ca6k_03125_custom(GM_ADDR x, GM_ADDR out0, GM_ADDR out1,
                                                       GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k03125Custom op;
    op.Init(x, out0, out1,
            tiling_data.totalLength0, tiling_data.totalLength1,
            tiling_data.tileLength, tiling_data.splitSize0, tiling_data.splitSize1,
            tiling_data.innerDim);
    op.Process();
}
