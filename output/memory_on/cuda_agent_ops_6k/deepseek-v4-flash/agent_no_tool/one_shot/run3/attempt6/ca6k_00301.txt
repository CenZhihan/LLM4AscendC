host_operator_src="""

#include "ca6k00301_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00301CustomTilingData tiling;

    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::StorageShape *y_storage = context->GetInputShape(1);
    const gert::Shape &x_shape = x_storage->GetOriginShape();
    const gert::Shape &y_shape = y_storage->GetOriginShape();
    if (x_shape.GetDimNum() != 3 || y_shape.GetDimNum() != 3)
        return ge::GRAPH_FAILED;

    uint32_t batch = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t M = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t K = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t N = static_cast<uint32_t>(y_shape.GetDim(2));

    // Compute conservative tile sizes that fit in UB
    uint32_t available = (UB_BUDGET_BYTES * 3 / 4) / sizeof(float);
    uint32_t tileM = 64;
    uint32_t tileN = 64;
    uint32_t tileK = 128;
    while (tileM * tileK + tileK * tileN + tileM * tileN > available && tileM > 16) {
        tileM /= 2; tileN /= 2; tileK /= 2;
    }
    if (tileM < 16) { tileM = 16; tileN = 16; tileK = 16; }

    tiling.set_batch(batch);
    tiling.set_m(M);
    tiling.set_n(N);
    tiling.set_k(K);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    uint32_t blockDim = batch;
    context->SetBlockDim(blockDim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *y_shape = context->GetInputShape(1);
    gert::Shape *z_shape = context->GetOutputShape(0);
    int64_t dims[3] = {x_shape->GetDim(0), x_shape->GetDim(1), y_shape->GetDim(2)};
    z_shape->SetDimNum(3);
    for (int64_t i = 0; i < 3; ++i)
        z_shape->SetDim(static_cast<int64_t>(i), dims[i]);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

}  // namespace ge
