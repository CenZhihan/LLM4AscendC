host_operator_src = '''
#include "ca6k_01135_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t MAX_TILE_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k01135CustomTilingData tiling;

    auto originShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(3));

    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    float scale = *(attrs->GetAttrPointer<float>(0));
    uint32_t H_out = static_cast<uint32_t>(H * scale);
    uint32_t W_out = static_cast<uint32_t>(W * scale);

    uint32_t totalOutputElems = N * C * H_out * W_out;
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * 2);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalOutputElems < tileLen) {
        tileLen = AlignDown(totalOutputElems, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalOutputElems;
    }

    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(totalOutputElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_inputH(H);
    tiling.set_inputW(W);
    tiling.set_outputH(H_out);
    tiling.set_outputW(W_out);
    tiling.set_scaleFactor(scale);
    tiling.set_totalOutputLength(totalOutputElems);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape* x_shape = context->GetInputShape(0);
    int64_t N = x_shape->GetDim(0);
    int64_t C = x_shape->GetDim(1);
    int64_t H = x_shape->GetDim(2);
    int64_t W = x_shape->GetDim(3);
    float scale = 2.0f;
    if (context->GetAttrs() != nullptr) {
        const gert::RuntimeAttrs *attrs = context->GetAttrs();
        scale = *(attrs->GetAttrPointer<float>(0));
    }
    int64_t H_out = static_cast<int64_t>(H * scale);
    int64_t W_out = static_cast<int64_t>(W * scale);
    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, H_out);
    y_shape->SetDim(3, W_out);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void ca6k_01135_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    float* __gm__ xGm = (__gm__ float*)x;
    float* __gm__ yGm = (__gm__ float*)y;

    uint32_t N = tiling_data.N;
    uint32_t C = tiling_data.C;
    uint32_t H_in = tiling_data.inputH;
    uint32_t W_in = tiling_data.inputW;
    uint32_t H_out = tiling_data.outputH;
    uint32_t W_out = tiling_data.outputW;
    float scale = tiling_data.scaleFactor;
    uint32_t totalOut = tiling_data.totalOutputLength;

    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t tileSize = (totalOut + blockNum - 1) / blockNum;
    uint32_t start = blockIdx * tileSize;
    uint32_t end = (blockIdx + 1) * tileSize;
    if (end > totalOut) end = totalOut;

    for (uint32_t idx = start; idx < end; idx++) {
        uint32_t tmp = idx;
        uint32_t ow = tmp % W_out; tmp /= W_out;
        uint32_t oh = tmp % H_out; tmp /= H_out;
        uint32_t c = tmp % C; tmp /= C;
        uint32_t n = tmp;
        // nearest neighbor mapping
        uint32_t iw = static_cast<uint32_t>(ow / scale);
        uint32_t ih = static_cast<uint32_t>(oh / scale);
        if (iw >= W_in) iw = W_in - 1;
        if (ih >= H_in) ih = H_in - 1;
        uint32_t inIdx = ((n * C + c) * H_in + ih) * W_in + iw;
        yGm[idx] = xGm[inIdx];
    }
}
'''
