#include "ca6k00285_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00285CustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const auto &origin = x_shape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t C = static_cast<uint32_t>(origin.GetDim(1));
    uint32_t H = static_cast<uint32_t>(origin.GetDim(2));
    uint32_t W = static_cast<uint32_t>(origin.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *factor_ptr = attrs->GetAttrPointer<int64_t>(0);
    uint32_t factor = static_cast<uint32_t>(*factor_ptr);

    uint32_t C_out = C * factor * factor;
    uint32_t H_out = H / factor;
    uint32_t W_out = W / factor;
    uint32_t totalLength = N * C_out * H_out * W_out;
    uint32_t inputLength = N * C * H * W;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = (maxTile / ALIGN_ELEMS) * ALIGN_ELEMS;
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = (totalLength / ALIGN_ELEMS) * ALIGN_ELEMS;
        if (tileLen == 0) tileLen = totalLength;
    }

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_inputLength(inputLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_factor(factor);
    tiling.set_C_in(C);
    tiling.set_H_in(H);
    tiling.set_W_in(W);
    tiling.set_C_out(C_out);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t C = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t H = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t W = static_cast<uint32_t>(x_shape->GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs) {
        const int64_t *factor_ptr = attrs->GetAttrPointer<int64_t>(0);
        uint32_t factor = static_cast<uint32_t>(*factor_ptr);
        uint32_t C_out = C * factor * factor;
        uint32_t H_out = H / factor;
        uint32_t W_out = W / factor;
        gert::Shape *y_shape = context->GetOutputShape(0);
        y_shape->SetDimNum(4);
        y_shape->SetDim(0, static_cast<int64_t>(N));
        y_shape->SetDim(1, static_cast<int64_t>(C_out));
        y_shape->SetDim(2, static_cast<int64_t>(H_out));
        y_shape->SetDim(3, static_cast<int64_t>(W_out));
    } else {
        gert::Shape *y_shape = context->GetOutputShape(0);
        *y_shape = *x_shape;
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge
