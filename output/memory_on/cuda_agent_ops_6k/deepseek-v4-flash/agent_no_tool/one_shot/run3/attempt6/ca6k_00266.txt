#include "ca6k_00266_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t TILE_HARD_CAP = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00266CustomTilingData tiling;

    const gert::StorageShape *indices_storage = context->GetInputShape(0);
    const gert::StorageShape *offsets_storage = context->GetInputShape(1);
    const gert::StorageShape *weight_storage = context->GetInputShape(2);

    uint32_t totalIndices = static_cast<uint32_t>(indices_storage->GetOriginShape().GetShapeSize());
    uint32_t numOffsets = static_cast<uint32_t>(offsets_storage->GetOriginShape().GetShapeSize());
    uint32_t numEmbeddings = static_cast<uint32_t>(weight_storage->GetOriginShape().GetDim(0));
    uint32_t embeddingDim = static_cast<uint32_t>(weight_storage->GetOriginShape().GetDim(1));
    uint32_t totalBags = numOffsets - 1;

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > TILE_HARD_CAP) maxTile = AlignDown(TILE_HARD_CAP, ALIGN_ELEMS);
    uint32_t tileDim = (embeddingDim < maxTile) ? embeddingDim : maxTile;
    tileDim = AlignDown(tileDim, ALIGN_ELEMS);
    if (tileDim == 0) tileDim = ALIGN_ELEMS;

    // Set block dimension (will be used by kernel for grid-stride)
    uint32_t blockDim = 8;
    if (totalBags < blockDim) blockDim = totalBags;
    if (blockDim < 1) blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalBags(totalBags);
    tiling.set_embeddingDim(embeddingDim);
    tiling.set_numEmbeddings(numEmbeddings);
    tiling.set_tileDim(tileDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *offsets_shape = context->GetInputShape(1);
    const gert::Shape *weight_shape = context->GetInputShape(2);
    uint32_t numOffsets = static_cast<uint32_t>(offsets_shape->GetDim(0));
    uint32_t totalBags = numOffsets - 1;
    uint32_t embeddingDim = static_cast<uint32_t>(weight_shape->GetDim(1));
    gert::Shape *output_shape = context->GetOutputShape(0);
    output_shape->SetDimNum(2);
    output_shape->SetDim(0, static_cast<int64_t>(totalBags));
    output_shape->SetDim(1, static_cast<int64_t>(embeddingDim));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
}  // namespace ge

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00266Custom {
public:
    __aicore__ inline KernelCa6k00266Custom() {}

    __aicore__ inline void Init(GM_ADDR indices, GM_ADDR offsets, GM_ADDR weight,
                                GM_ADDR output,
                                uint32_t totalBags, uint32_t embeddingDim,
                                uint32_t numEmbeddings, uint32_t tileDim) {
        totalBags_ = totalBags;
        embeddingDim_ = embeddingDim;
        numEmbeddings_ = numEmbeddings;
        tileDim_ = tileDim;

        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices, totalBags_ * (embeddingDim_ ? 1 : 0) + 1024); // rough max
        offsetsGm_.SetGlobalBuffer((__gm__ int64_t*)offsets, totalBags_ + 1);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numEmbeddings_ * embeddingDim_);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalBags_ * embeddingDim_);

        pipe_.InitBuffer(inQueueWeight_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(outQueueOut_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(accBuf_, 1, tileDim_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalBags_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // grid-stride over bags
        for (uint32_t bagIdx = blockIdx; bagIdx < totalBags_; bagIdx += blockNum) {
            ProcessBag(bagIdx);
        }
    }

private:
    __aicore__ inline void ProcessBag(uint32_t bagIdx) {
        // load start and end from offsets (scalar get)
        int64_t start = offsetsGm_.GetValue(bagIdx);
        int64_t end = offsetsGm_.GetValue(bagIdx + 1);
        uint32_t bagLength = static_cast<uint32_t>(end - start);
        if (bagLength == 0) {
            // fill with zeros
            for (uint32_t d = 0; d < embeddingDim_; d += tileDim_) {
                uint32_t curTile = (embeddingDim_ - d < tileDim_) ? (embeddingDim_ - d) : tileDim_;
                AscendC::LocalTensor<float> outLocal = outQueueOut_.AllocTensor<float>();
                AscendC::Duplicate<float>(outLocal, 0.0f, curTile);
                uint32_t outOffset = bagIdx * embeddingDim_ + d;
                AscendC::DataCopy(outputGm_[outOffset], outLocal, curTile);
                outQueueOut_.FreeTensor(outLocal);
            }
            return;
        }

        // accumulate per tile
        for (uint32_t d = 0; d < embeddingDim_; d += tileDim_) {
            uint32_t curTile = (embeddingDim_ - d < tileDim_) ? (embeddingDim_ - d) : tileDim_;
            AscendC::LocalTensor<float> acc = accBuf_.AllocTensor<float>();
            AscendC::Duplicate<float>(acc, 0.0f, curTile);

            for (uint32_t i = 0; i < bagLength; i++) {
                int64_t idx = indicesGm_.GetValue(static_cast<uint32_t>(start + i));
                uint32_t rowOffset = static_cast<uint32_t>(idx) * embeddingDim_ + d;
                AscendC::LocalTensor<float> wLocal = inQueueWeight_.AllocTensor<float>();
                AscendC::DataCopy(wLocal, weightGm_[rowOffset], curTile);
                AscendC::Add(acc, acc, wLocal, curTile);
                inQueueWeight_.FreeTensor(wLocal);
            }

            // write accumulated tile to output
            AscendC::LocalTensor<float> outLocal = outQueueOut_.AllocTensor<float>();
            AscendC::Copy(outLocal, acc, curTile);
            uint32_t outOffset = bagIdx * embeddingDim_ + d;
            AscendC::DataCopy(outputGm_[outOffset], outLocal, curTile);
            outQueueOut_.FreeTensor(outLocal);
            accBuf_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWeight_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accBuf_;

    AscendC::GlobalTensor<int64_t> indicesGm_;
    AscendC::GlobalTensor<int64_t> offsetsGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t totalBags_ {0};
    uint32_t embeddingDim_ {0};
    uint32_t numEmbeddings_ {0};
    uint32_t tileDim_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00266_custom(GM_ADDR indices, GM_ADDR offsets, GM_ADDR weight,
                                                         GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00266Custom op;
    op.Init(indices, offsets, weight, output,
            tiling_data.totalBags, tiling_data.embeddingDim,
            tiling_data.numEmbeddings, tiling_data.tileDim);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor ca6k_00266_impl_npu(
    const at::Tensor& indices,
    const at::Tensor& offsets,
    const at::Tensor& weight) {
    int64_t numBags = offsets.size(0) - 1;
    int64_t embeddingDim = weight.size(1);
    at::Tensor result = at::empty({numBags, embeddingDim}, weight.options());
    EXEC_NPU_CMD(aclnnCa6k00266Custom, indices, offsets, weight, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_00266_custom", &ca6k_00266_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("ca6k_00266_custom", &ca6k_00266_impl_npu,
          "Custom EmbeddingBag (sum) on NPU");
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00266_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_embeddings, embedding_dim, mode="sum"):
        super(ModelNew, self).__init__()
        self.num_embeddings = num_embeddings
        self.embedding_dim = embedding_dim
        # Weight parameter, default initialization same as nn.EmbeddingBag
        self.weight = torch.nn.Parameter(
            torch.randn(num_embeddings, embedding_dim, dtype=torch.float32)
        )
        nn.init.normal_(self.weight, mean=0.0, std=1.0)
        self.mode = mode

    def forward(self, input, offsets):
        fn = getattr(custom_ops_lib, _FN)
        return fn(input, offsets, self.weight)
'''
