kernel_src = '''

#include "kernel_operator.h"

class KernelPixelShuffle {
public:
    __aicore__ inline KernelPixelShuffle() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C_in, uint32_t H_in, uint32_t W_in,
                                uint32_t scaleFactor,
                                uint32_t C_out, uint32_t H_out, uint32_t W_out)
    {
        N_ = N;
        C_in_ = C_in;
        H_in_ = H_in;
        W_in_ = W_in;
        scale_ = scaleFactor;
        C_out_ = C_out;
        H_out_ = H_out;
        W_out_ = W_out;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_in_ * H_in_ * W_in_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_out_ * H_out_ * W_out_);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= N_) return;

        uint32_t n = blockIdx;
        uint32_t inBatchOff = n * C_in_ * H_in_ * W_in_;
        uint32_t outBatchOff = n * C_out_ * H_out_ * W_out_;

        AscendC::GlobalTensor<float> xBatch = xGm_.Sub(inBatchOff);
        AscendC::GlobalTensor<float> yBatch = yGm_.Sub(outBatchOff);

        for (uint32_t c_out = 0; c_out < C_out_; ++c_out) {
            for (uint32_t h_out = 0; h_out < H_out_; ++h_out) {
                for (uint32_t w_out = 0; w_out < W_out_; ++w_out) {
                    uint32_t c_in = c_out * scale_ * scale_ + (h_out % scale_) * scale_ + (w_out % scale_);
                    uint32_t h_in = h_out / scale_;
                    uint32_t w_in = w_out / scale_;

                    uint32_t inOffset = c_in * H_in_ * W_in_ + h_in * W_in_ + w_in;
                    uint32_t outOffset = c_out * H_out_ * W_out_ + h_out * W_out_ + w_out;

                    float val = xBatch(inOffset);
                    yBatch(outOffset) = val;
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_;
    uint32_t C_in_;
    uint32_t H_in_;
    uint32_t W_in_;
    uint32_t scale_;
    uint32_t C_out_;
    uint32_t H_out_;
    uint32_t W_out_;
};

extern "C" __global__ __aicore__ void ca6k_00303_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelPixelShuffle op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C_in, tiling_data.H_in, tiling_data.W_in,
            tiling_data.scaleFactor,
            tiling_data.C_out, tiling_data.H_out, tiling_data.W_out);
    op.Process();
}
'''

python_bind_src = '''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"
#include "aclnn_ca6k00303_custom.h"

at::Tensor pixel_shuffle_impl_npu(const at::Tensor& self, int64_t scale_factor) {
    // Compute output shape
    int64_t N = self.size(0);
    int64_t C_in = self.size(1);
    int64_t H_in = self.size(2);
    int64_t W_in = self.size(3);
    int64_t C_out = C_in / (scale_factor * scale_factor);
    int64_t H_out = H_in * scale_factor;
    int64_t W_out = W_in * scale_factor;
    std::vector<int64_t> out_shape = {N, C_out, H_out, W_out};
    at::Tensor result = at::empty(out_shape, self.options());
    EXEC_NPU_CMD(aclnnCa6k00303Custom, self, scale_factor, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_00303_custom", &pixel_shuffle_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("ca6k_00303_custom", &pixel_shuffle_impl_npu, "PixelShuffle operation (Ca6k00303Custom on NPU)");
}
