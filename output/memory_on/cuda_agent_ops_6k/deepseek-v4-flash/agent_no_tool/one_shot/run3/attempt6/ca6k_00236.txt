static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00236CustomTilingData tiling;

    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &originShape = storageShape->GetOriginShape();
    uint32_t batch = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t height = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t width = static_cast<uint32_t>(originShape.GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dimAttr;

    // Each column (across height) corresponds to one softmin operation
    uint32_t totalCols = batch * width;

    constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    // Buffers per column: input (height), output (height), exp (height), max scalar, sum scalar
    // Use 3 * height * sizeof(float) roughly, plus overhead. For height=512: 6144 bytes per column.
    constexpr uint32_t ALIGN = 8;
    uint32_t maxCols = UB_BUDGET_BYTES / (3 * height * sizeof(float));
    if (maxCols < 1) maxCols = 1;
    uint32_t tileCols = (maxCols / ALIGN) * ALIGN;
    if (tileCols == 0) tileCols = ALIGN;
    if (tileCols > totalCols) tileCols = totalCols;

    uint32_t numTiles = (totalCols + tileCols - 1) / tileCols;
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 32) blockDim = 32;

    context->SetBlockDim(blockDim);

    tiling.set_totalCols(totalCols);
    tiling.set_tileCols(tileCols);
    tiling.set_height(height);
    tiling.set_width(width);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00236Custom {
public:
    __aicore__ inline KernelCa6k00236Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalCols, uint32_t tileCols,
                                uint32_t height, uint32_t width, int32_t dim)
    {
        totalCols_ = totalCols;
        tileCols_ = tileCols;
        height_ = height;
        width_ = width;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Buffers for one column of size height_
        pipe_.InitBuffer(inQueue_, BUFFER_NUM, height_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, height_ * sizeof(float));
        pipe_.InitBuffer(expQueue_, 1, height_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalCols_ == 0 || tileCols_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t stride = blockNum * tileCols_;
        uint32_t colStart = blockIdx * tileCols_;
        while (colStart < totalCols_) {
            uint32_t curTileCols = totalCols_ - colStart;
            if (curTileCols > tileCols_) curTileCols = tileCols_;

            for (uint32_t c = 0; c < curTileCols; ++c) {
                uint32_t colIdx = colStart + c;
                ProcessOneColumn(colIdx);
            }

            colStart += stride;
        }
    }

private:
    __aicore__ inline void ProcessOneColumn(uint32_t colIdx)
    {
        // Decompose colIdx into batch and column within that image
        uint32_t batchIdx = colIdx / width_;
        uint32_t localCol = colIdx % width_;

        // Load column vector (height_ elements) with stride width_ from global
        AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
        for (uint32_t h = 0; h < height_; ++h) {
            uint32_t gmOffset = batchIdx * height_ * width_ + h * width_ + localCol;
            // Copy single element into local tensor position h
            AscendC::DataCopy(xLocal[h], xGm_[gmOffset], 1);
        }
        inQueue_.EnQue(xLocal);

        // Compute softmin on this vector xLocal
        AscendC::LocalTensor<float> x = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> y = outQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> expBuf = expQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> scalarBuf = scalarQueue_.AllocTensor<float>();

        // y = -x
        AscendC::Muls(y, x, -1.0f, height_);

        // max(y)
        AscendC::ReduceMax(scalarBuf, y, height_);  // scalar
        // y = y - max
        AscendC::Sub(y, y, scalarBuf, height_);

        // y = exp(y)
        AscendC::Exp(y, y, height_);

        // sum = sum(y)
        AscendC::ReduceSum(scalarBuf, y, height_);  // scalar
        // y = y / sum
        AscendC::Div(y, y, scalarBuf, height_);

        // Write result back to global memory (same layout)
        for (uint32_t h = 0; h < height_; ++h) {
            uint32_t gmOffset = batchIdx * height_ * width_ + h * width_ + localCol;
            AscendC::DataCopy(yGm_[gmOffset], y[h], 1);
        }

        outQueue_.EnQue(y);
        expQueue_.FreeTensor(expBuf);
        scalarQueue_.FreeTensor(scalarBuf);
        inQueue_.FreeTensor(x);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> expQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalCols_ {0};
    uint32_t tileCols_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00236_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00236Custom op;
    op.Init(x, y,
            tiling_data.totalCols,
            tiling_data.tileCols,
            tiling_data.height,
            tiling_data.width,
            tiling_data.dim);
    op.Process();
}
