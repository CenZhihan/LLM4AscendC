namespace optiling {

static constexpr uint32_t ALIGN = 64;

static uint32_t AlignDown(uint32_t val, uint32_t align)
{
    return (val / align) * align;
}

static uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00501CustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t totalBatch = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t totalChannels = static_cast<uint32_t>(xShape->GetDim(1));
    uint32_t H = static_cast<uint32_t>(xShape->GetDim(2));
    uint32_t W = static_cast<uint32_t>(xShape->GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(2);

    uint32_t kernelH = static_cast<uint32_t>(kernel_size[0]);
    uint32_t kernelW = static_cast<uint32_t>(kernel_size[1]);
    uint32_t strideH = static_cast<uint32_t>(stride[0]);
    uint32_t strideW = static_cast<uint32_t>(stride[1]);
    uint32_t padH = static_cast<uint32_t>(padding[0]);
    uint32_t padW = static_cast<uint32_t>(padding[1]);

    uint32_t Hout = (H + 2 * padH - kernelH) / strideH + 1;
    uint32_t Wout = (W + 2 * padW - kernelW) / strideW + 1;

    uint32_t tileHout = 64;
    uint32_t tileWout = 64;
    if (tileHout > Hout) tileHout = Hout;
    if (tileWout > Wout) tileWout = Wout;
    tileHout = AlignDown(tileHout, ALIGN);
    tileWout = AlignDown(tileWout, ALIGN);
    if (tileHout == 0) tileHout = Hout;
    if (tileWout == 0) tileWout = Wout;

    uint32_t totalSlices = totalBatch * totalChannels;
    uint32_t blockDim = totalSlices;
    if (blockDim > 64) blockDim = 64;
    if (blockDim < 8) blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_totalBatch(totalBatch);
    tiling.set_totalChannels(totalChannels);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Hout(Hout);
    tiling.set_Wout(Wout);
    tiling.set_kernelH(kernelH);
    tiling.set_kernelW(kernelW);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_tileHout(tileHout);
    tiling.set_tileWout(tileWout);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00501Custom {
public:
    __aicore__ inline KernelCa6k00501Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalBatch, uint32_t totalChannels,
                                uint32_t H, uint32_t W,
                                uint32_t Hout, uint32_t Wout,
                                uint32_t kernelH, uint32_t kernelW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t tileHout, uint32_t tileWout)
    {
        totalBatch_ = totalBatch;
        totalChannels_ = totalChannels;
        H_ = H;
        W_ = W;
        Hout_ = Hout;
        Wout_ = Wout;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        tileHout_ = tileHout;
        tileWout_ = tileWout;

        inputRowsPerTile_ = (tileHout - 1) * strideH + kernelH;
        inputColsPerTile_ = (tileWout - 1) * strideW + kernelW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        uint32_t inputBufSize = inputRowsPerTile_ * inputColsPerTile_;
        uint32_t outputBufSize = tileHout_ * tileWout_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputBufSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outputBufSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalBatch_ == 0 || totalChannels_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t sliceIdx = blockIdx;
        while (sliceIdx < totalBatch_ * totalChannels_) {
            uint32_t b = sliceIdx / totalChannels_;
            uint32_t c = sliceIdx % totalChannels_;

            for (uint32_t to = 0; to < Hout_; to += tileHout_) {
                for (uint32_t ti = 0; ti < Wout_; ti += tileWout_) {
                    uint32_t tileH = (to + tileHout_ > Hout_) ? Hout_ - to : tileHout_;
                    uint32_t tileW = (ti + tileWout_ > Wout_) ? Wout_ - ti : tileWout_;

                    int32_t inRowStart = static_cast<int32_t>(to * strideH_ - padH_);
                    int32_t inColStart = static_cast<int32_t>(ti * strideW_ - padW_);
                    uint32_t inRows = (tileH - 1) * strideH_ + kernelH_;
                    uint32_t inCols = (tileW - 1) * strideW_ + kernelW_;

                    AscendC::LocalTensor<float> inTile = inQueueX_.AllocTensor<float>();
                    AscendC::LocalTensor<float> outTile = outQueueY_.AllocTensor<float>();

                    // Fill input tile with -inf
                    AscendC::Fill<float>(inTile, -1.0f/0.0f, inRows * inCols);

                    // Copy valid input region into local tile
                    for (uint32_t r = 0; r < inRows; ++r) {
                        int32_t gRow = inRowStart + static_cast<int32_t>(r);
                        if (gRow < 0 || gRow >= static_cast<int32_t>(H_)) continue;
                        for (uint32_t c2 = 0; c2 < inCols; ++c2) {
                            int32_t gCol = inColStart + static_cast<int32_t>(c2);
                            if (gCol < 0 || gCol >= static_cast<int32_t>(W_)) continue;
                            uint32_t srcIdx = b * totalChannels_ * H_ * W_ + c * H_ * W_ + gRow * W_ + gCol;
                            uint32_t dstIdx = r * inCols + c2;
                            inTile.SetValue(dstIdx, xGm_.GetValue(srcIdx));
                        }
                    }

                    // Compute max pooling on local tile
                    for (uint32_t rOut = 0; rOut < tileH; ++rOut) {
                        for (uint32_t cOut = 0; cOut < tileW; ++cOut) {
                            float maxVal = -1.0f/0.0f;
                            for (uint32_t kr = 0; kr < kernelH_; ++kr) {
                                for (uint32_t kc = 0; kc < kernelW_; ++kc) {
                                    uint32_t inR = rOut * strideH_ + kr;
                                    uint32_t inC = cOut * strideW_ + kc;
                                    float v = inTile.GetValue(inR * inCols + inC);
                                    if (v > maxVal) maxVal = v;
                                }
                            }
                            outTile.SetValue(rOut * tileWout_ + cOut, maxVal);
                        }
                    }

                    // Copy output tile to global
                    for (uint32_t rOut = 0; rOut < tileH; ++rOut) {
                        for (uint32_t cOut = 0; cOut < tileW; ++cOut) {
                            uint32_t dstIdx = b * totalChannels_ * Hout_ * Wout_ + c * Hout_ * Wout_ +
                                              (to + rOut) * Wout_ + (ti + cOut);
                            uint32_t locIdx = rOut * tileWout_ + cOut;
                            yGm_.SetValue(dstIdx, outTile.GetValue(locIdx));
                        }
                    }

                    outQueueY_.FreeTensor(outTile);
                    inQueueX_.FreeTensor(inTile);
                }
            }

            sliceIdx += blockNum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalBatch_;
    uint32_t totalChannels_;
    uint32_t H_, W_;
    uint32_t Hout_, Wout_;
    uint32_t kernelH_, kernelW_;
    uint32_t strideH_, strideW_;
    uint32_t padH_, padW_;
    uint32_t tileHout_, tileWout_;
    uint32_t inputRowsPerTile_;
    uint32_t inputColsPerTile_;
};

extern "C" __global__ __aicore__ void ca6k_00501_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00501Custom op;
    op.Init(x, y,
            tiling_data.totalBatch, tiling_data.totalChannels,
            tiling_data.H, tiling_data.W,
            tiling_data.Hout, tiling_data.Wout,
            tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.tileHout, tiling_data.tileWout);
    op.Process();
}
'''

at::Tensor ca6k_00501_custom_npu(const at::Tensor& self,
                                 at::IntArrayRef kernel_size,
                                 at::IntArrayRef stride,
                                 at::IntArrayRef padding) {
    auto outSize = at::IntArrayRef({self.size(0), self.size(1),
                                    (self.size(2) + 2 * padding[0] - kernel_size[0]) / stride[0] + 1,
                                    (self.size(3) + 2 * padding[1] - kernel_size[1]) / stride[1] + 1});

    EXEC_NPU_CMD(aclnnCa6k00501Custom, self, kernel_size, stride, padding, result);
    return result;
