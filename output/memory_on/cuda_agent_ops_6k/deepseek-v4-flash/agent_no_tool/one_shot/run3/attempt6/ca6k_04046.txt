namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k04046CustomTilingData tiling;
  const gert::StorageShape *storageShape = context->GetInputShape(0);
  const auto& inShape = storageShape->GetOriginShape();
  uint32_t batch = inShape.GetDim(0);
  uint32_t channels = inShape.GetDim(1);
  uint32_t depth = inShape.GetDim(2);
  uint32_t height = inShape.GetDim(3);
  uint32_t width = inShape.GetDim(4);

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int32_t pad_front = attrs->GetAttrPointer<int32_t>(0)[0];
  int32_t pad_back = attrs->GetAttrPointer<int32_t>(1)[0];
  int32_t pad_top = attrs->GetAttrPointer<int32_t>(2)[0];
  int32_t pad_bottom = attrs->GetAttrPointer<int32_t>(3)[0];
  int32_t pad_left = attrs->GetAttrPointer<int32_t>(4)[0];
  int32_t pad_right = attrs->GetAttrPointer<int32_t>(5)[0];

  uint32_t dPad = depth + (uint32_t)pad_front + (uint32_t)pad_back;
  uint32_t hPad = height + (uint32_t)pad_top + (uint32_t)pad_bottom;
  uint32_t wPad = width + (uint32_t)pad_left + (uint32_t)pad_right;

  uint64_t totalOutput = (uint64_t)batch * channels * dPad * hPad * wPad;
  if (totalOutput >= 0xFFFFFFFFull) totalOutput = 0xFFFFFFFFul;
  uint32_t totalOutputLength = (uint32_t)totalOutput;

  uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = maxTile;
  if (totalOutputLength < tileLen) {
    tileLen = AlignDown(totalOutputLength, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = totalOutputLength;
  }

  uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutputLength, tileLen);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_batch(batch);
  tiling.set_channels(channels);
  tiling.set_depth(depth);
  tiling.set_height(height);
  tiling.set_width(width);
  tiling.set_pad_front(pad_front);
  tiling.set_pad_back(pad_back);
  tiling.set_pad_top(pad_top);
  tiling.set_pad_bottom(pad_bottom);
  tiling.set_pad_left(pad_left);
  tiling.set_pad_right(pad_right);
  tiling.set_totalOutputLength(totalOutputLength);
  tiling.set_tileLength(tileLen);
  tiling.set_blockDim(blockDim);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *workspace = context->GetWorkspaceSizes(1);
  workspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}
} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelReflectionPad3dCustom {
public:
    __aicore__ inline KernelReflectionPad3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t channels,
                                uint32_t depth, uint32_t height, uint32_t width,
                                int32_t pad_front, int32_t pad_back,
                                int32_t pad_top, int32_t pad_bottom,
                                int32_t pad_left, int32_t pad_right,
                                uint32_t totalOutputLength, uint32_t tileLength)
    {
        batch_ = batch;
        channels_ = channels;
        depth_ = depth;
        height_ = height;
        width_ = width;
        padFront_ = pad_front;
        padBack_ = pad_back;
        padTop_ = pad_top;
        padBottom_ = pad_bottom;
        padLeft_ = pad_left;
        padRight_ = pad_right;

        dPad_ = depth_ + (uint32_t)padFront_ + (uint32_t)padBack_;
        hPad_ = height_ + (uint32_t)padTop_ + (uint32_t)padBottom_;
        wPad_ = width_ + (uint32_t)padLeft_ + (uint32_t)padRight_;

        totalLen_ = totalOutputLength;
        tileLen_ = tileLength;

        // Strides for output tensor (padded)
        strideN_ = channels_ * dPad_ * hPad_ * wPad_;
        strideC_ = dPad_ * hPad_ * wPad_;
        strideD_ = hPad_ * wPad_;
        strideH_ = wPad_;
        strideW_ = 1;

        // Strides for input tensor (original)
        inStrideN_ = channels_ * depth_ * height_ * width_;
        inStrideC_ = depth_ * height_ * width_;
        inStrideD_ = height_ * width_;
        inStrideH_ = width_;
        inStrideW_ = 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, inStrideN_ * batch_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Process first tile
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        ComputeAndWrite(base, curLen);

        while (true) {
            uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                ComputeAndWrite(nextBase, nextLen);
            }

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline int32_t Reflect(int32_t idx, int32_t pad, uint32_t dim) const
    {
        int32_t delta = idx - pad;
        if (delta < 0) {
            return -delta;
        } else if (delta >= (int32_t)dim) {
            return (int32_t)(2 * dim) + pad - idx - 2;
        } else {
            return delta;
        }
    }

    __aicore__ inline void ComputeAndWrite(uint32_t startOutLinear, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; i++) {
            uint32_t outLinear = startOutLinear + i;
            // Decompose output linear index to (n,c,d,h,w) on padded tensor
            uint32_t n = outLinear / strideN_;
            uint32_t rem = outLinear % strideN_;
            uint32_t c = rem / strideC_;
            rem %= strideC_;
            uint32_t d = rem / strideD_;
            rem %= strideD_;
            uint32_t h = rem / strideH_;
            uint32_t w = rem % strideH_;

            // Reflect each spatial dimension
            int32_t d_src = Reflect((int32_t)d, padFront_, depth_);
            int32_t h_src = Reflect((int32_t)h, padTop_, height_);
            int32_t w_src = Reflect((int32_t)w, padLeft_, width_);

            // Compute input linear index
            uint32_t inLinear = n * inStrideN_ +
                                c * inStrideC_ +
                                (uint32_t)d_src * inStrideD_ +
                                (uint32_t)h_src * inStrideH_ +
                                (uint32_t)w_src * inStrideW_;

            yLocal(i) = xGm_.GetValue(inLinear);
        }

        // Write tile to global output
        AscendC::DataCopy(yGm_[startOutLinear], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t channels_;
    uint32_t depth_;
    uint32_t height_;
    uint32_t width_;

    int32_t padFront_;
    int32_t padBack_;
    int32_t padTop_;
    int32_t padBottom_;
    int32_t padLeft_;
    int32_t padRight_;

    uint32_t dPad_;
    uint32_t hPad_;
    uint32_t wPad_;

    uint32_t totalLen_;
    uint32_t tileLen_;

    // Output strides
    uint32_t strideN_;
    uint32_t strideC_;
    uint32_t strideD_;
    uint32_t strideH_;
    uint32_t strideW_;

    // Input strides
    uint32_t inStrideN_;
    uint32_t inStrideC_;
    uint32_t inStrideD_;
    uint32_t inStrideH_;
    uint32_t inStrideW_;
};

extern "C" __global__ __aicore__ void ca6k_04046_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelReflectionPad3dCustom op;
    op.Init(x, y,
            tiling_data.batch, tiling_data.channels,
            tiling_data.depth, tiling_data.height, tiling_data.width,
            tiling_data.pad_front, tiling_data.pad_back,
            tiling_data.pad_top, tiling_data.pad_bottom,
            tiling_data.pad_left, tiling_data.pad_right,
            tiling_data.totalOutputLength, tiling_data.tileLength);
    op.Process();
}
'''
