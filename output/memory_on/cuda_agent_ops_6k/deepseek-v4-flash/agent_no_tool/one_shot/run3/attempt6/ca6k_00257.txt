kernel_src="""
#include "kernel_operator.h"

class KernelCa6k00257Custom {
public:
    __aicore__ inline KernelCa6k00257Custom() {}

    __aicore__ inline void Init(GM_ADDR theta, GM_ADDR grid,
                                uint32_t totalElems, uint32_t height, uint32_t width,
                                float scale, int32_t align_corners)
    {
        totalElems_ = totalElems;
        height_ = height;
        width_ = width;
        scale_ = scale;
        alignCorners_ = align_corners;

        // theta shape: batch x 6, compute batch from totalElems
        batch_ = totalElems / (height_ * width_);
        thetaGm_.SetGlobalBuffer((__gm__ float*)theta, batch_ * 6);
        gridGm_.SetGlobalBuffer((__gm__ float*)grid, totalElems_ * 2);
    }

    __aicore__ inline void Process()
    {
        if (totalElems_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t idx = blockIdx; idx < totalElems_; idx += blockNum) {
            uint32_t b = idx / (height_ * width_);
            uint32_t rem = idx % (height_ * width_);
            uint32_t h = rem / width_;
            uint32_t w = rem % width_;

            float theta00 = thetaGm_[b * 6 + 0];
            float theta01 = thetaGm_[b * 6 + 1];
            float theta02 = thetaGm_[b * 6 + 2];
            float theta10 = thetaGm_[b * 6 + 3];
            float theta11 = thetaGm_[b * 6 + 4];
            float theta12 = thetaGm_[b * 6 + 5];

            float x_out = static_cast<float>(w);
            float y_out = static_cast<float>(h);

            float x_norm, y_norm;
            if (alignCorners_) {
                if (width_ > 1) {
                    x_norm = x_out / static_cast<float>(width_ - 1) * 2.0f - 1.0f;
                } else {
                    x_norm = 0.0f;
                }
                if (height_ > 1) {
                    y_norm = y_out / static_cast<float>(height_ - 1) * 2.0f - 1.0f;
                } else {
                    y_norm = 0.0f;
                }
            } else {
                x_norm = (x_out + 0.5f) / static_cast<float>(width_) * 2.0f - 1.0f;
                y_norm = (y_out + 0.5f) / static_cast<float>(height_) * 2.0f - 1.0f;
            }

            float x_src = theta00 * x_norm + theta01 * y_norm + theta02;
            float y_src = theta10 * x_norm + theta11 * y_norm + theta12;

            x_src *= scale_;
            y_src *= scale_;

            gridGm_[idx * 2 + 0] = x_src;
            gridGm_[idx * 2 + 1] = y_src;
        }
    }

private:
    AscendC::GlobalTensor<float> thetaGm_;
    AscendC::GlobalTensor<float> gridGm_;

    uint32_t totalElems_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    uint32_t batch_ {0};
    float scale_ {0.5f};
    int32_t alignCorners_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00257_custom(GM_ADDR theta, GM_ADDR grid, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00257Custom op;
    op.Init(theta, grid, tiling_data.totalElems, tiling_data.height, tiling_data.width,
            tiling_data.scale, tiling_data.align_corners);
    op.Process();
}

import torch

_FN = 'ca6k_00257_custom'

class ModelNew(torch.nn.Module):

        super(ModelNew, self).__init__()
        self.batch = size[0]
        self.height = size[2]
        self.width = size[3]
        self.scale = scale
        self.align_corners = align_corners

    def forward(self, theta):

        return fn(theta, self.batch, self.height, self.width, self.scale, self.align_corners)
