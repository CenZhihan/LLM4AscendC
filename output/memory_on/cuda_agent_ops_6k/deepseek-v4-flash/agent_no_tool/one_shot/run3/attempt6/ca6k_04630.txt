static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFuncGLU(gert::TilingContext *context) {
    Ca6k04630GLUTilingData tiling;
    uint32_t totalLength = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    uint32_t C = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));
    uint32_t halfC = C / 2;
    tiling.set_halfFeatures(halfC);

    uint32_t maxTile = UB_BUDGET_BYTES / (halfC * 4 * 3); // in, out, tmp
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = aligned > 0 ? aligned : totalLength;
    }

    uint32_t numTiles = tileLen == 0 ? 1 : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

kernel_src = '''
#include "kernel_operator.h"
#include "ca6k_04630_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGLU {
public:
    __aicore__ inline KernelGLU() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileLength, uint32_t halfFeatures) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        halfFeat_ = halfFeatures;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, halfFeat_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t strideElems = blockNum * tileLen_ * (halfFeat_ * 2); // total elements per tile (full channel)
        uint32_t base = blockIdx * tileLen_ * (halfFeat_ * 2);
        if (base >= totalLen_) return;

        AscendC::LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>();

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_ * halfFeat_ * 2) curLen = tileLen_ * halfFeat_ * 2;
            curLen = (curLen / (halfFeat_ * 2)) * (halfFeat_ * 2);
            if (curLen == 0) break;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // Process each vector of length 2*halfFeat
            uint32_t numVecs = curLen / (halfFeat_ * 2);
            for (uint32_t i = 0; i < numVecs; ++i) {
                AscendC::LocalTensor<float> vec = xLocal + i * halfFeat_ * 2;
                AscendC::LocalTensor<float> a = vec;
                AscendC::LocalTensor<float> b = vec + halfFeat_;
                // sigmoid: 1 / (1 + exp(-a))
                AscendC::Muls(tmp, a, -1.0f, halfFeat_);
                AscendC::Exp(tmp, tmp, halfFeat_);
                AscendC::Adds(tmp, tmp, 1.0f, halfFeat_);
                AscendC::Reciprocal(tmp, tmp, halfFeat_);
                // multiply with b
                AscendC::Mul(tmp, tmp, b, halfFeat_);
                // write to y
                AscendC::DataCopy(yLocal + i * halfFeat_, tmp, halfFeat_);
            }
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);

            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base / 2], yOut, curLen / 2);
            outQueueY_.FreeTensor(yOut);

            base += strideElems;
        }
        tmpBuf_.FreeTensor(tmp);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t halfFeat_ {0};
};

class KernelLinear {
public:
    __aicore__ inline KernelLinear() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength, uint32_t halfFeatures) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        halfFeat_ = halfFeatures;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, halfFeat_ * halfFeat_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, halfFeat_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(wBuf_, 1, halfFeat_ * halfFeat_ * sizeof(float));
        pipe_.InitBuffer(bBuf_, 1, halfFeat_ * sizeof(float));
        pipe_.InitBuffer(tmpDot_, 1, halfFeat_ * sizeof(float));
        pipe_.InitBuffer(tmpRow_, 1, halfFeat_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t strideElems = blockNum * tileLen_ * halfFeat_;
        uint32_t base = blockIdx * tileLen_ * halfFeat_;
        if (base >= totalLen_) return;

        // Load weight and bias to local
        AscendC::LocalTensor<float> wLocal = wBuf_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, halfFeat_ * halfFeat_);
        AscendC::LocalTensor<float> bLocal = bBuf_.AllocTensor<float>();
        AscendC::DataCopy(bLocal, bGm_, halfFeat_);

        AscendC::LocalTensor<float> dot = tmpDot_.AllocTensor<float>();
        AscendC::LocalTensor<float> row = tmpRow_.AllocTensor<float>();

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_ * halfFeat_) curLen = tileLen_ * halfFeat_;
            curLen = (curLen / halfFeat_) * halfFeat_;
            if (curLen == 0) break;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            uint32_t numRows = curLen / halfFeat_;
            for (uint32_t r = 0; r < numRows; ++r) {
                AscendC::LocalTensor<float> xrow = xLocal + r * halfFeat_;
                // For each output feature j
                for (uint32_t j = 0; j < halfFeat_; ++j) {
                    AscendC::LocalTensor<float> wrow = wLocal + j * halfFeat_;
                    AscendC::Mul(row, xrow, wrow, halfFeat_);
                    AscendC::ReduceAdd(dot, row, halfFeat_);
                    AscendC::Adds(dot, dot, bLocal(j), 1);
                    // store scalar to y row
                    yLocal(r * halfFeat_ + j) = dot(0);
                }
            }
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);

            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yOut, curLen);
            outQueueY_.FreeTensor(yOut);

            base += strideElems;
        }

        wBuf_.FreeTensor(wLocal);
        bBuf_.FreeTensor(bLocal);
        tmpDot_.FreeTensor(dot);
        tmpRow_.FreeTensor(row);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDot_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpRow_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t halfFeat_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04630_glu(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGLU op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.halfFeatures);
    op.Process();
}

extern "C" __global__ __aicore__ void ca6k_04630_linear(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLinear op;
    op.Init(x, weight, bias, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.halfFeatures);
    op.Process();
}

model_src = '''
import torch
import torch_npu
import llm4ascendc_ops_ca6k_04630

class ModelNew(torch.nn.Module):
    def __init__(self, in_features):
        super(ModelNew, self).__init__()
        self.in_features = in_features
        self.half_features = in_features // 2
        self.weight = torch.nn.Parameter(torch.empty(self.half_features, self.half_features))
        self.bias = torch.nn.Parameter(torch.empty(self.half_features))
        torch.nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        torch.nn.init.zeros_(self.bias)

    def forward(self, x):
        # GLU along dim=1
        x = llm4ascendc_ops_ca6k_04630.ca6k_04630_glu(x)
        # Linear
        b, c, h, w = x.shape
        x_flat = x.view(b * h * w, c)
        y_flat = llm4ascendc_ops_ca6k_04630.ca6k_04630_linear(x_flat, self.weight, self.bias)
        y = y_flat.view(b, c, h, w)
        return y
