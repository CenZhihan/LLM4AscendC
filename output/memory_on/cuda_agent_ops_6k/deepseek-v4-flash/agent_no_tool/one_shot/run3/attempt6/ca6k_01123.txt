kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class Ca6k01123CustomKernel {
public:
    __aicore__ inline Ca6k01123CustomKernel() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        batch_ = td.batch;
        channels_ = td.channels;
        inH_ = td.inH;
        inW_ = td.inW;
        outH_ = td.outH;
        outW_ = td.outW;
        kernelH_ = td.kernelH;
        kernelW_ = td.kernelW;
        strideH_ = td.strideH;
        strideW_ = td.strideW;
        padH_ = td.padH;
        padW_ = td.padW;
        ceilMode_ = td.ceilMode;
        countIncludePad_ = td.countIncludePad;
        oTileH_ = td.oTileH;
        oTileW_ = td.oTileW;
        blockDim_ = td.blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_*channels_*inH_*inW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch_*channels_*outH_*outW_);

        const uint32_t inTileH = (oTileH_ - 1) * strideH_ + kernelH_;
        const uint32_t inTileW = (oTileW_ - 1) * strideW_ + kernelW_;
        tileElems_ = inTileH * inTileW;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileElems_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileElems_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outH_ == 0 || outW_ == 0) return;

        const uint32_t totalPixels = batch_ * channels_ * outH_ * outW_;
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // grid-stride over flattened output pixels
        for (uint32_t idx = blockIdx; idx < totalPixels; idx += blockNum) {
            uint32_t tmp = idx;
            const uint32_t col = tmp % outW_;
            tmp /= outW_;
            const uint32_t row = tmp % outH_;
            tmp /= outH_;
            const uint32_t c = tmp % channels_;
            tmp /= channels_;
            const uint32_t b = tmp;

            const int32_t yInStart = static_cast<int32_t>(row * strideH_) - static_cast<int32_t>(padH_);
            const int32_t xInStart = static_cast<int32_t>(col * strideW_) - static_cast<int32_t>(padW_);

            float sum = 0.0f;
            uint32_t validCount = 0;

            for (uint32_t ky = 0; ky < kernelH_; ++ky) {
                int32_t iy = yInStart + static_cast<int32_t>(ky);
                if (iy < 0 || iy >= static_cast<int32_t>(inH_)) {
                    if (countIncludePad_) validCount++;
                    continue;
                }
                for (uint32_t kx = 0; kx < kernelW_; ++kx) {
                    int32_t ix = xInStart + static_cast<int32_t>(kx);
                    if (ix < 0 || ix >= static_cast<int32_t>(inW_)) {
                        if (countIncludePad_) validCount++;
                        continue;
                    }
                    uint32_t gmIdx = ((b * channels_ + c) * inH_ + iy) * inW_ + ix;
                    sum += xGm_[gmIdx];
                    validCount++;
                }
            }

            uint32_t denominator = countIncludePad_ ? (kernelH_ * kernelW_) : validCount;
            if (denominator == 0) denominator = 1;

            uint32_t outIdx = ((b * channels_ + c) * outH_ + row) * outW_ + col;
            yGm_[outIdx] = sum / static_cast<float>(denominator);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t channels_;
    uint32_t inH_;
    uint32_t inW_;
    uint32_t outH_;
    uint32_t outW_;
    uint32_t kernelH_;
    uint32_t kernelW_;
    uint32_t strideH_;
    uint32_t strideW_;
    uint32_t padH_;
    uint32_t padW_;
    uint32_t ceilMode_;
    uint32_t countIncludePad_;
    uint32_t oTileH_;
    uint32_t oTileW_;
    uint32_t blockDim_;
    uint32_t tileElems_;
};

extern "C" __global__ __aicore__ void ca6k_01123_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    Ca6k01123CustomKernel op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_01123_custom'

class ModelNew(torch.nn.Module):
    def __init__(
        self,
        kernel_size: tuple,
        stride: int = 1,
        padding: tuple = (0, 0),
        ceil_mode: bool = False,
        count_include_pad: bool = True,
    ):
        super().__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.ceil_mode = ceil_mode
        self.count_include_pad = count_include_pad

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        # Ensure stride is a tuple if it is an int
        stride = self.stride
        if isinstance(stride, int):
            stride = (stride, stride)
        return fn(
            x,
            self.kernel_size,
            stride,
            self.padding,
            self.ceil_mode,
            self.count_include_pad,
        )
'''
