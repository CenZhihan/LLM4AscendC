class KernelCa6k04593Custom {
public:

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t channels,
                                uint32_t depth, uint32_t height, uint32_t width,
                                float eps)
    {
        batch_ = batch;
        channels_ = channels;
        depth_ = depth;
        height_ = height;
        width_ = width;
        spatialSize_ = depth * height * width;
        totalRows_ = batch * channels;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * spatialSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * spatialSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, spatialSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, spatialSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, spatialSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || spatialSize_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            uint32_t offset = row * spatialSize_;

            // Copy in
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], spatialSize_);

            // Compute mean
            AscendC::LocalTensor<float> meanTensor = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(meanTensor, xLocal, spatialSize_); // meanTensor[0] = sum
            float mean = meanTensor(0) / static_cast<float>(spatialSize_);
            tmpQueue_.FreeTensor(meanTensor);

            // Subtract mean in-place: xLocal = xLocal - mean
            AscendC::LocalTensor<float> meanVec = tmpQueue_.AllocTensor<float>();
            AscendC::Duplicate(meanVec, mean, spatialSize_);
            AscendC::Sub(xLocal, xLocal, meanVec, spatialSize_);
            tmpQueue_.FreeTensor(meanVec);

            // Compute variance = sum(xLocal^2) / spatialSize
            AscendC::LocalTensor<float> sqLocal = tmpQueue_.AllocTensor<float>();
