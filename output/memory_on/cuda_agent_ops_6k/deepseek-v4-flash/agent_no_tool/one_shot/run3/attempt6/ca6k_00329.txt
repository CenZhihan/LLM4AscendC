static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const auto &origin_shape = x_shape->GetOriginShape();
    uint32_t dimNum = origin_shape.GetDimNum();
    uint32_t batch = 1, channels = 1, height = 1, width = 1;
    if (dimNum == 4) {
        batch = origin_shape.GetDim(0);
        channels = origin_shape.GetDim(1);
        height = origin_shape.GetDim(2);
        width = origin_shape.GetDim(3);
    } else if (dimNum == 3) {
        batch = origin_shape.GetDim(0);
        channels = origin_shape.GetDim(1);
        height = origin_shape.GetDim(2);
        width = 1;
    } else if (dimNum == 2) {
        batch = origin_shape.GetDim(0);
        channels = origin_shape.GetDim(1);
        height = 1;
        width = 1;
    }
    uint32_t totalLength = origin_shape.GetShapeSize();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps_ptr = attrs->GetAttrPointer<float>(0);

    uint32_t tileLen = maxTile;

kernel_src="""

#include "kernel_operator.h"
#include <math.h>

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00329Custom {
public:
    __aicore__ inline KernelCa6k00329Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR gamma, GM_ADDR beta,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batch, uint32_t channels,
                                uint32_t height, uint32_t width,
                                float eps)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batch_ = batch;
        channels_ = channels;
        height_ = height;
        width_ = width;
        spatialSize_ = batch * height * width;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, channels_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, channels_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || channels_ == 0) return;

        for (uint32_t c = 0; c < channels_; c++) {
            float sum = 0.0f;
            float sumSq = 0.0f;
            uint32_t spatialElems = spatialSize_;
            uint32_t baseOffset = c * spatialElems;

            // First pass: accumulate sum and sumSq
            uint32_t remaining = spatialElems;
            uint32_t offset = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[baseOffset + offset], curLen);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();
                for (uint32_t i = 0; i < curLen; i++) {
                    float val = xLocal(i);
                    sum += val;
                    sumSq += val * val;
                }
                inQueueX_.FreeTensor(xLocal);

                offset += curLen;
                remaining -= curLen;
            }

            // Compute mean, var, invStd
            float invSpatial = 1.0f / (float)spatialElems;
            float mean = sum * invSpatial;
            float var = (sumSq * invSpatial) - (mean * mean);
            float invStd = 1.0f / sqrtf(var + eps_);

            // Second pass: compute output
            float gammaVal = gammaGm_(c);
            float betaVal = betaGm_(c);
            remaining = spatialElems;
            offset = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[baseOffset + offset], curLen);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                for (uint32_t i = 0; i < curLen; i++) {
                    float normalized = (xLocal(i) - mean) * invStd;
                    yLocal(i) = normalized * gammaVal + betaVal;
                }
                outQueueY_.EnQue(yLocal);

                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[baseOffset + offset], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);

                offset += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t batch_ {1};
    uint32_t channels_ {1};
    uint32_t height_ {1};
    uint32_t width_ {1};
    uint32_t spatialSize_ {1};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_00329_custom(GM_ADDR x, GM_ADDR y, GM_ADDR gamma, GM_ADDR beta, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00329Custom op;
    op.Init(x, y, gamma, beta,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.batch, tiling_data.channels,
            tiling_data.height, tiling_data.width,
            tiling_data.eps);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00329_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_features):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.register_buffer('gamma', torch.ones(num_features))
        self.register_buffer('beta', torch.zeros(num_features))

    def forward(self, x):

        beta = self.beta.to(x.dtype).to(x.device)
        fn = getattr(custom_ops_lib, _FN)
