host_operator_src = """

#include "ca6k_04602_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04602CustomTilingData tiling;

    const gert::StorageShape *xShape = context->GetInputShape(0);
    const auto &xStorageShape = xShape->GetStorageShape();
    uint32_t totalN = static_cast<uint32_t>(xStorageShape.GetDim(0));
    uint32_t totalC = static_cast<uint32_t>(xStorageShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(xStorageShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(xStorageShape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = *attrs->GetAttrPointer<int32_t>(0);
    int32_t strideVal = *attrs->GetAttrPointer<int32_t>(1);
    int32_t paddingVal = *attrs->GetAttrPointer<int32_t>(2);

    // Estimate tileH from UB budget
    uint32_t outW = (W + 2 * static_cast<uint32_t>(paddingVal) - static_cast<uint32_t>(kernelSize)) / static_cast<uint32_t>(strideVal) + 1;
    uint32_t maxBytesPerTile = static_cast<uint32_t>(UB_BUDGET_BYTES * 0.9f);
    uint32_t estimatedTileH = maxBytesPerTile / (4 * (W + outW) + 8 * W);
    if (estimatedTileH > H) estimatedTileH = H;
    if (estimatedTileH > 64) estimatedTileH = 64;
    uint32_t tileH = (estimatedTileH < 1) ? 1 : estimatedTileH;

    uint32_t blockDim = totalN * totalC;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalN(totalN);
    tiling.set_totalC(totalC);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_tileH(tileH);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(strideVal);
    tiling.set_padding(paddingVal);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    gert::Shape *yShape = context->GetOutputShape(0);
    int32_t kernelSize = context->GetAttrs()->GetAttrPointer<int32_t>(0)[0];
    int32_t strideVal = context->GetAttrs()->GetAttrPointer<int32_t>(1)[0];
    int32_t paddingVal = context->GetAttrs()->GetAttrPointer<int32_t>(2)[0];
    int64_t N = xShape->GetDim(0);
    int64_t C = xShape->GetDim(1);
    int64_t H = xShape->GetDim(2);
    int64_t W = xShape->GetDim(3);
    int64_t outH = (H + 2 * paddingVal - kernelSize) / strideVal + 1;
    int64_t outW = (W + 2 * paddingVal - kernelSize) / strideVal + 1;
    yShape->SetDim(0, N);
    yShape->SetDim(1, C);
    yShape->SetDim(2, outH);
    yShape->SetDim(3, outW);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class Ca6k04602Custom : public OpDef {
public:
    explicit Ca6k04602Custom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("kernel_size").AttrType(OPTIONAL).Int(3);
        this->Attr("stride").AttrType(OPTIONAL).Int(1);
        this->Attr("padding").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(Ca6k04602Custom);
} // namespace ops
