kernel_src = """
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04600Custom {
public:
    __aicore__ inline KernelCa6k04600Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR running_mean, GM_ADDR running_var,
                                GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalLength, uint32_t tileLength, uint32_t blockDim,
                                float eps, uint32_t numFeatures, uint32_t spatialSize) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        eps_ = eps;
        numFeatures_ = numFeatures;
        spatialSize_ = spatialSize;
        numRows_ = totalLength / spatialSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        meanGm_.SetGlobalBuffer((__gm__ float*)running_mean, numFeatures_);
        varGm_.SetGlobalBuffer((__gm__ float*)running_var, numFeatures_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numFeatures_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, numFeatures_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));

        // Load per‑channel arrays into local memory for fast access
        pipe_.InitBuffer(meanLocal_, numFeatures_ * sizeof(float));
        pipe_.InitBuffer(varLocal_, numFeatures_ * sizeof(float));
        pipe_.InitBuffer(weightLocal_, numFeatures_ * sizeof(float));
        pipe_.InitBuffer(biasLocal_, numFeatures_ * sizeof(float));
        DataCopy(meanLocal_, meanGm_, numFeatures_);
        DataCopy(varLocal_, varGm_, numFeatures_);
        DataCopy(weightLocal_, weightGm_, numFeatures_);
        DataCopy(biasLocal_, biasGm_, numFeatures_);
    }

    __aicore__ inline void Process() {
        if (numRows_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        for (uint32_t row = blockIdx; row < numRows_; row += blockNum) {
            uint32_t channel = row % numFeatures_;

            float mean_val = meanLocal_(channel);
            float var_val = varLocal_(channel);
            float weight_val = weightLocal_(channel);
            float bias_val = biasLocal_(channel);
            float scale = weight_val / Sqrt(var_val + eps_);
            float shift = bias_val - scale * mean_val;

            uint32_t base = row * spatialSize_;
            uint32_t remaining = spatialSize_;

            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

                // CopyIn
                LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                DataCopy(xLocal, xGm_[base], curLen);
                inQueueX_.EnQue(xLocal);

                // Compute
                LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
                LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                Muls(yLocal, xDeque, scale, curLen);
                Adds(yLocal, yLocal, shift, curLen);
                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xDeque);

                // CopyOut
                LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
                DataCopy(yGm_[base], yDeque, curLen);
                outQueueY_.FreeTensor(yDeque);

                base += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    LocalTensor<float> meanLocal_;
    LocalTensor<float> varLocal_;
    LocalTensor<float> weightLocal_;
    LocalTensor<float> biasLocal_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> meanGm_;
    GlobalTensor<float> varGm_;
    GlobalTensor<float> weightGm_;
    GlobalTensor<float> biasGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
    float eps_ {1e-5f};
    uint32_t numFeatures_ {0};
    uint32_t spatialSize_ {0};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04600_custom(GM_ADDR x, GM_ADDR y, GM_ADDR running_mean, GM_ADDR running_var,
                                                         GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04600Custom op;
    op.Init(x, y, running_mean, running_var, weight, bias,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim,
            tiling_data.eps, tiling_data.numFeatures, tiling_data.spatialSize);
    op.Process();
}

        self.momentum = momentum
        self.running_mean = torch.nn.Parameter(torch.zeros(num_features), requires_grad=False)

        self.weight = torch.nn.Parameter(torch.ones(num_features))
        self.bias = torch.nn.Parameter(torch.zeros(num_features))

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.running_mean, self.running_var, self.weight, self.bias, self.eps)
