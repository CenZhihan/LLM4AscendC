kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelStdMean {
public:
    __aicore__ inline KernelStdMean() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR means, GM_ADDR stds,
                                uint32_t B, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t tileH, uint32_t tileW)
    {
        totalB_ = B;
        totalC_ = C;
        totalH_ = H;
        totalW_ = W;
        tileH_ = tileH;
        tileW_ = tileW;
        xGm_.SetGlobalBuffer((__gm__ float*)x, B * C * H * W);
        meansGm_.SetGlobalBuffer((__gm__ float*)means, C);
        stdsGm_.SetGlobalBuffer((__gm__ float*)stds, C);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileH_ * tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= totalC_) return;

        float sum = 0.0f;
        float sumSq = 0.0f;
        uint32_t count = 0;

        for (uint32_t b = 0; b < totalB_; ++b) {
            for (uint32_t h = 0; h < totalH_; h += tileH_) {
                uint32_t curTileH = tileH_;
                if (h + curTileH > totalH_) curTileH = totalH_ - h;

                uint32_t offset = (b * totalC_ + blockIdx) * totalH_ * totalW_
                                + h * totalW_;
                uint32_t tileSize = curTileH * totalW_;

                AscendC::LocalTensor<float> tile = inQueue_.AllocTensor<float>();
                AscendC::DataCopy(tile, xGm_[offset], tileSize);
                inQueue_.EnQue(tile);

                AscendC::LocalTensor<float> tileDeq = inQueue_.DeQue<float>();
                for (uint32_t i = 0; i < tileSize; ++i) {
                    float val = tileDeq(i);
                    sum += val;
                    sumSq += val * val;
                }
                count += tileSize;
                inQueue_.FreeTensor(tileDeq);
            }
        }

        float mean = sum / static_cast<float>(count);
        float var = (sumSq - sum * mean) / static_cast<float>(count - 1);
        float std = sqrtf(var);

        meansGm_.SetValue(blockIdx, mean);
        stdsGm_.SetValue(blockIdx, std);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> meansGm_;
    AscendC::GlobalTensor<float> stdsGm_;

    uint32_t totalB_ {0};
    uint32_t totalC_ {0};
    uint32_t totalH_ {0};
    uint32_t totalW_ {0};
    uint32_t tileH_ {64};
    uint32_t tileW_ {512};
};

extern "C" __global__ __aicore__ void ca6k_03129_custom(GM_ADDR x, GM_ADDR means, GM_ADDR stds,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelStdMean op;
    op.Init(x, means, stds,
            tiling_data.B, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.tileH, tiling_data.tileW);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

std::vector<at::Tensor> std_mean_impl_npu(const at::Tensor& x) {
    auto shape = x.sizes();
    int64_t C = shape[1];
    at::Tensor means = at::empty({C}, x.options());
    at::Tensor stds = at::empty({C}, x.options());
    EXEC_NPU_CMD(aclnnCa6k03129Custom, x, means, stds);
    return {stds, means};
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_03129_custom", &std_mean_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("ca6k_03129_custom", &std_mean_impl_npu, "fused std & mean per channel (Ca6k03129Custom on NPU)");
