kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr uint32_t BUFFER_NUM = 1;

class KernelCa6k04593Custom {
public:
    __aicore__ inline KernelCa6k04593Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t channels,
                                uint32_t depth, uint32_t height, uint32_t width,
                                float eps)
    {
        batch_ = batch;
        channels_ = channels;
        depth_ = depth;
        height_ = height;
        width_ = width;
        spatialSize_ = depth * height * width;
        totalRows_ = batch * channels;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * spatialSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * spatialSize_);

        pipe_.InitBuffer(rowBuf_, BUFFER_NUM, spatialSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            uint32_t offset = row * spatialSize_;

            LocalTensor<float> rowLocal = rowBuf_.AllocTensor<float>();
            DataCopy(rowLocal, xGm_[offset], spatialSize_);

            // Compute mean
            float sum = 0.0f;
            for (uint32_t i = 0; i < spatialSize_; i++) {
                sum += rowLocal(i);
            }
            float mean = sum / static_cast<float>(spatialSize_);

            // Compute variance (biased)
            float sqSum = 0.0f;
            for (uint32_t i = 0; i < spatialSize_; i++) {
                float diff = rowLocal(i) - mean;
                sqSum += diff * diff;
            }
            float var = sqSum / static_cast<float>(spatialSize_);

            // Normalize
            float invStd = 1.0f / sqrtf(var + eps_);
            for (uint32_t i = 0; i < spatialSize_; i++) {
                rowLocal(i) = (rowLocal(i) - mean) * invStd;
            }

            DataCopy(yGm_[offset], rowLocal, spatialSize_);
            rowBuf_.FreeTensor(rowLocal);
        }
    }

private:
    TPipe pipe_;
    TBuf<TPosition::VECCALC> rowBuf_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t channels_;
    uint32_t depth_;
    uint32_t height_;
    uint32_t width_;
    uint32_t spatialSize_;
    uint32_t totalRows_;
    float eps_;
};

extern "C" __global__ __aicore__ void ca6k_04593_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04593Custom op;
    op.Init(x, y,
            tiling_data.batch, tiling_data.channels,
            tiling_data.depth, tiling_data.height, tiling_data.width,
            tiling_data.eps);
    op.Process();
}
