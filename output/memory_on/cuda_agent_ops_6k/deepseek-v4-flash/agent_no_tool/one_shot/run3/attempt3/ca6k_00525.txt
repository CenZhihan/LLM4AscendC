#include "ca6k_00525_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;
static constexpr uint32_t MIN_TILE_ROWS = 1;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00525CustomTilingData tiling;

    const gert::StorageShape* inputStorage = context->GetInputShape(0);
    const gert::Shape& inputShape = inputStorage->GetOriginShape();
    const uint32_t ndim = inputShape.GetDimNum();
    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    bool keepdim = attrs->GetAttrPointer<bool>(1)[0];
    if (dim < 0) dim += ndim;

    uint32_t totalElements = 1;
    uint32_t reductionSize = 1;
    for (uint32_t i = 0; i < ndim; ++i) {
        uint32_t sz = static_cast<uint32_t>(inputShape.GetDim(i));
        totalElements *= sz;
        if (i == static_cast<uint32_t>(dim)) reductionSize = sz;
    }
    uint32_t totalRows = totalElements / reductionSize;

    uint32_t maxRows = UB_BUDGET_BYTES / (2 * reductionSize * sizeof(float));
    if (maxRows < MIN_TILE_ROWS) maxRows = MIN_TILE_ROWS;
    uint32_t tileRows = AlignDown(maxRows, 8);
    if (tileRows == 0) tileRows = MIN_TILE_ROWS;
    if (totalRows < tileRows) tileRows = totalRows;

    uint32_t numTiles = CeilDiv(totalRows, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_totalElements(totalElements);
    tiling.set_tileElements(tileRows * reductionSize);
    tiling.set_blockDim(blockDim);
    tiling.set_reductionSize(reductionSize);
    tiling.set_totalRows(totalRows);
    tiling.set_tileRows(tileRows);
    tiling.set_dim(dim);
    tiling.set_keepdim(keepdim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    bool keepdim = attrs->GetAttrPointer<bool>(1)[0];
    uint32_t ndim = x_shape->GetDimNum();
    if (dim < 0) dim += ndim;

    // Build output shape for both var and mean
    gert::Shape* outShape0 = context->GetOutputShape(0);
    gert::Shape* outShape1 = context->GetOutputShape(1);
    uint32_t outDimNum = keepdim ? ndim : ndim - 1;
    outShape0->SetDimNum(outDimNum);
    outShape1->SetDimNum(outDimNum);
    uint32_t outIdx = 0;
    for (uint32_t i = 0; i < ndim; ++i) {
        if (i == static_cast<uint32_t>(dim)) {
            if (keepdim) {
                outShape0->SetDim(outIdx, 1);
                outShape1->SetDim(outIdx, 1);
                outIdx++;
            }
        } else {
            int64_t d = static_cast<int64_t>(x_shape->GetDim(i));
            outShape0->SetDim(outIdx, d);
            outShape1->SetDim(outIdx, d);
            outIdx++;
        }
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    context->SetOutputDataType(1, dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00525Custom {
public:
    __aicore__ inline KernelCa6k00525Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR var, GM_ADDR mean,
                                uint32_t totalElements, uint32_t tileElements,
                                uint32_t reductionSize, uint32_t totalRows,
                                uint32_t tileRows, int32_t dim, bool keepdim) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElements);
        varGm_.SetGlobalBuffer((__gm__ float*)var, totalRows);
        meanGm_.SetGlobalBuffer((__gm__ float*)mean, totalRows);
        totalRows_ = totalRows;
        tileRows_ = tileRows;
        reductionSize_ = reductionSize;
        tileElements_ = tileElements;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, reductionSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueSq_, 1, reductionSize_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalRows_ == 0 || tileRows_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideRows = blockNum * tileRows_;
        uint32_t baseRow = blockIdx * tileRows_;
        if (baseRow >= totalRows_) return;

        while (true) {
            uint32_t curTileRows = totalRows_ - baseRow;
            if (curTileRows > tileRows_) curTileRows = tileRows_;

            if (curTileRows > 0) {
                CopyIn(baseRow, reductionSize_);
            }

            for (uint32_t k = 0; k < curTileRows; ++k) {
                uint32_t curRow = baseRow + k;
                if (k + 1 < curTileRows) {
                    CopyIn(curRow + 1, reductionSize_);
                }
                Compute(curRow);
                CopyOut(curRow);
            }

            baseRow += strideRows;
            if (baseRow >= totalRows_) break;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowIdx * reductionSize_], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rowIdx) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> sqLocal = tmpQueueSq_.AllocTensor<float>();

        AscendC::Mul(sqLocal, xLocal, xLocal, reductionSize_);
        float sum_x = AscendC::ReduceSum(xLocal, reductionSize_);
        float sum_x2 = AscendC::ReduceSum(sqLocal, reductionSize_);

        float n = static_cast<float>(reductionSize_);
        float mean_val = sum_x / n;
        float var_val = sum_x2 / n - mean_val * mean_val;

        meanGm_.SetValue(rowIdx, mean_val);
        varGm_.SetValue(rowIdx, var_val);

        tmpQueueSq_.FreeTensor(sqLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx) {
        // results already written
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueSq_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> varGm_;
    AscendC::GlobalTensor<float> meanGm_;

    uint32_t totalRows_ {0};
    uint32_t tileRows_ {0};
    uint32_t reductionSize_ {0};
    uint32_t tileElements_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00525_custom(GM_ADDR x, GM_ADDR var, GM_ADDR mean, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00525Custom op;
    op.Init(x, var, mean,
            tiling_data.totalElements, tiling_data.tileElements,
            tiling_data.reductionSize, tiling_data.totalRows,
            tiling_data.tileRows, tiling_data.dim, tiling_data.keepdim);
    op.Process();
}
'''
