class KernelCa6k01096Custom {
public:
    __aicore__ inline KernelCa6k01096Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                Ca6k01096CustomTilingData *tiling)
    {
        batch_ = tiling->get_batch();
        inChannels_ = tiling->get_inChannels();
        inHeight_ = tiling->get_inHeight();
        inWidth_ = tiling->get_inWidth();
        outChannels_ = tiling->get_outChannels();
        outHeight_ = tiling->get_outHeight();
        outWidth_ = tiling->get_outWidth();
        kH_ = tiling->get_kernelHeight();
        kW_ = tiling->get_kernelWidth();
        strideH_ = tiling->get_strideH();
        strideW_ = tiling->get_strideW();
        padH_ = tiling->get_padH();
        padW_ = tiling->get_padW();
        outputPadH_ = tiling->get_outputPadH();
        outputPadW_ = tiling->get_outputPadW();
        totalOutputElems_ = tiling->get_totalOutputElems();

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_ * inChannels_ * inHeight_ * inWidth_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outChannels_ * inChannels_ * kH_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElems_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inHeight_ * inWidth_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, kH_ * kW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outHeight_ * outWidth_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, outHeight_ * outWidth_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t tileLen_ = totalOutputElems_ / blockNum;
        if (tileLen_ == 0) tileLen_ = 1;
        uint32_t start = blockIdx * tileLen_;
        uint32_t end = (blockIdx == blockNum - 1) ? totalOutputElems_ : start + tileLen_;

        for (uint32_t idx = start; idx < end; idx++) {
            uint32_t temp1 = idx / (outChannels_ * outHeight_ * outWidth_);
            uint32_t n = temp1;
            uint32_t rem = idx - n * outChannels_ * outHeight_ * outWidth_;
            uint32_t co = rem / (outHeight_ * outWidth_);
            rem = rem - co * outHeight_ * outWidth_;
            uint32_t oh = rem / outWidth_;
            uint32_t ow = rem - oh * outWidth_;

            float sum = 0.0f;
            for (uint32_t ci = 0; ci < inChannels_; ci++) {
                for (uint32_t kh = 0; kh < kH_; kh++) {
                    for (uint32_t kw = 0; kw < kW_; kw++) {
                        int32_t ih = (int32_t)oh + padH_ - (int32_t)kh;
                        int32_t iw = (int32_t)ow + padW_ - (int32_t)kw;
                        if (ih % strideH_ != 0 || iw % strideW_ != 0) continue;
                        ih /= strideH_;
                        iw /= strideW_;
                        if (ih < 0 || (uint32_t)ih >= inHeight_ || iw < 0 || (uint32_t)iw >= inWidth_) continue;
                        float xVal = xGm_[n * inChannels_ * inHeight_ * inWidth_ + ci * inHeight_ * inWidth_ + (uint32_t)ih * inWidth_ + (uint32_t)iw];
                        float wVal = wGm_[co * inChannels_ * kH_ * kW_ + ci * kH_ * kW_ + kh * kW_ + kw];
                        sum += xVal * wVal;
                    }
                }
            }
            yGm_[idx] = sum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t inChannels_;
    uint32_t inHeight_;
    uint32_t inWidth_;
    uint32_t outChannels_;
    uint32_t outHeight_;
    uint32_t outWidth_;

    int32_t strideW_;
    int32_t padH_;
    int32_t padW_;
    int32_t outputPadH_;
    int32_t outputPadW_;
    uint32_t totalOutputElems_;
};
