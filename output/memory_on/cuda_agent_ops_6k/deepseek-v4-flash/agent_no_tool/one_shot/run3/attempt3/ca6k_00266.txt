#include "ca6k_00266_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t TILE_HARD_CAP = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00266CustomTilingData tiling;

    auto indices_shape = context->GetInputShape(0)->GetOriginShape();
    auto offsets_shape = context->GetInputShape(1)->GetOriginShape();
    auto weight_shape = context->GetInputShape(2)->GetOriginShape();

    uint32_t totalIndices = static_cast<uint32_t>(indices_shape.GetShapeSize());
    uint32_t numOffsets = static_cast<uint32_t>(offsets_shape.GetShapeSize());
    uint32_t numEmbeddings = static_cast<uint32_t>(weight_shape.GetDim(0));
    uint32_t embeddingDim = static_cast<uint32_t>(weight_shape.GetDim(1));
    uint32_t totalBags = numOffsets - 1;

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > TILE_HARD_CAP) maxTile = AlignDown(TILE_HARD_CAP, ALIGN_ELEMS);
    uint32_t tileDim = (embeddingDim < maxTile) ? embeddingDim : maxTile;
    tileDim = AlignDown(tileDim, ALIGN_ELEMS);
    if (tileDim == 0) tileDim = ALIGN_ELEMS;

    uint32_t blockDim = totalBags;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalBags(totalBags);
    tiling.set_embeddingDim(embeddingDim);
    tiling.set_numEmbeddings(numEmbeddings);
    tiling.set_tileDim(tileDim);
    tiling.set_totalIndices(totalIndices);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    auto weight_shape = context->GetInputShape(2);
    uint32_t embeddingDim = static_cast<uint32_t>(weight_shape->GetDim(1));
    auto offsets_shape = context->GetInputShape(1);
    uint32_t numOffsets = static_cast<uint32_t>(offsets_shape->GetDim(0));
    uint32_t totalBags = numOffsets - 1;

    gert::Shape *output_shape = context->GetOutputShape(0);
    output_shape->SetDimNum(2);
    output_shape->SetDim(0, totalBags);
    output_shape->SetDim(1, embeddingDim);

}

kernel_src = '''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00266Custom {
public:
    __aicore__ inline KernelCa6k00266Custom() {}

    __aicore__ inline void Init(GM_ADDR indices, GM_ADDR offsets, GM_ADDR weight, GM_ADDR output,
                                uint32_t totalBags, uint32_t embeddingDim,
                                uint32_t numEmbeddings, uint32_t tileDim, uint32_t totalIndices) {
        totalBags_ = totalBags;
        embeddingDim_ = embeddingDim;
        numEmbeddings_ = numEmbeddings;
        tileDim_ = tileDim;

        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices, static_cast<int64_t>(totalIndices));
        offsetsGm_.SetGlobalBuffer((__gm__ int64_t*)offsets, static_cast<int64_t>(totalBags + 1));
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, static_cast<int64_t>(numEmbeddings) * embeddingDim);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, static_cast<int64_t>(totalBags) * embeddingDim);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileDim_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalBags_ == 0 || embeddingDim_ == 0 || tileDim_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        for (uint32_t bag = blockIdx; bag < totalBags_; bag += blockNum) {
            int64_t bagStart = offsetsGm_[static_cast<int64_t>(bag)];
            int64_t bagEnd   = offsetsGm_[static_cast<int64_t>(bag) + 1];

            for (uint32_t t = 0; t < embeddingDim_; t += tileDim_) {
                uint32_t curTile = embeddingDim_ - t;
                if (curTile > tileDim_) curTile = tileDim_;

                LocalTensor<float> outTile = outQueueY_.AllocTensor<float>();
                AscendC::SetZero(outTile, curTile);

                for (int64_t i = bagStart; i < bagEnd; i++) {
                    int64_t idx = indicesGm_[i];
                    int64_t weightOffset = idx * static_cast<int64_t>(embeddingDim_) + static_cast<int64_t>(t);
                    LocalTensor<float> wTile = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(wTile, weightGm_[weightOffset], curTile);
                    AscendC::Add(outTile, outTile, wTile, curTile);
                    inQueueX_.FreeTensor(wTile);
                }

                int64_t outputOffset = static_cast<int64_t>(bag) * embeddingDim_ + static_cast<int64_t>(t);
                AscendC::DataCopy(outputGm_[outputOffset], outTile, curTile);
                outQueueY_.FreeTensor(outTile);
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    GlobalTensor<int64_t> indicesGm_;
    GlobalTensor<int64_t> offsetsGm_;
    GlobalTensor<float> weightGm_;
    GlobalTensor<float> outputGm_;

    uint32_t totalBags_ = 0;
    uint32_t embeddingDim_ = 0;
    uint32_t numEmbeddings_ = 0;
    uint32_t tileDim_ = 0;
};

extern "C" __global__ __aicore__ void ca6k_00266_custom(GM_ADDR indices, GM_ADDR offsets,
                                                        GM_ADDR weight, GM_ADDR output,
                                                        GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00266Custom op;
    op.Init(indices, offsets, weight, output,
            tiling_data.totalBags, tiling_data.embeddingDim,
            tiling_data.numEmbeddings, tiling_data.tileDim, tiling_data.totalIndices);
    op.Process();
}
'''
