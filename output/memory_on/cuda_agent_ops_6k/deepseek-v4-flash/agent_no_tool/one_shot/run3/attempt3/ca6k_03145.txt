kernel_src = """
#include "kernel_operator.h"

using namespace AscendC;

class KernelCa6k03145Custom {
public:
    __aicore__ inline KernelCa6k03145Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        xGm_.SetGlobalBuffer((__gm__ float*)x, M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M_ * N_);
    }

    __aicore__ inline void Process()
    {
        MatMul<float, float, float> matmul;
        matmul.SetTensorA(xGm_);
        matmul.SetTensorB(wGm_);
        matmul.SetBias(bGm_);
        matmul.SetResult(yGm_);
        matmul.SetConfig(M_, N_, K_, false, true, true);
        matmul.Process();
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;
    uint32_t M_;
    uint32_t N_;
    uint32_t K_;
};

extern "C" __global__ __aicore__ void ca6k03145_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k03145Custom op;
    op.Init(x, weight, bias, y, tiling_data.M, tiling_data.N, tiling_data.K);
    op.Process();
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k03145_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, bias=True):
        super(ModelNew, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.weight = torch.nn.Parameter(torch.randn(out_features, in_features))
        self.bias = torch.nn.Parameter(torch.zeros(out_features)) if bias else None

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias)
