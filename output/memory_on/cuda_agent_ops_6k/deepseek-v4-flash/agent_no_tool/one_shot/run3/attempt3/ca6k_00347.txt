    int32_t outputH = (height + 2 * padding - dilationH * (kernelH - 1) - 1) / stride + 1;
    int32_t outputW = (width  + 2 * padding - dilationW * (kernelW - 1) - 1) / stride + 1;

#include "kernel_operator.h"

class KernelCa6k00347Custom {
public:
    __aicore__ inline KernelCa6k00347Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                Ca6k00347CustomTilingData td)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        batchSize_ = td.batchSize_;
        channels_ = td.channels_;
        height_ = td.height_;
        width_ = td.width_;
        kernelH_ = td.kernelH_;
        kernelW_ = td.kernelW_;
        dilationH_ = td.dilationH_;
        dilationW_ = td.dilationW_;
        stride_ = static_cast<uint32_t>(td.stride_);
        padding_ = static_cast<uint32_t>(td.padding_);
        outputH_ = static_cast<uint32_t>(td.outputH_);
        outputW_ = static_cast<uint32_t>(td.outputW_);
        outputRows_ = static_cast<uint32_t>(td.outputRows_);
        totalOutputElems_ = td.totalOutputElems_;
        blockDim_ = td.blockDim_;
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        for (uint64_t elemIdx = blockIdx; elemIdx < totalOutputElems_; elemIdx += blockNum) {
            uint64_t inputIdx = computeInputIndex(elemIdx);
            float val = 0.0f;
            if (inputIdx != ~0ULL) {
                val = xGm_(inputIdx);
            }
            yGm_(elemIdx) = val;
        }
    }

private:
    __aicore__ inline uint64_t computeInputIndex(uint64_t elemIdx)
    {
        // elemIdx element in output (batch, r, h_out, w_out)
        uint64_t batch = elemIdx / (static_cast<uint64_t>(outputRows_) * outputH_ * outputW_);
        uint64_t rem = elemIdx % (static_cast<uint64_t>(outputRows_) * outputH_ * outputW_);
        uint64_t r = rem / (outputH_ * outputW_);         // output row (channel group)
        uint64_t p = rem % (outputH_ * outputW_);         // spatial position
        uint64_t h_out = p / outputW_;
        uint64_t w_out = p % outputW_;

        // r = c * kernelH_ * kernelW_ + kh * kernelW_ + kw
        uint64_t c = r / (static_cast<uint64_t>(kernelH_) * kernelW_);
        uint64_t kr = (r % (static_cast<uint64_t>(kernelH_) * kernelW_)) / kernelW_;
        uint64_t kc = (r % (static_cast<uint64_t>(kernelH_) * kernelW_)) % kernelW_;

        int64_t h_in = static_cast<int64_t>(h_out) * static_cast<int64_t>(stride_)
                        + static_cast<int64_t>(kr) * static_cast<int64_t>(dilationH_)
                        - static_cast<int64_t>(padding_);
        int64_t w_in = static_cast<int64_t>(w_out) * static_cast<int64_t>(stride_)
                        + static_cast<int64_t>(kc) * static_cast<int64_t>(dilationW_)
                        - static_cast<int64_t>(padding_);

        if (h_in >= 0 && h_in < static_cast<int64_t>(height_) &&
            w_in >= 0 && w_in < static_cast<int64_t>(width_)) {
            uint64_t inputIdx = batch * channels_ * height_ * width_
                                + c * height_ * width_
                                + static_cast<uint64_t>(h_in) * width_
                                + static_cast<uint64_t>(w_in);
            return inputIdx;
        }
        return ~0ULL; // signal invalid
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t channels_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    uint32_t kernelH_ {3};
    uint32_t kernelW_ {5};
    uint32_t dilationH_ {1};
    uint32_t dilationW_ {1};
    uint32_t stride_ {1};
    uint32_t padding_ {0};
    uint32_t outputH_ {0};
    uint32_t outputW_ {0};
    uint32_t outputRows_ {0};
    uint32_t totalOutputElems_ {0};
    uint32_t blockDim_ {24};
};

extern "C" __global__ __aicore__ void ca6k_00347_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00347Custom op;
    op.Init(x, y, tiling_data);
    op.Process();
}
'''

import torch

_FN = 'ca6k_00347_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        self.kernel_h = init_inputs[2]  # out_channels unused; use kernel_size from init_inputs
        self.kernel_w = init_inputs[3]
        self.dilation_h = init_inputs[4]
        self.dilation_w = init_inputs[5]
        # stride and padding default to 1 and 0 as per nn.Unfold defaults
        self.stride = 1
        self.padding = 0

    def forward(self, *inputs):
        x = inputs[0]
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_h, self.kernel_w, self.dilation_h, self.dilation_w, self.stride, self.padding)
