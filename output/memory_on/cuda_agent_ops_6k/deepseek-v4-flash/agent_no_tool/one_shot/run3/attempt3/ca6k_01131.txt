host_operator_src = '''
#include "ca6k_01131_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k01131CustomTilingData tiling;

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    int64_t inN = inputShape->GetOriginShape().GetDim(0);
    int64_t inC = inputShape->GetOriginShape().GetDim(1);
    int64_t inH = inputShape->GetOriginShape().GetDim(2);
    int64_t inW = inputShape->GetOriginShape().GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *paddings = attrs->GetAttrPointer<int64_t>(0);
    float value = *attrs->GetAttrPointer<float>(1);

    int64_t padFront = paddings[0];
    int64_t padBack  = paddings[1];
    int64_t padTop   = paddings[2];
    int64_t padBottom= paddings[3];
    int64_t padLeft  = paddings[4];
    int64_t padRight = paddings[5];

    int64_t outN = inN;
    int64_t outC = inC + padFront + padBack;
    int64_t outH = inH + padTop + padBottom;
    int64_t outW = inW + padLeft + padRight;

    uint64_t totalOutputElements = outN * outC * outH * outW;
    if (totalOutputElements == 0) {
        context->SetBlockDim(1);
        tiling.set_totalLength(0);
        tiling.set_tileLength(0);
        tiling.set_blockDim(1);
        tiling.set_outN(outN); tiling.set_outC(outC); tiling.set_outH(outH); tiling.set_outW(outW);
        tiling.set_inC(inC); tiling.set_inH(inH); tiling.set_inW(inW);
        tiling.set_padFront(padFront); tiling.set_padBack(padBack);
        tiling.set_padTop(padTop); tiling.set_padBottom(padBottom);
        tiling.set_padLeft(padLeft); tiling.set_padRight(padRight);
        tiling.set_value(value);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    uint32_t totalLen = static_cast<uint32_t>(totalOutputElements);
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLen < tileLen) {
        uint32_t aligned = AlignDown(totalLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLen;
    }

    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(totalLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLen);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_outN(outN); tiling.set_outC(outC); tiling.set_outH(outH); tiling.set_outW(outW);
    tiling.set_inC(inC); tiling.set_inH(inH); tiling.set_inW(inW);
    tiling.set_padFront(padFront); tiling.set_padBack(padBack);
    tiling.set_padTop(padTop); tiling.set_padBottom(padBottom);
    tiling.set_padLeft(padLeft); tiling.set_padRight(padRight);
    tiling.set_value(value);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *inputShape = context->GetInputShape(0);
    gert::Shape *outputShape = context->GetOutputShape(0);
    int64_t dims[4];
    dims[0] = inputShape->GetDim(0);
    dims[1] = inputShape->GetDim(1);
    dims[2] = inputShape->GetDim(2);
    dims[3] = inputShape->GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *paddings = attrs->GetAttrPointer<int64_t>(0);
    dims[1] += paddings[0] + paddings[1];
    dims[2] += paddings[2] + paddings[3];
    dims[3] += paddings[4] + paddings[5];

    outputShape->SetDimNum(4);
    for (int i = 0; i < 4; i++) {
        outputShape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01131Custom {
public:
    __aicore__ inline KernelCa6k01131Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int64_t outN, int64_t outC, int64_t outH, int64_t outW,
                                int64_t inC, int64_t inH, int64_t inW,
                                int64_t padFront, int64_t padBack,
                                int64_t padTop, int64_t padBottom,
                                int64_t padLeft, int64_t padRight,
                                float value) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        outN_ = outN; outC_ = outC; outH_ = outH; outW_ = outW;
        inC_ = inC; inH_ = inH; inW_ = inW;
        padFront_ = padFront; padBack_ = padBack;
        padTop_ = padTop; padBottom_ = padBottom;
        padLeft_ = padLeft; padRight_ = padRight;
        value_ = value;

        // Precompute strides
        outStrideC_ = outH * outW;
        outStrideH_ = outW;
        outStrideW_ = 1;
        inStrideC_ = inH * inW;
        inStrideH_ = inW;
        inStrideW_ = 1;
        batchStride_ = outC * outH * outW;
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t stride = blockNum * tileLen_;
        uint32_t start = blockIdx * tileLen_;
        if (start >= totalLen_) return;

        uint32_t end = start + tileLen_;
        if (end > totalLen_) end = totalLen_;

        for (uint32_t idx = start; idx < end; idx++) {
            int64_t outIndex = static_cast<int64_t>(idx);
            int64_t n = outIndex / batchStride_;
            int64_t rem = outIndex % batchStride_;
            int64_t c = rem / outStrideC_;
            rem = rem % outStrideC_;
            int64_t h = rem / outStrideH_;
            int64_t w = rem % outStrideW_;

            int64_t in_n = n;
            int64_t in_c = c - padFront_;
            int64_t in_h = h - padTop_;
            int64_t in_w = w - padLeft_;

            float val;
            if (in_c >= 0 && in_c < inC_ &&
                in_h >= 0 && in_h < inH_ &&
                in_w >= 0 && in_w < inW_) {
                int64_t inputOffset = in_n * inC_ * inH_ * inW_ +
                                      in_c * inStrideC_ +
                                      in_h * inStrideH_ +
                                      in_w;
                val = xGm_.GetValue(static_cast<uint32_t>(inputOffset));
            } else {
                val = value_;
            }
            yGm_.SetValue(idx, val);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int64_t outN_, outC_, outH_, outW_;
    int64_t inC_, inH_, inW_;
    int64_t padFront_, padBack_, padTop_, padBottom_, padLeft_, padRight_;
    float value_ {0.0f};
    int64_t outStrideC_, outStrideH_, outStrideW_;
    int64_t inStrideC_, inStrideH_, inStrideW_;
    int64_t batchStride_;
};

extern "C" __global__ __aicore__ void ca6k_01131_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01131Custom op;
    op.Init(x, y,
            tiling_data.get_totalLength(), tiling_data.get_tileLength(),
            tiling_data.get_outN(), tiling_data.get_outC(), tiling_data.get_outH(), tiling_data.get_outW(),
            tiling_data.get_inC(), tiling_data.get_inH(), tiling_data.get_inW(),
            tiling_data.get_padFront(), tiling_data.get_padBack(),
            tiling_data.get_padTop(), tiling_data.get_padBottom(),
            tiling_data.get_padLeft(), tiling_data.get_padRight(),
            tiling_data.get_value());
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_01131_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, pad_dims, padding_value):
        super(ModelNew, self).__init__()
        self.pad_dims = list(pad_dims)
        self.padding_value = float(padding_value)

    def forward(self, *inputs):
        x = inputs[0]
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.pad_dims, self.padding_value)
