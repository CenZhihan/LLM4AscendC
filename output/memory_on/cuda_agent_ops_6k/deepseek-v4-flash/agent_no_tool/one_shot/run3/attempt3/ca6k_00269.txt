    Ca6k00269CustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSizeH = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t kernelSizeW = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t strideH = attrs->GetAttrPointer<int32_t>(2)[0];
    int32_t strideW = attrs->GetAttrPointer<int32_t>(3)[0];

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    int32_t N = static_cast<int32_t>(x_shape->GetOriginShape().GetDim(0));
    int32_t C = static_cast<int32_t>(x_shape->GetOriginShape().GetDim(1));
    int32_t H = static_cast<int32_t>(x_shape->GetOriginShape().GetDim(2));
    int32_t W = static_cast<int32_t>(x_shape->GetOriginShape().GetDim(3));

    int32_t outH = (H - kernelSizeH) / strideH + 1;
    int32_t outW = (W - kernelSizeW) / strideW + 1;

    uint32_t totalLength = static_cast<uint32_t>(N) * static_cast<uint32_t>(C) *
                           static_cast<uint32_t>(outH) * static_cast<uint32_t>(outW);

    uint32_t tileLength = 512;
    if (tileLength > MAX_TILE_ELEMS) tileLength = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);
    if (tileLength < ALIGN_ELEMS) tileLength = ALIGN_ELEMS;
    if (totalLength < tileLength) {
        tileLength = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLength == 0) tileLength = totalLength;
    }

    uint32_t numTiles = CeilDiv(totalLength, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    int32_t inputNStride = C * H * W;
    int32_t inputCStride = H * W;
    int32_t inputHStride = W;
    int32_t inputWStride = 1;

    int32_t outputCStride = outH * outW;
    int32_t outputHStride = outW;

    tiling.set_outputWStride(outputWStride);

    tiling.set_kernelSizeW(kernelSizeW);

    tiling.set_strideW(strideW);

kernel_src="""

#include "kernel_operator.h"
#include "ca6k_00269_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00269Custom {
public:
    __aicore__ inline KernelCa6k00269Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(tiling_data, tiling);
        totalLen_ = tiling_data.totalLength;
        tileLen_ = tiling_data.tileLength;
        blockDim_ = tiling_data.blockDim;

        inputNStride_ = tiling_data.inputNStride;
        inputCStride_ = tiling_data.inputCStride;
        inputHStride_ = tiling_data.inputHStride;
        inputWStride_ = tiling_data.inputWStride;
        outputNStride_ = tiling_data.outputNStride;
        outputCStride_ = tiling_data.outputCStride;
        outputHStride_ = tiling_data.outputHStride;
        outputWStride_ = tiling_data.outputWStride;

        kernelSizeH_ = tiling_data.kernelSizeH;
        kernelSizeW_ = tiling_data.kernelSizeW;
        strideH_ = tiling_data.strideH;
        strideW_ = tiling_data.strideW;
        outH_ = tiling_data.outH;
        outW_ = tiling_data.outW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inRowBufQ_, BUFFER_NUM, static_cast<uint32_t>(kernelSizeH_) * static_cast<uint32_t>(kernelSizeW_) * sizeof(float));
        pipe_.InitBuffer(tmpBufQ_, BUFFER_NUM, static_cast<uint32_t>(kernelSizeW_) * sizeof(float));
        pipe_.InitBuffer(accumBufQ_, 1, static_cast<uint32_t>(kernelSizeW_) * sizeof(float));
        pipe_.InitBuffer(outBufQ_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = blockDim_;

        uint32_t startIdx = blockIdx * tileLen_;
        while (startIdx < totalLen_) {
            uint32_t curTileLen = (startIdx + tileLen_ <= totalLen_) ? tileLen_ : (totalLen_ - startIdx);
            ProcessTile(startIdx, curTileLen);
            startIdx += blockNum * tileLen_;
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t tileStart, uint32_t curLen)
    {
        for (uint32_t i = 0; i < curLen; i++) {
            uint32_t globalIdx = tileStart + i;

            uint32_t oh_ow = globalIdx % (static_cast<uint32_t>(outH_) * static_cast<uint32_t>(outW_));
            uint32_t n = globalIdx / outputNStride_;
            uint32_t rem_n = globalIdx - n * outputNStride_;
            uint32_t c = rem_n / outputCStride_;
            uint32_t rem_c = rem_n - c * outputCStride_;
            uint32_t oh = rem_c / outputHStride_;
            uint32_t ow = rem_c - oh * outputHStride_;

            int32_t base_h = static_cast<int32_t>(oh) * strideH_;
            int32_t base_w = static_cast<int32_t>(ow) * strideW_;

            float sumSq = 0.0f;

            AscendC::LocalTensor<float> accum = accumBufQ_.AllocTensor<float>();
            AscendC::Dup(accum, 0.0f, static_cast<uint32_t>(kernelSizeW_));

            for (int32_t dh = 0; dh < kernelSizeH_; dh++) {
                int32_t hIdx = base_h + dh;
                if (hIdx < 0 || hIdx >= inputHStride_ * outH_) continue;

                uint32_t rowOffset = n * inputNStride_ + c * inputCStride_ + static_cast<uint32_t>(hIdx) * inputHStride_ + static_cast<uint32_t>(base_w);
                AscendC::LocalTensor<float> rowBuf = inRowBufQ_.AllocTensor<float>();
                AscendC::DataCopy(rowBuf, xGm_[rowOffset], static_cast<uint32_t>(kernelSizeW_));

                AscendC::LocalTensor<float> tmp = tmpBufQ_.AllocTensor<float>();
                AscendC::Mul(tmp, rowBuf, rowBuf, static_cast<uint32_t>(kernelSizeW_));
                AscendC::Add(accum, accum, tmp, static_cast<uint32_t>(kernelSizeW_));

                tmpBufQ_.FreeTensor(tmp);
                inRowBufQ_.FreeTensor(rowBuf);
            }

            AscendC::LocalTensor<float> outScalar = outBufQ_.AllocTensor<float>();
            AscendC::ReduceSum(outScalar, accum, static_cast<uint32_t>(kernelSizeW_));
            float s = outScalar.GetValue(0);
            s = sqrtf(s);

            uint32_t outOffset = n * outputNStride_ + c * outputCStride_ + oh * outputHStride_ + ow;
            yGm_.SetValue(outOffset, s);

            outBufQ_.FreeTensor(outScalar);
            accumBufQ_.FreeTensor(accum);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inRowBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accumBufQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outBufQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};

    int32_t inputNStride_ {0};
    int32_t inputCStride_ {0};
    int32_t inputHStride_ {0};
    int32_t inputWStride_ {0};
    int32_t outputNStride_ {0};
    int32_t outputCStride_ {0};
    int32_t outputHStride_ {0};
    int32_t outputWStride_ {0};
    int32_t kernelSizeH_ {3};
    int32_t kernelSizeW_ {5};
    int32_t strideH_ {1};
    int32_t strideW_ {1};
    int32_t outH_ {0};
    int32_t outW_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00269_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelCa6k00269Custom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00269_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, norm_type: int, kernel_size: tuple, stride: int = 1, ceil_mode: bool = False):
        super(ModelNew, self).__init__()
        self.norm_type = norm_type
        self.kernel_size = kernel_size
        self.stride = stride
        self.ceil_mode = ceil_mode

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if self.norm_type != 2:
            raise NotImplementedError("Only norm_type=2 is supported")
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size[0], self.kernel_size[1],
                  self.stride, self.stride)
