#include "ca6k00378_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00378CustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::Shape &originShape = xShape->GetOriginShape();
    const uint32_t ndim = static_cast<uint32_t>(originShape.GetDimNum());
    uint32_t totalLength = 1;
    for (uint32_t i = 0; i < ndim; ++i) {
        totalLength *= static_cast<uint32_t>(originShape.GetDim(i));
    }
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(3));
    uint32_t spatialSize = N * H * W;
    uint32_t batchSize = N;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    const float *momentumPtr = attrs->GetAttrPointer<float>(1);

    uint32_t maxTile = UB_BUDGET_BYTES / static_cast<uint32_t>(sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (spatialSize < tileLen) {
        uint32_t aligned = AlignDown(spatialSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : spatialSize;
    }
    if (tileLen == 0) tileLen = 1;

    uint32_t blockDim = C;
    if (blockDim > 32) blockDim = 32;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_numChannels(C);
    tiling.set_spatialSize(spatialSize);
    tiling.set_batchSize(batchSize);
    tiling.set_eps(*epsPtr);
    tiling.set_momentum(*momentumPtr);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    gert::Shape *yShape = context->GetOutputShape(0);
    *yShape = *xShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00378Custom {
public:
    __aicore__ inline KernelCa6k00378Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR running_mean, GM_ADDR running_var,
                                GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t numChannels, uint32_t spatialSize,
                                uint32_t batchSize, float eps)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        numChannels_ = numChannels;
        spatialSize_ = spatialSize;
        batchSize_ = batchSize;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        meanGm_.SetGlobalBuffer((__gm__ float*)running_mean, numChannels_);
        varGm_.SetGlobalBuffer((__gm__ float*)running_var, numChannels_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numChannels_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, numChannels_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));

        pipe_.InitBuffer(meanLocal_, 1, numChannels_ * sizeof(float));
        pipe_.InitBuffer(varLocal_, 1, numChannels_ * sizeof(float));
        pipe_.InitBuffer(weightLocal_, 1, numChannels_ * sizeof(float));
        pipe_.InitBuffer(biasLocal_, 1, numChannels_ * sizeof(float));

        AscendC::LocalTensor<float> meanLocal = meanLocal_.AllocTensor<float>();
        AscendC::LocalTensor<float> varLocal = varLocal_.AllocTensor<float>();
        AscendC::LocalTensor<float> weightLocal = weightLocal_.AllocTensor<float>();
        AscendC::LocalTensor<float> biasLocal = biasLocal_.AllocTensor<float>();
        AscendC::DataCopy(meanLocal, meanGm_, numChannels_);
        AscendC::DataCopy(varLocal, varGm_, numChannels_);
        AscendC::DataCopy(weightLocal, weightGm_, numChannels_);
        AscendC::DataCopy(biasLocal, biasGm_, numChannels_);
        meanQue_.EnQue(meanLocal);
        varQue_.EnQue(varLocal);
        weightQue_.EnQue(weightLocal);
        biasQue_.EnQue(biasLocal);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t channelsPerBlock = (numChannels_ + blockNum - 1) / blockNum;
        uint32_t startChannel = blockIdx * channelsPerBlock;
        uint32_t endChannel = startChannel + channelsPerBlock;
        if (endChannel > numChannels_) endChannel = numChannels_;

        AscendC::LocalTensor<float> meanLocal = meanQue_.DeQue<float>();
        AscendC::LocalTensor<float> varLocal = varQue_.DeQue<float>();
        AscendC::LocalTensor<float> weightLocal = weightQue_.DeQue<float>();
        AscendC::LocalTensor<float> biasLocal = biasQue_.DeQue<float>();

        for (uint32_t c = startChannel; c < endChannel; ++c) {
            float mean = meanLocal(c);
            float var = varLocal(c);
            float weight = weightLocal(c);
            float bias = biasLocal(c);
            float invStd = 1.0f / AscendC::Sqrt(var + eps_);

            uint32_t channelOffset = c * spatialSize_;
            uint32_t channelEnd = channelOffset + spatialSize_;

            for (uint32_t base = channelOffset; base < channelEnd; base += tileLen_) {
                uint32_t curLen = channelEnd - base;
                if (curLen > tileLen_) curLen = tileLen_;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base], curLen);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                AscendC::Subs(yLocal, xLocal, mean, curLen);
                AscendC::Muls(yLocal, yLocal, invStd, curLen);
                AscendC::Muls(yLocal, yLocal, weight, curLen);
                AscendC::Adds(yLocal, yLocal, bias, curLen);

                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xLocal);

                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[base], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
            }
        }

        meanQue_.FreeTensor(meanLocal);
        varQue_.FreeTensor(varLocal);
        weightQue_.FreeTensor(weightLocal);
        biasQue_.FreeTensor(biasLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> meanQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> varQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> biasQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> meanGm_;
    AscendC::GlobalTensor<float> varGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numChannels_ {0};
    uint32_t spatialSize_ {0};
    uint32_t batchSize_ {0};
    float eps_ {1e-05f};
};

extern "C" __global__ __aicore__ void ca6k00378_custom(GM_ADDR x, GM_ADDR running_mean, GM_ADDR running_var,
                                                        GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00378Custom op;
    op.Init(x, running_mean, running_var, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.numChannels, tiling_data.spatialSize,
            tiling_data.batchSize, tiling_data.eps);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00378_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_features, eps=1e-05, momentum=0.1, affine=True, track_running_stats=True):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.eps = eps
        self.momentum = momentum
        self.affine = affine
        self.register_buffer('running_mean', torch.zeros(num_features))
        self.register_buffer('running_var', torch.ones(num_features))
        if affine:
            self.weight = torch.nn.Parameter(torch.ones(num_features))
            self.bias = torch.nn.Parameter(torch.zeros(num_features))
        else:
            self.register_buffer('weight', torch.ones(num_features))
            self.register_buffer('bias', torch.zeros(num_features))

    def forward(self, x):
        return getattr(custom_ops_lib, _FN)(
            x,
            self.running_mean,
            self.running_var,
            self.weight,
            self.bias,
            eps=self.eps,
            momentum=self.momentum
        )
