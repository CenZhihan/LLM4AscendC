namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00301CustomTilingData tiling;

    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::StorageShape *y_storage = context->GetInputShape(1);
    const gert::Shape &x_shape = x_storage->GetOriginShape();
    const gert::Shape &y_shape = y_storage->GetOriginShape();
    if (x_shape.GetDimNum() != 3 || y_shape.GetDimNum() != 3)
        return ge::GRAPH_FAILED;

    uint32_t batch = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t M = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t K = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t N = static_cast<uint32_t>(y_shape.GetDim(2));

    // Compute tile sizes based on UB budget
    uint32_t available = (UB_BUDGET_BYTES * 3 / 4) / sizeof(float);
    uint32_t tileM = 64;
    uint32_t tileN = 64;
    uint32_t tileK = 128;
    while (tileM * tileK + tileK * tileN + tileM * tileN > available && tileM > 16) {
        tileM /= 2; tileN /= 2; tileK /= 2;
    }
    if (tileM < 16) { tileM = 16; tileN = 16; tileK = 16; }

    tiling.set_batchSize(batch);
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    uint32_t blockDim = batch;
    context->SetBlockDim(blockDim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00301Custom {
public:
    __aicore__ inline KernelCa6k00301Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t batch, uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        batch_ = batch;
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileK_ = tileK;
        tileN_ = tileN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * M * K);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * K * N);
        zGm_.SetGlobalBuffer((__gm__ float*)z, batch * M * N);

        uint32_t aSize = tileM_ * tileK_ * sizeof(float);
        uint32_t bSize = tileK_ * tileN_ * sizeof(float);
        uint32_t cSize = tileM_ * tileN_ * sizeof(float);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, aSize);
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, bSize);
        pipe_.InitBuffer(outQueueZ_, BUFFER_NUM, cSize);
        pipe_.InitBuffer(tmpQueue_, 1, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = GetBlockIdx();
        if (blockIdx >= batch_) return;

        MatMulBatch(blockIdx);
    }

private:
    __aicore__ inline void MatMulBatch(uint32_t b)
    {
        uint32_t xBatchOffset = b * M_ * K_;
        uint32_t yBatchOffset = b * K_ * N_;
        uint32_t zBatchOffset = b * M_ * N_;

        for (uint32_t i = 0; i < M_; i += tileM_) {
            uint32_t curM = (i + tileM_ <= M_) ? tileM_ : M_ - i;
            for (uint32_t j = 0; j < N_; j += tileN_) {
                uint32_t curN = (j + tileN_ <= N_) ? tileN_ : N_ - j;

                LocalTensor<float> cLocal = outQueueZ_.AllocTensor<float>();
                SetZero(cLocal, curM * curN);

                for (uint32_t k = 0; k < K_; k += tileK_) {
                    uint32_t curK = (k + tileK_ <= K_) ? tileK_ : K_ - k;

                    // Load A tile (curM x curK)
                    LocalTensor<float> aLocal = inQueueX_.AllocTensor<float>();
                    for (uint32_t r = 0; r < curM; ++r) {
                        uint32_t srcOffset = xBatchOffset + (i + r) * K_ + k;
                        uint32_t dstOffset = r * tileK_;
                        DataCopy(aLocal[dstOffset], xGm_[srcOffset], curK);
                    }

                    // Load B tile (curK x curN)
                    LocalTensor<float> bLocal = inQueueY_.AllocTensor<float>();
                    for (uint32_t r = 0; r < curK; ++r) {
                        uint32_t srcOffset = yBatchOffset + (k + r) * N_ + j;
                        uint32_t dstOffset = r * tileN_;
                        DataCopy(bLocal[dstOffset], yGm_[srcOffset], curN);
                    }

                    // Compute cLocal += aLocal * bLocal
                    LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

                    for (uint32_t rA = 0; rA < curM; ++rA) {
                        uint32_t aRowOff = rA * tileK_;
                        uint32_t cRowOff = rA * tileN_;
                        for (uint32_t rB = 0; rB < curK; ++rB) {
                            float aScalar = aLocal.GetValue(aRowOff + rB);
                            uint32_t bRowOff = rB * tileN_;
                            Muls(tmp, bLocal[bRowOff], aScalar, curN);
                            Add(cLocal[cRowOff], cLocal[cRowOff], tmp, curN);
                        }
                    }
                    tmpQueue_.FreeTensor(tmp);

                    inQueueX_.FreeTensor(aLocal);
                    inQueueY_.FreeTensor(bLocal);
                }

                // Write output tile to global
                for (uint32_t r = 0; r < curM; ++r) {
                    uint32_t dstOffset = zBatchOffset + (i + r) * N_ + j;
                    uint32_t srcOffset = r * tileN_;
                    DataCopy(zGm_[dstOffset], cLocal[srcOffset], curN);
                }

                outQueueZ_.FreeTensor(cLocal);
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueY_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueZ_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> zGm_;

    uint32_t batch_ {0};
    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {64};
    uint32_t tileK_ {128};
    uint32_t tileN_ {64};
};

extern "C" __global__ __aicore__ void ca6k_00301_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00301Custom op;
    op.Init(x, y, z,
            tiling_data.batchSize, tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
