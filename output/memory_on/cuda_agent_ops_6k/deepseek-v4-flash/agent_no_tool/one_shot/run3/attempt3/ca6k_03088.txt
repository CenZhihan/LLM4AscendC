kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

extern "C" __global__ __aicore__ void ca6k_03088_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(td, tiling);
  uint32_t rows = td.rows;
  uint32_t cols = td.cols;
  uint32_t blockIdx = GetBlockIdx();
  if (blockIdx >= cols) return;

  GlobalTensor<float> xGm;
  xGm.SetGlobalBuffer((__gm__ float *)x, rows * cols);
  GlobalTensor<float> wsGm;
  wsGm.SetGlobalBuffer((__gm__ float *)workspace, rows * cols);
  GlobalTensor<float> yGm;
  yGm.SetGlobalBuffer((__gm__ float *)y, cols);

  uint32_t base = blockIdx * rows;
  const uint32_t tileSize = 2048;
  TPipe pipe;
  TQue<TPosition::VECIN, 1> queueIn;
  TQue<TPosition::VECOUT, 1> queueOut;
  pipe.InitBuffer(queueIn, 1, tileSize * sizeof(float));
  pipe.InitBuffer(queueOut, 1, tileSize * sizeof(float));

  for (uint32_t offset = 0; offset < rows; offset += tileSize) {
    uint32_t copyLen = (offset + tileSize <= rows) ? tileSize : (rows - offset);
    LocalTensor<float> localIn = queueIn.AllocTensor<float>();
    DataCopy(localIn, xGm[base + offset], copyLen);
    queueIn.EnQue(localIn);

    LocalTensor<float> localOut = queueIn.DeQue<float>();
    DataCopy(wsGm[base + offset], localOut, copyLen);
    queueOut.FreeTensor(localOut);
  }

  // QuickSelect on workspace column
  int32_t left = 0;
  int32_t right = static_cast<int32_t>(rows) - 1;
  int32_t k = static_cast<int32_t>((rows - 1) / 2);
  while (left < right) {
    float pivot = wsGm[base + left];
    int32_t i = left - 1;
    int32_t j = right + 1;
    while (true) {
      do { i++; } while (wsGm[base + i] < pivot);
      do { j--; } while (wsGm[base + j] > pivot);
      if (i >= j) break;
      float tmp = wsGm[base + i];
      wsGm[base + i] = wsGm[base + j];
      wsGm[base + j] = tmp;
    }
    if (k <= j) right = j;
    else left = j + 1;
  }

  float median = wsGm[base + k];
  yGm[blockIdx] = median;
}
"""
