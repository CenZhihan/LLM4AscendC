#include "ca6k04599_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04599CustomTilingData tiling;

    // Get origin shape
    const gert::Shape &xShape = context->GetInputShape(0)->GetOriginShape();

    const uint32_t inN = xShape.GetDim(0);
    const uint32_t inC = xShape.GetDim(1);
    const uint32_t inD = xShape.GetDim(2);
    const uint32_t inH = xShape.GetDim(3);
    const uint32_t inW = xShape.GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(0);

    uint32_t padLeft   = static_cast<uint32_t>(padding[0]);
    uint32_t padRight  = static_cast<uint32_t>(padding[1]);
    uint32_t padTop    = static_cast<uint32_t>(padding[2]);
    uint32_t padBottom = static_cast<uint32_t>(padding[3]);
    uint32_t padFront  = static_cast<uint32_t>(padding[4]);
    uint32_t padBack   = static_cast<uint32_t>(padding[5]);

    uint32_t outW = inW + padLeft + padRight;
    uint32_t outH = inH + padTop + padBottom;
    uint32_t outD = inD + padFront + padBack;
    uint32_t totalLength = inN * inC * outD * outH * outW;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_padLeft(padLeft);
    tiling.set_padRight(padRight);
    tiling.set_padTop(padTop);
    tiling.set_padBottom(padBottom);
    tiling.set_padFront(padFront);
    tiling.set_padBack(padBack);
    tiling.set_inN(inN);
    tiling.set_inC(inC);
    tiling.set_inD(inD);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}

    uint32_t padLeft   = static_cast<uint32_t>(padding[0]);
    uint32_t padRight  = static_cast<uint32_t>(padding[1]);
    uint32_t padTop    = static_cast<uint32_t>(padding[2]);
    uint32_t padBottom = static_cast<uint32_t>(padding[3]);
    uint32_t padFront  = static_cast<uint32_t>(padding[4]);

    yShape->SetDim(3, xShape->GetDim(3) + padTop + padBottom);
    yShape->SetDim(4, xShape->GetDim(4) + padLeft + padRight);
    return GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04599Custom {
public:
    __aicore__ inline KernelCa6k04599Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t padLeft, uint32_t padRight,
                                uint32_t padTop, uint32_t padBottom,
                                uint32_t padFront, uint32_t padBack,
                                uint32_t inN, uint32_t inC,
                                uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t outD, uint32_t outH, uint32_t outW) {
        xGm_.SetGlobalBuffer((__gm__ float *)x);
        yGm_.SetGlobalBuffer((__gm__ float *)y);
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        padLeft_ = padLeft;
        padRight_ = padRight;
        padTop_ = padTop;
        padBottom_ = padBottom;
        padFront_ = padFront;
        padBack_ = padBack;
        inN_ = inN;
        inC_ = inC;
        inD_ = inD;
        inH_ = inH;
        inW_ = inW;
        outD_ = outD;
        outH_ = outH;
        outW_ = outW;
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();
        uint32_t stride = blockNum * tileLen_;
        uint32_t start = blockIdx * tileLen_;
        if (start >= totalLen_) return;

        uint32_t outStrideH = outW_;
        uint32_t outStrideD = outH_ * outW_;
        uint32_t outStrideC = outD_ * outStrideD;
        uint32_t outStrideN = inC_ * outStrideC;

        uint32_t inStrideH = inW_;
        uint32_t inStrideD = inH_ * inW_;
        uint32_t inStrideC = inD_ * inStrideD;
        uint32_t inStrideN = inC_ * inStrideC;

        for (uint32_t i = start; i < totalLen_; i += stride) {
            uint32_t curLen = totalLen_ - i;
            if (curLen > tileLen_) curLen = tileLen_;

            for (uint32_t j = 0; j < curLen; j++) {
                uint32_t linearIdx = i + j;
                uint32_t n = linearIdx / outStrideN;
                uint32_t rem1 = linearIdx % outStrideN;
                uint32_t c = rem1 / outStrideC;
                uint32_t rem2 = rem1 % outStrideC;
                uint32_t d = rem2 / outStrideD;
                uint32_t rem3 = rem2 % outStrideD;
                uint32_t h = rem3 / outStrideH;
                uint32_t w = rem3 % outStrideH;

                int64_t in_w = static_cast<int64_t>(w) - static_cast<int64_t>(padLeft_);
                int64_t in_h = static_cast<int64_t>(h) - static_cast<int64_t>(padTop_);
                int64_t in_d = static_cast<int64_t>(d) - static_cast<int64_t>(padFront_);

                float val = 0.0f;
                if (in_w >= 0 && in_w < static_cast<int64_t>(inW_) &&
                    in_h >= 0 && in_h < static_cast<int64_t>(inH_) &&
                    in_d >= 0 && in_d < static_cast<int64_t>(inD_)) {
                    uint32_t u_in_w = static_cast<uint32_t>(in_w);
                    uint32_t u_in_h = static_cast<uint32_t>(in_h);
                    uint32_t u_in_d = static_cast<uint32_t>(in_d);
                    uint32_t inIdx = n * inStrideN +
                                     c * inStrideC +
                                     u_in_d * inStrideD +
                                     u_in_h * inStrideH +
                                     u_in_w;
                    val = xGm_.GetValue(inIdx);
                }
                yGm_.SetValue(linearIdx, val);
            }
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_ = 0;
    uint32_t tileLen_ = 0;
    uint32_t padLeft_ = 0, padRight_ = 0;
    uint32_t padTop_ = 0, padBottom_ = 0;
    uint32_t padFront_ = 0, padBack_ = 0;
    uint32_t inN_ = 0, inC_ = 0;
    uint32_t inD_ = 0, inH_ = 0, inW_ = 0;
    uint32_t outD_ = 0, outH_ = 0, outW_ = 0;
};

extern "C" __global__ __aicore__ void ca6k04599_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelCa6k04599Custom op;
    op.Init(x, y,
            tilingData.totalLength, tilingData.tileLength,
            tilingData.padLeft, tilingData.padRight,
            tilingData.padTop, tilingData.padBottom,
            tilingData.padFront, tilingData.padBack,
            tilingData.inN, tilingData.inC,
            tilingData.inD, tilingData.inH, tilingData.inW,
            tilingData.outD, tilingData.outH, tilingData.outW);
    op.Process();
}
'''
