#include "ca6k_00246_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t MAX_BLOCK = 24u;
static constexpr uint32_t TILE_ELEMS = 4096u;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00246CustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t r = attrs->GetAttrPointer<int64_t>(0)[0];
    if (r <= 0) r = 1;

    const auto& x_shape = context->GetInputShape(0)->GetOriginShape();
    int64_t N = x_shape.GetDim(0);
    int64_t inC = x_shape.GetDim(1);
    int64_t H = x_shape.GetDim(2);
    int64_t W = x_shape.GetDim(3);
    int64_t C = inC / (r * r);
    int64_t Hr = H * r;
    int64_t Wr = W * r;
    int64_t totalOut = N * C * Hr * Wr;

    uint32_t tileOut = TILE_ELEMS;
    if (tileOut > totalOut) tileOut = totalOut;
    uint32_t blockDim = (totalOut + tileOut - 1) / tileOut;
    if (blockDim > MAX_BLOCK) blockDim = MAX_BLOCK;
    if (blockDim < 8) blockDim = 8;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_r(r);
    tiling.set_totalOut(totalOut);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
