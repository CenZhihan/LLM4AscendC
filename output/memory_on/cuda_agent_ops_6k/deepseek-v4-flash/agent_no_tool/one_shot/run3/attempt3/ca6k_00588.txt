namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k00588CustomTilingData tiling;

  // Use GetOriginShape() to get a Shape object with GetDim()
  const gert::StorageShape *xStorage = context->GetInputShape(0);
  const auto &xShape = xStorage->GetOriginShape();
  uint32_t totalN = xShape.GetDim(0);
  uint32_t totalC = xShape.GetDim(1);
  uint32_t totalSpatial = xShape.GetDim(2);

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  const float *eps = attrs->GetAttrPointer<float>(0);

  // Conservative tile sizes
  uint32_t tileSpatial = 1024;
  uint32_t tileBatch = 16;
  uint32_t tileElems = tileBatch * tileSpatial;
  if (tileElems * sizeof(float) * 3 > UB_BUDGET_BYTES) {
    tileSpatial = 512;
    tileBatch = 16;
    tileElems = tileBatch * tileSpatial;
  }
  tileSpatial = AlignDown(tileSpatial, ALIGN_ELEMS);
  if (tileSpatial == 0) tileSpatial = ALIGN_ELEMS;
  if (tileSpatial > totalSpatial) tileSpatial = totalSpatial;

  context->SetBlockDim(1);

  tiling.set_totalN(totalN);
  tiling.set_totalC(totalC);
  tiling.set_totalSpatial(totalSpatial);
  tiling.set_tileBatch(tileBatch);
  tiling.set_tileSpatial(tileSpatial);
  tiling.set_eps(*eps);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00588Custom {
public:
  __aicore__ inline KernelCa6k00588Custom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                              uint32_t totalN, uint32_t totalC, uint32_t totalSpatial,
                              uint32_t tileBatch, uint32_t tileSpatial, float eps) {
    totalN_ = totalN;
    totalC_ = totalC;
    totalSpatial_ = totalSpatial;
    tileBatch_ = tileBatch;
    tileSpatial_ = tileSpatial;
    eps_ = eps;

    xGm_.SetGlobalBuffer((__gm__ float *)x, totalN_ * totalC_ * totalSpatial_);
    gammaGm_.SetGlobalBuffer((__gm__ float *)gamma, totalC_);
    betaGm_.SetGlobalBuffer((__gm__ float *)beta, totalC_);
    yGm_.SetGlobalBuffer((__gm__ float *)y, totalN_ * totalC_ * totalSpatial_);

    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileBatch_ * tileSpatial_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileBatch_ * tileSpatial_ * sizeof(float));
    pipe_.InitBuffer(sumQueue_, 1, totalC_ * sizeof(float));
    pipe_.InitBuffer(sumSqQueue_, 1, totalC_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalN_ == 0 || totalC_ == 0 || totalSpatial_ == 0) return;

    // First pass: compute per-channel sum and sum of squares
    AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
    AscendC::LocalTensor<float> sumSqLocal = sumSqQueue_.AllocTensor<float>();
    AscendC::SetZero(sumLocal, totalC_);
    AscendC::SetZero(sumSqLocal, totalC_);

    for (uint32_t c = 0; c < totalC_; c++) {
      for (uint32_t b = 0; b < totalN_; b += tileBatch_) {
        uint32_t curBatch = tileBatch_;
        if (b + curBatch > totalN_) curBatch = totalN_ - b;
        for (uint32_t s = 0; s < totalSpatial_; s += tileSpatial_) {
          uint32_t curSpatial = tileSpatial_;
          if (s + curSpatial > totalSpatial_) curSpatial = totalSpatial_ - s;

          AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
          uint32_t tileElems = curBatch * curSpatial;
          uint32_t srcOffset = b * totalC_ * totalSpatial_ + c * totalSpatial_ + s;
          AscendC::DataCopy(xTile, xGm_[srcOffset], tileElems);

          for (uint32_t i = 0; i < tileElems; i++) {
            float val = xTile(i);
            sumLocal(c) += val;
            sumSqLocal(c) += val * val;
          }

          inQueueX_.FreeTensor(xTile);
        }
      }
    }

    float nElemsPerChannel = (float)(totalN_ * totalSpatial_);
    AscendC::LocalTensor<float> meanLocal = sumLocal;
    AscendC::LocalTensor<float> varLocal = sumSqLocal;
    for (uint32_t c = 0; c < totalC_; c++) {
      meanLocal(c) = sumLocal(c) / nElemsPerChannel;
      varLocal(c) = sumSqLocal(c) / nElemsPerChannel - meanLocal(c) * meanLocal(c);
    }

    // Second pass: compute output
    for (uint32_t c = 0; c < totalC_; c++) {
      float gammaVal = gammaGm_(c);
      float betaVal = betaGm_(c);
      float normFactor = 1.0f / sqrtf(varLocal(c) + eps_);
      for (uint32_t b = 0; b < totalN_; b += tileBatch_) {
        uint32_t curBatch = tileBatch_;
        if (b + curBatch > totalN_) curBatch = totalN_ - b;
        for (uint32_t s = 0; s < totalSpatial_; s += tileSpatial_) {
          uint32_t curSpatial = tileSpatial_;
          if (s + curSpatial > totalSpatial_) curSpatial = totalSpatial_ - s;

          AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
          uint32_t tileElems = curBatch * curSpatial;
          uint32_t srcOffset = b * totalC_ * totalSpatial_ + c * totalSpatial_ + s;
          AscendC::DataCopy(xTile, xGm_[srcOffset], tileElems);

          AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();
          for (uint32_t i = 0; i < tileElems; i++) {
            float val = xTile(i);
            yTile(i) = gammaVal * (val - meanLocal(c)) * normFactor + betaVal;
          }

          uint32_t dstOffset = srcOffset;
          AscendC::DataCopy(yGm_[dstOffset], yTile, tileElems);

          inQueueX_.FreeTensor(xTile);
          outQueueY_.FreeTensor(yTile);
        }
      }
    }

    sumQueue_.FreeTensor(sumLocal);
    sumSqQueue_.FreeTensor(sumSqLocal);
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
  AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQueue_;
  AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumSqQueue_;

  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> gammaGm_;
  AscendC::GlobalTensor<float> betaGm_;
  AscendC::GlobalTensor<float> yGm_;

  uint32_t totalN_ {0};
  uint32_t totalC_ {0};
  uint32_t totalSpatial_ {0};
  uint32_t tileBatch_ {0};
  uint32_t tileSpatial_ {0};
  float eps_ {1e-5};
};

extern "C" __global__ __aicore__ void ca6k_00588_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                                         GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelCa6k00588Custom op;
  op.Init(x, gamma, beta, y,
          tiling_data.totalN, tiling_data.totalC, tiling_data.totalSpatial,
          tiling_data.tileBatch, tiling_data.tileSpatial, tiling_data.eps);
  op.Process();
}
'''

model_src = '''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00588_custom'

class ModelNew(nn.Module):
    def __init__(self, num_features, eps=1e-5, momentum=0.1, affine=True, track_running_stats=True):
        super().__init__()
        self.num_features = num_features
        self.eps = eps
        self.momentum = momentum
        self.track_running_stats = track_running_stats

        if affine:
            self.gamma = nn.Parameter(torch.ones(num_features))
            self.beta = nn.Parameter(torch.zeros(num_features))
        else:
            self.register_buffer('gamma', torch.ones(num_features))
            self.register_buffer('beta', torch.zeros(num_features))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.gamma, self.beta, self.eps)
'''
