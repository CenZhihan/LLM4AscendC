#include "ca6k_00271_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MAX_BLOCK_DIM = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00271CustomTilingData tiling;
    const gert::Shape &inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t dims = inShape.GetDimNum();
    uint32_t N = inShape.GetDim(0);
    uint32_t C = inShape.GetDim(1);
    uint32_t D = (dims > 2) ? inShape.GetDim(2) : 1;
    uint32_t H = (dims > 3) ? inShape.GetDim(3) : 1;
    uint32_t W = (dims > 4) ? inShape.GetDim(4) : 1;
    uint32_t spatialSize = D * H * W;

    // tileLength: each block loads and sums tiles of spatial data.
    // Two buffers for double buffering: 2 * tileLen * sizeof(float) = 8*tileLen <= UB_BUDGET
    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = maxTile;
    if (spatialSize < tileLen) {
        tileLen = AlignDown(spatialSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = spatialSize;
    }

    uint32_t blockDim = N * C;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_batch(N);
    tiling.set_channels(C);
    tiling.set_spatialSize(spatialSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00271Custom {
public:
    __aicore__ inline KernelCa6k00271Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t channels,
                                uint32_t spatialSize, uint32_t tileLength)
    {
        batch_ = batch;
        channels_ = channels;
        spatialSize_ = spatialSize;
        tileLen_ = tileLength;
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sumQueue_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        if (blockIdx >= batch_ * channels_) return;

        uint32_t n = blockIdx / channels_;
        uint32_t c = blockIdx % channels_;

        // Compute offset into input for this (n,c) slice
        uint32_t sliceOffset = (n * channels_ + c) * spatialSize_;

        // Initialize sum to 0
        float sum = 0.0f;

        // Tile loop over spatial dimension
        uint32_t remaining = spatialSize_;
        uint32_t offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining < tileLen_) ? remaining : tileLen_;

            // Prefetch using double buffering (simplified: single buffer for sum calculation)
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[sliceOffset + offset], curLen);
            inQueueX_.EnQue(xLocal);

            // Wait for data
            xLocal = inQueueX_.DeQue<float>();

            // Accumulate using vector reduce add
            // Use AscendC::ReduceAdd? Instead, manually sum in local tensor and then add to sum.
            // We'll use a simple loop over elements using tensor operations.
            AscendC::LocalTensor<float> tmp = pipe_.template CreateLocalTensor<float>(1);
            AscendC::SetVal(tmp, curLen, 0.0f); // actually we need sum, not set. Use ReduceAdd? Not directly.
            // Better: use AscendC::ReduceAdd?
            // For AscendC, we can use AscendC::ReduceAdd? Not standard. Simulate by:
            // We'll compute sum by repeatedly adding to scalar using pipeline.
            // Alternate approach: load into UB and perform scalar reduction.
            // Since tileLen is small enough, we can do a CPU-like loop? Not allowed.
            // Use AscendC's vector add to reduce? Not directly.
            // The simplest: use AscendC::Sum? Not available.
            // We'll implement a simple scalar loop in the kernel (slow but correct).
            // Actually AscendC supports AscendC::ReduceAdd? The documentation suggests using
            // AscendC::ReduceAdd for summation. We'll assume it exists.
            // For safety, write a manual reduction using a temporary buffer and loop.
            // Given timing, we'll use a for loop over the local tensor elements.
            // This may be inefficient but functional.
            for (uint32_t i = 0; i < curLen; ++i) {
                sum += xLocal.GetValue(i);
            }

            inQueueX_.FreeTensor(xLocal);

            offset += curLen;
            remaining -= curLen;
        }

        // Compute average and write to output
        float avg = sum / static_cast<float>(spatialSize_);
        yGm_.SetValue(sliceOffset / spatialSize_, avg); // output index is blockIdx (n*C+c)
        // Actually output has shape (N, C, 1, 1, 1), index = n*C + c
        yGm_.SetValue(blockIdx, avg);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> sumQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_ {0};
    uint32_t channels_ {0};
    uint32_t spatialSize_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00271_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00271Custom op;
    op.Init(x, y, tiling_data.batch, tiling_data.channels,
            tiling_data.spatialSize, tiling_data.tileLength);
    op.Process();
}
