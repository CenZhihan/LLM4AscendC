#include "ca6k_02661_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t MAX_TILE_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k02661CustomTilingData tiling;

    const auto &xShape = context->GetInputShape(0)->GetOriginShape();
    const auto &wShape = context->GetInputShape(1)->GetOriginShape();

    int32_t batch = static_cast<int32_t>(xShape.GetDim(0));
    int32_t inChannels = static_cast<int32_t>(xShape.GetDim(1));
    int32_t inputLength = static_cast<int32_t>(xShape.GetDim(2));
    int32_t outChannels = static_cast<int32_t>(wShape.GetDim(1));
    int32_t kernelSize = static_cast<int32_t>(wShape.GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t stride = 1, padding = 0, outputPadding = 0, groups = 1;
    if (attrs->GetAttrNum() >= 1) stride = attrs->GetAttrPointer<int32_t>(0)[0];
    if (attrs->GetAttrNum() >= 2) padding = attrs->GetAttrPointer<int32_t>(1)[0];
    if (attrs->GetAttrNum() >= 3) outputPadding = attrs->GetAttrPointer<int32_t>(2)[0];
    if (attrs->GetAttrNum() >= 4) groups = attrs->GetAttrPointer<int32_t>(3)[0];

    int32_t outputLength = (inputLength - 1) * stride - 2 * padding + (kernelSize - 1) + outputPadding + 1;
    if (outputLength < 0) outputLength = 0;

    uint32_t totalLength = static_cast<uint32_t>(batch) * outChannels * outputLength;

    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * (inChannels * kernelSize + 2 * outputLength));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batch(batch);
    tiling.set_inChannels(inChannels);
    tiling.set_outChannels(outChannels);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_outputPadding(outputPadding);
    tiling.set_inputLength(inputLength);
    tiling.set_outputLength(outputLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

class KernelCa6k02661Custom {
public:
    __aicore__ inline KernelCa6k02661Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                const Ca6k02661CustomTilingData &td) {
        xGm_ = (__gm__ float*)x;
        wGm_ = (__gm__ float*)weight;
        bGm_ = (__gm__ float*)bias;
        yGm_ = (__gm__ float*)y;
        batch_ = td.get_batch();
        inCh_ = td.get_inChannels();
        outCh_ = td.get_outChannels();
        kSize_ = td.get_kernelSize();
        stride_ = td.get_stride();
        padding_ = td.get_padding();
        outPad_ = td.get_outputPadding();
        inLen_ = td.get_inputLength();
        outLen_ = td.get_outputLength();
        totalLen_ = td.get_totalLength();
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t strideElems = blockNum;
        uint32_t base = blockIdx;

        while (base < totalLen_) {
            uint32_t b = base / (outCh_ * outLen_);
            uint32_t rem = base % (outCh_ * outLen_);
            uint32_t co = rem / outLen_;
            uint32_t i = rem % outLen_;

            const __gm__ float *xBatch = xGm_ + b * inCh_ * inLen_;
            const __gm__ float *wBase = wGm_;
            const __gm__ float *biasPtr = bGm_;
            __gm__ float *yBatch = yGm_ + b * outCh_ * outLen_;

            float sum = biasPtr[co];
            for (int32_t ci = 0; ci < inCh_; ++ci) {
                for (int32_t k = 0; k < kSize_; ++k) {
                    int32_t numer = static_cast<int32_t>(i) - k + padding_;
                    if (numer >= 0 && numer % stride_ == 0) {
                        int32_t i_in = numer / stride_;
                        if (i_in < inLen_) {
                            sum += wBase[ci * outCh_ * kSize_ + co * kSize_ + k] * xBatch[ci * inLen_ + i_in];
                        }
                    }
                }
            }
            yBatch[co * outLen_ + i] = sum;

            base += strideElems;
        }
    }

private:
    __gm__ float *xGm_;
    __gm__ float *wGm_;
    __gm__ float *bGm_;
    __gm__ float *yGm_;
    int32_t batch_;
    int32_t inCh_;
    int32_t outCh_;
    int32_t kSize_;
    int32_t stride_;
    int32_t padding_;
    int32_t outPad_;
    int32_t inLen_;
    int32_t outLen_;
    uint32_t totalLen_;
};

extern "C" __global__ __aicore__ void ca6k_02661_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                                         GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(td, tiling);
    KernelCa6k02661Custom op;
    op.Init(x, weight, bias, y, td);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_02661_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias):
        super().__init__()
        self.weight = torch.nn.Parameter(torch.randn(in_channels, out_channels, kernel_size))
        if bias:
            self.bias = torch.nn.Parameter(torch.zeros(out_channels))
        else:
            self.register_parameter('bias', None)
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.groups = 1

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias, self.stride, self.padding, self.output_padding, self.groups)
