    Ca6k00393CustomTilingData tiling;
    const gert::StorageShape *logProbShape = context->GetInputShape(0);

    uint32_t numClasses = logProbShape->GetOriginShape().GetDim(1);

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(int64_t);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);

    uint32_t tileLen = (maxTile < batchSize) ? maxTile : batchSize;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00393Custom {
public:
    __aicore__ inline KernelCa6k00393Custom() {}

    __aicore__ inline void Init(GM_ADDR logProb, GM_ADDR target, GM_ADDR loss,
                                uint32_t batchSize, uint32_t numClasses, uint32_t tileLen) {
        batchSize_ = batchSize;
        numClasses_ = numClasses;
        tileLen_ = tileLen;
        logProbGm_.SetGlobalBuffer((__gm__ float*)logProb, batchSize * numClasses);
        targetGm_.SetGlobalBuffer((__gm__ int64_t*)target, batchSize);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);
        pipe_.InitBuffer(inQueueTarget_, BUFFER_NUM, tileLen_ * sizeof(int64_t));
        pipe_.InitBuffer(tmpFloat_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        if (AscendC::GetBlockIdx() != 0) return;

        float sum = 0.0f;
        uint32_t remaining = batchSize_;
        uint32_t offset = 0;

        while (remaining > 0) {
            uint32_t curLen = (remaining < tileLen_) ? remaining : tileLen_;

            auto targetLocal = inQueueTarget_.AllocTensor<int64_t>();
            AscendC::DataCopy(targetLocal, targetGm_[offset], curLen);

            for (uint32_t i = 0; i < curLen; i++) {
                int64_t tidx = targetLocal(i);
                uint32_t globalIdx = (offset + i) * numClasses_ + static_cast<uint32_t>(tidx);

                auto tmp = tmpFloat_.AllocTensor<float>();
                AscendC::DataCopy(tmp, logProbGm_[globalIdx], 1);
                float prob = tmp(0);
                tmpFloat_.FreeTensor(tmp);
                sum -= prob;
            }

            inQueueTarget_.FreeTensor(targetLocal);
            offset += curLen;
            remaining -= curLen;
        }

        float mean = sum / static_cast<float>(batchSize_);
        auto outTmp = tmpFloat_.AllocTensor<float>();
        outTmp(0) = mean;
        AscendC::DataCopy(lossGm_[0], outTmp, 1);
        tmpFloat_.FreeTensor(outTmp);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTarget_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpFloat_;
    AscendC::GlobalTensor<float> logProbGm_;
    AscendC::GlobalTensor<int64_t> targetGm_;
    AscendC::GlobalTensor<float> lossGm_;
    uint32_t batchSize_{0};
    uint32_t numClasses_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void ca6k_00393_custom(GM_ADDR logProb, GM_ADDR target,
                                                        GM_ADDR loss,
                                                        GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelCa6k00393Custom op;
    op.Init(logProb, target, loss,
            tilingData.batchSize, tilingData.numClasses, tilingData.tileLen);
    op.Process();
}
