kernel_src = """
#include "kernel_operator.h"
#include "ca6k00414_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float BICUBIC_A = -0.75f;

class KernelCa6k00414Custom {
public:
    __aicore__ inline KernelCa6k00414Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t H, uint32_t W,
                                uint32_t H_out, uint32_t W_out) {
        N_ = N; C_ = C; H_ = H; W_ = W; H_out_ = H_out; W_out_ = W_out;
        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * H * W);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C * H_out * W_out);
        totalOut_ = N * C * H_out * W_out;
    }

    __aicore__ inline void Process() {
        uint32_t blkIdx = AscendC::GetBlockIdx();
        uint32_t blkNum = AscendC::GetBlockNum();
        for (uint32_t idx = blkIdx; idx < totalOut_; idx += blkNum) {
            uint32_t tmp = idx;
            uint32_t w = tmp % W_out_;
            tmp /= W_out_;
            uint32_t h = tmp % H_out_;
            tmp /= H_out_;
            uint32_t c = tmp % C_;
            uint32_t b = tmp / C_;

            float yF = h / 2.0f;
            float xF = w / 2.0f;
            int iy = (int)yF;
            int ix = (int)xF;
            float du = xF - ix;
            float dv = yF - iy;

            // Compute weights
            float wx[4], wy[4];
            for (int k = 0; k < 4; ++k) {
                float u = (float)(k - 1) - du;
                float v = (float)(k - 1) - dv;
                wx[k] = bicubicWeight(u, BICUBIC_A);
                wy[k] = bicubicWeight(v, BICUBIC_A);
            }

            float sum = 0.0f;
            for (int j = 0; j < 4; ++j) {
                int row = iy + j - 1;
                row = (row < 0) ? 0 : (row >= (int)H_ ? (int)H_ - 1 : row);
                float rowSum = 0.0f;
                for (int k = 0; k < 4; ++k) {
                    int col = ix + k - 1;
                    col = (col < 0) ? 0 : (col >= (int)W_ ? (int)W_ - 1 : col);
                    uint32_t gIdx = (b * C_ + c) * H_ * W_ + row * W_ + col;
                    rowSum += xGm_.GetValue(gIdx) * wx[k];
                }
                sum += rowSum * wy[j];
            }

            yGm_.SetValue(idx, sum);
        }
    }

private:
    __aicore__ inline float bicubicWeight(float x, float a) {
        float absX = x;
        if (absX < 0.0f) absX = -absX;
        if (absX <= 1.0f) {
            return (a + 2.0f) * absX * absX * absX - (a + 3.0f) * absX * absX + 1.0f;
        } else if (absX < 2.0f) {
            return a * absX * absX * absX - 5.0f * a * absX * absX + 8.0f * a * absX - 4.0f * a;
        } else {
            return 0.0f;
        }
    }

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t N_, C_, H_, W_, H_out_, W_out_;
    uint32_t totalOut_;
};

extern "C" __global__ __aicore__ void ca6k_00414_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00414Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C,
            tiling_data.H, tiling_data.W,
            tiling_data.H_out, tiling_data.W_out);
    op.Process();
}
"""

_FN = 'ca6k_00414_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        if len(init_inputs) >= 2:
            self.scale_factor = init_inputs[0]
            self.mode = init_inputs[1]
        else:
            self.scale_factor = 2.0
            self.mode = "bicubic"

    def forward(self, *inputs):
        fn = getattr(custom_ops_lib, _FN)
        x = inputs[0]
        return fn(x, self.scale_factor, self.mode)
