#include "ca6k_04334_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04334CustomTilingData tiling;
    const uint32_t totalLength = static_cast<uint32_t>(
        context->GetOutputShape(0)->GetOriginShape().GetShapeSize());
    uint32_t outChannels = static_cast<uint32_t>(
        context->GetInputShape(1)->GetOriginShape().GetDim(0));
    uint32_t spatialSize = totalLength / outChannels;

    uint32_t maxTile = UB_BUDGET_BYTES / (4 * 2);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }
    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_outChannels(outChannels);
    tiling.set_spatialSize(spatialSize);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04334Custom {
public:
    __aicore__ inline KernelCa6k04334Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t outChannels, uint32_t spatialSize) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        outChannels_ = outChannels;
        spatialSize_ = spatialSize;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outChannels_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueBias_, BUFFER_NUM, outChannels_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;
        // Load bias once (fits in UB)
        AscendC::LocalTensor<float> biasLocal = inQueueBias_.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, biasGm_, outChannels_);
        inQueueBias_.EnQue(biasLocal);

        while (base < totalLen_) {
            uint32_t curLen = (totalLen_ - base > tileLen_) ? tileLen_ : (totalLen_ - base);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], curLen);
            inQueueX_.EnQue(xLocal);

            Compute(curLen);

            AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);

            base += strideElems;
        }

        inQueueBias_.FreeTensor(biasLocal);
    }

private:
    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> biasLocal = inQueueBias_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // For each element, add bias based on channel index
        // Tile length is multiple of (outChannels * spatial elements)
        uint32_t channels = outChannels_;
        uint32_t spatialBlock = len / channels;
        for (uint32_t c = 0; c < channels; c++) {
            float biasVal = biasLocal(c);
            // Use Add with a scalar for the slice
            AscendC::LocalTensor<float> subX = xLocal[c * spatialBlock];
            AscendC::LocalTensor<float> subY = yLocal[c * spatialBlock];
            AscendC::Adds(subY, subX, biasVal, spatialBlock);
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueBias_.EnQue(biasLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBias_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t outChannels_ {0};
    uint32_t spatialSize_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04334_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04334Custom op;
    op.Init(x, bias, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.outChannels, tiling_data.spatialSize);
    op.Process();
}

import torch

class ModelNew(torch.nn.Module):
