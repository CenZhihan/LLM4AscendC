static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00212CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(1);

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00212Custom {
public:
    __aicore__ inline KernelCa6k00212Custom() {}

    __aicore__ inline void Init(GM_ADDR input, GM_ADDR target, GM_ADDR loss,
                                uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inputGm_.SetGlobalBuffer((__gm__ float*)input, totalLen_);
        targetGm_.SetGlobalBuffer((__gm__ float*)target, totalLen_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inQueueInput_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueTarget_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueLoss_, 1, sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        float sum = 0.0f;
        uint32_t offset = 0;
        while (offset < totalLen_) {
            uint32_t curLen = (offset + tileLen_ <= totalLen_) ? tileLen_ : (totalLen_ - offset);
            // Copy input and target tiles
            AscendC::LocalTensor<float> inLocal = inQueueInput_.AllocTensor<float>();
            AscendC::DataCopy(inLocal, inputGm_[offset], curLen);
            inQueueInput_.EnQue(inLocal);

            AscendC::LocalTensor<float> tarLocal = inQueueTarget_.AllocTensor<float>();
            AscendC::DataCopy(tarLocal, targetGm_[offset], curLen);
            inQueueTarget_.EnQue(tarLocal);

            // Compute absolute difference
            AscendC::LocalTensor<float> subLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(subLocal, inLocal, tarLocal, curLen);
            AscendC::Abs(subLocal, subLocal, curLen);

            // Accumulate sum using scalar loop over the tile
            float tileSum = 0.0f;
            for (uint32_t i = 0; i < curLen; i++) {
                tileSum += subLocal(i);
            }
            sum += tileSum;

            tmpQueue_.FreeTensor(subLocal);
            inQueueInput_.FreeTensor(inLocal);
            inQueueTarget_.FreeTensor(tarLocal);

            offset += curLen;
        }

        // Write scalar loss = sum / totalLen_
        float lossVal = sum / (float)totalLen_;
        AscendC::LocalTensor<float> lossLocal = outQueueLoss_.AllocTensor<float>();
        lossLocal(0) = lossVal;
        outQueueLoss_.EnQue(lossLocal);
        AscendC::LocalTensor<float> lossDeq = outQueueLoss_.DeQue<float>();
        AscendC::DataCopy(lossGm_, lossDeq, 1);
        outQueueLoss_.FreeTensor(lossDeq);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueInput_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTarget_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueLoss_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> inputGm_;
    AscendC::GlobalTensor<float> targetGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00212_custom(GM_ADDR input, GM_ADDR target, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00212Custom op;
    op.Init(input, target, loss, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
"""
