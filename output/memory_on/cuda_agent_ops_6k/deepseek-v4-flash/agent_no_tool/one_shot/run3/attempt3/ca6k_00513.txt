#include "ca6k00513_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00513CustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batch = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t channels = static_cast<uint32_t>(shape.GetDim(1));
    uint32_t width = static_cast<uint32_t>(shape.GetDim(2));
    uint32_t totalSequences = batch * channels;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];
    uint32_t outputLen = (width + 2 * padding - kernelSize) / stride + 1;

    uint32_t maxTile = UB_BUDGET_BYTES / (kernelSize * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalSequences, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalSequences(totalSequences);
    tiling.set_width(width);
    tiling.set_kernelSize(static_cast<uint32_t>(kernelSize));
    tiling.set_stride(static_cast<uint32_t>(stride));
    tiling.set_padding(static_cast<uint32_t>(padding));
    tiling.set_outputLen(outputLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00513Custom {
public:
    __aicore__ inline KernelCa6k00513Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalSequences, uint32_t width,
                                uint32_t kernelSize, uint32_t stride,
                                uint32_t padding, uint32_t outputLen,
                                uint32_t tileLen) {
        totalSequences_ = totalSequences;
        width_ = width;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        outputLen_ = outputLen;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalSequences * width);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalSequences * outputLen);

        pipe_.InitBuffer(tmpQueue_, 1, kernelSize_ * sizeof(float));
        pipe_.InitBuffer(scalarBufX_, 1, sizeof(float));
        pipe_.InitBuffer(scalarBufY_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t perBlock = (totalSequences_ + blockNum - 1) / blockNum;
        uint32_t startSeq = blockIdx * perBlock;
        uint32_t endSeq = startSeq + perBlock;
        if (startSeq >= totalSequences_) return;
        if (endSeq > totalSequences_) endSeq = totalSequences_;

        for (uint32_t seq = startSeq; seq < endSeq; ++seq) {
            ProcessSequence(seq);
        }
    }

private:
    __aicore__ inline void ProcessSequence(uint32_t seq) {
        AscendC::LocalTensor<float> window = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> oneVal = scalarBufX_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = scalarBufY_.AllocTensor<float>();

        for (uint32_t outIdx = 0; outIdx < outputLen_; ++outIdx) {
            int32_t start = static_cast<int32_t>(outIdx * stride_) - static_cast<int32_t>(padding_);
            for (uint32_t k = 0; k < kernelSize_; ++k) {
                int32_t idx = start + static_cast<int32_t>(k);
                if (idx >= 0 && idx < static_cast<int32_t>(width_)) {
                    AscendC::DataCopy(oneVal, xGm_[seq * width_ + idx], 1);
                    window(k) = oneVal(0);
                } else {
                    window(k) = 0.0f;
                }
            }
            float maxVal;
            AscendC::ReduceMax(maxVal, window, kernelSize_, AscendC::MASK_NZ);
            maxLocal(0) = maxVal;
            AscendC::DataCopy(yGm_[seq * outputLen_ + outIdx], maxLocal, 1);
        }

        tmpQueue_.FreeTensor(window);
        scalarBufX_.FreeTensor(oneVal);
        scalarBufY_.FreeTensor(maxLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBufX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBufY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalSequences_ {0};
    uint32_t width_ {0};
    uint32_t kernelSize_ {0};
    uint32_t stride_ {0};
    uint32_t padding_ {0};
    uint32_t outputLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k00513_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00513Custom op;
    op.Init(x, y, tiling_data.totalSequences, tiling_data.width,
            tiling_data.kernelSize, tiling_data.stride,
            tiling_data.padding, tiling_data.outputLen,
            tiling_data.tileLen);
    op.Process();
}
