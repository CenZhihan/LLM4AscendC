#include "ca6k_00271_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MAX_BLOCK_DIM = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00271CustomTilingData tiling;
    const gert::Shape *inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalInput = 1;
    uint32_t dims = inShape->GetDimNum();
    for (uint32_t i = 0; i < dims; ++i) {
        totalInput *= inShape->GetDim(i);
    }
    uint32_t B = inShape->GetDim(0);
    uint32_t C = inShape->GetDim(1);
    uint32_t outputLength = B * C;
    uint32_t spatialSize = totalInput / outputLength;

    // tileLength from UB: double buffering: 2 * tileLen * sizeof(float) = 8*tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = maxTile;
    if (spatialSize < tileLen) {
        tileLen = AlignDown(spatialSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = spatialSize;
    }

    uint32_t blockDim = outputLength;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;
    if (blockDim < 8) blockDim = 8;

    context->SetBlockDim(blockDim);

    tiling.set_totalInputLength(totalInput);
    tiling.set_outputLength(outputLength);
    tiling.set_spatialSize(spatialSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *inShape = context->GetInputShape(0);
    gert::Shape *outShape = context->GetOutputShape(0);
    uint32_t dims = inShape->GetDimNum();
    outShape->SetDimNum(dims);
    for (uint32_t i = 0; i < dims; ++i) {
        if (i < 2) {
            outShape->SetDim(i, inShape->GetDim(i));
        } else {
            outShape->SetDim(i, 1);
        }
    }
    return GRAPH_SUCCESS;
}

    const ge::DataType in_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, in_dtype);

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00271Custom {
public:
    __aicore__ inline KernelCa6k00271Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputLength,
                                uint32_t outputLength,
                                uint32_t spatialSize,
                                uint32_t tileLength) {
        totalInputLen_ = totalInputLength;
        outputLen_ = outputLength;
        spatialSize_ = spatialSize;
        tileLen_ = tileLength;
        invSpatial_ = 1.0f / (float)spatialSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (spatialSize_ == 0 || outputLen_ == 0) return;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t idx = blockIdx; idx < outputLen_; idx += blockNum) {
            uint32_t base = idx * spatialSize_;
            float sum = 0.0f;

            uint32_t remaining = spatialSize_;
            uint32_t offset = 0;

            // Prefetch first tile
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            CopyIn(base + offset, curLen);
            offset += curLen;
            remaining -= curLen;

            while (true) {
                // Prefetch next tile if any
                uint32_t nextLen = 0;
                if (remaining > 0) {
                    nextLen = (remaining > tileLen_) ? tileLen_ : remaining;
                    CopyIn(base + offset, nextLen);
                    offset += nextLen;
                    remaining -= nextLen;
                }

                // Compute current tile sum
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                sum += AscendC::VectorReduceAdd<float>(xLocal, curLen);
                inQueueX_.FreeTensor(xLocal);

                if (nextLen == 0) break;
                curLen = nextLen;
            }

            yGm_[idx] = sum * invSpatial_;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalInputLen_;
    uint32_t outputLen_;
    uint32_t spatialSize_;
    uint32_t tileLen_;
    float invSpatial_;
};

extern "C" __global__ __aicore__ void ca6k_00271_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00271Custom op;
    op.Init(x, y,
            tiling_data.totalInputLength,
            tiling_data.outputLength,
            tiling_data.spatialSize,
            tiling_data.tileLength);
    op.Process();
}
"""
