#include "ca6k_02887_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    auto *inputShape = context->GetInputShape(0);
    // Input shape: (batch, seq, in_features) -> totalM = batch*seq, totalK = in_features
    const uint32_t totalM = static_cast<uint32_t>(inputShape->GetOriginShape().GetShapeSize() / inputShape->GetOriginShape().GetDim(2));
    // weight shape: (out_features, in_features)
    auto *weightShape = context->GetInputShape(1);
    const uint32_t totalN = static_cast<uint32_t>(weightShape->GetOriginShape().GetDim(0));
    const uint32_t totalK = static_cast<uint32_t>(weightShape->GetOriginShape().GetDim(1));

    // UB budget: need 3 tiles (A tile, B tile, C tile) each of size tileM*tileK, tileN*tileK, tileM*tileN.
    // Approx: tileM*tileK + tileN*tileK + tileM*tileN floats => bytes*4.
    // We'll compute tileM and tileK from UB budget.
    // Set tileN = totalN (fixed)
    uint32_t tileN = totalN;
    uint32_t maxTileK = UB_BUDGET_BYTES / (4 * (tileN + 2 * 64)); // rough estimate
    uint32_t tileK = 64; // small fixed tileK
    if (tileK > totalK) tileK = totalK;
    tileK = AlignDown(tileK, 16); // alignment for vector ops

    uint32_t tileM = (UB_BUDGET_BYTES / (4 * (tileK + tileN + tileN*tileK/tileK?))) ? ... // simplified: use fixed tileM = 128
    tileM = 128;
    if (tileM > totalM) tileM = totalM;
    tileM = AlignDown(tileM, 16);
    if (tileM == 0) tileM = totalM;

    uint32_t blockDim = CeilDiv(totalM, tileM);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    Ca6k02887CustomTilingData tiling;
    tiling.totalM = totalM;
    tiling.totalN = totalN;
    tiling.totalK = totalK;
    tiling.tileM = tileM;
    tiling.tileK = tileK;
    tiling.blockDim = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    gert::StorageShape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape; // same shape as x? Actually output has same batch and seq but out_features dim.
    // Need to set shape from weight's out_features. We'll set output shape as (batch, seq, totalN)
    // We can derive from input shape and weight shape.
    // As a quick fix we just copy input shape and adjust last dim.
    // But harness expects exact shape. We'll infer by combining input and weight shapes.
    // For safety, we get input dims and set last dim to weight's dim0.
    auto x_origin = x_shape->GetOriginShape();

    y_shape->SetDim(1, seq);

    return GRAPH_SUCCESS;
}

kernel_src = """
#include "kernel_operator.h"
using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k02887Custom {
public:
    __aicore__ inline KernelCa6k02887Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias1, GM_ADDR bias2, GM_ADDR y,
                                uint32_t totalM, uint32_t totalN, uint32_t totalK,
                                uint32_t tileM, uint32_t tileK, uint32_t blockDim)
    {
        totalM_ = totalM;
        totalN_ = totalN;
        totalK_ = totalK;
        tileM_ = tileM;
        tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalM * totalK);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, totalN * totalK);
        b1Gm_.SetGlobalBuffer((__gm__ float*)bias1, totalN);
        b2Gm_.SetGlobalBuffer((__gm__ float*)bias2, totalN);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalM * totalN);
    }

    __aicore__ inline void Process()
    {
        if (totalM_ == 0 || totalN_ == 0 || totalK_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        // Each block processes a contiguous chunk of rows
        uint32_t rowsPerBlock = tileM_; // each block processes one tile
        uint32_t startRow = blockIdx * rowsPerBlock;
        if (startRow >= totalM_) return;
        uint32_t curRows = rowsPerBlock;
        if (startRow + curRows > totalM_) curRows = totalM_ - startRow;

        // Local buffers for one tile
        TPipe pipe;
        TQue<TPosition::VECIN, BUFFER_NUM> queA;
        TQue<TPosition::VECIN, BUFFER_NUM> queB;
        TQue<TPosition::VECOUT, BUFFER_NUM> queC;
        TQue<TPosition::VECCALC, 1> tmpQue;
        pipe.InitBuffer(queA, 1, tileM_ * totalK_ * sizeof(float)); // full tile of A rows
        pipe.InitBuffer(queB, 1, totalN_ * totalK_ * sizeof(float)); // full weight (could tile K but we load all)
        pipe.InitBuffer(queC, 1, tileM_ * totalN_ * sizeof(float));
        pipe.InitBuffer(tmpQue, 1, tileM_ * totalK_ * sizeof(float)); // temporary for dot product

        // Load entire tile of A (x)
        LocalTensor<float> aLocal = queA.AllocTensor<float>();
        DataCopy(aLocal, xGm_[startRow * totalK_], curRows * totalK_);
        queA.EnQue(aLocal);

        // Load entire weight matrix (might be large but within UB for typical sizes)
        LocalTensor<float> wLocal = queB.AllocTensor<float>();
        DataCopy(wLocal, wGm_[0], totalN_ * totalK_);
        queB.EnQue(wLocal);

        // Compute matmul: C(i,j) = sum_k A(i,k) * W(j,k)
        // We'll implement a simple loop over j and k for each row i using vector operations.
        // Better: use Mmad: but for simplicity we use nested loops.
        // Since tile sizes small, we can do double loop with vector multiply and reduce.
        // Actually we can use Matmul class but we'll keep manual.

        aLocal = queA.DeQue<float>();
        wLocal = queB.DeQue<float>();
        LocalTensor<float> cLocal = queC.AllocTensor<float>();
        LocalTensor<float> tLocal = tmpQue.AllocTensor<float>();

        // For each row in tile
        for (uint32_t i = 0; i < curRows; i++) {
            // A row start index in aLocal: i * totalK_
            // For each output column j
            for (uint32_t j = 0; j < totalN_; j++) {
                // Compute dot product of A[i, :] and W[j, :]
                // Use vector multiply and sum
                // We'll do elementwise multiply into tLocal and then reduce using a loop
                // Better: use AscendC::DotProduct? Not available; use VecMul + ReduceAdd?
                // We'll use a manual loop over K tiles
                float sum = 0.0f;
                for (uint32_t k = 0; k < totalK_; k++) {
                    sum += aLocal[i * totalK_ + k] * wLocal[j * totalK_ + k];
                }
                cLocal[i * totalN_ + j] = sum;
            }
        }

        // Add biases
        for (uint32_t i = 0; i < curRows; i++) {
            for (uint32_t j = 0; j < totalN_; j++) {
                cLocal[i * totalN_ + j] += b1Gm_.GetValue(j) + b2Gm_.GetValue(j);
            }
        }

        // Copy out
        DataCopy(yGm_[startRow * totalN_], cLocal, curRows * totalN_);

        queC.FreeTensor(cLocal);
        tmpQue.FreeTensor(tLocal);
        queB.FreeTensor(wLocal);
        queA.FreeTensor(aLocal);
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> b1Gm_;
    GlobalTensor<float> b2Gm_;
    GlobalTensor<float> yGm_;

    uint32_t totalM_ = 0;
    uint32_t totalN_ = 0;
    uint32_t totalK_ = 0;
    uint32_t tileM_ = 0;
    uint32_t tileK_ = 0;
};

extern "C" __global__ __aicore__ void ca6k_02887_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias1, GM_ADDR bias2, GM_ADDR y,
                                                         GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02887Custom op;
    op.Init(x, weight, bias1, bias2, y,
            tiling_data.totalM, tiling_data.totalN, tiling_data.totalK,
            tiling_data.tileM, tiling_data.tileK, tiling_data.blockDim);
    op.Process();
}

model_src = """
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_02887_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features):
        super(ModelNew, self).__init__()
        self.weight = torch.nn.Parameter(torch.randn(out_features, in_features))
        self.bias1 = torch.nn.Parameter(torch.randn(out_features))
        self.bias2 = torch.nn.Parameter(torch.randn(out_features))

    def forward(self, x):
        # Fused op: y = x @ weight.T + bias1 + bias2
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias1, self.bias2)
