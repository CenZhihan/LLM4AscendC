class KernelCa6k00391Cumsum {
public:
    __aicore__ inline void Init(GM_ADDR mask, GM_ADDR cumsum, uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        maskGm_.SetGlobalBuffer((__gm__ int8_t*)mask, totalLen_);
        cumsumGm_.SetGlobalBuffer((__gm__ int32_t*)cumsum, totalLen_);
        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(int8_t));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(int32_t));
    }
    __aicore__ inline void Process() {
        if (totalLen_==0 || tileLen_==0) return;
        uint32_t offset = 0;
        int32_t runningCount = 0;
        for (; offset < totalLen_; offset += tileLen_) {
            uint32_t curLen = totalLen_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;
            AscendC::LocalTensor<int8_t> maskLocal = inQueue_.AllocTensor<int8_t>();
            AscendC::DataCopy(maskLocal, maskGm_[offset], curLen);
            AscendC::LocalTensor<int32_t> cumsumLocal = outQueue_.AllocTensor<int32_t>();
            // compute prefix sum
            for (uint32_t i = 0; i < curLen; i++) {
                int8_t m = maskLocal.GetValue(i);
                cumsumLocal.SetValue(i, runningCount);
                if (m != 0) runningCount++;
            }
            AscendC::DataCopy(cumsumGm_[offset], cumsumLocal, curLen);
            outQueue_.FreeTensor(cumsumLocal);
            inQueue_.FreeTensor(maskLocal);
        }
    }
private:
    uint32_t totalLen_;
    uint32_t tileLen_;
    AscendC::GlobalTensor<int8_t> maskGm_;
    AscendC::GlobalTensor<int32_t> cumsumGm_;
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue_;
};

class KernelCa6k00391Scatter {
public:
    __aicore__ inline void Init(GM_ADDR input, GM_ADDR mask, GM_ADDR cumsum, GM_ADDR scatter, GM_ADDR output,
                                uint32_t totalLength, uint32_t tileLength, uint32_t scatterLen) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        scatterLen_ = scatterLen;
        inputGm_.SetGlobalBuffer((__gm__ float*)input, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ int8_t*)mask, totalLen_);
        cumsumGm_.SetGlobalBuffer((__gm__ int32_t*)cumsum, totalLen_);
        scatterGm_.SetGlobalBuffer((__gm__ float*)scatter, scatterLen_);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalLen_);
        pipe_.InitBuffer(inInputQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inMaskQueue_, BUFFER_NUM, tileLen_ * sizeof(int8_t));
        pipe_.InitBuffer(inCumsumQueue_, BUFFER_NUM, tileLen_ * sizeof(int32_t));
        pipe_.InitBuffer(outOutputQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }
    __aicore__ inline void Process() {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;
        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            // prefetch next? we'll just process sequentially per step
            CopyIn(base, curLen);
            Compute(curLen);
            CopyOut(base, curLen);
            base += strideElems;
            if (base >= totalLen_) break;
        }
    }
private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> inputLocal = inInputQueue_.AllocTensor<float>();
        AscendC::LocalTensor<int8_t> maskLocal = inMaskQueue_.AllocTensor<int8_t>();

        inInputQueue_.EnQue(inputLocal);
        inMaskQueue_.EnQue(maskLocal);
        inCumsumQueue_.EnQue(cumsumLocal);
    }
