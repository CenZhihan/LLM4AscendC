    if (spatialSize_ == 0 || outputLength_ == 0) return;
    uint32_t blockNum = GetBlockNum();
    uint32_t blockIdx = GetBlockIdx();
    float invSpatial = 1.0f / (float)spatialSize_;
    for (uint32_t idx = blockIdx; idx < outputLength_; idx += blockNum) {
        uint32_t base = idx * spatialSize_;
        float sum = 0.0f;
        uint32_t remaining = spatialSize_;
        uint32_t offset = 0;
        // Prefetch first tile
        uint32_t curLen = min(tileLen_, remaining);
        CopyIn(base + offset, curLen);
        offset += curLen;
