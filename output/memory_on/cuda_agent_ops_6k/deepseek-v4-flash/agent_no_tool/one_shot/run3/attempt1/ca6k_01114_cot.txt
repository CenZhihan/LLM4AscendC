class KernelTriuCustom {
public:
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLen, uint32_t rows, uint32_t cols, uint32_t tileLen, int32_t diagonal) {
        totalLen_ = totalLen;
        rows_ = rows;
        cols_ = cols;
        tileLen_ = tileLen;
        diagonal_ = diagonal;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = (base + tileLen_ > totalLen_) ? totalLen_ - base : tileLen_;
        CopyIn(base, curLen);
        while (true) {
            uint32_t nextBase = base + strideElems;
            uint32_t nextLen = (nextBase < totalLen_) ? ((nextBase + tileLen_ > totalLen_) ? totalLen_ - nextBase : tileLen_) : 0;
            if (nextLen > 0) CopyIn(nextBase, nextLen);
            Compute(curLen);
            CopyOut(base, curLen);
            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
