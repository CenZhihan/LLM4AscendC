#include "ca6k_04668_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04668CustomTilingData tiling;

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    auto originShape = inputShape->GetOriginShape();
    uint32_t inputN = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t inputC = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t inputD = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t inputH = static_cast<uint32_t>(originShape.GetDim(3));
    uint32_t inputW = static_cast<uint32_t>(originShape.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *outputSize = attrs->GetAttrPointer<int64_t>(0);
    uint32_t outputD = static_cast<uint32_t>(outputSize[0]);
    uint32_t outputH = static_cast<uint32_t>(outputSize[1]);
    uint32_t outputW = static_cast<uint32_t>(outputSize[2]);

    uint32_t totalOutputElems = inputN * inputC * outputD * outputH * outputW;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = (totalOutputElems < maxTile) ? totalOutputElems : maxTile;
    if (tileLen == 0) tileLen = 1;

    uint32_t numTiles = CeilDiv(totalOutputElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.totalOutputElems = totalOutputElems;
    tiling.tileLen = tileLen;
    tiling.blockDim = blockDim;
    tiling.inputN = inputN;
    tiling.inputC = inputC;
    tiling.inputD = inputD;
    tiling.inputH = inputH;
    tiling.inputW = inputW;
    tiling.outputD = outputD;
    tiling.outputH = outputH;
    tiling.outputW = outputW;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04668Custom {
public:
    __aicore__ inline KernelCa6k04668Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElems, uint32_t tileLen,
                                uint32_t inputN, uint32_t inputC,
                                uint32_t inputD, uint32_t inputH, uint32_t inputW,
                                uint32_t outputD, uint32_t outputH, uint32_t outputW)
    {
        totalOutputElems_ = totalOutputElems;
        tileLen_ = tileLen;
        inputN_ = inputN;
        inputC_ = inputC;
        inputD_ = inputD;
        inputH_ = inputH;
        inputW_ = inputW;
        outputD_ = outputD;
        outputH_ = outputH;
        outputW_ = outputW;

        // compute strides
        inputStrideD_ = inputH * inputW;
        inputStrideH_ = inputW;
        inputStrideW_ = 1;
        inputStrideC_ = inputD * inputStrideD_;
        inputStrideN_ = inputC * inputStrideC_;

        outputStrideD_ = outputH * outputW;
        outputStrideH_ = outputW;
        outputStrideW_ = 1;
        outputStrideC_ = outputD * outputStrideD_;
        outputStrideN_ = outputC_ * outputStrideC_;
        // outputC_ is same as inputC_
        outputC_ = inputC;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutputElems_) return;

        while (true) {
            uint32_t curLen = totalOutputElems_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            // allocate output local tensor
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t linearIdx = base + i;
                // compute batch and channel
                uint32_t tmp = linearIdx;
                uint32_t ow = tmp % outputW_;
                tmp /= outputW_;
                uint32_t oh = tmp % outputH_;
                tmp /= outputH_;
                uint32_t od = tmp % outputD_;
                tmp /= outputD_;
                uint32_t oc = tmp % outputC_;
                uint32_t n = tmp / outputC_;

                // compute pooling region in input
                float startD = od * inputD_ / outputD_;
                float endD = (od + 1) * inputD_ / outputD_;
                uint32_t sD = static_cast<uint32_t>(startD);
                uint32_t eD = static_cast<uint32_t>(endD);
                if (static_cast<float>(eD) < endD) eD++;

                float startH = oh * inputH_ / outputH_;
                float endH = (oh + 1) * inputH_ / outputH_;
                uint32_t sH = static_cast<uint32_t>(startH);
                uint32_t eH = static_cast<uint32_t>(endH);
                if (static_cast<float>(eH) < endH) eH++;

                float startW = ow * inputW_ / outputW_;
                float endW = (ow + 1) * inputW_ / outputW_;
                uint32_t sW = static_cast<uint32_t>(startW);
                uint32_t eW = static_cast<uint32_t>(endW);
                if (static_cast<float>(eW) < endW) eW++;

                float maxVal = -std::numeric_limits<float>::infinity();
                uint32_t inputBase = n * inputStrideN_ + oc * inputStrideC_;

                for (uint32_t d = sD; d < eD; d++) {
                    uint32_t dOffset = inputBase + d * inputStrideD_;
                    for (uint32_t h = sH; h < eH; h++) {
                        uint32_t hOffset = dOffset + h * inputStrideH_;
                        for (uint32_t w = sW; w < eW; w++) {
                            float val = xGm_[hOffset + w];
                            if (val > maxVal) maxVal = val;
                        }
                    }
                }
                yLocal.SetValue(i, maxVal);
            }

            // copy output to global
            AscendC::DataCopy(yGm_[base], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);

            base += stride;
            if (base >= totalOutputElems_) break;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutputElems_ {0};
    uint32_t tileLen_ {0};
    uint32_t inputN_ {0};
    uint32_t inputC_ {0};
    uint32_t inputD_ {0};
    uint32_t inputH_ {0};
    uint32_t inputW_ {0};
    uint32_t outputD_ {0};
    uint32_t outputH_ {0};
    uint32_t outputW_ {0};
    uint32_t outputC_ {0};

    uint32_t inputStrideN_, inputStrideC_, inputStrideD_, inputStrideH_, inputStrideW_;
    uint32_t outputStrideN_, outputStrideC_, outputStrideD_, outputStrideH_, outputStrideW_;
};

extern "C" __global__ __aicore__ void ca6k_04668_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04668Custom op;
    op.Init(x, y,
            tiling_data.totalOutputElems, tiling_data.tileLen,
            tiling_data.inputN, tiling_data.inputC,
            tiling_data.inputD, tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputD, tiling_data.outputH, tiling_data.outputW);
    op.Process();
}
'''
