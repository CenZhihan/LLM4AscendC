namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00236CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const gert::Shape *shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t dims = static_cast<uint32_t>(shape->GetDimNum());
    // Assume 3D: batch, height, width
    uint32_t batch = static_cast<uint32_t>(shape->GetDim(0));
    uint32_t height = static_cast<uint32_t>(shape->GetDim(1));
    uint32_t width = static_cast<uint32_t>(shape->GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dimAttr;

    // Number of independent softmin rows: batch * width (reduce over height)
    uint32_t totalRows = batch * width;

    // Determine tileRows from UB budget.
    // Each row needs space for height floats (input, max, sum, output)
    // Rough estimate: 4 * height * 4 bytes = 16*height bytes per row.
    // UB budget ~ 192KB.
    constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    uint32_t bytesPerRow = 4 * height * sizeof(float); // conservative
    uint32_t maxRows = UB_BUDGET_BYTES / bytesPerRow;
    if (maxRows < 1) maxRows = 1;
    // Align to 8
    constexpr uint32_t ALIGN = 8;
    uint32_t tileRows = (maxRows / ALIGN) * ALIGN;
    if (tileRows == 0) tileRows = ALIGN;
    if (tileRows > totalRows) tileRows = totalRows;

    uint32_t numTiles = (totalRows + tileRows - 1) / tileRows;
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 32) blockDim = 32; // typical cores

    context->SetBlockDim(blockDim);

    tiling.totalRows = totalRows;
    tiling.tileRows = tileRows;
    tiling.height = height;
    tiling.width = width;
    tiling.dim = dim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00236Custom {
public:
    __aicore__ inline KernelCa6k00236Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t tileRows,
                                uint32_t height, uint32_t width, int32_t dim)
    {
        totalRows_ = totalRows;
        tileRows_ = tileRows;
        height_ = height;
        width_ = width;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // We need space for one row of input, one row for max, one row for exp sum, one row for output
        // Use separate local tensors: xRow, maxRow, expSumRow, yRow
        // Each of size height_ floats.
        pipe_.InitBuffer(xBuf_, BUFFER_NUM, height_ * sizeof(float));
        pipe_.InitBuffer(maxBuf_, BUFFER_NUM, height_ * sizeof(float));
        pipe_.InitBuffer(expSumBuf_, 1, sizeof(float));
        // We can reuse tmpBuf for intermediate exp and log? Use separate tensors.
        // For simplicity, allocate a local tensor for computed row and directly write to yBuf.
        pipe_.InitBuffer(yBuf_, BUFFER_NUM, height_ * sizeof(float));
        pipe_.InitBuffer(negXBuf_, 1, height_ * sizeof(float)); // -x row
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || tileRows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each core processes a contiguous chunk of rows (grid-stride over tiles)
        uint32_t rowStart = blockIdx * tileRows_;
        if (rowStart >= totalRows_) return;

        uint32_t rowEnd = rowStart + tileRows_;
        if (rowEnd > totalRows_) rowEnd = totalRows_;

        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            // Compute (batch, width) index from row index r
            uint32_t outerIdx = r / width_; // batch index
            uint32_t innerIdx = r % width_; // width index

            // Load the height elements for this row (reduction dimension)
            // They are at positions: outerIdx * height_ * width_ + h * width_ + innerIdx
            // Use DataCopy with strides? Not directly. We'll loop using scalar load.
            // For performance, we can use a single local tensor and load each element.
            AscendC::LocalTensor<float> xLocal = xBuf_.AllocTensor<float>();
            AscendC::LocalTensor<float> negXLocal = negXBuf_.AllocTensor<float>();
            AscendC::LocalTensor<float> maxLocal = maxBuf_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = yBuf_.AllocTensor<float>();
            AscendC::LocalTensor<float> expSumLocal = expSumBuf_.AllocTensor<float>();

            // 1. Load x elements and compute -x
            uint32_t baseOffset = outerIdx * height_ * width_ + innerIdx;
            for (uint32_t h = 0; h < height_; ++h) {
                xLocal.SetValue(h, xGm_.GetValue(baseOffset + h * width_));
            }
            // Compute -x
            AscendC::Muls(negXLocal, xLocal, -1.0f, height_);

            // 2. Find max of negXLocal
            AscendC::ReduceMax(maxLocal, negXLocal, 1.0f, height_);
            float maxVal = maxLocal.GetValue(0);

            // 3. Compute exp(negXLocal - maxVal)
            AscendC::Subs(xLocal, negXLocal, maxVal, height_);
            AscendC::Exp(xLocal, xLocal, height_);

            // 4. Compute sum of exps
            AscendC::ReduceAdd(expSumLocal, xLocal, 1.0f, height_);
            float sumVal = expSumLocal.GetValue(0);

            // 5. Divide each element by sum to get softmin
            AscendC::Muls(yLocal, xLocal, 1.0f / sumVal, height_);

            // 6. Write back to global memory
            for (uint32_t h = 0; h < height_; ++h) {
                yGm_.SetValue(baseOffset + h * width_, yLocal.GetValue(h));
            }

            xBuf_.FreeTensor(xLocal);
            negXBuf_.FreeTensor(negXLocal);
            maxBuf_.FreeTensor(maxLocal);
            yBuf_.FreeTensor(yLocal);
            expSumBuf_.FreeTensor(expSumLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> maxBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> expSumBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> negXBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t tileRows_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00236_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00236Custom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.tileRows,
            tiling_data.height, tiling_data.width, tiling_data.dim);
    op.Process();
}
