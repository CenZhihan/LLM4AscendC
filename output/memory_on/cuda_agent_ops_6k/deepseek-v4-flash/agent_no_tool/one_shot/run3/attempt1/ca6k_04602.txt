#include "ca6k_04602_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per core

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04602CustomTilingData tiling;

    const gert::StorageShape *xShape = context->GetInputShape(0);
    const auto &xStorageShape = xShape->GetStorageShape();
    uint32_t totalN = static_cast<uint32_t>(xStorageShape.GetDim(0));
    uint32_t totalC = static_cast<uint32_t>(xStorageShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(xStorageShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(xStorageShape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = *attrs->GetAttrPointer<int32_t>(0);
    int32_t strideVal = *attrs->GetAttrPointer<int32_t>(1);
    int32_t paddingVal = *attrs->GetAttrPointer<int32_t>(2);

    // Compute tileH from UB budget:
    // Input tile: (tileH+2)*W*4 bytes (extra rows for neighbourhood)
    // Output tile: tileH*W*4 bytes
    // Double buffered: need 2 copies each => total = 2*(tileH+2)*W*4 + 2*tileH*W*4 = 8*W*(2*tileH+2)
    // Solve for tileH max <= UB_BUDGET_BYTES * 0.9 (margin)
    uint32_t maxBytesPerTile = static_cast<uint32_t>(UB_BUDGET_BYTES * 0.9f);
    uint32_t estimatedTileH = maxBytesPerTile / (8 * W) / 2;  // approximate (2tileH+2) ~ 2tileH
    if (estimatedTileH > H) estimatedTileH = H;
    if (estimatedTileH > 64) estimatedTileH = 64;  // cap for latency
    uint32_t tileH = estimatedTileH;
    if (tileH < 1) tileH = 1;

    uint32_t blockDim = totalN * totalC;
    if (blockDim > 24) blockDim = 24;  // limit per core

    context->SetBlockDim(blockDim);

    tiling.totalN = totalN;
    tiling.totalC = totalC;
    tiling.H = H;
    tiling.W = W;
    tiling.tileH = tileH;
    tiling.blockDim = blockDim;
    tiling.kernelSize = kernelSize;
    tiling.stride = strideVal;
    tiling.padding = paddingVal;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *xShape = context->GetInputShape(0);
    gert::StorageShape *yShape = context->GetOutputShape(0);
    *yShape = *xShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

        this->AICore().SetTiling(optiling::TilingFunc);
