static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

    const gert::StorageShape *xShape = context->GetInputShape(0);
    int32_t batch = static_cast<int32_t>(xShape->GetDim(0));
    int32_t channels = static_cast<int32_t>(xShape->GetDim(1));
    int32_t inH = static_cast<int32_t>(xShape->GetDim(2));
    int32_t inW = static_cast<int32_t>(xShape->GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float scaleFactor = *attrs->GetAttrPointer<float>(0);

    int32_t outH = static_cast<int32_t>(static_cast<float>(inH) * scaleFactor);
    int32_t outW = static_cast<int32_t>(static_cast<float>(inW) * scaleFactor);

    uint32_t totalLength = static_cast<uint32_t>(batch) *
                           static_cast<uint32_t>(channels) *
                           static_cast<uint32_t>(outH) *
                           static_cast<uint32_t>(outW);

    // Simple tiling: process one row of output per tile (worst-case)
    uint32_t tileLen = static_cast<uint32_t>(outW) * static_cast<uint32_t>(channels);

    tiling.inW = inW;

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelUpsampleBilinear {
public:
    __aicore__ inline KernelUpsampleBilinear() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                float scaleFactor,
                                int32_t batch, int32_t channels,
                                int32_t inH, int32_t inW,
                                int32_t outH, int32_t outW) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);

        totalLen_ = totalLength;
        tileLen_ = tileLength;
        scaleFactor_ = scaleFactor;
        batch_ = batch;
        channels_ = channels;
        inH_ = inH;
        inW_ = inW;
        outH_ = outH;
        outW_ = outW;

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride over output tiles
        uint32_t stride = blockNum * tileLen_;
        uint32_t start = blockIdx * tileLen_;

        if (start >= totalLen_) return;

        uint32_t end = start + tileLen_;
        if (end > totalLen_) end = totalLen_;

        // Process output tile
        uint32_t curLen = end - start;
        Compute(start, curLen);
    }

private:
    __aicore__ inline void Compute(uint32_t offset, uint32_t len) {
        // Allocate local tensor for output tile
        AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

        // For each element in this tile, compute bilinear interpolation
        for (uint32_t i = 0; i < len; i++) {
            uint32_t linearIdx = offset + i;

            // Decode linear index to (n, c, h_out, w_out)
            int32_t h_out = static_cast<int32_t>(linearIdx / (channels_ * outW_));
            int32_t temp = linearIdx % (channels_ * outW_);
            int32_t c = temp / outW_;
            int32_t w_out = temp % outW_;

            // Map to input coordinates
            float h_in_f = static_cast<float>(h_out) / scaleFactor_;
            float w_in_f = static_cast<float>(w_out) / scaleFactor_;

            int32_t h0 = static_cast<int32_t>(h_in_f);
            int32_t w0 = static_cast<int32_t>(w_in_f);
            int32_t h1 = h0 + 1;
            int32_t w1 = w0 + 1;

            // Clamp to [0, inH-1] and [0, inW-1]
            if (h0 < 0) h0 = 0;
            if (h0 >= inH_) h0 = inH_ - 1;
            if (h1 < 0) h1 = 0;
            if (h1 >= inH_) h1 = inH_ - 1;
            if (w0 < 0) w0 = 0;
            if (w0 >= inW_) w0 = inW_ - 1;
            if (w1 < 0) w1 = 0;
            if (w1 >= inW_) w1 = inW_ - 1;

            float h_frac = h_in_f - static_cast<float>(h0);
            float w_frac = w_in_f - static_cast<float>(w0);

            // Compute linear indices in input (NCHW)
            uint32_t base_n = linearIdx / (channels_ * outH_ * outW_); // not needed, but fine
            // Actually, we need batch index: n = linearIdx / (channels_ * outH_ * outW_)
            // We already have h_out, c, w_out, but we need batch index:
            // The linear index is global over all output elements: n * (channels*outH*outW) + ...
            // Since we only decoded h_out, c, w_out without n, we must get n separately.
            // Let's compute n from the full offset.
            // We have offset (starting point) and i, so total global index = offset + i.
            // We'll compute n from that.
            uint32_t globalIdx = offset + i;
            int32_t n = static_cast<int32_t>(globalIdx / (channels_ * outH_ * outW_));
            // Recompute h_out, c, w_out relative to n
            uint32_t perImage = static_cast<uint32_t>(channels_ * outH_ * outW_);
            uint32_t idxInImage = globalIdx % perImage;
            // But we already have h_out, c, w_out limited to image? Actually our earlier decode assumed no batch, so it's wrong.
            // Let's redo correct decode:
            // perImage = channels * outH * outW
            // n = globalIdx / perImage
            // idxInImage = globalIdx % perImage
            // h_out = idxInImage / (channels * outW)
            // temp = idxInImage % (channels * outW)
            // c = temp / outW
            // w_out = temp % outW
            // We'll do this now.
        }

        // We need to load input pixels for this tile? For bilinear, each output depends on 4 input pixels scattered across input.
        // This approach loops over output elements and reads input from global memory directly (no tiling of input).
        // That is okay but may be slow. For simplicity, we read each output's required input pixels from global.
        // To avoid excessive global reads, we could load a tile of input, but that is complex.
        // Given the constraints, we'll implement a naive loop with direct global loads for each output element.
        // However, AscendC kernel forbids global memory access from vector core? Actually __gm__ pointers can be accessed.
        // We'll use DataCopy to local tensors for input? That would require loading the whole input image, which is large.
        // So we'll just read directly from global tensor via index.
        // In AscendC, we can do xGm_[index] to get a float. That is allowed.
        // We'll compute value and store directly to yLocal.

        // We'll rewrite the loop properly.
        for (uint32_t i = 0; i < len; i++) {
            uint32_t globalIdx = offset + i;
            uint32_t perImage = static_cast<uint32_t>(channels_) * static_cast<uint32_t>(outH_) * static_cast<uint32_t>(outW_);
            int32_t n = static_cast<int32_t>(globalIdx / perImage);
            uint32_t idxInImage = globalIdx % perImage;

            int32_t h_out = static_cast<int32_t>(idxInImage / (channels_ * outW_));
            int32_t temp = idxInImage % (channels_ * outW_);
            int32_t c = temp / outW_;
            int32_t w_out = temp % outW_;

            float h_in_f = static_cast<float>(h_out) / scaleFactor_;
            float w_in_f = static_cast<float>(w_out) / scaleFactor_;

            int32_t h0 = static_cast<int32_t>(h_in_f);
            int32_t w0 = static_cast<int32_t>(w_in_f);
            int32_t h1 = h0 + 1;
            int32_t w1 = w0 + 1;

            // Clamp
            h0 = (h0 < 0) ? 0 : (h0 >= inH_ ? inH_-1 : h0);
            w0 = (w0 < 0) ? 0 : (w0 >= inW_ ? inW_-1 : w0);
            h1 = (h1 < 0) ? 0 : (h1 >= inH_ ? inH_-1 : h1);
            w1 = (w1 < 0) ? 0 : (w1 >= inW_ ? inW_-1 : w1);

            float h_frac = h_in_f - static_cast<float>(h0);
            float w_frac = w_in_f - static_cast<float>(w0);

            // Input linear indices (NCHW)
            uint32_t inPerImage = static_cast<uint32_t>(channels_) * static_cast<uint32_t>(inH_) * static_cast<uint32_t>(inW_);
            uint32_t baseN = static_cast<uint32_t>(n) * inPerImage;
            uint32_t baseC = baseN + static_cast<uint32_t>(c) * static_cast<uint32_t>(inH_) * static_cast<uint32_t>(inW_);
            uint32_t idx00 = baseC + static_cast<uint32_t>(h0) * static_cast<uint32_t>(inW_) + static_cast<uint32_t>(w0);
            uint32_t idx01 = baseC + static_cast<uint32_t>(h0) * static_cast<uint32_t>(inW_) + static_cast<uint32_t>(w1);
            uint32_t idx10 = baseC + static_cast<uint32_t>(h1) * static_cast<uint32_t>(inW_) + static_cast<uint32_t>(w0);
            uint32_t idx11 = baseC + static_cast<uint32_t>(h1) * static_cast<uint32_t>(inW_) + static_cast<uint32_t>(w1);

            float p00 = xGm_[idx00];
            float p01 = xGm_[idx01];
            float p10 = xGm_[idx10];
            float p11 = xGm_[idx11];

            float val = (1.0f - h_frac) * (1.0f - w_frac) * p00
                      + (1.0f - h_frac) * w_frac * p01
                      + h_frac * (1.0f - w_frac) * p10
                      + h_frac * w_frac * p11;

            yLocal.SetValue(i, val);
        }

        // Copy output tile to global
        AscendC::DataCopy(yGm_[offset], yLocal, len);

        outQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    float scaleFactor_;
    int32_t batch_;
    int32_t channels_;
    int32_t inH_;
    int32_t inW_;
    int32_t outH_;
    int32_t outW_;
};

extern "C" __global__ __aicore__ void ca6k_04615_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelUpsampleBilinear op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.scaleFactor,
            tiling_data.batch, tiling_data.channels,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW);
    op.Process();
}
'''

at::Tensor upsample_bilinear_npu(const at::Tensor& self, double scale_factor) {

    return result;

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_04615_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, scale_factor):
        super(ModelNew, self).__init__()
        self.scale_factor = scale_factor
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1)

    def forward(self, x):
        # Custom bilinear upsampling
        fn = getattr(custom_ops_lib, _FN)
        x = fn(x, self.scale_factor)
        x = self.conv(x)
        return x
'''
