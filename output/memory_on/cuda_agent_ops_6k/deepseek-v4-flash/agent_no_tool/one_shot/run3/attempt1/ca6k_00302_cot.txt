constexpr int32_t BUFFER_NUM = 2;

class Ca6k00302CustomKernel {
public:
    __aicore__ inline Ca6k00302CustomKernel() {}

    __aicore__ inline void Init(GM_ADDR input, GM_ADDR target, GM_ADDR output,
                                uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inputGm_.SetGlobalBuffer((__gm__ float*)input, totalLen_);
        targetGm_.SetGlobalBuffer((__gm__ float*)target, totalLen_);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, 1);

        pipe_.InitBuffer(inQueueInput_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueTarget_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sumQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        sum_ = 0.0f;
        uint32_t offset = 0;
        while (offset < totalLen_) {
            uint32_t curLen = tileLen_;
            if (offset + curLen > totalLen_) curLen = totalLen_ - offset;
            CopyIn(offset, curLen);
            Compute(curLen);
            offset += curLen;
        }
        // Write final result
        AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
        sumLocal.SetValue(0, sum_);
        AscendC::DataCopy(outputGm_, sumLocal, 1);
        sumQueue_.FreeTensor(sumLocal);
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> inputLocal = inQueueInput_.AllocTensor<float>();
        AscendC::LocalTensor<float> targetLocal = inQueueTarget_.AllocTensor<float>();
        AscendC::DataCopy(inputLocal, inputGm_[gmOffset], len);
        AscendC::DataCopy(targetLocal, targetGm_[gmOffset], len);
        inQueueInput_.EnQue(inputLocal);
        inQueueTarget_.EnQue(targetLocal);
    }

    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> inputLocal = inQueueInput_.DeQue<float>();
        AscendC::LocalTensor<float> targetLocal = inQueueTarget_.DeQue<float>();
        AscendC::LocalTensor<float> diffLocal = tmpQueue_.AllocTensor<float>();

        // diff = input - target
        AscendC::Sub(diffLocal, inputLocal, targetLocal, len);
        // abs(diff)
        AscendC::Abs(diffLocal, diffLocal, len);

        // Local reduce sum
        AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
        AscendC::ReduceAdd(sumLocal, diffLocal, len);  // sumLocal[0] = sum of diffLocal
        sum_ += sumLocal.GetValue(0);

        sumQueue_.FreeTensor(sumLocal);
        tmpQueue_.FreeTensor(diffLocal);
        inQueueInput_.FreeTensor(inputLocal);
        inQueueTarget_.FreeTensor(targetLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueInput_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTarget_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> sumQueue_;

    AscendC::GlobalTensor<float> inputGm_;
    AscendC::GlobalTensor<float> targetGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    float sum_{0.0f};
};

}
