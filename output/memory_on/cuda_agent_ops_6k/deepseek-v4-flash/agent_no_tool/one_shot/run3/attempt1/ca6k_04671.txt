#include "ca6k_04671_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04671CustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0);
    const auto &dims = xShape->GetOriginShape().GetDim();
    uint32_t totalLength = 1;
    for (auto d : dims) totalLength *= d;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = attrs->GetAttrPointer<float>(0) ? *attrs->GetAttrPointer<float>(0) : 1e-5f;
    float momentum = attrs->GetAttrPointer<float>(1) ? *attrs->GetAttrPointer<float>(1) : 0.1f;
    bool training = attrs->GetAttrPointer<bool>(2) ? *attrs->GetAttrPointer<bool>(2) : false;

    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * 5);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;
    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batchSize(dims[0]);
    tiling.set_channels(dims[1]);
    tiling.set_spatialSize(dims[2] * dims[3]);
    tiling.set_eps(eps);
    tiling.set_momentum(momentum);
    tiling.set_training(training);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04671Custom {
public:
    __aicore__ inline KernelCa6k04671Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                GM_ADDR running_mean, GM_ADDR running_var,
                                GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t channels,
                                uint32_t spatialSize, float eps, float momentum,
                                bool training)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        channels_ = channels;
        spatialSize_ = spatialSize;
        eps_ = eps;
        momentum_ = momentum;
        training_ = training;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        meanGm_.SetGlobalBuffer((__gm__ float*)running_mean, channels_);
        varGm_.SetGlobalBuffer((__gm__ float*)running_var, channels_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, channels_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, channels_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));

        // Copy per-channel arrays to local for fast access
        pipe_.InitBuffer(meanLocal_, channels_ * sizeof(float));
        pipe_.InitBuffer(varLocal_, channels_ * sizeof(float));
        pipe_.InitBuffer(weightLocal_, channels_ * sizeof(float));
        pipe_.InitBuffer(biasLocal_, channels_ * sizeof(float));
        AscendC::DataCopy(meanLocal_, meanGm_, channels_);
        AscendC::DataCopy(varLocal_, varGm_, channels_);
        AscendC::DataCopy(weightLocal_, weightGm_, channels_);
        AscendC::DataCopy(biasLocal_, biasGm_, channels_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = (totalLen_ - base) > tileLen_ ? tileLen_ : (totalLen_ - base);
        CopyIn(base, curLen);

        while (true) {
            uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = (totalLen_ - nextBase) > tileLen_ ? tileLen_ : (totalLen_ - nextBase);
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Process element-by-element using per-channel stats
        for (uint32_t i = 0; i < len; i++) {
            // Compute channel index for this element
            uint32_t globalIdx = currentOffset_ + i;  // we need to track current offset
            uint32_t batch = globalIdx / (channels_ * spatialSize_);
            uint32_t channel = (globalIdx / spatialSize_) % channels_;
            // Load stats for this channel from local buffers
            float mean = meanLocal_(channel);
            float var = varLocal_(channel);
            float weight = weightLocal_(channel);
            float bias = biasLocal_(channel);
            float xVal = xLocal(i);
            float yVal = weight * (xVal - mean) / sqrtf(var + eps_) + bias;
            yLocal(i) = yVal;
        }

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        currentOffset_ += len;  // update for next tile
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> meanGm_;
    AscendC::GlobalTensor<float> varGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    AscendC::LocalTensor<float> meanLocal_;
    AscendC::LocalTensor<float> varLocal_;
    AscendC::LocalTensor<float> weightLocal_;
    AscendC::LocalTensor<float> biasLocal_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t channels_ {0};
    uint32_t spatialSize_ {0};
    float eps_ {1e-5f};
    float momentum_ {0.1f};
    bool training_ {false};
    uint32_t currentOffset_ {0};  // tracks global offset for compute
};

extern "C" __global__ __aicore__ void ca6k_04671_custom(GM_ADDR x, GM_ADDR y,
    GM_ADDR running_mean, GM_ADDR running_var, GM_ADDR weight, GM_ADDR bias,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04671Custom op;
    op.Init(x, y, running_mean, running_var, weight, bias,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.batchSize, tiling_data.channels,
            tiling_data.spatialSize, tiling_data.eps,
            tiling_data.momentum, tiling_data.training);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_features, eps=1e-05, momentum=0.1, affine=True, track_running_stats=True):
        super().__init__()
        # This model uses the custom op directly; it does not maintain running stats itself.
        # The custom op expects running_mean and running_var as inputs.
        # We'll create dummy ones for initialization, but they will be overwritten at forward.
        self.running_mean = torch.zeros(num_features, device='npu')
        self.running_var = torch.ones(num_features, device='npu')
        self.weight = torch.nn.Parameter(torch.ones(num_features, device='npu'))
        self.bias = torch.nn.Parameter(torch.zeros(num_features, device='npu'))
        self.eps = eps
        self.momentum = momentum

    def forward(self, x):
        # For simplicity, we assume eval mode (training=False). The custom op will use
        # the provided running mean/var for normalization.
        return custom_ops_lib.ca6k_04671_custom(
            x, self.running_mean, self.running_var,
            self.weight, self.bias,
            self.eps, self.momentum, False
        )
