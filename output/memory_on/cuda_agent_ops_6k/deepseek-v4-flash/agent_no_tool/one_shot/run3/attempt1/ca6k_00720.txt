host_operator_src = '''
#include "ca6k_00720_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00720CustomTilingData tiling;

    // Get shapes
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::StorageShape *w_shape = context->GetInputShape(1);
    const gert::StorageShape *b_shape = context->GetInputShape(2);

    uint32_t batch = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t seq = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t inFeatures = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    uint32_t outFeatures = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));  // weight: [out, in]

    uint32_t totalRows = batch * seq;

    // Compute tile sizes
    // We will tile along both rows and output dimension.
    // UB budget: weight(tileOut*inFeatures) + x(tileRows*inFeatures) + y(tileRows*tileOut) + bias(tileOut) + temp (~x tile)
    // Try to set tileOut = outFeatures first; if too large, halve.
    uint32_t tileOut = outFeatures;
    uint32_t tileRows = 1;
    bool fit = false;
    while (!fit) {
        uint32_t weightBytes = tileOut * inFeatures * sizeof(float);
        uint32_t xBytes = tileRows * inFeatures * sizeof(float);
        uint32_t yBytes = tileRows * tileOut * sizeof(float);
        uint32_t biasBytes = tileOut * sizeof(float);
        uint32_t tempBytes = tileRows * inFeatures * sizeof(float);  // for copy temporary
        uint32_t totalBytes = weightBytes + xBytes + yBytes + biasBytes + tempBytes;
        if (totalBytes <= UB_BUDGET_BYTES) {
            fit = true;
        } else {
            // reduce tileRows first, then tileOut
            if (tileRows > 1) {
                tileRows = tileRows / 2;
                if (tileRows == 0) tileRows = 1;
            } else if (tileOut > 1) {
                tileOut = AlignDown(tileOut / 2, 8);
                if (tileOut == 0) tileOut = 1;
            } else {
                // minimal
                break;
            }
        }
    }

    // Align tileOut to 32 for better vectorization
    tileOut = AlignDown(tileOut, 32);
    if (tileOut == 0) tileOut = 32;
    tileRows = AlignDown(tileRows, 1);
    if (tileRows == 0) tileRows = 1;

    // Set block dim
    uint32_t blockDim = 8;  // fixed small
    context->SetBlockDim(blockDim);

    tiling.totalRows = totalRows;
    tiling.inFeatures = inFeatures;
    tiling.outFeatures = outFeatures;
    tiling.tileRows = tileRows;
    tiling.tileOut = tileOut;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00720Custom {
public:
    __aicore__ inline KernelCa6k00720Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalRows, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t tileRows, uint32_t tileOut) {
        totalRows_ = totalRows;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        tileRows_ = tileRows;
        tileOut_ = tileOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * inFeatures_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures_ * inFeatures_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * outFeatures_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileRows_ * inFeatures_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, 1, tileOut_ * inFeatures_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, tileOut_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileRows_ * inFeatures_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalRows_ == 0 || inFeatures_ == 0 || outFeatures_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        // Only one block for simplicity
        if (blockIdx != 0) return;

        // Process in tiles over output columns
        for (uint32_t oStart = 0; oStart < outFeatures_; oStart += tileOut_) {
            uint32_t curTileOut = tileOut_;
            if (oStart + curTileOut > outFeatures_) {
                curTileOut = outFeatures_ - oStart;
            }

            // Load weight and bias for this output tile
            AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[oStart * inFeatures_], curTileOut * inFeatures_);
            inQueueW_.EnQue(wLocal);

            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
            AscendC::DataCopy(bLocal, bGm_[oStart], curTileOut);
            inQueueB_.EnQue(bLocal);

            // Process rows in tiles
            for (uint32_t rStart = 0; rStart < totalRows_; rStart += tileRows_) {
                uint32_t curTileRows = tileRows_;
                if (rStart + curTileRows > totalRows_) {
                    curTileRows = totalRows_ - rStart;
                }

                // Load input x tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rStart * inFeatures_], curTileRows * inFeatures_);
                inQueueX_.EnQue(xLocal);

                // Compute output y tile
                AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> wDeq = inQueueW_.DeQue<float>();
                AscendC::LocalTensor<float> bDeq = inQueueB_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // For each row and output column, compute dot product
                for (uint32_t r = 0; r < curTileRows; ++r) {
                    for (uint32_t oc = 0; oc < curTileOut; ++oc) {
                        float sum = bDeq[oc];
                        for (uint32_t ic = 0; ic < inFeatures_; ++ic) {
                            sum += xDeq[r * inFeatures_ + ic] * wDeq[oc * inFeatures_ + ic];
                        }
                        yLocal[r * curTileOut + oc] = sum;
                    }
                }

                outQueueY_.EnQue<float>(yLocal);
                inQueueW_.EnQue(wDeq);  // re-enqueue for reuse across row tiles
                inQueueB_.EnQue(bDeq);
                inQueueX_.FreeTensor(xDeq);

                // Copy out
                AscendC::LocalTensor<float> yDeq = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[rStart * outFeatures_ + oStart], yDeq, curTileRows * curTileOut);
                outQueueY_.FreeTensor(yDeq);
            }

            // Free weight and bias local tensors after output tile is done
            AscendC::LocalTensor<float> wFinal = inQueueW_.DeQue<float>();
            inQueueW_.FreeTensor(wFinal);
            AscendC::LocalTensor<float> bFinal = inQueueB_.DeQue<float>();
            inQueueB_.FreeTensor(bFinal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_;
    uint32_t inFeatures_;
    uint32_t outFeatures_;
    uint32_t tileRows_;
    uint32_t tileOut_;
};

extern "C" __global__ __aicore__ void ca6k_00720_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00720Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalRows, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.tileRows, tiling_data.tileOut);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00720_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, bias_shape):
        super(ModelNew, self).__init__()
        # To match the original Model's parameters, we need to store weight and bias.
        # However, the original Model creates an nn.Linear and a separate bias parameter.
        # For simplicity, we expect the custom op to receive these as inputs.
        # The init_inputs are in_features, out_features, bias_shape.
        self.in_features = in_features
        self.out_features = out_features
        # We'll create dummy parameters to match the original's state dict? Not needed for forward.
        # The forward will receive x, weight, bias (the original model's linear.weight and self.bias).
        # To use like the original, we need to expose weight and bias as parameters.
        self.weight = torch.nn.Parameter(torch.randn(out_features, in_features))
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias)
'''
