kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k02348Custom {
public:
    __aicore__ inline KernelCa6k02348Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR y, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        n_ = td.n;
        c_in_ = td.c_in;
        h_in_ = td.h_in;
        w_in_ = td.w_in;
        c_out_ = td.c_out;
        k_h_ = td.k_h;
        k_w_ = td.k_w;
        stride_h_ = td.stride_h;
        stride_w_ = td.stride_w;
        pad_h_ = td.pad_h;
        pad_w_ = td.pad_w;
        out_pad_h_ = td.out_pad_h;
        out_pad_w_ = td.out_pad_w;
        h_out_ = td.h_out;
        w_out_ = td.w_out;
        totalElements_ = td.totalElements;
        tileLength_ = td.tileLength;
        hasBias_ = td.hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        bGm_.SetGlobalBuffer((__gm__ float*)bias);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, k_h_ * k_w_ * c_in_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, c_out_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalElements_ == 0) return;
        // Simple grid-stride loop over output elements
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        uint32_t stride = blockNum * tileLength_;
        uint32_t start = blockIdx * tileLength_;
        if (start >= totalElements_) return;

        // For each output element tile, compute via host-call reduction
        // This implementation is a naive scalar loop for clarity.
        // In practice, use vectorized accum.
        float* yLocal = outQueueY_.AllocTensor<float>();
        uint32_t remaining = totalElements_ - start;
        uint32_t len = (remaining > tileLength_) ? tileLength_ : remaining;
        if (len == 0) { outQueueY_.FreeTensor(yLocal); return; }

        // Load bias if present
        LocalTensor<float> biasLocal;
        if (hasBias_ > 0.5f) {
            biasLocal = inQueueB_.AllocTensor<float>();
            uint32_t biasLen = c_out_;
            pipe_barrier(PIPE_V);
            DataCopy(biasLocal, bGm_, biasLen);
            inQueueB_.EnQue(biasLocal);
        }

        // Process each output pixel in this tile
        for (uint32_t i = 0; i < len; i++) {
            uint32_t outIdx = start + i;
            // Convert linear index to (n, c_out, h_out, w_out)
            uint32_t tmp = outIdx;
            uint32_t ow = tmp % w_out_;
            tmp /= w_out_;
            uint32_t oh = tmp % h_out_;
            tmp /= h_out_;
            uint32_t co = tmp % c_out_;
            uint32_t n = tmp / c_out_;

            // Compute contribution: sum over input channels and kernel window
            float acc = 0.0f;
            for (uint32_t ci = 0; ci < c_in_; ci++) {
                for (uint32_t kh = 0; kh < k_h_; kh++) {
                    for (uint32_t kw = 0; kw < k_w_; kw++) {
                        // Input position
                        int32_t ih = static_cast<int32_t>(oh) + static_cast<int32_t>(pad_h_) - static_cast<int32_t>(kh);
                        int32_t iw = static_cast<int32_t>(ow) + static_cast<int32_t>(pad_w_) - static_cast<int32_t>(kw);
                        if (ih >= 0 && iw >= 0 && ih % stride_h_ == 0 && iw % stride_w_ == 0) {
                            uint32_t ihi = ih / stride_h_;
                            uint32_t iwi = iw / stride_w_;
                            if (ihi < h_in_ && iwi < w_in_) {
                                uint32_t xIdx = n * c_in_ * h_in_ * w_in_ +
                                                ci * h_in_ * w_in_ +
                                                ihi * w_in_ + iwi;
                                uint32_t wIdx = co * c_in_ * k_h_ * k_w_ +
                                                ci * k_h_ * k_w_ +
                                                kh * k_w_ + kw;
                                acc += xGm_.GetValue(xIdx) * wGm_.GetValue(wIdx);
                            }
                        }
                    }
                }
            }
            if (hasBias_ > 0.5f) {
                acc += bGm_.GetValue(co);
            }
            yLocal.SetValue(i, acc);
        }
        // Free bias tile
        if (hasBias_ > 0.5f) {
            LocalTensor<float> biasDeque = inQueueB_.DeQue<float>();
            inQueueB_.FreeTensor(biasDeque);
        }
        // Copy out
        outQueueY_.EnQue(yLocal);
        LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
        DataCopy(yGm_[start], yDeque, len);
        outQueueY_.FreeTensor(yDeque);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_, c_in_, h_in_, w_in_, c_out_, k_h_, k_w_;
    uint32_t stride_h_, stride_w_, pad_h_, pad_w_, out_pad_h_, out_pad_w_;
    uint32_t h_out_, w_out_, totalElements_, tileLength_;
    float hasBias_;
};

extern "C" __global__ __aicore__ void ca6k_02348_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                         GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    KernelCa6k02348Custom op;
    op.Init(x, weight, bias, y, tiling);
    op.Process();
}
