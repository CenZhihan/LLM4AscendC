host_operator_src = '''
#include "ca6k_00517_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00517CustomTilingData tiling;

    const gert::StorageShape *inShape = context->GetInputShape(0);
    const std::vector<int64_t> dims = inShape->GetStorageShape().GetDims();
    uint32_t N = static_cast<uint32_t>(dims[0]);
    uint32_t C = static_cast<uint32_t>(dims[1]);
    uint32_t L = static_cast<uint32_t>(dims[2]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(0);
    uint32_t pad = static_cast<uint32_t>(*padding);

    uint32_t totalSlices = N * C;
    uint32_t sliceLen = L;
    uint32_t sliceOutLen = L + 2 * pad;

    // Determine subtile size for the middle part (sliceLen)
    // We need: 2 * tileLen * sizeof(float) <= UB_BUDGET_BYTES (input tile + output tile)
    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = maxTile;
    if (sliceLen < tileLen) {
        tileLen = sliceLen;
        tileLen = AlignDown(tileLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = sliceLen;
    }

    tiling.totalSlices = totalSlices;
    tiling.sliceLen = sliceLen;
    tiling.sliceOutLen = sliceOutLen;
    tiling.padding = pad;
    tiling.tileLen = tileLen;

    // Block dim: at least 1, up to 24 for stability
    uint32_t numTiles = CeilDiv(totalSlices, 1u);
    uint32_t blockDim = (numTiles < 8u) ? 8u : numTiles;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::StorageShape *inShape = context->GetInputShape(0);
    std::vector<int64_t> dims = inShape->GetStorageShape().GetDims();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(0);
    dims[2] += 2 * (*padding);
    gert::StorageShape *outShape = context->GetOutputShape(0);
    outShape->GetStorageShape().SetDims(dims);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
}  // namespace ge

namespace ops {
REGISTER_CUSTOM_OP(Ca6k00517Custom)
    .Input("x")
    .ParamType(REQUIRED)
    .DataType({ge::DT_FLOAT})
    .Format({ge::FORMAT_ND})
    .UnknownShapeFormat({ge::FORMAT_ND})
    .Output("y")
    .ParamType(REQUIRED)
    .DataType({ge::DT_FLOAT})
    .Format({ge::FORMAT_ND})
    .UnknownShapeFormat({ge::FORMAT_ND})
    .Attr("padding", OPTIONAL, int64_t, 16)
    .SetInferShape(ge::InferShape)
    .SetInferDataType(ge::InferDataType)
    .AICore()
    .SetTiling(optiling::TilingFunc)
    .AddConfig("ascend910b");
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00517Custom {
public:
    __aicore__ inline KernelCa6k00517Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalSlices, uint32_t sliceLen,
                                uint32_t sliceOutLen, uint32_t padding,
                                uint32_t tileLen) {
        totalSlices_ = totalSlices;
        sliceLen_ = sliceLen;
        sliceOutLen_ = sliceOutLen;
        padding_ = padding;
        tileLen_ = tileLen;
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalSlices_ == 0 || sliceLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over slices
        for (uint32_t s = blockIdx; s < totalSlices_; s += blockNum) {
            ProcessSlice(s);
        }
    }

private:
    __aicore__ inline void ProcessSlice(uint32_t sliceIdx) {
        uint32_t inBase = sliceIdx * sliceLen_;
        uint32_t outBase = sliceIdx * sliceOutLen_;

        // Copy tail (last padding elements of slice) to output start
        if (padding_ > 0) {
            uint32_t tailOffset = inBase + sliceLen_ - padding_;
            AscendC::LocalTensor<float> tailBuf = pipe_.AllocTensor<float>();
            AscendC::DataCopy(tailBuf, xGm_[tailOffset], padding_);
            AscendC::DataCopy(yGm_[outBase], tailBuf, padding_);
            pipe_.FreeTensor(tailBuf);
        }

        // Copy middle (entire slice) to output offset padding
        uint32_t midInBase = inBase;
        uint32_t midOutBase = outBase + padding_;
        uint32_t midRemaining = sliceLen_;
        uint32_t midOffset = 0;
        while (midRemaining > 0) {
            uint32_t curTile = (midRemaining > tileLen_) ? tileLen_ : midRemaining;
            AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[midInBase + midOffset], curTile);
            AscendC::DataCopy(yGm_[midOutBase + midOffset], xLocal, curTile);
            outQueue_.FreeTensor(yLocal);
            inQueue_.FreeTensor(xLocal);
            midOffset += curTile;
            midRemaining -= curTile;
        }

        // Copy head (first padding elements of slice) to output end
        if (padding_ > 0) {
            uint32_t headOffset = outBase + sliceLen_ + padding_;  // actually outBase + sliceOutLen - padding_? Wait: output end starts at outBase + sliceLen_ + padding_? No: output length = sliceLen + 2*padding. After middle (sliceLen) placed at offset padding, the remaining last padding positions are from outBase + sliceLen + padding_ to outBase + sliceOutLen -1 (total length). So headBuf goes to outBase + sliceLen_ + padding_.
            uint32_t headOutStart = outBase + sliceLen_ + padding_; // Not padding? Actually we want to copy first padding elements of input slice to the end of output, which starts at outBase + sliceLen_ + padding_? Let's compute: Output layout: [0:padding] = tail, [padding:padding+sliceLen] = middle, [padding+sliceLen:padding+sliceLen+padding] = head. So head starts at outBase + padding_ + sliceLen_ = outBase + sliceLen_ + padding_. Since sliceOutLen = sliceLen + 2*padding, the head occupies last padding positions. So offset is outBase + sliceLen_ + padding_.
            AscendC::LocalTensor<float> headBuf = pipe_.AllocTensor<float>();
            AscendC::DataCopy(headBuf, xGm_[inBase], padding_);
            AscendC::DataCopy(yGm_[headOutStart], headBuf, padding_);
            pipe_.FreeTensor(headBuf);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalSlices_ {0};
    uint32_t sliceLen_ {0};
    uint32_t sliceOutLen_ {0};
    uint32_t padding_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00517_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00517Custom op;
    op.Init(x, y,
            tiling_data.totalSlices, tiling_data.sliceLen,
            tiling_data.sliceOutLen, tiling_data.padding,
            tiling_data.tileLen);
    op.Process();
}
'''

class ModelNew(torch.nn.Module):
