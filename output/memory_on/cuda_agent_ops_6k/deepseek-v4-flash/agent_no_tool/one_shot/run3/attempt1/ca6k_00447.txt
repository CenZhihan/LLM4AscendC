#include "ca6k_00447_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
  return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k00447CustomTilingData tiling;
  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int32_t pad_left = *attrs->GetAttrPointer<int32_t>(0);
  int32_t pad_right = *attrs->GetAttrPointer<int32_t>(1);
  int32_t pad_top = *attrs->GetAttrPointer<int32_t>(2);
  int32_t pad_bottom = *attrs->GetAttrPointer<int32_t>(3);
  int32_t pad_front = *attrs->GetAttrPointer<int32_t>(4);
  int32_t pad_back = *attrs->GetAttrPointer<int32_t>(5);

  const gert::Shape *x_shape = context->GetInputShape(0);
  uint32_t origN = x_shape->GetDim(0);
  uint32_t origC = x_shape->GetDim(1);
  uint32_t origD = x_shape->GetDim(2);
  uint32_t origH = x_shape->GetDim(3);
  uint32_t origW = x_shape->GetDim(4);

  uint32_t outD = origD + pad_front + pad_back;
  uint32_t outH = origH + pad_top + pad_bottom;
  uint32_t outW = origW + pad_left + pad_right;

  uint32_t totalOutLen = origN * origC * outD * outH * outW;

  // Tile length from UB: 2 buffers x sizeof(float) per element
  uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = maxTile;
  if (totalOutLen < tileLen) {
    uint32_t aligned = AlignDown(totalOutLen, ALIGN_ELEMS);
    tileLen = (aligned > 0) ? aligned : totalOutLen;
  }

  uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutLen, tileLen);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_totalOutLen(totalOutLen);
  tiling.set_tileLen(tileLen);
  tiling.set_blockDim(blockDim);
  tiling.set_pad_left(pad_left);
  tiling.set_pad_right(pad_right);
  tiling.set_pad_top(pad_top);
  tiling.set_pad_bottom(pad_bottom);
  tiling.set_pad_front(pad_front);
  tiling.set_pad_back(pad_back);
  tiling.set_origN(origN);
  tiling.set_origC(origC);
  tiling.set_origD(origD);
  tiling.set_origH(origH);
  tiling.set_origW(origW);
  tiling.set_outD(outD);
  tiling.set_outH(outH);
  tiling.set_outW(outW);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00447Custom {
public:
  __aicore__ inline KernelCa6k00447Custom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR bias,
                              uint32_t totalOutLen, uint32_t tileLen,
                              int32_t pad_left, int32_t pad_right,
                              int32_t pad_top, int32_t pad_bottom,
                              int32_t pad_front, int32_t pad_back,
                              uint32_t origN, uint32_t origC,
                              uint32_t origD, uint32_t origH, uint32_t origW,
                              uint32_t outD, uint32_t outH, uint32_t outW) {
    totalOutLen_ = totalOutLen;
    tileLen_ = tileLen;
    pad_left_ = pad_left;
    pad_right_ = pad_right;
    pad_top_ = pad_top;
    pad_bottom_ = pad_bottom;
    pad_front_ = pad_front;
    pad_back_ = pad_back;
    origN_ = origN;
    origC_ = origC;
    origD_ = origD;
    origH_ = origH;
    origW_ = origW;
    outD_ = outD;
    outH_ = outH;
    outW_ = outW;

    // Strides for output: (N, C, D, H, W)
    outStrideC_ = outD_ * outH_ * outW_;
    outStrideD_ = outH_ * outW_;
    outStrideH_ = outW_;
    outStrideW_ = 1;

    // Strides for input: (N, C, D, H, W)
    inStrideC_ = origD_ * origH_ * origW_;
    inStrideD_ = origH_ * origW_;
    inStrideH_ = origW_;
    inStrideW_ = 1;

    xGm_.SetGlobalBuffer((__gm__ float*)x, totalOutLen_);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);
    biasGm_.SetGlobalBuffer((__gm__ float*)bias, origC_);

    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalOutLen_ == 0 || tileLen_ == 0) return;

    uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    uint32_t strideElems = blockNum * tileLen_;
    uint32_t base = blockIdx * tileLen_;
    if (base >= totalOutLen_) return;

    uint32_t curLen = totalOutLen_ - base;
    if (curLen > tileLen_) curLen = tileLen_;

    // Prefetch first tile (no input data, we compute on the fly)
    // Actually we don't need to copy input because each output element is independent.
    // We'll compute directly and write output.
    // Use double buffering for output only.
    // For simplicity, we process tile without prefetch because input is accessed directly from global.
    // But we can still use double buffering for output to overlap compute and DMA.
    // We'll compute first tile and enqueue output.
    ComputeRange(base, curLen);

    while (true) {
      uint32_t nextBase = base + strideElems;
      uint32_t nextLen = 0;
      if (nextBase < totalOutLen_) {
        nextLen = totalOutLen_ - nextBase;
        if (nextLen > tileLen_) nextLen = tileLen_;
        ComputeRange(nextBase, nextLen);
      }

      CopyOut(base, curLen);

      if (nextLen == 0) break;
      base = nextBase;
      curLen = nextLen;
    }
  }

private:
  __aicore__ inline void ComputeRange(uint32_t outOffset, uint32_t len) {
    AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
    for (uint32_t i = 0; i < len; i++) {
      uint32_t flatIdx = outOffset + i;
      // Decompose flatIdx into (b, c, d, h, w) for output
      uint32_t b = flatIdx / (origC_ * outStrideC_);
      uint32_t rem = flatIdx % (origC_ * outStrideC_);
      uint32_t c = rem / outStrideC_;
      rem = rem % outStrideC_;
      uint32_t d = rem / outStrideD_;
      rem = rem % outStrideD_;
      uint32_t h = rem / outStrideH_;
      uint32_t w = rem % outStrideH_;

      float val;
      // Check if inside original region
      int32_t in_d = static_cast<int32_t>(d) - pad_front_;
      int32_t in_h = static_cast<int32_t>(h) - pad_top_;
      int32_t in_w = static_cast<int32_t>(w) - pad_left_;
      if (in_d >= 0 && in_d < static_cast<int32_t>(origD_) &&
          in_h >= 0 && in_h < static_cast<int32_t>(origH_) &&
          in_w >= 0 && in_w < static_cast<int32_t>(origW_)) {
        uint32_t inIdx = b * origC_ * inStrideC_ +
                         c * inStrideC_ +
                         static_cast<uint32_t>(in_d) * inStrideD_ +
                         static_cast<uint32_t>(in_h) * inStrideH_ +
                         static_cast<uint32_t>(in_w) * inStrideW_;
        val = xGm_.GetValue(inIdx);
      } else {
        val = 0.0f;
      }
      // Add bias per channel
      val += biasGm_.GetValue(c);
      yLocal.SetValue(i, val);
    }
    outQueueY_.EnQue<float>(yLocal);
  }

  __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
    AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
    AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
    outQueueY_.FreeTensor(yLocal);
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;

  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> yGm_;
  AscendC::GlobalTensor<float> biasGm_;

  uint32_t totalOutLen_{0};
  uint32_t tileLen_{0};
  int32_t pad_left_, pad_right_, pad_top_, pad_bottom_, pad_front_, pad_back_;
  uint32_t origN_, origC_, origD_, origH_, origW_;
  uint32_t outD_, outH_, outW_;
  uint32_t outStrideC_, outStrideD_, outStrideH_, outStrideW_;
  uint32_t inStrideC_, inStrideD_, inStrideH_, inStrideW_;
};

extern "C" __global__ __aicore__ void ca6k_00447_custom(GM_ADDR x, GM_ADDR y, GM_ADDR bias,
                                                        GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelCa6k00447Custom op;
  op.Init(x, y, bias,
          tiling_data.totalOutLen, tiling_data.tileLen,
          tiling_data.pad_left, tiling_data.pad_right,
          tiling_data.pad_top, tiling_data.pad_bottom,
          tiling_data.pad_front, tiling_data.pad_back,
          tiling_data.origN, tiling_data.origC,
          tiling_data.origD, tiling_data.origH, tiling_data.origW,
          tiling_data.outD, tiling_data.outH, tiling_data.outW);
  op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00447_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, pad_size, bias_shape):
        super(ModelNew, self).__init__()
        self.pad_size = pad_size  # tuple (left,right,top,bottom,front,back)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        # pad_size is a tuple of 6 ints
        return fn(x, self.bias, *self.pad_size)
