#include "ca6k_00536_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/utils/type_utils.h"
#include "graph/utils/tensor_utils.h"
#include "graph/operator_preload.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_UNIT = 32;
static constexpr uint32_t MIN_TILE_M = 4;
static constexpr uint32_t MIN_TILE_K = 16;

static uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00536CustomTilingData tiling;

    auto xShape = context->GetInputShape(0);
    auto weightShape = context->GetInputShape(1);
    auto biasShape = context->GetInputShape(2);

    uint32_t totalM = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(xShape->GetDim(1));
    uint32_t N = static_cast<uint32_t>(weightShape->GetDim(1));

    float alpha = 1.0f;
    // optionally get alpha from attributes
    if (context->GetAttrs()->GetAttrNum(0) > 0) {
        alpha = *(context->GetAttrs()->GetAttrPointer<float>(0));
    }

    // Estimate tile sizes from UB budget:
    // needed per tile: (tileM * tileK + tileK * N + tileM * N + N) floats * 4 bytes
    uint32_t bestTileM = MIN_TILE_M;
    uint32_t bestTileK = MIN_TILE_K;
    uint32_t maxBuffer = UB_BUDGET_BYTES / 4;

    for (uint32_t tm = MIN_TILE_M; tm <= 32; tm *= 2) {
        for (uint32_t tk = MIN_TILE_K; tk <= 128; tk *= 2) {
            uint32_t needed = tm * N + tm * tk + tk * N + N;
            if (needed <= maxBuffer) {
                bestTileM = tm;
                bestTileK = tk;
            }
        }
    }
    if (bestTileM > totalM) bestTileM = totalM;
    bestTileM = AlignDown(bestTileM, ALIGN_UNIT);
    if (bestTileM < MIN_TILE_M) bestTileM = MIN_TILE_M;
    if (bestTileK > K) bestTileK = K;
    bestTileK = AlignDown(bestTileK, ALIGN_UNIT);
    if (bestTileK < MIN_TILE_K) bestTileK = MIN_TILE_K;

    uint32_t numMTiles = CeilDiv(totalM, bestTileM);
    uint32_t blockDim = numMTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    tiling.totalM = totalM;
    tiling.K = K;
    tiling.N = N;
    tiling.tileM = bestTileM;
    tiling.tileK = bestTileK;
    tiling.blockDim = blockDim;
    tiling.alpha = alpha;

    context->SetBlockDim(blockDim);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

kernel_src = '''
#include "kernel_operator.h"
#include "matmul_api.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00536Custom {
public:
    __aicore__ inline KernelCa6k00536Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalM, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileK,
                                float alpha)
    {
        totalM_ = totalM;
        K_ = K;
        N_ = N;
        tileM_ = tileM;
        tileK_ = tileK;
        alpha_ = alpha;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalM_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, K_ * N_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalM_ * N_);

        // Pre-allocate buffers
        pipe_.InitBuffer(inQueBias_, 1, N_ * sizeof(float));
        pipe_.InitBuffer(inQueX_, 1, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueW_, 1, tileK_ * N_ * sizeof(float));
        pipe_.InitBuffer(outQueC_, 1, tileM_ * N_ * sizeof(float));
        pipe_.InitBuffer(tmpQueC_, 1, tileM_ * N_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalM_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Load bias once into UB
        auto biasLocal = inQueBias_.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, biasGm_, N_);
        inQueBias_.EnQue(biasLocal);

        uint32_t totalMTiles = (totalM_ + tileM_ - 1) / tileM_;
        for (uint32_t t = blockIdx; t < totalMTiles; t += blockNum) {
            uint32_t mStart = t * tileM_;
            uint32_t mEnd = mStart + tileM_;
            if (mEnd > totalM_) mEnd = totalM_;
            uint32_t curM = mEnd - mStart;

            // Initialize result tile C to zero
            auto cLocal = outQueC_.AllocTensor<float>();
            AscendC::SetZero(cLocal, curM * N_);
            outQueC_.EnQue(cLocal);

            // Loop over K tiles
            uint32_t totalKTiles = (K_ + tileK_ - 1) / tileK_;
            for (uint32_t kt = 0; kt < totalKTiles; ++kt) {
                uint32_t kStart = kt * tileK_;
                uint32_t kEnd = kStart + tileK_;
                if (kEnd > K_) kEnd = K_;
                uint32_t curK = kEnd - kStart;

                // Load A tile (x)
                auto xLocal = inQueX_.AllocTensor<float>();
                LoadTileX(xLocal, mStart, mEnd, kStart, kEnd);
                inQueX_.EnQue(xLocal);

                // Load B tile (weight)
                auto wLocal = inQueW_.AllocTensor<float>();
                LoadTileW(wLocal, kStart, kEnd);
                inQueW_.EnQue(wLocal);

                // Compute partial product C += A * B
                // Use MatmulCube
                AscendC::MatmulCube<AscendC::CubeFormat::ND, float, float, float> mm;
                auto pLocal = tmpQueC_.AllocTensor<float>();
                mm.SetTensorA(xLocal, AscendC::CubeFormat::ND, curM, curK);
                mm.SetTensorB(wLocal, AscendC::CubeFormat::ND, curK, N_);
                mm.SetTensorC(pLocal, AscendC::CubeFormat::ND, curM, N_);
                mm.IterateAll();

                // Accumulate to result C
                auto cLocalAcc = outQueC_.DeQue<float>();
                AscendC::Add(cLocalAcc, cLocalAcc, pLocal, curM * N_);
                outQueC_.EnQue(cLocalAcc);

                // Free temporary buffers
                tmpQueC_.FreeTensor(pLocal);
                inQueX_.FreeTensor(xLocal);
                inQueW_.FreeTensor(wLocal);
            }

            // After K loop, add bias and apply CELU
            auto cFinal = outQueC_.DeQue<float>();
            auto biasRead = inQueBias_.DeQue<float>();
            // Add bias: broadcast N dimension to each row
            for (uint32_t r = 0; r < curM; ++r) {
                AscendC::Add(cFinal, cFinal, biasRead, N_);
            }
            // Apply CELU: y = max(0,x) + min(0, alpha*(exp(x/alpha)-1))
            // Since alpha=1.0 hardcoded, we compute:
            //   for each element: if (x > 0) y = x; else y = exp(x) - 1;
            for (uint32_t i = 0; i < curM * N_; ++i) {
                float val = cFinal(i);
                if (val > 0.0f) {
                    // do nothing, val stays
                } else {
                    val = expf(val) - 1.0f;
                }
                cFinal(i) = val;
            }

            // Write result to global memory
            AscendC::DataCopy(yGm_[mStart * N_], cFinal, curM * N_);
            outQueC_.FreeTensor(cFinal);
            inQueBias_.EnQue(biasRead);
        }
    }

private:
    __aicore__ inline void LoadTileX(AscendC::LocalTensor<float> dst,
                                     uint32_t mStart, uint32_t mEnd,
                                     uint32_t kStart, uint32_t kEnd)
    {
        uint32_t curM = mEnd - mStart;
        uint32_t curK = kEnd - kStart;
        for (uint32_t i = 0; i < curM; ++i) {
            AscendC::DataCopy(dst[i * curK], xGm_[(mStart + i) * K_ + kStart], curK);
        }
    }

    __aicore__ inline void LoadTileW(AscendC::LocalTensor<float> dst,
                                     uint32_t kStart, uint32_t kEnd)
    {
        uint32_t curK = kEnd - kStart;
        AscendC::DataCopy(dst, wGm_[kStart * N_], curK * N_);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueBias_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueC_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalM_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {1};
    uint32_t tileK_ {1};
    float alpha_ {1.0f};
};

extern "C" __global__ __aicore__ void ca6k_00536_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00536Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalM, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileK,
            tiling_data.alpha);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = "ca6k_00536_custom"

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features):
        super().__init__()
        # No internal modules; weights are passed as arguments.
        self.in_features = in_features
        self.out_features = out_features

    def forward(self, x, weight, bias):
        # weight shape: (out_features, in_features), need transpose before custom op
        # custom op expects weight as [in_features, out_features]
        weight_t = weight.t().contiguous()
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, weight_t, bias, alpha=1.0)
