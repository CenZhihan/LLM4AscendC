#include "ca6k_01135_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;  // for float alignment
static constexpr uint32_t MAX_TILE_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k01135CustomTilingData tiling;

    const gert::Shape* inputShape = context->GetInputShape(0);
    const int64_t* dims = inputShape->GetOriginShape().GetDims();
    // Expect 4D: N, C, H, W
    uint32_t N = static_cast<uint32_t>(dims[0]);
    uint32_t C = static_cast<uint32_t>(dims[1]);
    uint32_t H = static_cast<uint32_t>(dims[2]);
    uint32_t W = static_cast<uint32_t>(dims[3]);

    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    float scale = *(attrs->GetAttrPointer<float>(0));
    uint32_t H_out = static_cast<uint32_t>(H * scale);
    uint32_t W_out = static_cast<uint32_t>(W * scale);

    uint32_t totalInputElems = N * C * H * W;
    uint32_t totalOutputElems = N * C * H_out * W_out;

    // Determine tile length for output elements
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * 2); // input + output
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalOutputElems < tileLen) {
        tileLen = AlignDown(totalOutputElems, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalOutputElems;
    }

    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(totalOutputElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalInputLength(totalInputElems);
    tiling.set_totalOutputLength(totalOutputElems);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_inputH(H);
    tiling.set_inputW(W);
    tiling.set_outputH(H_out);
    tiling.set_outputW(W_out);
    tiling.set_scaleFactor(scale);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stride = blockNum * tileLength_;
        uint32_t base = blockIdx * tileLength_;
        if (base >= totalOutputLength_) return;

        while (base < totalOutputLength_) {
            uint32_t curLen = totalOutputLength_ - base;
            if (curLen > tileLength_) curLen = tileLength_;

            CopyIn(base, curLen);
            Compute(base, curLen);
            CopyOut(base, curLen);

            base += stride;
