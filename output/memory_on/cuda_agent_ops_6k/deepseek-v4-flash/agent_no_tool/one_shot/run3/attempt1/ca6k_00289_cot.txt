kernel_src = """

#include "kernel_operator.h"

using namespace AscendC;

class KernelMaxPool3d {
public:
    __aicore__ inline KernelMaxPool3d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t D, uint32_t H, uint32_t W,
                                uint32_t Ho, uint32_t Wo,
                                int32_t kd, int32_t kh, int32_t kw,
                                int32_t sd, int32_t sh, int32_t sw,
                                int32_t pd, int32_t ph, int32_t pw)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        N_ = N; C_ = C; D_ = D; H_ = H; W_ = W;
        Ho_ = Ho; Wo_ = Wo;
        kd_ = kd; kh_ = kh; kw_ = kw;
        sd_ = sd; sh_ = sh; sw_ = sw;
        pd_ = pd; ph_ = ph; pw_ = pw;
        // compute total output elements for tiling
        totalOut_ = N_ * C_ * D_ * Ho_ * Wo_;
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        
        // grid-stride over output elements
        for (uint32_t idx = blockIdx; idx < totalOut_; idx += blockNum) {
            // decode output index to (n, c, d, h, w)
            uint32_t tmp = idx;
            uint32_t w = tmp % Wo_; tmp /= Wo_;
            uint32_t h = tmp % Ho_; tmp /= Ho_;
            uint32_t d = tmp % D_;  // depth dimension size is D (since output depth same as input)
            tmp /= D_;
            uint32_t c = tmp % C_;
            uint32_t n = tmp / C_;

            // compute input window bounds
            int32_t h_start = static_cast<int32_t>(h * sh_) - ph_;
            int32_t w_start = static_cast<int32_t>(w * sw_) - pw_;
            int32_t d_start = static_cast<int32_t>(d * sd_) - pd_;
            // d_start is 0 because pd=0, kd=1

            float maxVal = -FLT_MAX;
            for (int32_t kz = 0; kz < kd_; kz++) {
                int32_t in_d = d_start + kz;
                if (in_d < 0 || in_d >= static_cast<int32_t>(D_)) continue;
                for (int32_t ky = 0; ky < kh_; ky++) {
                    int32_t in_h = h_start + ky;
                    if (in_h < 0 || in_h >= static_cast<int32_t>(H_)) continue;
                    for (int32_t kx = 0; kx < kw_; kx++) {
                        int32_t in_w = w_start + kx;
                        if (in_w < 0 || in_w >= static_cast<int32_t>(W_)) continue;
                        // compute input linear index
                        uint32_t inIdx = ((n * C_ + c) * D_ + in_d) * H_ * W_ + in_h * W_ + in_w;
                        float val = xGm_.GetValue(inIdx);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }
            // write output
            yGm_.SetValue(idx, maxVal);
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    uint32_t N_, C_, D_, H_, W_;
    uint32_t Ho_, Wo_;
    uint32_t totalOut_;
    int32_t kd_, kh_, kw_;
    int32_t sd_, sh_, sw_;
    int32_t pd_, ph_, pw_;
};

extern "C" __global__ __aicore__ void ca6k_00289_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
