#include "ca6k_00343_custom_tiling.h"

namespace optiling {

static constexpr uint32_t MAX_WINDOW = 4096;  // elements
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    const gert::Shape *xShape = context->GetInputShape(0);
    const std::vector<int64_t> &dims = xShape->GetOriginShape().GetDims();
    uint32_t N = static_cast<uint32_t>(dims[0]);
    uint32_t C = static_cast<uint32_t>(dims[1]);
    uint32_t L = static_cast<uint32_t>(dims[2]);
    uint32_t totalInputSize = N * C * L;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *outputSizePtr = attrs->GetAttrPointer<int64_t>(0);
    uint32_t outputSize = static_cast<uint32_t>(*outputSizePtr);

    uint32_t blockDim = N * C;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // tileLen: the maximum window size for loading into UB
    uint32_t maxWindow = static_cast<uint32_t>(std::ceil(static_cast<double>(L) / outputSize));
    if (maxWindow < ALIGN_ELEMS) maxWindow = ALIGN_ELEMS;
    if (maxWindow > MAX_WINDOW) maxWindow = MAX_WINDOW;
    uint32_t tileLen = AlignDown(maxWindow, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = maxWindow;

    Ca6k00343CustomTilingData tiling;
    tiling.set_totalInputSize(totalInputSize);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_L(L);
    tiling.set_outputSize(outputSize);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

kernel_src = '''

#include "kernel_operator.h"
#include <limits>

constexpr int32_t BUFFER_NUM = 1;
constexpr int32_t MAX_WINDOW = 4096;

class KernelCa6k00343Custom {
public:
    __aicore__ inline KernelCa6k00343Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputSize, uint32_t N, uint32_t C, uint32_t L,
                                uint32_t outputSize, uint32_t tileLen)
    {
        totalInputSize_ = totalInputSize;
        N_ = N;
        C_ = C;
        L_ = L;
        outputSize_ = outputSize;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C * outputSize);

        pipe_.InitBuffer(windowBuf_, BUFFER_NUM, MAX_WINDOW * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalInputSize_ == 0) return;
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t totalPairs = N_ * C_;
        if (blockIdx >= totalPairs) return;

        // Each block handles batch and channel pair
        uint32_t b = blockIdx / C_;
        uint32_t c = blockIdx % C_;
        uint32_t inputOffset = b * C_ * L_ + c * L_;
        uint32_t outputOffset = b * C_ * outputSize_ + c * outputSize_;

        for (uint32_t k = 0; k < outputSize_; k++) {
            uint32_t start = (k * L_) / outputSize_;
            uint32_t end = ((k + 1) * L_) / outputSize_;
            if (end > L_) end = L_;
            uint32_t winLen = end - start;

            LocalTensor<float> winLocal = windowBuf_.AllocTensor<float>();
            DataCopy(winLocal, xGm_[inputOffset + start], winLen);

            float maxVal = -std::numeric_limits<float>::max();
            for (uint32_t i = 0; i < winLen; i++) {
                if (winLocal(i) > maxVal) maxVal = winLocal(i);
            }
            yGm_[outputOffset + k] = maxVal;

            windowBuf_.FreeTensor(winLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> windowBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputSize_ {0};
    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t L_ {0};
    uint32_t outputSize_ {1};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00343_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00343Custom op;
    op.Init(x, y,
            tiling_data.totalInputSize, tiling_data.N, tiling_data.C, tiling_data.L,
            tiling_data.outputSize, tiling_data.tileLen);
    op.Process();
}
'''
