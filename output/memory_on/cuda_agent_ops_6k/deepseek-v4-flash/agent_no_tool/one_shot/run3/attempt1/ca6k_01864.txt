kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01864Custom {
public:
    __aicore__ inline KernelCa6k01864Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias_conv, GM_ADDR bias_ext,
                                GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
        GET_TILING_DATA(td, tiling);
        N_ = td.N;
        C_in_ = td.C_in;
        C_out_ = td.C_out;
        L_in_ = td.L_in;
        L_out_ = td.L_out;
        kernel_size_ = td.kernel_size;
        stride_ = td.stride;
        padding_ = td.padding;
        output_padding_ = td.output_padding;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_in_ * L_in_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, C_in_ * C_out_ * kernel_size_);
        biasConvGm_.SetGlobalBuffer((__gm__ float*)bias_conv, C_out_);
        biasExtGm_.SetGlobalBuffer((__gm__ float*)bias_ext, C_out_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_out_ * L_out_);

        c_out_ = GetBlockIdx();
        if (c_out_ >= C_out_) return;

        // Allocate local weight (full) and input/output per batch
        uint32_t weightTotalSize = C_in_ * C_out_ * kernel_size_;
        pipe_.InitBuffer(weightBuf_, 1, weightTotalSize * sizeof(float));
        pipe_.InitBuffer(inBuf_, 1, C_in_ * L_in_ * sizeof(float));
        pipe_.InitBuffer(outBuf_, 1, L_out_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (c_out_ >= C_out_) return;

        // Load full weight
        AscendC::LocalTensor<float> weightFull = weightBuf_.AllocTensor<float>();
        AscendC::DataCopy(weightFull, weightGm_, C_in_ * C_out_ * kernel_size_);

        float biasSum = *(biasConvGm_.GetPtr() + c_out_) + *(biasExtGm_.GetPtr() + c_out_);

        for (uint32_t n = 0; n < N_; ++n) {
            // Load input for this batch
            AscendC::LocalTensor<float> xLocal = inBuf_.AllocTensor<float>();
            uint32_t batchOffset = n * C_in_ * L_in_;
            AscendC::DataCopy(xLocal, xGm_[batchOffset], C_in_ * L_in_);

            // Initialize output to zero
            AscendC::LocalTensor<float> yLocal = outBuf_.AllocTensor<float>();
            AscendC::DupTensor(yLocal, 0.0f, L_out_);

            // Transposed convolution
            for (uint32_t c_in = 0; c_in < C_in_; ++c_in) {
                uint32_t weightBase = c_in * C_out_ * kernel_size_ + c_out_ * kernel_size_;
                for (uint32_t k = 0; k < kernel_size_; ++k) {
                    float w = weightFull[weightBase + k];
                    for (uint32_t pos_in = 0; pos_in < L_in_; ++pos_in) {
                        int32_t pos_out = static_cast<int32_t>(pos_in) * stride_ - padding_ + k;
                        if (pos_out >= 0 && pos_out < static_cast<int32_t>(L_out_)) {
                            float xVal = xLocal[c_in * L_in_ + pos_in];
                            yLocal[pos_out] += w * xVal;
                        }
                    }
                }
            }

            // Add bias
            AscendC::Adds(yLocal, yLocal, biasSum, L_out_);

            // Store output
            uint32_t outBatchOffset = n * C_out_ * L_out_ + c_out_ * L_out_;
            AscendC::DataCopy(yGm_[outBatchOffset], yLocal, L_out_);

            outBuf_.FreeTensor(yLocal);
            inBuf_.FreeTensor(xLocal);
        }

        weightBuf_.FreeTensor(weightFull);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> weightBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasConvGm_;
    AscendC::GlobalTensor<float> biasExtGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0}, C_in_ {0}, C_out_ {0}, L_in_ {0}, L_out_ {0}, kernel_size_ {0};
    int32_t stride_ {1}, padding_ {0}, output_padding_ {0};
    uint32_t c_out_ {0};
};

extern "C" __global__ __aicore__ void ca6k_01864_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias_conv,
                                                         GM_ADDR bias_ext, GM_ADDR y, GM_ADDR workspace,
                                                         GM_ADDR tiling) {
    KernelCa6k01864Custom op;
    op.Init(x, weight, bias_conv, bias_ext, y, workspace, tiling);
    op.Process();
}

model_src = """
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_01864_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape):
        super(ModelNew, self).__init__()
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.weight = torch.nn.Parameter(torch.randn(in_channels, out_channels, kernel_size))
        self.bias_conv = torch.nn.Parameter(torch.randn(out_channels))
        self.bias_ext = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias_conv, self.bias_ext, self.stride, self.padding, self.output_padding)
