static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFuncCumsum(gert::TilingContext *context) {
  Ca6k00391CumsumTilingData tiling;
  uint32_t totalLength =
      static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

  uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) + sizeof(int32_t));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
  uint32_t tileLen = std::min(totalLength, maxTile);
  if (tileLen == 0) tileLen = 1;

  context->SetBlockDim(1);
  tiling.set_totalLength(totalLength);
  tiling.set_tileLength(tileLen);
  tiling.set_blockDim(1);
  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
  context->GetWorkspaceSizes(1)[0] = 0;
  return ge::GRAPH_SUCCESS;
}

  uint32_t scatterLen =

  uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float) + sizeof(int32_t) + sizeof(bool));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);

kernel_src = '''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00391Cumsum {
public:
    __aicore__ inline void Init(GM_ADDR mask, GM_ADDR cumsum,
                                uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        maskGm_.SetGlobalBuffer((__gm__ int8_t*)mask, totalLen_);
        cumsumGm_.SetGlobalBuffer((__gm__ int32_t*)cumsum, totalLen_);
        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(int8_t));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(int32_t));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        int32_t runningCount = 0;
        for (uint32_t offset = 0; offset < totalLen_; offset += tileLen_) {
            uint32_t curLen = totalLen_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;
            LocalTensor<int8_t> maskLocal = inQueue_.AllocTensor<int8_t>();
            DataCopy(maskLocal, maskGm_[offset], curLen);
            LocalTensor<int32_t> cumsumLocal = outQueue_.AllocTensor<int32_t>();
            for (uint32_t i = 0; i < curLen; i++) {
                int8_t m = maskLocal.GetValue(i);
                cumsumLocal.SetValue(i, runningCount);
                if (m != 0) runningCount++;
            }
            DataCopy(cumsumGm_[offset], cumsumLocal, curLen);
            outQueue_.FreeTensor(cumsumLocal);
            inQueue_.FreeTensor(maskLocal);
        }
    }

private:
    uint32_t totalLen_;
    uint32_t tileLen_;
    GlobalTensor<int8_t> maskGm_;
    GlobalTensor<int32_t> cumsumGm_;
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueue_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueue_;
};

class KernelCa6k00391Scatter {
public:
    __aicore__ inline void Init(GM_ADDR input, GM_ADDR mask, GM_ADDR cumsum,
                                GM_ADDR scatter, GM_ADDR output,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t scatterLen) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        scatterLen_ = scatterLen;
        inputGm_.SetGlobalBuffer((__gm__ float*)input, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ int8_t*)mask, totalLen_);
        cumsumGm_.SetGlobalBuffer((__gm__ int32_t*)cumsum, totalLen_);
        scatterGm_.SetGlobalBuffer((__gm__ float*)scatter, scatterLen_);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalLen_);
        pipe_.InitBuffer(inInputQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inMaskQueue_, BUFFER_NUM, tileLen_ * sizeof(int8_t));
        pipe_.InitBuffer(inCumsumQueue_, BUFFER_NUM, tileLen_ * sizeof(int32_t));
        pipe_.InitBuffer(outOutputQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;
        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            CopyIn(base, curLen);
            Compute(curLen);
            CopyOut(base, curLen);
            base += strideElems;
            if (base >= totalLen_) break;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        LocalTensor<float> inputLocal = inInputQueue_.AllocTensor<float>();
        LocalTensor<int8_t> maskLocal = inMaskQueue_.AllocTensor<int8_t>();
        LocalTensor<int32_t> cumsumLocal = inCumsumQueue_.AllocTensor<int32_t>();
        DataCopy(inputLocal, inputGm_[gmOffset], len);
        DataCopy(maskLocal, maskGm_[gmOffset], len);
        DataCopy(cumsumLocal, cumsumGm_[gmOffset], len);
        inInputQueue_.EnQue(inputLocal);
        inMaskQueue_.EnQue(maskLocal);
        inCumsumQueue_.EnQue(cumsumLocal);
    }

    __aicore__ inline void Compute(uint32_t len) {
        LocalTensor<float> inputLocal = inInputQueue_.DeQue<float>();
        LocalTensor<int8_t> maskLocal = inMaskQueue_.DeQue<int8_t>();
        LocalTensor<int32_t> cumsumLocal = inCumsumQueue_.DeQue<int32_t>();
        LocalTensor<float> outLocal = outOutputQueue_.AllocTensor<float>();
        for (uint32_t i = 0; i < len; i++) {
            int8_t m = maskLocal.GetValue(i);
            float val;
            if (m != 0) {
                int32_t idx = cumsumLocal.GetValue(i);
                val = scatterGm_.GetValue(idx);
            } else {
                val = inputLocal.GetValue(i);
            }
            outLocal.SetValue(i, val);
        }
        outOutputQueue_.EnQue(outLocal);
        inInputQueue_.FreeTensor(inputLocal);
        inMaskQueue_.FreeTensor(maskLocal);
        inCumsumQueue_.FreeTensor(cumsumLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
        LocalTensor<float> outLocal = outOutputQueue_.DeQue<float>();
        DataCopy(outputGm_[gmOffset], outLocal, len);
        outOutputQueue_.FreeTensor(outLocal);
    }

private:
    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t scatterLen_;
    GlobalTensor<float> inputGm_;
    GlobalTensor<int8_t> maskGm_;
    GlobalTensor<int32_t> cumsumGm_;
    GlobalTensor<float> scatterGm_;
    GlobalTensor<float> outputGm_;
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inInputQueue_;
    TQue<TPosition::VECIN, BUFFER_NUM> inMaskQueue_;
    TQue<TPosition::VECIN, BUFFER_NUM> inCumsumQueue_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outOutputQueue_;
};

extern "C" __global__ __aicore__ void ca6k_00391_cumsum(GM_ADDR mask, GM_ADDR cumsum,
                                                         GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00391Cumsum op;
    op.Init(mask, cumsum, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}

extern "C" __global__ __aicore__ void ca6k_00391_scatter(GM_ADDR input, GM_ADDR mask,
                                                          GM_ADDR cumsum, GM_ADDR scatter,
                                                          GM_ADDR output,
                                                          GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00391Scatter op;
    op.Init(input, mask, cumsum, scatter, output,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.scatterLen);
    op.Process();
}
'''
