#include "ca6k00390_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00390CustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(x_shape->GetShapeSize());
    uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t C = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t H = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t W = static_cast<uint32_t>(x_shape->GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *groupsPtr = attrs->GetAttrPointer<int64_t>(0);
    uint32_t groups = static_cast<uint32_t>(*groupsPtr);

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_groups(groups);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00390Custom {
public:
    __aicore__ inline KernelCa6k00390Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t groups, uint32_t N, uint32_t C,
                                uint32_t H, uint32_t W)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        groups_ = groups;
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        chPerGroup_ = C_ / groups_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inBuf_, 1, sizeof(float));
        pipe_.InitBuffer(outBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            Compute(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void Compute(uint32_t base, uint32_t len)
    {
        // Process each output element in this tile
        for (uint32_t i = 0; i < len; i++) {
            uint32_t outIdx = base + i;

            // Decompose flat output index into (n, c, h, w)
            uint32_t n = outIdx / (C_ * H_ * W_);
            uint32_t rem1 = outIdx % (C_ * H_ * W_);
            uint32_t c = rem1 / (H_ * W_);
            uint32_t rem2 = rem1 % (H_ * W_);
            uint32_t h = rem2 / W_;
            uint32_t w = rem2 % W_;

            // Compute input channel
            uint32_t inC = (c % groups_) * chPerGroup_ + (c / groups_);

            uint32_t inIdx = n * C_ * H_ * W_ + inC * H_ * W_ + h * W_ + w;

            // Load one float from input and store to output
            AscendC::LocalTensor<float> tmpIn = inBuf_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpOut = outBuf_.AllocTensor<float>();

            AscendC::DataCopy(tmpIn, xGm_[inIdx], 1);
            tmpOut(0) = tmpIn(0);
            AscendC::DataCopy(yGm_[outIdx], tmpOut, 1);

            inBuf_.FreeTensor(tmpIn);
            outBuf_.FreeTensor(tmpOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t groups_ {2};
    uint32_t N_ {1};
    uint32_t C_ {1};
    uint32_t H_ {1};
    uint32_t W_ {1};
    uint32_t chPerGroup_ {1};
};

extern "C" __global__ __aicore__ void ca6k00390_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00390Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.groups, tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W);
    op.Process();
}
'''

        if len(init_inputs) >= 1:

        else:
            self.num_channels = 64
        if len(init_inputs) >= 2:
            self.groups = init_inputs[1]
        else:
            self.groups = 2
        if len(init_inputs) >= 3:
            self.block_size = init_inputs[2]
