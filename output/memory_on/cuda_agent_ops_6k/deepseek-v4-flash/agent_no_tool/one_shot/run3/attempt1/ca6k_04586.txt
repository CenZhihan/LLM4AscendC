static constexpr uint32_t ALIGN_ELEMS = 1024;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04586CustomTilingData tiling;
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    uint32_t totalLength = 1;
    for (uint32_t i = 0; i < x_shape->GetDimNum(); i++) {
        totalLength *= x_shape->GetDim(i);
    }
    uint32_t numChannels = x_shape->GetDim(1);
    uint32_t spatialSize = totalLength / numChannels; // N*H*W

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    bool training = attrs->GetAttrPointer<bool>(0)[0];
    float momentum = attrs->GetAttrPointer<float>(1)[0];
    float eps = attrs->GetAttrPointer<float>(2)[0];

    // Tile sizing: each tile processes a part of spatial data for a channel
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float)); // 4 buffers (x, y, tmp, etc.)
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    uint32_t tileLen = (maxTile < spatialSize) ? maxTile : spatialSize;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = spatialSize;

    // Block dimension: one block per channel if possible, otherwise spread
    uint32_t blockDim = numChannels;
    if (blockDim > 32) blockDim = 32;
    if (blockDim < 8) blockDim = 8;

    context->SetBlockDim(blockDim);

    tiling.totalLength = totalLength;
    tiling.tileLength = tileLen;
    tiling.blockDim = blockDim;
    tiling.numChannels = numChannels;
    tiling.spatialSize = spatialSize;
    tiling.training = training;
    tiling.momentum = momentum;
    tiling.eps = eps;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_CHANNELS = 32;

class KernelCa6k04586Custom {
public:
    __aicore__ inline KernelCa6k04586Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR running_mean, GM_ADDR running_var,
                                GM_ADDR running_mean_out, GM_ADDR running_var_out,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t numChannels,
                                uint32_t spatialSize, bool training,
                                float momentum, float eps)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numChannels);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, numChannels);
        runningMeanGm_.SetGlobalBuffer((__gm__ float*)running_mean, numChannels);
        runningVarGm_.SetGlobalBuffer((__gm__ float*)running_var, numChannels);
        runningMeanOutGm_.SetGlobalBuffer((__gm__ float*)running_mean_out, numChannels);
        runningVarOutGm_.SetGlobalBuffer((__gm__ float*)running_var_out, numChannels);

        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        numChannels_ = numChannels;
        spatialSize_ = spatialSize;
        training_ = training;
        momentum_ = momentum;
        eps_ = eps;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = blockDim_;

        // Each block processes a contiguous set of channels
        uint32_t channelsPerBlock = (numChannels_ + blockNum - 1) / blockNum;
        uint32_t startChannel = blockIdx * channelsPerBlock;
        uint32_t endChannel = startChannel + channelsPerBlock;
        if (endChannel > numChannels_) endChannel = numChannels_;
        if (startChannel >= numChannels_) return;

        for (uint32_t c = startChannel; c < endChannel; c++) {
            float w = weightGm_.GetValue(c);
            float b = biasGm_.GetValue(c);
            float rm = runningMeanGm_.GetValue(c);
            float rv = runningVarGm_.GetValue(c);

            // Compute per-channel statistics if training
            float mean = 0.0f;
            float variance = 0.0f;
            if (training_) {
                // First pass: compute sum and sum_sq for this channel
                float sum = 0.0f;
                float sum_sq = 0.0f;
                uint32_t remaining = spatialSize_;
                uint32_t offset = c * spatialSize_;
                while (remaining > 0) {
                    uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                    // Load tile
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[offset], curLen);
                    inQueueX_.EnQue(xLocal);

                    xLocal = inQueueX_.DeQue<float>();
                    AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                    // sum
                    AscendC::ReduceAdd(tLocal, xLocal, curLen);
                    sum += tLocal.GetValue(0);
                    // sum_sq
                    AscendC::Mul(tLocal, xLocal, xLocal, curLen);
                    AscendC::ReduceAdd(tLocal, tLocal, curLen);
                    sum_sq += tLocal.GetValue(0);

                    tmpQueue_.FreeTensor(tLocal);
                    inQueueX_.FreeTensor(xLocal);
                    offset += curLen;
                    remaining -= curLen;
                }
                // compute mean and variance
                float inv_size = 1.0f / static_cast<float>(spatialSize_);
                mean = sum * inv_size;
                variance = sum_sq * inv_size - mean * mean;
                if (variance < 0.0f) variance = 0.0f; // clamp due to numerical error

                // Update running statistics
                float newMean = (1.0f - momentum_) * rm + momentum_ * mean;
                float newVar = (1.0f - momentum_) * rv + momentum_ * variance;
                runningMeanOutGm_.SetValue(c, newMean);
                runningVarOutGm_.SetValue(c, newVar);
            } else {
                mean = rm;
                variance = rv;
                // In inference, running stats unchanged
                runningMeanOutGm_.SetValue(c, rm);
                runningVarOutGm_.SetValue(c, rv);
            }

            // Second pass: normalize and apply scale/shift
            float inv_std = 1.0f / sqrt(variance + eps_);
            uint32_t remaining = spatialSize_;
            uint32_t offset = c * spatialSize_;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                // Load x
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curLen);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

                // (x - mean) * inv_std * weight + bias
                AscendC::Sub(tLocal, xLocal, mean, curLen);
                AscendC::Muls(tLocal, tLocal, inv_std, curLen);
                AscendC::Muls(tLocal, tLocal, w, curLen);
                AscendC::Add(yLocal, tLocal, b, curLen);

                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xLocal);

                // Store output
                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[offset], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);

                offset += curLen;
                remaining -= curLen;
            }
        }
    }

private:

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> yGm_;

    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> runningMeanGm_;
    AscendC::GlobalTensor<float> runningVarGm_;
    AscendC::GlobalTensor<float> runningMeanOutGm_;
    AscendC::GlobalTensor<float> runningVarOutGm_;

    uint32_t tileLen_;
