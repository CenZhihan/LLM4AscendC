host_operator_src = r'''
#include "ca6k_03088_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k03088CustomTilingData tiling;
  const auto& x_shape = context->GetInputShape(0)->GetOriginShape();
  uint32_t rows = static_cast<uint32_t>(x_shape.GetDim(0));
  uint32_t cols = static_cast<uint32_t>(x_shape.GetDim(1));
  uint32_t blockDim = cols; // one block per column
  context->SetBlockDim(blockDim);
  tiling.set_rows(rows);
  tiling.set_cols(cols);
  tiling.set_blockDim(blockDim);
  tiling.set_dim(0); // always dim=0 as per reference
  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
  size_t *workspace = context->GetWorkspaceSizes(1);
  workspace[0] = rows * cols * sizeof(float); // workspace for each column (shared)
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape *x_shape = context->GetInputShape(0);
  const int32_t dim = 0;
  // output shape: remove the dim dimension
  int64_t ndim = x_shape->GetDimNum();
  std::vector<int64_t> out_dims;
  for (int64_t i = 0; i < ndim; i++) {
    if (i != dim) out_dims.push_back(x_shape->GetDim(i));
  }
  gert::Shape *y_shape = context->GetOutputShape(0);
  *y_shape = gert::Shape(out_dims);
  return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
  const ge::DataType x_dtype = context->GetInputDataType(0);
  context->SetOutputDataType(0, x_dtype);
  return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class Ca6k03088Custom : public OpDef {
public:
  explicit Ca6k03088Custom(const char* name) : OpDef(name) {
    this->Input("x")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});
    this->Output("y")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});
    this->Attr("dim").AttrType(OPTIONAL).Int(0);
    this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
    this->AICore().SetTiling(optiling::TilingFunc);
    this->AICore().AddConfig("ascend910b");
  }
};
OP_ADD(Ca6k03088Custom);
} // namespace ops

kernel_src = r'''
#include "kernel_operator.h"

using namespace AscendC;

#define ROWS_PER_COLUMN 64000
#define COLS 512

extern "C" __global__ __aicore__ void ca6k_03088_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(td, tiling);
  uint32_t rows = td.rows;
  uint32_t cols = td.cols;
  uint32_t blockIdx = GetBlockIdx();
  if (blockIdx >= cols) return;

  // Global tensor for input, output, workspace
  GlobalTensor<float> xGm;
  xGm.SetGlobalBuffer((__gm__ float *)x, rows * cols);
  GlobalTensor<float> yGm;
  yGm.SetGlobalBuffer((__gm__ float *)y, cols);
  GlobalTensor<float> wsGm;
  wsGm.SetGlobalBuffer((__gm__ float *)workspace, rows * cols);

  // Copy the column data into workspace (global buffer)
  uint32_t colOffset = blockIdx * rows;
  DataCopy(wsGm[colOffset], xGm[colOffset], rows);

  // QuickSelect to find median (lower median)
  int32_t left = 0;
  int32_t right = rows - 1;
  int32_t k = (rows - 1) / 2;  // lower median index

  while (left < right) {
    float pivot = wsGm[colOffset + left];
    int32_t i = left - 1;
    int32_t j = right + 1;
    while (i < j) {
      do { i++; } while (wsGm[colOffset + i] < pivot);
      do { j--; } while (wsGm[colOffset + j] > pivot);
      if (i < j) {
        float tmp = wsGm[colOffset + i];
        wsGm[colOffset + i] = wsGm[colOffset + j];
        wsGm[colOffset + j] = tmp;
      }
    }
    if (k <= j)
      right = j;
    else
      left = j + 1;
  }
  float median = wsGm[colOffset + k];
  yGm[blockIdx] = median;
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor ca6k_03088_impl_npu(const at::Tensor& self, int64_t dim) {
    // Determine output shape: remove the dim dimension
    auto sizes = self.sizes().vec();
    sizes.erase(sizes.begin() + dim);
    at::Tensor result = at::empty(sizes, self.options());
    EXEC_NPU_CMD(aclnnCa6k03088Custom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_03088_custom", &ca6k_03088_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("ca6k_03088_custom", &ca6k_03088_impl_npu, "Median along dimension (Ca6k03088Custom on NPU)");
}
