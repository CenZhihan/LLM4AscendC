kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00613Custom {
public:
    __aicore__ inline KernelCa6k00613Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                GM_ADDR weight, GM_ADDR bias,
                                uint32_t batchSize, uint32_t numFeatures,
                                uint32_t seqLength, float eps) {
        batchSize_ = batchSize;
        numFeatures_ = numFeatures;
        seqLength_ = seqLength;
        eps_ = eps;
        totalInstances_ = batchSize_ * numFeatures_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInstances_ * seqLength_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalInstances_ * seqLength_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numFeatures_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, numFeatures_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, seqLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, seqLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, seqLength_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalInstances_ == 0 || seqLength_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t instanceIdx = blockIdx; instanceIdx < totalInstances_; instanceIdx += blockNum) {
            uint32_t channel = instanceIdx % numFeatures_;
            float weightVal = weightGm_.GetValue(channel);
            float biasVal = biasGm_.GetValue(channel);

            uint32_t baseOffset = instanceIdx * seqLength_;
            CopyIn(baseOffset, seqLength_);
            Compute(seqLength_, weightVal, biasVal, eps_);
            CopyOut(baseOffset, seqLength_);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len, float weight, float bias, float eps) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

        // Compute sum of x
        float sumX = 0.0f;
        for (uint32_t i = 0; i < len; ++i) {
            sumX += xLocal(i);
        }
        float mean = sumX / static_cast<float>(len);

        // Compute sum of squared differences
        float sumDiffSq = 0.0f;
        for (uint32_t i = 0; i < len; ++i) {
            float diff = xLocal(i) - mean;
            sumDiffSq += diff * diff;
        }
        float var = sumDiffSq / static_cast<float>(len);
        float invStd = 1.0f / sqrt(var + eps);

        // Normalize and apply affine
        for (uint32_t i = 0; i < len; ++i) {
            yLocal(i) = (xLocal(i) - mean) * invStd * weight + bias;
        }

        tmpQueue_.FreeTensor(tmpLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t batchSize_ {0};
    uint32_t numFeatures_ {0};
    uint32_t seqLength_ {0};
    uint32_t totalInstances_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_00613_custom(GM_ADDR x, GM_ADDR y,
                                                        GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00613Custom op;
    op.Init(x, y, weight, bias,
            tiling_data.batchSize, tiling_data.numFeatures,
            tiling_data.seqLength, tiling_data.eps);
    op.Process();
}

model_src = '''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00613_custom'

class ModelNew(nn.Module):
    def __init__(self, num_features, eps, momentum, affine):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.eps = eps
        self.affine = affine
        if affine:
            self.weight = nn.Parameter(torch.ones(num_features, dtype=torch.float))
            self.bias = nn.Parameter(torch.zeros(num_features, dtype=torch.float))
        else:
            self.register_buffer('weight', None)
            self.register_buffer('bias', None)

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        if self.affine:
            return fn(x, self.weight, self.bias, self.eps)
        else:
            # If not affine, use identity weight and zero bias
            dummy_weight = torch.ones(self.num_features, dtype=torch.float, device=x.device)
            dummy_bias = torch.zeros(self.num_features, dtype=torch.float, device=x.device)
            return fn(x, dummy_weight, dummy_bias, self.eps)
