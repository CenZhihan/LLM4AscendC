#include "ca6k_04630_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04630CustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    uint32_t totalElements = 1;
    for (int i = 0; i < xShape->GetDimNum(); ++i) {
        totalElements *= xShape->GetDim(i);
    }
    uint32_t C = xShape->GetDim(1);  // channels
    uint32_t H = xShape->GetDim(2);
    uint32_t W = xShape->GetDim(3);
    uint32_t halfC = C / 2;
    uint32_t spatialSize = H * W;
    uint32_t totalPoints = xShape->GetDim(0) * spatialSize;

    tiling.totalLength = totalPoints;
    tiling.inFeatures = C;
    tiling.halfFeatures = halfC;
    tiling.spatialStride = spatialSize;

    uint32_t maxTile = 1;  // one point per tile for simplicity
    uint32_t tileLen = maxTile;
    if (totalPoints < tileLen) tileLen = totalPoints;

    uint32_t numTiles = CeilDiv(totalPoints, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);
    tiling.blockDim = blockDim;
    tiling.tileLength = tileLen;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    // Output has half channels: adjust dim1
    y_shape->SetDim(1, x_shape->GetDim(1) / 2);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge
