void Process() {
    uint32_t blockNum = GetBlockNum();
    uint32_t blockIdx = GetBlockIdx();
    uint32_t numVectors = totalLength_ / dimSize_;
    for (uint32_t vecIdx = blockIdx; vecIdx < numVectors; vecIdx += blockNum) {
        uint32_t baseOffset = vecIdx * dimSize_;
        // Pass 1: find max
        float maxVal = -FLT_MAX;
        for (uint32_t tileOff = 0; tileOff < dimSize_; tileOff += tileLen_) {
            uint32_t curLen = min(tileLen_, dimSize_ - tileOff);
            // 加载x
            LocalTensor<float> xLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float)); // 需要tileLen_个元素
            DataCopy(xLocal, xGm_[baseOffset + tileOff], curLen);
            // 计算局部max
            LocalTensor<float> localMax = pipe_.AllocTensor<float>(sizeof(float));
            ReduceMax<float, float>(localMax, xLocal, true); // 沿所有元素减少
            float localMaxVal = localMax.GetValue(0);
            if (localMaxVal > maxVal) maxVal = localMaxVal;
            pipe_.FreeTensor(localMax);
            pipe_.FreeTensor(xLocal);
        }
        // pass 2: compute sum
        float sumVal = 0.0f;
        for (uint32_t tileOff = 0; tileOff < dimSize_; tileOff += tileLen_) {
            uint32_t curLen = min(tileLen_, dimSize_ - tileOff);
            LocalTensor<float> xLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
            DataCopy(xLocal, xGm_[baseOffset + tileOff], curLen);
            // yLocal = exp(x - maxVal)
            LocalTensor<float> diff = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
            Sub(diff, xLocal, maxVal, curLen);
            LocalTensor<float> expLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
            Exp(expLocal, diff, curLen);
            LocalTensor<float> localSum = pipe_.AllocTensor<float>(sizeof(float));
            ReduceSum<float, float>(localSum, expLocal, true);
            sumVal += localSum.GetValue(0);
            pipe_.FreeTensor(localSum);
            pipe_.FreeTensor(expLocal);
            pipe_.FreeTensor(diff);
            pipe_.FreeTensor(xLocal);
        }
        // pass 3: normalize and write out
        for (uint32_t tileOff = 0; tileOff < dimSize_; tileOff += tileLen_) {
            uint32_t curLen = min(tileLen_, dimSize_ - tileOff);
            LocalTensor<float> xLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
            DataCopy(xLocal, xGm_[baseOffset + tileOff], curLen);
            LocalTensor<float> diff = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
            Sub(diff, xLocal, maxVal, curLen);
            LocalTensor<float> expLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
            Exp(expLocal, diff, curLen);
            LocalTensor<float> yLocal = pipe_.AllocTensor<float>(tileLen_ * sizeof(float));
            // 除以sum
            Muls(yLocal, expLocal, 1.0f / sumVal, curLen);  // 使用Muls乘倒数
            // 或者使用Div，但Muls更高效
            DataCopy(yGm_[baseOffset + tileOff], yLocal, curLen);
            pipe_.FreeTensor(yLocal);
            pipe_.FreeTensor(expLocal);
            pipe_.FreeTensor(diff);
            pipe_.FreeTensor(xLocal);
        }
    }
}
