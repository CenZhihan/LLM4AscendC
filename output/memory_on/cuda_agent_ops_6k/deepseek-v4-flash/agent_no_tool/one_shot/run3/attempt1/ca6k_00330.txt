host_operator_src="""
#include "ca6k_00330_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00330CustomTilingData tiling;

    const gert::StorageShape *mat1Shape = context->GetInputShape(0);
    const gert::StorageShape *mat2Shape = context->GetInputShape(1);
    const gert::StorageShape *biasShape = context->GetInputShape(2);

    uint32_t batch = mat1Shape->GetOriginShape().GetDim(0);
    uint32_t outputDim = mat1Shape->GetOriginShape().GetDim(1);
    uint32_t reductionDim = mat1Shape->GetOriginShape().GetDim(2);
    uint32_t inputDim = mat2Shape->GetOriginShape().GetDim(2);

    uint32_t totalElements = batch * outputDim * inputDim;

    // Compute blockSize: each output element requires 2 * reductionDim floats in UB (mat1 row + mat2 col)
    uint32_t bytesPerElement = 2 * reductionDim * sizeof(float);
    uint32_t maxElements = UB_BUDGET_BYTES / bytesPerElement;
    if (maxElements < 1) maxElements = 1;
    // Allow multiple elements per block to reduce overhead
    uint32_t blockSize = maxElements * 8;  // batch 8 elements per block
    if (blockSize > totalElements) blockSize = totalElements;
    // Align to ALIGN_ELEMS for better DMA
    blockSize = (blockSize / ALIGN_ELEMS) * ALIGN_ELEMS;
    if (blockSize == 0) blockSize = ALIGN_ELEMS;
    if (blockSize > totalElements) blockSize = totalElements;

    uint32_t numTiles = CeilDiv(totalElements, blockSize);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.totalElements = totalElements;
    tiling.reductionDim = reductionDim;
    tiling.outputDim = outputDim;
    tiling.inputDim = inputDim;
    tiling.batch = batch;
    tiling.blockSize = blockSize;
    tiling.mat1StrideBatch = outputDim * reductionDim;
    tiling.mat2StrideBatch = reductionDim * inputDim;
    tiling.biasStrideBatch = outputDim * inputDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *biasShape = context->GetInputShape(2);
    gert::Shape *yShape = context->GetOutputShape(0);
    *yShape = *biasShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    ge::DataType dt = context->GetInputDataType(0);
    context->SetOutputDataType(0, dt);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

class KernelBaddbmmCustom {
public:
    __aicore__ inline KernelBaddbmmCustom() {}

    __aicore__ inline void Init(GM_ADDR mat1, GM_ADDR mat2, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalElements, uint32_t reductionDim,
                                uint32_t outputDim, uint32_t inputDim,
                                uint32_t batch, uint32_t blockSize,
                                uint32_t mat1StrideBatch, uint32_t mat2StrideBatch,
                                uint32_t biasStrideBatch) {
        totalElements_ = totalElements;
        reductionDim_ = reductionDim;
        outputDim_ = outputDim;
        inputDim_ = inputDim;
        batch_ = batch;
        blockSize_ = blockSize;
        mat1StrideBatch_ = mat1StrideBatch;
        mat2StrideBatch_ = mat2StrideBatch;
        biasStrideBatch_ = biasStrideBatch;

        mat1Gm_.SetGlobalBuffer((__gm__ float*)mat1);
        mat2Gm_.SetGlobalBuffer((__gm__ float*)mat2);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(mat1Buf_, 1, reductionDim_ * sizeof(float));
        pipe_.InitBuffer(mat2Buf_, 1, reductionDim_ * sizeof(float));
        pipe_.InitBuffer(prodBuf_, 1, reductionDim_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalElements_ == 0) return;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t startIdx = blockIdx * blockSize_;
        uint32_t endIdx = startIdx + blockSize_;
        if (endIdx > totalElements_) endIdx = totalElements_;

        for (uint32_t idx = startIdx; idx < endIdx; ++idx) {
            // decode batch, i, j
            uint32_t rem = idx;
            uint32_t b = rem / (outputDim_ * inputDim_);
            rem %= (outputDim_ * inputDim_);
            uint32_t i = rem / inputDim_;
            uint32_t j = rem % inputDim_;

            // compute offsets
            uint32_t mat1Off = b * mat1StrideBatch_ + i * reductionDim_;
            uint32_t mat2Off = b * mat2StrideBatch_ + j;
            uint32_t biasOff = b * biasStrideBatch_ + i * inputDim_ + j;

            // prefetch mat1 row
            AscendC::LocalTensor<float> localMat1 = mat1Buf_.AllocTensor<float>();
            AscendC::DataCopy(localMat1, mat1Gm_[mat1Off], reductionDim_);

            // prefetch mat2 column
            AscendC::LocalTensor<float> localMat2 = mat2Buf_.AllocTensor<float>();
            AscendC::DataCopy(localMat2, mat2Gm_[mat2Off], reductionDim_);

            // element-wise product
            AscendC::LocalTensor<float> prod = prodBuf_.AllocTensor<float>();
            AscendC::Mul(prod, localMat1, localMat2, reductionDim_);

            // reduction sum (scalar loop)
            float sum = 0.0f;
            for (uint32_t k = 0; k < reductionDim_; ++k) {
                sum += prod(k);
            }

            // add bias
            sum += biasGm_[biasOff];

            // store output
            yGm_[biasOff] = sum;

            // free local tensors
            prodBuf_.FreeTensor(prod);
            mat2Buf_.FreeTensor(localMat2);
            mat1Buf_.FreeTensor(localMat1);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> mat1Buf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> mat2Buf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> prodBuf_;

    AscendC::GlobalTensor<float> mat1Gm_;
    AscendC::GlobalTensor<float> mat2Gm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalElements_ {0};
    uint32_t reductionDim_ {0};
    uint32_t outputDim_ {0};
    uint32_t inputDim_ {0};
    uint32_t batch_ {0};
    uint32_t blockSize_ {0};
    uint32_t mat1StrideBatch_ {0};
    uint32_t mat2StrideBatch_ {0};
    uint32_t biasStrideBatch_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00330_custom(GM_ADDR mat1, GM_ADDR mat2, GM_ADDR bias,
                                                        GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBaddbmmCustom op;
    op.Init(mat1, mat2, bias, y,
            tiling_data.totalElements, tiling_data.reductionDim,
            tiling_data.outputDim, tiling_data.inputDim,
            tiling_data.batch, tiling_data.blockSize,
            tiling_data.mat1StrideBatch, tiling_data.mat2StrideBatch,
            tiling_data.biasStrideBatch);
    op.Process();
}
