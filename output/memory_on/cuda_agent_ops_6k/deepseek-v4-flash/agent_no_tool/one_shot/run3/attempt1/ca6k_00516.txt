host_operator_src = r'''
#include "ca6k_00516_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB size
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_SPATIAL = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00516CustomTilingData tiling;

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const std::vector<int64_t> dims = inputShape->GetOriginShape().GetDims();
    uint32_t N = static_cast<uint32_t>(dims[0]);
    uint32_t C = static_cast<uint32_t>(dims[1]);
    uint32_t H = static_cast<uint32_t>(dims[2]);
    uint32_t W = static_cast<uint32_t>(dims[3]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t numGroups = attrs->GetAttrPointer<int64_t>(0)[0];
    float epsilon = attrs->GetAttrPointer<float>(1)[0];

    // If numGroups is not set (default 1), use attribute default
    // but we treat it as given by user.

    uint32_t groupChannels = C / static_cast<uint32_t>(numGroups);
    uint32_t spatialSize = H * W;
    uint32_t totalElements = N * C * spatialSize;

    // Compute tileSpatial: number of spatial positions per tile
    // Per tile we need: input (groupChannels * tileSpatial floats), plus temporary accumulators.
    // Conservatively assume we need space for groupChannels * tileSpatial * 4 bytes (input) + some overhead.
    uint32_t maxTileSpatial = UB_BUDGET_BYTES / (groupChannels * sizeof(float) + 64);
    maxTileSpatial = AlignDown(maxTileSpatial, ALIGN_ELEMS);
    if (maxTileSpatial < ALIGN_ELEMS) maxTileSpatial = ALIGN_ELEMS;
    if (maxTileSpatial > HARD_CAP_SPATIAL) maxTileSpatial = AlignDown(HARD_CAP_SPATIAL, ALIGN_ELEMS);
    if (maxTileSpatial > spatialSize) maxTileSpatial = AlignDown(spatialSize, ALIGN_ELEMS);
    if (maxTileSpatial == 0) maxTileSpatial = spatialSize;

    uint32_t tileSpatial = maxTileSpatial;

    // Set block dim: we can use multiple blocks per batch or per group, but for simplicity use 1 block.
    context->SetBlockDim(1);

    tiling.batchSize = N;
    tiling.numChannels = C;
    tiling.height = H;
    tiling.width = W;
    tiling.numGroups = static_cast<uint32_t>(numGroups);
    tiling.groupChannels = groupChannels;
    tiling.spatialSize = spatialSize;
    tiling.tileSpatial = tileSpatial;
    tiling.epsilon = epsilon;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShapeCa6k00516Custom(gert::InferShapeContext *context) {
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    gert::StorageShape *outputShape = context->GetOutputShape(0);
    *outputShape = *inputShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataTypeCa6k00516Custom(gert::InferDataTypeContext *context) {
    ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src = r'''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00516Custom {
public:
    __aicore__ inline KernelCa6k00516Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                const Ca6k00516CustomTilingData *tiling) {
        xGm_.SetGlobalBuffer((__gm__ float *)x);
        yGm_.SetGlobalBuffer((__gm__ float *)y);
        batchSize_ = tiling->batchSize;
        numChannels_ = tiling->numChannels;
        height_ = tiling->height;
        width_ = tiling->width;
        numGroups_ = tiling->numGroups;
        groupChannels_ = tiling->groupChannels;
        spatialSize_ = tiling->spatialSize;
        tileSpatial_ = tiling->tileSpatial;
        epsilon_ = tiling->epsilon;

        // Align tileLen for data copy
        tileLen_ = groupChannels_ * tileSpatial_;
        // UB buffer for input tile
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        // UB buffer for output tile
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // Temporary buffer for intermediate sums (groupChannels float)
        pipe_.InitBuffer(tmpSumBuf_, 1, groupChannels_ * sizeof(float));
        pipe_.InitBuffer(tmpSumSqBuf_, 1, groupChannels_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (batchSize_ == 0 || spatialSize_ == 0 || groupChannels_ == 0) return;

        for (uint32_t n = 0; n < batchSize_; n++) {
            for (uint32_t g = 0; g < numGroups_; g++) {
                // First pass: compute sum and sum of squares for this group
                float sum = 0.0f;
                float sumSq = 0.0f;

                for (uint32_t sp = 0; sp < spatialSize_; sp += tileSpatial_) {
                    uint32_t curTileSpatial = tileSpatial_;
                    if (sp + curTileSpatial > spatialSize_) {
                        curTileSpatial = spatialSize_ - sp;
                    }
                    uint32_t curTileLen = groupChannels_ * curTileSpatial;

                    // Load input tile from GM to UB
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    uint32_t gmOffset = (n * numChannels_ * spatialSize_) + (g * groupChannels_ * spatialSize_) + sp;
                    // Because groupChannels contiguous in memory, we can copy whole tile
                    AscendC::DataCopy(xLocal, xGm_[gmOffset], curTileLen);

                    // Accumulate sum and sumSq over the tile
                    AscendC::LocalTensor<float> tmpSum = tmpSumBuf_.Get<float>();
                    AscendC::LocalTensor<float> tmpSumSq = tmpSumSqBuf_.Get<float>();

                    // For each channel in group, we could sum across spatial within the tile
                    // But easier: flatten and reduce over all elements of tile
                    // Use AscendC vector reduce
                    // Since reduction over whole tile is not directly available, we accumulate manually
                    // Better: use ReduceSum on the tile? AscendC does not have vector-level reduction.
                    // We'll do element-wise accumulation in a loop (suboptimal but works for small group)
                    // Alternatively, we can sum across channels first, then spatial, but that is complex.
                    // Given small groupChannels (likely 1-4), we can just loop over all elements.
                    for (uint32_t i = 0; i < curTileLen; i++) {
                        float val = xLocal(i);
                        sum += val;
                        sumSq += val * val;
                    }

                    inQueueX_.FreeTensor(xLocal);
                }

                // Compute mean and variance
                uint32_t groupElements = groupChannels_ * spatialSize_;
                float mean = sum / static_cast<float>(groupElements);
                float var = sumSq / static_cast<float>(groupElements) - mean * mean;
                float invStd = 1.0f / sqrtf(var + epsilon_);

                // Second pass: normalize and write output
                for (uint32_t sp = 0; sp < spatialSize_; sp += tileSpatial_) {
                    uint32_t curTileSpatial = tileSpatial_;
                    if (sp + curTileSpatial > spatialSize_) {
                        curTileSpatial = spatialSize_ - sp;
                    }
                    uint32_t curTileLen = groupChannels_ * curTileSpatial;

                    // Load input again
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    uint32_t gmOffset = (n * numChannels_ * spatialSize_) + (g * groupChannels_ * spatialSize_) + sp;
                    AscendC::DataCopy(xLocal, xGm_[gmOffset], curTileLen);

                    // Compute output
                    AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                    // y = (x - mean) * invStd
                    AscendC::Sub(yLocal, xLocal, mean, curTileLen);
                    AscendC::Mul(yLocal, yLocal, invStd, curTileLen);

                    // Write to GM
                    AscendC::DataCopy(yGm_[gmOffset], yLocal, curTileLen);

                    outQueueY_.FreeTensor(yLocal);
                    inQueueX_.FreeTensor(xLocal);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSumBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSumSqBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_;
    uint32_t numChannels_;
    uint32_t height_;
    uint32_t width_;
    uint32_t numGroups_;
    uint32_t groupChannels_;
    uint32_t spatialSize_;
    uint32_t tileSpatial_;
    uint32_t tileLen_;
    float epsilon_;
};

extern "C" __global__ __aicore__ void ca6k_00516_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00516Custom op;
    op.Init(x, y, &tiling_data);
    op.Process();
}
'''

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00516_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        # init_inputs: [num_channels, num_groups, num_channels_per_group]
        # We only need num_groups for the custom op
        self.num_groups = init_inputs[1]

    def forward(self, *inputs):
        fn = getattr(custom_ops_lib, _FN)
        x = inputs[0]
        return fn(x, self.num_groups, 1e-5)
