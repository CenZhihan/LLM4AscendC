    uint32_t L = xShape->GetDim(2);
    uint32_t O = wShape->GetDim(0);
    uint32_t K = wShape->GetDim(2);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t stride = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t dilation = attrs->GetAttrPointer<int32_t>(2)[0];
    uint32_t groups = static_cast<uint32_t>(attrs->GetAttrPointer<int32_t>(3)[0]);

    int32_t L_out = (L + 2 * padding - dilation * (static_cast<int32_t>(K) - 1) - 1) / stride + 1;
    uint32_t totalInput = N * C * L;
    uint32_t totalOutput = N * O * static_cast<uint32_t>(L_out);

    // UB budget: 192KB, each output element needs: input slice (C*K), weight slice (C*K), prod, acc => 4 * C*K * 4 bytes
    uint32_t sliceElems = C * K;
    uint32_t ubBytesPerOutput = 4 * sliceElems * sizeof(float); // input, weight, prod, acc
    uint32_t maxTile = (192 * 1024) / ubBytesPerOutput;
    if (maxTile > 1024) maxTile = 1024;
    if (maxTile < 1) maxTile = 1;

    uint32_t tileLength = maxTile;

    uint32_t blockDim = context->GetSocInfo()->GetCoreNum();

    if (blockDim < 8) blockDim = 8;

    tiling.set_C(C);
    tiling.set_L(L);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_dilation(dilation);
    tiling.set_groups(groups);

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01116Custom {
public:
    __aicore__ inline KernelCa6k01116Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t L, uint32_t O, uint32_t K,
                                int32_t stride, int32_t padding, int32_t dilation, uint32_t groups,
                                uint32_t L_out, uint32_t totalInput, uint32_t totalOutput,
                                uint32_t tileLength)
    {
        N_ = N;
        C_ = C;
        L_ = L;
        O_ = O;
        K_ = K;
        stride_ = stride;
        padding_ = padding;
        dilation_ = dilation;
        groups_ = groups;
        L_out_ = L_out;
        totalInput_ = totalInput;
        totalOutput_ = totalOutput;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInput_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, O_ * (C_ / groups_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutput_);

        uint32_t sliceElems = C_ * K_ / groups_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sliceElems * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * sliceElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, sliceElems * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutput_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid stride over output elements
        uint32_t totalTiles = (totalOutput_ + tileLen_ - 1) / tileLen_;
        uint32_t tilesPerBlock = (totalTiles + blockNum - 1) / blockNum;

        uint32_t startTile = blockIdx * tilesPerBlock;
        if (startTile >= totalTiles) return;
        uint32_t endTile = startTile + tilesPerBlock;
        if (endTile > totalTiles) endTile = totalTiles;

        for (uint32_t tile = startTile; tile < endTile; ++tile) {
            uint32_t base = tile * tileLen_;
            uint32_t curLen = totalOutput_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Prefetch first output tile
            Prefetch(base, curLen);

            // Compute and store
            Compute(base, curLen);

            // Wait for stores? Store is blocking? We'll keep simple double buffering.
        }
    }

private:
    __aicore__ inline void Prefetch(uint32_t outputStart, uint32_t len)
    {
        // For a tile of output indices, we need to load corresponding input slices and weight slices
        // We'll load input slice and weight slice for each output element in a loop
        // To simplify, we load iteratively in Compute loop, using double-buffer.
        // This is a placeholder; actual double-buffer loading is done inside Compute using manual pipeline
        // We'll implement a separate load function for the first tile.
        // For simplicity, we load inside Compute using single buffer.
    }

    __aicore__ inline void Compute(uint32_t outputStart, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t globalOutIdx = outputStart + i;
            // Compute indices
            uint32_t n = globalOutIdx / (O_ * L_out_);
            uint32_t rem = globalOutIdx % (O_ * L_out_);
            uint32_t o = rem / L_out_;
            uint32_t l_out = rem % L_out_;

            // Input start position for this output
            int32_t inputStart = static_cast<int32_t>(l_out) * stride_ - padding_;
            if (inputStart < 0 || inputStart + (static_cast<int32_t>(K_) - 1) * dilation_ >= static_cast<int32_t>(L_)) {
                // out of bounds, should not happen with valid padding? but handle with zero
                // actual conv: if inputStart<0, we need pad zeros, but with padding=0, inputStart should be non-negative.
                // Still handle safely.
                // For simplicity, assume valid
            }

            // Load bias
            float biasVal = 0.0f;
            if (bGm_.GetValue(o) != 0.0f) { // check if bias is present (we treat as non-zero if bias address valid)
                // bias might be present as input; we check via global pointer, but easier: use attribute? Hardcode.
            }
            // Actually we will add bias later; we load bias from per-channel address
            biasVal = bGm_.GetValue(o);

            AscendC::LocalTensor<float> xLocal = pipe_.LocalAlloc<float>(inQueueX_, C_ * K_ / groups_);
            AscendC::LocalTensor<float> wLocal = pipe_.LocalAlloc<float>(inQueueW_, C_ * K_ / groups_);
            AscendC::LocalTensor<float> tmp = pipe_.LocalAlloc<float>(tmpQueue_, C_ * K_ / groups_);
            AscendC::LocalTensor<float> acc = pipe_.LocalAlloc<float>(accQueue_, 1);
            acc.SetValue(0, biasVal);

            // Load input slice for one output element
            // Input shape: [n, c, inputStart + d*k]
            for (uint32_t c = 0; c < C_; ++c) {
                for (uint32_t k = 0; k < K_; ++k) {
                    int32_t idx = inputStart + static_cast<int32_t>(k) * dilation_;
                    float val = 0.0f;
                    if (idx >= 0 && idx < static_cast<int32_t>(L_)) {
                        val = xGm_.GetValue(n * C_ * L_ + c * L_ + static_cast<uint32_t>(idx));
                    }
                    xLocal.SetValue(c * K_ + k, val);
                }
            }

            // Load weight slice for output channel o
            for (uint32_t c = 0; c < C_; ++c) {
                for (uint32_t k = 0; k < K_; ++k) {
                    float val = wGm_.GetValue(o * (C_ / groups_) * K_ + c * K_ + k);
                    wLocal.SetValue(c * K_ + k, val);
                }
            }

            // Multiply-accumulate
            uint32_t sliceElems = C_ * K_ / groups_;
            for (uint32_t j = 0; j < sliceElems; ++j) {
                float xv = xLocal.GetValue(j);
                float wv = wLocal.GetValue(j);
                float prod = xv * wv;
                acc.SetValue(0, acc.GetValue(0) + prod);
            }

            // Write output
            yGm_.SetValue(globalOutIdx, acc.GetValue(0));

            pipe_.LocalFree(xLocal);
            pipe_.LocalFree(wLocal);
            pipe_.LocalFree(tmp);
            pipe_.LocalFree(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t L_ {0};
    uint32_t O_ {0};
    uint32_t K_ {0};
    int32_t stride_ {1};
    int32_t padding_ {0};
    int32_t dilation_ {1};
    uint32_t groups_ {1};
    uint32_t L_out_ {0};
    uint32_t totalInput_ {0};
    uint32_t totalOutput_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_01116_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01116Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.C, tiling_data.L, tiling_data.O, tiling_data.K,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation, tiling_data.groups,
            tiling_data.L_out, tiling_data.totalInput, tiling_data.totalOutput,
            tiling_data.tileLength);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>

at::Tensor ca6k_01116_npu(const at::Tensor& self, const at::Tensor& weight,
                          c10::optional<at::Tensor> bias,
                          int64_t stride, int64_t padding, int64_t dilation, int64_t groups) {
    at::Tensor result = at::empty_like(self); // shape will be set by op
    if (bias.has_value()) {
        EXEC_NPU_CMD(aclnnCa6k01116Custom, self, weight, bias.value(), result, stride, padding, dilation, groups);
    } else {
        at::Tensor emptyBias = at::tensor({}, self.options());
        EXEC_NPU_CMD(aclnnCa6k01116Custom, self, weight, emptyBias, result, stride, padding, dilation, groups);
    }
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_01116_custom", &ca6k_01116_npu);
}

}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0, dilation=1, groups=1):
        super(ModelNew, self).__init__()
        self.weight = torch.nn.Parameter(torch.randn(out_channels, in_channels // groups, kernel_size))
        self.bias = torch.nn.Parameter(torch.zeros(out_channels))
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups

    def forward(self, x):
        fn = getattr(custom_ops_lib, 'ca6k_01116_custom')
        return fn(x, self.weight, self.bias, self.stride, self.padding, self.dilation, self.groups)
'''
