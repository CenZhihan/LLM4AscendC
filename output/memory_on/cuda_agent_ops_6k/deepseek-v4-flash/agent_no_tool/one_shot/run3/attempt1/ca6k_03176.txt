host_operator_src = '''
#include "ca6k_03176_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MAX_TILE_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k03176CustomTilingData tiling;

    const gert::StorageShape *xShape = context->GetInputShape(0);
    uint32_t totalLength = 1;
    for (uint32_t d = 0; d < xShape->GetDimNum(); ++d)
        totalLength *= static_cast<uint32_t>(xShape->GetDim(d));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += xShape->GetDimNum();
    uint32_t rowSize = static_cast<uint32_t>(xShape->GetDim(dim));
    uint32_t columns = totalLength / rowSize;

    uint32_t tileLen = UB_BUDGET_BYTES / (2 * sizeof(float));  // two buffers: values + indices? we only need values
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen > MAX_TILE_ELEMS) tileLen = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;
    if (tileLen > rowSize) tileLen = AlignDown(rowSize, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = rowSize;

    uint32_t blockDim = columns;  // one block per column
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.totalLength = totalLength;
    tiling.columns = columns;
    tiling.rowSize = rowSize;
    tiling.tileLength = tileLen;
    tiling.dim = dim;
    tiling.blockDim = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += x_shape->GetDimNum();

    // Output values shape: same as x but with dimension dim removed
    gert::Shape values_shape;
    for (uint32_t d = 0; d < x_shape->GetDimNum(); ++d) {
        if (static_cast<int32_t>(d) != dim) {
            values_shape.SetDim(values_shape.GetDimNum(), x_shape->GetDim(d));
        }
    }
    // If dim was the only dimension, output shape is {1}
    if (values_shape.GetDimNum() == 0) {
        values_shape.SetDim(0, 1);
    }
    context->SetOutputShape(0, values_shape);

    // Indices shape same as values
    context->SetOutputShape(1, values_shape);

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_FLOAT);
    context->SetOutputDataType(1, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k03176Custom {
public:
    __aicore__ inline KernelCa6k03176Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR values, GM_ADDR indices,
                                uint32_t totalLength, uint32_t columns, uint32_t rowSize,
                                uint32_t tileLength, int32_t dim)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        valuesGm_.SetGlobalBuffer((__gm__ float*)values, columns);
        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices, columns);
        columns_ = columns;
        rowSize_ = rowSize;
        tileLen_ = tileLength;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (columns_ == 0 || rowSize_ == 0) return;
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= columns_) return;

        // Each block handles one column (blockIdx corresponds to column index)
        uint32_t colStart = blockIdx * rowSize_;

        // Step 1: count non-NaN elements
        uint32_t nonNanCount = 0;
        for (uint32_t offset = 0; offset < rowSize_; offset += tileLen_) {
            uint32_t curLen = (offset + tileLen_ > rowSize_) ? rowSize_ - offset : tileLen_;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[colStart + offset], curLen);
            // We'll count later in this loop; for now just count using scalar loop
            // Simple approach: dequeue and scan each element
            inQueueX_.EnQue(xLocal);
            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
            for (uint32_t i = 0; i < curLen; ++i) {
                float val = xDeque.GetValue(i);
                if (!AscendC::IsNan(val)) {
                    nonNanCount++;
                }
            }
            inQueueX_.FreeTensor(xDeque);
        }

        if (nonNanCount == 0) {
            // All NaN: set median to 0 and index to 0
            valuesGm_.SetValue(blockIdx, 0.0f);
            indicesGm_.SetValue(blockIdx, 0);
            return;
        }

        // k is zero-based index of median (lower middle for even count)
        uint32_t k = (nonNanCount - 1) / 2;

        // Quickselect to find k-th smallest among non-NaN elements
        // We'll loop until we find the exact element
        // Search range indices: we consider all positions [0, rowSize_)
        int64_t lowIndex = 0;
        int64_t highIndex = static_cast<int64_t>(rowSize_) - 1;
        float medianVal = 0.0f;
        int64_t medianIdx = 0;

        while (lowIndex <= highIndex) {
            // Choose pivot as element at lowIndex (skip if NaN)
            // But we need pivot from the column. We'll read pivot value.
            // Actually, we need to handle NaNs. We'll skip NaNs during counting.
            // We'll implement a standard quickselect that partitions based on value,
            // and we only count non-NaN elements. We'll scan the entire column each pass.

            // For simplicity, pick pivot from the middle of the range
            int64_t pivotPos = lowIndex + (highIndex - lowIndex) / 2;
            float pivotVal;
            // Read pivot value
            {
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                AscendC::DataCopy(tmp, xGm_[colStart + pivotPos], 1);
                pivotVal = tmp.GetValue(0);
                tmpQueue_.FreeTensor(tmp);
            }

            // Count non-NaN elements less than pivot, and total non-NaN in range
            uint32_t lessCount = 0;
            uint32_t eqCount = 0;
            int64_t eqPos = -1;
            for (int64_t pos = lowIndex; pos <= highIndex; ++pos) {
                // Read element
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                AscendC::DataCopy(tmp, xGm_[colStart + pos], 1);
                float val = tmp.GetValue(0);
                tmpQueue_.FreeTensor(tmp);
                if (AscendC::IsNan(val)) continue;
                if (val < pivotVal) {
                    lessCount++;
                } else if (val == pivotVal) {
                    eqCount++;
                    eqPos = pos;
                }
            }

            if (k < lessCount) {
                // median in left part (values < pivot)
                highIndex = pivotPos - 1;
                // But careful: we need to adjust low/high to the actual range of non-NaN values?
                // Since we are scanning the whole column, we only consider positions in [lowIndex, highIndex].
                // So we can continue with narrowed range.
            } else if (k < lessCount + eqCount) {
                // median is pivot value
                medianVal = pivotVal;
                medianIdx = eqPos;
                break;
            } else {
                k -= (lessCount + eqCount);
                lowIndex = pivotPos + 1;
            }
        }

        // After loop, we have medianVal and medianIdx
        valuesGm_.SetValue(blockIdx, medianVal);
        indicesGm_.SetValue(blockIdx, static_cast<int64_t>(colStart + medianIdx));
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> valuesGm_;
    AscendC::GlobalTensor<int64_t> indicesGm_;

    uint32_t columns_;
    uint32_t rowSize_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void ca6k_03176_custom(GM_ADDR x, GM_ADDR values, GM_ADDR indices,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k03176Custom op;
    op.Init(x, values, indices,
            tiling_data.totalLength, tiling_data.columns, tiling_data.rowSize,
            tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
'''
