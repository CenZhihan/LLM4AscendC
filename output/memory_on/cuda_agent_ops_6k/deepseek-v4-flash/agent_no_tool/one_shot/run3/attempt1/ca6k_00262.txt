    tiling.batch = batch;

    tiling.W = W;

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t ALIGN_ELEMS = 2048;

class KernelCa6k00262Custom {
public:
    __aicore__ inline KernelCa6k00262Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR src, GM_ADDR y,
                                uint32_t totalLength, uint32_t batch,
                                uint32_t H, uint32_t W,
                                uint32_t diagLen, int32_t offset)
    {
        totalLen_ = totalLength;
        batch_ = batch;
        H_ = H;
        W_ = W;
        diagLen_ = diagLen;
        offset_ = offset;
        tileLen_ = 0;

        // compute tile length from UB budget
        constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
        constexpr uint32_t ELEM_SIZE = sizeof(float);
        // We need at most two tiles (one for copy, one for src) plus a temporary
        uint32_t maxTile = UB_BUDGET_BYTES / (3 * ELEM_SIZE);
        maxTile = (maxTile / ALIGN_ELEMS) * ALIGN_ELEMS;
        if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
        tileLen_ = maxTile;

        uint32_t totalPerBatch = H_ * W_;
        if (tileLen_ > totalPerBatch) {
            tileLen_ = totalPerBatch;
        }

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        srcGm_.SetGlobalBuffer((__gm__ float*)src, batch_ * diagLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalPerBatch = H_ * W_;

        // grid-stride over batches
        for (uint32_t b = blockIdx; b < batch_; b += blockNum) {
            uint32_t batchOffset = b * totalPerBatch;
            uint32_t diagOffset = b * diagLen_;

            // Copy entire batch from x to y in tiles
            uint32_t remaining = totalPerBatch;
            uint32_t tileBase = batchOffset;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                // Align down to ALIGN_ELEMS for efficiency
                curLen = (curLen / ALIGN_ELEMS) * ALIGN_ELEMS;
                if (curLen == 0) curLen = ALIGN_ELEMS;
                if (curLen > remaining) curLen = remaining;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[tileBase], curLen);
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, xLocal, curLen);  // copy through local to output
                AscendC::DataCopy(yGm_[tileBase], yLocal, curLen);

                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);

                tileBase += curLen;
                remaining -= curLen;
            }

            // Overwrite diagonal elements with src values
            for (uint32_t i = 0; i < diagLen_; i++) {
                int32_t col = static_cast<int32_t>(i) + offset_;
                if (col >= 0 && static_cast<uint32_t>(col) < W_) {
                    uint32_t outIdx = batchOffset + i * W_ + static_cast<uint32_t>(col);
                    uint32_t srcIdx = diagOffset + i;
                    // Write directly to global memory (no vectorization needed for small loop)
                    yGm_.SetValue(outIdx, srcGm_.GetValue(srcIdx));
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> srcGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t batch_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t diagLen_ {0};
    int32_t offset_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00262_custom(GM_ADDR x, GM_ADDR src, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00262Custom op;
    op.Init(x, src, y,
            tiling_data.totalLength, tiling_data.batch,
            tiling_data.H, tiling_data.W,
            tiling_data.diagLen, tiling_data.offset);
    op.Process();
}
