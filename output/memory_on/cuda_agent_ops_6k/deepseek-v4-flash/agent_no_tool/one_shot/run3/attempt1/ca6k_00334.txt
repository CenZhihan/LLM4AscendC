host_operator_src = r"""
#include "ca6k_00334_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00334CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim_ptr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dim_ptr != nullptr) ? *dim_ptr : 0;

    // Assume dim=0 for contiguous processing
    const gert::Shape *shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t dimSize = static_cast<uint32_t>(shape->GetDim(static_cast<size_t>(dim)));

    // UB budget for one tile: each iteration uses 2 buffers (input + output) of tileLength floats
    // plus temporary for sequential cumprod (1 * tileLength floats)
    // total floats = 3 * tileLength => bytes = 12 * tileLength
    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // 16384 floats
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = (totalLength < maxTile) ? AlignDown(totalLength, ALIGN_ELEMS) : maxTile;
    if (tileLen == 0) tileLen = totalLength;

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.totalLength = totalLength;
    tiling.tileLength = tileLen;
    tiling.blockDim = blockDim;
    tiling.dimSize = dimSize;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    context->SetOutputDataType(0, x_dtype);

kernel_src = r"""
#include "kernel_operator.h"
#include "ca6k_00334_custom_tiling.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00334Custom {
public:
    __aicore__ inline KernelCa6k00334Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t dimSize) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        dimSize_ = dimSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        // Each block processes slices along dim=0 (contiguous)
        // slice length = dimSize, number of slices = totalLen / dimSize
        uint32_t numSlices = totalLen_ / dimSize_;

        // Grid-stride over slices
        for (uint32_t slice = blockIdx; slice < numSlices; slice += blockNum) {
            uint32_t base = slice * dimSize_;
            float cumulative = 1.0f;

            // Process the slice in tiles
            uint32_t offset = 0;
            while (offset < dimSize_) {
                uint32_t curLen = (dimSize_ - offset) > tileLen_ ? tileLen_ : (dimSize_ - offset);
                // Prefetch next tile (double buffering not needed for sequential per-slice)
                // but we use simple single tile
                CopyIn(base + offset, curLen);
                Compute(cumulative, curLen);
                CopyOut(base + offset, curLen);
                offset += curLen;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(float &cumulative, uint32_t len) {
        LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Compute cumulative product on the tile
        for (uint32_t i = 0; i < len; i++) {
            float val = xLocal(i);
            yLocal(i) = cumulative * val;
            cumulative *= val;
        }

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
        LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
    uint32_t dimSize_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00334_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00334Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim, tiling_data.dimSize);
    op.Process();
}
"""
