#include "ca6k_00333_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static uint32_t GetTilePixels(uint32_t C) {
    uint32_t pixelCost = 2 * C * 4 + C * 4; // in+out + scratch (approx)
    uint32_t maxPixels = UB_BUDGET_BYTES / pixelCost;
    maxPixels = AlignDown(maxPixels, ALIGN_ELEMS);
    if (maxPixels < 1) maxPixels = 1;
    return maxPixels;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00333CustomTilingData tiling;
    const uint32_t totalElements =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *sizePtr = attrs->GetAttrPointer<int32_t>(0);
    const float *alphaPtr = attrs->GetAttrPointer<float>(1);
    const float *betaPtr = attrs->GetAttrPointer<float>(2);
    const float *kPtr = attrs->GetAttrPointer<float>(3);

    // Determine C from shape (N, C, H, W)
    const gert::Shape *shape = context->GetInputShape(0);
    uint32_t C = static_cast<uint32_t>(shape->GetDim(1)); // channel dim
    uint32_t H = static_cast<uint32_t>(shape->GetDim(2));
    uint32_t W = static_cast<uint32_t>(shape->GetDim(3));
    uint32_t totalPixels = totalElements / C;

    uint32_t tilePixels = GetTilePixels(C);
    if (tilePixels > totalPixels) {
        tilePixels = AlignDown(totalPixels, ALIGN_ELEMS);
        if (tilePixels < 1) tilePixels = totalPixels;
    }
    uint32_t tileElements = tilePixels * C;

    uint32_t numTiles = CeilDiv(totalPixels, tilePixels);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.totalElements = totalElements;
    tiling.tileElements = tileElements;
    tiling.blockDim = blockDim;
    tiling.C = C;
    tiling.size = *sizePtr;
    tiling.alpha = *alphaPtr;
    tiling.beta = *betaPtr;
    tiling.k = *kPtr;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = r'''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00333Custom {
public:
    __aicore__ inline KernelCa6k00333Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalElements, uint32_t tileElements,
                                uint32_t blockDim, uint32_t C,
                                int32_t size, float alpha, float beta, float k) {
        totalElements_ = totalElements;
        tileElements_ = tileElements;
        C_ = C;
        size_ = size;
        alpha_ = alpha;
        beta_ = beta;
        k_ = k;
        halfWindow_ = size_ / 2;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElements_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElements_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileElements_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileElements_ * sizeof(float));
        // scratch buffer for one pixel's squares (size C)
        pipe_.InitBuffer(tmpQueue_, 1, C_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalElements_ == 0 || tileElements_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t tilePixels = tileElements_ / C_;
        uint32_t stridePixels = blockNum * tilePixels;
        uint32_t basePixel = blockIdx * tilePixels;
        if (basePixel * C_ >= totalElements_) return;

        // Prefetch first tile
        uint32_t curPixels = 0;
        uint32_t curBase = basePixel * C_;
        if (curBase < totalElements_) {
            curPixels = (totalElements_ - curBase) / C_;
            if (curPixels > tilePixels) curPixels = tilePixels;
            CopyIn(curBase, curPixels * C_);
        } else {
            return;
        }

        while (true) {
            uint32_t nextBase = (basePixel + stridePixels) * C_;
            uint32_t nextPixels = 0;
            if (nextBase < totalElements_) {
                nextPixels = (totalElements_ - nextBase) / C_;
                if (nextPixels > tilePixels) nextPixels = tilePixels;
                CopyIn(nextBase, nextPixels * C_);
            }

            Compute(curPixels);
            CopyOut(curBase, curPixels * C_);

            if (nextPixels == 0) break;
            curPixels = nextPixels;
            curBase = nextBase;
            basePixel += stridePixels;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t len) {
        AscendC::LocalTensor<float> local = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(local, xGm_[offset], len);
        inQueueX_.EnQue(local);
    }

    __aicore__ inline void Compute(uint32_t numPixels) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpSq = tmpQueue_.AllocTensor<float>();

        for (uint32_t p = 0; p < numPixels; p++) {
            // load channels for this pixel into local array (we'll use tmpSq as cache)
            uint32_t base = p * C_;
            for (uint32_t c = 0; c < C_; c++) {
                tmpSq(c) = xLocal(base + c);
            }
            // compute squares
            for (uint32_t c = 0; c < C_; c++) {
                float val = tmpSq(c);
                tmpSq(c) = val * val;
            }
            // for each output channel, compute sum of squares over window
            for (uint32_t c = 0; c < C_; c++) {
                float sumSq = 0.0f;
                int start = static_cast<int>(c) - halfWindow_;
                if (start < 0) start = 0;
                int end = static_cast<int>(c) + halfWindow_;
                if (end >= static_cast<int>(C_)) end = static_cast<int>(C_) - 1;
                for (int j = start; j <= end; j++) {
                    sumSq += tmpSq(static_cast<uint32_t>(j));
                }
                float denom = k_ + alpha_ * sumSq;
                float scale = 1.0f;
                // compute pow(denom, beta) using powf approximation? Use reciprocal of pow? We'll use AscendC's Pow.
                // But AscendC does not have powf; use Vln + Vexp? Better: compute exp(beta * ln(denom))
                // Use local tensor for scalars? We'll use inline function.
                // Since beta is constant per kernel, we can compute once per pixel per channel? Actually denom varies per channel.
                // We'll use a helper to compute power via logarithm.
                // AscendC provides VecPow? Not directly. We'll use a combination of Ln and Exp.
                // We'll use scalar intrinsic __expf(__logf(denom) * beta_) but we need to do vectorized? Use local tensor of size 1.
                // For simplicity, we'll compute on scalar using standard C++ math? Not allowed in kernel.
                // Use AscendC intrinsics: Ln, Exp on local tensors.
                float denom_ln = 0.0f;
                AscendC::Ln((AscendC::LocalTensor<float>)&denom_ln, (AscendC::LocalTensor<float>)&denom, 1);
                float temp = 0.0f;
                AscendC::Muls((AscendC::LocalTensor<float>)&temp, (AscendC::LocalTensor<float>)&denom_ln, beta_, 1);
                float denom_pow = 0.0f;
                AscendC::Exp((AscendC::LocalTensor<float>)&denom_pow, (AscendC::LocalTensor<float>)&temp, 1);
                float output = xLocal(base + c) / denom_pow;
                yLocal(base + c) = output;
            }
        }

        tmpQueue_.FreeTensor(tmpSq);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t offset, uint32_t len) {
        AscendC::LocalTensor<float> local = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], local, len);
        outQueueY_.FreeTensor(local);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalElements_ {0};
    uint32_t tileElements_ {0};
    uint32_t C_ {0};
    int32_t size_ {5};
    int32_t halfWindow_ {2};
    float alpha_ {1e-4f};
    float beta_ {0.75f};
    float k_ {1.0f};
};

extern "C" __global__ __aicore__ void ca6k_00333_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00333Custom op;
    op.Init(x, y,
            tiling_data.totalElements, tiling_data.tileElements,
            tiling_data.blockDim, tiling_data.C,
            tiling_data.size, tiling_data.alpha, tiling_data.beta, tiling_data.k);
    op.Process();
}
'''
