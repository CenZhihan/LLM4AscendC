#include "ca6k_01250_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k01250CustomTilingData tiling;
    const uint32_t totalLength = context->GetInputShape(0)->GetShapeSize();
    const uint32_t numParams = context->GetInputShape(1)->GetShapeSize();
    if (numParams == 0) {
        return ge::GRAPH_FAILED;
    }

    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) + sizeof(float)); // input x + output y, ignoring small scalar buffer
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.totalLength = totalLength;
    tiling.tileLength = tileLen;
    tiling.blockDim = blockDim;
    tiling.numParams = numParams;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    gert::StorageShape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t MAX_PARAMS = 32;

class KernelCa6k01250Custom {
public:
    __aicore__ inline KernelCa6k01250Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scalars, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t numParams) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        numParams_ = numParams;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        scalarsGm_.SetGlobalBuffer((__gm__ float*)scalars, numParams_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarsBuf_, 1, MAX_PARAMS * sizeof(float));

        // Preload scalars into local memory (once per block)
        AscendC::LocalTensor<float> scalarsLocal = scalarsBuf_.AllocTensor<float>();
        AscendC::DataCopy(scalarsLocal, scalarsGm_[0], numParams_);
        scalarsLocal_ = scalarsLocal;
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0 || numParams_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }

        scalarsBuf_.FreeTensor(scalarsLocal_);
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Start with xLocal as current value
        AscendC::Copy(tLocal, xLocal, len);
        for (uint32_t i = 0; i < numParams_; i++) {
            AscendC::Mul(tLocal, tLocal, scalarsLocal_[i], len);
        }
        AscendC::Copy(yLocal, tLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarsBuf_;
    AscendC::LocalTensor<float> scalarsLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> scalarsGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numParams_ {0};
};

extern "C" __global__ __aicore__ void ca6k_01250_custom(GM_ADDR x, GM_ADDR scalars, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k01250Custom op;
    op.Init(x, scalars, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.numParams);
    op.Process();
}
"""

model_src = """
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_01250_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        # init_inputs[0] is the list of scalar parameters from get_init_inputs()
        params_list = init_inputs[0]
        # Stack scalars to a 1D tensor
        self.scalars = torch.tensor([p.item() for p in params_list], dtype=torch.float32)

    def forward(self, *inputs):
        x = inputs[0]
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.scalars)
