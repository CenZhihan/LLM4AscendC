kernel_src="""

#include "kernel_operator.h"
#include <cmath>

class KernelMaxPool2d {
public:
    __aicore__ inline KernelMaxPool2d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputLength, uint32_t n,
                                uint32_t c, uint32_t h, uint32_t w)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        totalOutputLen_ = totalOutputLength;
        n_ = n;
        c_ = c;
        h_ = h;
        w_ = w;
        hOut_ = h / 2;
        wOut_ = w / 2;
        strideHW_ = w * h; // spatial size per channel
        strideC_ = strideHW_ * c; // total per batch? Actually per channel: strideHW_
        // per batch: strideC_ * n? we don't need that, we compute indices directly.
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output elements
        uint32_t start = blockIdx;
        uint32_t step = blockNum;

        for (uint32_t idx = start; idx < totalOutputLen_; idx += step) {
            // Convert linear output index to (n, c, outRow, outCol)
            uint32_t temp = idx;
            uint32_t outCol = temp % wOut_;
            temp /= wOut_;
            uint32_t outRow = temp % hOut_;
            temp /= hOut_;
            uint32_t ch = temp % c_;
            uint32_t batch = temp / c_;

            // Compute base offset for input: (batch, ch, 2*outRow, 2*outCol)
            uint32_t base = batch * n_ * strideHW_ + ch * strideHW_ + 2 * outRow * w_;
            uint32_t colBase = base + 2 * outCol;

            // Load the four input values as scalars
            float v00 = xGm_[colBase];
            float v01 = xGm_[colBase + 1];
            float v10 = xGm_[colBase + w_];
            float v11 = xGm_[colBase + w_ + 1];

            // Compute max
            float m0 = (v00 > v01) ? v00 : v01;
            float m1 = (v10 > v11) ? v10 : v11;
            float maxVal = (m0 > m1) ? m0 : m1;

            // Write output
            yGm_[idx] = maxVal;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalOutputLen_ {0};
    uint32_t n_ {1};
    uint32_t c_ {1};
    uint32_t h_ {1};
    uint32_t w_ {1};
    uint32_t hOut_ {0};
    uint32_t wOut_ {0};
    uint32_t strideHW_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00495_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPool2d op;
    op.Init(x, y, tiling_data.totalOutputLength, tiling_data.n,
            tiling_data.c, tiling_data.h, tiling_data.w);
    op.Process();
}
