#include "ca6k_04592_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ROWS = 16;               // align tile rows

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04592CustomTilingData tiling;
    const gert::Shape *inShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalElements = static_cast<uint32_t>(inShape->GetShapeSize());
    const uint32_t ndim = inShape->GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= ndim) dim = 1;

    uint32_t C = static_cast<uint32_t>(inShape->GetDim(dim));
    uint32_t rows = totalElements / C;

    // tileRows: each row requires 4 * C floats (double buffered x and y)
    uint32_t maxTileRows = UB_BUDGET_BYTES / (4 * C * sizeof(float));
    maxTileRows = AlignDown(maxTileRows, ALIGN_ROWS);
    if (maxTileRows < ALIGN_ROWS) maxTileRows = ALIGN_ROWS;
    if (maxTileRows > 1024) maxTileRows = 1024;  // hard cap

    uint32_t tileRows = maxTileRows;
    if (rows < tileRows) {
        tileRows = AlignDown(rows, ALIGN_ROWS);
        if (tileRows == 0) tileRows = rows;
    }

    const uint32_t numTiles = CeilDiv(rows, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalElements(totalElements);
    tiling.set_C(C);
    tiling.set_rows(rows);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04592Custom {
public:
    __aicore__ inline KernelCa6k04592Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t C,
                                uint32_t tileRows)
    {
        rows_ = rows;
        C_ = C;
        tileRows_ = tileRows;
        stride_ = C_;  // each row is C elements

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileRows_ * C_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * C_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileRows_ * C_ * sizeof(float));  // for exp subtract
        pipe_.InitBuffer(accQueue_, 1, C_ * sizeof(float));             // for row sum, temporary
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || C_ == 0 || tileRows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride over rows
        const uint32_t strideRows = blockNum * tileRows_;
        uint32_t baseRow = blockIdx * tileRows_;
        if (baseRow >= rows_) return;

        // first tile load
        uint32_t curRows = rows_ - baseRow;
        if (curRows > tileRows_) curRows = tileRows_;
        CopyIn(baseRow, curRows);

        while (true) {
            uint32_t nextBaseRow = baseRow + strideRows;
            uint32_t nextRows = 0;
            if (nextBaseRow < rows_) {
                nextRows = rows_ - nextBaseRow;
                if (nextRows > tileRows_) nextRows = tileRows_;
                CopyIn(nextBaseRow, nextRows);
            }

            Compute(curRows);
            CopyOut(baseRow, curRows);

            if (nextRows == 0) break;
            baseRow = nextBaseRow;
            curRows = nextRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowOffset, uint32_t numRows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        uint32_t len = numRows * C_;
        AscendC::DataCopy(xLocal, xGm_[rowOffset * stride_], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t numRows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> aLocal = accQueue_.AllocTensor<float>();  // reused for sum of exp per row

        // Process each row sequentially
        for (uint32_t r = 0; r < numRows; ++r) {
            float *rowX = reinterpret_cast<float*>(xLocal) + r * C_;
            float *rowY = reinterpret_cast<float*>(yLocal) + r * C_;
            // compute max
            float maxVal = rowX[0];
            for (uint32_t i = 1; i < C_; ++i) {
                if (rowX[i] > maxVal) maxVal = rowX[i];
            }
            // compute sum of exp(x - max)
            float sumExp = 0.0f;
            for (uint32_t i = 0; i < C_; ++i) {
                float tmp = expf(rowX[i] - maxVal);
                sumExp += tmp;
                // store exp result in tLocal (not needed after sum)
            }
            // logSum = log(sumExp)
            float logSum = logf(sumExp);
            // compute y = x - logSum
            for (uint32_t i = 0; i < C_; ++i) {
                rowY[i] = rowX[i] - logSum;
            }
        }

        tmpQueue_.FreeTensor(tLocal);
        accQueue_.FreeTensor(aLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowOffset, uint32_t numRows)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint32_t len = numRows * C_;
        AscendC::DataCopy(yGm_[rowOffset * stride_], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t C_ {0};
    uint32_t tileRows_ {0};
    uint32_t stride_ {0};
};

extern "C" __global__ __aicore__ void ca6k_04592_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04592Custom op;
    op.Init(x, y, tiling_data.rows, tiling_data.C, tiling_data.tileRows);
    op.Process();
}
