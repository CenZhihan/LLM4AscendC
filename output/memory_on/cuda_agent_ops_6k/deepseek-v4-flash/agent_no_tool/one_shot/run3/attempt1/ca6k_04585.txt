kernel_src="""

#include "kernel_operator.h"

class KernelCa6k04585Custom {
public:
    __aicore__ inline KernelCa6k04585Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t batch, uint32_t inC, uint32_t outC,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t sD, uint32_t sH, uint32_t sW,
                                uint32_t pD, uint32_t pH, uint32_t pW,
                                uint32_t dD, uint32_t dH, uint32_t dW,
                                uint32_t totalElems, uint32_t tileLen,
                                bool hasBias)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        if (hasBias) {
            bGm_.SetGlobalBuffer((__gm__ float*)bias);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        batch_   = batch;
        inC_     = inC;
        outC_    = outC;
        D_       = D;
        H_       = H;
        W_       = W;
        kD_      = kD;
        kH_      = kH;
        kW_      = kW;
        sD_      = sD;
        sH_      = sH;
        sW_      = sW;
        pD_      = pD;
        pH_      = pH;
        pW_      = pW;
        dD_      = dD;
        dH_      = dH;
        dW_      = dW;
        totalElems_ = totalElems;
        tileLen_    = tileLen;
        hasBias_    = hasBias;

        // precompute strides for x and w
        // x:(N,inC,D,H,W) contiguous
        xStrideN_ = inC_ * D_ * H_ * W_;
        xStrideC_ = D_ * H_ * W_;
        xStrideD_ = H_ * W_;
        xStrideH_ = W_;
        xStrideW_ = 1;
        // w:(outC,inC,kD,kH,kW) contiguous
        wStrideO_ = inC_ * kD_ * kH_ * kW_;
        wStrideC_ = kD_ * kH_ * kW_;
        wStrideD_ = kH_ * kW_;
        wStrideH_ = kW_;
        wStrideW_ = 1;
    }

    __aicore__ inline void Process()
    {
        if (totalElems_ == 0) return;
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t start = blockIdx * tileLen_;
        uint32_t end = start + tileLen_;
        if (end > totalElems_) end = totalElems_;
        uint32_t stride = blockNum * tileLen_;

        while (start < totalElems_) {
            for (uint32_t idx = start; idx < end; ++idx) {
                // compute output at flat index idx
                float result = computeOutputElement(idx);
                yGm_.SetValue(idx, result);
            }
            start += stride;
            end = start + tileLen_;
            if (end > totalElems_) end = totalElems_;
        }
    }

private:
    __aicore__ inline float computeOutputElement(uint32_t idx)
    {
        // decompose idx into (b, oc, od, oh, ow)
        uint32_t spatialSize = D_ * H_ * W_; // depth * height * width
        uint32_t ocStride = outC_ * spatialSize;
        uint32_t b = idx / ocStride;
        uint32_t rem = idx % ocStride;
        uint32_t oc = rem / spatialSize;
        rem = rem % spatialSize;
        uint32_t od = rem / (H_ * W_);
        rem = rem % (H_ * W_);
        uint32_t oh = rem / W_;
        uint32_t ow = rem % W_;

        float sum = 0.0f;
        // loop over kernel and input channels
        for (uint32_t kd = 0; kd < kD_; ++kd) {
            int32_t id = static_cast<int32_t>(od) * static_cast<int32_t>(sD_)
                         - static_cast<int32_t>(pD_)
                         + static_cast<int32_t>(kd) * static_cast<int32_t>(dD_);
            if (id < 0 || id >= static_cast<int32_t>(D_)) continue;
            for (uint32_t kh = 0; kh < kH_; ++kh) {
                int32_t ih = static_cast<int32_t>(oh) * static_cast<int32_t>(sH_)
                             - static_cast<int32_t>(pH_)
                             + static_cast<int32_t>(kh) * static_cast<int32_t>(dH_);
                if (ih < 0 || ih >= static_cast<int32_t>(H_)) continue;
                for (uint32_t kw = 0; kw < kW_; ++kw) {
                    int32_t iw = static_cast<int32_t>(ow) * static_cast<int32_t>(sW_)
                                 - static_cast<int32_t>(pW_)
                                 + static_cast<int32_t>(kw) * static_cast<int32_t>(dW_);
                    if (iw < 0 || iw >= static_cast<int32_t>(W_)) continue;
                    // input value
                    uint32_t xOffset = b * xStrideN_ + 0 * xStrideC_ // inC index = 0
                                       + static_cast<uint32_t>(id) * xStrideD_
                                       + static_cast<uint32_t>(ih) * xStrideH_
                                       + static_cast<uint32_t>(iw);
                    float xVal = xGm_.GetValue(xOffset);
                    // weight value
                    uint32_t wOffset = oc * wStrideO_ + 0 * wStrideC_ // inC index = 0
                                       + kd * wStrideD_
                                       + kh * wStrideH_
                                       + kw;
                    float wVal = wGm_.GetValue(wOffset);
                    sum += xVal * wVal;
                }
            }
        }
        // bias
        if (hasBias_) {
            sum += bGm_.GetValue(oc);
        }
        return sum;
    }

    // Global tensors
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    // dimensions
    uint32_t batch_;
    uint32_t inC_;
    uint32_t outC_;
    uint32_t D_, H_, W_;
    uint32_t kD_, kH_, kW_;
    uint32_t sD_, sH_, sW_;
    uint32_t pD_, pH_, pW_;
    uint32_t dD_, dH_, dW_;
    uint32_t totalElems_;
    uint32_t tileLen_;
    bool     hasBias_;

    // strides
    uint32_t xStrideN_, xStrideC_, xStrideD_, xStrideH_, xStrideW_;
    uint32_t wStrideO_, wStrideC_, wStrideD_, wStrideH_, wStrideW_;
};

extern "C" __global__ __aicore__ void ca6k_04585_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    KernelCa6k04585Custom op;
    op.Init(x, weight, bias, y,
            td.batch, td.inChannels, td.outChannels,
            td.depth, td.height, td.width,
            td.kD, td.kH, td.kW,
            td.strideD, td.strideH, td.strideW,
            td.padD, td.padH, td.padW,
            td.dilD, td.dilH, td.dilW,
            td.totalOutputElems, td.tileLen,
            td.hasBias);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size,
                 stride=1, padding=(0,0,0), dilation=(1,1,1), bias=True):
        super(ModelNew, self).__init__()
        # Custom operator does not own parameters; they are passed as inputs.
        # We store the convolution hyperparameters for the forward call.
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.bias_flag = bias
        # weight and bias will be created on first forward (like LazyConv3d)
        self.weight = None
        self.bias = None

    def forward(self, x):
        if self.weight is None:
            # lazy initialization: infer in_channels from x
            in_channels = x.size(1)
            self.weight = torch.nn.Parameter(
                torch.empty(self.out_channels, in_channels, *self.kernel_size,
                            dtype=x.dtype, device=x.device)
            )
            torch.nn.init.kaiming_uniform_(self.weight, a=5**0.5)
            if self.bias_flag:
                self.bias = torch.nn.Parameter(
                    torch.zeros(self.out_channels, dtype=x.dtype, device=x.device)
                )
            else:
                self.bias = None
        # call custom operator
        return custom_ops_lib.ca6k_04585_custom(
            x, self.weight,
            self.bias if self.bias_flag else None,
            stride=self.stride,
            padding=self.padding,
            dilation=self.dilation,
            has_bias=self.bias_flag
        )
