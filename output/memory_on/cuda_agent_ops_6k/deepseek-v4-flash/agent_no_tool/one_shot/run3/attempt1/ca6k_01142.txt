host_operator_src = '''
#include "ca6k_01142_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k01142CustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::StorageShape *y_shape = context->GetInputShape(1);

    auto x_dim = x_shape->GetOriginShape().GetDimNum();
    auto y_dim = y_shape->GetOriginShape().GetDimNum();

    uint32_t batchSize = x_shape->GetOriginShape().GetDim(0);
    uint32_t channels = x_shape->GetOriginShape().GetDim(1);
    uint32_t height = x_shape->GetOriginShape().GetDim(2);
    uint32_t width = x_shape->GetOriginShape().GetDim(3);

    uint32_t totalPixels = batchSize * height * width;

    // Tiling from UB budget: need to hold y_pred (float) and y_true (int64) data for a tile.
    // Each pixel we need to read 1 float from y_pred (indexed) and 1 int64 from y_true.
    // But in kernel we read y_true directly from global; no local copy for y_true.
    // We will copy a tile of y_pred indices (actually we need to read y_pred at computed indices).
    // Simpler: we only need a small local buffer to accumulate sums.
    // For each pixel we need to compute index and read y_pred value.
    // No large local buffer needed. So tile length can be large.
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);  // enough for accumulation
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLength = maxTile;
    if (totalPixels < tileLength) {
        uint32_t aligned = AlignDown(totalPixels, ALIGN_ELEMS);
        tileLength = (aligned > 0) ? aligned : totalPixels;
    }

    uint32_t numTiles = (tileLength == 0) ? 1u : CeilDiv(totalPixels, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.totalPixels = totalPixels;
    tiling.tileLength = tileLength;
    tiling.blockDim = blockDim;
    tiling.channels = channels;
    tiling.height = height;
    tiling.width = width;
    tiling.spatialSize = height * width;
    tiling.batchSize = batchSize;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    // output shape is scalar [1]
    gert::Shape *loss_shape = context->GetOutputShape(0);
    loss_shape->SetDimNum(1);
    loss_shape->SetDim(0, 1);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {
class Ca6k01142Custom : public OpDef {
public:
    explicit Ca6k01142Custom(const char* name) : OpDef(name) {
        this->Input("y_pred")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("y_true")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT64})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("loss")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(Ca6k01142Custom);
} // namespace ops

kernel_src = '''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01142Custom {
public:
    __aicore__ inline KernelCa6k01142Custom() {}

    __aicore__ inline void Init(GM_ADDR y_pred, GM_ADDR y_true, GM_ADDR loss,
                                uint32_t totalPixels, uint32_t tileLength,
                                uint32_t channels, uint32_t height, uint32_t width,
                                uint32_t spatialSize, uint32_t batchSize)
    {
        yPredGm_.SetGlobalBuffer((__gm__ float*)y_pred);
        yTrueGm_.SetGlobalBuffer((__gm__ int64_t*)y_true);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1); // scalar output

        totalPixels_ = totalPixels;
        tileLength_ = tileLength;
        channels_ = channels;
        height_ = height;
        width_ = width;
        spatialSize_ = spatialSize;
        batchSize_ = batchSize;
        blockDim_ = GetBlockNum();
        blockIdx_ = GetBlockIdx();
    }

    __aicore__ inline void Process()
    {
        if (totalPixels_ == 0 || tileLength_ == 0) return;

        // Initialize loss to zero (only block 0 does this)
        if (blockIdx_ == 0) {
            lossGm_.SetValue(0, 0.0f);
        }
        // Synchronize to ensure init is visible to others
        // Use a barrier? Not needed because atomic add will be based on initial value.

        uint32_t stride = blockDim_ * tileLength_;
        uint32_t base = blockIdx_ * tileLength_;

        while (base < totalPixels_) {
            uint32_t curLen = totalPixels_ - base;
            if (curLen > tileLength_) curLen = tileLength_;

            float localSum = 0.0f;
            // Process each pixel in this tile
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t pixelIdx = base + i; // linear index over [N, H, W]
                // Compute batch, h, w
                uint32_t b = pixelIdx / spatialSize_;
                uint32_t rem = pixelIdx % spatialSize_;
                uint32_t h = rem / width_;
                uint32_t w = rem % width_;

                // Get true class
                int64_t trueClass = yTrueGm_.GetValue(pixelIdx);
                // Compute index in y_pred: [b, trueClass, h, w]
                uint32_t predIdx = b * channels_ * spatialSize_ + trueClass * spatialSize_ + h * width_ + w;
                float logProb = yPredGm_.GetValue(predIdx);
                localSum -= logProb; // loss = -log(prob)
            }

            // Atomic add localSum to loss global
            AscendC::AtomicAdd<float>(lossGm_, localSum);

            base += stride;
        }
    }

private:
    GlobalTensor<float> yPredGm_;
    GlobalTensor<int64_t> yTrueGm_;
    GlobalTensor<float> lossGm_;

    uint32_t totalPixels_ {0};
    uint32_t tileLength_ {0};
    uint32_t channels_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    uint32_t spatialSize_ {0};
    uint32_t batchSize_ {0};
    uint32_t blockDim_ {0};
    uint32_t blockIdx_ {0};
};

extern "C" __global__ __aicore__ void ca6k_01142_custom(GM_ADDR y_pred, GM_ADDR y_true, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelCa6k01142Custom op;
    op.Init(y_pred, y_true, loss,
            tilingData.totalPixels, tilingData.tileLength,
            tilingData.channels, tilingData.height, tilingData.width,
            tilingData.spatialSize, tilingData.batchSize);
    op.Process();
}
'''
