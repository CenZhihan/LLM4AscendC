        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numEmbeddings * embeddingDim);
        indicesGm_.SetGlobalBuffer((__gm__ int32_t*)indices, totalIndices);
        offsetsGm_.SetGlobalBuffer((__gm__ int32_t*)offsets, totalBags+1);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalBags * embeddingDim);
        totalBags_ = totalBags;
        embeddingDim_ = embeddingDim;
        tileDim_ = tileDim;
        // Initialize pipe and queues
        pipe_.InitBuffer(inQueueX_, 1, tileDim_ * sizeof(float)); // weight row tile
        pipe_.InitBuffer(outQueueY_, 1, tileDim_ * sizeof(float)); // output tile accumulator
    }

    __aicore__ inline void Process() {
        // Each block processes a subset of bags
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        for (uint32_t bag = blockIdx; bag < totalBags_; bag += blockNum) {
            int32_t start = offsetsGm_[bag];
