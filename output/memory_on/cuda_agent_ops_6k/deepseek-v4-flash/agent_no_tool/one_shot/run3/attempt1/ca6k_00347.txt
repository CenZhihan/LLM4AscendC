kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;
constexpr int32_t MAX_PATCH_SIZE = 256; // enough for 3*5*64=960? we use small fixed size; safe

class KernelCa6k00347Custom {
public:
    __aicore__ inline KernelCa6k00347Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                Ca6k00347CustomTilingData td)
    {
        xGm_ = (__gm__ float*)x;
        yGm_ = (__gm__ float*)y;

        batchSize_ = td.batchSize;
        channels_ = td.channels;
        height_ = td.height;
        width_ = td.width;
        kernelH_ = td.kernelH;
        kernelW_ = td.kernelW;
        dilationH_ = td.dilationH;
        dilationW_ = td.dilationW;
        stride_ = td.stride;
        padding_ = td.padding;
        outputH_ = td.outputH;
        outputW_ = td.outputW;
        outputRows_ = td.outputRows;
        totalOutputElems_ = td.totalOutputElems;
        blockDim_ = td.blockDim;

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, sizeof(float) * MAX_PATCH_SIZE);
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, sizeof(float) * MAX_PATCH_SIZE);
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();

        for (uint64_t elemIdx = blockIdx; elemIdx < totalOutputElems_; elemIdx += blockNum) {
            // compute batch, row, col from flattened index
            uint32_t batch = elemIdx / (outputRows_ * outputH_ * outputW_);
            uint32_t remainder = elemIdx % (outputRows_ * outputH_ * outputW_);
            uint32_t row = remainder / outputW_;
            uint32_t col = remainder % outputW_;

            // compute output spatial position
            uint32_t h_out = col / outputW_;
            uint32_t w_out = col % outputW_;

            // copy patch into local buffer
            float *patchLocal = inQueue_.AllocTensor<float>();
            uint32_t patchOffset = 0;
            for (uint32_t c = 0; c < channels_; ++c) {
                for (int32_t dh = 0; dh < kernelH_; ++dh) {
                    for (int32_t dw = 0; dw < kernelW_; ++dw) {
                        // compute input index
                        uint32_t h_in = h_out * stride_ + dh * dilationH_ - padding_;
                        uint32_t w_in = w_out * stride_ + dw * dilationW_ - padding_;
                        float val = xGm_[batch * channels_ * height_ * width_
                                        + c * height_ * width_
                                        + h_in * width_
                                        + w_in];
                        patchLocal[patchOffset++] = val;
                    }
                }
            }
            inQueue_.EnQue(patchLocal);

            // dequeue and write to output
            float *patchOut = inQueue_.DeQue<float>();
            uint32_t outOffset = batch * outputRows_ * outputH_ * outputW_
                                + row * outputW_
                                + col;
            for (uint32_t i = 0; i < outputRows_; ++i) {
                yGm_[outOffset + i * outputH_ * outputW_] = patchOut[i];
            }
            inQueue_.FreeTensor(patchOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    __gm__ float *xGm_;
    __gm__ float *yGm_;

    uint32_t batchSize_;
    uint32_t channels_;
    uint32_t height_;
    uint32_t width_;
    int32_t kernelH_;
    int32_t kernelW_;
    int32_t dilationH_;
    int32_t dilationW_;
    int32_t stride_;
    int32_t padding_;
    uint32_t outputH_;
    uint32_t outputW_;
    uint32_t outputRows_;
    uint32_t totalOutputElems_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void ca6k_00347_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00347Custom op;
    op.Init(x, y, tiling_data);
    op.Process();
}
'''
