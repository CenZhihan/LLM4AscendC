#include "ca6k_00535_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t ALIGN_ELEMS = 8;  // float alignment for DMA (32 bytes)

static inline uint32_t AlignUp(uint32_t x, uint32_t a)
{
    return (x + a - 1) / a * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00535CustomTilingData tiling;

    const gert::StorageShape *xShape = context->GetInputShape(0);
    uint32_t rows = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t cols = static_cast<uint32_t>(xShape->GetDim(1));

    // tileLength aligned to 8 floats for DMA
    uint32_t tileLength = AlignUp(rows, ALIGN_ELEMS);
    if (tileLength > 49152) tileLength = 49152; // UB budget: 192KB / 4B = 49152

    uint32_t blockDim = cols;  // one block per column

    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00535Custom {
public:
    __aicore__ inline KernelCa6k00535Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t tileLength)
    {
        rows_ = rows;
        cols_ = cols;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, rows_ * cols_);

        // Buffer for one column (aligned)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= cols_) return; // only cols blocks active

        // Each block processes one column
        // Load entire column (row-major: stride = cols)
        CopyIn(blockIdx);
        Compute();
        CopyOut(blockIdx);
    }

private:
    __aicore__ inline void CopyIn(uint32_t colIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        // Load column: start at colIdx, stride = cols, count = rows_
        AscendC::DataCopy(xLocal, xGm_[colIdx], cols_, rows_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Sequential cumulative maximum
        float curMax = -3.4028235e+38f; // -FLT_MAX
        for (uint32_t i = 0; i < rows_; i++) {
            float val = xLocal.GetValue(i);
            if (val > curMax) {
                curMax = val;
            }
            yLocal.SetValue(i, curMax);
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t colIdx)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        // Write column
        AscendC::DataCopy(yGm_[colIdx], yLocal, cols_, rows_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00535_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00535Custom op;
    op.Init(x, y, tiling_data.rows, tiling_data.cols, tiling_data.tileLength);
    op.Process();
}
