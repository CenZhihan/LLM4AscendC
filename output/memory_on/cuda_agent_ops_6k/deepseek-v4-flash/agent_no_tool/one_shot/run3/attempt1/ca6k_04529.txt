kernel_src = """
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04529Custom {
public:
    __aicore__ inline KernelCa6k04529Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        totalLen_ = td.totalLength;
        tileLen_ = td.tileLength;
        blockDim_ = td.blockDim;
        N_ = td.N; C_in_ = td.C_in; H_in_ = td.H_in; W_in_ = td.W_in;
        C_out_ = td.C_out; H_out_ = td.H_out; W_out_ = td.W_out;
        kernelH_ = td.kernelH; kernelW_ = td.kernelW;
        stride_ = td.stride; padH_ = td.padH; padW_ = td.padW;
        dilationH_ = td.dilationH; dilationW_ = td.dilationW;
        poolKernelH_ = td.poolKernelH; poolKernelW_ = td.poolKernelW;
        poolStride_ = td.poolStride; poolPadH_ = td.poolPadH; poolPadW_ = td.poolPadW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmp2Queue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // For simplicity, we process the entire output in one block per block
        // We'll iterate over output spatial positions and compute conv+relu+pool
        // This is a naive implementation for demonstration
        if (totalLen_ == 0) return;

        uint32_t idx = blockIdx_ * tileLen_;
        if (idx >= totalLen_) return;

        uint32_t len = tileLen_;
        if (idx + len > totalLen_) len = totalLen_ - idx;

        // --- Convolution + ReLU ---
        // We compute the conv output element at output position (n, oc, h, w)
        // then apply ReLU, then later pool.
        // We'll store conv+relu result in a temporary buffer per tile.
        // For simplicity, we assume tileLen_ is small enough.

        LocalTensor<float> convReluLocal = tmpQueue_.AllocTensor<float>();
        for (uint32_t t = 0; t < len; t++) {
            uint32_t globalIdx = idx + t;
            // decode globalIdx to (n, oc, ph, pw) where ph,pw are pooled output indices
            uint32_t pw = globalIdx % W_out_;
            uint32_t ph = (globalIdx / W_out_) % H_out_;
            uint32_t oc = (globalIdx / (H_out_ * W_out_)) % C_out_;
            uint32_t n = globalIdx / (C_out_ * H_out_ * W_out_);

            // compute average pooling over a window in conv output
            // We need conv output at positions (n, oc, h_start..h_end, w_start..w_end)
            // h_start = ph * poolStride_ - poolPadH_, but we need max with 0 and min with H_out-1
            // Let's compute convolution result on the fly for each pooling window element
            float sum = 0.0f;
            int count = 0;
            for (uint32_t d_h = 0; d_h < poolKernelH_; d_h++) {
                int h_conv = static_cast<int>(ph * poolStride_ + d_h - poolPadH_);
                if (h_conv < 0 || static_cast<uint32_t>(h_conv) >= H_out_) continue;
                for (uint32_t d_w = 0; d_w < poolKernelW_; d_w++) {
                    int w_conv = static_cast<int>(pw * poolStride_ + d_w - poolPadW_);
                    if (w_conv < 0 || static_cast<uint32_t>(w_conv) >= W_out_) continue;
                    // compute conv value at (n, oc, h_conv, w_conv)
                    float convVal = computeConv(n, oc, h_conv, w_conv);
                    // ReLU
                    if (convVal < 0.0f) convVal = 0.0f;
                    sum += convVal;
                    count++;
                }
            }
            float avgVal = (count > 0) ? sum / count : 0.0f;
            // store to output global
            yGm_.SetValue(globalIdx, avgVal);
        }
        tmpQueue_.FreeTensor(convReluLocal);
    }

private:
    __aicore__ inline float computeConv(uint32_t n, uint32_t oc, uint32_t h, uint32_t w)
    {
        float val = 0.0f;
        uint32_t weightOffset = oc * C_in_ * kernelH_ * kernelW_;
        for (uint32_t ic = 0; ic < C_in_; ic++) {
            for (uint32_t kh = 0; kh < kernelH_; kh++) {
                int h_in = static_cast<int>(h * stride_ + kh * dilationH_ - padH_);
                if (h_in < 0 || static_cast<uint32_t>(h_in) >= H_in_) continue;
                for (uint32_t kw = 0; kw < kernelW_; kw++) {
                    int w_in = static_cast<int>(w * stride_ + kw * dilationW_ - padW_);
                    if (w_in < 0 || static_cast<uint32_t>(w_in) >= W_in_) continue;
                    float x_val = xGm_.GetValue(n * C_in_ * H_in_ * W_in_ + ic * H_in_ * W_in_ + h_in * W_in_ + w_in);
                    float w_val = wGm_.GetValue(weightOffset + ic * kernelH_ * kernelW_ + kh * kernelW_ + kw);
                    val += x_val * w_val;
                }
            }
        }
        return val;
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;
    TQue<TPosition::VECCALC, 1> tmp2Queue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> wGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t blockDim_{1};
    uint32_t blockIdx_{0};

    uint32_t N_, C_in_, H_in_, W_in_;
    uint32_t C_out_, H_out_, W_out_;
    uint32_t kernelH_, kernelW_, stride_;
    uint32_t padH_, padW_, dilationH_, dilationW_;
    uint32_t poolKernelH_, poolKernelW_, poolStride_;
    uint32_t poolPadH_, poolPadW_;
};

extern "C" __global__ __aicore__ void ca6k_04529_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelCa6k04529Custom op;
    op.Init(x, y, weight, workspace, tiling);
    op.Process();
}
