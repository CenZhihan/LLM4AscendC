```cpp
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k02874Custom {
public:
    __aicore__ inline KernelCa6k02874Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalRows, uint32_t totalCols, uint32_t tileRows, uint32_t blockDim)
    {
        totalRows_ = totalRows;
        totalCols_ = totalCols;
        blockDim_ = blockDim;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * totalCols_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, totalCols_ * totalCols_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, totalCols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * totalCols_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, totalCols_ * sizeof(float)); // for weight row
        pipe_.InitBuffer(tmpQueue_, 1, totalCols_ * sizeof(float)); // for output row
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, totalCols_ * sizeof(float)); // for result
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || totalCols_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        // Each block processes rows with stride blockNum
        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            ProcessOneRow(row);
        }
    }

private:
    __aicore__ inline void ProcessOneRow(uint32_t row)
    {
        // Load bias into tmp
        AscendC::LocalTensor<float> outRow = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(outRow, biasGm_, totalCols_);
        // For each k, accumulate weight[row k] * x[row][k]
        for (uint32_t k = 0; k < totalCols_; ++k) {
            float xVal = xGm_[row * totalCols_ + k];
            // Load weight row k (vector of length totalCols) into weightRow
            AscendC::LocalTensor<float> weightRow = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(weightRow, weightGm_ + k * totalCols_, totalCols_);
            // outRow += xVal * weightRow
            AscendC::Muls(weightRow, weightRow, xVal, totalCols_);
            AscendC::Add(outRow, outRow, weightRow, totalCols_);
            inQueueX_.FreeTensor(weightRow);
        }
        // Write result to global memory
        AscendC::DataCopy(yGm_ + row * totalCols_, outRow, totalCols_);
        tmpQueue_.FreeTensor(outRow);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalRows_ {0};
    uint32_t totalCols_ {0};
    uint32_t blockDim_ {1};
};

extern "C" __global__ __aicore__ void ca6k_02874_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{

    KernelCa6k02874Custom op;
    op.Init(x, weight, bias, y, tiling_data.totalRows, tiling_data.totalCols, tiling_data.tileRows, tiling_data.blockDim);
    op.Process();
}
```
