host_operator_src = '''
#include "ca6k_02258_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k02258CustomTilingData tiling;

  const gert::StorageShape *x_shape = context->GetInputShape(0);
  auto dims = x_shape->GetOriginShape();
  int32_t N = static_cast<int32_t>(dims.GetDim(0));
  int32_t C = static_cast<int32_t>(dims.GetDim(1));
  int32_t H_in = static_cast<int32_t>(dims.GetDim(2));
  int32_t W_in = static_cast<int32_t>(dims.GetDim(3));

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  const float *scale_factor = attrs->GetAttrPointer<float>(0);

  int32_t H_out = static_cast<int32_t>(static_cast<float>(H_in) * (*scale_factor));
  int32_t W_out = static_cast<int32_t>(static_cast<float>(W_in) * (*scale_factor));
  uint32_t totalLength = static_cast<uint32_t>(N) * static_cast<uint32_t>(C) *
                         static_cast<uint32_t>(H_out) * static_cast<uint32_t>(W_out);

  // Compute tileLength (output elements per tile) based on UB budget.
  // Each output element uses 1 float, but we need temporary space for input values.
  uint32_t maxTile = UB_BUDGET_BYTES / 20;
  maxTile = (maxTile / ALIGN_ELEMS) * ALIGN_ELEMS;
  uint32_t tileLen = (maxTile < totalLength) ? maxTile : totalLength;
  if (tileLen == 0) tileLen = 1;

  uint32_t numTiles = (totalLength + tileLen - 1) / tileLen;
  uint32_t blockDim = numTiles;
  if (blockDim < 8) blockDim = 8;
  if (blockDim > 24) blockDim = 24;

  context->SetBlockDim(blockDim);

  tiling.totalLength = totalLength;
  tiling.tileLength = tileLen;
  tiling.blockDim = blockDim;
  tiling.N = N;
  tiling.C = C;
  tiling.H_in = H_in;
  tiling.W_in = W_in;
  tiling.H_out = H_out;
  tiling.W_out = W_out;
  tiling.scaleFactor = *scale_factor;

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::StorageShape *x_shape = context->GetInputShape(0);
  gert::Shape *y_shape = context->GetOutputShape(0);
  auto dims = x_shape->GetOriginShape();
  int32_t H_in = static_cast<int32_t>(dims.GetDim(2));
  int32_t W_in = static_cast<int32_t>(dims.GetDim(3));
  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  const float *scale_factor = attrs->GetAttrPointer<float>(0);
  int32_t H_out = static_cast<int32_t>(static_cast<float>(H_in) * (*scale_factor));
  int32_t W_out = static_cast<int32_t>(static_cast<float>(W_in) * (*scale_factor));
  gert::Shape out_shape({dims.GetDim(0), dims.GetDim(1), H_out, W_out});
  *y_shape = out_shape;
  return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
  context->SetOutputDataType(0, context->GetInputDataType(0));
  return GRAPH_SUCCESS;
}
}

namespace ops {
class Ca6k02258Custom : public OpDef {
public:
  explicit Ca6k02258Custom(const char* name) : OpDef(name) {
    this->Input("x")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});

    this->Output("y")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});

    this->Attr("scale_factor").AttrType(OPTIONAL).Float(2.0);
    this->Attr("mode").AttrType(OPTIONAL).String("bilinear");
    this->Attr("align_corners").AttrType(OPTIONAL).Bool(false);

    this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

    this->AICore().SetTiling(optiling::TilingFunc);
    this->AICore().AddConfig("ascend910b");
  }
};

OP_ADD(Ca6k02258Custom);
}

kernel_src = '''
#include "kernel_operator.h"

using namespace AscendC;

class KernelCa6k02258Custom {
public:
    __aicore__ inline KernelCa6k02258Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t N, int32_t C, int32_t H_in, int32_t W_in,
                                int32_t H_out, int32_t W_out, float scaleFactor)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N;
        C_ = C;
        H_in_ = H_in;
        W_in_ = W_in;
        H_out_ = H_out;
        W_out_ = W_out;
        scaleFactor_ = scaleFactor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Allocate local buffers for output tile and input work area
        pipe_.InitBuffer(outBuf_, 1, tileLen_ * sizeof(float));
        // For simplicity, we compute directly using global reads.
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        // Grid-stride over output elements
        for (uint32_t start = blockIdx * tileLen_; start < totalLen_;
             start += blockNum * tileLen_) {
            uint32_t curLen = tileLen_;
            if (start + curLen > totalLen_) curLen = totalLen_ - start;

            // Fill output buffer for this tile
            LocalTensor<float> outLocal = outBuf_.AllocTensor<float>();
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t globalIdx = start + i;
                // Decode output index
                int32_t out_idx = static_cast<int32_t>(globalIdx);
                int32_t n = out_idx / (C_ * H_out_ * W_out_);
                out_idx %= (C_ * H_out_ * W_out_);
                int32_t c = out_idx / (H_out_ * W_out_);
                out_idx %= (H_out_ * W_out_);
                int32_t h_out = out_idx / W_out_;
                int32_t w_out = out_idx % W_out_;

                // Compute input coordinates (align_corners=False -> asymmetric)
                float h_in_f = static_cast<float>(h_out) / scaleFactor_;
                float w_in_f = static_cast<float>(w_out) / scaleFactor_;

                int32_t h_in_floor = static_cast<int32_t>(h_in_f);
                int32_t w_in_floor = static_cast<int32_t>(w_in_f);
                int32_t h_in_ceil = h_in_floor + 1;
                int32_t w_in_ceil = w_in_floor + 1;

                // Clamp to valid range
                if (h_in_ceil >= H_in_) h_in_ceil = H_in_ - 1;
                if (w_in_ceil >= W_in_) w_in_ceil = W_in_ - 1;

                float x_frac = h_in_f - static_cast<float>(h_in_floor);
                float y_frac = w_in_f - static_cast<float>(w_in_floor);

                // Read four corner values from global memory
                float v00 = xGm_[n * C_ * H_in_ * W_in_ + c * H_in_ * W_in_ + h_in_floor * W_in_ + w_in_floor];
                float v01 = xGm_[n * C_ * H_in_ * W_in_ + c * H_in_ * W_in_ + h_in_floor * W_in_ + w_in_ceil];
                float v10 = xGm_[n * C_ * H_in_ * W_in_ + c * H_in_ * W_in_ + h_in_ceil * W_in_ + w_in_floor];
                float v11 = xGm_[n * C_ * H_in_ * W_in_ + c * H_in_ * W_in_ + h_in_ceil * W_in_ + w_in_ceil];

                // Bilinear interpolation
                float top = v00 + (v01 - v00) * y_frac;
                float bottom = v10 + (v11 - v10) * y_frac;
                float value = top + (bottom - top) * x_frac;

                outLocal.SetValue(i, value);
            }

            // Copy output tile to global memory
            DataCopy(yGm_[start], outLocal, curLen);
            outBuf_.FreeTensor(outLocal);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECOUT, 1> outBuf_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    int32_t N_, C_, H_in_, W_in_, H_out_, W_out_;
    float scaleFactor_;
};

extern "C" __global__ __aicore__ void ca6k_02258_custom(GM_ADDR x, GM_ADDR y,
                                                         GM_ADDR workspace,
                                                         GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02258Custom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C, tiling_data.H_in, tiling_data.W_in,
            tiling_data.H_out, tiling_data.W_out, tiling_data.scaleFactor);
    op.Process();
}
'''

python_bind_src = '''
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor ca6k_02258_impl_npu(const at::Tensor& self,
                                double scale_factor,
                                const std::string& mode,
                                bool align_corners) {
    at::Tensor result = at::empty_like(self);
    // The custom op expects output shape derived from scale_factor.
    // We compute output shape here and let the op fill it.
    std::vector<int64_t> output_size = {
        self.size(0),
        self.size(1),
        static_cast<int64_t>(static_cast<double>(self.size(2)) * scale_factor),
        static_cast<int64_t>(static_cast<double>(self.size(3)) * scale_factor)
    };
    result = at::empty(output_size, self.options());
    EXEC_NPU_CMD(aclnnCa6k02258Custom, self, scale_factor, mode, align_corners, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_02258_custom", &ca6k_02258_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("ca6k_02258_custom", &ca6k_02258_impl_npu,
          "Bilinear upsample (Ca6k02258Custom on NPU)");
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_02258_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        # init_inputs: [scale_factor, mode, align_corners]
        self.scale_factor = init_inputs[0]
        self.mode = init_inputs[1]
        self.align_corners = init_inputs[2]

    def forward(self, *inputs):
        fn = getattr(custom_ops_lib, _FN)
        # inputs[0] is the tensor
        return fn(inputs[0], self.scale_factor, self.mode, self.align_corners)
