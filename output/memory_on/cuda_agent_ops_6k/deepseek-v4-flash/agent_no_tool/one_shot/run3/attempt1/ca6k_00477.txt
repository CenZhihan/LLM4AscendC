host_operator_src = """
#include "ca6k_00477_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 32;             // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 8192;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00477CustomTilingData tiling;
    const uint32_t totalLength = static_cast<uint32_t>(
        context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);

    // For InstanceNorm3d, spatial dimensions are D*H*W, numSlices = N*C
    // We need to infer numSlices from shape: shape is [N, C, D, H, W]
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = shape.GetDim(0);
    uint32_t C = shape.GetDim(1);
    uint32_t D = shape.GetDim(2);
    uint32_t H = shape.GetDim(3);
    uint32_t W = shape.GetDim(4);
    uint32_t numSlices = N * C;
    uint32_t spatialSize = D * H * W;

    // tileLen from UB budget: we need 2 input tiles + 2 output tiles = 4 tiles of floats
    // plus some scratch: double buffered sum accumulators etc.
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float)); // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // Ensure tileLen does not exceed spatialSize (slice size)
    uint32_t tileLen = maxTile;
    if (spatialSize < tileLen) {
        tileLen = AlignDown(spatialSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = spatialSize;
    }
    // Also ensure that tileLen is not too small
    if (tileLen < ALIGN_ELEMS && spatialSize > ALIGN_ELEMS) {
        tileLen = ALIGN_ELEMS;
    }

    // Set blockDim: number of slices per block, try to use many cores
    uint32_t blockDim = numSlices;
    if (blockDim > 24) blockDim = 24; // limit to available cores on 910B
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_numSlices(numSlices);
    tiling.set_spatialSize(spatialSize);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class Ca6k00477Custom : public OpDef {
public:
    explicit Ca6k00477Custom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("eps").AttrType(OPTIONAL).Float(1e-5);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00477Custom {
public:
    __aicore__ inline KernelCa6k00477Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t numSlices, uint32_t spatialSize,
                                float eps)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        numSlices_ = numSlices;
        spatialSize_ = spatialSize;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // double buffering: two input tiles, two output tiles, plus one scratch for sums
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scratchQueue_, 1, tileLen_ * sizeof(float)); // for x^2 or temp
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block handles a subset of slices (N*C)
        // slices are processed in grid-stride manner
        // For each slice we need two passes: sum pass and normalize pass
        // To reduce global reads, we can combine if tileLen_ covers entire slice, but skip for general case.

        for (uint32_t slice = blockIdx; slice < numSlices_; slice += blockNum) {
            // Pointer to start of this slice
            uint32_t sliceOffset = slice * spatialSize_;

            // ---- First pass: compute sum_x and sum_x2 ----
            float sum_x = 0.0f;
            float sum_x2 = 0.0f;
            uint32_t remaining = spatialSize_;
            uint32_t base = sliceOffset;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                // Read tile from global to local
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base], curLen);
                // Compute sums
                AscendC::LocalTensor<float> scratch = scratchQueue_.AllocTensor<float>();
                // x^2
                AscendC::Mul(scratch, xLocal, xLocal, curLen);
                // accumulate sum_x and sum_x2 using vector reduction? AscendC provides ReduceAdd?
                // We'll use a simple loop over elements since tile is small enough.
                // Alternatively, use AscendC::VecSum to compute partial sums into scalar.
                float partial_sum_x = 0.0f;
                float partial_sum_x2 = 0.0f;
                for (uint32_t i = 0; i < curLen; i++) {
                    partial_sum_x += xLocal(i);
                    partial_sum_x2 += scratch(i);
                }
                sum_x += partial_sum_x;
                sum_x2 += partial_sum_x2;

                // Free tensors
                scratchQueue_.FreeTensor(scratch);
                inQueueX_.FreeTensor(xLocal);
                base += curLen;
                remaining -= curLen;
            }

            // Compute mean and variance
            float mean = sum_x / spatialSize_;
            float var = (sum_x2 / spatialSize_) - mean * mean;
            float sqrtVar = AscendC::Sqrt(var + eps_);

            // ---- Second pass: normalize and write output ----
            remaining = spatialSize_;
            base = sliceOffset;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                // Read input tile into local
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base], curLen);
                // Normalize: (x - mean) / sqrtVar
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Subs(yLocal, xLocal, mean, curLen);
                AscendC::Divs(yLocal, yLocal, sqrtVar, curLen);
                // Write output
                AscendC::DataCopy(yGm_[base], yLocal, curLen);
                // Free tensors
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
                base += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scratchQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numSlices_ {0};
    uint32_t spatialSize_ {0};
    float eps_ {1e-5};
};

extern "C" __global__ __aicore__ void ca6k_00477_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00477Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.numSlices, tiling_data.spatialSize, tiling_data.eps);
    op.Process();
}
