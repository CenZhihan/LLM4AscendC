    int64_t N = input_shape.GetDim(0);
    int64_t C = input_shape.GetDim(1);
    int64_t D = input_shape.GetDim(2);
    int64_t H = input_shape.GetDim(3);
    int64_t W = input_shape.GetDim(4);

    int32_t padLeft   = static_cast<int32_t>(padding[0]);
    int32_t padRight  = static_cast<int32_t>(padding[1]);
    int32_t padTop    = static_cast<int32_t>(padding[2]);
    int32_t padBottom = static_cast<int32_t>(padding[3]);
    int32_t padFront  = static_cast<int32_t>(padding[4]);
    int32_t padBack   = static_cast<int32_t>(padding[5]);

    int64_t Wout = W + padLeft + padRight;
    int64_t Hout = H + padTop + padBottom;
    int64_t Dout = D + padFront + padBack;

    tiling.set_totalOutputLength(totalOutputLength);
    tiling.set_C(static_cast<uint32_t>(C));
    tiling.set_D(static_cast<uint32_t>(D));
    tiling.set_H(static_cast<uint32_t>(H));
    tiling.set_W(static_cast<uint32_t>(W));
    tiling.set_Dout(static_cast<uint32_t>(Dout));
    tiling.set_Hout(static_cast<uint32_t>(Hout));
    tiling.set_Wout(static_cast<uint32_t>(Wout));
    tiling.set_padLeft(padLeft);
    tiling.set_padRight(padRight);
    tiling.set_padTop(padTop);
    tiling.set_padBottom(padBottom);
    tiling.set_padFront(padFront);
    tiling.set_padBack(padBack);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    auto input_shape = x_shape->GetOriginShape();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(0);

    int64_t N = input_shape.GetDim(0);
    int64_t C = input_shape.GetDim(1);
    int64_t D = input_shape.GetDim(2);
    int64_t H = input_shape.GetDim(3);
    int64_t W = input_shape.GetDim(4);

    int64_t Wout = W + static_cast<int64_t>(padding[0]) + static_cast<int64_t>(padding[1]);
    int64_t Hout = H + static_cast<int64_t>(padding[2]) + static_cast<int64_t>(padding[3]);
    int64_t Dout = D + static_cast<int64_t>(padding[4]) + static_cast<int64_t>(padding[5]);

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> dims = {N, C, Dout, Hout, Wout};
    y_shape->SetDimCount(5);
    for (size_t i = 0; i < 5; ++i) {
        y_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src = r'''
#include "kernel_operator.h"

using namespace AscendC;

class KernelCa6k04616Custom {
public:
    __aicore__ inline KernelCa6k04616Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        totalOutputLength_ = td.totalOutputLength;
        C_ = td.C;
        D_ = td.D;
        H_ = td.H;
        W_ = td.W;
        Dout_ = td.Dout;
        Hout_ = td.Hout;
        Wout_ = td.Wout;
        padLeft_ = td.padLeft;
        padTop_  = td.padTop;
        padFront_ = td.padFront;
        blockDim_ = td.blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        if (totalOutputLength_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = blockDim_;
        uint32_t tileSize = (totalOutputLength_ + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * tileSize;
        uint32_t end = (blockIdx + 1) * tileSize;
        if (end > totalOutputLength_) end = totalOutputLength_;

        uint32_t strideH = Wout_;
        uint32_t strideD = Hout_ * Wout_;
        uint32_t strideC = Dout_ * strideD;
        uint32_t strideN = C_ * strideC;

        // Input strides
        uint32_t iStrideW = W_;
        uint32_t iStrideH = H_ * W_;
        uint32_t iStrideD = D_ * iStrideH;
        uint32_t iStrideC = C_ * iStrideD;
        uint32_t iStrideN = C_ * iStrideC;

        for (uint32_t linearIdx = start; linearIdx < end; ++linearIdx) {
            // Decode output indices
            uint32_t n = linearIdx / strideN;
            uint32_t rem1 = linearIdx % strideN;
            uint32_t c = rem1 / strideC;
            uint32_t rem2 = rem1 % strideC;
            uint32_t od = rem2 / strideD;
            uint32_t rem3 = rem2 % strideD;
            uint32_t oh = rem3 / strideH;
            uint32_t ow = rem3 % strideH;

            // Compute input indices with circular wrapping
            int32_t id = (static_cast<int32_t>(od) - padFront_) % static_cast<int32_t>(D_);
            if (id < 0) id += D_;
            int32_t ih = (static_cast<int32_t>(oh) - padTop_) % static_cast<int32_t>(H_);
            if (ih < 0) ih += H_;
            int32_t iw = (static_cast<int32_t>(ow) - padLeft_) % static_cast<int32_t>(W_);
            if (iw < 0) iw += W_;

            uint32_t inputIdx = n * iStrideN + c * iStrideC + static_cast<uint32_t>(id) * iStrideD
                              + static_cast<uint32_t>(ih) * iStrideH + static_cast<uint32_t>(iw);
            yGm_[linearIdx] = xGm_[inputIdx];
];
];
