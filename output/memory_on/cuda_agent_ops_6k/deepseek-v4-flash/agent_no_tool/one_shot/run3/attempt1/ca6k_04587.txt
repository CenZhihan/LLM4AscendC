static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04587CustomTilingData tiling;
    auto inputShape = context->GetInputShape(0);
    const uint32_t n = inputShape->GetDim(0);
    const uint32_t c = inputShape->GetDim(1);
    const uint32_t inD = inputShape->GetDim(2);
    const uint32_t inH = inputShape->GetDim(3);
    const uint32_t inW = inputShape->GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(2);
    uint32_t kd = static_cast<uint32_t>(kernel_size[0]);
    uint32_t kh = static_cast<uint32_t>(kernel_size[1]);
    uint32_t kw = static_cast<uint32_t>(kernel_size[2]);
    uint32_t sd = static_cast<uint32_t>(stride[0]);
    uint32_t sh = static_cast<uint32_t>(stride[1]);
    uint32_t sw = static_cast<uint32_t>(stride[2]);
    uint32_t pd = static_cast<uint32_t>(padding[0]);
    uint32_t ph = static_cast<uint32_t>(padding[1]);
    uint32_t pw = static_cast<uint32_t>(padding[2]);

    uint32_t od = (inD + 2 * pd - kd) / sd + 1;
    uint32_t oh = (inH + 2 * ph - kh) / sh + 1;
    uint32_t ow = (inW + 2 * pw - kw) / sw + 1;

    uint32_t totalLength = n * c * od * oh * ow;
    // tileLength: number of output elements processed per DMA burst
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * (kd * kh * kw + 1));
    if (maxTile > 4096) maxTile = 4096;
    if (totalLength < maxTile) maxTile = totalLength;
    uint32_t tileLength = maxTile;
    uint32_t blockDim = 24; // conservative

    context->SetBlockDim(blockDim);
    tiling.totalLength = totalLength;
    tiling.tileLength = tileLength;
    tiling.blockDim = blockDim;
    tiling.n = n;
    tiling.c = c;
    tiling.inD = inD;
    tiling.inH = inH;
    tiling.inW = inW;
    tiling.kd = kd;
    tiling.kh = kh;
    tiling.kw = kw;
    tiling.sd = sd;
    tiling.sh = sh;
    tiling.sw = sw;
    tiling.od = od;
    tiling.oh = oh;
    tiling.ow = ow;
    tiling.pad_d = pd;
    tiling.pad_h = ph;
    tiling.pad_w = pw;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src = '''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04587Custom {
public:
    __aicore__ inline KernelCa6k04587Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        totalLen_ = td.totalLength;
        tileLen_ = td.tileLength;
        n_ = td.n;
        c_ = td.c;
        inD_ = td.inD;
        inH_ = td.inH;
        inW_ = td.inW;
        kd_ = td.kd;
        kh_ = td.kh;
        kw_ = td.kw;
        sd_ = td.sd;
        sh_ = td.sh;
        sw_ = td.sw;
        od_ = td.od;
        oh_ = td.oh;
        ow_ = td.ow;
        pad_d_ = td.pad_d;
        pad_h_ = td.pad_h;
        pad_w_ = td.pad_w;
        windowSize_ = kd_ * kh_ * kw_;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, windowSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, windowSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        uint32_t stride = blockNum;
        uint32_t start = blockIdx;
        for (uint32_t idx = start; idx < totalLen_; idx += stride) {
            // Decompose idx into (n, c, od, oh, ow)
            uint32_t tmp = idx;
            uint32_t ow = tmp % ow_; tmp /= ow_;
            uint32_t oh = tmp % oh_; tmp /= oh_;
            uint32_t od = tmp % od_; tmp /= od_;
            uint32_t c = tmp % c_; tmp /= c_;
            uint32_t n = tmp;

            // Compute input position (top-left of window)
            int64_t id = static_cast<int64_t>(od) * static_cast<int64_t>(sd_) - static_cast<int64_t>(pad_d_);
            int64_t ih = static_cast<int64_t>(oh) * static_cast<int64_t>(sh_) - static_cast<int64_t>(pad_h_);
            int64_t iw = static_cast<int64_t>(ow) * static_cast<int64_t>(sw_) - static_cast<int64_t>(pad_w_);

            // Compute base offset for this (n, c)
            uint32_t planeSize = inD_ * inH_ * inW_;
            uint32_t baseOffset = (n * c_ + c) * planeSize;

            // Load window
            LocalTensor<float> winLocal = inQueueX_.AllocTensor<float>();
            uint32_t srcOffset = baseOffset + static_cast<uint32_t>(id) * inH_ * inW_ +
                                 static_cast<uint32_t>(ih) * inW_ + static_cast<uint32_t>(iw);
            DataCopy(winLocal, xGm_[srcOffset], windowSize_);
            inQueueX_.EnQue(winLocal);

            // Compute max
            LocalTensor<float> winDeque = inQueueX_.DeQue<float>();
            LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
            // ReduceMax expects output scalar tensor of length 1
            LocalTensor<float> maxLocal = outQueueY_.AllocTensor<float>();
            ReduceMax(maxLocal, winDeque, tmpLocal, windowSize_);
            tmpQueue_.FreeTensor(tmpLocal);
            outQueueY_.EnQue(maxLocal);
            inQueueX_.FreeTensor(winDeque);

            // Store result
            LocalTensor<float> maxOut = outQueueY_.DeQue<float>();
            yGm_[idx] = maxOut(0);
            outQueueY_.FreeTensor(maxOut);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t n_, c_;
    uint32_t inD_, inH_, inW_;
    uint32_t kd_, kh_, kw_;
    uint32_t sd_, sh_, sw_;
    uint32_t od_, oh_, ow_;
    uint32_t pad_d_, pad_h_, pad_w_;
    uint32_t windowSize_;
};

extern "C" __global__ __aicore__ void ca6k_04587_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelCa6k04587Custom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}
'''

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_04587_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_channels, pool_size, stride):
        super(ModelNew, self).__init__()
        # Store parameters for the custom op call
        self.pool_size = pool_size
        self.stride = stride
        self.padding = (0, 0, 0)

    def forward(self, *inputs):
        x = inputs[0]
        fn = getattr(custom_ops_lib, _FN)
        # kernel_size, stride, padding are passed as attributes to the custom op
        return fn(x, self.pool_size, self.stride, self.padding)
