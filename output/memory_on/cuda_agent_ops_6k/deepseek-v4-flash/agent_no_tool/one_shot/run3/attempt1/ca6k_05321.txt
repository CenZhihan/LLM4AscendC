static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k05321CustomTilingData tiling;

    const gert::StorageShape *xStorageShape = context->GetInputShape(0);
    const std::vector<int64_t> xShape = xStorageShape->GetOriginShape().GetDims();
    uint32_t N = static_cast<uint32_t>(xShape[0]);
    uint32_t C = static_cast<uint32_t>(xShape[1]);
    uint32_t H = static_cast<uint32_t>(xShape[2]);
    uint32_t W = static_cast<uint32_t>(xShape[3]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scalePtr = attrs->GetAttrPointer<float>(0);
    float scale = *scalePtr;

    uint32_t H_out = static_cast<uint32_t>(static_cast<double>(H) * scale);
    uint32_t W_out = static_cast<uint32_t>(static_cast<double>(W) * scale);

    uint32_t totalLength = N * C * H_out * W_out;

    // Determine tile size: 64x64 for input (so output 128x128 at scale=2)
    uint32_t tileH = 64;
    uint32_t tileW = 64;
    // Adjust if image smaller
    if (tileH > H) tileH = H;
    if (tileW > W) tileW = W;
    // UB budget: input tile with 1-pixel border: (tileH+1)*(tileW+1) floats
    // output tile: (tileH*scale)*(tileW*scale) floats
    // temporary: same size as output tile
    uint32_t inputTileSize = (tileH + 1) * (tileW + 1) * sizeof(float);
    uint32_t outputTileSize = static_cast<uint32_t>(tileH * scale) *
                              static_cast<uint32_t>(tileW * scale) * sizeof(float);
    uint32_t totalTileBytes = inputTileSize + 2 * outputTileSize;
    if (totalTileBytes > UB_BUDGET_BYTES) {
        // Reduce tile size
        tileH = 32;
        tileW = 32;
        inputTileSize = (tileH + 1) * (tileW + 1) * sizeof(float);
        outputTileSize = static_cast<uint32_t>(tileH * scale) *
                         static_cast<uint32_t>(tileW * scale) * sizeof(float);
        totalTileBytes = inputTileSize + 2 * outputTileSize;
    }

    uint32_t blockDim = 8; // fixed for this kernel
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_scale(scale);
    tiling.set_totalLength(totalLength);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k05321Custom {
public:
    __aicore__ inline KernelCa6k05321Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t H_out, uint32_t W_out, float scale,
                                uint32_t tileH, uint32_t tileW)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        H_out_ = H_out; W_out_ = W_out; scale_ = scale;
        tileH_ = tileH; tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * H * W);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C * H_out * W_out);

        inputTileH_ = tileH_ + 1;
        inputTileW_ = tileW_ + 1;
        uint32_t inputTileSize = inputTileH_ * inputTileW_;
        outputTileH_ = static_cast<uint32_t>(tileH_ * scale_);
        outputTileW_ = static_cast<uint32_t>(tileW_ * scale_);
        uint32_t outputTileSize = outputTileH_ * outputTileW_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputTileSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outputTileSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, outputTileSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || H_ == 0 || W_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Grid-stride over output tiles (in terms of input tile indices).
        // Each block processes a subset of input tile positions.
        uint32_t totalTilesX = (H_ + tileH_ - 1) / tileH_;
        uint32_t totalTilesY = (W_ + tileW_ - 1) / tileW_;
        uint32_t totalTiles = N_ * C_ * totalTilesX * totalTilesY;

        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            // Decode tile index
            uint32_t tmp = tileIdx;
            uint32_t tY = tmp % totalTilesY; tmp /= totalTilesY;
            uint32_t tX = tmp % totalTilesX; tmp /= totalTilesX;
            uint32_t cIdx = tmp % C_; tmp /= C_;
            uint32_t nIdx = tmp;

            uint32_t startRow = tX * tileH_;
            uint32_t startCol = tY * tileW_;
            uint32_t endRow = startRow + tileH_;
            uint32_t endCol = startCol + tileW_;
            if (endRow > H_) endRow = H_;
            if (endCol > W_) endCol = W_;

            // Load input tile with extra border
            uint32_t loadRows = (endRow - startRow) + 1; // +1 for top border? Actually need bottom border too; if endRow==H, no bottom.
            uint32_t loadCols = (endCol - startCol) + 1;
            // Clamp to image bounds
            if (startRow + inputTileH_ > H_) loadRows = H_ - startRow;
            if (startCol + inputTileW_ > W_) loadCols = W_ - startCol;
            // Copy input region to local
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            // We'll copy row by row? Use DataCopy2D? Use simple loop.
            // Simple: for each row, copy contiguous segment.
            for (uint32_t r = 0; r < loadRows; ++r) {
                uint32_t srcOffset = (nIdx * C_ + cIdx) * H_ * W_ + (startRow + r) * W_ + startCol;
                AscendC::LocalTensor<float> dst = xLocal[r * inputTileW_];
                AscendC::DataCopy(dst, xGm_[srcOffset], loadCols);
            }
            // Pad the rest with zeros? We'll leave as is, remaining entries uninitialized, but we'll only access valid region.

            // Compute output tile
            uint32_t outStartRow = static_cast<uint32_t>(startRow * scale_);
            uint32_t outStartCol = static_cast<uint32_t>(startCol * scale_);
            uint32_t outEndRow = static_cast<uint32_t>(endRow * scale_);
            uint32_t outEndCol = static_cast<uint32_t>(endCol * scale_);
            uint32_t outTileRows = outEndRow - outStartRow;
            uint32_t outTileCols = outEndCol - outStartCol;

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            // Bilinear interpolation for each output pixel in this tile
            for (uint32_t oy = 0; oy < outTileRows; ++oy) {
                float fy = (oy + 0.5f) / scale_ - 0.5f; // input coordinate in pixel space
                // Actually for bilinear, output pixel at (ox, oy) maps to input position (ox/scale, oy/scale)
                // But standard bilinear uses coordinate mapping: input_x = (ox+0.5)/scale - 0.5
                // We'll use simpler: input_x = ox / scale, input_y = oy / scale (align corners false)
                float inY = static_cast<float>(tY * tileW_ + oy) / scale_; // approximate, adjust
                // Better: compute exact mapping using output coordinates
                // We'll use standard PyTorch align_corners=False mapping
                // input_x = (ox + 0.5) / scale - 0.5
                // but for simplicity, use (ox) / scale assuming scale integer? scale=2.
                // Since scale=2, we can do direct: each output pixel corresponds to 2x2 block? But still need interpolation.
                // Let's use the exact formula.
                float inY = (outStartRow + oy + 0.5f) / scale_ - 0.5f;
                float inX_float = (outStartCol + 0.5f) / scale_ - 0.5f; // we'll loop over ox inside

                // Compute input integer coordinates and fractions
                int32_t iy0 = static_cast<int32_t>(inY);
                int32_t iy1 = iy0 + 1;
                float dy = inY - iy0;

                for (uint32_t ox = 0; ox < outTileCols; ++ox) {
                    float inX = (outStartCol + ox + 0.5f) / scale_ - 0.5f;
                    int32_t ix0 = static_cast<int32_t>(inX);
                    int32_t ix1 = ix0 + 1;
                    float dx = inX - ix0;

                    // Clamp to valid input region
                    if (iy0 < 0) iy0 = 0;
                    if (iy0 >= static_cast<int32_t>(H_)) iy0 = H_ - 1;
                    if (iy1 < 0) iy1 = 0;
                    if (iy1 >= static_cast<int32_t>(H_)) iy1 = H_ - 1;
                    if (ix0 < 0) ix0 = 0;
                    if (ix0 >= static_cast<int32_t>(W_)) ix0 = W_ - 1;
                    if (ix1 < 0) ix1 = 0;
                    if (ix1 >= static_cast<int32_t>(W_)) ix1 = W_ - 1;

                    // Load four values from local input tile (adjusted for local indexing)
                    // Local tile offset: (startRow, startCol) base; we stored rows from startRow with loadCols columns.
                    // But we need to map global indices to local. We'll compute local row = inY - startRow, col = inX - startCol.
                    int32_t localRow = iy0 - static_cast<int32_t>(startRow);
                    int32_t localCol = ix0 - static_cast<int32_t>(startCol);
                    float v00 = (localRow >=0 && localCol >=0 && localRow < static_cast<int32_t>(loadRows) && localCol < static_cast<int32_t>(loadCols)) ? xLocal[localRow * inputTileW_ + localCol] : 0.0f;
                    localRow = iy0 - static_cast<int32_t>(startRow);
                    localCol = ix1 - static_cast<int32_t>(startCol);
                    float v01 = (localRow >=0 && localCol >=0 && localRow < static_cast<int32_t>(loadRows) && localCol < static_cast<int32_t>(loadCols)) ? xLocal[localRow * inputTileW_ + localCol] : 0.0f;
                    localRow = iy1 - static_cast<int32_t>(startRow);
                    localCol = ix0 - static_cast<int32_t>(startCol);
                    float v10 = (localRow >=0 && localCol >=0 && localRow < static_cast<int32_t>(loadRows) && localCol < static_cast<int32_t>(loadCols)) ? xLocal[localRow * inputTileW_ + localCol] : 0.0f;
                    localRow = iy1 - static_cast<int32_t>(startRow);
                    localCol = ix1 - static_cast<int32_t>(startCol);
                    float v11 = (localRow >=0 && localCol >=0 && localRow < static_cast<int32_t>(loadRows) && localCol < static_cast<int32_t>(loadCols)) ? xLocal[localRow * inputTileW_ + localCol] : 0.0f;

                    float top = v00 * (1.0f - dx) + v01 * dx;
                    float bot = v10 * (1.0f - dx) + v11 * dx;
                    float val = top * (1.0f - dy) + bot * dy;

                    yLocal[oy * outTileCols + ox] = val;
                }
            }

            // Write output tile to global memory
            for (uint32_t oy = 0; oy < outTileRows; ++oy) {
                uint32_t dstOffset = (nIdx * C_ + cIdx) * H_out_ * W_out_ + (outStartRow + oy) * W_out_ + outStartCol;
                AscendC::DataCopy(yGm_[dstOffset], yLocal[oy * outTileCols], outTileCols);
            }

            tmpQueue_.FreeTensor(tLocal);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, H_, W_;
    uint32_t H_out_, W_out_;
    float scale_;
    uint32_t tileH_, tileW_;
    uint32_t inputTileH_, inputTileW_;
    uint32_t outputTileH_, outputTileW_;
};

extern "C" __global__ __aicore__ void ca6k_05321_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k05321Custom op;
    op.Init(x, y, tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.H_out, tiling_data.W_out, tiling_data.scale,
            tiling_data.tileH, tiling_data.tileW);
    op.Process();
}

#include <torch/library.h>

at::Tensor upsample_bilinear_impl_npu(const at::Tensor& self, double scale_factor) {
    at::Tensor result = at::empty_like(self);
    // For bilinear upsampling, output size is scale_factor * input spatial dims.
    // We'll compute output shape in host code.
    int64_t N = self.size(0);
    int64_t C = self.size(1);
    int64_t H = self.size(2);
    int64_t W = self.size(3);
    int64_t H_out = static_cast<int64_t>(static_cast<double>(H) * scale_factor);
    int64_t W_out = static_cast<int64_t>(static_cast<double>(W) * scale_factor);
    result = at::empty({N, C, H_out, W_out}, self.options());
    EXEC_NPU_CMD(aclnnCa6k05321Custom, self, scale_factor, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_05321_custom", &upsample_bilinear_impl_npu);

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_05321_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, scale_factor):
        super(ModelNew, self).__init__()
        self.scale_factor = scale_factor
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1)

    def forward(self, *inputs):
        x = inputs[0]
        x = getattr(custom_ops_lib, _FN)(x, self.scale_factor)
        x = self.conv(x)
