kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t MAX_CHUNK_J = 32;

class KernelCa6k04590Custom {
public:
    __aicore__ inline KernelCa6k04590Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t n, uint32_t d,
                                float p)
    {
        totalLen_ = totalLength;
        n_ = n;
        d_ = d;
        p_ = p;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n * d);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, d * sizeof(float));
        pipe_.InitBuffer(inQueueJ_, BUFFER_NUM, MAX_CHUNK_J * d * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, d * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || n_ == 0 || d_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Distribute i values among blocks
        uint32_t rowsPerBlock = (n_ + blockNum - 1) / blockNum;
        uint32_t startI = blockIdx * rowsPerBlock;
        uint32_t endI = startI + rowsPerBlock;
        if (endI > n_) endI = n_;

        // Compute base output index for this block: number of pairs before startI
        uint32_t baseIndex = 0;
        for (uint32_t i = 0; i < startI; i++) {
            baseIndex += (n_ - i - 1);
        }

        uint32_t outputOffset = baseIndex;

        for (uint32_t i = startI; i < endI; i++) {
            // Load row i (entire row) into local buffer
            AscendC::LocalTensor<float> rowIBuf = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(rowIBuf, xGm_[i * d_], d_);
            // We will process all j > i for this i
            uint32_t startJ = i + 1;
            uint32_t endJ = n_;
            uint32_t j = startJ;
            while (j < endJ) {
                uint32_t chunkEnd = j + MAX_CHUNK_J;
                if (chunkEnd > endJ) chunkEnd = endJ;
                uint32_t chunkLen = chunkEnd - j;
                // Load chunk of rows into local 2D buffer
                AscendC::LocalTensor<float> rowsJBuf = inQueueJ_.AllocTensor<float>();
                for (uint32_t k = 0; k < chunkLen; k++) {
                    AscendC::DataCopy(rowsJBuf[k * d_], xGm_[(j + k) * d_], d_);
                }
                // Process each j in chunk
                for (uint32_t k = 0; k < chunkLen; k++) {
                    uint32_t curJ = j + k;
                    // Compute distance between row i and row curJ
                    // diff = rowI - rowJ, square, sum, sqrt
                    AscendC::LocalTensor<float> diff = tmpQueue_.AllocTensor<float>();
                    AscendC::Sub(diff, rowIBuf, rowsJBuf[k * d_], d_);
                    AscendC::Mul(diff, diff, diff, d_);
                    float sum = 0.0f;
                    AscendC::LocalTensor<float> sumTensor = scalarQueue_.AllocTensor<float>();
                    AscendC::ReduceAdd(sumTensor, diff, d_);
                    sum = sumTensor.GetValue(0);
                    scalarQueue_.FreeTensor(sumTensor);
                    // sqrt
                    AscendC::LocalTensor<float> dist = scalarQueue_.AllocTensor<float>();
                    dist.SetValue(0, sum);
                    AscendC::Sqrt(dist, dist, 1);
                    float distVal = dist.GetValue(0);
                    scalarQueue_.FreeTensor(dist);
                    // Write to output
                    yGm_.SetValue(outputOffset, distVal);
                    outputOffset++;
                    tmpQueue_.FreeTensor(diff);
                }
                inQueueJ_.FreeTensor(rowsJBuf);
                j = chunkEnd;
            }
            inQueueX_.FreeTensor(rowIBuf);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueJ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t n_ {0};
    uint32_t d_ {0};
    float p_ {2.0f};
};

extern "C" __global__ __aicore__ void ca6k_04590_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04590Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.n, tiling_data.d, tiling_data.p);
    op.Process();
}
