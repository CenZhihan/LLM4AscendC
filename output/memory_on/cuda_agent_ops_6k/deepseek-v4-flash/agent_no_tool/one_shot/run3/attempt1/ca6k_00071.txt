static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  
static constexpr uint32_t ALIGN_ELEMS = 2048;            
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00071CustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(xShape->GetOriginShape().GetShapeSize());
    uint32_t ndim = static_cast<uint32_t>(xShape->GetOriginShape().GetDimNum());
    uint32_t numRows = (ndim >= 1) ? static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0)) : 0;
    uint32_t numCols = (ndim >= 2) ? static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1)) : 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t dim = *(attrs->GetAttrPointer<int32_t>(0));

    uint32_t tileLen = numCols;  // process all columns per tile
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (tileLen > maxTile) {
        tileLen = AlignDown(maxTile, ALIGN_ELEMS);
    }
    if (tileLen > HARD_CAP_ELEMS) tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.totalLength = totalLength;
    tiling.numRows = numRows;
    tiling.numCols = numCols;
    tiling.tileLength = tileLen;
    tiling.dim = dim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00071Custom {
public:
    __aicore__ inline KernelCa6k00071Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t numRows, uint32_t numCols,
                                uint32_t tileLength, int32_t dim)
    {
        totalLen_ = totalLength;
        N_ = numRows;
        C_ = numCols;
        tileLen_ = tileLength;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0u || tileLen_ == 0u) return;
        if (dim_ != 0) return; // only dim 0 implemented

        uint32_t colTile = (tileLen_ < C_) ? tileLen_ : C_;
        uint32_t cStart = 0u;
        while (cStart < C_) {
            uint32_t curColTile = (cStart + colTile <= C_) ? colTile : (C_ - cStart);

            // initialize accumulator to zero
            LocalTensor<float> acc = accQueue_.AllocTensor<float>();
            AscendC::ClearTensor(acc, curColTile);
            for (uint32_t i = 0u; i < N_; ++i) {
                uint32_t offset = i * C_ + cStart;

                // prefetch input for this row
                LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curColTile);
                inQueueX_.EnQue(xLocal);

                // compute: accum += x
                LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
                AscendC::Add(acc, acc, xDeq, curColTile);
                inQueueX_.FreeTensor(xDeq);

                // write output
                LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, acc, curColTile);
                outQueueY_.EnQue(yLocal);

                LocalTensor<float> yDeq = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[offset], yDeq, curColTile);
                outQueueY_.FreeTensor(yDeq);
            }
            accQueue_.FreeTensor(acc);

            cStart += colTile;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0u};
    uint32_t N_ {0u};
    uint32_t C_ {0u};
    uint32_t tileLen_ {0u};
    int32_t dim_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00071_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00071Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.numRows, tiling_data.numCols, tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
