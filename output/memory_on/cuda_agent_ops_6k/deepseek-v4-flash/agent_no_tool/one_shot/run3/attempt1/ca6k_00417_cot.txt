class KernelCa6k00417Custom {
public:
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, ...) {
        // set params
    }
    __aicore__ inline void Process() {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();
        uint32_t tileLen = tileLength_;
        uint32_t totalOut = totalLength_;
        for (uint32_t start = blockIdx * tileLen; start < totalOut; start += blockNum * tileLen) {
            uint32_t end = min(start + tileLen, totalOut);
            ComputeRange(start, end);
        }
    }
private:
    __aicore__ inline void ComputeRange(uint32_t start, uint32_t end) {
        for (uint32_t idx = start; idx < end; ++idx) {
            // compute n, c, h_out, w_out
            uint32_t w_out = idx % out_w_;
            uint32_t temp = idx / out_w_;
            uint32_t h_out = temp % out_h_;
            temp /= out_h_;
            uint32_t c = temp % c_;
            uint32_t n = temp / c_;
            // compute max over window
            int h_in_start = h_out * stride_h_ - pad_h_;
            int w_in_start = w_out * stride_w_ - pad_w_;
            float max_val = -FLT_MAX;
            for (int kh = 0; kh < kernel_h_; ++kh) {
                int h_in = h_in_start + kh;
                if (h_in < 0 || h_in >= h_) continue;
                for (int kw = 0; kw < kernel_w_; ++kw) {
                    int w_in = w_in_start + kw;
                    if (w_in < 0 || w_in >= w_) continue;
                    uint32_t in_idx = ((n * c_ + c) * h_ + h_in) * w_ + w_in;
                    float val = xGm_[in_idx];
                    if (val > max_val) max_val = val;
                }
            }
            yGm_[idx] = max_val;
        }
    }
};
