kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;
constexpr int32_t MAX_TILE_H = 64;
constexpr int32_t MAX_TILE_W = 64;

class KernelCa6k00346Custom {
public:
    __aicore__ inline KernelCa6k00346Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
        GET_TILING_DATA(td, tiling);
        N_ = td.N;
        C_ = td.C;
        inputH_ = td.inputH;
        inputW_ = td.inputW;
        outputH_ = td.outputH;
        outputW_ = td.outputW;
        tileH_ = td.tileH;
        tileW_ = td.tileW;
        scaleFactor_ = td.scaleFactor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * inputH_ * inputW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * outputH_ * outputW_);

        maxInputTileH_ = tileH_ + 1;
        maxInputTileW_ = tileW_ + 1;
        maxInputTileSize_ = maxInputTileH_ * maxInputTileW_;
        outTileSize_ = tileH_ * tileW_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, maxInputTileSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outTileSize_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t numTilesH = (outputH_ + tileH_ - 1) / tileH_;
        uint32_t numTilesW = (outputW_ + tileW_ - 1) / tileW_;
        uint32_t tilesPerChannel = numTilesH * numTilesW;

        uint32_t n = blockIdx / (C_ * tilesPerChannel);
        uint32_t rem = blockIdx % (C_ * tilesPerChannel);
        uint32_t c = rem / tilesPerChannel;
        uint32_t tileIdx = rem % tilesPerChannel;
        uint32_t th = tileIdx / numTilesW;
        uint32_t tw = tileIdx % numTilesW;

        uint32_t yStart = th * tileH_;
        uint32_t yEnd = (yStart + tileH_ > outputH_) ? outputH_ : yStart + tileH_;
        uint32_t xStart = tw * tileW_;
        uint32_t xEnd = (xStart + tileW_ > outputW_) ? outputW_ : xStart + tileW_;
        uint32_t oh = yEnd - yStart;
        uint32_t ow = xEnd - xStart;

        // Compute input tile region needed (with 1 extra row/col for interpolation)
        int32_t inRowStart = static_cast<int32_t>(yStart / scaleFactor_);
        int32_t inColStart = static_cast<int32_t>(xStart / scaleFactor_);
        int32_t inRowEnd   = static_cast<int32_t>((yEnd - 1) / scaleFactor_) + 2;
        int32_t inColEnd   = static_cast<int32_t>((xEnd - 1) / scaleFactor_) + 2;
        if (inRowStart < 0) inRowStart = 0;
        if (inColStart < 0) inColStart = 0;
        if (inRowEnd > static_cast<int32_t>(inputH_)) inRowEnd = static_cast<int32_t>(inputH_);
        if (inColEnd > static_cast<int32_t>(inputW_)) inColEnd = static_cast<int32_t>(inputW_);

        uint32_t ih = static_cast<uint32_t>(inRowEnd - inRowStart);
        uint32_t iw = static_cast<uint32_t>(inColEnd - inColStart);

        // Copy input tile to local
        LocalTensor<float> inLocal = inQueueX_.AllocTensor<float>();
        uint32_t inStride = inputW_;
        uint32_t inOffsetBase = n * C_ * inputH_ * inputW_ + c * inputH_ * inputW_;
        for (uint32_t r = 0; r < ih; r++) {
            uint32_t row = inRowStart + r;
            DataCopy(inLocal[r * iw], xGm_[inOffsetBase + row * inStride + inColStart], iw);
        }
        inQueueX_.EnQue(inLocal);

        // Compute output tile
        LocalTensor<float> inDeq = inQueueX_.DeQue<float>();
        LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t oy = 0; oy < oh; oy++) {
            float sy = (yStart + oy) / scaleFactor_;
            int32_t iy = static_cast<int32_t>(sy);
            float dy = sy - iy;
            int32_t iy1 = iy + 1;
            if (iy1 >= static_cast<int32_t>(inputH_)) iy1 = iy;

            int32_t local_iy = iy - inRowStart;
            int32_t local_iy1 = iy1 - inRowStart;
            if (local_iy < 0) local_iy = 0;
            if (local_iy >= static_cast<int32_t>(ih)) local_iy = ih - 1;
            if (local_iy1 < 0) local_iy1 = 0;
            if (local_iy1 >= static_cast<int32_t>(ih)) local_iy1 = ih - 1;

            for (uint32_t ox = 0; ox < (uint32_t)ow; ox++) {
                float sx = (xStart + ox) / scaleFactor_;
                int32_t ix = static_cast<int32_t>(sx);
                float dx = sx - ix;
                int32_t ix1 = ix + 1;
                if (ix1 >= static_cast<int32_t>(inputW_)) ix1 = ix;

                int32_t local_ix = ix - inColStart;
                int32_t local_ix1 = ix1 - inColStart;
                if (local_ix < 0) local_ix = 0;
                if (local_ix >= static_cast<int32_t>(iw)) local_ix = iw - 1;
                if (local_ix1 < 0) local_ix1 = 0;
                if (local_ix1 >= static_cast<int32_t>(iw)) local_ix1 = iw - 1;

                float v00 = inDeq[local_iy * iw + local_ix];
                float v10 = inDeq[local_iy1 * iw + local_ix];
                float v01 = inDeq[local_iy * iw + local_ix1];
                float v11 = inDeq[local_iy1 * iw + local_ix1];

                float result = (1.0f - dy) * (1.0f - dx) * v00 +
                               (1.0f - dy) * dx * v01 +
                               dy * (1.0f - dx) * v10 +
                               dy * dx * v11;
                outLocal[oy * ow + ox] = result;
            }
        }

        outQueueY_.EnQue(outLocal);
        inQueueX_.FreeTensor(inDeq);

        // Copy output tile to global
        LocalTensor<float> outDeq = outQueueY_.DeQue<float>();
        uint32_t outStride = outputW_;
        uint32_t outOffsetBase = n * C_ * outputH_ * outputW_ + c * outputH_ * outputW_;
        for (uint32_t r = 0; r < oh; r++) {
            uint32_t row = yStart + r;
            DataCopy(yGm_[outOffsetBase + row * outStride + xStart], outDeq[r * ow], ow);
        }
        outQueueY_.FreeTensor(outDeq);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, inputH_, inputW_, outputH_, outputW_;
    uint32_t tileH_, tileW_;
    float scaleFactor_;
    uint32_t maxInputTileH_, maxInputTileW_, maxInputTileSize_, outTileSize_;
};

extern "C" __global__ __aicore__ void ca6k_00346_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    KernelCa6k00346Custom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}
