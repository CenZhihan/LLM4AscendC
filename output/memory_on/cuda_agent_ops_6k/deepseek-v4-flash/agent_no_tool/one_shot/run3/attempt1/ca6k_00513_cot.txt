class KernelCa6k00513Custom {
public:
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalSequences, uint32_t width, uint32_t kernelSize, uint32_t stride, uint32_t padding, uint32_t outputLen, uint32_t tileLen) {
        // store params
        totalSequences_ = totalSequences;
        width_ = width;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        outputLen_ = outputLen;
        tileLen_ = tileLen; // not used much
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalSequences * width);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalSequences * outputLen);
        pipe_.InitBuffer(tmpQueue_, 1, kernelSize_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        // distribute sequences across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t perBlock = (totalSequences_ + blockNum - 1) / blockNum;
        uint32_t startSeq = blockIdx * perBlock;
        uint32_t endSeq = startSeq + perBlock;
        if (startSeq >= totalSequences_) return;
        if (endSeq > totalSequences_) endSeq = totalSequences_;

        for (uint32_t seq = startSeq; seq < endSeq; ++seq) {
            ProcessSequence(seq);
        }
    }

private:
    __aicore__ inline void ProcessSequence(uint32_t seq) {
        AscendC::LocalTensor<float> window = tmpQueue_.AllocTensor<float>();
        // iterate over output indices
        for (uint32_t outIdx = 0; outIdx < outputLen_; ++outIdx) {
            int32_t start = static_cast<int32_t>(outIdx * stride_) - static_cast<int32_t>(padding_);
            float maxVal = -1e38f; // negative large
            for (uint32_t k = 0; k < kernelSize_; ++k) {
                int32_t idx = start + static_cast<int32_t>(k);
                float val;
                if (idx >= 0 && idx < static_cast<int32_t>(width_)) {
                    val = xGm_[seq * width_ + idx];
                } else {
                    val = 0.0f;
                }
                window(k) = val;
                // We could do max in loop, but easier to use vector reduce
            }
            // use ReduceMax
            AscendC::ReduceMax(maxVal, window, AscendC::MASK_NMODE, kernelSize_);
            yGm_[seq * outputLen_ + outIdx] = maxVal;
        }
        tmpQueue_.FreeTensor(window);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalSequences_, width_, kernelSize_, stride_, padding_, outputLen_, tileLen_;
};
