#include "ca6k_00286_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00286CustomTilingData tiling;
    const gert::Shape *input_shape = context->GetInputShape(0);
    uint32_t ndim = static_cast<uint32_t>(input_shape->GetDimNum());
    uint32_t N = (ndim >= 4) ? static_cast<uint32_t>(input_shape->GetDim(0)) : 1;
    uint32_t C = (ndim >= 3) ? static_cast<uint32_t>(input_shape->GetDim(1)) : 1;
    uint32_t H = (ndim >= 2) ? static_cast<uint32_t>(input_shape->GetDim(2)) : 1;
    uint32_t W = (ndim >= 1) ? static_cast<uint32_t>(input_shape->GetDim(3)) : 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *scale_ptr = attrs->GetAttrPointer<int32_t>(0);
    int32_t scale = (scale_ptr != nullptr) ? *scale_ptr : 2;

    uint32_t H_out = H * static_cast<uint32_t>(scale);
    uint32_t W_out = W * static_cast<uint32_t>(scale);
    uint32_t totalOutputLength = N * C * H_out * W_out;

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalOutputLength < tileLen) {
        uint32_t aligned = AlignDown(totalOutputLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutputLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(totalOutputLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutputLength(totalOutputLength);
    tiling.set_tileOutputLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H_in(H);
    tiling.set_W_in(W);
    tiling.set_scaleFactor(scale);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    uint32_t ndim = static_cast<uint32_t>(x_shape->GetDimNum());
    for (uint32_t i = 0; i < ndim; i++) {
        y_shape->SetDim(i, x_shape->GetDim(i));
    }
    if (ndim >= 4) {
        const gert::RuntimeAttrs *attrs = context->GetAttrs();
        const int32_t *scale_ptr = attrs->GetAttrPointer<int32_t>(0);
        int32_t scale = (scale_ptr != nullptr) ? *scale_ptr : 2;
        int64_t H = x_shape->GetDim(2);
        int64_t W = x_shape->GetDim(3);
        y_shape->SetDim(2, H * scale);
        y_shape->SetDim(3, W * scale);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00286Custom {
public:
    __aicore__ inline KernelCa6k00286Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputLength,
                                uint32_t tileOutputLength,
                                uint32_t N, uint32_t C,
                                uint32_t H_in, uint32_t W_in,
                                int32_t scaleFactor) {
        totalLen_ = totalOutputLength;
        tileLen_ = tileOutputLength;
        N_ = N;
        C_ = C;
        H_in_ = H_in;
        W_in_ = W_in;
        scale_ = static_cast<uint32_t>(scaleFactor);
        H_out_ = H_in_ * scale_;
        W_out_ = W_in_ * scale_;
        stride_batch_input_ = C_ * H_in_ * W_in_;
        stride_c_input_ = H_in_ * W_in_;
        stride_h_input_ = W_in_;
        stride_batch_output_ = C_ * H_out_ * W_out_;
        stride_c_output_ = H_out_ * W_out_;
        stride_h_output_ = W_out_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        while (true) {
            ComputeAndCopyOut(base, curLen);

            base += strideElems;
            if (base >= totalLen_) break;
            curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
        }
    }

private:
    __aicore__ inline void ComputeAndCopyOut(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; i++) {
            uint32_t idx_out = gmOffset + i;
            uint32_t tmp = idx_out;
            uint32_t b = tmp / stride_batch_output_;
            tmp = tmp % stride_batch_output_;
            uint32_t c = tmp / stride_c_output_;
            tmp = tmp % stride_c_output_;
            uint32_t h_out = tmp / stride_h_output_;
            uint32_t w_out = tmp % stride_h_output_;

            uint32_t h_in = h_out / scale_;
            uint32_t w_in = w_out / scale_;

            uint32_t idx_in = b * stride_batch_input_ + c * stride_c_input_ + h_in * stride_h_input_ + w_in;
            float val = xGm_.GetValue(idx_in);
            val = (val > 0.0f) ? val : 0.0f;
            yLocal.SetValue(i, val);
        }

        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_ {1};
    uint32_t C_ {1};
    uint32_t H_in_ {1};
    uint32_t W_in_ {1};
    uint32_t scale_ {2};
    uint32_t H_out_ {2};
    uint32_t W_out_ {2};
    uint32_t stride_batch_input_ {1};
    uint32_t stride_c_input_ {1};
    uint32_t stride_h_input_ {1};
    uint32_t stride_batch_output_ {1};
    uint32_t stride_c_output_ {1};
    uint32_t stride_h_output_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00286_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00286Custom op;
    op.Init(x, y, tiling_data.totalOutputLength, tiling_data.tileOutputLength,
            tiling_data.N, tiling_data.C, tiling_data.H_in, tiling_data.W_in,
            tiling_data.scaleFactor);
    op.Process();
}
'''
