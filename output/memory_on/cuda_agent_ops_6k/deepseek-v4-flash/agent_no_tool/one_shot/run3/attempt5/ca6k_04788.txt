kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04788Custom {
public:
    __aicore__ inline KernelCa6k04788Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t n, int32_t c,
                                int32_t inD, int32_t inH, int32_t inW,
                                int32_t outD, int32_t outH, int32_t outW,
                                int32_t kernelD, int32_t kernelH, int32_t kernelW,
                                int32_t strideD, int32_t strideH, int32_t strideW,
                                int32_t padD, int32_t padH, int32_t padW,
                                float invKernelVol)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        n_ = n;
        c_ = c;
        inD_ = inD; inH_ = inH; inW_ = inW;
        outD_ = outD; outH_ = outH; outW_ = outW;
        kernelD_ = kernelD; kernelH_ = kernelH; kernelW_ = kernelW;
        strideD_ = strideD; strideH_ = strideH; strideW_ = strideW;
        padD_ = padD; padH_ = padH; padW_ = padW;
        invKernelVol_ = invKernelVol;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t linearIdx = blockIdx; linearIdx < totalLen_; linearIdx += blockNum)
        {
            int32_t ow = static_cast<int32_t>(linearIdx % outW_);
            int32_t tmp = static_cast<int32_t>(linearIdx / outW_);
            int32_t oh = tmp % outH_;
            tmp /= outH_;
            int32_t od = tmp % outD_;
            tmp /= outD_;
            int32_t cIdx = tmp % c_;
            int32_t nIdx = tmp / c_;

            float sum = 0.0f;
            for (int32_t kd = 0; kd < kernelD_; kd++) {
                int32_t id = od * strideD_ - padD_ + kd;
                if (id < 0 || id >= inD_) continue;
                for (int32_t kh = 0; kh < kernelH_; kh++) {
                    int32_t ih = oh * strideH_ - padH_ + kh;
                    if (ih < 0 || ih >= inH_) continue;
                    for (int32_t kw = 0; kw < kernelW_; kw++) {
                        int32_t iw = ow * strideW_ - padW_ + kw;
                        if (iw < 0 || iw >= inW_) continue;
                        uint32_t inIdx = ((static_cast<uint32_t>(nIdx) * static_cast<uint32_t>(c_) +
                                           static_cast<uint32_t>(cIdx)) *
                                           static_cast<uint32_t>(inD_) +
                                           static_cast<uint32_t>(id)) *
                                           static_cast<uint32_t>(inH_) *
                                           static_cast<uint32_t>(inW_) +
                                           static_cast<uint32_t>(ih) *
                                           static_cast<uint32_t>(inW_) +
                                           static_cast<uint32_t>(iw);
                        sum += xGm_.GetValue(inIdx);
                    }
                }
            }
            float result = sum * invKernelVol_;
            yGm_.SetValue(linearIdx, result);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t n_ {0}, c_ {0};
    int32_t inD_ {0}, inH_ {0}, inW_ {0};
    int32_t outD_ {0}, outH_ {0}, outW_ {0};
    int32_t kernelD_ {0}, kernelH_ {0}, kernelW_ {0};
    int32_t strideD_ {0}, strideH_ {0}, strideW_ {0};
    int32_t padD_ {0}, padH_ {0}, padW_ {0};
    float invKernelVol_ {0.0f};
};

extern "C" __global__ __aicore__ void ca6k04788_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04788Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.c,
            tiling_data.inD, tiling_data.inH, tiling_data.inW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW,
            tiling_data.invKernelVol);
    op.Process();
}
'''

import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k04788_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        self._init_inputs = init_inputs

    def forward(self, *inputs):
        x = inputs[0]
        init = self._init_inputs
        kernel_size = init[2]  # (kD, kH, kW)
        stride = init[3]
        padding = init[4]
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, kernel_size[0], kernel_size[1], kernel_size[2],
                  stride[0], stride[1], stride[2],
                  padding[0], padding[1], padding[2])
