#include "ca6k_00382_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t CHANNEL_TILE = 16;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00382CustomTilingData tiling;

    const gert::StorageShape* inputShape = context->GetInputShape(0);
    const gert::StorageShape* weightShape = context->GetInputShape(1);

    uint32_t n = inputShape->GetOriginShape().GetDim(0);
    uint32_t c = inputShape->GetOriginShape().GetDim(1);
    uint32_t h = inputShape->GetOriginShape().GetDim(2);
    uint32_t w = inputShape->GetOriginShape().GetDim(3);

    uint32_t outC = weightShape->GetOriginShape().GetDim(0);
    uint32_t kH = weightShape->GetOriginShape().GetDim(2);
    uint32_t kW = weightShape->GetOriginShape().GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t stride_ptr[2] = {1, 1};
    int64_t padding_ptr[2] = {0, 0};
    int64_t dilation_ptr[2] = {1, 1};
    int64_t groups = 1;

    if (attrs->GetAttrPointer<int64_t>(0) != nullptr) {
        stride_ptr[0] = attrs->GetAttrPointer<int64_t>(0)[0];
        stride_ptr[1] = attrs->GetAttrPointer<int64_t>(0)[1];
    }
    if (attrs->GetAttrPointer<int64_t>(1) != nullptr) {
        padding_ptr[0] = attrs->GetAttrPointer<int64_t>(1)[0];
        padding_ptr[1] = attrs->GetAttrPointer<int64_t>(1)[1];
    }
    if (attrs->GetAttrPointer<int64_t>(2) != nullptr) {
        dilation_ptr[0] = attrs->GetAttrPointer<int64_t>(2)[0];
        dilation_ptr[1] = attrs->GetAttrPointer<int64_t>(2)[1];
    }
    if (attrs->GetAttrPointer<int64_t>(3) != nullptr) {
        groups = *attrs->GetAttrPointer<int64_t>(3);
    }

    int64_t strideH = stride_ptr[0];
    int64_t strideW = stride_ptr[1];
    int64_t padH = padding_ptr[0];
    int64_t padW = padding_ptr[1];
    int64_t dilH = dilation_ptr[0];
    int64_t dilW = dilation_ptr[1];

    uint32_t outH = (h + 2 * padH - dilH * (kH - 1) - 1) / strideH + 1;
    uint32_t outW = (w + 2 * padW - dilW * (kW - 1) - 1) / strideW + 1;

    uint32_t tileC = CHANNEL_TILE;
    uint32_t tileH = 32;
    uint32_t tileW = 32;

    uint32_t input_tile_elems = c * tileH * tileW;
    uint32_t weight_tile_elems = tileC * c * kH * kW;
    uint32_t output_tile_elems = tileC * tileH * tileW;
    uint32_t total_elems = input_tile_elems + weight_tile_elems + output_tile_elems;
    uint32_t needed_bytes = total_elems * sizeof(float) * 3;
    while (needed_bytes > UB_BUDGET_BYTES && tileH > 1 && tileW > 1) {
        tileH /= 2;
        tileW /= 2;
        input_tile_elems = c * tileH * tileW;
        weight_tile_elems = tileC * c * kH * kW;
        output_tile_elems = tileC * tileH * tileW;
        total_elems = input_tile_elems + weight_tile_elems + output_tile_elems;
        needed_bytes = total_elems * sizeof(float) * 3;
    }
    if (tileH == 0) tileH = 1;
    if (tileW == 0) tileW = 1;

    uint32_t blockDim = CeilDiv(outC, tileC) * CeilDiv(outH, tileH) * CeilDiv(outW, tileW);
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_outC(outC);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_dilH(dilH);
    tiling.set_dilW(dilW);
    tiling.set_groups(groups);
    tiling.set_tileC(tileC);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}  // namespace optiling

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00382Custom {
public:
    __aicore__ inline KernelCa6k00382Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t outC, uint32_t outH, uint32_t outW,
                                uint32_t kH, uint32_t kW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilH, uint32_t dilW,
                                uint32_t groups,
                                uint32_t tileC, uint32_t tileH, uint32_t tileW,
                                uint32_t blockDim)
    {
        n_ = n; c_ = c; h_ = h; w_ = w;
        outC_ = outC; outH_ = outH; outW_ = outW;
        kH_ = kH; kW_ = kW;
        strideH_ = strideH; strideW_ = strideW;
        padH_ = padH; padW_ = padW;
        dilH_ = dilH; dilW_ = dilW;
        groups_ = groups;
        tileC_ = tileC; tileH_ = tileH; tileW_ = tileW;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Initialize convolution descriptor
        Conv2DDesc convDesc;
        convDesc.SetPadMode(1); // manual pad
        convDesc.SetPadH(padH);
        convDesc.SetPadW(padW);
        convDesc.SetStrideH(strideH);
        convDesc.SetStrideW(strideW);
        convDesc.SetDilationH(dilH);
        convDesc.SetDilationW(dilW);
        convDesc.SetGroups(groups);
        convDesc.SetInputFormat(FORMAT_NCHW);
        convDesc.SetInputDataType(DT_FLOAT);
        convDesc.SetFilterFormat(FORMAT_NCHW);
        convDesc.SetFilterDataType(DT_FLOAT);
        convDesc.SetOutputFormat(FORMAT_NCHW);
        convDesc.SetOutputDataType(DT_FLOAT);
        convDesc.SetFeatureMapSize(n, c, h, w);
        convDesc.SetFilterSize(outC, c / groups, kH, kW);
        convDesc.SetOutputSize(outH, outW);

        conv_.Init(convDesc);
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || h_ == 0 || w_ == 0) return;
        conv_.SetInputTensor(xGm_, FORMAT_NCHW, c_, h_, w_);
        conv_.SetFilterTensor(weightGm_, FORMAT_NCHW, outC_, kH_, kW_);
        if (biasGm_.GetSize() > 0) {
            conv_.SetBiasTensor(biasGm_);
        }
        conv_.SetOutputTensor(yGm_, FORMAT_NCHW, outC_, outH_, outW_);
        conv_.Forward(true); // synchronous
    }

private:
    AscendC::Conv2D conv_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_, c_, h_, w_;
    uint32_t outC_, outH_, outW_;
    uint32_t kH_, kW_;
    uint32_t strideH_, strideW_, padH_, padW_, dilH_, dilW_;
    uint32_t groups_;
    uint32_t tileC_, tileH_, tileW_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void ca6k_00382_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00382Custom op;
    op.Init(x, weight, bias,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.outC, tiling_data.outH, tiling_data.outW,
            tiling_data.kH, tiling_data.kW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilH, tiling_data.dilW,
            tiling_data.groups,
            tiling_data.tileC, tiling_data.tileH, tiling_data.tileW,
            tiling_data.blockDim);
    op.Process();
}
'''

import torch
import torch_npu
import custom_ops_lib

_FN = "ca6k_00382_custom"

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=(0,0), dilation=(1,1)):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        # Keep the same weight and bias as reference model for numerical match
        self.weight = torch.nn.Parameter(torch.randn(out_channels, in_channels, kernel_size[0], kernel_size[1]))
        self.bias = torch.nn.Parameter(torch.zeros(out_channels))
        # Register the custom op function
        self.custom_op = getattr(custom_ops_lib, _FN)

    def forward(self, x):
        return self.custom_op(x, self.weight, self.bias,
                               stride=self.stride,
                               padding=self.padding,
                               dilation=self.dilation,
                               groups=1)
