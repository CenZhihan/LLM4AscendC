```cpp
__aicore__ inline void Process()
{
    if (outputWidth_ == 0) return;
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t batch = blockIdx;
    uint32_t offsetX = batch * inputWidth_;
    uint32_t offsetY = batch * outputWidth_;

    uint32_t curStart = 0;
    while (curStart < outputWidth_) {
        uint32_t curTileLen = outputWidth_ - curStart;
        if (curTileLen > tileLen_) curTileLen = tileLen_;
        uint32_t inputSliceLen = (curTileLen > 1) ? ((curTileLen - 1) * stride_ + kernelSize_) : kernelSize_;
        uint32_t inputOff = curStart * stride_;

        // Allocate local tensors
        AscendC::LocalTensor<float> inLocal = pipe_.CreateLocalTensor<float>(inputSliceLen);
        AscendC::LocalTensor<float> yLocal = pipe_.CreateLocalTensor<float>(curTileLen);

        // Copy in
        AscendC::GlobalTensor<float> src;
        src.SetGlobalBuffer(xBase_ + offsetX + inputOff, inputSliceLen);
        AscendC::DataCopy(inLocal, src, inputSliceLen);
        pipe_.Barrier(); // ensure copy done

        // Compute
        AscendC::LocalTensor<float> tmp = pipe_.CreateLocalTensor<float>(kernelSize_);
        for (uint32_t i = 0; i < curTileLen; ++i) {
            // copy window to tmp (contiguous)
            AscendC::GlobalTensor<float> win;
            win.SetGlobalBuffer(xBase_ + offsetX + inputOff + i * stride_, kernelSize_);
            AscendC::DataCopy(tmp, win, kernelSize_);
            pipe_.Barrier();
            // square and sum
            float sum = 0.0f;
            for (uint32_t k = 0; k < kernelSize_; ++k) {
                float v = tmp(k);
                sum += v * v;
            }
            yLocal(i) = sqrtf(sum * invKernelSize_);
        }

        // Copy out
        AscendC::GlobalTensor<float> dst;
        dst.SetGlobalBuffer(yBase_ + offsetY + curStart, curTileLen);
        AscendC::DataCopy(dst, yLocal, curTileLen);
        pipe_.Barrier();

        curStart += curTileLen;
    }
}
```
