#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04608Custom {
public:
    __aicore__ inline KernelCa6k04608Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalElemsPerChannel, uint32_t tileLen,
                                float eps, uint32_t blockDim) {
        totalElemsPerChannel_ = totalElemsPerChannel;
        tileLen_ = tileLen;
        eps_ = eps;

        channelIdx_ = AscendC::GetBlockIdx();
        if (channelIdx_ >= blockDim) return;

        xGm_.SetGlobalBuffer((__gm__ float *)x);
        yGm_.SetGlobalBuffer((__gm__ float *)y);
        weightGm_.SetGlobalBuffer((__gm__ float *)weight, blockDim);
        biasGm_.SetGlobalBuffer((__gm__ float *)bias, blockDim);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));

        weight_ = weightGm_[channelIdx_];
        bias_ = biasGm_[channelIdx_];
    }

    __aicore__ inline void Process() {
        if (totalElemsPerChannel_ == 0 || tileLen_ == 0) return;

        float sum = 0.0f;
        float sumSq = 0.0f;

        uint32_t offset = channelIdx_ * totalElemsPerChannel_;
        uint32_t remaining = totalElemsPerChannel_;
        uint32_t curTileLen;
        AscendC::LocalTensor<float> xLocal;

        // First pass: compute sum and sum of squares
        while (remaining > 0) {
            curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;
            xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curTileLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            for (uint32_t i = 0; i < curTileLen; i++) {
                float v = xLocal(i);
                sum += v;
                sumSq += v * v;
            }
            inQueueX_.FreeTensor(xLocal);

            remaining -= curTileLen;
            offset += curTileLen;
        }

        float mean = sum / totalElemsPerChannel_;
        float var = sumSq / totalElemsPerChannel_ - mean * mean;
        float invStd = 1.0f / sqrt(var + eps_);

        // Second pass: normalize, scale, shift
        offset = channelIdx_ * totalElemsPerChannel_;
        remaining = totalElemsPerChannel_;
        while (remaining > 0) {
            curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;
            xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curTileLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            AscendC::Sub(yLocal, xLocal, mean, curTileLen);
            AscendC::Muls(yLocal, yLocal, invStd, curTileLen);
            AscendC::Muls(yLocal, yLocal, weight_, curTileLen);
            AscendC::Add(yLocal, yLocal, bias_, curTileLen);

            outQueueY_.EnQue<float>(yLocal);
            inQueueX_.FreeTensor(xLocal);

            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yLocal, curTileLen);
            outQueueY_.FreeTensor(yLocal);

            remaining -= curTileLen;
            offset += curTileLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalElemsPerChannel_ {0};
    uint32_t tileLen_ {0};
    uint32_t channelIdx_ {0};
    float weight_ {1.0f};
    float bias_ {0.0f};
    float eps_ {1e-5};
};

extern "C" __global__ __aicore__ void ca6k_04608_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                                         GM_ADDR running_mean, GM_ADDR running_var,
                                                         GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k04608Custom op;
    op.Init(x, y, weight, bias, tiling_data.totalElemsPerChannel, tiling_data.tileLen,
            tiling_data.eps, tiling_data.blockDim);
    op.Process();
}
