host_operator_src = '''
#include "ca6k_00266_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t TILE_HARD_CAP = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00266CustomTilingData tiling;

    auto indices_shape = context->GetInputShape(0)->GetOriginShape();
    auto offsets_shape = context->GetInputShape(1)->GetOriginShape();
    auto weight_shape = context->GetInputShape(2)->GetOriginShape();

    uint32_t totalIndices = static_cast<uint32_t>(indices_shape.GetShapeSize());
    uint32_t numOffsets = static_cast<uint32_t>(offsets_shape.GetShapeSize());
    uint32_t numEmbeddings = static_cast<uint32_t>(weight_shape.GetDim(0));
    uint32_t embeddingDim = static_cast<uint32_t>(weight_shape.GetDim(1));
    uint32_t totalBags = numOffsets - 1;

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > TILE_HARD_CAP) maxTile = AlignDown(TILE_HARD_CAP, ALIGN_ELEMS);
    uint32_t tileDim = (embeddingDim < maxTile) ? embeddingDim : maxTile;
    tileDim = AlignDown(tileDim, ALIGN_ELEMS);
    if (tileDim == 0) tileDim = ALIGN_ELEMS;

    // Distribute bags across blocks
    uint32_t blockNum = context->GetBlockNum();
    uint32_t batchBags = CeilDiv(totalBags, blockNum);
    uint32_t bagStartOffset = context->GetBlockIdx() * batchBags;
    if (bagStartOffset >= totalBags) {
        bagStartOffset = totalBags;
        batchBags = 0;
    }

    context->SetBlockDim(blockNum);

    tiling.set_totalBags(totalBags);
    tiling.set_embeddingDim(embeddingDim);
    tiling.set_numEmbeddings(numEmbeddings);
    tiling.set_tileDim(tileDim);
    tiling.set_totalIndices(totalIndices);
    tiling.set_batchBags(batchBags);
    tiling.set_bagStartOffset(bagStartOffset);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    auto offsets_shape = context->GetInputShape(1);
    auto weight_shape = context->GetInputShape(2);
    uint32_t numOffsets = static_cast<uint32_t>(offsets_shape->GetOriginShape().GetShapeSize());
    uint32_t totalBags = numOffsets - 1;
    uint32_t embeddingDim = static_cast<uint32_t>(weight_shape->GetDim(1));
    gert::Shape *output_shape = context->GetOutputShape(0);
    output_shape->SetDimNum(2);
    output_shape->SetDim(0, totalBags);
    output_shape->SetDim(1, embeddingDim);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
}  // namespace ge

namespace ops {

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00266Custom {
public:
    __aicore__ inline KernelCa6k00266Custom() {}

    __aicore__ inline void Init(GM_ADDR indices, GM_ADDR offsets, GM_ADDR weight,
                                uint32_t totalBags, uint32_t embeddingDim,
                                uint32_t numEmbeddings, uint32_t tileDim,
                                uint32_t totalIndices, uint32_t batchBags,
                                uint32_t bagStartOffset) {
        totalBags_ = totalBags;
        embeddingDim_ = embeddingDim;
        numEmbeddings_ = numEmbeddings;
        tileDim_ = tileDim;
        totalIndices_ = totalIndices;
        batchBags_ = batchBags;
        bagStartOffset_ = bagStartOffset;

        // set global memory pointers
        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices, totalIndices_);
        offsetsGm_.SetGlobalBuffer((__gm__ int64_t*)offsets, totalBags_ + 1);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numEmbeddings_ * embeddingDim_);

        // output: totalBags x embeddingDim
        outputGm_.SetGlobalBuffer((__gm__ float*)weight + numEmbeddings_ * embeddingDim_,
                                   totalBags_ * embeddingDim_);

        // allocate local buffers: for one tile of embeddingDim (float) and one row of weight
        pipe_.InitBuffer(inQueueIndices_, BUFFER_NUM, tileDim_ * sizeof(int64_t));  // not used as index vector
        pipe_.InitBuffer(inQueueWeight_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(outQueueOut_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileDim_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (batchBags_ == 0) return;
        // iterate over bags assigned to this block
        for (uint32_t b = 0; b < batchBags_; b++) {
            uint32_t bagIdx = bagStartOffset_ + b;
            if (bagIdx >= totalBags_) break;
            ProcessBag(bagIdx);
        }
    }

private:
    __aicore__ inline void ProcessBag(uint32_t bagIdx) {
        // get start and end index from offsets
        int64_t start = offsetsGm_.GetValue(bagIdx);
        int64_t end = offsetsGm_.GetValue(bagIdx + 1);
        uint32_t bagLength = static_cast<uint32_t>(end - start);
        if (bagLength == 0) {
            // output zero for this bag
            for (uint32_t d = 0; d < embeddingDim_; d += tileDim_) {
                uint32_t curTile = (embeddingDim_ - d < tileDim_) ? (embeddingDim_ - d) : tileDim_;
                AscendC::LocalTensor<float> outLocal = outQueueOut_.AllocTensor<float>();
                AscendC::Duplicate<float>(outLocal, 0.0f, curTile);
                uint32_t outOffset = bagIdx * embeddingDim_ + d;
                AscendC::DataCopy(outputGm_[outOffset], outLocal, curTile);
                outQueueOut_.FreeTensor(outLocal);
            }
            return;
        }

        // accumulate per-tile
        for (uint32_t d = 0; d < embeddingDim_; d += tileDim_) {
            uint32_t curTile = (embeddingDim_ - d < tileDim_) ? (embeddingDim_ - d) : tileDim_;
            AscendC::LocalTensor<float> accLocal = tmpBuf_.AllocTensor<float>();
            AscendC::Duplicate<float>(accLocal, 0.0f, curTile);

            // iterate over bag's indices
            for (uint32_t i = 0; i < bagLength; i++) {
                int64_t idx = indicesGm_.GetValue(static_cast<uint32_t>(start + i));
                uint32_t rowOffset = static_cast<uint32_t>(idx) * embeddingDim_ + d;
                // load tile row from weight
                AscendC::LocalTensor<float> wLocal = inQueueWeight_.AllocTensor<float>();
                AscendC::DataCopy(wLocal, weightGm_[rowOffset], curTile);
                AscendC::Add(accLocal, accLocal, wLocal, curTile);
                inQueueWeight_.FreeTensor(wLocal);
            }

            // write out
            AscendC::LocalTensor<float> outLocal = outQueueOut_.AllocTensor<float>();
            AscendC::Copy(outLocal, accLocal, curTile);
            uint32_t outOffset = bagIdx * embeddingDim_ + d;
            AscendC::DataCopy(outputGm_[outOffset], outLocal, curTile);
            outQueueOut_.FreeTensor(outLocal);
            tmpBuf_.FreeTensor(accLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIndices_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWeight_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<int64_t> indicesGm_;
    AscendC::GlobalTensor<int64_t> offsetsGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t totalBags_ {0};
    uint32_t embeddingDim_ {0};
    uint32_t numEmbeddings_ {0};
    uint32_t tileDim_ {0};
    uint32_t totalIndices_ {0};
    uint32_t batchBags_ {0};
    uint32_t bagStartOffset_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00266_custom(GM_ADDR indices, GM_ADDR offsets, GM_ADDR weight,
                                                         GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00266Custom op;
    op.Init(indices, offsets, weight,
            tiling_data.totalBags, tiling_data.embeddingDim,
            tiling_data.numEmbeddings, tiling_data.tileDim,
            tiling_data.totalIndices, tiling_data.batchBags,
            tiling_data.bagStartOffset);
    op.Process();
}
'''

#include <torch/library.h>

at::Tensor ca6k_00266_impl_npu(
    const at::Tensor& indices,
    const at::Tensor& offsets,
    const at::Tensor& weight) {
    // output shape: [numBags, embeddingDim]
    int64_t numBags = offsets.size(0) - 1;
    int64_t embeddingDim = weight.size(1);
    at::Tensor result = at::empty({numBags, embeddingDim}, weight.options());
    EXEC_NPU_CMD(aclnnCa6k00266Custom, indices, offsets, weight, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_00266_custom", &ca6k_00266_impl_npu);
}

    m.def("ca6k_00266_custom", &ca6k_00266_impl_npu,

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00266_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_embeddings, embedding_dim, mode="sum"):
        super(ModelNew, self).__init__()
        self.num_embeddings = num_embeddings
        self.embedding_dim = embedding_dim
        # We need to keep the weight from original EmbeddingBag.
        # However, the custom op does not automatically learn the weight.
        # So we store weight as a parameter.
        self.weight = torch.nn.Parameter(
            torch.randn(num_embeddings, embedding_dim, dtype=torch.float32)
        )
        # initialize similar to nn.EmbeddingBag default
        nn.init.normal_(self.weight, mean=0.0, std=1.0)
        self.mode = mode

    def forward(self, input, offsets):
        fn = getattr(custom_ops_lib, _FN)
        return fn(input, offsets, self.weight)
'''
