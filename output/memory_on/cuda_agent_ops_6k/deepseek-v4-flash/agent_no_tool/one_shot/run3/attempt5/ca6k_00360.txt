kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00360Custom {
public:
    __aicore__ inline KernelCa6k00360Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR output, GM_ADDR h_n,
                                GM_ADDR weight_ih_0, GM_ADDR weight_hh_0,
                                GM_ADDR bias_ih_0, GM_ADDR bias_hh_0,
                                GM_ADDR weight_ih_1, GM_ADDR weight_hh_1,
                                GM_ADDR bias_ih_1, GM_ADDR bias_hh_1,
                                uint32_t totalTimesteps, uint32_t batchSize,
                                uint32_t inputSize, uint32_t hiddenSize,
                                uint32_t numLayers, bool useBias)
    {
        totalTimesteps_ = totalTimesteps;
        batchSize_ = batchSize;
        inputSize_ = inputSize;
        hiddenSize_ = hiddenSize;
        numLayers_ = numLayers;
        useBias_ = useBias;
        seqLen_ = totalTimesteps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalTimesteps * batchSize * inputSize);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalTimesteps * batchSize * hiddenSize);
        h_nGm_.SetGlobalBuffer((__gm__ float*)h_n, numLayers * batchSize * hiddenSize);

        weightIH_[0].SetGlobalBuffer((__gm__ float*)weight_ih_0, hiddenSize * inputSize);
        weightHH_[0].SetGlobalBuffer((__gm__ float*)weight_hh_0, hiddenSize * hiddenSize);
        biasIH_[0].SetGlobalBuffer((__gm__ float*)bias_ih_0, hiddenSize);
        biasHH_[0].SetGlobalBuffer((__gm__ float*)bias_hh_0, hiddenSize);

        weightIH_[1].SetGlobalBuffer((__gm__ float*)weight_ih_1, hiddenSize * inputSize);
        weightHH_[1].SetGlobalBuffer((__gm__ float*)weight_hh_1, hiddenSize * hiddenSize);
        biasIH_[1].SetGlobalBuffer((__gm__ float*)bias_ih_1, hiddenSize);
        biasHH_[1].SetGlobalBuffer((__gm__ float*)bias_hh_1, hiddenSize);

        pipe_.InitBuffer(hiddenL0_, 1, batchSize * hiddenSize * sizeof(float));
        pipe_.InitBuffer(hiddenL1_, 1, batchSize * hiddenSize * sizeof(float));
        pipe_.InitBuffer(tempVec_, 1, batchSize * hiddenSize * sizeof(float));
        pipe_.InitBuffer(tempMat_, 1, batchSize * hiddenSize * sizeof(float));
        pipe_.InitBuffer(xStep_, 1, batchSize * inputSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        AscendC::LocalTensor<float> hL0 = hiddenL0_.Get<float>();
        AscendC::LocalTensor<float> hL1 = hiddenL1_.Get<float>();
        AscendC::LocalTensor<float> xStep = xStep_.Get<float>();
        AscendC::LocalTensor<float> tempVec = tempVec_.Get<float>();
        AscendC::LocalTensor<float> tempMat = tempMat_.Get<float>();

        uint32_t totalHSize = batchSize_ * hiddenSize_;
        uint32_t totalISize = batchSize_ * inputSize_;
        AscendC::SetZero(hL0, totalHSize);
        AscendC::SetZero(hL1, totalHSize);

        for (uint32_t t = 0; t < seqLen_; t++)
        {
            uint32_t xOffset = t * totalISize;
            AscendC::DataCopy(xStep, xGm_[xOffset], totalISize);

            computeLayer(xStep, weightIH_[0], weightHH_[0], biasIH_[0], biasHH_[0],
                         hL0, tempVec, tempMat, false);

            computeLayer(hL0, weightIH_[1], weightHH_[1], biasIH_[1], biasHH_[1],
                         hL1, tempVec, tempMat, true);

            uint32_t outOffset = t * totalHSize;
            AscendC::DataCopy(outputGm_[outOffset], hL1, totalHSize);
        }

        AscendC::DataCopy(h_nGm_[0], hL0, totalHSize);
        AscendC::DataCopy(h_nGm_[totalHSize], hL1, totalHSize);
    }

private:
    __aicore__ inline void computeLayer(
        AscendC::LocalTensor<float>& input,
        AscendC::GlobalTensor<float>& wIH,
        AscendC::GlobalTensor<float>& wHH,
        AscendC::GlobalTensor<float>& bIH,
        AscendC::GlobalTensor<float>& bHH,
        AscendC::LocalTensor<float>& hPrev,
        AscendC::LocalTensor<float>& tempVec,
        AscendC::LocalTensor<float>& tempMat,
        bool storeOutput)
    {
        AscendC::DataCopy(tempVec, input, batchSize_ * hiddenSize_);
        if (useBias_)
        {
            for (uint32_t b = 0; b < batchSize_; b++)
            {
                for (uint32_t j = 0; j < hiddenSize_; j++)
                {
                    tempVec(b * hiddenSize_ + j) += bIH(j);
                }
            }
        }
        AscendC::Tanh(hPrev, tempVec, batchSize_ * hiddenSize_);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> hiddenL0_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> hiddenL1_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> xStep_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tempVec_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tempMat_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> outputGm_;
    AscendC::GlobalTensor<float> h_nGm_;
    AscendC::GlobalTensor<float> weightIH_[2];
    AscendC::GlobalTensor<float> weightHH_[2];
    AscendC::GlobalTensor<float> biasIH_[2];
    AscendC::GlobalTensor<float> biasHH_[2];

    uint32_t totalTimesteps_ = 0;
    uint32_t batchSize_ = 0;
    uint32_t inputSize_ = 0;
    uint32_t hiddenSize_ = 0;
    uint32_t numLayers_ = 0;
    uint32_t seqLen_ = 0;
    bool useBias_ = false;
};

extern "C" __global__ __aicore__ void ca6k_00360_custom(
    GM_ADDR x, GM_ADDR output, GM_ADDR h_n,
    GM_ADDR weight_ih_0, GM_ADDR weight_hh_0,
    GM_ADDR bias_ih_0, GM_ADDR bias_hh_0,
    GM_ADDR weight_ih_1, GM_ADDR weight_hh_1,
    GM_ADDR bias_ih_1, GM_ADDR bias_hh_1,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00360Custom op;
    op.Init(x, output, h_n,
            weight_ih_0, weight_hh_0, bias_ih_0, bias_hh_0,
            weight_ih_1, weight_hh_1, bias_ih_1, bias_hh_1,
            tiling_data.get_totalTimesteps(), tiling_data.get_batchSize(),
            tiling_data.get_inputSize(), tiling_data.get_hiddenSize(),
            tiling_data.get_numLayers(), tiling_data.get_useBias());
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, bias=True):
        super().__init__()
        self.ref_rnn = torch.nn.RNN(input_size, hidden_size, num_layers, bias=bias)
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.bias = bias
        self.weight_ih_0 = torch.nn.Parameter(self.ref_rnn.weight_ih_l0.clone())
        self.weight_hh_0 = torch.nn.Parameter(self.ref_rnn.weight_hh_l0.clone())
        self.bias_ih_0 = torch.nn.Parameter(self.ref_rnn.bias_ih_l0.clone()) if bias else None
        self.bias_hh_0 = torch.nn.Parameter(self.ref_rnn.bias_hh_l0.clone()) if bias else None
        self.weight_ih_1 = torch.nn.Parameter(self.ref_rnn.weight_ih_l1.clone())
        self.weight_hh_1 = torch.nn.Parameter(self.ref_rnn.weight_hh_l1.clone())
        self.bias_ih_1 = torch.nn.Parameter(self.ref_rnn.bias_ih_l1.clone()) if bias else None
        self.bias_hh_1 = torch.nn.Parameter(self.ref_rnn.bias_hh_l1.clone()) if bias else None

    def forward(self, x):
        result = custom_ops_lib.ca6k_00360_custom(
            x,
            self.weight_ih_0, self.weight_hh_0,
            self.bias_ih_0, self.bias_hh_0,
            self.weight_ih_1, self.weight_hh_1,
            self.bias_ih_1, self.bias_hh_1
        )
        return result[0], result[1]
