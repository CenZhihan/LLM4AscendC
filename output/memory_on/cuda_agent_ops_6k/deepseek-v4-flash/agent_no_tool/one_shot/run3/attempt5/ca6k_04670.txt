static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k04670CustomTilingData tiling;

  const gert::StorageShape *xShape = context->GetInputShape(0);
  const uint32_t N = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
  const uint32_t C = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
  const uint32_t H = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
  const uint32_t W = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));
  const uint32_t totalLength = N * C * H * W;

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  const float *eps = attrs->GetAttrPointer<float>(0);

  constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
  constexpr uint32_t ALIGN_ELEMS = 2048;
  constexpr uint32_t HARD_CAP_ELEMS = 65536;

  uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));  // double buffer for x and y
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = maxTile;
  uint32_t spatialPerChannel = H * W;
  if (spatialPerChannel < tileLen) {
    tileLen = AlignDown(spatialPerChannel, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = spatialPerChannel;
  }

  uint32_t blockDim = 16;
  context->SetBlockDim(blockDim);

  tiling.set_totalLength(totalLength);
  tiling.set_tileLength(tileLen);
  tiling.set_blockDim(blockDim);
  tiling.set_N(N);
  tiling.set_C(C);
  tiling.set_H(H);
  tiling.set_W(W);
  tiling.set_eps(*eps);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}
