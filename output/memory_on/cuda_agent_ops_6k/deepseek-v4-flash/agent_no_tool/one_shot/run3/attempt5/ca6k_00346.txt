static constexpr uint32_t MAX_BLOCK_DIM = 24;

    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00346CustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::Shape &origin = x_shape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t C = static_cast<uint32_t>(origin.GetDim(1));
    uint32_t inputH = static_cast<uint32_t>(origin.GetDim(2));
    uint32_t inputW = static_cast<uint32_t>(origin.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scaleFactor = attrs->GetAttrPointer<float>(0);
    float sf = *scaleFactor;

    uint32_t outputH = static_cast<uint32_t>(inputH * sf);
    uint32_t outputW = static_cast<uint32_t>(inputW * sf);

    uint32_t tileH = 16;
    uint32_t tileW = 16;

    uint32_t numTilesH = CeilDiv(outputH, tileH);
    uint32_t numTilesW = CeilDiv(outputW, tileW);
    uint32_t totalTiles = N * C * numTilesH * numTilesW;

    uint32_t blockDim = (totalTiles < MAX_BLOCK_DIM) ? totalTiles : MAX_BLOCK_DIM;

    tiling.set_C(C);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_totalTiles(totalTiles);
    tiling.set_scaleFactor(sf);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

kernel_src='''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_H = 16;
constexpr int32_t TILE_W = 16;

class KernelCa6k00346Custom {
public:
    __aicore__ inline KernelCa6k00346Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t outputH, uint32_t outputW,
                                uint32_t tileH, uint32_t tileW,
                                uint32_t totalTiles,
                                float scaleFactor) {
        xGm_ = (__gm__ float*)x;
        yGm_ = (__gm__ float*)y;
        N_ = N;
        C_ = C;
        inputH_ = inputH;
        inputW_ = inputW;
        outputH_ = outputH;
        outputW_ = outputW;
        tileH_ = tileH;
        tileW_ = tileW;
        totalTiles_ = totalTiles;
        scaleFactor_ = scaleFactor;
        invScale_ = 1.0f / scaleFactor;
        numTilesH_ = (outputH_ + tileH_ - 1) / tileH_;
        numTilesW_ = (outputW_ + tileW_ - 1) / tileW_;
        tilesPerChannel_ = numTilesH_ * numTilesW_;

        // allocate reusable buffers
        // out tile buffer
        pipe_.InitBuffer(outTileBuf_, 1, tileH_ * tileW_ * sizeof(float));
        // two row buffers for sliding window (max cols needed per tile)
        maxCols_ = (tileW_ + 2) > inputW_ ? inputW_ : (tileW_ + 2);
        pipe_.InitBuffer(rowBuf0_, 1, maxCols_ * sizeof(float));
        pipe_.InitBuffer(rowBuf1_, 1, maxCols_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        // grid-stride: each block processes a range of tile indices
        uint32_t tilesPerBlock = (totalTiles_ + blockNum - 1) / blockNum;
        uint32_t tileStart = blockIdx * tilesPerBlock;
        uint32_t tileEnd = tileStart + tilesPerBlock;
        if (tileEnd > totalTiles_) tileEnd = totalTiles_;

        for (uint32_t linearTileIdx = tileStart; linearTileIdx < tileEnd; ++linearTileIdx) {
            ProcessOneTile(linearTileIdx);
        }
    }

private:
    __aicore__ inline void ProcessOneTile(uint32_t linearTileIdx) {
        uint32_t n = linearTileIdx / (C_ * tilesPerChannel_);
        uint32_t rem = linearTileIdx % (C_ * tilesPerChannel_);
        uint32_t c = rem / tilesPerChannel_;
        uint32_t tileIdxInChannel = rem % tilesPerChannel_;
        uint32_t tileRow = tileIdxInChannel / numTilesW_;
        uint32_t tileCol = tileIdxInChannel % numTilesW_;

        uint32_t oyStart = tileRow * tileH_;
        uint32_t oxStart = tileCol * tileW_;
        uint32_t oyEnd = oyStart + tileH_;
        if (oyEnd > outputH_) oyEnd = outputH_;
        uint32_t oxEnd = oxStart + tileW_;
        if (oxEnd > outputW_) oxEnd = outputW_;
        uint32_t actualTileH = oyEnd - oyStart;
        uint32_t actualTileW = oxEnd - oxStart;

        float inv = invScale_;
        uint32_t iyStart = (uint32_t)(oyStart * inv);
        uint32_t iyEnd = (uint32_t)((oyEnd - 1) * inv) + 2;
        if (iyEnd > inputH_) iyEnd = inputH_;
        uint32_t ixStart = (uint32_t)(oxStart * inv);
        uint32_t ixEnd = (uint32_t)((oxEnd - 1) * inv) + 2;
        if (ixEnd > inputW_) ixEnd = inputW_;
        uint32_t inputRows = iyEnd - iyStart;
        uint32_t inputCols = ixEnd - ixStart;

        // load first two input rows
        uint32_t currentIy = iyStart;
        LoadRow(rowBuf0_, n, c, currentIy, ixStart, inputCols);
        if (currentIy + 1 < inputH_) {
            LoadRow(rowBuf1_, n, c, currentIy + 1, ixStart, inputCols);
        } else {
            LoadRow(rowBuf1_, n, c, inputH_ - 1, ixStart, inputCols);
        }

        AscendC::LocalTensor<float> outTile = outTileBuf_.Get<float>();
        uint32_t outRowIdx = 0;
        for (uint32_t oy = oyStart; oy < oyEnd; ++oy) {
            float inY = oy * inv;
            uint32_t iy = (uint32_t)inY;
            float dy = inY - iy;

            if (iy > currentIy) {
                AscendC::LocalTensor<float> tmp = rowBuf0_;
                rowBuf0_ = rowBuf1_;
                rowBuf1_ = tmp;
                uint32_t nextIy = iy + 1;
                if (nextIy >= inputH_) nextIy = inputH_ - 1;
                LoadRow(rowBuf1_, n, c, nextIy, ixStart, inputCols);
                currentIy = iy;
            }

            AscendC::LocalTensor<float> row0 = rowBuf0_.Get<float>();
            AscendC::LocalTensor<float> row1 = rowBuf1_.Get<float>();

            for (uint32_t ox = oxStart; ox < oxEnd; ++ox) {
                float inX = ox * inv;
                uint32_t ix = (uint32_t)inX;
                float dx = inX - ix;
                uint32_t localIx0 = ix - ixStart;
                uint32_t localIx1 = localIx0 + 1;
                if (localIx1 >= inputCols) localIx1 = inputCols - 1;

                float v00 = row0(localIx0);
                float v01 = row0(localIx1);
                float v10 = row1(localIx0);
                float v11 = row1(localIx1);

                float top = v00 + dx * (v01 - v00);
                float bottom = v10 + dx * (v11 - v10);
                float val = top + dy * (bottom - top);

                uint32_t localOffset = outRowIdx * actualTileW + (ox - oxStart);
                outTile(localOffset) = val;
            }
            outRowIdx++;
        }

        // write output tile to global memory
        uint32_t outBase = n * C_ * outputH_ * outputW_ + c * outputH_ * outputW_ + oyStart * outputW_ + oxStart;
        for (uint32_t i = 0; i < actualTileH; ++i) {
            AscendC::DataCopy(yGm_[outBase + i * outputW_], outTile[i * actualTileW], actualTileW);
        }
    }

    __aicore__ inline void LoadRow(AscendC::LocalTensor<float> &localBuf,
                                   uint32_t n, uint32_t c,
                                   uint32_t row, uint32_t colStart, uint32_t len) {
        uint32_t inBase = n * C_ * inputH_ * inputW_ + c * inputH_ * inputW_ + row * inputW_ + colStart;
        AscendC::DataCopy(localBuf, xGm_[inBase], len);
    }

private:
    __gm__ float *xGm_;
    __gm__ float *yGm_;

    uint32_t N_, C_, inputH_, inputW_, outputH_, outputW_;
    uint32_t tileH_, tileW_, numTilesH_, numTilesW_, tilesPerChannel_, totalTiles_;
    uint32_t maxCols_;
    float scaleFactor_, invScale_;

    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> outTileBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> rowBuf0_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> rowBuf1_;
};

extern "C" __global__ __aicore__ void ca6k00346_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00346Custom op;
    op.Init(x, y,
            tiling_data.N,
            tiling_data.C,
            tiling_data.inputH,
            tiling_data.inputW,
            tiling_data.outputH,
            tiling_data.outputW,
            tiling_data.tileH,
            tiling_data.tileW,
            tiling_data.totalTiles,
            tiling_data.scaleFactor);
    op.Process();
}
'''

class ModelNew(torch.nn.Module):

        super(ModelNew, self).__init__()

        else:
            self.scale_factor = 2.0
