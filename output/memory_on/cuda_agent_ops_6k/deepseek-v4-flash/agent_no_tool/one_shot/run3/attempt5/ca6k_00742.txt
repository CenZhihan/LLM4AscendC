    return (x / a) * a;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;

kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00742Custom {
public:
    __aicore__ inline KernelCa6k00742Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t channels,
                                uint32_t spatialSize, uint32_t tileLength) {
        batchSize_ = batchSize;
        channels_ = channels;
        spatialSize_ = spatialSize;
        tileLen_ = tileLength;
        outputSize_ = batchSize * channels;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * channels_ * spatialSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (outputSize_ == 0 || spatialSize_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        for (uint32_t idx = blockIdx; idx < outputSize_; idx += blockNum) {
            uint32_t batchIdx = idx / channels_;
            uint32_t channelIdx = idx % channels_;
            uint32_t inputOffset = (batchIdx * channels_ + channelIdx) * spatialSize_;

            float sum = 0.0f;
            uint32_t remaining = spatialSize_;
            uint32_t curOffset = inputOffset;

            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[curOffset], curLen);
                inQueueX_.EnQue(xLocal);
                xLocal = inQueueX_.DeQue<float>();

                for (uint32_t i = 0; i < curLen; i++) {
                    sum += xLocal(i);
                }

                inQueueX_.FreeTensor(xLocal);

                curOffset += curLen;
                remaining -= curLen;
            }

            float avg = sum / static_cast<float>(spatialSize_);
            yGm_.SetValue(idx, avg);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ = 0;
    uint32_t channels_ = 0;
    uint32_t spatialSize_ = 0;
    uint32_t tileLen_ = 0;
    uint32_t outputSize_ = 0;
};

extern "C" __global__ __aicore__ void ca6k_00742_custom(GM_ADDR x, GM_ADDR y,
                                                          GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00742Custom op;
    op.Init(x, y,
            tiling_data.batchSize, tiling_data.channels,
            tiling_data.spatialSize, tiling_data.tileLength);
    op.Process();
}
