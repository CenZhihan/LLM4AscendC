class KernelCa6k00269Custom {
public:

                                uint32_t totalLength, uint32_t tileLength,
                                int32_t N, int32_t C, int32_t H, int32_t W,
                                int32_t outH, int32_t outW,
                                int32_t kernelSizeH, int32_t kernelSizeW,
                                int32_t strideH, int32_t strideW,
                                int32_t p)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N; C_ = C; H_ = H; W_ = W;
        outH_ = outH; outW_ = outW;
        kernelSizeH_ = kernelSizeH; kernelSizeW_ = kernelSizeW;
        strideH_ = strideH; strideW_ = strideW;
        p_ = p;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Allocate local tensor for output tile
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // Process each output element in this tile
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t flatIdx = base + i;
                // Decode flatIdx to (n, c, oh, ow)
                int32_t ow = flatIdx % outW_;
                int32_t oh = (flatIdx / outW_) % outH_;
                int32_t c = (flatIdx / (outW_ * outH_)) % C_;
                int32_t n = flatIdx / (outW_ * outH_ * C_);

                int32_t startH = oh * strideH_;
                int32_t startW = ow * strideW_;

                float sum = 0.0f;
                for (int32_t kh = 0; kh < kernelSizeH_; kh++) {
                    for (int32_t kw = 0; kw < kernelSizeW_; kw++) {
                        int32_t h = startH + kh;
                        int32_t w = startW + kw;
                        uint32_t inIdx = ((n * C_ + c) * H_ + h) * W_ + w;
                        float val = xGm_.GetValue(inIdx);
                        sum += val * val;
                    }
                }
                float outVal = sqrtf(sum);
                yLocal.SetValue(i, outVal);
            }

            outQueueY_.EnQue(yLocal);
            AscendC::LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yDeque, curLen);
            outQueueY_.FreeTensor(yDeque);

            base += strideElems;
        }
