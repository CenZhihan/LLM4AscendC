host_operator_src = """
#include "ca6k02887_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    auto *inputShape = context->GetInputShape(0);
    auto *weightShape = context->GetInputShape(1);

    uint32_t totalM = static_cast<uint32_t>(inputShape->GetOriginShape().GetShapeSize() /
                                             inputShape->GetOriginShape().GetDim(2));
    uint32_t totalN = static_cast<uint32_t>(weightShape->GetOriginShape().GetDim(0));
    uint32_t totalK = static_cast<uint32_t>(weightShape->GetOriginShape().GetDim(1));

    uint32_t tileK = 64;
    if (tileK > totalK) tileK = totalK;
    tileK = AlignDown(tileK, 16);
    if (tileK == 0) tileK = totalK;

    uint32_t tileM = 1;
    while (true) {
        uint32_t needed = tileM * tileK + totalN * tileK + tileM * totalN;
        if (needed * sizeof(float) <= UB_BUDGET_BYTES) {
            tileM *= 2;
            if (tileM > totalM) break;
        } else {
            tileM /= 2;
            if (tileM == 0) tileM = 1;
            break;
        }
    }
    if (tileM > totalM) tileM = totalM;
    tileM = AlignDown(tileM, 16);
    if (tileM == 0) tileM = 1;

    uint32_t blockDim = CeilDiv(totalM, tileM);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 64) blockDim = 64;
    context->SetBlockDim(blockDim);

    Ca6k02887CustomTilingData tiling;
    tiling.set_totalM(totalM);
    tiling.set_totalN(totalN);
    tiling.set_totalK(totalK);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    int64_t batch = x_shape->GetDim(0);
    int64_t seq = x_shape->GetDim(1);
    int64_t out_features = w_shape->GetDim(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, batch);
    y_shape->SetDim(1, seq);
    y_shape->SetDim(2, out_features);
    return GRAPH_SUCCESS;
}

{
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = """
#include "kernel_operator.h"
using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k02887Custom {
public:
    __aicore__ inline KernelCa6k02887Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias1, GM_ADDR bias2, GM_ADDR y,
                                uint32_t totalM, uint32_t totalN, uint32_t totalK,
                                uint32_t tileM, uint32_t tileK, uint32_t blockDim)
    {
        totalM_ = totalM;
        totalN_ = totalN;
        totalK_ = totalK;
        tileM_ = tileM;
        tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalM_ * totalK_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, totalN_ * totalK_);
        b1Gm_.SetGlobalBuffer((__gm__ float*)bias1, totalN_);
        b2Gm_.SetGlobalBuffer((__gm__ float*)bias2, totalN_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalM_ * totalN_);
    }

    __aicore__ inline void Process()
    {
        if (totalM_ == 0 || totalN_ == 0 || totalK_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        for (uint32_t rowStart = blockIdx * tileM_; rowStart < totalM_; rowStart += blockNum * tileM_) {
            uint32_t curRows = tileM_;
            if (rowStart + curRows > totalM_) curRows = totalM_ - rowStart;
            if (curRows == 0) break;

            TPipe pipe;
            TQue<TPosition::VECIN, 1> queXRow;
            TQue<TPosition::VECIN, 1> queWRow;
            TQue<TPosition::VECOUT, 1> queYRow;
            TQue<TPosition::VECCALC, 1> queTmp;
            uint32_t rowSize = totalK_ * sizeof(float);
            pipe.InitBuffer(queXRow, 1, rowSize);
            pipe.InitBuffer(queWRow, 1, rowSize);
            pipe.InitBuffer(queYRow, 1, totalN_ * sizeof(float));
            pipe.InitBuffer(queTmp, 1, rowSize);

            for (uint32_t i = 0; i < curRows; i++) {
                uint32_t globalRow = rowStart + i;

                LocalTensor<float> xRow = queXRow.AllocTensor<float>();
                DataCopy(xRow, xGm_[globalRow * totalK_], totalK_);
                queXRow.EnQue(xRow);
                xRow = queXRow.DeQue<float>();

                LocalTensor<float> yRow = queYRow.AllocTensor<float>();
                for (uint32_t j = 0; j < totalN_; j++) {
                    LocalTensor<float> wRow = queWRow.AllocTensor<float>();
                    DataCopy(wRow, wGm_[j * totalK_], totalK_);
                    queWRow.EnQue(wRow);
                    wRow = queWRow.DeQue<float>();

                    LocalTensor<float> tmp = queTmp.AllocTensor<float>();
                    for (uint32_t k = 0; k < totalK_; k++) {
                        tmp.SetValue(k, xRow.GetValue(k) * wRow.GetValue(k));
                    }
                    queTmp.EnQue(tmp);
                    tmp = queTmp.DeQue<float>();

                    float sum = 0.0f;
                    for (uint32_t k = 0; k < totalK_; k++) {
                        sum += tmp.GetValue(k);
                    }
                    queTmp.FreeTensor(tmp);

                    sum += b1Gm_.GetValue(j) + b2Gm_.GetValue(j);
                    yRow.SetValue(j, sum);

                    queWRow.FreeTensor(wRow);
                }
                queYRow.EnQue(yRow);
                yRow = queYRow.DeQue<float>();
                DataCopy(yGm_[globalRow * totalN_], yRow, totalN_);
                queYRow.FreeTensor(yRow);
                queXRow.FreeTensor(xRow);
            }
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> b1Gm_;
    GlobalTensor<float> b2Gm_;
    GlobalTensor<float> yGm_;

    uint32_t totalM_ = 0;
    uint32_t totalN_ = 0;
    uint32_t totalK_ = 0;
    uint32_t tileM_ = 0;
    uint32_t tileK_ = 0;
};

extern "C" __global__ __aicore__ void ca6k_02887_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias1, GM_ADDR bias2, GM_ADDR y,
                                                         GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02887Custom op;
    op.Init(x, weight, bias1, bias2, y,
            tiling_data.totalM_, tiling_data.totalN_, tiling_data.totalK_,
            tiling_data.tileM_, tiling_data.tileK_, tiling_data.blockDim_);
    op.Process();
}
"""
