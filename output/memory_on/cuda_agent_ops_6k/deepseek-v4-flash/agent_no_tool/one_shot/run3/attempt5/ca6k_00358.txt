kernel_src="""

#include "kernel_operator.h"
#include <cmath>
#include <float.h>

using namespace AscendC;

class KernelCa6k00358Custom {
public:
    __aicore__ inline KernelCa6k00358Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElems,
                                uint32_t inN,
                                uint32_t inC,
                                uint32_t inH,
                                uint32_t inW,
                                uint32_t outH,
                                uint32_t outW,
                                uint32_t blockDim)
    {
        xGm_ = (__gm__ float*)x;
        yGm_ = (__gm__ float*)y;
        totalOutElems_ = totalOutputElems;
        inN_ = inN;
        inC_ = inC;
        inH_ = inH;
        inW_ = inW;
        outH_ = outH;
        outW_ = outW;
        blockDim_ = blockDim;
    }

    __aicore__ inline void Process()
    {
        if (totalOutElems_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();

        for (uint32_t idx = blockIdx; idx < totalOutElems_; idx += blockDim_) {
            // Decode output index to (n, c, oh, ow)
            uint32_t temp = idx;
            uint32_t ow = temp % outW_;
            temp /= outW_;
            uint32_t oh = temp % outH_;
            temp /= outH_;
            uint32_t c = temp % inC_;
            temp /= inC_;
            uint32_t n = temp;

            // Compute adaptive pooling region
            float h_start_f = static_cast<float>(oh) * static_cast<float>(inH_) / static_cast<float>(outH_);
            float h_end_f = static_cast<float>(oh + 1) * static_cast<float>(inH_) / static_cast<float>(outH_);
            int32_t h_start = static_cast<int32_t>(floorf(h_start_f));
            int32_t h_end = static_cast<int32_t>(ceilf(h_end_f));
            if (h_start < 0) h_start = 0;
            if (h_end > static_cast<int32_t>(inH_)) h_end = static_cast<int32_t>(inH_);

            float w_start_f = static_cast<float>(ow) * static_cast<float>(inW_) / static_cast<float>(outW_);
            float w_end_f = static_cast<float>(ow + 1) * static_cast<float>(inW_) / static_cast<float>(outW_);
            int32_t w_start = static_cast<int32_t>(floorf(w_start_f));
            int32_t w_end = static_cast<int32_t>(ceilf(w_end_f));
            if (w_start < 0) w_start = 0;
            if (w_end > static_cast<int32_t>(inW_)) w_end = static_cast<int32_t>(inW_);

            // Compute max over region
            float max_val = -FLT_MAX;
            uint32_t base_NC = (n * inC_ + c);
            uint32_t base_row_stride = inW_;
            uint32_t base_channel_stride = inH_ * inW_;

            for (int32_t ih = h_start; ih < h_end; ++ih) {
                uint32_t row_offset = base_NC * base_channel_stride + ih * base_row_stride;
                for (int32_t iw = w_start; iw < w_end; ++iw) {
                    float val = xGm_[row_offset + iw];
                    if (val > max_val) max_val = val;
                }
            }

            // Write output
            uint32_t out_offset = ((n * inC_ + c) * outH_ + oh) * outW_ + ow;
            yGm_[out_offset] = max_val;
        }
    }

private:
    __gm__ float* xGm_;
    __gm__ float* yGm_;
    uint32_t totalOutElems_ {0};
    uint32_t inN_ {0};
    uint32_t inC_ {0};
    uint32_t inH_ {0};
    uint32_t inW_ {0};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
    uint32_t blockDim_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00358_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling, Ca6k00358CustomTilingData);
    KernelCa6k00358Custom op;
    op.Init(x, y,
            td.totalOutputElems,
            td.inN,
            td.inC,
            td.inH,
            td.inW,
            td.outH,
            td.outW,
            td.blockDim);
    op.Process();
}
