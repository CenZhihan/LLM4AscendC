#include "ca6k_00333_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00333CustomTilingData tiling;
    const uint32_t totalElements =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *sizePtr = attrs->GetAttrPointer<int32_t>(0);
    const float *alphaPtr = attrs->GetAttrPointer<float>(1);
    const float *betaPtr = attrs->GetAttrPointer<float>(2);
    const float *kPtr = attrs->GetAttrPointer<float>(3);

    const gert::Shape *shape = context->GetInputShape(0);
    uint32_t C = static_cast<uint32_t>(shape->GetDim(1));
    uint32_t totalPixels = totalElements / C;

    uint32_t maxTilePixels = UB_BUDGET_BYTES / ( (2*C + 1) * sizeof(float) );
    if (maxTilePixels < 1) maxTilePixels = 1;
    uint32_t tilePixels = maxTilePixels;
    uint32_t tileElements = AlignDown(tilePixels * C, ALIGN_ELEMS);
    if (tileElements < C) tileElements = C;
    tilePixels = tileElements / C;

    if (totalPixels < tilePixels) {
        tilePixels = totalPixels;
        tileElements = totalPixels * C;
    }

    uint32_t numTiles = CeilDiv(totalPixels, tilePixels);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalElements(totalElements);
    tiling.set_tileElements(tileElements);
    tiling.set_blockDim(blockDim);
    tiling.set_C(C);
    tiling.set_size(*sizePtr);
    tiling.set_alpha(*alphaPtr);
    tiling.set_beta(*betaPtr);
    tiling.set_k(*kPtr);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00333Custom {
public:
    __aicore__ inline KernelCa6k00333Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalElements, uint32_t tileElements,
                                uint32_t blockDim, uint32_t C,
                                int32_t size, float alpha, float beta, float k) {
        totalElements_ = totalElements;
        tileElements_ = tileElements;
        C_ = C;
        size_ = size;
        alpha_ = alpha;
        beta_ = beta;
        k_ = k;
        halfWindow_ = size_ / 2;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElements_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElements_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileElements_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileElements_ * sizeof(float));
        // tmpQueue: one buffer for per-pixel squares (C floats) and one for scratch (1 float)
        pipe_.InitBuffer(tmpQueue_, 1, (C_ + 1) * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalElements_ == 0 || tileElements_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t tilePixels = tileElements_ / C_;
        uint32_t stridePixels = blockNum * tilePixels;
        uint32_t basePixel = blockIdx * tilePixels;
        if (basePixel * C_ >= totalElements_) return;

        uint32_t curPixels = 0;
        uint32_t curBase = basePixel * C_;
        if (curBase < totalElements_) {
            curPixels = (totalElements_ - curBase) / C_;
            if (curPixels > tilePixels) curPixels = tilePixels;
            CopyIn(curBase, curPixels * C_);
        } else {
            return;
        }

        while (true) {
            uint32_t nextBase = (basePixel + stridePixels) * C_;
            uint32_t nextPixels = 0;
            if (nextBase < totalElements_) {
                nextPixels = (totalElements_ - nextBase) / C_;
                if (nextPixels > tilePixels) nextPixels = tilePixels;
                CopyIn(nextBase, nextPixels * C_);
            }

            Compute(curPixels);
            CopyOut(curBase, curPixels * C_);

            if (nextPixels == 0) break;
            curPixels = nextPixels;
            curBase = nextBase;
            basePixel += stridePixels;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t len) {
        AscendC::LocalTensor<float> local = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(local, xGm_[offset], len);
        inQueueX_.EnQue(local);
    }

    __aicore__ inline void Compute(uint32_t numPixels) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> sqBuffer = tmpQueue_.AllocTensor<float>();
        // use last element of sqBuffer as temp scalar
        AscendC::LocalTensor<float> tempScalar = sqBuffer[C_];

        for (uint32_t p = 0; p < numPixels; p++) {
            uint32_t offset = p * C_;

            // compute squares into sqBuffer[0:C_]
            for (uint32_t c = 0; c < C_; c++) {
                float val = xLocal(offset + c);
                sqBuffer(c) = val * val;
            }

            // compute LRN for each channel
            for (uint32_t c = 0; c < C_; c++) {
                float sumSq = 0.0f;
                int32_t start = static_cast<int32_t>(c) - halfWindow_;
                if (start < 0) start = 0;
                int32_t end = static_cast<int32_t>(c) + halfWindow_;
                if (end >= static_cast<int32_t>(C_)) end = static_cast<int32_t>(C_) - 1;
                for (int32_t j = start; j <= end; j++) {
                    sumSq += sqBuffer(static_cast<uint32_t>(j));
                }
                float denom = k_ + alpha_ * sumSq;
                // compute denom^(-beta_) = exp(-beta_ * log(denom))
                tempScalar(0) = denom;
                AscendC::Ln(tempScalar, tempScalar, 1);
                tempScalar(0) = tempScalar(0) * (-beta_);
                AscendC::Exp(tempScalar, tempScalar, 1);
                float scale = tempScalar(0);
                yLocal(offset + c) = xLocal(offset + c) * scale;
            }
        }

        tmpQueue_.FreeTensor(sqBuffer);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t offset, uint32_t len) {
        AscendC::LocalTensor<float> local = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], local, len);
        outQueueY_.FreeTensor(local);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalElements_ {0};
    uint32_t tileElements_ {0};
    uint32_t C_ {0};
    int32_t size_ {5};
    int32_t halfWindow_ {2};
    float alpha_ {1e-4f};
    float beta_ {0.75f};
    float k_ {1.0f};
};

extern "C" __global__ __aicore__ void ca6k_00333_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00333Custom op;
    op.Init(x, y,
            tiling_data.totalElements, tiling_data.tileElements,
            tiling_data.blockDim, tiling_data.C,
            tiling_data.size, tiling_data.alpha, tiling_data.beta, tiling_data.k);
    op.Process();
}
'''
