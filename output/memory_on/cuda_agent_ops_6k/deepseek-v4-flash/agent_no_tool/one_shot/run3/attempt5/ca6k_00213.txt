host_operator_src="""

#include "ca6k00213_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per core
static constexpr uint32_t ALIGN_ELEMS = 2048;            // alignment
static constexpr uint32_t HARD_CAP_ELEMS = 16384;        // limit tile size

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00213CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *threshold = attrs->GetAttrPointer<float>(0);
    const float *value = attrs->GetAttrPointer<float>(1);

    // UB usage: inQueue (x) + outQueue (y) + tmpQueue (mask) = 3 * tileLen * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }

    const uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_threshold(*threshold);
    tiling.set_val(*value);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;

class KernelCa6k00213Custom {
public:
    __aicore__ inline KernelCa6k00213Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                float threshold, float val)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        threshold_ = threshold;
        val_ = val;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockDim = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t stride = blockDim * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            CopyIn(base, curLen);
            Compute(curLen);
            CopyOut(base, curLen);

            base += stride;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        LocalTensor<float> mask = tmpQueue_.AllocTensor<float>();
        LocalTensor<float> diff = tmpQueue_.AllocTensor<float>(); // reuse same buffer but careful: need separate
        // Actually tmpQueue has only 1 buffer, so we need two temporaries. Let's allocate from tmpQueue twice.
        // Better: use two separate buffers from tmpQueue by allocating two tensors? But tmpQueue size is 1 * tileLen.
        // We can reuse the same buffer for diff after mask is used? Compute: mask = Vcmp; then diff = Sub(x, val_); then prod = Mul(diff, mask); then y = Add(prod, val_). So we need mask and diff simultaneously. But we can compute mask first, then use it to compute y directly without extra diff by doing: y = val_ + (x - val_) * mask.
        // That uses xLocal, mask, and we can compute in-place: Sub(xLocal, val_, len) -> diff in xLocal? No, we need x for later if we want to reuse. So we need two temporaries. Since tmpQueue has 1 buffer, we can allocate one from tmpQueue and one from inQueueX after we deque? But we have only one input buffer.
        // Let's use a different approach: use a single temporary for mask and compute y as:
        // y = val_ + (x - val_) * mask
        // mask computed first (1 buffer)
        // (x - val_) can be computed into yLocal directly, then multiply by mask, then add val_. That uses yLocal as temporary.
        // Steps:
        // 1. VCmp(mask, xLocal, threshold_, CMP_GT);   // mask = 1.0 if x>threshold else 0.0
        // 2. Sub(yLocal, xLocal, val_, len);             // diff = x - val_
        // 3. Mul(yLocal, yLocal, mask, len);             // diff * mask
        // 4. Add(yLocal, yLocal, val_, len);             // val_ + (x-val_)*mask
        // This uses only one temporary (mask) and yLocal. Perfect.

        VCmp(mask, xLocal, threshold_, CMP_GT);
        Sub(yLocal, xLocal, val_, len);
        Mul(yLocal, yLocal, mask, len);
        Add(yLocal, yLocal, val_, len);

        // tmpQueue only used for mask
        tmpQueue_.FreeTensor(mask);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    float threshold_ {0.5f};
    float val_ {0.0f};
};

extern "C" __global__ __aicore__ void ca6k_00213_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00213Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.threshold, tiling_data.val);
    op.Process();
}
