kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k02874Custom {
public:
    __aicore__ inline KernelCa6k02874Custom() {}

    __aicore__ inline void Init(GM_ADDR bias, GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalRows, uint32_t totalCols)
    {
        totalRows_ = totalRows;
        totalCols_ = totalCols;

        biasGm_.SetGlobalBuffer((__gm__ float*)bias, totalCols_);
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * totalCols_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, totalCols_ * totalCols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * totalCols_);
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || totalCols_ == 0) return;

        // Use cube Matmul unit for addmm: y = bias + x * weight
        // x: (totalRows_, totalCols_), weight: (totalCols_, totalCols_), bias: (1, totalCols_)
        AscendC::Matmul<AscendC::MatmulType<AscendC::TPosition::CUBE_MATMUL_LEFT,
                                             AscendC::TPosition::CUBE_MATMUL_RIGHT>> mm;
        AscendC::MatmulTiling tiling;
        tiling.SetM(totalRows_);
        tiling.SetN(totalCols_);
        tiling.SetK(totalCols_);
        tiling.SetBatch(1);

        mm.SetTensorA(xGm_, false);   // transA = false
        mm.SetTensorB(weightGm_, false); // transB = false (weight not transposed)
        mm.SetBias(biasGm_, true);    // broadcast bias across rows
        mm.SetResult(yGm_);
        mm.SetTiling(tiling);
        mm.IterateAll();
    }

private:
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t totalCols_ {0};
};

extern "C" __global__ __aicore__ void ca6k_02874_custom(GM_ADDR bias, GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k02874Custom op;
    op.Init(bias, x, weight, y,
            tiling_data.totalRows, tiling_data.totalCols);
    op.Process();
}
