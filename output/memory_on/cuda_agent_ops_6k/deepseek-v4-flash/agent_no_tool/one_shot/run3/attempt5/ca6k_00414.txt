#include "ca6k_00414_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00414CustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    tiling.n = shape.GetDim(0);
    tiling.c = shape.GetDim(1);
    tiling.h = shape.GetDim(2);
    tiling.w = shape.GetDim(3);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float scale = *attrs->GetAttrPointer<float>(0);
    tiling.hOut = static_cast<uint32_t>(tiling.h * scale);
    tiling.wOut = static_cast<uint32_t>(tiling.w * scale);

    // Simple blockDim: use one block per image channel * output tile (4 tiles)
    uint32_t blockDim = tiling.n * tiling.c * 4;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(x_shape->GetDimNum());
    y_shape->SetDim(0, x_shape->GetDim(0));

kernel_src = """
#include "kernel_operator.h"
#include "ca6k_00414_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float BICUBIC_A = -0.75f;

class KernelCa6k00414Custom {
public:
    __aicore__ inline KernelCa6k00414Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t h, uint32_t w,
                                uint32_t hOut, uint32_t wOut) {
        n_ = n; c_ = c; h_ = h; w_ = w; hOut_ = hOut; wOut_ = wOut;
        xGm_.SetGlobalBuffer((__gm__ float*)x, n * c * h * w);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n * c * hOut * wOut);
        totalOut_ = n * c * hOut * wOut;
    }

    __aicore__ inline void Process() {
        uint32_t blkIdx = AscendC::GetBlockIdx();
        uint32_t blkNum = AscendC::GetBlockNum();
        for (uint32_t idx = blkIdx; idx < totalOut_; idx += blkNum) {
            uint32_t tmp = idx;
            uint32_t wIdx = tmp % wOut_;
            tmp /= wOut_;
            uint32_t hIdx = tmp % hOut_;
            tmp /= hOut_;
            uint32_t cIdx = tmp % c_;
            uint32_t bIdx = tmp / c_;

            float yF = hIdx / 2.0f;
            float xF = wIdx / 2.0f;
            int iy = (int)yF;
            int ix = (int)xF;
            float du = xF - ix;
            float dv = yF - iy;

            float wx[4], wy[4];
            for (int k = 0; k < 4; ++k) {
                float u = (float)(k - 1) - du;
                float v = (float)(k - 1) - dv;
                wx[k] = bicubicWeight(u, BICUBIC_A);
                wy[k] = bicubicWeight(v, BICUBIC_A);
            }

            float sum = 0.0f;
            for (int j = 0; j < 4; ++j) {
                int row = iy + j - 1;
                row = (row < 0) ? 0 : (row >= (int)h_ ? (int)h_ - 1 : row);
                float rowSum = 0.0f;
                for (int k = 0; k < 4; ++k) {
                    int col = ix + k - 1;
                    col = (col < 0) ? 0 : (col >= (int)w_ ? (int)w_ - 1 : col);
                    uint32_t gIdx = (bIdx * c_ + cIdx) * h_ * w_ + row * w_ + col;
                    rowSum += xGm_.GetValue(gIdx) * wx[k];
                }
                sum += rowSum * wy[j];
            }

            yGm_.SetValue(idx, sum);
        }
    }

private:
    __aicore__ inline float bicubicWeight(float x, float a) {
        float absX = x;
        if (absX < 0.0f) absX = -absX;
        if (absX <= 1.0f) {
            return (a + 2.0f) * absX * absX * absX - (a + 3.0f) * absX * absX + 1.0f;
        } else if (absX < 2.0f) {
            return a * absX * absX * absX - 5.0f * a * absX * absX + 8.0f * a * absX - 4.0f * a;
        } else {
            return 0.0f;
        }
    }

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t n_, c_, h_, w_, hOut_, wOut_;
    uint32_t totalOut_;
};
