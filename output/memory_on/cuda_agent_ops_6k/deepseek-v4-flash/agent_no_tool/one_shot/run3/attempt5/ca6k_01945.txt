#include "ca6k01945_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k01945CustomTilingData tiling;

    const gert::StorageShape *inShape = context->GetInputShape(0);
    const gert::StorageShape *wShape = context->GetInputShape(1);

    uint32_t N = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(0));
    uint32_t C_in = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(1));
    uint32_t H_in = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(2));
    uint32_t W_in = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(3));

    uint32_t C_out = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(0));
    uint32_t kH = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(2));
    uint32_t kW = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t strideVal = *attrs->GetAttrPointer<int64_t>(0);
    int64_t paddingVal = *attrs->GetAttrPointer<int64_t>(1);
    int64_t outputPaddingVal = *attrs->GetAttrPointer<int64_t>(2);

    uint32_t H_out = (H_in - 1) * strideVal - 2 * paddingVal + kH + outputPaddingVal;
    uint32_t W_out = (W_in - 1) * strideVal - 2 * paddingVal + kW + outputPaddingVal;

    uint32_t totalInputLen = N * C_in * H_in * W_in;
    uint32_t totalOutputLen = N * C_out * H_out * W_out;

    // Estimate UB usage: input tile (one batch) + weight + output tile
    uint32_t inputSize = C_in * H_in * W_in * sizeof(float);
    uint32_t weightSize = C_out * C_in * kH * kW * sizeof(float);
    uint32_t remaining = UB_BUDGET_BYTES - inputSize - weightSize;
    uint32_t tileOutC = 4;
    if (tileOutC > C_out) tileOutC = C_out;
    uint32_t tileOutH = 32;
    if (tileOutH > H_out) tileOutH = H_out;
    // Adjust tile sizes to fit UB
    while (tileOutC * tileOutH * W_out * sizeof(float) > remaining && tileOutC > 1) {
        tileOutC /= 2;
    }
    while (tileOutH > 1 && tileOutC * tileOutH * W_out * sizeof(float) > remaining) {
        tileOutH = AlignDown(tileOutH / 2, ALIGN_ELEMS);
        if (tileOutH < ALIGN_ELEMS) tileOutH = ALIGN_ELEMS;
    }
    if (tileOutC * tileOutH * W_out * sizeof(float) > remaining) {
        tileOutH = 1;
    }

    uint32_t numOutCBlocks = CeilDiv(C_out, tileOutC);
    uint32_t numOutHBlocks = CeilDiv(H_out, tileOutH);
    uint32_t totalTiles = N * numOutCBlocks * numOutHBlocks;
    uint32_t blockDim = totalTiles;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    // Use setter methods generated by the TILING_DATA_FIELD_DEF macro
    tiling.set_totalInputLen(totalInputLen);
    tiling.set_totalOutputLen(totalOutputLen);
    tiling.set_inputN(N);
    tiling.set_inputC(C_in);
    tiling.set_inputH(H_in);
    tiling.set_inputW(W_in);
    tiling.set_outputC(C_out);
    tiling.set_outputH(H_out);
    tiling.set_outputW(W_out);
    tiling.set_kernelH(kH);
    tiling.set_kernelW(kW);
    tiling.set_stride(static_cast<uint32_t>(strideVal));
    tiling.set_padding(static_cast<uint32_t>(paddingVal));
    tiling.set_outputPadding(static_cast<uint32_t>(outputPaddingVal));
    tiling.set_blockDim(blockDim);
    tiling.set_tileOutC(tileOutC);
    tiling.set_tileOutH(tileOutH);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k01945Custom {
public:
    __aicore__ inline KernelCa6k01945Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
        GET_TILING_DATA(td, tiling);
        totalInputLen_ = td.totalInputLen;
        totalOutputLen_ = td.totalOutputLen;
        N_ = td.inputN;
        C_in_ = td.inputC;
        H_in_ = td.inputH;
        W_in_ = td.inputW;
        C_out_ = td.outputC;
        H_out_ = td.outputH;
        W_out_ = td.outputW;
        kH_ = td.kernelH;
        kW_ = td.kernelW;
        stride_ = td.stride;
        padding_ = td.padding;
        outputPadding_ = td.outputPadding;
        tileOutC_ = td.tileOutC;
        tileOutH_ = td.tileOutH;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, C_out_ * C_in_ * kH_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, C_in_ * H_in_ * W_in_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, C_out_ * C_in_ * kH_ * kW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOutC_ * tileOutH_ * W_out_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileOutC_ * tileOutH_ * W_out_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t inputLen = C_in_ * H_in_ * W_in_;
        uint32_t weightLen = C_out_ * C_in_ * kH_ * kW_;
        uint32_t outputStride = C_out_ * H_out_ * W_out_;

        uint32_t numOutCBlocks = (C_out_ + tileOutC_ - 1) / tileOutC_;
        uint32_t numOutHBlocks = (H_out_ + tileOutH_ - 1) / tileOutH_;
        uint32_t totalTiles = N_ * numOutCBlocks * numOutHBlocks;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, weightLen);

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            uint32_t batch = tileId / (numOutCBlocks * numOutHBlocks);
            uint32_t rem = tileId % (numOutCBlocks * numOutHBlocks);
            uint32_t cGroup = rem / numOutHBlocks;
            uint32_t hTile = rem % numOutHBlocks;

            uint32_t cStart = cGroup * tileOutC_;
            uint32_t cEnd = cStart + tileOutC_;
            if (cEnd > C_out_) cEnd = C_out_;

            uint32_t hStart = hTile * tileOutH_;
            uint32_t hEnd = hStart + tileOutH_;
            if (hEnd > H_out_) hEnd = H_out_;

            uint32_t tileC = cEnd - cStart;
            uint32_t tileH = hEnd - hStart;
            uint32_t tileSize = tileC * tileH * W_out_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_ + batch * inputLen, inputLen);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Muls(yLocal, yLocal, 0.0f, tileSize);

            for (uint32_t cIn = 0; cIn < C_in_; cIn++) {
                for (uint32_t hIn = 0; hIn < H_in_; hIn++) {
                    for (uint32_t wIn = 0; wIn < W_in_; wIn++) {
                        float inputVal = xLocal(cIn * H_in_ * W_in_ + hIn * W_in_ + wIn);
                        if (inputVal == 0.0f) continue;
                        for (uint32_t cOut = cStart; cOut < cEnd; cOut++) {
                            for (uint32_t kh = 0; kh < kH_; kh++) {
                                int32_t hOut = static_cast<int32_t>(hIn * stride_ - padding_ + kh);
                                if (hOut < static_cast<int32_t>(hStart) || hOut >= static_cast<int32_t>(hEnd)) continue;
                                for (uint32_t kw = 0; kw < kW_; kw++) {
                                    int32_t wOut = static_cast<int32_t>(wIn * stride_ - padding_ + kw);
                                    if (wOut < 0 || wOut >= static_cast<int32_t>(W_out_)) continue;
                                    float w = wLocal(cOut * C_in_ * kH_ * kW_ + cIn * kH_ * kW_ + kh * kW_ + kw);
                                    uint32_t localC = cOut - cStart;
                                    uint32_t localH = hOut - hStart;
                                    uint32_t yIdx = localC * tileH * W_out_ + localH * W_out_ + wOut;
                                    yLocal(yIdx) += inputVal * w;
                                }
                            }
                        }
                    }
                }
            }

            for (uint32_t cOut = cStart; cOut < cEnd; cOut++) {
                for (uint32_t hOut = hStart; hOut < hEnd; hOut++) {
                    uint32_t localC = cOut - cStart;
                    uint32_t localH = hOut - hStart;
                    uint32_t globalOffset = batch * outputStride + cOut * H_out_ * W_out_ + hOut * W_out_;
                    uint32_t localOffset = localC * tileH * W_out_ + localH * W_out_;
                    AscendC::DataCopy(yGm_ + globalOffset, yLocal + localOffset, W_out_);
                }
            }

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            tmpQueue_.FreeTensor(tLocal);
        }
        wQueue_.FreeTensor(wLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputLen_;
    uint32_t totalOutputLen_;
    uint32_t N_, C_in_, H_in_, W_in_;
    uint32_t C_out_, H_out_, W_out_;
    uint32_t kH_, kW_;
    uint32_t stride_, padding_, outputPadding_;
    uint32_t tileOutC_, tileOutH_;
};

extern "C" __global__ __aicore__ void ca6k_01945_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    KernelCa6k01945Custom op;
    op.Init(x, weight, y, workspace, tiling);
    op.Process();
}
'''

model_src = '''
import torch
import torch.nn as nn
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding):
        super(ModelNew, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size if isinstance(kernel_size, tuple) else (kernel_size, kernel_size)
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        # weight is created as a parameter (no bias)
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, *self.kernel_size))

    def forward(self, x):
        return custom_ops_lib.ca6k_01945_custom(
            x, self.weight,
            self.stride[0] if isinstance(self.stride, tuple) else self.stride,
            self.padding[0] if isinstance(self.padding, tuple) else self.padding,
            self.output_padding[0] if isinstance(self.output_padding, tuple) else self.output_padding
        )
