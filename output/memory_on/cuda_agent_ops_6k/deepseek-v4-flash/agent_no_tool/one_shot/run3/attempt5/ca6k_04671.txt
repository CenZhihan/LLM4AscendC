static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k04671CustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const auto &originShape = xShape->GetOriginShape();
    const uint32_t dims[4] = {
        static_cast<uint32_t>(originShape.GetDim(0)),
        static_cast<uint32_t>(originShape.GetDim(1)),
        static_cast<uint32_t>(originShape.GetDim(2)),
        static_cast<uint32_t>(originShape.GetDim(3))
    };
    uint32_t totalLength = dims[0] * dims[1] * dims[2] * dims[3];
    uint32_t batchSize = dims[0];
    uint32_t channels = dims[1];
    uint32_t spatialSize = dims[2] * dims[3];

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = attrs->GetAttrPointer<float>(0) ? *attrs->GetAttrPointer<float>(0) : 1e-5f;

    bool training = attrs->GetAttrPointer<bool>(2) ? *attrs->GetAttrPointer<bool>(2) : false;

    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * 5);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;
    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_tileLength(tileLen);

    tiling.set_batchSize(batchSize);
    tiling.set_channels(channels);
    tiling.set_spatialSize(spatialSize);
    tiling.set_eps(eps);
    tiling.set_momentum(momentum);
    tiling.set_training(training);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        AscendC::DataCopy(varLocal_, varGm_, channels_);

        AscendC::DataCopy(biasLocal_, biasGm_, channels_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = (totalLen_ - base) > tileLen_ ? tileLen_ : (totalLen_ - base);
            CopyIn(base, curLen);
            Compute(base, curLen);
            CopyOut(base, curLen);
            base += strideElems;

        inQueueX_.EnQue(xLocal);
    }

    {
