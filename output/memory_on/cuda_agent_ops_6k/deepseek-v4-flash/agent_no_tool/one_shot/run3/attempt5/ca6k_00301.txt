#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00301Custom {
public:
    __aicore__ inline KernelCa6k00301Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t batch, uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        batch_ = batch;
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileK_ = tileK;
        tileN_ = tileN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * M * K);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * K * N);
        zGm_.SetGlobalBuffer((__gm__ float*)z, batch * M * N);
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockIdx = GetBlockIdx();
        if (blockIdx >= batch_) return;

        const uint32_t xBatchOffset = blockIdx * M_ * K_;
        const uint32_t yBatchOffset = blockIdx * K_ * N_;
        const uint32_t zBatchOffset = blockIdx * M_ * N_;

        for (uint32_t i = 0; i < M_; i += tileM_) {
            uint32_t curM = (i + tileM_ <= M_) ? tileM_ : M_ - i;
            for (uint32_t j = 0; j < N_; j += tileN_) {
                uint32_t curN = (j + tileN_ <= N_) ? tileN_ : N_ - j;

                // Local tensor for C tile
                LocalTensor<float> cLocal;
                pipe_.InitBuffer(tmpC_, 1, curM * curN * sizeof(float));
                cLocal = tmpC_.AllocTensor<float>();
                // Fill C tile with zero
                Duplicate(cLocal, 0.0f, curM * curN);
                // Alternatively: for (uint32_t idx = 0; idx < curM*curN; ++idx) cLocal.SetValue(idx, 0.0f);

                for (uint32_t k = 0; k < K_; k += tileK_) {
                    uint32_t curK = (k + tileK_ <= K_) ? tileK_ : K_ - k;

                    // Allocate local tensors for A and B tiles
                    LocalTensor<float> aLocal;
                    pipe_.InitBuffer(tmpA_, 1, curM * curK * sizeof(float));
                    aLocal = tmpA_.AllocTensor<float>();
                    LocalTensor<float> bLocal;
                    pipe_.InitBuffer(tmpB_, 1, curK * curN * sizeof(float));
                    bLocal = tmpB_.AllocTensor<float>();

                    // Load A tile (curM x curK) from global
                    for (uint32_t r = 0; r < curM; ++r) {
                        for (uint32_t c = 0; c < curK; ++c) {
                            float val = xGm_.GetValue(xBatchOffset + (i + r) * K_ + k + c);
                            aLocal.SetValue(r * curK + c, val);
                        }
                    }

                    // Load B tile (curK x curN) from global (B is stored row-major: K rows, N cols)
                    for (uint32_t r = 0; r < curK; ++r) {
                        for (uint32_t c = 0; c < curN; ++c) {
                            float val = yGm_.GetValue(yBatchOffset + (k + r) * N_ + j + c);
                            bLocal.SetValue(r * curN + c, val);
                        }
                    }

                    // Compute C += A * B using element-wise operations
                    // For each row of A and column of B
                    for (uint32_t rA = 0; rA < curM; ++rA) {
                        for (uint32_t cB = 0; cB < curN; ++cB) {
                            float sum = 0.0f;
                            for (uint32_t kIdx = 0; kIdx < curK; ++kIdx) {
                                float aVal = aLocal.GetValue(rA * curK + kIdx);
                                float bVal = bLocal.GetValue(kIdx * curN + cB);
                                sum += aVal * bVal;
                            }
                            // Accumulate into C tile
                            float old = cLocal.GetValue(rA * curN + cB);
                            cLocal.SetValue(rA * curN + cB, old + sum);
                        }
                    }

                    // Free tile local tensors
                    tmpA_.FreeTensor(aLocal);
                    tmpB_.FreeTensor(bLocal);
                }

                // Write C tile to global
                for (uint32_t r = 0; r < curM; ++r) {
                    for (uint32_t c = 0; c < curN; ++c) {
                        float val = cLocal.GetValue(r * curN + c);
                        zGm_.SetValue(zBatchOffset + (i + r) * N_ + j + c, val);
                    }
                }

                tmpC_.FreeTensor(cLocal);
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, 1> tmpA_;
    TQue<TPosition::VECCALC, 1> tmpB_;
    TQue<TPosition::VECCALC, 1> tmpC_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> zGm_;

    uint32_t batch_ {0};
    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {64};
    uint32_t tileK_ {128};
    uint32_t tileN_ {64};
};

extern "C" __global__ __aicore__ void ca6k_00301_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00301Custom op;
    op.Init(x, y, z,
            tiling_data.batchSize, tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
