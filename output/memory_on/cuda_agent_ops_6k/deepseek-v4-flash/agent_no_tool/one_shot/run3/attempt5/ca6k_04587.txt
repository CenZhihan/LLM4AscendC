kernel_src = '''
#include "kernel_operator.h"
#include "ca6k_04587_custom_tiling.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k04587Custom {
public:
    __aicore__ inline KernelCa6k04587Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        totalLen_ = td.totalLength;
        tileLen_ = td.tileLength;
        n_ = td.n;
        c_ = td.c;
        inD_ = td.inD;
        inH_ = td.inH;
        inW_ = td.inW;
        kd_ = td.kd;
        kh_ = td.kh;
        kw_ = td.kw;
        sd_ = td.sd;
        sh_ = td.sh;
        sw_ = td.sw;
        od_ = td.od;
        oh_ = td.oh;
        ow_ = td.ow;
        pad_d_ = td.pad_d;
        pad_h_ = td.pad_h;
        pad_w_ = td.pad_w;
        windowSize_ = kd_ * kh_ * kw_;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, windowSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        uint32_t stride = blockNum;
        uint32_t start = blockIdx;
        for (uint32_t idx = start; idx < totalLen_; idx += stride) {
            // Decompose idx into (n, c, od, oh, ow)
            uint32_t tmp = idx;
            uint32_t ow = tmp % ow_; tmp /= ow_;
            uint32_t oh = tmp % oh_; tmp /= oh_;
            uint32_t od = tmp % od_; tmp /= od_;
            uint32_t c = tmp % c_; tmp /= c_;
            uint32_t n = tmp;

            // Compute start input indices (with padding)
            int32_t id_start = static_cast<int32_t>(od) * static_cast<int32_t>(sd_) - static_cast<int32_t>(pad_d_);
            int32_t ih_start = static_cast<int32_t>(oh) * static_cast<int32_t>(sh_) - static_cast<int32_t>(pad_h_);
            int32_t iw_start = static_cast<int32_t>(ow) * static_cast<int32_t>(sw_) - static_cast<int32_t>(pad_w_);

            // Base offset for (n, c) in input
            uint32_t planeSize = inD_ * inH_ * inW_;
            uint32_t baseOffset = (n * c_ + c) * planeSize;

            float maxVal = -FLT_MAX;
            // Iterate over window
            for (int32_t kd = 0; kd < static_cast<int32_t>(kd_); ++kd) {
                int32_t id = id_start + kd;
                if (id < 0 || id >= static_cast<int32_t>(inD_)) continue;
                for (int32_t kh = 0; kh < static_cast<int32_t>(kh_); ++kh) {
                    int32_t ih = ih_start + kh;
                    if (ih < 0 || ih >= static_cast<int32_t>(inH_)) continue;
                    for (int32_t kw = 0; kw < static_cast<int32_t>(kw_); ++kw) {
                        int32_t iw = iw_start + kw;
                        if (iw < 0 || iw >= static_cast<int32_t>(inW_)) continue;

                        // Compute linear index
                        uint32_t linearIdx = baseOffset +
                                             static_cast<uint32_t>(id) * inH_ * inW_ +
                                             static_cast<uint32_t>(ih) * inW_ +
                                             static_cast<uint32_t>(iw);
                        float val = xGm_.GetValue(linearIdx);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }
            if (maxVal == -FLT_MAX) maxVal = 0.0f;

            yGm_.SetValue(idx, maxVal);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t n_, c_;
    uint32_t inD_, inH_, inW_;
    uint32_t kd_, kh_, kw_;
    uint32_t sd_, sh_, sw_;
    uint32_t od_, oh_, ow_;
    uint32_t pad_d_, pad_h_, pad_w_;
    uint32_t windowSize_;
};

extern "C" __global__ __aicore__ void ca6k_04587_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelCa6k04587Custom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}
'''
