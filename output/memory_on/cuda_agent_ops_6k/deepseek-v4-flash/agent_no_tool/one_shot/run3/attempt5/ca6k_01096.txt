static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::StorageShape *w_storage = context->GetInputShape(1);
    const gert::Shape &x_shape = x_storage->GetOriginShape();
    const gert::Shape &w_shape = w_storage->GetOriginShape();

    uint32_t N = x_shape.GetDim(0);
    uint32_t C_in = x_shape.GetDim(1);
    uint32_t H_in = x_shape.GetDim(2);
    uint32_t W_in = x_shape.GetDim(3);
    uint32_t C_out = w_shape.GetDim(0);
    uint32_t kH = w_shape.GetDim(2);
    uint32_t kW = w_shape.GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int strideH = attrs->GetAttrPointer<int>(0)[0];
    int strideW = attrs->GetAttrPointer<int>(1)[0];
    int padH = attrs->GetAttrPointer<int>(2)[0];
    int padW = attrs->GetAttrPointer<int>(3)[0];
    int outputPadH = attrs->GetAttrPointer<int>(4)[0];
    int outputPadW = attrs->GetAttrPointer<int>(5)[0];

    int H_out = (H_in - 1) * strideH - 2 * padH + kH + outputPadH;
    int W_out = (W_in - 1) * strideW - 2 * padW + kW + outputPadW;
    if (H_out <= 0 || W_out <= 0) {
        return ge::GRAPH_FAILED;
    }
    uint32_t totalElems = N * C_out * (uint32_t)H_out * (uint32_t)W_out;

    Ca6k01096CustomTilingData tiling;
    tiling.batch = N;
    tiling.inChannels = C_in;
    tiling.inHeight = H_in;
    tiling.inWidth = W_in;
    tiling.outChannels = C_out;
    tiling.outHeight = (uint32_t)H_out;
    tiling.outWidth = (uint32_t)W_out;
    tiling.kernelHeight = kH;
    tiling.kernelWidth = kW;
    tiling.strideH = strideH;
    tiling.strideW = strideW;
    tiling.padH = padH;
    tiling.padW = padW;
    tiling.outputPadH = outputPadH;
    tiling.outputPadW = outputPadW;
    tiling.totalOutputElems = totalElems;

    uint32_t blockDim = 24;
    context->SetBlockDim(blockDim);

    // Write raw tiling data to buffer
    void *buffer = context->GetRawTilingData()->GetData();
    size_t capacity = context->GetRawTilingData()->GetCapacity();
    if (sizeof(tiling) > capacity) {
        return ge::GRAPH_FAILED;
    }
    memcpy(buffer, &tiling, sizeof(tiling));
    context->GetRawTilingData()->SetDataSize(sizeof(tiling));

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

#include "kernel_operator.h"

struct Ca6k01096CustomTilingData {
    uint32_t batch;
    uint32_t inChannels;
    uint32_t inHeight;
    uint32_t inWidth;
    uint32_t outChannels;
    uint32_t outHeight;
    uint32_t outWidth;
    uint32_t kernelHeight;
    uint32_t kernelWidth;
    int32_t strideH;
    int32_t strideW;
    int32_t padH;
    int32_t padW;
    int32_t outputPadH;
    int32_t outputPadW;
    uint32_t totalOutputElems;
};

class KernelCa6k01096Custom {
public:
    __aicore__ inline KernelCa6k01096Custom() {}

    __aicore__ inline void Init(
        uint32_t batch, uint32_t inChannels, uint32_t inHeight, uint32_t inWidth,
        uint32_t outChannels, uint32_t outHeight, uint32_t outWidth,
        uint32_t kH, uint32_t kW,
        int32_t strideH, int32_t strideW, int32_t padH, int32_t padW,
        uint32_t totalOutputElems,
        __gm__ float *x, __gm__ float *w, __gm__ float *y
    ) {
        batch_ = batch;
        inChannels_ = inChannels;
        inHeight_ = inHeight;
        inWidth_ = inWidth;
        outChannels_ = outChannels;
        outHeight_ = outHeight;
        outWidth_ = outWidth;
        kH_ = kH;
        kW_ = kW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        totalOutputElems_ = totalOutputElems;

        xGm_.SetGlobalBuffer(x, batch_ * inChannels_ * inHeight_ * inWidth_);
        wGm_.SetGlobalBuffer(w, outChannels_ * inChannels_ * kH_ * kW_);
        yGm_.SetGlobalBuffer(y, totalOutputElems_);
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        for (uint32_t idx = blockIdx; idx < totalOutputElems_; idx += blockNum) {
            uint32_t n = idx / (outChannels_ * outHeight_ * outWidth_);
            uint32_t rem = idx - n * outChannels_ * outHeight_ * outWidth_;
            uint32_t co = rem / (outHeight_ * outWidth_);
            rem = rem - co * outHeight_ * outWidth_;
            uint32_t oh = rem / outWidth_;
            uint32_t ow = rem - oh * outWidth_;

            float sum = 0.0f;
            for (uint32_t ci = 0; ci < inChannels_; ci++) {
                for (uint32_t kh = 0; kh < kH_; kh++) {
                    for (uint32_t kw = 0; kw < kW_; kw++) {
                        int32_t ih = (int32_t)(oh + padH_) - (int32_t)kh;
                        int32_t iw = (int32_t)(ow + padW_) - (int32_t)kw;
                        if (ih % strideH_ != 0 || iw % strideW_ != 0) continue;
                        ih /= strideH_;
                        iw /= strideW_;
                        if (ih < 0 || (uint32_t)ih >= inHeight_ || iw < 0 || (uint32_t)iw >= inWidth_) continue;
                        float xVal = xGm_[n * inChannels_ * inHeight_ * inWidth_ + ci * inHeight_ * inWidth_ + (uint32_t)ih * inWidth_ + (uint32_t)iw];
                        float wVal = wGm_[co * inChannels_ * kH_ * kW_ + ci * kH_ * kW_ + kh * kW_ + kw];
                        sum += xVal * wVal;
                    }
                }
            }
            yGm_[idx] = sum;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t inChannels_;
    uint32_t inHeight_;
    uint32_t inWidth_;
    uint32_t outChannels_;
    uint32_t outHeight_;
    uint32_t outWidth_;
    uint32_t kH_;
    uint32_t kW_;
    int32_t strideH_;
    int32_t strideW_;
    int32_t padH_;
    int32_t padW_;
    uint32_t totalOutputElems_;
};

extern "C" __global__ __aicore__ void ca6k_01096_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    const Ca6k01096CustomTilingData *tiling_data = (const Ca6k01096CustomTilingData *)tiling;
    KernelCa6k01096Custom op;
    op.Init(
        tiling_data->batch, tiling_data->inChannels, tiling_data->inHeight, tiling_data->inWidth,
        tiling_data->outChannels, tiling_data->outHeight, tiling_data->outWidth,
        tiling_data->kernelHeight, tiling_data->kernelWidth,
        tiling_data->strideH, tiling_data->strideW, tiling_data->padH, tiling_data->padW,
        tiling_data->totalOutputElems,
        (__gm__ float *)x, (__gm__ float *)weight, (__gm__ float *)y
    );
    op.Process();
}
