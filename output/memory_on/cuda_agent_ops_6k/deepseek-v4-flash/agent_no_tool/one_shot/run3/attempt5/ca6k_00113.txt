host_operator_src = """
#include "ca6k00113_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cstdint>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 4096;
static constexpr uint32_t MAX_TILE_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00113CustomTilingData tiling;
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(0));
    uint32_t C = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(1));
    uint32_t H = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(2));
    uint32_t W = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(3));

    uint32_t totalPixels = N * H * W;
    uint32_t spatialSize = H * W;

    // Each output tile: tilePixels int64 elements = 8*tilePixels bytes
    uint32_t maxTile = (UB_BUDGET_BYTES / (2 * static_cast<uint32_t>(sizeof(int64_t)))) / ALIGN_ELEMS * ALIGN_ELEMS;
    if (maxTile > MAX_TILE_ELEMS) {
        maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);
    }
    if (maxTile == 0) maxTile = ALIGN_ELEMS;

    uint32_t tilePixels = maxTile;
    if (totalPixels < tilePixels) {
        tilePixels = AlignDown(totalPixels, ALIGN_ELEMS);
        if (tilePixels == 0) tilePixels = totalPixels;
    }

    uint32_t numTiles = CeilDiv(totalPixels, tilePixels);
    uint32_t blockDim = numTiles;
    if (blockDim < 2) blockDim = 2;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    // Direct member assignment (no setter methods)
    tiling.totalPixels = totalPixels;
    tiling.tilePixels = tilePixels;
    tiling.blockDim = blockDim;
    tiling.channelNum = C;
    tiling.spatialSize = spatialSize;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *inputShape = context->GetInputShape(0);
    int64_t N = inputShape->GetDim(0);
    int64_t H = inputShape->GetDim(2);
    int64_t W = inputShape->GetDim(3);

    gert::Shape outputShape;
    outputShape.SetDimNum(3);
    outputShape.SetDim(0, N);
    outputShape.SetDim(1, H);
    outputShape.SetDim(2, W);
    *context->GetOutputShape(0) = outputShape;

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmax {
public:
    __aicore__ inline KernelArgmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalPixels, uint32_t tilePixels,
                                uint32_t channelNum, uint32_t spatialSize) {
        totalPixels_ = totalPixels;
        tilePixels_ = tilePixels;
        channelNum_ = channelNum;
        spatialSize_ = spatialSize;
        inputSize_ = totalPixels_ * channelNum_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, inputSize_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalPixels_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tilePixels_ * sizeof(int64_t));
    }

    __aicore__ inline void Process() {
        if (totalPixels_ == 0 || tilePixels_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t start = blockIdx * tilePixels_;
        if (start >= totalPixels_) return;

        uint32_t curStart = start;
        uint32_t curLen = totalPixels_ - curStart;
        if (curLen > tilePixels_) curLen = tilePixels_;

        AscendC::LocalTensor<int64_t> yLocal0 = outQueueY_.AllocTensor<int64_t>();
        Compute(curStart, curLen, yLocal0);
        outQueueY_.EnQue<int64_t>(yLocal0);

        while (true) {
            uint32_t nextStart = curStart + blockNum * tilePixels_;
            uint32_t nextLen = 0;
            if (nextStart < totalPixels_) {
                nextLen = totalPixels_ - nextStart;
                if (nextLen > tilePixels_) nextLen = tilePixels_;
                AscendC::LocalTensor<int64_t> yLocalNext = outQueueY_.AllocTensor<int64_t>();
                Compute(nextStart, nextLen, yLocalNext);
                outQueueY_.EnQue<int64_t>(yLocalNext);
            }

            AscendC::LocalTensor<int64_t> yLocalPrev = outQueueY_.DeQue<int64_t>();
            AscendC::DataCopy(yGm_[curStart], yLocalPrev, curLen);
            outQueueY_.FreeTensor(yLocalPrev);

            if (nextLen == 0) break;
            curStart = nextStart;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void Compute(uint32_t pixelOffset, uint32_t len,
                                   AscendC::LocalTensor<int64_t> &yLocal) {
        for (uint32_t i = 0; i < len; i++) {
            uint32_t globalPixelIdx = pixelOffset + i;
            uint32_t n = globalPixelIdx / spatialSize_;
            uint32_t hw = globalPixelIdx % spatialSize_;
            uint32_t base = n * channelNum_ * spatialSize_ + hw;

            float maxVal = xGm_[base];
            int64_t maxIdx = 0;
            for (uint32_t c = 1; c < channelNum_; c++) {
                float curVal = xGm_[base + c * spatialSize_];
                if (curVal > maxVal) {
                    maxVal = curVal;
                    maxIdx = static_cast<int64_t>(c);
                }
            }
            yLocal.SetValue(i, maxIdx);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalPixels_ {0};
    uint32_t tilePixels_ {0};
    uint32_t channelNum_ {0};
    uint32_t spatialSize_ {0};
    uint32_t inputSize_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00113_custom(GM_ADDR x, GM_ADDR y,
                                                         GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmax op;
    op.Init(x, y,
            tiling_data.totalPixels,
            tiling_data.tilePixels,
            tiling_data.channelNum,
            tiling_data.spatialSize);
    op.Process();
}
