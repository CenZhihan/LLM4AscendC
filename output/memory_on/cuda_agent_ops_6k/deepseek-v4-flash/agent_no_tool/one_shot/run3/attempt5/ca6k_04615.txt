#include "ca6k_04615_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k04615CustomTilingData tiling;

    const gert::Shape &xOrigin = context->GetInputShape(0)->GetOriginShape();
    int32_t batch = static_cast<int32_t>(xOrigin.GetDim(0));
    int32_t channels = static_cast<int32_t>(xOrigin.GetDim(1));
    int32_t inH = static_cast<int32_t>(xOrigin.GetDim(2));
    int32_t inW = static_cast<int32_t>(xOrigin.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float scaleFactor = *attrs->GetAttrPointer<float>(0);

    int32_t outH = static_cast<int32_t>(static_cast<float>(inH) * scaleFactor);
    int32_t outW = static_cast<int32_t>(static_cast<float>(inW) * scaleFactor);

    uint32_t totalLength = static_cast<uint32_t>(batch) *
                           static_cast<uint32_t>(channels) *
                           static_cast<uint32_t>(outH) *
                           static_cast<uint32_t>(outW);

    uint32_t tileLen = static_cast<uint32_t>(outW) * static_cast<uint32_t>(channels);
    if (tileLen > HARD_CAP_ELEMS) {
        tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    }
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_scaleFactor(scaleFactor);
    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *xShape = context->GetInputShape(0);
    gert::Shape *yShape = context->GetOutputShape(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float scaleFactor = *attrs->GetAttrPointer<float>(0);

    yShape->SetDimNum(4);
    yShape->SetDim(0, xShape->GetDim(0));
    yShape->SetDim(1, xShape->GetDim(1));
    yShape->SetDim(2, static_cast<int64_t>(static_cast<float>(xShape->GetDim(2)) * scaleFactor));
    yShape->SetDim(3, static_cast<int64_t>(static_cast<float>(xShape->GetDim(3)) * scaleFactor));

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelUpsampleBilinear {
public:
    __aicore__ inline KernelUpsampleBilinear() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                float scaleFactor,
                                int32_t batch, int32_t channels,
                                int32_t inH, int32_t inW,
                                int32_t outH, int32_t outW) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);

        totalLen_ = totalLength;
        tileLen_ = tileLength;
        scaleFactor_ = scaleFactor;
        batch_ = batch;
        channels_ = channels;
        inH_ = inH;
        inW_ = inW;
        outH_ = outH;
        outW_ = outW;

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output tiles
        for (uint32_t base = blockIdx * tileLen_; base < totalLen_; base += blockNum * tileLen_) {
            uint32_t curLen = base + tileLen_ <= totalLen_ ? tileLen_ : totalLen_ - base;
            Compute(base, curLen);
        }
    }

private:
    __aicore__ inline void Compute(uint32_t offset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; i++) {
            uint32_t globalIdx = offset + i;
            uint32_t perImage = static_cast<uint32_t>(channels_) * static_cast<uint32_t>(outH_) * static_cast<uint32_t>(outW_);
            int32_t n = static_cast<int32_t>(globalIdx / perImage);
            uint32_t idxInImage = globalIdx % perImage;

            int32_t h_out = static_cast<int32_t>(idxInImage / (channels_ * outW_));
            int32_t temp = static_cast<int32_t>(idxInImage % (channels_ * outW_));
            int32_t c = temp / outW_;
            int32_t w_out = temp % outW_;

            float h_in_f = static_cast<float>(h_out) / scaleFactor_;
            float w_in_f = static_cast<float>(w_out) / scaleFactor_;

            int32_t h0 = static_cast<int32_t>(h_in_f);
            int32_t w0 = static_cast<int32_t>(w_in_f);
            int32_t h1 = h0 + 1;
            int32_t w1 = w0 + 1;

            // Clamp to valid range
            h0 = (h0 < 0) ? 0 : (h0 >= inH_ ? inH_ - 1 : h0);
            w0 = (w0 < 0) ? 0 : (w0 >= inW_ ? inW_ - 1 : w0);
            h1 = (h1 < 0) ? 0 : (h1 >= inH_ ? inH_ - 1 : h1);
            w1 = (w1 < 0) ? 0 : (w1 >= inW_ ? inW_ - 1 : w1);

            float h_frac = h_in_f - static_cast<float>(h0);
            float w_frac = w_in_f - static_cast<float>(w0);

            // Input linear indices (NCHW)
            uint32_t inPerImage = static_cast<uint32_t>(channels_) * static_cast<uint32_t>(inH_) * static_cast<uint32_t>(inW_);
            uint32_t baseN = static_cast<uint32_t>(n) * inPerImage;
            uint32_t baseC = baseN + static_cast<uint32_t>(c) * static_cast<uint32_t>(inH_) * static_cast<uint32_t>(inW_);
            uint32_t idx00 = baseC + static_cast<uint32_t>(h0) * static_cast<uint32_t>(inW_) + static_cast<uint32_t>(w0);
            uint32_t idx01 = baseC + static_cast<uint32_t>(h0) * static_cast<uint32_t>(inW_) + static_cast<uint32_t>(w1);
            uint32_t idx10 = baseC + static_cast<uint32_t>(h1) * static_cast<uint32_t>(inW_) + static_cast<uint32_t>(w0);
            uint32_t idx11 = baseC + static_cast<uint32_t>(h1) * static_cast<uint32_t>(inW_) + static_cast<uint32_t>(w1);

            float p00 = xGm_[idx00];
            float p01 = xGm_[idx01];
            float p10 = xGm_[idx10];
            float p11 = xGm_[idx11];

            float val = (1.0f - h_frac) * (1.0f - w_frac) * p00
                      + (1.0f - h_frac) * w_frac * p01
                      + h_frac * (1.0f - w_frac) * p10
                      + h_frac * w_frac * p11;

            yLocal.SetValue(i, val);
        }

        AscendC::DataCopy(yGm_[offset], yLocal, len);
        outQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    float scaleFactor_;
    int32_t batch_;
    int32_t channels_;
    int32_t inH_;
    int32_t inW_;
    int32_t outH_;
    int32_t outW_;
};

extern "C" __global__ __aicore__ void ca6k_04615_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelUpsampleBilinear op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.scaleFactor,
            tiling_data.batch, tiling_data.channels,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor upsample_bilinear_npu(const at::Tensor& self, double scale_factor) {
    auto N = self.size(0);
    auto C = self.size(1);
    auto H = self.size(2);
    auto W = self.size(3);
    int64_t outH = static_cast<int64_t>(static_cast<double>(H) * scale_factor);
    int64_t outW = static_cast<int64_t>(static_cast<double>(W) * scale_factor);
    at::Tensor result = at::empty({N, C, outH, outW}, self.options());
    EXEC_NPU_CMD(aclnnCa6k04615Custom, self, static_cast<float>(scale_factor), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_04615_custom", &upsample_bilinear_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("ca6k_04615_custom", &upsample_bilinear_npu, "Bilinear upsampling (Ca6k04615Custom on NPU)");
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_04615_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, scale_factor):
        super(ModelNew, self).__init__()
        self.scale_factor = scale_factor
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1)

    def forward(self, x):
        # Custom bilinear upsampling
        fn = getattr(custom_ops_lib, _FN)
        x = fn(x, self.scale_factor)
        x = self.conv(x)
        return x
'''
