#include "ca6k_00266_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00266CustomTilingData tiling;
    auto *weight_shape = context->GetInputShape(0)->GetOriginShape();
    auto *indices_shape = context->GetInputShape(1)->GetOriginShape();
    auto *offsets_shape = context->GetInputShape(2)->GetOriginShape();

    uint32_t numEmbeddings = static_cast<uint32_t>(weight_shape.GetDim(0));
    uint32_t embeddingDim = static_cast<uint32_t>(weight_shape.GetDim(1));
    uint32_t totalIndices = static_cast<uint32_t>(indices_shape.GetShapeSize());
    uint32_t numBags = static_cast<uint32_t>(offsets_shape.GetShapeSize() - 1);

    // Determine block dimension: limit to number of bags and hardware maximum
    uint32_t maxBlocks = 24;  // conservative for 910B
    uint32_t blockDim = (numBags < maxBlocks) ? numBags : maxBlocks;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_numEmbeddings(numEmbeddings);
    tiling.set_embeddingDim(embeddingDim);
    tiling.set_totalIndices(totalIndices);
    tiling.set_numBags(numBags);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    auto *weight_shape = context->GetInputShape(0)->GetOriginShape();
    auto *offsets_shape = context->GetInputShape(2)->GetOriginShape();
    uint32_t numBags = static_cast<uint32_t>(offsets_shape.GetDim(0) - 1);
    uint32_t embeddingDim = static_cast<uint32_t>(weight_shape.GetDim(1));

    gert::Shape *out_shape = context->GetOutputShape(0);
    out_shape->SetDimNum(2);
    out_shape->SetDim(0, numBags);
    out_shape->SetDim(1, embeddingDim);

kernel_src = r"""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00266Custom {
public:
    __aicore__ inline KernelCa6k00266Custom() {}

    __aicore__ inline void Init(GM_ADDR weight, GM_ADDR indices, GM_ADDR offsets,
                                GM_ADDR output,
                                uint32_t numEmbeddings, uint32_t embeddingDim,
                                uint32_t totalIndices, uint32_t numBags,
                                uint32_t blockDim)
    {
        numEmbeddings_ = numEmbeddings;
        embeddingDim_ = embeddingDim;
        totalIndices_ = totalIndices;
        numBags_ = numBags;

        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numEmbeddings * embeddingDim);
        indicesGm_.SetGlobalBuffer((__gm__ int32_t*)indices, totalIndices);
        offsetsGm_.SetGlobalBuffer((__gm__ int32_t*)offsets, numBags + 1);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, numBags * embeddingDim);

        pipe_.InitBuffer(inQueueIdx_, BUFFER_NUM, 1 * sizeof(int32_t));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, embeddingDim * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1, embeddingDim * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (numBags_ == 0 || embeddingDim_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Distribute bags across blocks in interleaved fashion
        uint32_t bagStart = blockIdx;
        while (bagStart < numBags_) {
            ProcessBag(bagStart);
            bagStart += blockNum;
        }
    }

private:
    __aicore__ inline void ProcessBag(uint32_t bagIdx)
    {
        int32_t startOffset = offsetsGm_.GetValue(bagIdx);
        int32_t endOffset = offsetsGm_.GetValue(bagIdx + 1);
        uint32_t bagLen = static_cast<uint32_t>(endOffset - startOffset);

        // Allocate accumulator and initialize with 0
        AscendC::LocalTensor<float> accLocal = accQueue_.AllocTensor<float>();
        AscendC::Duplicated(accLocal, 0.0f, embeddingDim_);

        for (uint32_t i = 0; i < bagLen; ++i) {
            int32_t idx = indicesGm_.GetValue(startOffset + i);
            uint32_t rowOffset = static_cast<uint32_t>(idx) * embeddingDim_;

            // Load one row of weight
            AscendC::LocalTensor<float> rowLocal = inQueueW_.AllocTensor<float>();
            AscendC::DataCopy(rowLocal, weightGm_[rowOffset], embeddingDim_);
            inQueueW_.EnQue(rowLocal);

            rowLocal = inQueueW_.DeQue<float>();
            AscendC::Add(accLocal, accLocal, rowLocal, embeddingDim_);
            inQueueW_.FreeTensor(rowLocal);
        }

        // Write accumulator to output
        uint32_t outOffset = bagIdx * embeddingDim_;
        AscendC::DataCopy(outputGm_[outOffset], accLocal, embeddingDim_);
        accQueue_.FreeTensor(accLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdx_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<int32_t> indicesGm_;
    AscendC::GlobalTensor<int32_t> offsetsGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t numEmbeddings_ {0};
    uint32_t embeddingDim_ {0};
    uint32_t totalIndices_ {0};
    uint32_t numBags_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00266_custom(GM_ADDR weight, GM_ADDR indices,
                                                        GM_ADDR offsets, GM_ADDR output,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00266Custom op;
    op.Init(weight, indices, offsets, output,
            tiling_data.numEmbeddings, tiling_data.embeddingDim,
            tiling_data.totalIndices, tiling_data.numBags,
            tiling_data.blockDim);
    op.Process();
}
"""

python_bind_src = r"""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor ca6k_00266_custom_impl(const at::Tensor& weight,
                                  const at::Tensor& indices,
                                  const at::Tensor& offsets,
                                  const std::string& mode) {
    // mode is currently unused; only "sum" is supported
    int64_t numBags = offsets.size(0) - 1;
    int64_t embDim = weight.size(1);
    at::Tensor result = at::empty({numBags, embDim}, weight.options());
    EXEC_NPU_CMD(aclnnCa6k00266Custom, weight, indices, offsets, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_00266_custom", &ca6k_00266_custom_impl);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("ca6k_00266_custom", &ca6k_00266_custom_impl, "Custom EmbeddingBag (sum) on NPU");
}

model_src = r'''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00266_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_embeddings, embedding_dim):
        super(ModelNew, self).__init__()
        # weight is learned; we store it as a parameter inside the module
        self.weight = torch.nn.Parameter(torch.randn(num_embeddings, embedding_dim))

    def forward(self, input, offsets):
        fn = getattr(custom_ops_lib, _FN)
        return fn(self.weight, input, offsets)
