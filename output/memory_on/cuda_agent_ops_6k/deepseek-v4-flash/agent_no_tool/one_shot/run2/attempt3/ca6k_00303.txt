#include "ca6k00303_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00303CustomTilingData tiling;
    const auto x_shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t N = static_cast<uint32_t>(x_shape.GetDim(0));
    const uint32_t C_in = static_cast<uint32_t>(x_shape.GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape.GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *r_attr = attrs->GetAttrPointer<int64_t>(0);
    const uint32_t r = static_cast<uint32_t>(*r_attr);
    const uint32_t C_out = C_in / (r * r);
    const uint32_t outH = H * r;
    const uint32_t outW = W * r;
    const uint32_t totalElements = N * C_out * outH * outW;

    // Single block to avoid concurrency issues in element-by-element kernel
    context->SetBlockDim(1);

    tiling.set_totalElements(totalElements);
    tiling.set_batchSize(N);
    tiling.set_inChannels(C_in);
    tiling.set_inH(H);
    tiling.set_inW(W);
    tiling.set_r(r);
    tiling.set_outChannels(C_out);
    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

kernel_src = """
#include "kernel_operator.h"

class KernelCa6k00303Custom {
public:
    __aicore__ inline KernelCa6k00303Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalElements,
                                uint32_t batchSize, uint32_t inChannels,
                                uint32_t inH, uint32_t inW, uint32_t r,
                                uint32_t outChannels, uint32_t outH, uint32_t outW)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElements);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElements);
        totalElements_ = totalElements;
        batchSize_ = batchSize;
        inChannels_ = inChannels;
        inH_ = inH;
        inW_ = inW;
        r_ = r;
        outChannels_ = outChannels;
        outH_ = outH;
        outW_ = outW;
    }

    __aicore__ inline void Process()
    {
        if (totalElements_ == 0) return;
        // Single block, sequentially process all output elements
        for (uint32_t i = 0; i < totalElements_; i++) {
            // compute output coordinates (N, C_out, H_out, W_out)
            uint32_t tmp = i;
            uint32_t n = tmp / (outChannels_ * outH_ * outW_);
            tmp %= (outChannels_ * outH_ * outW_);
            uint32_t c_out = tmp / (outH_ * outW_);
            tmp %= (outH_ * outW_);
            uint32_t h_out = tmp / outW_;
            uint32_t w_out = tmp % outW_;

            // compute input coordinates
            uint32_t h_in = h_out / r_;
            uint32_t w_in = w_out / r_;
            uint32_t c_in = (h_out % r_) * r_ * outChannels_ + (w_out % r_) * outChannels_ + c_out;

            uint32_t inIdx = n * inChannels_ * inH_ * inW_ + c_in * inH_ * inW_ + h_in * inW_ + w_in;
            yGm_.SetValue(i, xGm_.GetValue(inIdx));
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalElements_{0};
    uint32_t batchSize_{0};
    uint32_t inChannels_{0};
    uint32_t inH_{0};
    uint32_t inW_{0};
    uint32_t r_{0};
    uint32_t outChannels_{0};
    uint32_t outH_{0};
    uint32_t outW_{0};
};

extern "C" __global__ __aicore__ void ca6k00303_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00303Custom op;
    op.Init(x, y, tiling_data.totalElements,
            tiling_data.batchSize, tiling_data.inChannels,
            tiling_data.inH, tiling_data.inW, tiling_data.r,
            tiling_data.outChannels, tiling_data.outH, tiling_data.outW);
    op.Process();
}
"""

#include <torch/library.h>

at::Tensor ca6k00303_custom_impl_npu(const at::Tensor& self, int64_t scale_factor) {
    auto shape = self.sizes();
    TORCH_CHECK(shape.size() == 4, "input must be 4D");
    int64_t N = shape[0];
    int64_t C_in = shape[1];
    int64_t H = shape[2];
    int64_t W = shape[3];
    int64_t r = scale_factor;
    int64_t C_out = C_in / (r * r);
    int64_t outH = H * r;
    int64_t outW = W * r;
    at::Tensor result = at::empty({N, C_out, outH, outW}, self.options());
    EXEC_NPU_CMD(aclnnCa6k00303Custom, self, scale_factor, result);
    return result;
}
