namespace optiling {

static constexpr uint32_t UB_BUDGET = 192 * 1024;  // bytes
static constexpr uint32_t ALIGN = 32;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00071CustomTilingData tiling;

    // Read dim from input tensor (host side)
    const int32_t *dimPtr = context->GetInputTensor(1)->GetData<int32_t>();
    int32_t dim = dimPtr[0];

    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &xShape = storageShape->GetOriginShape();
    uint32_t ndim = xShape.GetDimNum();
    uint32_t outerSize = 1;
    uint32_t innerSize = 1;

    if (ndim >= 2) {
        if (dim == 0) {
            outerSize = xShape.GetDim(1);
            innerSize = xShape.GetDim(0);
        } else {
            outerSize = xShape.GetDim(0);
            innerSize = xShape.GetDim(1);
        }
    } else if (ndim == 1) {
        outerSize = 1;
        innerSize = xShape.GetDim(0);
    }

    uint32_t totalLen = xShape.GetShapeSize();
    uint32_t tileLen = innerSize;
    if (tileLen * sizeof(float) > UB_BUDGET) {
        tileLen = UB_BUDGET / sizeof(float);
        tileLen = (tileLen / ALIGN) * ALIGN;
        if (tileLen > innerSize) tileLen = innerSize;
    }

    uint32_t blockDim = outerSize;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalLen(totalLen);
    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);
    tiling.set_dim(dim);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00071Custom {
public:
    __aicore__ inline KernelCa6k00071Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLen, uint32_t outerSize,
                                uint32_t innerSize, int32_t dim,
                                uint32_t tileLen, uint32_t blockDim)
    {
        totalLen_ = totalLen;
        outerSize_ = outerSize;
        innerSize_ = innerSize;
        dim_ = dim;
        tileLen_ = tileLen;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inBuf_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outBuf_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = blockDim_;

        uint32_t outerPerBlock = (outerSize_ + blockNum - 1) / blockNum;
        uint32_t outerStart = blockIdx * outerPerBlock;
        uint32_t outerEnd = outerStart + outerPerBlock;
        if (outerEnd > outerSize_) outerEnd = outerSize_;

        for (uint32_t outerIdx = outerStart; outerIdx < outerEnd; ++outerIdx) {
            if (dim_ == 0) {
                ProcessColumn(outerIdx);
            } else {
                ProcessRow(outerIdx);
            }
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t base = row * innerSize_;
        AscendC::LocalTensor<float> local = inBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> outLocal = outBuf_.AllocTensor<float>();

        AscendC::DataCopy(local, xGm_[base], innerSize_);

        float runningSum = 0.0f;
        for (uint32_t i = 0; i < innerSize_; ++i) {
            runningSum += local(i);
            outLocal(i) = runningSum;
        }

        AscendC::DataCopy(yGm_[base], outLocal, innerSize_);

        inBuf_.FreeTensor(local);
        outBuf_.FreeTensor(outLocal);
    }

    __aicore__ inline void ProcessColumn(uint32_t col)
    {
        AscendC::LocalTensor<float> local = inBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> outLocal = outBuf_.AllocTensor<float>();

        // load column data (non-contiguous)
        for (uint32_t row = 0; row < innerSize_; ++row) {
            local(row) = xGm_.GetValue()[row * outerSize_ + col];
        }

        float runningSum = 0.0f;
        for (uint32_t i = 0; i < innerSize_; ++i) {
            runningSum += local(i);
            outLocal(i) = runningSum;
        }

        // write back column
        for (uint32_t row = 0; row < innerSize_; ++row) {
            yGm_.GetValue()[row * outerSize_ + col] = outLocal(row);
        }

        inBuf_.FreeTensor(local);
        outBuf_.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t outerSize_;
    uint32_t innerSize_;
    int32_t dim_;
    uint32_t tileLen_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void ca6k_00071_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00071Custom op;
    op.Init(x, y,
            tiling_data.totalLen, tiling_data.outerSize,
            tiling_data.innerSize, tiling_data.dim,
            tiling_data.tileLen, tiling_data.blockDim);
    op.Process();
}
