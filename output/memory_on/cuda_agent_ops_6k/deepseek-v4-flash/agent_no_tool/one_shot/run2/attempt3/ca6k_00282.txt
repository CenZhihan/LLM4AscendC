#include "ca6k_00282_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_SIZE = 192 * 1024;
static constexpr uint32_t ALIGN = 8;
static constexpr uint32_t MAX_TILE = 16384;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00282CustomTilingData tiling;

    auto shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = shape.GetDim(0);
    uint32_t C = shape.GetDim(1);
    uint32_t D = shape.GetDim(2);
    uint32_t H = shape.GetDim(3);
    uint32_t W = shape.GetDim(4);
    uint32_t spatialSize = D * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps_ptr = attrs->GetAttrPointer<float>(0);
    float eps_val = (eps_ptr != nullptr) ? *eps_ptr : 1e-5f;

    uint32_t maxTile = (UB_SIZE / sizeof(float)) / 2;
    if (maxTile > MAX_TILE) maxTile = MAX_TILE;
    maxTile = (maxTile / ALIGN) * ALIGN;
    if (maxTile < ALIGN) maxTile = ALIGN;

    uint32_t tileLen = (spatialSize < maxTile) ? ((spatialSize / ALIGN) * ALIGN) : maxTile;
    if (tileLen == 0) tileLen = spatialSize;

    uint32_t blockDim = N * C;
    if (blockDim > 40) blockDim = 40;
    if (blockDim < 8) blockDim = 8;
    context->SetBlockDim(blockDim);

    // Use setter methods to avoid private member access errors
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_spatialSize(spatialSize);
    tiling.set_tileLength(tileLen);
    tiling.set_eps(eps_val);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
