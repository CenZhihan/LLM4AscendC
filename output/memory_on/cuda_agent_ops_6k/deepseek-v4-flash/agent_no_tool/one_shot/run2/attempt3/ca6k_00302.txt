static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00302CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Each tile: load x, target, compute abs diff (3 floats * 4B = 12B per element)
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00302Custom {
public:
    __aicore__ inline KernelCa6k00302Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR target, GM_ADDR y,
                                GM_ADDR workspace, uint32_t totalLength,
                                uint32_t tileLength, uint32_t blockNum)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockNum_ = blockNum;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        targetGm_.SetGlobalBuffer((__gm__ float*)target, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);
        // workspace layout: [partialSum0, partialSum1, ..., partialSum_{blockNum-1], blockCount]
        partialSumsGm_.SetGlobalBuffer((__gm__ float*)workspace, blockNum_);
        blockCountGm_.SetGlobalBuffer((__gm__ float*)workspace + blockNum_, 1);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueTarget_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueAbs_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t strideElems = blockNum_ * tileLen_;
        uint32_t base = blockIdx * tileLen_;

        if (base >= totalLen_) {
            // Write 0 partial sum for this block
            partialSumsGm_.SetValue(blockIdx, 0.0f);
            return;
        }

        float partialSum = 0.0f;
        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], curLen);
            AscendC::LocalTensor<float> targetLocal = inQueueTarget_.AllocTensor<float>();
            AscendC::DataCopy(targetLocal, targetGm_[base], curLen);

            AscendC::LocalTensor<float> absLocal = outQueueAbs_.AllocTensor<float>();
            AscendC::Sub(absLocal, xLocal, targetLocal, curLen);
            AscendC::Abs(absLocal, absLocal, curLen);

            for (uint32_t i = 0; i < curLen; i++) {
                partialSum += absLocal.GetValue(i);
            }

            inQueueX_.FreeTensor(xLocal);
            inQueueTarget_.FreeTensor(targetLocal);
            outQueueAbs_.FreeTensor(absLocal);

            base += strideElems;
        }

        // Write partial sum to workspace
        partialSumsGm_.SetValue(blockIdx, partialSum);

        // Block 0: accumulate all partial sums and compute mean
        if (blockIdx == 0) {
            float totalSum = 0.0f;
            for (uint32_t i = 0; i < blockNum_; i++) {
                totalSum += partialSumsGm_.GetValue(i);
            }
            float mean = totalSum / static_cast<float>(totalLen_);
            yGm_.SetValue(0, mean);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTarget_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueAbs_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> targetGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> partialSumsGm_;
    AscendC::GlobalTensor<float> blockCountGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockNum_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00302_custom(
    GM_ADDR x, GM_ADDR target, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00302Custom op;
    op.Init(x, target, y, workspace,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim);
    op.Process();
}
