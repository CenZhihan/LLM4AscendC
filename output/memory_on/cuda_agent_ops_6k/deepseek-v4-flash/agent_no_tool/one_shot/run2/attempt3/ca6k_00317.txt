kernel_src = r'''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00317Custom {
public:
    __aicore__ inline KernelCa6k00317Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                Ca6k00317CustomTilingData tiling)
    {
        tiling_ = tiling;

        n_ = tiling_.get_n();
        c_in_ = tiling_.get_c_in();
        c_out_ = tiling_.get_c_out();
        d_ = tiling_.get_d();
        h_ = tiling_.get_h();
        w_ = tiling_.get_w();
        kd_ = tiling_.get_kd();
        kh_ = tiling_.get_kh();
        kw_ = tiling_.get_kw();
        stride_d_ = tiling_.get_stride_d();
        stride_h_ = tiling_.get_stride_h();
        stride_w_ = tiling_.get_stride_w();
        pad_d_ = tiling_.get_pad_d();
        pad_h_ = tiling_.get_pad_h();
        pad_w_ = tiling_.get_pad_w();
        out_pad_d_ = tiling_.get_out_pad_d();
        out_pad_h_ = tiling_.get_out_pad_h();
        out_pad_w_ = tiling_.get_out_pad_w();
        dilation_d_ = tiling_.get_dilation_d();
        dilation_h_ = tiling_.get_dilation_h();
        dilation_w_ = tiling_.get_dilation_w();
        groups_ = tiling_.get_groups();

        out_d_ = (d_ - 1) * stride_d_ - 2 * pad_d_ + dilation_d_ * (kd_ - 1) + out_pad_d_ + 1;
        out_h_ = (h_ - 1) * stride_h_ - 2 * pad_h_ + dilation_h_ * (kh_ - 1) + out_pad_h_ + 1;
        out_w_ = (w_ - 1) * stride_w_ - 2 * pad_w_ + dilation_w_ * (kw_ - 1) + out_pad_w_ + 1;

        totalOutElems_ = n_ * c_out_ * out_d_ * out_h_ * out_w_;
        totalInElems_ = n_ * c_in_ * d_ * h_ * w_;
        totalWeightElems_ = c_in_ * c_out_ * kd_ * kh_ * kw_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInElems_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, totalWeightElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutElems_);

        // Use local buffers for vectorized operations
        uint32_t maxLocalSize = 1024; // conservative
        pipe_.InitBuffer(tmpBuf_, maxLocalSize * sizeof(float));
        pipe_.InitBuffer(weightBuf_, 1); // single weight scalar, reused
    }

    __aicore__ inline void Process()
    {
        // Naive loop: for each output element compute dot product across input channel and kernel.
        // This is correct but slow; for real use a tiled approach with vector DMA is needed.
        float* tmp = tmpBuf_.Get<float>();
        for (uint32_t on = 0; on < n_; on++) {
            for (uint32_t oc = 0; oc < c_out_; oc++) {
                for (uint32_t od = 0; od < out_d_; od++) {
                    for (uint32_t oh = 0; oh < out_h_; oh++) {
                        for (uint32_t ow = 0; ow < out_w_; ow++) {
                            float sum = 0.0f;
                            for (uint32_t ic = 0; ic < c_in_; ic++) {
                                for (uint32_t kdi = 0; kdi < kd_; kdi++) {
                                    int32_t id = static_cast<int32_t>(od) - static_cast<int32_t>(pad_d_) + kdi * static_cast<int32_t>(dilation_d_);
                                    if (id % static_cast<int32_t>(stride_d_) != 0) continue;
                                    id = id / static_cast<int32_t>(stride_d_);
                                    if (id < 0 || id >= static_cast<int32_t>(d_)) continue;
                                    for (uint32_t khi = 0; khi < kh_; khi++) {
                                        int32_t ih = static_cast<int32_t>(oh) - static_cast<int32_t>(pad_h_) + khi * static_cast<int32_t>(dilation_h_);
                                        if (ih % static_cast<int32_t>(stride_h_) != 0) continue;
                                        ih = ih / static_cast<int32_t>(stride_h_);
                                        if (ih < 0 || ih >= static_cast<int32_t>(h_)) continue;
                                        for (uint32_t kwi = 0; kwi < kw_; kwi++) {
                                            int32_t iw = static_cast<int32_t>(ow) - static_cast<int32_t>(pad_w_) + kwi * static_cast<int32_t>(dilation_w_);
                                            if (iw % static_cast<int32_t>(stride_w_) != 0) continue;
                                            iw = iw / static_cast<int32_t>(stride_w_);
                                            if (iw < 0 || iw >= static_cast<int32_t>(w_)) continue;
                                            float x_val = xGm_.GetValue(on * c_in_ * d_ * h_ * w_ + ic * d_ * h_ * w_ + id * h_ * w_ + ih * w_ + iw);
                                            float w_val = wGm_.GetValue(ic * c_out_ * kd_ * kh_ * kw_ + oc * kd_ * kh_ * kw_ + kdi * kh_ * kw_ + khi * kw_ + kwi);
                                            sum += x_val * w_val;
                                        }
                                    }
                                }
                            }
                            yGm_.SetValue(on * c_out_ * out_d_ * out_h_ * out_w_ + oc * out_d_ * out_h_ * out_w_ + od * out_h_ * out_w_ + oh * out_w_ + ow, sum);
                        }
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<TPosition::VECCALC> tmpBuf_;
    AscendC::TBuf<TPosition::VECCALC> weightBuf_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    Ca6k00317CustomTilingData tiling_;

    uint32_t n_, c_in_, c_out_, d_, h_, w_;
    uint32_t kd_, kh_, kw_;
    uint32_t stride_d_, stride_h_, stride_w_;
    uint32_t pad_d_, pad_h_, pad_w_;
    uint32_t out_pad_d_, out_pad_h_, out_pad_w_;
    uint32_t dilation_d_, dilation_h_, dilation_w_;
    uint32_t groups_;
    uint32_t out_d_, out_h_, out_w_;
    uint32_t totalOutElems_, totalInElems_, totalWeightElems_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void ca6k00317_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00317Custom op;
    op.Init(x, weight, y, tiling_data);
    op.Process();
}
'''

                               at::IntArrayRef output_padding, at::IntArrayRef dilation,
