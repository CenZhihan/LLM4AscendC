namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *in_shape = context->GetInputShape(0);
    gert::Shape *out_shape = context->GetOutputShape(0);
    // Copy input shape
    uint32_t dim_num = in_shape->GetDimNum();
    out_shape->SetDimNum(dim_num);
    for (uint32_t i = 0; i < dim_num; ++i) {
        out_shape->SetDim(i, in_shape->GetDim(i));
    }
    // Compute output length for last dimension
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t kernelSize = 3;
    int64_t stride = 2;
    if (attrs) {
        const int64_t *k_ptr = attrs->GetAttrPointer<int64_t>(1);
        if (k_ptr) kernelSize = *k_ptr;
        const int64_t *s_ptr = attrs->GetAttrPointer<int64_t>(2);
        if (s_ptr) stride = *s_ptr;
    }
    int64_t in_len = in_shape->GetDim(dim_num - 1);
    int64_t out_len = (in_len - kernelSize) / stride + 1;
    if (out_len < 0) out_len = 0;
    out_shape->SetDim(dim_num - 1, out_len);
    return GRAPH_SUCCESS;
}
