kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00294Custom {
public:
    __aicore__ inline KernelCa6k00294Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalInputLen, uint32_t embeddingDim,
                                uint32_t numEmbeddings, int64_t paddingIdx,
                                uint32_t tileLen)
    {
        totalInputLen_ = totalInputLen;
        embeddingDim_ = embeddingDim;
        numEmbeddings_ = numEmbeddings;
        paddingIdx_ = paddingIdx;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ int64_t*)x, totalInputLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, numEmbeddings_ * embeddingDim_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalInputLen_ * embeddingDim_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, sizeof(int64_t));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, embeddingDim_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalInputLen_ == 0 || embeddingDim_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t base = blockIdx;
        if (base >= totalInputLen_) return;

        while (base < totalInputLen_) {
            // Load input index
            AscendC::LocalTensor<int64_t> xLocal = inQueueX_.AllocTensor<int64_t>();
            AscendC::DataCopy(xLocal, xGm_[base], 1);
            int64_t idx = xLocal(0);
            inQueueX_.FreeTensor(xLocal);

            // Allocate output buffer for this row
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            if (idx == paddingIdx_) {
                // Fill with zeros
                AscendC::Muls(yLocal, yLocal, 0.0f, embeddingDim_);
            } else {
                // Load weight row
                uint32_t weightOffset = static_cast<uint32_t>(idx) * embeddingDim_;
                AscendC::DataCopy(yLocal, wGm_[weightOffset], embeddingDim_);
            }

            // Write output row
            uint32_t outOffset = base * embeddingDim_;
            AscendC::DataCopy(yGm_[outOffset], yLocal, embeddingDim_);

            outQueueY_.FreeTensor(yLocal);

            base += blockNum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<int64_t> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputLen_ {0};
    uint32_t embeddingDim_ {0};
    uint32_t numEmbeddings_ {0};
    int64_t paddingIdx_ {-1};
    uint32_t tileLen_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00294_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00294Custom op;
    op.Init(x, weight, y,
            tiling_data.totalInputLen, tiling_data.embeddingDim,
            tiling_data.numEmbeddings, tiling_data.paddingIdx,
            tiling_data.tileLen);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = "ca6k_00294_custom"

class ModelNew(torch.nn.Module):
    def __init__(self, num_embeddings, embedding_dim, padding_idx):
        super(ModelNew, self).__init__()
        self.embedding = torch.nn.Embedding(num_embeddings, embedding_dim, padding_idx=padding_idx)

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        pad = self.embedding.padding_idx if self.embedding.padding_idx is not None else -1
