#include "ca6k00113_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00113CustomTilingDataWrapper tiling;
    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &shape = storageShape->GetOriginShape();
    uint32_t ndim = shape.GetDimNum();
    uint32_t totalInputSize = 1;
    uint32_t inputDims[4] = {1, 1, 1, 1};
    for (uint32_t i = 0; i < ndim && i < 4; ++i) {
        int64_t d = shape.GetDim(i);
        inputDims[i] = static_cast<uint32_t>(d);
        totalInputSize *= d;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = *attrs->GetAttrPointer<int32_t>(0);
    if (dim < 0) dim += ndim;
    uint32_t outputDims[4];
    uint32_t outIdx = 0;
    for (uint32_t i = 0; i < ndim; ++i) {
        if (i == (uint32_t)dim) continue;
        outputDims[outIdx++] = inputDims[i];
    }
    uint32_t totalLength = 1;
    for (uint32_t i = 0; i < outIdx; ++i) totalLength *= outputDims[i];

    uint32_t maxTile = UB_BUDGET_BYTES / 4;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);
    tiling.set_N(inputDims[0]);
    tiling.set_C(inputDims[1]);
    tiling.set_H(inputDims[2]);
    tiling.set_W(inputDims[3]);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00113Custom {
public:
    __aicore__ inline KernelCa6k00113Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t dim,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        dim_ = dim;
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalLen_);

        strides_[3] = 1;
        strides_[2] = W_;
        strides_[1] = H_ * W_;
        strides_[0] = C_ * H_ * W_;
        reducedSize_ = (dim_ == 0) ? N_ : ((dim_ == 1) ? C_ : ((dim_ == 2) ? H_ : W_));
        reducedStride_ = strides_[dim_];
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            for (uint32_t i = 0; i < curLen; ++i) {
                int64_t outIdx = base + i;
                uint32_t b = outIdx / (H_ * W_);
                uint32_t remainder = outIdx % (H_ * W_);
                uint32_t h = remainder / W_;
                uint32_t w = remainder % W_;
                uint32_t baseIn = b * C_ * H_ * W_ + h * W_ + w;

                float maxVal = xGm_[baseIn];
                int64_t argMax = 0;
                for (uint32_t c = 1; c < C_; ++c) {
                    float cur = xGm_[baseIn + c * H_ * W_];
                    if (cur > maxVal) {
                        maxVal = cur;
                        argMax = c;
                    }
                }
                yGm_[outIdx] = argMax;
            }

            if (curLen < tileLen_) break;
            base += strideElems;
            if (base >= totalLen_) break;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ {1};
    uint32_t N_ {1}, C_ {1}, H_ {1}, W_ {1};
    uint32_t strides_[4];
    uint32_t reducedSize_ {1};
    uint32_t reducedStride_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00113_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00113Custom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.dim,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W);
    op.Process();
}
