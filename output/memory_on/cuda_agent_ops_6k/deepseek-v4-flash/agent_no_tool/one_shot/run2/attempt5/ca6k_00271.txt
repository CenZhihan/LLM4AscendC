namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00271CustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const auto &originShape = inputShape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t D = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(3));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(4));

    uint32_t spatialVolume = D * H * W;
    uint32_t blockDim = N * C;

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (spatialVolume < maxTile) {
        maxTile = AlignDown(spatialVolume, ALIGN_ELEMS);
        if (maxTile == 0) maxTile = spatialVolume;
    }
    uint32_t tileLength = maxTile;

    context->SetBlockDim(blockDim);
    tiling.set_totalLength(totalLength);
    tiling.set_spatialVolume(spatialVolume);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00271Custom {
public:
    __aicore__ inline KernelCa6k00271Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t spatialVolume,
                                uint32_t tileLength,
                                uint32_t blockDim) {
        totalLen_ = totalLength;
        spatialVol_ = spatialVolume;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, blockDim_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || spatialVol_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= blockDim_) return;

        uint32_t startOffset = blockIdx * spatialVol_;
        AscendC::LocalTensor<float> accLocal = pipe_.AllocTensor<float>(1);
        accLocal.SetValue(0, 0.0f);

        // Tile over spatial volume and accumulate
        for (uint32_t offset = 0; offset < spatialVol_; offset += tileLen_) {
            uint32_t curLen = spatialVol_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[startOffset + offset], curLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();
            // manual scalar accumulation (safe, compiles)
            for (uint32_t i = 0; i < curLen; i++) {
                float val = xTile.GetValue(i);
                float old = accLocal.GetValue(0);
                accLocal.SetValue(0, old + val);
            }
            inQueueX_.FreeTensor(xTile);
        }

        float sum = accLocal.GetValue(0);
        pipe_.FreeTensor(accLocal);
        float mean = sum / static_cast<float>(spatialVol_);

        // Write output
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        yLocal.SetValue(0, mean);
        AscendC::DataCopy(yGm_[blockIdx], yLocal, 1);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t spatialVol_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00271_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00271Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.spatialVolume, tiling_data.tileLength, tiling_data.blockDim);
    op.Process();
}
