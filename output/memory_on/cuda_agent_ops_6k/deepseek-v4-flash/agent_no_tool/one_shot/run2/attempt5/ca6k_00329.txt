kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00329Custom {
public:
    __aicore__ inline KernelCa6k00329Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                GM_ADDR running_mean, GM_ADDR running_var,
                                GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR batch_mean_out, GM_ADDR batch_var_out,
                                uint32_t totalRows, uint32_t C,
                                uint32_t tileRows, float momentum,
                                float eps, bool training)
    {
        totalRows_ = totalRows;
        C_ = C;
        tileRows_ = tileRows;
        momentum_ = momentum;
        eps_ = eps;
        training_ = training;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows * C);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows * C);
        runningMeanGm_.SetGlobalBuffer((__gm__ float*)running_mean, C);
        runningVarGm_.SetGlobalBuffer((__gm__ float*)running_var, C);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, C);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, C);
        batchMeanOutGm_.SetGlobalBuffer((__gm__ float*)batch_mean_out, C);
        batchVarOutGm_.SetGlobalBuffer((__gm__ float*)batch_var_out, C);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileRows_ * C_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * C_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileRows_ * C_ * sizeof(float));
        pipe_.InitBuffer(sumBuf_, 1, C_ * sizeof(float));
        pipe_.InitBuffer(sumsqBuf_, 1, C_ * sizeof(float));
        pipe_.InitBuffer(meanBuf_, 1, C_ * sizeof(float));
        pipe_.InitBuffer(varBuf_, 1, C_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || C_ == 0) return;

        AscendC::LocalTensor<float> sum = sumBuf_.Get<float>();
        AscendC::LocalTensor<float> sumsq = sumsqBuf_.Get<float>();
        AscendC::LocalTensor<float> mean = meanBuf_.Get<float>();
        AscendC::LocalTensor<float> var = varBuf_.Get<float>();

        AscendC::SetZero(sum, C_);
        AscendC::SetZero(sumsq, C_);

        uint32_t processed = 0;
        while (processed < totalRows_) {
            uint32_t curRows = (processed + tileRows_ <= totalRows_) ? tileRows_ : (totalRows_ - processed);

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[processed * C_], curRows * C_);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            for (uint32_t c = 0; c < C_; c++) {
                float s = 0.0f, sq = 0.0f;
                for (uint32_t r = 0; r < curRows; r++) {
                    float val = xLocal(r * C_ + c);
                    s += val;
                    sq += val * val;
                }
                sum(c) += s;
                sumsq(c) += sq;
            }
            inQueueX_.FreeTensor(xLocal);

            processed += curRows;
        }

        float invN = 1.0f / (float)totalRows_;
        for (uint32_t c = 0; c < C_; c++) {
            float m = sum(c) * invN;
            float v = sumsq(c) * invN - m * m;
            mean(c) = m;
            var(c) = v;
        }

        AscendC::DataCopy(batchMeanOutGm_, mean, C_);
        AscendC::DataCopy(batchVarOutGm_, var, C_);

        AscendC::LocalTensor<float> rm = sumBuf_.Get<float>();
        AscendC::LocalTensor<float> rv = sumsqBuf_.Get<float>();
        AscendC::LocalTensor<float> w = meanBuf_.Get<float>();
        AscendC::LocalTensor<float> b = varBuf_.Get<float>();
        AscendC::DataCopy(rm, runningMeanGm_, C_);
        AscendC::DataCopy(rv, runningVarGm_, C_);
        AscendC::DataCopy(w, weightGm_, C_);
        AscendC::DataCopy(b, biasGm_, C_);

        processed = 0;
        while (processed < totalRows_) {
            uint32_t curRows = (processed + tileRows_ <= totalRows_) ? tileRows_ : (totalRows_ - processed);

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[processed * C_], curRows * C_);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

            for (uint32_t r = 0; r < curRows; r++) {
                for (uint32_t c = 0; c < C_; c++) {
                    float xv = xLocal(r * C_ + c);
                    float m = (training_) ? mean(c) : rm(c);
                    float v = (training_) ? var(c) : rv(c);
                    float invSqrt = 1.0f / (sqrtf(v + eps_));
                    float normalized = (xv - m) * invSqrt;
                    yLocal(r * C_ + c) = w(c) * normalized + b(c);
                }
            }

            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);

            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[processed * C_], yLocal, curRows * C_);
            outQueueY_.FreeTensor(yLocal);

            processed += curRows;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumsqBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> meanBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> varBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> runningMeanGm_;
    AscendC::GlobalTensor<float> runningVarGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> batchMeanOutGm_;
    AscendC::GlobalTensor<float> batchVarOutGm_;

    uint32_t totalRows_ {0};
    uint32_t C_ {0};
    uint32_t tileRows_ {0};
    float momentum_ {0.1f};
    float eps_ {1e-5f};
    bool training_ {true};
};

extern "C" __global__ __aicore__ void ca6k_00329_custom(
    GM_ADDR x, GM_ADDR y,
    GM_ADDR running_mean, GM_ADDR running_var,
    GM_ADDR weight, GM_ADDR bias,
    GM_ADDR batch_mean_out, GM_ADDR batch_var_out,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00329Custom op;
    op.Init(x, y, running_mean, running_var, weight, bias,
            batch_mean_out, batch_var_out,
            tiling_data.totalRows, tiling_data.C,
            tiling_data.tileRows, tiling_data.momentum,
            tiling_data.eps, tiling_data.training);
    op.Process();
}
'''

model_src = """
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib as _ops

class ModelNew(nn.Module):
    def __init__(self, num_features: int):
        super().__init__()
        self.num_features = num_features
        self.affine = True
        self.track_running_stats = True
        self.momentum = 0.1
        self.eps = 1e-5

        if self.affine:
            self.weight = nn.Parameter(torch.ones(num_features))
            self.bias = nn.Parameter(torch.zeros(num_features))
        else:
            self.register_parameter('weight', None)
            self.register_parameter('bias', None)

        if self.track_running_stats:
            self.register_buffer('running_mean', torch.zeros(num_features))
            self.register_buffer('running_var', torch.ones(num_features))
        else:
            self.register_buffer('running_mean', None)
            self.register_buffer('running_var', None)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = x.to('npu').contiguous()

        rm = self.running_mean.to(x.device).contiguous()
        rv = self.running_var.to(x.device).contiguous()
        w = self.weight.to(x.device).contiguous() if self.weight is not None else torch.ones(self.num_features, device=x.device)
        b = self.bias.to(x.device).contiguous() if self.bias is not None else torch.zeros(self.num_features, device=x.device)

        result = _ops.ca6k_00329_custom(x, rm, rv, w, b, self.momentum, self.eps, self.training)
        y, batch_mean, batch_var = result[0], result[1], result[2]

        if self.training and self.track_running_stats:
            with torch.no_grad():
                self.running_mean = (1 - self.momentum) * self.running_mean + self.momentum * batch_mean.cpu()
                self.running_var = (1 - self.momentum) * self.running_var + self.momentum * batch_var.cpu()

        return y.cpu()
"""
