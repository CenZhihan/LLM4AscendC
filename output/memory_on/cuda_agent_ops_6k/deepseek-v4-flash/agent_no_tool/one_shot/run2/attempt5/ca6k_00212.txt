#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00212CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelL1Loss {
public:
    __aicore__ inline KernelL1Loss() {}

    __aicore__ inline void Init(GM_ADDR input, GM_ADDR target, GM_ADDR loss,
                                uint32_t totalLength, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inputGm_.SetGlobalBuffer((__gm__ float*)input, totalLen_);
        targetGm_.SetGlobalBuffer((__gm__ float*)target, totalLen_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        float sum = 0.0f;
        uint32_t offset = 0;
        while (offset < totalLen_)
        {
            uint32_t curLen = totalLen_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, inputGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, targetGm_[offset], curLen);
            inQueueY_.EnQue(yLocal);

            xLocal = inQueueX_.DeQue<float>();
            yLocal = inQueueY_.DeQue<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            AscendC::Sub(tLocal, xLocal, yLocal, curLen);
            AscendC::Abs(tLocal, tLocal, curLen);

            for (uint32_t i = 0; i < curLen; ++i)
            {
                sum += tLocal(i);
            }

            tmpQueue_.FreeTensor(tLocal);
            inQueueX_.FreeTensor(xLocal);
            inQueueY_.FreeTensor(yLocal);

            offset += curLen;
        }

        float invTotal = 1.0f / static_cast<float>(totalLen_);
        lossGm_.SetValue(0, sum * invTotal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> inputGm_;
    AscendC::GlobalTensor<float> targetGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00212_custom(GM_ADDR input, GM_ADDR target, GM_ADDR loss,
                                                       GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelL1Loss op;
    op.Init(input, target, loss, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
'''
