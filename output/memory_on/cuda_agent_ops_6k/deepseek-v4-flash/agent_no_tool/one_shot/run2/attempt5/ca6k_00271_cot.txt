    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00271CustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const auto &originShape = inputShape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t D = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(3));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(4));

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));

    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (spatialVolume < maxTile) {
        maxTile = AlignDown(spatialVolume, ALIGN_ELEMS);
        if (maxTile == 0) maxTile = spatialVolume;
