kernel_src="""
#include "kernel_operator.h"

class KernelCa6k00269Custom {
public:
    __aicore__ inline KernelCa6k00269Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H_in, uint32_t W_in,
                                uint32_t H_out, uint32_t W_out,
                                uint32_t kernelH, uint32_t kernelW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t ceil_mode)
    {
        N_ = N; C_ = C; H_in_ = H_in; W_in_ = W_in;
        H_out_ = H_out; W_out_ = W_out;
        kernelH_ = kernelH; kernelW_ = kernelW;
        strideH_ = strideH; strideW_ = strideW;
        ceil_mode_ = (ceil_mode != 0);
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        uint32_t totalOut = N_ * C_ * H_out_ * W_out_;
        if (totalOut == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t itemsPerBlock = (totalOut + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * itemsPerBlock;
        uint32_t end = (start + itemsPerBlock > totalOut) ? totalOut : start + itemsPerBlock;

        for (uint32_t idx = start; idx < end; idx++) {
            uint32_t ow = idx % W_out_;
            uint32_t oh = (idx / W_out_) % H_out_;
            uint32_t c = (idx / (W_out_ * H_out_)) % C_;
            uint32_t n = idx / (C_ * H_out_ * W_out_);

            uint32_t ih_start = oh * strideH_;
            uint32_t iw_start = ow * strideW_;
            float sum_sq = 0.0f;
            for (uint32_t kh = 0; kh < kernelH_; kh++) {
                for (uint32_t kw = 0; kw < kernelW_; kw++) {
                    uint32_t ih = ih_start + kh;
                    uint32_t iw = iw_start + kw;
                    if (ceil_mode_ && (ih >= H_in_ || iw >= W_in_)) continue;
                    float val = xGm_[n * C_ * H_in_ * W_in_ + c * H_in_ * W_in_ + ih * W_in_ + iw];
                    sum_sq += val * val;
                }
            }
            float scale = 1.0f / (kernelH_ * kernelW_);
            float result = sqrtf(sum_sq * scale);
            yGm_[n * C_ * H_out_ * W_out_ + c * H_out_ * W_out_ + oh * W_out_ + ow] = result;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t N_, C_, H_in_, W_in_, H_out_, W_out_;
    uint32_t kernelH_, kernelW_, strideH_, strideW_;
    bool ceil_mode_;
};

extern "C" __global__ __aicore__ void ca6k_00269_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00269Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C,
            tiling_data.H_in, tiling_data.W_in,
            tiling_data.H_out, tiling_data.W_out,
            tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.ceil_mode);
    op.Process();
}
