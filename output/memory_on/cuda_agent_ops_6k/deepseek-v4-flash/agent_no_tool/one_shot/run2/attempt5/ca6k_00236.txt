host_operator_src = '''
#include "ca6k_00236_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00236CustomTilingData tiling;

    const auto &origin_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t ndim = static_cast<uint32_t>(origin_shape.GetDimNum());
    if (ndim == 0) {
        return ge::GRAPH_FAILED;
    }

    uint32_t reductionLen = static_cast<uint32_t>(origin_shape.GetDim(ndim - 1));
    uint32_t totalRows = 1;
    for (uint32_t i = 0; i < ndim - 1; ++i) {
        totalRows *= static_cast<uint32_t>(origin_shape.GetDim(i));
    }
    if (totalRows == 0 || reductionLen == 0) {
        return ge::GRAPH_FAILED;
    }

    uint32_t maxRowsPerTile = UB_BUDGET_BYTES / (3 * reductionLen * sizeof(float));
    if (maxRowsPerTile == 0) maxRowsPerTile = 1;
    uint32_t tileRows = (totalRows < maxRowsPerTile) ? totalRows : maxRowsPerTile;
    tileRows = AlignDown(tileRows, 1);

    uint32_t blockDim = CeilDiv(totalRows, tileRows);
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_reductionLen(reductionLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

REGISTER_OP_TILING(Ca6k00236Custom, TilingFunc);
} // namespace optiling
'''

kernel_src = '''
#include "kernel_operator.h"
using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00236Custom {
public:
    __aicore__ inline KernelCa6k00236Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t reductionLen) {
        totalRows_ = totalRows;
        reductionLen_ = reductionLen;
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, reductionLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, reductionLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 3, reductionLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalRows_ == 0 || reductionLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        uint32_t rowsPerBlock = totalRows_ / blockNum;
        uint32_t rem = totalRows_ % blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock + (blockIdx < rem ? blockIdx : rem);
        uint32_t endRow = startRow + rowsPerBlock + (blockIdx < rem ? 1 : 0);

        for (uint32_t r = startRow; r < endRow; ++r) {
            uint32_t baseOffset = r * reductionLen_;
            ProcessRow(baseOffset);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t offset) {
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        DataCopy(xLocal, xGm_[offset], reductionLen_);

        // Negate
        LocalTensor<float> negLocal = tmpQueue_.AllocTensor<float>();
        Neg(negLocal, xLocal, reductionLen_);

        // Compute max of negLocal for stability
        LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
        ReduceMax(maxLocal, negLocal, reductionLen_);
        float maxVal = maxLocal(0);
        tmpQueue_.FreeTensor(maxLocal);

        // Subtract max
        LocalTensor<float> shiftedLocal = tmpQueue_.AllocTensor<float>();
        Sub(shiftedLocal, negLocal, maxVal, reductionLen_);
        tmpQueue_.FreeTensor(negLocal);

        // Exp
        LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
        Exp(expLocal, shiftedLocal, reductionLen_);
        tmpQueue_.FreeTensor(shiftedLocal);

        // Sum of exps
        LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
        ReduceSum(sumLocal, expLocal, reductionLen_);
        float sumVal = sumLocal(0);
        tmpQueue_.FreeTensor(sumLocal);

        // Divide
        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        Div(yLocal, expLocal, sumVal, reductionLen_);
        tmpQueue_.FreeTensor(expLocal);

        // Store
        DataCopy(yGm_[offset], yLocal, reductionLen_);

        outQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 3> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t reductionLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00236_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00236Custom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.reductionLen);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = "ca6k_00236_custom"

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, *inputs):
        fn = getattr(custom_ops_lib, _FN)
        x = inputs[0]
        dim = 1  # as in reference softmin(dim=1)
        ndim = x.dim()
        # move dim to last
        perm = list(range(ndim))
        perm.pop(dim)
        perm.append(dim)
        x_perm = x.permute(perm)
        y_perm = fn(x_perm)
        # inverse permute
        inv_perm = list(range(ndim))
        inv_perm.insert(dim, inv_perm.pop())
        y = y_perm.permute(inv_perm)
        return y
'''
