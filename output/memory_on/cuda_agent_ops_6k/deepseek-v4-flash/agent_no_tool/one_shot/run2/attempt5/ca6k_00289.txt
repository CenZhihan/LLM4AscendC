kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00289Custom {
public:
    __aicore__ inline KernelCa6k00289Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t D,
                                uint32_t H, uint32_t W,
                                uint32_t D_out, uint32_t H_out, uint32_t W_out,
                                uint32_t kernelD, uint32_t kernelH, uint32_t kernelW,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t padD, uint32_t padH, uint32_t padW,
                                uint32_t blockDim)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        N_ = N;
        C_ = C;
        D_ = D;
        H_ = H;
        W_ = W;
        D_out_ = D_out;
        H_out_ = H_out;
        W_out_ = W_out;
        kernelD_ = kernelD;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        padD_ = padD;
        padH_ = padH;
        padW_ = padW;
        blockDim_ = blockDim;

        inputSliceSize_ = H_ * W_;
        outputSliceSize_ = H_out_ * W_out_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputSliceSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outputSliceSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t numSlices = N_ * C_ * D_;
        uint32_t slicesPerBlock = (numSlices + blockDim_ - 1) / blockDim_;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startSlice = blockIdx * slicesPerBlock;
        uint32_t endSlice = (blockIdx + 1) * slicesPerBlock;
        if (endSlice > numSlices) endSlice = numSlices;

        for (uint32_t sliceIdx = startSlice; sliceIdx < endSlice; ++sliceIdx) {
            uint32_t n = sliceIdx / (C_ * D_);
            uint32_t rem = sliceIdx % (C_ * D_);
            uint32_t c = rem / D_;
            uint32_t d = rem % D_;

            uint32_t inputOffset = ((n * C_ + c) * D_ + d) * inputSliceSize_;
            uint32_t outputOffset = ((n * C_ + c) * D_out_ + d) * outputSliceSize_;

            // Load input slice (2D)
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[inputOffset], inputSliceSize_);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            constexpr float NEG_INF = -1e38f;
            for (uint32_t oh = 0; oh < H_out_; ++oh) {
                for (uint32_t ow = 0; ow < W_out_; ++ow) {
                    int32_t ihStart = static_cast<int32_t>(oh * strideH_) - static_cast<int32_t>(padH_);
                    int32_t iwStart = static_cast<int32_t>(ow * strideW_) - static_cast<int32_t>(padW_);
                    float maxVal = NEG_INF;
                    for (uint32_t kh = 0; kh < kernelH_; ++kh) {
                        for (uint32_t kw = 0; kw < kernelW_; ++kw) {
                            int32_t ih = ihStart + static_cast<int32_t>(kh);
                            int32_t iw = iwStart + static_cast<int32_t>(kw);
                            if (ih >= 0 && ih < static_cast<int32_t>(H_) &&
                                iw >= 0 && iw < static_cast<int32_t>(W_)) {
                                float val = xDeque[ih * W_ + iw];
                                if (val > maxVal) maxVal = val;
                            }
                        }
                    }
                    yLocal[oh * W_out_ + ow] = maxVal;
                }
            }

            outQueueY_.EnQue<float>(yLocal);
            inQueueX_.FreeTensor(xDeque);

            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[outputOffset], yOut, outputSliceSize_);
            outQueueY_.FreeTensor(yOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, D_, H_, W_;
    uint32_t D_out_, H_out_, W_out_;
    uint32_t kernelD_, kernelH_, kernelW_;
    uint32_t strideD_, strideH_, strideW_;
    uint32_t padD_, padH_, padW_;
    uint32_t blockDim_;

    uint32_t inputSliceSize_;
    uint32_t outputSliceSize_;
};

extern "C" __global__ __aicore__ void ca6k_00289_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00289Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.D,
            tiling_data.H, tiling_data.W,
            tiling_data.D_out, tiling_data.H_out, tiling_data.W_out,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW,
            tiling_data.blockDim);
    op.Process();
}
