#include "ca6k_00240_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00240CustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // UB buffers: 2 input tiles + 2 temp tiles = 4 * tileLength floats
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    gert::Shape *loss_shape = context->GetOutputShape(0);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelBCEWithLogits {
public:
    __aicore__ inline KernelBCEWithLogits() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR loss,
                                uint32_t totalLength, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        oneDivN_ = 1.0f / static_cast<float>(totalLength);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        float accumSum = 0.0f;
        uint32_t offset = 0;

        while (offset < totalLen_) {
            uint32_t curLen = totalLen_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;

            CopyIn(offset, curLen);
            float tileSum = Compute(curLen);
            accumSum += tileSum;

            offset += tileLen_;
        }

        float meanLoss = accumSum * oneDivN_;
        AscendC::DataCopy(lossGm_, &meanLoss, 1);
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline float Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = inQueueY_.DeQue<float>();

        AscendC::LocalTensor<float> t1 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2 = tmpQueue_.AllocTensor<float>();

        // t1 = log1p(exp(-abs(x)))
        AscendC::Abs(t1, xLocal, len);
        AscendC::Mul(t1, t1, -1.0f, len);
        AscendC::Exp(t1, t1, len);
        AscendC::Log1p(t1, t1, len);

        // t2 = max(x,0) - x*y
        AscendC::Max(t2, xLocal, 0.0f, len);
        AscendC::Mul(xLocal, xLocal, yLocal, len);
        AscendC::Sub(t2, t2, xLocal, len);

        // t2 = t2 + t1
        AscendC::Add(t2, t2, t1, len);

        // reduce sum
        float tileSum = AscendC::ReduceAdd(t2, len);

        tmpQueue_.FreeTensor(t1);
        tmpQueue_.FreeTensor(t2);
        inQueueX_.FreeTensor(xLocal);
        inQueueY_.FreeTensor(yLocal);

        return tileSum;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    float oneDivN_ {0.0f};
};

extern "C" __global__ __aicore__ void ca6k_00240_custom(GM_ADDR x, GM_ADDR y, GM_ADDR loss, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBCEWithLogits op;
    op.Init(x, y, loss, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
