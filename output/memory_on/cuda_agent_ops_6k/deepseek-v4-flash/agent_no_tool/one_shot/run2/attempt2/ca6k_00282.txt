#include "ca6k_00282_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_SIZE = 192 * 1024;
static constexpr uint32_t ALIGN = 8;
static constexpr uint32_t MAX_TILE = 16384;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00282CustomTilingData tiling;

    auto shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = shape.GetDim(0);
    uint32_t C = shape.GetDim(1);
    uint32_t D = shape.GetDim(2);
    uint32_t H = shape.GetDim(3);
    uint32_t W = shape.GetDim(4);
    uint32_t spatialSize = D * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps_ptr = attrs->GetAttrPointer<float>(0);
    float eps_val = (eps_ptr != nullptr) ? *eps_ptr : 1e-5f;

    uint32_t maxTile = (UB_SIZE / sizeof(float)) / 2;
    if (maxTile > MAX_TILE) maxTile = MAX_TILE;
    maxTile = (maxTile / ALIGN) * ALIGN;
    if (maxTile < ALIGN) maxTile = ALIGN;

    uint32_t tileLen = (spatialSize < maxTile) ? ((spatialSize / ALIGN) * ALIGN) : maxTile;
    if (tileLen == 0) tileLen = spatialSize;

    uint32_t blockDim = N * C;
    if (blockDim > 40) blockDim = 40;
    if (blockDim < 8) blockDim = 8;
    context->SetBlockDim(blockDim);

    // Direct assignment to underscored fields (generated by TILING_DATA_FIELD_DEF)
    tiling.N_ = N;
    tiling.C_ = C;
    tiling.spatialSize_ = spatialSize;
    tiling.tileLength_ = tileLen;
    tiling.eps_ = eps_val;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;

    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

extern "C" __global__ __aicore__ void ca6k_00282_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);

    uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
    uint32_t N = td.N_;
    uint32_t C = td.C_;
    uint32_t spatialSize = td.spatialSize_;
    uint32_t tileLen = td.tileLength_;
    float eps = td.eps_;

    if (blockIdx >= N * C) return;

    uint32_t n = blockIdx / C;
    uint32_t c = blockIdx % C;
    uint32_t baseOffset = (n * C + c) * spatialSize;

    float * __gm__ x_ptr = (float * __gm__)x + baseOffset;
    float * __gm__ y_ptr = (float * __gm__)y + baseOffset;

    // First pass: compute sum
    float sum = 0.0f;
    {
        AscendC::TPipe pipe;
        AscendC::TBuf<AscendC::TPosition::VECIN> xBuf;
        pipe.InitBuffer(xBuf, tileLen * sizeof(float));
        uint32_t remaining = spatialSize;
        uint32_t offset = 0;
        while (remaining > 0) {
            uint32_t curTile = (remaining > tileLen) ? tileLen : remaining;
            AscendC::LocalTensor<float> xLocal = xBuf.Get<float>();
            AscendC::DataCopy(xLocal, x_ptr + offset, curTile);
            for (uint32_t i = 0; i < curTile; i++) {
                sum += xLocal(i);
            }
            offset += curTile;
            remaining -= curTile;
        }
    }

    float mean = sum / static_cast<float>(spatialSize);

    // Second pass: compute sum of squared differences
    float sqSum = 0.0f;
    {
        AscendC::TPipe pipe;
        AscendC::TBuf<AscendC::TPosition::VECIN> xBuf;
        AscendC::TBuf<AscendC::TPosition::VECCALC> diffBuf;
        pipe.InitBuffer(xBuf, tileLen * sizeof(float));
        pipe.InitBuffer(diffBuf, tileLen * sizeof(float));
        uint32_t remaining = spatialSize;
        uint32_t offset = 0;
        while (remaining > 0) {
            uint32_t curTile = (remaining > tileLen) ? tileLen : remaining;
            AscendC::LocalTensor<float> xLocal = xBuf.Get<float>();
            AscendC::LocalTensor<float> diffLocal = diffBuf.Get<float>();
            AscendC::DataCopy(xLocal, x_ptr + offset, curTile);
            AscendC::Subs(diffLocal, xLocal, mean, curTile);
            for (uint32_t i = 0; i < curTile; i++) {
                sqSum += diffLocal(i) * diffLocal(i);
            }
            offset += curTile;
            remaining -= curTile;
        }
    }

    float variance = sqSum / static_cast<float>(spatialSize);
    float denom = 1.0f / sqrt(variance + eps);

    // Third pass: output normalized
    {
        AscendC::TPipe pipe;
        AscendC::TBuf<AscendC::TPosition::VECIN> xBuf;
        AscendC::TBuf<AscendC::TPosition::VECOUT> yBuf;
        AscendC::TBuf<AscendC::TPosition::VECCALC> diffBuf;
        pipe.InitBuffer(xBuf, tileLen * sizeof(float));
        pipe.InitBuffer(yBuf, tileLen * sizeof(float));
        pipe.InitBuffer(diffBuf, tileLen * sizeof(float));
        uint32_t remaining = spatialSize;
        uint32_t offset = 0;
        while (remaining > 0) {
            uint32_t curTile = (remaining > tileLen) ? tileLen : remaining;
            AscendC::LocalTensor<float> xLocal = xBuf.Get<float>();
            AscendC::LocalTensor<float> yLocal = yBuf.Get<float>();
            AscendC::LocalTensor<float> diffLocal = diffBuf.Get<float>();
            AscendC::DataCopy(xLocal, x_ptr + offset, curTile);
            AscendC::Subs(diffLocal, xLocal, mean, curTile);
            AscendC::Muls(yLocal, diffLocal, denom, curTile);
            AscendC::DataCopy(y_ptr + offset, yLocal, curTile);
            offset += curTile;
            remaining -= curTile;
        }
    }
}
