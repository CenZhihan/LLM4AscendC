host_operator_src = '''
#include "ca6k00262_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024u;  // 192KB per core
static constexpr uint32_t ALIGN_ELEMS = 2048u;
static constexpr uint32_t HARD_CAP_ELEMS = 32768u;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00262CustomTilingData tiling;

    const auto *xShape = context->GetInputShape(0);
    const auto &origin = xShape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t H = static_cast<uint32_t>(origin.GetDim(1));
    uint32_t W = static_cast<uint32_t>(origin.GetDim(2));
    uint32_t totalLength = N * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t offset = attrs->GetAttrPointer<int32_t>(0)[0];

    // Compute tile length: we need to copy src per batch (H floats) + 2 tiles (x, y).
    // srcLocal = H * sizeof(float) bytes.
    // Budget for two tiles = UB_BUDGET_BYTES - H * sizeof(float)
    uint32_t budgetForTiles = UB_BUDGET_BYTES - H * sizeof(float);
    uint32_t maxTile = budgetForTiles / (2u * sizeof(float));  // elements per tile
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (tileLen > W) tileLen = W;

    // blockDim = number of batches (each block handles one batch)
    uint32_t blockDim = N;
    if (blockDim > 256u) blockDim = 256u;  // safe upper bound

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_offset(offset);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *xShape = context->GetInputShape(0);
    gert::Shape *yShape = context->GetOutputShape(0);
    *yShape = *xShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}

}  // namespace ge

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00262Custom {
public:
    __aicore__ inline KernelCa6k00262Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR src, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t H, uint32_t W,
                                int32_t offset)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N;
        H_ = H;
        W_ = W;
        offset_ = offset;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        srcGm_.SetGlobalBuffer((__gm__ float*)src, N_ * H_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(srcBuf_, 1, H_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= N_) return;

        // Copy src for this batch to local
        AscendC::LocalTensor<float> srcLocal = srcBuf_.AllocTensor<float>();
        AscendC::DataCopy(srcLocal, srcGm_[blockIdx * H_], H_);

        // Process each row of the batch
        for (uint32_t r = 0; r < H_; r++) {
            uint32_t rowOffset = blockIdx * H_ * W_ + r * W_;
            for (uint32_t colStart = 0; colStart < W_; colStart += tileLen_) {
                uint32_t curLen = tileLen_;
                if (colStart + curLen > W_) curLen = W_ - colStart;

                // Load x tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + colStart], curLen);

                // Compute y tile
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                for (uint32_t i = 0; i < curLen; i++) {
                    uint32_t c = colStart + i;
                    if (c == r + offset_) {
                        yLocal(i) = srcLocal(r);
                    } else {
                        yLocal(i) = xLocal(i);
                    }
                }

                // Store y tile
                AscendC::DataCopy(yGm_[rowOffset + colStart], yLocal, curLen);

                inQueueX_.FreeTensor(xLocal);
                outQueueY_.FreeTensor(yLocal);
            }
        }

        srcBuf_.FreeTensor(srcLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> srcBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> srcGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    int32_t offset_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00262_custom(GM_ADDR x, GM_ADDR src, GM_ADDR y,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00262Custom op;
    op.Init(x, src, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.H, tiling_data.W,
            tiling_data.offset);
    op.Process();
}
'''

import torch

_FN = 'ca6k_00262_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, input_shape, src_shape, dim1, dim2, offset):

        self.src = torch.nn.Parameter(torch.randn(src_shape))
        self.dim1 = dim1
        self.dim2 = dim2
        self.offset = offset
