#include "ca6k_00333_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t MIN_TILE_CHANNELS = 64;
static constexpr uint32_t MAX_TILE_CHANNELS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00333CustomTilingData tiling;
    const uint32_t totalLength = static_cast<uint32_t>(
        context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t  size_  = *attrs->GetAttrPointer<int32_t>(0);
    const float    alpha_ = *attrs->GetAttrPointer<float>(1);
    const float    beta_  = *attrs->GetAttrPointer<float>(2);
    const float    k_     = *attrs->GetAttrPointer<float>(3);

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    int32_t ndim = shape.GetDimNum();
    uint32_t N = (ndim >= 1) ? shape.GetDim(0) : 1;
    uint32_t C = (ndim >= 2) ? shape.GetDim(1) : 1;
    uint32_t H = (ndim >= 3) ? shape.GetDim(2) : 1;
    uint32_t W = (ndim >= 4) ? shape.GetDim(3) : 1;

    uint32_t rows = N * H * W;
    uint32_t channels = C;
    uint32_t radius = static_cast<uint32_t>(size_ / 2);

    uint32_t maxExtended = static_cast<uint32_t>(0.8 * UB_BUDGET_BYTES / (6 * 4));
    uint32_t tileChannels = (maxExtended > 2 * radius) ? (maxExtended - 2 * radius) : MIN_TILE_CHANNELS;
    if (tileChannels > channels) tileChannels = channels;
    if (tileChannels > MAX_TILE_CHANNELS) tileChannels = MAX_TILE_CHANNELS;
    tileChannels = AlignDown(tileChannels, ALIGN_ELEMS);
    if (tileChannels < ALIGN_ELEMS) tileChannels = ALIGN_ELEMS;

    uint32_t tileRows = 1;

    uint32_t numTilesRows = CeilDiv(rows, tileRows);
    uint32_t blockDim = numTilesRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // Direct assignment instead of set_* methods
    tiling.totalLength = totalLength;
    tiling.rows = rows;
    tiling.channels = channels;
    tiling.tileRows = tileRows;
    tiling.tileChannels = tileChannels;
    tiling.channelSliceStart = 0;
    tiling.channelSliceSize = tileChannels;
    tiling.extendedSliceSize = tileChannels + 2 * radius;
    tiling.radius = radius;
    tiling.size = size_;
    tiling.alpha = alpha_;
    tiling.beta = beta_;
    tiling.k = k_;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00333Custom {
public:
    __aicore__ inline KernelCa6k00333Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t rows,
                                uint32_t channels,
                                uint32_t tileRows,
                                uint32_t tileChannels,
                                uint32_t channelSliceSize,
                                uint32_t extendedSliceSize,
                                int32_t radius,
                                int32_t size,
                                float alpha,
                                float beta,
                                float k)
    {
        totalLen_ = totalLength;
        rows_ = rows;
        channels_ = channels;
        tileRows_ = tileRows;
        tileChannels_ = tileChannels;
        channelSliceSize_ = channelSliceSize;
        extendedSliceSize_ = extendedSliceSize;
        radius_ = radius;
        size_ = size;
        alpha_ = alpha;
        beta_ = beta;
        k_ = k;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        uint32_t maxExtended = extendedSliceSize_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, maxExtended * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, maxExtended * sizeof(float));
        pipe_.InitBuffer(sqQueue_, 1, maxExtended * sizeof(float));
        pipe_.InitBuffer(sumQueue_, 1, maxExtended * sizeof(float));
        pipe_.InitBuffer(denomQueue_, 1, maxExtended * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 4, maxExtended * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t rowStride = blockNum;

        for (uint32_t r = blockIdx * tileRows_; r < rows_; r += rowStride * tileRows_) {
            for (uint32_t rr = 0; rr < tileRows_; ++rr) {
                uint32_t rowIdx = r + rr;
                if (rowIdx >= rows_) break;

                uint32_t channelStart = 0;
                while (channelStart < channels_) {
                    uint32_t sliceSize = (channelStart + channelSliceSize_ <= channels_) ?
                                         channelSliceSize_ : channels_ - channelStart;
                    uint32_t extStart = (channelStart >= radius_) ? channelStart - radius_ : 0;
                    uint32_t extEnd = (channelStart + sliceSize + radius_ <= channels_) ?
                                      channelStart + sliceSize + radius_ : channels_;
                    uint32_t extSize = extEnd - extStart;

                    if (extSize == 0) break;

                    uint32_t gmOffset = rowIdx * channels_ + extStart;
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[gmOffset], extSize);

                    AscendC::LocalTensor<float> sqLocal = sqQueue_.AllocTensor<float>();
                    AscendC::Mul(sqLocal, xLocal, xLocal, extSize);

                    AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
                    AscendC::SetZero(sumLocal, extSize);

                    // Manual sliding window sum using vector shifts (simplified)
                    for (int32_t d = -radius_; d <= radius_; ++d) {
                        AscendC::LocalTensor<float> shifted = tmpQueue_.AllocTensor<float>();
                        AscendC::SetZero(shifted, extSize);
                        if (d >= 0) {
                            uint32_t copyLen = (extSize > static_cast<uint32_t>(d)) ? extSize - static_cast<uint32_t>(d) : 0;
                            if (copyLen > 0) {
                                AscendC::DataCopy(shifted[d], sqLocal[0], copyLen);
                            }
                        } else {
                            uint32_t startIdx = static_cast<uint32_t>(-d);
                            uint32_t copyLen = (extSize > startIdx) ? extSize - startIdx : 0;
                            if (copyLen > 0) {
                                AscendC::DataCopy(shifted[0], sqLocal[startIdx], copyLen);
                            }
                        }
                        AscendC::Add(sumLocal, sumLocal, shifted, extSize);
                        tmpQueue_.FreeTensor(shifted);
                    }

                    AscendC::LocalTensor<float> denomLocal = denomQueue_.AllocTensor<float>();
                    AscendC::Muls(denomLocal, sumLocal, alpha_, extSize);
                    AscendC::Adds(denomLocal, denomLocal, k_, extSize);

                    AscendC::LocalTensor<float> powLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::LocalTensor<float> betaLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::BroadCast(betaLocal, beta_, extSize);
                    AscendC::Pow(powLocal, denomLocal, betaLocal, extSize);

                    AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                    AscendC::Div(yLocal, xLocal, powLocal, extSize);

                    uint32_t innerOffset = channelStart - extStart;
                    AscendC::DataCopy(yGm_[gmOffset + innerOffset], yLocal[innerOffset], sliceSize);

                    denomQueue_.FreeTensor(denomLocal);
                    tmpQueue_.FreeTensor(powLocal);
                    tmpQueue_.FreeTensor(betaLocal);
                    sqQueue_.FreeTensor(sqLocal);
                    sumQueue_.FreeTensor(sumLocal);
                    outQueueY_.FreeTensor(yLocal);
                    inQueueX_.FreeTensor(xLocal);

                    channelStart += sliceSize;
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sqQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> denomQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rows_ {0};
    uint32_t channels_ {0};
    uint32_t tileRows_ {1};
    uint32_t tileChannels_ {0};
    uint32_t channelSliceSize_ {0};
    uint32_t extendedSliceSize_ {0};
    int32_t radius_ {0};
    int32_t size_ {5};
    float alpha_ {1e-4f};
    float beta_ {0.75f};
    float k_ {1.0f};
};

extern "C" __global__ __aicore__ void ca6k_00333_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00333Custom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.rows,
            tiling_data.channels,
            tiling_data.tileRows,
            tiling_data.tileChannels,
            tiling_data.channelSliceSize,
            tiling_data.extendedSliceSize,
            tiling_data.radius,
            tiling_data.size,
            tiling_data.alpha,
            tiling_data.beta,
            tiling_data.k);
    op.Process();
}
