#include "ca6k_00301_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00301CustomTilingData tiling;

    auto xShape = context->GetInputShape(0)->GetOriginShape();
    auto yShape = context->GetInputShape(1)->GetOriginShape();
    // x: [B, M, K], y: [B, K, N]
    uint32_t B = static_cast<uint32_t>(xShape.GetDim(0));
    uint32_t M = static_cast<uint32_t>(xShape.GetDim(1));
    uint32_t K = static_cast<uint32_t>(xShape.GetDim(2));
    uint32_t N = static_cast<uint32_t>(yShape.GetDim(2));

    // tile sizes from UB budget: 3 tiles + workspace = ~ 4*tileM*tileK + 4*tileK*tileN + 4*tileM*tileN
    // budget 192KB => ~49152 floats, choose 64 as safe tile dimension
    constexpr uint32_t MAX_TILE = 64;

    uint32_t tileM = (M < MAX_TILE) ? M : MAX_TILE;
    uint32_t tileN = (N < MAX_TILE) ? N : MAX_TILE;
    uint32_t tileK = (K < MAX_TILE) ? K : MAX_TILE;

    // each batch handled by one block
    uint32_t blockDim = B;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.totalBatch_ = B;
    tiling.M_ = M;
    tiling.N_ = N;
    tiling.K_ = K;
    tiling.tileM_ = tileM;
    tiling.tileN_ = tileN;
    tiling.tileK_ = tileK;
    tiling.blockDim_ = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *y_shape = context->GetInputShape(1);
    gert::Shape *z_shape = context->GetOutputShape(0);

kernel_src="""

#include "kernel_operator.h"
#include "matmul.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00301Custom {
public:
    __aicore__ inline KernelCa6k00301Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t totalBatch, uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        totalBatch_ = totalBatch;
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        // set global base pointers (our batch offset is computed per call)
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        zGm_.SetGlobalBuffer((__gm__ float*)z);

        // allocate local buffers for a tile
        // A tile: tileM x tileK
        // B tile: tileK x tileN
        // C tile: tileM x tileN
        uint32_t aSize = tileM_ * tileK_ * sizeof(float);
        uint32_t bSize = tileK_ * tileN_ * sizeof(float);
        uint32_t cSize = tileM_ * tileN_ * sizeof(float);
        // double buffer for A and B to overlap load/compute
        pipe_.InitBuffer(inQueA_, BUFFER_NUM, aSize);
        pipe_.InitBuffer(inQueB_, BUFFER_NUM, bSize);
        pipe_.InitBuffer(outQueC_, BUFFER_NUM, cSize);
        // workspace for MatMul (internal)
        pipe_.InitBuffer(workspace_, 1, aSize + bSize + cSize); // rough estimate
    }

    __aicore__ inline void Process()
    {
        if (totalBatch_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        if (blockIdx >= totalBatch_) return;

        // compute batch offsets
        uint32_t batchOffsetA = blockIdx * M_ * K_;
        uint32_t batchOffsetB = blockIdx * K_ * N_;
        uint32_t batchOffsetC = blockIdx * M_ * N_;

        // loop over tiles
        for (uint32_t mi = 0; mi < M_; mi += tileM_) {
            uint32_t curM = (mi + tileM_ <= M_) ? tileM_ : (M_ - mi);
            for (uint32_t nj = 0; nj < N_; nj += tileN_) {
                uint32_t curN = (nj + tileN_ <= N_) ? tileN_ : (N_ - nj);
                // initialize C tile to zero
                LocalTensor<float> cTile = outQueC_.AllocTensor<float>();
                if (cTile.GetSize() < curM * curN * sizeof(float)) {
                    // should not happen
                    outQueC_.FreeTensor(cTile);
                    continue;
                }
                Duplicate<float>(cTile, 0.0f, curM * curN);

                for (uint32_t ki = 0; ki < K_; ki += tileK_) {
                    uint32_t curK = (ki + tileK_ <= K_) ? tileK_ : (K_ - ki);

                    // load A tile (curM x curK) from global
                    LocalTensor<float> aTile = inQueA_.AllocTensor<float>();
                    LocalTensor<float> bTile = inQueB_.AllocTensor<float>();

                    // copy A from (mi, ki) region
                    uint32_t aOffset = batchOffsetA + mi * K_ + ki;
                    // The global tensor is 3D, but we access linearized: index = mi*K_ + ki
                    // Actually we need to copy a rectangular region. Use DataCopy with stride.
                    // AscendC has DataCopyRect for 2D copy. We'll use vectorized copy in a loop.
                    // For simplicity, we assume tile sizes are powers of 2 and use DataCopy for each row.
                    for (uint32_t r = 0; r < curM; r++) {
                        uint32_t rowOff = aOffset + r * K_;
                        DataCopy(aTile[r * curK], xGm_[rowOff], curK);
                    }
                    // copy B tile (curK x curN) from (ki, nj)
                    uint32_t bOffset = batchOffsetB + ki * N_ + nj;
                    for (uint32_t r = 0; r < curK; r++) {
                        uint32_t rowOff = bOffset + r * N_;
                        DataCopy(bTile[r * curN], yGm_[rowOff], curN);
                    }

                    // compute C += A * B using MatMul
                    MatMul<MatMulType<ND, ND, ND, float>, CubeFormat::ND, false, false> mm;
                    mm.SetOrgShape(curM, curN, curK);
                    // set tensors
                    mm.SetTensorA(aTile);
                    mm.SetTensorB(bTile);
                    // use cTile as accumulator
                    LocalTensor<float> tmpC = workspace_.AllocTensor<float>();
                    Duplicate<float>(tmpC, 0.0f, curM * curN);
                    mm.SetBias(tmpC, false); // bias not used
                    mm.Calc(cTile, aTile, bTile);
                    // cTile now contains A*B; but we need to accumulate previous cTile?
                    // Actually mm.Calc writes to cTile. We need to add to previous content.
                    // We can use Add after calc. For simplicity, we redo inside loop as:
                    //   cTile = cTile + A*B
                    // We'll compute a new product and add.
                    LocalTensor<float> prod = workspace_.AllocTensor<float>();
                    Duplicate<float>(prod, 0.0f, curM * curN);
                    mm.Calc(prod, aTile, bTile);
                    Add(cTile, cTile, prod, curM * curN);
                    workspace_.FreeTensor(prod);
                    workspace_.FreeTensor(tmpC);

                    inQueB_.FreeTensor(bTile);
                    inQueA_.FreeTensor(aTile);
                }

                // write C tile to global
                uint32_t cOffset = batchOffsetC + mi * N_ + nj;
                for (uint32_t r = 0; r < curM; r++) {
                    uint32_t rowOff = cOffset + r * N_;
                    DataCopy(zGm_[rowOff], cTile[r * curN], curN);
                }
                outQueC_.FreeTensor(cTile);
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueA_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueB_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueC_;
    TQue<TPosition::VECCALC, 1> workspace_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> zGm_;

    uint32_t totalBatch_ {0};
    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
    uint32_t tileK_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00301_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00301Custom op;
    op.Init(x, y, z,
            tiling_data.totalBatch_, tiling_data.M_, tiling_data.N_, tiling_data.K_,
            tiling_data.tileM_, tiling_data.tileN_, tiling_data.tileK_);
    op.Process();
}
