#include "ca6k_00330_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t BLOCK_DIM_MAX = 24;
static constexpr uint32_t BLOCK_DIM_MIN = 1;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k00330CustomTilingData tiling;

  // Read dimensions directly from input shapes
  uint32_t B = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(0));
  uint32_t M = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));
  uint32_t N = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(2));
  uint32_t K = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetDim(2));

  uint32_t totalLength = B * M * N;

  // Compute blockDim based on totalLength; keep moderate.
  uint32_t blockDim = (totalLength + 1024*1024 - 1) / (1024*1024);
  if (blockDim < BLOCK_DIM_MIN) blockDim = BLOCK_DIM_MIN;
  if (blockDim > BLOCK_DIM_MAX) blockDim = BLOCK_DIM_MAX;

  context->SetBlockDim(blockDim);

  // Use setter methods as required by TILING_DATA_FIELD_DEF
  tiling.set_batch(B);
  tiling.set_M(M);
  tiling.set_N(N);
  tiling.set_K(K);
  tiling.set_totalLength(totalLength);
  tiling.set_tileLength(1); // not used in vector loop
  tiling.set_blockDim(blockDim);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape *bias_shape = context->GetInputShape(0);
  gert::Shape *y_shape = context->GetOutputShape(0);
  *y_shape = *bias_shape;
  return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
  const ge::DataType data_type = context->GetInputDataType(0);
  context->SetOutputDataType(0, data_type);
  return GRAPH_SUCCESS;
}

}  // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_K = 1024;

class KernelCa6k00330Custom {
 public:
  __aicore__ inline KernelCa6k00330Custom() {}

  __aicore__ inline void Init(GM_ADDR bias, GM_ADDR mat1, GM_ADDR mat2,
                              GM_ADDR y, uint32_t batch, uint32_t M,
                              uint32_t N, uint32_t K, uint32_t totalLength,
                              uint32_t tileLength, uint32_t blockDim) {
    batch_ = batch;
    M_ = M;
    N_ = N;
    K_ = K;
    totalLen_ = totalLength;
    blockDim_ = blockDim;

    biasGm_.SetGlobalBuffer((__gm__ float *)bias, batch_ * M_ * N_);
    mat1Gm_.SetGlobalBuffer((__gm__ float *)mat1, batch_ * M_ * K_);
    mat2Gm_.SetGlobalBuffer((__gm__ float *)mat2, batch_ * K_ * N_);
    yGm_.SetGlobalBuffer((__gm__ float *)y, batch_ * M_ * N_);

    // Allocate local buffers
    pipe_.InitBuffer(biasBuf_, BUFFER_NUM, M_ * N_ * sizeof(float));
    pipe_.InitBuffer(accBuf_, 1, N_ * sizeof(float));
    pipe_.InitBuffer(mat2Buf_, 1, TILE_K * N_ * sizeof(float));
    pipe_.InitBuffer(vecTmpBuf_, 1, N_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0) return;

    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t blockNum = blockDim_;

    for (uint32_t b = blockIdx; b < batch_; b += blockNum) {
      // Load bias for this batch
      AscendC::LocalTensor<float> biasLcl = biasBuf_.AllocTensor<float>();
      AscendC::DataCopy(biasLcl, biasGm_[b * M_ * N_], M_ * N_);

      // Process each row i
      for (uint32_t i = 0; i < M_; ++i) {
        // Start with bias[b, i, :]
        AscendC::LocalTensor<float> accLcl = accBuf_.AllocTensor<float>();
        AscendC::DataCopy(accLcl, biasLcl[i * N_], N_);

        // Accumulate over k tiles
        for (uint32_t kStart = 0; kStart < K_; kStart += TILE_K) {
          uint32_t curTileK = (K_ - kStart) < TILE_K ? (K_ - kStart) : TILE_K;

          // Load mat1[b, i, kStart:kStart+curTileK] as scalar vector
          AscendC::LocalTensor<float> mat1RowLcl = vecTmpBuf_.AllocTensor<float>();
          AscendC::DataCopy(mat1RowLcl,
                            mat1Gm_[b * M_ * K_ + i * K_ + kStart],
                            curTileK);

          // Load mat2[b, kStart:kStart+curTileK, :]   (size curTileK * N N N
