#include "ca6k_00257_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;            // smaller alignment for grid compute
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00257CustomTilingData tiling;
    const gert::Shape *theta_shape = context->GetInputShape(0);
    const uint32_t N = static_cast<uint32_t>(theta_shape->GetOriginShape().GetDim(0));
    const uint32_t H = static_cast<uint32_t>(context->GetAttrs()->GetAttr(1)->GetInt());  // H attr index 1?
    const uint32_t W = static_cast<uint32_t>(context->GetAttrs()->GetAttr(2)->GetInt());
    // Actually attrs order: 0: N, 1: H, 2: W, 3: scale, 4: align_corners
    // Use GeAttrParse in actual code; for simplicity use fixed known values
    // (they are passed as optional attrs)
    // We'll read them from context:
    const uint32_t N_attr = static_cast<uint32_t>(context->GetAttrs()->GetAttr(0)->GetInt());
    const uint32_t H_attr = static_cast<uint32_t>(context->GetAttrs()->GetAttr(1)->GetInt());
    const uint32_t W_attr = static_cast<uint32_t>(context->GetAttrs()->GetAttr(2)->GetInt());
    const float scale = context->GetAttrs()->GetAttr(3)->GetFloat();
    const bool align_corners = context->GetAttrs()->GetAttr(4)->GetBool();

    const uint32_t totalLength = N_attr * H_attr * W_attr * 2; // output elements

    // tileLen from UB budget: we need to compute grid coordinates for one tile
    // Each element computation uses several temps; conservatively assume 8 floats per output element
    uint32_t maxTile = UB_BUDGET_BYTES / (8 * sizeof(float)); // elements of output
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 4u) blockDim = 4u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N_attr);
    tiling.set_H(H_attr);
    tiling.set_W(W_attr);
    tiling.set_scale(scale);
    tiling.set_alignCornerFlag(align_corners ? 1 : 0);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00257Custom {
public:
    __aicore__ inline KernelCa6k00257Custom() {}

    __aicore__ inline void Init(GM_ADDR theta, GM_ADDR grid_scaled,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t H, uint32_t W,
                                float scale, int32_t alignCornerFlag)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N; H_ = H; W_ = W;
        scale_ = scale;
        alignCornerFlag_ = alignCornerFlag;

        thetaGm_.SetGlobalBuffer((__gm__ float*)theta, N_ * 6);
        gridGm_.SetGlobalBuffer((__gm__ float*)grid_scaled, totalLen_);

        // Temporary buffers: each tile processes tileLen output elements (each is 2 floats)
        // For each output element, we need to compute grid coordinates using theta matrix.
        // We will read theta and compute locally.
        pipe_.InitBuffer(inQueueTheta_, BUFFER_NUM, 6 * sizeof(float)); // one theta matrix per batch
        pipe_.InitBuffer(outQueueGrid_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Prefetch first tile
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        ComputeAndCopyOut(base, curLen);

        // Grid-stride loop to process remaining tiles
        while (true) {
            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalLen_) break;
            uint32_t nextLen = totalLen_ - nextBase;
            if (nextLen > tileLen_) nextLen = tileLen_;
            ComputeAndCopyOut(nextBase, nextLen);
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void ComputeAndCopyOut(uint32_t gmOffset, uint32_t len)
    {
        // gmOffset is linear index into output grid (each element is 2 floats)
        // Compute which batch, row, col this tile covers
        // We'll process each output element sequentially
        // For simplicity, we do one element at a time on scalar? Better to vectorize but for correctness we use scalar loop.
        // Use local tensor to accumulate results, then copy out.
        AscendC::LocalTensor<float> gridLocal = outQueueGrid_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Prefetch theta matrix for the relevant batch(es) - we do it on demand
        // Using series of inline computation
        for (uint32_t i = 0; i < len; i++) {
            uint32_t globalIdx = gmOffset + i; // element index in output
            // Determine batch, h, w
            uint32_t hw = globalIdx / 2;       // index within batch: globalIdx/2 -> batchIndex * H*W + ...
            uint32_t batchIdx = hw / (H_ * W_);
            uint32_t hwRem = hw % (H_ * W_);
            uint32_t h = hwRem / W_;
            uint32_t w = hwRem % W_;

            // Compute normalized coordinates in source grid
            float x = 0.0f;
            float y = 0.0f;
            if (alignCornerFlag_) {
                // align_corners: grid points at corners, range [-1, 1]
                if (W_ > 1) x = (2.0f * w) / (W_ - 1.0f) - 1.0f;
                if (H_ > 1) y = (2.0f * h) / (H_ - 1.0f) - 1.0f;
            } else {
                // not align_corners: grid points at pixel centers, range [-1+1/W, 1-1/W]
                x = (2.0f * w + 1.0f) / W_ - 1.0f;
                y = (2.0f * h + 1.0f) / H_ - 1.0f;
            }

            // Apply affine transformation: grid = A * [x, y, 1]^T
            // theta is (2,3) matrix stored row-major: [a,b,c; d,e,f]
            // grid_x = a*x + b*y + c
            // grid_y = d*x + e*y + f
            // fetch theta for this batch (6 floats)
            float theta[6];
            AscendC::DataCopy(gridLocal, thetaGm_[batchIdx * 6], 6); // copy 6 elements into gridLocal
            // Actually we need to read theta into a local tensor; use temporary
            AscendC::DataCopy(tLocal, thetaGm_[batchIdx * 6], 6);
            float a = tLocal.GetValue(0);
            float b = tLocal.GetValue(1);
            float c = tLocal.GetValue(2);
            float d = tLocal.GetValue(3);
            float e = tLocal.GetValue(4);
            float f = tLocal.GetValue(5);

            float gridX = a * x + b * y + c;
            float gridY = d * x + e * y + f;

            // Multiply by scale
            gridX *= scale_;
            gridY *= scale_;

            // Write to output
            gridLocal.SetValue(i * 2 + 0, gridX);
            gridLocal.SetValue(i * 2 + 1, gridY);
        }

        // Copy gridLocal to global output
        AscendC::DataCopy(gridGm_[gmOffset], gridLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        outQueueGrid_.FreeTensor(gridLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTheta_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueGrid_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> thetaGm_;
    AscendC::GlobalTensor<float> gridGm_;

    uint32_t totalLen_ = 0;
    uint32_t tileLen_ = 0;
    uint32_t N_ = 0;
    uint32_t H_ = 0;
    uint32_t W_ = 0;
    float scale_ = 0.5f;
    int32_t alignCornerFlag_ = 1;
};

extern "C" __global__ __aicore__ void ca6k_00257_custom(GM_ADDR theta, GM_ADDR grid_scaled, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00257Custom op;
    op.Init(theta, grid_scaled, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.H, tiling_data.W,
            tiling_data.scale, tiling_data.alignCornerFlag);
    op.Process();
}
'''
