namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 2048;
static constexpr uint32_t HARD_CAP = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00253CustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    tiling.set_N(static_cast<uint32_t>(x_shape->GetShapeSize()));
    tiling.set_inC(static_cast<uint32_t>(x_shape->GetShapeSize()));
    tiling.set_iH(static_cast<uint32_t>(x_shape->GetShapeSize()));
    tiling.set_iW(static_cast<uint32_t>(x_shape->GetShapeSize()));

    // Actually need per-dimension sizes; GetShapeSize() returns total elements.
    // Use GetDim(i) instead.
    // For simplicity, we assume shape is ND (4D) and use GetDim.
    // In real code, use GetDim(i).
    // But the error showed GetShapeSize expects 0 arguments, so we must use GetDim.
    // We'll redefine this section using GetDim.

    // Let's redo:
    // tiling.set_N(x_shape->GetDim(0));
    // etc.
    // But the above uses GetShapeSize which returns total size. We need per dimension.
    // Correct approach:
    tiling.set_N(x_shape->GetDim(0));
    tiling.set_inC(x_shape->GetDim(1));
    tiling.set_iH(x_shape->GetDim(2));
    tiling.set_iW(x_shape->GetDim(3));

    tiling.set_outC(w_shape->GetDim(0));
    tiling.set_convK(w_shape->GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    tiling.set_convS(static_cast<uint32_t>(attrs->GetAttrPointer<int>(3)[0]));
    tiling.set_poolK(static_cast<uint32_t>(attrs->GetAttrPointer<int>(4)[0]));
    tiling.set_poolS(static_cast<uint32_t>(attrs->GetAttrPointer<int>(5)[0]));

    uint32_t conv_oH = (tiling.get_iH() - tiling.get_convK()) / tiling.get_convS() + 1;
    uint32_t conv_oW = (tiling.get_iW() - tiling.get_convK()) / tiling.get_convS() + 1;
    tiling.set_oH(conv_oH / tiling.get_poolS());
    tiling.set_oW(conv_oW / tiling.get_poolS());

    uint32_t totalElems = tiling.get_N() * tiling.get_outC() * tiling.get_oH() * tiling.get_oW();

    // Simple tiling on total elements (output)
    uint32_t maxTile = UB_BUDGET_BYTES / (4);  // per element float
    maxTile = AlignDown(maxTile, ALIGN);
    if (maxTile < ALIGN) maxTile = ALIGN;
    if (maxTile > HARD_CAP) maxTile = AlignDown(HARD_CAP, ALIGN);

    uint32_t tileLen = (totalElems < maxTile) ? totalElems : maxTile;
    tiling.set_tileLen(tileLen);

    uint32_t numTiles = CeilDiv(totalElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);
    tiling.set_blockDim(blockDim);
    tiling.set_totalElements(totalElems);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00253Custom {
public:
    __aicore__ inline KernelCa6k00253Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t N, uint32_t inC, uint32_t outC,
                                uint32_t iH, uint32_t iW,
                                uint32_t convK, uint32_t convS,
                                uint32_t poolK, uint32_t poolS,
                                uint32_t oH, uint32_t oW,
                                uint32_t totalElements, uint32_t tileLen)
    {
        N_ = N; inC_ = inC; outC_ = outC;
        iH_ = iH; iW_ = iW;
        convK_ = convK; convS_ = convS;
        poolK_ = poolK; poolS_ = poolS;
        oH_ = oH; oW_ = oW;
        totalElems_ = totalElements;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * inC_ * iH_ * iW_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outC_ * inC_ * convK_ * convK_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, outC_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElems_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalElems_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stride = blockNum * tileLen_;
        uint32_t start = blockIdx * tileLen_;
        if (start >= totalElems_) return;

        for (uint32_t offset = start; offset < totalElems_; offset += stride) {
            uint32_t curLen = totalElems_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;
            ProcessTile(offset, curLen);
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t startIdx, uint32_t len) {
        for (uint32_t idx = startIdx; idx < startIdx + len; ++idx) {
            uint32_t ow = idx % oW_;
            uint32_t oh = (idx / oW_) % oH_;
            uint32_t oc = (idx / (oW_ * oH_)) % outC_;
            uint32_t b = idx / (outC_ * oH_ * oW_);

            uint32_t xBase = b * inC_ * iH_ * iW_;

            float sum = 0.0f;
            for (uint32_t ic = 0; ic < inC_; ++ic) {
                float best = -1e30f;
                for (uint32_t kh = 0; kh < convK_; ++kh) {
                    for (uint32_t kw = 0; kw < convK_; ++kw) {
                        for (uint32_t ph = 0; ph < poolK_; ++ph) {
                            for (uint32_t pw = 0; pw < poolK_; ++pw) {
                                uint32_t in_h = oh * poolS_ + ph + kh;
                                uint32_t in_w = ow * poolS_ + pw + kw;
                                uint32_t xIdx = xBase + ic * iH_ * iW_ + in_h * iW_ + in_w;
                                float xVal = xGm_.GetValue(xIdx);
                                float wVal = wGm_.GetValue(oc * inC_ * convK_ * convK_ + ic * convK_ * convK_ + kh * convK_ + kw);
                                float convVal = xVal * wVal + bGm_.GetValue(oc);
                                if (convVal < 0.0f) convVal = 0.0f;
                                if (convVal > best) best = convVal;
                            }
                        }
                    }
                }
                sum += best;
            }
            yGm_.SetValue(idx, sum);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t inC_ {0};
    uint32_t outC_ {0};
    uint32_t iH_ {0};
    uint32_t iW_ {0};
    uint32_t convK_ {0};
    uint32_t convS_ {0};
    uint32_t poolK_ {0};
    uint32_t poolS_ {0};
    uint32_t oH_ {0};
    uint32_t oW_ {0};
    uint32_t totalElems_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00253_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00253Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.inC, tiling_data.outC,
            tiling_data.iH, tiling_data.iW,
            tiling_data.convK, tiling_data.convS,
            tiling_data.poolK, tiling_data.poolS,
            tiling_data.oH, tiling_data.oW,
            tiling_data.totalElements, tiling_data.tileLen);
    op.Process();
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00253_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, container):
        super().__init__()
        self.weight = None
        self.bias = None
        if isinstance(container, torch.nn.Sequential):
            conv = container[0]
            self.weight = conv.weight.detach().clone().contiguous()
            self.bias = conv.bias.detach().clone().contiguous() if conv.bias is not None else None
            if self.bias is None:
                self.bias = torch.zeros(self.weight.size(0), dtype=self.weight.dtype, device=self.weight.device)
        else:
            raise ValueError("Container must be nn.Sequential")

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias)
