kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00334Custom {
public:
    __aicore__ inline KernelCa6k00334Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowSize,
                                uint32_t blockDim)
    {
        totalRows_ = totalRows;
        rowSize_ = rowSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * rowSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * rowSize_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, rowSize_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, rowSize_ * sizeof(float));
        pipe_.InitBuffer(prevQueue_, 1, rowSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, rowSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowSize_ == 0) return;

        // Process first row: copy x to y (identity for cumprod)
        {
            AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[0], rowSize_);
            AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, xLocal, rowSize_);
            // Store as previous output for next row
            AscendC::LocalTensor<float> prevLocal = prevQueue_.AllocTensor<float>();
            AscendC::DataCopy(prevLocal, yLocal, rowSize_);
            prevQueue_.EnQue(prevLocal);
            // Write output row 0 to global
            AscendC::DataCopy(yGm_[0], yLocal, rowSize_);
            outQueue_.FreeTensor(yLocal);
            inQueue_.FreeTensor(xLocal);
        }

        // Process remaining rows
        for (uint32_t row = 1; row < totalRows_; row++)
        {
            // Load current input row
            AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[row * rowSize_], rowSize_);

            // Get previous output row
            AscendC::LocalTensor<float> prevLocal = prevQueue_.DeQue<float>();

            // Compute: current output = previous output * current input (element-wise)
            AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
            AscendC::Mul(yLocal, prevLocal, xLocal, rowSize_);

            // Update prevQueue with this output for next iteration
            AscendC::LocalTensor<float> nextPrev = prevQueue_.AllocTensor<float>();
            AscendC::DataCopy(nextPrev, yLocal, rowSize_);
            prevQueue_.EnQue(nextPrev);

            // Write output row to global
            AscendC::DataCopy(yGm_[row * rowSize_], yLocal, rowSize_);

            // Free tensors
            outQueue_.FreeTensor(yLocal);
            inQueue_.FreeTensor(xLocal);
            prevQueue_.FreeTensor(prevLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> prevQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t rowSize_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00334_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00334Custom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.rowSize, tiling_data.blockDim);
    op.Process();
}
