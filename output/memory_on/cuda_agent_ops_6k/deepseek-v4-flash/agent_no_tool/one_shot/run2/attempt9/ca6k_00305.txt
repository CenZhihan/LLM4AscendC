#include "ca6k00305_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00305CustomTilingData tiling;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float p = 2.0f;
    uint32_t kernelSize = 3;
    uint32_t stride = 2;

    if (attrs) {
        const float *p_ptr = attrs->GetAttrPointer<float>(0);
        if (p_ptr) p = *p_ptr;
        const int64_t *k_ptr = attrs->GetAttrPointer<int64_t>(1);
        if (k_ptr) kernelSize = static_cast<uint32_t>(*k_ptr);
        const int64_t *s_ptr = attrs->GetAttrPointer<int64_t>(2);
        if (s_ptr) stride = static_cast<uint32_t>(*s_ptr);
    }

    const gert::StorageShape *storage_shape = context->GetInputShape(0);
    const gert::Shape &in_shape = storage_shape->GetOriginShape();
    const uint32_t in_len = static_cast<uint32_t>(in_shape.GetDim(in_shape.GetDimNum() - 1));
    const uint32_t batch_channels = static_cast<uint32_t>(in_shape.GetShapeSize()) / in_len;
    uint32_t out_len = (in_len >= kernelSize) ? ((in_len - kernelSize) / stride + 1) : 0;
    const uint32_t totalOutLen = batch_channels * out_len;
    const uint32_t totalInLen = in_shape.GetShapeSize();

    uint32_t maxTile = UB_BUDGET_BYTES / (4 * (stride + 1));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileOutLen = maxTile;
    if (totalOutLen < tileOutLen) {
        uint32_t aligned = AlignDown(totalOutLen, ALIGN_ELEMS);
        tileOutLen = (aligned > 0) ? aligned : totalOutLen;
    }
    if (tileOutLen == 0) tileOutLen = 1;

    uint32_t blockDim = CeilDiv(totalOutLen, tileOutLen);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutLen(totalOutLen);
    tiling.set_tileOutLen(tileOutLen);
    tiling.set_totalInLen(totalInLen);
    tiling.set_blockDim(blockDim);
    tiling.set_p(p);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *in_shape = context->GetInputShape(0);
    gert::Shape *out_shape = context->GetOutputShape(0);

    uint32_t dim_num = in_shape->GetDimNum();
    out_shape->SetDimNum(dim_num);
    for (uint32_t i = 0; i < dim_num - 1; ++i) {
        out_shape->SetDim(i, in_shape->GetDim(i));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t kernelSize = 3;
    int64_t stride = 2;
    if (attrs) {
        const int64_t *k_ptr = attrs->GetAttrPointer<int64_t>(1);
        if (k_ptr) kernelSize = *k_ptr;
        const int64_t *s_ptr = attrs->GetAttrPointer<int64_t>(2);
        if (s_ptr) stride = *s_ptr;
    }
    int64_t in_len = in_shape->GetDim(dim_num - 1);
    int64_t out_len = (in_len - kernelSize) / stride + 1;
    if (out_len < 0) out_len = 0;
    out_shape->SetDim(dim_num - 1, out_len);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00305Custom {
public:
    __aicore__ inline KernelCa6k00305Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutLen, uint32_t tileOutLen,
                                uint32_t totalInLen,
                                float p, uint32_t kernelSize, uint32_t stride)
    {
        totalOutLen_ = totalOutLen;
        tileOutLen_ = tileOutLen;
        totalInLen_ = totalInLen;
        p_ = p;
        kernelSize_ = kernelSize;
        stride_ = stride;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);

        inLenMax_ = ((tileOutLen_ - 1) * stride_ + kernelSize_ + 31) / 32 * 32;
        if (inLenMax_ == 0) inLenMax_ = tileOutLen_ * stride_ + kernelSize_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inLenMax_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOutLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileOutLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileOutLen_;
        uint32_t outStart = blockIdx * tileOutLen_;
        if (outStart >= totalOutLen_) return;

        uint32_t outLen = totalOutLen_ - outStart;
        if (outLen > tileOutLen_) outLen = tileOutLen_;

        uint32_t inStart = outStart * stride_;
        uint32_t inLen = (outStart + outLen - 1) * stride_ + kernelSize_ - inStart;
        CopyIn(inStart, inLen, outLen);

        while (true) {
            uint32_t nextOutStart = outStart + strideElems;
            uint32_t nextOutLen = 0;
            uint32_t nextInStart = 0, nextInLen = 0;
            if (nextOutStart < totalOutLen_) {
                nextOutLen = totalOutLen_ - nextOutStart;
                if (nextOutLen > tileOutLen_) nextOutLen = tileOutLen_;
                nextInStart = nextOutStart * stride_;
                nextInLen = (nextOutStart + nextOutLen - 1) * stride_ + kernelSize_ - nextInStart;
                CopyIn(nextInStart, nextInLen, nextOutLen);
            }

            Compute(outLen);
            CopyOut(outStart, outLen);

            if (nextOutLen == 0) break;
            outStart = nextOutStart;
            outLen = nextOutLen;
            inStart = nextInStart;
            inLen = nextInLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmInOffset, uint32_t inLen, uint32_t outLen)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmInOffset], inLen);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t outLen)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < outLen; i++) {
            float sum = 0.0f;
            for (uint32_t j = 0; j < kernelSize_; j++) {
                float val = xLocal.GetValue(i * stride_ + j);
                float absVal = (val >= 0) ? val : -val;
                sum += absVal * absVal;  // p=2: square instead of powf
            }
            yLocal.SetValue(i, sqrt(sum));
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOutOffset, uint32_t outLen)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOutOffset], yLocal, outLen);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLen_ = 0;
    uint32_t tileOutLen_ = 0;
    uint32_t totalInLen_ = 0;
    uint32_t inLenMax_ = 0;
    float p_ = 2.0f;
    uint32_t kernelSize_ = 3;
    uint32_t stride_ = 2;
};

extern "C" __global__ __aicore__ void ca6k00305_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00305Custom op;
    op.Init(x, y,
            tiling_data.totalOutLen, tiling_data.tileOutLen,
            tiling_data.totalInLen,
            tiling_data.p, tiling_data.kernelSize, tiling_data.stride);
    op.Process();
}
'''

#include <torch/library.h>

#include "pytorch_npu_helper.hpp"

at::Tensor ca6k00305_custom_impl_npu(const at::Tensor& self, double p, int64_t kernel_size, int64_t stride) {
    auto in_shape = self.sizes();
    int64_t in_len = in_shape[in_shape.size() - 1];
    int64_t out_len = (in_len - kernel_size) / stride + 1;
    if (out_len < 0) out_len = 0;
    auto out_sizes = in_shape.vec();
    out_sizes.back() = out_len;
    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnCa6k00305Custom, self, p, kernel_size, stride, result);
    return result;
}

    m.impl("ca6k00305_custom", &ca6k00305_custom_impl_npu);
