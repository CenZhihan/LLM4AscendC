kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00266Custom {
public:
    __aicore__ inline KernelCa6k00266Custom() {}

    __aicore__ inline void Init(GM_ADDR indices, GM_ADDR offsets, GM_ADDR weight, GM_ADDR output,
                                uint32_t bagCount, uint32_t embeddingDim, uint32_t vocabSize)
    {
        bagCount_ = bagCount;
        embeddingDim_ = embeddingDim;
        vocabSize_ = vocabSize;

        indicesGm_.SetGlobalBuffer((__gm__ int64_t*)indices);
        offsetsGm_.SetGlobalBuffer((__gm__ int64_t*)offsets);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, vocabSize * embeddingDim);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, bagCount * embeddingDim);

        pipe_.InitBuffer(outQueue_, BUFFER_NUM, embeddingDim * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, embeddingDim * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= bagCount_) return;

        int64_t start = offsetsGm_.GetValue(blockIdx);
        int64_t end = offsetsGm_.GetValue(blockIdx + 1);

        // Accumulate sums for this bag into local buffer
        AscendC::LocalTensor<float> localSum = outQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> localRow = tmpQueue_.AllocTensor<float>();
        for (uint32_t i = 0; i < embeddingDim_; i++) {
            localSum(i) = 0.0f;
        }

        for (int64_t idx = start; idx < end; idx++) {
            int64_t index = indicesGm_.GetValue(idx);
            // Load one embedding row into localRow
            AscendC::DataCopy(localRow, weightGm_[index * embeddingDim_], embeddingDim_);
            // Add to localSum
            for (uint32_t d = 0; d < embeddingDim_; d++) {
                localSum(d) += localRow(d);
            }
        }

        // Write result to output
        AscendC::DataCopy(outputGm_[blockIdx * embeddingDim_], localSum, embeddingDim_);

        tmpQueue_.FreeTensor(localRow);
        outQueue_.FreeTensor(localSum);
    }

private:
    AscendC::GlobalTensor<int64_t> indicesGm_;
    AscendC::GlobalTensor<int64_t> offsetsGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> outputGm_;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    uint32_t bagCount_;
    uint32_t embeddingDim_;
    uint32_t vocabSize_;
};

extern "C" __global__ __aicore__ void ca6k_00266_custom(GM_ADDR indices, GM_ADDR offsets, GM_ADDR weight, GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00266Custom op;
    op.Init(indices, offsets, weight, output,
            tiling_data.bagCount, tiling_data.embeddingDim, tiling_data.vocabSize);
    op.Process();
}
'''
