kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00334Custom {
public:
    __aicore__ inline KernelCa6k00334Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(prevQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t totalRows = totalLen_ / tileLen_;

        // First row: copy x to y (identity)
        {
            AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[0], tileLen_);

            AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, xLocal, tileLen_);

            AscendC::LocalTensor<float> prevLocal = prevQueue_.AllocTensor<float>();
            AscendC::DataCopy(prevLocal, yLocal, tileLen_);
            prevQueue_.EnQue(prevLocal);

            AscendC::DataCopy(yGm_[0], yLocal, tileLen_);

            outQueue_.FreeTensor(yLocal);
            inQueue_.FreeTensor(xLocal);
        }

        // Remaining rows
        for (uint32_t row = 1; row < totalRows; row++)
        {
            uint32_t offset = row * tileLen_;

            // Prefetch current row
            AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], tileLen_);

            // Dequeue previous row's result
            AscendC::LocalTensor<float> prevLocal = prevQueue_.DeQue<float>();

            // Compute y = prev * x (elementwise)
            AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
            AscendC::Mul(yLocal, prevLocal, xLocal, tileLen_);

            // Enqueue current result as next prev
            AscendC::LocalTensor<float> nextPrev = prevQueue_.AllocTensor<float>();
            AscendC::DataCopy(nextPrev, yLocal, tileLen_);
            prevQueue_.EnQue(nextPrev);

            // Write back to global
            AscendC::DataCopy(yGm_[offset], yLocal, tileLen_);

            // Free tensors
            outQueue_.FreeTensor(yLocal);
            inQueue_.FreeTensor(xLocal);
            prevQueue_.FreeTensor(prevLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> prevQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00334_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00334Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
