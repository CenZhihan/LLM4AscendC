#include "ca6k00257_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00257CustomTilingData tiling;

    const int *N_ptr = context->GetAttrs()->GetAttrPointer<int>(0);
    const int *H_ptr = context->GetAttrs()->GetAttrPointer<int>(1);
    const int *W_ptr = context->GetAttrs()->GetAttrPointer<int>(2);
    const float *scale_ptr = context->GetAttrs()->GetAttrPointer<float>(3);
    const bool *align_ptr = context->GetAttrs()->GetAttrPointer<bool>(4);

    uint32_t N_attr = static_cast<uint32_t>(*N_ptr);
    uint32_t H_attr = static_cast<uint32_t>(*H_ptr);
    uint32_t W_attr = static_cast<uint32_t>(*W_ptr);
    float scale = *scale_ptr;
    bool align_corners = *align_ptr;

    uint32_t totalLength = N_attr * H_attr * W_attr;

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float) + 6 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 4u) blockDim = 4u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N_attr);
    tiling.set_H(H_attr);
    tiling.set_W(W_attr);
    tiling.set_scale(scale);
    tiling.set_alignCornerFlag(align_corners ? 1 : 0);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
