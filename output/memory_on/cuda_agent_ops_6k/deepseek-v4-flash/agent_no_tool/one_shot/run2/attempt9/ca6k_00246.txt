kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00246Custom {
public:
    __aicore__ inline KernelCa6k00246Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t inC, uint32_t inH, uint32_t inW,
                                uint32_t upscaleFactor,
                                uint32_t outC, uint32_t outH, uint32_t outW,
                                uint32_t totalOutLen) {
        N_ = N;
        inC_ = inC;
        inH_ = inH;
        inW_ = inW;
        upscaleFactor_ = upscaleFactor;
        outC_ = outC;
        outH_ = outH;
        outW_ = outW;
        totalOutLen_ = totalOutLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * inC_ * inH_ * inW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);
    }

    __aicore__ inline void Process() {
        for (uint32_t outIdx = 0; outIdx < totalOutLen_; ++outIdx) {
            // compute output coordinates
            uint32_t n = outIdx / (outC_ * outH_ * outW_);
            uint32_t rest = outIdx % (outC_ * outH_ * outW_);
            uint32_t cOut = rest / (outH_ * outW_);
            rest = rest % (outH_ * outW_);
            uint32_t hOut = rest / outW_;
            uint32_t wOut = rest % outW_;

            // compute input coordinates
            uint32_t inCIdx = cOut * upscaleFactor_ * upscaleFactor_ +
                              (hOut % upscaleFactor_) * upscaleFactor_ +
                              (wOut % upscaleFactor_);
            uint32_t inHIdx = hOut / upscaleFactor_;
            uint32_t inWIdx = wOut / upscaleFactor_;

            uint32_t inIdx = ((n * inC_ + inCIdx) * inH_ + inHIdx) * inW_ + inWIdx;
            yGm_.SetValue(outIdx, xGm_.GetValue(inIdx));
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t inC_ {0};
    uint32_t inH_ {0};
    uint32_t inW_ {0};
    uint32_t upscaleFactor_ {1};
    uint32_t outC_ {0};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
    uint32_t totalOutLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00246_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00246Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.inC, tiling_data.inH, tiling_data.inW,
            tiling_data.upscaleFactor,
            tiling_data.outC, tiling_data.outH, tiling_data.outW,
            tiling_data.totalOutLen);
    op.Process();
}
"""
