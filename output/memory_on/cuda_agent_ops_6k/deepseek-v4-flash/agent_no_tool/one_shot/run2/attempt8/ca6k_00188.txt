static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00188CustomTilingData tiling;
    const auto &xShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalRows = static_cast<uint32_t>(xShape.GetDim(0));
    const uint32_t totalCols = static_cast<uint32_t>(xShape.GetDim(1));

    // Each block processes one column, max 24 cores
    uint32_t blockDim = totalCols;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;

    // Tile length: aim for 2048 floats per tile (8KB), tune for UB budget
    constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    // We need: input tile (tileLen*4) + output tile (tileLen*4) + min buffer (1 float) -> < 192KB
    uint32_t tileLen = 2048;
    if (tileLen > totalRows) tileLen = totalRows;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_totalCols(totalCols);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00188Custom {
public:
    __aicore__ inline KernelCa6k00188Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t totalCols,
                                uint32_t tileLength, float negSlope)
    {
        totalRows_ = totalRows;
        totalCols_ = totalCols;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * totalCols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * totalCols_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // One extra float for running minimum
        pipe_.InitBuffer(prevMinBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || totalCols_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= totalCols_) return;

        // Process one column per block
        const uint32_t col = blockIdx;
        const uint32_t colOffset = col * totalRows_;

        // Initialize running minimum to +infinity
        AscendC::LocalTensor<float> prevMinLocal = prevMinBuf_.Get<float>();
        prevMinLocal(0) = AscendC::FP32_INF_POS;

        uint32_t processed = 0;
        while (processed < totalRows_) {
            uint32_t curLen = totalRows_ - processed;
            if (curLen > tileLen_) curLen = tileLen_;

            // Copy in tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[colOffset + processed], curLen);
            inQueueX_.EnQue(xLocal);

            // Compute cumulative min for this tile
            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            for (uint32_t i = 0; i < curLen; i++) {
                float xi = xDeque(i);
                float prev = prevMinLocal(0);
                float newMin = (xi < prev) ? xi : prev;
                yLocal(i) = newMin;
                prevMinLocal(0) = newMin;
            }

            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xDeque);

            // Copy out tile
            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[colOffset + processed], yOut, curLen);
            outQueueY_.FreeTensor(yOut);

            processed += curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> prevMinBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t totalCols_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00188_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00188Custom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.totalCols, tiling_data.tileLength, 0.0f);
    op.Process();
}
