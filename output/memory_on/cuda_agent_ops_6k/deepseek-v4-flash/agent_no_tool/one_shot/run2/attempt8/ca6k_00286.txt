#include "ca6k_00286_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00286CustomTilingData tiling;

    auto input_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(input_shape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(input_shape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(input_shape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(input_shape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scale_factor = attrs->GetAttrPointer<float>(0);

    uint32_t H_out = static_cast<uint32_t>(H * (*scale_factor));
    uint32_t W_out = static_cast<uint32_t>(W * (*scale_factor));
    uint32_t totalLength = N * C * H_out * W_out;

    uint32_t maxTileOut = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTileOut = AlignDown(maxTileOut, ALIGN_ELEMS);
    if (maxTileOut < ALIGN_ELEMS) maxTileOut = ALIGN_ELEMS;
    if (maxTileOut > HARD_CAP_ELEMS) maxTileOut = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLenOut = maxTileOut;
    if (totalLength < tileLenOut) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLenOut = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = CeilDiv(totalLength, tileLenOut);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLenOut);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_scaleFactor(*scale_factor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    auto *attrs = context->GetAttrs();
    const float *scale_factor = attrs->GetAttrPointer<float>(0);

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, x_shape->GetDim(0));
    y_shape->SetDim(1, x_shape->GetDim(1));
    uint32_t H_out = static_cast<uint32_t>(x_shape->GetDim(2) * (*scale_factor));
    uint32_t W_out = static_cast<uint32_t>(x_shape->GetDim(3) * (*scale_factor));
    y_shape->SetDim(2, static_cast<int64_t>(H_out));
    y_shape->SetDim(3, static_cast<int64_t>(W_out));
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00286Custom {
public:
    __aicore__ inline KernelCa6k00286Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                float scaleFactor)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        scaleFactor_ = scaleFactor;
        H_out_ = static_cast<uint32_t>(static_cast<float>(H_) * scaleFactor_);
        W_out_ = static_cast<uint32_t>(static_cast<float>(W_) * scaleFactor_);
        invScale_ = 1.0f / scaleFactor_;
        rowLenIn_ = W_;
        rowLenOut_ = W_out_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * H_out_ * W_out_);

        // For reading a full input row (W floats)
        pipe_.InitBuffer(inQueueRow_, BUFFER_NUM, rowLenIn_ * sizeof(float));
        // For writing two elements at a time
        pipe_.InitBuffer(tmpBuf2_, 1, 2 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t totalRows = N_ * C_ * H_out_;

        // Each block processes a set of output rows (grid-stride)
        for (uint32_t rowBlockBase = blockIdx; rowBlockBase < totalRows; rowBlockBase += blockNum) {
            // Compute (n, c, h_out) from row index
            uint32_t tmp = rowBlockBase;
            uint32_t h_out = tmp % H_out_;
            tmp /= H_out_;
            uint32_t c = tmp % C_;
            uint32_t n = tmp / C_;

            uint32_t h_in = static_cast<uint32_t>(static_cast<float>(h_out) * invScale_);

            // Read the entire input row (n, c, h_in, 0) of length W
            uint32_t inRowOffset = n * C_ * H_ * W_ + c * H_ * W_ + h_in * W_;
            AscendC::LocalTensor<float> xRow = inQueueRow_.AllocTensor<float>();
            AscendC::DataCopy(xRow, xGm_[inRowOffset], rowLenIn_);

            // Process each input column and write two output elements
            AscendC::LocalTensor<float> tmp2 = tmpBuf2_.AllocTensor<float>();

            for (uint32_t w_in = 0; w_in < W_; ++w_in) {
                float val = xRow.GetValue(w_in);
                float reluVal = (val > 0.0f) ? val : 0.0f;
                // Write two copies
                tmp2.SetValue(0, reluVal);
                tmp2.SetValue(1, reluVal);
                uint32_t outColBase = w_in * 2;  // scale_factor=2, so exactly 2
                uint32_t outRowOffset = n * C_ * H_out_ * W_out_ + c * H_out_ * W_out_ + h_out * W_out_ + outColBase;
                AscendC::DataCopy(yGm_[outRowOffset], tmp2, 2);
            }

            tmpBuf2_.FreeTensor(tmp2);
            inQueueRow_.FreeTensor(xRow);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueRow_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_, C_, H_, W_;
    uint32_t H_out_, W_out_;
    float scaleFactor_;
    float invScale_;
    uint32_t rowLenIn_, rowLenOut_;
};

extern "C" __global__ __aicore__ void ca6k_00286_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00286Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.scaleFactor);
    op.Process();
}
