在kernel中，我们实现Process：

```
uint32_t blockNum = AscendC::GetBlockNum();
uint32_t blockIdx = AscendC::GetBlockIdx();
uint32_t strideElems = blockNum * tileLen_;
uint32_t base = blockIdx * tileLen_;
while (base < totalLen_) {
    uint32_t curLen = totalLen_ - base;
    if (curLen > tileLen_) curLen = tileLen_;
    // copy in
    LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
    DataCopy(xLocal, xGm_[base], curLen);
    inQueueX_.EnQue(xLocal);
    // compute
    xLocal = inQueueX_.DeQue<float>();
    LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
    for (uint32_t i = 0; i < curLen; ++i) {
        uint32_t globalIdx = base + i;
        uint32_t matIdx = globalIdx % (height_ * width_);
        uint32_t row = matIdx / width_;
        uint32_t col = matIdx % width_;
        float val = xLocal(i);
        if ((int)col - row <= offset_) {
            yLocal(i) = val;
        } else {
            yLocal(i) = 0.0f;
        }
    }
    outQueueY_.EnQue(yLocal);
    inQueueX_.FreeTensor(xLocal);
    // copy out
    yLocal = outQueueY_.DeQue<float>();
    DataCopy(yGm_[base], yLocal, curLen);
    outQueueY_.FreeTensor(yLocal);
    base += strideElems;
}
```
