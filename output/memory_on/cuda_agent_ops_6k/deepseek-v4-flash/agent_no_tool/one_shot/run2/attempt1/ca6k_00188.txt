kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00188Custom {
public:
    __aicore__ inline KernelCa6k00188Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t totalCols,
                                uint32_t blockDim)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows * totalCols);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows * totalCols);
        totalRows_ = totalRows;
        totalCols_ = totalCols;
        blockDim_ = blockDim;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t colsPerBlock = (totalCols + blockDim - 1) / blockDim;
        colStart_ = blockIdx * colsPerBlock;
        numCols_ = colsPerBlock;
        if (colStart_ + numCols_ > totalCols) {
            numCols_ = totalCols - colStart_;
        }

        pipe_.InitBuffer(xBuf_, numCols_ * sizeof(float));
        pipe_.InitBuffer(yBuf_, numCols_ * sizeof(float));
        pipe_.InitBuffer(runningMinBuf_, numCols_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, numCols_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (numCols_ == 0) return;

        AscendC::LocalTensor<float> xLocal = xBuf_.Get<float>();
        AscendC::LocalTensor<float> yLocal = yBuf_.Get<float>();
        AscendC::LocalTensor<float> runningMin = runningMinBuf_.Get<float>();
        AscendC::LocalTensor<float> tmp = tmpBuf_.Get<float>();

        for (uint32_t row = 0; row < totalRows_; ++row) {
            // copy input slice
            AscendC::DataCopy(xLocal, xGm_[row * totalCols_ + colStart_], numCols_);

            if (row == 0) {
                // first row: output = input, runningMin = input
                AscendC::DataCopy(runningMin, xLocal, numCols_);
                AscendC::DataCopy(yLocal, xLocal, numCols_);
            } else {
                // new_min = min(runningMin, input)
                AscendC::Min(tmp, runningMin, xLocal, numCols_);
                AscendC::DataCopy(yLocal, tmp, numCols_);
                AscendC::DataCopy(runningMin, tmp, numCols_);
            }

            // copy output slice
            AscendC::DataCopy(yGm_[row * totalCols_ + colStart_], yLocal, numCols_);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> xBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> yBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> runningMinBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t totalCols_ {0};
    uint32_t blockDim_ {1};
    uint32_t colStart_ {0};
    uint32_t numCols_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00188_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00188Custom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.totalCols, tiling_data.blockDim);
    op.Process();
}
