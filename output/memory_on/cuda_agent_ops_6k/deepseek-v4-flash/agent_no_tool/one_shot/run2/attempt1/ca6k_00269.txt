namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00269CustomTilingData tiling;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    uint32_t kernelH = attrs->GetAttrPointer<uint32_t>(1)[0];
    uint32_t kernelW = attrs->GetAttrPointer<uint32_t>(2)[0];
    uint32_t strideH = attrs->GetAttrPointer<uint32_t>(3)[0];
    uint32_t strideW = attrs->GetAttrPointer<uint32_t>(4)[0];
    bool ceil_mode = attrs->GetAttrPointer<bool>(5)[0];

    const auto &inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(inShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(inShape.GetDim(1));
    uint32_t H_in = static_cast<uint32_t>(inShape.GetDim(2));
    uint32_t W_in = static_cast<uint32_t>(inShape.GetDim(3));

    uint32_t H_out, W_out;
    if (ceil_mode) {
        H_out = CeilDiv(H_in - kernelH, strideH) + 1;
        W_out = CeilDiv(W_in - kernelW, strideW) + 1;
    } else {
        H_out = (H_in - kernelH) / strideH + 1;
        W_out = (W_in - kernelW) / strideW + 1;
    }

    uint32_t totalOutput = N * C * H_out * W_out;
    const uint32_t UB_BUDGET_BYTES = 192 * 1024;
    const uint32_t ALIGN = 64;
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * kernelH * kernelW + 16);
    if (maxTile > 1024) maxTile = 1024;
    maxTile = (maxTile / ALIGN) * ALIGN;
    if (maxTile == 0) maxTile = ALIGN;

    uint32_t tileLen = (totalOutput < maxTile) ? totalOutput : maxTile;
    uint32_t numTiles = CeilDiv(totalOutput, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.N = N;
    tiling.C = C;
    tiling.H_in = H_in;
    tiling.W_in = W_in;
    tiling.H_out = H_out;
    tiling.W_out = W_out;
    tiling.kernelH = kernelH;
    tiling.kernelW = kernelW;
    tiling.strideH = strideH;
    tiling.strideW = strideW;
    tiling.totalOutput = totalOutput;
    tiling.tileLen = tileLen;
    tiling.blockDim = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currWorkspace = context->GetWorkspaceSizes(1);
    currWorkspace[0] = 0;

kernel_src="""
#include "kernel_operator.h"
#include "ca6k_00269_custom_tiling.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00269Custom {
public:
    __aicore__ inline KernelCa6k00269Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, const Ca6k00269CustomTilingData &tiling)
    {
        N_ = tiling.N;
        C_ = tiling.C;
        H_in_ = tiling.H_in;
        W_in_ = tiling.W_in;
        H_out_ = tiling.H_out;
        W_out_ = tiling.W_out;
        kernelH_ = tiling.kernelH;
        kernelW_ = tiling.kernelW;
        strideH_ = tiling.strideH;
        strideW_ = tiling.strideW;
        totalOutput_ = tiling.totalOutput;
        tileLen_ = tiling.tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_in_ * W_in_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutput_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, kernelH_ * kernelW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutput_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;

        while (base < totalOutput_) {
            uint32_t curTile = tileLen_;
            if (base + curTile > totalOutput_)
                curTile = totalOutput_ - base;

            for (uint32_t i = 0; i < curTile; i++) {
                uint32_t outIdx = base + i;
                float sum = 0.0f;

                uint32_t tmp = outIdx;
                uint32_t n = tmp / (C_ * H_out_ * W_out_);
                tmp -= n * (C_ * H_out_ * W_out_);
                uint32_t c = tmp / (H_out_ * W_out_);
                tmp -= c * (H_out_ * W_out_);
                uint32_t r = tmp / W_out_;
                uint32_t col = tmp % W_out_;

                uint32_t startH = r * strideH_;
                uint32_t startW = col * strideW_;

                for (uint32_t kh = 0; kh < kernelH_; kh++) {
                    for (uint32_t kw = 0; kw < kernelW_; kw++) {
                        uint32_t inH = startH + kh;
                        uint32_t inW = startW + kw;
                        uint32_t inIdx = ((n * C_ + c) * H_in_ + inH) * W_in_ + inW;
                        float val = xGm_.GetValue(inIdx);
                        sum += val * val;
                    }
                }

                // sqrt into tmp buffer
                LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                tmpLocal.SetValue(0, sum);
                Sqrt(tmpLocal, tmpLocal, 1);
                float result = tmpLocal.GetValue(0);
                tmpQueue_.FreeTensor(tmpLocal);
                yGm_.SetValue(outIdx, result);
            }

            base += stride;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;   // unused but needed for pipe style
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_; // unused
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, H_in_, W_in_, H_out_, W_out_;
    uint32_t kernelH_, kernelW_, strideH_, strideW_;
    uint32_t totalOutput_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void ca6k_00269_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00269Custom op;
    op.Init(x, y, tiling_data);
    op.Process();
}
