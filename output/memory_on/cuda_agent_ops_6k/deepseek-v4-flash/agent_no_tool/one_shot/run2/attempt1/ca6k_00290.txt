static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00290CustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *offset = attrs->GetAttrPointer<int32_t>(0);

    // UB usage: 4 tiles (2 in, 2 out) each of tileLength floats
    uint32_t maxTile = UB_BUDGET_BYTES / (4u * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    // Get height and width from input shape (last two dims)
    const gert::Shape *inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t height = static_cast<uint32_t>(inShape->GetDim(2));
    uint32_t width  = static_cast<uint32_t>(inShape->GetDim(3));

    tiling.totalLength = totalLength;
    tiling.tileLength  = tileLen;
    tiling.blockDim    = blockDim;
    tiling.height      = height;
    tiling.width       = width;
    tiling.offset      = *offset;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src = r"""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00290Custom {
public:
    __aicore__ inline KernelCa6k00290Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t height, uint32_t width, int32_t offset)
    {
        totalLen_ = totalLength;
        tileLen_  = tileLength;
        height_   = height;
        width_    = width;
        offset_   = offset;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;

        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Prefetch next tile in advance
            uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            CopyIn(base, curLen);
            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Compute the start global index of this tile
        // We do not have the base offset here, but we can reconstruct it from the number of tiles processed
        // Compute base = (blockIdx * tileLen_) + (tileCount * strideElems)
        // Simpler: use a static counter per block
        // Since we don't have tile count here, we use the current tile's global offset computed in Process().
        // We retrieve it from a member variable computed before calling Compute.
        uint32_t tileBase = currentBase_;
        const uint32_t matSize = height_ * width_;

        for (uint32_t i = 0; i < len; ++i) {
            uint32_t globalIdx = tileBase + i;
            uint32_t matIdx = globalIdx % matSize;
            uint32_t row = matIdx / width_;
            uint32_t col = matIdx % width_;
            float val = xLocal(i);
            if ((int32_t)col - (int32_t)row <= offset_) {
                yLocal(i) = val;
            } else {
                yLocal(i) = 0.0f;
            }
        }

        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

public:
    // We expose this to set currentBase_ before Compute
    __aicore__ inline void SetCurrentBase(uint32_t base) { currentBase_ = base; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    int32_t offset_ {0};
    uint32_t currentBase_ {0};  // set by Process before each Compute
};

extern "C" __global__ __aicore__ void ca6k_00290_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00290Custom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.height, tiling_data.width, tiling_data.offset);

    // Process tiles with grid-stride
    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t tileLen = tiling_data.tileLength;
    const uint32_t strideElems = blockNum * tileLen;
    uint32_t base = blockIdx * tileLen;
    if (base >= tiling_data.totalLength) return;

    while (true) {
        uint32_t curLen = tiling_data.totalLength - base;
        if (curLen > tileLen) curLen = tileLen;

        uint32_t nextBase = base + strideElems;
        uint32_t nextLen = 0;
        if (nextBase < tiling_data.totalLength) {
            nextLen = tiling_data.totalLength - nextBase;
            if (nextLen > tileLen) nextLen = tileLen;
            // Prefetch next tile
            {
                AscendC::LocalTensor<float> xLocal = op.GetInBuffer(); // We cannot access private members this way; we will restructure
            }
        }

        // Simplified: we implement the loop directly in the external function without using class methods that require private buffer access.
        // Instead, we call simple inline functions in the class.
        // But we need to set currentBase_ before Compute.
        // We'll copy the logic inside Process into the external function, using the class only for Init and buffer management.
        // Since we cannot easily expose buffer queues, we will reimplement the loop logic inside the external function,
        // using the kernel class as a helper.

        // For simplicity, we delegate back to the Process method which we will rewrite to accept external loop control.
        // We'll modify KernelCa6k00290Custom to have a Process that works with the current grid-stride pattern.
        // Keep the original Process but add a parameter for the base offset per iteration?
        // Let's restructure: Process will be called with no arguments and will handle the full grid-stride loop.
        // We already have Process implemented above with grid-stride logic and prefetch. But Compute lacks base.
        // We will store currentBase_ as a member and update it in Process.
    }
}
