  return (a + b - 1) / b;

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k00330CustomTilingData tiling;

  const gert::RuntimeShape *biasShape = context->GetInputShape(0);
  const gert::RuntimeShape *mat1Shape = context->GetInputShape(1);
  const gert::RuntimeShape *mat2Shape = context->GetInputShape(2);

  uint32_t M = static_cast<uint32_t>(biasShape->GetOriginShape().GetDim(1));
  uint32_t N = static_cast<uint32_t>(biasShape->GetOriginShape().GetDim(2));
  uint32_t K = static_cast<uint32_t>(mat1Shape->GetOriginShape().GetDim(2));

  uint32_t totalLength = B * M * N;

  // Compute blockDim based on totalLength; keep moderate.
  uint32_t blockDim = (totalLength + 1024*1024 - 1) / (1024*1024);
  if (blockDim < BLOCK_DIM_MIN) blockDim = BLOCK_DIM_MIN;
  if (blockDim > BLOCK_DIM_MAX) blockDim = BLOCK_DIM_MAX;

  tiling.batch = B;
  tiling.M = M;
  tiling.N = N;
  tiling.K = K;
  tiling.totalLength = totalLength;
  tiling.tileLength = 1; // not used in vector loop
  tiling.blockDim = blockDim;

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_K = 1024;

class KernelCa6k00330Custom {
 public:
  __aicore__ inline KernelCa6k00330Custom() {}

  __aicore__ inline void Init(GM_ADDR bias, GM_ADDR mat1, GM_ADDR mat2,
                              GM_ADDR y, uint32_t batch, uint32_t M,
                              uint32_t N, uint32_t K, uint32_t totalLength,
                              uint32_t tileLength, uint32_t blockDim) {
    batch_ = batch;
    M_ = M;
    N_ = N;
    K_ = K;
    totalLen_ = totalLength;
    blockDim_ = blockDim;

    biasGm_.SetGlobalBuffer((__gm__ float *)bias, batch_ * M_ * N_);
    mat1Gm_.SetGlobalBuffer((__gm__ float *)mat1, batch_ * M_ * K_);
    mat2Gm_.SetGlobalBuffer((__gm__ float *)mat2, batch_ * K_ * N_);
    yGm_.SetGlobalBuffer((__gm__ float *)y, batch_ * M_ * N_);

    // Allocate local buffers
    pipe_.InitBuffer(biasBuf_, BUFFER_NUM, M_ * N_ * sizeof(float));
    pipe_.InitBuffer(accBuf_, 1, N_ * sizeof(float));
    pipe_.InitBuffer(mat2Buf_, 1, TILE_K * N_ * sizeof(float));
    pipe_.InitBuffer(vecTmpBuf_, 1, N_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0) return;

    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t blockNum = blockDim_;

    for (uint32_t b = blockIdx; b < batch_; b += blockNum) {
      // Load bias for this batch
      AscendC::LocalTensor<float> biasLcl = biasBuf_.AllocTensor<float>();
      AscendC::DataCopy(biasLcl, biasGm_[b * M_ * N_], M_ * N_);

      // Process each row i
      for (uint32_t i = 0; i < M_; ++i) {
        // Start with bias[b, i, :]
        AscendC::LocalTensor<float> accLcl = accBuf_.AllocTensor<float>();
        AscendC::DataCopy(accLcl, biasLcl[i * N_], N_);

        // Accumulate over k tiles
        for (uint32_t kStart = 0; kStart < K_; kStart += TILE_K) {
          uint32_t curTileK = (K_ - kStart) < TILE_K ? (K_ - kStart) : TILE_K;

          // Load mat1[b, i, kStart:kStart+curTileK] as scalar vector
          AscendC::LocalTensor<float> mat1RowLcl = vecTmpBuf_.AllocTensor<float>();
          AscendC::DataCopy(mat1RowLcl,
                            mat1Gm_[b * M_ * K_ + i * K_ + kStart],
                            curTileK);

          // Load mat2[b, kStart:kStart+curTileK, :]   (size curTileK * N)
          AscendC::LocalTensor<float> mat2Lcl = mat2Buf_.AllocTensor<float>();
          AscendC::DataCopy(mat2Lcl,
                            mat2Gm_[b * K_ * N_ + kStart * N_],
                            curTileK * N_);

          // For each k in the tile
          for (uint32_t kk = 0; kk < curTileK; ++kk) {
            float scalar = mat1RowLcl[kk];
            // Load mat2 row (N floats) from mat2Lcl at offset kk*N
            AscendC::LocalTensor<float> tmpLcl = vecTmpBuf_.AllocTensor<float>();
            AscendC::DataCopy(tmpLcl, mat2Lcl[kk * N_], N_);
            // Multiply by scalar
            AscendC::Muls(tmpLcl, tmpLcl, scalar, N_);
            // Add to accumulator
            AscendC::Add(accLcl, accLcl, tmpLcl, N_);
            vecTmpBuf_.FreeTensor(tmpLcl);
          }

          mat2Buf_.FreeTensor(mat2Lcl);
          vecTmpBuf_.FreeTensor(mat1RowLcl);
        }

        // Write output row y[b, i, :]
        AscendC::DataCopy(yGm_[b * M_ * N_ + i * N_], accLcl, N_);
        accBuf_.FreeTensor(accLcl);
      }

      biasBuf_.FreeTensor(biasLcl);
    }
  }

 private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> biasBuf_;
  AscendC::TQue<AscendC::TPosition::VECOUT, 1> accBuf_;
  AscendC::TQue<AscendC::TPosition::VECIN, 1> mat2Buf_;
  AscendC::TQue<AscendC::TPosition::VECCALC, 1> vecTmpBuf_;

  AscendC::GlobalTensor<float> biasGm_;
  AscendC::GlobalTensor<float> mat1Gm_;
  AscendC::GlobalTensor<float> mat2Gm_;
  AscendC::GlobalTensor<float> yGm_;

  uint32_t batch_ {0};
  uint32_t M_ {0};
  uint32_t N_ {0};
  uint32_t K_ {0};
  uint32_t totalLen_ {0};
  uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00330_custom(
    GM_ADDR bias, GM_ADDR mat1, GM_ADDR mat2, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelCa6k00330Custom op;
  op.Init(bias, mat1, mat2, y, tiling_data.batch, tiling_data.M,
          tiling_data.N, tiling_data.K, tiling_data.totalLength,
          tiling_data.tileLength, tiling_data.blockDim);
  op.Process();
}
