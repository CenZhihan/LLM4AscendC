#include "kernel_operator.h"

class KernelCa6k00303Custom {
public:
    __aicore__ inline KernelCa6k00303Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalElements, uint32_t tileLength,
                                uint32_t batchSize, uint32_t inChannels,
                                uint32_t inH, uint32_t inW, uint32_t r,
                                uint32_t outChannels, uint32_t outH, uint32_t outW)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElements);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElements);
        totalElements_ = totalElements;
        tileLength_ = tileLength;
        batchSize_ = batchSize;
        inChannels_ = inChannels;
        inH_ = inH;
        inW_ = inW;
        r_ = r;
        outChannels_ = outChannels;
        outH_ = outH;
        outW_ = outW;
    }

    __aicore__ inline void Process()
    {
        if (totalElements_ == 0 || tileLength_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t stride = blockNum * tileLength_;
        uint32_t start = blockIdx * tileLength_;
        for (uint32_t idx = start; idx < totalElements_; idx += stride) {
            uint32_t end = idx + tileLength_;
            if (end > totalElements_) end = totalElements_;
            for (uint32_t i = idx; i < end; i++) {
                // compute output coordinates
                uint32_t tmp = i;
                uint32_t n = tmp / (outChannels_ * outH_ * outW_);
                tmp %= (outChannels_ * outH_ * outW_);
                uint32_t c_out = tmp / (outH_ * outW_);
                tmp %= (outH_ * outW_);
                uint32_t h_out = tmp / outW_;
                uint32_t w_out = tmp % outW_;
                // compute input coordinates
                uint32_t h_in = h_out / r_;
                uint32_t w_in = w_out / r_;
                uint32_t c_in = (h_out % r_) * r_ * outChannels_ + (w_out % r_) * outChannels_ + c_out;
                // input linear index: n * inChannels_ * inH_ * inW_ + c_in * inH_ * inW_ + h_in * inW_ + w_in
                uint32_t inIdx = n * inChannels_ * inH_ * inW_ + c_in * inH_ * inW_ + h_in * inW_ + w_in;
                yGm_.SetValue(i, xGm_.GetValue(inIdx));
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalElements_{0};
    uint32_t tileLength_{0};
    uint32_t batchSize_{0};
    uint32_t inChannels_{0};
    uint32_t inH_{0};
    uint32_t inW_{0};
    uint32_t r_{0};
    uint32_t outChannels_{0};
    uint32_t outH_{0};
    uint32_t outW_{0};
};

extern "C" __global__ __aicore__ void ca6k00303_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00303Custom op;
    op.Init(x, y, tiling_data.totalElements, tiling_data.tileLength,
            tiling_data.batchSize, tiling_data.inChannels,
            tiling_data.inH, tiling_data.inW, tiling_data.r,
            tiling_data.outChannels, tiling_data.outH, tiling_data.outW);
