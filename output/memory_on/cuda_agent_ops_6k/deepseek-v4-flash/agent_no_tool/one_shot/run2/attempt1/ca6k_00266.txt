host_operator_src = r"""
#include "ca6k_00266_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00266CustomTilingData tiling;

    const gert::Shape *weight_shape = context->GetInputShape(0);
    const gert::Shape *indices_shape = context->GetInputShape(1);
    const gert::Shape *offsets_shape = context->GetInputShape(2);

    uint32_t embeddingDim = weight_shape->GetDim(1);
    uint32_t numBags = offsets_shape->GetDim(0) - 1;
    uint32_t totalIndices = indices_shape->GetDim(0);

    // Compute tile for embedding dimension based on UB budget
    // We need 2 tiles (accumulator + weight row) -> 2 * tileEmb * 4 bytes
    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileEmb = maxTile;
    if (embeddingDim < tileEmb) {
        tileEmb = AlignDown(embeddingDim, ALIGN_ELEMS);
        if (tileEmb == 0) tileEmb = embeddingDim;
    }

    // Block dim: use small fixed number for simplicity
    uint32_t blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.numBags = numBags;
    tiling.embeddingDim = embeddingDim;
    tiling.tileEmb = tileEmb;
    tiling.totalIndices = totalIndices;
    tiling.blockDim = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *weight_shape = context->GetInputShape(0);
    const gert::Shape *offsets_shape = context->GetInputShape(2);

    uint32_t numBags = offsets_shape->GetDim(0) - 1;
    uint32_t embeddingDim = weight_shape->GetDim(1);

    gert::Shape *output_shape = context->GetOutputShape(0);
    output_shape->SetDim(0, numBags);
    output_shape->SetDim(1, embeddingDim);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType weight_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, weight_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = r"""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00266Custom {
public:
    __aicore__ inline KernelCa6k00266Custom() {}

    __aicore__ inline void Init(GM_ADDR weight, GM_ADDR indices, GM_ADDR offsets, GM_ADDR output,
                                uint32_t numBags, uint32_t embeddingDim, uint32_t tileEmb,
                                uint32_t totalIndices, uint32_t blockDim) {
        numBags_ = numBags;
        embeddingDim_ = embeddingDim;
        tileEmb_ = tileEmb;
        totalIndices_ = totalIndices;
        blockDim_ = blockDim;

        weightGm_.SetGlobalBuffer((__gm__ float*)weight, numBags_ * embeddingDim_);  // actual numEmbeddings * embeddingDim
        indicesGm_.SetGlobalBuffer((__gm__ int32_t*)indices, totalIndices_);
        offsetsGm_.SetGlobalBuffer((__gm__ int32_t*)offsets, numBags_ + 1);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, numBags_ * embeddingDim_);

        pipe_.InitBuffer(inQueueAcc_, BUFFER_NUM, tileEmb_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileEmb_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (numBags_ == 0 || embeddingDim_ == 0) return;

        const uint32_t numTiles = (embeddingDim_ + tileEmb_ - 1) / tileEmb_;

        for (uint32_t bagIdx = 0; bagIdx < numBags_; ++bagIdx) {
            uint32_t startIdx = offsetsGm_.GetValue(bagIdx);
            uint32_t endIdx = offsetsGm_.GetValue(bagIdx + 1);

            for (uint32_t t = 0; t < numTiles; ++t) {
                uint32_t tileStart = t * tileEmb_;
                uint32_t curTileLen = (tileStart + tileEmb_ <= embeddingDim_) ? tileEmb_ : (embeddingDim_ - tileStart);

                // Allocate local accumulator and weight tile
                AscendC::LocalTensor<float> accLocal = inQueueAcc_.AllocTensor<float>();
                AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();

                // Zero accumulator
                AscendC::DupTensor(accLocal, 0.0f, curTileLen);

                // Accumulate each index in bag
                for (uint32_t pos = startIdx; pos < endIdx; ++pos) {
                    uint32_t idx = indicesGm_.GetValue(pos);
                    // Load weight row idx, tile t into wLocal
                    AscendC::DataCopy(wLocal, weightGm_[idx * embeddingDim_ + tileStart], curTileLen);
                    // Add to accumulator
                    AscendC::Add(accLocal, accLocal, wLocal, curTileLen);
                }

                // Write accumulator to output
                AscendC::DataCopy(outputGm_[bagIdx * embeddingDim_ + tileStart], accLocal, curTileLen);

                // Free local tensors
                inQueueAcc_.FreeTensor(accLocal);
                inQueueW_.FreeTensor(wLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> inQueueAcc_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> inQueueW_;

    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<int32_t> indicesGm_;
    AscendC::GlobalTensor<int32_t> offsetsGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t numBags_;
    uint32_t embeddingDim_;
    uint32_t tileEmb_;
    uint32_t totalIndices_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void ca6k_00266_custom(GM_ADDR weight, GM_ADDR indices, GM_ADDR offsets,
                                                         GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00266Custom op;
    op.Init(weight, indices, offsets, output,
            tiling_data.numBags, tiling_data.embeddingDim, tiling_data.tileEmb,
            tiling_data.totalIndices, tiling_data.blockDim);
    op.Process();
}
"""

python_bind_src = r"""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor embedding_bag_custom_impl_npu(const at::Tensor& weight,
                                          const at::Tensor& indices,
                                          const at::Tensor& offsets) {
    // weight shape: [num_embeddings, embedding_dim]
    // indices shape: [num_indices], int32
    // offsets shape: [num_bags + 1], int32
    auto batch_size = offsets.size(0) - 1;
    auto embedding_dim = weight.size(1);
    at::Tensor output = at::empty({batch_size, embedding_dim}, weight.options());
    EXEC_NPU_CMD(aclnnCa6k00266Custom, weight, indices, offsets, output);
    return output;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("ca6k_00266_custom", &embedding_bag_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("ca6k_00266_custom", &embedding_bag_custom_impl_npu,
          "Fused EmbeddingBag (sum mode) on NPU");
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_embeddings, embedding_dim, mode="sum"):
        super(ModelNew, self).__init__()
        self.weight = torch.nn.Parameter(
            torch.randn(num_embeddings, embedding_dim, dtype=torch.float32)
        )

    def forward(self, input, offsets):
        # input and offsets are int64 tensors; cast to int32 for the custom op
        input_i32 = input.int()
        offsets_i32 = offsets.int()
        return custom_ops_lib.ca6k_00266_custom(self.weight, input_i32, offsets_i32)
