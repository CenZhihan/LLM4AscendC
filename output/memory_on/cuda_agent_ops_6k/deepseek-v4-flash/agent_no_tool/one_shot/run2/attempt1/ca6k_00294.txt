static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00294CustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t paddingIdx = (attrs->GetAttrPointer<int64_t>(0) != nullptr) ? *attrs->GetAttrPointer<int64_t>(0) : -1;

    const uint32_t totalInputLen = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t embeddingDim = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetDim(1));

    // UB budget: 192KB
    // we will use: 1 int64 index (8 bytes) + 2 float rows (2 * embeddingDim * 4 bytes)
    const uint32_t bytesPerElement = 8 + 8 * embeddingDim;
    uint32_t tileLen = 1;  // process one input element per iteration per block
    // ensure fit in UB: limit based on budget
    if (bytesPerElement * tileLen > 192 * 1024) {
        tileLen = (192 * 1024) / bytesPerElement;
        if (tileLen < 1) tileLen = 1;
    }

    uint32_t blockDim = 16;
    if (totalInputLen > 256) blockDim = 24;
    else if (totalInputLen > 64) blockDim = 16;
    else blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.totalInputLen = totalInputLen;
    tiling.embeddingDim = embeddingDim;
    tiling.paddingIdx = paddingIdx;
    tiling.blockDim = blockDim;
    tiling.tileLen = tileLen;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    gert::Shape *y_shape = context->GetOutputShape(0);
    // output shape: [batch * seq, embeddingDim] flattened, or keep dims
    std::vector<int64_t> dims;

    *y_shape = out_shape;
    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00294Custom {
public:
    __aicore__ inline KernelCa6k00294Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalInputLen, uint32_t embeddingDim,
                                int64_t paddingIdx, uint32_t tileLen)
    {
        totalInputLen_ = totalInputLen;
        embeddingDim_ = embeddingDim;
        paddingIdx_ = paddingIdx;
        tileLen_ = tileLen;  // number of input elements per iteration (1)

        xGm_.SetGlobalBuffer((__gm__ int64_t*)x, totalInputLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, -1); // global buffer size unknown, but we offset manually
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalInputLen_ * embeddingDim_);

        // buffers: xBuf for 1 int64, wBuf for embeddingDim floats, yBuf for embeddingDim floats
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, sizeof(int64_t));
        pipe_.InitBuffer(tmpQueue_, 1, embeddingDim_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, embeddingDim_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalInputLen_ == 0 || embeddingDim_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t base = blockIdx;
        if (base >= totalInputLen_) return;

        while (base < totalInputLen_) {
            // load input index
            AscendC::LocalTensor<int64_t> xLocal = inQueueX_.AllocTensor<int64_t>();
            AscendC::DataCopy(xLocal, xGm_[base], 1);
            int64_t idx = xLocal(0);  // scalar value
            inQueueX_.FreeTensor(xLocal);

            // allocate output buffer
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

            if (idx == paddingIdx_) {
                // fill with zeros
                AscendC::Muls(yLocal, yLocal, 0.0f, embeddingDim_);
            } else {
                // load weight row: offset = idx * embeddingDim_
                uint32_t offset = static_cast<uint32_t>(idx) * embeddingDim_;
                AscendC::DataCopy(tmpLocal, wGm_[offset], embeddingDim_);
                AscendC::Copy(yLocal, tmpLocal, embeddingDim_);
            }

            // write to output: base * embeddingDim_
            uint32_t outOffset = base * embeddingDim_;
            AscendC::DataCopy(yGm_[outOffset], yLocal, embeddingDim_);

            tmpQueue_.FreeTensor(tmpLocal);
            outQueueY_.FreeTensor(yLocal);

            base += blockNum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<int64_t> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputLen_ {0};
    uint32_t embeddingDim_ {0};
    int64_t paddingIdx_ {-1};
    uint32_t tileLen_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00294_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00294Custom op;
    op.Init(x, weight, y, tiling_data.totalInputLen, tiling_data.embeddingDim, tiling_data.paddingIdx, tiling_data.tileLen);
    op.Process();
}
