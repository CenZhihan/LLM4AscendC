#include "ca6k_00322_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00322CustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t C = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetDim(3));
    const uint32_t totalLength = N * C * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *groupsPtr = attrs->GetAttrPointer<int32_t>(0);
    uint32_t groups = static_cast<uint32_t>(*groupsPtr);

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.N = N;
    tiling.C = C;
    tiling.H = H;
    tiling.W = W;
    tiling.groups = groups;
    tiling.totalLength = totalLength;
    tiling.tileLength = tileLen;
    tiling.blockDim = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00322Custom {
public:
    __aicore__ inline KernelCa6k00322Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t groups, uint32_t totalLength,
                                uint32_t tileLength)
    {
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        groups_ = groups;
        Cg_ = C / groups;
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Load a tile of output elements (contiguous)
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], curLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // Compute output for each element in tile
            for (uint32_t i = 0; i < curLen; i++) {
                // output index
                uint32_t idx = base + i;
                // (n, c_out, h, w) from linear index
                uint32_t c_out = (idx / (H_ * W_)) % C_;
                uint32_t n = idx / (C_ * H_ * W_);
                uint32_t rem = idx % (H_ * W_);
                uint32_t h = rem / W_;
                uint32_t w = rem % W_;

                uint32_t c_in = (c_out % groups_) * Cg_ + (c_out / groups_);
                uint32_t inIdx = n * C_ * H_ * W_ + c_in * H_ * W_ + h * W_ + w;

                // Read input value (scalar)
                float val = xGm_.GetValue(inIdx);
                yLocal.SetValue(i, val);
            }

            outQueueY_.EnQue(yLocal);
            // Wait for previous copy? Actually we need to get the local tensor for output
            AscendC::LocalTensor<float> yQueued = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yQueued, curLen);
            outQueueY_.FreeTensor(yQueued);

            // Free input
            AscendC::LocalTensor<float> xQueued = inQueueX_.DeQue<float>();
            inQueueX_.FreeTensor(xQueued);

            base += strideElems;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t groups_ {2};
    uint32_t Cg_ {0};
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00322_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00322Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.groups, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
