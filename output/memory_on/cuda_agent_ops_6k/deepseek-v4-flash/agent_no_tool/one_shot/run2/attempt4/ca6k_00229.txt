kernel_src="""

#include "kernel_operator.h"

class KernelCa6k00229Custom {
public:
    __aicore__ inline KernelCa6k00229Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t H_in, uint32_t W_in,
                                uint32_t H_out, uint32_t W_out,
                                uint32_t pad_top, uint32_t pad_left,
                                uint32_t blockDim)
    {
        N_ = N;
        C_ = C;
        H_in_ = H_in;
        W_in_ = W_in;
        H_out_ = H_out;
        W_out_ = W_out;
        pad_top_ = pad_top;
        pad_left_ = pad_left;
        blockDim_ = blockDim;

        stride_c_in_ = H_in_ * W_in_;
        stride_c_out_ = H_out_ * W_out_;
        stride_n_in_ = C_ * stride_c_in_;
        stride_n_out_ = C_ * stride_c_out_;
        total_in_ = N_ * stride_n_in_;
        total_out_ = N_ * stride_n_out_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, total_in_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, total_out_);
    }

    __aicore__ inline void Process()
    {
        uint32_t plane_count = N_ * C_;
        uint32_t block_idx = AscendC::GetBlockIdx();
        uint32_t block_num = blockDim_;

        for (uint32_t plane = block_idx; plane < plane_count; plane += block_num) {
            uint32_t n = plane / C_;
            uint32_t c = plane % C_;

            float * __gm__ x_plane = xGm_ + n * stride_n_in_ + c * stride_c_in_;
            float * __gm__ y_plane = yGm_ + n * stride_n_out_ + c * stride_c_out_;

            for (uint32_t r = 0; r < H_out_; r++) {
                for (uint32_t c_out = 0; c_out < W_out_; c_out++) {
                    float val = 0.0f;
                    if (r >= pad_top_ && r < (pad_top_ + H_in_)
                        && c_out >= pad_left_ && c_out < (pad_left_ + W_in_)) {
                        uint32_t r_in = r - pad_top_;
                        uint32_t c_in = c_out - pad_left_;
                        val = x_plane[r_in * W_in_ + c_in];
                    }
                    y_plane[r * W_out_ + c_out] = val;
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t H_in_ {0};
    uint32_t W_in_ {0};
    uint32_t H_out_ {0};
    uint32_t W_out_ {0};
    uint32_t pad_top_ {0};
    uint32_t pad_left_ {0};
    uint32_t blockDim_ {1};
    uint32_t stride_c_in_ {0};
    uint32_t stride_c_out_ {0};
    uint32_t stride_n_in_ {0};
    uint32_t stride_n_out_ {0};
    uint32_t total_in_ {0};
    uint32_t total_out_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00229_custom(GM_ADDR x, GM_ADDR y,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00229Custom op;
    op.Init(x, y,
            tiling_data.N_, tiling_data.C_,
            tiling_data.H_in_, tiling_data.W_in_,
            tiling_data.H_out_, tiling_data.W_out_,
            tiling_data.pad_top_, tiling_data.pad_left_,
            tiling_data.blockDim_);
    op.Process();
}
