#include "ca6k_00257_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00257CustomTilingData tiling;

    const int *N_ptr = context->GetAttrs()->GetAttrPointer<int>(0);
    const int *H_ptr = context->GetAttrs()->GetAttrPointer<int>(1);
    const int *W_ptr = context->GetAttrs()->GetAttrPointer<int>(2);
    const float *scale_ptr = context->GetAttrs()->GetAttrPointer<float>(3);
    const bool *align_ptr = context->GetAttrs()->GetAttrPointer<bool>(4);

    const uint32_t N_attr = static_cast<uint32_t>(*N_ptr);
    const uint32_t H_attr = static_cast<uint32_t>(*H_ptr);
    const uint32_t W_attr = static_cast<uint32_t>(*W_ptr);
    const float scale = *scale_ptr;
    const bool align_corners = *align_ptr;

    const uint32_t totalLength = N_attr * H_attr * W_attr * 2;

    uint32_t maxTile = UB_BUDGET_BYTES / (8 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 4u) blockDim = 4u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N_attr);
    tiling.set_H(H_attr);
    tiling.set_W(W_attr);
    tiling.set_scale(scale);
    tiling.set_alignCornerFlag(align_corners ? 1 : 0);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const int *N_ptr = context->GetAttrs()->GetAttrPointer<int>(0);
    const int *H_ptr = context->GetAttrs()->GetAttrPointer<int>(1);
    const int *W_ptr = context->GetAttrs()->GetAttrPointer<int>(2);

    uint32_t N = static_cast<uint32_t>(*N_ptr);
    uint32_t H = static_cast<uint32_t>(*H_ptr);
    uint32_t W = static_cast<uint32_t>(*W_ptr);

    std::vector<int64_t> dims = {static_cast<int64_t>(N), static_cast<int64_t>(H), static_cast<int64_t>(W), 2};
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    for (int i = 0; i < 4; i++) y_shape->SetDim(i, dims[i]);
    return GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00257Custom {
public:
    __aicore__ inline KernelCa6k00257Custom() {}

    __aicore__ inline void Init(GM_ADDR theta, GM_ADDR grid_scaled,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t H, uint32_t W,
                                float scale, int32_t alignCornerFlag)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N; H_ = H; W_ = W;
        scale_ = scale;
        alignCornerFlag_ = alignCornerFlag;

        thetaGm_.SetGlobalBuffer((__gm__ float*)theta, N_ * 6);
        gridGm_.SetGlobalBuffer((__gm__ float*)grid_scaled, totalLen_);

        pipe_.InitBuffer(inQueueTheta_, BUFFER_NUM, 6 * sizeof(float));
        pipe_.InitBuffer(outQueueGrid_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        ComputeAndCopyOut(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalLen_) break;
            uint32_t nextLen = totalLen_ - nextBase;
            if (nextLen > tileLen_) nextLen = tileLen_;
            ComputeAndCopyOut(nextBase, nextLen);
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void ComputeAndCopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> gridLocal = outQueueGrid_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; i++) {
            uint32_t globalIdx = gmOffset + i;
            uint32_t hw = globalIdx / 2;
            uint32_t batchIdx = hw / (H_ * W_);
            uint32_t hwRem = hw % (H_ * W_);
            uint32_t h = hwRem / W_;
            uint32_t w = hwRem % W_;

            float x = 0.0f, y = 0.0f;
            if (alignCornerFlag_) {
                if (W_ > 1) x = (2.0f * w) / (W_ - 1.0f) - 1.0f;
                if (H_ > 1) y = (2.0f * h) / (H_ - 1.0f) - 1.0f;
            } else {
                x = (2.0f * w + 1.0f) / W_ - 1.0f;
                y = (2.0f * h + 1.0f) / H_ - 1.0f;
            }

            // Get theta matrix for this batch
            AscendC::DataCopy(tLocal, thetaGm_[batchIdx * 6], 6);
            float a = tLocal(0);
            float b = tLocal(1);
            float c = tLocal(2);
            float d = tLocal(3);
            float e = tLocal(4);
            float f = tLocal(5);

            float gridX = a * x + b * y + c;
            float gridY = d * x + e * y + f;

            gridX *= scale_;
            gridY *= scale_;

            gridLocal(i * 2 + 0) = gridX;
            gridLocal(i * 2 + 1) = gridY;
        }

        AscendC::DataCopy(gridGm_[gmOffset], gridLocal, len);
        tmpQueue_.FreeTensor(tLocal);
        outQueueGrid_.FreeTensor(gridLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTheta_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueGrid_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> thetaGm_;
    AscendC::GlobalTensor<float> gridGm_;

    uint32_t totalLen_ = 0;
    uint32_t tileLen_ = 0;
    uint32_t N_ = 0;
    uint32_t H_ = 0;
    uint32_t W_ = 0;
    float scale_ = 0.5f;
    int32_t alignCornerFlag_ = 1;
};

extern "C" __global__ __aicore__ void ca6k_00257_custom(GM_ADDR theta, GM_ADDR grid_scaled, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00257Custom op;
    op.Init(theta, grid_scaled, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.H, tiling_data.W,
            tiling_data.scale, tiling_data.alignCornerFlag);
    op.Process();
}
'''
