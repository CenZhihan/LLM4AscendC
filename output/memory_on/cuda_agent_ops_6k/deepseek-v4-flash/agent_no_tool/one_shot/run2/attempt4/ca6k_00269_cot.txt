        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t itemsPerBlock = (totalOut + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * itemsPerBlock;
