host_operator_src = '''
#include "ca6k_00342_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00342CustomTilingData tiling;
    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::Shape &origin = x_storage->GetStorageShape();
    uint32_t N = origin.GetDim(0);
    uint32_t C = origin.GetDim(1);
    uint32_t H = origin.GetDim(2);
    uint32_t W = origin.GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t kernel_size = attrs->GetAttrPointer<uint32_t>(0)[0];
    uint32_t stride = attrs->GetAttrPointer<uint32_t>(1)[0];

    uint32_t H_out = (H - kernel_size) / stride + 1;
    uint32_t W_out = (W - kernel_size) / stride + 1;
    uint32_t totalLength = N * C * H_out * W_out;

    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_kernel_size(kernel_size);
    tiling.set_stride(stride);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::Shape &origin = x_storage->GetStorageShape();
    uint32_t N = origin.GetDim(0);
    uint32_t C = origin.GetDim(1);
    uint32_t H = origin.GetDim(2);
    uint32_t W = origin.GetDim(3);
    // For InferShape we assume default values (2,2) from operator definition.
    uint32_t kernel_size = 2;
    uint32_t stride = 2;
    uint32_t H_out = (H - kernel_size) / stride + 1;
    uint32_t W_out = (W - kernel_size) / stride + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDim(0, static_cast<int64_t>(N));
    y_shape->SetDim(1, static_cast<int64_t>(C));
    y_shape->SetDim(2, static_cast<int64_t>(H_out));
    y_shape->SetDim(3, static_cast<int64_t>(W_out));
    y_shape->SetDimNum(4);
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00342Custom {
public:
    __aicore__ inline KernelCa6k00342Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t H_out, uint32_t W_out,
                                uint32_t kernel_size, uint32_t stride)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        H_out_ = H_out;
        W_out_ = W_out;
        kernelSize_ = kernel_size;
        stride_ = stride;
        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * H * W);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        while (true) {
            uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
            }

            Compute(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void Compute(uint32_t startIdx, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            uint32_t globalIdx = startIdx + i;
            uint32_t n = globalIdx / (C_ * H_out_ * W_out_);
            uint32_t rem = globalIdx % (C_ * H_out_ * W_out_);
            uint32_t c = rem / (H_out_ * W_out_);
            uint32_t rem2 = rem % (H_out_ * W_out_);
            uint32_t h_out = rem2 / W_out_;
            uint32_t w_out = rem2 % W_out_;

            uint32_t h_in = h_out * stride_;
            uint32_t w_in = w_out * stride_;

            uint32_t baseOffset = ((n * C_ + c) * H_ + h_in) * W_ + w_in;

            float v00 = xGm_[baseOffset];
            float v01 = xGm_[baseOffset + 1];
            float v10 = xGm_[baseOffset + W_];
            float v11 = xGm_[baseOffset + W_ + 1];

            float max_val = v00;
            if (v01 > max_val) max_val = v01;
            if (v10 > max_val) max_val = v10;
            if (v11 > max_val) max_val = v11;

            yLocal.SetValue(i, max_val);
        }

        AscendC::DataCopy(yGm_[startIdx], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t H_out_ {0};
    uint32_t W_out_ {0};
    uint32_t kernelSize_ {2};
    uint32_t stride_ {2};
};

extern "C" __global__ __aicore__ void ca6k_00342_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00342Custom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.N,
            tiling_data.C,
            tiling_data.H,
            tiling_data.W,
            tiling_data.H_out,
            tiling_data.W_out,
            tiling_data.kernel_size,
            tiling_data.stride);
    op.Process();
}
'''
