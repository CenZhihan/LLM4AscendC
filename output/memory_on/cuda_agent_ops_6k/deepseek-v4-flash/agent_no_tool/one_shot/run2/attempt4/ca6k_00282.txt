#include "ca6k_00282_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;
static constexpr uint32_t MAX_TILE_ELEMS = 24576;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00282CustomTilingData tiling;

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = shape.GetDim(0);
    uint32_t C = shape.GetDim(1);
    uint32_t D = shape.GetDim(2);
    uint32_t H = shape.GetDim(3);
    uint32_t W = shape.GetDim(4);
    uint32_t spatialSize = D * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps_ptr = attrs->GetAttrPointer<float>(0);
    float eps_val = (eps_ptr != nullptr) ? *eps_ptr : 1e-5f;

    // UB budget: we need 2 buffers of tileLen each for input and output (single buffer per pass)
    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    if (maxTile > MAX_TILE_ELEMS) maxTile = MAX_TILE_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = (spatialSize < maxTile) ? AlignDown(spatialSize, ALIGN_ELEMS) : maxTile;
    if (tileLen == 0) tileLen = spatialSize;

    // One block per channel per sample
    uint32_t blockDim = N * C;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 40) blockDim = 40;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_spatialSize(spatialSize);
    tiling.set_tileLen(tileLen);
    tiling.set_eps(eps_val);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    gert::Shape *y_shape = context->GetOutputShape(0);

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00282Custom {
public:
    __aicore__ inline KernelCa6k00282Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t spatialSize, uint32_t tileLen,
                                float eps)
    {
        N_ = N;
        C_ = C;
        spatialSize_ = spatialSize;
        tileLen_ = tileLen;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * spatialSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * spatialSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (spatialSize_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= N_ * C_) return;

        // Determine channel index within (n,c) grid
        uint32_t n = blockIdx / C_;
        uint32_t c = blockIdx % C_;

        // Base offset for this channel
        uint32_t baseOffset = n * C_ * spatialSize_ + c * spatialSize_;

        // ---------- First pass: compute sum and sumSq ----------
        float sum = 0.0f;
        float sumSq = 0.0f;
        uint32_t remaining = spatialSize_;
        uint32_t offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[baseOffset + offset], curLen);

            // sum of tile
            AscendC::LocalTensor<float> sumLocal = tmpFloat_.AllocTensor<float>();
            AscendC::ReduceAdd(sumLocal, xLocal, curLen);
            sum += sumLocal.GetValue(0);
            tmpFloat_.FreeTensor(sumLocal);

            // squared sum of tile
            AscendC::LocalTensor<float> sqLocal = tmpFloat_.AllocTensor<float>();
            AscendC::Mul(sqLocal, xLocal, xLocal, curLen);
            AscendC::LocalTensor<float> sumSqLocal = tmpFloat_.AllocTensor<float>();
            AscendC::ReduceAdd(sumSqLocal, sqLocal, curLen);
            sumSq += sumSqLocal.GetValue(0);
            tmpFloat_.FreeTensor(sqLocal);
            tmpFloat_.FreeTensor(sumSqLocal);

            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }

        // Compute mean, variance, invStd
        float mean = sum / (float)spatialSize_;
        float variance = sumSq / (float)spatialSize_ - mean * mean;
        float invStd = 1.0f / sqrtf(variance + eps_);

        // ---------- Second pass: write normalized output ----------
        remaining = spatialSize_;
        offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[baseOffset + offset], curLen);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // y = (x - mean) * invStd
            AscendC::Sub(yLocal, xLocal, mean, curLen);
            AscendC::Mul(yLocal, yLocal, invStd, curLen);

            AscendC::DataCopy(yGm_[baseOffset + offset], yLocal, curLen);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpFloat_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t spatialSize_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void ca6k_00282_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00282Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C,
            tiling_data.spatialSize, tiling_data.tileLen,
            tiling_data.eps);
    op.Process();
}
