static ge::graphStatus TilingFunc(gert::TilingContext *context) {

    const auto &origin_shape = context->GetInputShape(0)->GetOriginShape();

    }

    uint32_t reductionLen = static_cast<uint32_t>(origin_shape.GetDim(ndim - 1));
    uint32_t totalRows = 1;
    for (uint32_t i = 0; i < ndim - 1; ++i) {
        totalRows *= static_cast<uint32_t>(origin_shape.GetDim(i));
    }
    if (totalRows == 0 || reductionLen == 0) {
        return ge::GRAPH_FAILED;
    }

    uint32_t maxRowsPerTile = UB_BUDGET_BYTES / (3 * reductionLen * sizeof(float));
    if (maxRowsPerTile == 0) maxRowsPerTile = 1;
    uint32_t tileRows = (totalRows < maxRowsPerTile) ? totalRows : maxRowsPerTile;
    tileRows = AlignDown(tileRows, 1);

    uint32_t blockDim = CeilDiv(totalRows, tileRows);
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;
