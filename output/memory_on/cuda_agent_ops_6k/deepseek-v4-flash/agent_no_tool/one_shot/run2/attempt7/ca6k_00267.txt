#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00267Custom {
public:
    __aicore__ inline KernelCa6k00267Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t stride, int32_t padding, int32_t outputPadding,
                                int32_t dilation, int32_t groups,
                                int32_t batch, int32_t inC, int32_t inH, int32_t inW,
                                int32_t outC, int32_t outH, int32_t outW,
                                int32_t kH, int32_t kW)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        stride_ = stride;
        padding_ = padding;
        outputPadding_ = outputPadding;
        dilation_ = dilation;
        groups_ = groups;
        batch_ = batch;
        inC_ = inC;
        inH_ = inH;
        inW_ = inW;
        outC_ = outC;
        outH_ = outH;
        outW_ = outW;
        kH_ = kH;
        kW_ = kW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * inC * inH * inW);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, inC * outC * kH * kW);
        if (bias != nullptr) {
            hasBias_ = true;
            bGm_.SetGlobalBuffer((__gm__ float*)bias, outC);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        const int32_t maxTileHW = tileLen_ / (batch * outC) + 1;
        const int32_t bufSize = maxTileHW * outC * batch * sizeof(float);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, bufSize);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, bufSize);
        pipe_.InitBuffer(tmpQueue_, 1, bufSize);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            ComputeTile(base, curLen);

            base += strideElems;
            if (base >= totalLen_) break;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t outOffset, uint32_t len)
    {
        for (uint32_t idx = 0; idx < len; idx++) {
            uint32_t globalIdx = outOffset + idx;
            int32_t b = globalIdx / (outC_ * outH_ * outW_);
            int32_t rest = globalIdx % (outC_ * outH_ * outW_);
            int32_t oc = rest / (outH_ * outW_);
            int32_t rest2 = rest % (outH_ * outW_);
            int32_t oh = rest2 / outW_;
            int32_t ow = rest2 % outW_;

            float sum = 0.0f;
            for (int32_t ic = 0; ic < inC_; ic++) {
                for (int32_t kh = 0; kh < kH_; kh++) {
                    int32_t ih = (oh + padding_ - (outputPadding_ - 1)) / stride_ - kh * dilation_;
                    if (ih < 0) continue;
                    if (ih >= inH_) continue;
                    for (int32_t kw = 0; kw < kW_; kw++) {
                        int32_t iw = (ow + padding_ - (outputPadding_ - 1)) / stride_ - kw * dilation_;
                        if (iw < 0) continue;
                        if (iw >= inW_) continue;

                        float w = wGm_.GetValue(ic * outC_ * kH_ * kW_ + oc * kH_ * kW_ + kh * kW_ + kw);
                        float x = xGm_.GetValue(b * inC_ * inH_ * inW_ + ic * inH_ * inW_ + ih * inW_ + iw);
                        sum += x * w;
                    }
                }
            }
            if (hasBias_) {
                sum += bGm_.GetValue(oc);
            }
            yGm_.SetValue(globalIdx, sum);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t stride_ {2};
    int32_t padding_ {1};
    int32_t outputPadding_ {1};
    int32_t dilation_ {1};
    int32_t groups_ {1};
    int32_t batch_ {1};
    int32_t inC_ {1};
    int32_t inH_ {1};
    int32_t inW_ {1};
    int32_t outC_ {1};
    int32_t outH_ {1};
    int32_t outW_ {1};
    int32_t kH_ {1};
    int32_t kW_ {1};
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void ca6k_00267_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00267Custom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.stride, tiling_data.padding, tiling_data.outputPadding,
            tiling_data.dilation, tiling_data.groups,
            tiling_data.batch, tiling_data.inC, tiling_data.inH, tiling_data.inW,
            tiling_data.outC, tiling_data.outH, tiling_data.outW,
            tiling_data.kH, tiling_data.kW);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import math

_FN = 'ca6k_00267_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias):
        super(ModelNew, self).__init__()
        self.out_channels = out_channels
        self.kernel_size = kernel_size if isinstance(kernel_size, tuple) else (kernel_size, kernel_size)
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.bias_flag = bias
        self.initialized = False

    def forward(self, x):
        if not self.initialized:
            in_channels = x.shape[1]
            self.weight = torch.nn.Parameter(
                torch.empty(in_channels, self.out_channels, *self.kernel_size, device=x.device, dtype=x.dtype)
            )
            torch.nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
            if self.bias_flag:
                self.bias = torch.nn.Parameter(torch.empty(self.out_channels, device=x.device, dtype=x.dtype))
                fan_in = self.weight.shape[1] * self.weight.shape[2] * self.weight.shape[3]
                bound = 1 / math.sqrt(fan_in)
                torch.nn.init.uniform_(self.bias, -bound, bound)
            else:
                self.bias = None
            self.initialized = True
        fn = getattr(custom_ops_lib, _FN)
        if self.bias is not None:
            return fn(x, self.weight, self.bias, self.stride, self.padding, self.output_padding, 1, 1)
        else:
            return fn(x, self.weight, None, self.stride, self.padding, self.output_padding, 1, 1)
