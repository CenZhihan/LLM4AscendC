#include "ca6k_00285_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00285CustomTilingData tiling;
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const uint32_t inputHeight = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t inputWidth = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));
    const uint32_t inputChannels = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t batch = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *upscaleFactorPtr = attrs->GetAttrPointer<int32_t>(0);
    uint32_t upscaleFactor = static_cast<uint32_t>(*upscaleFactorPtr);

    uint32_t outputHeight = inputHeight / upscaleFactor;
    uint32_t outputWidth = inputWidth / upscaleFactor;
    uint32_t outputChannels = inputChannels * upscaleFactor * upscaleFactor;
    uint32_t totalLength = batch * outputChannels * outputHeight * outputWidth;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_inputHeight(inputHeight);
    tiling.set_inputWidth(inputWidth);
    tiling.set_inputChannels(inputChannels);
    tiling.set_outputHeight(outputHeight);
    tiling.set_outputWidth(outputWidth);
    tiling.set_upscaleFactor(upscaleFactor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const uint32_t inputHeight = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t inputWidth = static_cast<uint32_t>(x_shape->GetDim(3));
    const uint32_t inputChannels = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t batch = static_cast<uint32_t>(x_shape->GetDim(0));

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00285Custom {
public:
    __aicore__ inline KernelCa6k00285Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inputHeight, uint32_t inputWidth,
                                uint32_t inputChannels, uint32_t outputHeight,
                                uint32_t outputWidth, uint32_t upscaleFactor)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inputHeight_ = inputHeight;
        inputWidth_ = inputWidth;
        inputChannels_ = inputChannels;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        upscaleFactor_ = upscaleFactor;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t outLinIdx = base + i;
                uint32_t n = outLinIdx / (outputHeight_ * outputWidth_ * (inputChannels_ * upscaleFactor_ * upscaleFactor_));
                uint32_t rem1 = outLinIdx % (outputHeight_ * outputWidth_ * (inputChannels_ * upscaleFactor_ * upscaleFactor_));
                uint32_t c_out = rem1 / (outputHeight_ * outputWidth_);
                uint32_t rem2 = rem1 % (outputHeight_ * outputWidth_);
                uint32_t h_out = rem2 / outputWidth_;
                uint32_t w_out = rem2 % outputWidth_;

                uint32_t c_in = c_out % inputChannels_;
                uint32_t c_accum = c_out / inputChannels_;
                uint32_t h_in = h_out * upscaleFactor_ + c_accum / upscaleFactor_;
                uint32_t w_in = w_out * upscaleFactor_ + c_accum % upscaleFactor_;

                uint32_t inLinIdx = ((n * inputChannels_ + c_in) * inputHeight_ + h_in) * inputWidth_ + w_in;

                float val = xGm_.GetValue(inLinIdx);
                yGm_.SetValue(base + i, val);
            }

            base += strideElems;
            if (base >= totalLen_) break;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t inputHeight_ {0};
    uint32_t inputWidth_ {0};
    uint32_t inputChannels_ {0};
    uint32_t outputHeight_ {0};
    uint32_t outputWidth_ {0};
    uint32_t upscaleFactor_ {1};
};

extern "C" __global__ __aicore__ void ca6k_00285_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00285Custom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.inputHeight,
            tiling_data.inputWidth,
            tiling_data.inputChannels,
            tiling_data.outputHeight,
            tiling_data.outputWidth,
            tiling_data.upscaleFactor);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor ca6k_00285_impl_npu(const at::Tensor& self, int64_t upscale_factor) {
    int64_t C = self.size(1);
    int64_t H = self.size(2);
    int64_t W = self.size(3);
    int64_t f = upscale_factor;
    int64_t outC = C * f * f;
    int64_t outH = H / f;
    int64_t outW = W / f;
    at::Tensor result = at::empty({self.size(0), outC, outH, outW}, self.options());
    EXEC_NPU_CMD(aclnnCa6k00285Custom, self, upscale_factor, result);
    return result;
}

    m.impl("ca6k_00285_custom", &ca6k_00285_impl_npu);
