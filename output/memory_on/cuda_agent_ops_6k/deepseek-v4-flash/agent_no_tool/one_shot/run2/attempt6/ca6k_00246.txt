#include "ca6k00246_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00246CustomTilingData tiling;
    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &inShape = storageShape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(inShape.GetDim(0));
    uint32_t inC = static_cast<uint32_t>(inShape.GetDim(1));
    uint32_t inH = static_cast<uint32_t>(inShape.GetDim(2));
    uint32_t inW = static_cast<uint32_t>(inShape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t upscaleFactor = static_cast<uint32_t>(attrs->GetAttrPointer<int>(0)[0]);

    uint32_t outC = inC / (upscaleFactor * upscaleFactor);
    uint32_t outH = inH * upscaleFactor;
    uint32_t outW = inW * upscaleFactor;
    uint32_t totalOutLen = N * outC * outH * outW;

    uint32_t maxTile = UB_BUDGET_BYTES / 4 / 2;  // read + write = 2 floats per element
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = (totalOutLen < maxTile) ? totalOutLen : maxTile;

    uint32_t blockDim = 1;  // single block for simplicity
    context->SetBlockDim(blockDim);

    // Direct member assignment (no setter methods)
    tiling.N = N;
    tiling.inC = inC;
    tiling.inH = inH;
    tiling.inW = inW;
    tiling.upscaleFactor = upscaleFactor;
    tiling.outC = outC;
    tiling.outH = outH;
    tiling.outW = outW;
    tiling.totalOutLen = totalOutLen;
    tiling.tileLen = tileLen;
    tiling.blockDim = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *inShape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    int64_t N = inShape->GetDim(0);
    int64_t inC = inShape->GetDim(1);
    int64_t inH = inShape->GetDim(2);
    int64_t inW = inShape->GetDim(3);
    int64_t outC = inC / (upscaleFactor * upscaleFactor);
    int64_t outH = inH * upscaleFactor;

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00246Custom {
public:
    __aicore__ inline KernelCa6k00246Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t inC, uint32_t inH, uint32_t inW,
                                uint32_t upscaleFactor,
                                uint32_t outC, uint32_t outH, uint32_t outW,
                                uint32_t totalOutLen, uint32_t tileLen) {
        N_ = N;
        inC_ = inC;
        inH_ = inH;
        inW_ = inW;
        upscaleFactor_ = upscaleFactor;
        outC_ = outC;
        outH_ = outH;
        outW_ = outW;
        totalOutLen_ = totalOutLen;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * inC_ * inH_ * inW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);
    }

    __aicore__ inline void Process() {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t stride = blockNum * tileLen_;
        uint32_t start = blockIdx * tileLen_;
        if (start >= totalOutLen_) return;

        for (uint32_t i = start; i < totalOutLen_; i += stride) {
            uint32_t curLen = (i + tileLen_ <= totalOutLen_) ? tileLen_ : (totalOutLen_ - i);
            for (uint32_t j = 0; j < curLen; ++j) {
                uint32_t outIdx = i + j;
                // compute output coordinates
                uint32_t n = outIdx / (outC_ * outH_ * outW_);
                uint32_t rest = outIdx % (outC_ * outH_ * outW_);
                uint32_t cOut = rest / (outH_ * outW_);
                rest = rest % (outH_ * outW_);
                uint32_t hOut = rest / outW_;
                uint32_t wOut = rest % outW_;

                // compute input coordinates
                uint32_t inC = cOut * upscaleFactor_ * upscaleFactor_ +
                               (hOut % upscaleFactor_) * upscaleFactor_ +
                               (wOut % upscaleFactor_);
                uint32_t inH = hOut / upscaleFactor_;
                uint32_t inW = wOut / upscaleFactor_;

                uint32_t inIdx = ((n * inC_ + inC) * inH_ + inH) * inW_ + inW;
                yGm_.SetValue(outIdx, xGm_.GetValue(inIdx));
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t inC_ {0};
    uint32_t inH_ {0};
    uint32_t inW_ {0};
    uint32_t upscaleFactor_ {1};
    uint32_t outC_ {0};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
    uint32_t totalOutLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00246_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00246Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.inC, tiling_data.inH, tiling_data.inW,
            tiling_data.upscaleFactor,
            tiling_data.outC, tiling_data.outH, tiling_data.outW,
            tiling_data.totalOutLen, tiling_data.tileLen);
    op.Process();
}
"""
