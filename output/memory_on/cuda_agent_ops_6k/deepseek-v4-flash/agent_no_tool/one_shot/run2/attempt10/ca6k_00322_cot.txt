{
    if (totalLen_ == 0 || tileLen_ == 0) return;
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t strideElems = blockNum * tileLen_;
    uint32_t base = blockIdx * tileLen_;
    if (base >= totalLen_) return;

    const uint32_t nc = C_ * H_ * W_;
    const uint32_t hw = H_ * W_;

    while (base < totalLen_) {
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        // Allocate output buffer
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Fill with permuted values
        for (uint32_t i = 0; i < curLen; i++) {
            uint32_t idx = base + i;
            uint32_t n = idx / nc;
            uint32_t rem = idx % nc;
            uint32_t c_out = rem / hw;
            uint32_t hwRem = rem % hw;
            uint32_t h = hwRem / W_;
            uint32_t w = hwRem % W_;

            uint32_t c_in = (c_out % groups_) * Cg_ + (c_out / groups_);
            uint32_t inIdx = n * nc + c_in * hw + h * W_ + w;
            float val = xGm_.GetValue(inIdx);
            yLocal.SetValue(i, val);
        }

        // Copy to global output
        AscendC::DataCopy(yGm_[base], yLocal, curLen);

        outQueueY_.FreeTensor(yLocal);

        base += strideElems;
    }
}
