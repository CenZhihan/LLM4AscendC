#include "ca6k_00271_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Ca6k00271CustomTilingData tiling;
    const gert::StorageShape *input_shape = context->GetInputShape(0);
    const uint32_t n = static_cast<uint32_t>(input_shape->GetOriginShape().GetDim(0));
    const uint32_t c = static_cast<uint32_t>(input_shape->GetOriginShape().GetDim(1));
    const uint32_t id = static_cast<uint32_t>(input_shape->GetOriginShape().GetDim(2));
    const uint32_t ih = static_cast<uint32_t>(input_shape->GetOriginShape().GetDim(3));
    const uint32_t iw = static_cast<uint32_t>(input_shape->GetOriginShape().GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const std::vector<int64_t> *output_size_vec = attrs->GetListInt(0);
    uint32_t od = static_cast<uint32_t>((*output_size_vec)[0]);
    uint32_t oh = static_cast<uint32_t>((*output_size_vec)[1]);
    uint32_t ow = static_cast<uint32_t>((*output_size_vec)[2]);

    uint32_t totalLength = n * c * id * ih * iw;
    uint32_t outputTotal = n * c * od * oh * ow;

    // Tile based on per-(batch,channel) processing: each tile processes one (n,c) slice
    // We can process multiple (n,c) slices per block; use simple block dim.
    uint32_t tileLen = totalLength;  // not used per (n,c) slice; we'll tile by slices
    // For UB budget: we need to handle one (n,c) slice of input and one slice of output.
    uint32_t sliceInputSize = id * ih * iw;
    uint32_t sliceOutputSize = od * oh * ow;
    uint32_t neededUB = (sliceInputSize + sliceOutputSize) * sizeof(float) * 2; // double buffer
    if (neededUB > UB_BUDGET_BYTES) {
        // fallback: reduce tile (not implemented for simplicity)
        tileLen = sliceInputSize;
    } else {
        tileLen = sliceInputSize;
    }
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = ALIGN_ELEMS;

    uint32_t numSlices = n * c;
    uint32_t blockDim = numSlices;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_id(id);
    tiling.set_ih(ih);
    tiling.set_iw(iw);
    tiling.set_od(od);
    tiling.set_oh(oh);
    tiling.set_ow(ow);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = r"""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00271Custom {
public:
    __aicore__ inline KernelCa6k00271Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t n, uint32_t c,
                                uint32_t id, uint32_t ih, uint32_t iw,
                                uint32_t od, uint32_t oh, uint32_t ow) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        n_ = n;
        c_ = c;
        id_ = id;
        ih_ = ih;
        iw_ = iw;
        od_ = od;
        oh_ = oh;
        ow_ = ow;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * c_ * od_ * oh_ * ow_);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, id_ * ih_ * iw_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, od_ * oh_ * ow_ * sizeof(float));
        pipe_.InitBuffer(scalarBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t totalSlices = n_ * c_;

        if (blockIdx >= totalSlices) return;

        // Each block processes a subset of (n,c) slices
        for (uint32_t idx = blockIdx; idx < totalSlices; idx += blockNum) {
            uint32_t n_idx = idx / c_;
            uint32_t c_idx = idx % c_;

            // Load input slice
            uint32_t inputOffset = (n_idx * c_ + c_idx) * id_ * ih_ * iw_;
            uint32_t outputOffset = (n_idx * c_ + c_idx) * od_ * oh_ * ow_;

            // Compute adaptive average pool
            ComputeSlice(inputOffset, outputOffset);
        }
    }

private:
    __aicore__ inline void ComputeSlice(uint32_t inputOffset, uint32_t outputOffset) {
        // Load entire input slice into local
        AscendC::LocalTensor<float> inLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(inLocal, xGm_[inputOffset], id_ * ih_ * iw_);
        inQueueX_.EnQue(inLocal);

        // Compute output
        AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t od_i = 0; od_i < od_; od_i++) {
            for (uint32_t oh_i = 0; oh_i < oh_; oh_i++) {
                for (uint32_t ow_i = 0; ow_i < ow_; ow_i++) {
                    // Compute input range
                    int32_t start_d = (od_i * id_ + 0) / od_;
                    int32_t end_d = ((od_i + 1) * id_ + od_ - 1) / od_;
                    int32_t start_h = (oh_i * ih_ + 0) / oh_;
                    int32_t end_h = ((oh_i + 1) * ih_ + oh_ - 1) / oh_;
                    int32_t start_w = (ow_i * iw_ + 0) / ow_;
                    int32_t end_w = ((ow_i + 1) * iw_ + ow_ - 1) / ow_;

                    if (end_d > id_) end_d = id_;
                    if (end_h > ih_) end_h = ih_;
                    if (end_w > iw_) end_w = iw_;

                    uint32_t count = (end_d - start_d) * (end_h - start_h) * (end_w - start_w);
                    float sum = 0.0f;

                    // Reduction using local memory
                    AscendC::LocalTensor<float> inDeque = inQueueX_.DeQue<float>();
                    for (int32_t d = start_d; d < end_d; d++) {
                        for (int32_t h = start_h; h < end_h; h++) {
                            for (int32_t w = start_w; w < end_w; w++) {
                                uint32_t idx = d * ih_ * iw_ + h * iw_ + w;
                                sum += inDeque(idx);
                            }
                        }
                    }
                    inQueueX_.EnQue(inDeque);

                    float avg = sum / (float)count;
                    uint32_t outIdx = od_i * oh_ * ow_ + oh_i * ow_ + ow_i;
                    outLocal(outIdx) = avg;
                }
            }
        }

        // Write output
        outQueueY_.EnQue(outLocal);
        AscendC::LocalTensor<float> outDeque = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[outputOffset], outDeque, od_ * oh_ * ow_);
        outQueueY_.FreeTensor(outDeque);

        // Free input
        AscendC::LocalTensor<float> inDeque2 = inQueueX_.DeQue<float>();
        inQueueX_.FreeTensor(inDeque2);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBuf_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t n_ {0};
    uint32_t c_ {0};
    uint32_t id_ {0};
    uint32_t ih_ {0};
    uint32_t iw_ {0};
    uint32_t od_ {0};
    uint32_t oh_ {0};
    uint32_t ow_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00271_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00271Custom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.c,
            tiling_data.id, tiling_data.ih, tiling_data.iw,
            tiling_data.od, tiling_data.oh, tiling_data.ow);
    op.Process();
}
"""
