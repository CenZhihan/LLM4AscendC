#include "kernel_operator.h"

using namespace AscendC;

class KernelCa6k00317Custom {
public:
    __aicore__ inline KernelCa6k00317Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                Ca6k00317CustomTilingData tiling)
    {
        tiling_ = tiling;

        n_ = tiling_.n;
        c_in_ = tiling_.c_in;
        c_out_ = tiling_.c_out;
        d_ = tiling_.d;
        h_ = tiling_.h;
        w_ = tiling_.w;
        kd_ = tiling_.kd;
        kh_ = tiling_.kh;
        kw_ = tiling_.kw;
        stride_d_ = tiling_.stride_d;
        stride_h_ = tiling_.stride_h;
        stride_w_ = tiling_.stride_w;
        pad_d_ = tiling_.pad_d;
        pad_h_ = tiling_.pad_h;
        pad_w_ = tiling_.pad_w;
        out_pad_d_ = tiling_.out_pad_d;
        out_pad_h_ = tiling_.out_pad_h;
        out_pad_w_ = tiling_.out_pad_w;
        dilation_d_ = tiling_.dilation_d;
        dilation_h_ = tiling_.dilation_h;
        dilation_w_ = tiling_.dilation_w;
        groups_ = tiling_.groups;

        out_d_ = tiling_.out_d;
        out_h_ = tiling_.out_h;
        out_w_ = tiling_.out_w;

        totalOutElems_ = n_ * c_out_ * out_d_ * out_h_ * out_w_;
        totalInElems_ = n_ * c_in_ * d_ * h_ * w_;
        totalWeightElems_ = c_out_ * c_in_ * kd_ * kh_ * kw_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInElems_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, totalWeightElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutElems_);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t on = 0; on < n_; on++) {
            for (uint32_t oc = 0; oc < c_out_; oc++) {
                for (uint32_t od = 0; od < out_d_; od++) {
                    for (uint32_t oh = 0; oh < out_h_; oh++) {
                        for (uint32_t ow = 0; ow < out_w_; ow++) {
                            float sum = 0.0f;
                            for (uint32_t ic = 0; ic < c_in_; ic++) {
                                for (uint32_t kdi = 0; kdi < kd_; kdi++) {
                                    int32_t id = static_cast<int32_t>(od) - static_cast<int32_t>(pad_d_) + static_cast<int32_t>(kdi) * static_cast<int32_t>(dilation_d_);
                                    if (id % static_cast<int32_t>(stride_d_) != 0) continue;
                                    id = id / static_cast<int32_t>(stride_d_);
                                    if (id < 0 || id >= static_cast<int32_t>(d_)) continue;
                                    for (uint32_t khi = 0; khi < kh_; khi++) {
                                        int32_t ih = static_cast<int32_t>(oh) - static_cast<int32_t>(pad_h_) + static_cast<int32_t>(khi) * static_cast<int32_t>(dilation_h_);
                                        if (ih % static_cast<int32_t>(stride_h_) != 0) continue;
                                        ih = ih / static_cast<int32_t>(stride_h_);
                                        if (ih < 0 || ih >= static_cast<int32_t>(h_)) continue;
                                        for (uint32_t kwi = 0; kwi < kw_; kwi++) {
                                            int32_t iw = static_cast<int32_t>(ow) - static_cast<int32_t>(pad_w_) + static_cast<int32_t>(kwi) * static_cast<int32_t>(dilation_w_);
                                            if (iw % static_cast<int32_t>(stride_w_) != 0) continue;
                                            iw = iw / static_cast<int32_t>(stride_w_);
                                            if (iw < 0 || iw >= static_cast<int32_t>(w_)) continue;
                                            float x_val = xGm_.GetValue(on * c_in_ * d_ * h_ * w_ + ic * d_ * h_ * w_ + static_cast<uint32_t>(id) * h_ * w_ + static_cast<uint32_t>(ih) * w_ + static_cast<uint32_t>(iw));
                                            // weight shape: (c_out, c_in, kd, kh, kw)
                                            float w_val = wGm_.GetValue(oc * c_in_ * kd_ * kh_ * kw_ + ic * kd_ * kh_ * kw_ + kdi * kh_ * kw_ + khi * kw_ + kwi);
                                            sum += x_val * w_val;
                                        }
                                    }
                                }
                            }
                            yGm_.SetValue(on * c_out_ * out_d_ * out_h_ * out_w_ + oc * out_d_ * out_h_ * out_w_ + od * out_h_ * out_w_ + oh * out_w_ + ow, sum);
                        }
                    }
                }
            }
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    Ca6k00317CustomTilingData tiling_;

    uint32_t n_, c_in_, c_out_, d_, h_, w_;
    uint32_t kd_, kh_, kw_;
    uint32_t stride_d_, stride_h_, stride_w_;
    uint32_t pad_d_, pad_h_, pad_w_;
    uint32_t out_pad_d_, out_pad_h_, out_pad_w_;
    uint32_t dilation_d_, dilation_h_, dilation_w_;
    uint32_t groups_;
    uint32_t out_d_, out_h_, out_w_;
    uint32_t totalOutElems_, totalInElems_, totalWeightElems_;
};

extern "C" __global__ __aicore__ void ca6k00317_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00317Custom op;
    op.Init(x, weight, y, tiling_data);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib
import math

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias):
        super(ModelNew, self).__init__()
        # Weight shape following PyTorch's ConvTranspose3d: (out_channels, in_channels, kd, kh, kw)

        # Use same initialization as nn.ConvTranspose3d (kaiming_uniform with a=sqrt(5))
        torch.nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.dilation = (1, 1, 1)
        self.groups = 1

    def forward(self, x):

            x, self.weight, self.stride, self.padding,
