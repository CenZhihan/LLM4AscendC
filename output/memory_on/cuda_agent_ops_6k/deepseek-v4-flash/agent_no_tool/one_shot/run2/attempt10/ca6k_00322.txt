#include "ca6k00322_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Ca6k00322CustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const auto &originShape = x_shape->GetOriginShape();
    const uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    const uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    const uint32_t H = static_cast<uint32_t>(originShape.GetDim(2));
    const uint32_t W = static_cast<uint32_t>(originShape.GetDim(3));
    const uint32_t totalLength = N * C * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *groupsPtr = attrs->GetAttrPointer<int32_t>(0);
    uint32_t groups = static_cast<uint32_t>(*groupsPtr);

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_groups(groups);
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

class KernelCa6k00322Custom {
public:
    __aicore__ inline KernelCa6k00322Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t groups, uint32_t totalLength,
                                uint32_t tileLength)
    {
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        groups_ = groups;
        Cg_ = C / groups;
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        const uint32_t nc = C_ * H_ * W_;
        const uint32_t hw = H_ * W_;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t idx = base + i;
                uint32_t n = idx / nc;
                uint32_t rem = idx % nc;
                uint32_t c_out = rem / hw;
                uint32_t hwRem = rem % hw;
                uint32_t h = hwRem / W_;
                uint32_t w = hwRem % W_;

                uint32_t c_in = (c_out % groups_) * Cg_ + (c_out / groups_);
                uint32_t inIdx = n * nc + c_in * hw + h * W_ + w;

                float val = xGm_.GetValue(inIdx);
                yLocal.SetValue(i, val);
            }

            AscendC::DataCopy(yGm_[base], yLocal, curLen);

            outQueueY_.FreeTensor(yLocal);

            base += strideElems;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t groups_ {2};
    uint32_t Cg_ {0};
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};

    static constexpr int32_t BUFFER_NUM = 2;
};

extern "C" __global__ __aicore__ void ca6k_00322_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00322Custom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.groups, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
