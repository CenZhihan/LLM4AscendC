kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00113Custom {
public:
    __aicore__ inline KernelCa6k00113Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t dimAttr, uint32_t innerDim,
                                uint32_t outputTotalLength) {
        dimAttr_ = dimAttr;
        innerDim_ = innerDim;
        outputTotalLen_ = outputTotalLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y);
    }

    __aicore__ inline void Process() {
        if (outputTotalLen_ == 0 || innerDim_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Each block processes a contiguous portion of output rows
        uint32_t rowsPerBlock = (outputTotalLen_ + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        if (startRow >= outputTotalLen_) return;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > outputTotalLen_) endRow = outputTotalLen_;

        // For each output row, scan innerDim columns to find argmax
        for (uint32_t r = startRow; r < endRow; ++r) {
            uint32_t baseOffset = r * innerDim_;
            float maxVal = xGm_[baseOffset];
            int64_t maxIdx = 0;
            for (uint32_t c = 1; c < innerDim_; ++c) {
                float val = xGm_[baseOffset + c];
                if (val > maxVal) {
                    maxVal = val;
                    maxIdx = static_cast<int64_t>(c);
                }
            }
            yGm_[r] = maxIdx;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t dimAttr_ {1};
    uint32_t innerDim_ {1};
    uint32_t outputTotalLen_ {0};
};

extern "C" __global__ __aicore__ void ca6k_00113_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00113Custom op;
    op.Init(x, y, tiling_data.dimAttr, tiling_data.innerDim, tiling_data.outputTotalLength);
    op.Process();
}
