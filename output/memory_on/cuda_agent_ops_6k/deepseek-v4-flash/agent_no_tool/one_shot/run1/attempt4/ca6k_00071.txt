#include "ca6k00071_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k00071CustomTilingData tiling;

  const gert::StorageShape *x_storage_shape = context->GetInputShape(0);
  const gert::Shape &x_shape = x_storage_shape->GetOriginShape();
  uint32_t ndim = x_shape.GetDimNum();
  uint32_t rows = (ndim >= 2) ? static_cast<uint32_t>(x_shape.GetDim(0)) : 1;
  uint32_t cols = (ndim >= 2) ? static_cast<uint32_t>(x_shape.GetDim(1)) : static_cast<uint32_t>(x_shape.GetDim(0));
  uint32_t totalLength = rows * cols;

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
  if (dim < 0) dim += ndim;
  uint32_t reductionDim = static_cast<uint32_t>(dim);

  uint32_t reductionSize, independentSize;
  if (reductionDim == 0) {
    reductionSize = rows;
    independentSize = cols;
  } else {
    reductionSize = cols;
    independentSize = rows;
  }

  uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
  uint32_t tileLen = maxTile;
  if (reductionSize < tileLen) {
    tileLen = AlignDown(reductionSize, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = reductionSize;
  }

  uint32_t blockDim;
  if (independentSize <= 24) {
    blockDim = independentSize;
  } else {
    blockDim = 24;
  }
  context->SetBlockDim(blockDim);

  tiling.set_totalLength(totalLength);
  tiling.set_rows(rows);
  tiling.set_cols(cols);
  tiling.set_reductionDim(reductionDim);
  tiling.set_reductionSize(reductionSize);
  tiling.set_independentSize(independentSize);
  tiling.set_tileLength(tileLen);
  tiling.set_blockDim(blockDim);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00071Custom {
public:
  __aicore__ inline KernelCa6k00071Custom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                              uint32_t totalLength, uint32_t rows, uint32_t cols,
                              uint32_t reductionDim, uint32_t reductionSize,
                              uint32_t independentSize, uint32_t tileLength,
                              uint32_t blockDim) {
    xGm_.SetGlobalBuffer((__gm__ float *)x, totalLength);
    yGm_.SetGlobalBuffer((__gm__ float *)y, totalLength);
    rows_ = rows;
    cols_ = cols;
    reductionDim_ = reductionDim;
    reductionSize_ = reductionSize;
    independentSize_ = independentSize;
    tileLen_ = tileLength;
    blockDim_ = blockDim;

    if (reductionDim_ == 1 && tileLen_ > 0) {
      pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
      pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }
  }

  __aicore__ inline void Process() {
    uint32_t blockIdx = AscendC::GetBlockIdx();
    if (blockIdx >= blockDim_) return;

    uint32_t indStart, indEnd;
    uint32_t indChunk = (independentSize_ + blockDim_ - 1) / blockDim_;
    indStart = blockIdx * indChunk;
    indEnd = indStart + indChunk;
    if (indEnd > independentSize_) indEnd = independentSize_;
    if (indStart >= independentSize_) return;

    if (reductionDim_ == 0) {
      for (uint32_t c = indStart; c < indEnd; c++) {
        float running = 0.0f;
        for (uint32_t r = 0; r < reductionSize_; r++) {
          uint32_t idx = r * cols_ + c;
          float val = xGm_.GetValue(idx);
          running += val;
          yGm_.SetValue(idx, running);
        }
      }
    } else {
      for (uint32_t r = indStart; r < indEnd; r++) {
        uint32_t row_base = r * cols_;
        float running = 0.0f;
        uint32_t remaining = cols_;
        uint32_t offset = 0;
        while (remaining > 0) {
          uint32_t curTile = (remaining < tileLen_) ? remaining : tileLen_;
          AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
          AscendC::DataCopy(xLocal, xGm_[row_base + offset], curTile);
          AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
          for (uint32_t i = 0; i < curTile; i++) {
            float v = xLocal(i);
            running += v;
            yLocal(i) = running;
          }
          AscendC::DataCopy(yGm_[row_base + offset], yLocal, curTile);
          inQueueX_.FreeTensor(xLocal);
          outQueueY_.FreeTensor(yLocal);
          offset += curTile;
          remaining -= curTile;
        }
      }
    }
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> yGm_;

  uint32_t rows_;
  uint32_t cols_;
  uint32_t reductionDim_;
  uint32_t reductionSize_;
  uint32_t independentSize_;
  uint32_t tileLen_;
  uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void ca6k00071_custom(GM_ADDR x, GM_ADDR y,
                                                        GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelCa6k00071Custom op;
  op.Init(x, y,
          tiling_data.totalLength, tiling_data.rows, tiling_data.cols,
          tiling_data.reductionDim, tiling_data.reductionSize,
          tiling_data.independentSize, tiling_data.tileLength,
          tiling_data.blockDim);
  op.Process();
}
"""
