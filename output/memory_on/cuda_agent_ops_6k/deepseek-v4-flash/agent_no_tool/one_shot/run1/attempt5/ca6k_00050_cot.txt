class KernelCa6k00050Custom {
public:
    __aicore__ inline KernelCa6k00050Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t N, uint32_t C_in, uint32_t C_out,
                                uint32_t H_in, uint32_t W_in,
                                uint32_t H_out, uint32_t W_out,
                                uint32_t kernel_h, uint32_t kernel_w,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t dilation_h, uint32_t dilation_w,
                                uint32_t patchSize)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C_in * H_in * W_in);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, C_out * C_in * kernel_h * kernel_w);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C_out * H_out * W_out);
        N_ = N; C_in_ = C_in; C_out_ = C_out;
        H_in_ = H_in; W_in_ = W_in;
        H_out_ = H_out; W_out_ = W_out;
        kernel_h_ = kernel_h; kernel_w_ = kernel_w;
        stride_h_ = stride_h; stride_w_ = stride_w;
        pad_h_ = pad_h; pad_w_ = pad_w;
        dilation_h_ = dilation_h; dilation_w_ = dilation_w;
        patchSize_ = patchSize;
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_out_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t totalOut = N_ * C_out_ * H_out_ * W_out_;
        uint32_t perBlock = (totalOut + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * perBlock;
        uint32_t end = (blockIdx + 1) * perBlock;
        if (start >= totalOut) return;
        if (end > totalOut) end = totalOut;

        for (uint32_t idx = start; idx < end; ++idx) {
            uint32_t remaining = idx;
            uint32_t ow = remaining % W_out_; remaining /= W_out_;
            uint32_t oh = remaining % H_out_; remaining /= H_out_;
            uint32_t co = remaining % C_out_; remaining /= C_out_;
            uint32_t n = remaining;

            float sum = 0.0f;
            for (uint32_t kh = 0; kh < kernel_h_; ++kh) {
                for (uint32_t kw = 0; kw < kernel_w_; ++kw) {
                    int32_t h_in = static_cast<int32_t>(oh * stride_h_) - static_cast<int32_t>(pad_h_)
                                   + static_cast<int32_t>(kh * dilation_h_);
                    int32_t w_in = static_cast<int32_t>(ow * stride_w_) - static_cast<int32_t>(pad_w_)
                                   + static_cast<int32_t>(kw * dilation_w_);
                    if (h_in < 0 || static_cast<uint32_t>(h_in) >= H_in_ ||
                        w_in < 0 || static_cast<uint32_t>(w_in) >= W_in_) continue;
                    uint32_t base_in = n * C_in_ * H_in_ * W_in_ + static_cast<uint32_t>(h_in) * W_in_ + static_cast<uint32_t>(w_in);
                    uint32_t base_w = co * C_in_ * kernel_h_ * kernel_w_ + kh * kernel_w_ * C_in_ + kw * C_in_;
                    for (uint32_t ci = 0; ci < C_in_; ++ci) {
                        sum += xGm_[base_in + ci * H_in_ * W_in_] * wGm_[base_w + ci];
                    }
                }
            }
            yGm_[idx] = sum;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_in_, C_out_, H_in_, W_in_, H_out_, W_out_;
    uint32_t kernel_h_, kernel_w_, stride_h_, stride_w_, pad_h_, pad_w_, dilation_h_, dilation_w_;
    uint32_t patchSize_;
};
