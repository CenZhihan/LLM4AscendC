static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  Ca6k00071CustomTilingData tiling;

  const gert::StorageShape *x_storage_shape = context->GetInputShape(0);
  const gert::Shape &x_shape = x_storage_shape->GetOriginShape();
  uint32_t ndim = x_shape.GetDimNum();
  uint32_t rows = (ndim >= 2) ? static_cast<uint32_t>(x_shape.GetDim(0)) : 1;
  uint32_t cols = (ndim >= 2) ? static_cast<uint32_t>(x_shape.GetDim(1)) : static_cast<uint32_t>(x_shape.GetDim(0));
  uint32_t totalLength = rows * cols;

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
  if (dim < 0) dim += ndim;
  uint32_t reductionDim = static_cast<uint32_t>(dim);

  // Use a single block to avoid parallelism issues
  uint32_t blockDim = 1;
  context->SetBlockDim(blockDim);

  uint32_t tileLen = 0;
  if (reductionDim == 1) {
    // contiguous along cols, load entire row per tile
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    if (cols <= maxTile) {
      tileLen = cols;
    } else {
      tileLen = maxTile;
    }
  }

  tiling.set_totalLength(totalLength);
  tiling.set_rows(rows);
  tiling.set_cols(cols);
  tiling.set_reductionDim(reductionDim);
  tiling.set_tileLen(tileLen);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00071Custom {
public:
  __aicore__ inline KernelCa6k00071Custom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                              uint32_t totalLength, uint32_t rows, uint32_t cols,
                              uint32_t reductionDim, uint32_t tileLen) {
    xGm_.SetGlobalBuffer((__gm__ float *)x, totalLength);
    yGm_.SetGlobalBuffer((__gm__ float *)y, totalLength);
    rows_ = rows;
    cols_ = cols;
    reductionDim_ = reductionDim;

    if (reductionDim_ == 1) {
      // For dim=1, we need a local buffer of tileLen floats
      tileLen_ = tileLen > 0 ? tileLen : cols_;
      pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
      pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }
  }

  __aicore__ inline void Process() {
    if (reductionDim_ == 0) {
      // cumsum along rows (dim=0) – use scalar loops for correctness
      for (uint32_t c = 0; c < cols_; c++) {
        float running = 0.0f;
        for (uint32_t r = 0; r < rows_; r++) {
          uint32_t idx = r * cols_ + c;
          running += xGm_.GetValue(idx);
          yGm_.SetValue(idx, running);
        }
      }
    } else {
      // cumsum along cols (dim=1) – contiguous, use tiled loads
      for (uint32_t r = 0; r < rows_; r++) {
        float running = 0.0f;
        uint32_t rowBase = r * cols_;
        uint32_t remaining = cols_;
        uint32_t offset = 0;
        while (remaining > 0) {
          uint32_t curTile = (remaining < tileLen_) ? remaining : tileLen_;
          AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
          AscendC::DataCopy(xLocal, xGm_[rowBase + offset], curTile);
          AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
          for (uint32_t i = 0; i < curTile; i++) {
            float v = xLocal(i);
            running += v;
            yLocal(i) = running;
          }
          AscendC::DataCopy(yGm_[rowBase + offset], yLocal, curTile);
          inQueueX_.FreeTensor(xLocal);
          outQueueY_.FreeTensor(yLocal);
          offset += curTile;
          remaining -= curTile;
        }
      }
    }
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> yGm_;

  uint32_t rows_;
  uint32_t cols_;
  uint32_t reductionDim_;
  uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void ca6k00071_custom(GM_ADDR x, GM_ADDR y,
                                                        GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelCa6k00071Custom op;
  op.Init(x, y,
          tiling_data.totalLength, tiling_data.rows, tiling_data.cols,
          tiling_data.reductionDim, tiling_data.tileLen);
  op.Process();
}
"""
