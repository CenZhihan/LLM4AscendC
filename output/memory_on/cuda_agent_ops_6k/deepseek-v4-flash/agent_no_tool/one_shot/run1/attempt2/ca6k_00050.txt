kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCa6k00050Custom {
public:
    __aicore__ inline KernelCa6k00050Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t N, uint32_t C_in, uint32_t C_out,
                                uint32_t H_in, uint32_t W_in,
                                uint32_t H_out, uint32_t W_out,
                                uint32_t kernel_h, uint32_t kernel_w,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t dilation_h, uint32_t dilation_w,
                                uint32_t patchSize)
    {
        N_ = N; C_in_ = C_in; C_out_ = C_out;
        H_in_ = H_in; W_in_ = W_in;
        H_out_ = H_out; W_out_ = W_out;
        kernel_h_ = kernel_h; kernel_w_ = kernel_w;
        stride_h_ = stride_h; stride_w_ = stride_w;
        pad_h_ = pad_h; pad_w_ = pad_w;
        dilation_h_ = dilation_h; dilation_w_ = dilation_w;
        patchSize_ = patchSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C_in * H_in * W_in);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, C_out * C_in * kernel_h * kernel_w);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C_out * H_out * W_out);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, patchSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, patchSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));  // output scalar for each output element
        pipe_.InitBuffer(tmpQueue_, 1, patchSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_out_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t totalOut = N_ * C_out_ * H_out_ * W_out_;
        uint32_t perBlock = (totalOut + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * perBlock;
        uint32_t end = (blockIdx + 1) * perBlock;
        if (start >= totalOut) return;
        if (end > totalOut) end = totalOut;

        for (uint32_t idx = start; idx < end; ++idx) {
            // Decompose idx into n, co, oh, ow
            uint32_t remaining = idx;
            uint32_t ow = remaining % W_out_;
            remaining /= W_out_;
            uint32_t oh = remaining % H_out_;
            remaining /= H_out_;
            uint32_t co = remaining % C_out_;
            uint32_t n = remaining / C_out_;

            float result = 0.0f;
            // Loop over kernel positions and input channels
            for (uint32_t kh = 0; kh < kernel_h_; ++kh) {
                for (uint32_t kw = 0; kw < kernel_w_; ++kw) {
                    uint32_t h_in = oh * stride_h_ - pad_h_ + kh * dilation_h_;
                    uint32_t w_in = ow * stride_w_ - pad_w_ + kw * dilation_w_;
                    if (h_in >= H_in_ || w_in >= W_in_) continue;
                    // For each input channel, load and accumulate
                    for (uint32_t ci = 0; ci < C_in_; ++ci) {
                        float x_val = xGm_[(n * C_in_ * H_in_ * W_in_) + (ci * H_in_ * W_in_) + (h_in * W_in_) + w_in];
                        float w_val = wGm_[(co * C_in_ * kernel_h_ * kernel_w_) + (ci * kernel_h_ * kernel_w_) + (kh * kernel_w_) + kw];
                        result += x_val * w_val;
                    }
                }
            }
            yGm_[idx] = result;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_in_, C_out_, H_in_, W_in_, H_out_, W_out_;
    uint32_t kernel_h_, kernel_w_, stride_h_, stride_w_, pad_h_, pad_w_, dilation_h_, dilation_w_;
    uint32_t patchSize_;
};

extern "C" __global__ __aicore__ void ca6k_00050_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCa6k00050Custom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.C_in, tiling_data.C_out,
            tiling_data.H_in, tiling_data.W_in,
            tiling_data.H_out, tiling_data.W_out,
            tiling_data.kernel_h, tiling_data.kernel_w,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.dilation_h, tiling_data.dilation_w,
            tiling_data.patchSize);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'ca6k_00050_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=(0,0), dilation=(1,1)):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, bias=False)
        # load pre-trained? For simplicity, we re-initialize
        torch.nn.init.normal_(self.conv.weight, mean=0.0, std=0.02)

    def forward(self, x):
        # call custom op with weight
        weight = self.conv.weight
        stride_h, stride_w = self.conv.stride if isinstance(self.conv.stride, tuple) else (self.conv.stride, self.conv.stride)
        pad_h, pad_w = self.conv.padding if isinstance(self.conv.padding, tuple) else (self.conv.padding, self.conv.padding)
        dilation_h, dilation_w = self.conv.dilation if isinstance(self.conv.dilation, tuple) else (self.conv.dilation, self.conv.dilation)
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, weight, stride_h, stride_w, pad_h, pad_w, dilation_h, dilation_w)
'''
