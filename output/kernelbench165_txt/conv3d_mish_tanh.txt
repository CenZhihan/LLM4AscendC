    if (elemsPerBlock == 0) elemsPerBlock = 1;

    tiling.set_block_dim(blockDim);
    tiling.set_elems_per_block(elemsPerBlock);

    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""

#include "kernel_operator.h"

// Fused activation: y = tanh(mish(x))
//   mish(x)   = x * tanh(softplus(x))
//   softplus  = max(x,0) + log(1 + exp(-|x|))
//   tanh(z)   = (1 - exp(-2z)) / (1 + exp(-2z))
// float32, ND contiguous flattened.
//
// Optimizations vs baseline:
// - Double-buffered I/O (BUFFER_NUM=2): overlap DataCopy with compute.
// - Bulk DataCopy for GM<->UB (replaces per-element GetValue/SetValue).
// - Coarse tile (2048 elts): ~256x fewer GM transactions.
// - max(x,0) via (x+|x|)/2 instead of scalar select.
// - 5 shared compute temporaries, reused across tiles.

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t TILE_ELEMS = 4096;

class KernelConv3dMishTanhCustom {
public:
    __aicore__ inline KernelConv3dMishTanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                               uint32_t total_x,
                               uint32_t chunk_elems)
    {
        totalX_ = total_x;
        chunkElems_ = chunk_elems;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(total_x));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(total_x));

        pipe_.InitBuffer(qX_,   BUFFER_NUM, TILE_ELEMS * sizeof(float));
        pipe_.InitBuffer(qY_,   BUFFER_NUM, TILE_ELEMS * sizeof(float));
        pipe_.InitBuffer(qAbs_, 1,          TILE_ELEMS * sizeof(float));
        pipe_.InitBuffer(qTmp_, 1,          TILE_ELEMS * sizeof(float));
        pipe_.InitBuffer(qExp_, 1,          TILE_ELEMS * sizeof(float));
        pipe_.InitBuffer(qNum_, 1,          TILE_ELEMS * sizeof(float));
        pipe_.InitBuffer(qDen_, 1,          TILE_ELEMS * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t bid = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t start = bid * chunkElems_;
        uint32_t end = start + chunkElems_;
        if (end > totalX_) end = totalX_;
        if (start >= end) return;

        uint32_t totalChunk = end - start;
        uint32_t fullTiles = totalChunk / TILE_ELEMS;
        uint32_t tailElems = totalChunk % TILE_ELEMS;

        AscendC::LocalTensor<float> absLocal = qAbs_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = qTmp_.AllocTensor<float>();
        AscendC::LocalTensor<float> expLocal = qExp_.AllocTensor<float>();
        AscendC::LocalTensor<float> numLocal = qNum_.AllocTensor<float>();
        AscendC::LocalTensor<float> denLocal = qDen_.AllocTensor<float>();

        for (uint32_t t = 0; t < fullTiles; ++t) {
            RunTile(start + t * TILE_ELEMS, TILE_ELEMS,
                    absLocal, tmpLocal, expLocal, numLocal, denLocal);
        }
        if (tailElems > 0) {
            RunTile(start + fullTiles * TILE_ELEMS, tailElems,
                    absLocal, tmpLocal, expLocal, numLocal, denLocal);
        }

        qDen_.FreeTensor(denLocal);
        qNum_.FreeTensor(numLocal);
        qExp_.FreeTensor(expLocal);
        qTmp_.FreeTensor(tmpLocal);
        qAbs_.FreeTensor(absLocal);
    }

private:
    __aicore__ inline void RunTile(
        uint32_t offset, uint32_t count,
        AscendC::LocalTensor<float>& absLocal,
        AscendC::LocalTensor<float>& tmpLocal,
        AscendC::LocalTensor<float>& expLocal,
        AscendC::LocalTensor<float>& numLocal,
        AscendC::LocalTensor<float>& denLocal)
    {
        // Double-buffered load
        auto xLocal = qX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offset], count);
        qX_.EnQue(xLocal);
        xLocal = qX_.DeQue<float>();

        auto yLocal = qY_.AllocTensor<float>();

        // ---- softplus(x) = max(x,0) + log(1 + exp(-|x|)) ----
        AscendC::Abs(absLocal, xLocal, count);
        AscendC::Muls(tmpLocal, absLocal, -1.0f, count);
        AscendC::Exp(expLocal, tmpLocal, count);
        AscendC::Adds(expLocal, expLocal, 1.0f, count);
        AscendC::Log(expLocal, expLocal, count);
        // max(x,0) = (x + |x|) / 2
        AscendC::Add(tmpLocal, xLocal, absLocal, count);
        AscendC::Muls(tmpLocal, tmpLocal, 0.5f, count);
        AscendC::Add(tmpLocal, tmpLocal, expLocal, count);  // softplus in tmpLocal

        // ---- tanh(sp) ----
        AscendC::Muls(expLocal, tmpLocal, -2.0f, count);
        AscendC::Exp(expLocal, expLocal, count);
        AscendC::Muls(numLocal, expLocal, -1.0f, count);
        AscendC::Adds(numLocal, numLocal, 1.0f, count);
        AscendC::Adds(denLocal, expLocal, 1.0f, count);
        AscendC::Div(tmpLocal, numLocal, denLocal, count);

        // ---- mish = x * tanh(sp) ----
        AscendC::Mul(yLocal, xLocal, tmpLocal, count);

        // ---- tanh(mish) ----
        AscendC::Muls(tmpLocal, yLocal, -2.0f, count);
        AscendC::Exp(expLocal, tmpLocal, count);
        AscendC::Muls(numLocal, expLocal, -1.0f, count);
        AscendC::Adds(numLocal, numLocal, 1.0f, count);
        AscendC::Adds(denLocal, expLocal, 1.0f, count);
        AscendC::Div(yLocal, numLocal, denLocal, count);

        // Double-buffered store
        qY_.EnQue(yLocal);
        yLocal = qY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], yLocal, count);

        qY_.FreeTensor(yLocal);
        qX_.FreeTensor(xLocal);
    }

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> qX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> qY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> qAbs_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> qTmp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> qExp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> qNum_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> qDen_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalX_;
    uint32_t chunkElems_;
};

extern "C" __global__ __aicore__ void conv3d_mish_tanh_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    (void)workspace;
    GET_TILING_DATA(t, tiling);

    KernelConv3dMishTanhCustom op;
    op.Init(x, y, t.total_x, t.elems_per_block);
    op.Process();
}

eval_src='''
#!/usr/bin/env python3
import json, os, time, torch, torch_npu

def _benchmark(fn, *args, synchronize, warmup=5, repeat=20):
    for _ in range(warmup):
        fn(*args)
    synchronize()
    start = time.perf_counter()
    for _ in range(repeat):
        fn(*args)
    synchronize()
    return (time.perf_counter() - start) * 1000.0 / repeat

def main() -> int:
    module_name = os.environ.get("LLM4ASCENDC_OP_MODULE", "llm4ascendc_ops_conv3d_mish_tanh")
    custom_ops = __import__(module_name)
    fn = getattr(custom_ops, "conv3d_mish_tanh_custom")

    device = torch.device("npu:0")
    synchronize = torch_npu.npu.synchronize

    # Typical Conv3D output shape: NCDHW
    N, C, D, H, W = 16, 64, 30, 62, 62
    x = torch.randn(N, C, D, H, W, dtype=torch.float32, device=device)

    y_custom = fn(x)

    # PyTorch reference: mish(x) = x * tanh(softplus(x)), then tanh
    sp = torch.nn.functional.softplus(x)
    mish = x * torch.tanh(sp)
    y_ref = torch.tanh(mish)

    if not torch.allclose(y_custom, y_ref, atol=1e-4, rtol=1e-4):
        md = (y_custom - y_ref).abs().max().item()
        print(f"[FAIL] Output mismatch: max_diff={md}")
        return 1
    print("[eval] allclose=True [ref=PyTorch mish+tanh]")

    ref_fn = lambda t: torch.tanh(t * torch.tanh(torch.nn.functional.softplus(t)))
    ref_ms = _benchmark(ref_fn, x, synchronize=synchronize)
    custom_ms = _benchmark(fn, x, synchronize=synchronize)
    speedup = ref_ms / custom_ms if custom_ms > 0 else None

    perf = {
        "ref_ms": ref_ms, "custom_ms": custom_ms,
        "speedup": speedup, "warmup": 5, "repeat": 20,
        "ref_device": str(device), "custom_device": str(device),
    }
    print("[perf] " + json.dumps(perf))
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
'''
