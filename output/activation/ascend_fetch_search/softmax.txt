#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t MAX_BLOCK_DIM = 32;
static constexpr uint32_t ELEMS_PER_ROW = 512;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint64_t AlignDown64(uint64_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::Shape &originShape = inputShape->GetOriginShape();

    uint32_t batchSize = 1;
    uint32_t dimSize = 1;
    
    if (originShape.GetDims() >= 2) {
        batchSize = static_cast<uint32_t>(originShape.GetDimValue(originShape.GetDims() - 2));
        dimSize = static_cast<uint32_t>(originShape.GetDimValue(originShape.GetDims() - 1));
    } else if (originShape.GetDims() == 1) {
        dimSize = static_cast<uint32_t>(originShape.GetDimValue(0));
    }

    uint64_t totalLength = 1;
    for (int i = 0; i < originShape.GetDims(); ++i) {
        totalLength *= static_cast<uint64_t>(originShape.GetDimValue(i));
    }

    int32_t softmaxDim = -1;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs != nullptr && attrs->GetAttrCount() > 0) {
        const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
        if (dimAttr != nullptr) {
            softmaxDim = *dimAttr;
        }
    }

    uint32_t blockDim = batchSize;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;

    uint32_t tileRowNum = 1;
    if (dimSize > ELEMS_PER_ROW) {
        tileRowNum = CeilDiv(dimSize, ELEMS_PER_ROW);
    }

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_batchSize(batchSize);
    tiling.set_dimSize(dimSize);
    tiling.set_tileRowNum(tileRowNum);
    tiling.set_blockDim(blockDim);
    tiling.set_softmaxDim(softmaxDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

        this->AICore().SetTiling(optiling::TilingFunc);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint64_t totalLength, uint32_t batchSize,
                                uint32_t dimSize, uint32_t tileRowNum,
                                uint32_t blockDim, int32_t softmaxDim)
    {
        totalLen_ = totalLength;
        batchSize_ = batchSize;
        dimSize_ = dimSize;
        tileRowNum_ = tileRowNum;
        blockDim_ = blockDim;
        softmaxDim_ = softmaxDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);

        if (tileRowNum_ > 1) {
            pipe_.InitBuffer(inQueueX_, BUFFER_NUM, ELEMENTS_PER_TILE * sizeof(float));
            pipe_.InitBuffer(outQueueY_, BUFFER_NUM, ELEMENTS_PER_TILE * sizeof(float));
            pipe_.InitBuffer(maxQueue_, BUFFER_NUM, tileRowNum_ * sizeof(float));
            pipe_.InitBuffer(sumQueue_, BUFFER_NUM, tileRowNum_ * sizeof(float));
            pipe_.InitBuffer(expQueue_, BUFFER_NUM, ELEMENTS_PER_TILE * sizeof(float));
            pipe_.InitBuffer(expLocalQueue_, BUFFER_NUM, ELEMENTS_PER_TILE * sizeof(float));
        } else {
            pipe_.InitBuffer(inQueueX_, BUFFER_NUM, dimSize_ * sizeof(float));
            pipe_.InitBuffer(outQueueY_, BUFFER_NUM, dimSize_ * sizeof(float));
            pipe_.InitBuffer(maxQueue_, BUFFER_NUM, sizeof(float));
            pipe_.InitBuffer(sumQueue_, BUFFER_NUM, sizeof(float));
            pipe_.InitBuffer(expQueue_, BUFFER_NUM, dimSize_ * sizeof(float));
        }
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();

        if (tileRowNum_ > 1) {
            ProcessLargeDim(blockIdx);
        } else {
            ProcessSmallDim(blockIdx);
        }
    }

private:
    __aicore__ inline void ProcessSmallDim(uint32_t blockIdx)
    {
        uint32_t rowStride = blockDim_;
        uint32_t rowIdx = blockIdx;

        while (rowIdx < batchSize_) {
            uint32_t offset = rowIdx * dimSize_;
            CopyInRow(offset, dimSize_);
            ComputeRowSoftmax(dimSize_);
            CopyOutRow(offset, dimSize_);
            rowIdx += rowStride;
        }
    }

    __aicore__ inline void ProcessLargeDim(uint32_t blockIdx)
    {
        uint32_t rowStride = blockDim_;
        uint32_t rowIdx = blockIdx;

        while (rowIdx < batchSize_) {
            uint32_t rowOffset = rowIdx * dimSize_;
            
            for (uint32_t tileIdx = 0; tileIdx < tileRowNum_; ++tileIdx) {
                uint32_t tileOffset = rowOffset + tileIdx * ELEMENTS_PER_TILE;
                uint32_t curLen = ELEMENTS_PER_TILE;
                if (tileOffset + curLen > (rowIdx + 1) * dimSize_) {
                    curLen = (rowIdx + 1) * dimSize_ - tileOffset;
                }
                
                if (tileIdx == 0) {
                    CopyIn(tileOffset, curLen);
                    ComputeTileMax(curLen, tileIdx);
                } else if (tileIdx < tileRowNum_ - 1) {
                    CopyIn(tileOffset, curLen);
                    ComputeTileMax(curLen, tileIdx);
                    ComputeTileExpAndSum(curLen, tileIdx);
                } else {
                    ComputeTileExpAndSum(curLen, tileIdx);
                    ComputeTileFinalExp(curLen, tileIdx);
                    ComputeTileNormalize(curLen, tileIdx);
                    CopyOut(tileOffset, curLen);
                }
            }
            rowIdx += rowStride;
        }
    }

    __aicore__ inline void CopyInRow(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyIn(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRowSoftmax(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maxLocal = maxQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> expLocal = expQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::ReduceMax(maxLocal, xLocal, 0, len, {len}, true);
        AscendC::Sub(expLocal, xLocal, maxLocal, len);
        AscendC::Exp(expLocal, expLocal, len);
        AscendC::ReduceSum(sumLocal, expLocal, 0, len, {len}, true);
        AscendC::Div(yLocal, expLocal, sumLocal, len);

        maxQueue_.FreeTensor(maxLocal);
        sumQueue_.FreeTensor(sumLocal);
        expQueue_.FreeTensor(expLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeTileMax(uint32_t len, uint32_t tileIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maxLocal = maxQueue_.AllocTensor<float>();
        
        AscendC::ReduceMax(maxLocal, xLocal, 0, len, {len}, true);
        
        maxQueue_.EnQue<float>(maxLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeTileExpAndSum(uint32_t len, uint32_t tileIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> expLocal = expQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = maxQueue_.DeQue<float>();
        AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
        
        AscendC::Sub(expLocal, xLocal, maxLocal, len);
        AscendC::Exp(expLocal, expLocal, len);
        AscendC::ReduceSum(sumLocal, expLocal, 0, len, {len}, true);
        
        expQueue_.EnQue<float>(expLocal);
        maxQueue_.FreeTensor(maxLocal);
        sumQueue_.FreeTensor(sumLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeTileFinalExp(uint32_t len, uint32_t tileIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maxLocal = maxQueue_.DeQue<float>();
        AscendC::LocalTensor<float> expLocal = expQueue_.AllocTensor<float>();
        
        AscendC::Sub(expLocal, xLocal, maxLocal, len);
        AscendC::Exp(expLocal, expLocal, len);
        
        maxQueue_.FreeTensor(maxLocal);
        expQueue_.FreeTensor(expLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeTileNormalize(uint32_t len, uint32_t tileIdx)
    {
        AscendC::LocalTensor<float> expLocal = expQueue_.DeQue<float>();
        AscendC::LocalTensor<float> sumLocal = sumQueue_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        
        AscendC::Div(yLocal, expLocal, sumLocal, len);
        
        sumQueue_.FreeTensor(sumLocal);
        expQueue_.FreeTensor(expLocal);
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    static constexpr uint32_t ELEMENTS_PER_TILE = 512;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> maxQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> sumQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> expQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> expLocalQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint64_t totalLen_ {0};
    uint32_t batchSize_ {1};
    uint32_t dimSize_ {1};
    uint32_t tileRowNum_ {1};
    uint32_t blockDim_ {1};
    int32_t softmaxDim_ {-1};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.batchSize, 
           tiling_data.dimSize, tiling_data.tileRowNum, 
           tiling_data.blockDim, tiling_data.softmaxDim);
    op.Process();
}
