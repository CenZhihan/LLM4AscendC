host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue (max, exp, sum): 3 * tile
// Total floats = 7 * tile => bytes = 28 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 512;             // align for DMA efficiency
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // hard cap for stability

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t axis = 1;  // default dim
    if (attrs != nullptr && attrs->GetAttrNum() > 0) {
        axis = static_cast<int32_t>(attrs->GetAttrPointer<int64_t>(0)[0]);
    }
    if (axis < 0) axis = 1;

    // Tile length from UB budget: bytes = 28 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 28;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    
    // Block count: process one batch per block
    uint32_t blockDim = 4096;
    if (tileLen > 0 && totalLength > tileLen) {
        uint32_t batchCount = CeilDiv(totalLength, tileLen);
        if (batchCount < blockDim) blockDim = batchCount;
        if (blockDim < 8u) blockDim = 8u;
    }

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_axis(axis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)

        this->Input("x")

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t axis)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        axis_ = axis;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueMax_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueExp_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueSum_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes one batch element
        const uint32_t batchCount = (tileLen_ > 0) ? (totalLen_ / tileLen_) : 1;
        if (blockIdx >= batchCount) return;

        const uint32_t offset = blockIdx * tileLen_;

        // Process all tiles for this batch
        const uint32_t tilesPerBatch = 1;
        for (uint32_t tileIdx = 0; tileIdx < tilesPerBatch; ++tileIdx) {
            uint32_t curOffset = offset + tileIdx * tileLen_;
            uint32_t curLen = tileLen_;
            
            CopyIn(curOffset, curLen);
            Compute(curLen);
            CopyOut(curOffset, curLen);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = tmpQueueMax_.AllocTensor<float>();
        AscendC::LocalTensor<float> expLocal = tmpQueueExp_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = tmpQueueSum_.AllocTensor<float>();

        // y = x - max - log(sum(exp(x - max)))
        // Step 1: max over all elements
        AscendC::ReduceMax(maxLocal, xLocal, 0, len, {1, 1, 8, 8});
        
        // Step 2: x - max
        AscendC::Sub(expLocal, xLocal, maxLocal, len);
        
        // Step 3: exp(x - max)
        AscendC::Exp(expLocal, expLocal, len);
        
        // Step 4: sum of exp
        AscendC::ReduceSum(sumLocal, expLocal, 0, len, {1, 1, 8, 8});
        
        // Step 5: log(sum)
        AscendC::Log(sumLocal, sumLocal, 1);
        
        // Step 6: y = x - max - log(sum)
        AscendC::Sub(yLocal, xLocal, maxLocal, len);
        AscendC::Sub(yLocal, yLocal, sumLocal, len);

        tmpQueueMax_.FreeTensor(maxLocal);
        tmpQueueExp_.FreeTensor(expLocal);
        tmpQueueSum_.FreeTensor(sumLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueExp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueSum_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t axis_ {1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.axis);
    op.Process();
}
