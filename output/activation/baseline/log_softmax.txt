host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue: 1 * tile (for exp values)
//   tmpQueue2: 1 * 1 (for scalar accum)
// Total floats = 5 * tile + 1 => bytes = 20 * tile + 4
static constexpr uint32_t UB_BUDGET_BYTES = 196608;     // 192KB per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;           // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;      // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    // Input shape: (batch_size, dim), dim=-1 means last dim
    const gert::Shape *inputShape = context->GetInputShape(0);
    const int32_t dimVal = static_cast<int32_t>(inputShape->GetShape().size()) + inputShape->GetShape().at(0);
    const uint32_t dim = (dimVal < 0) ?
        static_cast<uint32_t>(inputShape->GetShape().back()) :
        static_cast<uint32_t>(inputShape->GetShape().at(dimVal));
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // tileLen from UB budget: bytes = 20 * tileLen + 4
    uint32_t maxTile = (UB_BUDGET_BYTES - 4) / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // blockDim: one block per batch for best parallelism
    const uint32_t batchCount = totalLength / dim;
    uint32_t blockDim = batchCount;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t dim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLen;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // UB layout: 2*in + 2*out + 1*tmp + 1*scalar
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || dim_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes one batch element
        const uint32_t batchCount = totalLen_ / dim_;
        const uint32_t numBlocks = (batchCount + blockNum - 1) / blockNum;

        for (uint32_t b = 0; b < numBlocks; ++b) {
            uint32_t batchIdx = blockIdx + b * blockNum;
            if (batchIdx >= batchCount) break;

            uint32_t base = batchIdx * dim_;

            // Phase 1: Find max for this batch
            GetMax(base);

            // Phase 2: Compute softmax with loaded tiles
            ComputeSoftmax(base);

            // Copy out
            CopyOut(base);
        }
    }

private:
    __aicore__ inline void GetMax(uint32_t base)
    {
        const uint32_t numTiles = (dim_ + tileLen_ - 1) / tileLen_;
        float batchMax = -3.4e38f;  // -inf

        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = base + t * tileLen_;
            uint32_t len = tileLen_;
            if (offset + len > base + dim_) {
                len = base + dim_ - offset;
            }

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            float tileMax = AscendC::ReduceMax(xLocal, len);
            if (tileMax > batchMax) batchMax = tileMax;
            inQueueX_.FreeTensor(xLocal);
        }

        maxVal_ = batchMax;
    }

    __aicore__ inline void ComputeSoftmax(uint32_t base)
    {
        const uint32_t numTiles = (dim_ + tileLen_ - 1) / tileLen_;
        float scalarBuf = 0.0f;

        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = base + t * tileLen_;
            uint32_t len = tileLen_;
            if (offset + len > base + dim_) {
                len = base + dim_ - offset;
            }

            // Copy in
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            inQueueX_.EnQue(xLocal);

            // Compute: y = (x - max) - log(sum(exp(x - max)))
            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> wsLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> scalarLocal = tmpQueue2_.AllocTensor<float>();

            // y = x - max
            AscendC::Subs(yLocal, xLocal, maxVal_, len);
            // ws = exp(x - max)
            AscendC::Exp(wsLocal, yLocal, len);
            // scalar = sum(exp(x - max))
            AscendC::ReduceSum(wsLocal, scalarLocal, len);
            // scalar = log(sum(exp(x - max)))
            AscendC::Log(scalarLocal, scalarLocal, 1);
            // y = (x - max) - log(sum(...))
            AscendC::Subs(yLocal, yLocal, scalarLocal, len);

            tmpQueue2_.FreeTensor(scalarLocal);
            tmpQueue_.FreeTensor(wsLocal);
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

    __aicore__ inline void CopyOut(uint32_t base)
    {
        const uint32_t numTiles = (dim_ + tileLen_ - 1) / tileLen_;

        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = base + t * tileLen_;
            uint32_t len = tileLen_;
            if (offset + len > base + dim_) {
                len = base + dim_ - offset;
            }

            AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yLocal, len);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t dim_ {1};
    float maxVal_ {0.0f};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
