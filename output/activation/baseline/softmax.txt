host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   expQueue: 2 * tile
//   reduceQueue: 2 * tile (for multi-block reduction within a row)
//   outQueueY: 2 * tile
//   tmpReduce: 1 * tile
// Total floats = 9 * tile => bytes = 36 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const std::vector<int64_t> &shape = inputShape->GetStorageShape();
    
    // Handle both 2D and N-D tensors
    // dim=1 means softmax along the second dimension
    int32_t softmaxDim = 1;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs != nullptr && attrs->GetAttrNum() > 0) {
        int64_t dimVal = attrs->GetAttrInt(0);
        softmaxDim = static_cast<int32_t>(dimVal);
    }
    
    // Normalize dim to positive index
    int32_t dim = softmaxDim;
    if (dim < 0) {
        dim = static_cast<int32_t>(shape.size()) + dim;
    }
    
    uint32_t dimSize = 1;
    if (dim >= 0 && dim < static_cast<int32_t>(shape.size())) {
        dimSize = static_cast<uint32_t>(shape[dim]);
    }
    
    const uint32_t totalLength =
        static_cast<uint32_t>(inputShape->GetOriginShape().GetShapeSize());
    
    // Tile length should be at most dimSize (process one row at a time)
    uint32_t maxTile = UB_BUDGET_BYTES / 36;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > dimSize) maxTile = dimSize;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }
    
    // blockDim: number of rows to process in parallel
    uint32_t rows = 1;
    for (size_t i = 0; i < shape.size(); ++i) {
        if (static_cast<int32_t>(i) != dim) {
            rows *= static_cast<uint32_t>(shape[i]);
        }
    }
    
    uint32_t blockDim = rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;
    
    context->SetBlockDim(blockDim);
    
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_softmaxDim(dim);
    tiling.set_dimSize(dimSize);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t REDUCE_BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t softmaxDim, uint32_t dimSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        softmaxDim_ = softmaxDim;
        dimSize_ = dimSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // UB allocation: inQueueX(2*tile) + expQueue(2*tile) + reduceQueue(2*tile) + outQueueY(2*tile) + tmpReduce(1*tile)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(expQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(reduceQueue_, REDUCE_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpReduce_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes one complete row (dimSize elements)
        uint32_t rowSize = dimSize_;
        uint32_t numRows = (totalLen_ + rowSize - 1) / rowSize;
        
        // Grid-stride across rows
        uint32_t strideRows = blockNum;
        uint32_t rowStart = blockIdx;
        
        while (rowStart < numRows) {
            uint32_t rowOffset = rowStart * rowSize;
            uint32_t actualRowLen = rowSize;
            if (rowOffset + actualRowLen > totalLen_) {
                actualRowLen = totalLen_ - rowOffset;
            }
            
            ComputeRow(rowOffset, actualRowLen);
            
            rowStart += strideRows;
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t rowOffset, uint32_t rowLen)
    {
        // Step 1: Copy input row to UB
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowOffset], rowLen);
        inQueueX_.EnQue(xLocal);
        
        // Step 2: Compute exp(x)
        AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> expLocal = expQueue_.AllocTensor<float>();
        AscendC::Exp(expLocal, xDeq, rowLen);
        expQueue_.EnQue<float>(expLocal);
        inQueueX_.FreeTensor(xDeq);
        
        // Step 3: Reduce to get sum of exp(x)
        AscendC::LocalTensor<float> expDeq = expQueue_.DeQue<float>();
        AscendC::LocalTensor<float> sumReduce = reduceQueue_.AllocTensor<float>();
        AscendC::ReduceSum(sumReduce, expDeq, 0, rowLen);
        reduceQueue_.EnQue<float>(sumReduce);
        expQueue_.FreeTensor(expDeq);
        
        // Step 4: Normalize - divide each exp(x) by sum
        AscendC::LocalTensor<float> sumDeq = reduceQueue_.DeQue<float>();
        AscendC::LocalTensor<float> expDeq2 = expQueue_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Div(yLocal, expDeq2, sumDeq, rowLen);
        
        // Step 5: Copy output back to GM
        AscendC::DataCopy(yGm_[rowOffset], yLocal, rowLen);
        outQueueY_.FreeTensor(yLocal);
        expQueue_.FreeTensor(expDeq2);
        reduceQueue_.FreeTensor(sumDeq);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> expQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, REDUCE_BUFFER_NUM> reduceQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpReduce_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t softmaxDim_ {1};
    uint32_t dimSize_ {1};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.softmaxDim, tiling_data.dimSize);
    op.Process();
}
