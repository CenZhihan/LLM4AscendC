host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueueExp: 2 * tile (for exp values)
//   tmpQueueMax: 2 * tile (for max reduction)
//   tmpQueueSum: 1 * tile (for sum reduction)
// Total floats = 9 * tile => bytes = 36 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const int32_t softmaxDim = 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t dim = attrs->HasAttr(0) ? 
        static_cast<int32_t>(attrs->GetAttrPointer<int64_t>(0)[0]) : softmaxDim;

    const std::vector<int64_t>& shape = inputShape->GetStorageShape().GetDims();
    const int32_t nDim = static_cast<int32_t>(shape.size());

    int32_t actualDim = dim;
    if (actualDim < 0) {
        actualDim = nDim + actualDim;
    }

    uint32_t totalLength = 1;
    for (int32_t i = 0; i < nDim; ++i) {
        totalLength *= static_cast<uint32_t>(shape[i]);
    }

    uint32_t outerSize = 1;
    for (int32_t i = 0; i < actualDim; ++i) {
        outerSize *= static_cast<uint32_t>(shape[i]);
    }

    uint32_t innerSize = 1;
    for (int32_t i = actualDim + 1; i < nDim; ++i) {
        innerSize *= static_cast<uint32_t>(shape[i]);
    }

    uint32_t dimSize = static_cast<uint32_t>(shape[actualDim]);

    uint32_t maxTile = UB_BUDGET_BYTES / 36;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t blockDim = CeilDiv(totalLength, tileLen);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_softmaxDim(dim);
    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int64(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t softmaxDim, uint32_t outerSize, uint32_t innerSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLen;
        softmaxDim_ = softmaxDim;
        outerSize_ = outerSize;
        innerSize_ = innerSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueExp_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueMax_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueSum_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maxLocal = tmpQueueMax_.AllocTensor<float>();
        AscendC::LocalTensor<float> expLocal = tmpQueueExp_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = tmpQueueSum_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Step 1: Find max value for numerical stability
        // ReduceMax along the softmax dimension
        const uint32_t dimSize = len;
        AscendC::ReduceMax(maxLocal, xLocal, true, dimSize, 0);

        // Step 2: exp(x - max)
        AscendC::Subs(expLocal, xLocal, maxLocal, len);
        AscendC::Exp(expLocal, expLocal, len);

        // Step 3: Sum of exponentials
        AscendC::ReduceSum(sumLocal, expLocal, true, dimSize, 0);

        // Step 4: y = exp(x - max) / sum
        AscendC::Div(yLocal, expLocal, sumLocal, len);

        tmpQueueMax_.FreeTensor(maxLocal);
        tmpQueueExp_.FreeTensor(expLocal);
        tmpQueueSum_.FreeTensor(sumLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueExp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueSum_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t softmaxDim_ {1};
    uint32_t outerSize_ {1};
    uint32_t innerSize_ {1};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, 
            tiling_data.softmaxDim, tiling_data.outerSize, tiling_data.innerSize);
    op.Process();
}
