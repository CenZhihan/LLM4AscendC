host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 1 * row  (input row)
//   outQueueY: 1 * row (output softmax)
//   tmpQueue: 3 * row  (exp, max, sum - 32B each scalar in UB)
// Total floats = 5 * row => bytes = 20 * row
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t ELEMS_PER_BLOCK_MAX = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // For softmax on dim=1, each row has rowLength elements
    // We process complete rows per block for simplicity
    uint32_t rowLength = 1;
    if (context->GetInputShape(0)->GetOriginShape().GetDims().size() >= 2) {
        rowLength = static_cast<uint32_t>(
            context->GetInputShape(0)->GetOriginShape().GetDims()[1]);
    }
    if (rowLength == 0) rowLength = totalLength;

    uint32_t rowNum = CeilDiv(totalLength, rowLength);

    // Calculate tileRows (rows per block) based on UB budget
    // UB needed per row: 5 * rowLength floats = 20 * rowLength bytes
    uint32_t maxElemsPerBlock = UB_BUDGET_BYTES / 20;
    if (maxElemsPerBlock > ELEMS_PER_BLOCK_MAX) {
        maxElemsPerBlock = ELEMS_PER_BLOCK_MAX;
    }
    maxElemsPerBlock = AlignDown(maxElemsPerBlock, ALIGN_ELEMS);

    // How many complete rows can fit in UB
    uint32_t tileRows = maxElemsPerBlock / rowLength;
    if (tileRows == 0) tileRows = 1;
    if (tileRows > rowNum) tileRows = rowNum;

    uint32_t tileLength = tileRows * rowLength;

    // Block dim: enough blocks to cover all rows
    uint32_t blockDim = CeilDiv(rowNum, tileRows);
    if (blockDim < 8u) blockDim = 8u;
    // Limit blockDim for stability
    if (blockDim > 64u) blockDim = 64u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_rowNum(rowNum);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t rowNum, uint32_t tileRows)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        rowNum_ = rowNum;
        tileRows_ = tileRows;
        tileLen_ = tileRows_ * rowLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // UB layout per tile:
        // inQueueX_: tileLen_ (input row)
        // outQueueY_: tileLen_ (output softmax)
        // tmpQueue_: 3 scalars (max, sum, and exp workspace)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || rowNum_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t startRow = blockIdx * tileRows_;
        if (startRow >= rowNum_) return;

        uint32_t actualRows = tileRows_;
        if (startRow + actualRows > rowNum_) {
            actualRows = rowNum_ - startRow;
        }

        // Process each row in this tile
        for (uint32_t r = 0; r < actualRows; ++r) {
            uint32_t rowIdx = startRow + r;
            uint32_t rowOffset = rowIdx * rowLen_;
            ProcessRow(rowOffset, rowLen_);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowOffset, uint32_t len)
    {
        // Step 1: Copy input row to UB
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowOffset], len);
        inQueueX_.EnQue(xLocal);

        // Step 2: Compute max value for numerical stability
        AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> workspaceLocal = tmpQueue_.AllocTensor<float>();
        AscendC::ReduceMax(maxLocal, xLocal, workspaceLocal, len);
        tmpQueue_.EnQue(maxLocal);

        // Step 3: Subtract max (x - max) for numerical stability
        AscendC::LocalTensor<float> xDequeued = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> shiftedLocal = tmpQueue_.AllocTensor<float>();
        AscendC::Muls(shiftedLocal, maxLocal, -1.0f, 1);
        AscendC::Adds(shiftedLocal, shiftedLocal, xDequeued, len);
        inQueueX_.FreeTensor(xDequeued);

        // Step 4: Compute exp(x - max)
        AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
        AscendC::Exp(expLocal, shiftedLocal, len);

        // Step 5: Compute sum of exp values
        AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumWorkspace = tmpQueue_.AllocTensor<float>();
        AscendC::ReduceSum(sumLocal, expLocal, sumWorkspace, len);
        tmpQueue_.EnQue(sumLocal);

        // Step 6: Broadcast sum and divide exp by sum
        AscendC::LocalTensor<float> sumDequeued = tmpQueue_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        
        // Expand sum to same length as exp for division
        // Use Adds to create a broadcast-friendly tensor
        AscendC::LocalTensor<float> sumBroadcast = tmpQueue_.AllocTensor<float>();
        AscendC::Muls(sumBroadcast, sumDequeued, 0.0f, len);
        AscendC::Adds(sumBroadcast, sumBroadcast, sumDequeued, 1);
        
        // Division: y = exp / sum
        AscendC::Div(yLocal, expLocal, sumBroadcast, 0xFFFFFFFF, 1, 
                     AscendC::BinaryRepeatParams{8, 4, 8, 4});

        // Free intermediate tensors
        tmpQueue_.FreeTensor(sumBroadcast);
        tmpQueue_.FreeTensor(sumDequeued);
        tmpQueue_.FreeTensor(expLocal);
        tmpQueue_.FreeTensor(shiftedLocal);
        tmpQueue_.FreeTensor(maxLocal);

        outQueueY_.EnQue<float>(yLocal);

        // Step 7: Copy output to GM
        AscendC::LocalTensor<float> yDequeued = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[rowOffset], yDequeued, len);
        outQueueY_.FreeTensor(yDequeued);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t rowNum_ {0};
    uint32_t tileRows_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLength, 
            tiling_data.rowNum, tiling_data.tileRows);
    op.Process();
}
