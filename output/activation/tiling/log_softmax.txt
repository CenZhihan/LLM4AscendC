#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const int64_t *shapeData = inputShape->GetShapeData();
    uint32_t shapeSize = inputShape->GetShapeSize();
    
    uint64_t totalLength = 1;
    for (uint32_t i = 0; i < shapeSize; ++i) {
        totalLength *= static_cast<uint64_t>(shapeData[i]);
    }
    
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->HasAttr(0) ? attrs->GetAttr<int32_t>(0) : -1;
    if (dim < 0) dim += static_cast<int32_t>(shapeSize);
    
    uint32_t dimSize = static_cast<uint32_t>(shapeData[dim]);
    uint32_t outerSize = static_cast<uint32_t>(totalLength / dimSize);
    
    uint32_t maxTile = UB_BUDGET_BYTES / (7 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    
    uint32_t tileLen = maxTile;
    if (outerSize < tileLen) {
        const uint32_t aligned = AlignDown(outerSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : outerSize;
    }
    
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(outerSize, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;
    
    context->SetBlockDim(blockDim);
    
    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_dimSize(dimSize);
    tiling.set_outerSize(outerSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t REDUCE_BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t dimSize,
                                uint32_t outerSize, uint32_t tileLength,
                                int32_t dim)
    {
        totalLen_ = totalLength;
        dimSize_ = dimSize;
        outerSize_ = outerSize;
        tileLen_ = tileLength;
        dim_ = dim;
        
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * dimSize_ * sizeof(float));
        pipe_.InitBuffer(reduceMaxQueue_, REDUCE_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(reduceSumQueue_, REDUCE_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * dimSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * dimSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || dimSize_ == 0 || outerSize_ == 0) return;
        
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        
        const uint32_t strideTiles = blockNum * tileLen_;
        uint32_t tileBase = blockIdx * tileLen_;
        if (tileBase >= outerSize_) return;
        
        CopyIn(tileBase, tileLen_);
        
        while (true) {
            const uint32_t nextTileBase = tileBase + strideTiles;
            uint32_t curTileLen = tileLen_;
            
            if (nextTileBase >= outerSize_) {
                curTileLen = outerSize_ - tileBase;
            }
            
            if (tileBase < outerSize_) {
                uint32_t actualLen = curTileLen;
                if (nextTileBase < outerSize_) {
                    CopyIn(nextTileBase, curTileLen);
                }
                
                Compute(actualLen);
                CopyOut(tileBase, actualLen);
            }
            
            if (nextTileBase >= outerSize_) break;
            tileBase = nextTileBase;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t tileIdx, uint32_t tileLen)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        uint32_t gmOffset = tileIdx * dimSize_;
        AscendC::DataCopy(xLocal, xGm_[gmOffset], tileLen * dimSize_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t tileLen)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maxLocal = reduceMaxQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = reduceSumQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        
        // Step 1: Find max along dimSize dimension for each row
        AscendC::ReduceMax(maxLocal, xLocal, true, tileLen, dimSize_);
        
        // Step 2: Compute exp(x - max) for subtraction
        // y = x - max
        AscendC::Sub(tmpLocal, xLocal, maxLocal, tileLen * dimSize_);
        
        // Step 3: Compute exp(tmp)
        AscendC::Exp(tmpLocal, tmpLocal, tileLen * dimSize_);
        
        // Step 4: Sum exp values along dimSize dimension
        AscendC::ReduceSum(sumLocal, tmpLocal, true, tileLen, dimSize_);
        
        // Step 5: Compute log(sum)
        AscendC::Log(sumLocal, sumLocal, tileLen);
        
        // Step 6: y = x - max - log(sum)
        // tmpLocal currently has exp(x-max), need to reload x
        // Actually, let's do: y = x - max - log(sum_exp)
        // xLocal still has original x, maxLocal has max, sumLocal has log(sum)
        
        // Broadcast max and sum along dim dimension for each row
        AscendC::Sub(yLocal, xLocal, maxLocal, tileLen * dimSize_);
        AscendC::Sub(yLocal, yLocal, sumLocal, tileLen * dimSize_);
        
        outQueueY_.EnQue<float>(yLocal);
        
        reduceMaxQueue_.FreeTensor(maxLocal);
        reduceSumQueue_.FreeTensor(sumLocal);
        tmpQueue_.FreeTensor(tmpLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t tileIdx, uint32_t tileLen)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint32_t gmOffset = tileIdx * dimSize_;
        AscendC::DataCopy(yGm_[gmOffset], yLocal, tileLen * dimSize_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, REDUCE_BUFFER_NUM> reduceMaxQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, REDUCE_BUFFER_NUM> reduceSumQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t dimSize_ {0};
    uint32_t outerSize_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ {-1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.dimSize, 
            tiling_data.outerSize, tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
